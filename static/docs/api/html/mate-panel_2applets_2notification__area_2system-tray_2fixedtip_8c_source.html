50static gboolean
51button_press_handler (GtkWidget *fixedtip,
52 GdkEventButton *event,
53 gpointer data)
54{
55 if (event->button == 1 && event->type == GDK_BUTTON_PRESS)

61static gboolean
62na_fixed_tip_draw (GtkWidget *widget, cairo_t *cr)
63{
64 GtkStyleContext *context;
65 GtkStateFlags state;
66 int width, height;
67
68 width = gtk_widget_get_allocated_width (widget);
69 height = gtk_widget_get_allocated_height (widget);
70
71 state = gtk_widget_get_state_flags (widget);
72 context = gtk_widget_get_style_context (widget);
73 gtk_style_context_save (context);
74 gtk_style_context_add_class (context, GTK_STYLE_CLASS_TOOLTIP);
75 gtk_style_context_set_state (context, state);
76
77 cairo_save (cr);
78 gtk_render_background (context, cr,
79 0., 0.,
80 (gdouble)width,
81 (gdouble)height);
82 cairo_restore (cr);
83
84 gtk_style_context_restore (context);
85
86 return FALSE;
87}

137static void
138na_fixed_tip_position (NaFixedTip *fixedtip)
139{
140 GdkScreen *screen;
141 GdkWindow *parent_window;
142 GtkRequisition req;
143 int root_x;
144 int root_y;
145 int parent_width;
146 int parent_height;
147 int screen_width;
148 int screen_height;
149
150 screen = gtk_widget_get_screen (fixedtip->priv->parent);
151 parent_window = gtk_widget_get_window (fixedtip->priv->parent);
152
153 gtk_window_set_screen (GTK_WINDOW (fixedtip), screen);
154
155 gtk_widget_get_preferred_size (GTK_WIDGET (fixedtip), &req, NULL);
156
157 gdk_window_get_origin (parent_window, &root_x, &root_y);
158 parent_width = gdk_window_get_width(parent_window);
159 parent_height = gdk_window_get_height(parent_window);
160
161 screen_width = WidthOfScreen (gdk_x11_screen_get_xscreen (screen));
162 screen_height = HeightOfScreen (gdk_x11_screen_get_xscreen (screen));
163
164 /* pad between panel and message window */
165#define PAD 5
166
167 if (fixedtip->priv->orientation == GTK_ORIENTATION_VERTICAL)
168 {
169 if (root_x <= screen_width / 2)
170 root_x += parent_width + PAD;
171 else
172 root_x -= req.width + PAD;
173 }
174 else
175 {
176 if (root_y <= screen_height / 2)
177 root_y += parent_height + PAD;
178 else
179 root_y -= req.height + PAD;
180 }
181
182 /* Push onscreen */
183 if ((root_x + req.width) > screen_width)
184 root_x = screen_width - req.width;
185
186 if ((root_y + req.height) > screen_height)
187 root_y = screen_height - req.height;
188
189 gtk_window_move (GTK_WINDOW (fixedtip), root_x, root_y);
190}
