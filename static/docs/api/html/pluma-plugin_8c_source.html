138static void
139##(PLUGIN_ID.lower)_plugin_activate (PlumaWindowActivatable *activatable)
140{
141##ifdef WITH_MENU
142 ##(PLUGIN_ID.camel)PluginPrivate *data;
143 PlumaWindow *window;

145 GError *error = NULL;
146##endif
147
148 pluma_debug (DEBUG_PLUGINS);
149
150##ifdef WITH_MENU
151 data = ##(PLUGIN_ID.upper)_PLUGIN (activatable)->priv;
152 window = PLUMA_WINDOW (data->window);
153
154 manager = pluma_window_get_ui_manager (window);
155
156 data->action_group = gtk_action_group_new ("##(PLUGIN_ID.camel)PluginActions");
157 gtk_action_group_set_translation_domain (data->action_group,
158 GETTEXT_PACKAGE);
159 gtk_action_group_add_actions (data->action_group,
160 action_entries,
161 G_N_ELEMENTS (action_entries),
162 window);
163
164 gtk_ui_manager_insert_action_group (manager, data->action_group, -1);
165
166 data->ui_id = gtk_ui_manager_add_ui_from_string (manager,
167 submenu,
168 -1,
169 &error);
170 if (data->ui_id == 0)
171 {
172 g_warning ("%s", error->message);
173 return;
174 }
175 update_ui (data);
176##endif
