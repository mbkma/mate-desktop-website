111static gboolean
112entry_has_text (CajaSearchBar *bar)
113{
114 const char *text;
115
116 text = gtk_entry_get_text (GTK_ENTRY (bar->details->entry));

153static void
154caja_search_bar_init (CajaSearchBar *bar)
155{
156 GtkWidget *hbox;
157 GtkWidget *label;
158 GtkStyleContext *context;
159
160 context = gtk_widget_get_style_context (GTK_WIDGET (bar));
161 gtk_style_context_add_class (context, "caja-search-bar");
162
163 bar->details = g_new0 (CajaSearchBarDetails, 1);
164
165 gtk_event_box_set_visible_window (GTK_EVENT_BOX (bar), FALSE);
166
167 hbox = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
168 gtk_widget_set_margin_start (hbox, 6);
169 gtk_widget_set_margin_end (hbox, 6);
170 gtk_widget_show (hbox);
171 gtk_container_add (GTK_CONTAINER (bar), hbox);
172
173 label = gtk_label_new (_("Search:"));
174 gtk_widget_show (label);
175
176 gtk_box_pack_start (GTK_BOX (hbox), label, FALSE, FALSE, 0);
177
178 bar->details->entry = gtk_entry_new ();
179 gtk_entry_set_icon_from_icon_name (GTK_ENTRY (bar->details->entry),
180 GTK_ENTRY_ICON_SECONDARY,
181 "find");
182 gtk_box_pack_start (GTK_BOX (hbox), bar->details->entry, TRUE, TRUE, 0);
183
184 g_signal_connect (bar->details->entry, "activate",
185 G_CALLBACK (entry_activate_cb), bar);
186 g_signal_connect (bar->details->entry, "icon-release",
187 G_CALLBACK (entry_icon_release_cb), bar);
188 g_signal_connect (bar->details->entry, "focus-in-event",
189 G_CALLBACK (focus_in_event_callback), bar);
190
191 gtk_widget_show (bar->details->entry);
192}

235CajaQuery *
236caja_search_bar_get_query (CajaSearchBar *bar)
237{
238 const char *query_text;
239 CajaQuery *query;
240
241 query_text = gtk_entry_get_text (GTK_ENTRY (bar->details->entry));
242
243 /* Empty string is a NULL query */
244 if (query_text && query_text[0] == '\0')
245 {
246 return NULL;
247 }
248
249 query = caja_query_new ();
250 caja_query_set_text (query, query_text);
251
252 return query;
253}
