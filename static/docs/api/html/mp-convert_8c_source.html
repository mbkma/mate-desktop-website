179static int
180ends_with(const char *start, const char *end, const char *word)
181{
182 size_t word_len = strlen(word);
183
184 if (word_len > end - start)
185 return 0;
186
187 return strncmp(end - word_len, word, word_len) == 0;
188}

220bool
221mp_set_from_string(const char *str, int default_base, MPNumber *z)
222{
223 int i, base, negate = 0, multiplier = 0, base_multiplier = 1;
224 const char *c, *end;
225 gboolean has_fraction = FALSE;
226
227 const char *base_digits[] = {"‚ÇÄ", "‚ÇÅ", "‚ÇÇ", "‚ÇÉ", "‚ÇÑ", "‚ÇÖ", "‚ÇÜ", "‚Çá", "‚Çà", "‚Çâ", NULL};

229 int numerators[] = { 1, 1, 2, 1, 3, 1, 2, 3, 4, 1, 5, 1, 3, 5, 7};
230 int denominators[] = { 2, 3, 3, 4, 4, 5, 5, 5, 5, 6, 6, 8, 8, 8, 8};
231
232 if (strstr(str, "¬∞"))
233 return set_from_sexagesimal(str, strlen(str), z);
234
235 /* Find the base */
236 end = str;
237 while (*end != '\0')
238 end++;
239 base = 0;
240 while (1) {
241 for (i = 0; base_digits[i] != NULL; i++) {
242 if (ends_with(str, end, base_digits[i])) {
243 base += i * base_multiplier;
244 end -= strlen(base_digits[i]);
245 base_multiplier *= 10;
246 break;
247 }
248 }
249 if (base_digits[i] == NULL)
250 break;
251 }
252 if (base_multiplier == 1)
253 base = default_base;
254
255 /* Check if this has a sign */
256 c = str;
257 if (*c == '+') {
258 c++;
259 } else if (*c == '-') {
260 negate = 1;
261 c++;
262 } else if (strncmp(c, "‚àí", strlen("‚àí")) == 0) {
263 negate = 1;
264 c += strlen("‚àí");
265 }
266
267 /* Convert integer part */
268 mp_set_from_integer(0, z);
269 while ((i = char_val((char **)&c, base)) >= 0) {
270 if (i > base)
271 return true;
272 mp_multiply_integer(z, base, z);
273 mp_add_integer(z, i, z);
274 }
275
276 /* Look for fraction characters, e.g. ‚Öö */
277 for (i = 0; fractions[i] != NULL; i++) {
278 if (ends_with(str, end, fractions[i])) {
279 end -= strlen(fractions[i]);
280 break;
281 }
282 }
283 if (fractions[i] != NULL) {
284 MPNumber fraction = mp_new();
285 mp_set_from_fraction(numerators[i], denominators[i], &fraction);
286 mp_add(z, &fraction, z);
287 mp_clear(&fraction);
288 }
289
290 if (*c == '.') {
291 has_fraction = TRUE;
292 c++;
293 }
294
295 /* Convert fractional part */
296 if (has_fraction) {
297 MPNumber numerator = mp_new();
298 MPNumber denominator = mp_new();
299
300 mp_set_from_integer(0, &numerator);
301 mp_set_from_integer(1, &denominator);
302 while ((i = char_val((char **)&c, base)) >= 0) {
303 mp_multiply_integer(&denominator, base, &denominator);
304 mp_multiply_integer(&numerator, base, &numerator);
305 mp_add_integer(&numerator, i, &numerator);
306 }
307 mp_divide(&numerator, &denominator, &numerator);
308 mp_add(z, &numerator, z);
309 mp_clear(&numerator);
310 mp_clear(&denominator);
311 }
312
313 if (c != end) {
314 return true;
315 }
316

318 MPNumber t = mp_new();
319 mp_set_from_integer(10, &t);
320 mp_xpowy_integer(&t, multiplier, &t);
321 mp_multiply(z, &t, z);
322 mp_clear(&t);
323 }
324
325 if (negate == 1)
326 mp_invert_sign(z, z);
327
328 return false;
