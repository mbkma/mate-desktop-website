49/* These macros assumes the following things:
50 * a METHOD_NAME is defined with is a string
51 * a goto label called beach
52 * the return value is called ret
53 */
54
55#define CHECK_METHOD_NAME(self) \
56 if (!PyObject_HasAttrString(self, METHOD_NAME)) \
57 goto beach;
58
59#define CHECK_OBJECT(object) \
60 if (object->instance == NULL) \
61 { \
62 g_object_unref (object); \
63 goto beach; \
64 } \
65
66#define CONVERT_LIST(py_files, files) \
67 { \
68 GList *l; \
69 py_files = PyList_New(0); \
70 for (l = files; l; l = l->next) \
71 { \
72 PyList_Append(py_files, pygobject_new((GObject*)l->data)); \
73 } \
74 }
75
76#define HANDLE_RETVAL(py_ret) \
77 if (!py_ret) \
78 { \
79 PyErr_Print(); \
80 goto beach; \
81 } \
82 else if (py_ret == Py_None) \
83 { \
84 goto beach; \
85 }
86
87#define HANDLE_LIST(py_ret, type, type_name) \
88 { \
89 Py_ssize_t i = 0; \
90 if (!PySequence_Check(py_ret) || PyUnicode_Check(py_ret)) \
91 { \
92 PyErr_SetString(PyExc_TypeError, \
93 METHOD_NAME " must return a sequence"); \
94 goto beach; \
95 } \
96 for (i = 0; i < PySequence_Size (py_ret); i++) \
97 { \
98 PyGObject *py_item; \
99 py_item = (PyGObject*)PySequence_GetItem (py_ret, i); \
100 if (!pygobject_check(py_item, &Py##type##_Type)) \
101 { \
102 PyErr_SetString(PyExc_TypeError, \
103 METHOD_NAME \
104 " must return a sequence of " \
105 type_name); \
106 goto beach; \
107 } \
108 ret = g_list_append (ret, (type*) g_object_ref(py_item->obj)); \
109 Py_DECREF(py_item); \
110 } \
111 }

113static void
114free_pygobject_data(gpointer data, gpointer user_data)
115{
116 /* Some CajaFile objects are cached and not freed until caja
117 itself is closed. Since PyGObject stores data that must be freed by
118 the Python interpreter, we must always free it before the interpreter
119 is finalized. */
120 g_object_set_data((GObject *)data, "PyGObject::instance-data", NULL);
121}

123static void
124free_pygobject_data_list(GList *list)
125{
126 if (list == NULL)
127 return;
128
129 g_list_foreach(list, (GFunc)free_pygobject_data, NULL);
130}

388#define METHOD_NAME "update_file_info"
389static CajaOperationResult
390caja_python_object_update_file_info (CajaInfoProvider *provider,
391 CajaFile *file,
392 GClosure *update_complete,
393 CajaOperationHandle **handle)
394{
395 CajaPythonObject *object = (CajaPythonObject*)provider;
396 CajaOperationResult ret = CAJA_OPERATION_COMPLETE;
397 PyObject *py_ret = NULL;
398 PyGILState_STATE state = pyg_gil_state_ensure();
399 static volatile gssize handle_generator = 1;
400
401 debug_enter();
402
403 CHECK_OBJECT(object);
404
405 *handle = NULL;
406
407 if (PyObject_HasAttrString(object->instance, "update_file_info_full"))
408 {
409 PyObject *py_handle;
410 void *h;
411
412 /* Generate a new handle with a default value. */
413 do {
414 h = (CajaOperationHandle *) g_atomic_pointer_add (&handle_generator, 1);
415 } while (!h);
416 py_handle = caja_python_boxed_new (_PyCajaOperationHandle_Type,
417 h, FALSE);
418
419 py_ret = PyObject_CallMethod(object->instance,
420 METHOD_PREFIX "update_file_info_full", "(NNNN)",
421 pygobject_new((GObject*)provider),
422 py_handle,
423 pyg_boxed_new(G_TYPE_CLOSURE, update_complete, TRUE, TRUE),
424 pygobject_new((GObject*)file));
425 *handle = (void *) ((PyGBoxed *) py_handle)->boxed;
426 }
427 else if (PyObject_HasAttrString(object->instance, "update_file_info"))
428 {
429 py_ret = PyObject_CallMethod(object->instance,
430 METHOD_PREFIX METHOD_NAME, "(N)",
431 pygobject_new((GObject*)file));
432 }
433 else
434 {
435 goto beach;
436 }
437
438 HANDLE_RETVAL(py_ret);
439
440 if (!PyLong_Check(py_ret))
441 {
442 PyErr_SetString(PyExc_TypeError,
443 METHOD_NAME " must return None or a int");
444 goto beach;
445 }
446
447 ret = PyLong_AsLong(py_ret);
448
449 if (!*handle && ret == CAJA_OPERATION_IN_PROGRESS)
450 ret = CAJA_OPERATION_FAILED;
451
452 beach:
453 free_pygobject_data(file, NULL);
454 Py_XDECREF(py_ret);
455 pyg_gil_state_release(state);
456 return ret;
457}
458#undef METHOD_NAME
