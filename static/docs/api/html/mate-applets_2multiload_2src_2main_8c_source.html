94static void
95start_procman (MultiloadApplet *ma)
96{
97 GError *error = NULL;
98 GDesktopAppInfo *appinfo;
99 gchar *monitor;
100 GdkAppLaunchContext *launch_context;
101 GdkDisplay *display;
102 GAppInfo *app_info;
103 GdkScreen *screen;
104
105 g_return_if_fail (ma != NULL);
106
107 monitor = g_settings_get_string (ma->settings, "system-monitor");
108 if (monitor == NULL)
109 monitor = g_strdup ("mate-system-monitor.desktop");
110
111 screen = gtk_widget_get_screen (GTK_WIDGET (ma->applet));
112 appinfo = g_desktop_app_info_new (monitor);
113 if (appinfo) {
114 GdkAppLaunchContext *context;
115 display = gdk_screen_get_display (screen);
116 context = gdk_display_get_app_launch_context (display);
117 gdk_app_launch_context_set_screen (context, screen);
118 g_app_info_launch (G_APP_INFO (appinfo), NULL, G_APP_LAUNCH_CONTEXT (context), &error);
119 g_object_unref (context);
120 g_object_unref (appinfo);
121 }
122 else {
123 app_info = g_app_info_create_from_commandline ("mate-system-monitor",
124 _("Start system-monitor"),
125 G_APP_INFO_CREATE_NONE,
126 &error);
127
128 if (!error) {
129 display = gdk_screen_get_display (screen);
130 launch_context = gdk_display_get_app_launch_context (display);
131 gdk_app_launch_context_set_screen (launch_context, screen);
132 g_app_info_launch (app_info, NULL, G_APP_LAUNCH_CONTEXT (launch_context), &error);
133
134 g_object_unref (launch_context);
135 }
136 }
137 g_free (monitor);
138
139 if (error) {
140 GtkWidget *dialog;
141
142 dialog = gtk_message_dialog_new (NULL,
143 GTK_DIALOG_DESTROY_WITH_PARENT,
144 GTK_MESSAGE_ERROR,
145 GTK_BUTTONS_OK,
146 _("There was an error executing '%s': %s"),
147 "mate-system-monitor",
148 error->message);
149
150 g_signal_connect (dialog, "response",
151 G_CALLBACK (gtk_widget_destroy),
152 NULL);
153
154 gtk_window_set_resizable (GTK_WINDOW (dialog), FALSE);
155 gtk_window_set_screen (GTK_WINDOW (dialog), screen);
156
157 gtk_widget_show (dialog);
158
159 g_error_free (error);
160 }
161}

264/* update the tooltip to the graph's current "used" percentage */
265void
266multiload_applet_tooltip_update(LoadGraph *g)
267{
268 gchar *tooltip_text;
269 MultiloadApplet *multiload;
270 const char *tooltip_label [graph_n] = {
271 [graph_cpuload] = N_("Processor"),
272 [graph_memload] = N_("Memory"),
273 [graph_netload2] = N_("Network"),
274 [graph_swapload] = N_("Swap Space"),
275 [graph_loadavg] = N_("Load Average"),
276 [graph_diskload] = N_("Disk")
277 };
278 const char *name;
279
280 g_assert(g);
281
282 multiload = g->multiload;
283
284 /* label the tooltip intuitively */
285 name = gettext (tooltip_label [g->id]);
286
287 switch (g->id) {
288 case graph_memload: {
289 float user_percent, cache_percent;
290
291 user_percent = MIN ((float)(100 * multiload->memload_user) / (float)(multiload->memload_total), 100.0f);
292 cache_percent = MIN ((float)(100 * multiload->memload_cache) / (float)(multiload->memload_total), 100.0f);
293 tooltip_text = g_strdup_printf (_("%s:\n"
294 "%.01f%% in use by programs\n"
295 "%.01f%% in use as cache"),
296 name,
297 user_percent,
298 cache_percent);
299 break;
300 }
301 case graph_loadavg: {
302 tooltip_text = g_strdup_printf (_("The system load average is %0.02f"),
303 multiload->loadavg1);
304 break;
305 }
306 case graph_netload2: {
307 char *tx_in, *tx_out;
308
309 tx_in = netspeed_get(multiload->netspeed_in);
310 tx_out = netspeed_get(multiload->netspeed_out);
311 /* xgettext: same as in graphic tab of g-s-m */
312 tooltip_text = g_strdup_printf(_("%s:\n"
313 "Receiving %s\n"
314 "Sending %s"),
315 name, tx_in, tx_out);
316 g_free(tx_in);
317 g_free(tx_out);
318 break;
319 }
320 default: {
321 float ratio;
322 float percent;
323
324 if (g->id == graph_cpuload)
325 ratio = multiload->cpu_used_ratio;
326 else if (g->id == graph_swapload)
327 ratio = multiload->swapload_used_ratio;
328 else if (g->id == graph_diskload)
329 ratio = multiload->diskload_used_ratio;
330 else
331 g_assert_not_reached ();
332
333 percent = CLAMP (ratio * 100.0f, 0.0f, 100.0f);
334 tooltip_text = g_strdup_printf(_("%s:\n"
335 "%.01f%% in use"),
336 name,
337 percent);
338 }
339 }
340
341 gtk_widget_set_tooltip_text(g->disp, tooltip_text);
342
343 g_free(tooltip_text);
344}

346static void
347multiload_create_graphs(MultiloadApplet *ma)
348{
349 struct { const char *label;
350 const char *visibility_key;
351 const char *name;
352 guint num_colours;
353 LoadGraphDataFunc callback;
354 } graph_types [graph_n] = {
355 [graph_cpuload] = { _("CPU Load"), VIEW_CPULOAD_KEY, "cpuload", cpuload_n, GetLoad },
356 [graph_memload] = { _("Memory Load"), VIEW_MEMLOAD_KEY, "memload", memload_n, GetMemory },
357 [graph_netload2] = { _("Net Load"), VIEW_NETLOAD_KEY, "netload2", 6, GetNet },
358 [graph_swapload] = { _("Swap Load"), VIEW_SWAPLOAD_KEY, "swapload", swapload_n, GetSwap },
359 [graph_loadavg] = { _("Load Average"), VIEW_LOADAVG_KEY, "loadavg", 3, GetLoadAvg },
360 [graph_diskload] = { _("Disk Load"), VIEW_DISKLOAD_KEY, "diskload", diskload_n, GetDiskLoad }
361 };
362
363 guint size;
364 guint speed;
365 guint64 net_threshold1;
366 guint64 net_threshold2;
367 guint64 net_threshold3;
368 gint i;
369
370 speed = CLAMP (g_settings_get_uint (ma->settings, REFRESH_RATE_KEY), REFRESH_RATE_MIN, REFRESH_RATE_MAX);
371 size = CLAMP (g_settings_get_uint (ma->settings, GRAPH_SIZE_KEY), GRAPH_SIZE_MIN, GRAPH_SIZE_MAX);
372 net_threshold1 = CLAMP (g_settings_get_uint64 (ma->settings, KEY_NET_THRESHOLD1), MIN_NET_THRESHOLD1, MAX_NET_THRESHOLD1);
373 net_threshold2 = CLAMP (g_settings_get_uint64 (ma->settings, KEY_NET_THRESHOLD2), MIN_NET_THRESHOLD2, MAX_NET_THRESHOLD2);
374 net_threshold3 = CLAMP (g_settings_get_uint64 (ma->settings, KEY_NET_THRESHOLD3), MIN_NET_THRESHOLD3, MAX_NET_THRESHOLD3);
375 if (net_threshold1 >= net_threshold2)
376 {
377 net_threshold1 = net_threshold2 - 1;
378 }
379 if (net_threshold2 >= net_threshold3)
380 {
381 net_threshold3 = net_threshold2 + 1;
382 }
383
384 for (i = 0; i < graph_n; i++)
385 {
386 ma->graphs[i] = load_graph_new (ma,
387 graph_types[i].num_colours,
388 graph_types[i].label,
389 i,
390 speed,
391 size,
392 g_settings_get_boolean (ma->settings, graph_types[i].visibility_key),
393 graph_types[i].name,
394 graph_types[i].callback);
395 }
396 ma->nvme_diskstats = g_settings_get_boolean (ma->settings, "diskload-nvme-diskstats");
397 /* for Network graph, colors[4] is grid line color, it should not be used in loop in load-graph.c */
398 /* for Network graph, colors[5] is indicator color, it should not be used in loop in load-graph.c */
399 ma->graphs[graph_netload2]->n = 4;
400 ma->net_threshold1 = net_threshold1;
401 ma->net_threshold2 = net_threshold2;
402 ma->net_threshold3 = net_threshold3;
403 ma->netspeed_in = netspeed_new (ma->graphs [graph_netload2]);
404 ma->netspeed_out = netspeed_new (ma->graphs [graph_netload2]);
405 /* for Load graph, colors[2] is grid line color, it should not be used in loop in load-graph.c */
406 ma->graphs[graph_loadavg]->n = 2;
407}

409/* remove the old graphs and rebuild them */
410void
411multiload_applet_refresh(MultiloadApplet *ma)
412{
413 guint i;
414 MatePanelAppletOrient orientation;
415
416 /* stop and free the old graphs */
417 for (i = 0; i < graph_n; i++)
418 {
419 if (!ma->graphs[i])
420 continue;
421
422 load_graph_stop(ma->graphs[i]);
423 gtk_widget_destroy(ma->graphs[i]->main_widget);
424
425 load_graph_unalloc(ma->graphs[i]);
426 g_free(ma->graphs[i]);
427 }
428
429 if (ma->box)
430 gtk_widget_destroy(ma->box);
431
432 orientation = mate_panel_applet_get_orient(ma->applet);
433
434 if ( (orientation == MATE_PANEL_APPLET_ORIENT_UP) ||
435 (orientation == MATE_PANEL_APPLET_ORIENT_DOWN) ) {
436 ma->box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);
437 }
438 else
439 ma->box = gtk_box_new (GTK_ORIENTATION_VERTICAL, 0);
440
441 gtk_container_add(GTK_CONTAINER(ma->applet), ma->box);
442
443 /* create the N graphs, passing in their user-configurable properties with gsettings. */
444 multiload_create_graphs (ma);
445
446 /* only start and display the graphs the user has turned on */
447
448 for (i = 0; i < graph_n; i++) {
449 gtk_box_pack_start(GTK_BOX(ma->box),
450 ma->graphs[i]->main_widget,
451 TRUE, TRUE, 1);
452 if (ma->graphs[i]->visible) {
453 gtk_widget_show_all (ma->graphs[i]->main_widget);
454 load_graph_start(ma->graphs[i]);
455 }
456 }
457 gtk_widget_show (ma->box);
458
459 return;
460}

478static gboolean
479multiload_applet_new(MatePanelApplet *applet, const gchar *iid, gpointer data)
480{
481 GtkStyleContext *context;
482 MultiloadApplet *ma;
483 GSettings *lockdown_settings;
484 GtkActionGroup *action_group;
485 AtkObject *atk_obj;
486
487 context = gtk_widget_get_style_context (GTK_WIDGET (applet));
488 gtk_style_context_add_class (context, "multiload-applet");
489
490 ma = g_new0(MultiloadApplet, 1);
491
492 ma->applet = applet;
493
494 ma->about_dialog = NULL;
495 ma->prop_dialog = NULL;
496 ma->last_clicked = 0;
497
498#ifndef ENABLE_IN_PROCESS
499 g_set_application_name (_("System Monitor"));
500#endif
501
502 gtk_window_set_default_icon_name ("utilities-system-monitor");
503
504 ma->settings = mate_panel_applet_settings_new (applet, "org.mate.panel.applet.multiload");
505 mate_panel_applet_set_flags (applet, MATE_PANEL_APPLET_EXPAND_MINOR);
506
507 action_group = gtk_action_group_new ("Multiload Applet Actions");
508 gtk_action_group_set_translation_domain (action_group, GETTEXT_PACKAGE);
509 gtk_action_group_add_actions (action_group,
510 multiload_menu_actions,
511 G_N_ELEMENTS (multiload_menu_actions),
512 ma);
513 mate_panel_applet_setup_menu_from_resource (applet,
514 MULTILOAD_RESOURCE_PATH "multiload-applet-menu.xml",
515 action_group);
516
517 if (mate_panel_applet_get_locked_down (applet)) {
518 GtkAction *action;
519
520 action = gtk_action_group_get_action (action_group, "MultiLoadProperties");
521 gtk_action_set_visible (action, FALSE);
522 }
523
524 lockdown_settings = g_settings_new ("org.mate.lockdown");
525 if (g_settings_get_boolean (lockdown_settings, "disable-command-line") ||
526 mate_panel_applet_get_locked_down (applet)) {
527 GtkAction *action;
528
529 /* When the panel is locked down or when the command line is inhibited,
530 it seems very likely that running the procman would be at least harmful */
531 action = gtk_action_group_get_action (action_group, "MultiLoadRunProcman");
532 gtk_action_set_visible (action, FALSE);
533 }
534 g_object_unref (lockdown_settings);
535
536 g_object_unref (action_group);
537
538 g_signal_connect (applet, "change-size",
539 G_CALLBACK (multiload_change_size_cb), ma);
540 g_signal_connect (applet, "change-orient",
541 G_CALLBACK (multiload_change_orient_cb), ma);
542 g_signal_connect (applet, "destroy",
543 G_CALLBACK (multiload_destroy_cb), ma);
544 g_signal_connect (applet, "button-press-event",
545 G_CALLBACK (multiload_button_press_event_cb), ma);
546 g_signal_connect (applet, "key-press-event",
547 G_CALLBACK (multiload_key_press_event_cb), ma);
548
549 atk_obj = gtk_widget_get_accessible (GTK_WIDGET (applet));
550
551 if (GTK_IS_ACCESSIBLE (atk_obj)) {
552 atk_object_set_name (atk_obj, _("System Monitor"));
553 atk_object_set_description (atk_obj,
554 _("A system load monitor capable of displaying graphs "
555 "for CPU, ram, and swap space use, plus network "
556 "traffic."));
557 }
558
559 multiload_applet_refresh (ma);
560
561 gtk_widget_show(GTK_WIDGET(applet));
562
563 return TRUE;
564}
