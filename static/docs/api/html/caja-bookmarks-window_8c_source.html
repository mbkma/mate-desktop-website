122/* We store a pointer to the bookmark in a column so when an item is moved
123 with DnD we know which item it is. However we have to be careful to keep

143static void
144caja_bookmarks_window_response_callback (GtkDialog *dialog,
145 int response_id,
146 gpointer callback_data)
147{
148 if (response_id == GTK_RESPONSE_HELP)
149 {
150 GError *error = NULL;
151
152 gtk_show_uri_on_window (GTK_WINDOW (dialog),
153 "help:mate-user-guide/goscaja-36",
154 gtk_get_current_event_time (), &error);
155
156 if (error)
157 {
158 GtkWidget *err_dialog;
159 err_dialog = gtk_message_dialog_new (GTK_WINDOW (dialog),
160 GTK_DIALOG_DESTROY_WITH_PARENT,
161 GTK_MESSAGE_ERROR,
162 GTK_BUTTONS_OK,
163 _("There was an error displaying help: \n%s"),
164 error->message);
165
166 g_signal_connect (G_OBJECT (err_dialog),
167 "response", G_CALLBACK (gtk_widget_destroy),
168 NULL);
169 gtk_window_set_resizable (GTK_WINDOW (err_dialog), FALSE);
170 gtk_widget_show (err_dialog);
171 g_error_free (error);
172 }
173 }
174 else if (response_id == GTK_RESPONSE_CLOSE)
175 {
176 gtk_widget_hide (GTK_WIDGET (dialog));
177 }
178}

259{

261 GtkTreeViewColumn *col;
262 GtkCellRenderer *rend;
263 GtkBuilder *builder;
264
265 bookmarks = list;
266
267 builder = gtk_builder_new_from_resource ("/org/mate/caja/caja-bookmarks-window.ui");

269 window = (GtkWidget *)gtk_builder_get_object (builder, "bookmarks_dialog");
270 bookmark_list_widget = (GtkTreeView *)gtk_builder_get_object (builder, "bookmark_tree_view");
271 remove_button = (GtkWidget *)gtk_builder_get_object (builder, "bookmark_delete_button");
272 jump_button = (GtkWidget *)gtk_builder_get_object (builder, "bookmark_jump_button");
273
274 application = window_source->application;
275
276 if (CAJA_IS_NAVIGATION_WINDOW (window_source))
277 {
278 parent_is_browser_window = TRUE;
279 }
280 else
281 {
282 parent_is_browser_window = FALSE;
283 }
284
285 set_up_close_accelerator (window);
286
287 caja_bookmarks_window_restore_geometry (window);
288
289 g_object_weak_ref (G_OBJECT (window_source), edit_bookmarks_dialog_reset_signals,
290 window_source);
291
292 bookmark_list_widget = GTK_TREE_VIEW (gtk_builder_get_object (builder, "bookmark_tree_view"));
293
294 rend = gtk_cell_renderer_pixbuf_new ();
295 col = gtk_tree_view_column_new_with_attributes ("Icon",
296 rend,
297 "surface",
298 BOOKMARK_LIST_COLUMN_ICON,
299 NULL);
300 gtk_tree_view_append_column (bookmark_list_widget,
301 GTK_TREE_VIEW_COLUMN (col));
302 gtk_tree_view_column_set_fixed_width (GTK_TREE_VIEW_COLUMN (col),

304

307 "ellipsize", PANGO_ELLIPSIZE_END,
308 "ellipsize-set", TRUE,
309 NULL);
310
311 col = gtk_tree_view_column_new_with_attributes ("Icon",
312 rend,
313 "text",
314 BOOKMARK_LIST_COLUMN_NAME,
315 "style",
316 BOOKMARK_LIST_COLUMN_STYLE,
317 NULL);

432static CajaBookmark *
433get_selected_bookmark (void)
434{
435 g_return_val_if_fail(CAJA_IS_BOOKMARK_LIST(bookmarks), NULL);
436
437 if (!get_selection_exists())
438 return NULL;
439
440 if (caja_bookmark_list_length (bookmarks) < 1)
441 return NULL;
442
443 return caja_bookmark_list_item_at(bookmarks, get_selected_row ());
444}

446static guint
447get_selected_row (void)
448{
449 GtkTreeIter iter;
450 GtkTreePath *path;
451 GtkTreeModel *model;
452 gint *indices, row;
453
454 g_assert (get_selection_exists());
455
456 model = GTK_TREE_MODEL (bookmark_list_store);
457 gtk_tree_selection_get_selected (bookmark_selection,
458 &model,
459 &iter);
460
461 path = gtk_tree_model_get_path (model, &iter);
462 indices = gtk_tree_path_get_indices (path);
463 row = indices[0];
464 gtk_tree_path_free (path);
465 return row;
466}

474static void
475caja_bookmarks_window_restore_geometry (GtkWidget *window)
476{
477 const char *window_geometry;
478
479 g_return_if_fail (GTK_IS_WINDOW (window));
480 g_return_if_fail (CAJA_IS_BOOKMARK_LIST (bookmarks));
481
482 window_geometry = caja_bookmark_list_get_window_geometry (bookmarks);
483
484 if (window_geometry != NULL)
485 {
486 eel_gtk_window_set_initial_geometry_from_string
487 (GTK_WINDOW (window), window_geometry,
488 BOOKMARKS_WINDOW_MIN_WIDTH, BOOKMARKS_WINDOW_MIN_HEIGHT, FALSE);
489
490 }
491 else
492 {
493 /* use default since there was no stored geometry */
494 gtk_window_set_default_size (GTK_WINDOW (window),
495 BOOKMARKS_WINDOW_INITIAL_WIDTH,
496 BOOKMARKS_WINDOW_INITIAL_HEIGHT);
497
498 /* Let window manager handle default position if no position stored */
499 }
500}

561static void
562open_selected_bookmark (gpointer user_data, GdkScreen *screen)
563{
564 CajaBookmark *selected;
565 CajaWindow *window;
566 GFile *location;
567
568 selected = get_selected_bookmark ();
569
570 if (!selected)
571 {
572 return;
573 }
574
575 location = caja_bookmark_get_location (selected);
576 if (location == NULL)
577 {
578 return;
579 }
580
581 if (CAJA_IS_NAVIGATION_WINDOW (user_data))
582 {
583 window = user_data;
584 }
585 else if (CAJA_IS_SPATIAL_WINDOW (user_data))
586 {
587 window = caja_application_get_spatial_window (application,
588 NULL,
589 NULL,
590 location,
591 screen,
592 NULL);
593 } else { /* window that opened bookmarks window has been closed */
594 if (parent_is_browser_window || g_settings_get_boolean (caja_preferences, CAJA_PREFERENCES_ALWAYS_USE_BROWSER)) {
595 window = caja_application_create_navigation_window (application,
596 screen);
597 }
598 else
599 {
600 window = caja_application_get_spatial_window (application,
601 NULL,
602 NULL,
603 location,
604 screen,
605 NULL);
606 }
607 }
608
609 caja_window_go_to (window, location);
610
611 g_object_unref (location);
612}

624static void
625bookmarks_delete_bookmark (void)
626{
627 GtkTreeIter iter;
628 GtkTreePath *path;
629 gint *indices, row, rows;
630
631 g_assert (GTK_IS_TREE_VIEW (bookmark_list_widget));
632
633 if (!gtk_tree_selection_get_selected (bookmark_selection, NULL, &iter))
634 return;
635
636 /* Remove the selected item from the list store. on_row_deleted() will
637 remove it from the bookmark list. */
638 path = gtk_tree_model_get_path (GTK_TREE_MODEL (bookmark_list_store),
639 &iter);
640 indices = gtk_tree_path_get_indices (path);
641 row = indices[0];
642 gtk_tree_path_free (path);
643
644 gtk_list_store_remove (bookmark_list_store, &iter);
645
646 /* Try to select the same row, or the last one in the list. */
647 rows = gtk_tree_model_iter_n_children (GTK_TREE_MODEL (bookmark_list_store), NULL);
648 if (row >= rows)
649 row = rows - 1;
650
651 if (row < 0)
652 {
653 bookmarks_set_empty (TRUE);
654 }
655 else
656 {
657 gtk_tree_model_iter_nth_child (GTK_TREE_MODEL (bookmark_list_store),
658 &iter, NULL, row);
659 gtk_tree_selection_select_iter (bookmark_selection, &iter);
660 }
661}

670/* This is a bit of a kludge to get DnD to work. We check if the row in the
671 GtkListStore matches the one in the bookmark list. If it doesn't, we assume
672 the bookmark has just been dragged here and we insert it into the bookmark
673 list. */
674static void
675on_row_changed (GtkListStore *store,
676 GtkTreePath *path,
677 GtkTreeIter *iter,
678 gpointer user_data)
679{
680 CajaBookmark *bookmark = NULL;
681 gint *indices, row;
682 gboolean insert_bookmark = TRUE;
683
684 store = bookmark_list_store;
685
686 indices = gtk_tree_path_get_indices (path);
687 row = indices[0];
688 gtk_tree_model_get (GTK_TREE_MODEL (store), iter,
689 BOOKMARK_LIST_COLUMN_BOOKMARK, &bookmark,
690 -1);
691
692 /* If the bookmark in the list doesn't match the changed one, it must
693 have been dragged here, so we insert it into the list. */
694 if (row < (gint) caja_bookmark_list_length (bookmarks))
695 {
696 CajaBookmark *bookmark_in_list;
697

835static void
836update_bookmark_from_text (void)
837{
838 if (text_changed)
839 {
840 CajaBookmark *bookmark, *bookmark_in_list;
841 char *name;
842 cairo_surface_t *surface;
843 guint selected_row;
844 GtkTreeIter iter;
845 GFile *location;
846
847 g_assert (GTK_IS_ENTRY (name_field));
848 g_assert (GTK_IS_ENTRY (uri_field));
849
850 if (gtk_entry_get_text_length (GTK_ENTRY (uri_field)) == 0)
851 {
852 return;
853 }
854
855 location = g_file_parse_name
856 (gtk_entry_get_text (GTK_ENTRY (uri_field)));
857
858 bookmark = caja_bookmark_new (location, gtk_entry_get_text (GTK_ENTRY (name_field)),
859 name_text_changed, NULL);
860
861 g_object_unref (location);
862
863 selected_row = get_selected_row ();
864
865 /* turn off list updating 'cuz otherwise the list-reordering code runs
866 * after repopulate(), thus reordering the correctly-ordered list.
867 */
868 g_signal_handler_block (bookmarks,
869 bookmark_list_changed_signal_id);
870 caja_bookmark_list_delete_item_at (bookmarks, selected_row);
871 caja_bookmark_list_insert_item (bookmarks, bookmark, selected_row);
872 g_signal_handler_unblock (bookmarks,
873 bookmark_list_changed_signal_id);
874 g_object_unref (bookmark);
875
876 /* We also have to update the bookmark pointer in the list
877 store. */
878 gtk_tree_selection_get_selected (bookmark_selection,
879 NULL, &iter);
880 g_signal_handler_block (bookmark_list_store,
881 row_changed_signal_id);
882
883 bookmark_in_list = caja_bookmark_list_item_at (bookmarks,
884 selected_row);
885
886 name = caja_bookmark_get_name (bookmark_in_list);
887
888 surface = caja_bookmark_get_surface (bookmark_in_list, GTK_ICON_SIZE_MENU);
889
890 gtk_list_store_set (bookmark_list_store, &iter,
891 BOOKMARK_LIST_COLUMN_BOOKMARK, bookmark_in_list,
892 BOOKMARK_LIST_COLUMN_NAME, name,
893 BOOKMARK_LIST_COLUMN_ICON, surface,
894 -1);
895 g_signal_handler_unblock (bookmark_list_store,
896 row_changed_signal_id);
897
898 cairo_surface_destroy (surface);
899 g_free (name);
900 }

972static void
973repopulate (void)
974{
975 CajaBookmark *selected;
976 GtkListStore *store;
977 GtkTreeRowReference *reference;
978 guint index;
979 GtkTreePath *path = NULL;
980
981 g_assert (GTK_IS_TREE_VIEW (bookmark_list_widget));
982 g_assert (CAJA_IS_BOOKMARK_LIST (bookmarks));
983
984 store = GTK_LIST_STORE (bookmark_list_store);
985
986 selected = get_selected_bookmark ();
987
988 g_signal_handler_block (bookmark_selection,
989 selection_changed_id);
990 g_signal_handler_block (bookmark_list_store,
991 row_deleted_signal_id);
992 g_signal_handler_block (bookmark_list_widget,
993 row_activated_signal_id);
994 g_signal_handler_block (bookmark_list_widget,
995 key_pressed_signal_id);
996 g_signal_handler_block (bookmark_list_widget,
997 button_pressed_signal_id);
998
999 gtk_list_store_clear (store);
1000
1001 g_signal_handler_unblock (bookmark_list_widget,
1002 row_activated_signal_id);
1003 g_signal_handler_unblock (bookmark_list_widget,
1004 key_pressed_signal_id);
1005 g_signal_handler_unblock (bookmark_list_widget,
1006 button_pressed_signal_id);
1007 g_signal_handler_unblock (bookmark_list_store,
1008 row_deleted_signal_id);
1009 g_signal_handler_unblock (bookmark_selection,
1010 selection_changed_id);
1011
1012 /* Fill the list in with the bookmark names. */
1013 g_signal_handler_block (store, row_changed_signal_id);
1014
1015 reference = NULL;
1016
1017 for (index = 0; index < caja_bookmark_list_length (bookmarks); ++index)
1018 {
1019 CajaBookmark *bookmark;
1020 char *bookmark_name;
1021 cairo_surface_t *bookmark_surface;
1022 GtkTreeIter iter;
1023
1024 bookmark = caja_bookmark_list_item_at (bookmarks, index);
1025 bookmark_name = caja_bookmark_get_name (bookmark);
1026 bookmark_surface = caja_bookmark_get_surface (bookmark, GTK_ICON_SIZE_MENU);
1027
1028 gtk_list_store_append (store, &iter);
1029 gtk_list_store_set (store, &iter,
1030 BOOKMARK_LIST_COLUMN_ICON, bookmark_surface,
1031 BOOKMARK_LIST_COLUMN_NAME, bookmark_name,
1032 BOOKMARK_LIST_COLUMN_BOOKMARK, bookmark,
1033 BOOKMARK_LIST_COLUMN_STYLE, PANGO_STYLE_NORMAL,
1034 -1);
1035
1036 if (bookmark == selected)
1037 {
1038 /* save old selection */
1039 GtkTreePath *path;
1040
1041 path = gtk_tree_model_get_path (GTK_TREE_MODEL (store), &iter);
1042 reference = gtk_tree_row_reference_new (GTK_TREE_MODEL (store), path);
1043 gtk_tree_path_free (path);
1044 }
1045
1046 g_free (bookmark_name);
1047 cairo_surface_destroy (bookmark_surface);
1048
1049 }
1050 g_signal_handler_unblock (store, row_changed_signal_id);
1051
1052 if (reference != NULL)
1053 {
1054 /* restore old selection */
1055
1056 /* bookmarks_set_empty() will call the selection change handler,
1057 * so we block it here in case of selection change.
1058 */
1059 g_signal_handler_block (bookmark_selection, selection_changed_id);
1060
1061 g_assert (index != 0);
1062 g_assert (gtk_tree_row_reference_valid (reference));
1063
1064 path = gtk_tree_row_reference_get_path (reference);
1065 gtk_tree_selection_select_path (bookmark_selection, path);
1066 gtk_tree_row_reference_free (reference);
1067 gtk_tree_path_free (path);
1068
1069 g_signal_handler_unblock (bookmark_selection, selection_changed_id);
1070 }
1071
1072 bookmarks_set_empty (index == 0);
1073}
