156 manager->priv->categories = g_list_append(manager->priv->categories, category);

223gboolean
224unit_manager_convert_by_symbol(UnitManager *manager, const MPNumber *x, const char *x_symbol, const char *z_symbol, MPNumber *z)
225{
226 GList *iter;
227
228 g_return_val_if_fail(manager != NULL, FALSE);
229 g_return_val_if_fail(x != NULL, FALSE);
230 g_return_val_if_fail(x_symbol != NULL, FALSE);
231 g_return_val_if_fail(z_symbol != NULL, FALSE);
232 g_return_val_if_fail(z != NULL, FALSE);
233
234 for (iter = manager->priv->categories; iter; iter = iter->next) {
235 UnitCategory *c = iter->data;
236 Unit *x_units, *z_units;
237
238 x_units = unit_category_get_unit_by_symbol(c, x_symbol);
239 z_units = unit_category_get_unit_by_symbol(c, z_symbol);
240 if (x_units && z_units && unit_category_convert(c, x, x_units, z_units, z))
241 return TRUE;
242 }
243
244 return FALSE;
245}
