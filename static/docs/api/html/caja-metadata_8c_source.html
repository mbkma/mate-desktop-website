67guint
68caja_metadata_get_id (const char *metadata)
69{
70 static GHashTable *hash;
71
72 if (hash == NULL)
73 {
74 int i;
75
76 hash = g_hash_table_new (g_str_hash, g_str_equal);
77 for (i = 0; used_metadata_names[i] != NULL; i++)
78 g_hash_table_insert (hash,
79 used_metadata_names[i],
80 GINT_TO_POINTER (i + 1));
81 }
82
83 return GPOINTER_TO_INT (g_hash_table_lookup (hash, metadata));
84}
