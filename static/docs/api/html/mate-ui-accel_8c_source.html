136void
137mate_ui_accel_map_apply_to_app(MateUiAccelMap *map,
138 GtkApplication *app)
139{
140 g_return_if_fail(map != NULL);
141 g_return_if_fail(GTK_IS_APPLICATION(app));
142
143 GHashTableIter iter;
144 gpointer key, value;
145
146 g_hash_table_iter_init(&iter, map->accels);
147 while (g_hash_table_iter_next(&iter, &key, &value))
148 {
149 const gchar *action_name = key;
150 const gchar *accel = value;
151 const gchar *accels[] = { accel, NULL };
152
153 gtk_application_set_accels_for_action(app, action_name, accels);
154 }
155}

168gboolean
169mate_ui_accel_map_load(MateUiAccelMap *map,
170 const gchar *filename,
171 GError **error)
172{
173 g_return_val_if_fail(map != NULL, FALSE);
174 g_return_val_if_fail(filename != NULL, FALSE);
175
176 gchar *contents = NULL;
177 gsize length;
178
179 if (!g_file_get_contents(filename, &contents, &length, error))
180 return FALSE;
181
182 gchar **lines = g_strsplit(contents, "\n", -1);
183 g_free(contents);
184
185 for (gint i = 0; lines[i] != NULL; i++)
186 {
187 gchar *line = g_strstrip(lines[i]);
188
189 /* Skip empty lines and comments */
190 if (*line == '\0' || *line == '#')
191 continue;
192
193 gchar **parts = g_strsplit(line, "=", 2);
194 if (parts[0] != NULL && parts[1] != NULL)
195 {
196 gchar *action = g_strstrip(parts[0]);
197 gchar *accel = g_strstrip(parts[1]);
198
199 if (*action != '\0' && *accel != '\0')
200 {
201 mate_ui_accel_map_add(map, action, accel);
202 }
203 }
204 g_strfreev(parts);
205 }
206
207 g_strfreev(lines);
208 return TRUE;
209}

221gboolean
222mate_ui_accel_map_save(MateUiAccelMap *map,
223 const gchar *filename,
224 GError **error)
225{
226 g_return_val_if_fail(map != NULL, FALSE);
227 g_return_val_if_fail(filename != NULL, FALSE);
228
229 GString *content = g_string_new("# MATE UI Accelerator Map\n");
230 g_string_append(content, "# Format: action_name=accelerator\n\n");
231
232 GHashTableIter iter;
233 gpointer key, value;
234
235 g_hash_table_iter_init(&iter, map->accels);
236 while (g_hash_table_iter_next(&iter, &key, &value))
237 {
238 g_string_append_printf(content, "%s=%s\n", (gchar *)key, (gchar *)value);
239 }
240
241 gboolean result = g_file_set_contents(filename, content->str, content->len, error);
242 g_string_free(content, TRUE);
243
244 return result;
245}

297gboolean
298mate_ui_accel_group_add(GtkAccelGroup *accel_group,
299 const gchar *accel,
300 GCallback callback,
301 gpointer user_data)
302{
303 g_return_val_if_fail(GTK_IS_ACCEL_GROUP(accel_group), FALSE);
304 g_return_val_if_fail(accel != NULL, FALSE);
305
306 guint key;
307 GdkModifierType mods;
308 gtk_accelerator_parse(accel, &key, &mods);
309
310 if (key == 0)
311 return FALSE;
312
313 AccelCallbackData *data = g_new0(AccelCallbackData, 1);
314 data->callback = callback;
315 data->user_data = user_data;
316
317 GClosure *closure = g_cclosure_new(G_CALLBACK(accel_callback_wrapper),
318 data, accel_callback_data_free);
319
320 gtk_accel_group_connect(accel_group, key, mods, GTK_ACCEL_VISIBLE, closure);
321
322 return TRUE;
323}

465void
466mate_ui_accel_connect_to_widget(GtkWidget *widget,
467 const gchar *accel,
468 const gchar *signal_name)
469{
470 g_return_if_fail(GTK_IS_WIDGET(widget));
471 g_return_if_fail(accel != NULL);
472 g_return_if_fail(signal_name != NULL);
473
474 guint key;
475 GdkModifierType mods;
476 gtk_accelerator_parse(accel, &key, &mods);
477
478 if (key == 0)
479 return;
480
481 GtkWidget *toplevel = gtk_widget_get_toplevel(widget);
482 if (!GTK_IS_WINDOW(toplevel))
483 return;
484
485 GtkAccelGroup *accel_group = gtk_accel_group_new();
486 gtk_window_add_accel_group(GTK_WINDOW(toplevel), accel_group);
487
488 gtk_widget_add_accelerator(widget, signal_name, accel_group,
489 key, mods, GTK_ACCEL_VISIBLE);
490
491 g_object_unref(accel_group);
492}
