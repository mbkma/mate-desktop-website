40gboolean
41caja_navigation_window_slot_should_close_with_mount (CajaNavigationWindowSlot *slot,
42 GMount *mount)
43{
44 CajaBookmark *bookmark;
45 GFile *mount_location, *bookmark_location;
46 GList *l;
47 gboolean close_with_mount;
48
49 if (slot->parent.pane->window->details->initiated_unmount)
50 {
51 return FALSE;
52 }
53
54 mount_location = g_mount_get_root (mount);
55
56 close_with_mount = TRUE;
57
58 for (l = slot->back_list; l != NULL; l = l->next)
59 {
60 bookmark = CAJA_BOOKMARK (l->data);
61
62 bookmark_location = caja_bookmark_get_location (bookmark);
63 close_with_mount &= g_file_has_prefix (bookmark_location, mount_location) ||
64 g_file_equal (bookmark_location, mount_location);
65 g_object_unref (bookmark_location);
66
67 if (!close_with_mount)
68 {
69 break;
70 }
71 }
72
73 close_with_mount &= g_file_has_prefix (CAJA_WINDOW_SLOT (slot)->location, mount_location) ||
74 g_file_equal (CAJA_WINDOW_SLOT (slot)->location, mount_location);
75
76 /* we could also consider the forward list here, but since the “go home” request
77 * in caja-window-manager-views.c:mount_removed_callback() would discard those
78 * anyway, we don't consider them.
79 */
80
81 g_object_unref (mount_location);
82
83 return close_with_mount;
84}

104static void
105query_editor_changed_callback (CajaSearchBar *bar,
106 CajaQuery *query,
107 gboolean reload,
108 CajaWindowSlot *slot)
109{
110 CajaDirectory *directory;
111
112 g_assert (CAJA_IS_FILE (slot->viewed_file));
113
114 directory = caja_directory_get_for_file (slot->viewed_file);
115 g_assert (CAJA_IS_SEARCH_DIRECTORY (directory));
116
117 caja_search_directory_set_query (CAJA_SEARCH_DIRECTORY (directory),
118 query);
119 if (reload)
120 {
121 caja_window_slot_reload (slot);
122 }
123
124 caja_directory_unref (directory);
125}

127static void
128caja_navigation_window_slot_update_query_editor (CajaWindowSlot *slot)
129{
130 CajaDirectory *directory;
131 CajaSearchDirectory *search_directory;
132 GtkWidget *query_editor;
133
134 g_assert (slot->pane->window != NULL);
135
136 query_editor = NULL;
137
138 directory = caja_directory_get (slot->location);
139 if (CAJA_IS_SEARCH_DIRECTORY (directory))
140 {
141 search_directory = CAJA_SEARCH_DIRECTORY (directory);
142
143 if (caja_search_directory_is_saved_search (search_directory))
144 {
145 query_editor = caja_query_editor_new (TRUE,
146 caja_search_directory_is_indexed (search_directory));
147 }
148 else
149 {
150 query_editor = caja_query_editor_new_with_bar (FALSE,
151 caja_search_directory_is_indexed (search_directory),
152 slot->pane->window->details->active_pane->active_slot == slot,
153 CAJA_SEARCH_BAR (CAJA_NAVIGATION_WINDOW_PANE (slot->pane)->search_bar),
154 slot);
155 }
156 }
157
158 slot->query_editor = CAJA_QUERY_EDITOR (query_editor);
159
160 if (query_editor != NULL)
161 {
162 CajaQuery *query;
163
164 g_signal_connect_object (query_editor, "changed",
165 G_CALLBACK (query_editor_changed_callback), slot, 0);
166
167 query = caja_search_directory_get_query (search_directory);
168 if (query != NULL)
169 {
170 caja_query_editor_set_query (CAJA_QUERY_EDITOR (query_editor),
171 query);
172 g_object_unref (query);
173 }
174 else
175 {
176 caja_query_editor_set_default_query (CAJA_QUERY_EDITOR (query_editor));
177 }
178
179 caja_window_slot_add_extra_location_widget (slot, query_editor);
180 gtk_widget_show (query_editor);
181 caja_query_editor_grab_focus (CAJA_QUERY_EDITOR (query_editor));
182 }
183
184 caja_directory_unref (directory);
185}

187static void
188caja_navigation_window_slot_active (CajaWindowSlot *slot)
189{
190 CajaNavigationWindow *window;
191 CajaNavigationWindowPane *pane;
192 int page_num;
193
194 pane = CAJA_NAVIGATION_WINDOW_PANE (slot->pane);
195 window = CAJA_NAVIGATION_WINDOW (slot->pane->window);
196
197 page_num = gtk_notebook_page_num (GTK_NOTEBOOK (pane->notebook),
198 slot->content_box);
199 g_assert (page_num >= 0);
200
201 gtk_notebook_set_current_page (GTK_NOTEBOOK (pane->notebook), page_num);
202
203 EEL_CALL_PARENT (CAJA_WINDOW_SLOT_CLASS, active, (slot));
204
205 if (slot->viewed_file != NULL)
206 {
207 caja_navigation_window_load_extension_toolbar_items (window);
208 }
209}
