295static char *
296get_string_from_key_file (GKeyFile *key_file, const char *group, const char *key)
297{
298 GError *error;
299 char *str;
300
301 error = NULL;
302 str = NULL;
303
304 if (g_key_file_has_key (key_file, group, key, &error)) {

306 if (!str) {

320static void
321add_key_group_to_hashes (GKeyFile *key_file, const char *group)
322{
323 char *path;
324 char *comment;
325 char *acl;
326 gboolean is_writable;
327 char *guest_ok_str;
328 gboolean guest_ok;
329 ShareInfo *info;
330 ShareInfo *old_info;
331
332 /* Remove the old share based on the name */
333
334 old_info = lookup_share_by_share_name (group);
335 if (old_info) {
336 remove_share_info_from_hashes (old_info);
337 shares_free_share_info (old_info);
338 }
339
340 /* Start parsing, and remove the old share based on the path */
341
342 path = get_string_from_key_file (key_file, group, KEY_PATH);
343 if (!path) {
344 g_message ("group '%s' doesn't have a '%s' key! Ignoring group.", group, KEY_PATH);
345 return;
346 }
347
348 old_info = lookup_share_by_path (path);
349 if (old_info) {
350 remove_share_info_from_hashes (old_info);
351 shares_free_share_info (old_info);
352 }
353
354 /* Finish parsing */
355
356 comment = get_string_from_key_file (key_file, group, KEY_COMMENT);
357
358 acl = get_string_from_key_file (key_file, group, KEY_ACL);
359 if (acl) {
360 if (strstr (acl, "Everyone:R"))
361 is_writable = FALSE;
362 else if (strstr (acl, "Everyone:F"))
363 is_writable = TRUE;
364 else {
365 g_message ("unknown format for key '%s/%s' as it contains '%s'. Assuming that the share is read-only",
366 group, KEY_ACL, acl);
367 is_writable = FALSE;
368 }
369
370 g_free (acl);
371 } else {
372 g_message ("group '%s' doesn't have a '%s' key! Assuming that the share is read-only.", group, KEY_ACL);
373 is_writable = FALSE;
374 }
375
376 guest_ok_str = get_string_from_key_file (key_file, group, KEY_GUEST_OK);
377 if (guest_ok_str) {
378 if (strcmp (guest_ok_str, "n") == 0)
379 guest_ok = FALSE;
380 else if (strcmp (guest_ok_str, "y") == 0)
381 guest_ok = TRUE;
382 else {
383 g_message ("unknown format for key '%s/%s' as it contains '%s'. Assuming that the share is not guest accessible.",
384 group, KEY_GUEST_OK, guest_ok_str);
385 guest_ok = FALSE;
386 }
387
388 g_free (guest_ok_str);
389 } else {
390 g_message ("group '%s' doesn't have a '%s' key! Assuming that the share is not guest accessible.", group, KEY_GUEST_OK);
391 guest_ok = FALSE;
392 }
393
394 g_assert (path != NULL);
395 g_assert (group != NULL);
396
397 info = g_new (ShareInfo, 1);
398 info->path = path;
399 info->share_name = g_strdup (group);
400 info->comment = comment;
401 info->is_writable = is_writable;
402 info->guest_ok = guest_ok;
403
404 add_share_info_to_hashes (info);
405}

407static void
408replace_shares_from_key_file (GKeyFile *key_file)
409{
410 gsize num_groups;
411 char **group_names;
412 gsize i;
413
414 group_names = g_key_file_get_groups (key_file, &num_groups);
415
416 /* FIXME: In add_key_group_to_hashes(), we simply ignore key groups
417 * which have invalid data (i.e. no path). We could probably accumulate a
418 * GError with the list of invalid groups and propagate it upwards.
419 */
420 for (i = 0; i < num_groups; i++) {
421 g_assert (group_names[i] != NULL);
422 add_key_group_to_hashes (key_file, group_names[i]);
423 }
424
425 g_strfreev (group_names);
426}

462static gboolean
463refresh_if_needed (GError **error)
464{
465 gboolean retval;
466
467 if (refresh_timestamp_update_counter == 0) {
468 time_t new_timestamp;
469
470 refresh_timestamp_update_counter = NUM_CALLS_BETWEEN_TIMESTAMP_UPDATES;
471
472 new_timestamp = time (NULL);
473 if (new_timestamp - refresh_timestamp > TIMESTAMP_THRESHOLD) {
474 /* g_message ("REFRESHING SHARES"); */
475 retval = refresh_shares (error);
476 } else
477 retval = TRUE;
478
479 refresh_timestamp = new_timestamp;
480 } else {
481 refresh_timestamp_update_counter--;
482 retval = TRUE;
483 }
484
485 return retval;
486}

597static gboolean
598add_share (ShareInfo *info, GError **error)
599{
600 char *argv[7];
601 int argc;
602 ShareInfo *copy;
603 GKeyFile *key_file;
604 GError *real_error;
605 gboolean supports_success;
606 gboolean supports_guest_ok;
607 gboolean net_usershare_success;
608
609 /* g_message ("add_share() start"); */
610
611 if (throw_error_on_add) {
612 g_set_error (error,
613 SHARES_ERROR,
614 SHARES_ERROR_FAILED,
615 _("Failed"));
616 g_message ("add_share() end FAIL");
617 return FALSE;
618 }
619
620 supports_success = shares_supports_guest_ok (&supports_guest_ok, error);
621 if (!supports_success)
622 return FALSE;
623
624 argv[0] = "add";
625 argv[1] = "--long";
626 argv[2] = info->share_name;
627 argv[3] = info->path;
628 argv[4] = info->comment;
629 argv[5] = info->is_writable ? "Everyone:F" : g_strdup_printf ("Everyone:R,%s:F", g_get_user_name ());
630
631 if (supports_guest_ok) {
632 argv[6] = info->guest_ok ? "guest_ok=y" : "guest_ok=n";
633 argc = 7;
634 } else
635 argc = 6;
636
637 real_error = NULL;
638 net_usershare_success = net_usershare_run (argc, argv, &key_file, &real_error);

643 g_propagate_error (error, real_error);
644 return FALSE;
645 }
646
647 replace_shares_from_key_file (key_file);
648
649 copy = copy_share_info (info);
650 add_share_info_to_hashes (copy);
651
652 /* g_message ("add_share() end SUCCESS"); */
653
654 return TRUE;

659{
660 ShareInfo *old_info;
661 char *argv[2];
662 GError *real_error;
663
664 /* g_message ("remove_share() start"); */
665
666 if (throw_error_on_remove) {
667 g_set_error (error,

669 SHARES_ERROR_FAILED,
670 "Failed");
671 g_message ("remove_share() end FAIL");
672 return FALSE;
673 }
674
675 old_info = lookup_share_by_path (path);
676 if (!old_info) {
677 char *display_name;
678
679 display_name = g_filename_display_name (path);
680 g_set_error (error,
681 SHARES_ERROR,

689 }

693
694 real_error = NULL;
695 if (!net_usershare_run (G_N_ELEMENTS (argv), argv, NULL, &real_error)) {
696 g_message ("Called \"net usershare delete\" but it failed: %s", real_error->message);
697 g_propagate_error (error, real_error);
698 g_message ("remove_share() end FAIL");
699 return FALSE;
700 }
701
702 remove_share_info_from_hashes (old_info);
703 shares_free_share_info (old_info);
704
705 /* g_message ("remove_share() end SUCCESS"); */
706
707 return TRUE;
708}

710static gboolean
711modify_share (const char *old_path, ShareInfo *info, GError **error)
712{
713 ShareInfo *old_info;
714
715 /* g_message ("modify_share() start"); */
716
717 old_info = lookup_share_by_path (old_path);
718 if (old_info == NULL) {
719 /*g_message ("modify_share() end; calling add_share() instead");*/
720 return add_share (info, error);
721 }
722
723 g_assert (old_info != NULL);
724
725 if (strcmp (info->path, old_info->path) != 0) {
726 g_set_error (error,
727 SHARES_ERROR,
728 SHARES_ERROR_FAILED,
729 _("Cannot change the path of an existing share; please remove the old share first and add a new one"));
730 g_message ("modify_share() end FAIL: tried to change the path in a share!");
731 return FALSE;
732 }
733
734 if (throw_error_on_modify) {
735 g_set_error (error,
736 SHARES_ERROR,
737 SHARES_ERROR_FAILED,
738 "Failed");
739 g_message ("modify_share() end FAIL");
740 return FALSE;
741 }
742
743 /* Although "net usershare add" will modify an existing share if it has the same share name
744 * as the one that gets passed in, our semantics are different. We have a one-to-one mapping
745 * between paths and share names; "net usershare" supports a one-to-many mapping from paths
746 * to share names. So, we must first remove the old share and then add the new/modified one.
747 */
748
749 if (!remove_share (old_path, error)) {
750 g_message ("modify_share() end FAIL: error when removing old share");
751 return FALSE;
752 }
753
754 /* g_message ("modify_share() end: will call add_share() with the new share info"); */
755 return add_share (info, error);
756}
