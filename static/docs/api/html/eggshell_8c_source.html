56char *
57egg_shell (const char *shell)
58{
59 struct passwd *pw;
60 int i;
61 static const char shells [][14] =
62 {
63 /* Note that on some systems shells can also
64 * be installed in /usr/bin */
65 "/bin/bash", "/usr/bin/bash",
66 "/bin/zsh", "/usr/bin/zsh",
67 "/bin/tcsh", "/usr/bin/tcsh",
68 "/bin/ksh", "/usr/bin/ksh",
69 "/bin/csh", "/bin/sh"
70 };
71
72 if (geteuid () == getuid () &&
73 getegid () == getgid ())
74 {
75 /* only in non-setuid */
76 if (shell != NULL)
77 {
78 if (access (shell, X_OK) == 0)
79 {
80 return g_strdup (shell);
81 }
82 }
83 }
84 pw = getpwuid(getuid());
85 if (pw && pw->pw_shell)
86 {
87 if (access (pw->pw_shell, X_OK) == 0)
88 {
89 return g_strdup (pw->pw_shell);
90 }
91 }
92
93 for (i = 0; i != G_N_ELEMENTS (shells); i++)
94 {
95 if (access (shells [i], X_OK) == 0)
96 {
97 return g_strdup (shells[i]);
98 }
99 }
100
101 /* If /bin/sh doesn't exist, your system is truly broken. */
102 g_assert_not_reached ();
103
104 /* Placate compiler. */
105 return NULL;
106}
