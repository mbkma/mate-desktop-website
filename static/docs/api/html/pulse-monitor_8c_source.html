182static void
183pulse_monitor_finalize (GObject *object)
184{
185 PulseMonitor *monitor;
186
187 monitor = PULSE_MONITOR (object);
188
189 /* The pulse stream may exist if the monitor is running */
190 if (monitor->priv->stream != NULL) {
191 pa_stream_disconnect (monitor->priv->stream);
192 pa_stream_unref (monitor->priv->stream);
193 }
194
195 pa_context_unref (monitor->priv->context);
196 pa_proplist_free (monitor->priv->proplist);
197
198 G_OBJECT_CLASS (pulse_monitor_parent_class)->finalize (object);
199}

231gboolean
232pulse_monitor_set_enabled (PulseMonitor *monitor, gboolean enabled)
233{
234 g_return_val_if_fail (PULSE_IS_MONITOR (monitor), FALSE);
235
236 if (enabled == monitor->priv->enabled)
237 return TRUE;
238
239 if (enabled) {
240 monitor->priv->enabled = stream_connect (monitor);
241
242 if (monitor->priv->enabled == FALSE)
243 return FALSE;
244 } else {
245 pa_stream_disconnect (monitor->priv->stream);
246 pa_stream_unref (monitor->priv->stream);
247
248 monitor->priv->stream = NULL;
249 monitor->priv->enabled = FALSE;
250 }
251 g_object_notify_by_pspec (G_OBJECT (monitor), properties[PROP_ENABLED]);
252
253 return TRUE;
254}

256static gboolean
257stream_connect (PulseMonitor *monitor)
258{
259 pa_sample_spec spec;
260 pa_buffer_attr attr;
261 gchar *idx;
262 int ret;
263
264 attr.maxlength = (guint32) -1;
265 attr.tlength = 0;
266 attr.prebuf = 0;
267 attr.minreq = 0;
268 attr.fragsize = sizeof (gfloat);
269 spec.channels = 1;
270 spec.format = PA_SAMPLE_FLOAT32;
271 spec.rate = 25;
272
273 monitor->priv->stream =
274 pa_stream_new_with_proplist (monitor->priv->context,
275 _("Peak detect"),
276 &spec,
277 NULL,
278 monitor->priv->proplist);
279
280 if (G_UNLIKELY (monitor->priv->stream == NULL)) {
281 g_warning ("Failed to create peak monitor: %s",
282 pa_strerror (pa_context_errno (monitor->priv->context)));
283 return FALSE;
284 }
285
286 /* Set sink input index for the stream, source outputs are not supported */
287 if (monitor->priv->index_sink_input != PA_INVALID_INDEX)
288 pa_stream_set_monitor_stream (monitor->priv->stream,
289 monitor->priv->index_sink_input);
290
291 pa_stream_set_read_callback (monitor->priv->stream,
292 stream_read_cb,
293 monitor);
294
295 /* Source index must be passed as a string */
296 idx = g_strdup_printf ("%u", monitor->priv->index_source);
297 ret = pa_stream_connect_record (monitor->priv->stream,
298 idx,
299 &attr,
300 PA_STREAM_DONT_MOVE |
301 PA_STREAM_PEAK_DETECT |
302 PA_STREAM_ADJUST_LATENCY);
303 g_free (idx);
304
305 if (ret < 0) {
306 g_warning ("Failed to connect peak monitor: %s", pa_strerror (ret));
307 return FALSE;
308 }
309 return TRUE;
310}

312static void
313stream_read_cb (pa_stream *stream, size_t length, void *userdata)
314{
315 const void *data;
316
317 /* Read the next fragment from the buffer (for recording streams).
318 *
319 * If there is data at the current read index, data will point to the
320 * actual data and length will contain the size of the data in bytes
321 * (which can be less or more than a complete fragment).
322 *
323 * If there is no data at the current read index, it means that either
324 * the buffer is empty or it contains a hole (that is, the write index
325 * is ahead of the read index but there's no data where the read index
326 * points at). If the buffer is empty, data will be NULL and length will
327 * be 0. If there is a hole, data will be NULL and length will contain
328 * the length of the hole. */
329 if (pa_stream_peek (stream, &data, &length) < 0)
330 return;
331
332 if (data != NULL) {
333 gdouble v = ((const gfloat *) data)[length / sizeof (gfloat) - 1];
334
335 g_signal_emit (G_OBJECT (userdata),
336 signals[VALUE],
337 0,
338 CLAMP (v, 0, 1));
339 }
340
341 /* pa_stream_drop() should not be called if the buffer is empty, but it
342 * should be called if there is a hole */
343 if (length > 0)
344 pa_stream_drop (stream);
345}
