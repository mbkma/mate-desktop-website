303gchar *
304cact_export_ask_user( BaseWindow *parent, NAObjectItem *item, gboolean first )
305{
306 static const gchar *thisfn = "cact_export_ask_user";
307 CactExportAsk *editor;
308 gboolean are_locked, mandatory;
309 gboolean keep, keep_mandatory;
310 int code;
311 gchar *format;
312
313 g_return_val_if_fail( BASE_IS_WINDOW( parent ), NULL );
314
315 g_debug( "%s: parent=%p, item=%p (%s), first=%s",
316 thisfn,
317 ( void * ) parent,
318 ( void * ) item, G_OBJECT_TYPE_NAME( item ),
319 first ? "True":"False" );
320
321 format = na_settings_get_string( NA_IPREFS_EXPORT_ASK_USER_LAST_FORMAT, NULL, &mandatory );
322 keep = na_settings_get_boolean( NA_IPREFS_EXPORT_ASK_USER_KEEP_LAST_CHOICE, NULL, &keep_mandatory );
323
324 if( first || !keep ){
325 editor = g_object_new( CACT_TYPE_EXPORT_ASK,
326 BASE_PROP_PARENT, parent,
327 BASE_PROP_XMLUI_FILENAME, st_xmlui_filename,
328 BASE_PROP_TOPLEVEL_NAME, st_toplevel_name,
329 BASE_PROP_WSP_NAME, st_wsp_name,
330 NULL );
331
332 editor->private->format = g_strdup( format );
333 editor->private->format_mandatory = mandatory;
334 editor->private->keep_last_choice = keep;
335 editor->private->keep_last_choice_mandatory = keep_mandatory;
336 editor->private->parent = parent;
337 editor->private->item = item;
338
339 are_locked = na_settings_get_boolean( NA_IPREFS_ADMIN_PREFERENCES_LOCKED, NULL, &mandatory );
340 editor->private->preferences_locked = are_locked && mandatory;
341 code = base_window_run( BASE_WINDOW( editor ));
342
343 switch( code ){
344 case GTK_RESPONSE_OK:
345 g_free( format );
346 format = get_export_format( editor );
347 break;
348
349 case GTK_RESPONSE_CANCEL:
350 /* base_dialog::do_run only returns OK or CANCEL */
351 default:
352 g_free( format );
353 format = g_strdup( EXPORTER_FORMAT_NOEXPORT );
354 break;
355 }
356
357 g_object_unref( editor );
358 }
359
360 return( format );
361}

382on_base_initialize_window( CactExportAsk *editor, gpointer user_data )
383{
384 static const gchar *thisfn = "cact_export_ask_on_base_initialize_window";
385 gchar *item_label, *label;
386 GtkWidget *widget;
387 CactExportAskPrivate *priv;
388
389 g_return_if_fail( CACT_IS_EXPORT_ASK( editor ));
390
391 priv = editor->private;
392
393 if( !priv->dispose_has_run ){
394
395 g_debug( "%s: dialog=%p, user_data=%p", thisfn, ( void * ) editor, ( void * ) user_data );
396
397 item_label = na_object_get_label( priv->item );
398
399 if( NA_IS_OBJECT_ACTION( priv->item )){
400 /* Translators: The action <label> is about to be exported */
401 label = g_strdup_printf( _( "The action \"%s\" is about to be exported." ), item_label );
402 } else {
403 /* Translators: The menu <label> is about to be exported */
404 label = g_strdup_printf( _( "The menu \"%s\" is about to be exported." ), item_label );
405 }
406
407 widget = base_window_get_widget( BASE_WINDOW( editor ), "ExportAskLabel" );

409 g_free( label );

416 na_ioptions_list_set_default(
417 NA_IOPTIONS_LIST( editor ), widget,
418 priv->format );
419
420 base_gtk_utils_toggle_set_initial_state( BASE_WINDOW( editor ),
421 "AskKeepChoiceButton", G_CALLBACK( keep_choice_on_toggled ),
422 priv->keep_last_choice,
423 !priv->keep_last_choice_mandatory, !priv->preferences_locked );
424
425 base_window_signal_connect_by_name(
426 BASE_WINDOW( editor ),
427 "CancelButton",
428 "clicked",

468/*
469 * we have come here because preferred_export_format was 'Ask'
470 * in all cases, the chosen format is kept in 'ask_user_last_chosen_format'
471 * and so this will be the default format which will be proposed the next
472 * time we come here
473 * if the 'remember_my_choice' is cheecked, then we do not even ask the user
474 * but returns as soon as we enter
475 *
476 * not overrinding the preferred export format (i.e. letting it to 'Ask')
477 * let the user go back in ask dialog box the next time he will have some
478 * files to export
479 */
480static gchar *
481get_export_format( CactExportAsk *editor )
482{
483 GtkWidget *widget;
484 NAIOption *format;
485 gchar *format_id;
486
487 widget = base_window_get_widget( BASE_WINDOW( editor ), "ExportFormatAskVBox" );
488 format = na_ioptions_list_get_selected( NA_IOPTIONS_LIST( editor ), widget );
489 g_return_val_if_fail( NA_IS_EXPORT_FORMAT( format ), 0 );
490
491 if( !editor->private->keep_last_choice_mandatory ){
492 na_settings_set_boolean( NA_IPREFS_EXPORT_ASK_USER_KEEP_LAST_CHOICE, editor->private->keep_last_choice );
493 }
494
495 format_id = na_ioption_get_id( format );
496 na_settings_set_string( NA_IPREFS_EXPORT_ASK_USER_LAST_FORMAT, format_id );
497
498 return( format_id );
499}
