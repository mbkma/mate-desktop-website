41/* Based on rhythmbox/lib/rb-debug.c */
42/* Our own funky debugging function, should only be used when something
43 * is not going wrong, if something *is* wrong use g_warning.
44 */
45void
46gs_debug_real (const char *func,
47 const char *file,
48 const int line,
49 const char *format, ...)
50{
51 va_list args;
52 char buffer [1025];
53 char *str_time;
54 time_t the_time;
55
56 if (debugging == FALSE)
57 return;
58
59 va_start (args, format);
60
61 g_vsnprintf (buffer, 1024, format, args);
62
63 va_end (args);
64
65 time (&the_time);
66 str_time = g_new0 (char, 255);
67 strftime (str_time, 254, "%H:%M:%S", localtime (&the_time));
68
69 fprintf ((debug_out ? debug_out : stderr),
70 "[%s] %s:%d (%s):\t %s\n",
71 func, file, line, str_time, buffer);
72
73 if (debug_out)
74 fflush (debug_out);
75
76 g_free (str_time);
77}

85void
86gs_debug_init (gboolean debug,
87 gboolean to_file)
88{
89 /* return if already initialized */
90 if (debugging == TRUE)
91 {
92 return;
93 }
94
95 debugging = debug;
96
97 if (debug && to_file)
98 {
99 const char path [50] = "mate_screensaver_debug_XXXXXX";
100 int fd;
101
102 fd = g_file_open_tmp (path, NULL, NULL);
103
104 if (fd >= 0)
105 {
106 debug_out = fdopen (fd, "a");
107 }
108 }
109
110 gs_debug ("Debugging %s", (debug) ? "enabled" : "disabled");
111}

113void
114gs_debug_shutdown (void)
115{
116 if (! debugging)
117 return;
118
119 gs_debug ("Shutting down debugging");
120
121 debugging = FALSE;
122
123 if (debug_out != NULL)
124 {
125 fclose (debug_out);
126 debug_out = NULL;
127 }
128}

130void
131_gs_profile_log (const char *func,
132 const char *note,
133 const char *format,
134 ...)
135{
136 va_list args;
137 char *str;
138 char *formatted;
139
140 if (format == NULL)
141 {
142 formatted = g_strdup ("");
143 }
144 else
145 {
146 va_start (args, format);
147 formatted = g_strdup_vprintf (format, args);
148 va_end (args);
149 }
150
151 if (func != NULL)
152 {
153 str = g_strdup_printf ("MARK: %s %s: %s %s", g_get_prgname(), func, note ? note : "", formatted);
154 }
155 else
156 {
157 str = g_strdup_printf ("MARK: %s: %s %s", g_get_prgname(), note ? note : "", formatted);
158 }
159
160 g_free (formatted);
161
162 g_access (str, F_OK);
163 g_free (str);
164}
