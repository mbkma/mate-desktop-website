50void
51pluma_debug_init (void)
52{
53 if (g_getenv ("PLUMA_DEBUG") != NULL)
54 {
55 /* enable all debugging */
56 debug = ~PLUMA_NO_DEBUG;
57 goto out;
58 }
59
60 if (g_getenv ("PLUMA_DEBUG_VIEW") != NULL)
61 debug = debug | PLUMA_DEBUG_VIEW;
62 if (g_getenv ("PLUMA_DEBUG_SEARCH") != NULL)
63 debug = debug | PLUMA_DEBUG_SEARCH;
64 if (g_getenv ("PLUMA_DEBUG_PREFS") != NULL)
65 debug = debug | PLUMA_DEBUG_PREFS;
66 if (g_getenv ("PLUMA_DEBUG_PRINT") != NULL)
67 debug = debug | PLUMA_DEBUG_PRINT;
68 if (g_getenv ("PLUMA_DEBUG_PLUGINS") != NULL)
69 debug = debug | PLUMA_DEBUG_PLUGINS;
70 if (g_getenv ("PLUMA_DEBUG_TAB") != NULL)
71 debug = debug | PLUMA_DEBUG_TAB;
72 if (g_getenv ("PLUMA_DEBUG_DOCUMENT") != NULL)
73 debug = debug | PLUMA_DEBUG_DOCUMENT;
74 if (g_getenv ("PLUMA_DEBUG_COMMANDS") != NULL)
75 debug = debug | PLUMA_DEBUG_COMMANDS;
76 if (g_getenv ("PLUMA_DEBUG_APP") != NULL)
77 debug = debug | PLUMA_DEBUG_APP;
78 if (g_getenv ("PLUMA_DEBUG_SESSION") != NULL)
79 debug = debug | PLUMA_DEBUG_SESSION;
80 if (g_getenv ("PLUMA_DEBUG_UTILS") != NULL)
81 debug = debug | PLUMA_DEBUG_UTILS;
82 if (g_getenv ("PLUMA_DEBUG_METADATA") != NULL)
83 debug = debug | PLUMA_DEBUG_METADATA;
84 if (g_getenv ("PLUMA_DEBUG_WINDOW") != NULL)
85 debug = debug | PLUMA_DEBUG_WINDOW;
86 if (g_getenv ("PLUMA_DEBUG_LOADER") != NULL)
87 debug = debug | PLUMA_DEBUG_LOADER;
88 if (g_getenv ("PLUMA_DEBUG_SAVER") != NULL)
89 debug = debug | PLUMA_DEBUG_SAVER;
90
91out:
92
93#ifdef ENABLE_PROFILING
94 if (debug != PLUMA_NO_DEBUG)
95 timer = g_timer_new ();
96#endif
97 return;
98}

100void
101pluma_debug_message (PlumaDebugSection section,
102 const gchar *file,
103 gint line,
104 const gchar *function,
105 const gchar *format, ...)
106{
107 if (G_UNLIKELY (debug & section))
108 {
109#ifdef ENABLE_PROFILING
110 gdouble seconds;
111 g_return_if_fail (timer != NULL);
112#endif
113
114 va_list args;
115 gchar *msg;
116
117 g_return_if_fail (format != NULL);
118
119 va_start (args, format);
120 msg = g_strdup_vprintf (format, args);
121 va_end (args);
122
123#ifdef ENABLE_PROFILING
124 seconds = g_timer_elapsed (timer, NULL);
125 g_print ("[%f (%f)] %s:%d (%s) %s\n",
126 seconds, seconds - last, file, line, function, msg);
127 last = seconds;
128#else
129 g_print ("%s:%d (%s) %s\n", file, line, function, msg);
130#endif
131
132 fflush (stdout);
133
134 g_free (msg);
135 }
136}
