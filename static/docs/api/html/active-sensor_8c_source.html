37typedef enum {
38 VERY_LOW_SENSOR_VALUE = 0,
39 LOW_SENSOR_VALUE,
40 NORMAL_SENSOR_VALUE,
41 HIGH_SENSOR_VALUE,
42 VERY_HIGH_SENSOR_VALUE
43} SensorValueRange;
44
45/* Cast a given value to a valid SensorValueRange */
46#define SENSOR_VALUE_RANGE(x) ((SensorValueRange)(CLAMP(x, VERY_LOW_SENSOR_VALUE, VERY_HIGH_SENSOR_VALUE)))
47
48#define CAIRO_GRAPH_COLOR_GRADIENT 0.4

58static gdouble sensor_value_range_normalised(gdouble value,
59 gdouble low_value,
60 gdouble high_value) {
61
62 return ((value - low_value)/(high_value - low_value));
63}
64
65static SensorValueRange sensor_value_range(gdouble sensor_value,
66 gdouble low_value,
67 gdouble high_value) {
68
69 gdouble range;
70 range = sensor_value_range_normalised(sensor_value, low_value, high_value)*(gdouble)(VERY_HIGH_SENSOR_VALUE);
71
72 /* check if need to round up, otherwise let int conversion
73 * round down for us and make sure it is a valid range
74 * value */
75 return SENSOR_VALUE_RANGE(((gint)range + ((range - ((gint)range)) >= 0.5)));
76}

78static gboolean active_sensor_execute_alarm(ActiveSensor *active_sensor,
79 NotifType notif_type) {
80
81 gboolean ret;
82 GError *error = NULL;
83
84 sensors_applet_notify_active_sensor(active_sensor, notif_type);
85 g_debug("EXECUTING %s ALARM: %s",
86 (notif_type == LOW_ALARM ?
87 "LOW" : "HIGH"),
88 active_sensor->alarm_command[notif_type]);
89 ret = g_spawn_command_line_async (active_sensor->alarm_command[notif_type], &error);
90 g_debug("Command executed in shell");
91
92 if (error)
93 g_error_free (error);
94
95 return ret;
96}

106/* needs to be able to be called by the config dialog when the alarm command changes */
107void active_sensor_alarm_off(ActiveSensor *active_sensor,
108 NotifType notif_type) {
109
110 g_assert(active_sensor);
111
112 if (active_sensor->alarm_timeout_id[notif_type] != -1) {
113 g_debug("Disabling %s alarm.", (notif_type == LOW_ALARM ? "LOW" : "HIGH"));
114 if (!g_source_remove(active_sensor->alarm_timeout_id[notif_type])) {
115 g_debug("Error removing alarm source");
116 }
117 g_free(active_sensor->alarm_command[notif_type]);
118 active_sensor->alarm_timeout_id[notif_type] = -1;
119
120 }
121 sensors_applet_notify_end(active_sensor, notif_type);
122}

124static void active_sensor_all_alarms_off(ActiveSensor *active_sensor) {
125 /* turn off any alarms */
126 int i;

129 g_debug("-- turning off notif with type %d ---", i);
130 active_sensor_alarm_off(active_sensor, i);

135static void active_sensor_alarm_on(ActiveSensor *active_sensor,
136 NotifType notif_type) {
137
138 GtkTreeModel *model;
139 GtkTreePath *tree_path;
140 GtkTreeIter iter;
141
142 g_assert(active_sensor);
143
144 model = gtk_tree_row_reference_get_model(active_sensor->sensor_row);
145 tree_path = gtk_tree_row_reference_get_path(active_sensor->sensor_row);
146
147 if (gtk_tree_model_get_iter(model, &iter, tree_path)) {
148
149 if (active_sensor->alarm_timeout_id[notif_type] == -1) {
150 /* alarm is not currently on */
151 gtk_tree_model_get(model,
152 &iter,
153 (notif_type == LOW_ALARM ?
154 LOW_ALARM_COMMAND_COLUMN :
155 HIGH_ALARM_COMMAND_COLUMN),
156 &(active_sensor->alarm_command[notif_type]),
157 ALARM_TIMEOUT_COLUMN, &(active_sensor->alarm_timeout),
158 -1);
159
160 g_debug("Activating alarm to repeat every %d seconds", active_sensor->alarm_timeout);
161
162 /* execute alarm once, then add to time to keep repeating it */
163 active_sensor_execute_alarm(active_sensor, notif_type);
164 int timeout = (active_sensor->alarm_timeout <= 0 ? G_MAXINT : active_sensor->alarm_timeout);
165 switch (notif_type) {
166 case LOW_ALARM:
167 active_sensor->alarm_timeout_id[notif_type] = g_timeout_add_seconds(timeout,
168 (GSourceFunc)active_sensor_execute_low_alarm,
169 active_sensor);
170 break;
171
172 case HIGH_ALARM:
173 active_sensor->alarm_timeout_id[notif_type] = g_timeout_add_seconds(timeout,
174 (GSourceFunc)active_sensor_execute_high_alarm,
175 active_sensor);
176 break;
177
178 default:
179 g_debug("Unknown notif type: %d", notif_type);
180 }
181
182 }
183 }
184
185 gtk_tree_path_free(tree_path);
186}

192gint active_sensor_compare(ActiveSensor *a, ActiveSensor *b) {
193 GtkTreePath *a_tree_path, *b_tree_path;
194 gint ret_val;
195
196 g_assert(a);
197 g_assert(b);
198
199 a_tree_path = gtk_tree_row_reference_get_path(a->sensor_row);
200 b_tree_path = gtk_tree_row_reference_get_path(b->sensor_row);
201
202 ret_val = gtk_tree_path_compare(a_tree_path, b_tree_path);
203
204 gtk_tree_path_free(a_tree_path);
205 gtk_tree_path_free(b_tree_path);
206
207 return ret_val;
208}

210static void active_sensor_update_icon(ActiveSensor *active_sensor,
211 GdkPixbuf *base_icon,
212 SensorType sensor_type) {
213
214 GdkPixbuf *new_icon;
215 const gchar *overlay_icon_filename = NULL;
216 SensorValueRange value_range;
217
218 g_assert(active_sensor);
219
220 /* select overlay icon depending on sensor value */
221 value_range = sensor_value_range(active_sensor->sensor_values[0],
222 active_sensor->sensor_low_value,
223 active_sensor->sensor_high_value);
224
225 if (sensor_type == TEMP_SENSOR) {
226 overlay_icon_filename = temp_overlay_icons[value_range];
227 }
228
229 /* load base icon */
230 new_icon = gdk_pixbuf_copy(base_icon);
231
232 /* only load overlay if required */
233 if (overlay_icon_filename) {
234 GdkPixbuf *overlay_icon;
235
236 overlay_icon = gdk_pixbuf_new_from_file_at_size(overlay_icon_filename,
237 DEFAULT_ICON_SIZE,
238 DEFAULT_ICON_SIZE,
239 NULL);
240
241 if (overlay_icon) {
242 gdk_pixbuf_composite(overlay_icon, new_icon,
243 0, 0,
244 DEFAULT_ICON_SIZE, DEFAULT_ICON_SIZE,
245 0, 0,
246 1.0, 1.0,
247 GDK_INTERP_BILINEAR,
248 255);
249
250 g_object_unref(overlay_icon);
251 }
252 }
253
254 gtk_image_set_from_pixbuf(GTK_IMAGE(active_sensor->icon),
255 new_icon);
256 g_object_unref(new_icon);
257
258}

260static void active_sensor_update_graph(ActiveSensor *as, cairo_t *cr) {
261 GtkAllocation allocation;
262 gdouble line_height;
263 gdouble width, height;
264 gdouble x = 0.0, y = 0.0;
265 cairo_pattern_t *pattern;
266 gint i;
267
268 gtk_widget_get_allocation (as->graph, &allocation);
269 width = allocation.width;
270 height = allocation.height;
271
272 /* so we can set a clipping area, as well as fill the * back of the graph black */
273 cairo_rectangle(cr,
274 0, 0,
275 width,
276 height);
277 /* clip to rectangle and keep it as a path so can be
278 * filled below */
279 cairo_clip_preserve(cr);
280
281 /* use black for bg color of graphs */
282 cairo_set_source_rgb(cr, 0.0, 0.0, 0.0);
283 cairo_fill(cr);
284
285 /* determine height to scale line at for each value -
286 * only do as many as will fit or the number of
287 * samples that we have */
288 for (i = 0; i < MIN(as->num_samples, width); i++) {
289 /* need to remove one more to make it line up
290 * properly when drawing */
291 x = width - i - 1;
292 y = height;
293
294 line_height = sensor_value_range_normalised(as->sensor_values[i],
295 as->sensor_low_value,
296 as->sensor_high_value) * height;
297
298 if (line_height > 0) {
299 cairo_move_to(cr, x, y);
300 cairo_line_to(cr, x, y - line_height);
301 }
302 }
303
304 /* make lines a gradient from slightly darker than
305 * chosen color at bottom of graph, to slightly
306 * lighter than chosen color at top of graph */
307 pattern = cairo_pattern_create_linear(x, y, x, 0);
308 cairo_pattern_add_color_stop_rgb(pattern,
309 0,
310 as->graph_color.red - CAIRO_GRAPH_COLOR_GRADIENT,
311 as->graph_color.green - CAIRO_GRAPH_COLOR_GRADIENT,
312 as->graph_color.blue - CAIRO_GRAPH_COLOR_GRADIENT);
313
314 cairo_pattern_add_color_stop_rgb(pattern,
315 height,
316 as->graph_color.red + CAIRO_GRAPH_COLOR_GRADIENT,
317 as->graph_color.green + CAIRO_GRAPH_COLOR_GRADIENT,
318 as->graph_color.blue + CAIRO_GRAPH_COLOR_GRADIENT);
319
320 cairo_set_source(cr, pattern);
321 cairo_stroke(cr);
322 cairo_pattern_destroy(pattern);
323}

362static void active_sensor_set_graph_dimensions(ActiveSensor *as,
363 gint width,
364 gint height) {
365
366 gint num_samples, old_num_samples;
367 gint graph_width, graph_height;
368
369 /* dimensions are really for graph frame, so need to remove
370 * extra width added by graph frame - make sure not less than
371 * 1 - always need atleast 1 sample */
372 graph_width = CLAMP(width - GRAPH_FRAME_EXTRA_WIDTH, 1, width - GRAPH_FRAME_EXTRA_WIDTH);
373 graph_height = CLAMP(height - GRAPH_FRAME_EXTRA_WIDTH, 1 , height - GRAPH_FRAME_EXTRA_WIDTH);
374
375 g_debug("setting graph dimensions to %d x %d", graph_width, graph_height);
376 num_samples = graph_width;
377
378 if (as->sensor_values) {
379 gdouble *old_values;
380
381 old_values = as->sensor_values;
382 old_num_samples = as->num_samples;
383
384 as->num_samples = num_samples;
385 as->sensor_values = g_new0 (gdouble, as->num_samples);
386 memcpy(as->sensor_values,
387 old_values,
388 MIN(old_num_samples, as->num_samples)*sizeof(gdouble));
389
390 g_free(old_values);
391 } else {
392 as->sensor_values = g_new0 (gdouble, num_samples);
393 as->num_samples = num_samples;
394 }
395
396 /* update graph frame size request */
397 gtk_widget_set_size_request(as->graph,
398 graph_width,
399 graph_height);
400}

402void active_sensor_update_graph_dimensions(ActiveSensor *as,
403 gint sizes[2]) {
404
405 active_sensor_set_graph_dimensions(as, sizes[0], sizes[1]);
406 gtk_widget_queue_draw (as->graph);
407}

409ActiveSensor *active_sensor_new(SensorsApplet *sensors_applet,
410 GtkTreeRowReference *sensor_row) {
411
412 ActiveSensor *active_sensor;
413 MatePanelAppletOrient orient;
414 gint graph_size;
415 gboolean horizontal;
416
417 g_assert(sensors_applet);
418 g_assert(sensor_row);
419
420 g_debug("creating new active sensor");
421
422 active_sensor = g_new0(ActiveSensor, 1);
423 active_sensor->sensors_applet = sensors_applet;
424
425 active_sensor->sensor_row = sensor_row;
426
427#ifdef HAVE_LIBNOTIFY
428 /* set a base time, creation -5 minutes */
429 active_sensor->ierror_ts = time(NULL) - 300;
430#endif
431
432 int i;
433 for (i = 0; i < NUM_ALARMS; i++) {
434 active_sensor->alarm_timeout_id[i] = -1;
435 }
436
437 active_sensor->label = gtk_label_new("");
438 active_sensor->value = gtk_label_new("");
439 active_sensor->icon = gtk_image_new();
440
441 active_sensor->graph = gtk_drawing_area_new();
442 active_sensor->graph_frame = gtk_frame_new(NULL);
443 gtk_frame_set_shadow_type(GTK_FRAME(active_sensor->graph_frame),
444 GTK_SHADOW_IN);
445 gtk_container_add(GTK_CONTAINER(active_sensor->graph_frame),
446 active_sensor->graph);
447 gtk_widget_add_events(active_sensor->graph_frame,
448 GDK_ALL_EVENTS_MASK);
449
450 /* need to set size according to orientation */
451 orient = mate_panel_applet_get_orient(active_sensor->sensors_applet->applet);
452 graph_size = g_settings_get_int(active_sensor->sensors_applet->settings, GRAPH_SIZE);
453
454 horizontal = ((orient == MATE_PANEL_APPLET_ORIENT_UP) ||
455 (orient == MATE_PANEL_APPLET_ORIENT_DOWN));
456
457 active_sensor_set_graph_dimensions(active_sensor,
458 (horizontal ? graph_size : sensors_applet->size),
459 (horizontal ? sensors_applet->size : graph_size));
460
461 g_signal_connect(G_OBJECT(active_sensor->graph),
462 "draw",
463 G_CALLBACK(graph_draw_cb),
464 active_sensor);
465
466 active_sensor->updated = FALSE;
467 return active_sensor;
468}

470static void active_sensor_update_sensor_value(ActiveSensor *as,
471 gdouble sensor_value) {
472
473 /* only if have more than 1 sample stored */
474 if (as->num_samples > 1) {
475 memmove(&(as->sensor_values[1]),
476 as->sensor_values,
477 (as->num_samples - 1)*sizeof(gdouble));
478 }
479
480 as->sensor_values[0] = sensor_value;
481}

483void active_sensor_update(ActiveSensor *active_sensor,
484 SensorsApplet *sensors_applet) {
485
486 GtkTreeModel *model;
487 GtkTreeIter iter;
488 GtkTreePath *path;
489
490 /* instance data from the tree for this sensor */
491 gchar *sensor_path = NULL;
492 gchar *sensor_id = NULL;
493 gchar *sensor_label = NULL;
494 SensorType sensor_type;
495 gchar *sensor_interface;
496 gboolean sensor_enabled;
497 gdouble sensor_low_value;
498 gdouble sensor_high_value;
499 gboolean sensor_alarm_enabled;
500 gdouble sensor_multiplier;
501 gdouble sensor_offset;
502 gdouble sensor_value;
503 GdkPixbuf *icon_pixbuf;
504 gchar *graph_color;
505 gint label_min_width;
506 GtkRequisition req;
507
508 /* to build the list of labels as we go */
509 gchar *value_text = NULL;
510
511 TemperatureScale scale;
512 DisplayMode display_mode;
513
514 GError *error = NULL;
515
516 gchar *tooltip = NULL;
517 gchar *value_tooltip = NULL;
518
519 /* hidden gsettings options */
520 gint font_size = 0;
521 gboolean hide_units = FALSE;
522 gboolean hide_units_old = FALSE;
523
524 g_assert(active_sensor);
525 g_assert(active_sensor->sensor_row);
526 g_assert(sensors_applet);
527
528 model = gtk_tree_row_reference_get_model(active_sensor->sensor_row);
529 path = gtk_tree_row_reference_get_path(active_sensor->sensor_row);
530
531 /* if can successfully get iter can proceed */
532 if (gtk_tree_model_get_iter(model, &iter, path)) {
533 gtk_tree_path_free(path);
534 gtk_tree_model_get(GTK_TREE_MODEL(sensors_applet->sensors),
535 &iter,
536 PATH_COLUMN, &sensor_path,
537 ID_COLUMN, &sensor_id,
538 LABEL_COLUMN, &sensor_label,
539 INTERFACE_COLUMN, &sensor_interface,
540 SENSOR_TYPE_COLUMN, &sensor_type,
541 ENABLE_COLUMN, &sensor_enabled,
542 LOW_VALUE_COLUMN, &sensor_low_value,
543 HIGH_VALUE_COLUMN, &sensor_high_value,
544 ALARM_ENABLE_COLUMN, &sensor_alarm_enabled,
545 MULTIPLIER_COLUMN, &sensor_multiplier,
546 OFFSET_COLUMN, &sensor_offset,
547 ICON_PIXBUF_COLUMN, &icon_pixbuf,
548 GRAPH_COLOR_COLUMN, &graph_color,
549 -1);
550
551 SensorsAppletPluginGetSensorValue get_sensor_value;
552 /* only call function if is in hash table for plugin */
553 if ((get_sensor_value = sensors_applet_plugins_get_sensor_value_func(sensors_applet, sensor_interface)) != NULL) {
554 sensor_value = get_sensor_value(sensor_path,
555 sensor_id,
556 sensor_type,
557 &error);
558
559 if (error) {
560 g_debug("Error updating active sensor: %s", error->message);
561 sensors_applet_notify_active_sensor(active_sensor,
562 SENSOR_INTERFACE_ERROR);
563
564 /* hard code text as ERROR */
565 value_text = g_strdup(_("ERROR"));
566 value_tooltip = g_strdup_printf("- %s", error->message);
567 g_error_free(error);
568 error = NULL;
569
570 /* set sensor value to an error code -
571 * note this is not unique */
572 sensor_value = -1;
573 } else {
574 /* use hidden gsettings key for hide_units */
575 hide_units_old = hide_units;
576 hide_units = g_settings_get_boolean(sensors_applet->settings, HIDE_UNITS);
577
578 /* scale value and set text using this value */
579 switch (sensor_type) {
580 case TEMP_SENSOR:
581 scale = (TemperatureScale) g_settings_get_int(sensors_applet->settings, TEMPERATURE_SCALE);
582 /* scale value */
583 sensor_value = sensors_applet_convert_temperature(sensor_value, CELSIUS, scale);
584
585 sensor_value = (sensor_value * sensor_multiplier) + sensor_offset;
586 switch (scale) {
587 case FAHRENHEIT:
588 value_text = g_strdup_printf("%2.0f %s", sensor_value, (hide_units ? "" : UNITS_FAHRENHEIT));
589 /* tooltip should
590 * always display
591 * units */
592 value_tooltip = g_strdup_printf("%2.0f %s", sensor_value, UNITS_FAHRENHEIT);
593
594 break;
595 case CELSIUS:
596 value_text = g_strdup_printf("%2.0f %s", sensor_value, (hide_units ? "" : UNITS_CELSIUS));
597 value_tooltip = g_strdup_printf("%2.0f %s", sensor_value, UNITS_CELSIUS);
598 break;
599 case KELVIN:
600 value_text = g_strdup_printf("%2.0f", sensor_value);
601 value_tooltip = g_strdup(value_text);
602 break;
603 }
604 break;
605
606 case FAN_SENSOR:
607 sensor_value = (sensor_value * sensor_multiplier) + sensor_offset;
608 value_text = g_strdup_printf("%4.0f %s", sensor_value, (hide_units ? "" : UNITS_RPM));
609 value_tooltip = g_strdup_printf("%4.0f %s", sensor_value, UNITS_RPM);
610 break;
611
612 case VOLTAGE_SENSOR:
613 sensor_value = (sensor_value * sensor_multiplier) + sensor_offset;
614 value_text = g_strdup_printf("%4.2f %s", sensor_value, (hide_units ? "" : UNITS_VOLTAGE));
615 value_tooltip = g_strdup_printf("%4.2f %s", sensor_value, UNITS_VOLTAGE);
616 break;
617
618 case CURRENT_SENSOR:
619 sensor_value = (sensor_value * sensor_multiplier) + sensor_offset;
620 value_text = g_strdup_printf("%4.2f %s", sensor_value, (hide_units ? "" : UNITS_CURRENT));
621 value_tooltip = g_strdup_printf("%4.2f %s", sensor_value, UNITS_CURRENT);
622 break;
623
624 } /* end switch(sensor_type) */
625 } /* end else on error */
626
627 /* setup for tooltips */
628 if (sensors_applet->show_tooltip)
629 tooltip = g_strdup_printf("%s %s", sensor_label, value_tooltip);
630 g_free(value_tooltip);
631
632 /* only do icons and labels / graphs if needed */
633 display_mode = g_settings_get_int (sensors_applet->settings, DISPLAY_MODE);
634
635 /* most users wont have a font size set */
636 font_size = g_settings_get_int (sensors_applet->settings, FONT_SIZE);
637
638 /* do icon if needed */
639 if (display_mode == DISPLAY_ICON ||
640 display_mode == DISPLAY_ICON_WITH_VALUE) {
641
642 /* update icon if icon range has changed if no
643 * update has been done before */
644 if ((sensor_value_range(sensor_value, sensor_low_value, sensor_high_value) != sensor_value_range(active_sensor->sensor_values[0], active_sensor->sensor_low_value, active_sensor->sensor_high_value)) ||
645 !(active_sensor->updated)) {
646
647 active_sensor_update_sensor_value(active_sensor, sensor_value);
648 active_sensor->sensor_low_value = sensor_low_value;
649 active_sensor->sensor_high_value = sensor_high_value;
650 active_sensor_update_icon(active_sensor, icon_pixbuf, sensor_type);
651 }
652
653 if (tooltip) {
654 gtk_widget_set_tooltip_text(active_sensor->icon, tooltip);
655 }
656 }
657
658 active_sensor_update_sensor_value(active_sensor, sensor_value);
659 active_sensor->sensor_low_value = sensor_low_value;
660 active_sensor->sensor_high_value = sensor_high_value;
661
662 /* do graph if needed */
663 if (display_mode == DISPLAY_GRAPH) {
664 /* update graph color in case has changed */
665 gdk_rgba_parse(&(active_sensor->graph_color), graph_color);
666
667 gtk_widget_queue_draw (active_sensor->graph);
668 if (tooltip) {
669 gtk_widget_set_tooltip_text(active_sensor->graph, tooltip);
670 }
671 }
672

679 * activate alarm */

681 display_mode == DISPLAY_ICON_WITH_VALUE ||
682 display_mode == DISPLAY_VALUE) {
683
684 value_text = g_markup_printf_escaped("<span foreground=\"#FF0000\">%s</span>", old_value_text);
685 g_free(old_value_text);
686 }
687
688 /* could have both coditions at once */
689 if (sensor_value >= sensor_high_value) {
690 active_sensor_alarm_on(active_sensor, HIGH_ALARM);
691 }

693 if (sensor_value <= sensor_low_value) {
694 active_sensor_alarm_on(active_sensor, LOW_ALARM);
695 }
696
697 } else {
698 /* make sure alarms are off */
699 active_sensor_all_alarms_off(active_sensor);
700 }
701 } else {
702 /* else for if alarm enabled */
703 /* make sure all alarms are off */
704 active_sensor_all_alarms_off(active_sensor);
705 }
706
707 /* do value label */
708 if (display_mode == DISPLAY_LABEL_WITH_VALUE ||
709 display_mode == DISPLAY_ICON_WITH_VALUE ||
710 display_mode == DISPLAY_VALUE) {
711
712 if (font_size) {
713 old_value_text = value_text;
714
715 value_text = g_strdup_printf("<span font_desc=\"%d\">%s</span>", font_size, old_value_text);
716 g_free(old_value_text);
717 }
718

720 gtk_widget_get_preferred_size (GTK_WIDGET (active_sensor->value), &req, NULL);
721 if (!(hide_units_old == hide_units)) {
722 label_min_width = 0;
723 } else {
724 label_min_width = req.width;
725 }

728

730 gtk_widget_set_size_request (GTK_WIDGET (active_sensor->value), label_min_width, req.height);
731
732 if (tooltip) {
733 gtk_widget_set_tooltip_text(active_sensor->value, tooltip);
734 }
735 }
736 /* finished with value text */
737 g_free(value_text);
738
739 /* do label label */
740 if (display_mode == DISPLAY_LABEL_WITH_VALUE) {
741 if (font_size) {
742 old_value_text = sensor_label;
743 sensor_label = g_strdup_printf("<span font_desc=\"%d\">%s</span>", font_size, old_value_text);
744 g_free(old_value_text);
745 }
746
747 gtk_label_set_markup(GTK_LABEL(active_sensor->label), sensor_label);
748 if (tooltip) {
749 gtk_widget_set_tooltip_text(active_sensor->label, tooltip);
750 }
751 }
752
753 g_free(tooltip);
754 } else {
755 g_debug("no get_sensor_value function yet installed for interface %s.", sensor_interface);
756 }
757
758 g_free(sensor_path);
759 g_free(sensor_id);
760 g_free(sensor_label);
761 g_free(sensor_interface);
762 g_free(graph_color);
763 g_object_unref(icon_pixbuf);
764
765 } else {
766 g_debug("Error getting iter when updating sensor...");
767 }
768
769 active_sensor->updated = TRUE;
770
771}
