48gboolean
49ev_init (void)
50{
51 static gboolean have_backends;
52
53 if (ev_init_count++ > 0)
54 return have_backends;
55
56 _ev_debug_init ();
57 _ev_file_helpers_init ();
58 have_backends = _ev_backends_manager_init ();
59
60 return have_backends;
61}

68void
69ev_shutdown (void)
70{

73 if (--ev_init_count > 0)

77 _ev_file_helpers_shutdown ();
