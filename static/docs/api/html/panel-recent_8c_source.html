55static void
56recent_documents_activate_cb (GtkRecentChooser *chooser,
57 gpointer data)
58{
59 GtkRecentInfo *recent_info;
60 const char *uri;
61 const char *mime_type;
62 GdkScreen *screen;
63 GError *error = NULL;
64
65 screen = gtk_widget_get_screen (GTK_WIDGET (chooser));
66
67 recent_info = gtk_recent_chooser_get_current_item (chooser);
68 uri = gtk_recent_info_get_uri (recent_info);
69 mime_type = gtk_recent_info_get_mime_type (recent_info);
70 /* FIXME gtk_recent_info_get_application_info() could be useful */
71
72 if (show_uri (uri, mime_type, screen, &error) != TRUE) {
73 char *uri_utf8;
74
75 uri_utf8 = g_filename_to_utf8 (uri, -1, NULL, NULL, NULL);
76 /* FIXME this could fail... Maybe we want gtk_recent_info_get_display_name() */
77
78 if (error) {
79 char *primary;
80 primary = g_strdup_printf (_("Could not open recently used document \"%s\""),
81 uri_utf8);
82 panel_error_dialog (NULL, screen,
83 "cannot_open_recent_doc", TRUE,
84 primary, error->message);
85 g_free (primary);
86 g_error_free (error);
87 } else {
88 char *primary;
89 char *secondary;
90 primary = g_strdup_printf (_("Could not open recently used document \"%s\""),
91 uri_utf8);
92 secondary = g_strdup_printf (_("An unknown error occurred while trying to open \"%s\"."),
93 uri_utf8);
94 panel_error_dialog (NULL, screen,
95 "cannot_open_recent_doc", TRUE,
96 primary, secondary);
97 g_free (primary);
98 g_free (secondary);
99 }
100
101 g_free (uri_utf8);
102 }
103
104 /* we can unref it only after having used the data we fetched from it */
105 gtk_recent_info_unref (recent_info);
106}

132static void
133recent_documents_clear_cb (GtkMenuItem *menuitem,
134 GtkRecentManager *manager)
135{
136 gpointer tmp;
137
138 if (clear_recent_dialog != NULL) {
139 gtk_window_set_screen (GTK_WINDOW (clear_recent_dialog),
140 gtk_widget_get_screen (GTK_WIDGET (menuitem)));
141 gtk_window_present (GTK_WINDOW (clear_recent_dialog));

144
145 clear_recent_dialog = gtk_message_dialog_new (NULL,
146 0 /* flags */,
147 GTK_MESSAGE_WARNING,
148 GTK_BUTTONS_NONE,
149 _("Clear the Recent Documents list?"));
150 gtk_message_dialog_format_secondary_text (GTK_MESSAGE_DIALOG (clear_recent_dialog),
151 _("If you clear the Recent Documents list, you clear the following:\n"
152 "\342\200\242 All items from the Places \342\206\222 Recent Documents menu item.\n"
153 "\342\200\242 All items from the recent documents list in all applications."));
154
155 panel_dialog_add_button (GTK_DIALOG (clear_recent_dialog),
156 _("_Cancel"), "process-stop",
157 GTK_RESPONSE_CANCEL);

159 panel_dialog_add_button (GTK_DIALOG (clear_recent_dialog),
160 _("_Clear"), "edit-clear",
161 GTK_RESPONSE_ACCEPT);
162
163 gtk_container_set_border_width (GTK_CONTAINER (clear_recent_dialog), 6);
164
165 gtk_window_set_title (GTK_WINDOW (clear_recent_dialog),
166 _("Clear Recent Documents"));
167
168 gtk_dialog_set_default_response (GTK_DIALOG (clear_recent_dialog),
169 GTK_RESPONSE_ACCEPT);
170 gtk_window_set_skip_taskbar_hint (GTK_WINDOW (clear_recent_dialog),
171 FALSE);
172
173 g_signal_connect (clear_recent_dialog, "response",
174 G_CALLBACK (clear_dialog_response), manager);
175
176 g_signal_connect (clear_recent_dialog, "destroy",
177 G_CALLBACK (gtk_widget_destroyed),
178 &clear_recent_dialog);
179
180 tmp = &clear_recent_dialog;
181 g_object_add_weak_pointer (G_OBJECT (clear_recent_dialog), tmp);
182
183 gtk_window_set_screen (GTK_WINDOW (clear_recent_dialog),
184 gtk_widget_get_screen (GTK_WIDGET (menuitem)));
185 gtk_widget_show (clear_recent_dialog);
