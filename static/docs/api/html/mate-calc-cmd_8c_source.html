53static void
54str_adjust(char *str)
55{
56 int i, j = 0;
57
58 str[strlen(str)-1] = '\0'; /* Remove newline at end of string. */
59 for (i = 0; str[i] != '\0'; i++) { /* Remove whitespace. */
60 if (str[i] != ' ' && str[i] != '\t')
61 str[j++] = str[i];
62 }
63 str[j] = '\0';
64 if (j > 0 && str[j-1] == '=') /* Remove trailing '=' (if present). */
65 str[j-1] = '\0';
66}
