61void
62caja_navigation_window_set_spinner_active (CajaNavigationWindow *window,
63 gboolean allow)
64{
65 if (( window->details->spinner_active && allow) ||
66 (!window->details->spinner_active && !allow)) {
67 return;
68 }
69
70 window->details->spinner_active = allow;
71 if (allow) {
72 gtk_widget_show (window->details->spinner);
73 gtk_spinner_start (GTK_SPINNER (window->details->spinner));
74 } else {
75 gtk_widget_hide (window->details->spinner);
76 }
77}

79void
80caja_navigation_window_activate_spinner (CajaNavigationWindow *window)
81{
82 GtkToolItem *item;
83 GtkWidget *spinner;
84
85 if (window->details->spinner != NULL)
86 {
87 return;
88 }
89
90 item = gtk_tool_item_new ();
91 gtk_widget_show (GTK_WIDGET (item));
92 gtk_tool_item_set_expand (item, TRUE);
93 gtk_toolbar_insert (GTK_TOOLBAR (window->details->toolbar),
94 item, -1);
95
96 spinner = gtk_spinner_new ();
97 gtk_widget_show (GTK_WIDGET (spinner));
98
99 item = gtk_tool_item_new ();
100 gtk_container_add (GTK_CONTAINER (item), spinner);
101 gtk_widget_show (GTK_WIDGET (item));
102
103 gtk_toolbar_insert (GTK_TOOLBAR (window->details->toolbar),
104 item, -1);
105
106 window->details->spinner = spinner;
107}

115static GList *
116get_extension_toolbar_items (CajaNavigationWindow *window)
117{
118 CajaWindowSlot *slot;
119 GList *items;
120 GList *providers;
121 GList *l;
122
123 providers = caja_extensions_get_for_type (CAJA_TYPE_MENU_PROVIDER);
124 items = NULL;
125
126 slot = CAJA_WINDOW (window)->details->active_pane->active_slot;
127
128 for (l = providers; l != NULL; l = l->next)
129 {
130 CajaMenuProvider *provider;
131 GList *file_items;
132
133 provider = CAJA_MENU_PROVIDER (l->data);
134 file_items = caja_menu_provider_get_toolbar_items
135 (provider,
136 GTK_WIDGET (window),
137 slot->viewed_file);
138 items = g_list_concat (items, file_items);
139 }
140
141 caja_module_extension_list_free (providers);
142
143 return items;
144}

146void
147caja_navigation_window_load_extension_toolbar_items (CajaNavigationWindow *window)
148{
149 GtkUIManager *ui_manager;
150 GList *items;
151 GList *l;
152 guint merge_id;
153 GtkActionGroup *action_group;
154 GtkAction *action = NULL;
155 CajaMenuItem *item = NULL;
156 const gchar *action_name = NULL;
157
158 ui_manager = caja_window_get_ui_manager (CAJA_WINDOW (window));
159 if (window->details->extensions_toolbar_merge_id != 0)
160 {
161 gtk_ui_manager_remove_ui (ui_manager,
162 window->details->extensions_toolbar_merge_id);
163 window->details->extensions_toolbar_merge_id = 0;
164 }
165
166 if (window->details->extensions_toolbar_action_group != NULL)
167 {
168 gtk_ui_manager_remove_action_group (ui_manager,
169 window->details->extensions_toolbar_action_group);
170 window->details->extensions_toolbar_action_group = NULL;
171 }
172
173 merge_id = gtk_ui_manager_new_merge_id (ui_manager);
174 window->details->extensions_toolbar_merge_id = merge_id;
175 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
176 action_group = gtk_action_group_new ("ExtensionsToolbarGroup");
177 window->details->extensions_toolbar_action_group = action_group;
178#ifdef ENABLE_NLS
179 gtk_action_group_set_translation_domain (action_group, GETTEXT_PACKAGE);
180#endif /* ENABLE_NLS */
181 G_GNUC_END_IGNORE_DEPRECATIONS;
182 gtk_ui_manager_insert_action_group (ui_manager, action_group, -1);
183 g_object_unref (action_group); /* owned by ui manager */
184
185 items = get_extension_toolbar_items (window);
186
187 for (l = items; l != NULL; l = l->next)
188 {
189 item = CAJA_MENU_ITEM (l->data);
190
191 action = caja_toolbar_action_from_menu_item (item, GTK_WIDGET (window));
192
193 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
194 gtk_action_group_add_action (action_group,
195 GTK_ACTION (action));
196 g_object_unref (action);
197
198 action_name = gtk_action_get_name (action);
199 G_GNUC_END_IGNORE_DEPRECATIONS;
200
201 gtk_ui_manager_add_ui (ui_manager,
202 merge_id,
203 TOOLBAR_PATH_EXTENSION_ACTIONS,
204 action_name,
205 action_name,
206 GTK_UI_MANAGER_TOOLITEM,
207 FALSE);
208
209 g_object_unref (item);
210 }
211
212 g_list_free (items);
213}
