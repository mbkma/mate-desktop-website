167static gboolean
168caja_emblem_sidebar_button_press_cb (GtkWidget *widget,
169 GdkEventButton *event,
170 CajaEmblemSidebar *emblem_sidebar)
171{
172 char *keyword, *name;
173 GdkPixbuf *pixbuf;
174
175 if (event->button == 3)
176 {
177 keyword = g_object_get_data (G_OBJECT (widget),
178 "emblem-keyword");
179 name = g_object_get_data (G_OBJECT (widget),
180 "emblem-display-name");
181 pixbuf = g_object_get_data (G_OBJECT (widget),
182 "original-pixbuf");
183
184 emblem_sidebar->details->popup_emblem_keyword = keyword;
185 emblem_sidebar->details->popup_emblem_display_name = name;
186 emblem_sidebar->details->popup_emblem_pixbuf = pixbuf;
187
188 gtk_widget_set_sensitive (emblem_sidebar->details->popup_remove,
189 caja_emblem_can_remove_emblem (keyword));
190 gtk_widget_set_sensitive (emblem_sidebar->details->popup_rename,
191 caja_emblem_can_rename_emblem (keyword));
192
193 gtk_menu_popup_at_pointer (GTK_MENU (emblem_sidebar->details->popup),
194 (const GdkEvent*) event);
195 }
196
197 return TRUE;
198}

233static void
234rename_dialog_response_cb (GtkWidget *dialog, int response,
235 CajaEmblemSidebar *emblem_sidebar)
236{
237 GtkWidget *entry;
238 char *keyword, *name;
239
240 keyword = g_object_get_data (G_OBJECT (dialog), "emblem-keyword");
241
242 if (response == GTK_RESPONSE_CANCEL)
243 {
244 g_free (keyword);
245 gtk_widget_destroy (dialog);
246 return;
247 }
248 else if (response == GTK_RESPONSE_HELP)
249 {
250 g_message ("Implement me!");
251 return;
252 }
253
254 entry = g_object_get_data (G_OBJECT (dialog), "entry");
255
256 name = g_strdup (gtk_entry_get_text (GTK_ENTRY (entry)));
257
258 gtk_widget_destroy (dialog);
259
260 if (caja_emblem_rename_emblem (keyword, name))
261 {
262 send_emblems_changed ();
263 }
264 else
265 {
266 char *error;
267
268 error = g_strdup_printf (_("Could not rename emblem with name '%s'."), name);
269 eel_show_error_dialog (error, _("This is probably because the emblem is a permanent one, and not one that you added yourself."),
270 NULL);
271 g_free (error);
272 }
273
274 g_free (keyword);
275 g_free (name);
276}

388static GtkWidget *
389create_emblem_widget_with_pixbuf (CajaEmblemSidebar *emblem_sidebar,
390 const char *keyword,
391 const char *display_name,
392 GdkPixbuf *pixbuf)
393{
394 GtkWidget *image, *event_box;
395 GdkPixbuf *prelight_pixbuf;
396
397 image = eel_labeled_image_new (display_name, pixbuf);
398
399 eel_labeled_image_set_fixed_image_height (EEL_LABELED_IMAGE (image),
400 STANDARD_EMBLEM_HEIGHT);
401 eel_labeled_image_set_spacing (EEL_LABELED_IMAGE (image),
402 EMBLEM_LABEL_SPACING);
403 event_box = gtk_event_box_new ();
404 gtk_container_add (GTK_CONTAINER (event_box), image);
405
406 prelight_pixbuf = eel_create_spotlight_pixbuf (pixbuf);
407
408 gtk_drag_source_set (event_box, GDK_BUTTON1_MASK, drag_types,
409 G_N_ELEMENTS (drag_types),
410 GDK_ACTION_COPY | GDK_ACTION_MOVE);
411
412 gtk_drag_source_set_icon_pixbuf (event_box, pixbuf);
413
414 g_signal_connect (event_box, "button_press_event",
415 G_CALLBACK (caja_emblem_sidebar_button_press_cb),
416 emblem_sidebar);
417 g_signal_connect (event_box, "drag-data-get",
418 G_CALLBACK (caja_emblem_sidebar_drag_data_get_cb),
419 emblem_sidebar);
420 g_signal_connect (event_box, "enter-notify-event",
421 G_CALLBACK (caja_emblem_sidebar_enter_notify_cb),
422 emblem_sidebar);
423 g_signal_connect (event_box, "leave-notify-event",
424 G_CALLBACK (caja_emblem_sidebar_leave_notify_cb),
425 emblem_sidebar);
426
427 g_object_set_data_full (G_OBJECT (event_box),
428 "emblem-keyword",
429 g_strdup (keyword), g_free);
430 g_object_set_data_full (G_OBJECT (event_box),
431 "emblem-display-name",
432 g_strdup (display_name), g_free);
433 g_object_set_data_full (G_OBJECT (event_box),
434 "original-pixbuf",
435 pixbuf, g_object_unref);
436 g_object_set_data_full (G_OBJECT (event_box),
437 "prelight-pixbuf",
438 prelight_pixbuf, g_object_unref);
439 g_object_set_data (G_OBJECT (event_box),
440 "labeled-image", image);
441
442 return event_box;
443
444}

446static GtkWidget *
447create_emblem_widget (CajaEmblemSidebar *emblem_sidebar,
448 const char *name)
449{
450 GtkWidget *ret;

469 display_name, pixbuf);
470 g_free (keyword);
471 g_object_unref (info);
472 return ret;
473}

526static GtkWidget *
527create_add_emblems_dialog (CajaEmblemSidebar *emblem_sidebar,
528 GSList *emblems)
529{
530 GtkWidget *dialog, *label, *table;
531 GtkWidget *first_entry, *entry, *scroller;
532 GSList *list;
533 int num_emblems;
534 GtkWidget *image = NULL;
535 GtkWidget *hbox = NULL;
536 Emblem *emblem = NULL;
537
538 first_entry = NULL;
539
540 dialog = gtk_dialog_new ();
541 gtk_window_set_title (GTK_WINDOW (dialog), _("Add Emblems..."));
542 gtk_window_set_transient_for (GTK_WINDOW (dialog), NULL);
543
544 eel_dialog_add_button (GTK_DIALOG (dialog),
545 _("_Cancel"),
546 "process-stop",
547 GTK_RESPONSE_CANCEL);
548
549 eel_dialog_add_button (GTK_DIALOG (dialog),
550 _("_OK"),
551 "gtk-ok",
552 GTK_RESPONSE_OK);
553
554 eel_dialog_add_button (GTK_DIALOG (dialog),
555 _("_Help"),
556 "help-browser",
557 GTK_RESPONSE_HELP);
558
559 gtk_dialog_set_default_response (GTK_DIALOG (dialog),
560 GTK_RESPONSE_OK);
561
562 /* FIXME: make a better message */
563 if (g_slist_length (emblems) > 1)
564 {
565 label = gtk_label_new (_("Enter a descriptive name next to each emblem. This name will be used in other places to identify the emblem."));
566 }
567 else
568 {
569 label = gtk_label_new (_("Enter a descriptive name next to the emblem. This name will be used in other places to identify the emblem."));
570 }
571
572 gtk_label_set_line_wrap (GTK_LABEL (label), TRUE);
573 gtk_box_pack_start (GTK_BOX (gtk_dialog_get_content_area (GTK_DIALOG (dialog))),
574 label, FALSE, FALSE, 8);
575 gtk_widget_show (label);
576
577 scroller = eel_scrolled_wrap_table_new (TRUE, GTK_SHADOW_NONE, &table);
578 eel_wrap_table_set_x_spacing (EEL_WRAP_TABLE (table), 8);
579 eel_wrap_table_set_y_spacing (EEL_WRAP_TABLE (table), 8);
580
581 num_emblems=0;
582 list = emblems;
583 while (list != NULL)
584 {
585 /* walk through the list of emblems, and create an image
586 * and entry for each one
587 */
588
589 emblem = (Emblem *)list->data;
590 list = list->next;
591
592 image = gtk_image_new_from_pixbuf (emblem->pixbuf);
593
594 hbox = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);
595 gtk_box_pack_start (GTK_BOX (hbox), image, FALSE, FALSE, 0);
596
597 entry = gtk_entry_new ();
598 gtk_entry_set_activates_default (GTK_ENTRY (entry), TRUE);
599 g_signal_connect (entry, "changed",
600 G_CALLBACK (emblem_name_entry_changed_cb),
601 emblem);
602
603 gtk_box_pack_start (GTK_BOX (hbox), entry, FALSE, FALSE, 0);
604 gtk_container_add (GTK_CONTAINER (table), hbox);
605
606 if (num_emblems == 0)
607 {
608 first_entry = entry;
609 }
610
611 num_emblems++;
612 }
613
614 gtk_container_set_border_width (GTK_CONTAINER (dialog), 8);
615 gtk_box_pack_start (GTK_BOX (gtk_dialog_get_content_area (GTK_DIALOG (dialog))),
616 scroller, TRUE, TRUE, 8);
617 gtk_widget_show_all (scroller);
618
619 gtk_widget_grab_focus (first_entry);
620
621 /* we expand the window to hold up to about 4 emblems, but after that
622 * let the scroller do its thing. Is there a better way to do this?
623 */
624 gtk_window_set_default_size (GTK_WINDOW (dialog), 400,
625 MIN (120+(60*num_emblems), 350));
626
627 g_object_set_data_full (G_OBJECT (dialog), "emblems-to-add",
628 emblems, (GDestroyNotify)destroy_emblem_list);
629
630 return dialog;
631}

651static void
652add_emblems_dialog_response_cb (GtkWidget *dialog, int response,
653 CajaEmblemSidebar *emblem_sidebar)
654{
655 Emblem *emblem;
656 GSList *emblems;
657 GSList *l;
658
659 switch (response)
660 {
661 case GTK_RESPONSE_CANCEL:
662 gtk_widget_destroy (dialog);
663 break;
664
665 case GTK_RESPONSE_HELP:
666 g_message ("Implement me!");
667 break;
668
669 case GTK_RESPONSE_OK:
670 emblems = g_object_get_data (G_OBJECT (dialog),
671 "emblems-to-add");
672
673 for (l = emblems; l; l = l->next)
674 {
675 char *keyword;
676
677 emblem = (Emblem *)l->data;
678 if (emblem->keyword != NULL)
679 {
680 /* this one has already been verified */
681 continue;
682 }
683
684 keyword = caja_emblem_create_unique_keyword (emblem->name);
685 if (!caja_emblem_verify_keyword
686 (GTK_WINDOW (dialog), keyword, emblem->name))
687 {
688 g_free (keyword);
689 return;
690 }
691 else
692 {
693 emblem->keyword = keyword;
694 }
695
696 }
697
698 for (l = emblems; l; l = l->next)
699 {
700 emblem = (Emblem *)l->data;
701
702 caja_emblem_install_custom_emblem (emblem->pixbuf,
703 emblem->keyword,
704 emblem->name,
705 GTK_WINDOW (dialog));
706 }
707
708 gtk_widget_destroy (dialog);

737static void
738caja_emblem_sidebar_drag_received_cb (GtkWidget *widget,
739 GdkDragContext *drag_context,
740 gint x,
741 gint y,
742 GtkSelectionData *data,
743 guint info,
744 guint time,
745 CajaEmblemSidebar *emblem_sidebar)
746{
747 GSList *emblems;
748 Emblem *emblem;
749 GdkPixbuf *pixbuf;
750 char *uri, *error;
751 char **uris;
752 GFile *f;
753 int i;
754 gboolean had_failure;
755 gint data_format, data_length;
756 const guchar *data_data;
757
758 had_failure = FALSE;
759 emblems = NULL;
760 data_format = gtk_selection_data_get_format (data);
761 data_length = gtk_selection_data_get_length (data);
762 data_data = gtk_selection_data_get_data (data);
763
764 switch (info)
765 {
766 case TARGET_URI_LIST:
767 if (data_format != 8 ||
768 data_length == 0)
769 {
770 g_message ("URI list had wrong format (%d) or length (%d)\n",
771 data_format, data_length);
772 return;
773 }
774
775 uris = g_uri_list_extract_uris (data_data);
776 if (uris == NULL)
777 {
778 break;
779 }
780
781 for (i = 0; uris[i] != NULL; ++i)
782 {
783 f = g_file_new_for_uri (uris[i]);
784 pixbuf = caja_emblem_load_pixbuf_for_emblem (f);
785
786 if (pixbuf == NULL)
787 {
788 /* this one apparently isn't an image, or
789 * at least not one that we know how to read
790 */
791 had_failure = TRUE;
792 g_object_unref (f);
793 continue;
794 }
795
796 emblem = g_new (Emblem, 1);
797 emblem->uri = g_file_get_uri (f);
798 emblem->name = NULL; /* created later on by the user */
799 emblem->keyword = NULL;
800 emblem->pixbuf = pixbuf;
801
802 g_object_unref (f);
803
804 emblems = g_slist_prepend (emblems, emblem);
805 }
806
807 g_strfreev (uris);
808
809 if (had_failure && emblems != NULL)
810 {
811 eel_show_error_dialog (_("Some of the files could not be added as emblems."), _("The emblems do not appear to be valid images."), NULL);
812 }
813 else if (had_failure && emblems == NULL)
814 {
815 eel_show_error_dialog (_("None of the files could be added as emblems."), _("The emblems do not appear to be valid images."), NULL);
816
817 }
818
819 if (emblems != NULL)
820 {
821 show_add_emblems_dialog (emblem_sidebar, emblems);
822 }
823
824 break;
825
826 case TARGET_URI:
827 if (data_format != 8 ||
828 data_length == 0)
829 {
830 g_warning ("URI had wrong format (%d) or length (%d)\n",
831 data_format, data_length);
832 return;
833 }
834
835 uri = g_strndup (data_data, data_length);
836
837 f = g_file_new_for_uri (uri);
838 pixbuf = caja_emblem_load_pixbuf_for_emblem (f);
839
840 if (pixbuf != NULL)
841 {
842 emblem = g_new (Emblem, 1);
843 emblem->uri = uri;
844 emblem->name = NULL;
845 emblem->keyword = NULL;
846 emblem->pixbuf = pixbuf;
847
848 emblems = g_slist_prepend (NULL, emblem);
849
850 show_add_emblems_dialog (emblem_sidebar, emblems);
851 }
852 else
853 {
854 char *uri_utf8;
855
856 uri_utf8 = g_file_get_parse_name (f);
857
858 if (uri_utf8)
859 {
860 error = g_strdup_printf (_("The file '%s' does not appear to be a valid image."), uri_utf8);
861 }
862 else
863 {
864 error = g_strdup (_("The dragged file does not appear to be a valid image."));
865 }
866 eel_show_error_dialog (_("The emblem cannot be added."), error, NULL);
867 g_free (error);
868 g_free (uri_utf8);
869 }
870
871 g_object_unref (f);
872 g_free (uri);
873
874 break;
875
876 case TARGET_NETSCAPE_URL:
877 if (data_format != 8 ||
878 data_length == 0)
879 {
880 g_message ("URI had wrong format (%d) or length (%d)\n",
881 data_format, data_length);
882 return;
883 }
884
885 /* apparently, this is a URI/title pair? or just a pair
886 * of identical URIs? Regardless, this seems to work...
887 */
888
889 uris = g_uri_list_extract_uris (data_data);
890 if (uris == NULL)
891 {
892 break;
893 }
894
895 uri = uris[0];
896 if (uri == NULL)
897 {
898 g_strfreev (uris);
899 break;
900 }
901
902 f = g_file_new_for_uri (uri);
903 pixbuf = caja_emblem_load_pixbuf_for_emblem (f);
904 g_object_unref (f);
905
906 if (pixbuf != NULL)
907 {
908 emblem = g_new (Emblem, 1);
909 emblem->uri = g_strdup (uri);
910 emblem->name = NULL;
911 emblem->keyword = NULL;
912 emblem->pixbuf = pixbuf;
913
914 emblems = g_slist_prepend (NULL, emblem);
915
916 show_add_emblems_dialog (emblem_sidebar, emblems);
917 }
918 else
919 {
920 g_warning ("Tried to load '%s', but failed.\n",
921 uri);
922 error = g_strdup_printf (_("The file '%s' does not appear to be a valid image."), uri);
923 eel_show_error_dialog (_("The emblem cannot be added."), error, NULL);
924 g_free (error);
925 }
926
927 g_strfreev (uris);
928
929 break;
930 }

974{

979

983 if (path != NULL)
