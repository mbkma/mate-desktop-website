162guint
163na_iimporter_import_from_uri( const NAIImporter *importer, NAIImporterImportFromUriParmsv2 *parms )
164{
165 static const gchar *thisfn = "na_iimporter_import_from_uri";
166 guint code;
167
168 g_return_val_if_fail( NA_IS_IIMPORTER( importer ), IMPORTER_CODE_PROGRAM_ERROR );
169 g_return_val_if_fail( parms && parms->version == 2, IMPORTER_CODE_PROGRAM_ERROR );
170
171 code = IMPORTER_CODE_NOT_WILLING_TO;
172
173 g_debug( "%s: importer=%p (%s), parms=%p", thisfn,
174 ( void * ) importer, G_OBJECT_TYPE_NAME( importer), ( void * ) parms );
175
176 if( NA_IIMPORTER_GET_INTERFACE( importer )->import_from_uri ){
177 code = NA_IIMPORTER_GET_INTERFACE( importer )->import_from_uri( importer, parms );
178 }
179
180 return( code );
181}

217guint
218na_iimporter_manage_import_mode( NAIImporterManageImportModeParms *parms )
219{
220 static const gchar *thisfn = "na_iimporter_manage_import_mode";
221 guint code;
222 NAObjectItem *exists;
223 guint mode;
224 gchar *id;
225
226 g_return_val_if_fail( parms->imported != NULL, IMPORTER_CODE_CANCELLED );
227
228 code = IMPORTER_CODE_OK;
229 exists = NULL;
230 mode = 0;
231 parms->exist = FALSE;
232 parms->import_mode = parms->asked_mode;
233
234 if( parms->check_fn ){
235 exists = ( *parms->check_fn )( parms->imported, parms->check_fn_data );
236
237 } else {
238 renumber_label_item( parms );
239 na_core_utils_slist_add_message( &parms->messages, "%s", _( "Item was renumbered because the caller did not provide any check function." ));
240 parms->import_mode = IMPORTER_MODE_RENUMBER;
241 }
242
243 g_debug( "%s: exists=%p", thisfn, exists );
244
245 if( exists ){
246 parms->exist = TRUE;
247
248 if( parms->asked_mode == IMPORTER_MODE_ASK ){
249 if( parms->ask_fn ){
250 mode = ( *parms->ask_fn )( parms->imported, exists, parms->ask_fn_data );
251
252 } else {
253 renumber_label_item( parms );
254 na_core_utils_slist_add_message( &parms->messages, "%s", _( "Item was renumbered because the caller did not provide any ask user function." ));
255 parms->import_mode = IMPORTER_MODE_RENUMBER;
256 }
257
258 } else {
259 mode = parms->asked_mode;
260 }
261 }
262
263 /* mode is only set if asked mode is ask user and an ask function was provided
264 * or if asked mode was not ask user
265 */
266 if( mode ){
267 parms->import_mode = mode;
268
269 switch( mode ){
270 case IMPORTER_MODE_RENUMBER:
271 renumber_label_item( parms );
272 if( parms->asked_mode == IMPORTER_MODE_ASK ){
273 na_core_utils_slist_add_message( &parms->messages, "%s", _( "Item was renumbered due to user request." ));
274 }
275 break;
276
277 case IMPORTER_MODE_OVERRIDE:
278 if( parms->asked_mode == IMPORTER_MODE_ASK ){
279 na_core_utils_slist_add_message( &parms->messages, "%s", _( "Existing item was overridden due to user request." ));
280 }
281 break;
282
283 case IMPORTER_MODE_NO_IMPORT:
284 default:
285 id = na_object_get_id( parms->imported );
286 na_core_utils_slist_add_message( &parms->messages, _( "Item %s already exists." ), id );
287 if( parms->asked_mode == IMPORTER_MODE_ASK ){
288 na_core_utils_slist_add_message( &parms->messages, "%s", _( "Import was canceled due to user request." ));
289 }
290 g_free( id );
291 code = IMPORTER_CODE_CANCELLED;
292 }
293 }
294
295 return( code );
296}

299 * renumber the item, and set a new label
300 */
301static void
302renumber_label_item( NAIImporterManageImportModeParms *parms )
303{
304 gchar *label, *tmp;
305
306 na_object_set_new_id( parms->imported, NULL );
307
308 label = na_object_get_label( parms->imported );
309
310 /* Translators: the action has been renumbered during import operation */
311 tmp = g_strdup_printf( "%s %s", label, _( "(renumbered)" ));
312
313 na_object_set_label( parms->imported, tmp );
314
315 g_free( tmp );
316 g_free( label );
317}
