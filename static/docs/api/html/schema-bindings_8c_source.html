51static void
52single_stanza (gboolean is_window, const char *name,
53 const char *default_value,
54 gboolean can_reverse,
55 const char *description)
56{
57 char *keybinding_type = is_window? "window": "global";
58 char *escaped_default_value, *escaped_description;
59
60 if (description==NULL)
61 return; /* it must be undocumented, so it can't go in this table */
62
63 escaped_description = g_markup_escape_text (description, -1);
64 escaped_default_value = default_value==NULL? "disabled":
65 g_markup_escape_text (default_value, -1);
66
67 fprintf (target_file, " <schema>\n");
68 fprintf (target_file, " <key>/schemas/apps/marco/%s_keybindings/%s</key>\n",
69 keybinding_type, name);
70 fprintf (target_file, " <applyto>/apps/marco/%s_keybindings/%s</applyto>\n",
71 keybinding_type, name);
72 fprintf (target_file, " <owner>marco</owner>\n");
73 fprintf (target_file, " <type>string</type>\n");
74 fprintf (target_file, " <default>%s</default>\n", escaped_default_value);
75
76 fprintf (target_file, " <locale name=\"C\">\n");
77 fprintf (target_file, " <short>%s</short>\n", description);
78 fprintf (target_file, " <long>%s</long>\n",
79 can_reverse? about_reversible_keybindings:
80 about_keybindings);
81 fprintf (target_file, " </locale>\n");
82 fprintf (target_file, " </schema>\n\n");
83
84 g_free (escaped_description);
85
86 if (default_value!=NULL)
87 g_free (escaped_default_value);
88}

92static void
93produce_bindings ()
94{
95 /* 10240 is ridiculous overkill; we're writing the input file and
96 * the lines are always 80 chars or less.
97 */
98 char buffer[10240];
99
100 source_file = fopen(source_filename, "r");
101
102 if (!source_file)
103 {
104 g_error ("Cannot compile without %s: %s\n",
105 source_filename, strerror (errno));
106 }
107
108 target_file = fopen(target_filename, "w");
109
110 if (!target_file)
111 {
112 g_error ("Cannot create %s: %s\n",
113 target_filename, strerror (errno));
114 }
115
116 while (fgets (buffer, sizeof (buffer), source_file))
117 {
118 if (strstr (buffer, "<!-- GENERATED -->"))
119 break;
120
121 fprintf (target_file, "%s", buffer);
122 }
123
124 if (!feof (source_file))
125 {
126#define keybind(name, handler, param, flags, stroke, description) \
127 single_stanza ( \
128 flags & BINDING_PER_WINDOW, \
129 #name, \
130 stroke, \
131 flags & BINDING_REVERSES, \
132 description);
133#include "all-keybindings.h"
134#undef keybind
135 }
136
137 while (fgets (buffer, sizeof (buffer), source_file))
138 fprintf (target_file, "%s", buffer);
139
140 if (fclose (source_file)!=0)
141 {
142 g_error ("Cannot close %s: %s\n",
143 source_filename, strerror (errno));
144 }
145
146 if (fclose (target_file)!=0)
147 {
148 g_error ("Cannot close %s: %s\n",
149 target_filename, strerror (errno));
150 }
151}

164 /* Translators: Please don't translate "Control", "Shift", etc, since these
165 * are hardcoded (in gtk/gtkaccelgroup.c; it's not marco's fault).
166 * "disabled" must also stay as it is.
167 */
168 about_keybindings = g_markup_escape_text(_( \
169 "The format looks like \"<Control>a\" or \"<Shift><Alt>F1\".\n\n"\
170 "The parser is fairly liberal and allows "\
171 "lower or upper case, and also abbreviations such as \"<Ctl>\" and " \
172 "\"<Ctrl>\". If you set the option to the special string " \
173 "\"disabled\", then there will be no keybinding for this action."),
174 -1);
175
176 about_reversible_keybindings = g_markup_escape_text(_( \
177 "The format looks like \"<Control>a\" or \"<Shift><Alt>F1\".\n\n"\
178 "The parser is fairly liberal and allows "\
179 "lower or upper case, and also abbreviations such as \"<Ctl>\" and " \
180 "\"<Ctrl>\". If you set the option to the special string " \
181 "\"disabled\", then there will be no keybinding for this action.\n\n"\
182 "This keybinding may be reversed by holding down the \"shift\" key; "
183 "therefore, \"shift\" cannot be one of the keys it uses."),
184 -1);
185
186 produce_bindings ();
187
188 g_free (about_keybindings);
189 g_free (about_reversible_keybindings);
190
191 return 0;
192}
