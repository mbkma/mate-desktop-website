204static void
205mate_mixer_stream_control_get_property (GObject *object,
206 guint param_id,
207 GValue *value,
208 GParamSpec *pspec)
209{
210 MateMixerStreamControl *control;
211
212 control = MATE_MIXER_STREAM_CONTROL (object);
213
214 switch (param_id) {
215 case PROP_NAME:
216 g_value_set_string (value, control->priv->name);
217 break;
218 case PROP_LABEL:
219 g_value_set_string (value, control->priv->label);
220 break;
221 case PROP_FLAGS:
222 g_value_set_flags (value, control->priv->flags);
223 break;
224 case PROP_ROLE:
225 g_value_set_enum (value, control->priv->role);
226 break;
227 case PROP_MEDIA_ROLE:
228 g_value_set_enum (value, control->priv->media_role);
229 break;
230 case PROP_STREAM:
231 g_value_set_object (value, control->priv->stream);
232 break;
233
234 default:
235 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, param_id, pspec);
236 break;
237 }
238}

240static void
241mate_mixer_stream_control_set_property (GObject *object,
242 guint param_id,
243 const GValue *value,
244 GParamSpec *pspec)
245{
246 MateMixerStreamControl *control;
247
248 control = MATE_MIXER_STREAM_CONTROL (object);
249
250 switch (param_id) {
251 case PROP_NAME:
252 /* Construct-only string */
253 control->priv->name = g_value_dup_string (value);
254 break;
255 case PROP_LABEL:
256 /* Construct-only string */
257 control->priv->label = g_value_dup_string (value);
258 break;
259 case PROP_FLAGS:
260 control->priv->flags = g_value_get_flags (value);
261 break;
262 case PROP_ROLE:
263 control->priv->role = g_value_get_enum (value);
264 break;
265 case PROP_MEDIA_ROLE:
266 control->priv->media_role = g_value_get_enum (value);
267 break;
268 case PROP_STREAM:
269 /* Construct-only object */
270 control->priv->stream = g_value_get_object (value);
271
272 if (control->priv->stream != NULL)
273 g_object_add_weak_pointer (G_OBJECT (control->priv->stream),
274 (gpointer *) &control->priv->stream);
275 break;
276
277 default:
278 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, param_id, pspec);
279 break;
280 }
281}
