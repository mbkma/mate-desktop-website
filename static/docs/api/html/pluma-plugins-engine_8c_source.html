60static void
61pluma_plugins_engine_init (PlumaPluginsEngine *engine)
62{
63 GError *error = NULL;
64 char *user_plugins_dir;
65
66 pluma_debug (DEBUG_PLUGINS);
67
68 peas_engine_enable_loader (PEAS_ENGINE (engine), "python3");
69
70 engine->priv = pluma_plugins_engine_get_instance_private (engine);
71
72 engine->priv->plugin_settings = g_settings_new (PLUMA_SCHEMA_ID);
73
74 /* This should be moved to libpeas */
75#ifdef HAVE_GIREPOSITORY_2
76 if (!gi_repository_require (gi_repository_dup_default (),
77 "Peas", "1.0", 0, &error))
78#else
79 if (!g_irepository_require (g_irepository_get_default (),
80 "Peas", "1.0", 0, &error))
81#endif
82 {
83 g_warning ("Could not load Peas repository: %s", error->message);
84 g_clear_error (&error);
85 }
86
87#ifdef HAVE_GIREPOSITORY_2
88 if (!gi_repository_require (gi_repository_dup_default (),
89 "PeasGtk", "1.0", 0, &error))
90#else
91 if (!g_irepository_require (g_irepository_get_default (),
92 "PeasGtk", "1.0", 0, &error))
93#endif
94 {
95 g_warning ("Could not load PeasGtk repository: %s", error->message);
96 g_clear_error (&error);
97 }
98
99#ifdef HAVE_GIREPOSITORY_2
100 if (!gi_repository_require_private (gi_repository_dup_default (),
101 LIBDIR "/girepository-1.0",
102 "Pluma", "1.0", 0, &error))
103#else
104 if (!g_irepository_require_private (g_irepository_get_default (),
105 LIBDIR "/girepository-1.0",
106 "Pluma", "1.0", 0, &error))
107#endif
108 {
109 g_warning ("Could not load Pluma repository: %s", error->message);
110 g_clear_error (&error);
111 }
112
113 user_plugins_dir = pluma_dirs_get_user_plugins_dir ();
114 peas_engine_add_search_path (PEAS_ENGINE (engine),
115 user_plugins_dir,
116 user_plugins_dir);
117 g_free (user_plugins_dir);
118
119 peas_engine_add_search_path (PEAS_ENGINE (engine),
120 PLUMA_LIBDIR "/plugins",
121 PLUMA_DATADIR "/plugins");
122
123 g_settings_bind (engine->priv->plugin_settings,
124 PLUMA_SETTINGS_ACTIVE_PLUGINS,
125 engine,
126 "loaded-plugins",
127 G_SETTINGS_BIND_DEFAULT);
128}

130static void
131pluma_plugins_engine_dispose (GObject *object)
132{
133 PlumaPluginsEngine *engine = PLUMA_PLUGINS_ENGINE (object);
134
135 if (engine->priv->plugin_settings != NULL)
136 {
137 g_object_unref (engine->priv->plugin_settings);
138 engine->priv->plugin_settings = NULL;
139 }
140
141 G_OBJECT_CLASS (pluma_plugins_engine_parent_class)->dispose (object);
142}

144static void
145pluma_plugins_engine_class_init (PlumaPluginsEngineClass *klass)
146{
147 GObjectClass *object_class = G_OBJECT_CLASS (klass);
148
149 object_class->dispose = pluma_plugins_engine_dispose;
150}

152PlumaPluginsEngine *
153pluma_plugins_engine_get_default (void)
154{
155 if (default_engine != NULL)
156 return default_engine;
157
158 default_engine = PLUMA_PLUGINS_ENGINE (g_object_new (PLUMA_TYPE_PLUGINS_ENGINE, NULL));
159 g_object_add_weak_pointer (G_OBJECT (default_engine),
160 (gpointer) &default_engine);
161
162 return default_engine;
163}
