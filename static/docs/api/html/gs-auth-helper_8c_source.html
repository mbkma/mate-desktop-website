16/* The idea here is to be able to run mate-screensaver-dialog without any setuid bits.
17 * Password verification happens through an external program that you feed
18 * your password to on stdin. The external command is invoked with a user
19 * name argument.
20 *
21 * The external helper does whatever authentication is necessary. Currently,
22 * SuSE uses "unix2_chkpwd", which is a variation of "unix_chkpwd" from the
23 * PAM distribution.
24 *
25 * Normally, the password helper should just authenticate the calling user
26 * (i.e. based on the caller's real uid). This is in order to prevent
27 * brute-forcing passwords in a shadow environment. A less restrictive
28 * approach would be to allow verifying other passwords as well, but always
29 * with a 2 second delay or so. (Not sure what SuSE's "unix2_chkpwd"
30 * currently does.)
31 * -- Olaf Kirch <okir@suse.de>, 16-Dec-2003
32 */

83static gboolean
84ext_run (const char *user,
85 GSAuthMessageFunc func,
86 gpointer data)
87{
88 int pfd[2], r_pfd[2], status;
89 pid_t pid;
90 gboolean verbose = gs_auth_get_verbose ();
91
92 if (pipe (pfd) < 0 || pipe (r_pfd) < 0)
93 {
94 return FALSE;
95 }
96
97 if (verbose)
98 {
99 g_message ("ext_run (%s, %s)",
100 PASSWD_HELPER_PROGRAM, user);
101 }
102
103 block_sigchld ();
104
105 if ((pid = fork ()) < 0)
106 {
107 close (pfd [0]);
108 close (pfd [1]);
109 close (r_pfd [0]);
110 close (r_pfd [1]);
111 return FALSE;
112 }
113
114 if (pid == 0)
115 {
116 close (pfd [1]);
117 close (r_pfd [0]);
118 if (pfd [0] != 0)
119 {
120 dup2 (pfd [0], 0);
121 }
122 if (r_pfd [1] != 1)
123 {
124 dup2 (r_pfd [1], 1);
125 }
126
127 /* Helper is invoked as helper service-name [user] */
128 execlp (PASSWD_HELPER_PROGRAM, PASSWD_HELPER_PROGRAM, "mate-screensaver", user, NULL);
129 if (verbose)
130 {
131 g_message ("%s: %s", PASSWD_HELPER_PROGRAM, g_strerror (errno));
132 }
133
134 exit (1);
135 }
136
137 close (pfd [0]);
138 close (r_pfd [1]);
139
140 gboolean ret = FALSE;
141 while (waitpid (pid, &status, WNOHANG) == 0)
142 {
143 int msg_type;
144 char buf[MAXLEN];
145 size_t msg_len = MAXLEN;
146
147 msg_type = read_prompt (r_pfd [0], buf, &msg_len);
148 if (0 == msg_type) continue;
149 if (msg_type < 0)
150 {
151 g_message ("Error reading prompt (%d)", msg_type);
152 ret = FALSE;
153 goto exit;
154 }
155
156 char *input = NULL;
157 func (msg_type, buf, &input, data);
158
159 unsigned int input_len = input ? strlen (input) : 0;
160 ssize_t wt;
161
162 wt = write_msg (pfd [1], input, input_len);
163 if (wt < 0)
164 {
165 g_message ("Error writing prompt reply (%li)", wt);
166 ret = FALSE;
167 goto exit;
168 }
169 }
170
171 close (pfd [1]);
172 close (r_pfd [0]);
173 unblock_sigchld ();
174
175 if (! WIFEXITED (status) || WEXITSTATUS (status) != 0)
176 {
177 ret = FALSE;
178 }
179 else
180 ret = TRUE;
181
182 exit:
183 return ret;
184}
