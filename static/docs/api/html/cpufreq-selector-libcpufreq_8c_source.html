83static guint
84cpufreq_selector_libcpufreq_get_valid_frequency (CPUFreqSelectorLibcpufreq *selector,
85 guint frequency)
86{
87 guint cpu;
88 gint dist = G_MAXINT;
89 guint retval = 0;
90 CPUFreqFrequencyList *freqs, *freq;
91
92 g_object_get (G_OBJECT (selector),
93 "cpu", &cpu,
94 NULL);
95
96 freqs = cpufreq_get_available_frequencies (cpu);
97 if (!freqs)
98 return 0;
99
100 for (freq = freqs; freq; freq = freq->next) {
101 guint current_dist;
102
103 if (freq->frequency == frequency) {
104 cpufreq_put_available_frequencies (freqs);
105
106 return frequency;
107 }
108
109 current_dist = abs ((int)freq->frequency - (int)frequency);
110 if (current_dist < dist) {
111 dist = current_dist;
112 retval = freq->frequency;
113 }
114 }
115
116 return retval;
117}

146static gboolean
147cpufreq_selector_libcpufreq_validate_governor (CPUFreqSelectorLibcpufreq *selector,
148 const gchar *governor)
149{
150 guint cpu;
151 CPUFreqGovernorList *govs, *gov;
152
153 g_object_get (G_OBJECT (selector),
154 "cpu", &cpu,
155 NULL);
156
157 govs = cpufreq_get_available_governors (cpu);
158 if (!govs)
159 return FALSE;
160
161 for (gov = govs; gov; gov = gov->next) {
162 if (g_ascii_strcasecmp (gov->governor, governor) == 0) {
163 cpufreq_put_available_governors (govs);
164
165 return TRUE;
166 }
167 }
168
169 cpufreq_put_available_governors (govs);
170
171 return FALSE;
172}

174static gboolean
175cpufreq_selector_libcpufreq_set_governor (CPUFreqSelector *selector,
176 const gchar *governor,
177 GError **error)
178{
179 CPUFreqSelectorLibcpufreq *selector_libcpufreq;
180 guint cpu;
181
182 selector_libcpufreq = CPUFREQ_SELECTOR_LIBCPUFREQ (selector);
183
184 if (!cpufreq_selector_libcpufreq_validate_governor (selector_libcpufreq, governor)) {
185 g_set_error (error,
186 CPUFREQ_SELECTOR_ERROR,
187 SELECTOR_ERROR_INVALID_GOVERNOR,
188 "Invalid governor '%s'",
189 governor);
190
191 return FALSE;
192 }
193
194 g_object_get (G_OBJECT (selector),
195 "cpu", &cpu,
196 NULL);
197
198 if (cpufreq_modify_policy_governor (cpu, (gchar *)governor) != 0) {
199 g_set_error (error,
200 CPUFREQ_SELECTOR_ERROR,
201 SELECTOR_ERROR_INVALID_GOVERNOR,
202 "Invalid governor '%s'",
203 governor);
204
205 return FALSE;
206 }
207
208 return TRUE;
