24struct tm *GetCurrentTime(void)
25{
26 time_t tt;
27 tzset();
28 tt=time(NULL);
29
30 return localtime(&tt);
31}

33void
34ta_refresh_time (TimeAdmin *ta, struct tm *LocalTime)
35{
36 gchar *str;
37
38 gtk_spin_button_set_value (GTK_SPIN_BUTTON (ta->HourSpin), LocalTime->tm_hour);
39 gtk_spin_button_set_value (GTK_SPIN_BUTTON (ta->MinuteSpin), LocalTime->tm_min);
40 str = g_strdup_printf ("%02d", LocalTime->tm_sec);
41 gtk_entry_set_text (GTK_ENTRY (ta->SecondSpin), str);
42 g_free (str);
43}

45void
46ta_refresh_date (TimeAdmin *ta, struct tm *LocalTime)
47{
48 gtk_calendar_select_month (GTK_CALENDAR (ta->Calendar),
49 LocalTime->tm_mon,
50 LocalTime->tm_year+1900);
51 gtk_calendar_select_day (GTK_CALENDAR (ta->Calendar),
52 LocalTime->tm_mday);
53 gtk_calendar_clear_marks (GTK_CALENDAR (ta->Calendar));
54 gtk_calendar_mark_day (GTK_CALENDAR (ta->Calendar),
55 LocalTime->tm_mday);
56}

98gboolean GetNtpState(TimeAdmin *ta)
99{
100 GDBusProxy *proxy = NULL;
101 GError *error = NULL;
102 GVariant *ret;
103 GVariant *ntp;
104
105 proxy = g_dbus_proxy_new_sync (ta->Connection,
106 G_DBUS_PROXY_FLAGS_NONE,
107 NULL,
108 "org.freedesktop.timedate1",
109 "/org/freedesktop/timedate1",
110 "org.freedesktop.DBus.Properties",
111 NULL,
112 &error);
113 if(proxy == NULL)
114 {
115 goto EXIT;
116 }
117
118 ret = g_dbus_proxy_call_sync (proxy,
119 "Get",
120 g_variant_new ("(ss)",
121 "org.freedesktop.timedate1",
122 "NTP"),
123 G_DBUS_CALL_FLAGS_NONE,
124 -1,
125 NULL,
126 &error);
127 if(ret == NULL)
128 {
129 goto EXIT;
130 }
131 g_variant_get (ret, "(v)", &ntp);
132 return g_variant_get_boolean (ntp);
133
134EXIT:
135 ErrorMessage (_("GetNtpState"), error->message);
136 g_error_free(error);
137 return FALSE;
138
139}

214gboolean ChangeNtpSync (GtkSwitch *widget, gboolean state, gpointer data)
215{
216 TimeAdmin *ta = (TimeAdmin *)data;
217 GError *error = NULL;
218 GVariant *ret;
219
220 ret = g_dbus_proxy_call_sync (ta->proxy,
221 "SetNTP",
222 g_variant_new ("(bb)",state,state),
223 G_DBUS_CALL_FLAGS_NONE,
224 -1,
225 NULL,
226 &error);
227
228 if (ret == NULL)
229 {
230 ErrorMessage (_("Set Ntp sync"), error->message);
231 g_error_free(error);
232 return TRUE;
233 }
234 else
235 {
236 ta->NtpState = state;
237 ChangeSpinBttonState (ta);
238 Update_Clock_Start (ta);
239 UpdateDate (ta);
240 }
241 return FALSE;
242}

244static guint GetTimeStamp(TimeAdmin *ta)
245{
246 guint year,month,day,hour,min,sec;
247 GDateTime *dt;
248 char *st;
249
250 gtk_calendar_get_date(GTK_CALENDAR(ta->Calendar),&year,&month,&day);
251 hour = gtk_spin_button_get_value(GTK_SPIN_BUTTON(ta->HourSpin));
252 min = gtk_spin_button_get_value(GTK_SPIN_BUTTON(ta->MinuteSpin));
253 sec = atoi(gtk_entry_get_text(GTK_ENTRY(ta->SecondSpin)));
254
255 dt = g_date_time_new_local(year,month+1,day,hour,min,sec);
256 st = g_date_time_format(dt,"%s");
257 return atoi(st);
258
259}
