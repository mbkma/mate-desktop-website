58static void
59launch_process (char **argv, GtkWindow *parent)
60{
61 GError *error;
62 GtkWidget *dialog;
63
64 error = NULL;
65 if (!g_spawn_async (NULL,
66 argv, NULL,
67 0,
68 NULL, NULL,
69 NULL,
70 &error)) {
71
72
73 dialog = gtk_message_dialog_new (NULL, GTK_DIALOG_MODAL, GTK_MESSAGE_WARNING,
74 GTK_BUTTONS_OK, _("Unable to launch the Personal File Sharing preferences"));
75
76 gtk_message_dialog_format_secondary_text (GTK_MESSAGE_DIALOG (dialog), "%s", error->message);
77
78 gtk_dialog_run (GTK_DIALOG (dialog));
79 gtk_widget_destroy (dialog);
80
81 g_error_free (error);
82 }
83}

131static GtkWidget *
132caja_user_share_get_location_widget (CajaLocationWidgetProvider *iface,
133 const char *uri,
134 GtkWidget *window)
135{
136 GFile *file;
137 GtkWidget *bar;
138 guint i;
139 gboolean enable = FALSE;
140 GFile *home;
141#ifdef HAVE_BLUETOOTH
142 const GUserDirectory special_dirs[] = { G_USER_DIRECTORY_PUBLIC_SHARE, G_USER_DIRECTORY_DOWNLOAD };
143
144 gboolean is_dir[] = { FALSE, FALSE };
145#else
146 const GUserDirectory special_dirs[] = { G_USER_DIRECTORY_PUBLIC_SHARE };
147
148 gboolean is_dir[] = { FALSE };
149#endif /* HAVE_BLUETOOTH */
150 file = g_file_new_for_uri (uri);
151 home = g_file_new_for_path (g_get_home_dir ());
152
153 /* We don't show anything in $HOME */
154 if (g_file_equal (home, file)) {
155 g_object_unref (home);
156 g_object_unref (file);
157 return NULL;
158 }
159
160 g_object_unref (home);
161
162 for (i = 0; i < G_N_ELEMENTS (special_dirs); i++) {
163 GFile *dir;
164 dir = lookup_dir_with_fallback (special_dirs[i]);
165 if (g_file_equal (dir, file)) {
166 enable = TRUE;
167 is_dir[i] = TRUE;
168 }
169 g_object_unref (dir);
170 }
171
172 g_object_unref (file);
173
174 if (enable == FALSE)
175 return NULL;
176
177 if (is_dir[0] != FALSE) {
178#ifdef HAVE_BLUETOOTH
179 if (is_dir[1] != FALSE) {
180 bar = caja_share_bar_new (_("May be used to share or receive files"));
181 } else {
182 bar = caja_share_bar_new (_("May be shared over the network or Bluetooth"));
183 }
184#else
185 bar = caja_share_bar_new (_("May be shared over the network"));
186#endif /* HAVE_BLUETOOTH */
187 } else {
188#ifdef HAVE_BLUETOOTH
189 BluetoothClient *client;
190
191 bar = caja_share_bar_new (_("May be used to receive files over Bluetooth"));
192
193 gtk_widget_set_no_show_all (bar, TRUE);
194 client = bluetooth_client_new ();
195 g_object_set_data_full (G_OBJECT (bar), "bluetooth-client", client, g_object_unref);
196 g_signal_connect (G_OBJECT (client), "notify::default-adapter-powered",
197 G_CALLBACK (default_adapter_powered_cb), bar);
198 downloads_bar_set_from_bluetooth_status (bar);
199#endif /* HAVE_BLUETOOTH */
200 }
201
202 g_signal_connect (bar, "response",
203 G_CALLBACK (bar_response_cb),
204 window);
205
206 gtk_widget_show_all (bar);
207
208 return bar;
209}
