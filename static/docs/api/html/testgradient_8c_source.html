31static void
32draw_checkerboard (cairo_t *cr,
33 int width,
34 int height)
35{
36 gint i, j, xcount, ycount;
37 GdkRGBA color1, color2;
38
39#define CHECK_SIZE 10
40#define SPACING 2
41
42 color1.red = 30000. / 65535.;
43 color1.green = 30000. / 65535.;
44 color1.blue = 30000. / 65535.;
45 color1.alpha = 1.0;
46
47 color2.red = 50000. / 65535.;
48 color2.green = 50000. / 65535.;
49 color2.blue = 50000. / 65535.;
50 color2.alpha = 1.0;
51
52 xcount = 0;
53 i = SPACING;
54 while (i < width)
55 {
56 j = SPACING;
57 ycount = xcount % 2; /* start with even/odd depending on row */
58 while (j < height)
59 {
60 if (ycount % 2)
61 gdk_cairo_set_source_rgba (cr, &color1);
62 else
63 gdk_cairo_set_source_rgba (cr, &color2);
64
65 /* If we're outside event->area, this will do nothing.
66 * It might be mildly more efficient if we handled
67 * the clipping ourselves, but again we're feeling lazy.
68 */
69 cairo_rectangle (cr, i, j, CHECK_SIZE, CHECK_SIZE);
70 cairo_fill (cr);
71
72 j += CHECK_SIZE + SPACING;
73 ++ycount;
74 }
75
76 i += CHECK_SIZE + SPACING;
77 ++xcount;
78 }
79}

81static void
82render_simple (
83 cairo_t *cr,
84 int width, int height,
85 MetaGradientType type,
86 gboolean with_alpha)
87{
88 GdkPixbuf *pixbuf;
89 GdkRGBA from, to;
90
91 gdk_rgba_parse (&from, "blue");
92 gdk_rgba_parse (&to, "green");
93
94 pixbuf = meta_gradient_create_simple (width, height,
95 &from, &to,
96 type);
97
98 if (with_alpha)
99 {
100 const unsigned char alphas[] = { 0xff, 0xaa, 0x2f, 0x0, 0xcc, 0xff, 0xff };
101
102 if (!gdk_pixbuf_get_has_alpha (pixbuf))
103 {
104 GdkPixbuf *new_pixbuf;
105
106 new_pixbuf = gdk_pixbuf_add_alpha (pixbuf, FALSE, 0, 0, 0);
107 g_object_unref (G_OBJECT (pixbuf));
108 pixbuf = new_pixbuf;
109 }
110
111 meta_gradient_add_alpha (pixbuf,
112 alphas, G_N_ELEMENTS (alphas),
113 META_GRADIENT_HORIZONTAL);
114
115 draw_checkerboard (cr, width, height);
116 }
117
118 gdk_cairo_set_source_pixbuf (cr, pixbuf, 0, 0);
119 cairo_rectangle (cr, 0, 0, width, height);
120 cairo_fill (cr);
121
122 g_object_unref (G_OBJECT (pixbuf));
123}

236static gboolean
237draw_callback (GtkWidget *widget,
238 cairo_t *cr,
239 gpointer data)
240{
241 RenderGradientFunc func = data;
242 GtkAllocation allocation;
243 GtkStyleContext *style;
244 GdkRGBA color;
245
246 style = gtk_widget_get_style_context (widget);
247
248 gtk_style_context_save (style);
249 gtk_style_context_set_state (style, gtk_widget_get_state_flags (widget));
250 gtk_style_context_lookup_color (style, "foreground-color", &color);
251 gtk_style_context_restore (style);
252
253 gtk_widget_get_allocation (widget, &allocation);
254
255 cairo_set_source_rgba (cr, color.red, color.green, color.blue, color.alpha);
256
257 (* func) (
258 cr,
259 allocation.width,
260 allocation.height);
261
262 return FALSE;
263}

265static GtkWidget*
266create_gradient_window (const char *title,
267 RenderGradientFunc func)
268{
269 GtkWidget *window;
270 GtkWidget *drawing_area;
271
272 window = gtk_window_new (GTK_WINDOW_TOPLEVEL);
273
274 gtk_window_set_title (GTK_WINDOW (window), title);
275
276 drawing_area = gtk_drawing_area_new ();
277
278 gtk_widget_set_size_request (drawing_area, 1, 1);
279
280 gtk_window_set_default_size (GTK_WINDOW (window), 175, 175);
281
282 g_signal_connect (G_OBJECT (drawing_area),
283 "draw",
284 G_CALLBACK (draw_callback),
285 func);
286
287 gtk_container_add (GTK_CONTAINER (window), drawing_area);
288
289 gtk_widget_show_all (window);
290
291 return window;
292}
