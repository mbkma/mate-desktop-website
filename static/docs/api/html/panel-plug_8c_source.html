36static gboolean
37panel_plug_draw (GtkWidget *widget,
38 cairo_t *cr)
39{
40 GdkWindow *window;
41 cairo_pattern_t *pattern;
42
43 if (!gtk_widget_get_realized (widget))
44 return GTK_WIDGET_CLASS (panel_plug_parent_class)->draw (widget, cr);
45
46 window = gtk_widget_get_window (widget);
47 pattern = gdk_window_get_background_pattern (window);
48
49 if (!pattern)
50 {
51 GtkStyleContext *context;
52 gint width;
53 gint height;
54
55 context = gtk_widget_get_style_context (widget);
56 width = gtk_widget_get_allocated_width (widget);
57 height = gtk_widget_get_allocated_height (widget);
58
59 gtk_render_background (context, cr, 0, 0, width, height);
60 }
61
62 return GTK_WIDGET_CLASS (panel_plug_parent_class)->draw (widget, cr);
63}

65static void
66panel_plug_realize (GtkWidget *widget)
67{
68 GdkScreen *screen;
69 GdkVisual *visual;
70
71 screen = gdk_screen_get_default ();
72 visual = gdk_screen_get_rgba_visual (screen);
73
74 if (!visual)
75 visual = gdk_screen_get_system_visual (screen);
76
77 gtk_widget_set_visual (widget, visual);
78
79 GTK_WIDGET_CLASS (panel_plug_parent_class)->realize (widget);
80}
