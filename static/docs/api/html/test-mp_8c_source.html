119static void
120try(const char *string, bool result, bool expected)
121{
122 if ((result && !expected) || (!result && expected))
123 fail("%s -> %s, expected %s", string, expected ? "true" : "false", result ? "true" : "false");
124 else
125 pass("%s -> %s", string, result ? "true" : "false");
126}
