59static gpointer
60_check_strptime_updates_wday (gpointer data)
61{
62 struct tm tm;
63
64 memset (&tm, '\0', sizeof (tm));
65 strptime ("2008:12:24 20:30:45", "%Y:%m:%d %T", &tm);

78static void
79_calculate_wday_yday (struct tm *tm)
80{
81 GDate *exif_date;
82 struct tm tmp_tm;
83
84 exif_date = g_date_new_dmy (tm->tm_mday,
85 tm->tm_mon+1,
86 tm->tm_year+1900);
87
88 g_return_if_fail (exif_date != NULL && g_date_valid (exif_date));
89
90 // Use this to get GLib <-> libc corrected values
91 g_date_to_struct_tm (exif_date, &tmp_tm);
92 g_date_free (exif_date);
93
94 tm->tm_wday = tmp_tm.tm_wday;
95 tm->tm_yday = tmp_tm.tm_yday;
96}

104#ifdef HAVE_STRPTIME
105static gchar *
106eom_exif_util_format_date_with_strptime (const gchar *date, const gchar* format)
107{
108 static GOnce strptime_updates_wday = G_ONCE_INIT;
109 gchar *new_date = NULL;
110 gchar tmp_date[DATE_BUF_SIZE];
111 gchar *p;
112 gsize dlen;
113 struct tm tm;
114
115 memset (&tm, '\0', sizeof (tm));
116 p = strptime (date, "%Y:%m:%d %T", &tm);
117
118 if (p == date + strlen (date)) {
119 g_once (&strptime_updates_wday,
120 _check_strptime_updates_wday,
121 NULL);
122
123 // Ensure tm.tm_wday and tm.tm_yday are set
124 if (!GPOINTER_TO_BOOLEAN (strptime_updates_wday.retval))
125 _calculate_wday_yday (&tm);
126
127 /* A strftime-formatted string, to display the date the image was taken. */
128 dlen = strftime (tmp_date, DATE_BUF_SIZE * sizeof(gchar), format, &tm);
129 new_date = g_strndup (tmp_date, dlen);
130 }
131
132 return new_date;
133}
134#else
135static gchar *
136eom_exif_util_format_date_by_hand (const gchar *date, const gchar* format)
137{
138 int year, month, day, hour, minutes, seconds;
139 int result;
140 gchar *new_date = NULL;
141
142 result = sscanf (date, "%d:%d:%d %d:%d:%d",
143 &year, &month, &day, &hour, &minutes, &seconds);
144
145 if (result < 3 || !g_date_valid_dmy (day, month, year)) {
146 return NULL;
147 } else {
148 gchar tmp_date[DATE_BUF_SIZE];
149 gsize dlen;
150 struct tm tm;
151
152 memset (&tm, '\0', sizeof (tm));
153 tm.tm_mday = day;
154 tm.tm_mon = month-1;
155 tm.tm_year = year-1900;
156 // Calculate tm.tm_wday
157 _calculate_wday_yday (&tm);
158
159 tm.tm_sec = result < 6 ? 0 : seconds;
160 tm.tm_min = result < 5 ? 0 : minutes;
161 tm.tm_hour = result < 4 ? 0 : hour;
162 dlen = strftime (tmp_date, DATE_BUF_SIZE * sizeof(gchar), format, &tm);
163
164 if (dlen == 0)
165 return NULL;
166 else
167 new_date = g_strndup (tmp_date, dlen);
168 }
169 return new_date;
170}
171#endif /* HAVE_STRPTIME */

200 EomExifData *exif_data,
201 gint tag_id)
202{
203 gchar exif_buffer[512];
204 const gchar *buf_ptr;
205 gchar *label_text = NULL;
206
207 g_return_if_fail (GTK_IS_LABEL (label));
208
209 if (exif_data) {
210 buf_ptr = eom_exif_data_get_value (exif_data, tag_id,
211 exif_buffer, 512);
212
213 if (tag_id == EXIF_TAG_DATE_TIME_ORIGINAL && buf_ptr)
214 label_text = eom_exif_util_format_date (buf_ptr);
215 else
216 label_text = eom_util_make_valid_utf8 (buf_ptr);
217 }
218
219 gtk_label_set_text (label, label_text);
220 g_free (label_text);

250void
251eom_exif_util_set_focal_length_label_text (GtkLabel *label,
252 EomExifData *exif_data)
253{
254 ExifEntry *entry = NULL, *entry35mm = NULL;
255 ExifByteOrder byte_order;
256 gfloat f_val = 0.0;
257 gchar *fl_text = NULL,*fl35_text = NULL;
258
259 /* If no ExifData is supplied the label will be
260 * cleared later as fl35_text is NULL. */
261 if (exif_data != NULL) {
262 entry = exif_data_get_entry (exif_data, EXIF_TAG_FOCAL_LENGTH);
263 entry35mm = exif_data_get_entry (exif_data,
264 EXIF_TAG_FOCAL_LENGTH_IN_35MM_FILM);
265 byte_order = exif_data_get_byte_order (exif_data);
266 }
267
268 if (entry && G_LIKELY (entry->format == EXIF_FORMAT_RATIONAL)) {
269 ExifRational value;
270
271 /* Decode value by hand as libexif is not necessarily returning
272 * it in the format we want it to be.
273 */
274 value = exif_get_rational (entry->data, byte_order);
275 /* Guard against div by zero */
276 if (G_LIKELY(value.denominator != 0))
277 f_val = (gfloat)value.numerator/
278 (gfloat)value.denominator;
279
280 /* TRANSLATORS: This is the actual focal length used when
281 the image was taken.*/
282 fl_text = g_strdup_printf (_("%.1f (lens)"), f_val);
283
284 }
285 if (entry35mm && G_LIKELY (entry35mm->format == EXIF_FORMAT_SHORT)) {
286 ExifShort s_val;
287
288 s_val = exif_get_short (entry35mm->data, byte_order);
289
290 /* Print as float to get a similar look as above. */
291 /* TRANSLATORS: This is the equivalent focal length assuming
292 a 35mm film camera. */
293 fl35_text = g_strdup_printf(_("%.1f (35mm film)"),(float)s_val);
294 }
295
296 if (fl_text) {
297 if (fl35_text) {
298 gchar *merged_txt;
299
300 merged_txt = g_strconcat (fl35_text,", ", fl_text, NULL);
301 gtk_label_set_text (label, merged_txt);
302 g_free (merged_txt);
303 } else {
304 gtk_label_set_text (label, fl_text);
305 }
306 } else {
307 /* This will also clear the label if no ExifData was supplied */
308 gtk_label_set_text (label, fl35_text);
309 }
310
311 g_free (fl35_text);
312 g_free (fl_text);
313}
