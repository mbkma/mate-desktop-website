33static char *
34uid_gid_string (uid_t uid,
35 gid_t gid)
36{
37 static char *buf;
38 struct passwd *p = NULL;
39 struct group *g = NULL;
40
41 p = getpwuid (uid);
42 g = getgrgid (gid);
43
44 buf = g_strdup_printf ("%s/%s (%ld/%ld)",
45 (p && p->pw_name ? p->pw_name : "???"),
46 (g && g->gr_name ? g->gr_name : "???"),
47 (long) uid, (long) gid);
48
49 return buf;
50}

52static gboolean
53set_ids_by_number (uid_t uid,
54 gid_t gid,
55 char **message_ret)
56{
57 int uid_errno = 0;
58 int gid_errno = 0;
59 int sgs_errno = 0;
60 struct passwd *p = getpwuid (uid);
61 struct group *g = getgrgid (gid);
62
63 if (message_ret)
64 *message_ret = NULL;
65
66 /* Rumor has it that some implementations of of setuid() do nothing
67 when called with -1; therefore, if the "nobody" user has a uid of
68 -1, then that would be Really Bad. Rumor further has it that such
69 systems really ought to be using -2 for "nobody", since that works.
70 So, if we get a uid (or gid, for good measure) of -1, switch to -2
71 instead. Note that this must be done after we've looked up the
72 user/group names with getpwuid(-1) and/or getgrgid(-1).
73 */
74 if (gid == (gid_t) -1) gid = (gid_t) -2;
75 if (uid == (uid_t) -1) uid = (uid_t) -2;
76
77 errno = 0;
78 if (setgroups (1, &gid) < 0)
79 sgs_errno = errno ? errno : -1;
80
81 errno = 0;
82 if (setgid (gid) != 0)
83 gid_errno = errno ? errno : -1;
84
85 errno = 0;
86 if (setuid (uid) != 0)
87 uid_errno = errno ? errno : -1;
88
89 if (uid_errno == 0 && gid_errno == 0 && sgs_errno == 0)
90 {
91 static char *reason;
92 reason = g_strdup_printf ("changed uid/gid to %s/%s (%ld/%ld).",
93 (p && p->pw_name ? p->pw_name : "???"),
94 (g && g->gr_name ? g->gr_name : "???"),
95 (long) uid, (long) gid);
96 if (message_ret)
97 *message_ret = g_strdup (reason);
98
99 g_free (reason);
100
101 return TRUE;
102 }
103 else
104 {
105 char *reason = NULL;
106
107 if (sgs_errno)
108 {
109 reason = g_strdup_printf ("couldn't setgroups to %s (%ld)",
110 (g && g->gr_name ? g->gr_name : "???"),
111 (long) gid);
112 if (sgs_errno == -1)
113 fprintf (stderr, "%s: unknown error\n", reason);
114 else
115 {
116 errno = sgs_errno;
117 perror (reason);
118 }
119 g_free (reason);
120 reason = NULL;
121 }
122
123 if (gid_errno)
124 {
125 reason = g_strdup_printf ("couldn't set gid to %s (%ld)",
126 (g && g->gr_name ? g->gr_name : "???"),
127 (long) gid);
128 if (gid_errno == -1)
129 fprintf (stderr, "%s: unknown error\n", reason);
130 else
131 {
132 errno = gid_errno;
133 perror (reason);
134 }
135 g_free (reason);
136 reason = NULL;
137 }
138
139 if (uid_errno)
140 {
141 reason = g_strdup_printf ("couldn't set uid to %s (%ld)",
142 (p && p->pw_name ? p->pw_name : "???"),
143 (long) uid);
144 if (uid_errno == -1)
145 fprintf (stderr, "%s: unknown error\n", reason);
146 else
147 {
148 errno = uid_errno;
149 perror (reason);
150 }
151 g_free (reason);
152 reason = NULL;
153 }
154 return FALSE;
155 }
156 return FALSE;
157}

159/* If we've been run as setuid or setgid to someone else (most likely root)
160 turn off the extra permissions so that random user-specified programs
161 don't get special privileges. (On some systems it is necessary to install
162 this program as setuid root in order to read the passwd file to implement
163 lock-mode.)

165 *** WARNING: DO NOT DISABLE ANY OF THE FOLLOWING CODE!
166 If you do so, you will open a security hole. See the sections
167 of the xscreensaver manual titled "LOCKING AND ROOT LOGINS",

172gboolean
173hack_uid (char **nolock_reason,
174 char **orig_uid,
175 char **uid_message)
176{
177 char *reason;
178 gboolean ret;
179
180 ret = TRUE;
181 reason = NULL;
182
183 if (nolock_reason != NULL)
184 {
185 *nolock_reason = NULL;
186 }
187 if (orig_uid != NULL)
188 {
189 *orig_uid = NULL;
190 }
191 if (uid_message != NULL)
192 {
193 *uid_message = NULL;
194 }
195
196 /* Discard privileges, and set the effective user/group ids to the
197 real user/group ids. That is, give up our "chmod +s" rights.
198 */
199 {
200 uid_t euid = geteuid ();
201 gid_t egid = getegid ();
202 uid_t uid = getuid ();
203 gid_t gid = getgid ();
204
205 if (orig_uid != NULL)
206 {
207 *orig_uid = uid_gid_string (euid, egid);
208 }
209
210 if (uid != euid || gid != egid)
211 {
212#ifdef HAVE_BSDAUTH /* we need to setgid auth to run the bsd_auth(3) login_* helpers */
213 struct group *authg = getgrnam("auth");
214 if (!authg || !authg->gr_name || !*authg->gr_name) {
215 reason = g_strdup ("no such group as \"auth\" for bsdauth.");
216 ret = FALSE;
217 goto out;
218 }
219 gid = authg->gr_gid;
220#endif /* !HAVE_BSDAUTH */
221 if (! set_ids_by_number (uid, gid, uid_message))
222 {
223 reason = g_strdup ("unable to discard privileges.");
224
225 ret = FALSE;
226 goto out;
227 }
228 }
229 }
230
231 /* Locking can't work when running as root, because we have no way of
232 knowing what the user id of the logged in user is (so we don't know
233 whose password to prompt for.)
234
235 *** WARNING: DO NOT DISABLE THIS CODE!
236 If you do so, you will open a security hole. See the sections
237 of the xscreensaver manual titled "LOCKING AND ROOT LOGINS",
238 and "USING XDM".
239 */
240 if (getuid () == (uid_t) 0)
241 {
242 reason = g_strdup ("running as root");
243 ret = FALSE;
244 goto out;
245 }
246
247out:
248 if (nolock_reason != NULL)
249 {
250 *nolock_reason = g_strdup (reason);
251 }
252 g_free (reason);
253
254 return ret;
255}
