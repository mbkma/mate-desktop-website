63static void
64ev_sidebar_layers_dispose (GObject *object)
65{
66 EvSidebarLayers *sidebar = EV_SIDEBAR_LAYERS (object);
67
68 if (sidebar->priv->job) {
69 g_signal_handlers_disconnect_by_func (sidebar->priv->job,
70 job_finished_callback,
71 sidebar);
72 ev_job_cancel (sidebar->priv->job);
73 g_object_unref (sidebar->priv->job);
74 sidebar->priv->job = NULL;
75 }
76
77 if (sidebar->priv->document) {
78 g_object_unref (sidebar->priv->document);
79 sidebar->priv->document = NULL;
80 }
81
82 G_OBJECT_CLASS (ev_sidebar_layers_parent_class)->dispose (object);
83}

177static void
178ev_sidebar_layers_visibility_changed (GtkCellRendererToggle *cell,
179 gchar *path_str,
180 EvSidebarLayers *ev_layers)
181{
182 GtkTreeModel *model;
183 GtkTreePath *path;
184 GtkTreeIter iter;
185 gboolean visible;
186 EvLayer *layer;
187
188 model = gtk_tree_view_get_model (ev_layers->priv->tree_view);
189
190 path = gtk_tree_path_new_from_string (path_str);
191 gtk_tree_model_get_iter (model, &iter, path);
192 gtk_tree_model_get (model, &iter,
193 EV_DOCUMENT_LAYERS_COLUMN_VISIBLE, &visible,
194 EV_DOCUMENT_LAYERS_COLUMN_LAYER, &layer,
195 -1);
196
197 visible = !visible;
198 if (visible) {
199 gint rb_group;
200
201 ev_document_layers_show_layer (EV_DOCUMENT_LAYERS (ev_layers->priv->document),
202 layer);
203
204 rb_group = ev_layer_get_rb_group (layer);
205 if (rb_group) {
206 gtk_tree_model_foreach (model,
207 (GtkTreeModelForeachFunc)clear_rb_group,
208 &rb_group);
209 }
210 } else {
211 ev_document_layers_hide_layer (EV_DOCUMENT_LAYERS (ev_layers->priv->document),
212 layer);
213 }
214
215 gtk_tree_store_set (GTK_TREE_STORE (model), &iter,
216 EV_DOCUMENT_LAYERS_COLUMN_VISIBLE, visible,
217 -1);
218
219 if (ev_layer_is_parent (layer)) {
220 gtk_tree_model_foreach (model,
221 (GtkTreeModelForeachFunc)update_kids,
222 &iter);
223 }
224
225 gtk_tree_path_free (path);
226
227 g_signal_emit (ev_layers, signals[LAYERS_VISIBILITY_CHANGED], 0);
228}

232{

241

243
244 renderer = gtk_cell_renderer_toggle_new ();

248 "activatable", EV_DOCUMENT_LAYERS_COLUMN_ENABLED,
249 "visible", EV_DOCUMENT_LAYERS_COLUMN_SHOWTOGGLE,
250 "sensitive", EV_DOCUMENT_LAYERS_COLUMN_ENABLED,
251 NULL);
252 g_object_set (G_OBJECT (renderer),
253 "xpad", 4,
254 "ypad", 0,
255 NULL);
256 g_signal_connect (renderer, "toggled",
257 G_CALLBACK (ev_sidebar_layers_visibility_changed),
258 (gpointer)ev_layers);
259
260 renderer = gtk_cell_renderer_text_new ();
261 gtk_tree_view_column_pack_start (column, renderer, TRUE);
262 gtk_tree_view_column_set_attributes (column, renderer,

330static void
331update_layers_state (GtkTreeModel *model,
332 GtkTreeIter *iter,
333 EvDocumentLayers *document_layers)
334{
335 EvLayer *layer;
336 gboolean visible;
337 GtkTreeIter child_iter;
338
339 do {
340 gtk_tree_model_get (model, iter,
341 EV_DOCUMENT_LAYERS_COLUMN_VISIBLE, &visible,
342 EV_DOCUMENT_LAYERS_COLUMN_LAYER, &layer,
343 -1);
344 if (layer) {
345 gboolean layer_visible;
346
347 layer_visible = ev_document_layers_layer_is_visible (document_layers, layer);
348 if (layer_visible != visible) {
349 gtk_tree_store_set (GTK_TREE_STORE (model), iter,
350 EV_DOCUMENT_LAYERS_COLUMN_VISIBLE, layer_visible,
351 -1);
352 }
353 }
354
355 if (gtk_tree_model_iter_children (model, &child_iter, iter))
356 update_layers_state (model, &child_iter, document_layers);
357 } while (gtk_tree_model_iter_next (model, iter));
358}

360void
361ev_sidebar_layers_update_layers_state (EvSidebarLayers *sidebar_layers)
362{
363 GtkTreeModel *model;
364 GtkTreeIter iter;
365 EvDocumentLayers *document_layers;
366
367 document_layers = EV_DOCUMENT_LAYERS (sidebar_layers->priv->document);
368 model = gtk_tree_view_get_model (GTK_TREE_VIEW (sidebar_layers->priv->tree_view));
369 if (gtk_tree_model_get_iter_first (model, &iter))
370 update_layers_state (model, &iter, document_layers);
371}

387static void
388ev_sidebar_layers_document_changed_cb (EvDocumentModel *model,
389 GParamSpec *pspec,
390 EvSidebarLayers *sidebar_layers)
391{
392 EvDocument *document = ev_document_model_get_document (model);
393 EvSidebarLayersPrivate *priv = sidebar_layers->priv;
394
395 if (!EV_IS_DOCUMENT_LAYERS (document))
396 return;
397
398 if (priv->document) {
399 gtk_tree_view_set_model (GTK_TREE_VIEW (priv->tree_view), NULL);
400 g_object_unref (priv->document);
401 }
402
403 priv->document = g_object_ref (document);
404
405 if (priv->job) {
406 g_signal_handlers_disconnect_by_func (priv->job,
407 job_finished_callback,
408 sidebar_layers);
409 g_object_unref (priv->job);
410 }
411
412 priv->job = ev_job_layers_new (document);
413 g_signal_connect (priv->job, "finished",
414 G_CALLBACK (job_finished_callback),
415 sidebar_layers);
416 /* The priority doesn't matter for this job */
417 ev_job_scheduler_push_job (priv->job, EV_JOB_PRIORITY_NONE);
418}
