61static gboolean parse_arguments (int *argc, char ***argv)
62{
63 GOptionContext *context;
64 GError *error;
65
66 context = g_option_context_new (NULL);
67
68 g_option_context_add_main_entries (context, entries, NULL);
69
70 error = NULL;
71 if (g_option_context_parse (context, argc, argv, &error) == FALSE)
72 {
73 g_option_context_free (context);
74 g_warning ("Failed to parse command line arguments: %s", error->message);
75 g_error_free (error);
76
77 return FALSE;
78 }
79
80 g_option_context_free (context);
81
82 if (debug)
83 g_setenv ("G_MESSAGES_DEBUG", "all", FALSE);
84
85 return TRUE;
86}
