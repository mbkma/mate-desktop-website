33char*
34meta_rectangle_to_string (const MetaRectangle *rect,
35 char *output)
36{
37 /* 25 chars: 2 commas, space, plus, trailing \0 + 5 for each digit.
38 * Should be more than enough space. Note that of this space, the
39 * trailing \0 will be overwritten for all but the last rectangle.
40 */
41 g_snprintf (output, RECT_LENGTH, "%d,%d +%d,%d",
42 rect->x, rect->y, rect->width, rect->height);
43
44 return output;
45}

47char*
48meta_rectangle_region_to_string (GList *region,
49 const char *separator_string,
50 char *output)
51{
52 /* 27 chars: 2 commas, 2 square brackets, space, plus, trailing \0 + 5
53 * for each digit. Should be more than enough space. Note that of this
54 * space, the trailing \0 will be overwritten for all but the last
55 * rectangle.
56 */
57 char rect_string[RECT_LENGTH];
58
59 GList *tmp = region;
60 char *cur = output;
61
62 if (region == NULL)
63 g_snprintf (output, 10, "(EMPTY)");
64
65 while (tmp)
66 {
67 MetaRectangle *rect = tmp->data;
68 g_snprintf (rect_string, RECT_LENGTH, "[%d,%d +%d,%d]",
69 rect->x, rect->y, rect->width, rect->height);
70 cur = g_stpcpy (cur, rect_string);
71 tmp = tmp->next;
72 if (tmp)
73 cur = g_stpcpy (cur, separator_string);
74 }
75
76 return output;
77}

79char*
80meta_rectangle_edge_to_string (const MetaEdge *edge,
81 char *output)
82{
83 /* 25 chars: 2 commas, space, plus, trailing \0 + 5 for each digit.
84 * Should be more than enough space. Note that of this space, the
85 * trailing \0 will be overwritten for all but the last rectangle.

87 * Plus 2 for parenthesis, 4 for 2 more numbers, 2 more commas, and

90 g_snprintf (output, EDGE_LENGTH, "[%d,%d +%d,%d], %2d, %2d",
91 edge->rect.x, edge->rect.y, edge->rect.width, edge->rect.height,
92 edge->side_type, edge->edge_type);
93
94 return output;

153gboolean
154meta_rectangle_intersect (const MetaRectangle *src1,
155 const MetaRectangle *src2,
156 MetaRectangle *dest)
157{
158 int dest_x, dest_y;
159 int dest_w, dest_h;
160 int return_val;
161
162 g_return_val_if_fail (src1 != NULL, FALSE);
163 g_return_val_if_fail (src2 != NULL, FALSE);
164 g_return_val_if_fail (dest != NULL, FALSE);
165
166 return_val = FALSE;
167
168 dest_x = MAX (src1->x, src2->x);
169 dest_y = MAX (src1->y, src2->y);
170 dest_w = MIN (src1->x + src1->width, src2->x + src2->width) - dest_x;
171 dest_h = MIN (src1->y + src1->height, src2->y + src2->height) - dest_y;
172
173 if (dest_w > 0 && dest_h > 0)
174 {
175 dest->x = dest_x;
176 dest->y = dest_y;
177 dest->width = dest_w;
178 dest->height = dest_h;
179 return_val = TRUE;
180 }
181 else
182 {
183 dest->width = 0;
184 dest->height = 0;
185 }
186
187 return return_val;
188}

200void
201meta_rectangle_union (const MetaRectangle *rect1,
202 const MetaRectangle *rect2,
203 MetaRectangle *dest)
204{
205 int dest_x, dest_y;
206 int dest_w, dest_h;
207
208 dest_x = rect1->x;
209 dest_y = rect1->y;
210 dest_w = rect1->width;
211 dest_h = rect1->height;
212
213 if (rect2->x < dest_x)
214 {
215 dest_w += dest_x - rect2->x;
216 dest_x = rect2->x;
217 }
218 if (rect2->y < dest_y)
219 {
220 dest_h += dest_y - rect2->y;
221 dest_y = rect2->y;
222 }
223 if (rect2->x + rect2->width > dest_x + dest_w)
224 dest_w = rect2->x + rect2->width - dest_x;
225 if (rect2->y + rect2->height > dest_y + dest_h)
226 dest_h = rect2->y + rect2->height - dest_y;
227
228 dest->x = dest_x;
229 dest->y = dest_y;
230 dest->width = dest_w;
231 dest->height = dest_h;
232}

282void
283meta_rectangle_resize_with_gravity (const MetaRectangle *old_rect,
284 MetaRectangle *rect,
285 int gravity,
286 int new_width,
287 int new_height)
288{
289 /* FIXME: I'm too deep into this to know whether the below comment is
290 * still clear or not now that I've moved it out of constraints.c.
291 * boxes.h has a good comment, but I'm not sure if the below info is also
292 * helpful on top of that (or whether it has superfluous info).
293 */
294
295 /* These formulas may look overly simplistic at first but you can work
296 * everything out with a left_frame_with, right_frame_width,
297 * border_width, and old and new client area widths (instead of old total
298 * width and new total width) and you come up with the same formulas.
299 *
300 * Also, note that the reason we can treat NorthWestGravity and
301 * StaticGravity the same is because we're not given a location at
302 * which to place the window--the window was already placed
303 * appropriately before. So, NorthWestGravity for this function
304 * means to just leave the upper left corner of the outer window
305 * where it already is, and StaticGravity for this function means to
306 * just leave the upper left corner of the inner window where it
307 * already is. But leaving either of those two corners where they
308 * already are will ensure that the other corner is fixed as well
309 * (since frame size doesn't change)--thus making the two
310 * equivalent.
311 */
312
313 /* First, the x direction */
314 switch (gravity)
315 {
316 case NorthWestGravity:
317 case WestGravity:
318 case SouthWestGravity:
319 rect->x = old_rect->x;
320 break;
321
322 case NorthGravity:
323 case CenterGravity:
324 case SouthGravity:
325 /* FIXME: Needing to adjust new_width kind of sucks, but not doing so
326 * would cause drift.
327 */
328 new_width -= (old_rect->width - new_width) % 2;
329 rect->x = old_rect->x + (old_rect->width - new_width)/2;
330 break;
331
332 case NorthEastGravity:
333 case EastGravity:
334 case SouthEastGravity:
335 rect->x = old_rect->x + (old_rect->width - new_width);
336 break;
337
338 case StaticGravity:
339 default:
340 rect->x = old_rect->x;
341 break;
342 }
343 rect->width = new_width;
344
345 /* Next, the y direction */
346 switch (gravity)
347 {
348 case NorthWestGravity:
349 case NorthGravity:
350 case NorthEastGravity:
351 rect->y = old_rect->y;
352 break;
353
354 case WestGravity:
355 case CenterGravity:
356 case EastGravity:
357 /* FIXME: Needing to adjust new_height kind of sucks, but not doing so
358 * would cause drift.
359 */
360 new_height -= (old_rect->height - new_height) % 2;
361 rect->y = old_rect->y + (old_rect->height - new_height)/2;
362 break;
363
364 case SouthWestGravity:
365 case SouthGravity:
366 case SouthEastGravity:
367 rect->y = old_rect->y + (old_rect->height - new_height);
368 break;
369
370 case StaticGravity:
371 default:
372 rect->y = old_rect->y;
373 break;
374 }
375 rect->height = new_height;
376}

378/* Not so simple helper function for get_minimal_spanning_set_for_region() */
379static GList*
380merge_spanning_rects_in_region (GList *region)
381{
382 /* NOTE FOR ANY OPTIMIZATION PEOPLE OUT THERE: Please see the
383 * documentation of get_minimal_spanning_set_for_region() for performance
384 * considerations that also apply to this function.
385 */
386
387 GList* compare;
388 compare = region;
389
390 if (region == NULL)
391 {
392 meta_warning ("Region to merge was empty! Either you have a some "
393 "pathological STRUT list or there's a bug somewhere!\n");
394 return NULL;
395 }
396
397 while (compare && compare->next)
398 {
399 MetaRectangle *a = compare->data;
400 GList *other = compare->next;
401
402 g_assert (a->width > 0 && a->height > 0);
403
404 while (other)
405 {
406 MetaRectangle *b = other->data;
407 GList *delete_me = NULL;
408
409 g_assert (b->width > 0 && b->height > 0);
410
411 /* If a contains b, just remove b */
412 if (meta_rectangle_contains_rect (a, b))
413 {
414 delete_me = other;
415 }
416 /* If b contains a, just remove a */
417 else if (meta_rectangle_contains_rect (b, a))
418 {
419 delete_me = compare;
420 }
421 /* If a and b might be mergeable horizontally */
422 else if (a->y == b->y && a->height == b->height)
423 {
424 /* If a and b overlap */
425 if (meta_rectangle_overlap (a, b))
426 {
427 int new_x = MIN (a->x, b->x);
428 a->width = MAX (a->x + a->width, b->x + b->width) - new_x;
429 a->x = new_x;
430 delete_me = other;
431 }
432 /* If a and b are adjacent */
433 else if (a->x + a->width == b->x || a->x == b->x + b->width)
434 {
435 int new_x = MIN (a->x, b->x);
436 a->width = MAX (a->x + a->width, b->x + b->width) - new_x;
437 a->x = new_x;
438 delete_me = other;
439 }
440 }
441 /* If a and b might be mergeable vertically */
442 else if (a->x == b->x && a->width == b->width)
443 {
444 /* If a and b overlap */
445 if (meta_rectangle_overlap (a, b))
446 {
447 int new_y = MIN (a->y, b->y);
448 a->height = MAX (a->y + a->height, b->y + b->height) - new_y;
449 a->y = new_y;
450 delete_me = other;
451 }
452 /* If a and b are adjacent */
453 else if (a->y + a->height == b->y || a->y == b->y + b->height)
454 {
455 int new_y = MIN (a->y, b->y);
456 a->height = MAX (a->y + a->height, b->y + b->height) - new_y;
457 a->y = new_y;
458 delete_me = other;
459 }
460 }
461
462 other = other->next;
463
464 /* Delete any rectangle in the list that is no longer wanted */
465 if (delete_me != NULL)
466 {
467 /* Deleting the rect we compare others to is a little tricker */
468 if (compare == delete_me)
469 {
470 compare = compare->next;
471 other = compare->next;
472 a = compare->data;
473 }
474
475 /* avoid false positive warning in Clang static analyzer */
476 g_assert (a != delete_me->data);
477
478 /* Okay, we can free it now */
479 g_free (delete_me->data);
480 region = g_list_delete_link (region, delete_me);
481 }
482
483 }
484
485 compare = compare->next;
486 }
487
488 return region;
489}

491/* Simple helper function for get_minimal_spanning_set_for_region()... */
492static gint
493compare_rect_areas (gconstpointer a, gconstpointer b)
494{
495 const MetaRectangle *a_rect = (gconstpointer) a;
496 const MetaRectangle *b_rect = (gconstpointer) b;
497
498 int a_area = meta_rectangle_area (a_rect);
499 int b_area = meta_rectangle_area (b_rect);
500
501 return b_area - a_area; /* positive ret value denotes b > a, ... */
502}

504/* This function is trying to find a "minimal spanning set (of rectangles)"
505 * for a given region.
506 *
507 * The region is given by taking basic_rect, then removing the areas
508 * covered by all the rectangles in the all_struts list, and then expanding
509 * the resulting region by the given number of pixels in each direction.
510 *
511 * A "minimal spanning set (of rectangles)" is the best name I could come
512 * up with for the concept I had in mind. Basically, for a given region, I
513 * want a set of rectangles with the property that a window is contained in
514 * the region if and only if it is contained within at least one of the
515 * rectangles.
516 *
517 * The GList* returned will be a list of (allocated) MetaRectangles.
518 * The list will need to be freed by calling
519 * meta_rectangle_free_spanning_set() on it (or by manually
520 * implementing that function...)
521 */
522GList*
523meta_rectangle_get_minimal_spanning_set_for_region (
524 const MetaRectangle *basic_rect,
525 const GSList *all_struts,
526 gboolean skip_middle_struts)
527
528{
529 /* NOTE FOR OPTIMIZERS: This function *might* be somewhat slow,
530 * especially due to the call to merge_spanning_rects_in_region() (which
531 * is O(n^2) where n is the size of the list generated in this function).
532 * This is made more onerous due to the fact that it involves a fair
533 * number of memory allocation and deallocation calls. However, n is 1
534 * for default installations of Mate (because partial struts aren't used
535 * by default and only partial struts increase the size of the spanning
536 * set generated). With one partial strut, n will be 2 or 3. With 2
537 * partial struts, n will probably be 4 or 5. So, n probably isn't large
538 * enough to make this worth bothering. Further, it is only called from
539 * workspace.c:ensure_work_areas_validated (at least as of the time of
540 * writing this comment), which in turn should only be called if the
541 * strut list changes or the screen or xinerama size changes. If it ever
542 * does show up on profiles (most likely because people start using
543 * ridiculously huge numbers of partial struts), possible optimizations
544 * include:
545 *
546 * (1) rewrite merge_spanning_rects_in_region() to be O(n) or O(nlogn).
547 * I'm not totally sure it's possible, but with a couple copies of
548 * the list and sorting them appropriately, I believe it might be.
549 * (2) only call merge_spanning_rects_in_region() with a subset of the
550 * full list of rectangles. I believe from some of my preliminary
551 * debugging and thinking about it that it is possible to figure out
552 * apriori groups of rectangles which are only merge candidates with
553 * each other. (See testboxes.c:get_screen_region() when which==2
554 * and track the steps of this function carefully to see what gave
555 * me the hint that this might work)
556 * (3) figure out how to avoid merge_spanning_rects_in_region(). I think
557 * it might be possible to modify this function to make that
558 * possible, and I spent just a little while thinking about it, but n
559 * wasn't large enough to convince me to care yet.
560 * (4) Some of the stuff Rob mentioned at http://mail.gnome.org/archives\
561 * /marco-devel-list/2005-November/msg00028.html. (Sorry for the
562 * URL splitting.)
563 */
564
565 GList *ret;
566 GList *tmp_list;
567 const GSList *strut_iter;
568 MetaRectangle *temp_rect;
569
570 /* The algorithm is basically as follows:
571 * Initialize rectangle_set to basic_rect
572 * Foreach strut:
573 * Foreach rectangle in rectangle_set:
574 * - Split the rectangle into new rectangles that don't overlap the
575 * strut (but which are as big as possible otherwise)
576 * - Remove the old (pre-split) rectangle from the rectangle_set,
577 * and replace it with the new rectangles generated from the
578 * splitting
579 */
580
581 temp_rect = g_new (MetaRectangle, 1);
582 *temp_rect = *basic_rect;
583 ret = g_list_prepend (NULL, temp_rect);
584
585 for (strut_iter = all_struts; strut_iter; strut_iter = strut_iter->next)
586 {
587 GList *rect_iter;
588 MetaStrut *strut = (MetaStrut *) strut_iter->data;
589 MetaRectangle *strut_rect = &strut->rect;
590
591 if (skip_middle_struts && strut->edge == META_EDGE_XINERAMA)
592 {
593 if ((strut->side == META_SIDE_LEFT &&
594 strut_rect->x != basic_rect->x) ||
595 (strut->side == META_SIDE_RIGHT &&
596 strut_rect->x + strut_rect->width != basic_rect->width) ||
597 (strut->side == META_SIDE_TOP &&
598 strut_rect->y != basic_rect->y) ||
599 (strut->side == META_SIDE_BOTTOM &&
600 strut_rect->y + strut_rect->height != basic_rect->height))
601 {
602 continue;
603 }
604 }
605
606 tmp_list = ret;
607 ret = NULL;
608 rect_iter = tmp_list;
609 while (rect_iter)
610 {
611 MetaRectangle *rect = (MetaRectangle*) rect_iter->data;
612 if (!meta_rectangle_overlap (rect, strut_rect))
613 ret = g_list_prepend (ret, rect);
614 else
615 {
616 /* If there is area in rect left of strut */
617 if (BOX_LEFT (*rect) < BOX_LEFT (*strut_rect))
618 {
619 temp_rect = g_new (MetaRectangle, 1);
620 *temp_rect = *rect;
621 temp_rect->width = BOX_LEFT (*strut_rect) - BOX_LEFT (*rect);
622 ret = g_list_prepend (ret, temp_rect);
623 }
624 /* If there is area in rect right of strut */
625 if (BOX_RIGHT (*rect) > BOX_RIGHT (*strut_rect))
626 {
627 int new_x;
628 temp_rect = g_new (MetaRectangle, 1);
629 *temp_rect = *rect;
630 new_x = BOX_RIGHT (*strut_rect);
631 temp_rect->width = BOX_RIGHT(*rect) - new_x;
632 temp_rect->x = new_x;
633 ret = g_list_prepend (ret, temp_rect);
634 }
635 /* If there is area in rect above strut */
636 if (BOX_TOP (*rect) < BOX_TOP (*strut_rect))
637 {
638 temp_rect = g_new (MetaRectangle, 1);
639 *temp_rect = *rect;
640 temp_rect->height = BOX_TOP (*strut_rect) - BOX_TOP (*rect);
641 ret = g_list_prepend (ret, temp_rect);
642 }
643 /* If there is area in rect below strut */
644 if (BOX_BOTTOM (*rect) > BOX_BOTTOM (*strut_rect))
645 {
646 int new_y;
647 temp_rect = g_new (MetaRectangle, 1);
648 *temp_rect = *rect;
649 new_y = BOX_BOTTOM (*strut_rect);
650 temp_rect->height = BOX_BOTTOM (*rect) - new_y;
651 temp_rect->y = new_y;
652 ret = g_list_prepend (ret, temp_rect);
653 }
654 g_free (rect);
655 }
656 rect_iter = rect_iter->next;
657 }
658 g_list_free (tmp_list);
659 }
660
661 /* Sort by maximal area, just because I feel like it... */
662 ret = g_list_sort (ret, compare_rect_areas);
663
664 /* Merge rectangles if possible so that the list really is minimal */
665 ret = merge_spanning_rects_in_region (ret);
666
667 return ret;
668}

686GList*
687meta_rectangle_expand_region_conditionally (GList *region,
688 const int left_expand,
689 const int right_expand,
690 const int top_expand,
691 const int bottom_expand,
692 const int min_x,
693 const int min_y)
694{
695 GList *tmp_list = region;
696 while (tmp_list)
697 {
698 MetaRectangle *rect = (MetaRectangle*) tmp_list->data;
699 if (rect->width >= min_x)
700 {
701 rect->x -= left_expand;
702 rect->width += (left_expand + right_expand);
703 }
704 if (rect->height >= min_y)
705 {
706 rect->y -= top_expand;
707 rect->height += (top_expand + bottom_expand);
708 }
709 tmp_list = tmp_list->next;
710 }
711
712 return region;
713}

716meta_rectangle_expand_to_avoiding_struts (MetaRectangle *rect,
717 const MetaRectangle *expand_to,
718 const MetaDirection direction,
719 const GSList *all_struts)
720{
721 const GSList *strut_iter;
722
723 /* If someone wants this function to handle more fine-grained
724 * direction expanding in the future (e.g. only left, or fully
725 * horizontal plus upward), feel free. But I'm hard-coding for both
726 * horizontal directions (exclusive-)or both vertical directions.
727 */
728 g_assert ((direction == META_DIRECTION_HORIZONTAL) ^
729 (direction == META_DIRECTION_VERTICAL ));
730
731 if (direction == META_DIRECTION_HORIZONTAL)
732 {
733 rect->x = expand_to->x;
734 rect->width = expand_to->width;
735 }
736 else
737 {
738 rect->y = expand_to->y;
739 rect->height = expand_to->height;
740 }
741
742 /* Run over all struts */
743 for (strut_iter = all_struts; strut_iter; strut_iter = strut_iter->next)
744 {
745 MetaStrut *strut = (MetaStrut*) strut_iter->data;
746
747 /* Skip struts that don't overlap */
748 if (!meta_rectangle_overlap (&strut->rect, rect))
749 continue;
750
751 if (direction == META_DIRECTION_HORIZONTAL)
752 {
753 if (strut->side == META_SIDE_LEFT)
754 {
755 int offset = BOX_RIGHT(strut->rect) - BOX_LEFT(*rect);
756 rect->x += offset;
757 rect->width -= offset;
758 }
759 else if (strut->side == META_SIDE_RIGHT)
760 {
761 int offset = BOX_RIGHT (*rect) - BOX_LEFT(strut->rect);
762 rect->width -= offset;
763 }
764 /* else ignore the strut */
765 }
766 else /* direction == META_DIRECTION_VERTICAL */
767 {
768 if (strut->side == META_SIDE_TOP)
769 {
770 int offset = BOX_BOTTOM(strut->rect) - BOX_TOP(*rect);
771 rect->y += offset;
772 rect->height -= offset;
773 }
774 else if (strut->side == META_SIDE_BOTTOM)
775 {
776 int offset = BOX_BOTTOM(*rect) - BOX_TOP(strut->rect);
777 rect->height -= offset;
778 }
779 /* else ignore the strut */
780 }
781 } /* end loop over struts */
782} /* end meta_rectangle_expand_to_avoiding_struts */

838void
839meta_rectangle_clamp_to_fit_into_region (const GList *spanning_rects,
840 FixedDirections fixed_directions,
841 MetaRectangle *rect,
842 const MetaRectangle *min_size)
843{
844 const GList *temp;
845 const MetaRectangle *best_rect = NULL;
846 int best_overlap = 0;
847
848 /* First, find best rectangle from spanning_rects to which we can clamp
849 * rect to fit into.
850 */
851 for (temp = spanning_rects; temp; temp = temp->next)
852 {
853 MetaRectangle *compare_rect = temp->data;
854 int maximal_overlap_amount_for_compare;
855
856 /* If x is fixed and the entire width of rect doesn't fit in compare,
857 * skip this rectangle.
858 */
859 if ((fixed_directions & FIXED_DIRECTION_X) &&
860 (compare_rect->x > rect->x ||
861 compare_rect->x + compare_rect->width < rect->x + rect->width))
862 continue;
863
864 /* If y is fixed and the entire height of rect doesn't fit in compare,
865 * skip this rectangle.
866 */
867 if ((fixed_directions & FIXED_DIRECTION_Y) &&
868 (compare_rect->y > rect->y ||
869 compare_rect->y + compare_rect->height < rect->y + rect->height))
870 continue;
871
872 /* If compare can't hold the min_size window, skip this rectangle. */
873 if (compare_rect->width < min_size->width ||
874 compare_rect->height < min_size->height)
875 continue;
876
877 /* Determine maximal overlap amount */
878 maximal_overlap_amount_for_compare =
879 MIN (rect->width, compare_rect->width) *
880 MIN (rect->height, compare_rect->height);
881
882 /* See if this is the best rect so far */
883 if (maximal_overlap_amount_for_compare > best_overlap)
884 {
885 best_rect = compare_rect;
886 best_overlap = maximal_overlap_amount_for_compare;
887 }
888 }
889
890 /* Clamp rect appropriately */
891 if (best_rect == NULL)
892 {
893 meta_warning ("No rect whose size to clamp to found!\n");
894
895 /* If it doesn't fit, at least make it no bigger than it has to be */
896 if (!(fixed_directions & FIXED_DIRECTION_X))
897 rect->width = min_size->width;
898 if (!(fixed_directions & FIXED_DIRECTION_Y))
899 rect->height = min_size->height;
900 }
901 else
902 {
903 rect->width = MIN (rect->width, best_rect->width);
904 rect->height = MIN (rect->height, best_rect->height);
905 }
906}

909meta_rectangle_clip_to_region (const GList *spanning_rects,
910 FixedDirections fixed_directions,
911 MetaRectangle *rect)
912{
913 const GList *temp;
914 const MetaRectangle *best_rect = NULL;
915 int best_overlap = 0;
916
917 /* First, find best rectangle from spanning_rects to which we will clip
918 * rect into.
919 */
920 for (temp = spanning_rects; temp; temp = temp->next)
921 {
922 MetaRectangle *compare_rect = temp->data;
923 MetaRectangle overlap;
924 int maximal_overlap_amount_for_compare;
925
926 /* If x is fixed and the entire width of rect doesn't fit in compare,
927 * skip the rectangle.
928 */
929 if ((fixed_directions & FIXED_DIRECTION_X) &&
930 (compare_rect->x > rect->x ||
931 compare_rect->x + compare_rect->width < rect->x + rect->width))
932 continue;

935 * skip the rectangle.
936 */
937 if ((fixed_directions & FIXED_DIRECTION_Y) &&
938 (compare_rect->y > rect->y ||
939 compare_rect->y + compare_rect->height < rect->y + rect->height))
940 continue;
941
942 /* Determine maximal overlap amount */
943 meta_rectangle_intersect (rect, compare_rect, &overlap);
944 maximal_overlap_amount_for_compare = meta_rectangle_area (&overlap);
945
946 /* See if this is the best rect so far */
947 if (maximal_overlap_amount_for_compare > best_overlap)
948 {
949 best_rect = compare_rect;
950 best_overlap = maximal_overlap_amount_for_compare;
951 }
952 }
953
954 /* Clip rect appropriately */
955 if (best_rect == NULL)
956 meta_warning ("No rect to clip to found!\n");
957 else
958 {
959 /* Extra precaution with checking fixed direction shouldn't be needed
960 * due to logic above, but it shouldn't hurt either.
961 */
962 if (!(fixed_directions & FIXED_DIRECTION_X))
963 {
964 /* Find the new left and right */
965 int new_x = MAX (rect->x, best_rect->x);
966 rect->width = MIN ((rect->x + rect->width) - new_x,
967 (best_rect->x + best_rect->width) - new_x);
968 rect->x = new_x;
969 }
970
971 /* Extra precaution with checking fixed direction shouldn't be needed
972 * due to logic above, but it shouldn't hurt either.
973 */
974 if (!(fixed_directions & FIXED_DIRECTION_Y))
975 {
976 /* Clip the top, if needed */
977 int new_y = MAX (rect->y, best_rect->y);
978 rect->height = MIN ((rect->y + rect->height) - new_y,
979 (best_rect->y + best_rect->height) - new_y);
980 rect->y = new_y;
981 }
982 }

985void
986meta_rectangle_shove_into_region (const GList *spanning_rects,
987 FixedDirections fixed_directions,
988 MetaRectangle *rect)
989{
990 const GList *temp;
991 const MetaRectangle *best_rect = NULL;
992 int best_overlap = 0;
993 int shortest_distance = G_MAXINT;
994
995 /* First, find best rectangle from spanning_rects to which we will shove
996 * rect into.
997 */
998
999 for (temp = spanning_rects; temp; temp = temp->next)
1000 {
1001 MetaRectangle *compare_rect = temp->data;
1002 int maximal_overlap_amount_for_compare;
1003 int dist_to_compare;
1004
1005 /* If x is fixed and the entire width of rect doesn't fit in compare,
1006 * skip this rectangle.
1007 */
1008 if ((fixed_directions & FIXED_DIRECTION_X) &&
1009 (compare_rect->x > rect->x ||
1010 compare_rect->x + compare_rect->width < rect->x + rect->width))
1011 continue;
1012
1013 /* If y is fixed and the entire height of rect doesn't fit in compare,
1014 * skip this rectangle.
1015 */
1016 if ((fixed_directions & FIXED_DIRECTION_Y) &&
1017 (compare_rect->y > rect->y ||
1018 compare_rect->y + compare_rect->height < rect->y + rect->height))
1019 continue;
1020
1021 /* Determine maximal overlap amount between rect & compare_rect */
1022 maximal_overlap_amount_for_compare =
1023 MIN (rect->width, compare_rect->width) *
1024 MIN (rect->height, compare_rect->height);
1025
1026 /* Determine distance necessary to put rect into compare_rect */
1027 dist_to_compare = 0;
1028 if (compare_rect->x > rect->x)
1029 dist_to_compare += compare_rect->x - rect->x;
1030 if (compare_rect->x + compare_rect->width < rect->x + rect->width)
1031 dist_to_compare += (rect->x + rect->width) -
1032 (compare_rect->x + compare_rect->width);
1033 if (compare_rect->y > rect->y)
1034 dist_to_compare += compare_rect->y - rect->y;
1035 if (compare_rect->y + compare_rect->height < rect->y + rect->height)
1036 dist_to_compare += (rect->y + rect->height) -
1037 (compare_rect->y + compare_rect->height);
1038
1039 /* See if this is the best rect so far */
1040 if ((maximal_overlap_amount_for_compare > best_overlap) ||
1041 (maximal_overlap_amount_for_compare == best_overlap &&
1042 dist_to_compare < shortest_distance))
1043 {
1044 best_rect = compare_rect;
1045 best_overlap = maximal_overlap_amount_for_compare;
1046 shortest_distance = dist_to_compare;
1047 }
1048 }
1049
1050 /* Shove rect appropriately */
1051 if (best_rect == NULL)
1052 meta_warning ("No rect to shove into found!\n");
1053 else
1054 {
1055 /* Extra precaution with checking fixed direction shouldn't be needed
1056 * due to logic above, but it shouldn't hurt either.
1057 */
1058 if (!(fixed_directions & FIXED_DIRECTION_X))
1059 {
1060 /* Shove to the right, if needed */
1061 if (best_rect->x > rect->x)
1062 rect->x = best_rect->x;
1063
1064 /* Shove to the left, if needed */
1065 if (best_rect->x + best_rect->width < rect->x + rect->width)
1066 rect->x = (best_rect->x + best_rect->width) - rect->width;
1067 }
1068
1069 /* Extra precaution with checking fixed direction shouldn't be needed
1070 * due to logic above, but it shouldn't hurt either.
1071 */
1072 if (!(fixed_directions & FIXED_DIRECTION_Y))
1073 {
1074 /* Shove down, if needed */
1075 if (best_rect->y > rect->y)
1076 rect->y = best_rect->y;
1077
1078 /* Shove up, if needed */
1079 if (best_rect->y + best_rect->height < rect->y + rect->height)
1080 rect->y = (best_rect->y + best_rect->height) - rect->height;
1081 }
1082 }
1083}

1085void
1086meta_rectangle_find_linepoint_closest_to_point (double x1,
1087 double y1,
1088 double x2,
1089 double y2,
1090 double px,
1091 double py,
1092 double *valx,
1093 double *valy)
1094{
1095 /* I'll use the shorthand rx, ry for the return values, valx & valy.
1096 * Now, we need (rx,ry) to be on the line between (x1,y1) and (x2,y2).
1097 * For that to happen, we first need the slope of the line from (x1,y1)
1098 * to (rx,ry) must match the slope of (x1,y1) to (x2,y2), i.e.:
1099 * (ry-y1) (y2-y1)
1100 * ------- = -------
1101 * (rx-x1) (x2-x1)
1102 * If x1==x2, though, this gives divide by zero errors, so we want to
1103 * rewrite the equation by multiplying both sides by (rx-x1)*(x2-x1):
1104 * (ry-y1)(x2-x1) = (y2-y1)(rx-x1)
1105 * This is a valid requirement even when x1==x2 (when x1==x2, this latter
1106 * equation will basically just mean that rx must be equal to both x1 and
1107 * x2)
1108 *
1109 * The other requirement that we have is that the line from (rx,ry) to
1110 * (px,py) must be perpendicular to the line from (x1,y1) to (x2,y2). So
1111 * we just need to get a vector in the direction of each line, take the
1112 * dot product of the two, and ensure that the result is 0:
1113 * (rx-px)*(x2-x1) + (ry-py)*(y2-y1) = 0.
1114 *
1115 * This gives us two equations and two unknowns:
1116 *
1117 * (ry-y1)(x2-x1) = (y2-y1)(rx-x1)
1118 * (rx-px)*(x2-x1) + (ry-py)*(y2-y1) = 0.
1119 *
1120 * This particular pair of equations is always solvable so long as
1121 * (x1,y1) and (x2,y2) are not the same point (and note that anyone who
1122 * calls this function that way is braindead because it means that they
1123 * really didn't specify a line after all). However, the caller should
1124 * be careful to avoid making (x1,y1) and (x2,y2) too close (e.g. like
1125 * 10^{-8} apart in each coordinate), otherwise roundoff error could
1126 * cause issues. Solving these equations by hand (or using Maple(TM) or
1127 * Mathematica(TM) or whatever) results in slightly messy expressions,
1128 * but that's all the below few lines do.
1129 */
1130
1131 double diffx, diffy, den;
1132 diffx = x2 - x1;
1133 diffy = y2 - y1;
1134 den = diffx * diffx + diffy * diffy;
1135
1136 *valx = (py * diffx * diffy + px * diffx * diffx +
1137 y2 * x1 * diffy - y1 * x2 * diffy) / den;
1138 *valy = (px * diffx * diffy + py * diffy * diffy +
1139 x2 * y1 * diffx - x1 * y2 * diffx) / den;
1140}

1149meta_rectangle_edge_aligns (const MetaRectangle *rect, const MetaEdge *edge)
1150{
1151 /* The reason for the usage of <= below instead of < is because we are
1152 * interested in in-the-way-or-adject'ness. So, a left (i.e. vertical
1153 * edge) occupying y positions 0-9 (which has a y of 0 and a height of
1154 * 10) and a rectangle with top at y=10 would be considered to "align" by
1155 * this function.
1156 */
1157 switch (edge->side_type)
1158 {
1159 case META_SIDE_LEFT:
1160 case META_SIDE_RIGHT:
1161 return BOX_TOP (*rect) <= BOX_BOTTOM (edge->rect) &&
1162 BOX_TOP (edge->rect) <= BOX_BOTTOM (*rect);
1163 case META_SIDE_TOP:
1164 case META_SIDE_BOTTOM:
1165 return BOX_LEFT (*rect) <= BOX_RIGHT (edge->rect) &&
1166 BOX_LEFT (edge->rect) <= BOX_RIGHT (*rect);
1167 default:
1168 g_assert_not_reached ();
1169 }

1217static GList*
1218replace_rect_with_list (GList *old_element,
1219 GList *new_list)
1220{
1221 GList *ret;
1222 g_assert (old_element != NULL);
1223
1224 if (!new_list)
1225 {
1226 /* If there is no new list, just remove the old_element */
1227 ret = g_list_remove_link (old_element, old_element);
1228 }
1229 else
1230 {
1231 /* Fix up the prev and next pointers everywhere */
1232 ret = new_list;
1233 if (old_element->prev)
1234 {
1235 old_element->prev->next = new_list;
1236 new_list->prev = old_element->prev;
1237 }
1238 if (old_element->next)
1239 {

1241 old_element->next->prev = tmp;
1242 tmp->next = old_element->next;
1243 }
1244 }

1247 g_free (old_element->data);
1248 g_list_free_1 (old_element);

1252/* Make a copy of the strut list, make sure that copy only contains parts
1253 * of the old_struts that intersect with the region rect, and then do some
1254 * magic to make all the new struts disjoint (okay, we we break up struts
1255 * that aren't disjoint in a way that the overlapping part is only included
1256 * once, so it's not really magic...).
1257 */
1258static GList*
1259get_disjoint_strut_rect_list_in_region (const GSList *old_struts,
1260 const MetaRectangle *region)
1261{
1262 GList *strut_rects;
1263 GList *tmp;
1264
1265 /* First, copy the list */
1266 strut_rects = NULL;
1267 while (old_struts)
1268 {
1269 MetaRectangle *cur = &((MetaStrut*)old_struts->data)->rect;
1270 MetaRectangle *copy = g_new (MetaRectangle, 1);
1271 *copy = *cur;
1272 if (meta_rectangle_intersect (copy, region, copy))
1273 strut_rects = g_list_prepend (strut_rects, copy);
1274 else
1275 g_free (copy);
1276
1277 old_struts = old_struts->next;
1278 }
1279
1280 /* Now, loop over the list and check for intersections, fixing things up
1281 * where they do intersect.
1282 */
1283 tmp = strut_rects;
1284 while (tmp)
1285 {
1286 GList *compare;
1287
1288 MetaRectangle *cur = tmp->data;
1289
1290 compare = tmp->next;
1291 while (compare)
1292 {
1293 MetaRectangle *comp = compare->data;
1294 MetaRectangle overlap;
1295
1296 if (meta_rectangle_intersect (cur, comp, &overlap))
1297 {
1298 /* Get a list of rectangles for each strut that don't overlap
1299 * the intersection region.
1300 */
1301 GList *cur_leftover = get_rect_minus_overlap (tmp, &overlap);
1302 GList *comp_leftover = get_rect_minus_overlap (compare, &overlap);
1303
1304 /* Add the intersection region to cur_leftover */
1305 MetaRectangle *overlap_allocated = g_new (MetaRectangle, 1);
1306 *overlap_allocated = overlap;
1307 cur_leftover = g_list_prepend (cur_leftover, overlap_allocated);
1308
1309 /* Fix up tmp, compare, and cur -- maybe struts too */
1310 if (strut_rects == tmp)
1311 {
1312 strut_rects = replace_rect_with_list (tmp, cur_leftover);
1313 tmp = strut_rects;
1314 }
1315 else
1316 tmp = replace_rect_with_list (tmp, cur_leftover);
1317 compare = replace_rect_with_list (compare, comp_leftover);
1318
1319 if (compare == NULL)
1320 break;
1321
1322 cur = tmp->data;
1323 }
1324
1325 compare = compare->next;
1326 }
1327
1328 tmp = tmp->next;
1329 }
1330
1331 return strut_rects;
1332}

1334gint
1335meta_rectangle_edge_cmp_ignore_type (gconstpointer a, gconstpointer b)
1336{
1337 const MetaEdge *a_edge_rect = a;
1338 const MetaEdge *b_edge_rect = b;
1339 int a_compare = 0;
1340 int b_compare = 0;
1341
1342 /* Edges must be both vertical or both horizontal, or it doesn't make
1343 * sense to compare them.
1344 */
1345 g_assert ((a_edge_rect->rect.width == 0 && b_edge_rect->rect.width == 0) ||
1346 (a_edge_rect->rect.height == 0 && b_edge_rect->rect.height == 0));
1347
1348 switch (a_edge_rect->side_type)
1349 {
1350 case META_SIDE_LEFT:
1351 case META_SIDE_RIGHT:
1352 a_compare = a_edge_rect->rect.x;
1353 b_compare = b_edge_rect->rect.x;
1354 if (a_compare == b_compare)
1355 {
1356 a_compare = a_edge_rect->rect.y;
1357 b_compare = b_edge_rect->rect.y;
1358 }
1359 break;
1360 case META_SIDE_TOP:
1361 case META_SIDE_BOTTOM:
1362 a_compare = a_edge_rect->rect.y;
1363 b_compare = b_edge_rect->rect.y;
1364 if (a_compare == b_compare)
1365 {
1366 a_compare = a_edge_rect->rect.x;
1367 b_compare = b_edge_rect->rect.x;
1368 }
1369 break;
1370 default:
1371 g_assert_not_reached ();
1372 }
1373 return a_compare - b_compare; /* positive value denotes a > b ... */
1374}

1376/* To make things easily testable, provide a nice way of sorting edges */
1377gint
1378meta_rectangle_edge_cmp (gconstpointer a, gconstpointer b)
1379{
1380 const MetaEdge *a_edge_rect = (gconstpointer) a;
1381 const MetaEdge *b_edge_rect = (gconstpointer) b;
1382
1383 int a_compare, b_compare;

1415static gboolean
1416rectangle_and_edge_intersection (const MetaRectangle *rect,
1417 const MetaEdge *edge,
1418 MetaEdge *overlap,
1419 int *handle_type)
1420{
1421 const MetaRectangle *rect2 = &edge->rect;
1422 MetaRectangle *result = &overlap->rect;
1423 gboolean intersect = TRUE;
1424
1425 /* We don't know how to set these, so set them to invalid values */
1426 overlap->edge_type = -1;
1427 overlap->side_type = -1;
1428
1429 /* Figure out what the intersection is */
1430 result->x = MAX (rect->x, rect2->x);
1431 result->y = MAX (rect->y, rect2->y);
1432 result->width = MIN (BOX_RIGHT (*rect), BOX_RIGHT (*rect2)) - result->x;
1433 result->height = MIN (BOX_BOTTOM (*rect), BOX_BOTTOM (*rect2)) - result->y;
1434
1435 /* Find out if the intersection is empty; have to do it this way since
1436 * edges have a thickness of 0
1437 */
1438 if ((result->width < 0 || result->height < 0) ||
1439 (result->width == 0 && result->height == 0))
1440 {
1441 result->width = 0;
1442 result->height = 0;
1443 intersect = FALSE;
1444 }
1445 else
1446 {
1447 /* Need to figure out the handle_type, a somewhat weird quantity:
1448 * 0 - overlap is in middle of rect
1449 * -1 - overlap is at the side of rect, and is on the opposite side
1450 * of rect than the edge->side_type side
1451 * 1 - overlap is at the side of rect, and the side of rect it is
1452 * on is the edge->side_type side
1453 */
1454 switch (edge->side_type)
1455 {
1456 case META_SIDE_LEFT:
1457 if (result->x == rect->x)
1458 *handle_type = 1;
1459 else if (result->x == BOX_RIGHT (*rect))
1460 *handle_type = -1;
1461 else
1462 *handle_type = 0;
1463 break;
1464 case META_SIDE_RIGHT:
1465 if (result->x == rect->x)
1466 *handle_type = -1;
1467 else if (result->x == BOX_RIGHT (*rect))
1468 *handle_type = 1;
1469 else
1470 *handle_type = 0;
1471 break;
1472 case META_SIDE_TOP:
1473 if (result->y == rect->y)
1474 *handle_type = 1;
1475 else if (result->y == BOX_BOTTOM (*rect))
1476 *handle_type = -1;
1477 else
1478 *handle_type = 0;
1479 break;
1480 case META_SIDE_BOTTOM:
1481 if (result->y == rect->y)
1482 *handle_type = -1;
1483 else if (result->y == BOX_BOTTOM (*rect))
1484 *handle_type = 1;
1485 else
1486 *handle_type = 0;
1487 break;
1488 default:
1489 g_assert_not_reached ();
1490 }
1491 }
1492 return intersect;
1493}

1603/* Split up edge and remove preliminary edges from strut_edges depending on
1604 * if and how rect and edge intersect.

1610{

1717/* This function is trying to find all the edges of an onscreen region. */
1718GList*
1719meta_rectangle_find_onscreen_edges (const MetaRectangle *basic_rect,
1720 const GSList *all_struts)
1721{
1722 GList *ret;
1723 GList *fixed_strut_rects;
1724 GList *edge_iter;
1725 const GList *strut_rect_iter;
1726
1727 /* The algorithm is basically as follows:
1728 * Make sure the struts are disjoint
1729 * Initialize the edge_set to the edges of basic_rect
1730 * Foreach strut:
1731 * Put together a preliminary new edge from the edges of the strut
1732 * Foreach edge in edge_set:
1733 * - Split the edge if it is partially contained inside the strut
1734 * - If the edge matches an edge of the strut (i.e. a strut just
1735 * against the edge of the screen or a not-next-to-edge-of-screen
1736 * strut adjacent to another), then both the edge from the
1737 * edge_set and the preliminary edge for the strut will need to
1738 * be split
1739 * Add any remaining "preliminary" strut edges to the edge_set
1740 */
1741
1742 /* Make sure the struts are disjoint */
1743 fixed_strut_rects =
1744 get_disjoint_strut_rect_list_in_region (all_struts, basic_rect);
1745
1746 /* Start off the list with the edges of basic_rect */
1747 ret = add_edges (NULL, basic_rect, TRUE);
1748
1749 strut_rect_iter = fixed_strut_rects;
1750 while (strut_rect_iter)
1751 {
1752 MetaRectangle *strut_rect = (MetaRectangle*) strut_rect_iter->data;
1753
1754 /* Get the new possible edges we may need to add from the strut */
1755 GList *new_strut_edges = add_edges (NULL, strut_rect, FALSE);
1756
1757 edge_iter = ret;
1758 while (edge_iter)
1759 {
1760 MetaEdge *cur_edge = edge_iter->data;
1761 GList *splits_of_cur_edge = NULL;
1762 gboolean edge_needs_removal = FALSE;
1763
1764 fix_up_edges (strut_rect, cur_edge,
1765 &new_strut_edges, &splits_of_cur_edge,
1766 &edge_needs_removal);
1767
1768 if (edge_needs_removal)
1769 {
1770 /* Delete the old edge */
1771 GList *delete_me = edge_iter;
1772 edge_iter = edge_iter->next;
1773 g_free (cur_edge);
1774 ret = g_list_delete_link (ret, delete_me);
1775
1776 /* Add the new split parts of the edge */
1777 ret = g_list_concat (splits_of_cur_edge, ret);
1778 }
1779 else
1780 {
1781 edge_iter = edge_iter->next;
1782 }
1783
1784 /* edge_iter was already advanced above */
1785 }
1786
1787 ret = g_list_concat (new_strut_edges, ret);
1788 strut_rect_iter = strut_rect_iter->next;
1789 }
1790
1791 /* Sort the list */
1792 ret = g_list_sort (ret, meta_rectangle_edge_cmp);
1793
1794 /* Free the fixed struts list */
1795 g_list_free_full (fixed_strut_rects, g_free);

1797 return ret;
1798}
