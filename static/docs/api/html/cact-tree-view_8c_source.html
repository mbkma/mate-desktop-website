426static void
427instance_get_property( GObject *object, guint property_id, GValue *value, GParamSpec *spec )
428{
429 CactTreeView *self;
430
431 g_return_if_fail( CACT_IS_TREE_VIEW( object ));
432 self = CACT_TREE_VIEW( object );
433
434 if( !self->private->dispose_has_run ){
435
436 switch( property_id ){
437 case TREE_PROP_WINDOW_ID:
438 g_value_set_pointer( value, self->private->window );
439 break;
440
441 case TREE_PROP_PARENT_ID:
442 g_value_set_pointer( value, self->private->parent );
443 break;
444
445 case TREE_PROP_WIDGET_NAME_ID:
446 g_value_set_string( value, self->private->widget_name );
447 break;
448
449 case TREE_PROP_MODE_ID:
450 g_value_set_uint( value, self->private->mode );
451 break;
452
453 case TREE_PROP_NOTIFY_ALLOWED_ID:
454 g_value_set_boolean( value, self->private->notify_allowed );
455 break;
456
457 default:
458 G_OBJECT_WARN_INVALID_PROPERTY_ID( object, property_id, spec );
459 break;
460 }
461 }

504instance_constructed( GObject *object )
505{
506 static const gchar *thisfn = "cact_tree_view_instance_constructed";
507 CactTreeViewPrivate *priv;
508
509 g_return_if_fail( CACT_IS_TREE_VIEW( object ));

514

521

531 BASE_SIGNAL_SHOW_WIDGETS,

533
534 g_object_set_data( G_OBJECT( priv->window ), WINDOW_DATA_TREE_VIEW, object );
535
536 initialize_gtk( CACT_TREE_VIEW( object ));
537 }

622static void
623initialize_gtk( CactTreeView *items_view )
624{
625 static const gchar *thisfn = "cact_tree_view_initialize_gtk";
626 GtkTreeView *treeview;
627 CactTreeModel *model;
628 GtkWidget *label;
629 GtkTreeViewColumn *column;
630 GtkCellRenderer *renderer;
631 GtkTreeSelection *selection;
632 GList *renderers;
633
634 g_debug( "%s: items_view=%p", thisfn, ( void * ) items_view );
635
636 treeview = GTK_TREE_VIEW( get_tree_view( items_view ));
637 model = cact_tree_model_new( items_view->private->window, treeview, items_view->private->mode );
638 gtk_tree_view_set_model( treeview, GTK_TREE_MODEL( model ));
639 g_object_unref( model );
640 g_debug( "%s: cact_tree_model_ref_count=%d", thisfn, G_OBJECT( model )->ref_count );
641
642 /* associates the ItemsView to the label */
643 label = na_gtk_utils_find_widget_by_name( items_view->private->parent, "ActionsListLabel" );
644 gtk_label_set_mnemonic_widget( GTK_LABEL( label ), GTK_WIDGET( treeview ));
645
646 /* create visible columns on the tree view
647 */
648 column = gtk_tree_view_column_new_with_attributes(
649 "icon",
650 gtk_cell_renderer_pixbuf_new(),
651 "pixbuf", TREE_COLUMN_ICON,
652 NULL );
653 gtk_tree_view_append_column( treeview, column );
654
655 renderer = gtk_cell_renderer_text_new();
656 column = gtk_tree_view_column_new_with_attributes(
657 "label",
658 renderer,
659 "text", TREE_COLUMN_LABEL,
660 NULL );
661 gtk_tree_view_column_set_sort_column_id( column, TREE_COLUMN_LABEL );
662 gtk_tree_view_append_column( treeview, column );
663
664 /* allow multiple selection
665 */
666 selection = gtk_tree_view_get_selection( treeview );
667 gtk_tree_selection_set_mode( selection, GTK_SELECTION_MULTIPLE );
668
669 /* misc properties
670 */
671 gtk_tree_view_set_enable_tree_lines( treeview, TRUE );
672
673 if( items_view->private->mode == TREE_MODE_EDITION ){
674 column = gtk_tree_view_get_column( treeview, TREE_COLUMN_LABEL );
675 renderers = gtk_cell_layout_get_cells( GTK_CELL_LAYOUT( column ));
676 renderer = GTK_CELL_RENDERER( renderers->data );
677 gtk_tree_view_column_set_cell_data_func(
678 column, renderer, ( GtkTreeCellDataFunc ) display_label, items_view, NULL );
679 }
680}

682static void
683on_base_initialize_view( BaseWindow *window, gpointer user_data )
684{
685 static const gchar *thisfn = "cact_tree_view_on_base_initialize_view";
686 CactTreeView *items_view;
687 GtkTreeView *treeview;
688 GtkTreeSelection *selection;
689
690 items_view = CACT_TREE_VIEW( g_object_get_data( G_OBJECT( window ), WINDOW_DATA_TREE_VIEW ));
691
692 if( !items_view->private->dispose_has_run ){
693
694 g_debug( "%s: window=%p (%s), user_data=%p",
695 thisfn,
696 ( void * ) window, G_OBJECT_TYPE_NAME( window ),
697 ( void * ) user_data );
698
699 treeview = GTK_TREE_VIEW( get_tree_view( items_view ));
700 items_view->private->tree_view = treeview;
701
702 /* monitors the selection */
703 selection = gtk_tree_view_get_selection( treeview );
704 base_window_signal_connect(
705 window,
706 G_OBJECT( selection ),
707 "changed",
708 G_CALLBACK( on_selection_changed ));
709
710 /* delay all other signal connections until the widget be realized
711 */
712 base_window_signal_connect(
713 window,
714 G_OBJECT( treeview ),
715 "realize",
716 G_CALLBACK( on_tree_view_realized ));
717
718 if( items_view->private->mode == TREE_MODE_EDITION ){
719 cact_tree_ieditable_initialize( CACT_TREE_IEDITABLE( items_view ), treeview, window );
720 }
721 }
