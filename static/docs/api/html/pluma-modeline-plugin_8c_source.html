168static void
169connect_handlers (PlumaView *view)
170{
171 DocumentData *data;
172 GtkTextBuffer *doc;
173
174 doc = gtk_text_view_get_buffer (GTK_TEXT_VIEW (view));
175
176 data = g_slice_new (DocumentData);
177
178 data->document_loaded_handler_id =
179 g_signal_connect (doc, "loaded",
180 G_CALLBACK (on_document_loaded_or_saved),
181 view);
182 data->document_saved_handler_id =
183 g_signal_connect (doc, "saved",
184 G_CALLBACK (on_document_loaded_or_saved),
185 view);
186
187 g_object_set_data_full (G_OBJECT (doc), DOCUMENT_DATA_KEY,
188 data, (GDestroyNotify) document_data_free);
189}
190
191static void
192disconnect_handlers (PlumaView *view)
193{
194 DocumentData *data;
195 GtkTextBuffer *doc;
196
197 doc = gtk_text_view_get_buffer (GTK_TEXT_VIEW (view));
198
199 data = g_object_steal_data (G_OBJECT (doc), DOCUMENT_DATA_KEY);
200
201 if (data)
202 {
203 g_signal_handler_disconnect (doc, data->document_loaded_handler_id);
204 g_signal_handler_disconnect (doc, data->document_saved_handler_id);
205
206 document_data_free (data);
207 }
208 else
209 {
210 g_warning ("Modeline handlers not found");
211 }
212}

230static void
231pluma_modeline_plugin_activate (PlumaWindowActivatable *activatable)
232{
233 PlumaModelinePluginPrivate *data;
234 PlumaWindow *window;
235 GList *views;
236 GList *l;
237
238 pluma_debug (DEBUG_PLUGINS);
239
240 data = PLUMA_MODELINE_PLUGIN (activatable)->priv;
241 window = PLUMA_WINDOW (data->window);
242
243 views = pluma_window_get_views (window);
244 for (l = views; l != NULL; l = l->next)
245 {
246 connect_handlers (PLUMA_VIEW (l->data));
247 modeline_parser_apply_modeline (GTK_SOURCE_VIEW (l->data));
248 }
249 g_list_free (views);
250
251 data->tab_added_handler_id =
252 g_signal_connect (window, "tab-added",
253 G_CALLBACK (on_window_tab_added), NULL);
254
255 data->tab_removed_handler_id =
256 g_signal_connect (window, "tab-removed",
257 G_CALLBACK (on_window_tab_removed), NULL);
258}

260static void
261pluma_modeline_plugin_deactivate (PlumaWindowActivatable *activatable)
262{
263 PlumaModelinePluginPrivate *data;
264 PlumaWindow *window;
265 GList *views;
266 GList *l;
267
268 pluma_debug (DEBUG_PLUGINS);
269
270 data = PLUMA_MODELINE_PLUGIN (activatable)->priv;
271 window = PLUMA_WINDOW (data->window);
272
273 g_signal_handler_disconnect (window, data->tab_added_handler_id);
274 g_signal_handler_disconnect (window, data->tab_removed_handler_id);
275
276 views = pluma_window_get_views (window);
277
278 for (l = views; l != NULL; l = l->next)
279 {
280 disconnect_handlers (PLUMA_VIEW (l->data));
281
282 modeline_parser_deactivate (GTK_SOURCE_VIEW (l->data));
283 }
284
285 g_list_free (views);
286}
