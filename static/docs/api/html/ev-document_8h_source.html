224/* convenience macro to ease interface addition in the CODE
225 * section of EV_BACKEND_REGISTER_WITH_CODE (this macro relies on
226 * the g_define_type_id present within EV_BACKEND_REGISTER_WITH_CODE()).
227 * usage example:
228 * EV_BACKEND_REGISTER_WITH_CODE (PdfDocument, pdf_document,
229 * EV_BACKEND_IMPLEMENT_INTERFACE (EV_TYPE_DOCUMENT_THUMBNAILS,
230 * pdf_document_document_thumbnails_iface_init));
231 */
232#define EV_BACKEND_IMPLEMENT_INTERFACE(TYPE_IFACE, iface_init) { \
233 const GInterfaceInfo g_implement_interface_info = { \
234 (GInterfaceInitFunc) iface_init, NULL, NULL \
235 }; \
236 g_type_module_add_interface (module, \
237 g_define_type_id, \
238 TYPE_IFACE, \
239 &g_implement_interface_info); \
240}

242/*
243 * Utility macro used to register backends
244 *
245 * use: EV_BACKEND_REGISTER_WITH_CODE(BackendName, backend_name, CODE)
246 */
247#define EV_BACKEND_REGISTER_WITH_CODE(BackendName, backend_name, CODE) \
248 \
249static GType g_define_type_id = 0; \
250 \
251GType \
252backend_name##_get_type (void) \
253{ \
254 return g_define_type_id; \
255} \
256 \
257static void backend_name##_init (BackendName *self); \
258static void backend_name##_class_init (BackendName##Class *klass); \
259static gpointer backend_name##_parent_class = NULL; \
260static void backend_name##_class_intern_init (gpointer klass) \
261{ \
262 backend_name##_parent_class = g_type_class_peek_parent (klass); \
263 backend_name##_class_init ((BackendName##Class *) klass); \
264} \
265 \
266G_MODULE_EXPORT GType \
267register_atril_backend (GTypeModule *module) \
268{ \
269 const GTypeInfo our_info = { \
270 sizeof (BackendName##Class), \
271 NULL, /* base_init */ \
272 NULL, /* base_finalize */ \
273 (GClassInitFunc) backend_name##_class_intern_init, \
274 NULL, \
275 NULL, /* class_data */ \
276 sizeof (BackendName), \
277 0, /* n_preallocs */ \
278 (GInstanceInitFunc) backend_name##_init \
279 }; \
280 \
281 g_define_type_id = g_type_module_register_type (module, \
282 EV_TYPE_DOCUMENT, \
283 #BackendName, \
284 &our_info, \
285 (GTypeFlags)0); \
286 \
287 CODE \
288 \
289 return g_define_type_id; \
290}
