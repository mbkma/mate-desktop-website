31static char *
32met_reprocess (char *x, int len)
33{
34 char *p = x;
35 char *o;
36 int spacing = 0;
37 static gchar *buf;
38 static gint buflen = 0;
39 gchar *lastspace = NULL;
40 int count = 0;
41
42 if (buflen < len)
43 {
44 if (buf)
45 g_free (buf);
46 buf = g_malloc (len + 1);
47 buflen = len;
48 }
49
50 o = buf;
51 x += len; /* End mark */
52
53 while (*p && p < x) {
54 if (g_ascii_isspace (*p)) {
55 if (!spacing) {
56 spacing = 1;
57 lastspace = o;
58 count++;
59 *o++ = ' ';
60 }
61 p++;
62 continue;
63 }
64 spacing = 0;
65 if (count > 75 && lastspace) {
66 count = o - lastspace - 1;
67 *lastspace = '\n';
68 lastspace = NULL;
69 }
70
71 if (*p == '&') {
72 if (g_ascii_strncasecmp (p, "&amp;", 5) == 0) {
73 *o++ = '&';
74 count++;
75 p += 5;
76 continue;
77 }
78 if (g_ascii_strncasecmp (p, "&lt;", 4) == 0) {
79 *o++ = '<';
80 count++;
81 p += 4;
82 continue;
83 }
84 if (g_ascii_strncasecmp (p, "&gt;", 4) == 0) {
85 *o++ = '>';
86 count++;
87 p += 4;
88 continue;
89 }
90 }
91 if (*p == '<') {
92 if (g_ascii_strncasecmp (p, "<BR>", 4) == 0) {
93 *o++ = '\n';
94 count = 0;
95 }
96 if (g_ascii_strncasecmp (p, "<B>", 3) == 0) {
97 *o++ = '\n';
98 *o++ = '\n';
99 count = 0;
100 }
101 p++;
102 while (*p && *p != '>')
103 p++;
104 if (*p)
105 p++;
106 continue;
107 }
108 *o++ = *p++;
109 count++;
110 }
111 *o = 0;
112 return buf;
113}

121static gchar *
122met_parse (const gchar *meto)
123{
124 gchar *p;
125 gchar *rp;
126 gchar *r = g_strdup ("Met Office Forecast\n");
127 gchar *t;
128
129 g_return_val_if_fail (meto != NULL, r);
130
131 p = strstr (meto, "Summary: </b>");
132 g_return_val_if_fail (p != NULL, r);
133
134 rp = strstr (p, "Text issued at:");
135 g_return_val_if_fail (rp != NULL, r);
136
137 p += 13;
138 /* p to rp is the text block we want but in HTML malformat */
139 t = g_strconcat (r, met_reprocess (p, rp - p), NULL);
140 g_free (r);
141
142 return t;
143}

145static void
146met_finish (SoupSession *session, SoupMessage *msg, gpointer data)
147{
148 WeatherInfo *info = (WeatherInfo *)data;
149
150 g_return_if_fail (info != NULL);
151
152 if (!SOUP_STATUS_IS_SUCCESSFUL (msg->status_code)) {
153 g_warning ("Failed to get Met Office forecast data: %d %s.\n",
154 msg->status_code, msg->reason_phrase);
155 request_done (info, FALSE);
156 return;
157 }
158
159 info->forecast = met_parse (msg->response_body->data);
160 request_done (info, TRUE);
161}
