102static void
103launch_url (Launcher *launcher)
104{
105 char *url;
106 GdkScreen *screen;
107
108 g_return_if_fail (launcher != NULL);
109 g_return_if_fail (launcher->key_file != NULL);
110
111 /* FIXME panel_ditem_launch() should be enough for this! */
112 url = panel_key_file_get_string (launcher->key_file, "URL");
113
114 screen = launcher_get_screen (launcher);
115
116 if (!url || *url == 0) {
117 GtkWidget *error_dialog;
118
119 error_dialog = panel_error_dialog (NULL, screen,
120 "no_url_dialog", TRUE,
121 _("Could not show this URL"),
122 _("No URL was specified."));
123 launcher_register_error_dialog (launcher, error_dialog);
124 g_free (url);
125 return;
126 }
127
128 panel_show_uri (screen, url, gtk_get_current_event_time (), NULL);
129
130 g_free (url);
131}

133void
134launcher_launch (Launcher *launcher,
135 const gchar *action)
136{
137 char *type;
138
139 g_return_if_fail (launcher != NULL);
140 g_return_if_fail (launcher->key_file != NULL);
141
142 if (action == NULL) {
143 type = panel_key_file_get_string (launcher->key_file, "Type");
144 } else {
145 type = NULL;
146 }
147
148 if (type && !strcmp (type, "Link"))
149 launch_url (launcher);
150 else {
151 GError *error = NULL;
152
153 panel_launch_key_file (launcher->key_file, NULL,
154 launcher_get_screen (launcher), action, &error);
155 if (error) {
156 GtkWidget *error_dialog;
157
158 error_dialog = panel_error_dialog (
159 NULL,
160 launcher_get_screen (launcher),
161 "cannot_launch_application",
162 TRUE,
163 _("Could not launch application"),
164 error->message);
165 launcher_register_error_dialog (launcher, error_dialog);
166 g_clear_error (&error);
167 }
168 }
169 g_free (type);
170}

172static void
173drag_data_received_cb (GtkWidget *widget,
174 GdkDragContext *context,
175 gint x,
176 gint y,
177 GtkSelectionData *selection_data,
178 guint info,
179 guint time,
180 Launcher *launcher)
181{
182 GError *error = NULL;
183 char **uris;
184 int i;
185 GList *file_list;
186
187 /* The animation uses X specific functionality */
188#ifdef HAVE_X11
189 if (is_using_x11 () && panel_global_config_get_enable_animations ()) {
190 cairo_surface_t *surface;
191 surface = button_widget_get_surface (BUTTON_WIDGET (widget));
192 xstuff_zoom_animate (widget,
193 surface,
194 button_widget_get_orientation (BUTTON_WIDGET (widget)),
195 NULL);
196 cairo_surface_destroy (surface);
197 }
198#endif
199
200 file_list = NULL;
201 uris = g_uri_list_extract_uris ((const char *) gtk_selection_data_get_data (selection_data));
202 for (i = 0; uris[i]; i++)
203 file_list = g_list_prepend (file_list, uris[i]);
204 file_list = g_list_reverse (file_list);
205
206 panel_launch_key_file (launcher->key_file, file_list,
207 launcher_get_screen (launcher), NULL, &error);
208
209 g_list_free (file_list);
210 g_strfreev (uris);
211
212 if (error) {
213 GtkWidget *error_dialog;
214 error_dialog = panel_error_dialog (NULL,
215 launcher_get_screen (launcher),
216 "cannot_use_dropped_item",
217 TRUE,
218 _("Could not use dropped item"),
219 error->message);
220 launcher_register_error_dialog (launcher, error_dialog);
221 g_clear_error (&error);
222 }
223
224 gtk_drag_finish (context, TRUE, FALSE, time);
225}

291static gboolean
292is_this_drop_ok (GtkWidget *widget,
293 GdkDragContext *context)
294{
295 static GdkAtom text_uri_list = GDK_NONE;
296 GList *l;
297 GtkWidget *source;
298
299 source = gtk_drag_get_source_widget (context);
300
301 if (source == widget)
302 return FALSE;
303
304 if (!(gdk_drag_context_get_actions (context) & GDK_ACTION_COPY))
305 return FALSE;
306
307 if (!text_uri_list)
308 text_uri_list = gdk_atom_intern_static_string ("text/uri-list");
309
310 for (l = gdk_drag_context_list_targets (context); l; l = l->next) {
311 if (GDK_POINTER_TO_ATOM (l->data) == text_uri_list)
312 break;
313 }
314
315 return l ? TRUE : FALSE;
316}

440static Launcher *
441create_launcher (const char *location)
442{
443 GKeyFile *key_file;
444 g_autoptr(GFile) file = NULL;
445 gboolean loaded = FALSE;
446 Launcher *launcher;
447 GError *error = NULL;
448 char *new_location;
449
450 if (!location) {
451 g_printerr (_("No URI provided for panel launcher desktop file\n"));
452 return NULL;
453 }
454
455 new_location = NULL;
456 key_file = g_key_file_new ();
457
458 if (!strchr (location, G_DIR_SEPARATOR)) {
459 /* try to first load a file in our config directory, and if it
460 * doesn't exist there, try to find it in the xdg data dirs */
461 char *path;
462
463 path = panel_make_full_path (NULL, location);
464
465 if (!g_file_test (path, G_FILE_TEST_EXISTS)) {
466 g_free (path);
467 path = panel_g_lookup_in_applications_dirs (location);
468 /* it's important to keep the full path if the desktop
469 * file comes from a data dir: when the user will edit
470 * it, we'll want to save it in PANEL_LAUNCHERS_PATH
471 * with a random name (and not evolution.desktop, eg)
472 * and having only a basename as location will make
473 * this impossible */
474 if (path)
475 new_location = g_strdup (path);
476 }
477
478 if (path) {
479 loaded = g_key_file_load_from_file (key_file, path,
480 G_KEY_FILE_KEEP_COMMENTS|G_KEY_FILE_KEEP_TRANSLATIONS,
481 &error);
482 g_free (path);
483 }
484 } else
485 loaded = panel_key_file_load_from_uri (key_file, location,
486 G_KEY_FILE_KEEP_COMMENTS|G_KEY_FILE_KEEP_TRANSLATIONS,
487 &error);
488
489 if (!loaded) {
490 g_printerr (_("Unable to open desktop file %s for panel launcher%s%s\n"),
491 location,
492 error ? ": " : "",
493 error ? error->message : "");
494 if (error)
495 g_error_free (error);
496
497 g_key_file_free (key_file);
498 g_free (new_location);
499 return NULL; /*button is null*/
500 }
501
502 if (!new_location)
503 new_location = g_strdup (location);
504 file = g_file_new_for_path (new_location);
505
506 launcher = g_new0 (Launcher, 1);
507
508 launcher->info = NULL;
509 launcher->button = NULL;
510 launcher->location = new_location;
511 launcher->key_file = key_file;
512 launcher->prop_dialog = NULL;
513 launcher->destroy_handler = 0;
514
515 /* Icon will be setup later */
516 launcher->button = button_widget_new (NULL /* icon */,
517 FALSE,
518 PANEL_ORIENTATION_TOP);
519
520 launcher->monitor = g_file_monitor_file (file,
521 G_FILE_MONITOR_NONE,
522 NULL,
523 NULL);
524 gtk_widget_show (launcher->button);
525
526 /*gtk_drag_dest_set (GTK_WIDGET (launcher->button),
527 GTK_DEST_DEFAULT_ALL,
528 dnd_targets, 2,
529 GDK_ACTION_COPY);*/
530 gtk_drag_dest_set (GTK_WIDGET (launcher->button),
531 0, NULL, 0, 0);
532
533 g_signal_connect (launcher->button, "drag-data-get",
534 G_CALLBACK (drag_data_get_cb), launcher);
535 g_signal_connect (launcher->button, "drag-data-received",
536 G_CALLBACK (drag_data_received_cb), launcher);
537 g_signal_connect (launcher->button, "drag-motion",
538 G_CALLBACK (drag_motion_cb), launcher);
539 g_signal_connect (launcher->button, "drag-drop",
540 G_CALLBACK (drag_drop_cb), launcher);
541 g_signal_connect (launcher->button, "drag-leave",
542 G_CALLBACK (drag_leave_cb), launcher);
543 g_signal_connect_swapped (launcher->button, "clicked",
544 G_CALLBACK (clicked_cb), launcher);
545
546 launcher->destroy_handler =
547 g_signal_connect (launcher->button, "destroy",
548 G_CALLBACK (destroy_launcher),
549 launcher);
550
551 return launcher;
552}

554static void
555setup_actions (Launcher *launcher)
556{
557 GDesktopAppInfo *app_info;
558 const gchar * const *actions;
559 const gchar * const *ptr;
560
561 app_info = g_desktop_app_info_new_from_keyfile (launcher->key_file);
562 if (app_info == NULL)
563 return;
564
565 actions = g_desktop_app_info_list_actions (app_info);
566 ptr = actions;
567 for(; *ptr != NULL; ptr++) {
568 const gchar *action = *ptr;
569 gchar *callback = g_strdup_printf("launch-action_%s", action);
570 gchar *action_name = g_desktop_app_info_get_action_name (app_info, action);
571 mate_panel_applet_add_callback (launcher->info,
572 callback,
573 NULL,
574 action_name,
575 NULL);
576 g_free (callback);
577 g_free (action_name);
578 }
579
580 g_object_unref (app_info);
581}

583static void
584setup_button (Launcher *launcher)
585{
586 char *comment;
587 char *name;
588 char *str;
589 char *icon;
590 char *unescaped_str;
591
592 g_return_if_fail (launcher != NULL);
593
594 mate_panel_applet_clear_user_menu (launcher->info);
595
596 mate_panel_applet_add_callback (launcher->info,
597 "launch",
598 "system-run",
599 _("_Launch"),
600 NULL);
601
602 setup_actions (launcher);
603
604 mate_panel_applet_add_callback (launcher->info,
605 "properties",
606 "document-properties",
607 _("_Properties"),
608 launcher_properties_enabled);
609
610 name = panel_key_file_get_locale_string (launcher->key_file, "Name");
611 comment = panel_key_file_get_locale_string (launcher->key_file,
612 "Comment");
613
614 /* Setup tooltip */
615 if (!PANEL_GLIB_STR_EMPTY (name) && !PANEL_GLIB_STR_EMPTY (comment))
616 str = g_strdup_printf ("%s\n%s", name, comment);
617 else if (!PANEL_GLIB_STR_EMPTY (name))
618 str = g_strdup (name);
619 else
620 str = g_strdup (comment);
621
622 /* If we can unescape the string, then we probably have an escaped
623 * string (a location e.g.). If we can't, then it most probably means
624 * we have a % that is not here to encode a character, and we don't
625 * want to unescape in this case. See bug #170516 for details. */
626 unescaped_str = g_uri_unescape_string (str, NULL);
627 if (unescaped_str) {
628 g_free (str);
629 str = unescaped_str;
630 }
631
632 panel_util_set_tooltip_text (launcher->button, str);
633
634 /* Setup accessible name */
635 panel_a11y_set_atk_name_desc (launcher->button, str, NULL);
636
637 g_free (str);
638
639 /* Setup icon */
640 icon = panel_key_file_get_locale_string (launcher->key_file, "Icon");
641 if (icon && icon[0] == '\0')
642 g_clear_pointer (&icon, g_free);
643
644 if (!icon) {
645 gchar *exec;
646 exec = panel_key_file_get_string (launcher->key_file, "Exec");
647 if (exec && exec[0] != '\0') {
648 icon = guess_icon_from_exec (button_widget_get_icon_theme (BUTTON_WIDGET (launcher->button)),
649 exec);
650 }
651
652 g_free (exec);
653 }
654
655 if (!icon)
656 icon = g_strdup (PANEL_ICON_LAUNCHER);
657
658 button_widget_set_icon_name (BUTTON_WIDGET (launcher->button), icon);
659 g_free (icon);
660 g_free (name);
661 g_free (comment);
662}

664static char *
665panel_launcher_find_writable_uri (const char *launcher_location,
666 const char *source)
667{
668 if (!launcher_location)
669 return panel_make_unique_desktop_uri (NULL, source);
670
671 if (!strchr (launcher_location, G_DIR_SEPARATOR)) {
672 char *path = panel_make_full_path (NULL, launcher_location);
673 char *uri = g_filename_to_uri (path, NULL, NULL);
674 g_free (path);
675 return uri;
676 }
677
678 char *filename = panel_launcher_get_filename (launcher_location);
679 if (filename != NULL) {
680 g_free (filename);
681 /* we have a file in the user directory. We either have a path
682 * or an URI */
683 if (g_path_is_absolute (launcher_location))
684 return g_filename_to_uri (launcher_location,
685 NULL, NULL);
686 else
687 return g_strdup (launcher_location);
688 }
689
690 return panel_make_unique_desktop_uri (NULL, source);
691}

720static char *
721launcher_save_uri (PanelDItemEditor *dialog,
722 gpointer data)
723{
724 GKeyFile *key_file;
725 char *type;
726 char *exec_or_uri;
727 Launcher *launcher;
728 char *new_uri;
729 const char *uri;
730
731 key_file = panel_ditem_editor_get_key_file (dialog);
732 type = panel_key_file_get_string (key_file, "Type");
733 if (type && !strcmp (type, "Application"))
734 exec_or_uri = panel_key_file_get_string (key_file, "Exec");
735 else if (type && !strcmp (type, "Link"))
736 exec_or_uri = panel_key_file_get_string (key_file, "URL");
737 else
738 exec_or_uri = panel_key_file_get_string (key_file, "Name");
739 g_free (type);
740
741 launcher = (Launcher *) data;
742
743 if (launcher) {
744 new_uri = panel_launcher_find_writable_uri (launcher->location,
745 exec_or_uri);
746 }
747 else
748 new_uri = panel_launcher_find_writable_uri (NULL, exec_or_uri);
749
750 g_free (exec_or_uri);
751
752 uri = panel_ditem_editor_get_uri (dialog);
753
754 if (!uri || (new_uri && strcmp (new_uri, uri)))
755 return new_uri;
756
757 g_free (new_uri);
758
759 return NULL;
760}

762static void
763launcher_saved (GtkWidget *dialog,
764 Launcher *launcher)
765{
766 const char *uri;
767 char *filename;
768
769 uri = panel_ditem_editor_get_uri (PANEL_DITEM_EDITOR (dialog));
770 filename = panel_launcher_get_filename(uri);
771 if (filename)
772 uri = filename;
773
774 if (uri && launcher->location && strcmp (uri, launcher->location)) {
