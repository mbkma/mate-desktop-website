54void
55eel_report_check_failure (char *result, char *expected)
56{
57 if (!failed)
58 {
59 fprintf (stderr, "\n");
60 }
61
62 fprintf (stderr, "FAIL: check failed in %s, line %d\n", current_file_name, current_line_number);
63 fprintf (stderr, " evaluated: %s\n", current_expression);
64 fprintf (stderr, " expected: %s\n", expected == NULL ? "NULL" : expected);
65 fprintf (stderr, " got: %s\n", result == NULL ? "NULL" : result);
66
67 failed = TRUE;
68
69 g_free (result);
70 g_free (expected);
71}

198void
199eel_check_string_result (char *result, const char *expected)
200{
201 gboolean match;
202
203 /* Stricter than eel_strcmp.
204 * NULL does not match "" in this test.
205 */
206 if (expected == NULL)
207 {
208 match = result == NULL;
209 }
210 else
211 {
212 match = result != NULL && strcmp (result, expected) == 0;
213 }
214
215 if (!match)
216 {
217 eel_report_check_failure (result, g_strdup (expected));
218 }
219 else
220 {
221 g_free (result);
222 }
223 eel_after_check ();
224}
