145static void show_help(FishApplet* fish, const char* link_id)
146{
147 GError *error = NULL;
148 char *uri;
149#define FISH_HELP_DOC "mate-fish"
150
151 if (link_id)
152 uri = g_strdup_printf ("help:%s/%s", FISH_HELP_DOC, link_id);
153 else
154 uri = g_strdup_printf ("help:%s", FISH_HELP_DOC);
155
156 gtk_show_uri_on_window (NULL, uri,
157 gtk_get_current_event_time (), &error);
158 g_free (uri);
159
160 if (error &&
161 g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
162 g_error_free (error);
163 else if (error) {
164 GtkWidget *dialog;
165 char *primary;
166
167 primary = g_markup_printf_escaped (
168 _("Could not display help document '%s'"),
169 FISH_HELP_DOC);
170 dialog = gtk_message_dialog_new (
171 NULL,
172 GTK_DIALOG_DESTROY_WITH_PARENT,
173 GTK_MESSAGE_ERROR,
174 GTK_BUTTONS_CLOSE,
175 "%s", primary);
176
177 gtk_message_dialog_format_secondary_text (
178 GTK_MESSAGE_DIALOG (dialog),
179 "%s", error->message);
180
181 g_error_free (error);
182 g_free (primary);
183
184 g_signal_connect (dialog, "response",
185 G_CALLBACK (gtk_widget_destroy),
186 NULL);
187
188 gtk_window_set_icon_name (GTK_WINDOW (dialog), FISH_ICON);
189 gtk_window_set_screen (GTK_WINDOW (dialog),
190 gtk_widget_get_screen (GTK_WIDGET (fish)));
191 /* we have no parent window */
192 gtk_window_set_skip_taskbar_hint (GTK_WINDOW (dialog), FALSE);
193 gtk_window_set_title (GTK_WINDOW (dialog),
194 _("Error displaying help document"));
195
196 gtk_widget_show (dialog);
197 }

212static void image_value_changed(GtkFileChooser* chooser, FishApplet* fish)
213{
214 char *path;
215 char *image;
216 char *path_gsettings;
217
218 path = gtk_file_chooser_get_filename (GTK_FILE_CHOOSER (chooser));
219
220 if (!path || !path[0]) {
221 g_free (path);
222 return;
223 }
224
225 path_gsettings = get_image_path (fish);
226 if (!strcmp (path, path_gsettings)) {
227 g_free (path);
228 g_free (path_gsettings);
229 return;
230 }
231 g_free (path_gsettings);
232
233 if (!strncmp (path, FISH_ICONDIR, strlen (FISH_ICONDIR))) {
234 image = path + strlen (FISH_ICONDIR);
235 while (*image && *image == G_DIR_SEPARATOR)
236 image++;
237 } else
238 image = path;
239
240 g_settings_set_string (fish->settings, FISH_IMAGE_KEY, image);
241
242 g_free (path);
243}

245static void command_value_changed(GtkEntry* entry, FishApplet *fish)
246{
247 const char *text;
248
249 text = gtk_entry_get_text (entry);
250
251 if (!text || !text [0]) {
252 g_settings_set_string (fish->settings, FISH_COMMAND_KEY, "");
253 return;
254 }
255
256 if (!strncmp (text, "ps ", 3) ||
257 !strcmp (text, "ps") ||
258 !strncmp (text, "who ", 4) ||
259 !strcmp (text, "who") ||
260 !strcmp (text, "uptime") ||
261 !strncmp (text, "tail ", 5)) {
262 static gboolean message_given = FALSE;
263 const char *warning_format =
264 _("Warning: The command "
265 "appears to be something actually useful.\n"
266 "Since this is a useless applet, you "
267 "may not want to do this.\n"
268 "We strongly advise you against "
269 "using %s for anything\n"
270 "which would make the applet "
271 "\"practical\" or useful.");
272
273 if (!message_given) {
274 char *message = g_strdup_printf (warning_format, fish->name);
275 something_fishy_going_on (fish, message);
276 g_free (message);
277 message_given = TRUE;
278 }
279 }
280
281 g_settings_set_string (fish->settings, FISH_COMMAND_KEY, text);
282}

613static gboolean locate_fortune_command (FishApplet* fish, int* argcp, char*** argvp)
614{
615 char *prog = NULL;
616
617 if (fish->command
618 && g_shell_parse_argv (fish->command, argcp, argvp, NULL)) {
619 prog = g_find_program_in_path ((*argvp)[0]);
620 if (prog) {
621 g_free (prog);
622 return TRUE;
623 }
624
625 g_strfreev (*argvp);
626 }
627
628 prog = g_find_program_in_path ("fortune");
629 if (prog) {
630 g_free (prog);
631 if (g_shell_parse_argv ("fortune", argcp, argvp, NULL))
632 return FALSE;
633 }
634
635 if (g_file_test ("/usr/games/fortune", G_FILE_TEST_IS_EXECUTABLE)
636 && g_shell_parse_argv ("/usr/games/fortune", argcp, argvp, NULL))
637 return FALSE;
638
639 something_fishy_going_on (fish,
640 _("Unable to locate the command to execute"));
641 *argvp = NULL;
642 return FALSE;
643}

724static gboolean fish_read_output(GIOChannel* source, GIOCondition condition, gpointer data)
725{
726 char output[4096];
727 gsize bytes_read;
728 GError *error = NULL;
729 GIOStatus status;
730 FishApplet *fish;
731
732 fish = (FishApplet *) data;
733
734 if (!(condition & G_IO_IN)) {
735 fish->source_id = 0;
736 fish_close_channel (fish);
737 return FALSE;
738 }
739
740 status = g_io_channel_read_chars (source, output, 4096, &bytes_read,
741 &error);
742
743 if (error) {
744 char *message;
745
746 message = g_strdup_printf (_("Unable to read output from command\n\nDetails: %s"),
747 error->message);
748 something_fishy_going_on (fish, message);
749 g_free (message);
750 g_error_free (error);
751 fish->source_id = 0;
752 fish_close_channel (fish);
753 return FALSE;
754 }
755
756 if (status == G_IO_STATUS_AGAIN)
757 return TRUE;
758
759 if (bytes_read > 0) {
760 char *utf8_output;
761
762 /* The output is not guarantied to be in UTF-8 format, most
763 * likely it's just in ASCII-7 or in the user locale
764 */
765 if (!g_utf8_validate (output, -1, NULL))
766 utf8_output = g_locale_to_utf8 (output, bytes_read,
767 NULL, NULL, NULL);
768 else
769 utf8_output = g_strndup (output, bytes_read);
770
771 if (utf8_output)
772 insert_fortune_text (fish, utf8_output);
773
774 g_free (utf8_output);
775 }
776
777 if (status == G_IO_STATUS_EOF) {
778 fish->source_id = 0;
779 fish_close_channel (fish);
780 }
781 return (status != G_IO_STATUS_EOF);
782}

1094static char* get_location(void)
1095{
1096 static char location [256];
1097 char *buffer;
1098 FILE *zone;
1099 int i, len, count;
1100
1101 /* Old method : works for glibc < 2.2 */
1102 zone = fopen("/etc/timezone", "r");
1103 if (zone) {
1104 count = fscanf (zone, "%255s", location);
1105 fclose (zone);
1106 /* if we could read it, we return what we got */
1107 if (count == 1)
1108 return location;
1109 }
1110
1111 /* New method : works for glibc 2.2 */
1112 /* FIXME: this is broken for many distros, see the clock code */
1113 buffer = g_file_read_link ("/etc/localtime", NULL);
1114 if (!buffer)
1115 return NULL;
1116
1117 len = strlen (buffer);
1118 for (i = len, count = 0; (i > 0) && (count != 2); i--)
1119 if (buffer [i] == '/')
1120 count++;
1121
1122 if (count != 2) {
1123 g_free (buffer);
1124 return NULL;
1125 }
1126
1127 memcpy (location, &buffer [i + 2], len - i - 2);
1128 g_free (buffer);
1129
1130 return location;
1131}

1133static void init_fools_day(void)
1134{
1135 const char *spanish_timezones [] = {
1136 "Europe/Madrid",
1137 "Africa/Ceuta",
1138 "Atlantic/Canary",
1139 "America/Mexico_City",

1157 if (!g_ascii_strcasecmp (spanish_timezones [i], location)) {
1158 /* Hah!, We are in Spain or Mexico

1355 cairo_pattern_t *pattern;
1356
1357 gtk_widget_get_allocation (widget, &allocation);
1358
1359 if (!gtk_widget_get_realized (widget) ||
1360 allocation.width <= 0 ||
1361 allocation.height <= 0)
1362 return;

1364 if (!fish->pixbuf && !load_fish_image (fish))
1365 return;

1368 (fish->orientation == MATE_PANEL_APPLET_ORIENT_LEFT ||

1373 pixbuf_height = gdk_pixbuf_get_height (fish->pixbuf);
1374
1375 prev_requisition = fish->requisition;
1376
1377 if (fish->orientation == MATE_PANEL_APPLET_ORIENT_UP ||
1378 fish->orientation == MATE_PANEL_APPLET_ORIENT_DOWN) {
1379 height = allocation.height;
1380 width = pixbuf_width * ((gdouble) height / pixbuf_height);
1381 fish->requisition.width = width / fish->n_frames;
1382 fish->requisition.height = height;
1383 } else {
