72/* Returns the name of the media player.
73 * User must free. */
74static gchar*
75get_player_name(const gchar *name)
76{
77 gchar **tokens;
78 gchar *player_name;
79
80 /* max_tokens is 4 because a player could have additional instances,
81 * like org.mpris.MediaPlayer2.vlc.instance7389 */
82 tokens = g_strsplit (name, ".", 4);
83 player_name = g_strdup (tokens[3]);
84 g_strfreev (tokens);
85
86 return player_name;
87}

89/* A media player was just run and should be
90 * added to the head of media_player_queue. */
91static void
92mp_name_appeared (GDBusConnection *connection,
93 const gchar *name,
94 const gchar *name_owner,
95 gpointer user_data)
96{
97 MsdMprisManager *manager = user_data;
98 gchar *player_name;
99
100 g_debug ("MPRIS Name acquired: %s\n", name);
101
102 player_name = get_player_name(name);
103 g_queue_push_head (manager->priv->media_player_queue,
104 player_name);
105}

107/* A media player quit running and should be
108 * removed from media_player_queue. */
109static void
110mp_name_vanished (GDBusConnection *connection,
111 const gchar *name,
112 gpointer user_data)
113{
114 MsdMprisManager *manager = user_data;
115 gchar *player_name;
116 GList *player_list;
117
118 if (g_queue_is_empty (manager->priv->media_player_queue))
119 return;
120
121 g_debug ("MPRIS Name vanished: %s\n", name);
122
123 player_name = get_player_name(name);
124
125 player_list = g_queue_find_custom (manager->priv->media_player_queue,
126 player_name, (GCompareFunc) g_strcmp0);
127
128 if (player_list)
129 g_queue_remove (manager->priv->media_player_queue, player_list->data);
130
131 g_free (player_name);
132}

134/* Code copied from Totem media player
135 * src/plugins/media-player-keys/totem-media-player-keys.c */
136static void
137on_media_player_key_pressed (MsdMprisManager *manager,
138 const gchar *key)
139{
140 GDBusProxyFlags flags = G_DBUS_PROXY_FLAGS_DO_NOT_AUTO_START;
141 GDBusProxy *proxy = NULL;
142 GError *error = NULL;
143 const char *mpris_key = NULL;
144 const char *mpris_head = NULL;
145 char *mpris_name = NULL;
146
147 if (g_queue_is_empty (manager->priv->media_player_queue))
148 return;
149
150 if (strcmp ("Play", key) == 0)
151 mpris_key = "PlayPause";
152 else if (strcmp ("Pause", key) == 0)
153 mpris_key = "Pause";
154 else if (strcmp ("Previous", key) == 0)
155 mpris_key = "Previous";
156 else if (strcmp ("Next", key) == 0)
157 mpris_key = "Next";
158 else if (strcmp ("Stop", key) == 0)
159 mpris_key = "Stop";
160
161 if (mpris_key != NULL)
162 {
163 mpris_head = g_queue_peek_head (manager->priv->media_player_queue);
164 mpris_name = g_strdup_printf (MPRIS_PREFIX "%s", mpris_head);
165
166 g_debug ("MPRIS Sending '%s' to '%s'!", mpris_key, mpris_head);
167
168 proxy = g_dbus_proxy_new_for_bus_sync(G_BUS_TYPE_SESSION,
169 flags,
170 NULL,
171 mpris_name,
172 MPRIS_OBJECT_PATH,
173 MPRIS_INTERFACE,
174 NULL,
175 &error);
176 g_free (mpris_name);
177 if (proxy == NULL)
178 {
179 g_printerr("Error creating proxy: %s\n", error->message);
180 g_error_free(error);
181 return;
182 }
183
184 g_dbus_proxy_call (proxy, mpris_key, NULL,
185 G_DBUS_CALL_FLAGS_NONE,
186 -1, NULL, NULL, NULL);
187 g_object_unref (proxy);
188
189 }
190}

192static void
193grab_media_player_keys_cb (GDBusProxy *proxy,
194 GAsyncResult *res,
195 MsdMprisManager *manager)
196{
197 GVariant *variant;
198 GError *error = NULL;
199
200 variant = g_dbus_proxy_call_finish (proxy, res, &error);
201
202 if (variant == NULL) {
203 if (!g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
204 g_warning ("Failed to call \"GrabMediaPlayerKeys\": %s", error->message);
205 g_error_free (error);
206 return;
207 }
208 g_variant_unref (variant);
209}

211static void
212grab_media_player_keys (MsdMprisManager *manager)
213{
214 if (manager->priv->media_keys_proxy == NULL)
215 return;
216
217 g_dbus_proxy_call (manager->priv->media_keys_proxy,
218 "GrabMediaPlayerKeys",
219 g_variant_new ("(su)", "MsdMpris", 0),
220 G_DBUS_CALL_FLAGS_NONE,
221 -1, NULL,
222 (GAsyncReadyCallback) grab_media_player_keys_cb,
223 manager);
224}

226static void
227key_pressed (GDBusProxy *proxy G_GNUC_UNUSED,
228 gchar *sender_name G_GNUC_UNUSED,
229 gchar *signal_name,
230 GVariant *parameters,
231 MsdMprisManager *manager)
232{
233 char *app, *cmd;
234
235 if (g_strcmp0 (signal_name, "MediaPlayerKeyPressed") != 0)
236 return;
237 g_variant_get (parameters, "(ss)", &app, &cmd);
238 if (g_strcmp0 (app, "MsdMpris") == 0) {
239 on_media_player_key_pressed (manager, cmd);
240 }
241 g_free (app);
242 g_free (cmd);
243}

245static void
246got_proxy_cb (GObject *source_object,
247 GAsyncResult *res,
248 MsdMprisManager *manager)
249{
250 GError *error = NULL;
251
252 manager->priv->media_keys_proxy = g_dbus_proxy_new_for_bus_finish (res, &error);
253
254 if (manager->priv->media_keys_proxy == NULL) {
255 if (!g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
256 g_warning ("Failed to contact settings daemon: %s", error->message);
257 g_error_free (error);
258 return;
259 }
260
261 grab_media_player_keys (manager);
262
263 g_signal_connect (G_OBJECT (manager->priv->media_keys_proxy), "g-signal",
264 G_CALLBACK (key_pressed), manager);
265}

296gboolean
297msd_mpris_manager_start (MsdMprisManager *manager,
298 GError **error)
299{
300 g_debug ("Starting mpris manager");
301 mate_settings_profile_start (NULL);
302
303 manager->priv->media_player_queue = g_queue_new();
304
305 /* Register the namespace we wish to watch. */
306 manager->priv->namespace_watcher_id = bus_watch_namespace (G_BUS_TYPE_SESSION,
307 "org.mpris.MediaPlayer2",
308 mp_name_appeared,
309 mp_name_vanished,
310 manager,
311 NULL);
312
313 manager->priv->watch_id = g_bus_watch_name (G_BUS_TYPE_SESSION,
314 "org.mate.SettingsDaemon",
315 G_BUS_NAME_WATCHER_FLAGS_NONE,
316 (GBusNameAppearedCallback) msd_name_appeared,
317 (GBusNameVanishedCallback) msd_name_vanished,
318 manager, NULL);
319
320 mate_settings_profile_end (NULL);
321 return TRUE;
322}
