83static void
84view_as_menu_switch_views_callback (GtkComboBox *combo_box, CajaNavigationWindow *window)
85{
86 int active;
87
88 g_assert (GTK_IS_COMBO_BOX (combo_box));
89 g_assert (CAJA_IS_NAVIGATION_WINDOW (window));
90
91 active = gtk_combo_box_get_active (combo_box);
92
93 if (active < 0)
94 {
95 return;
96 }
97 else if (active < GPOINTER_TO_INT (g_object_get_data (G_OBJECT (combo_box), "num viewers")))
98 {
99 activate_nth_short_list_item (CAJA_WINDOW (window), active);
100 }
101 else
102 {
103 activate_extra_viewer (CAJA_WINDOW (window));
104 }
105}

107static void
108view_as_changed_callback (CajaWindow *window,
109 GtkComboBox *combo_box)
110{
111 CajaWindowSlot *slot;
112 GList *node;
113 int index;
114 int selected_index = -1;
115 GtkTreeModel *model;
116 GtkListStore *store;
117 const CajaViewInfo *info;
118
119 /* Clear the contents of ComboBox in a wacky way because there
120 * is no function to clear all items and also no function to obtain
121 * the number of items in a combobox.
122 */
123 model = gtk_combo_box_get_model (combo_box);
124 g_return_if_fail (GTK_IS_LIST_STORE (model));
125 store = GTK_LIST_STORE (model);
126 gtk_list_store_clear (store);
127
128 slot = caja_window_get_active_slot (window);
129
130 /* Add a menu item for each view in the preferred list for this location. */
131 for (node = window->details->short_list_viewers, index = 0;
132 node != NULL;
133 node = node->next, ++index)
134 {
135 info = caja_view_factory_lookup (node->data);
136 gtk_combo_box_text_append_text (GTK_COMBO_BOX_TEXT (combo_box), _(info->view_combo_label));
137
138 if (caja_window_slot_content_view_matches_iid (slot, (char *)node->data))
139 {
140 selected_index = index;
141 }
142 }
143 g_object_set_data (G_OBJECT (combo_box), "num viewers", GINT_TO_POINTER (index));
144
145 if (g_list_length (window->details->short_list_viewers) == 1)
146 {
147 selected_index = 0;
148 }
149
150 if (selected_index == -1)
151 {
152 const char *id;
153 /* We're using an extra viewer, add a menu item for it */
154
155 id = caja_window_slot_get_content_view_id (slot);
156 info = caja_view_factory_lookup (id);
157 gtk_combo_box_text_append_text (GTK_COMBO_BOX_TEXT (combo_box),
158 _(info->view_combo_label));
159 selected_index = index;
160 }
161 gtk_combo_box_set_active (combo_box, selected_index);
162 if (g_list_length (window->details->short_list_viewers) == 1) {
163 gtk_widget_hide(GTK_WIDGET(combo_box));
164 } else {
165 gtk_widget_show(GTK_WIDGET(combo_box));
166 }
167}

169static void
170connect_proxy (GtkAction *action,
171 GtkWidget *proxy)
172{
173 if (GTK_IS_TOOL_ITEM (proxy))
174 {
175 GtkToolItem *item = GTK_TOOL_ITEM (proxy);
176 CajaViewAsAction *vaction = CAJA_VIEW_AS_ACTION (action);
177 CajaNavigationWindow *window = vaction->priv->window;
178 GtkWidget *view_as_menu_vbox;
179 GtkWidget *view_as_combo_box;
180
181 /* Option menu for content view types; it's empty here, filled in when a uri is set.
182 * Pack it into vbox so it doesn't grow vertically when location bar does.
183 */
184 view_as_menu_vbox = gtk_box_new (GTK_ORIENTATION_VERTICAL, 4);
185 gtk_widget_show (view_as_menu_vbox);
186
187 gtk_container_add (GTK_CONTAINER (item), view_as_menu_vbox);
188
189 view_as_combo_box = gtk_combo_box_text_new ();
190
191 gtk_widget_set_focus_on_click (view_as_combo_box, FALSE);
192 gtk_box_pack_end (GTK_BOX (view_as_menu_vbox), view_as_combo_box, TRUE, FALSE, 0);
193 gtk_widget_show (view_as_combo_box);
194 g_signal_connect_object (view_as_combo_box, "changed",
195 G_CALLBACK (view_as_menu_switch_views_callback), window, 0);
196
197 g_signal_connect (window, "view-as-changed",
198 G_CALLBACK (view_as_changed_callback),
199 view_as_combo_box);
200 }
201
202 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
203 (* GTK_ACTION_CLASS (parent_class)->connect_proxy) (action, proxy);
204 G_GNUC_END_IGNORE_DEPRECATIONS;
205}

207static void
208disconnect_proxy (GtkAction *action,
209 GtkWidget *proxy)
210{
211 if (GTK_IS_TOOL_ITEM (proxy))
212 {
213 CajaViewAsAction *vaction = CAJA_VIEW_AS_ACTION (action);
214 CajaNavigationWindow *window = vaction->priv->window;
215
216 g_signal_handlers_disconnect_matched (window,
217 G_SIGNAL_MATCH_FUNC,
218 0, 0, NULL, G_CALLBACK (view_as_changed_callback), NULL);
219 }
220
221 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
222 (* GTK_ACTION_CLASS (parent_class)->disconnect_proxy) (action, proxy);
223 G_GNUC_END_IGNORE_DEPRECATIONS;
