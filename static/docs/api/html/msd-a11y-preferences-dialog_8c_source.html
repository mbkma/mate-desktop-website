59/* X servers sometimes lie about the screen's physical dimensions, so we cannot
60 * compute an accurate DPI value. When this happens, the user gets fonts that
61 * are too huge or too tiny. So, we see what the server returns: if it reports
62 * something outside of the range [DPI_LOW_REASONABLE_VALUE,
63 * DPI_HIGH_REASONABLE_VALUE], then we assume that it is lying and we use
64 * DPI_FALLBACK instead.
65 *
66 * See get_dpi_from_gsettings_or_server() below, and also
67 * https://bugzilla.novell.com/show_bug.cgi?id=217790
68 */
69#define DPI_LOW_REASONABLE_VALUE 50
70#define DPI_HIGH_REASONABLE_VALUE 500
71
72#define DPI_FACTOR_LARGE 1.25
73#define DPI_FACTOR_LARGER 1.5
74#define DPI_FACTOR_LARGEST 2.0
75#define DPI_DEFAULT 96

151static double
152dpi_from_pixels_and_mm (int pixels,
153 int mm)
154{
155 double dpi;
156
157 if (mm >= 1) {
158 dpi = pixels / (mm / 25.4);
159 } else {
160 dpi = 0;
161 }
162
163 return dpi;
164}

166static double
167get_dpi_from_x_server (void)
168{
169 GdkScreen *screen;
170 double dpi;
171 int scale;
172
173 screen = gdk_screen_get_default ();
174 if (screen != NULL) {
175 double width_dpi;
176 double height_dpi;
177
178 Screen *xscreen = gdk_x11_screen_get_xscreen (screen);
179
180 scale = gdk_window_get_scale_factor (gdk_screen_get_root_window (screen));
181 width_dpi = dpi_from_pixels_and_mm (WidthOfScreen (xscreen), WidthMMOfScreen (xscreen));
182 height_dpi = dpi_from_pixels_and_mm (HeightOfScreen (xscreen), HeightMMOfScreen (xscreen));
183
184 if (width_dpi < DPI_LOW_REASONABLE_VALUE
185 || width_dpi > DPI_HIGH_REASONABLE_VALUE
186 || height_dpi < DPI_LOW_REASONABLE_VALUE
187 || height_dpi > DPI_HIGH_REASONABLE_VALUE) {
188 dpi = DPI_DEFAULT;
189 } else {
190 dpi = (width_dpi + height_dpi) / 2.0;
191 }
192
193 dpi *= scale;
194
195 } else {
196 /* Huh!? No screen? */
197 dpi = DPI_DEFAULT;
198 }
199
200 return dpi;
201}

203static gboolean
204config_get_large_print (gboolean *is_writable)
205{
206 GSettings *settings;
207 gboolean ret;
208 gdouble x_dpi;
209 gdouble u_dpi;
210 gdouble gs_dpi;
211
212 settings = g_settings_new (FONT_RENDER_SCHEMA);
213
214 gs_dpi = g_settings_get_double (settings, KEY_FONT_DPI);
215
216 if (gs_dpi != 0) {
217 u_dpi = gs_dpi;
218 } else {
219 u_dpi = DPI_DEFAULT;
220 }
221
222 x_dpi = get_dpi_from_x_server ();
223
224 g_object_unref (settings);
225
226 g_debug ("MsdA11yPreferences: got x-dpi=%f user-dpi=%f", x_dpi, u_dpi);
227
228 ret = (((double)DPI_FACTOR_LARGE * x_dpi) < u_dpi);
229
230 return ret;
231}

233static void
234config_set_large_print (gboolean enabled)
235{
236 GSettings *settings;
237
238 settings = g_settings_new (FONT_RENDER_SCHEMA);
239
240 if (enabled) {
241 gdouble x_dpi;
242 gdouble u_dpi;
243
244 x_dpi = get_dpi_from_x_server ();
245 u_dpi = (double)DPI_FACTOR_LARGER * x_dpi;
246
247 g_debug ("MsdA11yPreferences: setting x-dpi=%f user-dpi=%f", x_dpi, u_dpi);
248
249 g_settings_set_double (settings, KEY_FONT_DPI, u_dpi);
250 } else {
251 g_settings_reset (settings, KEY_FONT_DPI);
252 }
253
254 g_object_unref (settings);
255}

257static gboolean
258config_get_high_contrast (MsdA11yPreferencesDialog *dialog, gboolean *is_writable)
259{
260 gboolean ret;
261 char *gtk_theme;
262
263 ret = FALSE;
264
265 *is_writable = g_settings_is_writable (dialog->priv->settings_interface, KEY_GTK_THEME);
266 gtk_theme = g_settings_get_string (dialog->priv->settings_interface, KEY_GTK_THEME);
267
268 if (gtk_theme != NULL && strcmp (gtk_theme, HIGH_CONTRAST_THEME) == 0) {
269 ret = TRUE;
270 }
271 g_free (gtk_theme);
272
273 return ret;
274}

340static gboolean
341config_have_at_gsettings_condition (const char *condition)
342{
343 GDBusProxy *proxy = NULL;
344 GError *error = NULL;
345 GVariant *ret;
346 gboolean is_handled;
347
348 proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SYSTEM,
349 G_DBUS_PROXY_FLAGS_NONE,
350 NULL,
351 SM_DBUS_NAME,
352 SM_DBUS_PATH,
353 SM_DBUS_INTERFACE,
354 NULL,
355 &error);
356 if (proxy == NULL) {
357 g_warning ("Unable to connect to session bus: %s", error->message);
358 g_error_free (error);
359 return FALSE;
360 }
361
362 is_handled = FALSE;
363 ret = g_dbus_proxy_call_sync (proxy,
364 "IsAutostartConditionHandled",
365 g_variant_new ("(s)", condition),
366 G_DBUS_CALL_FLAGS_NONE,
367 -1,
368 NULL,
369 &error);
370 if (ret == NULL) {
371 g_warning ("Unable to call IsAutostartConditionHandled (%s): %s",
372 condition,
373 error->message);
374 g_error_free (error);
375 } else {
376 g_variant_get (ret, "(b)", &is_handled);
377 g_variant_unref (ret);
378 }
379
380 g_object_unref (proxy);
381
382 return is_handled;
383}

596static void
597key_changed_cb (GSettings *settings,
598 gchar *key,
599 MsdA11yPreferencesDialog *dialog)
600{
601 if (g_strcmp0 (key, KEY_STICKY_KEYS_ENABLED) == 0) {
602 gboolean enabled;
603 enabled = g_settings_get_boolean (settings, key);
604 ui_set_sticky_keys (dialog, enabled);
605 } else if (g_strcmp0 (key, KEY_BOUNCE_KEYS_ENABLED) == 0) {
606 gboolean enabled;
607 enabled = g_settings_get_boolean (settings, key);
608 ui_set_bounce_keys (dialog, enabled);
609 } else if (g_strcmp0 (key, KEY_SLOW_KEYS_ENABLED) == 0) {
610 gboolean enabled;
611 enabled = g_settings_get_boolean (settings, key);
612 ui_set_slow_keys (dialog, enabled);
613#ifdef HAVE_LIBATSPI
614 } else if (g_strcmp0 (key, KEY_CAPSLOCK_BEEP_ENABLED) == 0) {
615 gboolean enabled;
616 enabled = g_settings_get_boolean (settings, key);
617 ui_set_capslock_beep (dialog, enabled);
618#endif
619 } else if (g_strcmp0 (key, KEY_AT_SCREEN_READER_ENABLED) == 0) {
620 gboolean enabled;
621 enabled = g_settings_get_boolean (settings, key);
622 ui_set_at_screen_reader (dialog, enabled);
623 } else if (g_strcmp0 (key, KEY_AT_SCREEN_KEYBOARD_ENABLED) == 0) {
624 gboolean enabled;
625 enabled = g_settings_get_boolean (settings, key);
626 ui_set_at_screen_keyboard (dialog, enabled);
627 } else if (strcmp (key, KEY_AT_SCREEN_MAGNIFIER_ENABLED) == 0) {
628 gboolean enabled;
629 enabled = g_settings_get_boolean (settings, key);
630 ui_set_at_screen_magnifier (dialog, enabled);
631 } else {
632 g_debug ("Config key not handled: %s", key);
633 }
634}

636static void
637setup_dialog (MsdA11yPreferencesDialog *dialog,
638 GtkBuilder *builder)
639{
640 GtkWidget *widget;
641 gboolean enabled;
642 gboolean is_writable;
643
644 widget = GTK_WIDGET (gtk_builder_get_object (builder,
645 "sticky_keys_checkbutton"));
646 dialog->priv->sticky_keys_checkbutton = widget;
647 g_signal_connect (widget,
648 "toggled",
649 G_CALLBACK (on_sticky_keys_checkbutton_toggled),
650 dialog);
651 enabled = config_get_sticky_keys (dialog, &is_writable);
652 ui_set_sticky_keys (dialog, enabled);
653 if (! is_writable) {
654 gtk_widget_set_sensitive (widget, FALSE);
655 }
656
657 widget = GTK_WIDGET (gtk_builder_get_object (builder,
658 "bounce_keys_checkbutton"));
659 dialog->priv->bounce_keys_checkbutton = widget;
660 g_signal_connect (widget,
661 "toggled",
662 G_CALLBACK (on_bounce_keys_checkbutton_toggled),
663 dialog);
664 enabled = config_get_bounce_keys (dialog, &is_writable);
665 ui_set_bounce_keys (dialog, enabled);
666 if (! is_writable) {
667 gtk_widget_set_sensitive (widget, FALSE);
668 }
669
670 widget = GTK_WIDGET (gtk_builder_get_object (builder,
671 "slow_keys_checkbutton"));
672 dialog->priv->slow_keys_checkbutton = widget;
