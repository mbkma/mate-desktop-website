234static cairo_surface_t *
235get_eject_icon (gboolean highlighted)
236{
237 GdkPixbuf *eject;
238 cairo_surface_t *eject_surface;
239 CajaIconInfo *eject_icon_info;
240 int icon_size, icon_scale;
241
242 icon_size = caja_get_icon_size_for_stock_size (GTK_ICON_SIZE_MENU);
243 icon_scale = gdk_window_get_scale_factor (gdk_get_default_root_window ());
244
245 eject_icon_info = caja_icon_info_lookup_from_name ("media-eject", icon_size, icon_scale);
246 eject = caja_icon_info_get_pixbuf_at_size (eject_icon_info, icon_size);
247
248 if (highlighted) {
249 GdkPixbuf *high;
250 high = eel_create_spotlight_pixbuf (eject);
251 g_object_unref (eject);
252 eject = high;
253 }
254
255 eject_surface = gdk_cairo_surface_create_from_pixbuf (eject, icon_scale, NULL);
256
257 g_object_unref (eject_icon_info);
258 g_object_unref (eject);
259
260 return eject_surface;
261}

263static gboolean
264is_built_in_bookmark (CajaFile *file)
265{
266 gboolean built_in;
267 gint idx;
268
269 built_in = FALSE;
270
271 for (idx = 0; idx < G_USER_N_DIRECTORIES; idx++) {
272 /* PUBLIC_SHARE and TEMPLATES are not in our built-in list */
273 if (caja_file_is_user_special_directory (file, idx)) {
274 if (idx != G_USER_DIRECTORY_PUBLIC_SHARE && idx != G_USER_DIRECTORY_TEMPLATES) {
275 built_in = TRUE;
276 }
277
278 break;
279 }
280 }
281
282 return built_in;
283}

309static void
310check_heading_for_section (CajaPlacesSidebar *sidebar,
311 SectionType section_type)
312{
313 switch (section_type) {
314 case SECTION_DEVICES:
315 if (!sidebar->devices_header_added) {
316 add_heading (sidebar, SECTION_DEVICES,
317 _("Devices"));
318 sidebar->devices_header_added = TRUE;
319 }
320
321 break;
322 case SECTION_BOOKMARKS:
323 if (!sidebar->bookmarks_header_added) {
324 add_heading (sidebar, SECTION_BOOKMARKS,
325 _("Bookmarks"));
326 sidebar->bookmarks_header_added = TRUE;
327 }
328
329 break;
330 default:
331 break;
332 }
333}

335static GtkTreeIter
336add_place (CajaPlacesSidebar *sidebar,
337 PlaceType place_type,
338 SectionType section_type,
339 const char *name,
340 GIcon *icon,
341 const char *uri,
342 GDrive *drive,
343 GVolume *volume,
344 GMount *mount,
345 const int index,
346 const char *tooltip)
347{
348 GdkPixbuf *pixbuf;
349 cairo_surface_t *surface;
350 GtkTreeIter iter, child_iter;
351 cairo_surface_t *eject;
352 CajaIconInfo *icon_info;
353 int icon_size;
354 int icon_scale;
355 gboolean show_eject;
356 gboolean show_unmount;
357 gboolean show_eject_button;
358
359 check_heading_for_section (sidebar, section_type);
360
361 icon_size = caja_get_icon_size_for_stock_size (GTK_ICON_SIZE_MENU);
362 icon_scale = gtk_widget_get_scale_factor (GTK_WIDGET (sidebar));
363 icon_info = caja_icon_info_lookup (icon, icon_size, icon_scale);
364
365 pixbuf = caja_icon_info_get_pixbuf_at_size (icon_info, icon_size);
366 g_object_unref (icon_info);
367
368 if (pixbuf != NULL)
369 {
370 surface = gdk_cairo_surface_create_from_pixbuf (pixbuf, icon_scale, NULL);
371 g_object_unref (pixbuf);
372 }
373 else
374 {
375 surface = NULL;
376 }
377
378 check_unmount_and_eject (mount, volume, drive,
379 &show_unmount, &show_eject);
380
381 if (show_unmount || show_eject)
382 {
383 g_assert (place_type != PLACES_BOOKMARK);
384 }
385
386 if (mount == NULL)
387 {
388 show_eject_button = FALSE;
389 }
390 else
391 {
392 show_eject_button = (show_unmount || show_eject);
393 }
394
395 if (show_eject_button) {
396 eject = get_eject_icon (FALSE);
397 } else {
398 eject = NULL;
399 }
400
401 gtk_list_store_append (sidebar->store, &iter);
402 gtk_list_store_set (sidebar->store, &iter,
403 PLACES_SIDEBAR_COLUMN_ICON, surface,
404 PLACES_SIDEBAR_COLUMN_NAME, name,
405 PLACES_SIDEBAR_COLUMN_URI, uri,
406 PLACES_SIDEBAR_COLUMN_DRIVE, drive,
407 PLACES_SIDEBAR_COLUMN_VOLUME, volume,
408 PLACES_SIDEBAR_COLUMN_MOUNT, mount,
409 PLACES_SIDEBAR_COLUMN_ROW_TYPE, place_type,
410 PLACES_SIDEBAR_COLUMN_INDEX, index,
411 PLACES_SIDEBAR_COLUMN_EJECT, show_eject_button,
412 PLACES_SIDEBAR_COLUMN_NO_EJECT, !show_eject_button,
413 PLACES_SIDEBAR_COLUMN_BOOKMARK, place_type != PLACES_BOOKMARK,
414 PLACES_SIDEBAR_COLUMN_TOOLTIP, tooltip,
415 PLACES_SIDEBAR_COLUMN_EJECT_ICON, eject,
416 PLACES_SIDEBAR_COLUMN_SECTION_TYPE, section_type,
417 -1);
418
419 if (surface != NULL)
420 {
421 cairo_surface_destroy (surface);
422 }
423 gtk_tree_model_filter_refilter (GTK_TREE_MODEL_FILTER (sidebar->filter_model));
424 gtk_tree_model_filter_convert_child_iter_to_iter (GTK_TREE_MODEL_FILTER (sidebar->filter_model),
425 &child_iter,
426 &iter);
427 return child_iter;
428}

430static void
431compare_for_selection (CajaPlacesSidebar *sidebar,
432 const gchar *location,
433 const gchar *added_uri,
434 const gchar *last_uri,
435 GtkTreeIter *iter,
436 GtkTreePath **path)
437{
438 int res;
439
440 res = g_strcmp0 (added_uri, last_uri);
441
442 if (res == 0)
443 {
444 /* last_uri always comes first */
445 if (*path != NULL)
446 {
447 gtk_tree_path_free (*path);
448 }
449 *path = gtk_tree_model_get_path (sidebar->filter_model,
450 iter);
451 }
452 else if (g_strcmp0 (location, added_uri) == 0)
453 {
454 if (*path == NULL)
455 {
456 *path = gtk_tree_model_get_path (sidebar->filter_model,
457 iter);
458 }
459 }
460}

462static void
463update_places (CajaPlacesSidebar *sidebar)
464{
465 CajaBookmark *bookmark;
466 GtkTreeSelection *selection;
467 GtkTreeIter last_iter;
468 GtkTreePath *select_path;
469 GtkTreeModel *model;
470 GVolumeMonitor *volume_monitor;
471 GList *mounts, *l, *ll;
472 GMount *mount;
473 GList *drives;
474 GDrive *drive;
475 GList *volumes;
476 GVolume *volume;
477 int bookmark_count, index;
478 char *location, *mount_uri, *name, *desktop_path, *last_uri;
479 const gchar *path;
480 GIcon *icon;
481 GFile *root;
482 CajaWindowSlotInfo *slot;
483 char *tooltip;
484 GList *network_mounts;
485 GList *xdg_dirs;
486 CajaFile *file;
487
488 model = NULL;
489 last_uri = NULL;
490 select_path = NULL;
491 bookmark = NULL;
492
493 selection = gtk_tree_view_get_selection (sidebar->tree_view);
494 if (gtk_tree_selection_get_selected (selection, &model, &last_iter))
495 {
496 gtk_tree_model_get (model,
497 &last_iter,
498 PLACES_SIDEBAR_COLUMN_URI, &last_uri, -1);
499 }
500 gtk_list_store_clear (sidebar->store);
501
502 sidebar->devices_header_added = FALSE;
503 sidebar->bookmarks_header_added = FALSE;
504
505 slot = caja_window_info_get_active_slot (sidebar->window);
506 location = caja_window_slot_info_get_current_location (slot);
507
508 volume_monitor = sidebar->volume_monitor;
509
510 /* COMPUTER */
511 last_iter = add_heading (sidebar, SECTION_COMPUTER,
512 _("Computer"));
513
514 /* add built in bookmarks */
515
516 /* home folder */
517 char *display_name;
518
519 mount_uri = caja_get_home_directory_uri ();
520 display_name = g_filename_display_basename (g_get_home_dir ());
521 icon = g_themed_icon_new (CAJA_ICON_HOME);
522 last_iter = add_place (sidebar, PLACES_BUILT_IN,
523 SECTION_COMPUTER,
524 display_name, icon,
525 mount_uri, NULL, NULL, NULL, 0,
526 _("Open your personal folder"));
527 g_object_unref (icon);
528 g_free (display_name);
529 compare_for_selection (sidebar,
530 location, mount_uri, last_uri,
531 &last_iter, &select_path);
532 g_free (mount_uri);
533
534 /* desktop */
535 desktop_path = caja_get_desktop_directory ();
536 if (strcmp (g_get_home_dir(), desktop_path) != 0) {
537 mount_uri = g_filename_to_uri (desktop_path, NULL, NULL);
538 icon = g_themed_icon_new (CAJA_ICON_DESKTOP);
539 last_iter = add_place (sidebar, PLACES_BUILT_IN,
540 SECTION_COMPUTER,
541 _("Desktop"), icon,
542 mount_uri, NULL, NULL, NULL, 0,
543 _("Open the contents of your desktop in a folder"));
544 g_object_unref (icon);
545 compare_for_selection (sidebar,
546 location, mount_uri, last_uri,
547 &last_iter, &select_path);
548 g_free (mount_uri);
549 }
550 g_free (desktop_path);
551
552 /* file system root */
553 mount_uri = "file:///"; /* No need to strdup */
554 icon = g_themed_icon_new (CAJA_ICON_FILESYSTEM);
555 last_iter = add_place (sidebar, PLACES_BUILT_IN,
556 SECTION_COMPUTER,
557 _("File System"), icon,
558 mount_uri, NULL, NULL, NULL, 0,
559 _("Open the contents of the File System"));
560 g_object_unref (icon);
561 compare_for_selection (sidebar,
562 location, mount_uri, last_uri,
563 &last_iter, &select_path);
564
565 /* XDG directories */
566 xdg_dirs = NULL;
567 for (index = 0; index < G_USER_N_DIRECTORIES; index++) {
568
569 if (index == G_USER_DIRECTORY_DESKTOP ||
570 index == G_USER_DIRECTORY_TEMPLATES ||
571 index == G_USER_DIRECTORY_PUBLIC_SHARE) {
572 continue;
573 }
574
575 path = g_get_user_special_dir (index);
576
577 /* xdg resets special dirs to the home directory in case
578 * it's not finiding what it expects. We don't want the home
579 * to be added multiple times in that weird configuration.
580 */
581 if (path == NULL
582 || g_strcmp0 (path, g_get_home_dir ()) == 0
583 || g_list_find_custom (xdg_dirs, path, (GCompareFunc) g_strcmp0) != NULL) {
584 continue;
585 }
586
587 root = g_file_new_for_path (path);
588 name = g_file_get_basename (root);
589 icon = caja_user_special_directory_get_gicon (index);
590 mount_uri = g_file_get_uri (root);
591 tooltip = g_file_get_parse_name (root);
592
593 last_iter = add_place (sidebar, PLACES_BUILT_IN,
594 SECTION_COMPUTER,
595 name, icon, mount_uri,
596 NULL, NULL, NULL, 0,
597 tooltip);
598 compare_for_selection (sidebar,
599 location, mount_uri, last_uri,
600 &last_iter, &select_path);
601 g_free (name);
602 g_object_unref (root);
603 g_object_unref (icon);
604 g_free (mount_uri);
605 g_free (tooltip);
606
607 xdg_dirs = g_list_prepend (xdg_dirs, (char *)path);
608 }
609 g_list_free (xdg_dirs);
610
611 mount_uri = "trash:///"; /* No need to strdup */
612 icon = caja_trash_monitor_get_icon ();
613 last_iter = add_place (sidebar, PLACES_BUILT_IN,
614 SECTION_COMPUTER,
615 _("Trash"), icon, mount_uri,
616 NULL, NULL, NULL, 0,
617 _("Open the trash"));
618 compare_for_selection (sidebar,
619 location, mount_uri, last_uri,
620 &last_iter, &select_path);
621 g_object_unref (icon);
622
623 /* first go through all connected drives */
624 drives = g_volume_monitor_get_connected_drives (volume_monitor);
625
626 for (l = drives; l != NULL; l = l->next)
627 {
628 drive = l->data;
629
630 volumes = g_drive_get_volumes (drive);
631 if (volumes != NULL)
632 {
633 for (ll = volumes; ll != NULL; ll = ll->next)
634 {
635 volume = ll->data;
636 mount = g_volume_get_mount (volume);
637 if (mount != NULL)
638 {
639 /* Show mounted volume in the sidebar */
640 icon = g_mount_get_icon (mount);
641 root = g_mount_get_default_location (mount);
642 mount_uri = g_file_get_uri (root);
643 name = g_mount_get_name (mount);
644 tooltip = g_file_get_parse_name (root);
645
646 last_iter = add_place (sidebar, PLACES_MOUNTED_VOLUME,
647 SECTION_DEVICES,
648 name, icon, mount_uri,
649 drive, volume, mount, 0, tooltip);
650 compare_for_selection (sidebar,
651 location, mount_uri, last_uri,
652 &last_iter, &select_path);
653 g_object_unref (root);
654 g_object_unref (mount);
655 g_object_unref (icon);
656 g_free (tooltip);
657 g_free (name);
658 g_free (mount_uri);
659 }
660 else
661 {
662 /* Do show the unmounted volumes in the sidebar;
663 * this is so the user can mount it (in case automounting
664 * is off).
665 *
666 * Also, even if automounting is enabled, this gives a visual
667 * cue that the user should remember to yank out the media if
668 * he just unmounted it.
669 */
670 icon = g_volume_get_icon (volume);
671 name = g_volume_get_name (volume);
672 tooltip = g_strdup_printf (_("Mount and open %s"), name);
673
674 last_iter = add_place (sidebar, PLACES_MOUNTED_VOLUME,
675 SECTION_DEVICES,
676 name, icon, NULL,
677 drive, volume, NULL, 0, tooltip);
678 g_object_unref (icon);
679 g_free (name);
680 g_free (tooltip);
681 }
682 g_object_unref (volume);
683 }
684 g_list_free (volumes);
685 }
686 else
687 {
688 if (g_drive_is_media_removable (drive) && !g_drive_is_media_check_automatic (drive))
689 {
690 /* If the drive has no mountable volumes and we cannot detect media change.. we
691 * display the drive in the sidebar so the user can manually poll the drive by
692 * right clicking and selecting "Rescan..."
693 *
694 * This is mainly for drives like floppies where media detection doesn't
695 * work.. but it's also for human beings who like to turn off media detection
696 * in the OS to save battery juice.
697 */
698 icon = g_drive_get_icon (drive);
699 name = g_drive_get_name (drive);
700 tooltip = g_strdup_printf (_("Mount and open %s"), name);
701
702 last_iter = add_place (sidebar, PLACES_BUILT_IN,
703 SECTION_DEVICES,
704 name, icon, NULL,
705 drive, NULL, NULL, 0, tooltip);
706 g_object_unref (icon);
707 g_free (tooltip);
708 g_free (name);
709 }
710 }
711 g_object_unref (drive);
712 }
713 g_list_free (drives);
714
715 /* add all volumes that is not associated with a drive */
716 volumes = g_volume_monitor_get_volumes (volume_monitor);
717 for (l = volumes; l != NULL; l = l->next)
718 {
719 volume = l->data;
720 drive = g_volume_get_drive (volume);
721 if (drive != NULL)
722 {
723 g_object_unref (volume);
724 g_object_unref (drive);
725 continue;
726 }
727 mount = g_volume_get_mount (volume);
728 if (mount != NULL)
729 {
730 icon = g_mount_get_icon (mount);
731 root = g_mount_get_default_location (mount);
732 mount_uri = g_file_get_uri (root);
733 tooltip = g_file_get_parse_name (root);
734 g_object_unref (root);
735 name = g_mount_get_name (mount);
736 last_iter = add_place (sidebar, PLACES_MOUNTED_VOLUME,
737 SECTION_DEVICES,
738 name, icon, mount_uri,
739 NULL, volume, mount, 0, tooltip);
740 compare_for_selection (sidebar,
741 location, mount_uri, last_uri,
742 &last_iter, &select_path);
743 g_object_unref (mount);
744 g_object_unref (icon);
745 g_free (name);
746 g_free (tooltip);
747 g_free (mount_uri);
748 }
749 else
750 {
751 /* see comment above in why we add an icon for an unmounted mountable volume */
752 icon = g_volume_get_icon (volume);
753 name = g_volume_get_name (volume);
754 last_iter = add_place (sidebar, PLACES_MOUNTED_VOLUME,
755 SECTION_DEVICES,
756 name, icon, NULL,
757 NULL, volume, NULL, 0, name);
758 g_object_unref (icon);
759 g_free (name);
760 }
761 g_object_unref (volume);
762 }
763 g_list_free (volumes);
764
765 /* add mounts that has no volume (/etc/mtab mounts, ftp, sftp,...) */
766 network_mounts = NULL;
767 mounts = g_volume_monitor_get_mounts (volume_monitor);
768
769 for (l = mounts; l != NULL; l = l->next)
770 {
771 mount = l->data;
772 if (g_mount_is_shadowed (mount))
773 {
774 g_object_unref (mount);
775 continue;
776 }
777 volume = g_mount_get_volume (mount);
778 if (volume != NULL)
779 {
780 g_object_unref (volume);
781 g_object_unref (mount);
782 continue;
783 }
784 root = g_mount_get_default_location (mount);
785
786 if (!g_file_is_native (root)) {
787 network_mounts = g_list_prepend (network_mounts, g_object_ref (mount));
788 continue;
789 }
790
791 icon = g_mount_get_icon (mount);
792 mount_uri = g_file_get_uri (root);
793 name = g_mount_get_name (mount);
794 tooltip = g_file_get_parse_name (root);
795 last_iter = add_place (sidebar, PLACES_MOUNTED_VOLUME,
796 SECTION_COMPUTER,
797 name, icon, mount_uri,
798 NULL, NULL, mount, 0, tooltip);
799 compare_for_selection (sidebar,
800 location, mount_uri, last_uri,
801 &last_iter, &select_path);
802 g_object_unref (root);
803 g_object_unref (mount);
804 g_object_unref (icon);
805 g_free (name);
806 g_free (mount_uri);
807 g_free (tooltip);
808 }
809 g_list_free (mounts);
810
811 /* add bookmarks */
812 bookmark_count = caja_bookmark_list_length (sidebar->bookmarks);
813
814 for (index = 0; index < bookmark_count; ++index) {
815 bookmark = caja_bookmark_list_item_at (sidebar->bookmarks, index);
816
817 if (caja_bookmark_uri_known_not_to_exist (bookmark)) {
818 continue;
819 }
820
821 root = caja_bookmark_get_location (bookmark);
822 file = caja_file_get (root);
823
824 if (is_built_in_bookmark (file)) {
825 g_object_unref (root);
826 caja_file_unref (file);
827 continue;
828 }
829
830 name = caja_bookmark_get_name (bookmark);
831 icon = caja_bookmark_get_icon (bookmark);
832 mount_uri = caja_bookmark_get_uri (bookmark);
833 tooltip = g_file_get_parse_name (root);
834
835 last_iter = add_place (sidebar, PLACES_BOOKMARK,
836 SECTION_BOOKMARKS,
837 name, icon, mount_uri,
838 NULL, NULL, NULL, index,
839 tooltip);
840 compare_for_selection (sidebar,
841 location, mount_uri, last_uri,
842 &last_iter, &select_path);
843 g_free (name);
844 g_object_unref (root);
845 g_object_unref (icon);
846 g_free (mount_uri);
847 g_free (tooltip);
848 }
849
850 /* network */
851 last_iter = add_heading (sidebar, SECTION_NETWORK,
852 _("Network"));
853
854 network_mounts = g_list_reverse (network_mounts);
855 for (l = network_mounts; l != NULL; l = l->next) {
856 mount = l->data;
857 root = g_mount_get_default_location (mount);
858 icon = g_mount_get_icon (mount);
859 mount_uri = g_file_get_uri (root);
860 name = g_mount_get_name (mount);
861 tooltip = g_file_get_parse_name (root);
862 last_iter = add_place (sidebar, PLACES_MOUNTED_VOLUME,
863 SECTION_NETWORK,
864 name, icon, mount_uri,
865 NULL, NULL, mount, 0, tooltip);
866 compare_for_selection (sidebar,
867 location, mount_uri, last_uri,
868 &last_iter, &select_path);
869 g_object_unref (root);
870 g_object_unref (mount);
871 g_object_unref (icon);
872 g_free (name);
873 g_free (mount_uri);
874 g_free (tooltip);
875 }
876
877 g_list_free_full (network_mounts, g_object_unref);
878
879 /* network:// */
880 mount_uri = "network:///"; /* No need to strdup */
881 icon = g_themed_icon_new (CAJA_ICON_NETWORK);
882 last_iter = add_place (sidebar, PLACES_BUILT_IN,
883 SECTION_NETWORK,
884 _("Browse Network"), icon,
885 mount_uri, NULL, NULL, NULL, 0,
886 _("Browse the contents of the network"));
887 g_object_unref (icon);
888 compare_for_selection (sidebar,
889 location, mount_uri, last_uri,
890 &last_iter, &select_path);
891
892 g_free (location);
893
894 if (select_path != NULL) {
895 gtk_tree_selection_select_path (selection, select_path);
896 }
897
898 if (select_path != NULL) {
899 gtk_tree_path_free (select_path);
900 }
901
902 g_free (last_uri);
903}

1105/* Computes the appropriate row and position for dropping */
1106static gboolean
1107compute_drop_position (GtkTreeView *tree_view,
1108 int x,
1109 int y,
1110 GtkTreePath **path,
1111 GtkTreeViewDropPosition *pos,
1112 CajaPlacesSidebar *sidebar)
1113{
1114 GtkTreeModel *model;
1115 GtkTreeIter iter;
1116 PlaceType place_type;
1117 SectionType section_type;
1118
1119 if (!gtk_tree_view_get_dest_row_at_pos (tree_view,
1120 x,
1121 y,
1122 path,
1123 pos)) {
1124 return FALSE;
1125 }
1126
1127 model = gtk_tree_view_get_model (tree_view);
1128
1129 gtk_tree_model_get_iter (model, &iter, *path);
1130 gtk_tree_model_get (model, &iter,
1131 PLACES_SIDEBAR_COLUMN_ROW_TYPE, &place_type,
1132 PLACES_SIDEBAR_COLUMN_SECTION_TYPE, &section_type,
1133 -1);
1134
1135 if (place_type == PLACES_HEADING &&
1136 section_type != SECTION_BOOKMARKS &&
1137 section_type != SECTION_NETWORK) {
1138 /* never drop on headings, but the bookmarks or network heading
1139 * is a special case, so we can create new bookmarks by dragging
1140 * at the beginning or end of the bookmark list.
1141 */
1142 gtk_tree_path_free (*path);
1143 *path = NULL;
1144
1145 return FALSE;
1146 }
1147
1148 if (section_type != SECTION_BOOKMARKS &&
1149 sidebar->drag_data_received &&
1150 sidebar->drag_data_info == GTK_TREE_MODEL_ROW) {
1151 /* don't allow dropping bookmarks into non-bookmark areas */
1152 gtk_tree_path_free (*path);
1153 *path = NULL;
1154
1155 return FALSE;
1156 }
1157
1158 /* drag to top or bottom of bookmark list to add a bookmark */
1159 if (place_type == PLACES_HEADING && section_type == SECTION_BOOKMARKS) {
1160 *pos = GTK_TREE_VIEW_DROP_AFTER;
1161 } else if (place_type == PLACES_HEADING && section_type == SECTION_NETWORK) {
1162 *pos = GTK_TREE_VIEW_DROP_BEFORE;
1163 } else {
1164 /* or else you want to drag items INTO the existing bookmarks */
1165 *pos = GTK_TREE_VIEW_DROP_INTO_OR_BEFORE;
1166 }
1167
1168 if (*pos != GTK_TREE_VIEW_DROP_BEFORE &&
1169 sidebar->drag_data_received &&
1170 sidebar->drag_data_info == GTK_TREE_MODEL_ROW) {
1171 /* bookmark rows are never dragged into other bookmark rows */
1172 *pos = GTK_TREE_VIEW_DROP_AFTER;
1173 }
1174
1175 return TRUE;
1176}

1245static gboolean
1246drag_motion_callback (GtkTreeView *tree_view,
1247 GdkDragContext *context,
1248 int x,
1249 int y,
1250 unsigned int time,
1251 CajaPlacesSidebar *sidebar)
1252{
1253 GtkTreePath *path;
1254 GtkTreeViewDropPosition pos;
1255 int action = 0;
1256 GtkTreeIter iter;
1257 char *uri;
1258 gboolean res;
1259
1260 if (!sidebar->drag_data_received)
1261 {
1262 if (!get_drag_data (tree_view, context, time))
1263 {
1264 return FALSE;
1265 }
1266 }
1267
1268 path = NULL;
1269 res = compute_drop_position (tree_view, x, y, &path, &pos, sidebar);
1270
1271 if (!res) {
1272 goto out;
1273 }
1274
1275 if (pos == GTK_TREE_VIEW_DROP_BEFORE ||
1276 pos == GTK_TREE_VIEW_DROP_AFTER )
1277 {
1278 if (sidebar->drag_data_received &&
1279 sidebar->drag_data_info == GTK_TREE_MODEL_ROW)
1280 {
1281 action = GDK_ACTION_MOVE;
1282 }
1283 else if (can_accept_items_as_bookmarks (sidebar->drag_list))
1284 {
1285 action = GDK_ACTION_COPY;
1286 }
1287 else
1288 {
1289 action = 0;
1290 }
1291 }
1292 else
1293 {
1294 if (sidebar->drag_list == NULL)
1295 {
1296 action = 0;
1297 }
1298 else
1299 {
1300 gtk_tree_model_get_iter (sidebar->filter_model,
1301 &iter, path);
1302 gtk_tree_model_get (GTK_TREE_MODEL (sidebar->filter_model),
1303 &iter,
1304 PLACES_SIDEBAR_COLUMN_URI, &uri,
1305 -1);
1306 caja_drag_default_drop_action_for_icons (context, uri,
1307 sidebar->drag_list,
1308 &action);
1309 g_free (uri);
1310 }
1311 }
1312
1313 if (action != 0) {
1314 gtk_tree_view_set_drag_dest_row (tree_view, path, pos);
1315 }
1316
1317 if (path != NULL) {
1318 gtk_tree_path_free (path);
1319 }
1320
1321 out:
1322 g_signal_stop_emission_by_name (tree_view, "drag-motion");
1323
1324 if (action != 0)
1325 {
1326 gdk_drag_status (context, action, time);
1327 }
1328 else
1329 {
1330 gdk_drag_status (context, 0, time);
1331 }
1332
1333 return TRUE;
1334}

1487static void
1488drag_data_received_callback (GtkWidget *widget,
1489 GdkDragContext *context,
1490 int x,
1491 int y,
1492 GtkSelectionData *selection_data,
1493 unsigned int info,
1494 unsigned int time,
1495 CajaPlacesSidebar *sidebar)
1496{
1497 GtkTreeView *tree_view;
1498 GtkTreePath *tree_path;
1499 GtkTreeViewDropPosition tree_pos;
1500 GtkTreeIter iter;
1501 int position;
1502 GtkTreeModel *model;
1503 char *drop_uri;
1504 GList *selection_list, *uris;
1505 PlaceType place_type;
1506 SectionType section_type;
1507 gboolean success;
1508
1509 tree_view = GTK_TREE_VIEW (widget);
1510
1511 if (!sidebar->drag_data_received)
1512 {
1513 if (gtk_selection_data_get_target (selection_data) != GDK_NONE &&
1514 info == TEXT_URI_LIST)
1515 {
1516 sidebar->drag_list = build_selection_list (gtk_selection_data_get_data (selection_data));
1517 }
1518 else
1519 {
1520 sidebar->drag_list = NULL;
1521 }
1522 sidebar->drag_data_received = TRUE;
1523 sidebar->drag_data_info = info;
1524 }
1525
1526 g_signal_stop_emission_by_name (widget, "drag-data-received");
1527
1528 if (!sidebar->drop_occured)
1529 {
1530 return;
1531 }
1532
1533 /* Compute position */
1534 success = compute_drop_position (tree_view, x, y, &tree_path, &tree_pos, sidebar);
1535 if (!success)
1536 goto out;
1537
1538 success = FALSE;
1539
1540 if (tree_pos == GTK_TREE_VIEW_DROP_BEFORE ||
1541 tree_pos == GTK_TREE_VIEW_DROP_AFTER)
1542 {
1543 model = gtk_tree_view_get_model (tree_view);
1544
1545 if (!gtk_tree_model_get_iter (model, &iter, tree_path))
1546 {
1547 goto out;
1548 }
1549
1550 gtk_tree_model_get (model, &iter,
1551 PLACES_SIDEBAR_COLUMN_SECTION_TYPE, &section_type,
1552 PLACES_SIDEBAR_COLUMN_ROW_TYPE, &place_type,
1553 PLACES_SIDEBAR_COLUMN_INDEX, &position,
1554 -1);
1555
1556 if (section_type != SECTION_BOOKMARKS &&
1557 !(section_type == SECTION_NETWORK && place_type == PLACES_HEADING)) {
1558 goto out;
1559 }
1560
1561 if (section_type == SECTION_NETWORK && place_type == PLACES_HEADING &&
1562 tree_pos == GTK_TREE_VIEW_DROP_BEFORE) {
1563 position = caja_bookmark_list_length (sidebar->bookmarks);
1564 }
1565
1566 if (tree_pos == GTK_TREE_VIEW_DROP_AFTER && place_type != PLACES_HEADING) {
1567 /* heading already has position 0 */
1568 position++;
1569 }
1570
1571 switch (info)
1572 {
1573 case TEXT_URI_LIST:
1574 bookmarks_drop_uris (sidebar, selection_data, position);
1575 success = TRUE;
1576 break;
1577 case GTK_TREE_MODEL_ROW:
1578 reorder_bookmarks (sidebar, position);
1579 success = TRUE;
1580 break;
1581 default:
1582 g_assert_not_reached ();
1583 break;
1584 }
1585 }
1586 else
1587 {
1588 GdkDragAction real_action;
1589
1590 /* file transfer requested */
1591 real_action = gdk_drag_context_get_selected_action (context);
1592
1593 if (real_action == GDK_ACTION_ASK)
1594 {
1595 real_action =
1596 caja_drag_drop_action_ask (GTK_WIDGET (tree_view),
1597 gdk_drag_context_get_actions (context));
1598 }
1599
1600 if (real_action > 0)
1601 {
1602 model = gtk_tree_view_get_model (tree_view);
1603
1604 gtk_tree_model_get_iter (model, &iter, tree_path);
1605 gtk_tree_model_get (model, &iter,
1606 PLACES_SIDEBAR_COLUMN_URI, &drop_uri,
1607 -1);
1608
1609 switch (info)
1610 {
1611 case TEXT_URI_LIST:
1612 selection_list = build_selection_list (gtk_selection_data_get_data (selection_data));
1613 uris = uri_list_from_selection (selection_list);
1614 caja_file_operations_copy_move (uris, NULL, drop_uri,
1615 real_action, GTK_WIDGET (tree_view),
1616 NULL, NULL);
1617 caja_drag_destroy_selection_list (selection_list);
1618 g_list_free (uris);
1619 success = TRUE;
1620 break;
1621 case GTK_TREE_MODEL_ROW:
1622 success = FALSE;
1623 break;
1624 default:
1625 g_assert_not_reached ();
1626 break;
1627 }
1628
1629 g_free (drop_uri);
1630 }
1631 }
1632
1633out:
1634 sidebar->drop_occured = FALSE;
1635 free_drag_data (sidebar);
1636 gtk_drag_finish (context, success, FALSE, time);
1637
1638 gtk_tree_path_free (tree_path);
1639}
