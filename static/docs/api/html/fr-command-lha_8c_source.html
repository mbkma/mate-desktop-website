45static time_t
46mktime_from_string (char *month,
47 char *mday,
48 char *time_or_year)
49{
50 struct tm tm = {0, };
51
52 tm.tm_isdst = -1;
53
54 /* date */
55
56 if (month != NULL) {
57 static const char *months[] = { "Jan", "Feb", "Mar", "Apr", "May", "Jun",
58 "Jul", "Aug", "Sep", "Oct", "Nov", "Dec" };
59 int i;
60
61 for (i = 0; i < 12; i++) {
62 if (strcmp (months[i], month) == 0) {
63 tm.tm_mon = i;
64 break;
65 }
66 }
67 }
68 tm.tm_mday = atoi (mday);
69 if (strchr (time_or_year, ':') == NULL)
70 tm.tm_year = atoi (time_or_year) - 1900;
71 else {
72 char **fields;
73 time_t now;
74 struct tm *tm_now;
75
76 now = time (NULL);
77 tm_now = localtime (&now);
78 if (tm_now != NULL)
79 tm.tm_year = tm_now->tm_year;
80
81 /* time */
82
83 fields = g_strsplit (time_or_year, ":", 2);
84 if (fields[0] != NULL) {
85 tm.tm_hour = atoi (fields[0]);
86 if (fields[1] != NULL)
87 tm.tm_min = atoi (fields[1]);
88 }
89 g_strfreev (fields);
90 }
91
92 return mktime (&tm);
93}

95static char **
96split_line_lha (char *line)
97{
98 char **fields;
99 int n_fields = 7;
100 const char *scan;
101 int i;
102
103 fields = g_new0 (char *, n_fields + 1);
104 fields[n_fields] = NULL;
105
106 i = 0;
107
108 if (strncmp (line, "[MS-DOS]", 8) == 0) {
109 fields[i++] = g_strdup ("");
110 fields[i++] = g_strdup ("");
111 line += strlen ("[MS-DOS]");
112 }
113 else if (strncmp (line, "[generic]", 9) == 0) {
114 fields[i++] = g_strdup ("");
115 fields[i++] = g_strdup ("");
116 line += strlen ("[generic]");
117 }
118 else if (strncmp (line, "[unknown]", 9) == 0) {
119 fields[i++] = g_strdup ("");
120 fields[i++] = g_strdup ("");
121 line += strlen ("[unknown]");
122 }
123 else if (strncmp (line, "[Amiga]", 7) == 0) {
124 fields[i++] = g_strdup ("");
125 fields[i++] = g_strdup ("");
126 line += strlen ("[Amiga]");
127 }
128
129 scan = eat_spaces (line);
130 for (; i < n_fields; i++) {
131 const char *field_end;
132
133 if (NULL != (field_end = strchr (scan, ' '))) {
134 fields[i] = g_strndup (scan, field_end - scan);
135 scan = eat_spaces (field_end);
136 }
137 }
138
139 return fields;
140}

142static const char *
143get_last_field_lha (char *line)
144{
145 int i;
146 const char *field;
147 int n = 7;
148
149 if (strncmp (line, "[MS-DOS]", 8) == 0)
150 n--;
151
152 if (strncmp (line, "[generic]", 9) == 0)
153 n--;
154
155 if (strncmp (line, "[unknown]", 9) == 0)
156 n--;
157
158 if (strncmp (line, "[Amiga]", 7) == 0)
159 n--;
160
161 field = eat_spaces (line);
162 for (i = 0; i < n; i++) {
163 field = strchr (field, ' ');
164 field = eat_spaces (field);
165 }
166
167 return field;
168}

170static void
171process_line (char *line,
172 gpointer data)
173{
174 FileData *fdata;
175 FrCommand *comm = FR_COMMAND (data);
176 char **fields;
177 const char *name_field;
178
179 g_return_if_fail (line != NULL);
180
181 fdata = file_data_new ();
182
183 fields = split_line_lha (line);
184 fdata->size = g_ascii_strtoull (fields[2], NULL, 10);
185 fdata->modified = mktime_from_string (fields[4],
186 fields[5],
187 fields[6]);
188 g_strfreev (fields);
189
190 /* Full path */
191
192 name_field = get_last_field_lha (line);
193
194 if (name_field && *name_field == '/') {
195 fdata->full_path = g_strdup (name_field);
196 fdata->original_path = fdata->full_path;
197 } else {
198 fdata->full_path = g_strconcat ("/", name_field, NULL);
199 fdata->original_path = fdata->full_path + 1;
200 }
201
202 fdata->link = NULL;
203
204 fdata->dir = line[0] == 'd';
205 if (fdata->dir)
206 fdata->name = dir_name_from_path (fdata->full_path);
207 else
208 fdata->name = g_strdup (file_name_from_path (fdata->full_path));
209
210 fdata->path = remove_level_from_path (fdata->full_path);
211
212 if (*fdata->name == 0)
213 file_data_free (fdata);
214 else
215 fr_command_add_file (comm, fdata);
216}

268static void
269fr_command_lha_extract (FrCommand *comm,
270 const char *from_file,
271 GList *file_list,
272 const char *dest_dir,
273 gboolean overwrite,
274 gboolean skip_older,
275 gboolean junk_paths)
276{
277 GList *scan;
278 char options[5];
279 int i = 0;
280
281 fr_process_begin_command (comm->process, "lha");
282
283 if (dest_dir != NULL)
284 fr_process_set_working_dir (comm->process, dest_dir);
285
286 options[i++] = 'x';
287 options[i++] = 'f'; /* Always overwrite.
288 * The overwrite option is handled in
289 * fr_archive_extract,
290 * this is because lha asks the user whether he
291 * wants to overwrite a file. */
292
293 if (junk_paths)
294 options[i++] = 'i';
295
296 options[i++] = 0;
297 fr_process_add_arg (comm->process, options);
298 fr_process_add_arg (comm->process, comm->filename);
299
300 for (scan = file_list; scan; scan = scan->next)
301 fr_process_add_arg (comm->process, scan->data);
302
303 fr_process_end_command (comm->process);
304}
