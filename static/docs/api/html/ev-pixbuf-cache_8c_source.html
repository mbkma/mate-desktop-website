311static void
312job_finished_cb (EvJob *job,
313 EvPixbufCache *pixbuf_cache)
314{
315 CacheJobInfo *job_info;
316 EvJobRender *job_render = EV_JOB_RENDER (job);
317
318 /* If the job is outside of our interest, we silently discard it */
319 if ((job_render->page < (pixbuf_cache->start_page - pixbuf_cache->preload_cache_size)) ||
320 (job_render->page > (pixbuf_cache->end_page + pixbuf_cache->preload_cache_size))) {
321 g_object_unref (job);
322 return;
323 }
324
325 job_info = find_job_cache (pixbuf_cache, job_render->page);
326
327 if (ev_job_is_failed (job)) {
328 job_info->job = NULL;
329 g_object_unref (job);
330 return;
331 }
332
333 copy_job_to_job_info (job_render, job_info, pixbuf_cache);
334 g_signal_emit (pixbuf_cache, signals[JOB_FINISHED], 0, job_info->region);
335}

342 CacheJobInfo *job_info,

344{

368/* Do all function that copies a job from an older cache to it's position in the
369 * new cache. It clears the old job if it doesn't have a place.
370 */
371static void
372move_one_job (CacheJobInfo *job_info,
373 EvPixbufCache *pixbuf_cache,
374 int page,
375 CacheJobInfo *new_job_list,
376 CacheJobInfo *new_prev_job,
377 CacheJobInfo *new_next_job,
378 int new_preload_cache_size,
379 int start_page,
380 int end_page,
381 gint priority)
382{
383 CacheJobInfo *target_page = NULL;
384 int page_offset;
385 gint new_priority;
386
387 if (page < (start_page - new_preload_cache_size) ||
388 page > (end_page + new_preload_cache_size)) {
389 dispose_cache_job_info (job_info, pixbuf_cache);
390 return;
391 }
392
393 /* find the target page to copy it over to. */
394 if (page < start_page) {
395 page_offset = (page - (start_page - new_preload_cache_size));
396
397 g_assert (page_offset >= 0 &&
398 page_offset < new_preload_cache_size);
399 target_page = new_prev_job + page_offset;
400 new_priority = EV_JOB_PRIORITY_LOW;
401 } else if (page > end_page) {
402 page_offset = (page - (end_page + 1));
403
404 g_assert (page_offset >= 0 &&
405 page_offset < new_preload_cache_size);
406 target_page = new_next_job + page_offset;
407 new_priority = EV_JOB_PRIORITY_LOW;
408 } else {
409 page_offset = page - start_page;
410 g_assert (page_offset >= 0 &&
411 page_offset <= ((end_page - start_page) + 1));
412 new_priority = EV_JOB_PRIORITY_URGENT;
413 target_page = new_job_list + page_offset;
414 }
415
416 *target_page = *job_info;
417 job_info->job = NULL;
418 job_info->region = NULL;
419 job_info->surface = NULL;
420
421 if (new_priority != priority && target_page->job) {
422 ev_job_scheduler_update_job (target_page->job, new_priority);
423 }
424}

440static gint
441ev_pixbuf_cache_get_preload_size (EvPixbufCache *pixbuf_cache,
442 gint start_page,
443 gint end_page,
444 gdouble scale,
445 gint rotation)
446{
447 gsize range_size = 0;
448 gint new_preload_cache_size = 0;
449 gint i;
450 guint n_pages = ev_document_get_n_pages (pixbuf_cache->document);
451
452 /* Get the size of the current range */
453 for (i = start_page; i <= end_page; i++) {
454 range_size += ev_pixbuf_cache_get_page_size (pixbuf_cache, i, scale, rotation);
455 }
456
457 if (range_size >= pixbuf_cache->max_size)
458 return new_preload_cache_size;
459
460 i = 1;
461 while (((start_page - i > 0) || (end_page + i < n_pages)) &&
462 new_preload_cache_size < MAX_PRELOADED_PAGES) {
463 gsize page_size;
464 gboolean updated = FALSE;
465
466 if (end_page + i < n_pages) {
467 page_size = ev_pixbuf_cache_get_page_size (pixbuf_cache, end_page + i,
468 scale, rotation);
469 if (page_size + range_size <= pixbuf_cache->max_size) {
470 range_size += page_size;
471 new_preload_cache_size++;
472 updated = TRUE;
473 } else {
474 break;
475 }
476 }
477
478 if (start_page - i > 0) {
479 page_size = ev_pixbuf_cache_get_page_size (pixbuf_cache, start_page - i,
480 scale, rotation);
481 if (page_size + range_size <= pixbuf_cache->max_size) {
482 range_size += page_size;
483 if (!updated)
484 new_preload_cache_size++;
485 } else {
486 break;
487 }
488 }
489 i++;
490 }
491
492 return new_preload_cache_size;
493}

495static void
496ev_pixbuf_cache_update_range (EvPixbufCache *pixbuf_cache,
497 gint start_page,
498 gint end_page,
499 guint rotation,
500 gdouble scale)
501{
502 CacheJobInfo *new_job_list;
503 CacheJobInfo *new_prev_job = NULL;
504 CacheJobInfo *new_next_job = NULL;
505 gint new_preload_cache_size;
506 guint new_job_list_len;
507 int i, page;
508
509 new_preload_cache_size = ev_pixbuf_cache_get_preload_size (pixbuf_cache,
510 start_page,
511 end_page,
512 scale,
513 rotation);
514 if (pixbuf_cache->start_page == start_page &&
515 pixbuf_cache->end_page == end_page &&
516 pixbuf_cache->preload_cache_size == new_preload_cache_size)
517 return;
518
519 new_job_list_len = (end_page - start_page) + 1;
520 new_job_list = g_slice_alloc0 (sizeof (CacheJobInfo) * new_job_list_len);
521 if (new_preload_cache_size > 0) {
522 new_prev_job = g_slice_alloc0 (sizeof (CacheJobInfo) * new_preload_cache_size);
523 new_next_job = g_slice_alloc0 (sizeof (CacheJobInfo) * new_preload_cache_size);
524 }
525
526 /* We go through each job in the old cache and either clear it or move
527 * it to a new location. */
528
529 /* Start with the prev cache. */
530 page = pixbuf_cache->start_page - pixbuf_cache->preload_cache_size;
531 for (i = 0; i < pixbuf_cache->preload_cache_size; i++) {
532 if (page < 0) {
533 dispose_cache_job_info (pixbuf_cache->prev_job + i, pixbuf_cache);
534 } else {
535 move_one_job (pixbuf_cache->prev_job + i,
536 pixbuf_cache, page,
537 new_job_list, new_prev_job, new_next_job,
538 new_preload_cache_size,
539 start_page, end_page, EV_JOB_PRIORITY_LOW);
540 }
541 page ++;
542 }
543
544 page = pixbuf_cache->start_page;
545 for (i = 0; i < PAGE_CACHE_LEN (pixbuf_cache) && page >= 0; i++) {
546 move_one_job (pixbuf_cache->job_list + i,
547 pixbuf_cache, page,
548 new_job_list, new_prev_job, new_next_job,
549 new_preload_cache_size,
550 start_page, end_page, EV_JOB_PRIORITY_URGENT);
551 page ++;
552 }
553
554 for (i = 0; i < pixbuf_cache->preload_cache_size; i++) {
555 if (page >= ev_document_get_n_pages (pixbuf_cache->document)) {
556 dispose_cache_job_info (pixbuf_cache->next_job + i, pixbuf_cache);
557 } else {
558 move_one_job (pixbuf_cache->next_job + i,
559 pixbuf_cache, page,
560 new_job_list, new_prev_job, new_next_job,
561 new_preload_cache_size,
562 start_page, end_page, EV_JOB_PRIORITY_LOW);
563 }
564 page ++;
565 }
566
567 if (pixbuf_cache->job_list) {
568 g_slice_free1 (sizeof (CacheJobInfo) * pixbuf_cache->job_list_len,
569 pixbuf_cache->job_list);
570 }
571 if (pixbuf_cache->prev_job) {
572 g_slice_free1 (sizeof (CacheJobInfo) * pixbuf_cache->preload_cache_size,
573 pixbuf_cache->prev_job);
574 }
575 if (pixbuf_cache->next_job) {
576 g_slice_free1 (sizeof (CacheJobInfo) * pixbuf_cache->preload_cache_size,
577 pixbuf_cache->next_job);
578 }
579
580 pixbuf_cache->preload_cache_size = new_preload_cache_size;
581 pixbuf_cache->job_list_len = new_job_list_len;
582
583 pixbuf_cache->job_list = new_job_list;
584 pixbuf_cache->prev_job = new_prev_job;
585 pixbuf_cache->next_job = new_next_job;
586
587 pixbuf_cache->start_page = start_page;
588 pixbuf_cache->end_page = end_page;
589}

591static CacheJobInfo *
592find_job_cache (EvPixbufCache *pixbuf_cache,
593 int page)
594{
595 int page_offset;
596
597 if (page < (pixbuf_cache->start_page - pixbuf_cache->preload_cache_size) ||
598 page > (pixbuf_cache->end_page + pixbuf_cache->preload_cache_size))
599 return NULL;
600
601 if (page < pixbuf_cache->start_page) {
602 page_offset = (page - (pixbuf_cache->start_page - pixbuf_cache->preload_cache_size));
603
604 g_assert (page_offset >= 0 &&
605 page_offset < pixbuf_cache->preload_cache_size);
606 return pixbuf_cache->prev_job + page_offset;
607 }
608
609 if (page > pixbuf_cache->end_page) {
610 page_offset = (page - (pixbuf_cache->end_page + 1));
611
612 g_assert (page_offset >= 0 &&
613 page_offset < pixbuf_cache->preload_cache_size);
614 return pixbuf_cache->next_job + page_offset;
615 }
616
617 page_offset = page - pixbuf_cache->start_page;
618 g_assert (page_offset >= 0 &&
619 page_offset <= PAGE_CACHE_LEN(pixbuf_cache));
620 return pixbuf_cache->job_list + page_offset;
621}

623static void
624ev_pixbuf_cache_clear_job_sizes (EvPixbufCache *pixbuf_cache,
625 gfloat scale)
626{
627 int i;
628
629 for (i = 0; i < PAGE_CACHE_LEN (pixbuf_cache); i++) {
630 check_job_size_and_unref (pixbuf_cache, pixbuf_cache->job_list + i, scale);

632
633 for (i = 0; i < pixbuf_cache->preload_cache_size; i++) {

635 check_job_size_and_unref (pixbuf_cache, pixbuf_cache->next_job + i, scale);
636 }
637}

657static void
658add_job (EvPixbufCache *pixbuf_cache,
659 CacheJobInfo *job_info,
660 cairo_region_t *region,
661 gint width,
662 gint height,
663 gint page,
664 gint rotation,
665 gfloat scale,
666 EvJobPriority priority)
667{
668 job_info->device_scale = get_device_scale (pixbuf_cache);
669 job_info->page_ready = FALSE;
670
671 if (job_info->region)
672 cairo_region_destroy (job_info->region);
673 job_info->region = region ? cairo_region_reference (region) : NULL;
674
675 if (job_info->job)
676 end_job (job_info, pixbuf_cache);
677
678 job_info->job = ev_job_render_new (pixbuf_cache->document,

685 GdkColor text, base;

688 ev_job_render_set_selection_info (EV_JOB_RENDER (job_info->job),
689 &(job_info->target_points),
690 job_info->selection_style,
691 &text, &base);

1028{

1031 /* the document does not implement the selection interface */

1033 return NULL;
