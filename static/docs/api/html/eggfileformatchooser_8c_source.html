135static GtkFileFilter*
136egg_file_format_filter_new (const gchar *name,
137 gboolean show_extensions)
138{
139 GtkFileFilter *filter;
140 EggFileFormatFilterInfo *info;
141
142 filter = gtk_file_filter_new ();
143 gtk_file_filter_set_name (filter, name);
144
145 info = egg_file_format_filter_info_new (name, show_extensions);
146
147 gtk_file_filter_add_custom (filter, GTK_FILE_FILTER_FILENAME,
148 egg_file_format_filter_filter,
149 info, NULL);
150 g_object_set_qdata_full (G_OBJECT (filter),
151 egg_file_format_filter_info_quark (),
152 info, egg_file_format_filter_info_free);
153
154 return filter;

157static void
158egg_file_format_filter_add_extensions (GtkFileFilter *filter,
159 const gchar *extensions)
160{
161 EggFileFormatFilterInfo *info;
162 GString *filter_name;
163 gchar **strings;
164 gchar **ptr;
165
166 g_assert (NULL != extensions);
167
168 info = g_object_get_qdata (G_OBJECT (filter),
169 egg_file_format_filter_info_quark ());
170
171 info->extension_list = g_slist_prepend (info->extension_list,
172 g_strdup (extensions));
173
174 if (info->show_extensions)
175 {
176 filter_name = g_string_new (info->name);
177 g_string_append (filter_name, " (");
178 }
179 else
180 filter_name = NULL;
181
182 strings = g_strsplit (extensions, ", ", -1);
183 for (ptr = strings; *ptr; ptr++)
184 {
185 gchar *pattern = g_strdup_printf ("*%s", *ptr);
186
187 if (filter_name)
188 {
189 if (ptr != strings)
190 g_string_append (filter_name, ", ");
191
192 g_string_append (filter_name, pattern);
193 }
194
195 g_hash_table_replace (info->extension_set, pattern, pattern);
196 }
197 g_strfreev (strings);
198
199 if (filter_name)
200 {
201 g_string_append (filter_name, ")");
202 gtk_file_filter_set_name (filter, filter_name->str);
203 g_string_free (filter_name, TRUE);
204 }
205}

207static void
208selection_changed_cb (GtkTreeSelection *selection,
209 EggFileFormatChooser *self)
210{
211 gchar *name;
212
213 GtkFileFilter *filter;
214 GtkTreeModel *model;
215 GtkTreeIter parent;
216 GtkTreeIter iter;
217
218 if (gtk_tree_selection_get_selected (selection, &model, &iter))
219 {
220 gchar *label;
221
222 gtk_tree_model_get (model, &iter, MODEL_COLUMN_NAME, &name, -1);
223
224 label = g_strdup_printf (_("File _Format: %s"), name);
225 gtk_expander_set_use_underline (GTK_EXPANDER (self), TRUE);
226 gtk_expander_set_label (GTK_EXPANDER (self), label);
227
228 g_free (name);
229 g_free (label);
230
231 if (self->priv->chooser)
232 {
233 while (gtk_tree_model_iter_parent (model, &parent, &iter))
234 iter = parent;
235
236 gtk_tree_model_get (model, &iter, MODEL_COLUMN_FILTER, &filter, -1);
237 gtk_file_chooser_set_filter (self->priv->chooser, filter);
238 g_object_unref (filter);
239 }
240
241 g_signal_emit (self, signals[SIGNAL_SELECTION_CHANGED], 0);
242 }

276static gboolean
277find_in_list (gchar *list,
278 const gchar *needle)
279{
280 gchar *saveptr;
281 gchar *token;
282
283 for (token = strtok_r (list, ",", &saveptr); NULL != token;
284 token = strtok_r (NULL, ",", &saveptr))
285 {
286 token = g_strstrip (token);
287
288 if (strcasecmp (needle, token) == 0)
289 return TRUE;
290 }
291
292 return FALSE;
293}

295static gboolean
296accept_filename (gchar *extensions,
297 const gchar *filename)
298{
299 gchar *saveptr;
300 gchar *token;
301 gsize length;
302
303 length = strlen (filename);
304
305 for (token = strtok_r (extensions, ",", &saveptr); NULL != token;
306 token = strtok_r (NULL, ",", &saveptr))
307 {
308 const gchar *extptr;
309
310 token = g_strstrip (token);
311 extptr = filename + length - strlen (token) - 1;
312
313 if (extptr > filename && '.' == *extptr &&
314 !strcmp (extptr + 1, token))
315 return TRUE;
316 }
317
318 return FALSE;
319}

321static gboolean
322find_by_extension (GtkTreeModel *model,
323 GtkTreePath *path G_GNUC_UNUSED,
324 GtkTreeIter *iter,
325 gpointer data)
326{
327 EggFileFormatSearch *search = data;
328
329 gchar *extensions = NULL;
330 guint format = 0;
331
332 gtk_tree_model_get (model, iter,
333 MODEL_COLUMN_EXTENSIONS, &extensions,
334 MODEL_COLUMN_ID, &format,
335 -1);
336
337 if (extensions && find_in_list (extensions, search->extension))
338 {
339 search->format = format;
340 search->success = TRUE;
341 search->iter = *iter;
342 }
343
344 g_free (extensions);
345 return search->success;

368static void
369egg_file_format_chooser_init (EggFileFormatChooser *self)
370{
371 GtkWidget *scroller;
372 GtkWidget *view;
373
374 GtkTreeViewColumn *column;
375 GtkCellRenderer *cell;
376 GtkTreeIter iter;
377
378 self->priv = egg_file_format_chooser_get_instance_private(self);
379
380 self->priv->size_changed_event = 0;
381
382/* file filters */
383
384 self->priv->all_files = g_object_ref_sink (gtk_file_filter_new ());
385 gtk_file_filter_set_name (self->priv->all_files, _("All Files"));
386 self->priv->supported_files = egg_file_format_filter_new (_("All Supported Files"), FALSE);
387
388/* tree model */
389
390 self->priv->model = gtk_tree_store_new (7, G_TYPE_UINT, G_TYPE_STRING, G_TYPE_STRING, G_TYPE_STRING,
391 GTK_TYPE_FILE_FILTER, G_TYPE_POINTER, G_TYPE_POINTER);
392
393 gtk_tree_store_append (self->priv->model, &iter, NULL);
394 gtk_tree_store_set (self->priv->model, &iter,
395 MODEL_COLUMN_NAME, _("By Extension"),
396 MODEL_COLUMN_FILTER, self->priv->supported_files,
397 MODEL_COLUMN_ID, 0,
398 -1);
399
400/* tree view */
401
402 view = gtk_tree_view_new_with_model (GTK_TREE_MODEL (self->priv->model));
403 self->priv->selection = gtk_tree_view_get_selection (GTK_TREE_VIEW (view));
404
405/* file format column */
406
407 column = gtk_tree_view_column_new ();
408 gtk_tree_view_column_set_expand (column, TRUE);
409 gtk_tree_view_column_set_title (column, _("File Format"));
410 gtk_tree_view_append_column (GTK_TREE_VIEW (view), column);
411
412 cell = gtk_cell_renderer_pixbuf_new ();
413 gtk_tree_view_column_pack_start (column, cell, FALSE);
414 gtk_tree_view_column_set_attributes (column, cell,
415 "icon-name", MODEL_COLUMN_ICON,
416 NULL);
417
418 cell = gtk_cell_renderer_text_new ();
419 gtk_tree_view_column_pack_start (column, cell, TRUE);
420 gtk_tree_view_column_set_attributes (column, cell,
421 "text", MODEL_COLUMN_NAME,
422 NULL);
423
424/* extensions column */
425
426 column = gtk_tree_view_column_new_with_attributes (
427 _("Extension(s)"), gtk_cell_renderer_text_new (),
428 "text", MODEL_COLUMN_EXTENSIONS, NULL);
429 gtk_tree_view_column_set_expand (column, FALSE);
430 gtk_tree_view_append_column (GTK_TREE_VIEW (view), column);
431
432/* selection */
433
434 gtk_tree_selection_set_mode (self->priv->selection, GTK_SELECTION_BROWSE);
435 g_signal_connect (self->priv->selection, "changed",
436 G_CALLBACK (selection_changed_cb), self);
437 self->priv->idle_hack = g_idle_add (select_default_file_format, self);
438
439/* scroller */
440
441 scroller = gtk_scrolled_window_new (NULL, NULL);
442 gtk_scrolled_window_set_policy (GTK_SCROLLED_WINDOW (scroller),
443 GTK_POLICY_NEVER, GTK_POLICY_AUTOMATIC);
444 gtk_scrolled_window_set_shadow_type (GTK_SCROLLED_WINDOW (scroller),
445 GTK_SHADOW_IN);
446 gtk_widget_set_size_request (scroller, -1, 150);
447 gtk_container_add (GTK_CONTAINER (scroller), view);
448 gtk_widget_show_all (scroller);
449
450 gtk_container_add (GTK_CONTAINER (self), scroller);
451
452 g_signal_connect_after (scroller, "unmap", G_CALLBACK (expander_unmap_cb), self);
453}

526static void
527filter_changed_cb (GObject *object,
528 GParamSpec *spec,
529 gpointer data)
530{
531 EggFileFormatChooser *self;
532
533 GtkFileFilter *current_filter;
534 GtkFileFilter *format_filter;
535
536 GtkTreeModel *model;
537 GtkTreeIter iter;
538 GtkTreeIter parent;
539
540 self = EGG_FILE_FORMAT_CHOOSER (data);
541
542 format_filter = NULL;
543 current_filter = gtk_file_chooser_get_filter (GTK_FILE_CHOOSER (object));
544 model = GTK_TREE_MODEL (self->priv->model);
545
546 if (gtk_tree_selection_get_selected (self->priv->selection, &model, &iter))
547 {
548 while (gtk_tree_model_iter_parent (model, &parent, &iter))
549 iter = parent;
550
551 gtk_tree_model_get (model, &iter,
552 MODEL_COLUMN_FILTER,
553 &format_filter, -1);
554 g_object_unref (format_filter);
555 }
556
557 if (current_filter && current_filter != format_filter &&
558 gtk_tree_model_get_iter_first (model, &iter))
559 {
560 if (current_filter == self->priv->all_files)
561 format_filter = current_filter;
562 else
563 {
564 format_filter = NULL;
565
566 do
567 {
568 gtk_tree_model_get (model, &iter,
569 MODEL_COLUMN_FILTER,
570 &format_filter, -1);
571 g_object_unref (format_filter);
572
573 if (format_filter == current_filter)
574 break;
575 }
576 while (gtk_tree_model_iter_next (model, &iter));
577 }
578
579 if (format_filter)
580 gtk_tree_selection_select_iter (self->priv->selection, &iter);
581 }
582}

632static void
633chooser_response_cb (GtkDialog *dialog,
634 gint response_id,
635 gpointer data)
636{
637 EggFileFormatChooser *self;
638
639 self = EGG_FILE_FORMAT_CHOOSER (data);
640
641 if (EGG_IS_POSITIVE_RESPONSE (response_id))
642 {
643 gchar *filename, *basename;
644 guint format;
645
646 filename = gtk_file_chooser_get_filename (self->priv->chooser);
647 basename = g_filename_display_basename (filename);
648 g_free (filename);
649
650 format = egg_file_format_chooser_get_format (self, basename);
651 g_print ("%s: %s - %d\n", G_STRFUNC, basename, format);
652
653 if (0 == format)
654 {
655 gchar *message;
656
657 message = g_strdup_printf (
658 _("The program was not able to find out the file format "
659 "you want to use for `%s'. Please make sure to use a "
660 "known extension for that file or manually choose a "
661 "file format from the list below."),
662 basename);
663
664 error_message (self,
665 _("File format not recognized"),
666 message);
667
668 g_free (message);
669
670 g_signal_stop_emission_by_name (dialog, "response");
671 }
672 else
673 {
674 filename = egg_file_format_chooser_append_extension (self, basename, format);
675
676 if (strcmp (filename, basename))
677 {
678 gtk_file_chooser_set_current_name (self->priv->chooser, filename);
679 g_signal_stop_emission_by_name (dialog, "response");
680 }
681
682 g_free (filename);
683 }
684
685 g_free (basename);
686 }

690static void
691egg_file_format_chooser_realize (GtkWidget *widget)
692{
693 EggFileFormatChooser *self;
694 GtkWidget *parent;
695
696 GtkFileFilter *filter;
697 GtkTreeModel *model;
698 GtkTreeIter iter;
699
700 GTK_WIDGET_CLASS (egg_file_format_chooser_parent_class)->realize (widget);
701
702 self = EGG_FILE_FORMAT_CHOOSER (widget);
703
704 g_return_if_fail (NULL == self->priv->chooser);
705
706 parent = gtk_widget_get_parent (widget);
707 while ((parent != NULL) && !GTK_IS_FILE_CHOOSER (parent))
708 parent = gtk_widget_get_parent (parent);
709
710 self->priv->chooser = GTK_FILE_CHOOSER (parent);
711
712 g_return_if_fail (GTK_IS_FILE_CHOOSER (self->priv->chooser));
713 g_return_if_fail (gtk_file_chooser_get_action (self->priv->chooser) ==
714 GTK_FILE_CHOOSER_ACTION_SAVE);
715
716 g_object_ref (self->priv->chooser);
717
718 g_signal_connect (self->priv->chooser, "notify::filter",
719 G_CALLBACK (filter_changed_cb), self);
720 gtk_file_chooser_add_filter (self->priv->chooser, self->priv->all_files);
721
722 model = GTK_TREE_MODEL (self->priv->model);
723
724 if (gtk_tree_model_get_iter_first (model, &iter))
725 {
726 do
727 {
728 gtk_tree_model_get (model, &iter, MODEL_COLUMN_FILTER, &filter, -1);
729 gtk_file_chooser_add_filter (self->priv->chooser, filter);
730 g_object_unref (filter);
731 }
732 while (gtk_tree_model_iter_next (model, &iter));
733 }
734
735 gtk_file_chooser_set_filter (self->priv->chooser,
736 self->priv->supported_files);
737
738 if (GTK_IS_DIALOG (self->priv->chooser))
739 g_signal_connect (self->priv->chooser, "response",
740 G_CALLBACK (chooser_response_cb), self);
741}

743static void
744egg_file_format_chooser_unrealize (GtkWidget *widget)
745{
746 EggFileFormatChooser *self;
747
748 GtkFileFilter *filter;
749 GtkTreeModel *model;
750 GtkTreeIter iter;
751
752 GTK_WIDGET_CLASS (egg_file_format_chooser_parent_class)->unrealize (widget);
753
754 self = EGG_FILE_FORMAT_CHOOSER (widget);
755 model = GTK_TREE_MODEL (self->priv->model);
756
757 g_signal_handlers_disconnect_by_func (self->priv->chooser,
758 filter_changed_cb, self);
759 g_signal_handlers_disconnect_by_func (self->priv->chooser,
760 chooser_response_cb, self);
761
762 if (gtk_tree_model_get_iter_first (model, &iter))
763 {
764 do

959 extensions = g_strjoinv (", ", extension_list);
960 g_strfreev (extension_list);

962 description = gdk_pixbuf_format_get_description (format);
963 name = gdk_pixbuf_format_get_name (format);

965 id = egg_file_format_chooser_add_format_impl (self, parent, description,
966 icon, extensions);

970 g_free (icon);
971
972 egg_file_format_chooser_set_format_data (self, id, name, g_free);
973
974 if (formats)

976 }

1139 const gchar *filename,

1145

1148
1149 g_return_val_if_fail (EGG_IS_FILE_FORMAT_CHOOSER (self), NULL);
1150 g_return_val_if_fail (NULL != filename, NULL);
1151

1162 search.format = format;
1163

1167 g_return_val_if_fail (search.success, NULL);
1168
1169 gtk_tree_model_get (model, &search.iter,
1170 MODEL_COLUMN_EXTENSIONS, &extensions,
1171 -1);
1172
1173 if (NULL == extensions &&
1174 gtk_tree_model_iter_nth_child (model, &child, &search.iter, 0))
1175 {
1176 gtk_tree_model_get (model, &child,
1177 MODEL_COLUMN_EXTENSIONS, &extensions,
1178 -1);
1179 }
1180

1186 }
1187
1188 if (accept_filename (extensions, filename))
1189 result = g_strdup (filename);
1190 else
1191 result = g_strconcat (filename, ".", extensions, NULL);
1192
