31static MetaGroup*
32meta_group_new (MetaDisplay *display,
33 Window group_leader)
34{
35 MetaGroup *group;
36#define N_INITIAL_PROPS 3
37 Atom initial_props[N_INITIAL_PROPS];
38 int i;
39
40 g_assert (N_INITIAL_PROPS == (int) G_N_ELEMENTS (initial_props));
41
42 group = g_new0 (MetaGroup, 1);
43
44 group->display = display;
45 group->windows = NULL;
46 group->group_leader = group_leader;
47 group->refcount = 1; /* owned by caller, hash table has only weak ref */
48
49 if (display->groups_by_leader == NULL)
50 display->groups_by_leader = g_hash_table_new (meta_unsigned_long_hash,
51 meta_unsigned_long_equal);
52
53 g_assert (g_hash_table_lookup (display->groups_by_leader, &group_leader) == NULL);
54
55 g_hash_table_insert (display->groups_by_leader,
56 &group->group_leader,
57 group);
58
59 /* Fill these in the order we want them to be gotten */
60 i = 0;
61 initial_props[i++] = display->atom_WM_CLIENT_MACHINE;
62 initial_props[i++] = display->atom__NET_WM_PID;
63 initial_props[i++] = display->atom__NET_STARTUP_ID;
64 g_assert (N_INITIAL_PROPS == i);
65
66 meta_group_reload_properties (group, initial_props, N_INITIAL_PROPS);
67
68 meta_topic (META_DEBUG_GROUPS,
69 "Created new group with leader 0x%lx\n",
70 group->group_leader);
71
72 return group;
73}

75static void
76meta_group_unref (MetaGroup *group)
77{
78 g_return_if_fail (group->refcount > 0);
79
80 group->refcount -= 1;
81 if (group->refcount == 0)
82 {
83 meta_topic (META_DEBUG_GROUPS,
84 "Destroying group with leader 0x%lx\n",
85 group->group_leader);
86
87 g_assert (group->display->groups_by_leader != NULL);
88
89 g_hash_table_remove (group->display->groups_by_leader,
90 &group->group_leader);
91
92 /* mop up hash table, this is how it gets freed on display close */
93 if (g_hash_table_size (group->display->groups_by_leader) == 0)
94 {
95 g_hash_table_destroy (group->display->groups_by_leader);
96 group->display->groups_by_leader = NULL;
97 }
98
99 g_free (group->wm_client_machine);
100 g_free (group->startup_id);
101
102 g_free (group);
103 }
104}

115void
116meta_window_compute_group (MetaWindow* window)
117{
118 MetaGroup *group;
119 MetaWindow *ancestor;
120
121 /* use window->xwindow if no window->xgroup_leader */
122
123 group = NULL;
124
125 /* Determine the ancestor of the window; its group setting will override the
126 * normal grouping rules; see bug 328211.
127 */
128 ancestor = meta_window_find_root_ancestor (window);
129
130 if (window->display->groups_by_leader)
131 {
132 if (ancestor != window)
133 group = ancestor->group;
134 else if (window->xgroup_leader != None)
135 group = g_hash_table_lookup (window->display->groups_by_leader,
136 &window->xgroup_leader);
137 else
138 group = g_hash_table_lookup (window->display->groups_by_leader,
139 &window->xwindow);
140 }
141
142 if (group != NULL)
143 {
144 window->group = group;
145 group->refcount += 1;
146 }
147 else
148 {
149 if (ancestor != window && ancestor->xgroup_leader != None)
150 group = meta_group_new (window->display,
151 ancestor->xgroup_leader);
152 else if (window->xgroup_leader != None)
153 group = meta_group_new (window->display,
154 window->xgroup_leader);
155 else
156 group = meta_group_new (window->display,
157 window->xwindow);
158
159 window->group = group;
160 }
161
162 window->group->windows = g_slist_prepend (window->group->windows, window);
163
164 meta_topic (META_DEBUG_GROUPS,
165 "Adding %s to group with leader 0x%lx\n",
166 window->desc, group->group_leader);
167
168}

170static void
171remove_window_from_group (MetaWindow *window)
172{
173 if (window->group != NULL)
174 {
175 meta_topic (META_DEBUG_GROUPS,
176 "Removing %s from group with leader 0x%lx\n",
177 window->desc, window->group->group_leader);
178
179 window->group->windows =
180 g_slist_remove (window->group->windows,
181 window);
182 meta_group_unref (window->group);
183 window->group = NULL;
184 }

221void
222meta_group_update_layers (MetaGroup *group)
223{
224 GSList *tmp;
225 GSList *frozen_stacks;
226
227 if (group->windows == NULL)
228 return;
229
230 frozen_stacks = NULL;
231 tmp = group->windows;
232 while (tmp != NULL)
233 {
234 MetaWindow *window = tmp->data;
235
236 /* we end up freezing the same stack a lot of times,
237 * but doesn't hurt anything. have to handle
238 * groups that span 2 screens.
239 */
240 meta_stack_freeze (window->screen->stack);
241 frozen_stacks = g_slist_prepend (frozen_stacks, window->screen->stack);
242
243 meta_stack_update_layer (window->screen->stack,
244 window);
245
246 tmp = tmp->next;
247 }
248
249 tmp = frozen_stacks;
250 while (tmp != NULL)
251 {
252 meta_stack_thaw (tmp->data);
253 tmp = tmp->next;
254 }
255
256 g_slist_free (frozen_stacks);
257}
