238static cairo_surface_t *
239tile_surface (cairo_surface_t *surface,
240 int width,
241 int height)
242{
243 cairo_surface_t *copy;
244 cairo_t *cr;
245
246 if (surface == NULL)
247 {
248 copy = gdk_window_create_similar_surface (gdk_get_default_root_window (),
249 CAIRO_CONTENT_COLOR,
250 width, height);
251 }
252 else
253 {
254 copy = cairo_surface_create_similar (surface,
255 cairo_surface_get_content (surface),
256 width, height);
257 }
258
259 cr = cairo_create (copy);
260
261 if (surface != NULL)
262 {
263 cairo_pattern_t *pattern;
264 cairo_set_source_surface (cr, surface, 0.0, 0.0);
265 pattern = cairo_get_source (cr);
266 cairo_pattern_set_extend (pattern, CAIRO_EXTEND_REPEAT);
267 }
268 else
269 {
270 static GtkCssProvider *provider = NULL;
271 GtkStyleContext *context;
272 GdkRGBA bg;
273
274 if (provider == NULL)
275 provider = gtk_css_provider_new ();
276
277 context = gtk_style_context_new ();
278 gtk_style_context_add_provider (context,
279 GTK_STYLE_PROVIDER (provider),
280 GTK_STYLE_PROVIDER_PRIORITY_THEME);
281 gtk_style_context_get (context, GTK_STATE_FLAG_NORMAL, "background-color", &bg, NULL);
282 gdk_cairo_set_source_rgba(cr, &bg);
283 g_object_unref (G_OBJECT (context));
284 }
285
286 cairo_paint (cr);
287
288 if (cairo_status (cr) != CAIRO_STATUS_SUCCESS)
289 {
290 cairo_surface_destroy (copy);
291 copy = NULL;
292 }
293
294 cairo_destroy(cr);
295
296 return copy;
297}

421static void
422draw_background (MateBGCrossfade *fade)
423{
424 if (fade->priv->widget != NULL) {
425 gtk_widget_queue_draw (fade->priv->widget);
426 } else if (gdk_window_get_window_type (fade->priv->window) != GDK_WINDOW_ROOT) {
427 cairo_t *cr;
428 cairo_region_t *region;
429 GdkDrawingContext *draw_context;
430
431 region = gdk_window_get_visible_region (fade->priv->window);
432 draw_context = gdk_window_begin_draw_frame (fade->priv->window,
433 region);
434 cr = gdk_drawing_context_get_cairo_context (draw_context);
435 cairo_set_operator (cr, CAIRO_OPERATOR_OVER);
436 cairo_set_source_surface (cr, fade->priv->fading_surface, 0, 0);
437 cairo_paint (cr);
438 gdk_window_end_draw_frame (fade->priv->window, draw_context);
439 cairo_region_destroy (region);
440 } else {
441 Display *xdisplay = GDK_WINDOW_XDISPLAY (fade->priv->window);
442 GdkDisplay *display;
443 display = gdk_display_get_default ();
444 gdk_x11_display_error_trap_push (display);
445 XGrabServer (xdisplay);
446 XClearWindow (xdisplay, GDK_WINDOW_XID (fade->priv->window));
447 send_root_property_change_notification (fade);
448 XFlush (xdisplay);
449 XUngrabServer (xdisplay);
450 gdk_x11_display_error_trap_pop_ignored (display);
451 }
452}

468static gboolean
469on_tick (MateBGCrossfade *fade)
470{
471 gdouble now, percent_done;
472 cairo_t *cr;
473 cairo_status_t status;
474
475 g_return_val_if_fail (MATE_IS_BG_CROSSFADE (fade), FALSE);
476
477 now = get_current_time ();
478
479 percent_done = (now - fade->priv->start_time) / fade->priv->total_duration;
480 percent_done = CLAMP (percent_done, 0.0, 1.0);
481
482 /* If it's taking a long time to get to the first frame,
483 * then lengthen the duration, so the user will get to see
484 * the effect.
485 */
486 if (fade->priv->is_first_frame && percent_done > .33) {
487 fade->priv->is_first_frame = FALSE;
488 fade->priv->total_duration *= 1.5;
489 return on_tick (fade);
490 }
491
492 if (fade->priv->fading_surface == NULL ||
493 fade->priv->end_surface == NULL) {
494 return FALSE;
495 }
496
497 if (animations_are_disabled (fade)) {
498 return FALSE;
499 }
500
501 /* We accumulate the results in place for performance reasons.
502 *
503 * This means 1) The fade is exponential, not linear (looks good!)
504 * 2) The rate of fade is not independent of frame rate. Slower machines
505 * will get a slower fade (but never longer than .75 seconds), and
506 * even the fastest machines will get *some* fade because the framerate
507 * is capped.
508 */
509 cr = cairo_create (fade->priv->fading_surface);
510
511 cairo_set_source_surface (cr, fade->priv->end_surface,
512 0.0, 0.0);
513 cairo_paint_with_alpha (cr, percent_done);
514
515 status = cairo_status (cr);
516 cairo_destroy (cr);
517
518 if (status == CAIRO_STATUS_SUCCESS) {
519 draw_background (fade);
520 }
521 return percent_done <= .99;
522}

524static void
525on_finished (MateBGCrossfade *fade)
526{
527 cairo_t *cr;
528
529 if (fade->priv->timeout_id == 0)
530 return;
531
532 g_assert (fade->priv->fading_surface != NULL);
533 g_assert (fade->priv->end_surface != NULL);
534
535 cr = cairo_create (fade->priv->fading_surface);
536 cairo_set_source_surface (cr, fade->priv->end_surface, 0, 0);
537 cairo_paint (cr);
538 cairo_destroy (cr);
539 draw_background (fade);
540
541 cairo_surface_destroy (fade->priv->fading_surface);
542 fade->priv->fading_surface = NULL;
543
544 cairo_surface_destroy (fade->priv->end_surface);
545 fade->priv->end_surface = NULL;
546
547 g_assert (fade->priv->start_surface != NULL);
548
549 cairo_surface_destroy (fade->priv->start_surface);
550 fade->priv->start_surface = NULL;
551
552 if (fade->priv->widget != NULL) {
553 g_signal_handlers_disconnect_by_func (fade->priv->widget,
554 (GCallback) on_widget_draw,
555 fade);
556 }
557 fade->priv->widget = NULL;
558
559 fade->priv->timeout_id = 0;
560 g_signal_emit (fade, signals[FINISHED], 0, fade->priv->window);
561}

563/* This function queries the _XROOTPMAP_ID property from the root window
564 * to determine the current root window background pixmap and returns a
565 * surface to draw directly to it.
566 * If _XROOTPMAP_ID is not set, then NULL returned.
567 */
568static cairo_surface_t *
569get_root_pixmap_id_surface (GdkDisplay *display)
570{
571 GdkScreen *screen;
572 Display *xdisplay;
573 Visual *xvisual;
574 Window xroot;
575 Atom type;
576 int format, result;
577 unsigned long nitems, bytes_after;
578 unsigned char *data;
579 cairo_surface_t *surface = NULL;
580
581 g_return_val_if_fail (display != NULL, NULL);
582
583 screen = gdk_display_get_default_screen (display);
584 xdisplay = GDK_DISPLAY_XDISPLAY (display);
585 xvisual = GDK_VISUAL_XVISUAL (gdk_screen_get_system_visual (screen));
586 xroot = RootWindow (xdisplay, GDK_SCREEN_XNUMBER (screen));
587
588 result = XGetWindowProperty (xdisplay, xroot,
589 gdk_x11_get_xatom_by_name ("_XROOTPMAP_ID"),
590 0L, 1L, False, XA_PIXMAP,
591 &type, &format, &nitems, &bytes_after,
592 &data);
593
594 if (result != Success || type != XA_PIXMAP ||
595 format != 32 || nitems != 1) {
596 XFree (data);
597 data = NULL;
598 }
599
600 if (data != NULL) {
601 Pixmap pixmap = *(Pixmap *) data;
602 Window root_ret;
603 int x_ret, y_ret;
604 unsigned int w_ret, h_ret, bw_ret, depth_ret;
605
606 gdk_x11_display_error_trap_push (display);
607 if (XGetGeometry (xdisplay, pixmap, &root_ret,
608 &x_ret, &y_ret, &w_ret, &h_ret,
609 &bw_ret, &depth_ret))
610 {
611 surface = cairo_xlib_surface_create (xdisplay,
612 pixmap, xvisual,
613 w_ret, h_ret);
614 }
615
616 gdk_x11_display_error_trap_pop_ignored (display);
617 XFree (data);
618 }
619
620 gdk_display_flush (display);
621 return surface;
622}

636void
637mate_bg_crossfade_start (MateBGCrossfade *fade,
638 GdkWindow *window)
639{
640 GSource *source;
641 GMainContext *context;
642
643 g_return_if_fail (MATE_IS_BG_CROSSFADE (fade));
644 g_return_if_fail (window != NULL);
645 g_return_if_fail (fade->priv->start_surface != NULL);
646 g_return_if_fail (fade->priv->end_surface != NULL);
647 g_return_if_fail (!mate_bg_crossfade_is_started (fade));
648 g_return_if_fail (gdk_window_get_window_type (window) != GDK_WINDOW_FOREIGN);
649
650 /* If drawing is done on the root window,
651 * it is essential to have the root pixmap.
652 */
653 if (gdk_window_get_window_type (window) == GDK_WINDOW_ROOT) {
654 GdkDisplay *display = gdk_window_get_display (window);
655 cairo_surface_t *surface = get_root_pixmap_id_surface (display);
656
657 g_return_if_fail (surface != NULL);
658 cairo_surface_destroy (surface);
659 }
660
661 if (fade->priv->fading_surface != NULL) {
662 cairo_surface_destroy (fade->priv->fading_surface);
663 fade->priv->fading_surface = NULL;
664 }
665
666 fade->priv->window = window;
667 if (gdk_window_get_window_type (fade->priv->window) != GDK_WINDOW_ROOT) {
668 fade->priv->fading_surface = tile_surface (fade->priv->start_surface,
669 fade->priv->width,
670 fade->priv->height);
671 if (fade->priv->widget != NULL) {
672 g_signal_connect (fade->priv->widget, "draw",
673 (GCallback) on_widget_draw, fade);
674 }
675 } else {
676 cairo_t *cr;
677 GdkDisplay *display = gdk_window_get_display (fade->priv->window);
678
679 fade->priv->fading_surface = get_root_pixmap_id_surface (display);
680 cr = cairo_create (fade->priv->fading_surface);
681 cairo_set_source_surface (cr, fade->priv->start_surface, 0, 0);
682 cairo_paint (cr);
683 cairo_destroy (cr);
684 }
685 draw_background (fade);
686
687 source = g_timeout_source_new (1000 / 60.0);
688 g_source_set_callback (source,
689 (GSourceFunc) on_tick,
690 fade,
691 (GDestroyNotify) on_finished);
692 context = g_main_context_default ();
693 fade->priv->timeout_id = g_source_attach (source, context);
694 g_source_unref (source);
695
696 fade->priv->is_first_frame = TRUE;
697 fade->priv->total_duration = .75;
698 fade->priv->start_time = get_current_time ();
699}
