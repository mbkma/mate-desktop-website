117static gboolean
118panel_lockdown_load_bool (PanelLockdown *lockdown,
119 GSettings *settings,
120 const char *key,
121 GCallback notify_func)
122{
123 gboolean retval;
124 gchar *signal_name;
125
126 retval = g_settings_get_boolean (settings, key);
127
128 signal_name = g_strdup_printf ("changed::%s", key);
129
130 g_signal_connect (settings,
131 signal_name,
132 G_CALLBACK (notify_func),
133 lockdown);
134
135 g_free (signal_name);
136
137 return retval;
138}

157void
158panel_lockdown_init (void)
159{
160 panel_lockdown.panel_settings = g_settings_new (PANEL_SCHEMA);
161 panel_lockdown.lockdown_settings = g_settings_new (LOCKDOWN_SCHEMA);
162
163 panel_lockdown.locked_down =
164 (panel_lockdown_load_bool (&panel_lockdown,
165 panel_lockdown.panel_settings,
166 PANEL_LOCKED_DOWN_KEY,
167 G_CALLBACK (locked_down_notify)) != FALSE);
168
169 panel_lockdown.disable_command_line =
170 (panel_lockdown_load_bool (&panel_lockdown,
171 panel_lockdown.lockdown_settings,
172 LOCKDOWN_DISABLE_COMMAND_LINE_KEY,
173 G_CALLBACK (disable_command_line_notify)) != FALSE);
174
175 panel_lockdown.disable_lock_screen =
176 (panel_lockdown_load_bool (&panel_lockdown,
177 panel_lockdown.lockdown_settings,
178 LOCKDOWN_DISABLE_LOCK_SCREEN_KEY,
179 G_CALLBACK (disable_lock_screen_notify)) != FALSE);
180
181 panel_lockdown.disable_log_out =
182 (panel_lockdown_load_bool (&panel_lockdown,
183 panel_lockdown.lockdown_settings,
184 LOCKDOWN_DISABLE_LOG_OUT_KEY,
185 G_CALLBACK (disable_log_out_notify)) != FALSE);
186
187 panel_lockdown.disable_force_quit =
188 (panel_lockdown_load_bool (&panel_lockdown,
189 panel_lockdown.panel_settings,
190 PANEL_DISABLE_FORCE_QUIT_KEY,
191 G_CALLBACK (disable_force_quit_notify)) != FALSE);
192
193 panel_lockdown.disabled_applets =
194 panel_lockdown_load_disabled_applets (&panel_lockdown,
195 panel_lockdown.panel_settings);
196
197 panel_lockdown.initialized = TRUE;
198}

200void
201panel_lockdown_finalize (void)
202{
203 GSList *l;
204
205 g_assert (panel_lockdown.initialized != FALSE);
206
207 if (panel_lockdown.disabled_applets) {
208 g_strfreev (panel_lockdown.disabled_applets);
209 panel_lockdown.disabled_applets = NULL;
210 }
211
212 if (panel_lockdown.panel_settings) {
213 g_object_unref (panel_lockdown.panel_settings);
214 panel_lockdown.panel_settings = NULL;
215 }
216 if (panel_lockdown.lockdown_settings) {
217 g_object_unref (panel_lockdown.lockdown_settings);
218 panel_lockdown.lockdown_settings = NULL;
219 }
220
221 for (l = panel_lockdown.closures; l; l = l->next)
222 g_closure_unref (l->data);
223 g_slist_free (panel_lockdown.closures);
224 panel_lockdown.closures = NULL;
225
226 panel_lockdown.initialized = FALSE;
227}

269gboolean
270panel_lockdown_is_applet_disabled (const char *iid)
271{
272 g_assert (panel_lockdown.initialized != FALSE);
273
274 if (panel_lockdown.disabled_applets)
275 for (gint i = 0; panel_lockdown.disabled_applets[i]; i++)
276 if (!strcmp (panel_lockdown.disabled_applets[i], iid))
277 return TRUE;
278
279 return FALSE;
280}

282static GClosure *
283panel_lockdown_notify_find (GSList *closures,
284 GCallback callback_func,
285 gpointer user_data)
286{
287 for (GSList *l = closures; l; l = l->next) {
288 GCClosure *cclosure = l->data;
289 GClosure *closure = l->data;
290
291 if (closure->data == user_data &&
292 cclosure->callback == callback_func)
293 return closure;
294 }
295
296 return NULL;
297}

299static void
300marshal_user_data (GClosure *closure,
301 GValue *return_value,
302 guint n_param_values,
303 const GValue *param_values,
304 gpointer invocation_hint,
305 gpointer marshal_data)
306{
307 GCClosure *cclosure = (GCClosure*) closure;
308
309 g_return_if_fail (cclosure->callback != NULL);
310 g_return_if_fail (n_param_values == 0);
311
312 ((void (*) (gpointer *))cclosure->callback) (closure->data);
313}

315void
316panel_lockdown_notify_add (GCallback callback_func,
317 gpointer user_data)
318{
319 GClosure *closure;
320
321 g_assert (panel_lockdown_notify_find (panel_lockdown.closures,
322 callback_func,
323 user_data) == NULL);
324
325 closure = g_cclosure_new (callback_func, user_data, NULL);
326 g_closure_set_marshal (closure, marshal_user_data);
327
328 panel_lockdown.closures = g_slist_append (panel_lockdown.closures,
329 closure);
330}

332void
333panel_lockdown_notify_remove (GCallback callback_func,
334 gpointer user_data)
335{

341
342 g_assert (closure != NULL);
343
344 panel_lockdown.closures = g_slist_remove (panel_lockdown.closures,
345 closure);
346
347 g_closure_unref (closure);
