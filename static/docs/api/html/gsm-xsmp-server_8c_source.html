136static gboolean
137auth_iochannel_watch (GIOChannel *source,
138 GIOCondition condition,
139 IceConn ice_conn)
140{
141
142 GsmIceConnectionWatch *data;
143 gboolean keep_going;
144
145 data = ice_conn->context;
146
147 switch (IceProcessMessages (ice_conn, NULL, NULL)) {
148 case IceProcessMessagesSuccess:
149 keep_going = TRUE;
150 break;
151 case IceProcessMessagesIOError:
152 g_debug ("GsmXsmpServer: IceProcessMessages returned IceProcessMessagesIOError");
153 free_ice_connection_watch (data);
154 disconnect_ice_connection (ice_conn);
155 keep_going = FALSE;
156 break;
157 case IceProcessMessagesConnectionClosed:
158 g_debug ("GsmXsmpServer: IceProcessMessages returned IceProcessMessagesConnectionClosed");
159 free_ice_connection_watch (data);
160 keep_going = FALSE;
161 break;
162 default:
163 g_assert_not_reached ();
164 }
165
166 return keep_going;
167}

169/* IceAcceptConnection returns a new ICE connection that is in a "pending" state,
170 * this is because authentification may be necessary.
171 * So we've to authenticate it, before accept_xsmp_connection() is called.
172 * Then each GsmXSMPClient will have its own IceConn watcher
173 */
174static void
175auth_ice_connection (IceConn ice_conn)
176{
177 GIOChannel *channel;
178 GsmIceConnectionWatch *data;
179 int fd;
180
181 g_debug ("GsmXsmpServer: auth_ice_connection()");
182
183 fd = IceConnectionNumber (ice_conn);
184 fcntl (fd, F_SETFD, fcntl (fd, F_GETFD, 0) | FD_CLOEXEC);
185 channel = g_io_channel_unix_new (fd);
186
187 data = g_new0 (GsmIceConnectionWatch, 1);
188 ice_conn->context = data;
189
190 data->protocol_timeout = g_timeout_add_seconds (5,
191 (GSourceFunc)ice_protocol_timeout,
192 ice_conn);
193 data->watch_id = g_io_add_watch (channel,
194 G_IO_IN | G_IO_ERR,
195 (GIOFunc)auth_iochannel_watch,
196 ice_conn);
197 g_io_channel_unref (channel);
198}

200/* This is called (by glib via xsmp->ice_connection_watch) when a
201 * connection is first received on the ICE listening socket.
202 */
203static gboolean
204accept_ice_connection (GIOChannel *source,
205 GIOCondition condition,
206 GsmIceConnectionData *data)
207{
208 IceConn ice_conn;
209 IceAcceptStatus status;
210
211 g_debug ("GsmXsmpServer: accept_ice_connection()");
212
213 ice_conn = IceAcceptConnection (data->listener, &status);
214 if (status != IceAcceptSuccess) {
215 g_debug ("GsmXsmpServer: IceAcceptConnection returned %d", status);
216 return TRUE;
217 }
218
219 auth_ice_connection (ice_conn);
220
221 return TRUE;
222}

224void
225gsm_xsmp_server_start (GsmXsmpServer *server)
226{
227 GIOChannel *channel;
228 int i;
229
230 for (i = 0; i < server->num_local_xsmp_sockets; i++) {
231 GsmIceConnectionData *data;
232
233 data = g_new0 (GsmIceConnectionData, 1);
234 data->server = server;
235 data->listener = server->xsmp_sockets[i];
236
237 channel = g_io_channel_unix_new (IceGetListenConnectionNumber (server->xsmp_sockets[i]));
238 g_io_add_watch_full (channel,
239 G_PRIORITY_DEFAULT,
240 G_IO_IN | G_IO_HUP | G_IO_ERR,
241 (GIOFunc)accept_ice_connection,
242 data,
243 (GDestroyNotify)g_free);
244 g_io_channel_unref (channel);
245 }
246}

305/* This is called (by libSM) when XSMP is initiated on an ICE
306 * connection that was already accepted by accept_ice_connection.
307 */
308static Status
309accept_xsmp_connection (SmsConn sms_conn,
310 GsmXsmpServer *server,
311 unsigned long *mask_ret,
312 SmsCallbacks *callbacks_ret,
313 char **failure_reason_ret)
314{
315 IceConn ice_conn;
316 GsmClient *client;
317 GsmIceConnectionWatch *data;
318
319 /* FIXME: what about during shutdown but before gsm_xsmp_shutdown? */
320 if (server->xsmp_sockets == NULL) {
321 g_debug ("GsmXsmpServer: In shutdown, rejecting new client");
322
323 *failure_reason_ret = strdup (_("Refusing new client connection because the session is currently being shut down\n"));
324 return FALSE;
325 }
326
327 ice_conn = SmsGetIceConnection (sms_conn);
328 data = ice_conn->context;
329
330 /* Each GsmXSMPClient has its own IceConn watcher */
331 free_ice_connection_watch (data);
332
333 client = gsm_xsmp_client_new (ice_conn);
334
335 gsm_store_add (server->client_store, gsm_client_peek_id (client), G_OBJECT (client));
336 /* the store will own the ref */
337 g_object_unref (client);
338
339 gsm_xsmp_client_connect (GSM_XSMP_CLIENT (client), sms_conn, mask_ret, callbacks_ret);
340
341 return TRUE;
342}

344static void
345ice_error_handler (IceConn conn,
346 Bool swap,
347 int offending_minor_opcode,
348 unsigned long offending_sequence,
349 int error_class,
350 int severity,
351 IcePointer values)
352{
353 g_debug ("GsmXsmpServer: ice_error_handler (%p, %s, %d, %lx, %d, %d)",
354 conn, swap ? "TRUE" : "FALSE", offending_minor_opcode,
355 offending_sequence, error_class, severity);
356
357 if (severity == IceCanContinue) {

361 /* FIXME: the ICElib docs are completely vague about what we're

365 */
366 IceCloseConnection (conn);

398static IceAuthFileEntry *
399auth_entry_new (const char *protocol,
400 const char *network_id)
401{
402 IceAuthFileEntry *file_entry;
403 IceAuthDataEntry data_entry;
404
405 file_entry = malloc (sizeof (IceAuthFileEntry));
406
407 file_entry->protocol_name = strdup (protocol);
408 file_entry->protocol_data = NULL;
409 file_entry->protocol_data_length = 0;
410 file_entry->network_id = strdup (network_id);
411 file_entry->auth_name = strdup (GSM_ICE_MAGIC_COOKIE_AUTH_NAME);
412 file_entry->auth_data = IceGenerateMagicCookie (GSM_ICE_MAGIC_COOKIE_LEN);
413 file_entry->auth_data_length = GSM_ICE_MAGIC_COOKIE_LEN;
414
415 /* Also create an in-memory copy, which is what the server will
416 * actually use for checking client auth.
417 */
418 data_entry.protocol_name = file_entry->protocol_name;
419 data_entry.network_id = file_entry->network_id;
420 data_entry.auth_name = file_entry->auth_name;
421 data_entry.auth_data = file_entry->auth_data;
422 data_entry.auth_data_length = file_entry->auth_data_length;
423 IceSetPaAuthData (1, &data_entry);
424
425 return file_entry;
426}

428static gboolean
429update_iceauthority (GsmXsmpServer *server,
430 gboolean adding)
431{
432 char *filename;
433 char **our_network_ids;
434 FILE *fp;
435 IceAuthFileEntry *auth_entry;
436 GSList *entries;
437 GSList *e;
438 int i;
439 gboolean ok = FALSE;
440
441 filename = IceAuthFileName ();
442 if (IceLockAuthFile (filename,
443 GSM_ICE_AUTH_RETRIES,
444 GSM_ICE_AUTH_INTERVAL,
445 GSM_ICE_AUTH_LOCK_TIMEOUT) != IceAuthLockSuccess) {
446 return FALSE;
447 }
448
449 our_network_ids = g_malloc (server->num_local_xsmp_sockets * sizeof (char *));
450 for (i = 0; i < server->num_local_xsmp_sockets; i++) {
451 our_network_ids[i] = IceGetListenConnectionString (server->xsmp_sockets[i]);
452 }
453
454 entries = NULL;
455
456 fp = fopen (filename, "r+");
457 if (fp != NULL) {
458 while ((auth_entry = IceReadAuthFileEntry (fp)) != NULL) {
459 /* Skip/delete entries with no network ID (invalid), or with
460 * our network ID; if we're starting up, an entry with our
461 * ID must be a stale entry left behind by an old process,
462 * and if we're shutting down, it won't be valid in the
463 * future, so either way we want to remove it from the list.
464 */
465 if (!auth_entry->network_id) {
466 IceFreeAuthFileEntry (auth_entry);
467 continue;
468 }
469
470 for (i = 0; i < server->num_local_xsmp_sockets; i++) {
471 if (!strcmp (auth_entry->network_id, our_network_ids[i])) {
472 IceFreeAuthFileEntry (auth_entry);
473 break;
474 }
475 }
476 if (i != server->num_local_xsmp_sockets) {
477 continue;
478 }
479
480 entries = g_slist_prepend (entries, auth_entry);
481 }
482
483 rewind (fp);
484 } else {
485 int fd;
486
487 if (g_file_test (filename, G_FILE_TEST_EXISTS)) {
488 g_warning ("Unable to read ICE authority file: %s", filename);
489 goto cleanup;
490 }
491
492 fd = open (filename, O_CREAT | O_WRONLY, 0600);
493 fp = fdopen (fd, "w");
494 if (!fp) {
495 g_warning ("Unable to write to ICE authority file: %s", filename);
496 if (fd != -1) {
497 close (fd);
498 }
499 goto cleanup;
500 }
501 }
502
503 if (adding) {
504 for (i = 0; i < server->num_local_xsmp_sockets; i++) {
505 entries = g_slist_append (entries,
506 auth_entry_new ("ICE", our_network_ids[i]));
507 entries = g_slist_prepend (entries,
508 auth_entry_new ("XSMP", our_network_ids[i]));
509 }
510 }
511
512 for (e = entries; e; e = e->next) {
513 IceAuthFileEntry *auth_file_entry = e->data;
514 IceWriteAuthFileEntry (fp, auth_file_entry);
515 IceFreeAuthFileEntry (auth_file_entry);
516 }
517 g_slist_free (entries);
518
519 fclose (fp);
520 ok = TRUE;
521
522 cleanup:
523 IceUnlockAuthFile (filename);
524 for (i = 0; i < server->num_local_xsmp_sockets; i++) {
525 free (our_network_ids[i]);
526 }
527 g_free (our_network_ids);
528
529 return ok;
530}

532static void
533setup_listener (GsmXsmpServer *server)
534{
535 char error[256];
536 mode_t saved_umask;
537 char *network_id_list;
538 int i;
539 int res;
540
541 /* Set up sane error handlers */
542 IceSetErrorHandler (ice_error_handler);
543 IceSetIOErrorHandler (ice_io_error_handler);
544 SmsSetErrorHandler (sms_error_handler);
545
546 /* Initialize libSM; we pass NULL for hostBasedAuthProc to disable
547 * host-based authentication.
548 */
549 res = SmsInitialize (PACKAGE,
550 VERSION,
551 (SmsNewClientProc)accept_xsmp_connection,
552 server,
553 NULL,
554 sizeof (error),
555 error);
556 if (! res) {
557 gsm_util_init_error (TRUE, "Could not initialize libSM: %s", error);
558 }
559
560#if HAVE_XTRANS
561 /* By default, IceListenForConnections will open one socket for each
562 * transport type known to X. We don't want connections from remote
563 * hosts, so for security reasons it would be best if ICE didn't
564 * even open any non-local sockets. So we use an internal ICElib
565 * method to disable them here. Unfortunately, there is no way to
566 * ask X what transport types it knows about, so we're forced to
567 * guess.
568 */
569 _IceTransNoListen ("tcp");
570#endif
571
572 /* Create the XSMP socket. Older versions of IceListenForConnections
573 * have a bug which causes the umask to be set to 0 on certain types
574 * of failures. Probably not an issue on any modern systems, but
575 * we'll play it safe.
576 */
577 saved_umask = umask (0);
578 umask (saved_umask);
579 res = IceListenForConnections (&server->num_xsmp_sockets,
580 &server->xsmp_sockets,
581 sizeof (error),
582 error);
583 if (! res) {
584 gsm_util_init_error (TRUE, _("Could not create ICE listening socket: %s"), error);
585 }
586
587 umask (saved_umask);
588
589 /* Find the local sockets in the returned socket list and move them
590 * to the start of the list.
591 */
592 for (i = server->num_local_xsmp_sockets = 0; i < server->num_xsmp_sockets; i++) {
593 char *id = IceGetListenConnectionString (server->xsmp_sockets[i]);
594
595 if (!strncmp (id, "local/", sizeof ("local/") - 1) ||
596 !strncmp (id, "unix/", sizeof ("unix/") - 1)) {
597 if (i > server->num_local_xsmp_sockets) {
598 IceListenObj tmp;
599 tmp = server->xsmp_sockets[i];
600 server->xsmp_sockets[i] = server->xsmp_sockets[server->num_local_xsmp_sockets];
601 server->xsmp_sockets[server->num_local_xsmp_sockets] = tmp;
602 }
603 server->num_local_xsmp_sockets++;
604 }
605 free (id);
606 }
607
608 if (server->num_local_xsmp_sockets == 0) {
609 gsm_util_init_error (TRUE, "IceListenForConnections did not return a local listener!");
610 }
611
612#ifdef HAVE_XTRANS
613 if (server->num_local_xsmp_sockets != server->num_xsmp_sockets) {
614 /* Xtrans was apparently compiled with support for some
615 * non-local transport besides TCP (which we disabled above); we
616 * won't create IO watches on those extra sockets, so
617 * connections to them will never be noticed, but they're still
618 * there, which is inelegant.
619 *
620 * If the g_warning below is triggering for you and you want to
621 * stop it, the fix is to add additional _IceTransNoListen()
622 * calls above.
623 */
624 network_id_list = IceComposeNetworkIdList (server->num_xsmp_sockets - server->num_local_xsmp_sockets,
625 server->xsmp_sockets + server->num_local_xsmp_sockets);
626 g_warning ("IceListenForConnections returned %d non-local listeners: %s",
627 server->num_xsmp_sockets - server->num_local_xsmp_sockets,
628 network_id_list);
629 free (network_id_list);
630 }
631#endif
632
633 /* Update .ICEauthority with new auth entries for our socket */
634 if (!update_iceauthority (server, TRUE)) {
635 /* FIXME: is this really fatal? Hm... */
636 gsm_util_init_error (TRUE,
637 "Could not update ICEauthority file %s",
638 IceAuthFileName ());
639 }
640
641 network_id_list = IceComposeNetworkIdList (server->num_local_xsmp_sockets,
642 server->xsmp_sockets);
643
644 gsm_util_setenv ("SESSION_MANAGER", network_id_list);
645 g_debug ("GsmXsmpServer: SESSION_MANAGER=%s\n", network_id_list);
646 free (network_id_list);
647}
