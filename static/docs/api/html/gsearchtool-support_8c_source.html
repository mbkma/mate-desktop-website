53gboolean
54is_path_hidden (const gchar * path)
55{
56 gint results = FALSE;
57 gchar * sub_str;
58 gchar * hidden_path_substr = g_strconcat (G_DIR_SEPARATOR_S, ".", NULL);
59
60 sub_str = g_strstr_len (path, strlen (path), hidden_path_substr);
61
62 if (sub_str != NULL) {
63 gchar * mate_desktop_str;
64
65 mate_desktop_str = g_strconcat (G_DIR_SEPARATOR_S, ".mate-desktop", G_DIR_SEPARATOR_S, NULL);
66
67 /* exclude the .mate-desktop folder */
68 if (strncmp (sub_str, mate_desktop_str, strlen (mate_desktop_str)) == 0) {
69 sub_str++;
70 results = (g_strstr_len (sub_str, strlen (sub_str), hidden_path_substr) != NULL);
71 }
72 else {
73 results = TRUE;
74 }
75
76 g_free (mate_desktop_str);
77 }
78
79 g_free (hidden_path_substr);
80 return results;
81}

83gboolean
84is_quick_search_excluded_path (const gchar * path)
85{
86 GSettings * settings;
87 gchar ** exclude_path_list;
88 gchar * dir;
89 gboolean results = FALSE;
90 gint i;
91
92 dir = g_strdup (path);
93
94 /* Remove trailing G_DIR_SEPARATOR. */
95 if ((strlen (dir) > 1) && (g_str_has_suffix (dir, G_DIR_SEPARATOR_S) == TRUE)) {
96 dir[strlen (dir) - 1] = '\0';
97 }
98
99 /* Always exclude a path that is symbolic link. */
100 if (g_file_test (dir, G_FILE_TEST_IS_SYMLINK)) {
101 g_free (dir);
102
103 return TRUE;
104 }
105 g_free (dir);
106
107 settings = g_settings_new ("org.mate.search-tool");
108
109 /* Check path against the Quick-Search-Excluded-Paths list. */
110 exclude_path_list = g_settings_get_strv (settings, "quick-search-excluded-paths");
111
112 if (exclude_path_list) {
113 for (i = 0; exclude_path_list[i]; i++) {
114
115 /* Skip empty or null values. */
116 if (strlen (exclude_path_list[i]) == 0) {
117 continue;
118 }
119
120 dir = g_strdup (exclude_path_list[i]);
121
122 /* Wild-card comparisons. */
123 if (g_strstr_len (dir, strlen (dir), "*") != NULL) {
124
125 if (g_pattern_match_simple (dir, path) == TRUE) {
126
127 results = TRUE;
128 g_free (dir);
129 break;
130 }
131 }
132 /* Non-wild-card comparisons. */
133 else {
134 /* Add a trailing G_DIR_SEPARATOR. */
135 if (g_str_has_suffix (dir, G_DIR_SEPARATOR_S) == FALSE) {
136
137 gchar *tmp;
138
139 tmp = dir;
140 dir = g_strconcat (dir, G_DIR_SEPARATOR_S, NULL);
141 g_free (tmp);
142 }
143
144 if (strcmp (path, dir) == 0) {
145
146 results = TRUE;
147 g_free (dir);
148 break;
149 }
150 }
151 g_free (dir);
152 }
153 g_strfreev (exclude_path_list);
154 }
155
156 g_object_unref (settings);
157 return results;
158}

237gboolean
238compare_regex (const gchar * regex,
239 const gchar * string)
240{
241 regex_t regexec_pattern;
242
243 if (regex == NULL) {
244 return TRUE;
245 }
246
247 if (!regcomp (&regexec_pattern, regex, REG_EXTENDED|REG_NOSUB)) {
248 if (regexec (&regexec_pattern, string, 0, 0, 0) != REG_NOMATCH) {
249 regfree (&regexec_pattern);
250 return TRUE;
251 }
252 regfree (&regexec_pattern);
253 }
254 return FALSE;
255}

288{
289 GString * gs;
290
291 if (string == NULL) {
292 return NULL;
293 }
294
295 if (count_of_char_in_string (string, '\'') == 0) {
296 return g_strdup(string);
297 }
298 gs = g_string_new ("");
299 for(; *string; string++) {
300 if (*string == '\'') {
301 g_string_append(gs, "'\\''");
302 }
303 else {
304 g_string_append_c(gs, *string);
305 }
306 }
307 return g_string_free (gs, FALSE);

407gchar *
408get_readable_date (const CajaDateFormat date_format_enum,
409 GDateTime *file_date_time)
410{
411 const gchar *format;
412 GDateTime *now;
413 GTimeSpan file_date_age;
414
415 /* Base format of date column on caja date-format key */
416 if (date_format_enum == CAJA_DATE_FORMAT_LOCALE) {
417 return g_date_time_format (file_date_time, "%c");
418 } else if (date_format_enum == CAJA_DATE_FORMAT_ISO) {
419 return g_date_time_format (file_date_time, "%Y-%m-%d %H:%M:%S");
420 }
421
422 now = g_date_time_new_now_local ();
423 file_date_age = g_date_time_difference (now, file_date_time);
424 g_date_time_unref (now);
425
426 if (file_date_age < G_TIME_SPAN_DAY) {
427 format = _("today at %-I:%M %p");
428 } else if (file_date_age < 2 * G_TIME_SPAN_DAY) {
429 format = _("yesterday at %-I:%M %p");
430 } else {
431 format = _("%A, %B %-d %Y at %-I:%M:%S %p");
432 }
433
434 return g_date_time_format (file_date_time, format);
435}

437gchar *
438get_file_type_description (const gchar * file,
439 GFileInfo * file_info)
440{
441 const char * content_type = NULL;
442 gchar * desc;
443
444 if (file != NULL) {
445 content_type = g_file_info_get_content_type (file_info);
446 }
447
448 if (content_type == NULL || g_content_type_is_unknown (content_type) == TRUE) {
449 return g_content_type_get_description ("application/octet-stream");
450 }
451
452 desc = g_content_type_get_description (content_type);
453
454 if (g_file_info_get_is_symlink (file_info) == TRUE) {
455
456 const gchar * symlink_target;
457 gchar * absolute_symlink = NULL;
458 gchar * str = NULL;
459
460 symlink_target = g_file_info_get_symlink_target (file_info);
461
462 if (g_path_is_absolute (symlink_target) != TRUE) {
463 gchar *dirname;
464
465 dirname = g_path_get_dirname (file);
466 absolute_symlink = g_strconcat (dirname, G_DIR_SEPARATOR_S, symlink_target, NULL);
467 g_free (dirname);
468 }
469 else {
470 absolute_symlink = g_strdup (symlink_target);
471 }
472
473 if (g_file_test (absolute_symlink, G_FILE_TEST_EXISTS) != TRUE) {
474 if ((g_ascii_strcasecmp (content_type, "x-special/socket") != 0) &&
475 (g_ascii_strcasecmp (content_type, "x-special/fifo") != 0)) {
476 g_free (absolute_symlink);
477 g_free (desc);
478 return g_strdup (_("link (broken)"));
479 }
480 }
481
482 str = g_strdup_printf (_("link to %s"), (desc != NULL) ? desc : content_type);
483 g_free (absolute_symlink);
484 g_free (desc);
485 return str;
486 }
487 return desc;
488}

518static void
519gsearchtool_draw_frame_row (GdkPixbuf * frame_image,
520 gint target_width,
521 gint source_width,
522 gint source_v_position,
523 gint dest_v_position,
524 GdkPixbuf * result_pixbuf,
525 gint left_offset,
526 gint height)
527{
528 gint remaining_width;
529 gint h_offset;
530 gint slab_width;
531
532 remaining_width = target_width;
533 h_offset = 0;
534 while (remaining_width > 0) {
535 slab_width = remaining_width > source_width ? source_width : remaining_width;
536 gdk_pixbuf_copy_area (frame_image, left_offset, source_v_position, slab_width,
537 height, result_pixbuf, left_offset + h_offset, dest_v_position);
538 remaining_width -= slab_width;
539 h_offset += slab_width;
540 }
541}

568static GdkPixbuf *
569gsearchtool_stretch_frame_image (GdkPixbuf *frame_image,
570 gint left_offset,
571 gint top_offset,
572 gint right_offset,
573 gint bottom_offset,
574 gint dest_width,
575 gint dest_height,
576 gboolean fill_flag)
577{
578 GdkPixbuf * result_pixbuf;
579 gint frame_width, frame_height;
580 gint target_width, target_frame_width;
581 gint target_height, target_frame_height;
582
583 frame_width = gdk_pixbuf_get_width (frame_image);
584 frame_height = gdk_pixbuf_get_height (frame_image);
585
586 if (fill_flag) {
587 result_pixbuf = gdk_pixbuf_scale_simple (frame_image, dest_width, dest_height, GDK_INTERP_NEAREST);
588 } else {
589 result_pixbuf = gdk_pixbuf_new (GDK_COLORSPACE_RGB, TRUE, 8, dest_width, dest_height);
590 }
591
592 /* clear the new pixbuf */
593 if (fill_flag == FALSE) {
594 gdk_pixbuf_fill (result_pixbuf, 0xffffffff);
595 }
596
597 target_width = dest_width - left_offset - right_offset;
598 target_frame_width = frame_width - left_offset - right_offset;
599
600 target_height = dest_height - top_offset - bottom_offset;
601 target_frame_height = frame_height - top_offset - bottom_offset;
602
603 /* Draw the left top corner and top row */
604 gdk_pixbuf_copy_area (frame_image, 0, 0, left_offset, top_offset, result_pixbuf, 0, 0);
605 gsearchtool_draw_frame_row (frame_image, target_width, target_frame_width, 0, 0,
606 result_pixbuf, left_offset, top_offset);
607
608 /* Draw the right top corner and left column */
609 gdk_pixbuf_copy_area (frame_image, frame_width - right_offset, 0, right_offset, top_offset,
610 result_pixbuf, dest_width - right_offset, 0);
611 gsearchtool_draw_frame_column (frame_image, target_height, target_frame_height, 0, 0,
612 result_pixbuf, top_offset, left_offset);
613
614 /* Draw the bottom right corner and bottom row */
615 gdk_pixbuf_copy_area (frame_image, frame_width - right_offset, frame_height - bottom_offset,
616 right_offset, bottom_offset, result_pixbuf, dest_width - right_offset,
617 dest_height - bottom_offset);
618 gsearchtool_draw_frame_row (frame_image, target_width, target_frame_width,
619 frame_height - bottom_offset, dest_height - bottom_offset,
620 result_pixbuf, left_offset, bottom_offset);
621
622 /* Draw the bottom left corner and the right column */
623 gdk_pixbuf_copy_area (frame_image, 0, frame_height - bottom_offset, left_offset, bottom_offset,
624 result_pixbuf, 0, dest_height - bottom_offset);
625 gsearchtool_draw_frame_column (frame_image, target_height, target_frame_height,
626 frame_width - right_offset, dest_width - right_offset,
627 result_pixbuf, top_offset, right_offset);
628 return result_pixbuf;
629}

675static GdkPixbuf *
676gsearchtool_get_thumbnail_image (const gchar * thumbnail)
677{
678 GdkPixbuf * pixbuf = NULL;
679
680 if (thumbnail != NULL) {
681 if (g_file_test (thumbnail, G_FILE_TEST_EXISTS)) {
682
683 GdkPixbuf * thumbnail_pixbuf = NULL;
684 gfloat scale_factor_x = 1.0;
685 gfloat scale_factor_y = 1.0;
686 gint scale_x;
687 gint scale_y;
688 gint width;
689 gint height;
690
691 thumbnail_pixbuf = gdk_pixbuf_new_from_file (thumbnail, NULL);
692 gsearchtool_thumbnail_frame_image (&thumbnail_pixbuf);
693
694 width = gdk_pixbuf_get_width (thumbnail_pixbuf);
695 if (width > ICON_SIZE) {
696 scale_factor_x = (gfloat) ICON_SIZE / (gfloat) width;
697 }
698 height = gdk_pixbuf_get_height (thumbnail_pixbuf);
699 if (height > ICON_SIZE) {
700 scale_factor_y = (gfloat) ICON_SIZE / (gfloat) height;
701 }
702
703 if (width > height) {
704 scale_x = ICON_SIZE;
705 scale_y = (gint) (height * scale_factor_x);
706 }
707 else {
708 scale_x = (gint) (width * scale_factor_y);
709 scale_y = ICON_SIZE;
710 }
711
712 pixbuf = gdk_pixbuf_scale_simple (thumbnail_pixbuf, scale_x, scale_y, GDK_INTERP_BILINEAR);
713 g_object_unref (thumbnail_pixbuf);
714 }
715 }
716 return pixbuf;
717}

1163static gchar *
1164make_valid_utf8 (const gchar * name)
1165{
1166 GString *string;
1167 const char *remainder, *invalid;
1168 int remaining_bytes, valid_bytes;
1169
1170 string = NULL;
1171 remainder = name;
1172 remaining_bytes = strlen (name);
1173
1174 while (remaining_bytes != 0) {
1175 if (g_utf8_validate (remainder, remaining_bytes, &invalid)) {
1176 break;
1177 }
1178 valid_bytes = invalid - remainder;
1179
1180 if (string == NULL) {
1181 string = g_string_sized_new (remaining_bytes);
1182 }
1183 g_string_append_len (string, remainder, valid_bytes);
1184 g_string_append_c (string, '?');
1185
1186 remaining_bytes -= valid_bytes + 1;
1187 remainder = invalid + 1;
1188 }
1189
1190 if (string == NULL) {
1191 return g_strdup (name);
1192 }
1193
1194 g_string_append (string, remainder);
1195 g_string_append (string, _(" (invalid Unicode)"));
1196 g_assert (g_utf8_validate (string->str, -1, NULL));
1197
1198 return g_string_free (string, FALSE);
1199}

1201static gchar *
1202extract_string_until (const gchar * original,
1203 const gchar * until_substring)
1204{
1205 gchar * result;
1206
1207 g_assert ((gint) strlen (original) >= until_substring - original);
1208 g_assert (until_substring - original >= 0);
1209
1210 result = g_malloc (until_substring - original + 1);
1211 strncpy (result, original, until_substring - original);
1212 result[until_substring - original] = '\0';
1213
1214 return result;
1215}

1217/* Dismantle a file name, separating the base name, the file suffix and removing any
1218 * (xxxcopy), etc. string. Figure out the count that corresponds to the given
1219 * (xxxcopy) substring.
1220 */
1221static void
1222parse_previous_duplicate_name (const gchar * name,
1223 gchar ** name_base,
1224 const gchar ** suffix,
1225 gint * count)
1226{
1227 const gchar * tag;
1228
1229 g_assert (name[0] != '\0');
1230
1231 *suffix = strchr (name + 1, '.');
1232 if (*suffix == NULL || (*suffix)[1] == '\0') {
1233 /* no suffix */
1234 *suffix = "";
1235 }
1236
1237 tag = strstr (name, COPY_DUPLICATE_TAG);
1238 if (tag != NULL) {
1239 if (tag > *suffix) {
1240 /* handle case "foo. (copy)" */
1241 *suffix = "";
1242 }
1243 *name_base = extract_string_until (name, tag);
1244 *count = 1;
1245 return;
1246 }
1247
1248 tag = strstr (name, ANOTHER_COPY_DUPLICATE_TAG);
1249 if (tag != NULL) {
1250 if (tag > *suffix) {
1251 /* handle case "foo. (another copy)" */
1252 *suffix = "";
1253 }
1254 *name_base = extract_string_until (name, tag);
1255 *count = 2;
1256 return;
1257 }
1258
1259 /* Check to see if we got one of st, nd, rd, th. */
1260 tag = strstr (name, X11TH_COPY_DUPLICATE_TAG);
1261
1262 if (tag == NULL) {
1263 tag = strstr (name, X12TH_COPY_DUPLICATE_TAG);
1264 }
1265 if (tag == NULL) {
1266 tag = strstr (name, X13TH_COPY_DUPLICATE_TAG);
1267 }
1268 if (tag == NULL) {
1269 tag = strstr (name, ST_COPY_DUPLICATE_TAG);
1270 }
1271 if (tag == NULL) {
1272 tag = strstr (name, ND_COPY_DUPLICATE_TAG);
1273 }
1274 if (tag == NULL) {
1275 tag = strstr (name, RD_COPY_DUPLICATE_TAG);
1276 }
1277 if (tag == NULL) {
1278 tag = strstr (name, TH_COPY_DUPLICATE_TAG);
1279 }
1280
1281 /* If we got one of st, nd, rd, th, fish out the duplicate number. */
1282 if (tag != NULL) {
1283 /* localizers: opening parentheses to match the "th copy)" string */
1284 tag = strstr (name, _(" ("));
1285 if (tag != NULL) {
1286 if (tag > *suffix) {
1287 /* handle case "foo. (22nd copy)" */
1288 *suffix = "";
1289 }
1290 *name_base = extract_string_until (name, tag);
1291 /* localizers: opening parentheses of the "th copy)" string */
1292 if (sscanf (tag, _(" (%d"), count) == 1) {
1293 if (*count < 1 || *count > 1000000) {
1294 /* keep the count within a reasonable range */
1295 *count = 0;
1296 }
1297 return;
1298 }
1299 *count = 0;
1300 return;
1301 }
1302 }
1303
1304 *count = 0;
1305 if (**suffix != '\0') {
1306 *name_base = extract_string_until (name, *suffix);
1307 } else {
1308 *name_base = g_strdup (name);
1309 }
1310}

1312static gchar *
1313make_next_duplicate_name (const gchar *base,
1314 const gchar *suffix,
1315 gint count)
1316{
1317 const gchar * format;
1318 gchar * result;
1319
1320 if (count < 1) {
1321 g_warning ("bad count %d in make_next_duplicate_name()", count);
1322 count = 1;
1323 }
1324
1325 if (count <= 2) {
1326
1327 /* Handle special cases for low numbers.
1328 * Perhaps for some locales we will need to add more.
1329 */
1330 switch (count) {
1331 default:
1332 g_assert_not_reached ();
1333 /* fall through */
1334 case 1:
1335 format = FIRST_COPY_DUPLICATE_FORMAT;
1336 break;
1337 case 2:
1338 format = SECOND_COPY_DUPLICATE_FORMAT;
1339 break;
1340
1341 }
1342 result = g_strdup_printf (format, base, suffix);
1343 } else {
1344
1345 /* Handle special cases for the first few numbers of each ten.
1346 * For locales where getting this exactly right is difficult,
1347 * these can just be made all the same as the general case below.
1348 */
1349
1350 /* Handle special cases for x11th - x20th.
1351 */
1352 switch (count % 100) {
1353 case 11:
1354 format = X11TH_COPY_DUPLICATE_FORMAT;
1355 break;
1356 case 12:
1357 format = X12TH_COPY_DUPLICATE_FORMAT;
1358 break;
1359 case 13:
1360 format = X13TH_COPY_DUPLICATE_FORMAT;
1361 break;
1362 default:
1363 format = NULL;
1364 break;
1365 }
1366
1367 if (format == NULL) {
1368 switch (count % 10) {
1369 case 1:
1370 format = ST_COPY_DUPLICATE_FORMAT;
1371 break;
1372 case 2:
1373 format = ND_COPY_DUPLICATE_FORMAT;
1374 break;
1375 case 3:
1376 format = RD_COPY_DUPLICATE_FORMAT;
1377 break;
1378 default:
1379 /* The general case. */
1380 format = TH_COPY_DUPLICATE_FORMAT;
1381 break;
1382 }
1383 }
1384 result = g_strdup_printf (format, base, count, suffix);
1385 }
1386 return result;
1387}

1389static gchar *
1390get_duplicate_name (const gchar *name)
1391{
1392 const gchar * suffix;
1393 gchar * name_base;
1394 gchar * result;
1395 gint count;
1396
1397 parse_previous_duplicate_name (name, &name_base, &suffix, &count);
1398 result = make_next_duplicate_name (name_base, suffix, count + 1);
1399 g_free (name_base);
1400
1401 return result;
1402}

1405gsearchtool_get_next_duplicate_name (const gchar * basename)
1406{
1407 gchar * utf8_name;
1408 gchar * utf8_result;
1409 gchar * result;
1410
1411 utf8_name = g_filename_to_utf8 (basename, -1, NULL, NULL, NULL);
1412
1413 if (utf8_name == NULL) {

1415 * G_BROKEN_FILENAMES not set when it should be.
1416 * Try converting from the locale */
