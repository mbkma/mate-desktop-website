81static void
82clear_children (EvViewAccessible *self)
83{
84 gint i;
85 AtkObject *child;
86
87 if (self->priv->children == NULL)
88 return;
89
90 for (i = 0; i < self->priv->children->len; i++) {

98static void
99ev_view_accessible_finalize (GObject *object)
100{
101 EvViewAccessiblePrivate *priv = EV_VIEW_ACCESSIBLE (object)->priv;
102 int i;
103
104 if (priv->model) {
105 g_signal_handlers_disconnect_by_data (priv->model, object);
106 g_object_unref (priv->model);
107 priv->model = NULL;
108 }
109 if (priv->action_idle_handler)
110 g_source_remove (priv->action_idle_handler);
111 for (i = 0; i < LAST_ACTION; i++)
112 g_free (priv->action_descriptions [i]);
113
114 clear_children (EV_VIEW_ACCESSIBLE (object));
115
116 G_OBJECT_CLASS (ev_view_accessible_parent_class)->finalize (object);
117}

138static AtkObject *
139ev_view_accessible_ref_child (AtkObject *obj,
140 gint i)
141{
142 EvViewAccessible *self;
143 EvView *view;
144
145 g_return_val_if_fail (EV_IS_VIEW_ACCESSIBLE (obj), NULL);
146 self = EV_VIEW_ACCESSIBLE (obj);
147 g_return_val_if_fail (i >= 0 || i < ev_view_accessible_get_n_pages (self), NULL);
148
149 view = EV_VIEW (gtk_accessible_get_widget (GTK_ACCESSIBLE (obj)));
150 if (view == NULL)
151 return NULL;
152
153 /* If a given page is requested, we assume that the text would
154 * be requested soon, so we need to be sure that is cached.*/
155 if (view->page_cache)
156 ev_page_cache_ensure_page (view->page_cache, i);
157
158 return g_object_ref (g_ptr_array_index (self->priv->children, i));
159}

227static gboolean
228ev_view_accessible_action_do_action (AtkAction *action,
229 gint i)
230{
231 EvViewAccessiblePrivate* priv = EV_VIEW_ACCESSIBLE (action)->priv;
232
233 if (gtk_accessible_get_widget (GTK_ACCESSIBLE (action)) == NULL)
234 return FALSE;
235
236 if (priv->action_idle_handler)
237 return FALSE;
238
239 switch (i) {
240 case ACTION_SCROLL_UP:
241 priv->idle_scroll = GTK_SCROLL_PAGE_BACKWARD;
242 break;
243 case ACTION_SCROLL_DOWN:
244 priv->idle_scroll = GTK_SCROLL_PAGE_FORWARD;
245 break;
246 default:
247 return FALSE;
248 }
249 priv->action_idle_handler = g_idle_add (ev_view_accessible_idle_do_action,
250 action);
251 return TRUE;
252}

313static void
314ev_view_accessible_cursor_moved (EvView *view,
315 gint page,
316 gint offset,
317 EvViewAccessible *accessible)
318{
319 EvViewAccessiblePrivate* priv = accessible->priv;
320 EvPageAccessible *page_accessible = NULL;
321
322 if (priv->previous_cursor_page != page) {
323 AtkObject *previous_page = NULL;
324 AtkObject *current_page = NULL;
325
326 previous_page = g_ptr_array_index (priv->children,
327 priv->previous_cursor_page);
328 atk_object_notify_state_change (previous_page, ATK_STATE_FOCUSED, FALSE);
329 priv->previous_cursor_page = page;
330 current_page = g_ptr_array_index (priv->children, page);
331 atk_object_notify_state_change (current_page, ATK_STATE_FOCUSED, TRUE);
332
333 /* +1 as user start to count on 1, but evince starts on 0 */
334 g_signal_emit_by_name (accessible, "page-changed", page + 1);
335 }
336
337 page_accessible = g_ptr_array_index (priv->children, page);
338 g_signal_emit_by_name (page_accessible, "text-caret-moved", offset);
339}

365static void
366initialize_children (EvViewAccessible *self)
367{
368 gint i;
369 EvPageAccessible *child;
370 gint n_pages;
371 EvDocument *ev_document;
372
373 ev_document = ev_document_model_get_document (self->priv->model);
374 n_pages = ev_document_get_n_pages (ev_document);
375
376 self->priv->children = g_ptr_array_new_full (n_pages, (GDestroyNotify) g_object_unref);
377 for (i = 0; i < n_pages; i++) {
378 child = ev_page_accessible_new (self, i);
379 g_ptr_array_add (self->priv->children, child);
380 }
381 /* When a document is reloaded, it may have less pages.
382 * We need to update the end page accordingly to avoid
383 * invalid access to self->priv->children
384 * See https://bugzilla.gnome.org/show_bug.cgi?id=735744
385 */
386 if (self->priv->end_page >= n_pages)
387 self->priv->end_page = n_pages - 1;
388}

390static void
391document_changed_cb (EvDocumentModel *model,
392 GParamSpec *pspec,
393 EvViewAccessible *accessible)
394{
395 EvDocument *document = ev_document_model_get_document (model);
396
397 clear_children (accessible);
398
399 if (document == NULL)
400 return;
401
402 initialize_children (accessible);
403
404 /* Inside this callback the document is already loaded. We
405 * don't have here an "just before" and "just after"
406 * signal. We emit both in a row, as usual ATs uses reload to
407 * know that current content has changed, and load-complete to
408 * know that the content is already available.
409 */
410 g_signal_emit_by_name (accessible, "reload");
411 g_signal_emit_by_name (accessible, "load-complete");
412}

414void
415ev_view_accessible_set_model (EvViewAccessible *accessible,
416 EvDocumentModel *model)
417{
418 EvViewAccessiblePrivate* priv = accessible->priv;
419
420 if (priv->model == model)
421 return;
422
423 if (priv->model) {
424 g_signal_handlers_disconnect_by_data (priv->model, accessible);
425 g_object_unref (priv->model);
426 }
427
428 priv->model = g_object_ref (model);
429
430 document_changed_cb (model, NULL, accessible);
431 g_signal_connect (priv->model, "page-changed",
432 G_CALLBACK (page_changed_cb),
433 accessible);
434 g_signal_connect (priv->model, "notify::document",
435 G_CALLBACK (document_changed_cb),
436 accessible);
437}

504void
505_transform_doc_rect_to_atk_rect (EvViewAccessible *accessible,
506 gint page,
507 EvRectangle *doc_rect,
508 EvRectangle *atk_rect,
509 AtkCoordType coord_type)
510{
511 EvView *view;
512 GdkRectangle view_rect;
513 GtkWidget *widget, *toplevel;
514 gint x_widget, y_widget;
515
516 view = EV_VIEW (gtk_accessible_get_widget (GTK_ACCESSIBLE (accessible)));
517 _ev_view_transform_doc_rect_to_view_rect (view, page, doc_rect, &view_rect);
518 view_rect.x -= view->scroll_x;
519 view_rect.y -= view->scroll_y;
520
521 widget = GTK_WIDGET (view);
522 toplevel = gtk_widget_get_toplevel (widget);
523 gtk_widget_translate_coordinates (widget, toplevel, 0, 0, &x_widget, &y_widget);
524 view_rect.x += x_widget;
525 view_rect.y += y_widget;
526
527 if (coord_type == ATK_XY_SCREEN) {
528 gint x_window, y_window;
529 gdk_window_get_origin (gtk_widget_get_window (toplevel), &x_window, &y_window);
530 view_rect.x += x_window;
531 view_rect.y += y_window;
532 }
533
534 atk_rect->x1 = view_rect.x;
535 atk_rect->y1 = view_rect.y;
536 atk_rect->x2 = view_rect.x + view_rect.width;
537 atk_rect->y2 = view_rect.y + view_rect.height;
538}

540gboolean
541ev_view_accessible_is_doc_rect_showing (EvViewAccessible *accessible,
542 gint page,
543 EvRectangle *doc_rect)
544{
545 EvView *view;
546 GdkRectangle view_rect;
547 GtkAllocation allocation;
548 gint x, y;
549 gboolean hidden;
550
551 view = EV_VIEW (gtk_accessible_get_widget (GTK_ACCESSIBLE (accessible)));
552 if (page < view->start_page || page > view->end_page)
553 return FALSE;
554
555 gtk_widget_get_allocation (GTK_WIDGET (view), &allocation);
556 x = gtk_adjustment_get_value (view->hadjustment);
557 y = gtk_adjustment_get_value (view->vadjustment);
558
559 _ev_view_transform_doc_rect_to_view_rect (view, page, doc_rect, &view_rect);
560 hidden = view_rect.x + view_rect.width < x || view_rect.x > x + allocation.width ||
561 view_rect.y + view_rect.height < y || view_rect.y > y + allocation.height;
562
563 return !hidden;
564}

566void
567ev_view_accessible_set_page_range (EvViewAccessible *accessible,
568 gint start,
569 gint end)
570{
571 gint i;
572 AtkObject *page;
573
574 g_return_if_fail (EV_IS_VIEW_ACCESSIBLE (accessible));
575
576 for (i = accessible->priv->start_page; i <= accessible->priv->end_page; i++) {
577 if (i < start || i > end) {
578 page = g_ptr_array_index (accessible->priv->children, i);
579 atk_object_notify_state_change (page, ATK_STATE_SHOWING, FALSE);
580 }
581 }
582
583 for (i = start; i <= end; i++) {
584 if (i < accessible->priv->start_page || i > accessible->priv->end_page) {
585 page = g_ptr_array_index (accessible->priv->children, i);
586 atk_object_notify_state_change (page, ATK_STATE_SHOWING, TRUE);
587 }
588 }
589
590 accessible->priv->start_page = start;
591 accessible->priv->end_page = end;
