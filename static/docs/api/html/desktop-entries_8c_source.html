93static const char *
94get_current_desktop (void)
95{
96 static char *current_desktop = NULL;
97
98 /* Support XDG_CURRENT_DESKTOP environment variable; this can be used
99 * to abuse mate-menus in non-MATE desktops. */
100 if (!current_desktop)
101 {
102 const char *desktop;
103
104 desktop = g_getenv ("XDG_CURRENT_DESKTOP");
105
106 /* Note: if XDG_CURRENT_DESKTOP is set but empty, do as if it
107 * was not set */
108 if (!desktop || desktop[0] == '\0')
109 current_desktop = g_strdup ("MATE");
110 else
111 current_desktop = g_strdup (desktop);
112 }
113
114 /* Using "*" means skipping desktop-related checks */
115 if (g_strcmp0 (current_desktop, "*") == 0)
116 return NULL;
117
118 return current_desktop;
119}

121static GIcon *
122key_file_get_icon (GKeyFile *key_file)
123{
124 GIcon *icon = NULL;
125 gchar *icon_name;
126
127 icon_name = g_key_file_get_locale_string (key_file, DESKTOP_ENTRY_GROUP,
128 "Icon", NULL, NULL);
129 if (!icon_name)
130 return NULL;
131
132 if (g_path_is_absolute (icon_name)) {
133 GFile *file;
134
135 file = g_file_new_for_path (icon_name);
136 icon = g_file_icon_new (file);
137 g_object_unref (file);
138 } else {
139 char *p;
140
141 /* Work around a common mistake in desktop files */
142 if ((p = strrchr (icon_name, '.')) != NULL &&
143 (strcmp (p, ".png") == 0 ||
144 strcmp (p, ".xpm") == 0 ||
145 strcmp (p, ".svg") == 0))
146 *p = 0;
147
148 icon = g_themed_icon_new (icon_name);
149 }
150
151 g_free (icon_name);
152
153 return icon;
154}

156static gboolean
157key_file_get_show_in (GKeyFile *key_file)
158{
159 const gchar *current_desktop;
160 gchar **strv;
161 gboolean show_in = TRUE;
162 int i;
163
164 current_desktop = get_current_desktop ();
165 if (!current_desktop)
166 return TRUE;
167
168 strv = g_key_file_get_string_list (key_file,
169 DESKTOP_ENTRY_GROUP,
170 "OnlyShowIn",
171 NULL,
172 NULL);
173 if (strv)
174 {
175 show_in = FALSE;
176 for (i = 0; strv[i]; i++)
177 {
178 if (!strcmp (strv[i], current_desktop))
179 {
180 show_in = TRUE;
181 break;
182 }
183 }
184 }
185 else
186 {
187 strv = g_key_file_get_string_list (key_file,
188 DESKTOP_ENTRY_GROUP,
189 "NotShowIn",
190 NULL,
191 NULL);
192 if (strv)
193 {
194 show_in = TRUE;
195 for (i = 0; strv[i]; i++)
196 {
197 if (!strcmp (strv[i], current_desktop))
198 {
199 show_in = FALSE;
200 }
201 }
202 }
203 }
204 g_strfreev (strv);
205
206 return show_in;
207}

209static gboolean
210desktop_entry_load_directory (DesktopEntry *entry,
211 GKeyFile *key_file,
212 GError **error)
213{
214 DesktopEntryDirectory *entry_directory = (DesktopEntryDirectory*)entry;
215 char *type_str;
216
217 type_str = g_key_file_get_string (key_file, DESKTOP_ENTRY_GROUP, "Type", error);
218 if (!type_str)
219 return FALSE;
220
221 if (strcmp (type_str, "Directory") != 0)
222 {
223 g_set_error (error,
224 G_KEY_FILE_ERROR,
225 G_KEY_FILE_ERROR_INVALID_VALUE,
226 "\"%s\" does not contain the correct \"Type\" value\n", entry->path);
227 g_free (type_str);
228 return FALSE;
229 }
230
231 g_free (type_str);
232
233 entry_directory->name = g_key_file_get_locale_string (key_file, DESKTOP_ENTRY_GROUP, "Name", NULL, error);
234 if (entry_directory->name == NULL)
235 return FALSE;
236
237 entry_directory->generic_name = g_key_file_get_locale_string (key_file, DESKTOP_ENTRY_GROUP, "GenericName", NULL, NULL);
238 entry_directory->comment = g_key_file_get_locale_string (key_file, DESKTOP_ENTRY_GROUP, "Comment", NULL, NULL);
239 entry_directory->icon = key_file_get_icon (key_file);
240 entry_directory->nodisplay = g_key_file_get_boolean (key_file, DESKTOP_ENTRY_GROUP, "NoDisplay", NULL) != FALSE;
241 entry_directory->hidden = g_key_file_get_boolean (key_file, DESKTOP_ENTRY_GROUP, "Hidden", NULL) != FALSE;
242 entry_directory->showin = key_file_get_show_in (key_file) != FALSE;
243
244 return TRUE;
245}

247static gboolean
248desktop_entry_load (DesktopEntry *entry)
249{
250 if (entry->type == DESKTOP_ENTRY_DESKTOP)
251 {
252 DesktopEntryDesktop *entry_desktop = (DesktopEntryDesktop*)entry;
253 const char *categories_str;
254
255 entry_desktop->appinfo = g_desktop_app_info_new_from_filename (entry->path);
256 if (!entry_desktop->appinfo ||
257 !g_app_info_get_name (G_APP_INFO (entry_desktop->appinfo)) ||
258 !g_app_info_get_executable (G_APP_INFO (entry_desktop->appinfo)))
259 {
260 menu_verbose ("Failed to load \"%s\"\n", entry->path);
261 return FALSE;
262 }
263
264 categories_str = g_desktop_app_info_get_categories (entry_desktop->appinfo);
265 if (categories_str)
266 {
267 char **categories;
268 int i;
269
270 categories = g_strsplit (categories_str, ";", -1);
271 entry_desktop->categories = g_new0 (GQuark, g_strv_length (categories) + 1);
272
273 for (i = 0; categories[i]; i++)
274 entry_desktop->categories[i] = g_quark_from_string (categories[i]);
275
276 g_strfreev (categories);
277 }
278
279 return TRUE;
280 }
281 else if (entry->type == DESKTOP_ENTRY_DIRECTORY)
282 {
283 GKeyFile *key_file = NULL;
284 GError *error = NULL;
285 gboolean retval = FALSE;
286
287 key_file = g_key_file_new ();
288
289 if (!g_key_file_load_from_file (key_file, entry->path, 0, &error))
290 goto out;
291
292 if (!desktop_entry_load_directory (entry, key_file, &error))
293 goto out;
294
295 retval = TRUE;
296
297 out:
298 g_key_file_free (key_file);
299
300 if (!retval)
301 {
302 if (error)
303 {
304 menu_verbose ("Failed to load \"%s\": %s\n", entry->path, error->message);
305 g_error_free (error);
306 }
307 else
308 {
309 menu_verbose ("Failed to load \"%s\"\n", entry->path);
310 }
311 }
312
313 return retval;
314 }
315 else
316 g_assert_not_reached ();
317
318 return FALSE;
319}

321DesktopEntry* desktop_entry_new(const char* path)
322{
323 DesktopEntryType type;
324 DesktopEntry *retval;
325
326 menu_verbose ("Loading desktop entry \"%s\"\n", path);
327
328 if (g_str_has_suffix (path, ".desktop"))
329 {
330 type = DESKTOP_ENTRY_DESKTOP;
331 retval = (DesktopEntry*)g_new0 (DesktopEntryDesktop, 1);
332 }
333 else if (g_str_has_suffix (path, ".directory"))
334 {
335 type = DESKTOP_ENTRY_DIRECTORY;
336 retval = (DesktopEntry*)g_new0 (DesktopEntryDirectory, 1);
337 }
338 else
339 {
340 menu_verbose ("Unknown desktop entry suffix in \"%s\"\n",
341 path);
342 return NULL;
343 }
344
345 retval->refcount = 1;
346 retval->type = type;
347 retval->path = g_strdup (path);
348 retval->basename = unix_basename_from_path (retval->path);
349
350 if (!desktop_entry_load (retval))
351 {
352 desktop_entry_unref (retval);
353 return NULL;
354 }
355
356 return retval;
357}

359DesktopEntry* desktop_entry_reload(DesktopEntry* entry)
360{
361 g_return_val_if_fail (entry != NULL, NULL);
362
363 menu_verbose ("Re-loading desktop entry \"%s\"\n", entry->path);
364
365 if (entry->type == DESKTOP_ENTRY_DESKTOP)
366 {
367 DesktopEntryDesktop *entry_desktop = (DesktopEntryDesktop *) entry;
368
369 g_object_unref (entry_desktop->appinfo);
370 entry_desktop->appinfo = NULL;
371
372 g_free (entry_desktop->categories);
373 entry_desktop->categories = NULL;
374 }
375 else if (entry->type == DESKTOP_ENTRY_DIRECTORY)
376 {
377 DesktopEntryDirectory *entry_directory = (DesktopEntryDirectory*) entry;
378
379 g_free (entry_directory->name);
380 entry_directory->name = NULL;
381
382 g_free (entry_directory->comment);
383 entry_directory->comment = NULL;
384
385 g_object_unref (entry_directory->icon);
386 entry_directory->icon = NULL;
387 }
388 else
389 g_assert_not_reached ();
390
391 if (!desktop_entry_load (entry))
392 {
393 desktop_entry_unref (entry);
394 return NULL;
395 }
396
397 return entry;
398}

410DesktopEntry* desktop_entry_copy(DesktopEntry* entry)
411{
412 DesktopEntry *retval;
413
414 menu_verbose ("Copying desktop entry \"%s\"\n",
415 entry->basename);
416
417 if (entry->type == DESKTOP_ENTRY_DESKTOP)
418 retval = (DesktopEntry*)g_new0 (DesktopEntryDesktop, 1);
419 else if (entry->type == DESKTOP_ENTRY_DIRECTORY)
420 retval = (DesktopEntry*)g_new0 (DesktopEntryDirectory, 1);
421 else
422 g_assert_not_reached ();
423
424 retval->refcount = 1;
425 retval->type = entry->type;
426 retval->path = g_strdup (entry->path);
427 retval->basename = unix_basename_from_path (retval->path);
428
429 if (retval->type == DESKTOP_ENTRY_DESKTOP)
430 {
431 DesktopEntryDesktop *desktop_entry = (DesktopEntryDesktop*) entry;
432 DesktopEntryDesktop *retval_desktop_entry = (DesktopEntryDesktop*) retval;
433
434 retval_desktop_entry->appinfo = g_object_ref (desktop_entry->appinfo);
435
436 if (desktop_entry->categories != NULL)
437 {
438 gsize i;
439
440 i = 0;
441 for (; desktop_entry->categories[i]; i++);
442
443 retval_desktop_entry->categories = g_new0 (GQuark, i + 1);
444
445 i = 0;
446 for (; desktop_entry->categories[i]; i++)
447 retval_desktop_entry->categories[i] = desktop_entry->categories[i];
448 }
449 else
450 retval_desktop_entry->categories = NULL;
451 }
452 else if (entry->type == DESKTOP_ENTRY_DIRECTORY)
453 {
454 DesktopEntryDirectory *entry_directory = (DesktopEntryDirectory*)entry;
455 DesktopEntryDirectory *retval_directory = (DesktopEntryDirectory*)retval;
456
457 retval_directory->name = g_strdup (entry_directory->name);
458 retval_directory->comment = g_strdup (entry_directory->comment);
459 retval_directory->icon = g_object_ref (entry_directory->icon);
460 retval_directory->nodisplay = entry_directory->nodisplay;
461 retval_directory->hidden = entry_directory->hidden;
462 retval_directory->showin = entry_directory->showin;
463 }
464
465 return retval;
466}

468void desktop_entry_unref(DesktopEntry* entry)
469{
470 g_return_if_fail (entry != NULL);
471 g_return_if_fail (entry->refcount > 0);
472
473 entry->refcount -= 1;
474 if (entry->refcount != 0)
475 return;
476
477 g_free (entry->path);
478 entry->path = NULL;
479
480 if (entry->type == DESKTOP_ENTRY_DESKTOP)
481 {
482 DesktopEntryDesktop *desktop_entry = (DesktopEntryDesktop*) entry;
483 g_free (desktop_entry->categories);
484 if (desktop_entry->appinfo)
485 g_object_unref (desktop_entry->appinfo);
486 }
487 else if (entry->type == DESKTOP_ENTRY_DIRECTORY)
488 {
489 DesktopEntryDirectory *entry_directory = (DesktopEntryDirectory*) entry;
490
491 g_free (entry_directory->name);
492 entry_directory->name = NULL;
493
494 g_free (entry_directory->comment);
495 entry_directory->comment = NULL;
496
497 if (entry_directory->icon != NULL)
498 {
499 g_object_unref (entry_directory->icon);
500 entry_directory->icon = NULL;
501 }
502 }
503 else
504 g_assert_not_reached ();
505
506 g_free (entry);
507}

600gboolean desktop_entry_has_category(DesktopEntry* entry, const char* category)
601{
602 GQuark quark;
603 int i;
604 DesktopEntryDesktop *desktop_entry;
605
606 if (entry->type != DESKTOP_ENTRY_DESKTOP)
607 return FALSE;
608
609 desktop_entry = (DesktopEntryDesktop*) entry;
610
611 if (desktop_entry->categories == NULL)
612 return FALSE;
613
614 if (!(quark = g_quark_try_string (category)))
615 return FALSE;
616
617 for (i = 0; desktop_entry->categories[i]; i++)
618 {
619 if (quark == desktop_entry->categories[i])
620 return TRUE;
621 }
622
623 return FALSE;
624}

706void desktop_entry_set_add_entry(DesktopEntrySet* set, DesktopEntry* entry, const char* file_id)
707{
708 menu_verbose (" Adding to set %p entry %s\n", set, file_id);
709
710 if (set->hash == NULL)
711 {
712 set->hash = g_hash_table_new_full (g_str_hash,
713 g_str_equal,
714 g_free,
715 (GDestroyNotify) desktop_entry_unref);
716 }
717
718 g_hash_table_replace (set->hash,
719 g_strdup (file_id),
720 desktop_entry_ref (entry));
721}

731typedef struct {

734} EntryHashForeachData;

736static void entry_hash_foreach(const char* file_id, DesktopEntry* entry, EntryHashForeachData* fd)
737{
738 fd->func(file_id, entry, fd->user_data);
739}

741void desktop_entry_set_foreach(DesktopEntrySet* set, DesktopEntrySetForeachFunc func, gpointer user_data)
742{
743 g_return_if_fail (set != NULL);
744 g_return_if_fail (func != NULL);
745
746 if (set->hash != NULL)
747 {
748 EntryHashForeachData fd;
749
750 fd.func = func;
751 fd.user_data = user_data;
752
753 g_hash_table_foreach (set->hash,
754 (GHFunc) entry_hash_foreach,
755 &fd);
756 }
757}

779static void union_foreach(const char* file_id, DesktopEntry* entry, DesktopEntrySet* set)
780{
781 /* we are iterating over "with" adding anything not
782 * already in "set". We unconditionally overwrite
783 * the stuff in "set" because we can assume
784 * two entries with the same name are equivalent.
785 */
786 desktop_entry_set_add_entry(set, entry, file_id);
787}
788
789void desktop_entry_set_union(DesktopEntrySet* set, DesktopEntrySet* with)
790{
791 menu_verbose (" Union of %p and %p\n", set, with);
792
793 if (desktop_entry_set_get_count (with) == 0)
794 return; /* A fast simple case */
795
796 g_hash_table_foreach (with->hash,
797 (GHFunc) union_foreach,
798 set);
799}

801typedef struct {
802 DesktopEntrySet *set;
803 DesktopEntrySet *with;
804} IntersectData;
805
806static gboolean intersect_foreach_remove(const char* file_id, DesktopEntry* entry, IntersectData* id)
807{
808 /* Remove everything in "set" which is not in "with" */
809
810 if (g_hash_table_lookup (id->with->hash, file_id) != NULL)
811 return FALSE;
812
813 menu_verbose (" Removing from %p entry %s\n", id->set, file_id);
814
815 return TRUE; /* return TRUE to remove */
816}
817
818void desktop_entry_set_intersection(DesktopEntrySet* set, DesktopEntrySet* with)
819{
820 IntersectData id;
821
822 menu_verbose (" Intersection of %p and %p\n", set, with);
823
824 if (desktop_entry_set_get_count (set) == 0 ||
825 desktop_entry_set_get_count (with) == 0)
826 {
827 /* A fast simple case */
828 desktop_entry_set_clear (set);
829 return;
830 }
831
832 id.set = set;
833 id.with = with;
834
835 g_hash_table_foreach_remove (set->hash,
836 (GHRFunc) intersect_foreach_remove,
