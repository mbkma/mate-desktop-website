47/* Sets the idle percent limit, i.e. how hard the computer can work
48 while considered "at idle" */
49#define GPM_IDLE_CPU_LIMIT 5
50#define GPM_IDLE_IDLETIME_ID 1
51
52struct GpmIdlePrivate
53{
54 EggIdletime *idletime;
55 GpmLoad *load;
56 GpmSession *session;
57 GpmIdleMode mode;
58 guint timeout_dim; /* in seconds */
59 guint timeout_blank; /* in seconds */
60 guint timeout_sleep; /* in seconds */
61 guint timeout_blank_id;
62 guint timeout_sleep_id;
63 gboolean x_idle;
64 gboolean check_type_cpu;
65};

80static const gchar *
81gpm_idle_mode_to_string (GpmIdleMode mode)
82{
83 if (mode == GPM_IDLE_MODE_NORMAL)
84 return "normal";
85 if (mode == GPM_IDLE_MODE_DIM)
86 return "dim";
87 if (mode == GPM_IDLE_MODE_BLANK)
88 return "blank";
89 if (mode == GPM_IDLE_MODE_SLEEP)
90 return "sleep";
91 return "unknown";
92}

98static void
99gpm_idle_set_mode (GpmIdle *idle, GpmIdleMode mode)
100{
101 g_return_if_fail (GPM_IS_IDLE (idle));
102
103 if (mode != idle->priv->mode) {
104 idle->priv->mode = mode;
105 g_debug ("Doing a state transition: %s", gpm_idle_mode_to_string (mode));
106 g_signal_emit (idle, signals [IDLE_CHANGED], 0, mode);
107 }
108}

136static gboolean
137gpm_idle_blank_cb (GpmIdle *idle)
138{
139 if (idle->priv->mode > GPM_IDLE_MODE_BLANK) {
140 g_debug ("ignoring current mode %s", gpm_idle_mode_to_string (idle->priv->mode));
141 return FALSE;
142 }
143 gpm_idle_set_mode (idle, GPM_IDLE_MODE_BLANK);
144 return FALSE;
145}

150static gboolean
151gpm_idle_sleep_cb (GpmIdle *idle)
152{
153 gdouble load;
154 gboolean ret = FALSE;
155
156 /* get our computed load value */
157 if (idle->priv->check_type_cpu) {
158 load = gpm_load_get_current (idle->priv->load);
159 if (load > GPM_IDLE_CPU_LIMIT) {
160 /* check if system is "idle" enough */
161 g_debug ("Detected that the CPU is busy");
162 ret = TRUE;
163 goto out;
164 }
165 }
166 gpm_idle_set_mode (idle, GPM_IDLE_MODE_SLEEP);
167out:
168 return ret;
169}

174static void
175gpm_idle_evaluate (GpmIdle *idle)
176{
177 gboolean is_idle;
178 gboolean is_idle_inhibited;
179 gboolean is_suspend_inhibited;
180
181 is_idle = gpm_session_get_idle (idle->priv->session);
182 is_idle_inhibited = gpm_session_get_idle_inhibited (idle->priv->session);
183 is_suspend_inhibited = gpm_session_get_suspend_inhibited (idle->priv->session);
184 g_debug ("session_idle=%i, idle_inhibited=%i, suspend_inhibited=%i, x_idle=%i", is_idle, is_idle_inhibited, is_suspend_inhibited, idle->priv->x_idle);
185
186 /* check we are really idle */
187 if (!idle->priv->x_idle) {
188 gpm_idle_set_mode (idle, GPM_IDLE_MODE_NORMAL);
189 g_debug ("X not idle");
190 if (idle->priv->timeout_blank_id != 0) {
191 g_source_remove (idle->priv->timeout_blank_id);
192 idle->priv->timeout_blank_id = 0;
193 }
194 if (idle->priv->timeout_sleep_id != 0) {
195 g_source_remove (idle->priv->timeout_sleep_id);
196 idle->priv->timeout_sleep_id = 0;
197 }
198 goto out;
199 }
200
201 /* are we inhibited from going idle */
202 if (is_idle_inhibited) {
203 g_debug ("inhibited, so using normal state");
204 gpm_idle_set_mode (idle, GPM_IDLE_MODE_NORMAL);
205 if (idle->priv->timeout_blank_id != 0) {
206 g_source_remove (idle->priv->timeout_blank_id);
207 idle->priv->timeout_blank_id = 0;
208 }
209 if (idle->priv->timeout_sleep_id != 0) {
210 g_source_remove (idle->priv->timeout_sleep_id);
211 idle->priv->timeout_sleep_id = 0;
212 }
213 goto out;
214 }
215
216 /* normal to dim */
217 if (idle->priv->mode == GPM_IDLE_MODE_NORMAL) {
218 g_debug ("normal to dim");
219 gpm_idle_set_mode (idle, GPM_IDLE_MODE_DIM);
220 }
221
222 /* set up blank callback even when session is not idle,
223 * but only if we actually want to blank. */
224 if (idle->priv->timeout_blank_id == 0 &&
225 idle->priv->timeout_blank != 0) {
226 g_debug ("setting up blank callback for %us", idle->priv->timeout_blank);
227 idle->priv->timeout_blank_id = g_timeout_add_seconds (idle->priv->timeout_blank,
228 (GSourceFunc) gpm_idle_blank_cb, idle);
229 g_source_set_name_by_id (idle->priv->timeout_blank_id, "[GpmIdle] blank");
230 }
231
232 /* are we inhibited from sleeping */
233 if (is_suspend_inhibited) {
234 g_debug ("suspend inhibited");
235 if (idle->priv->timeout_sleep_id != 0) {
236 g_source_remove (idle->priv->timeout_sleep_id);
237 idle->priv->timeout_sleep_id = 0;
238 }
239 } else if (is_idle) {
240 /* only do the sleep timeout when the session is idle and we aren't inhibited from sleeping */
241 if (idle->priv->timeout_sleep_id == 0 &&
242 idle->priv->timeout_sleep != 0) {
243 g_debug ("setting up sleep callback %us", idle->priv->timeout_sleep);
244 idle->priv->timeout_sleep_id = g_timeout_add_seconds (idle->priv->timeout_sleep,
245 (GSourceFunc) gpm_idle_sleep_cb, idle);
246 g_source_set_name_by_id (idle->priv->timeout_sleep_id, "[GpmIdle] sleep");
247 }
248 }
249out:
250 return;
251}

268static guint
269gpm_idle_adjust_timeout_dim (guint idle_time, guint timeout)
270{
271 /* allow 2 sec margin for messaging delay. */
272 idle_time += 2;
273
274 /* Double timeout until it's larger than current idle time.
275 * Give up for ultra slow machines. (86400 sec = 24 hours) */
276 while (timeout < idle_time && timeout < 86400 && timeout > 0) {
277 timeout *= 2;
278 }
279 return timeout;
280}

286gboolean
287gpm_idle_set_timeout_dim (GpmIdle *idle, guint timeout)
288{
289 gint64 idle_time_in_msec;
290 guint timeout_adjusted;
291
292 g_return_val_if_fail (GPM_IS_IDLE (idle), FALSE);
293
294 idle_time_in_msec = egg_idletime_get_time (idle->priv->idletime);
295 timeout_adjusted = gpm_idle_adjust_timeout_dim (idle_time_in_msec / 1000, timeout);
296 g_debug ("Current idle time=%lldms, timeout was %us, becomes %us after adjustment",
297 (long long int)idle_time_in_msec, timeout, timeout_adjusted);
298 timeout = timeout_adjusted;
299
300 g_debug ("Setting dim idle timeout: %us", timeout);
301 if (idle->priv->timeout_dim != timeout) {
302 idle->priv->timeout_dim = timeout;
303
304 if (timeout > 0)
305 egg_idletime_alarm_set (idle->priv->idletime, GPM_IDLE_IDLETIME_ID, timeout * 1000);
306 else
307 egg_idletime_alarm_remove (idle->priv->idletime, GPM_IDLE_IDLETIME_ID);
308 }
309 return TRUE;
310}

403static void
404gpm_idle_finalize (GObject *object)
405{
406 GpmIdle *idle;
407
408 g_return_if_fail (object != NULL);
409 g_return_if_fail (GPM_IS_IDLE (object));
410
411 idle = GPM_IDLE (object);
412
413 g_return_if_fail (idle->priv != NULL);
414
415 if (idle->priv->timeout_blank_id != 0) {
416 g_source_remove (idle->priv->timeout_blank_id);
417 idle->priv->timeout_blank_id = 0;
418 }
419
420 if (idle->priv->timeout_sleep_id != 0) {
421 g_source_remove (idle->priv->timeout_sleep_id);
422 idle->priv->timeout_sleep_id = 0;
423 }
424
425 g_object_unref (idle->priv->load);
426 g_object_unref (idle->priv->session);
427
428 egg_idletime_alarm_remove (idle->priv->idletime, GPM_IDLE_IDLETIME_ID);
429 g_object_unref (idle->priv->idletime);
430
431 G_OBJECT_CLASS (gpm_idle_parent_class)->finalize (object);
432}

438static void
439gpm_idle_class_init (GpmIdleClass *klass)
440{
441 GObjectClass *object_class = G_OBJECT_CLASS (klass);
442
443 object_class->finalize = gpm_idle_finalize;
444
445 signals [IDLE_CHANGED] =
446 g_signal_new ("idle-changed",
447 G_TYPE_FROM_CLASS (object_class),
448 G_SIGNAL_RUN_LAST,
449 G_STRUCT_OFFSET (GpmIdleClass, idle_changed),
450 NULL, NULL, g_cclosure_marshal_VOID__INT,
451 G_TYPE_NONE, 1, G_TYPE_INT);
452}

461static void
462gpm_idle_init (GpmIdle *idle)
463{
464 idle->priv = gpm_idle_get_instance_private (idle);
465
466 idle->priv->timeout_dim = G_MAXUINT;
467 idle->priv->timeout_blank = G_MAXUINT;
468 idle->priv->timeout_sleep = G_MAXUINT;
469 idle->priv->timeout_blank_id = 0;
470 idle->priv->timeout_sleep_id = 0;
471 idle->priv->x_idle = FALSE;
472 idle->priv->load = gpm_load_new ();
473 idle->priv->session = gpm_session_new ();
474 g_signal_connect (idle->priv->session, "idle-changed", G_CALLBACK (gpm_idle_session_idle_changed_cb), idle);
475 g_signal_connect (idle->priv->session, "inhibited-changed", G_CALLBACK (gpm_idle_session_inhibited_changed_cb), idle);
476
477 idle->priv->idletime = egg_idletime_new ();
478 g_signal_connect (idle->priv->idletime, "reset", G_CALLBACK (gpm_idle_idletime_reset_cb), idle);
479 g_signal_connect (idle->priv->idletime, "alarm-expired", G_CALLBACK (gpm_idle_idletime_alarm_expired_cb), idle);
480
481 gpm_idle_evaluate (idle);
482}
