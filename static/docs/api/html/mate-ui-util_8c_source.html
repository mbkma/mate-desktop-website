61gboolean
62mate_ui_util_show_help(GdkScreen *screen,
63 const gchar *doc_id,
64 const gchar *link_id,
65 GError **error)
66{
67 g_return_val_if_fail(doc_id != NULL, FALSE);
68
69 gchar *uri;
70 if (link_id != NULL)
71 uri = g_strdup_printf("help:%s/%s", doc_id, link_id);
72 else
73 uri = g_strdup_printf("help:%s", doc_id);
74
75 guint32 timestamp = gtk_get_current_event_time();

77 g_free(uri);

189gchar *
190mate_ui_util_icon_name_for_mimetype(const gchar *mimetype)
191{
192 g_return_val_if_fail(mimetype != NULL, NULL);
193
194 GIcon *icon = g_content_type_get_icon(mimetype);
195 if (icon == NULL)
196 return g_strdup("application-octet-stream");
197
198 gchar *icon_name = NULL;
199 if (G_IS_THEMED_ICON(icon))
200 {
201 const gchar * const *names = g_themed_icon_get_names(G_THEMED_ICON(icon));
202 if (names != NULL && names[0] != NULL)
203 icon_name = g_strdup(names[0]);
204 }
205
206 g_object_unref(icon);
207
208 if (icon_name == NULL)
209 icon_name = g_strdup("application-octet-stream");
210
211 return icon_name;
212}

319GtkCssProvider *
320mate_ui_util_load_css_file(const gchar *filename,
321 guint priority,
322 GError **error)
323{
324 g_return_val_if_fail(filename != NULL, NULL);
325
326 GtkCssProvider *provider = gtk_css_provider_new();
327 GFile *file = g_file_new_for_path(filename);
328
329 if (!gtk_css_provider_load_from_file(provider, file, error))
330 {
331 g_object_unref(file);
332 g_object_unref(provider);
333 return NULL;
334 }
335
336 g_object_unref(file);
337
338 GdkScreen *screen = gdk_screen_get_default();
339 gtk_style_context_add_provider_for_screen(screen,
340 GTK_STYLE_PROVIDER(provider),
341 priority);
342
343 return provider;
344}

358{

363 GError *error = NULL;
364
365 GdkPixbuf *pixbuf = gtk_icon_theme_load_icon(theme, icon_name, size,

367 if (error != NULL)
368 {
369 g_warning("Failed to load icon '%s': %s", icon_name, error->message);
370 g_error_free(error);
371 }
372
373 return pixbuf;
374}

458gchar *
459mate_ui_util_format_time(guint seconds)
460{
461 if (seconds < 60)
462 return g_strdup_printf("%u sec", seconds);
463 else if (seconds < 3600)
464 return g_strdup_printf("%u min %u sec", seconds / 60, seconds % 60);
465 else
466 {
467 guint hours = seconds / 3600;
468 guint mins = (seconds % 3600) / 60;
469 guint secs = seconds % 60;
470 return g_strdup_printf("%u hr %u min %u sec", hours, mins, secs);
471 }
472}

483gboolean
484mate_ui_util_spawn_command_async(const gchar *command,
485 GError **error)
486{
487 g_return_val_if_fail(command != NULL, FALSE);
488
489 gint argc;
490 gchar **argv;
491
492 if (!g_shell_parse_argv(command, &argc, &argv, error))
493 return FALSE;
494
495 gboolean result = g_spawn_async(NULL, argv, NULL,
496 G_SPAWN_SEARCH_PATH | G_SPAWN_STDOUT_TO_DEV_NULL | G_SPAWN_STDERR_TO_DEV_NULL,
497 NULL, NULL, NULL, error);
498
499 g_strfreev(argv);
500 return result;
501}
