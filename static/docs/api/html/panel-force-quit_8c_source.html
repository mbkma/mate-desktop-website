137static gboolean
138wm_state_set (Display *xdisplay,
139 Window window)
140{
141 GdkDisplay *display;
142 gulong nitems;
143 gulong bytes_after;
144 gulong *prop;
145 Atom ret_type = None;
146 int ret_format;
147 int result;
148
149 display = gdk_display_get_default ();
150 gdk_x11_display_error_trap_push (display);
151 result = XGetWindowProperty (xdisplay, window, wm_state_atom,
152 0, G_MAXLONG, False, wm_state_atom,
153 &ret_type, &ret_format, &nitems,
154 &bytes_after, (gpointer) &prop);
155
156 if (gdk_x11_display_error_trap_pop (display))
157 return FALSE;
158
159 if (result != Success)
160 return FALSE;
161
162 XFree (prop);
163
164 if (ret_type != wm_state_atom)
165 return FALSE;
166
167 return TRUE;
168}

170static Window
171find_managed_window (Display *xdisplay,
172 Window window)
173{
174 GdkDisplay *display;
175 Window root;
176 Window parent;
177 Window *kids = NULL;
178 Window retval;
179 guint nkids;
180 int i, result;
181
182 if (wm_state_set (xdisplay, window))
183 return window;
184
185 display = gdk_display_get_default ();
186 gdk_x11_display_error_trap_push (display);
187 result = XQueryTree (xdisplay, window, &root, &parent, &kids, &nkids);
188 if (gdk_x11_display_error_trap_pop (display) || !result)
189 return None;
190
191 retval = None;
192
193 for (i = 0; i < nkids; i++) {
194 if (wm_state_set (xdisplay, kids [i])) {
195 retval = kids [i];
196 break;
197 }
198
199 retval = find_managed_window (xdisplay, kids [i]);
200 if (retval != None)
201 break;
202 }
203
204 if (kids)
205 XFree (kids);
206
207 return retval;
208}

267static void
268handle_button_press_event (GtkWidget *popup,
269 Display *display,
270 Window subwindow)
271{
272 Window window;
273
274 remove_popup (popup);
275
276 if (subwindow == None)
277 return;
278
279 if (wm_state_atom == None)
280 wm_state_atom = XInternAtom (display, "WM_STATE", FALSE);
281
282 window = find_managed_window (display, subwindow);
283
284 if (window != None) {
285 if (!gdk_x11_window_lookup_for_display (gdk_x11_lookup_xdisplay (display), window))
286 kill_window_question ((gpointer) window);
287 }

290static GdkFilterReturn
291popup_filter (GdkXEvent *gdk_xevent,
292 GdkEvent *event,
293 GtkWidget *popup)
294{
295 XEvent *xevent = (XEvent *) gdk_xevent;
296 XIEvent *xiev;
297 XIDeviceEvent *xidev;
298
299 switch (xevent->type) {
300 case ButtonPress:
301 handle_button_press_event (popup, xevent->xbutton.display, xevent->xbutton.subwindow);
302 return GDK_FILTER_REMOVE;
303 case KeyPress:
304 if (xevent->xkey.keycode == XKeysymToKeycode (xevent->xany.display, XK_Escape)) {
305 remove_popup (popup);
306 return GDK_FILTER_REMOVE;
307 }
308 break;
309 case GenericEvent:
310 xiev = (XIEvent *) xevent->xcookie.data;
311 xidev = (XIDeviceEvent *) xiev;
312 switch (xiev->evtype) {
313 case XI_KeyPress:
314 if (xidev->detail == XKeysymToKeycode (xevent->xany.display, XK_Escape)) {
315 remove_popup (popup);
316 return GDK_FILTER_REMOVE;
317 }
318 break;
319 case XI_ButtonPress:
320 handle_button_press_event (popup, xidev->display, xidev->child);
321 return GDK_FILTER_REMOVE;
322 }
323 break;
324 default:
325 break;
326 }
327
328 return GDK_FILTER_CONTINUE;
329}

339void
340panel_force_quit (GdkScreen *screen,
341 guint time)
342{
343 GdkGrabStatus status;
344 GdkCursor *cross;
345 GdkSeatCapabilities caps;
346 GtkWidget *popup;
347 GdkWindow *root;
348 GdkDisplay *display;
349 GdkSeat *seat;
350
351 g_return_if_fail (GDK_IS_X11_DISPLAY (gdk_screen_get_display (screen)));
352
353 popup = display_popup_window (screen);
354
355 root = gdk_screen_get_root_window (screen);
356
357 gdk_window_add_filter (root, (GdkFilterFunc) popup_filter, popup);
358 cross = gdk_cursor_new_for_display (gdk_display_get_default (),
359 GDK_CROSS);
360 caps = GDK_SEAT_CAPABILITY_POINTER | GDK_SEAT_CAPABILITY_KEYBOARD;
361 display = gdk_window_get_display (root);
362 seat = gdk_display_get_default_seat (display);
363
364 status = gdk_seat_grab (seat, root,
365 caps, FALSE,
366 cross, NULL,
367 prepare_root_window,
368 NULL);
369
370 g_object_unref (cross);
371
372 if (status != GDK_GRAB_SUCCESS) {
373 g_warning ("Pointer grab failed\n");
374 remove_popup (popup);
375 return;
376 }
377
378 gdk_display_flush (display);
379}
