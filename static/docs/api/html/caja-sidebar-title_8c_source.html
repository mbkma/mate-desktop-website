117static void
118caja_sidebar_title_init (CajaSidebarTitle *sidebar_title)
119{
120 sidebar_title->details = caja_sidebar_title_get_instance_private (sidebar_title);
121
122 gtk_orientable_set_orientation (GTK_ORIENTABLE (sidebar_title), GTK_ORIENTATION_VERTICAL);
123
124 /* Create the icon */
125 sidebar_title->details->icon = gtk_image_new ();
126 gtk_box_pack_start (GTK_BOX (sidebar_title), sidebar_title->details->icon, 0, 0, 0);
127 gtk_widget_show (sidebar_title->details->icon);
128
129 /* Create the title label */
130 sidebar_title->details->title_label = sidebar_title_create_title_label ();
131 gtk_box_pack_start (GTK_BOX (sidebar_title), sidebar_title->details->title_label, 0, 0, 0);
132 gtk_widget_show (sidebar_title->details->title_label);
133
134 /* Create the more info label */
135 sidebar_title->details->more_info_label = sidebar_title_create_more_info_label ();
136 gtk_box_pack_start (GTK_BOX (sidebar_title), sidebar_title->details->more_info_label, 0, 0, 0);
137 gtk_widget_show (sidebar_title->details->more_info_label);
138
139 sidebar_title->details->emblem_box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);
140 gtk_widget_show (sidebar_title->details->emblem_box);
141 gtk_box_pack_start (GTK_BOX (sidebar_title), sidebar_title->details->emblem_box, 0, 0, 0);
142
143 sidebar_title->details->best_icon_size = get_best_icon_size (sidebar_title);
144 /* Keep track of changes in graphics trade offs */
145 update_all (sidebar_title);
146
147 /* initialize the label colors & fonts */
148 style_updated (GTK_WIDGET (sidebar_title));
149
150 g_signal_connect_swapped (caja_preferences,
151 "changed::" CAJA_PREFERENCES_SHOW_DIRECTORY_ITEM_COUNTS,
152 G_CALLBACK(update_more_info),
153 sidebar_title);
154}

156/* destroy by throwing away private storage */
157static void
158release_file (CajaSidebarTitle *sidebar_title)
159{
160 if (sidebar_title->details->file_changed_connection != 0)
161 {
162 if (g_signal_handler_is_connected(G_OBJECT (sidebar_title->details->file),
163 sidebar_title->details->file_changed_connection)){
164 g_signal_handler_disconnect (sidebar_title->details->file,
165 sidebar_title->details->file_changed_connection);
166 }
167 sidebar_title->details->file_changed_connection = 0;
168 }
169
170 if (sidebar_title->details->file != NULL)
171 {
172 caja_file_monitor_remove (sidebar_title->details->file, sidebar_title);
173 caja_file_unref (sidebar_title->details->file);
174 sidebar_title->details->file = NULL;
175 }
176}

237void
238caja_sidebar_title_select_text_color (CajaSidebarTitle *sidebar_title,
239 EelBackground *background)
240{
241 GdkRGBA *light_info_color, *dark_info_color;
242 GtkStyleContext *style;
243 GdkRGBA color;
244 GdkRGBA *c;
245
246 g_assert (CAJA_IS_SIDEBAR_TITLE (sidebar_title));
247 g_return_if_fail (gtk_widget_get_realized (GTK_WIDGET (sidebar_title)));
248
249 /* read the info colors from the current theme; use a reasonable default if undefined */
250 style = gtk_widget_get_style_context (GTK_WIDGET (sidebar_title));
251 gtk_style_context_get_style (style,
252 "light_info_color", &light_info_color,
253 "dark_info_color", &dark_info_color,
254 NULL);
255
256 if (!light_info_color)
257 {
258 light_info_color = g_malloc (sizeof (GdkRGBA));
259 gdk_rgba_parse (light_info_color, DEFAULT_LIGHT_INFO_COLOR);
260 }
261
262 if (!dark_info_color)
263 {
264 dark_info_color = g_malloc (sizeof (GdkRGBA));
265 gdk_rgba_parse (dark_info_color, DEFAULT_DARK_INFO_COLOR);
266 }
267
268 gtk_style_context_get_color (style, GTK_STATE_FLAG_SELECTED, &color);
269 setup_gc_with_fg (sidebar_title, LABEL_COLOR_HIGHLIGHT, &color);
270
271 gtk_style_context_get_color (style, GTK_STATE_FLAG_ACTIVE, &color);
272 setup_gc_with_fg (sidebar_title, LABEL_COLOR_ACTIVE, &color);
273
274 gtk_style_context_get_color (style, GTK_STATE_FLAG_PRELIGHT, &color);
275 setup_gc_with_fg (sidebar_title, LABEL_COLOR_PRELIGHT, &color);
276
277 gtk_style_context_get (style, GTK_STATE_FLAG_SELECTED,
278 GTK_STYLE_PROPERTY_BACKGROUND_COLOR,
279 &c, NULL);
280 color = *c;
281
282 setup_gc_with_fg (sidebar_title, LABEL_INFO_COLOR_HIGHLIGHT,
283 eel_gdk_rgba_is_dark (&color) ? light_info_color : dark_info_color);
284
285 gtk_style_context_get (style, GTK_STATE_FLAG_ACTIVE,
286 GTK_STYLE_PROPERTY_BACKGROUND_COLOR,
287 &c, NULL);
288 color = *c;
289
290 setup_gc_with_fg (sidebar_title, LABEL_INFO_COLOR_ACTIVE,
291 eel_gdk_rgba_is_dark (&color) ? light_info_color : dark_info_color);
292
293 /* If EelBackground is not set in the widget, we can safely
294 * use the foreground color from the theme, because it will
295 * always be displayed against the gtk background */
296 if (!eel_background_is_set(background))
297 {
298 gtk_style_context_get_color (style, GTK_STATE_FLAG_NORMAL, &color);
299 setup_gc_with_fg (sidebar_title, LABEL_COLOR, &color);
300
301 gtk_style_context_get (style, GTK_STATE_FLAG_NORMAL,
302 GTK_STYLE_PROPERTY_BACKGROUND_COLOR,
303 &color, NULL);
304 color = *c;
305
306 setup_gc_with_fg (sidebar_title, LABEL_INFO_COLOR,
307 eel_gdk_rgba_is_dark (&color) ?
308 light_info_color : dark_info_color);
309 }
310 else if (eel_background_is_dark (background))
311 {
312 GdkRGBA tmp;
313
314 gdk_rgba_parse (&tmp, "EFEFEF");
315 setup_gc_with_fg (sidebar_title, LABEL_COLOR, &tmp);
316 setup_gc_with_fg (sidebar_title, LABEL_INFO_COLOR, light_info_color);
317 }
318 else /* converse */
319 {
320 GdkRGBA tmp;
321
322 gdk_rgba_parse (&tmp, "000000");
323 setup_gc_with_fg (sidebar_title, LABEL_COLOR, &tmp);
324 setup_gc_with_fg (sidebar_title, LABEL_INFO_COLOR, dark_info_color);
325 }
326
327 gdk_rgba_free (c);
328 gdk_rgba_free (dark_info_color);
329 gdk_rgba_free (light_info_color);
330}

361/* set up the icon image */
362static void
363update_icon (CajaSidebarTitle *sidebar_title)
364{
365 cairo_surface_t *surface;
366 char *icon_name;
367 gboolean leave_surface_unchanged;
368 gint icon_scale;
369
370 leave_surface_unchanged = FALSE;
371
372 /* see if the current content view is specifying an icon */
373 icon_name = get_property_from_component (sidebar_title, "icon_name");
374 icon_scale = gtk_widget_get_scale_factor (GTK_WIDGET (sidebar_title));
375
376 surface = NULL;
377
378 if (icon_name != NULL && icon_name[0] != '\0')
379 {
380 CajaIconInfo *info;
381
382 info = caja_icon_info_lookup_from_name (icon_name, CAJA_ICON_SIZE_LARGE, icon_scale);
383 surface = caja_icon_info_get_surface_at_size (info, CAJA_ICON_SIZE_LARGE);
384 g_object_unref (info);
385 }
386 else if (sidebar_title->details->file != NULL &&
387 caja_file_check_if_ready (sidebar_title->details->file,
388 CAJA_FILE_ATTRIBUTES_FOR_ICON))
389 {
390 surface = caja_file_get_icon_surface (sidebar_title->details->file,
391 sidebar_title->details->best_icon_size * icon_scale,
392 TRUE,
393 icon_scale,
394 CAJA_FILE_ICON_FLAGS_USE_THUMBNAILS |
395 CAJA_FILE_ICON_FLAGS_USE_MOUNT_ICON_AS_EMBLEM);
396 }
397 else if (sidebar_title->details->determined_icon)
398 {
399 /* We used to know the icon for this file, but now the file says it isn't
400 * ready. This means that some file info has been invalidated, which
401 * doesn't necessarily mean that the previously-determined icon is
402 * wrong (in fact, in practice it usually doesn't mean that). Keep showing
403 * the one we last determined for now.
404 */
405 leave_surface_unchanged = TRUE;
406 }
407
408 g_free (icon_name);
409
410 if (!leave_surface_unchanged)
411 {
412 gtk_image_set_from_surface (GTK_IMAGE (sidebar_title->details->icon), surface);
413 }
414
415 if (surface != NULL)
416 {
417 sidebar_title->details->determined_icon = TRUE;
418 cairo_surface_destroy (surface);
419 }
420}

422static void
423override_title_font (GtkWidget *widget,
424 const gchar *font)
425{
426 gchar *css;
427 GtkCssProvider *provider;
428 gchar *tempsize;
429
430 provider = gtk_css_provider_new ();
431 tempsize = g_strdup (font);
432
433 g_strreverse (tempsize);
434 g_strcanon (tempsize, "1234567890", '\0');
435 g_strreverse (tempsize);
436
437 gchar tempfont [strlen (font)];
438 g_strlcpy (tempfont, font, sizeof (tempfont));
439 tempfont [strlen (font) - strlen (tempsize)] = 0;
440
441 css = g_strdup_printf ("label { font-family: %s; font-size: %spt; }", tempfont, tempsize);
442 gtk_css_provider_load_from_data (provider, css, -1, NULL);
443 g_free (tempsize);
444 g_free (css);
445
446 gtk_style_context_add_provider (gtk_widget_get_style_context (widget),
447 GTK_STYLE_PROVIDER (provider),
448 GTK_STYLE_PROVIDER_PRIORITY_APPLICATION);
449 g_object_unref (provider);
450}

452static void
453update_title_font (CajaSidebarTitle *sidebar_title)
454{
455 int available_width, width;
456 int max_fit_font_size, max_style_font_size;
457 GtkStyleContext *context;
458 GtkStateFlags state;
459 GtkAllocation allocation;
460 PangoFontDescription *title_font, *tmp_font;
461 PangoLayout *layout;
462 char *title_font_str;
463
464 /* Make sure theres work to do */
465 if (sidebar_title->details->title_text == NULL
466 || strlen (sidebar_title->details->title_text) < 1)
467 {
468 return;
469 }
470
471 gtk_widget_get_allocation (GTK_WIDGET (sidebar_title), &allocation);
472 available_width = allocation.width - TITLE_PADDING;
473
474 /* No work to do */
475 if (available_width <= 0)
476 {
477 return;
478 }
479
480 context = gtk_widget_get_style_context (GTK_WIDGET (sidebar_title));
481 state = gtk_widget_get_state_flags (GTK_WIDGET (sidebar_title));
482 gtk_style_context_get (context, state, GTK_STYLE_PROPERTY_FONT, &title_font, NULL);
483 max_style_font_size = pango_font_description_get_size (title_font) * 1.8 / PANGO_SCALE;
484 if (max_style_font_size < MIN_TITLE_FONT_SIZE + 1)
485 {
486 max_style_font_size = MIN_TITLE_FONT_SIZE + 1;
487 }
488
489 /* Calculate largest-fitting font size */
490 layout = pango_layout_new (gtk_widget_get_pango_context (sidebar_title->details->title_label));
491 pango_layout_set_text (layout, sidebar_title->details->title_text, -1);
492 pango_layout_set_font_description (layout, title_font);
493 tmp_font = pango_font_description_new ();
494
495 max_fit_font_size = max_style_font_size;
496 for (; max_fit_font_size >= MIN_TITLE_FONT_SIZE; max_fit_font_size--)
497 {
498 pango_font_description_set_size (tmp_font, max_fit_font_size * PANGO_SCALE);
499 pango_layout_set_font_description (layout, tmp_font);
500 pango_layout_get_pixel_size (layout, &width, NULL);
501
502 if (width <= available_width)
503 break;
504 }
505
506 pango_font_description_free (tmp_font);
507 g_object_unref (layout);
508
509 pango_font_description_set_size (title_font, max_fit_font_size * PANGO_SCALE);
510 pango_font_description_set_weight (title_font, PANGO_WEIGHT_BOLD);
511
512 title_font_str = pango_font_description_to_string (title_font);
513 override_title_font (sidebar_title->details->title_label, title_font_str);
514 g_free (title_font_str);
515
516 pango_font_description_free (title_font);
517}

520update_title (CajaSidebarTitle *sidebar_title)

536static void
537append_and_eat (GString *string, const char *separator, char *new_string)
538{
539 if (new_string == NULL)
540 {

542 }

545 g_string_append (string, separator);
546 }
547 g_string_append (string, new_string);

563static void
564update_more_info (CajaSidebarTitle *sidebar_title)
565{
566 CajaFile *file;
567 GString *info_string;
568 char *component_info;
569 GtkAllocation allocation;
570
571 file = sidebar_title->details->file;
572
573 /* allow components to specify the info if they wish to */
574 component_info = get_property_from_component (sidebar_title, "summary_info");
575 if (component_info != NULL && strlen (component_info) > 0)
576 {
577 info_string = g_string_new (component_info);
578 g_free (component_info);
579 }
580 else
581 {
582 char *type_string;
583 int sidebar_width;
584
585 info_string = g_string_new (NULL);
586
587 type_string = NULL;
588
589 if (file != NULL && caja_file_should_show_type (file))
590 {
591 type_string = caja_file_get_string_attribute (file, "type");
592 }
593
594 if (type_string != NULL)
595 {
596 append_and_eat (info_string, NULL, type_string);
597 append_and_eat (info_string, ", ",
598 caja_file_get_string_attribute (file, "size"));
599 }
600 else
601 {
602 append_and_eat (info_string, NULL,
603 caja_file_get_string_attribute (file, "size"));
604 }
605
606 gtk_widget_get_allocation (GTK_WIDGET (sidebar_title), &allocation);
607 sidebar_width = allocation.width - 2 * SIDEBAR_INFO_MARGIN;
608 if (sidebar_width > MINIMUM_INFO_WIDTH)
609 {
610 char *date_modified_str;
611 PangoLayout *layout;
612
613 layout = pango_layout_copy (gtk_label_get_layout (GTK_LABEL (sidebar_title->details->more_info_label)));
614 pango_layout_set_width (layout, -1);
615 date_modified_str = caja_file_fit_modified_date_as_string
616 (file, sidebar_width, measure_width_callback, NULL, layout);
617 g_object_unref (layout);
618 append_and_eat (info_string, "\n", date_modified_str);
619 }
620 }
621 gtk_label_set_text (GTK_LABEL (sidebar_title->details->more_info_label),
622 info_string->str);
623
624 g_string_free (info_string, TRUE);
625}

638static void
639update_emblems (CajaSidebarTitle *sidebar_title)
640{
641 GList *pixbufs, *p;
642 GdkPixbuf *pixbuf = NULL;
643
644 /* exit if we don't have the file yet */
645 if (sidebar_title->details->file == NULL)
646 {
647 return;
648 }
649
650 /* First, deallocate any existing ones */
651 gtk_container_foreach (GTK_CONTAINER (sidebar_title->details->emblem_box),
652 (GtkCallback) gtk_widget_destroy,
653 NULL);
654
655 /* fetch the emblem icons from metadata */
656 pixbufs = caja_file_get_emblem_pixbufs (sidebar_title->details->file,
657 caja_icon_get_emblem_size_for_icon_size (CAJA_ICON_SIZE_STANDARD),
658 FALSE,
659 NULL);
660
661 /* loop through the list of emblems, installing them in the box */
662 for (p = pixbufs; p != NULL; p = p->next)
663 {
664 pixbuf = p->data;
665 add_emblem (sidebar_title, pixbuf);
666 g_object_unref (pixbuf);
667 }
668 g_list_free (pixbufs);
669}

706static void
707monitor_add (CajaSidebarTitle *sidebar_title)
708{
709 CajaFileAttributes attributes;
710
711 /* Monitor the things needed to get the right icon. Don't
712 * monitor a directory's item count at first even though the
713 * "size" attribute is based on that, because the main view
714 * will get it for us in most cases, and in other cases it's
715 * OK to not show the size -- if we did monitor it, we'd be in
716 * a race with the main view and could cause it to have to
717 * load twice. Once we have a size, though, we want to monitor
718 * the size to guarantee it stays up to date.
719 */
720
721 sidebar_title->details->monitoring_count = item_count_ready (sidebar_title);
722
723 attributes = CAJA_FILE_ATTRIBUTES_FOR_ICON | CAJA_FILE_ATTRIBUTE_INFO;
724 if (sidebar_title->details->monitoring_count)
725 {
726 attributes |= CAJA_FILE_ATTRIBUTE_DIRECTORY_ITEM_COUNT;
727 }
728
729 caja_file_monitor_add (sidebar_title->details->file, sidebar_title, attributes);
730}

732static void
733update_all (CajaSidebarTitle *sidebar_title)
734{
735 update_icon (sidebar_title);
736
737 update_title (sidebar_title);
738 update_more_info (sidebar_title);
739
740 update_emblems (sidebar_title);
741
742 /* Redo monitor once the count is ready. */
743 if (!sidebar_title->details->monitoring_count && item_count_ready (sidebar_title))
744 {
745 caja_file_monitor_remove (sidebar_title->details->file, sidebar_title);
746 monitor_add (sidebar_title);
747 }

750void
751caja_sidebar_title_set_file (CajaSidebarTitle *sidebar_title,
752 CajaFile *file,
753 const char *initial_text)
754{
755 if (file != sidebar_title->details->file)
756 {
757 release_file (sidebar_title);
758 sidebar_title->details->file = file;
759 sidebar_title->details->determined_icon = FALSE;
760 caja_file_ref (sidebar_title->details->file);
761
762 /* attach file */
763 if (file != NULL)
764 {
765 sidebar_title->details->file_changed_connection =
766 g_signal_connect_object
767 (sidebar_title->details->file, "changed",
768 G_CALLBACK (update_all), sidebar_title, G_CONNECT_SWAPPED);
769 monitor_add (sidebar_title);
770 }
771 }
772
773 g_free (sidebar_title->details->title_text);
774 sidebar_title->details->title_text = g_strdup (initial_text);
775
776 update_all (sidebar_title);
777}
