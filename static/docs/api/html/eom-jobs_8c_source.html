73{
74 EomJob *job;
75
76 job = EOM_JOB (object);
77
78 if (job->error) {
79 g_error_free (job->error);

99eom_job_class_init (EomJobClass *class)
100{
101 GObjectClass *oclass;
102
103 oclass = G_OBJECT_CLASS (class);
104
105 oclass->dispose = eom_job_dispose;

110 g_signal_new ("finished",
111 EOM_TYPE_JOB,
112 G_SIGNAL_RUN_LAST,

114 NULL, NULL,
115 g_cclosure_marshal_VOID__VOID,
116 G_TYPE_NONE, 0);

120 EOM_TYPE_JOB,
121 G_SIGNAL_RUN_LAST,

123 NULL, NULL,
124 g_cclosure_marshal_VOID__FLOAT,

144void
145eom_job_run (EomJob *job)
146{
147 EomJobClass *class;
148
149 g_return_if_fail (EOM_IS_JOB (job));
150
151 class = EOM_JOB_GET_CLASS (job);
152 if (class->run)
153 class->run (job);
154 else
155 eom_job_run_default (job);

157static gboolean
158notify_progress (gpointer data)
159{
160 EomJob *job = EOM_JOB (data);
161
162 g_signal_emit (job, job_signals[SIGNAL_PROGRESS], 0, job->progress);
163
164 return FALSE;
165}

167void
168eom_job_set_progress (EomJob *job, float progress)
169{
170 g_return_if_fail (EOM_IS_JOB (job));
171 g_return_if_fail (progress >= 0.0 && progress <= 1.0);
172
173 g_mutex_lock (job->mutex);
174 job->progress = progress;
175 g_mutex_unlock (job->mutex);
176
177 g_idle_add (notify_progress, job);
178}

228static void
229eom_job_thumbnail_run (EomJob *ejob)
230{
231 gchar *orig_width, *orig_height;
232 gint width, height;
233 GdkPixbuf *pixbuf;
234 EomJobThumbnail *job;
235
236 g_return_if_fail (EOM_IS_JOB_THUMBNAIL (ejob));
237
238 job = EOM_JOB_THUMBNAIL (ejob);
239
240 if (ejob->error) {
241 g_error_free (ejob->error);
242 ejob->error = NULL;
243 }
244
245 job->thumbnail = eom_thumbnail_load (job->image,
246 &ejob->error);
247
248 if (!job->thumbnail) {
249 ejob->finished = TRUE;
250 return;
251 }
252
253 orig_width = g_strdup (gdk_pixbuf_get_option (job->thumbnail, "tEXt::Thumb::Image::Width"));
254 orig_height = g_strdup (gdk_pixbuf_get_option (job->thumbnail, "tEXt::Thumb::Image::Height"));
255
256 pixbuf = eom_thumbnail_fit_to_size (job->thumbnail, EOM_LIST_STORE_THUMB_SIZE);
257 g_object_unref (job->thumbnail);
258 job->thumbnail = eom_thumbnail_add_frame (pixbuf);
259 g_object_unref (pixbuf);
260
261 if (orig_width) {
262 sscanf (orig_width, "%i", &width);
263 g_object_set_data (G_OBJECT (job->thumbnail),
264 EOM_THUMBNAIL_ORIGINAL_WIDTH,
265 GINT_TO_POINTER (width));
266 g_free (orig_width);
267 }
268 if (orig_height) {
269 sscanf (orig_height, "%i", &height);
270 g_object_set_data (G_OBJECT (job->thumbnail),
271 EOM_THUMBNAIL_ORIGINAL_HEIGHT,
272 GINT_TO_POINTER (height));
273 g_free (orig_height);
274 }
275
276 if (ejob->error) {
277 g_warning ("%s", ejob->error->message);
278 }
279
280 ejob->finished = TRUE;
281}

375static void
376filter_files (GSList *files, GList **file_list, GList **error_list)
377{
378 GSList *it;
379 GFileInfo *file_info;
380
381 for (it = files; it != NULL; it = it->next) {
382 GFile *file;
383 GFileType type = G_FILE_TYPE_UNKNOWN;
384
385 file = (GFile *) it->data;
386
387 if (file != NULL) {
388 file_info = g_file_query_info (file,
389 G_FILE_ATTRIBUTE_STANDARD_TYPE","
390 G_FILE_ATTRIBUTE_STANDARD_CONTENT_TYPE","
391 G_FILE_ATTRIBUTE_STANDARD_FAST_CONTENT_TYPE,
392 0, NULL, NULL);
393 if (file_info == NULL) {
394 type = G_FILE_TYPE_UNKNOWN;
395 } else {
396 type = g_file_info_get_file_type (file_info);
397
398 /* Workaround for gvfs backends that
399 don't set the GFileType. */
400 if (G_UNLIKELY (type == G_FILE_TYPE_UNKNOWN)) {
401 const gchar *ctype;
402
403 ctype = eom_util_get_content_type_with_fallback (file_info);
404
405 /* If the content type is supported
406 adjust the file_type */
407 if (eom_image_is_supported_mime_type (ctype))
408 type = G_FILE_TYPE_REGULAR;
409 }
410
411 g_object_unref (file_info);
412 }
413 }
414
415 switch (type) {
416 case G_FILE_TYPE_REGULAR:
417 case G_FILE_TYPE_DIRECTORY:
418 *file_list = g_list_prepend (*file_list, g_object_ref (file));
419 break;
420 default:
421 *error_list = g_list_prepend (*error_list,
422 g_file_get_uri (file));
423 break;
424 }
425 }
426
427 *file_list = g_list_reverse (*file_list);
428 *error_list = g_list_reverse (*error_list);
429}

431static void
432eom_job_model_run (EomJob *ejob)
433{
434 GList *filtered_list = NULL;
435 GList *error_list = NULL;
436 EomJobModel *job;
437
438 g_return_if_fail (EOM_IS_JOB_MODEL (ejob));
439
440 job = EOM_JOB_MODEL (ejob);
441
442 filter_files (job->file_list, &filtered_list, &error_list);
443
444 job->store = EOM_LIST_STORE (eom_list_store_new ());
445
446 eom_list_store_add_files (job->store, filtered_list, job->preserve_order);
447
448 g_list_free_full (filtered_list, g_object_unref);
449 g_list_free_full (error_list, g_free);
450
451 ejob->finished = TRUE;
452}

524void
525eom_job_transform_run (EomJob *ejob)
526{
527 EomJobTransform *job;
528 GList *it;
529
530 g_return_if_fail (EOM_IS_JOB_TRANSFORM (ejob));
531
532 job = EOM_JOB_TRANSFORM (ejob);
533
534 if (ejob->error) {
535 g_error_free (ejob->error);
536 ejob->error = NULL;
537 }
538
539 for (it = job->images; it != NULL; it = it->next) {
540 EomImage *image = EOM_IMAGE (it->data);
541
542 if (job->trans == NULL) {
543 eom_image_undo (image);
544 } else {
545 eom_image_transform (image, job->trans, ejob);
546 }
547
548 if (eom_image_is_modified (image) || job->trans == NULL) {
549 g_object_ref (image);
550 g_idle_add (eom_job_transform_image_modified, image);
551 }
552 }
553
554 ejob->finished = TRUE;
555}

603static void
604save_progress_handler (EomImage *image, gfloat progress, gpointer data)
605{
606 EomJobSave *job = EOM_JOB_SAVE (data);

615static void
616eom_job_save_run (EomJob *ejob)
617{
618 EomJobSave *job;
619 GList *it;
620
621 g_return_if_fail (EOM_IS_JOB_SAVE (ejob));
622
623 job = EOM_JOB_SAVE (ejob);
624
625 job->current_pos = 0;
626
627 for (it = job->images; it != NULL; it = it->next, job->current_pos++) {
628 EomImage *image = EOM_IMAGE (it->data);
629 EomImageSaveInfo *save_info = NULL;
630 gulong handler_id = 0;
631 gboolean success = FALSE;
632
633 job->current_image = image;
634
635 /* Make sure the image doesn't go away while saving */
636 eom_image_data_ref (image);
637
638 if (!eom_image_has_data (image, EOM_IMAGE_DATA_ALL)) {
639 EomImageMetadataStatus m_status;
640 gint data2load = 0;
641
642 m_status = eom_image_get_metadata_status (image);
643 if (!eom_image_has_data (image, EOM_IMAGE_DATA_IMAGE)) {
644 // Queue full read in this case
645 data2load = EOM_IMAGE_DATA_ALL;
646 } else if (m_status == EOM_IMAGE_METADATA_NOT_READ) {
647 // Load only if we haven't read it yet
648 data2load = EOM_IMAGE_DATA_EXIF | EOM_IMAGE_DATA_XMP;
649 }
650
651 if (data2load != 0) {
652 eom_image_load (image,
653 data2load,
654 NULL,
655 &ejob->error);
656 }
657 }
658
659 handler_id = g_signal_connect (image, "save-progress",
660 G_CALLBACK (save_progress_handler),
661 job);
662
663 save_info = eom_image_save_info_new_from_image (image);
664
665 success = eom_image_save_by_info (image,
666 save_info,
667 &ejob->error);
668
669 if (save_info)
670 g_object_unref (save_info);
671
672 if (handler_id != 0)
673 g_signal_handler_disconnect (image, handler_id);
674
675 eom_image_data_unref (image);
676
677 if (!success) break;
678 }
679
680 ejob->finished = TRUE;
681}

737static void
738eom_job_save_as_run (EomJob *ejob)
739{
740 EomJobSave *job;
741 EomJobSaveAs *saveas_job;
742 GList *it;
743 guint n_images;
744
745 g_return_if_fail (EOM_IS_JOB_SAVE_AS (ejob));
746
747 job = EOM_JOB_SAVE (ejob);
748
749 n_images = g_list_length (job->images);
750
751 saveas_job = EOM_JOB_SAVE_AS (job);
752
753 job->current_pos = 0;
754
755 for (it = job->images; it != NULL; it = it->next, job->current_pos++) {
756 GdkPixbufFormat *format;
757 EomImageSaveInfo *src_info, *dest_info;
758 EomImage *image = EOM_IMAGE (it->data);
759 gboolean success = FALSE;
760 gulong handler_id = 0;
761
762 job->current_image = image;
763
764 eom_image_data_ref (image);
765
766 if (!eom_image_has_data (image, EOM_IMAGE_DATA_ALL)) {
767 EomImageMetadataStatus m_status;
768 gint data2load = 0;
769
770 m_status = eom_image_get_metadata_status (image);
771 if (!eom_image_has_data (image, EOM_IMAGE_DATA_IMAGE)) {
772 // Queue full read in this case
773 data2load = EOM_IMAGE_DATA_ALL;
774 } else if (m_status == EOM_IMAGE_METADATA_NOT_READ) {
775 // Load only if we haven't read it yet
776 data2load = EOM_IMAGE_DATA_EXIF | EOM_IMAGE_DATA_XMP;
777 }
778
779 if (data2load != 0) {
780 eom_image_load (image,
781 data2load,
782 NULL,
783 &ejob->error);
784 }
785 }
786
787 g_assert (ejob->error == NULL);
788
789 handler_id = g_signal_connect (image, "save-progress",
790 G_CALLBACK (save_progress_handler),
791 job);
792
793 src_info = eom_image_save_info_new_from_image (image);
794
795 if (n_images == 1) {
796 g_assert (saveas_job->file != NULL);
797
798 format = eom_pixbuf_get_format (saveas_job->file);
799
