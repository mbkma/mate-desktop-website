168static void

170{
171 Tile *tile = TILE (g_object);

173

187static void
188tile_dispose (GObject * g_object)
189{
190 Tile *tile = TILE (g_object);
191
192 /* free the TileAction object */
193 if (tile->n_actions)
194 {
195 gint x;
196 for (x = 0; x < tile->n_actions; x++)
197 {
198 if (tile->actions[x] != NULL) {
199 g_object_unref (tile->actions[x]);
200 tile->actions[x] = NULL;
201 }
202 }
203 }
204
205 /* free the GtkMenu object */
206 if (tile->context_menu != NULL) {
207 gtk_widget_destroy (GTK_WIDGET (tile->context_menu));
208 tile->context_menu = NULL;
209 }
210
211 (*G_OBJECT_CLASS (tile_parent_class)->dispose) (g_object);
212}

235static void
236tile_set_property (GObject * g_obj, guint prop_id, const GValue * value, GParamSpec * param_spec)
237{
238 Tile *tile;
239 GtkMenu *menu;
240
241 if (!IS_TILE (g_obj))
242 return;
243
244 tile = TILE (g_obj);
245
246 switch (prop_id)
247 {
248 case PROP_TILE_URI:
249 tile->uri = g_strdup (g_value_get_string (value));
250 break;
251
252 case PROP_TILE_CONTEXT_MENU:
253 menu = g_value_get_object (value);
254
255 if (menu == tile->context_menu)
256 break;
257
258 if (tile->context_menu)
259 gtk_menu_detach (tile->context_menu);
260
261 tile->context_menu = menu;
262
263 if (tile->context_menu)
264 gtk_menu_attach_to_widget (tile->context_menu, GTK_WIDGET (tile), NULL);
265
266 break;
267
268 default:
269 break;
270 }

361static gboolean
362tile_button_release (GtkWidget * widget, GdkEventButton * event)
363{
364 Tile *tile = TILE (widget);
365 TilePrivate *priv = tile_get_instance_private (tile);
366
367 TileEvent *tile_event;
368 gboolean handled;
369
370 if (priv->is_dragging)
371 {
372 priv->is_dragging = FALSE;
373
374 return TRUE;
375 }
376
377 switch (event->button)
378 {
379 case 1:
380 tile_event = g_new0 (TileEvent, 1);
381 tile_event->time = event->time;
382
383 if (double_click_detector_is_double_click (priv->double_click_detector, event->time,
384 TRUE))
385 tile_event->type = TILE_EVENT_ACTIVATED_DOUBLE_CLICK;
386 else
387 tile_event->type = TILE_EVENT_ACTIVATED_SINGLE_CLICK;
388
389 g_signal_emit (tile, tile_signals[TILE_ACTIVATED_SIGNAL], 0, tile_event);
390 g_signal_emit_by_name (GTK_BUTTON (widget), "button-release-event", &event, &handled);
391
392 g_free (tile_event);
393
394 break;
395
396 case 3:
397 if (GTK_IS_MENU (tile->context_menu))
398 gtk_menu_popup_at_widget (GTK_MENU (tile->context_menu),
399 widget,
400 GDK_GRAVITY_SOUTH_WEST,
401 GDK_GRAVITY_NORTH_WEST,
402 (const GdkEvent*) event);
403
404 break;
405
406 default:
407 break;
408 }
409
410 return TRUE;
411}
