105static void
106interface_base_init( NAIFactoryProviderInterface *klass )
107{
108 static const gchar *thisfn = "na_ifactory_provider_interface_base_init";
109
110 if( !st_initializations ){
111
112 g_debug( "%s: klass=%p (%s)", thisfn, ( void * ) klass, G_OBJECT_CLASS_NAME( klass ));
113
114 klass->private = g_new0( NAIFactoryProviderInterfacePrivate, 1 );

116 klass->get_version = ifactory_provider_get_version;
117 klass->read_start = NULL;
118 klass->read_data = NULL;
119 klass->read_done = NULL;
120 klass->write_start = NULL;
121 klass->write_data = NULL;
122 klass->write_done = NULL;
123 }
124
125 st_initializations += 1;
126}

213guint
214na_ifactory_provider_write_item( const NAIFactoryProvider *writer, void *writer_data, NAIFactoryObject *object, GSList **messages )
215{
216 static const gchar *thisfn = "na_ifactory_provider_write_item";
217 guint code;
218
219 g_return_val_if_fail( NA_IS_IFACTORY_PROVIDER( writer ), NA_IIO_PROVIDER_CODE_PROGRAM_ERROR );
220 g_return_val_if_fail( NA_IS_IFACTORY_OBJECT( object ), NA_IIO_PROVIDER_CODE_PROGRAM_ERROR );
221
222 g_debug( "%s: writer=%p, writer_data=%p, object=%p (%s)",
223 thisfn, ( void * ) writer, ( void * ) writer_data, ( void * ) object, G_OBJECT_TYPE_NAME( object ));
224
225 code = v_factory_provider_write_start( writer, writer_data, object, messages );
226
227 if( code == NA_IIO_PROVIDER_CODE_OK ){
228 code = na_factory_object_write_item( object, writer, writer_data, messages );
229 }
230
231 if( code == NA_IIO_PROVIDER_CODE_OK ){
232 code = v_factory_provider_write_done( writer, writer_data, object, messages );
233 }
234
235 return( code );
236}
