44static void
45prefs_changed_callback (MetaPreference preference,
46 gpointer data)
47{

61void
62meta_window_ensure_frame (MetaWindow *window)
63{
64 MetaFrame *frame;
65 XSetWindowAttributes attrs;
66
67 if (window->frame)
68 return;
69
70 /* See comment below for why this is required. */
71 meta_display_grab (window->display);
72
73 frame = g_new (MetaFrame, 1);
74
75 frame->window = window;
76 frame->xwindow = None;
77
78 frame->rect = window->rect;
79 frame->child_x = 0;
80 frame->child_y = 0;
81 frame->bottom_height = 0;
82 frame->right_width = 0;
83 frame->current_cursor = 0;
84
85 frame->mapped = FALSE;
86 frame->need_reapply_frame_shape = TRUE;
87 frame->is_flashing = FALSE;
88 frame->borders_cached = FALSE;
89
90 meta_verbose ("Frame geometry %d,%d %dx%d\n",
91 frame->rect.x, frame->rect.y,
92 frame->rect.width, frame->rect.height);
93
94 frame->xwindow = meta_ui_create_frame_window (window->screen->ui,
95 window->display->xdisplay,
96 window->xvisual,
97 frame->rect.x,
98 frame->rect.y,
99 frame->rect.width,
100 frame->rect.height);
101
102 meta_verbose ("Frame for %s is 0x%lx\n", frame->window->desc, frame->xwindow);
103 attrs.event_mask = EVENT_MASK;
104 XChangeWindowAttributes (window->display->xdisplay,
105 frame->xwindow, CWEventMask, &attrs);
106
107 meta_display_register_x_window (window->display, &frame->xwindow, window);
108
109 /* Reparent the client window; it may be destroyed,
110 * thus the error trap. We'll get a destroy notify later
111 * and free everything. Comment in FVWM source code says
112 * we need a server grab or the child can get its MapNotify
113 * before we've finished reparenting and getting the decoration
114 * window onscreen, so ensure_frame must be called with
115 * a grab.
116 */
117 meta_error_trap_push (window->display);
118 if (window->mapped)
119 {
120 window->mapped = FALSE; /* the reparent will unmap the window,
121 * we don't want to take that as a withdraw
122 */
123 meta_topic (META_DEBUG_WINDOW_STATE,
124 "Incrementing unmaps_pending on %s for reparent\n", window->desc);
125 window->unmaps_pending += 1;
126 }
127 /* window was reparented to this position */
128 window->rect.x = 0;
129 window->rect.y = 0;
130
131 XReparentWindow (window->display->xdisplay,
132 window->xwindow,
133 frame->xwindow,
134 window->rect.x,
135 window->rect.y);
136 /* FIXME handle this error */
137 meta_error_trap_pop (window->display, FALSE);
138
139 /* stick frame to the window */
140 window->frame = frame;
141
142 /* Now that frame->xwindow is registered with window, we can set its
143 * style.
144 */
145 meta_ui_update_frame_style (window->screen->ui, frame->xwindow);
146 if (window->title)
147 meta_ui_set_frame_title (window->screen->ui,
148 window->frame->xwindow,
149 window->title);
150
151 /* Move keybindings to frame instead of window */
152 meta_window_grab_keys (window);
153
154 /* Shape mask */
155 meta_ui_apply_frame_shape (frame->window->screen->ui,
156 frame->window->display,
157 frame->xwindow,
158 frame->rect.width,
159 frame->rect.height,
160 frame->window->has_shape);
161 frame->need_reapply_frame_shape = FALSE;
162
163 meta_display_ungrab (window->display);
164
165 meta_prefs_add_listener (prefs_changed_callback, frame);
166}

168void
169meta_window_destroy_frame (MetaWindow *window)
170{
171 MetaFrame *frame;
172 MetaFrameBorders borders;
173
174 if (window->frame == NULL)
175 return;
176
177 meta_verbose ("Unframing window %s\n", window->desc);
178
179 frame = window->frame;
180
181 meta_prefs_remove_listener (prefs_changed_callback, frame);
182
183 meta_frame_calc_borders (frame, &borders);
184
185 meta_bell_notify_frame_destroy (frame);
186
187 /* Unparent the client window; it may be destroyed,
188 * thus the error trap.
189 */
190 meta_error_trap_push (window->display);
191 if (window->mapped)
192 {
193 window->mapped = FALSE; /* Keep track of unmapping it, so we
194 * can identify a withdraw initiated
195 * by the client.
196 */
197 meta_topic (META_DEBUG_WINDOW_STATE,
198 "Incrementing unmaps_pending on %s for reparent back to root\n", window->desc);
199 window->unmaps_pending += 1;
200 }
201 XReparentWindow (window->display->xdisplay,
202 window->xwindow,
203 window->screen->xroot,
204 /* Using anything other than meta_window_get_position()
205 * coordinates here means we'll need to ensure a configure
206 * notify event is sent; see bug 399552.
207 */
208 window->frame->rect.x + borders.invisible.left,
209 window->frame->rect.y + borders.invisible.top);
210 meta_error_trap_pop (window->display, FALSE);
211
212 meta_ui_destroy_frame_window (window->screen->ui, frame->xwindow);
213
214 meta_display_unregister_x_window (window->display,
215 frame->xwindow);
216
217 window->frame = NULL;
218 if (window->frame_bounds)
219 {
220 cairo_region_destroy (window->frame_bounds);
221 window->frame_bounds = NULL;
222 }
223
224 /* Move keybindings to window instead of frame */
225 meta_window_grab_keys (window);
226
227 g_free (frame);
228
229 /* Put our state back where it should be */
230 meta_window_queue (window, META_QUEUE_CALC_SHOWING);
231 meta_window_queue (window, META_QUEUE_MOVE_RESIZE);
232}

234MetaFrameFlags
235meta_frame_get_flags (MetaFrame *frame)
236{
237 MetaFrameFlags flags;
238
239 flags = 0;
240
241 if (frame->window->border_only)
242 {
243 ; /* FIXME this may disable the _function_ as well as decor
244 * in some cases, which is sort of wrong.
245 */
246 }
247 else
248 {
249 flags |= META_FRAME_ALLOWS_MENU;
250
251 if (frame->window->has_close_func)
252 flags |= META_FRAME_ALLOWS_DELETE;
253
254 if (frame->window->has_maximize_func)
255 flags |= META_FRAME_ALLOWS_MAXIMIZE;
256
257 if (frame->window->has_minimize_func)
258 flags |= META_FRAME_ALLOWS_MINIMIZE;
259
260 if (frame->window->has_shade_func)
261 flags |= META_FRAME_ALLOWS_SHADE;
262 }
263
264 if (META_WINDOW_ALLOWS_MOVE (frame->window))
265 flags |= META_FRAME_ALLOWS_MOVE;
266
267 if (META_WINDOW_ALLOWS_HORIZONTAL_RESIZE (frame->window))
268 flags |= META_FRAME_ALLOWS_HORIZONTAL_RESIZE;
269
270 if (META_WINDOW_ALLOWS_VERTICAL_RESIZE (frame->window))
271 flags |= META_FRAME_ALLOWS_VERTICAL_RESIZE;
272
273 if (meta_window_appears_focused (frame->window))
274 flags |= META_FRAME_HAS_FOCUS;
275
276 if (frame->window->shaded)
277 flags |= META_FRAME_SHADED;
278
279 if (frame->window->on_all_workspaces)
280 flags |= META_FRAME_STUCK;
281
282 /* FIXME: Should we have some kind of UI for windows that are just vertically
283 * maximized or just horizontally maximized?
284 */
285 if (META_WINDOW_MAXIMIZED (frame->window))
286 flags |= META_FRAME_MAXIMIZED;
287
288 if (META_WINDOW_TILED_LEFT (frame->window))
289 flags |= META_FRAME_TILED_LEFT;
290
291 if (META_WINDOW_TILED_RIGHT (frame->window))
292 flags |= META_FRAME_TILED_RIGHT;
293
294 if (frame->window->fullscreen)
295 flags |= META_FRAME_FULLSCREEN;
296
297 if (frame->is_flashing)
298 flags |= META_FRAME_IS_FLASHING;
299
300 if (frame->window->wm_state_above)
301 flags |= META_FRAME_ABOVE;
302
303 return flags;
304}

384gboolean
385meta_frame_sync_to_window (MetaFrame *frame,
386 int resize_gravity,
387 gboolean need_move,
388 gboolean need_resize)
389{
390 if (!(need_move || need_resize))
391 {
392 return update_shape (frame);
393 }
394
395 meta_topic (META_DEBUG_GEOMETRY,
396 "Syncing frame geometry %d,%d %dx%d (SE: %d,%d)\n",
397 frame->rect.x, frame->rect.y,
398 frame->rect.width, frame->rect.height,
399 frame->rect.x + frame->rect.width,
400 frame->rect.y + frame->rect.height);
401
402 /* set bg to none to avoid flicker */
403 if (need_resize)
404 {
405 /* we need new shape if we're resized */
406 frame->need_reapply_frame_shape = TRUE;
407 }
408
409 /* Done before the window resize, because doing it before means
410 * part of the window being resized becomes unshaped, which may
411 * be sort of hard to see with bg = None. If we did it after
412 * window resize, part of the window being resized would become
413 * shaped, which might be more visible.
414 */
415 update_shape (frame);
416
417 meta_ui_move_resize_frame (frame->window->screen->ui,
418 frame->xwindow,
419 frame->rect.x,
420 frame->rect.y,
421 frame->rect.width,
422 frame->rect.height);
423
424 if (need_resize)
425 {
426 /* If we're interactively resizing the frame, repaint
427 * it immediately so we don't start to lag.
428 */
429 if (frame->window->display->grab_window ==
430 frame->window)
431 meta_ui_repaint_frame (frame->window->screen->ui,
432 frame->xwindow);
433 }
434 return need_resize;
435}
