71static void
72create_scale_box (GvcBalanceBar *bar)
73{
74 bar->priv->scale_box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
75 bar->priv->start_box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
76 bar->priv->end_box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
77 bar->priv->scale = gtk_scale_new (GTK_ORIENTATION_HORIZONTAL,
78 bar->priv->adjustment);
79
80 /* Balance and fade scales do not have an origin */
81 if (bar->priv->btype != BALANCE_TYPE_LFE)
82 gtk_scale_set_has_origin (GTK_SCALE (bar->priv->scale), FALSE);
83
84 gtk_widget_set_size_request (bar->priv->scale, SCALE_SIZE, -1);
85
86 gtk_box_pack_start (GTK_BOX (bar->priv->scale_box),
87 bar->priv->start_box,
88 FALSE, FALSE, 0);
89 gtk_box_pack_start (GTK_BOX (bar->priv->start_box),
90 bar->priv->label,
91 FALSE, FALSE, 0);
92 gtk_box_pack_start (GTK_BOX (bar->priv->scale_box),
93 bar->priv->scale,
94 TRUE, TRUE, 0);
95 gtk_box_pack_start (GTK_BOX (bar->priv->scale_box),
96 bar->priv->end_box,
97 FALSE, FALSE, 0);
98
99 ca_gtk_widget_disable_sounds (bar->priv->scale, FALSE);
100
101 gtk_widget_add_events (bar->priv->scale, GDK_SCROLL_MASK);
102
103 g_signal_connect (G_OBJECT (bar->priv->scale),
104 "scroll-event",
105 G_CALLBACK (on_scale_scroll_event),
106 bar);
107
108 if (bar->priv->size_group != NULL) {
109 gtk_size_group_add_widget (bar->priv->size_group,
110 bar->priv->start_box);
111
112 if (bar->priv->symmetric)
113 gtk_size_group_add_widget (bar->priv->size_group,
114 bar->priv->end_box);
115 }
116
117 gtk_scale_set_draw_value (GTK_SCALE (bar->priv->scale), FALSE);
118}

120static void
121update_scale_marks (GvcBalanceBar *bar)
122{
123 gchar *str_lower = NULL,
124 *str_upper = NULL;
125 gdouble lower,
126 upper;
127
128 gtk_scale_clear_marks (GTK_SCALE (bar->priv->scale));
129
130 switch (bar->priv->btype) {
131 case BALANCE_TYPE_RL:
132 str_lower = g_strdup_printf ("<small>%s</small>", C_("balance", "Left"));
133 str_upper = g_strdup_printf ("<small>%s</small>", C_("balance", "Right"));
134 break;
135 case BALANCE_TYPE_FR:
136 str_lower = g_strdup_printf ("<small>%s</small>", C_("balance", "Rear"));
137 str_upper = g_strdup_printf ("<small>%s</small>", C_("balance", "Front"));
138 break;
139 case BALANCE_TYPE_LFE:
140 str_lower = g_strdup_printf ("<small>%s</small>", C_("balance", "Minimum"));
141 str_upper = g_strdup_printf ("<small>%s</small>", C_("balance", "Maximum"));
142 break;
143 }
144
145 lower = gtk_adjustment_get_lower (bar->priv->adjustment);
146 gtk_scale_add_mark (GTK_SCALE (bar->priv->scale),
147 lower,
148 GTK_POS_BOTTOM,
149 str_lower);
150 upper = gtk_adjustment_get_upper (bar->priv->adjustment);
151 gtk_scale_add_mark (GTK_SCALE (bar->priv->scale),
152 upper,
153 GTK_POS_BOTTOM,
154 str_upper);
155 g_free (str_lower);
156 g_free (str_upper);
157
158 if (bar->priv->btype != BALANCE_TYPE_LFE)
159 gtk_scale_add_mark (GTK_SCALE (bar->priv->scale),
160 (upper - lower) / 2 + lower,
161 GTK_POS_BOTTOM,
162 NULL);
163}

187static void
188update_balance_value (GvcBalanceBar *bar)
189{
190 gdouble value = 0;
191
192 switch (bar->priv->btype) {
193 case BALANCE_TYPE_RL:
194 value = mate_mixer_stream_control_get_balance (bar->priv->control);
195 g_debug ("Balance value changed to %.2f", value);
196 break;
197 case BALANCE_TYPE_FR:
198 value = mate_mixer_stream_control_get_fade (bar->priv->control);
199 g_debug ("Fade value changed to %.2f", value);
200 break;
201 case BALANCE_TYPE_LFE:
202 value = mate_mixer_stream_control_get_channel_volume (bar->priv->control,
203 bar->priv->lfe_channel);
204
205 g_debug ("Subwoofer volume changed to %.0f", value);
206 break;
207 }
208
209 gtk_adjustment_set_value (bar->priv->adjustment, value);
210}

220static gint
221find_stream_lfe_channel (MateMixerStreamControl *control)
222{
223 guint i;
224
225 for (i = 0; i < mate_mixer_stream_control_get_num_channels (control); i++) {
226 MateMixerChannelPosition position;
227
228 position = mate_mixer_stream_control_get_channel_position (control, i);
229 if (position == MATE_MIXER_CHANNEL_LFE)
230 return i;
231 }
232
233 return -1;
234}

236static void
237gvc_balance_bar_set_control (GvcBalanceBar *bar, MateMixerStreamControl *control)
238{
239 g_return_if_fail (GVC_IS_BALANCE_BAR (bar));
240 g_return_if_fail (MATE_MIXER_IS_STREAM_CONTROL (control));
241
242 if (bar->priv->control != NULL) {
243 g_signal_handlers_disconnect_by_func (G_OBJECT (bar->priv->control),
244 on_balance_value_changed,
245 bar);
246 g_object_unref (bar->priv->control);
247 }
248
249 bar->priv->control = g_object_ref (control);
250
251 if (bar->priv->btype == BALANCE_TYPE_LFE) {
252 gdouble minimum;
253 gdouble maximum;
254
255 minimum = mate_mixer_stream_control_get_min_volume (bar->priv->control);
256 maximum = mate_mixer_stream_control_get_normal_volume (bar->priv->control);
257
258 /* Configure the adjustment for the volume limits of the current
259 * stream.
260 * Only subwoofer scale uses volume, balance and fade use fixed
261 * limits which do not need to be updated as balance type is
262 * only set during construction. */
263 gtk_adjustment_configure (GTK_ADJUSTMENT (bar->priv->adjustment),
264 gtk_adjustment_get_value (bar->priv->adjustment),
265 minimum,
266 maximum,
267 (maximum - minimum) / 100.0,
268 (maximum - minimum) / 10.0,
269 0.0);
270
271 bar->priv->lfe_channel = find_stream_lfe_channel (bar->priv->control);
272
273 if (G_LIKELY (bar->priv->lfe_channel > -1))
274 g_debug ("Found LFE channel at position %d", bar->priv->lfe_channel);
275 else
276 g_warn_if_reached ();
277 } else
278 bar->priv->lfe_channel = -1;
279
280 switch (bar->priv->btype) {
281 case BALANCE_TYPE_RL:
282 g_signal_connect (G_OBJECT (bar->priv->control),
283 "notify::balance",
284 G_CALLBACK (on_balance_value_changed),
285 bar);
286 break;
287 case BALANCE_TYPE_FR:
288 g_signal_connect (G_OBJECT (bar->priv->control),
289 "notify::fade",
290 G_CALLBACK (on_balance_value_changed),
291 bar);
292 break;
293 case BALANCE_TYPE_LFE:
294 g_signal_connect (G_OBJECT (bar->priv->control),
295 "notify::volume",
296 G_CALLBACK (on_balance_value_changed),
297 bar);
298 break;
299 }
300
301 update_balance_value (bar);
302 update_scale_marks (bar);
303
304 g_object_notify_by_pspec (G_OBJECT (bar), properties[PROP_CONTROL]);
305}

307static void
308gvc_balance_bar_set_balance_type (GvcBalanceBar *bar, GvcBalanceType btype)
309{
310 GtkWidget *frame;
311 GtkAdjustment *adjustment;
312
313 /* Create adjustment with limits for balance and fade types because
314 * some limits must be provided.
315 * If subwoofer type is used instead, the limits will be changed when
316 * stream is set. */
317 adjustment = GTK_ADJUSTMENT (gtk_adjustment_new (0.0, -1.0, 1.0, 0.05, 0.5, 0.0));
318
319 bar->priv->btype = btype;
320 bar->priv->adjustment = GTK_ADJUSTMENT (g_object_ref_sink (adjustment));
321
322 g_signal_connect (G_OBJECT (adjustment),
323 "value-changed",
324 G_CALLBACK (on_adjustment_value_changed),
325 bar);
326
327 switch (btype) {
328 case BALANCE_TYPE_RL:
329 bar->priv->label = gtk_label_new_with_mnemonic (_("_Balance:"));
330 break;
331 case BALANCE_TYPE_FR:
332 bar->priv->label = gtk_label_new_with_mnemonic (_("_Fade:"));
333 break;
334 case BALANCE_TYPE_LFE:
335 bar->priv->label = gtk_label_new_with_mnemonic (_("_Subwoofer:"));
336 break;
337 }
338
339 gtk_label_set_xalign (GTK_LABEL (bar->priv->label), 0.0);
340 gtk_label_set_yalign (GTK_LABEL (bar->priv->label), 0.0);
341
342 /* Frame */
343 frame = gtk_frame_new (NULL);
344 gtk_frame_set_shadow_type (GTK_FRAME (frame), GTK_SHADOW_NONE);
345 gtk_box_pack_start (GTK_BOX (bar), frame, TRUE, TRUE, 0);
346
347 /* Box with scale */
348 create_scale_box (bar);
349 gtk_container_add (GTK_CONTAINER (frame), bar->priv->scale_box);
350
351 gtk_label_set_mnemonic_widget (GTK_LABEL (bar->priv->label),
352 bar->priv->scale);
353
354 gtk_widget_set_direction (bar->priv->scale, GTK_TEXT_DIR_LTR);
355 gtk_widget_show_all (frame);
356
357 g_object_notify_by_pspec (G_OBJECT (bar), properties[PROP_BALANCE_TYPE]);
358}

430static gboolean
431on_scale_scroll_event (GtkWidget *widget,
432 GdkEventScroll *event,
433 GvcBalanceBar *bar)
434{
435 gdouble value;
436 gdouble minimum;
437 gdouble maximum;
438 gdouble step;
439
440 value = gtk_adjustment_get_value (bar->priv->adjustment);
441 minimum = gtk_adjustment_get_lower (bar->priv->adjustment);
442 maximum = gtk_adjustment_get_upper (bar->priv->adjustment);
443
444 // XXX fix this for GTK3
445
446 if (bar->priv->btype == BALANCE_TYPE_LFE)
447 step = (maximum - minimum) / 100.0;
448 else
449 step = 0.05;
450
451 if (event->direction == GDK_SCROLL_UP) {
452 if (value + step > maximum)
453 value = maximum;
454 else
455 value = value + step;
456 } else if (event->direction == GDK_SCROLL_DOWN) {
457 if (value - step < minimum)
458 value = minimum;
459 else
460 value = value - step;
461 }
462
463 gtk_adjustment_set_value (bar->priv->adjustment, value);
464 return TRUE;
465}

467static void
468on_adjustment_value_changed (GtkAdjustment *adjustment, GvcBalanceBar *bar)
469{
470 gdouble value;
471
472 if (bar->priv->control == NULL)
473 return;
474
475 value = gtk_adjustment_get_value (adjustment);
476
477 switch (bar->priv->btype) {
478 case BALANCE_TYPE_RL:
479 mate_mixer_stream_control_set_balance (bar->priv->control,
480 (gfloat) value);
481 break;
482 case BALANCE_TYPE_FR:
483 mate_mixer_stream_control_set_fade (bar->priv->control,
484 (gfloat) value);
485 break;
486 case BALANCE_TYPE_LFE:
487 mate_mixer_stream_control_set_channel_volume (bar->priv->control,
488 bar->priv->lfe_channel,
489 (guint) value);
490 break;
491 }
492}
