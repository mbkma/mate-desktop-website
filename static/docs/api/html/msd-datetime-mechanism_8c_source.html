166static gboolean
167msd_datetime_mechanism_adjust_time_handler (MateSettingsDateTimeMechanism *object,
168 GDBusMethodInvocation *invocation,
169 gint64 seconds_to_add,
170 gpointer user_data)
171{
172 struct timeval tv;
173 gboolean ret = FALSE;
174 GError *error = NULL;
175 MsdDatetimeMechanism *mechanism;
176
177 mechanism = MSD_DATETIME_MECHANISM (user_data);
178
179 reset_killtimer (mechanism->priv->loop);
180 g_debug ("AdjustTime(%ld) called", seconds_to_add);
181
182 if (gettimeofday (&tv, NULL) != 0) {
183 error = g_error_new (MSD_DATETIME_MECHANISM_ERROR,
184 MSD_DATETIME_MECHANISM_ERROR_GENERAL,
185 "Error calling gettimeofday(): %s", strerror (errno));
186 g_dbus_method_invocation_return_gerror (invocation, error);
187 g_error_free (error);
188 return FALSE;
189 }
190
191 if (!_check_polkit_for_action (mechanism,
192 "org.mate.settingsdaemon.datetimemechanism.settime",
193 g_dbus_method_invocation_get_sender (invocation),
194 &error)) {
195 g_dbus_method_invocation_return_gerror (invocation, error);
196 g_error_free (error);
197 return FALSE;
198 }
199
200 tv.tv_sec += (time_t) seconds_to_add;
201 ret = _set_time (mechanism, &tv, &error);
202 if (!ret) {
203 g_dbus_method_invocation_return_gerror (invocation, error);
204 g_error_free (error);
205 } else {
206 mate_settings_date_time_mechanism_complete_adjust_time (object, invocation);
207 }
208
209 return ret;
210}

212static gint
213check_can_do (MsdDatetimeMechanism *mechanism,
214 const char *action,
215 const char *sender,
216 GError **error)
217{
218 gint value = -1;
219 PolkitSubject *subject;
220 PolkitAuthorizationResult *result;
221
222 /* Check that caller is privileged */
223 subject = polkit_system_bus_name_new (sender);
224 result = polkit_authority_check_authorization_sync (mechanism->priv->auth,
225 subject,
226 action,
227 NULL,
228 0,
229 NULL,
230 error);
231 g_object_unref (subject);
232
233 if (*error != NULL) {
234 return value;
235 }
236
237 if (polkit_authorization_result_get_is_authorized (result)) {
238 value = 2;
239 }
240 else if (polkit_authorization_result_get_is_challenge (result)) {
241 value = 1;
242 }
243 else {
244 value = 0;
245 }
246
247 g_object_unref (result);
248 return value;
249}

307static gboolean
308msd_datetime_mechanism_get_hardware_clock_using_utc_handler (MateSettingsDateTimeMechanism *object,
309 GDBusMethodInvocation *invocation,
310 gpointer user_data G_GNUC_UNUSED)
311{
312 char **lines;
313 char *data;
314 gsize len;
315 gboolean is_utc;
316 GError *error = NULL;
317
318 if (!g_file_get_contents ("/etc/adjtime", &data, &len, &error)) {
319 GError *error2;
320 error2 = g_error_new (MSD_DATETIME_MECHANISM_ERROR,
321 MSD_DATETIME_MECHANISM_ERROR_GENERAL,
322 "Error reading /etc/adjtime file: %s", error->message);
323 g_error_free (error);
324 g_dbus_method_invocation_return_gerror (invocation, error2);
325 g_error_free (error2);
326 return FALSE;
327 }
328
329 lines = g_strsplit (data, "\n", 0);
330 g_free (data);
331
332 if (g_strv_length (lines) < 3) {
333 error = g_error_new (MSD_DATETIME_MECHANISM_ERROR,
334 MSD_DATETIME_MECHANISM_ERROR_GENERAL,
335 "Cannot parse /etc/adjtime");
336 g_dbus_method_invocation_return_gerror (invocation, error);
337 g_error_free (error);
338 g_strfreev (lines);
339 return FALSE;
340 }
341
342 if (strcmp (lines[2], "UTC") == 0) {
343 is_utc = TRUE;
344 } else if (strcmp (lines[2], "LOCAL") == 0) {
345 is_utc = FALSE;
346 } else {
347 error = g_error_new (MSD_DATETIME_MECHANISM_ERROR,
348 MSD_DATETIME_MECHANISM_ERROR_GENERAL,
349 "Expected UTC or LOCAL at line 3 of /etc/adjtime; found '%s'",
350 lines[2]);
351 g_dbus_method_invocation_return_gerror (invocation, error);
352 g_error_free (error);
353 g_strfreev (lines);
354 return FALSE;
355 }
356 g_strfreev (lines);
357 mate_settings_date_time_mechanism_complete_get_hardware_clock_using_utc (object, invocation, is_utc);
358 return TRUE;
359}

380static gboolean
381msd_datetime_mechanism_set_hardware_clock_using_utc_handler (MateSettingsDateTimeMechanism *object,
382 GDBusMethodInvocation *invocation,
383 gboolean using_utc,
384 gpointer user_data)
385{
386 GError *error = NULL;
387 MsdDatetimeMechanism *mechanism;
388
389 mechanism = MSD_DATETIME_MECHANISM (user_data);
390 if (!_check_polkit_for_action (mechanism,
391 "org.mate.settingsdaemon.datetimemechanism.configurehwclock",
392 g_dbus_method_invocation_get_sender (invocation),
393 &error)) {
394 g_dbus_method_invocation_return_gerror (invocation, error);
395 g_error_free (error);
396 return FALSE;
397 }
398
399 if (g_file_test ("/sbin/hwclock",
400 G_FILE_TEST_EXISTS | G_FILE_TEST_IS_REGULAR | G_FILE_TEST_IS_EXECUTABLE)) {
401 int exit_status;
402 char *cmd;
403 cmd = g_strdup_printf ("/sbin/hwclock %s --systohc", using_utc ? "--utc" : "--localtime");
404 if (!g_spawn_command_line_sync (cmd, NULL, NULL, &exit_status, &error)) {
405 GError *error2;
406 error2 = g_error_new (MSD_DATETIME_MECHANISM_ERROR,
407 MSD_DATETIME_MECHANISM_ERROR_GENERAL,
408 "Error spawning /sbin/hwclock: %s", error->message);
409 g_error_free (error);
410 g_dbus_method_invocation_return_gerror (invocation, error2);
411 g_error_free (error2);
412 g_free (cmd);
413 return FALSE;
414 }
415 g_free (cmd);
416 if (WEXITSTATUS (exit_status) != 0) {
417 error = g_error_new (MSD_DATETIME_MECHANISM_ERROR,
418 MSD_DATETIME_MECHANISM_ERROR_GENERAL,
419 "/sbin/hwclock returned %d", exit_status);
420 g_dbus_method_invocation_return_gerror (invocation, error);
421 g_error_free (error);
422 return FALSE;
423 }
424
425 if (!_rh_update_etc_sysconfig_clock ("UTC=", using_utc ? "true" : "false", &error)) {
426 g_dbus_method_invocation_return_gerror (invocation, error);
427 g_error_free (error);
428 return FALSE;
429 }
430
431 }
432 mate_settings_date_time_mechanism_complete_set_hardware_clock_using_utc (object, invocation);
433 return TRUE;
434}

474static gboolean
475msd_datetime_mechanism_set_timezone_handler (MateSettingsDateTimeMechanism *object,
476 GDBusMethodInvocation *invocation,
477 const gchar *zonefile,
478 gpointer user_data)
479{
480 GError *error = NULL;
481 MsdDatetimeMechanism *mechanism;
482
483 mechanism = MSD_DATETIME_MECHANISM (user_data);
484 reset_killtimer (mechanism->priv->loop);
485 g_debug ("SetTimezone('%s') called", zonefile);
486
487 if (!_check_polkit_for_action (mechanism,
488 "org.mate.settingsdaemon.datetimemechanism.settimezone",
489 g_dbus_method_invocation_get_sender (invocation),
490 &error)) {
491 g_dbus_method_invocation_return_gerror (invocation, error);
492 g_error_free (error);
493 return FALSE;
494 }
495
496 if (!system_timezone_set_from_file (zonefile, &error)) {
497 GError *error2;
498 int code;
499
500 if (error->code == SYSTEM_TIMEZONE_ERROR_INVALID_TIMEZONE_FILE)
501 code = MSD_DATETIME_MECHANISM_ERROR_INVALID_TIMEZONE_FILE;

503 code = MSD_DATETIME_MECHANISM_ERROR_GENERAL;
504
505 error2 = g_error_new (MSD_DATETIME_MECHANISM_ERROR,
506 code, "%s", error->message);
507
508
509 g_error_free (error);
510
511 g_dbus_method_invocation_return_gerror (invocation, error2);
512 g_error_free (error2);
513 return FALSE;
514 }
515
516 mate_settings_date_time_mechanism_complete_set_timezone (object, invocation);
517 return TRUE;
518}

547 mechanism);

549 "handle-set-time",
550 G_CALLBACK (msd_datetime_mechanism_set_time_handler),
551 mechanism);
552
553 g_signal_connect (mechanism->priv->skeleton,
554 "handle-adjust-time",
555 G_CALLBACK (msd_datetime_mechanism_adjust_time_handler),
556 mechanism);
557
558 g_signal_connect (mechanism->priv->skeleton,
559 "handle-get-hardware-clock-using-utc",
560 G_CALLBACK (msd_datetime_mechanism_get_hardware_clock_using_utc_handler),

562 g_signal_connect (mechanism->priv->skeleton,
563 "handle-set-hardware-clock-using-utc",
564 G_CALLBACK (msd_datetime_mechanism_set_hardware_clock_using_utc_handler),
565 mechanism);
566
567 exported = g_dbus_interface_skeleton_export (
568 G_DBUS_INTERFACE_SKELETON (mechanism->priv->skeleton),
569 connection,
570 MSD_DATETIME_DBUS_PATH,
571 &error);
572 if (!exported)
573 {
574 g_warning ("Failed to export interface: %s", error->message);
575 g_error_free (error);
576 g_main_loop_quit (mechanism->priv->loop);
577 }

656static void
657msd_datetime_mechanism_dispose (GObject *object)
658{
659 MsdDatetimeMechanism *mechanism;
660
661 g_return_if_fail (object != NULL);
662 g_return_if_fail (MSD_DATETIME_IS_MECHANISM (object));
663
664 mechanism = MSD_DATETIME_MECHANISM (object);
665
666 g_return_if_fail (mechanism->priv != NULL);
667
668 if (mechanism->priv->skeleton != NULL)
669 {
670 GDBusInterfaceSkeleton *skeleton;
671
672 skeleton = G_DBUS_INTERFACE_SKELETON (mechanism->priv->skeleton);
673 g_dbus_interface_skeleton_unexport (skeleton);
674 g_clear_object (&mechanism->priv->skeleton);
675 }
676
677 if (mechanism->priv->bus_name_id > 0)
678 {
679 g_bus_unown_name (mechanism->priv->bus_name_id);
680 mechanism->priv->bus_name_id = 0;
681 }
682
683
684 G_OBJECT_CLASS (msd_datetime_mechanism_parent_class)->dispose (object);
685}

800static gboolean
801_rh_update_etc_sysconfig_clock (const char *key,
802 const char *value,
803 GError **error)
804{
805 /* On Red Hat / Fedora, the /etc/sysconfig/clock file needs to be kept in sync */
806 if (g_file_test ("/etc/sysconfig/clock", G_FILE_TEST_EXISTS | G_FILE_TEST_IS_REGULAR)) {
807 char **lines;
808 int n;
809 gboolean replaced;
810 char *data;
811 gsize len;
812
813 if (!g_file_get_contents ("/etc/sysconfig/clock", &data, &len, error)) {
814 GError *error2;
815 error2 = g_error_new (MSD_DATETIME_MECHANISM_ERROR,
816 MSD_DATETIME_MECHANISM_ERROR_GENERAL,
817 "Error reading /etc/sysconfig/clock file: %s", (*error)->message);
818 g_error_free (*error);
819 g_propagate_error (error, error2);
820 return FALSE;
821 }
822 replaced = FALSE;
823 lines = g_strsplit (data, "\n", 0);
824 g_free (data);
825
826 for (n = 0; lines[n] != NULL; n++) {
827 if (g_str_has_prefix (lines[n], key)) {
828 g_free (lines[n]);
829 lines[n] = g_strdup_printf ("%s%s", key, value);
830 replaced = TRUE;
831 }
832 }
833 if (replaced) {
834 GString *str;
835
836 str = g_string_new (NULL);
837 for (n = 0; lines[n] != NULL; n++) {
838 g_string_append (str, lines[n]);
839 if (lines[n + 1] != NULL)
840 g_string_append_c (str, '\n');
841 }
842 data = g_string_free (str, FALSE);
843 len = strlen (data);
844 if (!g_file_set_contents ("/etc/sysconfig/clock", data, len, error)) {
845 GError *error2;
846 error2 = g_error_new (MSD_DATETIME_MECHANISM_ERROR,
847 MSD_DATETIME_MECHANISM_ERROR_GENERAL,
848 "Error updating /etc/sysconfig/clock: %s", (*error)->message);
849 g_error_free (*error);
850 g_propagate_error (error, error2);
851 g_free (data);
852 return FALSE;
853 }
854 g_free (data);
855 }
856 g_strfreev (lines);
857 }
858
859 return TRUE;
860}
