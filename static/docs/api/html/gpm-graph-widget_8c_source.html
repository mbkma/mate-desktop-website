303static void
304gpm_graph_widget_init (GpmGraphWidget *graph)
305{
306 PangoFontMap *fontmap;
307 PangoContext *context;
308 PangoFontDescription *desc;
309
310 graph->priv = gpm_graph_widget_get_instance_private (graph);
311 graph->priv->start_x = 0;
312 graph->priv->start_y = 0;
313 graph->priv->stop_x = 60;
314 graph->priv->stop_y = 100;
315 graph->priv->use_grid = TRUE;
316 graph->priv->use_legend = FALSE;
317 graph->priv->data_list = g_ptr_array_new_with_free_func ((GDestroyNotify) g_ptr_array_unref);
318 graph->priv->plot_list = g_ptr_array_new ();
319 graph->priv->key_data = NULL;
320 graph->priv->type_x = GPM_GRAPH_WIDGET_TYPE_TIME;
321 graph->priv->type_y = GPM_GRAPH_WIDGET_TYPE_PERCENTAGE;
322
323 /* do pango stuff */
324 fontmap = pango_cairo_font_map_get_default ();
325 context = pango_font_map_create_context (PANGO_FONT_MAP (fontmap));
326 pango_context_set_base_gravity (context, PANGO_GRAVITY_AUTO);
327
328 graph->priv->layout = pango_layout_new (context);
329 desc = pango_font_description_from_string (GPM_GRAPH_WIDGET_FONT);
330 pango_layout_set_font_description (graph->priv->layout, desc);
331 pango_font_description_free (desc);
332}

379gboolean
380gpm_graph_widget_data_assign (GpmGraphWidget *graph, GpmGraphWidgetPlot plot, GPtrArray *data)
381{
382 GPtrArray *copy;
383 GpmPointObj *obj;
384 guint i;
385
386 g_return_val_if_fail (data != NULL, FALSE);
387 g_return_val_if_fail (GPM_IS_GRAPH_WIDGET (graph), FALSE);
388
389 /* make a deep copy */
390 copy = g_ptr_array_new_with_free_func ((GDestroyNotify) gpm_point_obj_free);
391 for (i=0; i<data->len; i++) {
392 obj = gpm_point_obj_copy (g_ptr_array_index (data, i));
393 g_ptr_array_add (copy, obj);
394 }
395
396 /* get the new data */
397 g_ptr_array_add (graph->priv->data_list, copy);
398 g_ptr_array_add (graph->priv->plot_list, GUINT_TO_POINTER(plot));
399
400 /* refresh */
401 gtk_widget_queue_draw (GTK_WIDGET (graph));
402
403 return TRUE;
404}

418static gchar *
419gpm_get_axis_label (GpmGraphWidgetType axis, gfloat value)
420{
421 gchar *text = NULL;
422 if (axis == GPM_GRAPH_WIDGET_TYPE_TIME) {
423 gint time_s = abs((gint) value);
424 gint minutes = time_s / 60;
425 gint seconds = time_s - (minutes * 60);
426 gint hours = minutes / 60;
427 gint days = hours / 24;
428 minutes = minutes - (hours * 60);
429 hours = hours - (days * 24);
430 if (days > 0) {
431 if (hours == 0) {
432 /*Translators: This is %i days*/
433 text = g_strdup_printf (_("%id"), days);
434 } else {
435 /*Translators: This is %i days %02i hours*/
436 text = g_strdup_printf (_("%id%02ih"), days, hours);
437 }
438 } else if (hours > 0) {
439 if (minutes == 0) {
440 /*Translators: This is %i hours*/
441 text = g_strdup_printf (_("%ih"), hours);
442 } else {
443 /*Translators: This is %i hours %02i minutes*/
444 text = g_strdup_printf (_("%ih%02im"), hours, minutes);
445 }
446 } else if (minutes > 0) {
447 if (seconds == 0) {
448 /*Translators: This is %2i minutes*/
449 text = g_strdup_printf (_("%2im"), minutes);
450 } else {
451 /*Translators: This is %2i minutes %02i seconds*/
452 text = g_strdup_printf (_("%2im%02i"), minutes, seconds);
453 }
454 } else {
455 /*Translators: This is %2i seconds*/
456 text = g_strdup_printf (_("%2is"), seconds);
457 }
458 } else if (axis == GPM_GRAPH_WIDGET_TYPE_PERCENTAGE) {
459 /*Translators: This is %i Percentage*/
460 text = g_strdup_printf (_("%i%%"), (gint) value);
461 } else if (axis == GPM_GRAPH_WIDGET_TYPE_POWER) {
462 /*Translators: This is %.1f Watts*/
463 text = g_strdup_printf (_("%.1fW"), value);
464 } else if (axis == GPM_GRAPH_WIDGET_TYPE_FACTOR) {
465 text = g_strdup_printf ("%.1f", value);
466 } else if (axis == GPM_GRAPH_WIDGET_TYPE_VOLTAGE) {
467 /*Translators: This is %.1f Volts*/
468 text = g_strdup_printf (_("%.1fV"), value);
469 } else {
470 text = g_strdup_printf ("%i", (gint) value);
471 }
472 return text;
473}

482static void
483gpm_graph_widget_draw_grid (GpmGraphWidget *graph, cairo_t *cr)
484{
485 guint i;
486 gfloat b;
487 gdouble dotted[] = {1., 2.};
488 gfloat divwidth = (gfloat)graph->priv->box_width / 10.0f;
489 gfloat divheight = (gfloat)graph->priv->box_height / 10.0f;
490
491 cairo_save (cr);
492
493 cairo_set_line_width (cr, 1);
494 cairo_set_dash (cr, dotted, 2, 0.0);
495
496 /* do vertical lines */
497 cairo_set_source_rgb (cr, 0.1, 0.1, 0.1);
498 for (i=1; i<10; i++) {
499 b = graph->priv->box_x + ((gfloat) i * divwidth);
500 cairo_move_to (cr, (gint)b + 0.5f, graph->priv->box_y);
501 cairo_line_to (cr, (gint)b + 0.5f, graph->priv->box_y + graph->priv->box_height);
502 cairo_stroke (cr);
503 }
504
505 /* do horizontal lines */
506 for (i=1; i<10; i++) {
507 b = graph->priv->box_y + ((gfloat) i * divheight);
508 cairo_move_to (cr, graph->priv->box_x, (gint)b + 0.5f);
509 cairo_line_to (cr, graph->priv->box_x + graph->priv->box_width, (int)b + 0.5f);
510 cairo_stroke (cr);
511 }
512
513 cairo_restore (cr);
514}

523static void
524gpm_graph_widget_draw_labels (GpmGraphWidget *graph, cairo_t *cr)
525{
526 guint i;
527 gfloat b;
528 gchar *text;
529 gfloat value;
530 gfloat divwidth = (gfloat)graph->priv->box_width / 10.0f;
531 gfloat divheight = (gfloat)graph->priv->box_height / 10.0f;
532 gint length_x = graph->priv->stop_x - graph->priv->start_x;
533 gint length_y = graph->priv->stop_y - graph->priv->start_y;
534 PangoRectangle ink_rect, logical_rect;
535 gfloat offsetx = 0;
536 gfloat offsety = 0;
537
538 cairo_save (cr);
539
540 /* do x text */
541 cairo_set_source_rgb (cr, 0, 0, 0);
542 for (i=0; i<11; i++) {
543 b = graph->priv->box_x + ((gfloat) i * divwidth);
544 value = ((length_x / 10.0f) * (gfloat) i) + (gfloat) graph->priv->start_x;
545 text = gpm_get_axis_label (graph->priv->type_x, value);
546
547 pango_layout_set_text (graph->priv->layout, text, -1);
548 pango_layout_get_pixel_extents (graph->priv->layout, &ink_rect, &logical_rect);
549 /* have data points 0 and 10 bounded, but 1..9 centered */
550 if (i == 0)
551 offsetx = 2.0;
552 else if (i == 10)
553 offsetx = ink_rect.width;
554 else
555 offsetx = (ink_rect.width / 2.0f);
556
557 cairo_move_to (cr, b - offsetx,
558 graph->priv->box_y + graph->priv->box_height + 2.0);
559
560 pango_cairo_show_layout (cr, graph->priv->layout);
561 g_free (text);
562 }
563
564 /* do y text */
565 for (i=0; i<11; i++) {
566 b = graph->priv->box_y + ((gfloat) i * divheight);
567 value = ((gfloat) length_y / 10.0f) * (10 - (gfloat) i) + graph->priv->start_y;
568 text = gpm_get_axis_label (graph->priv->type_y, value);
569
570 pango_layout_set_text (graph->priv->layout, text, -1);
571 pango_layout_get_pixel_extents (graph->priv->layout, &ink_rect, &logical_rect);
572
573 /* have data points 0 and 10 bounded, but 1..9 centered */
574 if (i == 10)
575 offsety = 0;
576 else if (i == 0)
577 offsety = ink_rect.height;
578 else
579 offsety = (ink_rect.height / 2.0f);
580 offsetx = ink_rect.width + 7;
581 offsety -= 10;
582 cairo_move_to (cr, graph->priv->box_x - offsetx - 2, b + offsety);
583 pango_cairo_show_layout (cr, graph->priv->layout);
584 g_free (text);
585 }
586
587 cairo_restore (cr);
588}

597static guint
598gpm_graph_widget_get_y_label_max_width (GpmGraphWidget *graph, cairo_t *cr)
599{
600 guint i;
601 gchar *text;
602 gint value;
603 gint length_y = graph->priv->stop_y - graph->priv->start_y;
604 PangoRectangle ink_rect, logical_rect;
605 guint biggest = 0;
606
607 /* do y text */
608 for (i=0; i<11; i++) {
609 value = (length_y / 10) * (10 - (gfloat) i) + graph->priv->start_y;
610 text = gpm_get_axis_label (graph->priv->type_y, value);
611 pango_layout_set_text (graph->priv->layout, text, -1);
612 pango_layout_get_pixel_extents (graph->priv->layout, &ink_rect, &logical_rect);
613 if (ink_rect.width > (gint) biggest)
614 biggest = ink_rect.width;
615 g_free (text);
616 }
617 return biggest;
618}

628static void
629gpm_graph_widget_autorange_x (GpmGraphWidget *graph)
630{
631 gfloat biggest_x = G_MINFLOAT;
632 gfloat smallest_x = G_MAXFLOAT;
633 guint rounding_x = 1;
634 GPtrArray *data;
635 GpmPointObj *point;
636 guint i, j;
637 guint len = 0;
638 GPtrArray *array;
639
640 array = graph->priv->data_list;
641
642 /* find out if we have no data */
643 for (j=0; j<array->len; j++) {
644 data = g_ptr_array_index (array, j);
645 len = data->len;
646 if (len > 0)
647 break;
648 }
649
650 /* no data in any array */
651 if (len == 0) {
652 g_debug ("no data");
653 graph->priv->start_x = 0;
654 graph->priv->stop_x = 10;
655 return;
656 }
657
658 /* get the range for the graph */
659 for (j=0; j<array->len; j++) {
660 data = g_ptr_array_index (array, j);
661 for (i=0; i < data->len; i++) {
662 point = (GpmPointObj *) g_ptr_array_index (data, i);
663 if (point->x > biggest_x)
664 biggest_x = point->x;
665 if (point->x < smallest_x)
666 smallest_x = point->x;
667 }
668 }
669 g_debug ("Data range is %f<x<%f", smallest_x, biggest_x);
670 /* don't allow no difference */
671 if (biggest_x - smallest_x < 0.0001) {
672 biggest_x++;
673 smallest_x--;
674 }
675
676 if (graph->priv->type_x == GPM_GRAPH_WIDGET_TYPE_PERCENTAGE) {
677 rounding_x = 10;
678 } else if (graph->priv->type_x == GPM_GRAPH_WIDGET_TYPE_FACTOR) {
679 rounding_x = 1;
680 } else if (graph->priv->type_x == GPM_GRAPH_WIDGET_TYPE_POWER) {
681 rounding_x = 10;
682 } else if (graph->priv->type_x == GPM_GRAPH_WIDGET_TYPE_VOLTAGE) {
683 rounding_x = 1000;
684 } else if (graph->priv->type_x == GPM_GRAPH_WIDGET_TYPE_TIME) {
685 if (biggest_x-smallest_x < 150)
686 rounding_x = 150;
687 else if (biggest_x-smallest_x < 5*60)
688 rounding_x = 5 * 60;
689 else
690 rounding_x = 10 * 60;
691 }
692
693 graph->priv->start_x = egg_precision_round_down (smallest_x, rounding_x);
694 graph->priv->stop_x = egg_precision_round_up (biggest_x, rounding_x);
695
696 g_debug ("Processed(1) range is %i<x<%i",
697 graph->priv->start_x, graph->priv->stop_x);
698
699 /* if percentage, and close to the end points, then extend */
700 if (graph->priv->type_x == GPM_GRAPH_WIDGET_TYPE_PERCENTAGE) {
701 if (graph->priv->stop_x >= 90)
702 graph->priv->stop_x = 100;
703 if (graph->priv->start_x > 0 && graph->priv->start_x <= 10)
704 graph->priv->start_x = 0;
705 } else if (graph->priv->type_x == GPM_GRAPH_WIDGET_TYPE_TIME) {
706 if (graph->priv->start_x > 0 && graph->priv->start_x <= 60*10)
707 graph->priv->start_x = 0;
708 }
709
710 g_debug ("Processed range is %i<x<%i",
711 graph->priv->start_x, graph->priv->stop_x);
712}

722static void
723gpm_graph_widget_autorange_y (GpmGraphWidget *graph)
724{
725 gfloat biggest_y = G_MINFLOAT;
726 gfloat smallest_y = G_MAXFLOAT;
727 guint rounding_y = 1;
728 GPtrArray *data;
729 GpmPointObj *point;
730 guint i, j;
731 guint len = 0;
732 GPtrArray *array;
733
734 array = graph->priv->data_list;
735
736 /* find out if we have no data */
737 for (j=0; j<array->len; j++) {
738 data = g_ptr_array_index (array, j);
739 len = data->len;
740 if (len > 0)
741 break;
742 }
743
744 /* no data in any array */
745 if (len == 0) {
746 g_debug ("no data");
747 graph->priv->start_y = 0;
748 graph->priv->stop_y = 10;
749 return;
750 }
751
752 /* get the range for the graph */
753 for (j=0; j<array->len; j++) {
754 data = g_ptr_array_index (array, j);
755 for (i=0; i < data->len; i++) {
756 point = (GpmPointObj *) g_ptr_array_index (data, i);
757 if (point->y > biggest_y)
758 biggest_y = point->y;
759 if (point->y < smallest_y)
760 smallest_y = point->y;
761 }
762 }
763 g_debug ("Data range is %f<y<%f", smallest_y, biggest_y);
764 /* don't allow no difference */
765 if (biggest_y - smallest_y < 0.0001) {
766 biggest_y++;
767 smallest_y--;
768 }
769
770 if (graph->priv->type_y == GPM_GRAPH_WIDGET_TYPE_PERCENTAGE) {
771 rounding_y = 10;
772 } else if (graph->priv->type_y == GPM_GRAPH_WIDGET_TYPE_FACTOR) {
773 rounding_y = 1;
774 } else if (graph->priv->type_y == GPM_GRAPH_WIDGET_TYPE_POWER) {
775 rounding_y = 10;
776 } else if (graph->priv->type_y == GPM_GRAPH_WIDGET_TYPE_VOLTAGE) {
777 rounding_y = 1000;
778 } else if (graph->priv->type_y == GPM_GRAPH_WIDGET_TYPE_TIME) {
779 if (biggest_y-smallest_y < 150)
780 rounding_y = 150;
781 else if (biggest_y < 5*60)
782 rounding_y = 5 * 60;
783 else
784 rounding_y = 10 * 60;
785 }
786
787 graph->priv->start_y = egg_precision_round_down (smallest_y, rounding_y);
788 graph->priv->stop_y = egg_precision_round_up (biggest_y, rounding_y);
789
790 /* a factor graph always is centered around zero */
791 if (graph->priv->type_y == GPM_GRAPH_WIDGET_TYPE_FACTOR) {
792 if (abs (graph->priv->stop_y) > abs (graph->priv->start_y))
793 graph->priv->start_y = -graph->priv->stop_y;
794 else
795 graph->priv->stop_y = -graph->priv->start_y;
796 }
797
798 g_debug ("Processed(1) range is %i<y<%i",
799 graph->priv->start_y, graph->priv->stop_y);
800
801 if (graph->priv->type_y == GPM_GRAPH_WIDGET_TYPE_PERCENTAGE) {
802 if (graph->priv->stop_y >= 90)
803 graph->priv->stop_y = 100;
804 if (graph->priv->start_y > 0 && graph->priv->start_y <= 10)
805 graph->priv->start_y = 0;
806 } else if (graph->priv->type_y == GPM_GRAPH_WIDGET_TYPE_TIME) {
807 if (graph->priv->start_y <= 60*10)
808 graph->priv->start_y = 0;
809 }
810
811 g_debug ("Processed range is %i<y<%i",
812 graph->priv->start_y, graph->priv->stop_y);
813}

894static void
895gpm_graph_widget_draw_line (GpmGraphWidget *graph, cairo_t *cr)
896{
897 gfloat oldx, oldy;
898 gfloat newx, newy;
899 GPtrArray *data;
900 GPtrArray *array;
901 GpmGraphWidgetPlot plot;
902 GpmPointObj *point;
903 guint i, j;
904
905 if (graph->priv->data_list->len == 0) {
906 g_debug ("no data");
907 return;
908 }
909 cairo_save (cr);
910
911 array = graph->priv->data_list;
912
913 /* do each line */
914 for (j=0; j<array->len; j++) {
915 data = g_ptr_array_index (array, j);
916 if (data->len == 0)
917 continue;
918 plot = GPOINTER_TO_UINT (g_ptr_array_index (graph->priv->plot_list, j));
919
920 /* get the very first point so we can work out the old */
921 point = (GpmPointObj *) g_ptr_array_index (data, 0);
922 oldx = 0;
923 oldy = 0;
924 gpm_graph_widget_get_pos_on_graph (graph, point->x, point->y, &oldx, &oldy);
925 if (plot == GPM_GRAPH_WIDGET_PLOT_POINTS || plot == GPM_GRAPH_WIDGET_PLOT_BOTH)
926 gpm_graph_widget_draw_dot (cr, oldx, oldy, point->color);
927
928 for (i=1; i < data->len; i++) {
929 point = (GpmPointObj *) g_ptr_array_index (data, i);
930
931 gpm_graph_widget_get_pos_on_graph (graph, point->x, point->y, &newx, &newy);
932
933 /* ignore white lines */
934 if (point->color == 0xffffff) {
935 oldx = newx;
936 oldy = newy;
937 continue;
938 }
939
940 /* draw line */
941 if (plot == GPM_GRAPH_WIDGET_PLOT_LINE || plot == GPM_GRAPH_WIDGET_PLOT_BOTH) {
942 cairo_move_to (cr, oldx, oldy);
943 cairo_line_to (cr, newx, newy);
944 cairo_set_line_width (cr, 1.5);
945 gpm_graph_widget_set_color (cr, point->color);
946 cairo_stroke (cr);
947 }
948
949 /* draw data dot */
950 if (plot == GPM_GRAPH_WIDGET_PLOT_POINTS || plot == GPM_GRAPH_WIDGET_PLOT_BOTH)
951 gpm_graph_widget_draw_dot (cr, newx, newy, point->color);
952
953 /* save old */
954 oldx = newx;
955 oldy = newy;
956 }
957 }
958
959 cairo_restore (cr);
960}
