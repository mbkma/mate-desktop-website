312static void
313instance_get_property( GObject *object, guint property_id, GValue *value, GParamSpec *spec )
314{
315 NAImportMode *self;
316
317 g_return_if_fail( NA_IS_IMPORT_MODE( object ));
318 self = NA_IMPORT_MODE( object );
319
320 if( !self->private->dispose_has_run ){
321
322 switch( property_id ){
323 case NA_IMPORT_PROP_MODE_ID:
324 g_value_set_string( value, self->private->mode );
325 break;
326
327 case NA_IMPORT_PROP_LABEL_ID:
328 g_value_set_string( value, self->private->label );
329 break;
330
331 case NA_IMPORT_PROP_DESCRIPTION_ID:
332 g_value_set_string( value, self->private->description );
333 break;
334
335 case NA_IMPORT_PROP_IMAGE_ID:
336 g_value_set_pointer( value, self->private->image );
337 break;
338
339 default:
340 G_OBJECT_WARN_INVALID_PROPERTY_ID( object, property_id, spec );
341 break;
342 }
343 }

346static void
347instance_set_property( GObject *object, guint property_id, const GValue *value, GParamSpec *spec )
348{
349 NAImportMode *self;
350
351 g_return_if_fail( NA_IS_IMPORT_MODE( object ));
352 self = NA_IMPORT_MODE( object );
353
354 if( !self->private->dispose_has_run ){
355
356 switch( property_id ){
357 case NA_IMPORT_PROP_MODE_ID:
358 g_free( self->private->mode );
359 self->private->mode = g_value_dup_string( value );
360 break;
361
362 case NA_IMPORT_PROP_LABEL_ID:
363 g_free( self->private->label );
364 self->private->label = g_value_dup_string( value );
365 break;
366
367 case NA_IMPORT_PROP_DESCRIPTION_ID:
368 g_free( self->private->description );
369 self->private->description = g_value_dup_string( value );
370 break;
371
372 case NA_IMPORT_PROP_IMAGE_ID:
373 self->private->image = g_value_get_pointer( value );
374 break;
375
376 default:
377 G_OBJECT_WARN_INVALID_PROPERTY_ID( object, property_id, spec );
378 break;
379 }
380 }
