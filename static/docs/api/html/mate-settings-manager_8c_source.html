96static void
97maybe_activate_plugin (MateSettingsPluginInfo *info,
98 MateSettingsManager *manager)
99{
100 if (mate_settings_plugin_info_get_enabled (info)) {
101 int plugin_priority;
102 plugin_priority = mate_settings_plugin_info_get_priority (info);
103
104 if (manager->priv->load_init_flag == PLUGIN_LOAD_ALL ||
105 (manager->priv->load_init_flag == PLUGIN_LOAD_INIT && plugin_priority <= manager->priv->init_load_priority) ||
106 (manager->priv->load_init_flag == PLUGIN_LOAD_DEFER && plugin_priority > manager->priv->init_load_priority)) {
107 gboolean res;
108 res = mate_settings_plugin_info_activate (info);
109 if (res) {
110 g_debug ("Plugin %s: active", mate_settings_plugin_info_get_location (info));
111 } else {
112 g_debug ("Plugin %s: activation failed", mate_settings_plugin_info_get_location (info));
113 }
114 } else {
115 g_debug ("Plugin %s: loading deferred or previously loaded", mate_settings_plugin_info_get_location (info));
116 }
117 } else {
118 g_debug ("Plugin %s: inactive", mate_settings_plugin_info_get_location (info));
119 }
120}

122static gint
123compare_location (MateSettingsPluginInfo *a,
124 MateSettingsPluginInfo *b)
125{
126 const char *loc_a;
127 const char *loc_b;
128
129 loc_a = mate_settings_plugin_info_get_location (a);
130 loc_b = mate_settings_plugin_info_get_location (b);
131
132 if (loc_a == NULL || loc_b == NULL) {
133 return -1;
134 }
135
136 return strcmp (loc_a, loc_b);

177static gboolean
178is_schema (const char *schema)
179{
180 GSettingsSchemaSource *source = NULL;
181 gchar **non_relocatable = NULL;
182 gchar **relocatable = NULL;
183 gboolean in_schema;
184
185 source = g_settings_schema_source_get_default ();
186 if (!source)
187 return FALSE;
188
189 g_settings_schema_source_list_schemas (source, TRUE, &non_relocatable, &relocatable);
190
191 in_schema = (is_item_in_schema (non_relocatable, schema) ||
192 is_item_in_schema (relocatable, schema));
193
194 g_strfreev (non_relocatable);
195 g_strfreev (relocatable);
196
197 return in_schema;
198}

200static void
201_load_file (MateSettingsManager *manager,
202 const char *filename)
203{
204 MateSettingsPluginInfo *info;
205 char *schema;
206 GSList *l;
207
208 g_debug ("Loading plugin: %s", filename);
209 mate_settings_profile_start ("%s", filename);
210
211 info = mate_settings_plugin_info_new_from_file (filename);
212 if (info == NULL) {
213 goto out;
214 }
215
216 l = g_slist_find_custom (manager->priv->plugins,
217 info,
218 (GCompareFunc) compare_location);
219 if (l != NULL) {
220 goto out;
221 }
222
223 schema = g_strdup_printf ("%s.plugins.%s",
224 DEFAULT_SETTINGS_PREFIX,
225 mate_settings_plugin_info_get_location (info));
226
227 /* Ignore unknown schemas or else we'll assert */
228 if (is_schema (schema)) {
229 manager->priv->plugins = g_slist_prepend (manager->priv->plugins,
230 g_object_ref (info));
231
232 g_signal_connect (info, "activated",
233 G_CALLBACK (on_plugin_activated), manager);
234 g_signal_connect (info, "deactivated",
235 G_CALLBACK (on_plugin_deactivated), manager);
236
237 /* Also sets priority for plugins */
238 mate_settings_plugin_info_set_schema (info, schema);
239 } else {
240 g_warning ("Ignoring unknown module '%s'", schema);
241 }
242
243 g_free (schema);
244
245 out:
246 if (info != NULL) {
247 g_object_unref (info);
248 }
249
250 mate_settings_profile_end ("%s", filename);
251}

253static void
254_load_dir (MateSettingsManager *manager,
255 const char *path)
256{
257 GError *error;
258 GDir *d;
259 const char *name;
260
261 g_debug ("Loading settings plugins from dir: %s", path);
262 mate_settings_profile_start (NULL);
263
264 error = NULL;
265 d = g_dir_open (path, 0, &error);
266 if (d == NULL) {
267 g_warning ("%s", error->message);
268 g_error_free (error);
269 return;
270 }
271
272 while ((name = g_dir_read_name (d))) {
273 char *filename;
274
275 if (!g_str_has_suffix (name, PLUGIN_EXT)) {
276 continue;
277 }
278
279 filename = g_build_filename (path, name, NULL);
280 if (g_file_test (filename, G_FILE_TEST_IS_REGULAR)) {
281 _load_file (manager, filename);
282 }
283 g_free (filename);
284 }
285
286 g_dir_close (d);
287
288 mate_settings_profile_end (NULL);
289}

291static void
292_load_all (MateSettingsManager *manager)
293{
294 mate_settings_profile_start (NULL);
295
296 /* load system plugins */
297 _load_dir (manager, MATE_SETTINGS_PLUGINDIR G_DIR_SEPARATOR_S);
298
299 manager->priv->plugins = g_slist_sort (manager->priv->plugins, (GCompareFunc) compare_priority);
300 g_slist_foreach (manager->priv->plugins, (GFunc) maybe_activate_plugin, manager);
301 mate_settings_profile_end (NULL);
302}

336static void
337handle_method_call (GDBusConnection *connection,
338 const gchar *sender,
339 const gchar *object_path,
340 const gchar *interface_name,
341 const gchar *method_name,
342 GVariant *parameters,
343 GDBusMethodInvocation *invocation,
344 gpointer user_data)
345{
346 MateSettingsManager *manager = (MateSettingsManager *) user_data;
347 GError *error = NULL;
348
349 g_debug ("Calling method '%s' for settings daemon", method_name);
350
351 if (g_strcmp0 (method_name, "Awake") == 0) {
352 if (mate_settings_manager_awake (manager, &error) == FALSE)
353 g_dbus_method_invocation_return_gerror (invocation, error);
354 else
355 g_dbus_method_invocation_return_value (invocation, NULL);
356 } else if (g_strcmp0 (method_name, "Start") == 0) {
357 if (mate_settings_manager_start (manager, PLUGIN_LOAD_INIT, &error) == FALSE)
358 g_dbus_method_invocation_return_gerror (invocation, error);
359 else
360 g_dbus_method_invocation_return_value (invocation, NULL);
361 }
362}

369static void
370on_bus_gotten (GObject *source_object,
371 GAsyncResult *res,
372 MateSettingsManager *manager)
373{
374 GDBusConnection *connection;
375 GError *error = NULL;
376
377 connection = g_bus_get_finish (res, &error);
378 if (connection == NULL) {
379 if (!g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
380 g_warning ("Could not get session bus: %s", error->message);
381 g_error_free (error);
382 return;
383 }
384 manager->priv->connection = connection;
385
386 manager->priv->dbus_register_object_id = g_dbus_connection_register_object (connection,
387 MSD_MANAGER_DBUS_PATH,
388 manager->priv->introspection_data->interfaces[0],
389 &interface_vtable,
390 manager,
391 NULL,
392 NULL);
393 g_assert (manager->priv->dbus_register_object_id > 0);
394}

410gboolean
411mate_settings_manager_start (MateSettingsManager *manager,
412 gint load_init_flag,
413 GError **error)
414{
415 gboolean ret;
416
417 g_debug ("Starting settings manager");
418
419 ret = FALSE;
420
421 mate_settings_profile_start (NULL);
422
423 if (!g_module_supported ()) {
424 g_warning ("mate-settings-daemon is not able to initialize the plugins.");
425 g_set_error (error,
426 MATE_SETTINGS_MANAGER_ERROR,
427 MATE_SETTINGS_MANAGER_ERROR_GENERAL,
428 "Plugins not supported");
429
430 goto out;
431 }
432
433 manager->priv->load_init_flag = load_init_flag;
434 _load_all (manager);
435 mate_settings_profile_end ("initializing plugins");
436
437 ret = TRUE;
438 out:
439 mate_settings_profile_end (NULL);
440
441 return ret;
442}
