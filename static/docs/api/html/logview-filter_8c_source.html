64static void
65logview_filter_set_property (GObject *object, guint prop_id, const GValue *value, GParamSpec *pspec)
66{
67 LogviewFilterPrivate* priv = LOGVIEW_FILTER (object)->priv;
68
69 switch (prop_id) {
70 case PROP_NAME:
71 priv->name = g_value_dup_string (value);
72 break;
73 case PROP_REGEX: {
74 GError* err;
75 const gchar* regex;
76
77 err = NULL;
78
79 regex = g_value_get_string (value);
80 priv->regex = g_regex_new (regex, 0, 0, &err);
81
82 if (err) {
83 g_regex_unref (priv->regex);
84 priv->regex = NULL;
85 g_warning ("Couldn't create GRegex object: %s", err->message);
86 g_error_free (err);
87 }
88
89 break;
90 }
91 case PROP_TEXTTAG: {
92 if (priv->tag) {
93 g_object_unref (priv->tag);
94 }
95
96 priv->tag = g_value_dup_object (value);
97 break;
98 }
99 default:
100 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
101 break;
102 }
103}

171gboolean
172logview_filter_filter (LogviewFilter *filter, const gchar *line)
173{
174 GMatchInfo* match_info;
175 LogviewFilterPrivate* priv;
176 gboolean retval;
177
178 g_return_val_if_fail (LOGVIEW_IS_FILTER (filter), FALSE);
179 g_return_val_if_fail (line != NULL, FALSE);
180
181 priv = filter->priv;
182
183 g_regex_match (priv->regex, line, 0, &match_info);
184
185 retval = g_match_info_matches (match_info);
186
187 g_match_info_free (match_info);
188
189 return retval;
190}
