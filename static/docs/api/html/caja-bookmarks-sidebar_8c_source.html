92static gboolean
93is_built_in_bookmark (CajaFile *file)
94{
95 gboolean built_in;
96 gint idx;
97
98 built_in = FALSE;
99
100 for (idx = 0; idx < G_USER_N_DIRECTORIES; idx++) {
101 /* PUBLIC_SHARE and TEMPLATES are not in our built-in list */
102 if (caja_file_is_user_special_directory (file, idx)) {
103 if (idx != G_USER_DIRECTORY_PUBLIC_SHARE && idx != G_USER_DIRECTORY_TEMPLATES) {
104 built_in = TRUE;
105 }
106
107 break;
108 }
109 }
110
111 return built_in;
112}

114static void
115update_bookmarks (CajaBookmarksSidebar *sidebar)
116{
117 GtkListStore *store;
118 GtkTreeSelection *selection;
119 GtkTreeIter iter;
120 CajaBookmark *bookmark = NULL;
121 int bookmark_count = 0;
122 int index = 0;
123 GFile *root;
124 CajaFile *file;
125 char *bookmark_uri;
126 cairo_surface_t *surface = NULL;
127 char *name;
128 gboolean bookmarks_list_empty = TRUE;
129
130 store = GTK_LIST_STORE (gtk_tree_view_get_model (sidebar->tree_view));
131
132 gtk_list_store_clear (store);
133
134 selection = GTK_TREE_SELECTION (gtk_tree_view_get_selection (sidebar->tree_view));
135
136 bookmark_count = caja_bookmark_list_length (sidebar->bookmarks);
137
138 for (index = 0; index < bookmark_count; index++)
139 {
140 bookmark = caja_bookmark_list_item_at (sidebar->bookmarks, index);
141
142 if (caja_bookmark_uri_known_not_to_exist (bookmark)) {
143 continue;
144 }
145
146 bookmark_uri = caja_bookmark_get_uri (bookmark);
147
148 root = caja_bookmark_get_location (bookmark);
149 file = caja_file_get (root);
150
151 if (is_built_in_bookmark (file)) {
152 g_object_unref (root);
153 caja_file_unref (file);
154 continue;
155 }
156
157 bookmarks_list_empty = FALSE;
158 surface = caja_bookmark_get_surface (bookmark, GTK_ICON_SIZE_MENU);
159 name = caja_bookmark_get_name (bookmark);
160 gtk_list_store_append (store, &iter);
161 gtk_list_store_set (store, &iter,
162 BOOKMARKS_SIDEBAR_COLUMN_ICON, surface,
163 BOOKMARKS_SIDEBAR_COLUMN_NAME, name,
164 BOOKMARKS_SIDEBAR_COLUMN_STYLE, PANGO_STYLE_NORMAL,
165 BOOKMARKS_SIDEBAR_COLUMN_ICON_VISIBLE, TRUE,
166 BOOKMARKS_SIDEBAR_COLUMN_BOOKMARK, bookmark,
167 -1);
168
169 /* Select the bookmark if we're in the directory the bookmark points to. */
170 if (g_strcmp0 (bookmark_uri, sidebar->current_uri) == 0)
171 gtk_tree_selection_select_iter (selection, &iter);
172
173 g_object_unref (root);
174 caja_file_unref (file);
175 g_free (bookmark_uri);
176
177 if (surface != NULL)
178 {
179 cairo_surface_destroy (surface);
180 }
181 g_free (name);
182 }
183
184 /* If the user has no user-defined bookmarks, then add one row to the list
185 * which says that no bookmarks have been defined, rather than leaving the
186 * user scratching their head wondering why the list is blank.
187 */
188 if (bookmarks_list_empty)
189 {
190 gtk_list_store_append (store, &iter);
191 gtk_list_store_set (store, &iter,
192 BOOKMARKS_SIDEBAR_COLUMN_ICON, NULL,
193 BOOKMARKS_SIDEBAR_COLUMN_NAME, _("No bookmarks defined"),
194 BOOKMARKS_SIDEBAR_COLUMN_STYLE, PANGO_STYLE_ITALIC,
195 BOOKMARKS_SIDEBAR_COLUMN_ICON_VISIBLE, FALSE,
196 BOOKMARKS_SIDEBAR_COLUMN_BOOKMARK, NULL,
197 -1);
198 gtk_tree_selection_select_iter (selection, &iter);
199 }
200}

202static void
203open_selected_item (CajaBookmarksSidebar *sidebar,
204 GtkTreePath *path,
205 CajaWindowOpenFlags flags)
206{
207 CajaWindowSlotInfo *slot;
208 GtkTreeModel *model;
209 GtkTreeIter iter;
210 CajaBookmark *bookmark;
211 GFile *location;
212
213 model = gtk_tree_view_get_model (sidebar->tree_view);
214
215 if (!gtk_tree_model_get_iter (model, &iter, path))
216 return;
217
218 gtk_tree_model_get (model, &iter, BOOKMARKS_SIDEBAR_COLUMN_BOOKMARK,
219 &bookmark, -1);
220
221 if (bookmark == NULL)
222 return;
223
224 /* Navigate to the clicked location. */
225 location = caja_bookmark_get_location (CAJA_BOOKMARK (bookmark));
226
227 slot = caja_window_info_get_active_slot (sidebar->window);
228
229 caja_window_slot_info_open_location (slot, location,
230 CAJA_WINDOW_OPEN_ACCORDING_TO_MODE,
231 flags, NULL);
232
233 g_object_unref (location);
234}

250static gboolean
251button_press_event_callback (GtkWidget *widget,
252 GdkEventButton *event,
253 gpointer user_data)
254{
255 CajaBookmarksSidebar *sidebar;
256 GtkTreePath *path;
257
258 int open_flags = 0;
259
260 if (event->type == GDK_BUTTON_PRESS)
261 {
265 if (event->button == GDK_BUTTON_MIDDLE)
266 open_flags = CAJA_WINDOW_OPEN_FLAG_NEW_TAB;
267
273 else
274 return FALSE;
275 }
276
277 sidebar = CAJA_BOOKMARKS_SIDEBAR (user_data);
278 g_assert (sidebar->tree_view == GTK_TREE_VIEW (widget));
279
280 if (gtk_tree_view_get_path_at_pos (sidebar->tree_view,
281 event->x, event->y,
282 &path, NULL, NULL, NULL))
283 {
284 open_selected_item (sidebar,
285 path,
286 open_flags);
287
288 gtk_tree_path_free (path);
289 }
290
291 /*
292 * If the middle button was pressed, don't let the event bubble up to the
293 * row_activated_callback() (above), because we do not want the row to
294 * be selected: The location of the bookmark was opened in a new tab, not
295 * the current tab.
296 */
297 return TRUE;
298}

300static void
301loading_uri_callback (CajaWindowInfo *window,
302 char *location,
303 CajaBookmarksSidebar *sidebar)
304{
305 GtkTreeModel *model;
306 GtkTreeIter iter;
307 gboolean valid;
308 CajaBookmark *bookmark;
309 char *uri;
310
311 if (strcmp (sidebar->current_uri, location) != 0)
312 {
313 GtkTreeSelection *selection;
314
315 g_free (sidebar->current_uri);
316 sidebar->current_uri = g_strdup (location);
317
318 /* set selection if any place matches location */
319 selection = gtk_tree_view_get_selection (sidebar->tree_view);
320 gtk_tree_selection_unselect_all (selection);
321
322 model = gtk_tree_view_get_model (sidebar->tree_view);
323
324 valid = gtk_tree_model_get_iter_first (model, &iter);
325
326 while (valid)
327 {
328 gtk_tree_model_get (model, &iter,
329 BOOKMARKS_SIDEBAR_COLUMN_BOOKMARK, &bookmark,
330 -1);
331 if (bookmark && ((uri = caja_bookmark_get_uri (bookmark)) != NULL))
332 {
333 if (strcmp (uri, location) == 0)
334 {
335 g_free (uri);
336 gtk_tree_selection_select_iter (selection, &iter);
337 break;
338 }
339 g_free (uri);
340 }
341 valid = gtk_tree_model_iter_next (model, &iter);
342 }
343 }
344}

346/*
347 * If no bookmarks have been defined by the user, then we create one row
348 * in the bookmark list sidebar which says "No bookmarks defined". We
349 * do not want the user to be able to select the row, however, so if the
350 * bookmark is NULL, then refuse selection; if the bookmark is not NULL,
351 * allow selection.
352*/
353static gboolean
354is_row_selectable (GtkTreeSelection *selection,
355 GtkTreeModel *model,
356 GtkTreePath *path,
357 gboolean path_currently_selected,
358 gpointer data)
359{
360 GtkTreeIter iter;
361 CajaBookmark *bookmark;
362
363 if (!gtk_tree_model_get_iter (model, &iter, path))
364 return FALSE;
365
366 gtk_tree_model_get (model, &iter, BOOKMARKS_SIDEBAR_COLUMN_BOOKMARK,
367 &bookmark, -1);
368
369 if (bookmark == NULL)
370 return FALSE;
371 else
372 return TRUE;
373}

375static void
376caja_bookmarks_sidebar_init (CajaBookmarksSidebar *sidebar)
377{
378 GtkTreeView *tree_view;
379 GtkTreeViewColumn *col;
380 GtkCellRenderer *cell;
381 GtkListStore *store;
382 GtkTreeSelection *selection;
383
384 tree_view = GTK_TREE_VIEW (gtk_tree_view_new ());
385 gtk_tree_view_set_headers_visible (tree_view, FALSE);
386 gtk_widget_show (GTK_WIDGET (tree_view));
387
388 col = GTK_TREE_VIEW_COLUMN (gtk_tree_view_column_new ());
389
390 cell = gtk_cell_renderer_pixbuf_new ();
391 gtk_tree_view_column_pack_start (col, cell, FALSE);
392 gtk_tree_view_column_set_attributes (col, cell,
393 "surface", BOOKMARKS_SIDEBAR_COLUMN_ICON,
394 "visible", BOOKMARKS_SIDEBAR_COLUMN_ICON_VISIBLE,
395 NULL);
396
397 cell = gtk_cell_renderer_text_new ();
398 gtk_tree_view_column_pack_start (col, cell, TRUE);
399 gtk_tree_view_column_set_attributes (col, cell,
400 "text", BOOKMARKS_SIDEBAR_COLUMN_NAME,
401 "style", BOOKMARKS_SIDEBAR_COLUMN_STYLE,
402 NULL);
403
404 gtk_tree_view_column_set_fixed_width (col, CAJA_ICON_SIZE_SMALLER);
405 gtk_tree_view_append_column (tree_view, col);
406
407 store = gtk_list_store_new (BOOKMARKS_SIDEBAR_COLUMN_COUNT,
408 CAIRO_GOBJECT_TYPE_SURFACE,
409 G_TYPE_STRING,
410 PANGO_TYPE_STYLE,
411 G_TYPE_BOOLEAN,
412 CAJA_TYPE_BOOKMARK);
413
414 gtk_tree_view_set_model (tree_view, GTK_TREE_MODEL (store));
415 g_object_unref (store);
416
417 gtk_scrolled_window_set_policy (GTK_SCROLLED_WINDOW (sidebar),
418 GTK_POLICY_AUTOMATIC,
419 GTK_POLICY_AUTOMATIC);
420
421 gtk_scrolled_window_set_hadjustment (GTK_SCROLLED_WINDOW (sidebar), NULL);
422 gtk_scrolled_window_set_vadjustment (GTK_SCROLLED_WINDOW (sidebar), NULL);
423 gtk_scrolled_window_set_shadow_type (GTK_SCROLLED_WINDOW (sidebar), GTK_SHADOW_IN);
424 gtk_scrolled_window_set_overlay_scrolling (GTK_SCROLLED_WINDOW (sidebar), FALSE);
425
426 gtk_container_add (GTK_CONTAINER (sidebar), GTK_WIDGET (tree_view));
427 gtk_widget_show (GTK_WIDGET (sidebar));
428
429 sidebar->tree_view = tree_view;
430
431 selection = gtk_tree_view_get_selection (tree_view);
432 gtk_tree_selection_set_mode (selection, GTK_SELECTION_SINGLE);
433
434 /* Please see the comment associated with the function is_row_selectable()
435 * (above).
436 */
437 gtk_tree_selection_set_select_function (selection,
438 is_row_selectable,
439 NULL,
440 NULL);
441
442 g_signal_connect_object
443 (tree_view, "row_activated",
444 G_CALLBACK (row_activated_callback), sidebar, 0);
445
446 g_signal_connect (tree_view, "button-press-event",
447 G_CALLBACK (button_press_event_callback), sidebar);
448
449 eel_gtk_tree_view_set_activate_on_single_click (sidebar->tree_view, TRUE);
450}
