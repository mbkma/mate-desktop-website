34static void
35set_gdk_window_struts (GdkWindow *window,
36 int left,
37 int right,
38 int top,
39 int bottom)
40{
41 long vals[12];
42
43 vals[0] = left;
44 vals[1] = right;
45 vals[2] = top;
46 vals[3] = bottom;
47 vals[4] = 000;
48 vals[5] = 400;
49 vals[6] = 200;
50 vals[7] = 600;
51 vals[8] = 76;
52 vals[9] = 676;
53 vals[10] = 200;
54 vals[11] = 800;

57 GDK_WINDOW_XID (window),
58 XInternAtom (GDK_WINDOW_XDISPLAY (window),
59 "_NET_WM_STRUT_PARTIAL", False),
60 XA_CARDINAL, 32, PropModeReplace,
61 (guchar *)vals, 12);
62}

217int
218main (int argc, char **argv)
219{
220 GList *list;
221 GdkPixbuf *pixbuf;
222 GError *err;
223
224 gtk_init (&argc, &argv);
225
226 err = NULL;
227 pixbuf = gdk_pixbuf_new_from_file (MARCO_ICON_DIR"/marco-window-demo.png",
228 &err);
229 if (pixbuf)
230 {
231 list = g_list_prepend (NULL, pixbuf);
232
233 gtk_window_set_default_icon_list (list);
234 g_list_free (list);
235 g_object_unref (G_OBJECT (pixbuf));
236 }
237 else
238 {
239 g_printerr ("Could not load icon: %s\n", err->message);
240 g_error_free (err);
241 }
242
243 do_appwindow (NULL, NULL, NULL);
244
245 gtk_main ();
246
247 return 0;
248}

255static void
256make_dialog (GtkWidget *parent,
257 int depth)
258{
259 GtkWidget *dialog;
260 char *str;
261
262 dialog = gtk_message_dialog_new (parent ? GTK_WINDOW (parent) : NULL,
263 GTK_DIALOG_DESTROY_WITH_PARENT,
264 GTK_MESSAGE_INFO,
265 GTK_BUTTONS_CLOSE,
266 parent ? "Here is a dialog %d" :
267 "Here is a dialog %d with no transient parent",
268 depth);
269
270 str = g_strdup_printf ("%d dialog", depth);
271 gtk_window_set_title (GTK_WINDOW (dialog), str);
272 g_free (str);
273
274 gtk_dialog_add_button (GTK_DIALOG (dialog),
275 "Open child dialog",
276 GTK_RESPONSE_ACCEPT);
277
278 /* Close dialog on user response */
279 g_signal_connect (G_OBJECT (dialog),
280 "response",
281 G_CALLBACK (response_cb),
282 NULL);
283
284 g_object_set_data (G_OBJECT (dialog), "depth",
285 GINT_TO_POINTER (depth));
286
287 gtk_widget_show (dialog);
288}
289
290static void
291response_cb (GtkDialog *dialog,
292 int response_id,
293 void *data)
294{
295 switch (response_id)
296 {
297 case GTK_RESPONSE_ACCEPT:
298 make_dialog (GTK_WIDGET (dialog),
299 GPOINTER_TO_INT (g_object_get_data (G_OBJECT (dialog),
300 "depth")) + 1);
301 break;
302
303 default:
304 gtk_widget_destroy (GTK_WIDGET (dialog));
305 break;
306 }
307}

554static void
555make_dock (int type)
556{
557 GtkWidget *window;
558 GtkWidget *image;
559 GtkWidget *box;
560 GtkWidget *button;
561
562 g_return_if_fail (type != DOCK_ALL);
563
564 box = NULL;
565 switch (type)
566 {
567 case DOCK_LEFT:
568 case DOCK_RIGHT:
569 box = gtk_box_new (GTK_ORIENTATION_VERTICAL, 0);
570 break;
571 case DOCK_TOP:
572 case DOCK_BOTTOM:
573 box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);
574 break;
575 case DOCK_ALL:
576 break;
577 }
578
579 window = gtk_window_new (GTK_WINDOW_TOPLEVEL);
580 set_gtk_window_type (GTK_WINDOW (window), "_NET_WM_WINDOW_TYPE_DOCK");
581
582 image = gtk_image_new_from_icon_name ("dialog-information", GTK_ICON_SIZE_DIALOG);
583 gtk_box_pack_start (GTK_BOX (box), image, FALSE, FALSE, 0);
584
585 gtk_box_pack_start (GTK_BOX (box), focus_label (window), FALSE, FALSE, 0);
586
587 button = gtk_button_new_with_label ("Close");
588 gtk_box_pack_start (GTK_BOX (box), button, FALSE, FALSE, 0);
589
590 g_signal_connect_swapped (G_OBJECT (button), "clicked",
591 G_CALLBACK (gtk_widget_destroy), window);
592
593 gtk_container_add (GTK_CONTAINER (window), box);
594
595#define DOCK_SIZE 48
596 switch (type)
597 {
598 case DOCK_LEFT:
599 gtk_widget_set_size_request (window, DOCK_SIZE, 400);
600 gtk_window_move (GTK_WINDOW (window), 0, 000);
601 set_gtk_window_struts (window, DOCK_SIZE, 0, 0, 0);
602 gtk_window_set_title (GTK_WINDOW (window), "LeftDock");
603 break;
604 case DOCK_RIGHT:
605 gtk_widget_set_size_request (window, DOCK_SIZE, 400);
606 gtk_window_move (GTK_WINDOW (window), WidthOfScreen (gdk_x11_screen_get_xscreen (gdk_screen_get_default ())) - DOCK_SIZE, 200);
607 set_gtk_window_struts (window, 0, DOCK_SIZE, 0, 0);
608 gtk_window_set_title (GTK_WINDOW (window), "RightDock");
609 break;
610 case DOCK_TOP:
611 gtk_widget_set_size_request (window, 600, DOCK_SIZE);
612 gtk_window_move (GTK_WINDOW (window), 76, 0);
613 set_gtk_window_struts (window, 0, 0, DOCK_SIZE, 0);
614 gtk_window_set_title (GTK_WINDOW (window), "TopDock");
615 break;
616 case DOCK_BOTTOM:
617 gtk_widget_set_size_request (window, 600, DOCK_SIZE);
618 gtk_window_move (GTK_WINDOW (window), 200, HeightOfScreen (gdk_x11_screen_get_xscreen (gdk_screen_get_default ())) - DOCK_SIZE);
619 set_gtk_window_struts (window, 0, 0, 0, DOCK_SIZE);
620 gtk_window_set_title (GTK_WINDOW (window), "BottomDock");
621 break;
622 case DOCK_ALL:
623 break;
624 }
625
626 gtk_widget_show_all (window);
627}

629static void
630dock_cb (GSimpleAction *action,
631 GVariant *parameter,
632 gpointer callback_data)
633{
634 guint callback_action;
635 const gchar *name;
636
637 g_object_get (G_OBJECT (action), "name", &name, NULL);
638
639 if (!g_strcmp0 (name, "top-dock"))
640 callback_action = DOCK_TOP;
641 else if (!g_strcmp0 (name, "bottom-dock"))
642 callback_action = DOCK_BOTTOM;
643 else if (!g_strcmp0 (name, "left-dock"))
644 callback_action = DOCK_LEFT;
645 else if (!g_strcmp0 (name, "right-dock"))
646 callback_action = DOCK_RIGHT;
647 else if (!g_strcmp0 (name, "all-docks"))
648 callback_action = DOCK_ALL;
649 else
650 return;
651
652 if (callback_action == DOCK_ALL)
653 {
654 make_dock (DOCK_TOP);
655 make_dock (DOCK_BOTTOM);
656 make_dock (DOCK_LEFT);
657 make_dock (DOCK_RIGHT);
658 }
659 else
660 {
661 make_dock (callback_action);
662 }
663}

665static void
666override_background_color (GtkWidget *widget,

668{
669 gchar *css;

671
672 provider = gtk_css_provider_new ();
673

675 gdk_rgba_to_string (rgba));
676 gtk_css_provider_load_from_data (provider, css, -1, NULL);

678
679 gtk_style_context_add_provider (gtk_widget_get_style_context (widget),
680 GTK_STYLE_PROVIDER (provider),
681 GTK_STYLE_PROVIDER_PRIORITY_APPLICATION);
682 g_object_unref (provider);

724static void
725toggle_aspect_ratio (GSimpleAction *action,
726 GVariant *parameter,
727 gpointer data)
728{
729 GtkWidget *window;
730 GdkGeometry geom;
731 GtkWidget *widget = GTK_WIDGET (data);
732
733 if (aspect_on)
