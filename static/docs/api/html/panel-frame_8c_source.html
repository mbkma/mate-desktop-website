41 gint *minimal_width,
42 gint *natural_width)
43{
44 PanelFrame *frame = (PanelFrame *) widget;
45 GtkBin *bin = (GtkBin *) widget;
46 GtkStyleContext *context;
47 GtkWidget *child;
48 GtkBorder padding;
49 int border_width;
50
51 context = gtk_widget_get_style_context (widget);

53 border_width = gtk_container_get_border_width (GTK_CONTAINER (widget));
54

61

63 *natural_width += border_width;
64
65 if (frame->edges & PANEL_EDGE_LEFT) {
66 *minimal_width += padding.left;
67 *natural_width += padding.left;
68 }
69
70 if (frame->edges & PANEL_EDGE_RIGHT) {
71 *minimal_width += padding.right;
72 *natural_width += padding.right;

112static void
113panel_frame_size_allocate (GtkWidget *widget,
114 GtkAllocation *allocation)
115{
116 PanelFrame *frame = (PanelFrame *) widget;
117 GtkBin *bin = (GtkBin *) widget;
118 GtkStyleContext *context;
119 GtkBorder padding;
120 GtkAllocation child_allocation;
121 GtkAllocation child_allocation_current;
122 GtkWidget *child;
123 int border_width;
124
125 gtk_widget_set_allocation (widget, allocation);
126
127 context = gtk_widget_get_style_context (widget);
128 gtk_style_context_get_padding (context, gtk_widget_get_state_flags (widget), &padding);
129 border_width = gtk_container_get_border_width (GTK_CONTAINER (widget));
130
131 child_allocation.x = allocation->x + border_width;
132 child_allocation.y = allocation->y + border_width;
133 child_allocation.width = allocation->width - 2 * border_width;
134 child_allocation.height = allocation->height - 2 * border_width;
135
136 if (frame->edges & PANEL_EDGE_LEFT) {
137 child_allocation.x += padding.left;
138 child_allocation.width -= padding.left;
139 }
140
141 if (frame->edges & PANEL_EDGE_TOP) {
142 child_allocation.y += padding.top;
143 child_allocation.height -= padding.top;
144 }
145
146 if (frame->edges & PANEL_EDGE_RIGHT)
147 child_allocation.width -= padding.left;
148
149 if (frame->edges & PANEL_EDGE_BOTTOM)
150 child_allocation.height -= padding.top;
151
152 child = gtk_bin_get_child (bin);
153 gtk_widget_get_allocation (child, &child_allocation_current);
154
155 if (gtk_widget_get_mapped (widget) &&
156 (child_allocation.x != child_allocation_current.x ||
157 child_allocation.y != child_allocation_current.y ||
158 child_allocation.width != child_allocation_current.width ||
159 child_allocation.height != child_allocation_current.height))
160 gdk_window_invalidate_rect (gtk_widget_get_window (widget), allocation, FALSE);
161
162 if (child && gtk_widget_get_visible (child))
163 gtk_widget_size_allocate (child, &child_allocation);
164}

166void
167panel_frame_draw (GtkWidget *widget,
168 cairo_t *cr,
169 PanelFrameEdge edges)
170{
171 PanelFrame *frame = (PanelFrame *) widget;
172 GtkStyleContext *context;
173 GtkStateFlags state;
174 GdkRGBA *bg;
175 GdkRGBA dark, light;
176 GtkBorder padding;
177 int x, y, width, height;
178
179 if (edges == PANEL_EDGE_NONE)
180 return;
181
182 context = gtk_widget_get_style_context (widget);
183 state = gtk_widget_get_state_flags (widget);
184
185 width = gtk_widget_get_allocated_width (widget);
186 height = gtk_widget_get_allocated_height (widget);
187
188 gtk_style_context_get (context, state,
189 "background-color", &bg,
190 NULL);
191
192 gtk_style_shade (bg, &dark, 0.7);
193 gtk_style_shade (bg, &light, 1.3);
194
195 gtk_style_context_get_padding (context, state, &padding);
196
197 /* Copied from gtk_default_draw_shadow() */
198
199 x = y = 0;
200
201 cairo_set_line_width (cr, 1);
202
203 if (frame->edges & PANEL_EDGE_BOTTOM && padding.bottom > 0) {
204 if (padding.bottom > 1) {
205 gdk_cairo_set_source_rgba (cr, &dark);
206 cairo_move_to (cr, x + .5, y + height - 2 + .5);
207 cairo_line_to (cr, x + width - 1 - .5, y + height - 2 + .5);
208 cairo_stroke (cr);
209
210 cairo_set_source_rgb (cr, 0., 0., 0.);
211 cairo_move_to (cr, x + .5, y + height - 1 - .5);
212 cairo_line_to (cr, x + width - 1 - .5, y + height - 1 - .5);
213 cairo_stroke (cr);
214 } else {
215 gdk_cairo_set_source_rgba (cr, &dark);
216 cairo_move_to (cr, x + .5, y + height - 1 - .5);
217 cairo_line_to (cr, x + width - 1 - .5, y + height - 1 - .5);
218 cairo_stroke (cr);
219 }
220 }
221
222 if (frame->edges & PANEL_EDGE_RIGHT && padding.right > 0) {
223 if (padding.right > 1) {
224 gdk_cairo_set_source_rgba (cr, &dark);
225 cairo_move_to (cr, x + width - 2 - .5, y + .5);
226 cairo_line_to (cr, x + width - 2 - .5, y + height - 1 - .5);
227 cairo_stroke (cr);
228
229 cairo_set_source_rgb (cr, 0., 0., 0.);
230 cairo_move_to (cr, x + width - 1 - .5, y + .5);
231 cairo_line_to (cr, x + width - 1 - .5, y + height - 1 - .5);
232 cairo_stroke (cr);
233 } else {
234 gdk_cairo_set_source_rgba (cr, &dark);
235 cairo_move_to (cr, x + width - 1 - .5, y + .5);
236 cairo_line_to (cr, x + width - 1 - .5, y + height - 1 - .5);
237 cairo_stroke (cr);
238 }
239 }
240
241 if (frame->edges & PANEL_EDGE_TOP && padding.top > 0) {
242 gdk_cairo_set_source_rgba (cr, &light);
243 cairo_move_to (cr, x + .5, y + .5);
244 cairo_line_to (cr, x + width - 1 - .5, y + .5);
245 cairo_stroke (cr);
246
247 if (padding.top > 1) {
248 gdk_cairo_set_source_rgba (cr, bg);
249 cairo_move_to (cr, x + .5, y + 1 + .5);
250 cairo_line_to (cr, x + width - 1 - .5, y + 1 + .5);
251 cairo_stroke (cr);
252 }
253 }
254
255 if (frame->edges & PANEL_EDGE_LEFT && padding.left > 0) {
256 gdk_cairo_set_source_rgba (cr, &light);
257 cairo_move_to (cr, x + .5, y + .5);
258 cairo_line_to (cr, x + .5, y + height - 1 - .5);
259 cairo_stroke (cr);
260
261 if (padding.left > 1) {
262 gdk_cairo_set_source_rgba (cr, bg);
263 cairo_move_to (cr, x + 1 + .5, y + .5);
264 cairo_line_to (cr, x + 1 + .5, y + height - 1 - .5);
265 cairo_stroke (cr);
266 }
267 }
268
269 gdk_rgba_free (bg);
270}

272static gboolean panel_frame_expose(GtkWidget* widget, cairo_t* cr)
273{
274 PanelFrame *frame = (PanelFrame *) widget;
275 gboolean retval = FALSE;
276
277 if (!gtk_widget_is_drawable (widget))
278 return retval;
279
280 if (GTK_WIDGET_CLASS (panel_frame_parent_class)->draw)
281 retval = GTK_WIDGET_CLASS (panel_frame_parent_class)->draw (widget, cr);
282
283 panel_frame_draw (widget, cr, frame->edges);
284
285 return retval;
286}
