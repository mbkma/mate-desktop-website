95static void
96fr_restore_session (EggSMClient *client)
97{
98 GKeyFile *state = NULL;
99 guint i;
100
101 state = egg_sm_client_get_state_file (client);
102
103 i = g_key_file_get_integer (state, "Session", "archives", NULL);
104
105 for (; i > 0; i--) {
106 GtkWidget *window;
107 gchar *key, *archive;
108
109 key = g_strdup_printf ("archive%d", i);
110 archive = g_key_file_get_string (state, "Session", key, NULL);
111 g_free (key);
112
113 window = fr_window_new ();
114 gtk_widget_show (window);
115 if (strlen (archive))
116 fr_window_archive_open (FR_WINDOW (window), archive, GTK_WINDOW (window));
117
118 g_free (archive);
119 }
120}

135static void
136prepare_app (void)
137{
138 char *extract_to_uri = NULL;
139 char *add_to_uri = NULL;
140 EggSMClient *client = NULL;
141
142 client = egg_sm_client_get ();
143 if (egg_sm_client_is_resumed (client)) {
144 fr_restore_session (client);
145 return;
146 }
147
148 /**/
149
150 if (remaining_args == NULL) { /* No archive specified. */
151 fr_window_new ();
152 return;
153 }
154
155 if (extract_to != NULL)
156 extract_to_uri = get_uri_from_command_line (extract_to);
157
158 if (add_to != NULL)
159 add_to_uri = get_uri_from_command_line (add_to);
160
161 if ((add_to != NULL) || (add == 1)) { /* Add files to an archive */
162 GtkWidget *window;
163 GList *file_list = NULL;
164 const char *filename;
165 int i = 0;
166
167 window = fr_window_new ();
168 if (default_url != NULL)
169 fr_window_set_default_dir (FR_WINDOW (window), default_url, TRUE);
170
171 while ((filename = remaining_args[i++]) != NULL)
172 file_list = g_list_prepend (file_list, get_uri_from_command_line (filename));
173 file_list = g_list_reverse (file_list);
174
175 fr_window_new_batch (FR_WINDOW (window), _("Compress"));
176 fr_window_set_batch__add (FR_WINDOW (window), add_to_uri, file_list);
177 fr_window_append_batch_action (FR_WINDOW (window),
178 FR_BATCH_ACTION_QUIT,
179 NULL,
180 NULL);
181 fr_window_start_batch (FR_WINDOW (window));
182 }
183 else if ((extract_to != NULL) || (extract == 1) || (extract_here == 1)) {
184
185 /* Extract all archives. */
186
187 GtkWidget *window;
188 const char *archive;
189 int i = 0;
190
191 window = fr_window_new ();
192 if (default_url != NULL)
193 fr_window_set_default_dir (FR_WINDOW (window), default_url, TRUE);
194
195 fr_window_new_batch (FR_WINDOW (window), _("Extract archive"));
196 while ((archive = remaining_args[i++]) != NULL) {
197 char *archive_uri;
198
199 archive_uri = get_uri_from_command_line (archive);
200 if (extract_here == 1)
201 fr_window_set_batch__extract_here (FR_WINDOW (window),
202 archive_uri);
203 else
204 fr_window_set_batch__extract (FR_WINDOW (window),
205 archive_uri,
206 extract_to_uri);
207 g_free (archive_uri);
208 }
209 fr_window_append_batch_action (FR_WINDOW (window),
210 FR_BATCH_ACTION_QUIT,
211 NULL,
212 NULL);
213
214 fr_window_start_batch (FR_WINDOW (window));
215 }
216 else { /* Open each archive in a window */
217 const char *filename = NULL;
218
219 int i = 0;
220 while ((filename = remaining_args[i++]) != NULL) {
221 GtkWidget *window;
222 GFile *file;
223 char *uri;
224
225 window = fr_window_new ();
226
227 file = g_file_new_for_commandline_arg (filename);
228 uri = g_file_get_uri (file);
229 fr_window_archive_open (FR_WINDOW (window), uri, GTK_WINDOW (window));
230 g_free (uri);
231 g_object_unref (file);
232 }
233 }
234
235 g_free (add_to_uri);
236 g_free (extract_to_uri);
237}

260static void
261fr_save_state (EggSMClient *client, GKeyFile *state, gpointer user_data)
262{
263 /* discard command is automatically set by EggSMClient */
264
265 const char *argv[2] = { NULL };
266 GApplication *application;
267 guint i = 0;
268
269 /* restart command */
270 argv[0] = program_argv0;
271 argv[1] = NULL;
272
273 egg_sm_client_set_restart_command (client, 1, argv);
274
275 /* state */
276 application = g_application_get_default ();
277 if (application != NULL) {
278 GList *window;
279
280 for (window = gtk_application_get_windows (GTK_APPLICATION (application)), i = 0;
281 window != NULL;
282 window = window->next, i++)
283 {
284 FrWindow *session = window->data;
285 gchar *key;
286
287 key = g_strdup_printf ("archive%d", i+1);
288 if ((session->archive == NULL) || (session->archive->file == NULL)) {
289 g_key_file_set_string (state, "Session", key, "");
290 }
291 else {
292 gchar *uri;
293
294 uri = g_file_get_uri (session->archive->file);
295 g_key_file_set_string (state, "Session", key, uri);
296 g_free (uri);
297 }
298 g_free (key);
299 }
300 }
301
302 g_key_file_set_integer (state, "Session", "archives", i);
303}

319 textdomain (GETTEXT_PACKAGE);
320#endif /* ENABLE_NLS */
321
322 context = g_option_context_new (N_("- Create and modify an archive"));
323#ifdef ENABLE_NLS
324 g_option_context_set_translation_domain (context, GETTEXT_PACKAGE);
325#endif /* ENABLE_NLS */
326 g_option_context_add_main_entries (context, options, GETTEXT_PACKAGE);
327
328 g_option_context_add_group (context, gtk_get_option_group (TRUE));
329 g_option_context_add_group (context, egg_sm_client_get_option_group ());
330
331 if (! g_option_context_parse (context, &argc, &argv, &error)) {
332 g_critical ("Failed to parse arguments: %s", error->message);
333 g_error_free (error);
334 g_option_context_free (context);
335 return EXIT_FAILURE;
336 }
337
338 g_option_context_free (context);
339
340 g_set_application_name (_("Engrampa"));
341 gtk_window_set_default_icon_name ("engrampa");
342
343 client = egg_sm_client_get ();
344 g_signal_connect (client, "save-state", G_CALLBACK (fr_save_state), NULL);
345
346 gtk_icon_theme_append_search_path (gtk_icon_theme_get_default (),
347 PKG_DATA_DIR G_DIR_SEPARATOR_S "icons");
348
349 app = gtk_application_new (NULL, G_APPLICATION_FLAGS_NONE);
350 g_signal_connect (app, "startup", G_CALLBACK (startup_cb), NULL);
351 g_signal_connect (app, "activate", G_CALLBACK (activate_cb), NULL);
352
353 status = g_application_run (G_APPLICATION (app), argc, argv);
354
355 release_data ();
356
357 return status;
