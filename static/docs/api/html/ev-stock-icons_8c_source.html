56static void
57ev_stock_icons_add_icons_path_for_screen (GdkScreen *screen)
58{
59 GtkIconTheme *icon_theme;
60
61 icon_theme = screen ? gtk_icon_theme_get_for_screen (screen) : gtk_icon_theme_get_default ();
62 if (icon_theme) {
63 gchar **path = NULL;
64 gint n_paths;
65 gint i;
66
67 /* GtkIconTheme will then look in Atril custom hicolor dir
68 * for icons as well as the standard search paths
69 */
70 gtk_icon_theme_get_search_path (icon_theme, &path, &n_paths);
71 for (i = n_paths - 1; i >= 0; i--) {
72 if (g_ascii_strcasecmp (EV_ICONS_PATH, path[i]) == 0)
73 break;
74 }
75
76 if (i < 0)
77 gtk_icon_theme_append_search_path (icon_theme,
78 EV_ICONS_PATH);
79
80 g_strfreev (path);
81 }
82}

89void
90ev_stock_icons_init (void)
91{
92 GtkIconFactory *factory;
93 GtkIconSource *source;
94 gint i;
95
96 factory = gtk_icon_factory_new ();
97 gtk_icon_factory_add_default (factory);
98
99 source = gtk_icon_source_new ();
100
101 for (i = 0; i < G_N_ELEMENTS (stock_icons); i++) {
102 GtkIconSet *set;
103
104 gtk_icon_source_set_icon_name (source, stock_icons [i].icon);
105
106 set = gtk_icon_set_new ();
107 gtk_icon_set_add_source (set, source);
108
109 gtk_icon_factory_add (factory, stock_icons [i].stock_id, set);
110 gtk_icon_set_unref (set);
111 }
112
113 gtk_icon_source_free (source);
114
115 g_object_unref (G_OBJECT (factory));
116
117 ev_stock_icons_add_icons_path_for_screen (gdk_screen_get_default ());
118}

124
125 ev_stock_icons_add_icons_path_for_screen (screen);
