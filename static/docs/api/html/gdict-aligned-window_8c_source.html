114static void
115gdict_aligned_window_get_property (GObject *object,
116 guint prop_id,
117 GValue *value,
118 GParamSpec *pspec)
119{
120 GdictAlignedWindow *aligned_window = GDICT_ALIGNED_WINDOW (object);
121
122 switch (prop_id)
123 {
124 case PROP_ALIGN_WIDGET:
125 g_value_set_object (value, aligned_window->priv->align_widget);

127 default:
128 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
129 break;
130 }

153static void
154gdict_aligned_window_position (GdictAlignedWindow *window)
155{
156 GdictAlignedWindowPrivate *priv;
157 GtkWidget *align_widget;
158 gint our_width, our_height;
159 gint entry_x, entry_y, entry_width, entry_height;
160 gint x, y;
161 GdkGravity gravity = GDK_GRAVITY_NORTH_WEST;
162 GdkWindow *gdk_window;
163 GdkDisplay *display;
164 GdkMonitor *monitor;
165 GdkRectangle geometry = {0};
166
167 g_assert (GDICT_IS_ALIGNED_WINDOW (window));
168 priv = window->priv;
169
170 if (!priv->align_widget)
171 return;
172
173 align_widget = priv->align_widget;
174 gdk_window = gtk_widget_get_window (align_widget);
175
176 display = gdk_display_get_default ();
177 gdk_display_flush (display);
178
179 gdk_window_get_geometry (gtk_widget_get_window (GTK_WIDGET (window)), NULL, NULL, &our_width, &our_height);
180
181 /* stick, skip taskbar and pager */
182 gtk_window_stick (GTK_WINDOW (window));
183 gtk_window_set_skip_taskbar_hint (GTK_WINDOW (window), TRUE);
184 gtk_window_set_skip_pager_hint (GTK_WINDOW (window), TRUE);
185
186 /* make sure the align_widget is realized before we do anything */
187 gtk_widget_realize (align_widget);
188
189 /* get the positional and dimensional attributes of the align widget */
190 gdk_window_get_origin (gdk_window,
191 &entry_x,
192 &entry_y);
193 gdk_window_get_geometry (gdk_window, NULL, NULL, &entry_width, &entry_height);
194
195 /*Get the monitor dimensions*/
196 display= gdk_screen_get_display (gdk_screen_get_default ());
197 monitor = gdk_display_get_monitor (display, 0);
198 gdk_monitor_get_geometry (monitor, &geometry);
199
200 if (entry_x + our_width < geometry.width)
201 x = entry_x + 1;
202 else
203 {
204 x = entry_x + entry_width - our_width - 1;
205
206 gravity = GDK_GRAVITY_NORTH_EAST;
207 }
208
209 if (entry_y + entry_height + our_height < geometry.height)
210 y = entry_y + entry_height - 1;
211 else
212 {
213 y = entry_y - our_height + 1;
214
215 if (gravity == GDK_GRAVITY_NORTH_EAST)
216 gravity = GDK_GRAVITY_SOUTH_EAST;
217 else
218 gravity = GDK_GRAVITY_SOUTH_WEST;
219 }
220
221 gtk_window_set_gravity (GTK_WINDOW (window), gravity);
222 gtk_window_move (GTK_WINDOW (window), x, y);
223
224#if defined (ENABLE_WAYLAND) && defined (GDK_WINDOWING_WAYLAND)
225 if (GDK_IS_WAYLAND_DISPLAY (display))
226 {
227 gboolean top, bottom, left, right;
228 GtkWidget *toplevel = gtk_widget_get_toplevel (GTK_WIDGET(priv->align_widget));
229
230 top = gtk_layer_get_anchor (GTK_WINDOW (toplevel), GTK_LAYER_SHELL_EDGE_TOP);
231 bottom = gtk_layer_get_anchor (GTK_WINDOW (toplevel), GTK_LAYER_SHELL_EDGE_BOTTOM);
232 left = gtk_layer_get_anchor (GTK_WINDOW (toplevel), GTK_LAYER_SHELL_EDGE_LEFT);
233 right = gtk_layer_get_anchor (GTK_WINDOW (toplevel), GTK_LAYER_SHELL_EDGE_RIGHT);
234
235 /*Set anchors to the edges (will hold to panel edge) and position along the panel
236 *Confine the dialog to the screen dimensions if the applet is near right or bottom
237 *Unset margins and anchors from any other position so as to avoid rendering issues
238 *when orientation changes as when the panel is moved
239 */
240 if (top && left && right)
241 {
242 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, TRUE);
243 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_BOTTOM, FALSE);
244 if (entry_x + our_width >= geometry.width)
245 {
246 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, FALSE);
247 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_RIGHT,TRUE);
248 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, 0);
249 }
250 else
251 {
252 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, TRUE);
253 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_RIGHT, FALSE);
254 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, entry_x);
255 }
256 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, 0);
257 }
258
259 if (bottom && left && right)
260 {
261 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_BOTTOM, TRUE);
262 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, FALSE);
263 if (entry_x + our_width >= geometry.width)
264 {
265 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, FALSE);
266 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_RIGHT,TRUE);
267 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, 0);
268 }
269 else
270 {
271 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, TRUE);
272 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_RIGHT, FALSE);
273 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, entry_x);
274 }
275
276 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, 0);
277 }
278 if (left && bottom && top && !right)
279 {
280 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, TRUE);
281 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_RIGHT, FALSE);
282 if (entry_y + our_height >= geometry.height)
283 {
284 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, FALSE);
285 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_BOTTOM, TRUE);
286 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, 0);
287 }
288 else
289 {
290 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, TRUE);
291 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_BOTTOM, FALSE);
292 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, entry_y);
293 }
294 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, 0);
295 }
296 if (right && bottom && top && !left)
297 {
298 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_RIGHT, TRUE);
299 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, FALSE);
300 if (entry_y + our_height >= geometry.height)
301 {
302 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, FALSE);
303 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_BOTTOM, TRUE);
304 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, 0);
305 }
306 else
307 {
308 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, TRUE);
309 gtk_layer_set_anchor (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_BOTTOM, FALSE);
310 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, entry_y);
311 }
312 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_TOP, entry_y);
313 gtk_layer_set_margin (GTK_WINDOW (window), GTK_LAYER_SHELL_EDGE_LEFT, 0);
314 }
315 }
316#endif
317}

341static gboolean
342gdict_aligned_window_motion_notify_cb (GtkWidget *widget,
343 GdkEventMotion *event,
344 GdictAlignedWindow *aligned_window)
345{
346 GtkAllocation alloc;
347 GdkRectangle rect;
348
349 gtk_widget_get_allocation (GTK_WIDGET (aligned_window), &alloc);
350
351 rect.x = 0;
352 rect.y = 0;
353 rect.width = alloc.width;
354 rect.height = alloc.height;
355
356 gdk_window_invalidate_rect (gtk_widget_get_window (GTK_WIDGET (aligned_window)),
357 &rect,
358 FALSE);
359
360 return FALSE;
361}

416 priv->align_widget = NULL;

420 priv->motion_id = g_signal_connect (priv->align_widget, "motion-notify-event",
421 G_CALLBACK (gdict_aligned_window_motion_notify_cb),
422 aligned_window);
423}
