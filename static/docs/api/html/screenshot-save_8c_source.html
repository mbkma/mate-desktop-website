33static char *tmp_filename = NULL;

35static void
36clean_up_temporary_dir (void)
37{
38 char *message;
39 gboolean error_occurred = FALSE;
40 if (tmp_filename && g_file_test (tmp_filename, G_FILE_TEST_EXISTS))
41 error_occurred = unlink (tmp_filename);
42 if (parent_dir && g_file_test (parent_dir, G_FILE_TEST_EXISTS))
43 error_occurred = rmdir (parent_dir) || error_occurred;
44
45 if (error_occurred)
46 {
47 message = g_strdup_printf (_("Unable to clear the temporary folder:\n%s"),
48 tmp_filename);
49 g_warning ("%s", message);
50 g_free (message);
51 }
52 g_free (tmp_filename);
53 g_free (parent_dir);
54 tmp_filename = NULL;
55 parent_dir = NULL;
56}

58static char *
59make_temp_directory (void)
60{
61 gchar *dir_name;
62
63 // mkdtemp uses XXXXXX as a template to create a unique name for the temp dir
64 dir_name = g_build_filename (g_get_tmp_dir (),
65 "mate-screenshot.XXXXXX",
66 NULL);
67
68 if (mkdtemp (dir_name) == NULL)
69 {
70 g_warning ("Failed to create temporary directory: %s", g_strerror (errno));
71 g_free (dir_name);
72 return NULL;
73 }
74
75 return dir_name;
76}

84void
85screenshot_save_start (GdkPixbuf *pixbuf,
86 SaveFunction callback,
87 gpointer user_data)
88{
89 GError *error = NULL;
90
91 static gboolean cleanup_registered = FALSE;
92 if (!cleanup_registered)
93 {
94 atexit (cleanup_handler);
95 cleanup_registered = TRUE;
96 }
97
98 clean_up_temporary_dir ();
99
100 parent_dir = make_temp_directory ();
101 if (parent_dir == NULL)
102 {
103 if (callback)
104 callback (user_data);
105 return;
106 }
107
108 tmp_filename = g_build_filename (parent_dir,
109 _("Screenshot.png"),
110 NULL);
111
112 if (! gdk_pixbuf_save (pixbuf, tmp_filename,
113 "png", &error,
114 "tEXt::Software", "mate-screenshot",
115 NULL))
116 {
117 if (error && error->message)
118 g_warning ("Failed to save screenshot: %s", error->message);
119 else
120 g_warning ("Failed to save screenshot: Unknown error");
121
122 g_error_free (error);
123
124 clean_up_temporary_dir ();
125 }
126
127 if (callback)
128 callback (user_data);
129}
