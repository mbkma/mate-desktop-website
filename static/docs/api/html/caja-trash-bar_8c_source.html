53static void
54restore_button_clicked_cb (GtkWidget *button,
55 CajaTrashBar *bar)
56{
57 GList *locations, *files, *l;
58
59 locations = caja_window_info_get_selection (CAJA_WINDOW_INFO (bar->priv->window));
60 files = NULL;
61
62 for (l = locations; l != NULL; l = l->next)
63 {
64 files = g_list_prepend (files, caja_file_get (l->data));
65 }
66
67 caja_restore_files_from_trash (files, GTK_WINDOW (gtk_widget_get_toplevel (button)));
68
69 caja_file_list_free (files);
70 g_list_free_full (locations, g_object_unref);
71}

94static void
95caja_trash_bar_set_property (GObject *object,
96 guint prop_id,
97 const GValue *value,
98 GParamSpec *pspec)
99{
100 CajaTrashBar *bar;
101
102 bar = CAJA_TRASH_BAR (object);
103
104 switch (prop_id)
105 {
106 case PROP_WINDOW:
107 bar->priv->window = g_value_get_object (value);
108 connect_window_and_update_button (bar);

111 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
112 break;
