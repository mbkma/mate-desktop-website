96static void
97entry_changed_cb (GtkEditable *editable,
98 gpointer user_data)
99{
100 LogviewFindbar *findbar = user_data;
101 const char *text;
102
103 text = gtk_entry_get_text (GTK_ENTRY (editable));
104
105 if (g_strcmp0 (text, "") == 0) {
106 return;
107 }
108
109 if (g_strcmp0 (findbar->priv->string, text) != 0) {
110 g_free (findbar->priv->string);
111 findbar->priv->string = g_strdup (text);
112
113 g_signal_emit (findbar, signals[TEXT_CHANGED], 0);
114 }
115}

147static void
148logview_findbar_init (LogviewFindbar *findbar)
149{
150 GtkWidget *label, *w, *box;
151 GtkToolbar *gtoolbar;
152 GtkToolItem *item;
153 LogviewFindbarPrivate *priv;
154
155 priv = findbar->priv = logview_findbar_get_instance_private (findbar);
156
157 gtoolbar = GTK_TOOLBAR (findbar);
158
159 gtk_toolbar_set_style (gtoolbar, GTK_TOOLBAR_BOTH_HORIZ);
160
161 priv->status_bold_id = 0;
162
163 /* Find: |_______| */
164 box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 12);
165 gtk_widget_set_halign (box, GTK_ALIGN_START);
166 gtk_widget_set_margin_start (box, 2);
167 gtk_widget_set_margin_end (box, 2);
168
169 label = gtk_label_new_with_mnemonic (_("_Find:"));
170 gtk_box_pack_start (GTK_BOX (box), label, FALSE, FALSE, 0);
171
172 priv->entry = gtk_entry_new ();
173 gtk_entry_set_width_chars (GTK_ENTRY (priv->entry), 32);
174 gtk_label_set_mnemonic_widget (GTK_LABEL (label), priv->entry);
175 gtk_box_pack_start (GTK_BOX (box), priv->entry, TRUE, TRUE, 0);
176
177 item = gtk_tool_item_new ();
178 gtk_container_add (GTK_CONTAINER (item), box);
179 gtk_toolbar_insert (gtoolbar, item, -1);
180 gtk_widget_show_all (GTK_WIDGET (item));
181
182 /* "Previous" and "Next" buttons */
183 w = gtk_image_new_from_icon_name ("pan-start-symbolic", GTK_ICON_SIZE_BUTTON);
184 priv->back_button = gtk_tool_button_new (w, _("Find Previous"));
185 gtk_tool_item_set_is_important (priv->back_button, TRUE);
186 gtk_tool_item_set_tooltip_text (priv->back_button,
187 _("Find previous occurrence of the search string"));
188 gtk_toolbar_insert (gtoolbar, priv->back_button, -1);
189 gtk_widget_show_all (GTK_WIDGET (priv->back_button));
190
191 w = gtk_image_new_from_icon_name ("pan-end-symbolic", GTK_ICON_SIZE_BUTTON);
192 priv->forward_button = gtk_tool_button_new (w, _("Find Next"));
193 gtk_tool_item_set_is_important (priv->forward_button, TRUE);
194 gtk_tool_item_set_tooltip_text (priv->forward_button,
195 _("Find next occurrence of the search string"));
196 gtk_toolbar_insert (gtoolbar, priv->forward_button, -1);
197 gtk_widget_show_all (GTK_WIDGET (priv->forward_button));
198
199 /* clear button */
200 w = gtk_image_new_from_icon_name ("edit-clear-all", GTK_ICON_SIZE_BUTTON);
201 priv->clear_button = gtk_tool_button_new (w, NULL);
202 gtk_tool_item_set_tooltip_text (priv->clear_button,
203 _("Clear the search string"));
204 gtk_toolbar_insert (gtoolbar, priv->clear_button, -1);
205 gtk_widget_show_all (GTK_WIDGET (priv->clear_button));
206
207 /* separator */
208 priv->separator = gtk_separator_tool_item_new ();
209 gtk_toolbar_insert (gtoolbar, priv->separator, -1);
210
211 /* message */
212 priv->status_item = gtk_tool_item_new ();
213 gtk_tool_item_set_expand (priv->status_item, TRUE);
214 priv->message = gtk_label_new ("");
215 gtk_label_set_use_markup (GTK_LABEL (priv->message), TRUE);
216 gtk_label_set_xalign (GTK_LABEL (priv->message), 0.0);
217 gtk_label_set_yalign (GTK_LABEL (priv->message), 0.5);
218 gtk_container_add (GTK_CONTAINER (priv->status_item), priv->message);
219 gtk_widget_show (priv->message);
220 gtk_toolbar_insert (gtoolbar, priv->status_item, -1);
221
222 priv->string = NULL;
223
224 /* signal handlers */
225 g_signal_connect (priv->back_button, "clicked",
226 G_CALLBACK (back_button_clicked_cb), findbar);
227 g_signal_connect (priv->forward_button, "clicked",
228 G_CALLBACK (forward_button_clicked_cb), findbar);
229 g_signal_connect (priv->clear_button, "clicked",
230 G_CALLBACK (clear_button_clicked_cb), findbar);
231 g_signal_connect (priv->entry, "activate",
232 G_CALLBACK (entry_activate_cb), findbar);
233 g_signal_connect (priv->entry, "changed",
234 G_CALLBACK (entry_changed_cb), findbar);
235 g_signal_connect (priv->entry, "key-press-event",
236 G_CALLBACK (entry_key_press_event_cb), findbar);
237}

327void
328logview_findbar_set_message (LogviewFindbar *findbar,
329 const char *text)
330{
331 PangoFontDescription *desc;
332
333 g_assert (LOGVIEW_IS_FINDBAR (findbar));
334
335 if (text) {

338 gtk_widget_override_font (findbar->priv->message, desc);

344 gtk_label_set_text (GTK_LABEL (findbar->priv->message),
345 text != NULL ? text : "");
346 g_object_set (findbar->priv->separator, "visible", text != NULL, NULL);
347 g_object_set (findbar->priv->status_item, "visible", text != NULL, NULL);
