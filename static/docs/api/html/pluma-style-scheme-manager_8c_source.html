106GSList *
107pluma_style_scheme_manager_list_schemes_sorted (GtkSourceStyleSchemeManager *manager)
108{
109 const gchar * const * scheme_ids;
110 GSList *schemes = NULL;
111
112 g_return_val_if_fail (GTK_SOURCE_IS_STYLE_SCHEME_MANAGER (manager), NULL);
113
114 scheme_ids = gtk_source_style_scheme_manager_get_scheme_ids (manager);
115
116 while (*scheme_ids != NULL)
117 {
118 GtkSourceStyleScheme *scheme;
119
120 scheme = gtk_source_style_scheme_manager_get_scheme (manager,
121 *scheme_ids);
122
123 schemes = g_slist_prepend (schemes, scheme);
124
125 ++scheme_ids;
126 }
127
128 if (schemes != NULL)
129 schemes = g_slist_sort (schemes, (GCompareFunc)schemes_compare);
130
131 return schemes;
132}

177static gboolean
178file_copy (const gchar *name,
179 const gchar *dest_name,
180 GError **error)
181{
182 gchar *contents;
183 gsize length;
184 gchar *dest_dir;
185
186 /* FIXME - Paolo (Aug. 13, 2007):
187 * Since the style scheme files are relatively small, we can implement
188 * file copy getting all the content of the source file in a buffer and
189 * then write the content to the destination file. In this way we
190 * can use the g_file_get_contents and g_file_set_contents and avoid to
191 * write custom code to copy the file (with sane error management).
192 * If needed we can improve this code later. */
193
194 g_return_val_if_fail (name != NULL, FALSE);
195 g_return_val_if_fail (dest_name != NULL, FALSE);
196 g_return_val_if_fail (error == NULL || *error == NULL, FALSE);
197
198 /* Note: we allow to copy a file to itself since this is not a problem
199 * in our use case */
200
201 /* Ensure the destination directory exists */
202 dest_dir = g_path_get_dirname (dest_name);
203
204 errno = 0;
205 if (g_mkdir_with_parents (dest_dir, 0755) != 0)
206 {
207 gint save_errno = errno;
208 gchar *display_filename = g_filename_display_name (dest_dir);
209
210 g_set_error (error,
211 G_FILE_ERROR,
212 g_file_error_from_errno (save_errno),
213 _("Directory '%s' could not be created: g_mkdir_with_parents() failed: %s"),
214 display_filename,
215 g_strerror (save_errno));
216
217 g_free (dest_dir);
218 g_free (display_filename);
219
220 return FALSE;
221 }
222
223 g_free (dest_dir);
224
225 if (!g_file_get_contents (name, &contents, &length, error))
226 return FALSE;
227
228 if (!g_file_set_contents (dest_name, contents, length, error))
229 return FALSE;
230
231 g_free (contents);
232
233 return TRUE;
234}

251const gchar *
252_pluma_style_scheme_manager_install_scheme (GtkSourceStyleSchemeManager *manager,
253 const gchar *fname)
254{
255 gchar *new_file_name = NULL;
256 gchar *dirname;
257 gchar *styles_dir;
258 GError *error = NULL;
259 gboolean copied = FALSE;
260
261 const gchar* const *ids;
262
263 g_return_val_if_fail (GTK_SOURCE_IS_STYLE_SCHEME_MANAGER (manager), NULL);
264 g_return_val_if_fail (fname != NULL, NULL);
265
266 dirname = g_path_get_dirname (fname);
267 styles_dir = get_pluma_styles_path();
268
269 if (strcmp (dirname, styles_dir) != 0)
270 {
271 gchar *basename;
272
273 basename = g_path_get_basename (fname);
274 new_file_name = g_build_filename (styles_dir, basename, NULL);
275 g_free (basename);
276
277 /* Copy the style scheme file into PLUMA_STYLES_DIR */
278 if (!file_copy (fname, new_file_name, &error))
279 {
280 g_free (new_file_name);
281
282 g_message ("Cannot install style scheme:\n%s",
283 error->message);
284
285 g_free (dirname);
286 g_free (styles_dir);
287 return NULL;
288 }
289
290 copied = TRUE;
291 }
292 else
293 {
294 new_file_name = g_strdup (fname);
295 }
296
297 g_free (dirname);
298 g_free (styles_dir);
299
300 /* Reload the available style schemes */
301 gtk_source_style_scheme_manager_force_rescan (manager);
302
303 /* Check the new style scheme has been actually installed */
304 ids = gtk_source_style_scheme_manager_get_scheme_ids (manager);
305
306 while (*ids != NULL)
307 {
308 GtkSourceStyleScheme *scheme;
309 const gchar *filename;
310
311 scheme = gtk_source_style_scheme_manager_get_scheme (
312 pluma_get_style_scheme_manager (), *ids);
313
314 filename = gtk_source_style_scheme_get_filename (scheme);
315
316 if (filename && (strcmp (filename, new_file_name) == 0))
317 {
318 /* The style scheme has been correctly installed */
319 g_free (new_file_name);
320
321 return gtk_source_style_scheme_get_id (scheme);
322 }
323 ++ids;
324 }
325
326 /* The style scheme has not been correctly installed */
327 if (copied)
328 g_unlink (new_file_name);
329
330 g_free (new_file_name);
331
332 return NULL;
333}

347gboolean
348_pluma_style_scheme_manager_uninstall_scheme (GtkSourceStyleSchemeManager *manager,
349 const gchar *id)
350{
351 GtkSourceStyleScheme *scheme;
352 const gchar *filename;
353
354 g_return_val_if_fail (GTK_SOURCE_IS_STYLE_SCHEME_MANAGER (manager), FALSE);
355 g_return_val_if_fail (id != NULL, FALSE);
356
357 scheme = gtk_source_style_scheme_manager_get_scheme (manager, id);
358 if (scheme == NULL)
359 return FALSE;
360
361 filename = gtk_source_style_scheme_get_filename (scheme);
362 if (filename == NULL)
363 return FALSE;
364
365 if (g_unlink (filename) == -1)
366 return FALSE;
367
368 /* Reload the available style schemes */
369 gtk_source_style_scheme_manager_force_rescan (manager);
370
371 return TRUE;
