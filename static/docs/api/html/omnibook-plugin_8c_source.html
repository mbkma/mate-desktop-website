38static void omnibook_plugin_setup_manually(GList **sensors) {

62static gdouble omnibook_plugin_get_sensor_value(const gchar *path,
63 const gchar *id,
64 SensorType type,
65 GError **error) {
66
67 /* to open and access the value of each sensor */
68 FILE *fp;
69 gfloat sensor_value;
70
71 if (NULL == (fp = fopen(path, "r"))) {
72 g_set_error(error, SENSORS_APPLET_PLUGIN_ERROR, OMNIBOOK_DEVICE_FILE_OPEN_ERROR, "Error opening sensor device file %s", path);
73 return -1.0;
74 }
75
76 if (fscanf(fp, "CPU temperature: %f", &sensor_value) != 1) {
77 g_set_error(error, SENSORS_APPLET_PLUGIN_ERROR, OMNIBOOK_DEVICE_FILE_READ_ERROR, "Error reading from sensor device file %s", path);
78 fclose(fp);
79 return -1.0;
80 }
81
82 fclose(fp);
83
84 return (gdouble)sensor_value;
85}
