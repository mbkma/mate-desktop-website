26void test_one_locale(const gchar *locale)
27{
28 char *lang, *country, *norm_locale;
29 char *language_code, *country_code, *codeset, *modifier;
30
31 lang = mate_get_language_from_locale (locale, locale);
32 country = mate_get_country_from_locale (locale, locale);
33 norm_locale = mate_normalize_locale (locale);
34
35 printf("Current locale: %s\n", locale);
36 printf("[locale]:\t\tlang=%s, country=%s, locale=%s\n", lang, country, norm_locale);
37 g_free(lang);
38 g_free(country);
39 g_free(norm_locale);
40
41 if (mate_parse_locale (locale, &language_code, &country_code, &codeset, &modifier)) {
42 lang = mate_get_language_from_code (language_code, locale);
43 country = mate_get_country_from_code (country_code, locale);
44 if (mate_language_has_translations(language_code)) {
45 printf("[mate_parse_locale]:\tlang_code=%s, country_code=%s, code=%s, modifier=%s\n"
46 "[code]:\t\t\tlang=%s, country=%s, Has translation\n",
47 language_code, country_code, codeset, modifier,
48 lang, country);
49 } else {
50 printf("[mate_parse_locale]:\tlang_code=%s, country_code=%s, code=%s, modifier=%s\n"
51 "[code]:\t\t\tlang=%s, country=%s\n",
52 language_code, country_code, codeset, modifier,
53 lang, country);
54 }
55 g_free(lang);
56 g_free(country);
57 }
58 putchar('\n');
59
60 g_free(language_code);
61 g_free(country_code);
62 g_free(codeset);
63 g_free(modifier);
64}
