138static void
139automount_all_volumes (CajaApplication *application)
140{
141 GList *volumes, *l;
142
143 if (g_settings_get_boolean (caja_media_preferences, CAJA_PREFERENCES_MEDIA_AUTOMOUNT))
144 {
145 /* automount all mountable volumes at start-up */
146
147 GVolume *volume = NULL;
148 GMount *mount = NULL;
149
150 volumes = g_volume_monitor_get_volumes (application->priv->volume_monitor);
151
152 for (l = volumes; l != NULL; l = l->next)
153 {
154 volume = l->data;
155
156 if (!g_volume_should_automount (volume) ||
157 !g_volume_can_mount (volume))
158 {
159 continue;
160 }
161
162 mount = g_volume_get_mount (volume);
163 if (mount != NULL)
164 {
165 g_object_unref (mount);
166 continue;
167 }
168
169 /* pass NULL as GMountOperation to avoid user interaction */
170 g_volume_mount (volume, 0, NULL, NULL, startup_volume_mount_cb, NULL);
171 }
172 g_list_free_full (volumes, g_object_unref);
173 }

228static void
229open_window (CajaApplication *application,
230 GFile *location,
231 GdkScreen *screen,
232 const char *geometry,
233 gboolean browser_window)
234{
235 CajaApplication *self = CAJA_APPLICATION (application);
236 CajaWindow *window;
237 gchar *uri;
238
239 uri = g_file_get_uri (location);
240 g_debug ("Opening new window at uri %s", uri);
241
242 /*monitor the preference to use browser or spatial windows */
243 /*connect before trying to read or this preference won't be read by root or after change*/
244 g_signal_connect_swapped(caja_preferences, "changed::"CAJA_PREFERENCES_ALWAYS_USE_BROWSER,
245 G_CALLBACK (caja_empty_callback_to_ensure_read),
246 self);
247
248 if (browser_window ||g_settings_get_boolean (caja_preferences, CAJA_PREFERENCES_ALWAYS_USE_BROWSER)) {
249 window = caja_application_create_navigation_window (application,
250 screen);
251 } else {
252 window = caja_application_get_spatial_window (application,
253 NULL,
254 NULL,
255 location,
256 screen,
257 NULL);
258 }
259
260 caja_window_go_to (window, location);
261
262 if (geometry != NULL && !gtk_widget_get_visible (GTK_WIDGET (window))) {
263 /* never maximize windows opened from shell if a
264 * custom geometry has been requested. */
265 gtk_window_unmaximize (GTK_WINDOW (window));
266 eel_gtk_window_set_initial_geometry_from_string (GTK_WINDOW (window),
267 geometry,
268 APPLICATION_WINDOW_MIN_WIDTH,
269 APPLICATION_WINDOW_MIN_HEIGHT,
270 FALSE);
271 }
272
273 g_free (uri);
274}

276static void
277open_tabs (CajaApplication *application,
278 GFile **locations,
279 guint n_files,
280 GdkScreen *screen,
281 const char *geometry,
282 gboolean browser_window)
283{
284 CajaApplication *self = CAJA_APPLICATION (application);
285 CajaWindow *window;
286 gchar *uri = NULL;
287 guint i;
288
289 /* monitor the preference to use browser or spatial windows */
290 /* connect before trying to read or this preference won't be read by root or after change */
291 g_signal_connect_swapped (caja_preferences,
292 "changed::"CAJA_PREFERENCES_ALWAYS_USE_BROWSER,
293 G_CALLBACK (caja_empty_callback_to_ensure_read),
294 self);
295
296 if (browser_window ||g_settings_get_boolean (caja_preferences, CAJA_PREFERENCES_ALWAYS_USE_BROWSER)) {
297 window = caja_application_create_navigation_window (application, screen);
298 } else {
299 window = caja_application_get_spatial_window (application,
300 NULL,
301 NULL,
302 locations[0],
303 screen,
304 NULL);
305 }
306
307 /* open all locations */
308 uri = g_file_get_uri (locations[0]);
309 g_debug ("Opening new tab at uri %s\n", uri);
310 caja_window_go_to (window, locations[0]);
311 g_free (uri);
312 for (i = 1; i < n_files; i++) {
313 /* open tabs in reverse order because each
314 * tab is opened before the previous one */
315 guint tab = n_files-i;
316 uri = g_file_get_uri (locations[tab]);
317 g_debug ("Opening new tab at uri %s\n", uri);
318 caja_window_go_to_tab (window, locations[tab]);
319 g_free (uri);
320 }
321
322 if (geometry != NULL && !gtk_widget_get_visible (GTK_WIDGET (window))) {
323 /* never maximize windows opened from shell if a
324 * custom geometry has been requested. */
325 gtk_window_unmaximize (GTK_WINDOW (window));
326 eel_gtk_window_set_initial_geometry_from_string (GTK_WINDOW (window),
327 geometry,
328 APPLICATION_WINDOW_MIN_WIDTH,
329 APPLICATION_WINDOW_MIN_HEIGHT,
330 FALSE);
331 }
332}

334static void
335open_windows (CajaApplication *application,
336 GFile **files,
337 GdkScreen *screen,
338 const char *geometry,
339 guint n_files,
340 gboolean browser_window,
341 gboolean open_in_tabs)
342{
343 guint i;
344
345 if (files == NULL || files[0] == NULL) {
346 /* Open a window pointing at the default location. */
347 open_window (application, NULL, screen, geometry, browser_window);
348 } else {
349 if (open_in_tabs) {
350 /* Open one window with one tab at each requested location */
351 open_tabs (application, files, n_files, screen, geometry, browser_window);
352 } else {
353 /* Open windows at each requested location. */
354 i = 0;
355 while (i < n_files) {
356 open_window (application, files[i], screen, geometry, browser_window);
357 i++ ;
358 }
359 }
360 }

363static void
364caja_application_open (GApplication *app,
365 GFile **files,
366 gint n_files,
367 const gchar *options)
368{
369 CajaApplication *self = CAJA_APPLICATION (app);
370 gboolean browser_window = FALSE;
371 gboolean open_in_tabs = FALSE;
372 const gchar *geometry = NULL;
373
374 g_debug ("Open called on the GApplication instance; %d files", n_files);
375
376 /* Check if local command line passed --browser, --geometry or --tabs */
377 if (strlen (options) > 0) {
378 gchar** splitedOptions = g_strsplit (options, "=", 3);
379 sscanf (splitedOptions[0], "%d", &browser_window);
380 if (strcmp (splitedOptions[1], "NULL") != 0) {
381 geometry = splitedOptions[1];
382 }
383 sscanf (splitedOptions[2], "%d", &open_in_tabs);
384
385 open_windows (self, files,
386 gdk_screen_get_default (),
387 geometry,
388 n_files,
389 browser_window,
390 open_in_tabs);
391
392 /* Reset this or 3ed and later invocations will use same
393 * geometry even if the user has resized open window */
394 self->priv->geometry = NULL;
395 g_strfreev (splitedOptions);
396 }
397 else
398 open_windows (self, files,
399 gdk_screen_get_default (),
400 geometry,
401 n_files,
402 browser_window,
403 open_in_tabs);
404}

509static gboolean
510check_required_directories (CajaApplication *application)
511{
512 char *user_directory;
513 char *desktop_directory;
514 GSList *directories;
515 gboolean ret;
516
517 g_assert (CAJA_IS_APPLICATION (application));
518
519 ret = TRUE;
520
521 user_directory = caja_get_user_directory ();
522 desktop_directory = caja_get_desktop_directory ();
523
524 directories = NULL;
525
526 if (!g_file_test (user_directory, G_FILE_TEST_IS_DIR))
527 {
528 directories = g_slist_prepend (directories, user_directory);
529 }
530
531 if (!g_file_test (desktop_directory, G_FILE_TEST_IS_DIR))
532 {
533 directories = g_slist_prepend (directories, desktop_directory);
534 }
535
536 if (directories != NULL)
537 {
538 int failed_count;
539 GString *directories_as_string;
540 GSList *l;
541 char *error_string;
542 const char *detail_string;
543 GtkDialog *dialog;
544
545 ret = FALSE;
546
547 failed_count = g_slist_length (directories);
548
549 directories_as_string = g_string_new ((const char *)directories->data);
550 for (l = directories->next; l != NULL; l = l->next)
551 {
552 g_string_append_printf (directories_as_string, ", %s", (const char *)l->data);
553 }
554
555 if (failed_count == 1)
556 {
557 error_string = g_strdup_printf (_("Caja could not create the required folder \"%s\"."),
558 directories_as_string->str);
559 detail_string = _("Before running Caja, please create the following folder, or "
560 "set permissions such that Caja can create it.");
561 }
562 else
563 {
564 error_string = g_strdup_printf (_("Caja could not create the following required folders: "
565 "%s."), directories_as_string->str);
566 detail_string = _("Before running Caja, please create these folders, or "
567 "set permissions such that Caja can create them.");
568 }
569
570 dialog = eel_show_error_dialog (error_string, detail_string, NULL);
571 /* We need the main event loop so the user has a chance to see the dialog. */
572
573 gtk_application_add_window (GTK_APPLICATION (application),
574 GTK_WINDOW (dialog));
575
576 g_string_free (directories_as_string, TRUE);
577 g_free (error_string);
578 }
579
580 g_slist_free (directories);
581 g_free (user_directory);
582 g_free (desktop_directory);
583
584 return ret;
585}

703static void
704caja_application_create_desktop_windows (CajaApplication *application)
705{
706 GdkDisplay *display;
707 GtkWidget *selection_widget;
708
709 g_return_if_fail (caja_application_desktop_windows == NULL);
710 g_return_if_fail (CAJA_IS_APPLICATION (application));
711 display = gdk_display_get_default ();
712
713 selection_widget = get_desktop_manager_selection (display);
714
715 if (selection_widget != NULL)
716 {
717 CajaDesktopWindow *window;
718
719 window = caja_desktop_window_new (application, gdk_display_get_default_screen (display));
720
721 g_signal_connect (selection_widget, "selection_clear_event",
722 G_CALLBACK (selection_clear_event_cb), window);
723

726
727 /* We realize it immediately so that the CAJA_DESKTOP_WINDOW_ID
728 property is set so mate-settings-daemon doesn't try to set the
729 background. And we do a gdk_display_flush() to be sure X gets it. */
730 gtk_widget_realize (GTK_WIDGET (window));
731 gdk_display_flush (display);
732
733 caja_application_desktop_windows =
734 g_list_prepend (caja_application_desktop_windows, window);
735 gtk_application_add_window (GTK_APPLICATION (application),
736 GTK_WINDOW (window));

1297/* Called whenever a mount is unmounted. Check and see if there are
1298 * any windows open displaying contents on the mount. If there are,
1299 * close them. It would also be cool to save open window and position
1300 * info.
1301 *
1302 * This is also called on pre_unmount.
1303 */
1304static void
1305mount_removed_callback (GVolumeMonitor *monitor,
1306 GMount *mount,
1307 CajaApplication *application)
1308{
1309 GList *window_list, *node, *close_list;
1310 CajaWindow *window;
1311 CajaWindowSlot *slot;
1312 CajaWindowSlot *force_no_close_slot;
1313 GFile *root, *computer;
1314 gboolean unclosed_slot;
1315
1316 close_list = NULL;
1317 force_no_close_slot = NULL;
1318 unclosed_slot = FALSE;
1319
1320 /* Check and see if any of the open windows are displaying contents from the unmounted mount */
1321 window_list = gtk_application_get_windows (GTK_APPLICATION (application));
1322 root = g_mount_get_root (mount);
1323 /* Construct a list of windows to be closed. Do not add the non-closable windows to the list. */
1324 for (node = window_list; node != NULL; node = node->next)
1325 {
1326 window = CAJA_WINDOW (node->data);
1327 if (window != NULL && window_can_be_closed (window))
1328 {
1329 GList *l;
1330 GList *lp;
1331 GFile *location;
1332
1333 for (lp = window->details->panes; lp != NULL; lp = lp->next)
1334 {
1335 CajaWindowPane *pane;
1336 pane = (CajaWindowPane*) lp->data;
1337 for (l = pane->slots; l != NULL; l = l->next)
1338 {
1339 slot = l->data;
1340 location = slot->location;
1341 if (g_file_has_prefix (location, root) ||
1342 g_file_equal (location, root))
1343 {
1344 close_list = g_list_prepend (close_list, slot);
1345
1346 if (!should_close_slot_with_mount (window, slot, mount))
1347 {
1348 /* We'll be redirecting this, not closing */
1349 unclosed_slot = TRUE;
1350 }
1351 }
1352 else
1353 {
1354 unclosed_slot = TRUE;
1355 }
1356 } /* for all slots */
1357 } /* for all panes */
1358 }
1359 }
1360
1361 if (caja_application_desktop_windows == NULL &&
1362 !unclosed_slot)
1363 {
1364 /* We are trying to close all open slots. Keep one navigation slot open. */
1365 force_no_close_slot = get_first_navigation_slot (close_list);
1366 }
1367
1368 /* Handle the windows in the close list. */
1369 for (node = close_list; node != NULL; node = node->next)
1370 {
1371 slot = node->data;
1372 window = slot->pane->window;
1373
1374 if (should_close_slot_with_mount (window, slot, mount) &&
1375 slot != force_no_close_slot)
1376 {
1377 caja_window_slot_close (slot);
1378 }
1379 else
1380 {
1381 computer = g_file_new_for_uri ("computer:///");
1382 caja_window_slot_go_to (slot, computer, FALSE);
1383 g_object_unref(computer);
1384 }
1385 }
1386
1387 g_list_free (close_list);
1388}

1390static char *

1392{
1393 xmlDocPtr doc;

1395 GList *l, *window_list;
1396
1397 char *data;
1398 unsigned n_processed;
1399 xmlSaveCtxtPtr ctx;
1400 xmlBufferPtr buffer;
1401
1402 doc = xmlNewDoc ((const xmlChar *) "1.0");
1403
1404 root_node = xmlNewNode (NULL, (const xmlChar *) "session");
1405 xmlDocSetRootElement (doc, root_node);
1406
1407 history_node = xmlNewChild (root_node, NULL, (const xmlChar *) "history", NULL);
1408
1409 n_processed = 0;
1410 for (l = caja_get_history_list (); l != NULL; l = l->next) {
1411 CajaBookmark *bookmark;
1412 xmlNodePtr bookmark_node;
1413 GIcon *icon;
1414 char *tmp;
1415
1416 bookmark = l->data;
1417
1418 bookmark_node = xmlNewChild (history_node, NULL, (const xmlChar *) "bookmark", NULL);
1419

1421 xmlNewProp (bookmark_node, (const xmlChar *) "name", (const xmlChar *) tmp);
1422 g_free (tmp);
1423
1424 icon = caja_bookmark_get_icon (bookmark);
1425
1426 tmp = g_icon_to_string (icon);
1427 g_object_unref (icon);
1428 if (tmp) {
1429 xmlNewProp (bookmark_node, (const xmlChar *) "icon", (const xmlChar *) tmp);
1430 g_free (tmp);
1431 }
1432
1433 tmp = caja_bookmark_get_uri (bookmark);
1434 xmlNewProp (bookmark_node, (const xmlChar *) "uri", (const xmlChar *) tmp);
1435 g_free (tmp);
1436
1437 if (caja_bookmark_get_has_custom_name (bookmark)) {
1438 xmlNewProp (bookmark_node, (const xmlChar *) "has_custom_name", (const xmlChar *) "TRUE");
1439 }
1440
1441 if (++n_processed > 50) { /* prevent history list from growing arbitrarily large. */
1442 break;
1443 }
1444 }
1445 window_list = gtk_application_get_windows (GTK_APPLICATION (self));
1446 for (l = window_list; l != NULL; l = l->next) {
1447 xmlNodePtr win_node, slot_node;
1448 CajaWindow *window;
1449 GList *slots, *m;
1450 char *tmp;
1451 CajaWindowSlot *active_slot;
1452 CajaWindowSlot *slot = NULL;
1453

1455

1457 active_slot = caja_window_get_active_slot (window);
1458

1460 * older Caja versions will bail when reading the file. */
1461 tmp = caja_window_slot_get_location_uri (active_slot);
1462
1463 if (eel_uri_is_desktop (tmp)) {
1464 g_list_free (slots);
1465 g_free (tmp);
1466 continue;
1467 }
1468
1469 win_node = xmlNewChild (root_node, NULL, (const xmlChar *) "window", NULL);
1470
1471 xmlNewProp (win_node, (const xmlChar *) "location", (const xmlChar *) tmp);
1472 g_free (tmp);
1473
1474 xmlNewProp (win_node, (const xmlChar *) "type", CAJA_IS_NAVIGATION_WINDOW (window) ? (const xmlChar *) "navigation" : (const xmlChar *) "spatial");
1475
1476 if (CAJA_IS_NAVIGATION_WINDOW (window)) { /* spatial windows store their state as file metadata */
1477 GdkWindow *gdk_window;
1478
1479 tmp = eel_gtk_window_get_geometry_string (GTK_WINDOW (window));
1480 xmlNewProp (win_node, (const xmlChar *) "geometry", (const xmlChar *) tmp);
1481 g_free (tmp);

1484
1485 if (gdk_window &&
1486 gdk_window_get_state (gdk_window) & GDK_WINDOW_STATE_MAXIMIZED) {
1487 xmlNewProp (win_node, (const xmlChar *) "maximized", (const xmlChar *) "TRUE");
1488 }
1489
1490 if (gdk_window &&
1491 gdk_window_get_state (gdk_window) & GDK_WINDOW_STATE_STICKY) {
1492 xmlNewProp (win_node, (const xmlChar *) "sticky", (const xmlChar *) "TRUE");
1493 }
1494

1538{

1542 GKeyFile *state_file;
1543 char *data;

1547 return;
1548 }

1550 state_file = egg_sm_client_get_state_file (application->smclient);
1551 if (!state_file)

1553 return;
1554 }
1555
1556 data = g_key_file_get_string (state_file,
1557 "Caja",
1558 "documents",
1559 NULL);
1560 if (data == NULL)
1561 {
1562 return;

1583 {

1586
1587 emit_change = FALSE;
1588
1589 for (bookmark_node = node->children; bookmark_node != NULL; bookmark_node = bookmark_node->next)
1590 {

1592 xmlStrcmp (bookmark_node->name, (const xmlChar *) "text") == 0)
1593 {
1594 continue;

1652 g_message ("empty type node while parsing session data");
1653 bail = TRUE;
1654 continue;
1655 }

1660 g_message ("empty location node while parsing session data");
1661 bail = TRUE;

1664 }

1686 }

1688 {
1689 gtk_window_unmaximize (GTK_WINDOW (window));
1690 }
1691

1693 {
1694 gtk_window_stick (GTK_WINDOW (window));
1695 }
1696 else
1697 {
1698 gtk_window_unstick (GTK_WINDOW (window));
1699 }
1700

1704 }
1705 else
1706 {
1707 gtk_window_set_keep_above (GTK_WINDOW (window), FALSE);
1708 }

1711 {

1713 {

1736 }
1737
1738 i++;
1739 }
1740 xmlFree (slot_uri);
1741 }
1742 }
1743
1744 if (i == 0)
1745 {
1746 /* This may be an old session file */
1747 location = g_file_new_for_uri ((const char *) location_uri);
1748 caja_window_slot_open_location (window->details->active_pane->active_slot, location, FALSE);
1749 g_object_unref (location);
1750 }
1751 }
1752 else if (xmlStrcmp (type, (const xmlChar *) "spatial") == 0)
1753 {

1769 }

1771 {
1772 g_message ("unexpected node %s while parsing session data", node->name);
1773 bail = TRUE;
1774 continue;
1775 }
1776 }
1777 }
1778
1779 if (doc != NULL)
1780 {
1781 xmlFreeDoc (doc);
1782 }
1783
1784 g_free (data);

1786 if (bail)
1787 {
1788 g_message ("failed to load session");
1789 }
