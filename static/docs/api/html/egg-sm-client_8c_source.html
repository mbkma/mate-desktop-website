192void
193egg_sm_client_startup (void)
194{
195 if( has_startup_run )
196 return;
197
198 EggSMClient *client = egg_sm_client_get ();
199
200 if (sm_client_id == NULL)
201 {
202 const gchar *desktop_autostart_id;
203
204 desktop_autostart_id = g_getenv ("DESKTOP_AUTOSTART_ID");
205
206 if (desktop_autostart_id != NULL)
207 sm_client_id = g_strdup (desktop_autostart_id);
208 }
209
210 /* Unset DESKTOP_AUTOSTART_ID in order to avoid child processes to
211 * use the same client id. */
212 g_unsetenv ("DESKTOP_AUTOSTART_ID");
213
214 if (EGG_SM_CLIENT_GET_CLASS (client)->startup)
215 EGG_SM_CLIENT_GET_CLASS (client)->startup (client, sm_client_id);
216
217 has_startup_run = TRUE;
218}

326EggSMClient *
327egg_sm_client_get (void)
328{
329 if (!global_client)
330 {
331 if (global_client_mode != EGG_SM_CLIENT_MODE_DISABLED &&
332 !sm_client_disable)
333 {
334#if defined (GDK_WINDOWING_WIN32)
335 global_client = egg_sm_client_win32_new ();
336#elif defined (GDK_WINDOWING_QUARTZ)
337 global_client = egg_sm_client_osx_new ();
338#else
339 /* If both D-Bus and XSMP are compiled in, try XSMP first
340 * (since it supports state saving) and fall back to D-Bus
341 * if XSMP isn't available.
342 */
343# ifdef EGG_SM_CLIENT_BACKEND_XSMP
344 g_debug( "egg_sm_client_get: egg_sm_client_xsmp_new" );
345 global_client = egg_sm_client_xsmp_new ();
346# endif
347# ifdef EGG_SM_CLIENT_BACKEND_DBUS
348 if (!global_client)
349 global_client = egg_sm_client_dbus_new ();
350# endif
351#endif
352 }
353
354 /* Fallback: create a dummy client, so that callers don't have
355 * to worry about a %NULL return value.
356 */
357 if (!global_client) {
358 g_debug( "egg_sm_client_get: allocating dummy client" );
359 global_client = g_object_new (EGG_TYPE_SM_CLIENT, NULL);
360 }
361 }
362
363 return global_client;
364}

404GKeyFile *
405egg_sm_client_get_state_file (EggSMClient *client)
406{
407 EggSMClientPrivate *priv = egg_sm_client_get_instance_private (client);
408 char *state_file_path;
409 GError *err = NULL;
410
411 g_return_val_if_fail (client == global_client, NULL);
412
413 if (!sm_client_state_file)
414 return NULL;
415 if (priv->state_file)
416 return priv->state_file;
417
418 if (!strncmp (sm_client_state_file, "file://", 7))
419 state_file_path = g_filename_from_uri (sm_client_state_file, NULL, NULL);
420 else
421 state_file_path = g_strdup (sm_client_state_file);
422
423 priv->state_file = g_key_file_new ();
424 if (!g_key_file_load_from_file (priv->state_file, state_file_path, 0, &err))
425 {
426 g_warning ("Could not load SM state file '%s': %s",
427 sm_client_state_file, err->message);
428 g_clear_error (&err);
429 g_key_file_free (priv->state_file);
430 priv->state_file = NULL;
431 }
432
433 g_free (state_file_path);
434 return priv->state_file;
435}
