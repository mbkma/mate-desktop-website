238/*
239 * na_tokens_new_for_example:
240 *
241 * Returns: a new #NATokens object initialized with fake values for two
242 * regular files, in order to be used as an example of an expanded command
243 * line.
244 */
245NATokens *
246na_tokens_new_for_example( void )
247{
248 NATokens *tokens;
249 const gchar *ex_uri1 = _( "file:///path/to/file1.mid" );
250 const gchar *ex_uri2 = _( "file:///path/to/file2.jpeg" );
251 const gchar *ex_mimetype1 = _( "audio/x-midi" );
252 const gchar *ex_mimetype2 = _( "image/jpeg" );
253 const gint ex_port = 8080;
254 const gchar *ex_host = _( "test.example.net" );
255 const gchar *ex_user = _( "user" );
256 gchar *dirname, *bname, *bname_woext, *ext;
257 GSList *is;
258 gboolean first;
259
260 tokens = g_object_new( NA_TYPE_TOKENS, NULL );
261 first = TRUE;
262 tokens->private->count = 2;
263
264 tokens->private->uris = g_slist_append( tokens->private->uris, g_strdup( ex_uri1 ));
265 tokens->private->uris = g_slist_append( tokens->private->uris, g_strdup( ex_uri2 ));
266
267 for (is = tokens->private->uris; is; is = is->next) {
268 gchar *path = NULL;
269 gchar *scheme = NULL;
270
271 g_uri_split (is->data,
272 G_URI_FLAGS_NONE,
273 &scheme,
274 NULL, /* userinfo */
275 NULL, /* host */
276 NULL, /* port */
277 &path,
278 NULL, /* query */
279 NULL, /* fragment*/
280 NULL); /* error */
281
282 tokens->private->filenames = g_slist_append (tokens->private->filenames, g_strdup (path));
283 dirname = g_path_get_dirname (path);
284 tokens->private->basedirs = g_slist_append( tokens->private->basedirs, dirname );
285 bname = g_path_get_basename (path);
286 tokens->private->basenames = g_slist_append( tokens->private->basenames, bname );
287 na_core_utils_dir_split_ext( bname, &bname_woext, &ext );
288 tokens->private->basenames_woext = g_slist_append( tokens->private->basenames_woext, bname_woext );
289 tokens->private->exts = g_slist_append( tokens->private->exts, ext );
290
291 if( first ){
292 tokens->private->scheme = g_strdup (scheme);
293 first = FALSE;
294 }
295
296 g_clear_pointer (&scheme, g_free);
297 g_clear_pointer (&path, g_free);
298 }
299
300 tokens->private->mimetypes = g_slist_append( tokens->private->mimetypes, g_strdup( ex_mimetype1 ));
301 tokens->private->mimetypes = g_slist_append( tokens->private->mimetypes, g_strdup( ex_mimetype2 ));
302
303 tokens->private->hostname = g_strdup( ex_host );
304 tokens->private->username = g_strdup( ex_user );
305 tokens->private->port = ex_port;
306
307 return( tokens );
308}

310/*
311 * na_tokens_new_from_selection:
312 * @selection: a #GList list of #NASelectedInfo objects.
313 *
314 * Returns: a new #NATokens object which holds all possible tokens.
315 */
316NATokens *
317na_tokens_new_from_selection( GList *selection )
318{
319 static const gchar *thisfn = "na_tokens_new_from_selection";
320 NATokens *tokens;
321 GList *it;
322 gchar *uri, *filename, *basedir, *basename, *bname_woext, *ext, *mimetype;
323 gboolean first;
324
325 g_debug( "%s: selection=%p (count=%d)", thisfn, ( void * ) selection, g_list_length( selection ));
326
327 first = TRUE;
328 tokens = g_object_new( NA_TYPE_TOKENS, NULL );
329
330 tokens->private->count = g_list_length( selection );
331
332 for( it = selection ; it ; it = it->next ){
333 mimetype = na_selected_info_get_mime_type( NA_SELECTED_INFO( it->data ));
334
335 uri = na_selected_info_get_uri( NA_SELECTED_INFO( it->data ));
336 filename = na_selected_info_get_path( NA_SELECTED_INFO( it->data ));
337 basedir = na_selected_info_get_dirname( NA_SELECTED_INFO( it->data ));
338 basename = na_selected_info_get_basename( NA_SELECTED_INFO( it->data ));
339 na_core_utils_dir_split_ext( basename, &bname_woext, &ext );
340
341 if( first ){
342 tokens->private->hostname = na_selected_info_get_uri_host( NA_SELECTED_INFO( it->data ));
343 tokens->private->username = na_selected_info_get_uri_user( NA_SELECTED_INFO( it->data ));
344 tokens->private->port = na_selected_info_get_uri_port( NA_SELECTED_INFO( it->data ));
345 tokens->private->scheme = na_selected_info_get_uri_scheme( NA_SELECTED_INFO( it->data ));
346 first = FALSE;
347 }
348
349 tokens->private->uris = g_slist_append( tokens->private->uris, uri );
350 tokens->private->filenames = g_slist_append( tokens->private->filenames, filename );
351 tokens->private->basedirs = g_slist_append( tokens->private->basedirs, basedir );
352 tokens->private->basenames = g_slist_append( tokens->private->basenames, basename );
353 tokens->private->basenames_woext = g_slist_append( tokens->private->basenames_woext, bname_woext );
354 tokens->private->exts = g_slist_append( tokens->private->exts, ext );
355 tokens->private->mimetypes = g_slist_append( tokens->private->mimetypes, mimetype );
356 }
357
358 return( tokens );
359}

381/*
382 * na_tokens_execute_action:
383 * @tokens: a #NATokens object.
384 * @profile: the #NAObjectProfile to be executed.
385 *
386 * Execute the given action, regarding the context described by @tokens.
387 */
388void
389na_tokens_execute_action( const NATokens *tokens, const NAObjectProfile *profile )
390{
391 gchar *path, *parameters, *exec;
392 gboolean singular;
393 guint i;
394 gchar *command;
395
396 path = na_object_get_path( profile );
397 parameters = na_object_get_parameters( profile );
398 exec = g_strdup_printf( "%s %s", path, parameters );
399 g_free( parameters );
400 g_free( path );
401
402 singular = is_singular_exec( tokens, exec );
403
404 if( singular ){
405 for( i = 0 ; i < tokens->private->count ; ++i ){
406 command = parse_singular( tokens, exec, i, FALSE, TRUE );
407 execute_action_command( command, profile, tokens );
408 g_free( command );
409 }
410
411 } else {
412 command = parse_singular( tokens, exec, 0, FALSE, TRUE );
413 execute_action_command( command, profile, tokens );
414 g_free( command );
415 }
416
417 g_free( exec );
418}

460static gchar *
461display_output_get_content( int fd )
462{
463 static const gchar *thisfn = "na_tokens_display_output_get_content";
464 GInputStream *stream;
465 GString *string;
466 gchar buf[1024];
467 GError *error;
468 gchar *msg;
469
470 string = g_string_new( "" );
471 memset( buf, '\0', sizeof( buf ));
472
473 if( fd > 0 ){
474 stream = g_unix_input_stream_new( fd, TRUE );
475 error = NULL;
476
477 while( g_input_stream_read( stream, buf, sizeof( buf )-1, NULL, &error )){
478 string = g_string_append( string, buf );
479 memset( buf, '\0', sizeof( buf ));
480 }
481 if( error ){
482 g_warning( "%s: g_input_stream_read: %s", thisfn, error->message );
483 g_error_free( error );
484 }
485 g_input_stream_close( stream, NULL, NULL );
486 }
487
488 msg = g_locale_to_utf8( string->str, -1, NULL, NULL, NULL );
489 g_string_free( string, TRUE );
490
491 return( msg );
492}

494/*
495 * Execution environment:
496 * - Normal: just execute the specified command
497 * - Terminal: use the user preference to have a terminal which stays openeded
498 * - Embedded: id. Terminal
499 * - DisplayOutput: execute in a shell
500 */
501static void
502execute_action_command( gchar *command, const NAObjectProfile *profile, const NATokens *tokens )
503{
504 static const gchar *thisfn = "caja_actions_execute_action_command";
505 GError *error;
506 gchar *execution_mode, *run_command;
507 gchar **argv;
508 gint argc;
509 gchar *wdir, *wdir_nq;
510 GPid child_pid;
511 ChildStr *child_str;
512
513 g_debug( "%s: profile=%p", thisfn, ( void * ) profile );
514
515 error = NULL;
516 run_command = NULL;
517 child_str = g_new0( ChildStr, 1 );
518 child_pid = ( GPid ) 0;
519 execution_mode = na_object_get_execution_mode( profile );
520
521 if( !strcmp( execution_mode, "Normal" )){
522 run_command = get_command_execution_normal( command );
523
524 } else if( !strcmp( execution_mode, "Terminal" )){
525 run_command = get_command_execution_terminal( command );
526
527 } else if( !strcmp( execution_mode, "Embedded" )){
528 run_command = get_command_execution_embedded( command );
529
530 } else if( !strcmp( execution_mode, "DisplayOutput" )){
531 child_str->is_output_displayed = TRUE;
532 run_command = get_command_execution_display_output( command );
533
534 } else {
535 g_warning( "%s: unknown execution mode: %s", thisfn, execution_mode );
536 }
537
538 if( run_command ){
539 child_str->command = g_strdup( run_command );
540
541 if( !g_shell_parse_argv( run_command, &argc, &argv, &error )){
542 g_warning( "%s: g_shell_parse_argv: %s", thisfn, error->message );
543 g_error_free( error );
544
545 } else {
546 wdir = na_object_get_working_dir( profile );
547 wdir_nq = parse_singular( tokens, wdir, 0, FALSE, FALSE );
548 g_debug( "%s: run_command=%s, wdir=%s", thisfn, run_command, wdir_nq );
549
550 /* it appears that at least mplayer does not support g_spawn_async_with_pipes
551 * (at least when not run in '-quiet' mode) while, e.g., totem and vlc rightly
552 * support this function
553 * So only use g_spawn_async_with_pipes when we really need to get back
554 * the content of output and error streams
555 * See https://bugzilla.gnome.org/show_bug.cgi?id=644289.
556 */
557 if( child_str->is_output_displayed ){
558 g_spawn_async_with_pipes(
559 wdir_nq,
560 argv,
561 NULL,
562 G_SPAWN_SEARCH_PATH | G_SPAWN_DO_NOT_REAP_CHILD,
563 NULL,
564 NULL,
565 &child_pid,
566 NULL,
567 &child_str->child_stdout,
568 &child_str->child_stderr,
569 &error );
570
571 } else {
572 g_spawn_async(
573 wdir_nq,
574 argv,
575 NULL,
576 G_SPAWN_SEARCH_PATH | G_SPAWN_DO_NOT_REAP_CHILD,
577 NULL,
578 NULL,
579 &child_pid,
580 &error );
581 }
582
583 if( error ){
584 g_warning( "%s: g_spawn_async: %s", thisfn, error->message );
585 g_error_free( error );
586 child_pid = ( GPid ) 0;
587
588 } else {
589 g_child_watch_add( child_pid, ( GChildWatchFunc ) child_watch_fn, child_str );
590 }
591
592 g_free( wdir );
593 g_free( wdir_nq );
594 g_strfreev( argv );
595 }
596
597 g_free( run_command );
598 }
599
600 g_free( execution_mode );
601
602 if( child_pid == ( GPid ) 0 ){
603 g_free( child_str->command );
604 g_free( child_str );
605 }
606}

649na_tokens_command_for_terminal( const gchar *pattern, const gchar *command )
650{
651 gchar *run_command;

653
654 if( pattern && strlen( pattern )){

657 g_free( quoted );
658
659 } else {
660 run_command = g_strdup( command );
661 }
662
663 return( run_command );

666/*
667 * na_tokens_is_singular_exec:
668 * @tokens: the current #NATokens object.
669 * @exec: the to be executed command-line before having been parsed
670 *
671 * Returns: %TRUE if the first relevant parameter found in @exec
672 * command-line is of singular form, %FALSE else.
673 */
674static gboolean
675is_singular_exec( const NATokens *tokens, const gchar *exec )
676{
677 gboolean singular;
678 gboolean found;
679 gchar *iter;
680
681 singular = FALSE;
682 found = FALSE;
683 iter = ( gchar * ) exec;
684
685 while(( iter = g_strstr_len( iter, -1, "%" )) != NULL && !found ){
686
687 switch( iter[1] ){
688 case 'b':
689 case 'd':
690 case 'f':
691 case 'm':
692 case 'o':
693 case 'u':
694 case 'w':
695 case 'x':
696 found = TRUE;
697 singular = TRUE;
698 break;
699
700 case 'B':
701 case 'D':
702 case 'F':

729 * parse_singular:
730 * @tokens: a #NATokens object.
731 * @input: the input string, may or may not contain tokens.
732 * @i: the number of the iteration in a multiple selection, starting with zero.
733 * @utf8: whether the @input string is UTF-8 encoded, or a standard ASCII
734 * string.
735 * @quoted: whether the filenames have to be quoted (should be %TRUE when
736 * about to execute a command).
737 *
738 * A command is said of 'singular form' when its first parameter is not
739 * of plural form. In the case of a multiple selection, singular form
740 * commands are executed one time for each element of the selection
741 *
742 * Returns: a #GSList which contains two fields: the command and its parameters.
743 * The returned #GSList should be na_core_utils_slist_free() by the caller.
744 */
745static gchar *
746parse_singular( const NATokens *tokens, const gchar *input, guint i, gboolean utf8, gboolean quoted )
747{
748 GString *output;
749 gchar *iter, *prev_iter;
750 const gchar *nth;
751
752 output = g_string_new( "" );
753
754 /* return NULL if input is NULL
755 */
756 if( !input ){
757 return( g_string_free( output, TRUE ));
758 }
759
760 /* return an empty string if input is empty
761 */
762 if( utf8 ){
763 if( !g_utf8_strlen( input, -1 )){
764 return( g_string_free( output, FALSE ));
765 }
766 } else {
767 if( !strlen( input )){
768 return( g_string_free( output, FALSE ));
769 }
770 }
771
772 iter = ( gchar * ) input;
773 prev_iter = iter;
774
775 while(( iter = g_strstr_len( iter, -1, "%" ))){
776 output = g_string_append_len( output, prev_iter, strlen( prev_iter ) - strlen( iter ));
777
778 switch( iter[1] ){
779 case 'b':
780 if( tokens->private->basenames ){
781 nth = ( const gchar * ) g_slist_nth_data( tokens->private->basenames, i );
782 if( nth ){
783 output = quote_string( output, nth, quoted );
784 }
785 }
786 break;
787
788 case 'B':
789 if( tokens->private->basenames ){
790 output = quote_string_list( output, tokens->private->basenames, quoted );
791 }
792 break;
793
794 case 'c':
795 g_string_append_printf( output, "%d", tokens->private->count );
796 break;
797
798 case 'd':
799 if( tokens->private->basedirs ){
800 nth = ( const gchar * ) g_slist_nth_data( tokens->private->basedirs, i );
801 if( nth ){
802 output = quote_string( output, nth, quoted );
803 }
804 }
805 break;
806
807 case 'D':
808 if( tokens->private->basedirs ){
809 output = quote_string_list( output, tokens->private->basedirs, quoted );
810 }
811 break;
812
813 case 'f':
814 if( tokens->private->filenames ){
815 nth = ( const gchar * ) g_slist_nth_data( tokens->private->filenames, i );
816 if( nth ){
817 output = quote_string( output, nth, quoted );
818 }
819 }
820 break;
821
822 case 'F':
823 if( tokens->private->filenames ){
824 output = quote_string_list( output, tokens->private->filenames, quoted );
825 }
826 break;
827
828 case 'h':
829 if( tokens->private->hostname ){
830 output = quote_string( output, tokens->private->hostname, quoted );
831 }
832 break;
833
834 /* mimetypes are never quoted
835 */
836 case 'm':
837 if( tokens->private->mimetypes ){
838 nth = ( const gchar * ) g_slist_nth_data( tokens->private->mimetypes, i );
839 if( nth ){
840 output = quote_string( output, nth, FALSE );
841 }
842 }
843 break;
844
845 case 'M':
846 if( tokens->private->mimetypes ){
847 output = quote_string_list( output, tokens->private->mimetypes, FALSE );
848 }
849 break;
850
851 /* no-op operators */
852 case 'o':
853 case 'O':
854 break;
855
856 case 'n':
857 if( tokens->private->username ){
858 output = quote_string( output, tokens->private->username, quoted );
859 }
860 break;
861
862 /* port number is never quoted
863 */
864 case 'p':
865 if( tokens->private->port > 0 ){
866 g_string_append_printf( output, "%d", tokens->private->port );
867 }
868 break;
869
870 case 's':
871 if( tokens->private->scheme ){
872 output = quote_string( output, tokens->private->scheme, quoted );
873 }
874 break;
875
876 case 'u':
877 if( tokens->private->uris ){
878 nth = ( const gchar * ) g_slist_nth_data( tokens->private->uris, i );
879 if( nth ){
880 output = quote_string( output, nth, quoted );
881 }
882 }
883 break;
884
885 case 'U':
886 if( tokens->private->uris ){
887 output = quote_string_list( output, tokens->private->uris, quoted );
888 }
889 break;
890
891 case 'w':
892 if( tokens->private->basenames_woext ){
893 nth = ( const gchar * ) g_slist_nth_data( tokens->private->basenames_woext, i );
894 if( nth ){
895 output = quote_string( output, nth, quoted );
896 }
897 }
898 break;
899
900 case 'W':
901 if( tokens->private->basenames_woext ){
902 output = quote_string_list( output, tokens->private->basenames_woext, quoted );
903 }
904 break;
905
906 case 'x':
907 if( tokens->private->exts ){
908 nth = ( const gchar * ) g_slist_nth_data( tokens->private->exts, i );
909 if( nth ){
910 output = quote_string( output, nth, quoted );
911 }
912 }
913 break;
914
915 case 'X':
916 if( tokens->private->exts ){
917 output = quote_string_list( output, tokens->private->exts, quoted );
918 }
919 break;
920
921 /* a percent sign
922 */
923 case '%':
924 output = g_string_append_c( output, '%' );
925 break;
926 }
927
928 iter += 2; /* skip the % sign and the character after */
929 prev_iter = iter; /* store the new start of the string */
930 }
931
932 output = g_string_append_len( output, prev_iter, strlen( prev_iter ));
933
934 return( g_string_free( output, FALSE ));
935}
