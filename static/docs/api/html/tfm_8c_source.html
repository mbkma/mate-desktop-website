74/*
75 * Although it does not seem that way, this conversion is independent of the
76 * shrinking factors, within roundoff (that's because `conv' and `vconv'
77 * have already been scaled by hshrink and vshrink, repsectively). We
78 * should really use `dviconv' and `dvivconv', but I'm not so sure those
79 * should be moved to the DviParams structure.
80 */
81#define XCONV(x) FROUND(params->conv * (x) * params->hshrink)
82#define YCONV(y) FROUND(params->vconv * (y) * params->vshrink)

84/* this is used quite often in several places, so I made it standalone */
85int get_tfm_chars(DviParams *params, DviFont *font, TFMInfo *info, int loaded)
86{
87 Int32 z, alpha, beta;
88 int n;
89 DviFontChar *ch;
90 TFMChar *ptr;
91
92 n = info->hic - info->loc + 1;
93 if(n != FONT_GLYPH_COUNT(font)) {
94 font->chars = mdvi_realloc(font->chars,
95 n * sizeof(DviFontChar));
96 }
97 font->loc = info->loc;
98 font->hic = info->hic;
99 ch = font->chars;
100 ptr = info->chars;
101
102 /* Prepare z, alpha and beta for TFM width computation */
103 TFMPREPARE(font->scale, z, alpha, beta);
104
105 /* get the character metrics */
106 for(n = info->loc; n <= info->hic; ch++, ptr++, n++) {
107 int a, b, c, d;
108
109 ch->offset = ptr->present;
110 if(ch->offset == 0)
111 continue;
112 /* this is what we came here for */
113 ch->tfmwidth = TFMSCALE(z, ptr->advance, alpha, beta);
114 /* scale all other TFM units (so they are in DVI units) */
115 a = TFMSCALE(z, ptr->left, alpha, beta);
116 b = TFMSCALE(z, ptr->right, alpha, beta);
117 c = TFMSCALE(z, ptr->height, alpha, beta);
118 d = TFMSCALE(z, ptr->depth, alpha, beta);
119
120 /* now convert to unscaled pixels */
121 ch->width = XCONV(b - a);
122 ch->height = YCONV(c - d);
123 if(ch->height < 0) ch->height = -ch->height;
124 ch->x = XCONV(a);
125 ch->y = YCONV(c);
126 /*
127 * the offset is not used, but we might as well set it to
128 * something meaningful (and it MUST be non-zero)
129 */
130 ch->flags = 0;
131 ch->code = n;
132 ch->glyph.data = NULL;
133 ch->grey.data = NULL;
134 ch->shrunk.data = NULL;
135 ch->loaded = loaded;
136 }
137
138 return 0;
139}

141/*
142 * We use this function as a last resort to find the character widths in a
143 * font The DVI rendering code can correctly skip over a glyph if it knows
144 * its TFM width, which is what we try to find here.
145 */
146static int tfm_load_font(DviParams *params, DviFont *font)
147{
148 TFMInfo *tfm;
149 int type;
150
151 switch(font->search.info->kpse_type) {
152 case kpse_tfm_format:
153 type = DviFontTFM;
154 break;
155 case kpse_afm_format:
156 type = DviFontAFM;
157 break;
158 case kpse_ofm_format:
159 type = DviFontOFM;
160 break;
161 default:
162 return -1;
163 }
164
165 /* we don't need this */
166 if(font->in) {
167 fclose(font->in);
168 font->in = NULL;
169 }
170 tfm = get_font_metrics(font->fontname, type, font->filename);
171 if(tfm == NULL)
172 return -1;
173
174 if(tfm->checksum && font->checksum && tfm->checksum != font->checksum) {
175 mdvi_warning(_("%s: Checksum mismatch (got %u, expected %u)\n"),
176 font->fontname, (unsigned)tfm->checksum,
177 (unsigned)font->checksum);
178 }
179 font->checksum = tfm->checksum;
180 font->design = tfm->design;
181 font->loc = 0;
182 font->hic = 0;
183 font->chars = NULL;
184 get_tfm_chars(params, font, tfm, 1);
185
186 /* free everything */
187 free_font_metrics(tfm);
188
189 return 0;
190}

192static int tfm_font_get_glyph(DviParams *params, DviFont *font, int code)
193{
194 DviFontChar *ch;
195
196 ch = FONTCHAR(font, code);
197 if(!glyph_present(ch))
198 return -1;
199 ch->glyph.x = ch->x;
200 ch->glyph.y = ch->y;
201 ch->glyph.w = ch->width;
202 ch->glyph.h = ch->height;
203 /*
204 * This has two purposes: (1) avoid unnecessary calls to this function,
205 * and (2) detect when the glyph data for a TFM font is actually used
206 * (we'll get a SEGV). Any occurrence of that is a bug.
207 */
208 ch->glyph.data = MDVI_GLYPH_EMPTY;
209
210 return 0;
211}
