185static const char *
186apm_readinfo (BatteryStatus *status)
187{
188 int fd;
189
190 if (DEBUG) g_print ("apm_readinfo () (FreeBSD)\n");
191
192 if (using_acpi) {
193 if (acpi_count <= 0) {
194 acpi_count = 30;
195 acpi_process_event (&acpiinfo);
196 if (acpi_freebsd_read (&apminfo, &acpiinfo) == FALSE)
197 return ERR_FREEBSD_ACPI;
198 }
199 acpi_count--;
200 }
201 else
202 {
203 /* This is how I read the information from the APM subsystem under
204 FreeBSD. Each time this functions is called (once every second)
205 the APM device is opened, read from and then closed.
206 */
207 fd = open (APMDEVICE, O_RDONLY);
208 if (fd == -1) {
209 return ERR_OPEN_APMDEV;
210 }
211
212 if (ioctl (fd, APMIO_GETINFO, &apminfo) == -1)
213 err (1, "ioctl (APMIO_GETINFO)");
214
215 close (fd);
216
217 if (apminfo.ai_status == 0)
218 return ERR_APM_E;
219 }
220
221 status->present = TRUE;
222 status->on_ac_power = apminfo.ai_acline ? 1 : 0;
223 status->percent = apminfo.ai_batt_life;
224 status->charging = (apminfo.ai_batt_stat) ? TRUE : FALSE;
225 if (using_acpi)
226 status->minutes = apminfo.ai_batt_time;
227 else
228 status->minutes = (int) (apminfo.ai_batt_time/60.0);
229
230 return NULL;
231}

293static gboolean acpi_callback (GIOChannel * chan, GIOCondition cond, gpointer data)
294{
295 if (cond & (G_IO_ERR | G_IO_HUP)) {
296 acpi_linux_cleanup (&acpiinfo);
297 apminfo.battery_percentage = -1;
298 return FALSE;
299 }
300
301 if (acpi_process_event (&acpiinfo)) {
302 acpi_linux_read (&apminfo, &acpiinfo);
303 }
304 return TRUE;
305}

307static const char *
308apm_readinfo (BatteryStatus *status)
309{
310 /* Code for Linux by Thomas Hood <jdthood@mail.com>. apm_read () will
311 read from /proc/... instead and we do not need to open the device
312 ourselves.
313 */
314 if (DEBUG) g_print ("apm_readinfo () (Linux)\n");
315
316 /* ACPI support added by Lennart Poettering <lennart@poettering.de> 10/27/2001
317 * Updated by David Moore <dcm@acm.org> 5/29/2003 to poll less and
318 * use ACPI events. */
319 if (using_acpi && acpiinfo.event_fd >= 0) {
320 if (acpi_count <= 0) {
321 /* Only call this one out of 30 calls to apm_readinfo () (every 30 seconds)
322 * since reading the ACPI system takes CPU cycles. */
323 acpi_count=30;
324 acpi_linux_read (&apminfo, &acpiinfo);
325 }
326 acpi_count--;
327 }
328 /* If we lost the file descriptor with ACPI events, try to get it back. */
329 else if (using_acpi) {
330 if (acpi_linux_init (&acpiinfo)) {
331 acpiwatch = g_io_add_watch (acpiinfo.channel,
332 G_IO_IN | G_IO_ERR | G_IO_HUP,
333 acpi_callback, NULL);
334 acpi_linux_read (&apminfo, &acpiinfo);
335 }
336 }
337 else
338 apm_read (&apminfo);
339
340 status->present = TRUE;
341 status->on_ac_power = apminfo.ac_line_status ? 1 : 0;
342 status->percent = (guint) apminfo.battery_percentage;
343 status->charging = (apminfo.battery_flags & 0x8) ? TRUE : FALSE;
344 status->minutes = apminfo.battery_time;
345
346 return NULL;
347}

369/*
370 * power_management_getinfo
371 *
372 * Main interface to the power management code. Fills 'status' for you so
373 * you don't have to worry about platform-specific details.
374 *
375 * In the case of success, NULL is returned. In case of failure, a
376 * localised error message is returned to give the user hints about what
377 * the problem might be. This error message is not to be freed.
378 */
379const char *
380power_management_getinfo (BatteryStatus *status)
381{
382 const char *retval;
383
384 if (!pm_initialised)
385 {
386 status->on_ac_power = TRUE;
387 status->minutes = -1;
388 status->percent = 0;
389 status->charging = FALSE;
390 status->present = FALSE;
391
392 return NULL;
393 }
394
395#ifdef HAVE_UPOWER
396 if (using_upower)
397 {
398 battstat_upower_get_battery_info (status);
399 return NULL;
400 }
401#endif
402
403 retval = apm_readinfo (status);
404
405 if (status->percent == -1) {
406 status->percent = 0;
407 status->present = FALSE;
408 }
409
410 if (status->percent > 100)
411 status->percent = 100;
412
413 if (status->percent == 100)
414 status->charging = FALSE;
415
416 if (!status->on_ac_power)
417 status->charging = FALSE;
418
419 return retval;
420}

422/*
423 * power_management_initialise
424 *
425 * Initialise the power management code. Call this before you call anything
426 * else.
427 *
428 * In the case of success, NULL is returned. In case of failure, a
429 * localised error message is returned to give the user hints about what
430 * the problem might be. This error message is not to be freed.
431 */
432const char *
433power_management_initialise (void (*callback) (void))
434{
435#ifdef __linux__
436 struct stat statbuf;
437#endif /* __linux__ */
438#ifdef HAVE_UPOWER
439 char *err;
440
441 err = battstat_upower_initialise (callback);
442
443 if (err == NULL) /* UPOWER is up */
444 {
445 pm_initialised = TRUE;
446 using_upower = TRUE;
447 return NULL;
448 }
449 else
450 /* fallback to legacy methods */
451 g_free (err);
452#endif /* HAVE_UPOWER */
453
454#ifdef __linux__
455
456 if (acpi_linux_init (&acpiinfo)) {
457 using_acpi = TRUE;
458 acpi_count = 0;
459 }
460 else
461 using_acpi = FALSE;
462
463 /* If neither ACPI nor APM could be read, but ACPI does seem to be
464 * installed, warn the user how to get ACPI working. */
465 if (!using_acpi && (apm_exists () == 1) &&
466 (stat ("/proc/acpi", &statbuf) == 0)) {
467 using_acpi = TRUE;
468 acpi_count = 0;
469 return ERR_ACPID;
470 }
471
472 /* Watch for ACPI events and handle them immediately with acpi_callback (). */
473 if (using_acpi && acpiinfo.event_fd >= 0) {
474 acpiwatch = g_io_add_watch (acpiinfo.channel,
475 G_IO_IN | G_IO_ERR | G_IO_HUP,
476 acpi_callback, NULL);
477 }
478#elif defined(__FreeBSD__) || defined(__FreeBSD_kernel__)
479 if (acpi_freebsd_init (&acpiinfo)) {
480 using_acpi = TRUE;
481 acpi_count = 0;
482 }
483 else
484 using_acpi = FALSE;
485#endif
486 pm_initialised = TRUE;
487
488 return NULL;
489}

506#endif
507
508#ifdef __linux__

513 acpiwatch = 0;
514 acpi_linux_cleanup (&acpiinfo);
515 }
516#elif defined(__FreeBSD__) || defined(__FreeBSD_kernel__)
517 if (using_acpi) {
518 acpi_freebsd_cleanup (&acpiinfo);
519 }
520#endif
521
522 pm_initialised = FALSE;
523}
