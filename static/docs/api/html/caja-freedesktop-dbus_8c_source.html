58static gboolean
59skeleton_handle_show_items_cb (CajaFreedesktopFileManager1 *object,
60 GDBusMethodInvocation *invocation,
61 const gchar *const *uris,
62 const gchar *startup_id,
63 CajaFreedesktopDBus *fdb)
64{
65 CajaApplication *application;
66 int i;
67
68 application = CAJA_APPLICATION (fdb->application);
69
70 for (i = 0; uris[i] != NULL; i++) {
71 GFile *file;
72 GFile *parent;
73
74 file = g_file_new_for_uri (uris[i]);
75 parent = g_file_get_parent (file);
76
77 if (parent != NULL) {
78 caja_application_open_location (application, parent, file, startup_id, 0);
79 g_object_unref (parent);
80 } else {
81 caja_application_open_location (application, file, NULL, startup_id, 0);
82 }
83
84 g_object_unref (file);
85 }
86
87 caja_freedesktop_file_manager1_complete_show_items (object, invocation);
88 return TRUE;
89}

117static gboolean
118skeleton_handle_show_item_properties_cb (CajaFreedesktopFileManager1 *object,
119 GDBusMethodInvocation *invocation,
120 const gchar *const *uris,
121 const gchar *startup_id,
122 CajaFreedesktopDBus *fdb)
123{
124 CajaApplication *application;
125 GList *files;
126 int i;
127
128 application = CAJA_APPLICATION (fdb->application);
129 files = NULL;
130
131 for (i = 0; uris[i] != NULL; i++) {
132 files = g_list_prepend (files, caja_file_get_by_uri (uris[i]));
133 }
134
135 files = g_list_reverse (files);
136
137 if (uris[0] != NULL) {
138 GFile *file;
139 CajaWindow *window;
140
141 file = g_file_new_for_uri (uris[i]);
142 window = caja_application_get_spatial_window (application,
143 NULL,
144 startup_id,
145 file,
146 gdk_screen_get_default (),
147 NULL);
148 fm_properties_window_present (files, GTK_WIDGET (window));
149 g_object_unref (file);
150 }
151
152 caja_file_list_free (files);
153
154 caja_freedesktop_file_manager1_complete_show_item_properties (object, invocation);
155 return TRUE;
156}

158static void
159bus_acquired_cb (GDBusConnection *conn,
160 const gchar *name,
161 gpointer user_data)
162{
163 CajaFreedesktopDBus *fdb = user_data;
164
165 caja_debug_log (FALSE, CAJA_DEBUG_LOG_DOMAIN_USER, "Bus acquired at %s", name);
166
167 fdb->object_manager = g_dbus_object_manager_server_new (CAJA_FDO_DBUS_PATH);
168
169 fdb->skeleton = caja_freedesktop_file_manager1_skeleton_new ();
170
171 g_signal_connect (fdb->skeleton, "handle-show-items",
172 G_CALLBACK (skeleton_handle_show_items_cb), fdb);
173 g_signal_connect (fdb->skeleton, "handle-show-folders",
174 G_CALLBACK (skeleton_handle_show_folders_cb), fdb);
175 g_signal_connect (fdb->skeleton, "handle-show-item-properties",
176 G_CALLBACK (skeleton_handle_show_item_properties_cb), fdb);
177
178 g_dbus_interface_skeleton_export (G_DBUS_INTERFACE_SKELETON (fdb->skeleton), conn, CAJA_FDO_DBUS_PATH, NULL);
179
180 g_dbus_object_manager_server_set_connection (fdb->object_manager, conn);
181}

199static void
200caja_freedesktop_dbus_dispose (GObject *object)
201{
202 CajaFreedesktopDBus *fdb = (CajaFreedesktopDBus *) object;
203
204 if (fdb->owner_id != 0) {
205 g_bus_unown_name (fdb->owner_id);
206 fdb->owner_id = 0;
207 }
208
209 if (fdb->skeleton != NULL) {
210 g_dbus_interface_skeleton_unexport (G_DBUS_INTERFACE_SKELETON (fdb->skeleton));
211 g_object_unref (fdb->skeleton);
212 fdb->skeleton = NULL;
213 }
214
215 g_clear_object (&fdb->object_manager);
216
217 G_OBJECT_CLASS (caja_freedesktop_dbus_parent_class)->dispose (object);
218}
