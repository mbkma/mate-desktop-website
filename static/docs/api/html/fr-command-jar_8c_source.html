47static void
48fr_command_jar_add (FrCommand *comm,
49 const char *from_file,
50 GList *file_list,
51 const char *base_dir,
52 gboolean update,
53 gboolean recursive)
54{
55 FrProcess *proc = comm->process;
56 GList *zip_list = NULL, *jardata_list = NULL, *jar_list = NULL;
57 GList *scan;
58 char *tmp_dir;
59
60 for (scan = file_list; scan; scan = scan->next) {
61 char *filename = scan->data;
62 char *path = build_uri (base_dir, filename, NULL);
63 char *package = NULL;
64
65 if (file_extension_is (filename, ".java"))
66 package = get_package_name_from_java_file (path);
67 else if (file_extension_is (filename, ".class"))
68 package = get_package_name_from_class_file (path);
69
70 if ((package == NULL) || (strlen (package) == 0))
71 zip_list = g_list_append (zip_list, g_strdup (filename));
72 else {
73 JarData *newdata = g_new0 (JarData, 1);
74
75 newdata->package_minus_one_level = remove_level_from_path (package);
76 newdata->link_name = g_strdup (file_name_from_path (package));
77 newdata->rel_path = remove_level_from_path (filename);
78 newdata->filename = g_strdup (file_name_from_path (filename));
79 jardata_list = g_list_append (jardata_list, newdata);
80 }
81
82 g_free (package);
83 g_free (path);
84 }
85
86 tmp_dir = get_temp_work_dir (NULL);
87 for (scan = jardata_list; scan ; scan = scan->next) {
88 JarData *jdata = scan->data;
89 char *pack_path;
90 char *old_link;
91 char *link_name;
92 int retval;
93
94 pack_path = build_uri (tmp_dir, jdata->package_minus_one_level, NULL);
95 if (! make_directory_tree_from_path (pack_path, 0755, NULL)) {
96 g_free (pack_path);
97 continue;
98 }
99
100 old_link = build_uri (base_dir, jdata->rel_path, NULL);
101 link_name = g_build_filename (pack_path, jdata->link_name, NULL);
102
103 retval = symlink (old_link, link_name);
104 if ((retval != -1) || (errno == EEXIST))
105 jar_list = g_list_append (jar_list,
106 g_build_filename (jdata->package_minus_one_level,
107 jdata->link_name,
108 jdata->filename,
109 NULL));
110
111 g_free (link_name);
112 g_free (old_link);
113 g_free (pack_path);
114 }
115
116 if (zip_list != NULL)
117 parent_class->add (comm, NULL, zip_list, base_dir, update, FALSE);
118
119 if (jar_list != NULL)
120 parent_class->add (comm, NULL, jar_list, tmp_dir, update, FALSE);
121
122 fr_process_begin_command (proc, "rm");
123 fr_process_set_working_dir (proc, "/");
124 fr_process_add_arg (proc, "-r");
125 fr_process_add_arg (proc, "-f");
126 fr_process_add_arg (proc, tmp_dir);
127 fr_process_end_command (proc);
128 fr_process_set_sticky (proc, TRUE);
129
130 for (scan = jardata_list; scan ; scan = scan->next) {
131 JarData *jdata = scan->data;
132 g_free (jdata->filename);
133 g_free (jdata->package_minus_one_level);
134 g_free (jdata->link_name);
135 g_free (jdata->rel_path);
136 }
137
138 path_list_free (jardata_list);
139 path_list_free (jar_list);
140 path_list_free (zip_list);
141 g_free (tmp_dir);
142}
