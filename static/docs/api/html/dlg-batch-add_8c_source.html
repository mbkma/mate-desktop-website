200 GTK_DIALOG_DESTROY_WITH_PARENT,
201 NULL,
202 _("Could not create the archive"),

209 g_free (archive_name);

243
244 d = _gtk_error_dialog_new (GTK_WINDOW (window),
245 GTK_DIALOG_DESTROY_WITH_PARENT,

247 _("Could not create the archive"),

261

263 GTK_DIALOG_DESTROY_WITH_PARENT,

265 _("Archive not created"),

323 if (r == GTK_RESPONSE_YES) {

349 g_free (archive_name);

354static void
355update_sensitivity_for_mime_type (DialogData *data,
356 const char *mime_type)
357{
358 int i;
359
360 if (mime_type == NULL) {
361 gtk_widget_set_sensitive (GET_WIDGET ("a_password_entry"), FALSE);
362 gtk_widget_set_sensitive (GET_WIDGET ("a_password_label"), FALSE);
363 gtk_widget_set_sensitive (GET_WIDGET ("a_encrypt_header_checkbutton"), FALSE);
364 gtk_toggle_button_set_inconsistent (GET_TOGGLE_BUTTON ("a_encrypt_header_checkbutton"), TRUE);
365 gtk_widget_set_sensitive (GET_WIDGET ("a_volume_box"), FALSE);
366 return;
367 }
368
369 for (i = 0; mime_type_desc[i].mime_type != NULL; i++) {
370 if (strcmp (mime_type_desc[i].mime_type, mime_type) == 0) {
371 gboolean sensitive;
372
373 sensitive = mime_type_desc[i].capabilities & FR_COMMAND_CAN_ENCRYPT;
374 gtk_widget_set_sensitive (GET_WIDGET ("a_password_entry"), sensitive);
375 gtk_widget_set_sensitive (GET_WIDGET ("a_password_label"), sensitive);
376
377 sensitive = mime_type_desc[i].capabilities & FR_COMMAND_CAN_ENCRYPT_HEADER;
378 can_encrypt_header = sensitive;
379 gtk_widget_set_sensitive (GET_WIDGET ("a_encrypt_header_checkbutton"), sensitive ? has_password : FALSE);
380 gtk_toggle_button_set_inconsistent (GET_TOGGLE_BUTTON ("a_encrypt_header_checkbutton"), sensitive ? (!has_password) : TRUE);
381
382 sensitive = mime_type_desc[i].capabilities & FR_COMMAND_CAN_CREATE_VOLUMES;
383 gtk_widget_set_sensitive (GET_WIDGET ("a_volume_box"), sensitive);
384
385 break;
386 }
387 }
