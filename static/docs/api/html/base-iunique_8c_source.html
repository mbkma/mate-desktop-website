137static IUniqueData *
138get_iunique_data( BaseIUnique *instance )
139{
140 IUniqueData *data;
141
142 data = ( IUniqueData * ) g_object_get_data( G_OBJECT( instance ), BASE_PROP_IUNIQUE_DATA );
143
144 if( !data ){
145 data = g_new0( IUniqueData, 1 );
146 g_object_set_data( G_OBJECT( instance ), BASE_PROP_IUNIQUE_DATA, data );
147 g_object_weak_ref( G_OBJECT( instance ), ( GWeakNotify ) on_instance_finalized, NULL );

171gboolean
172base_iunique_init_with_name( BaseIUnique *instance, const gchar *unique_app_name )
173{
174 static const gchar *thisfn = "base_iunique_init_with_name";
175 IUniqueData *data;
176 gboolean ret;
177 gboolean is_first;
178 gchar *msg;
179 GError *error = NULL;
180
181 g_return_val_if_fail( BASE_IS_IUNIQUE( instance ), FALSE );
182
183 g_debug( "%s: instance=%p, unique_app_name=%s", thisfn, ( void * ) instance, unique_app_name );
184
185 ret = TRUE;
186 data = get_iunique_data( instance );
187
188 if( unique_app_name && strlen( unique_app_name )){
189
190 data->handle = g_application_new( unique_app_name, G_APPLICATION_FLAGS_NONE );
191
192 if (!g_application_register (data->handle, NULL, &error))
193 {
194 g_warning ("%s", error->message);
195 g_error_free (error);
196 return 1;
197 }
198
199 is_first = !g_application_get_is_remote( data->handle );
200
201 if( !is_first ){
202 /* Translators: application name */
203 msg = g_strdup_printf(
204 _( "Another instance of %s is already running.\n"
205 "Please switch back to it." ),
206 m_get_application_name( instance ));
207 base_window_display_error_dlg( NULL, _( "The application is not unique" ), msg );
208 g_free( msg );
209 ret = FALSE;
210 } else {
211 data->unique_app_name = g_strdup( unique_app_name );
212 }
213 }
214
215 return( ret );
216}
