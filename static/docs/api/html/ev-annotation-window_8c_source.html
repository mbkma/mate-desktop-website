79static void
80send_focus_change (GtkWidget *widget,
81 gboolean in)
82{
83 GdkEvent *fevent = gdk_event_new (GDK_FOCUS_CHANGE);
84
85 fevent->focus_change.type = GDK_FOCUS_CHANGE;
86 fevent->focus_change.window = gtk_widget_get_window (widget);
87 fevent->focus_change.in = in;
88 if (fevent->focus_change.window)
89 g_object_ref (fevent->focus_change.window);
90
91 gtk_widget_send_focus_change (widget, fevent);
92
93 gdk_event_free (fevent);
94}

110static void
111ev_annotation_window_sync_contents (EvAnnotationWindow *window)
112{
113 gchar *contents;
114 GtkTextIter start, end;
115 GtkTextBuffer *buffer;
116 EvAnnotation *annot = window->annotation;
117
118 buffer = gtk_text_view_get_buffer (GTK_TEXT_VIEW (window->text_view));
119 gtk_text_buffer_get_bounds (buffer, &start, &end);
120 contents = gtk_text_buffer_get_text (buffer, &start, &end, FALSE);
121 ev_annotation_set_contents (annot, contents);
122 g_free (contents);
123}

242static void
243ev_annotation_window_set_resize_cursor (GtkWidget *widget,
244 EvAnnotationWindow *window)
245{
246 GdkWindow *gdk_window = gtk_widget_get_window (widget);
247
248 if (!gdk_window)
249 return;
250
251 if (gtk_widget_is_sensitive (widget)) {
252 GdkDisplay *display = gtk_widget_get_display (widget);
253 GdkCursor *cursor;
254
255 cursor = gdk_cursor_new_for_display (display,
256 widget == window->resize_sw ?
257 GDK_BOTTOM_LEFT_CORNER :
258 GDK_BOTTOM_RIGHT_CORNER);
259 gdk_window_set_cursor (gdk_window, cursor);
260 g_object_unref (cursor);
261 } else {
262 gdk_window_set_cursor (gdk_window, NULL);
263 }
264}

283static gboolean
284ev_annotation_window_button_press_event (GtkWidget *widget,
285 GdkEventButton *event)
286{
287 EvAnnotationWindow *window = EV_ANNOTATION_WINDOW (widget);
288
289 if (event->type == GDK_BUTTON_PRESS && event->button == 1) {
290 window->in_move = TRUE;
291 window->x = event->x_root - event->x;
292 window->y = event->y_root - event->y;
293 gtk_window_begin_move_drag (GTK_WINDOW (widget),
294 event->button,
295 event->x_root,
296 event->y_root,
297 event->time);

429static GObject *
430ev_annotation_window_constructor (GType type,
431 guint n_construct_properties,
432 GObjectConstructParam *construct_params)
433{
434 GObject *object;
435 EvAnnotationWindow *window;
436 EvAnnotation *annot;
437 EvAnnotationMarkup *markup;
438 const gchar *contents;
439 const gchar *label;
440 GdkRGBA color;
441 EvRectangle *rect;
442 gdouble scale;
443 gdouble opacity;
444
445 object = G_OBJECT_CLASS (ev_annotation_window_parent_class)->constructor (type,
446 n_construct_properties,
447 construct_params);
448 window = EV_ANNOTATION_WINDOW (object);
449 annot = window->annotation;
450 markup = EV_ANNOTATION_MARKUP (annot);
451
452 gtk_window_set_transient_for (GTK_WINDOW (window), window->parent);
453 gtk_window_set_destroy_with_parent (GTK_WINDOW (window), FALSE);
454
455 label = ev_annotation_markup_get_label (markup);
456 window->is_open = ev_annotation_markup_get_popup_is_open (markup);
457 ev_annotation_markup_get_rectangle (markup, &window->rect);
458
459 rect = &window->rect;
460
461 /* Rectangle is at doc resolution (72.0) */
462 scale = get_monitor_dpi (window) / 72.0;
463 gtk_window_resize (GTK_WINDOW (window),
464 (gint)((rect->x2 - rect->x1) * scale),
465 (gint)((rect->y2 - rect->y1) * scale));
466
467 ev_annotation_get_rgba (annot, &color);
468 ev_annotation_window_set_color (window, &color);
469
470 opacity = ev_annotation_markup_get_opacity (markup);
471 ev_annotation_window_set_opacity (window, opacity);
472
473 gtk_widget_set_name (GTK_WIDGET (window), ev_annotation_get_name (annot));
474 gtk_window_set_title (GTK_WINDOW (window), label);
475 gtk_label_set_text (GTK_LABEL (window->title), label);
476
477 contents = ev_annotation_get_contents (annot);
478 if (contents) {
479 GtkTextBuffer *buffer;
480
481 buffer = gtk_text_view_get_buffer (GTK_TEXT_VIEW (window->text_view));
482 gtk_text_buffer_set_text (buffer, contents, -1);
483 }
484
485 g_signal_connect (annot, "notify::label",
486 G_CALLBACK (ev_annotation_window_label_changed),
487 window);
488 g_signal_connect (annot, "notify::rgba",
489 G_CALLBACK (ev_annotation_window_color_changed),
490 window);
491 g_signal_connect (annot, "notify::opacity",
492 G_CALLBACK (ev_annotation_window_opacity_changed),
493 window);
494
495 return object;
496}

513static gboolean
514ev_annotation_window_focus_in_event (GtkWidget *widget,
515 GdkEventFocus *event)
516{
517 EvAnnotationWindow *window = EV_ANNOTATION_WINDOW (widget);
518
519 if (window->in_move) {
520 if (window->orig_x != window->x || window->orig_y != window->y) {
521 window->orig_x = window->x;
522 window->orig_y = window->y;
523 g_signal_emit (window, signals[MOVED], 0, window->x, window->y);
524 }
525 window->in_move = FALSE;
526 }
527
528 gtk_widget_grab_focus (window->text_view);
529 send_focus_change (window->text_view, TRUE);
530 gtk_text_view_set_cursor_visible (GTK_TEXT_VIEW (window->text_view), TRUE);
531
532 return FALSE;
533}
