176static void
177calculate_info (PlumaDocument *doc,
178 GtkTextIter *start,
179 GtkTextIter *end,
180 gint *chars,
181 gint *words,
182 gint *white_chars,
183 gint *bytes)
184{
185 gchar *text;
186
187 pluma_debug (DEBUG_PLUGINS);
188
189 text = gtk_text_buffer_get_slice (GTK_TEXT_BUFFER (doc),
190 start,
191 end,
192 TRUE);
193
194 *chars = g_utf8_strlen (text, -1);
195 *bytes = strlen (text);
196
197 if (*chars > 0)
198 {
199 PangoLogAttr *attrs;
200 gint i;
201
202 attrs = g_new0 (PangoLogAttr, *chars + 1);
203
204 pango_get_log_attrs (text,
205 -1,
206 0,
207 pango_language_from_string ("C"),
208 attrs,
209 *chars + 1);
210
211 for (i = 0; i < (*chars); i++)
212 {
213 if (attrs[i].is_white)
214 ++(*white_chars);
215
216 if (attrs[i].is_word_start)
217 ++(*words);
218 }
219
220 g_free (attrs);
221 }
222 else
223 {
224 *white_chars = 0;
225 *words = 0;
226 }
227
228 g_free (text);
229}

231static void
232docinfo_real (PlumaDocument *doc,
233 DocInfoDialog *dialog)
234{
235 GtkTextIter start, end;
236 gint words = 0;
237 gint chars = 0;
238 gint white_chars = 0;
239 gint lines = 0;
240 gint bytes = 0;
241 gchar *tmp_str;
242 gchar *doc_name;
243
244 pluma_debug (DEBUG_PLUGINS);
245
246 gtk_text_buffer_get_bounds (GTK_TEXT_BUFFER (doc),
247 &start,
248 &end);
249
250 lines = gtk_text_buffer_get_line_count (GTK_TEXT_BUFFER (doc));
251
252 calculate_info (doc,
253 &start, &end,
254 &chars, &words, &white_chars, &bytes);
255
256 if (chars == 0)
257 lines = 0;
258
259 pluma_debug_message (DEBUG_PLUGINS, "Chars: %d", chars);
260 pluma_debug_message (DEBUG_PLUGINS, "Lines: %d", lines);
261 pluma_debug_message (DEBUG_PLUGINS, "Words: %d", words);
262 pluma_debug_message (DEBUG_PLUGINS, "Chars non-space: %d", chars - white_chars);
263 pluma_debug_message (DEBUG_PLUGINS, "Bytes: %d", bytes);
264
265 doc_name = pluma_document_get_short_name_for_display (doc);
266 tmp_str = g_strdup_printf ("<span weight=\"bold\">%s</span>", doc_name);
267 gtk_label_set_markup (GTK_LABEL (dialog->file_name_label), tmp_str);
268 g_free (doc_name);
269 g_free (tmp_str);
270
271 tmp_str = g_strdup_printf("%d", lines);
272 gtk_label_set_text (GTK_LABEL (dialog->lines_label), tmp_str);
273 g_free (tmp_str);
274
275 tmp_str = g_strdup_printf("%d", words);
276 gtk_label_set_text (GTK_LABEL (dialog->words_label), tmp_str);
277 g_free (tmp_str);
278
279 tmp_str = g_strdup_printf("%d", chars);
280 gtk_label_set_text (GTK_LABEL (dialog->chars_label), tmp_str);
281 g_free (tmp_str);
282
283 tmp_str = g_strdup_printf("%d", chars - white_chars);
284 gtk_label_set_text (GTK_LABEL (dialog->chars_ns_label), tmp_str);
285 g_free (tmp_str);
286
287 tmp_str = g_strdup_printf("%d", bytes);
288 gtk_label_set_text (GTK_LABEL (dialog->bytes_label), tmp_str);
289 g_free (tmp_str);
290}

292static void
293selectioninfo_real (PlumaDocument *doc,
294 DocInfoDialog *dialog)
295{
296 gboolean sel;
297 GtkTextIter start, end;
298 gint words = 0;
299 gint chars = 0;
300 gint white_chars = 0;
301 gint lines = 0;
302 gint bytes = 0;
303 gchar *tmp_str;
304
305 pluma_debug (DEBUG_PLUGINS);
306
307 sel = gtk_text_buffer_get_selection_bounds (GTK_TEXT_BUFFER (doc),
308 &start,
309 &end);
310
311 if (sel)
312 {
313 lines = gtk_text_iter_get_line (&end) - gtk_text_iter_get_line (&start) + 1;
314
315 calculate_info (doc,
316 &start, &end,
317 &chars, &words, &white_chars, &bytes);
318
319 pluma_debug_message (DEBUG_PLUGINS, "Selected chars: %d", chars);
320 pluma_debug_message (DEBUG_PLUGINS, "Selected lines: %d", lines);
321 pluma_debug_message (DEBUG_PLUGINS, "Selected words: %d", words);
322 pluma_debug_message (DEBUG_PLUGINS, "Selected chars non-space: %d", chars - white_chars);
323 pluma_debug_message (DEBUG_PLUGINS, "Selected bytes: %d", bytes);
324
325 gtk_widget_set_sensitive (dialog->selection_vbox, TRUE);
326 }
327 else
328 {
329 gtk_widget_set_sensitive (dialog->selection_vbox, FALSE);
330
331 pluma_debug_message (DEBUG_PLUGINS, "Selection empty");
332 }
333
334 if (chars == 0)
335 lines = 0;
336
337 tmp_str = g_strdup_printf("%d", lines);
338 gtk_label_set_text (GTK_LABEL (dialog->selected_lines_label), tmp_str);
339 g_free (tmp_str);
340
341 tmp_str = g_strdup_printf("%d", words);
342 gtk_label_set_text (GTK_LABEL (dialog->selected_words_label), tmp_str);
343 g_free (tmp_str);
344
345 tmp_str = g_strdup_printf("%d", chars);
346 gtk_label_set_text (GTK_LABEL (dialog->selected_chars_label), tmp_str);
347 g_free (tmp_str);
348
349 tmp_str = g_strdup_printf("%d", chars - white_chars);
350 gtk_label_set_text (GTK_LABEL (dialog->selected_chars_ns_label), tmp_str);
351 g_free (tmp_str);
352
353 tmp_str = g_strdup_printf("%d", bytes);
354 gtk_label_set_text (GTK_LABEL (dialog->selected_bytes_label), tmp_str);
355 g_free (tmp_str);
356}

358static void
359docinfo_cb (GtkAction *action,
360 PlumaDocInfoPlugin *plugin)
361{
362 PlumaDocInfoPluginPrivate *data;
363 PlumaWindow *window;
364 PlumaDocument *doc;
365
366 pluma_debug (DEBUG_PLUGINS);
367
368 data = plugin->priv;
369 window = PLUMA_WINDOW (data->window);
370 doc = pluma_window_get_active_document (window);
371 g_return_if_fail (doc != NULL);
372
373 if (data->dialog != NULL)
374 {
375 gtk_window_present (GTK_WINDOW (data->dialog->dialog));
376 gtk_widget_grab_focus (GTK_WIDGET (data->dialog->dialog));
377 }
378 else
379 {
380 DocInfoDialog *dialog;
381
382 dialog = get_docinfo_dialog (plugin);
383 g_return_if_fail (dialog != NULL);
384
385 data->dialog = dialog;
386
387 gtk_widget_show (GTK_WIDGET (dialog->dialog));
388 }
389
390 docinfo_real (doc,
391 data->dialog);
392 selectioninfo_real (doc,
393 data->dialog);
394}

567

569
570 gtk_ui_manager_add_ui (manager,
571 data->ui_id,
572 MENU_PATH,
573 "DocumentStatistics",
574 "DocumentStatistics",
575 GTK_UI_MANAGER_MENUITEM,
576 FALSE);
577
578 update_ui (data);
