31static int get_bit(int in, int bit)
32{
33 return (in & (1 << bit)) >> bit;
34}

36static int get_bits(int in, int begin, int end)
37{
38 int mask = (1 << (end - begin + 1)) - 1;
39
40 return (in >> begin) & mask;
41}

53static int decode_vendor_and_product_identification(const uchar* edid, MonitorInfo* info)
54{
55 int is_model_year;
56
57 /* Manufacturer Code */
58 info->manufacturer_code[0] = get_bits(edid[0x08], 2, 6);
59 info->manufacturer_code[1] = get_bits(edid[0x08], 0, 1) << 3;
60 info->manufacturer_code[1] |= get_bits(edid[0x09], 5, 7);
61 info->manufacturer_code[2] = get_bits(edid[0x09], 0, 4);
62 info->manufacturer_code[3] = '\0';
63
64 info->manufacturer_code[0] += 'A' - 1;
65 info->manufacturer_code[1] += 'A' - 1;
66 info->manufacturer_code[2] += 'A' - 1;
67
68 /* Product Code */
69 info->product_code = edid[0x0b] << 8 | edid[0x0a];
70
71 /* Serial Number */
72 info->serial_number = edid[0x0c] | edid[0x0d] << 8 | edid[0x0e] << 16 | edid[0x0f] << 24;
73
74 /* Week and Year */
75 is_model_year = FALSE;
76
77 switch (edid[0x10])
78 {
79 case 0x00:
80 info->production_week = -1;
81 break;
82
83 case 0xff:
84 info->production_week = -1;
85 is_model_year = TRUE;
86 break;
87
88 default:
89 info->production_week = edid[0x10];
90 break;
91 }
92
93 if (is_model_year)
94 {
95 info->production_year = -1;
96 info->model_year = 1990 + edid[0x11];
97 }
98 else
99 {
100 info->production_year = 1990 + edid[0x11];
101 info->model_year = -1;
102 }
103
104 return TRUE;
105}

112 return TRUE;

115static int decode_display_parameters(const uchar* edid, MonitorInfo* info)
116{
117 /* Digital vs Analog */
118 info->is_digital = get_bit(edid[0x14], 7);
119
120 if (info->is_digital)
121 {
122 int bits;
123
124 static const int bit_depth[8] = {
125 -1, 6, 8, 10, 12, 14, 16, -1
126 };
127
128 static const Interface interfaces[6] = {
129 UNDEFINED, DVI, HDMI_A, HDMI_B, MDDI, DISPLAY_PORT
130 };
131
132 bits = get_bits(edid[0x14], 4, 6);
133 info->connector.digital.bits_per_primary = bit_depth[bits];
134
135 bits = get_bits(edid[0x14], 0, 3);
136
137 if (bits <= 5)
138 {
139 info->connector.digital.interface = interfaces[bits];
140 }
141 else
142 {
143 info->connector.digital.interface = UNDEFINED;
144 }
145 }
146 else
147 {
148 int bits = get_bits (edid[0x14], 5, 6);
149
150 static const double levels[][3] = {
151 { 0.7, 0.3, 1.0 },
152 { 0.714, 0.286, 1.0 },
153 { 1.0, 0.4, 1.4 },
154 { 0.7, 0.0, 0.7 },
155 };
156
157 info->connector.analog.video_signal_level = levels[bits][0];
158 info->connector.analog.sync_signal_level = levels[bits][1];
159 info->connector.analog.total_signal_level = levels[bits][2];
160
161 info->connector.analog.blank_to_black = get_bit (edid[0x14], 4);
162
163 info->connector.analog.separate_hv_sync = get_bit (edid[0x14], 3);
164 info->connector.analog.composite_sync_on_h = get_bit (edid[0x14], 2);
165 info->connector.analog.composite_sync_on_green = get_bit (edid[0x14], 1);
166
167 info->connector.analog.serration_on_vsync = get_bit (edid[0x14], 0);
168 }
169
170 /* Screen Size / Aspect Ratio */
171 if (edid[0x15] == 0 && edid[0x16] == 0)
172 {
173 info->width_mm = -1;
174 info->height_mm = -1;
175 info->aspect_ratio = -1.0;
176 }
177 else if (edid[0x16] == 0)
178 {
179 info->width_mm = -1;
180 info->height_mm = -1;
181 info->aspect_ratio = 100.0 / (edid[0x15] + 99);
182 }
183 else if (edid[0x15] == 0)
184 {
185 info->width_mm = -1;
186 info->height_mm = -1;
187 info->aspect_ratio = 100.0 / (edid[0x16] + 99);
188 info->aspect_ratio = 1/info->aspect_ratio; /* portrait */
189 }
190 else
191 {
192 info->width_mm = 10 * edid[0x15];
193 info->height_mm = 10 * edid[0x16];
194 }
195
196 /* Gamma */
197 if (edid[0x17] == 0xFF)
198 info->gamma = -1.0;
199 else
200 info->gamma = (edid[0x17] + 100.0) / 100.0;
201
202 /* Features */
203 info->standby = get_bit(edid[0x18], 7);
204 info->suspend = get_bit(edid[0x18], 6);
205 info->active_off = get_bit(edid[0x18], 5);
206
207 if (info->is_digital)
208 {
209 info->connector.digital.rgb444 = TRUE;
210
211 if (get_bit(edid[0x18], 3))
212 info->connector.digital.ycrcb444 = 1;
213 if (get_bit(edid[0x18], 4))
214 info->connector.digital.ycrcb422 = 1;
215 }
216 else
217 {
218 int bits = get_bits(edid[0x18], 3, 4);
219
220 ColorType color_type[4] = {
221 MONOCHROME, RGB, OTHER_COLOR, UNDEFINED_COLOR
222 };
223
224 info->connector.analog.color_type = color_type[bits];
225 }
226
227 info->srgb_is_standard = get_bit(edid[0x18], 2);
228
229 /* In 1.3 this is called "has preferred timing" */
230 info->preferred_timing_includes_native = get_bit(edid[0x18], 1);
231
232 /* FIXME: In 1.3 this indicates whether the monitor accepts GTF */
233 info->continuous_frequency = get_bit(edid[0x18], 0);
234
235 return TRUE;
236}

238static double decode_fraction(int high, int low)
239{
240 double result = 0.0;
241 int i;
242
243 high = (high << 2) | low;
244
245 for (i = 0; i < 10; ++i)
246 {
247 result += get_bit (high, i) * pow (2, i - 10);
248 }
249
250 return result;
251}

253static int decode_color_characteristics(const uchar* edid, MonitorInfo* info)
254{
255 info->red_x = decode_fraction(edid[0x1b], get_bits(edid[0x19], 6, 7));
256 info->red_y = decode_fraction(edid[0x1c], get_bits(edid[0x19], 5, 4));
257 info->green_x = decode_fraction(edid[0x1d], get_bits(edid[0x19], 2, 3));
258 info->green_y = decode_fraction(edid[0x1e], get_bits(edid[0x19], 0, 1));
259 info->blue_x = decode_fraction(edid[0x1f], get_bits(edid[0x1a], 6, 7));
260 info->blue_y = decode_fraction(edid[0x20], get_bits(edid[0x1a], 4, 5));
261 info->white_x = decode_fraction(edid[0x21], get_bits(edid[0x1a], 2, 3));
262 info->white_y = decode_fraction(edid[0x22], get_bits(edid[0x1a], 0, 1));
263
264 return TRUE;
265}

267static int decode_established_timings(const uchar* edid, MonitorInfo* info)
268{
269 static const Timing established[][8] = {
270 {
271 {800, 600, 60},
272 {800, 600, 56},
273 {640, 480, 75},
274 {640, 480, 72},
275 {640, 480, 67},
276 {640, 480, 60},
277 {720, 400, 88},
278 {720, 400, 70}
279 },
280 {
281 {1280, 1024, 75},
282 {1024, 768, 75},
283 {1024, 768, 70},
284 {1024, 768, 60},
285 {1024, 768, 87},
286 {832, 624, 75},
287 {800, 600, 75},
288 {800, 600, 72}
289 },
290 {
291 {0, 0, 0},
292 {0, 0, 0},
293 {0, 0, 0},
294 {0, 0, 0},
295 {0, 0, 0},
296 {0, 0, 0},
297 {0, 0, 0},
298 {1152, 870, 75}
299 },
300 };
301
302 int i, j, idx;
303
304 idx = 0;
305
306 for (i = 0; i < 3; ++i)
307 {
308 for (j = 0; j < 8; ++j)
309 {
310 int byte = edid[0x23 + i];
311
312 if (get_bit (byte, j) && established[i][j].frequency != 0)
313 {
314 info->established[idx++] = established[i][j];
315 }
316 }
317 }
318
319 return TRUE;
320}

322static int decode_standard_timings(const uchar* edid, MonitorInfo* info)
323{
324 int i;
325
326 for (i = 0; i < 8; i++)
327 {
328 int first = edid[0x26 + 2 * i];
329 int second = edid[0x27 + 2 * i];
330
331 if (first != 0x01 && second != 0x01)
332 {
333 int w = 8 * (first + 31);
334 int h = 0;
335
336 switch (get_bits(second, 6, 7))
337 {
338 case 0x00:
339 h = (w / 16) * 10;
340 break;
341 case 0x01:
342 h = (w / 4) * 3;
343 break;
344 case 0x02:
345 h = (w / 5) * 4;
346 break;
347 case 0x03:
348 h = (w / 16) * 9;
349 break;
350 }
351
352 info->standard[i].width = w;
353 info->standard[i].height = h;
354 info->standard[i].frequency = get_bits(second, 0, 5) + 60;
355 }
356 }
357
358 return TRUE;
359}

361static void decode_lf_string(const uchar* s, int n_chars, char* result)
362{
363 int i;
364
365 for (i = 0; i < n_chars; ++i)
366 {
367 if (s[i] == 0x0a)
368 {
369 *result++ = '\0';
370 break;
371 }
372 else if (s[i] == 0x00)
373 {
374 /* Convert embedded 0's to spaces */
375 *result++ = ' ';
376 }
377 else
378 {
379 *result++ = s[i];
380 }
381 }
382}

384static void decode_display_descriptor(const uchar* desc, MonitorInfo* info)
385{
386 switch (desc[0x03])
387 {
388 case 0xFC:
389 decode_lf_string (desc + 5, 13, info->dsc_product_name);
390 break;
391 case 0xFF:
392 decode_lf_string (desc + 5, 13, info->dsc_serial_number);
393 break;
394 case 0xFE:
395 decode_lf_string (desc + 5, 13, info->dsc_string);
396 break;
397 case 0xFD:
398 /* Range Limits */
399 break;
400 case 0xFB:
401 /* Color Point */
402 break;
403 case 0xFA:
404 /* Timing Identifications */
405 break;
406 case 0xF9:
407 /* Color Management */
408 break;
409 case 0xF8:
410 /* Timing Codes */
411 break;
412 case 0xF7:
413 /* Established Timings */
414 break;
415 case 0x10:
416 break;
417 }

420static void decode_detailed_timing(const uchar* timing, DetailedTiming* detailed)
421{
422 int bits;
423
424 StereoType stereo[] = {
425 NO_STEREO, NO_STEREO, FIELD_RIGHT, FIELD_LEFT,
426 TWO_WAY_RIGHT_ON_EVEN, TWO_WAY_LEFT_ON_EVEN,
427 FOUR_WAY_INTERLEAVED, SIDE_BY_SIDE
428 };
429
430 detailed->pixel_clock = (timing[0x00] | timing[0x01] << 8) * 10000;
431 detailed->h_addr = timing[0x02] | ((timing[0x04] & 0xf0) << 4);
432 detailed->h_blank = timing[0x03] | ((timing[0x04] & 0x0f) << 8);
433 detailed->v_addr = timing[0x05] | ((timing[0x07] & 0xf0) << 4);
434 detailed->v_blank = timing[0x06] | ((timing[0x07] & 0x0f) << 8);
435 detailed->h_front_porch = timing[0x08] | get_bits(timing[0x0b], 6, 7) << 8;
436 detailed->h_sync = timing[0x09] | get_bits(timing[0x0b], 4, 5) << 8;
437 detailed->v_front_porch =
438 get_bits (timing[0x0a], 4, 7) | get_bits(timing[0x0b], 2, 3) << 4;
439 detailed->v_sync =
440 get_bits (timing[0x0a], 0, 3) | get_bits(timing[0x0b], 0, 1) << 4;
441 detailed->width_mm = timing[0x0c] | get_bits(timing[0x0e], 4, 7) << 8;
442 detailed->height_mm = timing[0x0d] | get_bits(timing[0x0e], 0, 3) << 8;
443 detailed->right_border = timing[0x0f];
444 detailed->top_border = timing[0x10];
445
446 detailed->interlaced = get_bit(timing[0x11], 7);
447
448 /* Stereo */
449 bits = get_bits (timing[0x11], 5, 6) << 1 | get_bit(timing[0x11], 0);
450 detailed->stereo = stereo[bits];
451
452 /* Sync */
453 bits = timing[0x11];
454
455 detailed->digital_sync = get_bit(bits, 4);
456
457 if (detailed->digital_sync)
458 {
459 detailed->connector.digital.composite = !get_bit(bits, 3);
460
461 if (detailed->connector.digital.composite)
462 {
463 detailed->connector.digital.serrations = get_bit(bits, 2);
464 detailed->connector.digital.negative_vsync = FALSE;
465 }
466 else
467 {
468 detailed->connector.digital.serrations = FALSE;
469 detailed->connector.digital.negative_vsync = !get_bit(bits, 2);
470 }
471
472 detailed->connector.digital.negative_hsync = !get_bit(bits, 0);
473 }
474 else
475 {
476 detailed->connector.analog.bipolar = get_bit(bits, 3);
477 detailed->connector.analog.serrations = get_bit(bits, 2);
478 detailed->connector.analog.sync_on_green = !get_bit(bits, 1);
479 }
480}

482static int decode_descriptors(const uchar* edid, MonitorInfo* info)
483{
484 int i;
485 int timing_idx;
486
487 timing_idx = 0;
488
489 for (i = 0; i < 4; ++i)
490 {
491 int index = 0x36 + i * 18;
492
493 if (edid[index + 0] == 0x00 && edid[index + 1] == 0x00)
494 {
495 decode_display_descriptor(edid + index, info);
496 }
497 else
498 {
499 decode_detailed_timing(edid + index, &(info->detailed_timings[timing_idx++]));
500 }
501 }
502
503 info->n_detailed_timings = timing_idx;
504
505 return TRUE;
506}

521MonitorInfo* decode_edid(const uchar* edid)
522{
523 MonitorInfo* info = g_new0(MonitorInfo, 1);
524
525 decode_check_sum(edid, info);
526
527 if (decode_header(edid) && decode_vendor_and_product_identification(edid, info) && decode_edid_version(edid, info) && decode_display_parameters(edid, info) && decode_color_characteristics(edid, info) && decode_established_timings(edid, info) && decode_standard_timings(edid, info) && decode_descriptors(edid, info))
528 {
529 return info;
530 }
531 else
532 {
533 g_free(info);
534 return NULL;
535 }
