529static void
530caja_property_browser_drag_begin (GtkWidget *widget,
531 GdkDragContext *context)
532{
533 CajaPropertyBrowser *property_browser;
534 GtkWidget *child;
535 GdkPixbuf *pixbuf;
536 char *element_name;
537
538 property_browser = CAJA_PROPERTY_BROWSER (widget);
539
540 child = g_object_steal_data (G_OBJECT (property_browser), "dragged-image");
541 g_return_if_fail (child != NULL);
542
543 element_name = g_object_get_data (G_OBJECT (child), "property-name");
544 g_return_if_fail (child != NULL);
545
546 /* compute the offsets for dragging */
547 if (strcmp (drag_types[0].target, "application/x-color") != 0)
548 {
549 /* it's not a color, so, for now, it must be an image */
550 /* fiddle with the category to handle the "reset" case properly */
551 char * save_category = property_browser->details->category;
552 if (g_strcmp0 (property_browser->details->category, "colors") == 0)
553 {
554 property_browser->details->category = "patterns";
555 }
556 pixbuf = make_drag_image (property_browser, element_name);
557 property_browser->details->category = save_category;
558 }
559 else
560 {
561 pixbuf = make_color_drag_image (property_browser, element_name, TRUE);
562 }
563
564 /* set the pixmap and mask for dragging */
565 if (pixbuf != NULL)
566 {
567 int x_delta, y_delta;
568
569 x_delta = gdk_pixbuf_get_width (pixbuf) / 2;
570 y_delta = gdk_pixbuf_get_height (pixbuf) / 2;
571
572 gtk_drag_set_icon_pixbuf
573 (context,
574 pixbuf,
575 x_delta, y_delta);
576 g_object_unref (pixbuf);
577 }
578
579}

581/* drag and drop data get handler */
582
583static void
584caja_property_browser_drag_data_get (GtkWidget *widget,
585 GdkDragContext *context,
586 GtkSelectionData *selection_data,
587 guint info,
588 guint32 time)
589{
590 char *image_file_name, *image_file_uri;
591 gboolean is_reset;
592 CajaPropertyBrowser *property_browser = CAJA_PROPERTY_BROWSER(widget);
593 GdkAtom target;
594
595 g_return_if_fail (widget != NULL);
596 g_return_if_fail (context != NULL);
597
598 target = gtk_selection_data_get_target (selection_data);
599
600 switch (info)
601 {
602 case PROPERTY_TYPE:
603 /* formulate the drag data based on the drag type. Eventually, we will
604 probably select the behavior from properties in the category xml definition,
605 but for now we hardwire it to the drag_type */
606
607 is_reset = FALSE;
608 if (strcmp (property_browser->details->drag_type,
609 "property/keyword") == 0)
610 {
611 char *keyword_str = eel_filename_strip_extension (property_browser->details->dragged_file);
612 gtk_selection_data_set (selection_data, target, 8, keyword_str, strlen (keyword_str));
613 g_free (keyword_str);
614 return;
615 }
616 else if (strcmp (property_browser->details->drag_type,
617 "application/x-color") == 0)
618 {
619 /* handle the "reset" case as an image */
620 if (g_strcmp0 (property_browser->details->dragged_file, RESET_IMAGE_NAME) != 0)
621 {
622 GdkRGBA color;
623 guint16 colorArray [4];
624
625 gdk_rgba_parse (&color, property_browser->details->dragged_file);
626
627 colorArray [0] = (guint16) (color.red * 65535.0);
628 colorArray [1] = (guint16) (color.green * 65535.0);
629 colorArray [2] = (guint16) (color.blue * 65535.0);
630 colorArray [3] = USHRT_MAX;

1359static void
1360add_color_to_file (CajaPropertyBrowser *property_browser,
1361 const xmlChar *color_spec,
1362 const xmlChar *color_name)
1363{
1364 xmlNodePtr cur_node, new_color_node, children_node;
1365 xmlDocPtr document;

1367
1368 document = read_browser_xml (property_browser);
