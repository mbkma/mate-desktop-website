54static gdouble
55unix_time_to_julian_date (gint unix_time)
56{
57 return UNIX_EPOCH + (double) unix_time / (60 * 60 * 24);

60/* Finds an iterative solution for [ E - e sin (E) = M ] for values of e less
61 than 0.1. Page 90 */
62
63#define ERROR_ACCURACY 1e-6 /* radians */
64
65static gdouble
66solve_keplers_equation (gdouble e,
67 gdouble M)
68{
69 gdouble d, E;
70
71 /* start with an initial estimate */
72 E = M;
73
74 d = E - e * sin (E) - M;
75
76 while (ABS (d) > ERROR_ACCURACY)
77 {
78 E = E - (d / (1 - e * cos (E)));
79 d = E - e * sin (E) - M;
80 }
81
82 return E;
83}

85 /* convert the ecliptic longitude to right ascension and declination. Section 27. */
86static void
87ecliptic_to_equatorial (gdouble lambda,
88 gdouble beta,
89 gdouble *ra,
90 gdouble *dec)
91{
92 gdouble cos_mo;
93 gdouble sin_mo;
94
95 g_assert (ra != NULL);
96 g_assert (dec != NULL);
97
98 sin_mo = sin (DEG_TO_RADS (MEAN_OBLIQUITY));
99 cos_mo = cos (DEG_TO_RADS (MEAN_OBLIQUITY));
100
101 *ra = atan2 (sin (lambda) * cos_mo - tan (beta) * sin_mo, cos (lambda));
102 *dec = asin (sin (beta) * cos_mo + cos (beta) * sin_mo * sin (lambda));
103}

105/* calculate GST. Section 12 */
106static gdouble
107greenwich_sidereal_time (gdouble unix_time)
108{
109 gdouble u, JD, T, T0, UT;
110
111 u = fmod (unix_time, 24 * 60 * 60);
112 JD = unix_time_to_julian_date (unix_time - u);
113 T = (JD - 2451545) / 36525;
114 T0 = 6.697374558 + (2400.051336 * T) + (0.000025862 * T * T);
115 T0 = fmod (T0, 24);
116 UT = u / (60 * 60);
117 T0 = T0 + UT * 1.002737909;
118 T0 = fmod (T0, 24);
119
120 return T0;
121}

123/* Calculate the position of the sun at a given time. pages 89-91 */
124void
125sun_position (time_t unix_time, gdouble *lat, gdouble *lon)
126{
127 gdouble jd, D, N, M, E, x, v, lambda;
128 gdouble ra, dec;
129 jd = unix_time_to_julian_date (unix_time);
130
131 /* Calculate number of days since the epoch */
132 D = jd - EPOCH;
133
134 N = D*360/365.242191;
135
136 /* normalize to 0 - 360 degrees */
137 NORMALIZE (N);
138
139 /* Step 4: */
140 M = N + EPSILON_G - MU_G;
141 NORMALIZE (M);
142
143 /* Step 5: convert to radians */
144 M = DEG_TO_RADS (M);
145
146 /* Step 6: */
147 E = solve_keplers_equation (ECCENTRICITY, M);
148
149 /* Step 7: */
150 x = sqrt ((1 + ECCENTRICITY)/(1 - ECCENTRICITY)) * tan (E/2);
151
152 /* Step 8, 9 */
153 v = 2 * RADS_TO_DEG (atan (x));
154 NORMALIZE (v);
155
156 /* Step 10 */
157 lambda = v + MU_G;
158 NORMALIZE (lambda);
159
160 /* convert the ecliptic longitude to right ascension and declination */
161 ecliptic_to_equatorial (DEG_TO_RADS (lambda), 0.0, &ra, &dec);
162
163 ra = ra - (G_PI/12) * greenwich_sidereal_time (unix_time);
164 ra = RADS_TO_DEG (ra);
165 dec = RADS_TO_DEG (dec);
166 NORMALIZE (ra);
167 NORMALIZE (dec);
168
169 *lat = dec;
170 *lon = ra;
171}
