72static void
73add_variable_cb(GtkWidget *widget, MathVariablePopup *popup)
74{
75 const gchar *name;
76 MPNumber z = mp_new();
77
78 name = gtk_entry_get_text(GTK_ENTRY(popup->priv->variable_name_entry));
79 if (name[0] == '\0')
80 return;
81
82 if (math_equation_get_number(popup->priv->equation, &z))
83 math_variables_set(math_equation_get_variables(popup->priv->equation), name, &z);
84 else if (math_equation_is_result(popup->priv->equation))
85 math_variables_set(math_equation_get_variables(popup->priv->equation), name, math_equation_get_answer(popup->priv->equation));
86 else
87 g_warning("Can't add variable %s, the display is not a number", name);
88
89 gtk_widget_destroy(gtk_widget_get_toplevel(widget));
90 mp_clear(&z);
91}

122static GtkWidget *
123make_variable_entry(MathVariablePopup *popup, const gchar *name, const MPNumber *value, gboolean writable)
124{
125 GtkWidget *hbox, *button, *label;
126 gchar *text;
127
128 hbox = gtk_box_new(GTK_ORIENTATION_HORIZONTAL, 6);
129
130 if (value)
131 {
132 gchar *value_text = mp_serializer_to_string(math_equation_get_serializer(popup->priv->equation), value);
133 text = g_strdup_printf("<b>%s</b> = %s", name, value_text);
134 g_free (value_text);
135 }
136 else
137 text = g_strdup_printf("<b>%s</b>", name);
138
139 button = gtk_button_new();
140 g_object_set_data(G_OBJECT(button), "variable_name", g_strdup(name)); // FIXME: These will all leak memory
141 g_signal_connect(G_OBJECT(button), "clicked", G_CALLBACK(insert_variable_cb), popup);
142 gtk_button_set_relief(GTK_BUTTON(button), GTK_RELIEF_NONE);
143 gtk_box_pack_start(GTK_BOX(hbox), button, TRUE, TRUE, 0);
144 gtk_widget_show(button);
145
146 label = gtk_label_new(text);
147 g_free(text);
148 gtk_label_set_use_markup(GTK_LABEL(label), TRUE);
149 gtk_widget_set_halign (label, GTK_ALIGN_START);
150 gtk_container_add(GTK_CONTAINER(button), label);
151 gtk_widget_show(label);
152
153 if (writable)
154 {
155 GtkWidget *image;
156
157 button = gtk_button_new();
158 g_object_set_data(G_OBJECT(button), "variable_name", g_strdup(name));
159 image = gtk_image_new_from_icon_name("document-save", GTK_ICON_SIZE_BUTTON);
160 gtk_container_add(GTK_CONTAINER(button), image);
161 gtk_box_pack_start(GTK_BOX(hbox), button, FALSE, TRUE, 0);
162 g_signal_connect(G_OBJECT(button), "clicked", G_CALLBACK(save_variable_cb), popup);
163 gtk_widget_show(image);
164 gtk_widget_show(button);
165
166 button = gtk_button_new();
167 g_object_set_data(G_OBJECT(button), "variable_name", g_strdup(name));
168 image = gtk_image_new_from_icon_name("edit-delete", GTK_ICON_SIZE_BUTTON);
169 gtk_container_add(GTK_CONTAINER(button), image);
170 gtk_box_pack_start(GTK_BOX(hbox), button, FALSE, TRUE, 0);
171 g_signal_connect(G_OBJECT(button), "clicked", G_CALLBACK(delete_variable_cb), popup);
172 gtk_widget_show(image);
173 gtk_widget_show(button);
174 }
175
176 return hbox;
177}

179static void
180math_variable_popup_set_property(GObject *object,
181 guint prop_id,
182 const GValue *value,
183 GParamSpec *pspec)
184{
185 MathVariablePopup *self;
186 gchar **names;
187 int i;
188 GtkWidget *entry, *image;
189
190 self = MATH_VARIABLE_POPUP(object);
191
192 switch (prop_id) {
193 case PROP_EQUATION:
194 self->priv->equation = g_value_get_object(value);
195
196 names = math_variables_get_names(math_equation_get_variables(self->priv->equation));
197 for (i = 0; names[i]; i++) {
198 MPNumber *aux;
199
200 aux = math_variables_get(math_equation_get_variables(self->priv->equation), names[i]);
201 entry = make_variable_entry(self, names[i], aux, TRUE);
202 gtk_widget_show(entry);
203 gtk_box_pack_start(GTK_BOX(self->priv->vbox), entry, FALSE, TRUE, 0);
204 }
205 g_strfreev(names);
206
207 entry = gtk_box_new(GTK_ORIENTATION_HORIZONTAL, 6);
208 gtk_widget_show(entry);
209
210 // TODO: Show greyed "variable name" text to give user a hint how to use
211 self->priv->variable_name_entry = gtk_entry_new();
212 g_signal_connect(G_OBJECT(self->priv->variable_name_entry), "key-press-event", G_CALLBACK(variable_name_key_press_cb), self);
213 g_signal_connect(G_OBJECT(self->priv->variable_name_entry), "changed", G_CALLBACK(variable_name_changed_cb), self);
214 g_signal_connect(G_OBJECT(self->priv->variable_name_entry), "activate", G_CALLBACK(add_variable_cb), self);
215 gtk_box_pack_start(GTK_BOX(entry), self->priv->variable_name_entry, TRUE, TRUE, 0);
216 gtk_widget_show(self->priv->variable_name_entry);
217
218 self->priv->add_variable_button = gtk_button_new();
219 gtk_widget_set_sensitive(self->priv->add_variable_button, FALSE);
220 g_signal_connect(G_OBJECT(self->priv->add_variable_button), "clicked", G_CALLBACK(add_variable_cb), self);
221 image = gtk_image_new_from_icon_name("list-add", GTK_ICON_SIZE_BUTTON);
222 gtk_container_add(GTK_CONTAINER(self->priv->add_variable_button), image);
223 gtk_box_pack_start(GTK_BOX(entry), self->priv->add_variable_button, FALSE, TRUE, 0);
224 gtk_widget_show(image);
225 gtk_widget_show(self->priv->add_variable_button);
226 gtk_box_pack_end(GTK_BOX(self->priv->vbox), entry, FALSE, TRUE, 0);
227
228 entry = make_variable_entry(self, "rand", NULL, FALSE);
229 gtk_widget_show(entry);
230 gtk_box_pack_end(GTK_BOX(self->priv->vbox), entry, FALSE, TRUE, 0);
231
232 entry = make_variable_entry(self, "ans", math_equation_get_answer(self->priv->equation), FALSE);
233 gtk_widget_show(entry);
234 gtk_box_pack_end(GTK_BOX(self->priv->vbox), entry, FALSE, TRUE, 0);
235 break;
236 default:
237 G_OBJECT_WARN_INVALID_PROPERTY_ID(object, prop_id, pspec);
238 break;
239 }
240}
