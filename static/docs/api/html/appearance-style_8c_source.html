59static gchar *
60find_string_in_model (GtkTreeModel *model, const gchar *value, gint column)
61{
62 GtkTreeIter iter;
63 gboolean valid;
64 gchar *path = NULL, *test;
65
66 if (!value)
67 return NULL;
68
69 for (valid = gtk_tree_model_get_iter_first (model, &iter); valid;
70 valid = gtk_tree_model_iter_next (model, &iter))
71 {
72 gtk_tree_model_get (model, &iter, column, &test, -1);
73
74 if (test)
75 {
76 gint cmp = strcmp (test, value);
77 g_free (test);
78
79 if (!cmp)
80 {
81 path = gtk_tree_model_get_string_from_iter (model, &iter);
82 break;
83 }
84 }
85 }
86
87 return path;
88}

90static void
91treeview_gsettings_changed_callback (GSettings *settings, gchar *key, GtkTreeView *list)
92{
93 GtkTreeModel *store;
94 gchar *curr_value;
95 gchar *path;
96
97 /* find value in model */
98 curr_value = g_settings_get_string (settings, key);
99 store = gtk_tree_view_get_model (list);
100
101 path = find_string_in_model (store, curr_value, COL_NAME);
102
103 /* Add a temporary item if we can't find a match
104 * TODO: delete this item if it is no longer selected?
105 */
106 if (!path)
107 {
108 GtkListStore *list_store;
109 GtkTreeIter iter, sort_iter;
110 ThemeConvData *conv;
111
112 list_store = GTK_LIST_STORE (gtk_tree_model_sort_get_model (GTK_TREE_MODEL_SORT (store)));
113
114 conv = g_object_get_data (G_OBJECT(list), THEME_DATA);
115 gtk_list_store_insert_with_values (list_store, &iter, 0,
116 COL_LABEL, curr_value,
117 COL_NAME, curr_value,
118 COL_THUMBNAIL, conv->thumbnail,
119 -1);
120 /* convert the tree store iter for use with the sort model */
121 gtk_tree_model_sort_convert_child_iter_to_iter (GTK_TREE_MODEL_SORT (store),
122 &sort_iter, &iter);
123 path = gtk_tree_model_get_string_from_iter (store, &sort_iter);
124
125 create_thumbnail (curr_value, conv->thumbnail, conv->data);
126 }
127 /* select the new gsettings theme in treeview */
128 GtkTreeSelection *selection = gtk_tree_view_get_selection (list);
129 GtkTreePath *treepath = gtk_tree_path_new_from_string (path);
130 gtk_tree_selection_select_path (selection, treepath);
131 gtk_tree_view_scroll_to_cell (list, treepath, NULL, FALSE, 0, 0);
132 gtk_tree_path_free (treepath);
133}

135static void
136treeview_selection_changed_callback (GtkTreeSelection *selection, guint data)
137{
138 GSettings *settings;
139 gchar *key;
140 GtkTreeIter iter;
141 GtkTreeModel *model;
142 GtkTreeView *list;
143
144 list = gtk_tree_selection_get_tree_view (selection);
145
146 settings = g_object_get_data (G_OBJECT (list), GSETTINGS_SETTINGS);
147 key = g_object_get_data (G_OBJECT (list), GSETTINGS_KEY);
148
149 if (gtk_tree_selection_get_selected (selection, &model, &iter)) {
150 gchar *list_value;
151
152 gtk_tree_model_get (model, &iter, COL_NAME, &list_value, -1);
153
154 if (list_value) {
155 g_settings_set_string (settings, key, list_value);
156
157 /*Load the gnome interface schema if we are running under wayland and it is present*/
158 if (!(GDK_IS_X11_DISPLAY (gdk_display_get_default())))
159 {
160 GSettingsSchemaSource *source = g_settings_schema_source_get_default ();
161
162 if (source)
163 {
164 GSettingsSchema *schema = g_settings_schema_source_lookup (source, INTERFACE_GNOME_SCHEMA, TRUE);
165 {
166 if (schema)
167 {
168 GSettings *interface_gnome_settings = NULL;
169 interface_gnome_settings = g_settings_new_full (schema, NULL, NULL);
170
171 if ((strcmp (key, GTK_THEME_KEY) == 0) ||
172 (strcmp (key, ICON_THEME_KEY) == 0) ||
173 (strcmp (key, CURSOR_THEME_KEY) == 0))
174 g_settings_set_string (interface_gnome_settings, key, list_value);
175
176 }
177 }
178 }
179 }
180 }

184static gint
185cursor_theme_sort_func (GtkTreeModel *model,
186 GtkTreeIter *a,
187 GtkTreeIter *b,
188 gpointer user_data)
189{
190 gchar *a_label = NULL;
191 gchar *b_label = NULL;
192 const gchar *default_label;
193 gint result;
194
195 gtk_tree_model_get (model, a, COL_LABEL, &a_label, -1);
196 gtk_tree_model_get (model, b, COL_LABEL, &b_label, -1);
197
198 default_label = _("Default Pointer");
199
200 if (!strcmp (a_label, default_label))
201 result = -1;
202 else if (!strcmp (b_label, default_label))
203 result = 1;
204 else
205 result = strcmp (a_label, b_label);
206
207 g_free (a_label);
208 g_free (b_label);
209
210 return result;
211}

234static void update_message_area(AppearanceData* data)
235{
236 GtkSettings* settings = gtk_settings_get_default();
237 gchar* theme = NULL;
238 gchar* engine;
239
240 g_object_get(settings, "gtk-theme-name", &theme, NULL);
241 engine = gtk_theme_info_missing_engine(theme, TRUE);
242 g_free(theme);
243
244 if (data->style_message_area == NULL)
245 {
246 GtkWidget* hbox;
247 GtkWidget* parent;
248 GtkWidget* icon;
249 GtkWidget* content;
250
251 if (engine == NULL)
252 {
253 return;
254 }
255
256 data->style_message_area = gtk_info_bar_new ();
257
258 g_signal_connect (data->style_message_area, "response", (GCallback) style_message_area_response_cb, data);
259
260 data->style_install_button = gtk_info_bar_add_button(GTK_INFO_BAR (data->style_message_area), _("Install"), GTK_RESPONSE_APPLY);
261
262 data->style_message_label = gtk_label_new (NULL);
263 gtk_label_set_line_wrap (GTK_LABEL (data->style_message_label), TRUE);
264 gtk_label_set_xalign (GTK_LABEL (data->style_message_label), 0.0);
265
266 hbox = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 9);
267 icon = gtk_image_new_from_icon_name ("dialog-warning", GTK_ICON_SIZE_DIALOG);
268 gtk_widget_set_halign (icon, GTK_ALIGN_CENTER);
269 gtk_widget_set_valign (icon, GTK_ALIGN_START);
270 gtk_box_pack_start (GTK_BOX (hbox), icon, FALSE, FALSE, 0);
271 gtk_box_pack_start (GTK_BOX (hbox), data->style_message_label, TRUE, TRUE, 0);
272 content = gtk_info_bar_get_content_area (GTK_INFO_BAR (data->style_message_area));
273 gtk_container_add (GTK_CONTAINER (content), hbox);
274 gtk_widget_show_all (data->style_message_area);
275 gtk_widget_set_no_show_all (data->style_message_area, TRUE);
276
277 parent = appearance_capplet_get_widget (data, "gtk_themes_vbox");
278 gtk_box_pack_start (GTK_BOX (parent), data->style_message_area, FALSE, FALSE, 0);
279 }
280
281 g_free (engine);
282 gtk_widget_hide(data->style_message_area);
283}

327static void
328check_color_schemes_enabled (GtkSettings *settings,
329 AppearanceData *data)
330{
331 gchar *theme = NULL;
332 gchar *filename;
333 GSList *symbolic_colors = NULL;
334 gboolean enable_colors = FALSE;
335 gint i;
336
337 g_object_get (settings, "gtk-theme-name", &theme, NULL);
338 filename = gtkrc_find_named (theme);
339 g_free (theme);
340
341 gtkrc_get_details (filename, NULL, &symbolic_colors);
342 g_free (filename);
343
344 for (i = 0; i < NUM_SYMBOLIC_COLORS; ++i) {
345 gboolean found;
346
347 found = (g_slist_find_custom (symbolic_colors, symbolic_names[i], (GCompareFunc) strcmp) != NULL);
348 gtk_widget_set_sensitive (appearance_capplet_get_widget (data, symbolic_names[i]), found);
349
350 enable_colors |= found;
351 }
352
353 g_slist_free_full (symbolic_colors, g_free);
354
355 gtk_widget_set_sensitive (appearance_capplet_get_widget (data, "color_scheme_table"), enable_colors);
356 gtk_widget_set_sensitive (appearance_capplet_get_widget (data, "color_scheme_defaults_button"), enable_colors);
357
358 if (enable_colors)
359 gtk_widget_hide (appearance_capplet_get_widget (data, "color_scheme_message_hbox"));
360 else
361 gtk_widget_show (appearance_capplet_get_widget (data, "color_scheme_message_hbox"));
362}

364static void
365color_button_clicked_cb (GtkWidget *colorbutton, AppearanceData *data)
366{
367 GtkWidget *widget;
368 GdkRGBA color;
369 GString *scheme = g_string_new (NULL);
370 gchar *colstr;
371 gchar *old_scheme = NULL;
372 gint i;
373
374 for (i = 0; i < NUM_SYMBOLIC_COLORS; ++i) {
375 widget = appearance_capplet_get_widget (data, symbolic_names[i]);
376 gtk_color_chooser_get_rgba (GTK_COLOR_CHOOSER (widget), &color);
377
378 colstr = gdk_rgba_to_string (&color);
379 g_string_append_printf (scheme, "%s:%s\n", symbolic_names[i], colstr);
380 g_free (colstr);
381 }
382 /* remove the last newline */
383 g_string_truncate (scheme, scheme->len - 1);
384
385 /* verify that the scheme really has changed */
386 g_object_get (gtk_settings_get_default (), "gtk-color-scheme", &old_scheme, NULL);
387
388 if (!mate_theme_color_scheme_equal (old_scheme, scheme->str)) {
389 g_settings_set_string (data->interface_settings, COLOR_SCHEME_KEY, scheme->str);
390
391 gtk_widget_set_sensitive (appearance_capplet_get_widget (data, "color_scheme_defaults_button"), TRUE);
392 }
393 g_free (old_scheme);
394 g_string_free (scheme, TRUE);
395}

414static void
415gtk_theme_changed (GSettings *settings, gchar *key, AppearanceData *data)
416{
417 MateThemeInfo *theme = NULL;
418 gchar *name;
419 GtkSettings *gtksettings = gtk_settings_get_default ();
420
421 name = g_settings_get_string (settings, key);
422 if (name) {
423 gchar *current;
424
425 theme = mate_theme_info_find (name);
426
427 /* Manually update GtkSettings to new gtk+ theme.
428 * This will eventually happen anyway, but we need the
429 * info for the color scheme updates already. */
430 g_object_get (gtksettings, "gtk-theme-name", &current, NULL);
431
432 if (strcmp (current, name) != 0) {
433 g_object_set (gtksettings, "gtk-theme-name", name, NULL);
434 update_message_area (data);
435 }
436
437 g_free (current);
438
439 check_color_schemes_enabled (gtksettings, data);
440 update_color_buttons_from_settings (gtksettings, data);
441 g_free (name);
442 }
443
444 gtk_widget_set_sensitive (appearance_capplet_get_widget (data, "gtk_themes_delete"),
445 theme_is_writable (theme));
446}

531static void
532update_cursor_size_scale (MateThemeCursorInfo *theme,
533 AppearanceData *data)
534{
535 GtkWidget *cursor_size_scale;
536 GtkWidget *cursor_size_label;
537 GtkWidget *cursor_size_small_label;
538 GtkWidget *cursor_size_large_label;
539 gboolean sensitive;
540 gint size, gsettings_size;
541
542 cursor_size_scale = appearance_capplet_get_widget (data, "cursor_size_scale");
543 cursor_size_label = appearance_capplet_get_widget (data, "cursor_size_label");
544 cursor_size_small_label = appearance_capplet_get_widget (data, "cursor_size_small_label");
545 cursor_size_large_label = appearance_capplet_get_widget (data, "cursor_size_large_label");
546
547 sensitive = theme && theme->sizes->len > 1;
548 gtk_widget_set_sensitive (cursor_size_scale, sensitive);
549 gtk_widget_set_sensitive (cursor_size_label, sensitive);
550 gtk_widget_set_sensitive (cursor_size_small_label, sensitive);
551 gtk_widget_set_sensitive (cursor_size_large_label, sensitive);
552
553 gsettings_size = g_settings_get_int (data->mouse_settings, CURSOR_SIZE_KEY);
554
555 if (sensitive) {
556 GtkAdjustment *adjustment;
557 gint i, index;
558 GtkRange *range = GTK_RANGE (cursor_size_scale);
559
560 adjustment = gtk_range_get_adjustment (range);
561 g_object_set (adjustment, "upper", (gdouble) theme->sizes->len - 1, NULL);
562
563 /* fallback if the gsettings value is bigger than all available sizes;
564 use the largest we have */
565 index = theme->sizes->len - 1;
566
567 /* set the slider to the cursor size which matches the gsettings setting best */
568 for (i = 0; i < theme->sizes->len; i++) {
569 size = g_array_index (theme->sizes, gint, i);
570
571 if (size == gsettings_size) {
572 index = i;
573 break;
574 } else if (size > gsettings_size) {
575 if (i == 0) {
576 index = 0;
577 } else {
578 gint diff, diff_to_last;
579
580 diff = size - gsettings_size;
581 diff_to_last = gsettings_size - g_array_index (theme->sizes, gint, i - 1);
582
583 index = (diff < diff_to_last) ? i : i - 1;
584 }
585 break;
586 }
587 }
588
589 gtk_range_set_value (range, (gdouble) index);
590
591 size = g_array_index (theme->sizes, gint, index);
592 } else {
593 if (theme && theme->sizes->len > 0)
594 size = g_array_index (theme->sizes, gint, 0);
595 else
596 size = 24;
597 }
598
599 if (size != gsettings_size)
600 cursor_size_changed_cb (size, data);
601}

622static void
623generic_theme_delete (const gchar *tv_name, ThemeType type, AppearanceData *data)
624{
625 GtkTreeView *treeview = GTK_TREE_VIEW (appearance_capplet_get_widget (data, tv_name));
626 GtkTreeSelection *selection = gtk_tree_view_get_selection (treeview);
627 GtkTreeModel *model;
628 GtkTreeIter iter;
629
630 if (gtk_tree_selection_get_selected (selection, &model, &iter)) {
631 gchar *name;
632
633 gtk_tree_model_get (model, &iter, COL_NAME, &name, -1);
634
635 if (name != NULL && theme_delete (name, type)) {
636 /* remove theme from the model, too */
637 GtkTreeIter child;
638 GtkTreePath *path;
639
640 path = gtk_tree_model_get_path (model, &iter);
641 gtk_tree_model_sort_convert_iter_to_child_iter (
642 GTK_TREE_MODEL_SORT (model), &child, &iter);
643 gtk_list_store_remove (GTK_LIST_STORE (
644 gtk_tree_model_sort_get_model (GTK_TREE_MODEL_SORT (model))), &child);
645
646 if (gtk_tree_model_get_iter (model, &iter, path) ||
647 theme_model_iter_last (model, &iter)) {
648 gtk_tree_path_free (path);
649 path = gtk_tree_model_get_path (model, &iter);
650 gtk_tree_selection_select_path (selection, path);
651 gtk_tree_view_scroll_to_cell (treeview, path, NULL, FALSE, 0, 0);
652 }
653 gtk_tree_path_free (path);
654 }
655 g_free (name);
656 }
657}

822static void
823changed_on_disk_cb (MateThemeCommonInfo *theme,
824 MateThemeChangeType change_type,
825 MateThemeElement element_type,
826 AppearanceData *data)
827{
828 if (theme->type == MATE_THEME_TYPE_REGULAR) {
829 MateThemeInfo *info = (MateThemeInfo *) theme;
830
831 if (change_type == MATE_THEME_CHANGE_DELETED) {
832 if (element_type & MATE_THEME_GTK_2)
833 remove_from_treeview ("gtk_themes_list", info->name, data);
834 if (element_type & MATE_THEME_MARCO)
835 remove_from_treeview ("window_themes_list", info->name, data);
836
837 } else {
838 if (element_type & MATE_THEME_GTK_2) {
839 if (change_type == MATE_THEME_CHANGE_CREATED)
840 add_to_treeview ("gtk_themes_list", info->name, info->name, data->gtk_theme_icon, data);
841 else if (change_type == MATE_THEME_CHANGE_CHANGED)
842 update_in_treeview ("gtk_themes_list", info->name, info->name, data);
843
844 generate_gtk_theme_thumbnail_async (info,
845 (ThemeThumbnailFunc) gtk_theme_thumbnail_cb, data, NULL);
846 }
847
848 if (element_type & MATE_THEME_MARCO) {
849 if (change_type == MATE_THEME_CHANGE_CREATED)
850 add_to_treeview ("window_themes_list", info->name, info->name, data->window_theme_icon, data);
851 else if (change_type == MATE_THEME_CHANGE_CHANGED)
852 update_in_treeview ("window_themes_list", info->name, info->name, data);
853
854 generate_marco_theme_thumbnail_async (info,
855 (ThemeThumbnailFunc) marco_theme_thumbnail_cb, data, NULL);
856 }
857 }
858
859 } else if (theme->type == MATE_THEME_TYPE_ICON) {
860 MateThemeIconInfo *info = (MateThemeIconInfo *) theme;
861
862 if (change_type == MATE_THEME_CHANGE_DELETED) {
863 remove_from_treeview ("icon_themes_list", info->name, data);
864 } else {
865 if (change_type == MATE_THEME_CHANGE_CREATED)
866 add_to_treeview ("icon_themes_list", info->name, info->readable_name, data->icon_theme_icon, data);
867 else if (change_type == MATE_THEME_CHANGE_CHANGED)
868 update_in_treeview ("icon_themes_list", info->name, info->readable_name, data);
869
870 generate_icon_theme_thumbnail_async (info,
871 (ThemeThumbnailFunc) icon_theme_thumbnail_cb, data, NULL);
872 }
873
874 } else if (theme->type == MATE_THEME_TYPE_CURSOR) {
875 MateThemeCursorInfo *info = (MateThemeCursorInfo *) theme;
876
877 if (change_type == MATE_THEME_CHANGE_DELETED) {
878 remove_from_treeview ("cursor_themes_list", info->name, data);
879 } else {
880 if (change_type == MATE_THEME_CHANGE_CREATED)
881 add_to_treeview ("cursor_themes_list", info->name, info->readable_name, info->thumbnail, data);
882 else if (change_type == MATE_THEME_CHANGE_CHANGED)
883 update_in_treeview ("cursor_themes_list", info->name, info->readable_name, data);
884 }
885 }

888static void
889prepare_list (AppearanceData *data, GtkWidget *list, ThemeType type, GCallback callback)
890{
891 GtkListStore *store;
892 GList *l, *themes = NULL;
893 GtkCellRenderer *renderer;
894 GtkTreeViewColumn *column;
895 GtkTreeModel *sort_model;
896 GdkPixbuf *thumbnail;
897 const gchar *key;
898 ThumbnailGenFunc generator;
899 ThemeThumbnailFunc thumb_cb;
900 ThemeConvData *conv_data;
901 GSettings *settings;
902
903 switch (type)
904 {
905 case THEME_TYPE_GTK:
906 themes = mate_theme_info_find_by_type (MATE_THEME_GTK_2);
907 thumbnail = data->gtk_theme_icon;
908 settings = data->interface_settings;
909 key = GTK_THEME_KEY;
910 generator = (ThumbnailGenFunc) generate_gtk_theme_thumbnail_async;
911 thumb_cb = (ThemeThumbnailFunc) gtk_theme_thumbnail_cb;
912 break;
913
914 case THEME_TYPE_WINDOW:
915 themes = mate_theme_info_find_by_type (MATE_THEME_MARCO);
916 thumbnail = data->window_theme_icon;
917 settings = data->marco_settings;
918 key = MARCO_THEME_KEY;
919 generator = (ThumbnailGenFunc) generate_marco_theme_thumbnail_async;
920 thumb_cb = (ThemeThumbnailFunc) marco_theme_thumbnail_cb;
921 break;
922
923 case THEME_TYPE_ICON:
924 themes = mate_theme_icon_info_find_all ();
925 thumbnail = data->icon_theme_icon;
926 settings = data->interface_settings;
927 key = ICON_THEME_KEY;
928 generator = (ThumbnailGenFunc) generate_icon_theme_thumbnail_async;
929 thumb_cb = (ThemeThumbnailFunc) icon_theme_thumbnail_cb;
930 break;
931
932 case THEME_TYPE_CURSOR:
933 themes = mate_theme_cursor_info_find_all ();
934 thumbnail = NULL;
935 settings = data->mouse_settings;
936 key = CURSOR_THEME_KEY;
937 generator = NULL;
938 thumb_cb = NULL;
939 break;
940
941 default:
942 /* we don't deal with any other type of themes here */
943 return;
944 }
945
946 store = gtk_list_store_new (NUM_COLS, GDK_TYPE_PIXBUF, G_TYPE_STRING, G_TYPE_STRING);
947
948 for (l = themes; l; l = g_list_next (l))
949 {
950 MateThemeCommonInfo *theme = (MateThemeCommonInfo *) l->data;
951 GtkTreeIter i;
952
953 if (type == THEME_TYPE_CURSOR) {
954 thumbnail = ((MateThemeCursorInfo *) theme)->thumbnail;
955 } else {
956 generator (theme, thumb_cb, data, NULL);
957 }
958
959 gtk_list_store_insert_with_values (store, &i, 0,
960 COL_LABEL, theme->readable_name,
961 COL_NAME, theme->name,
962 COL_THUMBNAIL, thumbnail,
963 -1);
964
965 if (type == THEME_TYPE_CURSOR && thumbnail) {
966 g_object_unref (thumbnail);
967 thumbnail = NULL;
968 }
969 }
970 g_list_free (themes);
971
972 sort_model = gtk_tree_model_sort_new_with_model (GTK_TREE_MODEL (store));
973 gtk_tree_sortable_set_sort_column_id (GTK_TREE_SORTABLE (sort_model),
974 COL_LABEL, GTK_SORT_ASCENDING);
975
976 if (type == THEME_TYPE_CURSOR)
977 gtk_tree_sortable_set_sort_func (GTK_TREE_SORTABLE (sort_model), COL_LABEL,
978 (GtkTreeIterCompareFunc) cursor_theme_sort_func,
979 NULL, NULL);
980
981 gtk_tree_view_set_model (GTK_TREE_VIEW (list), GTK_TREE_MODEL (sort_model));
982
983 renderer = gtk_cell_renderer_pixbuf_new ();
984 g_object_set (renderer, "xpad", 3, "ypad", 3, NULL);
985
986 column = gtk_tree_view_column_new ();
987 gtk_tree_view_column_pack_start (column, renderer, FALSE);
988 gtk_tree_view_column_add_attribute (column, renderer, "pixbuf", COL_THUMBNAIL);
989 gtk_tree_view_append_column (GTK_TREE_VIEW (list), column);
990
991 renderer = gtk_cell_renderer_text_new ();
992
993 column = gtk_tree_view_column_new ();
994 gtk_tree_view_column_pack_start (column, renderer, FALSE);
995 gtk_tree_view_column_add_attribute (column, renderer, "text", COL_LABEL);
996 gtk_tree_view_append_column (GTK_TREE_VIEW (list), column);
997
998 conv_data = g_new (ThemeConvData, 1);
999 conv_data->data = data;
1000 conv_data->thumbnail = thumbnail;
1001
1002 /* set useful data for callbacks */
1003 g_object_set_data_full (G_OBJECT (list), THEME_DATA, conv_data, g_free);
1004 g_object_set_data (G_OBJECT (list), GSETTINGS_SETTINGS, settings);
1005 g_object_set_data_full (G_OBJECT (list), GSETTINGS_KEY, g_strdup(key), g_free);
1006
1007 /* select in treeview the theme set in gsettings */
1008 GtkTreeModel *treemodel;
1009 treemodel = gtk_tree_view_get_model (GTK_TREE_VIEW (list));
1010 gchar *theme = g_settings_get_string (settings, key);
1011 gchar *path = find_string_in_model (treemodel, theme, COL_NAME);
1012 if (path)
1013 {
1014 GtkTreeSelection *selection = gtk_tree_view_get_selection (GTK_TREE_VIEW (list));
1015 GtkTreePath *treepath = gtk_tree_path_new_from_string (path);
1016 gtk_tree_selection_select_path (selection, treepath);
1017 gtk_tree_view_scroll_to_cell (GTK_TREE_VIEW (list), treepath, NULL, FALSE, 0, 0);
1018 gtk_tree_path_free (treepath);
1019 g_free (path);
1020 }
1021 if (theme)
1022 g_free (theme);
1023
1024 /* connect to gsettings change event */
1025 gchar *signal_name = g_strdup_printf("changed::%s", key);
1026 g_signal_connect (settings, signal_name,
1027 G_CALLBACK (treeview_gsettings_changed_callback), list);
1028 g_signal_connect (settings, signal_name, callback, data);
1029 g_free (signal_name);
1030
1031 /* connect to treeview change event */
1032 g_signal_connect (gtk_tree_view_get_selection (GTK_TREE_VIEW (list)),
1033 "changed", G_CALLBACK (treeview_selection_changed_callback), list);
1034}
