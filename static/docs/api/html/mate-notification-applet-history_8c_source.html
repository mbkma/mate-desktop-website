32static GtkWidget *
33create_notification_icon (const gchar *icon_name)
34{
35 GdkPixbuf *pixbuf = NULL;
36 GtkWidget *image;
37
38 if (icon_name && *icon_name) {
39 if (g_path_is_absolute (icon_name)) {
40 pixbuf = gdk_pixbuf_new_from_file_at_scale (icon_name, IMAGE_SIZE, IMAGE_SIZE, TRUE, NULL);
41 } else {
42 GtkIconTheme *theme = gtk_icon_theme_get_default ();
43 pixbuf = gtk_icon_theme_load_icon (theme, icon_name, IMAGE_SIZE, GTK_ICON_LOOKUP_USE_BUILTIN, NULL);
44 }
45 }
46
47 if (pixbuf) {
48 image = gtk_image_new_from_pixbuf (pixbuf);
49 g_object_unref (pixbuf);
50 } else {
51 image = gtk_image_new_from_icon_name ("mate-notification-properties", GTK_ICON_SIZE_DIALOG);
52 }
53
54 gtk_widget_set_valign (image, GTK_ALIGN_CENTER);
55 return image;
56}

81static GtkWidget *
82create_notification_row (guint id, const gchar *app_name, const gchar *app_icon,
83 const gchar *summary, const gchar *body, gint64 timestamp)
84{
85 GtkWidget *row, *hbox, *icon_image, *content_box;
86 GtkWidget *title_label, *body_label, *time_label;
87 GDateTime *dt;
88 gchar *time_str, *markup;
89
90 /* Format timestamp */
91 dt = g_date_time_new_from_unix_local (timestamp / G_TIME_SPAN_SECOND);
92 time_str = g_date_time_format (dt, "%H:%M");
93 g_date_time_unref (dt);
94
95 /* Create row container for the entire notification */
96 row = gtk_list_box_row_new ();
97 hbox = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 12);
98 gtk_container_set_border_width (GTK_CONTAINER (hbox), 6);
99
100 icon_image = create_notification_icon (app_icon);
101
102 content_box = gtk_box_new (GTK_ORIENTATION_VERTICAL, 3);
103
104 /* Title */
105 markup = g_markup_printf_escaped ("<b>%s</b>", summary ? summary : _("(No summary)"));
106 title_label = gtk_label_new (NULL);
107 gtk_label_set_markup (GTK_LABEL (title_label), markup);
108 gtk_widget_set_halign (title_label, GTK_ALIGN_START);
109 gtk_label_set_ellipsize (GTK_LABEL (title_label), PANGO_ELLIPSIZE_END);
110 gtk_label_set_max_width_chars (GTK_LABEL (title_label), 40);
111 g_free (markup);
112
113 /* Body */
114 body_label = NULL;
115 if (body && *body) {
116 body_label = gtk_label_new (NULL);
117 gtk_label_set_markup (GTK_LABEL (body_label), body);
118 gtk_widget_set_halign (body_label, GTK_ALIGN_START);
119 gtk_label_set_ellipsize (GTK_LABEL (body_label), PANGO_ELLIPSIZE_END);
120 gtk_label_set_max_width_chars (GTK_LABEL (body_label), 40);
121
122 /* Set tooltip in case the body overflows */
123 gchar *tooltip_text = g_strdup_printf ("%s\n%s", summary, body);
124 gtk_widget_set_tooltip_text (row, tooltip_text);
125 g_free (tooltip_text);
126 }
127
128 /* Time and app label */
129 markup = g_markup_printf_escaped ("<small>%s - %s</small>",
130 app_name ? app_name : _("Unknown"),
131 time_str);
132 time_label = gtk_label_new (NULL);
133 gtk_label_set_markup (GTK_LABEL (time_label), markup);
134 gtk_widget_set_halign (time_label, GTK_ALIGN_START);
135 g_free (markup);
136 g_free (time_str);
137
138 /* Pack content box */
139 gtk_box_pack_start (GTK_BOX (content_box), title_label, FALSE, FALSE, 0);
140 if (body_label)
141 gtk_box_pack_start (GTK_BOX (content_box), body_label, FALSE, FALSE, 0);
142 gtk_box_pack_start (GTK_BOX (content_box), time_label, FALSE, FALSE, 0);
143
144 /* Pack horizontal box */
145 gtk_box_pack_start (GTK_BOX (hbox), icon_image, FALSE, FALSE, 0);
146 gtk_box_pack_start (GTK_BOX (hbox), content_box, TRUE, TRUE, 0);
147
148 gtk_container_add (GTK_CONTAINER (row), hbox);
149
150 return row;
151}

222static void
223position_popup_window (GtkWidget *popup,
224 GtkWidget *reference_widget,
225 GdkScreen *screen,
226 gint popup_width,
227 gint popup_height)
228{
229 gint x, y;
230 GdkWindow *window;
231
232 window = gtk_widget_get_window (reference_widget);
233 if (!window) {
234 return;
235 }
236
237 gdk_window_get_origin (window, &x, &y);
238
239 /* Calculate popup dimensions */
240 gint applet_height = gtk_widget_get_allocated_height (reference_widget);
241
242 /* Get screen dimensions */
243 gint screen_width = gdk_screen_get_width (screen);
244 gint screen_height = gdk_screen_get_height (screen);
245
246 /* Calculate initial position below applet */
247 gint popup_x = x;
248 gint popup_y = y + applet_height;
249
250 /* Check if popup extends beyond screen boundaries and adjust */
251 if (popup_x + popup_width > screen_width) {
252 popup_x = screen_width - popup_width;
253 }
254 if (popup_x < 0) {
255 popup_x = 0;
256 }
257
258 /* If popup extends below screen, place it above the applet instead */
259 if (popup_y + popup_height > screen_height) {
260 popup_y = y - popup_height;
261 /* If it still doesn't fit above, center it vertically */
262 if (popup_y < 0) {
263 popup_y = (screen_height - popup_height) / 2;
264 if (popup_y < 0) popup_y = 0;
265 }
266 }
267
268 gtk_window_move (GTK_WINDOW (popup), popup_x, popup_y);
269}

271void
272show_notification_history (MateNotificationHistoryContext *context)
273{
274 GtkWidget *popup;
275 GtkWidget *vbox;
276 GtkWidget *scrolled_window;
277 GtkWidget *list_box;
278 GtkWidget *button_box;
279 GtkWidget *dnd_toggle;
280 GtkWidget *clear_button;
281 GtkWidget *close_button;
282 GVariant *result;
283 GVariantIter *iter;
284 guint id, urgency;
285 gchar *app_name, *app_icon, *summary, *body;
286 gint64 timestamp, closed_timestamp;
287 guint reason;
288 gboolean read;
289
290 if (!dbus_context_is_available (context->dbus_context)) {
291 g_warning ("Cannot show history: daemon not available");
292 return;
293 }
294
295 /* Check if history is enabled */
296 if (context->settings && !g_settings_get_boolean (context->settings, GSETTINGS_KEY_HISTORY_ENABLED)) {
297 g_warning ("Cannot show history: history is disabled for privacy");
298 return;
299 }
300
301 /* If popup already exists, destroy it (basically toggle off) */
302 if (context->history_popup) {
303 gtk_widget_destroy (context->history_popup);
304 context->history_popup = NULL;
305 return;
306 }
307
308 /* Get notification history from daemon */
309 result = dbus_context_get_notification_history (context->dbus_context);
310
311 if (!result) {
312 g_warning ("Failed to get notification history");
313 return;
314 }
315
316 /* Trigger count update since accessing history marks all as read */
317 if (context->count_update_callback) {
318 ((void (*)(gpointer)) context->count_update_callback) (context->count_update_user_data);
319 }
320
321 popup = create_popup_window ();
322
323 context->history_popup = popup;
324 g_signal_connect (popup, "destroy", G_CALLBACK (popup_destroyed_cb), context);
325
326 /* Create main container */
327 vbox = gtk_box_new (GTK_ORIENTATION_VERTICAL, 6);
328 gtk_container_set_border_width (GTK_CONTAINER (vbox), 12);
329 gtk_container_add (GTK_CONTAINER (popup), vbox);
330
331 /* Create list box for all notifications */
332 list_box = gtk_list_box_new ();
333 gtk_list_box_set_selection_mode (GTK_LIST_BOX (list_box), GTK_SELECTION_NONE);
334
335 /* Get history data and add to list */
336 gint notification_count = 0;
337 if (result) {
338 g_variant_get (result, "(a(ussssxxuub))", &iter);
339
340 while (g_variant_iter_loop (iter, "(ussssxxuub)",
341 &id, &app_name, &app_icon, &summary, &body,
342 &timestamp, &closed_timestamp, &reason, &urgency, &read)) {
343 notification_count++;
344 /* Add each notification as a new row in the list */
345 GtkWidget *row = create_notification_row (id, app_name, app_icon, summary, body, timestamp);
346 gtk_list_box_insert (GTK_LIST_BOX (list_box), row, -1);
347 }
348
349 g_variant_iter_free (iter);
350 g_variant_unref (result);
351 }
352
353 /* Add message if list is empty */
354 if (gtk_list_box_get_row_at_index (GTK_LIST_BOX (list_box), 0) == NULL) {
355 GtkWidget *row = gtk_list_box_row_new ();
356 GtkWidget *label = gtk_label_new (_("No notifications"));
357 gtk_widget_set_sensitive (label, FALSE);
358 gtk_container_set_border_width (GTK_CONTAINER (row), 12);
359 gtk_container_add (GTK_CONTAINER (row), label);
360 gtk_list_box_insert (GTK_LIST_BOX (list_box), row, -1);
361 }
362
363 /* Add this window for the list of notifications */
364 scrolled_window = gtk_scrolled_window_new (NULL, NULL);
365 gtk_scrolled_window_set_policy (GTK_SCROLLED_WINDOW (scrolled_window),
366 GTK_POLICY_NEVER, GTK_POLICY_AUTOMATIC);
367
368 /* Calculate to ensure the popup is the right height:
369 * 80% of screen height, with some room for buttons */
370 GdkScreen *screen = gtk_widget_get_screen (context->main_widget);
371 gint max_content_height = (gdk_screen_get_height (screen) * 0.8) - 100;
372
373 gint content_height = 100; /* 100px minimum */
374 if (notification_count > 0)
375 content_height = MIN(notification_count * content_height, max_content_height);
376
377 /* Now force the scrollable window to be the calculated height */
378 gtk_scrolled_window_set_max_content_height (GTK_SCROLLED_WINDOW (scrolled_window), max_content_height);
379 gtk_scrolled_window_set_min_content_height (GTK_SCROLLED_WINDOW (scrolled_window), content_height);
380 gtk_container_add (GTK_CONTAINER (scrolled_window), list_box);
381
382 gtk_box_pack_start (GTK_BOX (vbox), scrolled_window, TRUE, TRUE, 0);
383
384 /* Add a box for action buttons */
385 button_box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
386 gtk_container_set_border_width (GTK_CONTAINER (button_box), 6);
387
388 /* DND toggle first */
389 dnd_toggle = gtk_check_button_new_with_label (_("Do not disturb"));
390 gtk_widget_set_halign (dnd_toggle, GTK_ALIGN_START);
391
392 if (context->settings) {
393 gboolean dnd_enabled = g_settings_get_boolean (context->settings, "do-not-disturb");
394 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (dnd_toggle), dnd_enabled);
395 }
396
397 g_signal_connect (dnd_toggle, "toggled", G_CALLBACK (dnd_toggle_clicked), context);
398 gtk_box_pack_start (GTK_BOX (button_box), dnd_toggle, FALSE, FALSE, 0);
399
400 /* Spacer to push buttons to the right */
401 GtkWidget *spacer = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);
402 gtk_box_pack_start (GTK_BOX (button_box), spacer, TRUE, TRUE, 0);
403
404 /* Then the action buttons */
405 clear_button = gtk_button_new_with_label (_("Clear All"));
406 close_button = gtk_button_new_with_label (_("Close"));
407
408 g_signal_connect (clear_button, "clicked", G_CALLBACK (clear_history), context);
409 g_signal_connect_swapped (close_button, "clicked", G_CALLBACK (gtk_widget_destroy), popup);
410
411 gtk_box_pack_end (GTK_BOX (button_box), close_button, FALSE, FALSE, 0);
412 gtk_box_pack_end (GTK_BOX (button_box), clear_button, FALSE, FALSE, 0);
413 gtk_box_pack_start (GTK_BOX (vbox), button_box, FALSE, FALSE, 0);
414
415 /* Position popup window with boundary checking */
416 position_popup_window (popup, context->main_widget, screen, 450, content_height + 100);
417
418 /* Set popup size based on content (with space for buttons) */
419 gtk_window_set_default_size (GTK_WINDOW (popup), 450, content_height + 100);
420 gtk_widget_show_all (popup);
421}
