133static void
134instance_dispose( GObject *instance )
135{
136 static const gchar *thisfn = "base_builder_instance_dispose";
137 BaseBuilder *self;
138 GSList *objects;
139
140 g_return_if_fail( BASE_IS_BUILDER( instance ));
141
142 self = BASE_BUILDER( instance );
143
144 if( !self->private->dispose_has_run ){
145
146 g_debug( "%s: instance=%p (%s)", thisfn, ( void * ) instance, G_OBJECT_TYPE_NAME( instance ));
147
148 objects = gtk_builder_get_objects( GTK_BUILDER( instance ));
149 g_slist_foreach( objects, ( GFunc ) free_built_object, NULL );
150 g_slist_free( objects );
151
152 self->private->dispose_has_run = TRUE;
153
154 /* chain up to the parent class */
155 if( G_OBJECT_CLASS( st_parent_class )->dispose ){
156 G_OBJECT_CLASS( st_parent_class )->dispose( instance );
157 }
158 }
159}

227gboolean
228base_builder_add_from_file( BaseBuilder *builder, const gchar *filename, GError **error )
229{
230 static const gchar *thisfn = "base_builder_add_from_file";
231 gboolean ret = FALSE;
232 guint retint;
233
234 g_return_val_if_fail( BASE_IS_BUILDER( builder ), FALSE );
235
236 if( !builder->private->dispose_has_run ){
237
238 if( already_loaded( builder, filename )){
239 g_debug( "%s: %s already loaded", thisfn, filename );
240 ret = TRUE;
241
242 } else {
243 retint = gtk_builder_add_from_file( GTK_BUILDER( builder ), filename, error );
244 if( retint > 0 && !( *error )){
245 builder->private->fnames = g_slist_prepend( builder->private->fnames, g_strdup( filename ));
246 ret = TRUE;
247 }
248 }
249 }
250
251 return( ret );
252}

254static gboolean
255already_loaded( BaseBuilder *builder, const gchar *filename )
256{
257 gboolean loaded = FALSE;
258 GSList *it;
259
260 for( it = builder->private->fnames ; it && !loaded ; it = it->next ){
261 if( !na_core_utils_str_collate(( const gchar * ) it->data, filename )){
262 loaded = TRUE;
263 }
264 }
265
266 return( loaded );
267}
