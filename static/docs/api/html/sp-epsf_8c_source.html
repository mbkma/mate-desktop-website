54/* Note: the given strings are modified in place */
55static char *parse_epsf_special(EpsfBox *box, char **ret,
56 char *prefix, char *arg)
57{
58 static struct {
59 char *name;
60 int has_arg;
61 char *value;
62 } keys[] = {
63 {"llx", 1, "0"},
64 {"lly", 1, "0"},
65 {"urx", 1, "0"},
66 {"ury", 1, "0"},
67 {"rwi", 1, "0"},
68 {"rhi", 1, "0"},
69 {"hoffset", 1, "0"},
70 {"voffset", 1, "0"},
71 {"hsize", 1, "612"},
72 {"vsize", 1, "792"},
73 {"hscale", 1, "100"},
74 {"vscale", 1, "100"},
75 {"angle", 1, "0"},
76 {"clip", 0, "0"}
77 };
78#define NKEYS (sizeof(keys) / sizeof(keys[0]))
79 char *ptr;
80 char *filename;
81 double value[NKEYS];
82 Uchar present[NKEYS];
83 Buffer buffer;
84 char *name;
85 int i;
86 double originx;
87 double originy;
88 double hsize;
89 double vsize;
90 double hscale;
91 double vscale;
92
93 /* this special has the form
94 * ["]file.ps["] [key=valye]*
95 */
96
97 /* scan the filename */
98 while(*arg == ' ' || *arg == '\t')
99 arg++;
100
101 /* make a copy of the string */
102 ptr = arg;
103
104 if(*ptr == '"')
105 for(name = ++ptr; *ptr && *ptr != '"'; ptr++);
106 else
107 for(name = ptr; *ptr && *ptr != ' ' && *ptr != '\t'; ptr++);
108 if(ptr == name)
109 return NULL;
110 *ptr++ = 0;
111 filename = name;
112
113 /* reset values to defaults */
114 for(i = 0; i < NKEYS; i++) {
115 value[i] = atof(keys[i].value);
116 present[i] = 0;
117 }
118
119 buff_init(&buffer);
120 buff_add(&buffer, "@beginspecial ", 0);
121
122 while(*ptr) {
123 const char *keyname;
124 char *val;
125 char *p;
126
127 while(*ptr == ' ' || *ptr == '\t')
128 ptr++;
129 keyname = ptr;
130
131 /* get the whole key=value pair */
132 for(; *ptr && *ptr != ' ' && *ptr != '\t'; ptr++);
133
134 if(*ptr) *ptr++ = 0;
135 /* now we shouldn't touch `ptr' anymore */
136
137 /* now work on this pair */
138 p = strchr(keyname, '=');
139 if(p == NULL)
140 val = NULL;
141 else {
142 *p++ = 0;
143 if(*p == '"') {
144 val = ++p;
145 /* skip until closing quote */
146 while(*p && *p != '"')
147 p++;
148 if(*p != '"')
149 mdvi_warning(
150 _("%s: malformed value for key `%s'\n"),
151 filename, keyname);
152 } else
153 val = p;
154 }
155
156 /* lookup the key */
157 for(i = 0; i < NKEYS; i++)
158 if(STRCEQ(keys[i].name, keyname))
159 break;
160 if(i == NKEYS) {
161 mdvi_warning(_("%s: unknown key `%s' ignored\n"),
162 filename, keyname);
163 continue;
164 }
165 if(keys[i].has_arg && val == NULL) {
166 mdvi_warning(_("%s: no argument for key `%s', using defaults\n"),
167 filename, keyname);
168 val = keys[i].value;
169 } else if(!keys[i].has_arg && val) {
170 mdvi_warning(_("%s: argument `%s' ignored for key `%s'\n"),
171 filename, val, keyname);
172 val = NULL;
173 }
174 if(val)
175 value[i] = atof(val);
176
177 /* put the argument */
178 buff_add(&buffer, val, 0);
179 buff_add(&buffer, " @", 2);
180 buff_add(&buffer, keyname, 0);
181 buff_add(&buffer, " ", 1);
182
183 /* remember that this option was given */
184 present[i] = 0xff;
185 }
186 buff_add(&buffer, " @setspecial", 0);
187
188 /* now compute the bounding box (code comes from dvips) */
189 originx = 0;
190 originy = 0;
191 hscale = 1;
192 vscale = 1;
193 hsize = 0;
194 vsize = 0;
195
196 if(present[HSIZE])
197 hsize = value[HSIZE];
198 if(present[VSIZE])
199 vsize = value[VSIZE];
200 if(present[HOFF])
201 originx = value[HOFF];
202 if(present[VOFF])
203 originy = value[VOFF];
204 if(present[HSCALE])
205 hscale = value[HSCALE] / 100.0;
206 if(present[VSCALE])
207 vscale = value[VSCALE] / 100.0;
208 if(present[URX] && present[LLX])
209 hsize = value[URX] - value[LLX];
210 if(present[URY] && present[LLY])
211 vsize = value[URY] - value[LLY];
212 if(present[RWI] || present[RHI]) {
213 if(present[RWI] && !present[RHI])
214 hscale = vscale = value[RWI] / (10.0 * hsize);
215 else if(present[RHI] && !present[RWI])
216 hscale = vscale = value[RHI] / (10.0 * vsize);
217 else {
218 hscale = value[RWI] / (10.0 * hsize);
219 vscale = value[RHI] / (10.0 * vsize);
220 }
221 }
222
223 box->ox = originx;
224 box->oy = originy;
225 box->bw = hsize * hscale;
226 box->bh = vsize * vscale;
227 box->angle = value[ANGLE];
228
229 *ret = buffer.data;
230
231 return filename;
232}

234void epsf_special(DviContext *dvi, char *prefix, char *arg)
235{
236 char *file;
237 char *special;
238 char *psfile;
239 char *tmp;
240 EpsfBox box = {0, 0, 0, 0};
241 int x, y;
242 int w, h;
243 double xf, vf;
244 struct stat buf;
245
246 file = parse_epsf_special(&box, &special, prefix, arg);
247 if (file != NULL)
248 mdvi_free (special);
249
250 xf = dvi->params.dpi * dvi->params.mag / (72.0 * dvi->params.hshrink);
251 vf = dvi->params.vdpi * dvi->params.mag / (72.0 * dvi->params.vshrink);
252 w = FROUND(box.bw * xf);
253 h = FROUND(box.bh * vf);
254 x = FROUND(box.ox * xf) + dvi->pos.hh;
255 y = FROUND(box.oy * vf) + dvi->pos.vv - h + 1;
256
257 if (!file || !dvi->device.draw_ps) {
258 dvi->device.draw_rule (dvi, x, y, w, h, 0);
259 return;
260 }
261
262 if (file[0] == '/') { /* Absolute path */
263 if (stat (file, &buf) == 0)
264 dvi->device.draw_ps (dvi, file, x, y, w, h);
265 else
266 dvi->device.draw_rule (dvi, x, y, w, h, 0);
267 return;
268 }
269
270 tmp = mdvi_strrstr (dvi->filename, "/");
271 if (tmp) { /* Document directory */
272 int path_len = strlen (dvi->filename) - strlen (tmp + 1);
273 int file_len = strlen (file);
274
275 psfile = mdvi_malloc (path_len + file_len + 1);
276 psfile[0] = '\0';
277 strncat (psfile, dvi->filename, path_len);
278 strncat (psfile, file, file_len);
279
280 if (stat (psfile, &buf) == 0) {
281 dvi->device.draw_ps (dvi, psfile, x, y, w, h);
282 mdvi_free (psfile);
283
284 return;
285 }
286
287 mdvi_free (psfile);
288 }
289
290 psfile = mdvi_build_path_from_cwd (file);
291 if (stat (psfile, &buf) == 0) { /* Current working dir */
292 dvi->device.draw_ps (dvi, psfile, x, y, w, h);
293 mdvi_free (psfile);
294
295 return;
296 }
297
298 mdvi_free (psfile);
299
300 psfile = kpse_find_pict (file);
301 if (psfile) { /* kpse */
302 dvi->device.draw_ps (dvi, psfile, x, y, w, h);
303 } else {
304 dvi->device.draw_rule(dvi, x, y, w, h, 0);
305 }
306
307 free (psfile);
308}
