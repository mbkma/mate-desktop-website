113static void
114on_network_device_combo_changed (GtkComboBox *combo,
115 NetspeedPreferences *preferences)
116{
117 gint active;
118 gboolean old_auto_change_device, auto_change_device;
119
120 active = gtk_combo_box_get_active (combo);
121 g_assert (active > -1);
122
123 old_auto_change_device = auto_change_device = g_settings_get_boolean (preferences->settings,
124 "auto-change-device");
125 if (0 == active) {
126 if (auto_change_device)
127 return;
128 auto_change_device = TRUE;
129 } else {
130 const gchar *current_device_name;
131 const gchar *selected_device_name;
132
133 current_device_name = netspeed_applet_get_current_device_name (preferences->netspeed);
134 auto_change_device = FALSE;
135 selected_device_name = g_list_nth_data (preferences->devices, (guint) (active - 1));
136 if (!g_strcmp0 (selected_device_name, current_device_name))
137 return;
138 g_settings_set_string (preferences->settings,
139 "device", selected_device_name);
140 }
141 if (old_auto_change_device != auto_change_device)
142 g_settings_set_boolean (preferences->settings,
143 "auto-change-device", auto_change_device);
144}

188static void
189fill_device_combo (NetspeedPreferences *preferences, GSettings *settings)
190{
191 GList *ptr;
192 int i, active = -1;
193 const gchar *current_device_name;
194 gboolean auto_change_device;
195
196 /* Default means device with default route set */
197 gtk_combo_box_text_append_text (GTK_COMBO_BOX_TEXT (preferences->network_device_combo),
198 _("Default"));
199 ptr = preferences->devices = get_available_devices ();
200 current_device_name = netspeed_applet_get_current_device_name (preferences->netspeed);
201 auto_change_device = g_settings_get_boolean (settings, "auto-change-device");
202 for (i = 0; ptr; ptr = g_list_next (ptr)) {
203 gtk_combo_box_text_append_text (GTK_COMBO_BOX_TEXT (preferences->network_device_combo),
204 ptr->data);
205 if (g_str_equal (ptr->data, current_device_name))
206 active = (i + 1);
207 ++i;
208 }
209 if (active < 0 || auto_change_device)
210 active = 0;
211 gtk_combo_box_set_active (GTK_COMBO_BOX (preferences->network_device_combo), active);
212}
