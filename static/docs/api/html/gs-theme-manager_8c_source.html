71/* Returns the full path to the queried command */
72static char *
73find_command (const char *command)
74{
75 int i;
76
77 if (g_path_is_absolute (command))
78 {
79 char *dirname;
80
81 dirname = g_path_get_dirname (command);
82 for (i = 0; known_engine_locations [i]; i++)
83 {
84 if (strcmp (dirname, known_engine_locations [i]) == 0)
85 {
86 if (g_file_test (command, G_FILE_TEST_IS_EXECUTABLE)
87 && ! g_file_test (command, G_FILE_TEST_IS_DIR))
88 {
89 g_free (dirname);
90 return g_strdup (command);
91 }
92 }
93 }
94 g_free (dirname);
95 }
96 else
97 {
98 for (i = 0; known_engine_locations [i]; i++)
99 {
100 char *path;
101
102 path = g_build_filename (known_engine_locations [i], command, NULL);
103
104 if (g_file_test (path, G_FILE_TEST_IS_EXECUTABLE)
105 && ! g_file_test (path, G_FILE_TEST_IS_DIR))
106 {
107 return path;
108 }
109
110 g_free (path);
111 }
112 }
113
114 return NULL;
115}

117static gboolean
118check_command (const char *command)
119{
120 char *path;
121 char **argv;
122
123 g_return_val_if_fail (command != NULL, FALSE);
124
125 g_shell_parse_argv (command, NULL, &argv, NULL);
126 path = find_command (argv [0]);
127 g_strfreev (argv);
128
129 if (path != NULL)
130 {
131 g_free (path);
132 return TRUE;
133 }
134
135 return FALSE;
136}

138static void
139add_known_engine_locations_to_path (void)
140{
141 static gboolean already_added;
142 int i;
143 GString *str;
144
145 /* We only want to add the items to the path once */
146 if (already_added)
147 {
148 return;
149 }
150
151 already_added = TRUE;
152
153 /* TODO: set a default PATH ? */
154
155 str = g_string_new (g_getenv ("PATH"));
156 for (i = 0; known_engine_locations [i]; i++)
157 {
158 /* TODO: check that permissions are safe */
159 if (g_file_test (known_engine_locations [i], G_FILE_TEST_IS_DIR))
160 {
161 g_string_append_printf (str, ":%s", known_engine_locations [i]);
162 }
163 }
164
165 g_setenv ("PATH", str->str, TRUE);
166 g_string_free (str, TRUE);
167}

231static GSThemeInfo *
232gs_theme_info_new_from_matemenu_tree_entry (MateMenuTreeEntry *entry)
233{
234 GSThemeInfo *info;
235 const char *str;
236 char *pos;
237 GDesktopAppInfo *ginfo;
238
239 info = g_new0 (GSThemeInfo, 1);
240 ginfo = matemenu_tree_entry_get_app_info (entry);
241
242 info->refcount = 1;
243 info->name = g_strdup (g_app_info_get_name(G_APP_INFO(ginfo)));
244 info->exec = g_strdup (g_app_info_get_commandline(G_APP_INFO(ginfo)));
245
246 /* remove the .desktop suffix */
247 str = matemenu_tree_entry_get_desktop_file_id (entry);
248 pos = g_strrstr (str, ".desktop");
249 if (pos)
250 {
251 info->file_id = g_strndup (str, pos - str);
252 }
253 else
254 {
255 info->file_id = g_strdup (str);
256 }
257
258 return info;
259}

261static GSThemeInfo *
262find_info_for_id (MateMenuTree *tree,
263 const char *id)
264{
265 GSThemeInfo *info;
266 MateMenuTreeDirectory *root;
267 MateMenuTreeIter *iter;
268 MateMenuTreeItemType type;
269
270 root = matemenu_tree_get_root_directory (tree);
271 if (root == NULL)
272 {
273 return NULL;
274 }
275
276 info = NULL;
277 iter = matemenu_tree_directory_iter (root);
278 while ((type = matemenu_tree_iter_next (iter)) != MATEMENU_TREE_ITEM_INVALID) {
279 if (info == NULL && type == MATEMENU_TREE_ITEM_ENTRY) {
280 MateMenuTreeEntry *entry;
281 const char *file_id;
282
283 entry = matemenu_tree_iter_get_entry(iter);
284 file_id = matemenu_tree_entry_get_desktop_file_id (entry);
285 if (file_id && id && strcmp (file_id, id) == 0)
286 {
287 info = gs_theme_info_new_from_matemenu_tree_entry (entry);
288 }
289 matemenu_tree_item_unref (entry);
290 }
291 }
292 matemenu_tree_iter_unref (iter);
293 matemenu_tree_item_unref (root);
294 return info;
295}

326static void
327make_theme_list (GSList **parent_list,
328 MateMenuTreeDirectory *directory,
329 const char *filename)
330{
331 MateMenuTreeIter *iter;
332 MateMenuTreeItemType type;
333
334 iter = matemenu_tree_directory_iter (directory);
335 while ((type = matemenu_tree_iter_next (iter)) != MATEMENU_TREE_ITEM_INVALID) {
336 if (type == MATEMENU_TREE_ITEM_ENTRY) {
337 MateMenuTreeEntry *item;
338 item = matemenu_tree_iter_get_entry (iter);
339 theme_prepend_entry (parent_list, (MateMenuTreeEntry*)item, filename);
340 matemenu_tree_item_unref (item);
341 }
342 }
343 matemenu_tree_iter_unref (iter);
344 *parent_list = g_slist_reverse (*parent_list);
345}

374static MateMenuTree *
375get_themes_tree (void)
376{
377 MateMenuTree *themes_tree = NULL;
378 GError *error = NULL;
379
380 /* we only need to add the locations to the path once
381 and since this is only run once we'll do it here */
382 add_known_engine_locations_to_path ();
383
384 themes_tree = matemenu_tree_new ("mate-screensavers.menu", MATEMENU_TREE_FLAGS_NONE);
385 if (!matemenu_tree_load_sync (themes_tree, &error)) {
386 g_debug("Load matemenu tree got error: %s\n", error->message);
387 g_error_free(error);
388 g_object_unref(themes_tree);
389 return NULL;
390 }
391
392 return themes_tree;
393}
