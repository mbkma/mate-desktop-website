68static void
69sample_info_cb (pa_context *c, const pa_sample_info *i, int eol, void *userdata)
70{
71 pa_operation *o;
72
73 if (!i)
74 return;
75
76 g_debug ("Found sample %s", i->name);
77
78 /* We only flush those samples which have an XDG sound name
79 * attached, because only those originate from themeing */
80 if (!(pa_proplist_gets (i->proplist, PA_PROP_EVENT_ID)))
81 return;
82
83 g_debug ("Dropping sample %s from cache", i->name);
84
85 if (!(o = pa_context_remove_sample (c, i->name, NULL, NULL))) {
86 g_debug ("pa_context_remove_sample (): %s", pa_strerror (pa_context_errno (c)));
87 return;
88 }
89
90 pa_operation_unref (o);
91
92 /* We won't wait until the operation is actually executed to
93 * speed things up a bit.*/
94}

96static void
97flush_cache (void)
98{
99 pa_mainloop *ml = NULL;
100 pa_context *c = NULL;
101 pa_proplist *pl = NULL;
102 pa_operation *o = NULL;
103
104 g_debug ("Flushing sample cache");
105
106 if (!(ml = pa_mainloop_new ())) {
107 g_debug ("Failed to allocate pa_mainloop");
108 goto fail;
109 }
110
111 if (!(pl = pa_proplist_new ())) {
112 g_debug ("Failed to allocate pa_proplist");
113 goto fail;
114 }
115
116 pa_proplist_sets (pl, PA_PROP_APPLICATION_NAME, PACKAGE_NAME);
117 pa_proplist_sets (pl, PA_PROP_APPLICATION_VERSION, PACKAGE_VERSION);
118 pa_proplist_sets (pl, PA_PROP_APPLICATION_ID, "org.mate.SettingsDaemon");
119
120 if (!(c = pa_context_new_with_proplist (pa_mainloop_get_api (ml), PACKAGE_NAME, pl))) {
121 g_debug ("Failed to allocate pa_context");
122 goto fail;
123 }
124
125 pa_proplist_free (pl);
126 pl = NULL;
127
128 if (pa_context_connect (c, NULL, PA_CONTEXT_NOAUTOSPAWN, NULL) < 0) {
129 g_debug ("pa_context_connect(): %s", pa_strerror (pa_context_errno (c)));
130 goto fail;
131 }
132
133 /* Wait until the connection is established */
134 while (pa_context_get_state (c) != PA_CONTEXT_READY) {
135
136 if (!PA_CONTEXT_IS_GOOD (pa_context_get_state (c))) {
137 g_debug ("Connection failed: %s", pa_strerror (pa_context_errno (c)));
138 goto fail;
139 }
140
141 if (pa_mainloop_iterate (ml, TRUE, NULL) < 0) {
142 g_debug ("pa_mainloop_iterate() failed");
143 goto fail;
144 }
145 }
146
147 /* Enumerate all cached samples */
148 if (!(o = pa_context_get_sample_info_list (c, sample_info_cb, NULL))) {
149 g_debug ("pa_context_get_sample_info_list(): %s", pa_strerror (pa_context_errno (c)));
150 goto fail;
151 }
152
153 /* Wait until our operation is finished and there's nothing
154 * more queued to send to the server */
155 while (pa_operation_get_state (o) == PA_OPERATION_RUNNING || pa_context_is_pending (c)) {
156
157 if (!PA_CONTEXT_IS_GOOD (pa_context_get_state (c))) {
158 g_debug ("Connection failed: %s", pa_strerror (pa_context_errno (c)));
159 goto fail;
160 }
161
162 if (pa_mainloop_iterate (ml, TRUE, NULL) < 0) {
163 g_debug ("pa_mainloop_iterate() failed");
164 goto fail;
165 }
166 }
167
168 g_debug ("Sample cache flushed");
169
170fail:
171 if (o) {
172 pa_operation_cancel (o);
173 pa_operation_unref (o);
174 }
175
176 if (c) {
177 pa_context_disconnect (c);
178 pa_context_unref (c);
179 }
180
181 if (pl)
182 pa_proplist_free (pl);
183
184 if (ml)
185 pa_mainloop_free (ml);
186}

196static void
197trigger_flush (MsdSoundManager *manager)
198{
199
200 if (manager->timeout)
201 g_source_remove (manager->timeout);
202
203 /* We delay the flushing a bit so that we can coalesce
204 * multiple changes into a single cache flush */
205 manager->timeout = g_timeout_add (500, (GSourceFunc) flush_cb, manager);
206}

227static gboolean
228register_directory_callback (MsdSoundManager *manager,
229 const char *path,
230 GError **error)
231{
232 GFile *f;
233 GFileMonitor *m;
234 gboolean succ = FALSE;
235
236 g_debug ("Registering directory monitor for %s", path);
237
238 f = g_file_new_for_path (path);
239
240 m = g_file_monitor_directory (f, 0, NULL, error);
241
242 if (m != NULL) {
243 g_signal_connect (m, "changed", G_CALLBACK (file_monitor_changed_cb), manager);
244
245 manager->monitors = g_list_prepend (manager->monitors, m);
246
247 succ = TRUE;
248 }
249
250 g_object_unref (f);
251
252 return succ;
253}

257gboolean
258msd_sound_manager_start (MsdSoundManager *manager,
259 GError **error)
260{
261
262#ifdef HAVE_PULSE
263 char *p, **ps, **k;
264 const char *env, *dd;
265#endif
266
267 g_debug ("Starting sound manager");
268 mate_settings_profile_start (NULL);
269
270#ifdef HAVE_PULSE
271
272 /* We listen for change of the selected theme ... */
273 manager->settings = g_settings_new (MATE_SOUND_SCHEMA);
274
275 g_signal_connect (manager->settings, "changed", G_CALLBACK (gsettings_notify_cb), manager);
276
277 /* ... and we listen to changes of the theme base directories
278 * in $HOME ...*/
279
280 if ((env = g_getenv ("XDG_DATA_HOME")) && *env == '/')
281 p = g_build_filename (env, "sounds", NULL);
282 else if (((env = g_getenv ("HOME")) && *env == '/') || (env = g_get_home_dir ()))
283 p = g_build_filename (env, ".local", "share", "sounds", NULL);
284 else
285 p = NULL;
286
287 if (p) {
288 register_directory_callback (manager, p, NULL);
289 g_free (p);
290 }
291
292 /* ... and globally. */
293 if (!(dd = g_getenv ("XDG_DATA_DIRS")) || *dd == 0)
294 dd = "/usr/local/share:/usr/share";
295
296 ps = g_strsplit (dd, ":", 0);
297
298 for (k = ps; *k; ++k)
299 register_directory_callback (manager, *k, NULL);
300
301 g_strfreev (ps);
302#endif
303
304 mate_settings_profile_end (NULL);
305
306 return TRUE;
307}

309void
310msd_sound_manager_stop (MsdSoundManager *manager)
311{
312 g_debug ("Stopping sound manager");
313
314#ifdef HAVE_PULSE
315 if (manager->settings != NULL) {
316 g_object_unref (manager->settings);
317 manager->settings = NULL;
318 }
319
320 if (manager->timeout) {
321 g_source_remove (manager->timeout);
322 manager->timeout = 0;
323 }
324
325 while (manager->monitors) {
326 g_file_monitor_cancel (G_FILE_MONITOR (manager->monitors->data));
327 g_object_unref (manager->monitors->data);
328 manager->monitors = g_list_delete_link (manager->monitors, manager->monitors);
329 }
330#endif
331}
