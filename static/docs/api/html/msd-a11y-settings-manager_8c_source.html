57static void
58apps_settings_changed (GSettings *settings G_GNUC_UNUSED,
59 const char *key,
60 MsdA11ySettingsManager *manager)
61{
62 gboolean screen_reader, keyboard;
63
64 if (g_str_equal (key, "screen-reader-enabled") == FALSE &&
65 g_str_equal (key, "screen-keyboard-enabled") == FALSE)
66 return;
67
68 g_debug ("screen reader or OSK enablement changed");
69
70 screen_reader = g_settings_get_boolean (manager->priv->a11y_apps_settings, "screen-reader-enabled");
71 keyboard = g_settings_get_boolean (manager->priv->a11y_apps_settings, "screen-keyboard-enabled");
72
73 if (screen_reader || keyboard) {
74 g_debug ("Enabling accessibility, screen reader or OSK enabled");
75 g_settings_set_boolean (manager->priv->interface_settings, "accessibility", TRUE);
76 } else if (screen_reader == FALSE && keyboard == FALSE) {
77 g_debug ("Disabling accessibility, screen reader and OSK disabled");
78 g_settings_set_boolean (manager->priv->interface_settings, "accessibility", FALSE);
79 }
80}

82gboolean
83msd_a11y_settings_manager_start (MsdA11ySettingsManager *manager,
84 GError **error)
85{
86 g_debug ("Starting a11y_settings manager");
87 mate_settings_profile_start (NULL);
88
89 manager->priv->interface_settings = g_settings_new ("org.mate.interface");
90 manager->priv->a11y_apps_settings = g_settings_new ("org.gnome.desktop.a11y.applications");
91
92 g_signal_connect (G_OBJECT (manager->priv->a11y_apps_settings), "changed",
93 G_CALLBACK (apps_settings_changed), manager);
94
95 /* If any of the screen reader or on-screen keyboard are enabled,
96 * make sure a11y is enabled for the toolkits.
97 * We don't do the same thing for the reverse so it's possible to
98 * enable AT-SPI for the toolkits without using an a11y app */
99 if (g_settings_get_boolean (manager->priv->a11y_apps_settings, "screen-keyboard-enabled") ||
100 g_settings_get_boolean (manager->priv->a11y_apps_settings, "screen-reader-enabled"))
101 g_settings_set_boolean (manager->priv->interface_settings, "accessibility", TRUE);
102
103 mate_settings_profile_end (NULL);
104 return TRUE;
105}
