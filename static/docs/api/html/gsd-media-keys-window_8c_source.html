78static void
79action_changed (MsdMediaKeysWindow *window)
80{
81 if (!msd_osd_window_is_composited (MSD_OSD_WINDOW (window))) {
82 switch (window->priv->action) {
83 case MSD_MEDIA_KEYS_WINDOW_ACTION_VOLUME:
84 volume_controls_set_visible (window, TRUE);
85
86 if (window->priv->volume_muted) {
87 window_set_icon_name (window, "audio-volume-muted");
88 } else {
89 window_set_icon_name (window, "audio-volume-high");
90 }
91
92 break;
93 case MSD_MEDIA_KEYS_WINDOW_ACTION_CUSTOM:
94 volume_controls_set_visible (window, window->priv->show_level);
95 window_set_icon_name (window, window->priv->icon_name);
96 break;
97 default:
98 g_assert_not_reached ();
99 break;
100 }
101 }
102
103 msd_osd_window_update_and_hide (MSD_OSD_WINDOW (window));
104}

165 action_changed (window);
166 } else {
167 msd_osd_window_update_and_hide (MSD_OSD_WINDOW (window));
168 }

218static void
219draw_eject (cairo_t *cr,
220 double _x0,
221 double _y0,
222 double width,
223 double height)
224{
225 int box_height;
226 int tri_height;
227 int separation;
228
229 box_height = height * 0.2;
230 separation = box_height / 3;
231 tri_height = height - box_height - separation;
232
233 cairo_rectangle (cr, _x0, _y0 + height - box_height, width, box_height);
234
235 cairo_move_to (cr, _x0, _y0 + tri_height);
236 cairo_rel_line_to (cr, width, 0);
237 cairo_rel_line_to (cr, -width / 2, -tri_height);
238 cairo_rel_line_to (cr, -width / 2, tri_height);
239 cairo_close_path (cr);
240 cairo_set_source_rgba (cr, 1.0, 1.0, 1.0, MSD_OSD_WINDOW_FG_ALPHA);
241 cairo_fill_preserve (cr);
242
243 cairo_set_source_rgba (cr, 0.6, 0.6, 0.6, MSD_OSD_WINDOW_FG_ALPHA / 2);
244 cairo_set_line_width (cr, 2);
245 cairo_stroke (cr);
246}

248static void
249draw_waves (cairo_t *cr,
250 double cx,
251 double cy,
252 double max_radius,
253 int volume_level)
254{
255 const int n_waves = 3;
256 int last_wave;
257 int i;
258
259 last_wave = n_waves * volume_level / 100;
260
261 for (i = 0; i < n_waves; i++) {
262 double angle1;
263 double angle2;
264 double radius;
265 double alpha;
266
267 angle1 = -M_PI / 4;
268 angle2 = M_PI / 4;
269
270 if (i < last_wave)
271 alpha = 1.0;
272 else if (i > last_wave)
273 alpha = 0.1;
274 else alpha = 0.1 + 0.9 * (n_waves * volume_level % 100) / 100.0;
275
276 radius = (i + 1) * (max_radius / n_waves);
277 cairo_arc (cr, cx, cy, radius, angle1, angle2);
278 cairo_set_source_rgba (cr, 0.6, 0.6, 0.6, alpha / 2);
279 cairo_set_line_width (cr, 14);
280 cairo_set_line_cap (cr, CAIRO_LINE_CAP_ROUND);
281 cairo_stroke_preserve (cr);
282
283 cairo_set_source_rgba (cr, 1.0, 1.0, 1.0, alpha);
284 cairo_set_line_width (cr, 10);
285 cairo_set_line_cap (cr, CAIRO_LINE_CAP_ROUND);
286 cairo_stroke (cr);
287 }
288}

303 cairo_set_line_width (cr, 14);
304 cairo_set_line_cap (cr, CAIRO_LINE_CAP_ROUND);
305 cairo_stroke_preserve (cr);

313static void
314draw_speaker (cairo_t *cr,
315 double cx,
316 double cy,
317 double width,
318 double height)
319{
320 double box_width;
321 double box_height;
322 double _x0;
323 double _y0;
324
325 box_width = width / 3;
326 box_height = height / 3;
327
328 _x0 = cx - (width / 2) + box_width;
329 _y0 = cy - box_height / 2;
330
331 cairo_move_to (cr, _x0, _y0);
332 cairo_rel_line_to (cr, - box_width, 0);
333 cairo_rel_line_to (cr, 0, box_height);
334 cairo_rel_line_to (cr, box_width, 0);
335
336 cairo_line_to (cr, cx + box_width, cy + height / 2);
337 cairo_rel_line_to (cr, 0, -height);
338 cairo_line_to (cr, _x0, _y0);
339 cairo_close_path (cr);
340
341 cairo_set_source_rgba (cr, 1.0, 1.0, 1.0, MSD_OSD_WINDOW_FG_ALPHA);
342 cairo_fill_preserve (cr);
343
344 cairo_set_source_rgba (cr, 0.6, 0.6, 0.6, MSD_OSD_WINDOW_FG_ALPHA / 2);
345 cairo_set_line_width (cr, 2);
346 cairo_stroke (cr);
347}

349static gboolean
350render_speaker (MsdMediaKeysWindow *window,
351 cairo_t *cr,
352 double _x0,
353 double _y0,
354 double width,
355 double height)
356{
357 GdkPixbuf *pixbuf;
358 int icon_size;
359 int n;
360 static const char *icon_names[] = {
361 "audio-volume-muted",
362 "audio-volume-low",
363 "audio-volume-medium",
364 "audio-volume-high",
365 NULL
366 };
367
368 if (window->priv->volume_muted) {
369 n = 0;
370 } else {
371 /* select image */
372 n = 3 * window->priv->volume_level / 100 + 1;
373 if (n < 1) {
374 n = 1;
375 } else if (n > 3) {
376 n = 3;
377 }
378 }
379
380 icon_size = (int)width;
381
382 pixbuf = load_pixbuf (window, icon_names[n], icon_size);
383
384 if (pixbuf == NULL) {
385 return FALSE;
386 }
387
388 gdk_cairo_set_source_pixbuf (cr, pixbuf, _x0, _y0);
389 cairo_paint_with_alpha (cr, MSD_OSD_WINDOW_FG_ALPHA);
390
391 g_object_unref (pixbuf);
392
393 return TRUE;
394}

396static void
397draw_volume_boxes (MsdMediaKeysWindow *window,
398 cairo_t *cr,
399 double percentage,
400 double _x0,
401 double _y0,
402 double width,
403 double height)
404{
405 gdouble x1;
406 GtkStyleContext *context;
407
408 height = round (height) - 1;
409 width = round (width) - 1;
410 x1 = round ((width - 1) * percentage);
411 context = gtk_widget_get_style_context (GTK_WIDGET (window));
412
413 /* bar background */
414 gtk_style_context_save (context);
415 gtk_style_context_add_class (context, GTK_STYLE_CLASS_TROUGH);
416
417 gtk_render_background (context, cr, _x0, _y0, width, height);
418 gtk_render_frame (context, cr, _x0, _y0, width, height);
419
420 gtk_style_context_restore (context);
421
422 /* bar progress */
423 if (percentage < 0.01)
424 return;
425
426 gtk_style_context_save (context);
427 gtk_style_context_add_class (context, GTK_STYLE_CLASS_PROGRESSBAR);
428
429 gtk_render_background (context, cr, _x0 + 0.5, _y0 + 0.5, x1, height -1 );
430 gtk_render_frame (context, cr, _x0 + 0.5, _y0 + 0.5, x1, height -1 );
431
432 gtk_style_context_restore (context);
433}

435static void
436draw_action_volume (MsdMediaKeysWindow *window,
437 cairo_t *cr)
438{
439 int window_width;
440 int window_height;
441 double icon_box_width;
442 double icon_box_height;
443 double icon_box_x0;
444 double icon_box_y0;
445 double volume_box_x0;
446 double volume_box_y0;
447 double volume_box_width;
448 double volume_box_height;
449 gboolean res;
450
451 gtk_window_get_size (GTK_WINDOW (window), &window_width, &window_height);
452
453 icon_box_width = round (window_width * ICON_SCALE);
454 icon_box_height = round (window_height * ICON_SCALE);
455 volume_box_width = icon_box_width;
456 volume_box_height = round (window_height * 0.05);
457
458 icon_box_x0 = round ((window_width - icon_box_width) / 2);
459 icon_box_y0 = round ((window_height - icon_box_height) / 2);
460 volume_box_x0 = round (icon_box_x0);
461 volume_box_y0 = round (window_height - icon_box_y0 / 2 - volume_box_height);
462
463 res = render_speaker (window,
464 cr,
465 icon_box_x0, icon_box_y0,
466 icon_box_width, icon_box_height);
467 if (! res) {
468 double speaker_width;
469 double speaker_height;
470 double speaker_cx;
471 double speaker_cy;
472
473 speaker_width = icon_box_width * 0.5;
474 speaker_height = icon_box_height * 0.75;
475 speaker_cx = icon_box_x0 + speaker_width / 2;
476 speaker_cy = icon_box_y0 + speaker_height / 2;
477
478#if 0
479 g_message ("speaker box: w=%f h=%f cx=%f cy=%f",
480 speaker_width,
481 speaker_height,
482 speaker_cx,
483 speaker_cy);
484#endif
485
486 /* draw speaker symbol */
487 draw_speaker (cr, speaker_cx, speaker_cy, speaker_width, speaker_height);
488
489 if (! window->priv->volume_muted) {
490 /* draw sound waves */
491 double wave_x0;
492 double wave_y0;
493 double wave_radius;
494
495 wave_x0 = window_width / 2;
496 wave_y0 = speaker_cy;
497 wave_radius = icon_box_width / 2;
498
499 draw_waves (cr, wave_x0, wave_y0, wave_radius, window->priv->volume_level);
500 } else {
501 /* draw 'mute' cross */
502 double cross_x0;
503 double cross_y0;
504 double cross_size;
505
506 cross_size = speaker_width * 3 / 4;
507 cross_x0 = icon_box_x0 + icon_box_width - cross_size;
508 cross_y0 = speaker_cy;
509
510 draw_cross (cr, cross_x0, cross_y0, cross_size);
511 }
512 }
513
514 /* draw volume meter */
515 draw_volume_boxes (window,
516 cr,
517 (double)window->priv->volume_level / 100.0,
518 volume_box_x0,
519 volume_box_y0,
520 volume_box_width,
521 volume_box_height);
522}

524static gboolean
525render_custom (MsdMediaKeysWindow *window,
526 cairo_t *cr,
527 double _x0,
528 double _y0,
529 double width,
530 double height)
531{
