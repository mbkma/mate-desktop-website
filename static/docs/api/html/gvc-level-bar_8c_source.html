83static gboolean
84layout_changed (LevelBarLayout *layout1, LevelBarLayout *layout2)
85{
86 if (layout1->area.x != layout2->area.x)
87 return TRUE;
88 if (layout1->area.y != layout2->area.y)
89 return TRUE;
90 if (layout1->area.width != layout2->area.width)
91 return TRUE;
92 if (layout1->area.height != layout2->area.height)
93 return TRUE;
94
95 if (layout1->delta != layout2->delta)
96 return TRUE;
97 if (layout1->peak_num != layout2->peak_num)
98 return TRUE;
99 if (layout1->max_peak_num != layout2->max_peak_num)
100 return TRUE;
101
102 if (!gdk_rgba_equal (&layout1->color_fg, &layout2->color_fg))
103 return TRUE;
104 if (!gdk_rgba_equal (&layout1->color_bg, &layout2->color_bg))
105 return TRUE;
106 if (!gdk_rgba_equal (&layout1->color_dark, &layout2->color_dark))
107 return TRUE;
108
109 return FALSE;
110}

112static gdouble
113fraction_from_adjustment (GvcLevelBar *bar,
114 GtkAdjustment *adjustment)
115{
116 gdouble level;
117 gdouble fraction = 0.0;
118 gdouble min;
119 gdouble max;
120
121 level = gtk_adjustment_get_value (adjustment);
122 min = gtk_adjustment_get_lower (adjustment);
123 max = gtk_adjustment_get_upper (adjustment);
124
125 switch (bar->priv->scale) {
126 case GVC_LEVEL_SCALE_LINEAR:
127 fraction = (level - min) / (max - min);
128 break;
129 case GVC_LEVEL_SCALE_LOG:
130 fraction = log10 ((level - min + 1) / (max - min + 1));
131 break;
132 }
133
134 return fraction;
135}

137static gboolean
138reset_max_peak (GvcLevelBar *bar)
139{
140 bar->priv->max_peak = gtk_adjustment_get_lower (bar->priv->peak_adjustment);

142 bar->priv->layout.max_peak_num = 0;

144 gtk_widget_queue_draw (GTK_WIDGET (bar));
145
146 bar->priv->max_peak_id = 0;
147 return FALSE;

150static void
151bar_calc_layout (GvcLevelBar *bar)
152{
153 gdouble peak_level;
154 gdouble max_peak_level;
155 GtkAllocation allocation;
156
157 GtkStyleContext *context;
158
159 context = gtk_widget_get_style_context (GTK_WIDGET (bar));
160
161 gtk_style_context_save (context);
162 gtk_style_context_set_state (context, GTK_STATE_FLAG_NORMAL);
163 gtk_style_context_get_background_color (context,
164 gtk_style_context_get_state (context),
165 &bar->priv->layout.color_bg);
166 mate_desktop_gtk_style_get_dark_color (context,
167 gtk_style_context_get_state (context),
168 &bar->priv->layout.color_dark);
169
170 gtk_style_context_set_state (context, GTK_STATE_FLAG_SELECTED);
171 gtk_style_context_get_background_color (context,
172 gtk_style_context_get_state (context),
173 &bar->priv->layout.color_fg);
174 gtk_style_context_restore (context);
175
176 gtk_widget_get_allocation (GTK_WIDGET (bar), &allocation);
177
178 bar->priv->layout.area.width = allocation.width - 2;
179 bar->priv->layout.area.height = allocation.height - 2;
180
181 if (bar->priv->orientation == GTK_ORIENTATION_VERTICAL) {
182 peak_level = bar->priv->peak_fraction * (gdouble) bar->priv->layout.area.height;
183 max_peak_level = bar->priv->max_peak * (gdouble) bar->priv->layout.area.height;
184
185 bar->priv->layout.delta = bar->priv->layout.area.height / NUM_BOXES;
186 bar->priv->layout.area.x = 0;
187 bar->priv->layout.area.y = 0;
188 bar->priv->layout.box_height = bar->priv->layout.delta / 2;
189 bar->priv->layout.box_width = bar->priv->layout.area.width;
190 bar->priv->layout.box_radius = bar->priv->layout.box_width / 2;
191 } else {
192 peak_level = bar->priv->peak_fraction * (gdouble) bar->priv->layout.area.width;
193 max_peak_level = bar->priv->max_peak * (gdouble) bar->priv->layout.area.width;
194
195 bar->priv->layout.delta = bar->priv->layout.area.width / NUM_BOXES;
196 bar->priv->layout.area.x = 0;
197 bar->priv->layout.area.y = 0;
198 bar->priv->layout.box_width = bar->priv->layout.delta / 2;
199 bar->priv->layout.box_height = bar->priv->layout.area.height;
200 bar->priv->layout.box_radius = bar->priv->layout.box_height / 2;
201 }
202
203 bar->priv->layout.peak_num = (int) (peak_level / (gdouble) bar->priv->layout.delta);
204 bar->priv->layout.max_peak_num = (int) (max_peak_level / (gdouble) bar->priv->layout.delta);
205}

207static void
208update_peak_value (GvcLevelBar *bar)
209{
210 gdouble value;
211 LevelBarLayout layout;
212
213 value = fraction_from_adjustment (bar, bar->priv->peak_adjustment);
214
215 bar->priv->peak_fraction = value;
216
217 if (value > bar->priv->max_peak) {
218 AtkObject *acc_obj;
219 char *acc_string;
220 gdouble peak_percentage;
221
222 if (bar->priv->max_peak_id > 0)
223 g_source_remove (bar->priv->max_peak_id);
224
225 bar->priv->max_peak_id =
226 g_timeout_add_seconds (1, (GSourceFunc) reset_max_peak, bar);
227 bar->priv->max_peak = value;
228 /* Providing an accessible based on peaks, so that
229 * it doesnâ€™t get updated too often. */
230 acc_obj = gtk_widget_get_accessible (GTK_WIDGET (bar));
231 peak_percentage = round (value * 100.0);
232 acc_string = g_strdup_printf (_("Peak: %.0lf"), peak_percentage);
233 atk_object_set_name (acc_obj, acc_string);
234 g_free (acc_string);
235 }
236
237 layout = bar->priv->layout;
238
239 bar_calc_layout (bar);
240
241 if (layout_changed (&bar->priv->layout, &layout))
242 gtk_widget_queue_draw (GTK_WIDGET (bar));
243}

294void
295gvc_level_bar_set_peak_adjustment (GvcLevelBar *bar,
296 GtkAdjustment *adjustment)
297{
298 g_return_if_fail (GVC_IS_LEVEL_BAR (bar));
299 g_return_if_fail (GTK_IS_ADJUSTMENT (adjustment));
300
301 if (bar->priv->peak_adjustment != NULL) {
302 g_signal_handlers_disconnect_by_func (G_OBJECT (bar->priv->peak_adjustment),
303 G_CALLBACK (on_peak_adjustment_value_changed),
304 bar);
305 g_object_unref (bar->priv->peak_adjustment);
306 }
307
308 bar->priv->peak_adjustment = g_object_ref_sink (adjustment);
309
310 g_signal_connect (G_OBJECT (bar->priv->peak_adjustment),
311 "value-changed",
312 G_CALLBACK (on_peak_adjustment_value_changed),
313 bar);
314
315 update_peak_value (bar);
316
317 g_object_notify_by_pspec (G_OBJECT (bar), properties[PROP_PEAK_ADJUSTMENT]);
318}

513static void
514curved_rectangle (cairo_t *cr,
515 double x0,
516 double y0,
517 double width,
518 double height,
519 double radius)
520{
521 double x1;
522 double y1;
523
524 x1 = x0 + width;
525 y1 = y0 + height;
526
527 if (width == 0.0 || height == 0.0)
528 return;
529
530 if (width / 2 < radius) {
531 if (height / 2 < radius) {
532 cairo_move_to (cr, x0, (y0 + y1) / 2);
533 cairo_curve_to (cr, x0 ,y0, x0, y0, (x0 + x1) / 2, y0);
534 cairo_curve_to (cr, x1, y0, x1, y0, x1, (y0 + y1) / 2);
535 cairo_curve_to (cr, x1, y1, x1, y1, (x1 + x0) / 2, y1);
536 cairo_curve_to (cr, x0, y1, x0, y1, x0, (y0 + y1) / 2);
537 } else {
538 cairo_move_to (cr, x0, y0 + radius);
539 cairo_curve_to (cr, x0, y0, x0, y0, (x0 + x1) / 2, y0);
540 cairo_curve_to (cr, x1, y0, x1, y0, x1, y0 + radius);
541 cairo_line_to (cr, x1, y1 - radius);
542 cairo_curve_to (cr, x1, y1, x1, y1, (x1 + x0) / 2, y1);
543 cairo_curve_to (cr, x0, y1, x0, y1, x0, y1 - radius);
544 }
545 } else {
546 if (height / 2 < radius) {
547 cairo_move_to (cr, x0, (y0 + y1) / 2);
548 cairo_curve_to (cr, x0, y0, x0 , y0, x0 + radius, y0);
549 cairo_line_to (cr, x1 - radius, y0);
550 cairo_curve_to (cr, x1, y0, x1, y0, x1, (y0 + y1) / 2);
551 cairo_curve_to (cr, x1, y1, x1, y1, x1 - radius, y1);
552 cairo_line_to (cr, x0 + radius, y1);
553 cairo_curve_to (cr, x0, y1, x0, y1, x0, (y0 + y1) / 2);
554 } else {
555 cairo_move_to (cr, x0, y0 + radius);
556 cairo_curve_to (cr, x0 , y0, x0 , y0, x0 + radius, y0);
557 cairo_line_to (cr, x1 - radius, y0);
558 cairo_curve_to (cr, x1, y0, x1, y0, x1, y0 + radius);
559 cairo_line_to (cr, x1, y1 - radius);
560 cairo_curve_to (cr, x1, y1, x1, y1, x1 - radius, y1);
561 cairo_line_to (cr, x0 + radius, y1);
562 cairo_curve_to (cr, x0, y1, x0, y1, x0, y1 - radius);
563 }
564 }
565
566 cairo_close_path (cr);
567}

569static int
570gvc_level_bar_draw (GtkWidget *widget, cairo_t *cr)
571{
572 GvcLevelBar *bar;
573
574 bar = GVC_LEVEL_BAR (widget);
575
576 cairo_save (cr);
577
578 if (bar->priv->orientation == GTK_ORIENTATION_VERTICAL) {
579 int i;
580
581 for (i = 0; i < NUM_BOXES; i++) {
582 int by;
583
584 by = i * bar->priv->layout.delta;
585 curved_rectangle (cr,
586 bar->priv->layout.area.x + 0.5,
587 by + 0.5,
588 bar->priv->layout.box_width - 1,
589 bar->priv->layout.box_height - 1,
590 bar->priv->layout.box_radius);
591 if ((bar->priv->layout.max_peak_num - 1) == i) {
592 /* fill peak foreground */
593 gdk_cairo_set_source_rgba (cr, &bar->priv->layout.color_fg);
594 cairo_fill_preserve (cr);
595 } else if ((bar->priv->layout.peak_num - 1) >= i) {
596 /* fill background */
597 gdk_cairo_set_source_rgba (cr, &bar->priv->layout.color_bg);
598 cairo_fill_preserve (cr);
599
600 /* fill foreground */
601 cairo_set_source_rgba (cr,
602 bar->priv->layout.color_fg.red,
603 bar->priv->layout.color_fg.green,
604 bar->priv->layout.color_fg.blue,
605 0.5);
606 cairo_fill_preserve (cr);
607 } else {
608 /* fill background */
609 gdk_cairo_set_source_rgba (cr, &bar->priv->layout.color_bg);
610 cairo_fill_preserve (cr);
611 }
612
613 /* stroke border */
614 gdk_cairo_set_source_rgba (cr, &bar->priv->layout.color_dark);
615 cairo_set_line_width (cr, 1);
616 cairo_stroke (cr);
617 }
618 } else {
619 int i;
620
621 if (gtk_widget_get_direction (widget) == GTK_TEXT_DIR_RTL) {
622 GtkAllocation allocation;
623
624 gtk_widget_get_allocation (widget, &allocation);
625
626 cairo_scale (cr, -1, 1);
627 cairo_translate (cr, -allocation.width, 0);
628 }
629
630 for (i = 0; i < NUM_BOXES; i++) {
631 int bx;
632
633 bx = i * bar->priv->layout.delta;
634 curved_rectangle (cr,
635 bx + 0.5,
636 bar->priv->layout.area.y + 0.5,
637 bar->priv->layout.box_width - 1,
638 bar->priv->layout.box_height - 1,
639 bar->priv->layout.box_radius);
640
641 if ((bar->priv->layout.max_peak_num - 1) == i) {
642 /* fill peak foreground */
643 gdk_cairo_set_source_rgba (cr, &bar->priv->layout.color_fg);
644 cairo_fill_preserve (cr);
645 } else if ((bar->priv->layout.peak_num - 1) >= i) {
646 /* fill background */
647 gdk_cairo_set_source_rgba (cr, &bar->priv->layout.color_bg);
648 cairo_fill_preserve (cr);
649
650 /* fill foreground */
651 cairo_set_source_rgba (cr,
652 bar->priv->layout.color_fg.red,
653 bar->priv->layout.color_fg.green,
654 bar->priv->layout.color_fg.blue,
655 0.5);
656 cairo_fill_preserve (cr);
657 } else {
658 /* fill background */
659 gdk_cairo_set_source_rgba (cr, &bar->priv->layout.color_bg);
660 cairo_fill_preserve (cr);
661 }
662
663 /* stroke border */
664 gdk_cairo_set_source_rgba (cr, &bar->priv->layout.color_dark);
665 cairo_set_line_width (cr, 1);
666 cairo_stroke (cr);
667 }
668 }
669
670 cairo_restore (cr);
671
672 return FALSE;
673}

675static void
676gvc_level_bar_class_init (GvcLevelBarClass *klass)
