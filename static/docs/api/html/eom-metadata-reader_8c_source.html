35EomMetadataReader*
36eom_metadata_reader_new (EomMetadataFileType type)
37{
38 EomMetadataReader *emr;
39
40 switch (type) {
41 case EOM_METADATA_JPEG:
42 emr = EOM_METADATA_READER (g_object_new (EOM_TYPE_METADATA_READER_JPG, NULL));
43 break;
44 case EOM_METADATA_PNG:
45 emr = EOM_METADATA_READER (g_object_new (EOM_TYPE_METADATA_READER_PNG, NULL));
46 break;
47 default:
48 emr = NULL;
49 break;
50 }
51
52 return emr;
53}

104/* Default vfunc that simply clears the output if not overriden by the
105 implementing class. This mimics the old behavour of get_exif_chunk(). */
106static void
107_eom_metadata_reader_default_get_raw_exif (EomMetadataReader *emr,
108 guchar **data, guint *length)
109{
110 g_return_if_fail (data != NULL && length != NULL);
111 *data = NULL;
112 *length = 0;
