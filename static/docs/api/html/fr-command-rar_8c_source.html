67static const gchar *
68get_rar_command (void)
69{
70 for (guint i = 0; i < G_N_ELEMENTS(candidate_commands); i++) {
71 if (is_program_in_path (candidate_commands[i].command))
72 return candidate_commands[i].command;
73 }
74
75 /* this should never happen as we shouldn't get called if none of the
76 * candidate exist, but better safe than sorry */
77 return "unrar";
78}

133static time_t
134mktime_from_string (const char *date_s,
135 const char *time_s)
136{
137 struct tm tm = {0, };
138 char *date_time_s;
139
140 tm.tm_isdst = -1;
141 date_time_s = g_strjoin (" ", date_s, time_s, NULL);
142 strptime (date_time_s, date_newstyle ? "%Y-%m-%d %H:%M" : "%d-%m-%y %H:%M", &tm);
143 g_free (date_time_s);
144
145 return mktime (&tm);
146}

148static gboolean
149attribute_field_with_space (const char *line)
150{
151 /* sometimes when the archive is encrypted the attributes field is
152 * like this: "* ..A...."
153 * */
154 return ((line[0] != ' ') && (line[1] == ' '));
155}

157static gboolean
158attr_field_is_dir (const char *attr_field,
159 FrCommandRar *rar_comm)
160{
161 if ((attr_field[0] == 'd') ||
162 (IS_OUTPUT_TYPE (rar_comm, FR_COMMAND_RAR_TYPE_RAR5) && attr_field[3] == 'D') ||
163 (! IS_OUTPUT_TYPE (rar_comm, FR_COMMAND_RAR_TYPE_RAR5) && attr_field[1] == 'D'))
164 return TRUE;
165
166 return FALSE;
167}

169static void
170parse_name_field (char *line,
171 FrCommandRar *rar_comm)
172{
173 char *name_field;
174 FileData *fdata;
175
176 rar_comm->fdata = fdata = file_data_new ();
177
178 /* read file name. */
179
180 fdata->encrypted = (line[0] == '*') ? TRUE : FALSE;
181
182 if (IS_OUTPUT_TYPE (rar_comm, FR_COMMAND_RAR_TYPE_RAR5)) {
183 int field_num = attribute_field_with_space (line) ? 9 : 8;
184 const char *field;
185
186 /* in RAR7 format, directories have a blank CRC32, not the
187 * "????????" of the RAR5 format */
188 if (IS_OUTPUT_TYPE (rar_comm, FR_COMMAND_RAR_TYPE_RAR7)) {
189 field = get_last_field (line, field_num - 8);
190 if (attr_field_is_dir (field, rar_comm))
191 field_num--;
192 }
193
194 field = get_last_field (line, field_num);
195 /* rar-5 output adds trailing spaces to short file names :( */
196 name_field = field ? g_strchomp (g_strdup (field)) : NULL;
197 } else
198 name_field = g_strdup (line + 1);
199
200 if (name_field == NULL)
201 return;
202
203 if (*name_field == '/') {
204 fdata->full_path = g_strdup (name_field);
205 fdata->original_path = fdata->full_path;
206 }
207 else {
208 fdata->full_path = g_strconcat ("/", name_field, NULL);
209 fdata->original_path = fdata->full_path + 1;
210 }
211
212 fdata->link = NULL;
213 fdata->path = remove_level_from_path (fdata->full_path);
214
215 g_free (name_field);
216}

218static void
219process_line (char *line,
220 gpointer data)
221{
222 FrCommand *comm = FR_COMMAND (data);
223 FrCommandRar *rar_comm = FR_COMMAND_RAR (comm);
224 char **fields;
225
226 g_return_if_fail (line != NULL);
227
228 if (! rar_comm->list_started) {
229 int version = 0;
230
231 if (sscanf (line, "RAR %d.", &version) == 1 || sscanf (line, "UNRAR %d.", &version) == 1) {
232 if (version >= 7)
233 rar_comm->output_type = FR_COMMAND_RAR_TYPE_RAR7;
234 else if (version >= 5)
235 rar_comm->output_type = FR_COMMAND_RAR_TYPE_RAR5;
236 else
237 rar_comm->output_type = FR_COMMAND_RAR_TYPE_RAR4;
238
239 if (version > 5)
240 date_newstyle = TRUE;
241 else if (version == 5 && (sscanf (line, "RAR 5.%d ", &version) == 1 ||
242 sscanf (line, "UNRAR 5.%d ", &version) == 1))
243 {
244 if (version >= 30)
245 date_newstyle = TRUE;
246 }
247 }
248 else if (g_str_has_prefix (line, "unrar-free ")) {
249 rar_comm->output_type = FR_COMMAND_RAR_TYPE_UNRAR_FREE;
250 date_newstyle = FALSE;
251 }
252 else if (strncmp (line, "--------", 8) == 0) {
253 rar_comm->list_started = TRUE;
254 if (! IS_OUTPUT_TYPE (rar_comm, FR_COMMAND_RAR_TYPE_RAR5))
255 rar_comm->rar4_odd_line = TRUE;
256 }
257 else if (strncmp (line, "Volume ", 7) == 0)
258 comm->multi_volume = TRUE;
259 return;
260 }
261
262 if (strncmp (line, "--------", 8) == 0) {
263 rar_comm->list_started = FALSE;
264 return;
265 }
266
267 if (rar_comm->rar4_odd_line || IS_OUTPUT_TYPE (rar_comm, FR_COMMAND_RAR_TYPE_RAR5))
268 parse_name_field (line, rar_comm);
269
270 if (! rar_comm->rar4_odd_line && rar_comm->fdata && rar_comm->fdata->full_path) {
271 FileData *fdata;
272 const char *size_field, *ratio_field, *date_field, *time_field, *attr_field;
273 int n_fields;
274
275 if (IS_OUTPUT_TYPE (rar_comm, FR_COMMAND_RAR_TYPE_RAR5))
276 n_fields = 6 + (attribute_field_with_space (line) != 0);
277 else if (IS_OUTPUT_TYPE (rar_comm, FR_COMMAND_RAR_TYPE_UNRAR_FREE))
278 n_fields = 4;
279 else
280 n_fields = 6;
281
282 fdata = rar_comm->fdata;
283
284 /* read file info. */
285
286 fields = split_line (line, n_fields);
287 if (g_strv_length (fields) < (guint) n_fields) {
288 /* wrong line format, treat this line as a filename line */
289 g_strfreev (fields);
290 file_data_free (rar_comm->fdata);
291 rar_comm->fdata = NULL;
292 rar_comm->rar4_odd_line = TRUE;
293 parse_name_field (line, rar_comm);
294 }
295 else {
296 if (IS_OUTPUT_TYPE (rar_comm, FR_COMMAND_RAR_TYPE_RAR5)) {
297 int offset = attribute_field_with_space (line) ? 1 : 0;
298
299 size_field = fields[1+offset];
300 ratio_field = fields[3+offset];
301 date_field = fields[4+offset];
302 time_field = fields[5+offset];
303 attr_field = fields[0+offset];
304 }
305 else if (IS_OUTPUT_TYPE (rar_comm, FR_COMMAND_RAR_TYPE_UNRAR_FREE)) {
306 size_field = fields[0];
307 date_field = fields[1];
308 time_field = fields[2];
309 attr_field = fields[3];
310 ratio_field = "";
311 }
312 else {
313 size_field = fields[0];
314 ratio_field = fields[2];
315 date_field = fields[3];
316 time_field = fields[4];
317 attr_field = fields[5];
318 }
319
320 if ((strcmp (ratio_field, "<->") == 0)
321 || (strcmp (ratio_field, "<--") == 0))
322 {
323 /* ignore files that span more volumes */
324
325 file_data_free (rar_comm->fdata);
326 rar_comm->fdata = NULL;
327 }
328 else {
329 fdata->size = g_ascii_strtoull (size_field, NULL, 10);
330 fdata->modified = mktime_from_string (date_field, time_field);
331
332 if (attr_field_is_dir (attr_field, rar_comm)) {
333 char *tmp;
334
335 tmp = fdata->full_path;
336 fdata->full_path = g_strconcat (fdata->full_path, "/", NULL);
337
338 fdata->original_path = g_strdup (fdata->original_path);
339 fdata->free_original_path = TRUE;
340
341 g_free (tmp);
342
343 fdata->name = dir_name_from_path (fdata->full_path);
344 fdata->dir = TRUE;
345 }
346 else {
347 fdata->name = g_strdup (file_name_from_path (fdata->full_path));
348 if (attr_field[0] == 'l')
349 fdata->link = g_strdup (file_name_from_path (fdata->full_path));
350 }
351
352 fr_command_add_file (comm, fdata);
353 rar_comm->fdata = NULL;
354 }
355
356 g_strfreev (fields);
357 }
358 }
359
360 if (! IS_OUTPUT_TYPE (rar_comm, FR_COMMAND_RAR_TYPE_RAR5))
361 rar_comm->rar4_odd_line = ! rar_comm->rar4_odd_line;
362}

364static void
365add_password_arg (FrCommand *comm,
366 const char *password,
367 gboolean disable_query)
368{
369 if ((password != NULL) && (password[0] != '\0')) {
370 if (comm->encrypt_header)
371 fr_process_add_arg_concat (comm->process, "-hp", password, NULL);
372 else
373 fr_process_add_arg_concat (comm->process, "-p", password, NULL);
374 }
375 else if (disable_query)
376 fr_process_add_arg (comm->process, "-p-");
377}

421static void
422process_line__add (char *line,
423 gpointer data)
424{
425 FrCommand *comm = FR_COMMAND (data);
426
427 if (strncmp (line, "Creating archive ", 17) == 0) {
428 const char *archive_filename = line + 17;
429 char *uri;
430
431 uri = g_filename_to_uri (archive_filename, NULL, NULL);
432 if ((comm->volume_size > 0)
433 && g_regex_match_simple ("^.*\\.part(0)*2\\.rar$", uri, G_REGEX_CASELESS, 0))
434 {
435 char *volume_filename;
436 GFile *volume_file;
437
438 volume_filename = g_strdup (archive_filename);
439 volume_filename[strlen (volume_filename) - 5] = '1';
440 volume_file = g_file_new_for_path (volume_filename);
441 fr_command_set_multi_volume (comm, volume_file);
442
443 g_object_unref (volume_file);
444 g_free (volume_filename);
445 }
446 fr_command_working_archive (comm, uri);
447
448 g_free (uri);
449 return;
450 }
451
452 if (comm->n_files != 0)
453 parse_progress_line (comm, "Adding ", _("Adding file: "), line);
454}

456static void
457fr_command_rar_add (FrCommand *comm,
458 const char *from_file,
459 GList *file_list,
460 const char *base_dir,
461 gboolean update,
462 gboolean recursive)
463{
464 GList *scan;
465
466 fr_process_use_standard_locale (comm->process, TRUE);
467 fr_process_set_out_line_func (comm->process,
468 process_line__add,
469 comm);
470
471 fr_process_begin_command (comm->process, "rar");
472
473 if (base_dir != NULL)
474 fr_process_set_working_dir (comm->process, base_dir);
475
476 if (update)
477 fr_process_add_arg (comm->process, "u");
478 else
479 fr_process_add_arg (comm->process, "a");
480
481 switch (comm->compression) {
482 case FR_COMPRESSION_VERY_FAST:
483 fr_process_add_arg (comm->process, "-m1"); break;
484 case FR_COMPRESSION_FAST:
485 fr_process_add_arg (comm->process, "-m2"); break;
486 case FR_COMPRESSION_NORMAL:
487 fr_process_add_arg (comm->process, "-m3"); break;
488 case FR_COMPRESSION_MAXIMUM:
489 fr_process_add_arg (comm->process, "-m5"); break;
490 }
491
492 add_password_arg (comm, comm->password, FALSE);
493
494 if (comm->volume_size > 0)
495 fr_process_add_arg_printf (comm->process, "-v%ub", comm->volume_size);
496
497 /* disable percentage indicator */
498 fr_process_add_arg (comm->process, "-Idp");
499
500 fr_process_add_arg (comm->process, "--");
501 fr_process_add_arg (comm->process, comm->filename);
502
503 if (from_file == NULL)
504 for (scan = file_list; scan; scan = scan->next)
505 fr_process_add_arg (comm->process, scan->data);
506 else
507 fr_process_add_arg_concat (comm->process, "@", from_file, NULL);
508
509 fr_process_end_command (comm->process);
510}

652static void
653fr_command_rar_handle_error (FrCommand *comm,
654 FrProcError *error)
655{
656 GList *scan;
657
658#if 0
659 {
660 GList *scan;
661
662 for (scan = g_list_last (comm->process->err.raw); scan; scan = scan->prev)
663 g_print ("%s\n", (char*)scan->data);
664 }
665#endif
666
667 if (error->type == FR_PROC_ERROR_NONE)
668 return;
669
670 /*if (error->status == 3)
671 error->type = FR_PROC_ERROR_ASK_PASSWORD;
672 else */
673 if (error->status <= 1)
674 error->type = FR_PROC_ERROR_NONE;
675
676 for (scan = g_list_last (comm->process->err.raw); scan; scan = scan->prev) {
677 char *line = scan->data;
678
679 if ((strstr (line, "Incorrect password") != NULL) ||
680 (strstr (line, "password incorrect") != NULL) ||
681 (strstr (line, "password is incorrect") != NULL) ||
682 (strstr (line, "wrong password") != NULL)) {
683 error->type = FR_PROC_ERROR_ASK_PASSWORD;
684 break;
685 }
686
687 if (strncmp (line, "Unexpected end of archive", 25) == 0) {
688 /* FIXME: handle this type of errors at a higher level when the freeze is over. */
689 }
690
691 if (strncmp (line, "Cannot find volume", 18) == 0) {
692 char *volume_filename;
693
694 g_clear_error (&error->gerror);
695
696 error->type = FR_PROC_ERROR_MISSING_VOLUME;
697 volume_filename = g_path_get_basename (line + strlen ("Cannot find volume "));
698 error->gerror = g_error_new (FR_ERROR, error->status, _("Could not find the volume: %s"), volume_filename);
699 g_free (volume_filename);
700
701 /* RAR7 complains about missing volume for incorrect passwords
702 * as well, so don't make this a definite error in case we also
703 * had an incorrect password one earlier */
704 if (! IS_OUTPUT_TYPE (FR_COMMAND_RAR (comm), FR_COMMAND_RAR_TYPE_RAR7))
705 break;
706 }
707 }

725 FrCommandCaps capabilities;
726
727 capabilities = FR_COMMAND_CAN_ARCHIVE_MANY_FILES | FR_COMMAND_CAN_ENCRYPT | FR_COMMAND_CAN_ENCRYPT_HEADER;
728 /* add command-specific capabilities depending on availability */
729 for (guint i = 0; i < G_N_ELEMENTS (candidate_commands); i++) {
730 if (is_program_available (candidate_commands[i].command, check_command)) {
731 capabilities |= candidate_commands[i].capabilities;
732 break;
733 }
734 }

737 if ((comm->files->len > 0) && comm->multi_volume && (capabilities & FR_COMMAND_CAN_WRITE))
738 capabilities ^= FR_COMMAND_CAN_WRITE;
739
740 return capabilities;
741}
