196static gboolean
197fm_list_model_get_iter (GtkTreeModel *tree_model, GtkTreeIter *iter, GtkTreePath *path)
198{
199 FMListModel *model;
200 GSequence *files;
201 GSequenceIter *ptr;
202 FileEntry *file_entry;
203 int d;
204
205 model = (FMListModel *)tree_model;
206 ptr = NULL;
207
208 files = model->details->files;
209 for (d = 0; d < gtk_tree_path_get_depth (path); d++)
210 {
211 int i;
212
213 i = gtk_tree_path_get_indices (path)[d];
214
215 if (files == NULL || i >= g_sequence_get_length (files))
216 {
217 return FALSE;
218 }
219
220 ptr = g_sequence_get_iter_at_pos (files, i);
221 file_entry = g_sequence_get (ptr);
222 files = file_entry->files;
223 }
224
225 fm_list_model_ptr_to_iter (model, ptr, iter);
226
227 return TRUE;
228}

282static void
283fm_list_model_get_value (GtkTreeModel *tree_model, GtkTreeIter *iter, int column, GValue *value)
284{
285 FMListModel *model;
286 FileEntry *file_entry;
287 CajaFile *file;
288 GdkPixbuf *rendered_icon;
289 GIcon *emblemed_icon;
290 GList *emblem_icons, *l;
291 CajaZoomLevel zoom_level;
292 CajaFileIconFlags flags;
293
294 model = (FMListModel *)tree_model;
295
296 g_return_if_fail (model->details->stamp == iter->stamp);
297 g_return_if_fail (!g_sequence_iter_is_end (iter->user_data));
298
299 file_entry = g_sequence_get (iter->user_data);
300 file = file_entry->file;
301
302 switch (column)
303 {
304 case FM_LIST_MODEL_FILE_COLUMN:
305 g_value_init (value, CAJA_TYPE_FILE);
306
307 g_value_set_object (value, file);
308 break;
309 case FM_LIST_MODEL_SUBDIRECTORY_COLUMN:
310 g_value_init (value, CAJA_TYPE_DIRECTORY);
311
312 g_value_set_object (value, file_entry->subdirectory);
313 break;
314 case FM_LIST_MODEL_SMALLEST_ICON_COLUMN:
315 case FM_LIST_MODEL_SMALLER_ICON_COLUMN:
316 case FM_LIST_MODEL_SMALL_ICON_COLUMN:
317 case FM_LIST_MODEL_STANDARD_ICON_COLUMN:
318 case FM_LIST_MODEL_LARGE_ICON_COLUMN:
319 case FM_LIST_MODEL_LARGER_ICON_COLUMN:
320 case FM_LIST_MODEL_LARGEST_ICON_COLUMN:
321 if (!g_settings_get_boolean (caja_preferences, CAJA_PREFERENCES_SHOW_ICONS_IN_LIST_VIEW)) {
322 cairo_surface_t *surface;
323 int icon_size;
324
325 g_value_init (value, CAIRO_GOBJECT_TYPE_SURFACE);
326
327 zoom_level = fm_list_model_get_zoom_level_from_column_id (column);
328 icon_size = caja_get_icon_size_for_zoom_level (zoom_level);
329
330 surface = cairo_image_surface_create (CAIRO_FORMAT_ARGB32, icon_size, icon_size);
331 g_value_take_boxed (value, surface);
332 break;
333 }
334 g_value_init (value, CAIRO_GOBJECT_TYPE_SURFACE);
335
336 if (file != NULL)
337 {
338 GdkPixbuf *icon;
339 GIcon *gicon;
340 CajaIconInfo *icon_info;
341 GEmblem *emblem;
342 int icon_size, icon_scale;
343 CajaFile *parent_file;
344 char *emblems_to_ignore[3];
345 int i;
346 cairo_surface_t *surface;
347 const char *icon_name;
348
349 zoom_level = fm_list_model_get_zoom_level_from_column_id (column);
350 icon_size = caja_get_icon_size_for_zoom_level (zoom_level);
351 icon_scale = fm_list_model_get_icon_scale (model);
352
353 flags = CAJA_FILE_ICON_FLAGS_USE_THUMBNAILS |
354 CAJA_FILE_ICON_FLAGS_FORCE_THUMBNAIL_SIZE |
355 CAJA_FILE_ICON_FLAGS_USE_MOUNT_ICON_AS_EMBLEM;
356 if (model->details->drag_view != NULL)
357 {
358 GtkTreePath *path_a;
359
360 gtk_tree_view_get_drag_dest_row (model->details->drag_view,
361 &path_a,
362 NULL);
363 if (path_a != NULL)
364 {
365 GtkTreePath *path_b;
366
367 path_b = gtk_tree_model_get_path (tree_model, iter);
368
369 if (gtk_tree_path_compare (path_a, path_b) == 0)
370 {
371 flags |= CAJA_FILE_ICON_FLAGS_FOR_DRAG_ACCEPT;
372 }
373
374 gtk_tree_path_free (path_a);
375 gtk_tree_path_free (path_b);
376 }
377 }
378
379 gicon = caja_file_get_gicon (file, flags);
380
381 /* render emblems with GEmblemedIcon */
382 parent_file = caja_file_get_parent (file);
383 i = 0;
384 emblems_to_ignore[i++] = CAJA_FILE_EMBLEM_NAME_TRASH;
385 if (parent_file) {
386 if (!caja_file_can_write (parent_file)) {
387 emblems_to_ignore[i++] = CAJA_FILE_EMBLEM_NAME_CANT_WRITE;
388 }
389 caja_file_unref (parent_file);
390 }
391 emblems_to_ignore[i++] = NULL;
392
393 emblem = NULL;
394 emblem_icons = caja_file_get_emblem_icons (file,
395 emblems_to_ignore);
396
397 if (emblem_icons != NULL) {
398 GIcon *emblem_icon;
399
400 emblem_icon = emblem_icons->data;
401 emblem = g_emblem_new (emblem_icon);
402 emblemed_icon = g_emblemed_icon_new (gicon, emblem);
403
404 g_object_unref (emblem);
405
406 for (l = emblem_icons->next; l != NULL; l = l->next) {
407 emblem_icon = l->data;
408 emblem = g_emblem_new (emblem_icon);
409 g_emblemed_icon_add_emblem
410 (G_EMBLEMED_ICON (emblemed_icon), emblem);
411
412 g_object_unref (emblem);
413 }
414
415 g_list_free_full (emblem_icons, g_object_unref);
416
417 g_object_unref (gicon);
418 gicon = emblemed_icon;
419 }
420
421 icon_info = caja_file_get_icon (file, icon_size, icon_scale, flags);
422 icon_name = caja_icon_info_get_used_name (icon_info);
423
424 if (icon_name != NULL) {
425 g_object_unref (icon_info);
426 icon_info = caja_icon_info_lookup (gicon, icon_size, icon_scale);
427 }
428 icon = caja_icon_info_get_pixbuf_at_size (icon_info, icon_size);
429
430 g_object_unref (icon_info);
431 g_object_unref (gicon);
432
433 if (model->details->highlight_files != NULL &&
434 g_list_find_custom (model->details->highlight_files,
435 file, (GCompareFunc) caja_file_compare_location))
436 {
437 rendered_icon = eel_create_spotlight_pixbuf (icon);
438
439 if (rendered_icon != NULL)
440 {
441 g_object_unref (icon);
442 icon = rendered_icon;
443 }
444 }
445
446 surface = gdk_cairo_surface_create_from_pixbuf (icon, icon_scale, NULL);
447 g_value_take_boxed (value, surface);
448 g_object_unref (icon);
449 }
450 break;
451 case FM_LIST_MODEL_FILE_NAME_IS_EDITABLE_COLUMN:
452 g_value_init (value, G_TYPE_BOOLEAN);
453
454 g_value_set_boolean (value, file != NULL && caja_file_can_rename (file));
455 break;
456 default:
457 if (column >= FM_LIST_MODEL_NUM_COLUMNS || column < FM_LIST_MODEL_NUM_COLUMNS + model->details->columns->len)
458 {
459 CajaColumn *caja_column;
460 GQuark attribute;
461 caja_column = model->details->columns->pdata[column - FM_LIST_MODEL_NUM_COLUMNS];
462
463 g_value_init (value, G_TYPE_STRING);
464 g_object_get (caja_column,
465 "attribute_q", &attribute,
466 NULL);
467 if (file != NULL)
468 {
469 char *str;
470
471 str = caja_file_get_string_attribute_with_default_q (file,
472 attribute);
473 g_value_take_string (value, str);
474 }
475 else if (attribute == attribute_name_q)
476 {
477 if (file_entry->parent->loaded)
478 {
479 g_value_set_string (value, _("(Empty)"));
480 }
481 else
482 {
483 g_value_set_string (value, _("Loading..."));
484 }
485 }
486 }
487 else
488 {
489 g_assert_not_reached ();
490 }
491 }
492}

629static GSequenceIter *
630lookup_file (FMListModel *model, CajaFile *file,
631 CajaDirectory *directory)
632{
633 GSequenceIter *ptr, *parent_ptr;
634
635 parent_ptr = NULL;
636 if (directory)
637 {
638 parent_ptr = g_hash_table_lookup (model->details->directory_reverse_map,
639 directory);
640 }
641
642 if (parent_ptr)
643 {
644 FileEntry *file_entry;
645
646 file_entry = g_sequence_get (parent_ptr);
647 ptr = g_hash_table_lookup (file_entry->reverse_map, file);
648 }
649 else
650 {
651 ptr = g_hash_table_lookup (model->details->top_reverse_map, file);
652 }
653
654 if (ptr)
655 {
656 g_assert (((FileEntry *)g_sequence_get (ptr))->file == file);
657 }
658
659 return ptr;
660}

753static int
754fm_list_model_file_entry_compare_func (gconstpointer a,
755 gconstpointer b,
756 gpointer user_data)
757{
758 FileEntry *file_entry1;
759 FileEntry *file_entry2;
760 FMListModel *model;
761 int result;
762
763 model = (FMListModel *)user_data;
764
765 file_entry1 = (FileEntry *)a;

802static void
803fm_list_model_sort_file_entries (FMListModel *model, GSequence *files, GtkTreePath *path)
804{
805 GSequenceIter **old_order;
806 GtkTreeIter iter;
807 int *new_order;
808 int length;
809 int i;
810 gboolean has_iter;
811 FileEntry *file_entry = NULL;
812
813 length = g_sequence_get_length (files);
814
815 if (length <= 1)
816 {
817 return;
818 }
819
820 /* generate old order of GSequenceIter's */
821 old_order = g_new (GSequenceIter *, length);
822 for (i = 0; i < length; ++i)
823 {
824 GSequenceIter *ptr = g_sequence_get_iter_at_pos (files, i);
825
826 file_entry = g_sequence_get (ptr);
827
828 if (file_entry->files != NULL)
829 {
830 gtk_tree_path_append_index (path, i);
831 fm_list_model_sort_file_entries (model, file_entry->files, path);
832 gtk_tree_path_up (path);
833 }
834
835 old_order[i] = ptr;
836 }
837
838 /* sort */
839 g_sequence_sort (files, fm_list_model_file_entry_compare_func, model);
840
841 /* generate new order */
842 new_order = g_new (int, length);
843 /* Note: new_order[newpos] = oldpos */
844 for (i = 0; i < length; ++i)
845 {
846 new_order[g_sequence_iter_get_position (old_order[i])] = i;
847 }
848
849 /* Let the world know about our new order */
850
851 g_assert (new_order != NULL);
852
853 has_iter = FALSE;
854 if (gtk_tree_path_get_depth (path) != 0)
855 {
856 gboolean get_iter_result;
857 has_iter = TRUE;
858 get_iter_result = gtk_tree_model_get_iter (GTK_TREE_MODEL (model), &iter, path);
859 g_assert (get_iter_result);
860 }
861
862 gtk_tree_model_rows_reordered (GTK_TREE_MODEL (model),
863 path, has_iter ? &iter : NULL, new_order);
864
865 g_free (old_order);
866 g_free (new_order);
867}

1054gboolean
1055fm_list_model_add_file (FMListModel *model, CajaFile *file,
1056 CajaDirectory *directory)
1057{
1058 GtkTreeIter iter;
1059 GtkTreePath *path;
1060 FileEntry *file_entry;
1061 GSequenceIter *ptr, *parent_ptr;
1062 GSequence *files;
1063 gboolean replace_dummy;
1064 GHashTable *parent_hash;
1065
1066 parent_ptr = g_hash_table_lookup (model->details->directory_reverse_map,
1067 directory);
1068 if (parent_ptr)
1069 {
1070 file_entry = g_sequence_get (parent_ptr);
1071 ptr = g_hash_table_lookup (file_entry->reverse_map, file);
1072 }
1073 else
1074 {
1075 file_entry = NULL;
1076 ptr = g_hash_table_lookup (model->details->top_reverse_map, file);
1077 }
1078
1079 if (ptr != NULL)
1080 {
1081 g_warning ("file already in tree (parent_ptr: %p)!!!\n", parent_ptr);
1082 return FALSE;
1083 }
1084
1085 file_entry = g_new0 (FileEntry, 1);
1086 file_entry->file = caja_file_ref (file);
1087 file_entry->parent = NULL;
1088 file_entry->subdirectory = NULL;
1089 file_entry->files = NULL;
1090
1091 files = model->details->files;
1092 parent_hash = model->details->top_reverse_map;
1093
1094 replace_dummy = FALSE;
1095
1096 if (parent_ptr != NULL)
1097 {
1098 file_entry->parent = g_sequence_get (parent_ptr);
1099 /* At this point we set loaded. Either we saw
1100 * "done" and ignored it waiting for this, or we do this
1101 * earlier, but then we replace the dummy row anyway,
1102 * so it doesn't matter */
1103 file_entry->parent->loaded = 1;
1104 parent_hash = file_entry->parent->reverse_map;
1105 files = file_entry->parent->files;
1106 if (g_sequence_get_length (files) == 1)
1107 {
1108 GSequenceIter *dummy_ptr = g_sequence_get_iter_at_pos (files, 0);
1109 FileEntry *dummy_entry = g_sequence_get (dummy_ptr);
1110 if (dummy_entry->file == NULL)
1111 {
1112 /* replace the dummy loading entry */
1113 model->details->stamp++;
1114 g_sequence_remove (dummy_ptr);
1115
1116 replace_dummy = TRUE;
1117 }
1118 }
1119 }
1120
1121 file_entry->ptr = g_sequence_insert_sorted (files, file_entry,
1122 fm_list_model_file_entry_compare_func, model);
1123
1124 g_hash_table_insert (parent_hash, file, file_entry->ptr);
1125
1126 iter.stamp = model->details->stamp;
1127 iter.user_data = file_entry->ptr;
1128
1129 path = gtk_tree_model_get_path (GTK_TREE_MODEL (model), &iter);
1130 if (replace_dummy)
1131 {
1132 gtk_tree_model_row_changed (GTK_TREE_MODEL (model), path, &iter);
1133 }
1134 else
1135 {
1136 gtk_tree_model_row_inserted (GTK_TREE_MODEL (model), path, &iter);
1137 }
1138
1139 if (caja_file_is_directory (file))
1140 {
1141 file_entry->files = g_sequence_new ((GDestroyNotify)file_entry_free);
1142
1143 add_dummy_row (model, file_entry);
1144
1145 gtk_tree_model_row_has_child_toggled (GTK_TREE_MODEL (model),
1146 path, &iter);
1147 }
1148 gtk_tree_path_free (path);
1149
1150 return TRUE;
1151}

1153void
1154fm_list_model_file_changed (FMListModel *model, CajaFile *file,
1155 CajaDirectory *directory)
1156{
1157 FileEntry *parent_file_entry;
1158 GtkTreeIter iter;
1159 GtkTreePath *path;
1160 GSequenceIter *ptr;
1161 int pos_before, pos_after;
1162 gboolean has_iter;
1163
1164 ptr = lookup_file (model, file, directory);
1165 if (!ptr)
1166 {
1167 return;
1168 }
1169
1170 pos_before = g_sequence_iter_get_position (ptr);
1171
1172 g_sequence_sort_changed (ptr, fm_list_model_file_entry_compare_func, model);
1173
1174 pos_after = g_sequence_iter_get_position (ptr);
1175
1176 if (pos_before != pos_after)
1177 {
1178 GtkTreePath *parent_path;
1179 int length, i, old;
1180 int *new_order;
1181 GSequence *files;
1182
1183 /* The file moved, we need to send rows_reordered */
1184
1185 parent_file_entry = ((FileEntry *)g_sequence_get (ptr))->parent;
1186
1187 if (parent_file_entry == NULL)
1188 {
1189 has_iter = FALSE;
1190 parent_path = gtk_tree_path_new ();
1191 files = model->details->files;
1192 }
1193 else
1194 {
1195 has_iter = TRUE;
1196 fm_list_model_ptr_to_iter (model, parent_file_entry->ptr, &iter);
1197 parent_path = gtk_tree_model_get_path (GTK_TREE_MODEL (model), &iter);
1198 files = parent_file_entry->files;
1199 }
1200
1201 length = g_sequence_get_length (files);
1202 new_order = g_new (int, length);
1203 /* Note: new_order[newpos] = oldpos */
1204 for (i = 0, old = 0; i < length; ++i)
1205 {
1206 if (i == pos_after)
1207 {
1208 new_order[i] = pos_before;
1209 }
1210 else
1211 {
1212 if (old == pos_before)
1213 old++;
1214 new_order[i] = old++;
1215 }
1216 }
1217
1218 gtk_tree_model_rows_reordered (GTK_TREE_MODEL (model),
1219 parent_path, has_iter ? &iter : NULL, new_order);
1220
1221 gtk_tree_path_free (parent_path);
1222 g_free (new_order);
1223 }
1224
1225 fm_list_model_ptr_to_iter (model, ptr, &iter);
1226 path = gtk_tree_model_get_path (GTK_TREE_MODEL (model), &iter);
1227 gtk_tree_model_row_changed (GTK_TREE_MODEL (model), path, &iter);
1228 gtk_tree_path_free (path);
1229}

1243static void
1244fm_list_model_remove (FMListModel *model, GtkTreeIter *iter)
1245{
1246 GSequenceIter *ptr;
1247 FileEntry *file_entry, *parent_file_entry;
1248 GtkTreePath *path;
1249 GtkTreeIter parent_iter;
1250
1251 ptr = iter->user_data;
1252 file_entry = g_sequence_get (ptr);
1253
1254 if (file_entry->files != NULL)
1255 {
1256 GSequenceIter *child_ptr = NULL;
1257 FileEntry *child_file_entry = NULL;
1258
1259 while (g_sequence_get_length (file_entry->files) > 0)
1260 {
1261 child_ptr = g_sequence_get_begin_iter (file_entry->files);
1262 child_file_entry = g_sequence_get (child_ptr);
1263
1264 if (child_file_entry->file != NULL)
1265 {
1266 fm_list_model_remove_file (model,
1267 child_file_entry->file,
1268 file_entry->subdirectory);
1269 }
1270 else
1271 {
1272 path = gtk_tree_model_get_path (GTK_TREE_MODEL (model), iter);
1273 gtk_tree_path_append_index (path, 0);
1274 model->details->stamp++;
1275 g_sequence_remove (child_ptr);
1276 gtk_tree_model_row_deleted (GTK_TREE_MODEL (model), path);
1277 gtk_tree_path_free (path);
1278 }
1279
1280 /* the parent iter didn't actually change */
1281 iter->stamp = model->details->stamp;
1282 }
1283
1284 }
1285
1286 if (file_entry->file != NULL) /* Don't try to remove dummy row */
1287 {
1288 if (file_entry->parent != NULL)
1289 {
1290 g_hash_table_remove (file_entry->parent->reverse_map, file_entry->file);
1291 }
1292 else
1293 {
1294 g_hash_table_remove (model->details->top_reverse_map, file_entry->file);
1295 }
1296 }
1297
1298 parent_file_entry = file_entry->parent;
1299 if (parent_file_entry && g_sequence_get_length (parent_file_entry->files) == 1 &&
1300 file_entry->file != NULL)
1301 {
1302 /* this is the last non-dummy child, add a dummy node */
1303 /* We need to do this before removing the last file to avoid
1304 * collapsing the row.
1305 */
1306 add_dummy_row (model, parent_file_entry);
1307 }
1308
1309 if (file_entry->subdirectory != NULL)
1310 {
1311 g_signal_emit (model,
1312 list_model_signals[SUBDIRECTORY_UNLOADED], 0,
1313 file_entry->subdirectory);
1314 g_hash_table_remove (model->details->directory_reverse_map,
1315 file_entry->subdirectory);
1316 }
1317
1318 path = gtk_tree_model_get_path (GTK_TREE_MODEL (model), iter);
1319
1320 g_sequence_remove (ptr);
1321 model->details->stamp++;
1322 gtk_tree_model_row_deleted (GTK_TREE_MODEL (model), path);
1323
1324 gtk_tree_path_free (path);
1325
1326 if (parent_file_entry && g_sequence_get_length (parent_file_entry->files) == 0)
1327 {
1328 parent_iter.stamp = model->details->stamp;
1329 parent_iter.user_data = parent_file_entry->ptr;
1330 path = gtk_tree_model_get_path (GTK_TREE_MODEL (model), &parent_iter);
1331 gtk_tree_model_row_has_child_toggled (GTK_TREE_MODEL (model),
1332 path, &parent_iter);
1333 gtk_tree_path_free (path);
1334 }
1335}
