43static gboolean
44panel_separator_draw (GtkWidget *widget, cairo_t *cr)
45{
46 PanelSeparator *separator;
47 GtkStyleContext *context;
48 GtkStateFlags state;
49 GtkBorder padding;
50 int width;
51 int height;
52
53 separator = PANEL_SEPARATOR (widget);
54
55 if (!gtk_widget_is_drawable(widget))
56 return FALSE;
57
58 GTK_WIDGET_CLASS(panel_separator_parent_class)->draw(widget, cr);
59
60 state = gtk_widget_get_state_flags (widget);
61 width = gtk_widget_get_allocated_width (widget);
62 height = gtk_widget_get_allocated_height (widget);
63
64 context = gtk_widget_get_style_context (widget);
65 gtk_style_context_get_padding (context, state, &padding);
66
67 gtk_style_context_save (context);
68 gtk_style_context_set_state (context, state);
69
70 cairo_save (cr);
71
72 if (separator->priv->orientation == GTK_ORIENTATION_HORIZONTAL) {
73 int x;
74
75 x = (width - padding.left - padding.right) / 2 + padding.left;
76 x = MIN (x, width - padding.right);
77
78 gtk_render_line (context, cr,
79 x, padding.top,
80 x, height - padding.bottom);
81 } else {
82 int y;
83
84 y = (height - padding.top - padding.bottom) / 2 + padding.top;
85 y = MIN (y, height - padding.bottom);
86
87 gtk_render_line (context, cr,
88 padding.left, y,
89 width - padding.right, y);
90 }
91 cairo_restore (cr);
92
93 gtk_style_context_restore (context);
94
95 return FALSE;
96}

98static void
99panel_separator_get_preferred_width (GtkWidget *widget,
100 gint *minimal_width,
101 gint *natural_width)
102{
103 PanelSeparator *separator;
104 int size;
105
106 separator = PANEL_SEPARATOR (widget);
107
108 size = panel_toplevel_get_size (separator->priv->panel->toplevel);
109
110 if (separator->priv->orientation == GTK_ORIENTATION_VERTICAL)
111 *minimal_width = *natural_width = size;
112 else
113 *minimal_width = *natural_width = SEPARATOR_SIZE;

134static void
135panel_separator_size_allocate (GtkWidget *widget,
136 GtkAllocation *allocation)
137{
138 GtkAllocation old_allocation;
139 GtkAllocation widget_allocation;
140 PanelBackground *background;
141
142 gtk_widget_get_allocation (widget, &widget_allocation);
143
144 old_allocation.x = widget_allocation.x;
145 old_allocation.y = widget_allocation.y;
146 old_allocation.width = widget_allocation.width;
147 old_allocation.height = widget_allocation.height;
148
149 GTK_WIDGET_CLASS (panel_separator_parent_class)->size_allocate (widget, allocation);
150
151 if (old_allocation.x == allocation->x &&
152 old_allocation.y == allocation->y &&
153 old_allocation.width == allocation->width &&
154 old_allocation.height == allocation->height)
155 return;
156
157 background = &PANEL_SEPARATOR (widget)->priv->panel->toplevel->background;
158 if (background->type == PANEL_BACK_NONE ||
159 (background->type == PANEL_BACK_COLOR && !background->has_alpha))
160 return;
161
162 panel_separator_change_background (PANEL_SEPARATOR (widget));
163}

204void
205panel_separator_set_orientation (PanelSeparator *separator,
206 PanelOrientation orientation)
207{
208 GtkOrientation orient = GTK_ORIENTATION_HORIZONTAL;
209
210 g_return_if_fail (PANEL_IS_SEPARATOR (separator));
211
212 switch (orientation) {
213 case PANEL_ORIENTATION_TOP:
214 case PANEL_ORIENTATION_BOTTOM:
215 orient = GTK_ORIENTATION_HORIZONTAL;
216 break;
217 case PANEL_ORIENTATION_RIGHT:
218 case PANEL_ORIENTATION_LEFT:
219 orient = GTK_ORIENTATION_VERTICAL;
220 break;
221 }
222
223 if (orient == separator->priv->orientation)
224 return;
225
226 separator->priv->orientation = orient;
227
228 gtk_widget_queue_draw (GTK_WIDGET (separator));
229}

231void
232panel_separator_load_from_gsettings (PanelWidget *panel,
233 gboolean locked,

238
239 separator = g_object_new (PANEL_TYPE_SEPARATOR, NULL);
240
241 separator->priv->info = mate_panel_applet_register (GTK_WIDGET (separator),

243 panel, locked, position,
244 TRUE,
245 PANEL_OBJECT_SEPARATOR,
246 id);
247
248 if (!separator->priv->info) {
249 gtk_widget_destroy (GTK_WIDGET (separator));
250 return;
251 }
252
253 panel_widget_set_applet_expandable (panel, GTK_WIDGET (separator),
254 FALSE, TRUE);
255 panel_widget_set_applet_size_constrained (panel,
256 GTK_WIDGET (separator), TRUE);
257}
