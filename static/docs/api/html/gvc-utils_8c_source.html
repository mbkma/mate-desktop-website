98const gchar *
99gvc_channel_map_to_pretty_string (MateMixerStreamControl *control)
100{
101 g_return_val_if_fail (MATE_MIXER_IS_STREAM_CONTROL (control), NULL);
102
103#define HAS_POSITION(p) (mate_mixer_stream_control_has_channel_position (control, (p)))
104
105 /* Modeled after PulseAudio 5.0, probably could be extended with other combinations */
106 switch (mate_mixer_stream_control_get_num_channels (control)) {
107 case 1:
108 if (HAS_POSITION (MATE_MIXER_CHANNEL_MONO))
109 return _("Mono");
110 break;
111 case 2:
112 if (HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_LEFT) &&
113 HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_RIGHT))
114 return _("Stereo");
115 break;
116 case 4:
117 if (HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_LEFT) &&
118 HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_RIGHT) &&
119 HAS_POSITION (MATE_MIXER_CHANNEL_BACK_LEFT) &&
120 HAS_POSITION (MATE_MIXER_CHANNEL_BACK_RIGHT))
121 return _("Surround 4.0");
122 break;
123 case 5:
124 if (HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_LEFT) &&
125 HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_RIGHT) &&
126 HAS_POSITION (MATE_MIXER_CHANNEL_BACK_LEFT) &&
127 HAS_POSITION (MATE_MIXER_CHANNEL_BACK_RIGHT)) {
128 if (HAS_POSITION (MATE_MIXER_CHANNEL_LFE))
129 return _("Surround 4.1");
130 if (HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_CENTER))
131 return _("Surround 5.0");
132 }
133 break;
134 case 6:
135 if (HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_LEFT) &&
136 HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_RIGHT) &&
137 HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_CENTER) &&
138 HAS_POSITION (MATE_MIXER_CHANNEL_BACK_LEFT) &&
139 HAS_POSITION (MATE_MIXER_CHANNEL_BACK_RIGHT) &&
140 HAS_POSITION (MATE_MIXER_CHANNEL_LFE))
141 return _("Surround 5.1");
142 break;
143 case 8:
144 if (HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_LEFT) &&
145 HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_RIGHT) &&
146 HAS_POSITION (MATE_MIXER_CHANNEL_FRONT_CENTER) &&
147 HAS_POSITION (MATE_MIXER_CHANNEL_BACK_LEFT) &&
148 HAS_POSITION (MATE_MIXER_CHANNEL_BACK_RIGHT) &&
149 HAS_POSITION (MATE_MIXER_CHANNEL_SIDE_LEFT) &&
150 HAS_POSITION (MATE_MIXER_CHANNEL_SIDE_RIGHT) &&
151 HAS_POSITION (MATE_MIXER_CHANNEL_LFE))
152 return _("Surround 7.1");
153 break;
154 }
155
156#undef HAS_POSITION
157
158 return NULL;
159}
