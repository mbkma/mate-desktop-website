38static void
39_panel_launch_error_dialog (const gchar *name,
40 GdkScreen *screen,
41 const gchar *message)
42{
43 char *primary;
44
45 if (name)
46 primary = g_markup_printf_escaped (_("Could not launch '%s'"),
47 name);
48 else
49 primary = g_strdup (_("Could not launch application"));
50
51 panel_error_dialog (NULL, screen, "cannot_launch", TRUE,

56static gboolean
57_panel_launch_handle_error (const gchar *name,
58 GdkScreen *screen,
59 GError *local_error,
60 GError **error)
61{
62 if (g_error_matches (local_error,
63 G_IO_ERROR, G_IO_ERROR_CANCELLED)) {
64 g_error_free (local_error);
65 return TRUE;
66 }
67
68 else if (error != NULL)
69 g_propagate_error (error, local_error);
70
71 else {
72 _panel_launch_error_dialog (name, screen, local_error->message);
73 g_error_free (local_error);
74 }
75
76 return FALSE;
77}

98gboolean
99panel_app_info_launch_uris (GDesktopAppInfo *appinfo,
100 GList *uris,
101 GdkScreen *screen,
102 const gchar *action,
103 guint32 timestamp,
104 GError **error)
105{
106 GdkAppLaunchContext *context;
107 GError *local_error;
108 gboolean retval;
109
110 g_return_val_if_fail (G_IS_DESKTOP_APP_INFO (appinfo), FALSE);
111 g_return_val_if_fail (GDK_IS_SCREEN (screen), FALSE);
112 g_return_val_if_fail (error == NULL || *error == NULL, FALSE);
113
114 GdkDisplay *display = gdk_display_get_default ();
115 context = gdk_display_get_app_launch_context (display);
116 gdk_app_launch_context_set_screen (context, screen);
117 gdk_app_launch_context_set_timestamp (context, timestamp);
118
119 local_error = NULL;
120 if (action == NULL) {
121 retval = g_desktop_app_info_launch_uris_as_manager (appinfo, uris,
122 G_APP_LAUNCH_CONTEXT (context),
123 G_SPAWN_SEARCH_PATH | G_SPAWN_DO_NOT_REAP_CHILD,
124 NULL, NULL, gather_pid_callback, appinfo,
125 &local_error);
126 } else {
127 g_desktop_app_info_launch_action (appinfo, action, G_APP_LAUNCH_CONTEXT (context));
128 retval = TRUE;
129 }
130
131 g_object_unref (context);
132
133 if ((local_error == NULL) && (retval == TRUE))
134 return TRUE;
135
136 return _panel_launch_handle_error (g_app_info_get_name (G_APP_INFO(appinfo)),
137 screen, local_error, error);
138}

193gboolean
194panel_launch_desktop_file (const char *desktop_file,
195 GdkScreen *screen,
196 GError **error)
197{
198 GDesktopAppInfo *appinfo;
199 gboolean retval;
200
201 g_return_val_if_fail (desktop_file != NULL, FALSE);
202 g_return_val_if_fail (GDK_IS_SCREEN (screen), FALSE);
203 g_return_val_if_fail (error == NULL || *error == NULL, FALSE);
204
205 appinfo = NULL;
206
207 if (g_path_is_absolute (desktop_file))
208 appinfo = g_desktop_app_info_new_from_filename (desktop_file);
209 else {
210 char *full;
211
212 full = panel_g_lookup_in_applications_dirs (desktop_file);
213 if (full) {
214 appinfo = g_desktop_app_info_new_from_filename (full);
215 g_free (full);
216 }
217 }
218
219 if (appinfo == NULL)
220 return FALSE;
221
222 retval = panel_app_info_launch_uris (appinfo, NULL, screen, NULL,
223 gtk_get_current_event_time (),
224 error);
225
226 g_object_unref (appinfo);
227
228 return retval;
229}

240gboolean
241panel_launch_desktop_file_with_fallback (const char *desktop_file,
242 const char *fallback_exec,
243 GdkScreen *screen,
244 GError **error)
245{
246 char *argv[2] = { (char *) fallback_exec, NULL };
247 GError *local_error;
248 gboolean retval;
249 GPid pid;
250 GdkDisplay *display;
251 char *display_name;
252
253 g_return_val_if_fail (desktop_file != NULL, FALSE);
254 g_return_val_if_fail (fallback_exec != NULL, FALSE);
255 g_return_val_if_fail (GDK_IS_SCREEN (screen), FALSE);
256 g_return_val_if_fail (error == NULL || *error == NULL, FALSE);
257
258 /* need to pass a non-NULL error to avoid getting a dialog */
259 local_error = NULL;
260 if (panel_launch_desktop_file (desktop_file, screen, &local_error))
261 return TRUE;
262
263 if (local_error) {
264 g_error_free (local_error);
265 local_error = NULL;
266 }
267
268 display = gdk_screen_get_display (screen);
269 display_name = g_strdup (gdk_display_get_name (display));
270 retval = g_spawn_async (NULL, /* working directory */
271 argv,
272 NULL, /* envp */
273 G_SPAWN_SEARCH_PATH,
274 set_environment,
275 &display_name,
276 &pid,
277 &local_error);
278 g_free (display_name);
279
280 if (local_error == NULL && retval == TRUE) {
281 g_child_watch_add (pid, dummy_child_watch, NULL);
282 }
283 return TRUE;
284
285 return _panel_launch_handle_error (fallback_exec,
286 screen, local_error, error);
287}
