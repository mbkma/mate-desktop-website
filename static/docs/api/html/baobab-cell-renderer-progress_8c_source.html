97static void
98baobab_cell_renderer_progress_get_size (GtkCellRenderer *cell,
99 GtkWidget *widget,
100 const GdkRectangle *cell_area,
101 gint *x_offset,
102 gint *y_offset,
103 gint *width,
104 gint *height)
105{
106 gint calc_width;
107 gint calc_height;
108 gint xpad;
109 gint ypad;
110 gfloat xalign;
111 gfloat yalign;
112
113 gtk_cell_renderer_get_padding (cell, &xpad, &ypad);
114 calc_width = (gint) xpad * 2 + FIXED_WIDTH;
115 calc_height = (gint) ypad * 2 + FIXED_HEIGHT;
116
117 if (width)
118 *width = calc_width;
119
120 if (height)
121 *height = calc_height;
122
123 if (cell_area)
124 {
125 gtk_cell_renderer_get_alignment (cell, &xalign, &yalign);
126 if (x_offset)
127 {
128 *x_offset = (gint) (xalign * (gfloat) (cell_area->width - calc_width));
129 *x_offset = MAX (*x_offset, 0);
130 }
131
132 if (y_offset)
133 {
134 *y_offset = (gint) (yalign * (gfloat) (cell_area->height - calc_height));
135 *y_offset = MAX (*y_offset, 0);
136 }
137 }
138}

140static void
141set_color_according_to_perc (cairo_t *cr, double value)
142{
143 static GdkRGBA red;
144 static GdkRGBA yellow;
145 static GdkRGBA green;
146 static gboolean colors_initialized = FALSE;
147
148 if (!colors_initialized)
149 {
150 /* hardcoded tango colors */
151 gdk_rgba_parse (&red, "#cc0000");
152 gdk_rgba_parse (&yellow, "#edd400");
153 gdk_rgba_parse (&green, "#73d216");
154
155 colors_initialized = TRUE;
156 }
157
158 if (value <= 0)
159 {
160 cairo_set_source_rgb (cr, 1, 1, 1);
161 return;
162 }
163 else if (value <= 33.33)
164 {
165 gdk_cairo_set_source_rgba (cr, &green);
166 return;
167 }
168 else if (value <= 66.66)
169 {
170 gdk_cairo_set_source_rgba (cr, &yellow);
171 return;
172 }
173 else if (value <= 100.0)
174 {
175 gdk_cairo_set_source_rgba (cr, &red);
176 return;
177 }
178 else
179 g_assert_not_reached ();
180}

182static void
183baobab_cell_renderer_progress_render (GtkCellRenderer *cell,
184 cairo_t *cr,
185 GtkWidget *widget,
186 const GdkRectangle *background_area,
187 const GdkRectangle *cell_area,
188 guint flags)
189{
190 BaobabCellRendererProgress *cellprogress = BAOBAB_CELL_RENDERER_PROGRESS (cell);
191 gint x, y, w, h, perc_w;
192 gboolean is_rtl;
193 gint xpad;
194 gint ypad;
195 GtkStyle *style;
196
197 is_rtl = gtk_widget_get_direction (widget) == GTK_TEXT_DIR_RTL;
198
199 gtk_cell_renderer_get_padding (cell, &xpad, &ypad);
200
201 x = cell_area->x + xpad;
202 y = cell_area->y + ypad;
203
204 w = cell_area->width - xpad * 2;
205 h = cell_area->height - ypad * 2;
206
207 /*
208 * we always use a white bar with black
209 * border and green/yellow/red progress...
210 * I know it's not theme friendly, but we don't
211 * want a plain progress bar
212 */
213
214 cairo_rectangle (cr, x, y, w, h);
215 cairo_set_source_rgb (cr, 0, 0, 0);
216 cairo_fill (cr);
217
218 style = gtk_widget_get_style (widget);
219 x += style->xthickness;
220 y += style->ythickness;
221 w -= style->xthickness * 2;
222 h -= style->ythickness * 2;
223
224 cairo_rectangle (cr, x, y, w, h);
225 cairo_set_source_rgb (cr, 1, 1, 1);
226 cairo_fill (cr);
227
228 perc_w = (gint) (w * MAX (0, cellprogress->priv->perc) / 100);
229
230 cairo_rectangle (cr, is_rtl ? (x + w - perc_w) : x, y, perc_w, h);
231 set_color_according_to_perc (cr, cellprogress->priv->perc);
232 cairo_fill (cr);
233}
