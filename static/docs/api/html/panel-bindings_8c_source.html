68static void
69panel_binding_set_from_string (PanelBinding *binding,
70 const char *str)
71{
72 g_assert (binding->keyval == 0);
73 g_assert (binding->modifiers == 0);
74
75 if (!str || !str [0] || !strcmp (str, "disabled")) {
76 binding->keyval = 0;
77 binding->modifiers = 0;
78 return;
79 }
80
81 gtk_accelerator_parse (str, &binding->keyval, &binding->modifiers);
82 if (binding->keyval == 0 && binding->modifiers == 0) {
83 g_warning ("Unable to parse binding '%s'\n", str);
84 return;
85 }
86}

88static inline GtkBindingSet *
89get_binding_set (GtkBindingSet *binding_set)
90{
91 if (!binding_set) {
92 PanelToplevelClass *toplevel_class;
93
94 toplevel_class = g_type_class_peek (PANEL_TYPE_TOPLEVEL);
95 if (!toplevel_class)
96 return NULL;
97
98 g_assert (PANEL_IS_TOPLEVEL_CLASS (toplevel_class));
99
100 binding_set = gtk_binding_set_by_class (toplevel_class);
101 }
102
103 return binding_set;
104}

128static void
129panel_binding_changed (GSettings *settings,
130 gchar *key,
131 PanelBinding *binding)
132{
133 if (binding->keyval)
134 panel_binding_clear_entry (binding, NULL);
135
136 binding->keyval = 0;
137 binding->modifiers = 0;
138
139 gchar *value = g_settings_get_string (settings, key);
140 panel_binding_set_from_string (binding, value);
141 g_free (value);
142
143 if (!binding->keyval)
144 return;
145
146 panel_binding_set_entry (binding, NULL);
147}

149static void
150panel_binding_watch (PanelBinding *binding,
151 const char *key)
152{
153 gchar *signal_name;
154 signal_name = g_strdup_printf ("changed::%s", key);
155 g_signal_connect (marco_keybindings_settings,
156 signal_name,
157 G_CALLBACK (panel_binding_changed),

159 g_free (signal_name);

162static void
163panel_bindings_mouse_modifier_set_from_string (const char *str)
164{
165 guint modifier_keysym;
166 guint modifier_keymask;
167
168 gtk_accelerator_parse (str, &modifier_keysym, &modifier_keymask);
169
170 if (modifier_keysym == 0 && modifier_keymask == 0) {
171 g_warning ("Unable to parse mouse modifier '%s'\n", str);
172 return;
173 }
174
175 if (modifier_keymask)
176 mouse_button_modifier_keymask = modifier_keymask;
177 else
178 mouse_button_modifier_keymask = DEFAULT_MOUSE_MODIFIER;
179}

191static void
192panel_bindings_initialise (void)
193{
194 gsize i;
195 char *str;
196
197 if (initialised)
198 return;
199
200 if (!mate_gsettings_schema_exists (MARCO_SCHEMA)) {
201 initialised = TRUE;
202 return;
203 }
204
205 marco_settings = g_settings_new (MARCO_SCHEMA);
206 marco_keybindings_settings = g_settings_new (MARCO_KEYBINDINGS_SCHEMA);
207
208 for (i = 0; i < G_N_ELEMENTS (bindings); i++) {
209 str = g_settings_get_string (marco_keybindings_settings, bindings [i].key);
210 panel_binding_set_from_string (&bindings [i], str);
211 panel_binding_watch (&bindings [i], bindings [i].key);
212 g_free (str);
213 }
214
215 /* mouse button modifier */
216
217 g_signal_connect (marco_settings,
218 "changed::" MARCO_MOUSE_BUTTON_MODIFIER_KEY,
219 G_CALLBACK (panel_bindings_mouse_modifier_changed),
220 NULL);
221
222 str = g_settings_get_string (marco_settings, MARCO_MOUSE_BUTTON_MODIFIER_KEY);
223 panel_bindings_mouse_modifier_set_from_string (str);
224 g_free (str);
225
226 initialised = TRUE;
227}

229void
230panel_bindings_set_entries (GtkBindingSet *binding_set)
231{
232 gsize i;
233
234 if (!initialised)
235 panel_bindings_initialise ();
236
237 for (i = 0; i < G_N_ELEMENTS (bindings); i++) {
238 if (!bindings [i].keyval)
239 continue;
240
241 panel_binding_set_entry (&bindings [i], binding_set);
242 }
243}
