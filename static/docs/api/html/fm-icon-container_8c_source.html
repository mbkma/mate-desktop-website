51static CajaIconInfo *
52fm_icon_container_get_icon_images (CajaIconContainer *container,
53 CajaIconData *data,
54 int size,
55 GList **emblem_pixbufs,
56 char **embedded_text,
57 gboolean for_drag_accept,
58 gboolean need_large_embeddded_text,
59 gboolean *embedded_text_needs_loading,
60 gboolean *has_window_open)
61{
62 FMIconView *icon_view;
63 CajaFile *file;
64 gboolean use_embedding;
65 CajaFileIconFlags flags;
66 guint emblem_size;
67 gint scale;
68
69 file = (CajaFile *) data;
70
71 g_assert (CAJA_IS_FILE (file));
72 icon_view = get_icon_view (container);
73 g_return_val_if_fail (icon_view != NULL, NULL);
74
75 use_embedding = FALSE;
76 if (embedded_text)
77 {
78 *embedded_text = caja_file_peek_top_left_text (file, need_large_embeddded_text, embedded_text_needs_loading);
79 use_embedding = *embedded_text != NULL;
80 }
81
82 if (emblem_pixbufs != NULL)
83 {
84 emblem_size = caja_icon_get_emblem_size_for_icon_size (size);
85 /* don't return images larger than the actual icon size */
86 emblem_size = MIN (emblem_size, size);
87
88 if (emblem_size > 0)
89 {
90 char **emblems_to_ignore;
91
92 emblems_to_ignore = fm_directory_view_get_emblem_names_to_exclude
93 (FM_DIRECTORY_VIEW (icon_view));
94 *emblem_pixbufs = caja_file_get_emblem_pixbufs (file,
95 emblem_size,
96 FALSE,
97 emblems_to_ignore);
98 g_strfreev (emblems_to_ignore);
99 }
100 }
101
102 *has_window_open = caja_file_has_open_window (file);
103
104 flags = CAJA_FILE_ICON_FLAGS_USE_MOUNT_ICON_AS_EMBLEM;
105 if (!fm_icon_view_is_compact (icon_view) ||
106 caja_icon_container_get_zoom_level (container) > CAJA_ZOOM_LEVEL_STANDARD)
107 {
108 flags |= CAJA_FILE_ICON_FLAGS_USE_THUMBNAILS;
109 if (fm_icon_view_is_compact (icon_view))
110 {
111 flags |= CAJA_FILE_ICON_FLAGS_FORCE_THUMBNAIL_SIZE;
112 }
113 }
114
115 if (use_embedding)
116 {
117 flags |= CAJA_FILE_ICON_FLAGS_EMBEDDING_TEXT;
118 }
119 if (for_drag_accept)
120 {
121 flags |= CAJA_FILE_ICON_FLAGS_FOR_DRAG_ACCEPT;
122 }
123
124 scale = gtk_widget_get_scale_factor (GTK_WIDGET (icon_view));
125
126 return caja_file_get_icon (file, size, scale, flags);
127}

206/*
207 * Get the preference for which caption text should appear
208 * beneath icons.
209 */
210static GQuark *
211fm_icon_container_get_icon_text_attributes_from_preferences (void)
212{
213 static GQuark *attributes = NULL;
214
215 if (attributes == NULL)
216 {
217 eel_g_settings_add_auto_strv_as_quarks (caja_icon_view_preferences,
218 CAJA_PREFERENCES_ICON_VIEW_CAPTIONS,
219 &attributes);
220 }
221
222 /* We don't need to sanity check the attributes list even though it came
223 * from preferences.
224 *
225 * There are 2 ways that the values in the list could be bad.
226 *
227 * 1) The user picks "bad" values. "bad" values are those that result in
228 * there being duplicate attributes in the list.
229 *
230 * 2) Value stored in MateConf are tampered with. Its possible physically do
231 * this by pulling the rug underneath MateConf and manually editing its
232 * config files. Its also possible to use a third party MateConf key
233 * editor and store garbage for the keys in question.
234 *
235 * Thankfully, the Caja preferences machinery deals with both of
236 * these cases.
237 *
238 * In the first case, the preferences dialog widgetry prevents
239 * duplicate attributes by making "bad" choices insensitive.
240 *
241 * In the second case, the preferences getter (and also the auto storage) for
242 * string_array values are always valid members of the enumeration associated
243 * with the preference.
244 *
245 * So, no more error checking on attributes is needed here and we can return
246 * a the auto stored value.
247 */
248 return attributes;
249}

273fm_icon_container_get_icon_text_attribute_names (CajaIconContainer *container,
274 int *len)
275{

302static char *
303get_git_branch (const char *git_path) {
304 gchar *head_content = NULL;
305 gchar *branch_name = NULL;
306 gchar *head_path = NULL;
307
308 if (g_file_test (git_path, G_FILE_TEST_IS_REGULAR))
309 {
310 /* It's a file, thus we are probably dealing with a submodule, so read it to resolve the real git dir */
311 gchar *contents = NULL;
312 if (g_file_get_contents (git_path, &contents, NULL, NULL) && contents)
313 {
314 const char *git_dir_prefix = "gitdir: ";
315 if (g_str_has_prefix (contents, git_dir_prefix))
316 {
317 size_t git_dir_prefix_len = strlen (git_dir_prefix);
318 gchar *relative = g_strstrip (contents + git_dir_prefix_len);
319 gchar *base = g_path_get_dirname (git_path);
320 g_free (git_path);
321 git_path = g_build_filename (base, relative, NULL);
322 g_free (base);
323 }
324 g_free (contents);
325 }
326 }
327
328 /* Extract the current git branch name of the repository from file HEAD within .git directory */
329 if (g_file_test (git_path, G_FILE_TEST_IS_DIR))
330 {
331 head_path = g_build_filename (git_path, "HEAD", NULL);
332 if (g_file_get_contents (head_path, &head_content, NULL, NULL))
333 {
334 g_strstrip (head_content);
335 if (g_str_has_prefix (head_content, "ref: "))
336 {
337 gchar **parts = g_strsplit (head_content, "/", -1);
338 if (parts != NULL)
339 {
340 branch_name = g_strdup (parts[g_strv_length(parts) - 1]);
341 g_strchomp (branch_name);
342 g_strfreev (parts);
343 }
344 }
345 else
346 {
347 /* Repository is in a detached HEAD state */
348 branch_name = g_strdup_printf (_("detached: %.7s"), head_content);
349 }
350 }
351 }
352
353 g_free (head_content);
354 g_free (head_path);
355
356 /* branch_name is a newly allocated string, so the caller is responsible for freeing it */
357 return branch_name;
358}

360/* This callback returns the text, both the editable part, and the
361 * part below that is not editable.
362 */
363static void
364fm_icon_container_get_icon_text (CajaIconContainer *container,
365 CajaIconData *data,
366 char **editable_text,
367 char **additional_text,
368 gboolean include_invisible)
369{
370 GQuark *attributes;
371 char *text_array[4];
372 int i, j, num_attributes;
373 FMIconView *icon_view;
374 CajaFile *file;
375 gboolean use_additional;
376
377 file = CAJA_FILE (data);
378
379 g_assert (CAJA_IS_FILE (file));
380 g_assert (editable_text != NULL);
381 icon_view = get_icon_view (container);
382 g_return_if_fail (icon_view != NULL);
383
384 use_additional = (additional_text != NULL);
385
386 /* In the smallest zoom mode, no text is drawn. */
387 if (caja_icon_container_get_zoom_level (container) == CAJA_ZOOM_LEVEL_SMALLEST &&
388 !include_invisible)
389 {
390 *editable_text = NULL;
391 }
392 else
393 {
394 /* Strip the suffix for caja object xml files. */
395 *editable_text = caja_file_get_display_name (file);
396 }
397
398 if (!use_additional)
399 {
400 return;
401 }
402
403 if (fm_icon_view_is_compact (icon_view))
404 {
405 *additional_text = NULL;
406 return;
407 }
408
409 if (CAJA_IS_DESKTOP_ICON_FILE (file))
410 {
411 /* Don't show the normal extra information for desktop icons, it doesn't
412 * make sense. */
413 *additional_text = NULL;
414 return;
415 }
416
417 /* Handle link files specially. */
418 if (caja_file_is_caja_link (file))
419 {
420 /* FIXME bugzilla.gnome.org 42531: Does sync. I/O and works only locally. */
421 *additional_text = NULL;
422 if (caja_file_is_local (file))
423 {
424 char *actual_uri;
425 gchar *description;
426
427 actual_uri = caja_file_get_uri (file);
428 description = caja_link_local_get_additional_text (actual_uri);
429
430 if (description)
431 *additional_text = g_strdup_printf (" \n%s\n ", description);
432
433 g_free (description);
434 g_free (actual_uri);
435 }
436 /* Don't show the normal extra information for desktop files, it doesn't
437 * make sense. */
438 return;
439 }
440
441 /* Find out what attributes go below each icon. */
442 attributes = fm_icon_container_get_icon_text_attribute_names (container,
443 &num_attributes);
444
445 /* Get the attributes. */
446 j = 0;
447 for (i = 0; i < num_attributes; ++i)
448 {
449 if (attributes[i] == attribute_none_q)
450 {
451 continue;
452 }
453
454 text_array[j++] =
455 caja_file_get_string_attribute_with_default_q (file, attributes[i]);
456 }
457 text_array[j] = NULL;
458
459 /* Return them. */
460 if (j == 0)
461 {
462 *additional_text = NULL;
463 if (container->details->display_git_branch == CAJA_DISPLAY_GIT_BRANCH_ENABLED)
464 {
465 /* If we have a directory, check if it's a git repository or submodule */
466 GFileType file_type = caja_file_get_file_type (file);
467 if (file_type == G_FILE_TYPE_DIRECTORY)
468 {
469 GFile *location = caja_file_get_location (file);
470 char *path = g_file_get_path (location);
471 if (path != NULL)
472 {
473 char *git_path = g_build_path (G_DIR_SEPARATOR_S, path, ".git", NULL);
474 if (git_path != NULL && G_UNLIKELY(g_file_test (git_path, G_FILE_TEST_EXISTS)))
475 {
476 char *branch = get_git_branch (git_path);
477 if (branch != NULL)
478 {
479 *additional_text = g_strdup_printf ("[%s]", branch);
480 g_free (branch);
481 }
482 }
483 g_free (git_path);
484 g_free (path);
485 }
486 g_object_unref (location);
487 }
488 }
489 }
490 else if (j == 1)
491 {
492 /* Only one item, avoid the strdup + free */
493 *additional_text = text_array[0];
494 }
495 else
496 {
497 *additional_text = g_strjoinv ("\n", text_array);
498
499 for (i = 0; i < j; i++)
500 {
501 g_free (text_array[i]);
502 }
503 }
504}

506/* Sort as follows:
507 * 0) computer link
508 * 1) home link
509 * 2) network link
510 * 3) mount links
511 * 4) other
512 * 5) trash link
513 */
514typedef enum
515{
516 SORT_COMPUTER_LINK,
517 SORT_HOME_LINK,
518 SORT_NETWORK_LINK,
519 SORT_MOUNT_LINK,
520 SORT_OTHER,
521 SORT_TRASH_LINK
522} SortCategory;
523
524static SortCategory
525get_sort_category (CajaFile *file)
526{
527 SortCategory category;
528
529 category = SORT_OTHER;
530
531 if (CAJA_IS_DESKTOP_ICON_FILE (file))
532 {
533 CajaDesktopLink *link;
534
535 link = caja_desktop_icon_file_get_link (CAJA_DESKTOP_ICON_FILE (file));
536
537 if (link != NULL)
538 {
539 switch (caja_desktop_link_get_link_type (link))
540 {
541 case CAJA_DESKTOP_LINK_COMPUTER:
542 category = SORT_COMPUTER_LINK;
543 break;
544 case CAJA_DESKTOP_LINK_HOME:
545 category = SORT_HOME_LINK;
546 break;
547 case CAJA_DESKTOP_LINK_MOUNT:
548 category = SORT_MOUNT_LINK;
549 break;
550 case CAJA_DESKTOP_LINK_TRASH:
551 category = SORT_TRASH_LINK;
552 break;
553 case CAJA_DESKTOP_LINK_NETWORK:
554 category = SORT_NETWORK_LINK;
555 break;
556 default:
557 category = SORT_OTHER;
558 break;
559 }
560 g_object_unref (link);
561 }
562 }
563
564 return category;
565}

567static int
568fm_desktop_icon_container_icons_compare (CajaIconContainer *container,
569 CajaIconData *data_a,
570 CajaIconData *data_b)
571{
572 CajaFile *file_a;
573 CajaFile *file_b;
574 FMDirectoryView *directory_view;
575 SortCategory category_a, category_b;
576
577 file_a = (CajaFile *) data_a;
578 file_b = (CajaFile *) data_b;
579
580 directory_view = FM_DIRECTORY_VIEW (FM_ICON_CONTAINER (container)->view);
581 g_return_val_if_fail (directory_view != NULL, 0);
582
583 category_a = get_sort_category (file_a);
584 category_b = get_sort_category (file_b);
585
586 if (category_a == category_b)
587 {
588 return caja_file_compare_for_sort
589 (file_a, file_b, CAJA_FILE_SORT_BY_DISPLAY_NAME,
590 fm_directory_view_should_sort_directories_first (directory_view),
591 FALSE);
592 }
593
594 if (category_a < category_b)
595 {
596 return -1;
597 }
598 else
599 {
600 return +1;
601 }
602}
