53static void
54statusbar_set_date (GtkStatusbar *statusbar,
55 EomThumbView *view)
56{
57 EomImage *image;
58 gchar *date = NULL;
59 gchar time_buffer[32];
60 ExifData *exif_data;
61
62 if (eom_thumb_view_get_n_selected (view) == 0)
63 {
64 return;
65 }
66
67 image = eom_thumb_view_get_first_selected_image (view);
68
69 gtk_statusbar_pop (statusbar, 0);
70
71 if (!eom_image_has_data (image, EOM_IMAGE_DATA_EXIF))
72 {
73 if (!eom_image_load (image, EOM_IMAGE_DATA_EXIF, NULL, NULL))
74 {
75 gtk_widget_hide (GTK_WIDGET (statusbar));
76 }
77 }
78
79 exif_data = eom_image_get_exif_info (image);
80
81 if (exif_data)
82 {
83 date = eom_exif_util_format_date (eom_exif_data_get_value (exif_data, EXIF_TAG_DATE_TIME_ORIGINAL, time_buffer, 32));
84 eom_exif_data_free (exif_data);
85 }
86
87 if (date)
88 {
89 gtk_statusbar_push (statusbar, 0, date);
90 gtk_widget_show (GTK_WIDGET (statusbar));
91 g_free (date);
92 }
93 else
94 {
95 gtk_widget_hide (GTK_WIDGET (statusbar));
96 }
97}

167static void
168eom_statusbar_date_plugin_activate (EomWindowActivatable *activatable)
169{
170 EomStatusbarDatePlugin *plugin = EOM_STATUSBAR_DATE_PLUGIN (activatable);
171 EomWindow *window = plugin->window;
172 GtkWidget *statusbar = eom_window_get_statusbar (window);
173 GtkWidget *thumbview = eom_window_get_thumb_view (window);
174
175 eom_debug (DEBUG_PLUGINS);
176
177 plugin->statusbar_date = gtk_statusbar_new ();
178 gtk_widget_set_size_request (plugin->statusbar_date, 200, 10);
179 gtk_widget_set_margin_top (GTK_WIDGET (plugin->statusbar_date), 0);
180 gtk_widget_set_margin_bottom (GTK_WIDGET (plugin->statusbar_date), 0);
181 gtk_box_pack_end (GTK_BOX (statusbar), plugin->statusbar_date, FALSE, FALSE, 0);
182
183 plugin->signal_id = g_signal_connect_after (G_OBJECT (thumbview), "selection_changed",
184 G_CALLBACK (selection_changed_cb), plugin);
185
186 statusbar_set_date (GTK_STATUSBAR (plugin->statusbar_date),
187 EOM_THUMB_VIEW (eom_window_get_thumb_view (window)));
188}

190static void

192{
193 EomStatusbarDatePlugin *plugin = EOM_STATUSBAR_DATE_PLUGIN (activatable);
194 EomWindow *window = plugin->window;
195 GtkWidget *statusbar = eom_window_get_statusbar (window);
196 GtkWidget *view = eom_window_get_thumb_view (window);
197
198#if GLIB_CHECK_VERSION(2,62,0)
199 g_clear_signal_handler (&plugin->signal_id, view);
200#else
201 if (plugin->signal_id != 0) {
202 g_signal_handler_disconnect (view, plugin->signal_id);
203 plugin->signal_id = 0;
204 }
205#endif
206
207 gtk_container_remove (GTK_CONTAINER (statusbar), plugin->statusbar_date);
208}
