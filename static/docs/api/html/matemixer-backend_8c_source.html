351MateMixerDevice *
352mate_mixer_backend_get_device (MateMixerBackend *backend, const gchar *name)
353{
354 const GList *list;
355
356 g_return_val_if_fail (MATE_MIXER_IS_BACKEND (backend), NULL);
357 g_return_val_if_fail (name != NULL, NULL);
358
359 list = mate_mixer_backend_list_devices (backend);
360 while (list != NULL) {
361 MateMixerDevice *device = MATE_MIXER_DEVICE (list->data);
362
363 if (strcmp (name, mate_mixer_device_get_name (device)) == 0)
364 return device;
365
366 list = list->next;
367 }
368 return NULL;
369}

526static void
527device_added (MateMixerBackend *backend, const gchar *name)
528{
529 MateMixerDevice *device;
530
531 device = mate_mixer_backend_get_device (backend, name);
532 if (G_UNLIKELY (device == NULL)) {
533 g_warn_if_reached ();
534 return;
535 }
536
537 /* Keep the device in a hash table as it won't be possible to retrieve
538 * it when the remove signal is received */
539 g_hash_table_insert (backend->priv->devices,
540 g_strdup (name),
541 g_object_ref (device));
542
543 /* Connect to the stream signals from devices so we can forward them on
544 * the backend */
545 g_signal_connect_swapped (G_OBJECT (device),
546 "stream-added",
547 G_CALLBACK (device_stream_added),
548 backend);
549 g_signal_connect_swapped (G_OBJECT (device),
550 "stream-removed",
551 G_CALLBACK (device_stream_removed),
552 backend);
553}
554
555static void
556device_removed (MateMixerBackend *backend, const gchar *name)
557{
558 MateMixerDevice *device;
559
560 device = g_hash_table_lookup (backend->priv->devices, name);
561 if (G_UNLIKELY (device == NULL)) {
562 g_warn_if_reached ();
563 return;
564 }
565
566 g_signal_handlers_disconnect_by_func (G_OBJECT (device),
567 G_CALLBACK (device_stream_added),
568 backend);
569 g_signal_handlers_disconnect_by_func (G_OBJECT (device),
570 G_CALLBACK (device_stream_removed),
571 backend);
572
573 g_hash_table_remove (backend->priv->devices, name);
574}

608void
609_mate_mixer_backend_set_default_input_stream (MateMixerBackend *backend,
610 MateMixerStream *stream)
611{
612 g_return_if_fail (MATE_MIXER_IS_BACKEND (backend));
613 g_return_if_fail (stream == NULL || MATE_MIXER_IS_STREAM (stream));
614
615 if (backend->priv->default_input == stream)
616 return;
617
618 if (backend->priv->default_input != NULL)
619 g_object_unref (backend->priv->default_input);
620
621 if (stream != NULL)
622 backend->priv->default_input = g_object_ref (stream);
623 else
624 backend->priv->default_input = NULL;
625
626 g_debug ("Default input stream changed to %s",
627 (stream != NULL) ? mate_mixer_stream_get_name (stream) : "none");
628
629 g_object_notify_by_pspec (G_OBJECT (backend),
630 properties[PROP_DEFAULT_INPUT_STREAM]);
631}
