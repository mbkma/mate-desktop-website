35/* CajaDirectory is a class that manages the model for a directory,
36 real or virtual, for Caja, mainly the file-manager component. The directory is
37 responsible for managing both real data and cached metadata. On top of
38 the file system independence provided by gio, the directory
39 object also provides:
40
41 1) A synchronization framework, which notifies via signals as the
42 set of known files changes.
43 2) An abstract interface for getting attributes and performing
44 operations on files.
45*/
46

83 /* The files_added signal is emitted as the directory model
84 * discovers new files.
85 */
86 void (* files_added) (CajaDirectory *directory,
87 GList *added_files);
88
89 /* The files_changed signal is emitted as changes occur to
90 * existing files that are noticed by the synchronization framework,
91 * including when an old file has been deleted. When an old file
92 * has been deleted, this is the last chance to forget about these
93 * file objects, which are about to be unref'd. Use a call to
94 * caja_file_is_gone () to test for this case.
95 */
96 void (* files_changed) (CajaDirectory *directory,
97 GList *changed_files);
98
99 /* The done_loading signal is emitted when a directory load
100 * request completes. This is needed because, at least in the
101 * case where the directory is empty, the caller will receive
102 * no kind of notification at all when a directory load
103 * initiated by `caja_directory_file_monitor_add' completes.
104 */
105 void (* done_loading) (CajaDirectory *directory);
