32hook_on_connect(DropboxClient *dc) {
33 dc->hook_connect_called = TRUE;
34
35 if (dc->command_connect_called) {
36 debug("client connection");
37 g_hook_list_invoke(&(dc->onconnect_hooklist), FALSE);

91void
92dropbox_client_force_reconnect(DropboxClient *dc) {
93 if (dropbox_client_is_connected(dc) == TRUE) {
94 debug("forcing client to reconnect");
95 dropbox_command_client_force_reconnect(&(dc->dcc));

100/* should only be called once on initialization */
101void
102dropbox_client_setup(DropboxClient *dc) {
103 caja_dropbox_hooks_setup(&(dc->hookserv));
104 dropbox_command_client_setup(&(dc->dcc));
105
106 g_hook_list_init(&(dc->ondisconnect_hooklist), sizeof(GHook));
107 g_hook_list_init(&(dc->onconnect_hooklist), sizeof(GHook));
108
109 dc->hook_disconnect_called = dc->command_disconnect_called = FALSE;
110 dc->hook_connect_called = dc->command_connect_called = FALSE;
111
112 caja_dropbox_hooks_add_on_connect_hook(&(dc->hookserv),
113 (DropboxHookClientConnectHook)
114 hook_on_connect, dc);
115
116 dropbox_command_client_add_on_connect_hook(&(dc->dcc),
117 (DropboxCommandClientConnectHook)
118 command_on_connect, dc);
119
120 caja_dropbox_hooks_add_on_disconnect_hook(&(dc->hookserv),
121 (DropboxHookClientConnectHook)
122 hook_on_disconnect, dc);
123
124 dropbox_command_client_add_on_disconnect_hook(&(dc->dcc),
125 (DropboxCommandClientConnectHook)
126 command_on_disconnect, dc);
127}
