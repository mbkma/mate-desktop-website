226GtkWidget *
227make_title_label (const char *text)
228{
229 GtkWidget *label;
230 char *full;
231
232 full = g_strdup_printf ("<span weight=\"bold\">%s</span>", text);
233 label = gtk_label_new (full);

235
236 gtk_label_set_xalign (GTK_LABEL (label), 0.0);

244create_sys_view (ProcData *procdata, GtkBuilder * builder)
245{
246 GtkWidget *cpu_graph_box, *mem_graph_box, *net_graph_box;

253 gint i;

269 TRUE,
270 0);

273 gtk_box_reorder_child (GTK_BOX (cpu_graph_box), flowbox, 1);

276
277 temp_hbox = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);

289 label_text = g_strdup_printf (_("CPU%d"), i+1);
290 }
291 title_text = g_strdup_printf(title_template, label_text);
292 label = gtk_label_new (label_text);
293 gsm_color_button_set_title(GSM_COLOR_BUTTON(color_picker), title_text);
294 g_free(title_text);
295 gtk_box_pack_start (GTK_BOX (temp_hbox), label, FALSE, FALSE, 6);
296 g_free (label_text);
297
298 cpu_label = gtk_label_new (NULL);
299 gtk_label_set_width_chars (GTK_LABEL (cpu_label), 7);
300 gtk_label_set_xalign (GTK_LABEL (cpu_label), 0.0);
301
302 gtk_box_pack_start (GTK_BOX (temp_hbox), cpu_label, TRUE, TRUE, 0);
303 load_graph_get_labels(cpu_graph)->cpu[i] = cpu_label;
304
305 }
306
307 procdata->cpu_graph = cpu_graph;
308
310 mem_graph_box = GTK_WIDGET (gtk_builder_get_object (builder, "mem_graph_box"));
311
312 mem_graph = new LoadGraph(LOAD_GRAPH_MEM);
313 gtk_box_pack_start (GTK_BOX (mem_graph_box),
314 load_graph_get_widget(mem_graph),
315 TRUE,
316 TRUE,
317 0);
318
319 table = GTK_WIDGET (gtk_builder_get_object (builder, "mem_table"));
320
321 color_picker = load_graph_get_mem_color_picker(mem_graph);
322 g_signal_connect (G_OBJECT (color_picker), "color_set",
323 G_CALLBACK (cb_mem_color_changed), procdata);
324 title_text = g_strdup_printf(title_template, _("Memory"));
325 gsm_color_button_set_title(GSM_COLOR_BUTTON(color_picker), title_text);
326 g_free(title_text);
327
328 gtk_grid_attach (GTK_GRID (table), color_picker, 0, 0, 1, 2);
329
330 label = load_graph_get_labels(mem_graph)->memory;
331 gtk_label_set_xalign(GTK_LABEL(label), 0);
332 gtk_grid_attach (GTK_GRID (table), label, 1, 1, 1, 1);
333
334 color_picker = load_graph_get_swap_color_picker(mem_graph);
335 g_signal_connect (G_OBJECT (color_picker), "color_set",
336 G_CALLBACK (cb_swap_color_changed), procdata);
337 title_text = g_strdup_printf(title_template, _("Swap"));
338 gsm_color_button_set_title(GSM_COLOR_BUTTON(color_picker), title_text);
339 g_free(title_text);
340
341 gtk_grid_attach (GTK_GRID (table), color_picker, 2, 0, 1, 2);
342
343 label = load_graph_get_labels(mem_graph)->swap;
344 gtk_label_set_xalign(GTK_LABEL(label), 0);
345 gtk_grid_attach (GTK_GRID (table), label, 3, 1, 1, 1);
346
347 procdata->mem_graph = mem_graph;
348
349 /* The net box */
350 net_graph_box = GTK_WIDGET (gtk_builder_get_object (builder, "net_graph_box"));
351
352 net_graph = new LoadGraph(LOAD_GRAPH_NET);

355 TRUE,
356 TRUE,
357 0);

361 color_picker = gsm_color_button_new (

399 procdata->net_graph = net_graph;
400 g_free(title_template);
401}

403static void
404menu_item_select_cb (GtkMenuItem *proxy,
405 ProcData *procdata)
406{
407 GtkAction *action;
408 char *message;
409
410 action = gtk_activatable_get_related_action (GTK_ACTIVATABLE(proxy));
411 g_assert(action);
412
413 g_object_get (G_OBJECT (action), "tooltip", &message, NULL);
414 if (message)
415 {
416 gtk_statusbar_push (GTK_STATUSBAR (procdata->statusbar),
417 procdata->tip_message_cid, message);
418 g_free (message);
419 }
420}
421
422static void
423menu_item_deselect_cb (GtkMenuItem *proxy,
424 ProcData *procdata)
425{
426 gtk_statusbar_pop (GTK_STATUSBAR (procdata->statusbar),
427 procdata->tip_message_cid);
428}

430static void
431connect_proxy_cb (GtkUIManager *manager,
432 GtkAction *action,
433 GtkWidget *proxy,
434 ProcData *procdata)
435{
436 if (GTK_IS_MENU_ITEM (proxy)) {
437 g_signal_connect (proxy, "select",
438 G_CALLBACK (menu_item_select_cb), procdata);
439 g_signal_connect (proxy, "deselect",
440 G_CALLBACK (menu_item_deselect_cb), procdata);
441 }
442}
443
444static void
445disconnect_proxy_cb (GtkUIManager *manager,
446 GtkAction *action,
447 GtkWidget *proxy,
448 ProcData *procdata)
449{
450 if (GTK_IS_MENU_ITEM (proxy)) {
451 g_signal_handlers_disconnect_by_func
452 (proxy, (void*)(G_CALLBACK(menu_item_select_cb)), procdata);
453 g_signal_handlers_disconnect_by_func
454 (proxy, (void*)(G_CALLBACK(menu_item_deselect_cb)), procdata);
455 }
456}

458void
459create_main_window (ProcData *procdata)
460{
461 gint i;
462 gint width, height, xpos, ypos;
463 GtkWidget *app;
464 GtkAction *action;
465 GtkWidget *menubar;
466 GtkWidget *main_box;
467 GtkWidget *notebook;
468 GtkBuilder *builder;
469
470 builder = gtk_builder_new_from_resource("/org/mate/mate-system-monitor/interface.ui");
471
472 app = GTK_WIDGET (gtk_builder_get_object (builder, "main_window"));
473 main_box = GTK_WIDGET (gtk_builder_get_object (builder, "main_box"));
474
475 GdkScreen* screen = gtk_widget_get_screen(app);
476 /* use visual, if available */
477 GdkVisual* visual = gdk_screen_get_rgba_visual(screen);
478 if (visual)
479 gtk_widget_set_visual(app, visual);
480
481 width = procdata->config.width;
482 height = procdata->config.height;
483 xpos = procdata->config.xpos;
484 ypos = procdata->config.ypos;
485 gtk_window_set_default_size (GTK_WINDOW (app), width, height);
486 gtk_window_move(GTK_WINDOW (app), xpos, ypos);
487
488 if (procdata->config.maximized) {
489 gtk_window_maximize(GTK_WINDOW(app));
490 }
491
492 /* create the menubar */
493 procdata->uimanager = gtk_ui_manager_new ();
494
495 /* show tooltips in the statusbar */
496 g_signal_connect (procdata->uimanager, "connect_proxy",
497 G_CALLBACK (connect_proxy_cb), procdata);
498 g_signal_connect (procdata->uimanager, "disconnect_proxy",
499 G_CALLBACK (disconnect_proxy_cb), procdata);
500
501 gtk_window_add_accel_group (GTK_WINDOW (app),
502 gtk_ui_manager_get_accel_group (procdata->uimanager));
503
504 if (!gtk_ui_manager_add_ui_from_string (procdata->uimanager,
505 ui_info,
506 -1,
507 NULL)) {
508 g_error("building menus failed");
509 }
510
511 procdata->action_group = gtk_action_group_new ("ProcmanActions");
512 gtk_action_group_set_translation_domain (procdata->action_group, NULL);
513 gtk_action_group_add_actions (procdata->action_group,
514 menu_entries,
515 G_N_ELEMENTS (menu_entries),
516 procdata);
517 gtk_action_group_add_toggle_actions (procdata->action_group,
518 toggle_menu_entries,
519 G_N_ELEMENTS (toggle_menu_entries),
520 procdata);
521
522 gtk_action_group_add_radio_actions (procdata->action_group,
523 radio_menu_entries,
524 G_N_ELEMENTS (radio_menu_entries),
525 procdata->config.whose_process,
526 G_CALLBACK(cb_radio_processes),
527 procdata);
528
529 gtk_action_group_add_radio_actions (procdata->action_group,
530 priority_menu_entries,
531 G_N_ELEMENTS (priority_menu_entries),
532 NORMAL_PRIORITY,
533 G_CALLBACK(cb_renice),
534 procdata);
535
536 gtk_ui_manager_insert_action_group (procdata->uimanager,
537 procdata->action_group,
538 0);
539
540 menubar = gtk_ui_manager_get_widget (procdata->uimanager, "/MenuBar");
541 gtk_box_pack_start (GTK_BOX (main_box), menubar, FALSE, FALSE, 0);
542 gtk_box_reorder_child (GTK_BOX (main_box), menubar, 0);
543
544 /* create the main notebook */
545 procdata->notebook = notebook = GTK_WIDGET (gtk_builder_get_object (builder, "notebook"));
546
547 create_proc_view(procdata, builder);
548 create_sys_view (procdata, builder);
549 create_disk_view (procdata, builder);
550
551 g_signal_connect (G_OBJECT (notebook), "switch-page",
552 G_CALLBACK (cb_switch_page), procdata);
553 g_signal_connect (G_OBJECT (notebook), "change-current-page",
554 G_CALLBACK (cb_change_current_page), procdata);
555
556 gtk_widget_show_all(notebook); // need to make page switch work
557 gtk_notebook_set_current_page (GTK_NOTEBOOK (notebook), procdata->config.current_tab);
558 cb_change_current_page (GTK_NOTEBOOK (notebook), procdata->config.current_tab, procdata);
559 g_signal_connect (G_OBJECT (app), "delete_event",
560 G_CALLBACK (cb_app_delete),
561 procdata);
562
563 GtkAccelGroup *accel_group;
564 GClosure *goto_tab_closure[4];
565 accel_group = gtk_accel_group_new ();
566 gtk_window_add_accel_group (GTK_WINDOW(app), accel_group);
567 for (i = 0; i < 4; ++i) {
568 goto_tab_closure[i] = g_cclosure_new_swap (G_CALLBACK (cb_proc_goto_tab),
569 GINT_TO_POINTER (i), NULL);
570 gtk_accel_group_connect (accel_group, '0'+(i+1),
571 GDK_MOD1_MASK, GTK_ACCEL_VISIBLE,
572 goto_tab_closure[i]);
573 }
574
575 /* create the statusbar */
576 procdata->statusbar = GTK_WIDGET (gtk_builder_get_object (builder, "statusbar"));
