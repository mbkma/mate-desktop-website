27MateWPInfo* mate_wp_info_new(const char* uri, MateDesktopThumbnailFactory* thumbs)
28{
29 MateWPInfo* wp;
30
31 GFile* file = g_file_new_for_commandline_arg(uri);
32
33 GFileInfo* info = g_file_query_info(file,
34 G_FILE_ATTRIBUTE_STANDARD_NAME ","
35 G_FILE_ATTRIBUTE_STANDARD_SIZE ","
36 G_FILE_ATTRIBUTE_STANDARD_CONTENT_TYPE ","
37 G_FILE_ATTRIBUTE_TIME_MODIFIED,
38 G_FILE_QUERY_INFO_NONE,
39 NULL, NULL);
40
41 g_object_unref(file);
42
43 if (info == NULL || g_file_info_get_content_type (info) == NULL)
44 {
45 if (!strcmp (uri, "(none)"))
46 {
47 wp = g_new0(MateWPInfo, 1);
48
49 wp->mime_type = g_strdup("image/x-no-data");
50 wp->uri = g_strdup(uri);
51 wp->name = g_strdup(_("No Desktop Background"));
52 wp->size = 0;
53 }
54 else
55 {
56 wp = NULL;
57 }
58 }
59 else
60 {
61 wp = g_new0 (MateWPInfo, 1);
62
63 wp->uri = g_strdup(uri);
64
65 wp->name = g_strdup(g_file_info_get_name(info));
66
67 if (g_file_info_get_content_type(info) != NULL)
68 {
69 wp->mime_type = g_strdup(g_file_info_get_content_type(info));
70 }
71
72 wp->size = g_file_info_get_size(info);
73 wp->mtime = g_file_info_get_attribute_uint64(info, G_FILE_ATTRIBUTE_TIME_MODIFIED);
74
75 wp->thumburi = mate_desktop_thumbnail_factory_lookup(thumbs, uri, wp->mtime);
76 }
77
78 if (info != NULL)
79 {
80 g_object_unref(info);
81 }
82
83 return wp;
84}
