29static gboolean
30on_button_press (GtkWidget *button,
31 GdkEventButton *event,
32 EomWindow *window)
33{
34 if (event->button == 1 && event->type == GDK_2BUTTON_PRESS)
35 {
36 EomWindowMode mode = eom_window_get_mode (window);
37
38 if (mode == EOM_WINDOW_MODE_SLIDESHOW || mode == EOM_WINDOW_MODE_FULLSCREEN)
39 {
40 eom_window_set_mode (window, EOM_WINDOW_MODE_NORMAL);
41 }
42 else if (mode == EOM_WINDOW_MODE_NORMAL)
43 {
44 eom_window_set_mode (window, EOM_WINDOW_MODE_FULLSCREEN);
45 }
46
47 return TRUE;
48 }
49
50 return FALSE;
51}
