32static GVariant *
33add_to_dict (GVariant *dict, const gchar *schema, const gchar *path)
34{
35 GVariantIter iter;
36 GVariantBuilder builder;
37 gboolean is_schema_found;
38 gboolean is_incorrect_schema;
39 gint path_counter;
40
41 gchar *key;
42 gchar *value;
43
44 g_variant_builder_init (&builder, (const GVariantType *) "a{ss}");
45 g_variant_iter_init (&iter, dict);
46
47 is_schema_found = FALSE;
48 is_incorrect_schema = FALSE;
49 path_counter = 0;
50
51 while (g_variant_iter_next (&iter, "{ss}", &key, &value)) {
52 gboolean path_is_found = FALSE;
53 if (g_strcmp0 (value, path) == 0) {
54 path_is_found = TRUE;
55 path_counter++;
56 if (g_strcmp0 (key, schema) == 0) {
57 is_schema_found = TRUE;
58 } else {
59 // skip incoorect schema for path
60 is_incorrect_schema = TRUE;
61 g_free (key);
62 g_free (value);
63 continue;
64 }
65 }
66
67 gboolean need_add_to_dict = !path_is_found || path_counter < 2;
68
69 if (need_add_to_dict) {
70 g_variant_builder_add (&builder, "{ss}", key, value);
71 }
72
73 g_free (key);
74 g_free (value);
75 }
76
77 if (!is_schema_found) {
78 g_variant_builder_add (&builder, "{ss}", schema, path);
79 }
80
81 if (!is_schema_found || is_incorrect_schema || (path_counter > 1)) {
82 return g_variant_ref_sink (g_variant_builder_end (&builder));
83 } else {
84 g_variant_builder_clear (&builder);
85 // no changes
86 return NULL;
87 }
88}

90static void
91register_dconf_editor_relocatable_schema (const gchar *schema, const gchar *path)
92{
93 GSettingsSchemaSource *source;
94 GSettingsSchema *dconf_editor_schema;
95 GSettings *dconf_editor_settings;
96
97 source = g_settings_schema_source_get_default ();
98
99 if (! source)
100 return;
101
102 dconf_editor_schema = g_settings_schema_source_lookup (source, "ca.desrt.dconf-editor.Settings", FALSE);
103
104 if (! dconf_editor_schema)
105 return;
106
107 dconf_editor_settings = g_settings_new_full (dconf_editor_schema, NULL, NULL);
108
109 if (dconf_editor_settings && g_settings_is_writable (dconf_editor_settings, "relocatable-schemas-user-paths")) {
110 GVariant *relocatable_schemas = g_settings_get_value (dconf_editor_settings, "relocatable-schemas-user-paths");
111
112 if (g_variant_is_of_type (relocatable_schemas, G_VARIANT_TYPE_DICTIONARY)) {
113 GVariant * new_relocatable_schemas = add_to_dict (relocatable_schemas, schema, path);
114 if (new_relocatable_schemas) {
115 g_settings_set_value (dconf_editor_settings, "relocatable-schemas-user-paths", new_relocatable_schemas);
116 g_variant_unref (new_relocatable_schemas);
117 }
118 }
119
120 g_variant_unref (relocatable_schemas);
121 }
122
123 g_object_unref (dconf_editor_settings);
124 g_settings_schema_unref (dconf_editor_schema);
125}

127GSettings *
128mate_panel_applet_settings_new (MatePanelApplet *applet, gchar *schema)
129{
130 GSettings *settings = NULL;
131 gchar *path;
132
133 g_return_val_if_fail (MATE_PANEL_IS_APPLET (applet), NULL);
134
135 path = mate_panel_applet_get_preferences_path (applet);
136
137 if (path) {
138 settings = g_settings_new_with_path (schema, path);
139 register_dconf_editor_relocatable_schema (schema, path);
140 g_free (path);
141 }
142
143 return settings;
144}

146GList*
147mate_panel_applet_settings_get_glist (GSettings *settings, gchar *key)
148{
149 gchar **array;
150 GList *list = NULL;
151
152 array = g_settings_get_strv (settings, key);
153 if (array != NULL) {
154 for (gint i = 0; array[i]; i++) {
155 list = g_list_prepend (list, array[i]);
156 }
157 g_free (array);
158 }
159 return g_list_reverse (list);
160}
