191static void
192test_basic_fitting (void)
193{
194 MetaRectangle temp1, temp2, temp3;
195 int i;
196 /* Four cases:
197 * case temp1 fits temp2 temp1 could fit temp2
198 * 1 Y Y
199 * 2 N Y
200 * 3 Y N
201 * 4 N N
202 * Of the four cases, case 3 is impossible. An alternate way of looking
203 * at this table is that either the middle column must be no, or the last
204 * column must be yes. So we test that. Also, we can repeat the test
205 * reversing temp1 and temp2.
206 */
207 for (i = 0; i < NUM_RANDOM_RUNS; i++)
208 {
209 get_random_rect (&temp1);
210 get_random_rect (&temp2);
211 g_assert (!meta_rectangle_contains_rect (&temp1, &temp2) ||
212 meta_rectangle_could_fit_rect (&temp1, &temp2));
213 g_assert (!meta_rectangle_contains_rect (&temp2, &temp1) ||
214 meta_rectangle_could_fit_rect (&temp2, &temp1));
215 }
216
217 temp1 = meta_rect ( 0, 0, 10, 10);
218 temp2 = meta_rect ( 5, 5, 5, 5);
219 temp3 = meta_rect ( 8, 2, 3, 7);
220 g_assert ( meta_rectangle_contains_rect (&temp1, &temp2));
221 g_assert (!meta_rectangle_contains_rect (&temp2, &temp1));
222 g_assert (!meta_rectangle_contains_rect (&temp1, &temp3));
223 g_assert ( meta_rectangle_could_fit_rect (&temp1, &temp3));
224 g_assert (!meta_rectangle_could_fit_rect (&temp3, &temp2));
225
226 printf ("%s passed.\n", G_STRFUNC);
227}

978static void
979verify_edge_lists_are_equal (GList *code, GList *answer)
980{
981 int which = 0;
982
983 while (code && answer)
984 {
985 MetaEdge *a = code->data;
986 MetaEdge *b = answer->data;
987
988 if (!meta_rectangle_equal (&a->rect, &b->rect) ||
989 a->side_type != b->side_type ||
990 a->edge_type != b->edge_type)
991 {
992 g_error ("%dth item in code answer answer lists do not match; "
993 "code rect: %d,%d + %d,%d; answer rect: %d,%d + %d,%d\n",
994 which,
995 a->rect.x, a->rect.y, a->rect.width, a->rect.height,
996 b->rect.x, b->rect.y, b->rect.width, b->rect.height);
997 }
998
999 code = code->next;
1000 answer = answer->next;
1001
1002 which++;
1003 }
1004
1005 /* Ought to be at the end of both lists; check if we aren't */
1006 if (code)
1007 {
1008 MetaEdge *tmp = code->data;
1009 g_error ("code list longer than answer list by %d items; "
1010 "first extra item rect: %d,%d +%d,%d\n",
1011 g_list_length (code),
1012 tmp->rect.x, tmp->rect.y, tmp->rect.width, tmp->rect.height);
1013 }
1014
1015 if (answer)
1016 {
1017 MetaEdge *tmp = answer->data;
1018 g_error ("answer list longer than code list by %d items; "
1019 "first extra item rect: %d,%d +%d,%d\n",
1020 g_list_length (answer),
1021 tmp->rect.x, tmp->rect.y, tmp->rect.width, tmp->rect.height);
1022 }
