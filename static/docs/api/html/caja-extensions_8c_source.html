36static Extension *
37extension_new (gchar *filename, gboolean state, gboolean python, GObject *module)
38{
39 GError *error = NULL;
40 Extension *ext;
41 GKeyFile *extension_file;
42 gchar *extension_filename;
43
44 ext = g_new0 (Extension, 1);
45 ext->filename = filename;
46 ext->name = NULL;
47 ext->description = NULL;
48 ext->author = NULL;
49 ext->copyright = NULL;
50 ext->version = NULL;
51 ext->website = NULL;
52 ext->state = state;
53 ext->module = module;
54
55 extension_file = g_key_file_new ();
56 extension_filename = g_strdup_printf(CAJA_DATADIR "/extensions/%s.caja-extension", filename);
57 if (g_key_file_load_from_file (extension_file, extension_filename, G_KEY_FILE_NONE, &error))
58 {
59 ext->name = g_key_file_get_locale_string (extension_file, CAJA_EXTENSION_GROUP, "Name", NULL, NULL);
60 ext->description = g_key_file_get_locale_string (extension_file, CAJA_EXTENSION_GROUP, "Description", NULL, NULL);
61 ext->icon = g_key_file_get_string (extension_file, CAJA_EXTENSION_GROUP, "Icon", NULL);
62 ext->author = g_key_file_get_string_list (extension_file, CAJA_EXTENSION_GROUP, "Author", NULL, NULL);
63 ext->copyright = g_key_file_get_locale_string (extension_file, CAJA_EXTENSION_GROUP, "Copyright", NULL, NULL);
64 ext->version = g_key_file_get_string (extension_file, CAJA_EXTENSION_GROUP, "Version", NULL);
65 ext->website = g_key_file_get_string (extension_file, CAJA_EXTENSION_GROUP, "Website", NULL);
66 g_key_file_free (extension_file);
67 }
68 else
69 {
70 caja_debug_log (FALSE, CAJA_DEBUG_LOG_DOMAIN_USER, "Error loading keys from file \"%s\": %s\n", extension_filename, error->message);
71 g_error_free (error);
72 }
73 g_free (extension_filename);
74
75 if (python)
76 {
77 ext->name = g_strconcat("Python: ", filename, NULL);
78 ext->description = "Python-caja extension";
79 }
80
81 return ext;
82}

86static gboolean
87gsettings_key_has_value (const gchar *value)
88{
89 gchar **list;
90 gint i;
91
92 list = g_settings_get_strv (caja_extension_preferences,
93 CAJA_PREFERENCES_DISABLED_EXTENSIONS);
94
95 if (list != NULL)
96 {
97 for (i = 0; list[i]; i++)
98 {
99 if (g_ascii_strcasecmp (value, list[i]) == 0)
100 {
101 g_strfreev (list);
102 return TRUE;
103 }
104 }
105 }
106 g_strfreev (list);
107 return FALSE;
108}

110static gboolean
111gsettings_append_to_list (const char *value)
112{
113 gchar **current;
114 gchar **new;
115 gint size;
116 gboolean retval;
117
118 current = g_settings_get_strv (caja_extension_preferences,
119 CAJA_PREFERENCES_DISABLED_EXTENSIONS);
120
121 for (size = 0; current[size] != NULL; size++);
122
123 size += 1;
124 size += 1;
125
126 new = g_realloc_n (current, size, sizeof (gchar *));
127
128 new[size - 2] = g_strdup (value);
129 new[size - 1] = NULL;
130
131 retval = g_settings_set_strv (caja_extension_preferences,
132 CAJA_PREFERENCES_DISABLED_EXTENSIONS,
133 (const gchar **) new);
134
135 g_strfreev (new);
136 return retval;
137}

139static gboolean
140gsettings_remove_from_list (const char *value)
141{
142 gchar **current;
143 GArray *array;
144 gint i;
145 gboolean retval;
146
147 current = g_settings_get_strv (caja_extension_preferences,
148 CAJA_PREFERENCES_DISABLED_EXTENSIONS);
149
150 array = g_array_new (TRUE, TRUE, sizeof (gchar *));
151
152 for (i = 0; current[i] != NULL; i++)
153 {
154 if (g_strcmp0 (current[i], value) != 0)
155 array = g_array_append_val (array, current[i]);
156 }
157
158 retval = g_settings_set_strv (caja_extension_preferences,
159 CAJA_PREFERENCES_DISABLED_EXTENSIONS,
160 (const gchar **) array->data);
161
162 g_strfreev (current);
163 g_array_free (array, TRUE);
164 return retval;
165}

175GList *
176caja_extensions_get_for_type (GType type)
177{
178 GList *l;
179 GList *ret = NULL;
180
181 for (l = caja_extensions; l != NULL; l = l->next)
182 {
183 Extension *ext = l->data;
184 ext->state = caja_extension_get_state (ext->filename);
185 if (ext->state) // only load enabled extensions
186 {
187 if (G_TYPE_CHECK_INSTANCE_TYPE (G_OBJECT (ext->module), type))
188 {
189 g_object_ref (ext->module);
190 ret = g_list_prepend (ret, ext->module);
191 }
192 }
193 }
194
195 return ret;
196}

204void
205caja_extension_register (gchar *filename, GObject *module)
206{
207 gboolean ext_state = TRUE; // new extensions are enabled by default.
208 gboolean ext_python = FALSE;
209 gchar *ext_filename;
210
211 ext_filename = g_strndup (filename, strlen(filename) - 3);
212 ext_state = caja_extension_get_state (ext_filename);
213
214 if (g_str_has_suffix (filename, ".py")) {
215 ext_python = TRUE;
216 }
217
218 Extension *ext = extension_new (ext_filename, ext_state, ext_python, module);
219 caja_extensions = g_list_append (caja_extensions, ext);
220}

222gboolean
223caja_extension_set_state (Extension *ext, gboolean new_state)
224{
225 if (ext)
226 {
227 g_return_val_if_fail (ext->state != new_state, FALSE);
228 ext->state = new_state;
229 }
230
231 gboolean retval;
232 if (new_state) {
233 retval = gsettings_remove_from_list (ext->filename);
234 }
235 else {
236 retval = gsettings_append_to_list (ext->filename);
237 }
238
239 g_return_val_if_fail (retval == TRUE, FALSE);
240 return TRUE;
241}
