189static GHashTable* pnp_ids = NULL;
190
191static void read_pnp_ids(void)
192{
193 gchar* contents;
194 gchar** lines;
195 gchar* line;
196 gchar* code, *name;
197 gint i;
198
199 if (pnp_ids)
200 return;
201
202 pnp_ids = g_hash_table_new_full(g_str_hash, g_str_equal, g_free, NULL);
203
204 if (g_file_get_contents(PNP_IDS, &contents, NULL, NULL))
205 {
206 lines = g_strsplit(contents, "\n", -1);
207
208 for (i = 0; lines[i]; i++)
209 {
210 line = lines[i];
211
212 if (line[0] && line[1] && line[2] && line[3] == '\t' && line[4])
213 {
214 code = line;
215 line[3] = '\0';
216 name = line + 4;
217 g_hash_table_insert(pnp_ids, code, name);
218 }
219 }
220
221 g_free(lines);
222 g_free(contents);
223 }
224}

226static const char* find_vendor(const char* code)
227{
228 const char* vendor_name;
229 gsize i;
230
231 read_pnp_ids();
232
233 vendor_name = g_hash_table_lookup(pnp_ids, code);
234
235 if (vendor_name)
236 return vendor_name;
237
238 for (i = 0; i < G_N_ELEMENTS (vendors); ++i)
239 {
240 const Vendor* v = &(vendors[i]);
241
242 if (strcmp(v->vendor_id, code) == 0)
243 return v->vendor_name;
244 }
245
246 return code;
247}

249char* make_display_name(const MonitorInfo* info)
250{
251 const char* vendor;
252 int width_mm, height_mm, inches;
253
254 if (info)
255 {
256 vendor = find_vendor(info->manufacturer_code);
257 }
258 else
259 {
260 /* Translators: "Unknown" here is used to identify a monitor for which
261 * we don't know the vendor. When a vendor is known, the name of the
262 * vendor is used. */
263 vendor = C_("Monitor vendor", "Unknown");
264 }
265
266 if (info && info->width_mm != -1 && info->height_mm)
267 {
268 width_mm = info->width_mm;
269 height_mm = info->height_mm;
270 }
271 else if (info && info->n_detailed_timings)
272 {
273 width_mm = info->detailed_timings[0].width_mm;
274 height_mm = info->detailed_timings[0].height_mm;
275 }
276 else
277 {
278 width_mm = -1;
279 height_mm = -1;
280 }
281
282 if (width_mm != -1 && height_mm != -1)
283 {
284 double d = sqrt (width_mm * width_mm + height_mm * height_mm);
285
286 inches = (int) (d / 25.4 + 0.5);
287 }
288 else
289 {
290 inches = -1;
291 }
292
293 if (inches > 0)
294 {
295 return g_strdup_printf("%s %d\"", vendor, inches);
296 }
297 else
298 {
299 return g_strdup(vendor);
300 }
301}
