62static void
63simulate_user_activity (MsdSmartcardPlugin *plugin)
64{
65 GError *error = NULL;
66 GVariant *ret;
67
68 g_debug ("MsdSmartcardPlugin telling screensaver about smart card insertion");
69 ret = g_dbus_proxy_call_sync (plugin->priv->screensaver_proxy,
70 "SimulateUserActivity",
71 g_variant_new ("()"),
72 G_DBUS_CALL_FLAGS_NONE,
73 -1,
74 NULL,
75 &error);
76 if (ret == NULL) {
77 g_warning ("MsdSmartcardPlugin Unable to force logout: %s", error->message);
78 g_error_free (error);
79 } else {
80 g_variant_unref (ret);
81 }
82}

107static void
108force_logout (MsdSmartcardPlugin *plugin)
109{
110 GDBusProxy *sm_proxy;
111 GError *error = NULL;
112 GVariant *ret;
113
114 g_debug ("MsdSmartcardPlugin telling session manager to force logout");
115 sm_proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SESSION,
116 G_DBUS_PROXY_FLAGS_NONE,
117 NULL,
118 SM_DBUS_NAME,
119 SM_DBUS_PATH,
120 SM_DBUS_INTERFACE,
121 NULL,
122 &error);
123 if (sm_proxy == NULL) {
124 g_warning ("Unable to contact session manager daemon: %s\n", error->message);
125 g_error_free (error);
126 return;
127 }
128
129 ret = g_dbus_proxy_call_sync (sm_proxy,
130 "Logout",
131 g_variant_new ("(u", SM_LOGOUT_MODE_FORCE),
132 G_DBUS_CALL_FLAGS_NONE,
133 -1,
134 NULL,
135 &error);
136 if (ret == NULL) {
137 g_warning ("MsdSmartcardPlugin Unable to force logout: %s", error->message);
138 g_error_free (error);
139 } else {
140 g_variant_unref (ret);
141 }
142
143 g_object_unref (sm_proxy);
144}

197static MsdSmartcardRemoveAction
198get_configured_remove_action (MsdSmartcardPlugin *plugin)
199{
200 GSettings *settings;
201 char *remove_action_string;
202 MsdSmartcardRemoveAction remove_action;
203
204 settings = g_settings_new (MSD_SMARTCARD_SCHEMA);
205 remove_action_string = g_settings_get_string (settings,
206 KEY_REMOVE_ACTION);
207
208 if (remove_action_string == NULL) {
209 g_warning ("MsdSmartcardPlugin unable to get smartcard remove action");
210 remove_action = MSD_SMARTCARD_REMOVE_ACTION_NONE;
211 } else if (strcmp (remove_action_string, "none") == 0) {
212 remove_action = MSD_SMARTCARD_REMOVE_ACTION_NONE;
213 } else if (strcmp (remove_action_string, "lock_screen") == 0) {
214 remove_action = MSD_SMARTCARD_REMOVE_ACTION_LOCK_SCREEN;
215 } else if (strcmp (remove_action_string, "force_logout") == 0) {
216 remove_action = MSD_SMARTCARD_REMOVE_ACTION_FORCE_LOGOUT;
217 } else {
218 g_warning ("MsdSmartcardPlugin unknown smartcard remove action");
219 remove_action = MSD_SMARTCARD_REMOVE_ACTION_NONE;
220 }
221
222 g_object_unref (settings);
223
224 return remove_action;
225}

227static void
228process_smartcard_removal (MsdSmartcardPlugin *plugin)
229{
230 MsdSmartcardRemoveAction remove_action;
231
232 g_debug ("MsdSmartcardPlugin processing smartcard removal");
233 remove_action = get_configured_remove_action (plugin);
234
235 switch (remove_action)
236 {
237 case MSD_SMARTCARD_REMOVE_ACTION_NONE:
238 return;
239 case MSD_SMARTCARD_REMOVE_ACTION_LOCK_SCREEN:
240 lock_screen (plugin);
241 break;
242 case MSD_SMARTCARD_REMOVE_ACTION_FORCE_LOGOUT:
243 force_logout (plugin);
244 break;
245 }
246}

268static void
269impl_activate (MateSettingsPlugin *plugin)
270{
271 GError *error;
272 MsdSmartcardPlugin *smartcard_plugin = MSD_SMARTCARD_PLUGIN (plugin);
273
274 if (smartcard_plugin->priv->is_active) {
275 g_debug ("MsdSmartcardPlugin Not activating smartcard plugin, because it's "
276 "already active");
277 return;
278 }
279
280 if (!user_logged_in_with_smartcard ()) {
281 g_debug ("MsdSmartcardPlugin Not activating smartcard plugin, because user didn't use "
282 " smartcard to log in");
283 smartcard_plugin->priv->is_active = FALSE;
284 return;
285 }
286
287 g_debug ("MsdSmartcardPlugin Activating smartcard plugin");
288
289 error = NULL;
290 smartcard_plugin->priv->screensaver_proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SESSION,
291 G_DBUS_PROXY_FLAGS_NONE,
292 NULL,
293 SCREENSAVER_DBUS_NAME,
294 SCREENSAVER_DBUS_PATH,
295 SCREENSAVER_DBUS_INTERFACE,
296 NULL,
297 &error);
298
299 if (smartcard_plugin->priv->screensaver_proxy == NULL) {
300 g_warning ("MsdSmartcardPlugin Unable to connect to session bus: %s", error->message);
301 g_error_free (error);
302 return;
303 }
304
305 if (!msd_smartcard_manager_start (smartcard_plugin->priv->manager, &error)) {
306 g_warning ("MsdSmartcardPlugin Unable to start smartcard manager: %s", error->message);
307 g_error_free (error);
308 }
309
310 g_signal_connect (smartcard_plugin->priv->manager,
311 "smartcard-removed",
312 G_CALLBACK (smartcard_removed_cb), smartcard_plugin);
313
314 g_signal_connect (smartcard_plugin->priv->manager,
315 "smartcard-inserted",
316 G_CALLBACK (smartcard_inserted_cb), smartcard_plugin);
317
318 if (!msd_smartcard_manager_login_card_is_inserted (smartcard_plugin->priv->manager)) {
319 g_debug ("MsdSmartcardPlugin processing smartcard removal immediately user logged in with smartcard "
320 "and it's not inserted");
321 process_smartcard_removal (smartcard_plugin);
322 }
323
324 smartcard_plugin->priv->is_active = TRUE;
325}

327static void
328impl_deactivate (MateSettingsPlugin *plugin)
329{
330 MsdSmartcardPlugin *smartcard_plugin = MSD_SMARTCARD_PLUGIN (plugin);
331
332 if (!smartcard_plugin->priv->is_active) {
333 g_debug ("MsdSmartcardPlugin Not deactivating smartcard plugin, "
334 "because it's already inactive");
335 return;
336 }
337
338 g_debug ("MsdSmartcardPlugin Deactivating smartcard plugin");
339
340 msd_smartcard_manager_stop (smartcard_plugin->priv->manager);
341
342 g_signal_handlers_disconnect_by_func (smartcard_plugin->priv->manager,
343 smartcard_removed_cb, smartcard_plugin);
344
345 g_signal_handlers_disconnect_by_func (smartcard_plugin->priv->manager,
346 smartcard_inserted_cb, smartcard_plugin);
347 if (smartcard_plugin->priv->screensaver_proxy != NULL)
348 g_object_unref (smartcard_plugin->priv->screensaver_proxy);
349 smartcard_plugin->priv->is_active = FALSE;
350}
