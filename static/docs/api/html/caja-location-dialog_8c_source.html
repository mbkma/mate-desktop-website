67static void
68open_current_location (CajaLocationDialog *dialog)
69{
70 GFile *location;
71 char *user_location;
72
73 user_location = gtk_editable_get_chars (GTK_EDITABLE (dialog->details->entry), 0, -1);
74 location = g_file_parse_name (user_location);
75 caja_window_go_to (dialog->details->window, location);
76 g_object_unref (location);
77 g_free (user_location);
78}

80static void
81response_callback (CajaLocationDialog *dialog,
82 int response_id,
83 gpointer data)
84{
85 GError *error;
86
87 switch (response_id)
88 {
89 case RESPONSE_OPEN :
90 open_current_location (dialog);
91 gtk_widget_destroy (GTK_WIDGET (dialog));
92 break;
93 case GTK_RESPONSE_NONE :
94 case GTK_RESPONSE_DELETE_EVENT :
95 case GTK_RESPONSE_CANCEL :
96 gtk_widget_destroy (GTK_WIDGET (dialog));
97 break;
98 case GTK_RESPONSE_HELP :
99 error = NULL;
100 gtk_show_uri_on_window (GTK_WINDOW (dialog),
101 "help:mate-user-guide/caja-open-location",
102 gtk_get_current_event_time (), &error);
103 if (error)
104 {
105 eel_show_error_dialog (_("There was an error displaying help."), error->message,
106 GTK_WINDOW (dialog));
107 g_error_free (error);
108 }
109 break;
110 default :
111 g_assert_not_reached ();
112 }
113}

152static void
153caja_location_dialog_init (CajaLocationDialog *dialog)
154{
155 GtkWidget *box;
156 GtkWidget *label;
157
158 dialog->details = g_new0 (CajaLocationDialogDetails, 1);
159
160 gtk_window_set_title (GTK_WINDOW (dialog), _("Open Location"));
161 gtk_window_set_default_size (GTK_WINDOW (dialog), 300, -1);
162 gtk_window_set_destroy_with_parent (GTK_WINDOW (dialog), TRUE);
163 gtk_container_set_border_width (GTK_CONTAINER (dialog), 5);
164 gtk_box_set_spacing (GTK_BOX (gtk_dialog_get_content_area (GTK_DIALOG (dialog))), 2);
165
166 box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 12);
167 gtk_container_set_border_width (GTK_CONTAINER (box), 5);
168 gtk_widget_show (box);
169
170 label = gtk_label_new_with_mnemonic (_("_Location:"));
171 gtk_widget_show (label);
172 gtk_box_pack_start (GTK_BOX (box), label, FALSE, FALSE, 0);
173
174 dialog->details->entry = caja_location_entry_new ();
175 gtk_entry_set_width_chars (GTK_ENTRY (dialog->details->entry), 30);
176 g_signal_connect_after (dialog->details->entry,
177 "activate",
178 G_CALLBACK (entry_activate_callback),
179 dialog);
180
181 gtk_widget_show (dialog->details->entry);
182
183 gtk_label_set_mnemonic_widget (GTK_LABEL (label), dialog->details->entry);
184
185 gtk_box_pack_start (GTK_BOX (box), dialog->details->entry,
186 TRUE, TRUE, 0);
187
188 gtk_box_pack_start (GTK_BOX (gtk_dialog_get_content_area (GTK_DIALOG (dialog))),
189 box, FALSE, TRUE, 0);
190
191 eel_dialog_add_button (GTK_DIALOG (dialog),
192 _("_Help"),
193 "help-browser",
194 GTK_RESPONSE_HELP);
195
196 eel_dialog_add_button (GTK_DIALOG (dialog),
197 _("_Cancel"),

201 eel_dialog_add_button (GTK_DIALOG (dialog),
202 _("_Open"),
203 "document-open",
204 RESPONSE_OPEN);
205
206 gtk_dialog_set_default_response (GTK_DIALOG (dialog),
207 RESPONSE_OPEN);
208
209 g_signal_connect (dialog->details->entry, "notify::text",
210 G_CALLBACK (entry_text_changed), dialog);
211
212 g_signal_connect (dialog, "response",
213 G_CALLBACK (response_callback),

217GtkWidget *
218caja_location_dialog_new (CajaWindow *window)
219{
220 CajaLocationDialog *loc_dialog;
221 GtkWidget *dialog;
222 GFile *location;
223
224 dialog = gtk_widget_new (CAJA_TYPE_LOCATION_DIALOG, NULL);
225 loc_dialog = CAJA_LOCATION_DIALOG (dialog);
226
227 if (window)
228 {
229 gtk_window_set_transient_for (GTK_WINDOW (dialog), GTK_WINDOW (window));
230 gtk_window_set_screen (GTK_WINDOW (dialog),
231 gtk_window_get_screen (GTK_WINDOW (window)));
232 loc_dialog->details->window = window;
233 location = window->details->active_pane->active_slot->location;
234 }
235 else
236 location = NULL;
237
238 if (location != NULL)
239 {
240 char *formatted_location;
241
242 if (CAJA_IS_DESKTOP_WINDOW (window))
243 {
244 formatted_location = g_strdup_printf ("%s/", g_get_home_dir ());
245 }
246 else
247 {
248 formatted_location = g_file_get_parse_name (location);
249 }
250 caja_location_entry_update_current_location (CAJA_LOCATION_ENTRY (loc_dialog->details->entry),
251 formatted_location);
252 g_free (formatted_location);
253 }
254
255 gtk_widget_grab_focus (loc_dialog->details->entry);
256
257 return dialog;
258}
