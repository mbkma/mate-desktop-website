66void
67orientation_change (AppletInfo *info,
68 PanelWidget *panel)
69{
70 PanelOrientation orientation;
71
72 orientation = panel_widget_get_applet_orientation (panel);
73
74 switch (info->type) {
75 case PANEL_OBJECT_APPLET:
76 mate_panel_applet_frame_change_orientation (
77 MATE_PANEL_APPLET_FRAME (info->widget), orientation);
78 break;
79 case PANEL_OBJECT_MENU:
80 case PANEL_OBJECT_LAUNCHER:
81 case PANEL_OBJECT_ACTION:
82 button_widget_set_orientation (BUTTON_WIDGET (info->widget), orientation);
83 break;
84 case PANEL_OBJECT_MENU_BAR:
85 panel_menu_bar_set_orientation (PANEL_MENU_BAR (info->widget), orientation);
86 break;
87 case PANEL_OBJECT_DRAWER: {
88 Drawer *drawer = info->data;
89 PanelWidget *panel_widget;
90
91 panel_widget = panel_toplevel_get_panel_widget (drawer->toplevel);
92
93 button_widget_set_orientation (BUTTON_WIDGET (info->widget), orientation);
94
95 gtk_widget_queue_resize (GTK_WIDGET (drawer->toplevel));
96 gtk_container_foreach (GTK_CONTAINER (panel_widget),
97 orient_change_foreach,
98 panel_widget);
99 }
100 break;
101 case PANEL_OBJECT_SEPARATOR:
102 panel_separator_set_orientation (PANEL_SEPARATOR (info->widget),
103 orientation);
104 break;
105 default:
106 break;
107 }
108}

339static gboolean
340panel_popup_menu (PanelToplevel *toplevel,
341 guint button,
342 guint32 activate_time)
343{
344 PanelWidget *panel_widget;
345 GtkWidget *menu;
346 PanelData *panel_data;
347 GdkEvent *current_event;
348
349 panel_widget = panel_toplevel_get_panel_widget (toplevel);
350 panel_data = g_object_get_data (G_OBJECT (toplevel), "PanelData");
351
352 current_event = gtk_get_current_event ();
353 if (current_event && current_event->type == GDK_BUTTON_PRESS)
354 panel_data->insertion_pos = panel_widget_get_cursorloc (panel_widget);
355 else
356 panel_data->insertion_pos = -1;
357
358 if (current_event)
359 gdk_event_free (current_event);
360
361 menu = make_popup_panel_menu (panel_widget);
362 if (!menu)
363 return FALSE;
364
365 gtk_menu_set_screen (GTK_MENU (menu),
366 gtk_window_get_screen (GTK_WINDOW (toplevel)));
367
368 gtk_window_set_attached_to (GTK_WINDOW (gtk_widget_get_toplevel (menu)),
369 GTK_WIDGET (toplevel));
370 gtk_menu_popup_at_pointer (GTK_MENU (menu), NULL);
371
372 return TRUE;
373}

391static gboolean
392panel_key_press_event (GtkWidget *widget,
393 GdkEventKey *event)
394{
395#ifdef HAVE_X11
396 /*
397 * If the focus widget is a GtkSocket, i.e. the
398 * focus is in an applet in another process, then key
399 * bindings do not work. We get around this by
400 * activating the key bindings here.
401 *
402 * Will always be false when not using X
403 */
404 if (GTK_IS_SOCKET (gtk_window_get_focus (GTK_WINDOW (widget))) &&
405 event->keyval == GDK_KEY_F10 &&
406 (event->state & gtk_accelerator_get_default_mod_mask ()) == GDK_CONTROL_MASK)
407 return gtk_bindings_activate (G_OBJECT (widget),
408 event->keyval,
409 event->state);
410#endif
411
412 return FALSE;
413}

481static gboolean
482drop_url (PanelWidget *panel,
483 int position,
484 const char *url)
485{
486 enum {
487 NETSCAPE_URL_URL,
488 NETSCAPE_URL_NAME
489 };
490 char **netscape_url;
491 char *name;
492 char *comment;
493
494 g_return_val_if_fail (url != NULL, FALSE);
495
496 if (!panel_profile_id_lists_are_writable ())
497 return FALSE;
498
499 netscape_url = g_strsplit (url, "\n", 2);
500 if (!netscape_url ||
501 PANEL_GLIB_STR_EMPTY (netscape_url[NETSCAPE_URL_URL])) {
502 g_strfreev (netscape_url);
503 return FALSE;
504 }
505
506 comment = g_strdup_printf (_("Open URL: %s"),
507 netscape_url[NETSCAPE_URL_URL]);
508
509 if (PANEL_GLIB_STR_EMPTY (netscape_url[NETSCAPE_URL_NAME]))
510 name = netscape_url[NETSCAPE_URL_URL];
511 else
512 name = netscape_url[NETSCAPE_URL_NAME];
513
514 panel_launcher_create_from_info (panel->toplevel, position, FALSE,
515 netscape_url[NETSCAPE_URL_URL],
516 name, comment, PANEL_ICON_REMOTE);
517
518 g_free (comment);
519 g_strfreev (netscape_url);
520
521 return TRUE;
522}

542static gboolean
543drop_uri (PanelWidget *panel,
544 int position,
545 const char *uri,
546 const char *fallback_icon)
547{
548 char *name;
549 char *comment;
550 char *buf;
551 char *icon;
552 GFile *file;
553
554 if (!panel_profile_id_lists_are_writable ())
555 return FALSE;
556
557 name = panel_util_get_label_for_uri (uri);
558 icon = panel_util_get_icon_for_uri (uri);
559 if (!icon)
560 icon = g_strdup (fallback_icon);
561
562 /* FIXME: we might get icons like "folder-music" that might not exist in
563 * the icon theme. This would usually be okay if we could use fallback
564 * icons (and get "folder" this way). However, this is not possible for
565 * launchers: this could be an application that uses an icon named
566 * folder-magic-app, for which we don't want fallbacks. We just want to
567 * go to hicolor. */
568
569 file = g_file_new_for_uri (uri);
570 buf = g_file_get_parse_name (file);
571 g_object_unref (file);
572 /* Translators: %s is a URI */
573 comment = g_strdup_printf (_("Open '%s'"), buf);
574 g_free (buf);
575
576 panel_launcher_create_from_info (panel->toplevel, position, FALSE,
577 uri, name, comment, icon);
578
579 g_free (name);
580 g_free (comment);
581 g_free (icon);
582
583 return TRUE;
584}

586static gboolean
587drop_caja_desktop_uri (PanelWidget *panel,
588 int pos,
589 const char *uri)
590{
591 gboolean success;
592 const char *id;
593 const char *basename;
594
595 if (g_ascii_strncasecmp (uri, "x-caja-desktop:///",
596 strlen ("x-caja-desktop:///")) != 0)
597 return FALSE;
598
599 success = TRUE;
600 id = panel_profile_get_toplevel_id (panel->toplevel);
601 basename = uri + strlen ("x-caja-desktop:///");
602
603 if (strncmp (basename, "trash", strlen ("trash")) == 0)
604 mate_panel_applet_frame_create (panel->toplevel, pos,
605 "OAFIID:MATE_Panel_TrashApplet");
606 else if (strncmp (basename, "home", strlen ("home")) == 0)
607 panel_launcher_create_with_id (id, pos,
608 "caja-home.desktop");
609 else if (strncmp (basename, "computer", strlen ("computer")) == 0)
610 panel_launcher_create_with_id (id, pos,
611 "caja-computer.desktop");
612 else if (strncmp (basename, "network", strlen ("network")) == 0)
613 panel_launcher_create_with_id (id, pos,
614 "caja-scheme.desktop");
615 else
616 success = FALSE;
617
618 return success;
619}

710 toplevel);
711

741static gboolean
742drop_urilist (PanelWidget *panel,
743 int pos,
744 char *urilist)
745{
746 char **uris;
747 guint n_uris;
748 gboolean success;
749 int i;
750
751 uris = g_uri_list_extract_uris (urilist);
752 n_uris = g_strv_length (uris);
753
754 success = TRUE;
755 for (i = 0; uris[i]; i++) {
756 GFile *file;
757 GFileInfo *info;
758 const char *uri;
759
760 uri = uris[i];
761
762 if (g_ascii_strncasecmp (uri, "http:", strlen ("http:")) == 0 ||
763 g_ascii_strncasecmp (uri, "https:", strlen ("https:")) == 0 ||
764 g_ascii_strncasecmp (uri, "ftp:", strlen ("ftp:")) == 0 ||
765 g_ascii_strncasecmp (uri, "gopher:", strlen ("gopher:")) == 0 ||
766 g_ascii_strncasecmp (uri, "ghelp:", strlen ("ghelp:")) == 0 ||
767 g_ascii_strncasecmp (uri, "help:", strlen ("help:")) == 0 ||
768 g_ascii_strncasecmp (uri, "man:", strlen ("man:")) == 0 ||
769 g_ascii_strncasecmp (uri, "info:", strlen ("info:")) == 0) {
770 /* FIXME: probably do this only on link,
771 * in fact, on link always set up a link,
772 * on copy do all the other stuff. Or something. */
773 if ( ! drop_url (panel, pos, uri))
774 success = FALSE;
775 continue;
776 }
777
778 if (g_ascii_strncasecmp (uri, "x-caja-desktop:",
779 strlen ("x-caja-desktop:")) == 0) {
780 success = drop_caja_desktop_uri (panel, pos, uri);
781 continue;
782 }
783
784 file = g_file_new_for_uri (uri);
785 info = g_file_query_info (file,
786 "standard::type,"
787 "standard::content-type,"
788 "access::can-execute",
789 G_FILE_QUERY_INFO_NONE,
790 NULL, NULL);
791
792 if (info) {
793 const char *mime;
794 GFileType type;
795 gboolean can_exec;
796
797 mime = g_file_info_get_content_type (info);
798 type = g_file_info_get_file_type (info);
799 can_exec = g_file_info_get_attribute_boolean (info,
800 G_FILE_ATTRIBUTE_ACCESS_CAN_EXECUTE);
801
802 if (n_uris == 1 &&
803 mime && g_str_has_prefix (mime, "image")) {
804 if (! ask_about_image_drop (panel, pos, uri))
805 success = FALSE;
806 } else if (mime &&
807 (!strcmp (mime, "application/x-mate-app-info") ||
808 !strcmp (mime, "application/x-desktop") ||
809 !strcmp (mime, "application/x-kde-app-info"))) {
810 if (panel_profile_id_lists_are_writable ())
811 panel_launcher_create (panel->toplevel, pos, uri);
812 else
813 success = FALSE;
814 } else if (type != G_FILE_TYPE_DIRECTORY && can_exec) {
815 char *filename;
816
817 filename = g_file_get_path (file);
818
819 if (panel_profile_id_lists_are_writable ())
820 /* executable and local, so add a
821 * launcher with it */
822 ask_about_launcher (filename, panel,
823 pos, TRUE);
824 else
825 success = FALSE;
826 g_free (filename);
827 } else {
828 if (!drop_uri (panel, pos, uri,
829 PANEL_ICON_UNKNOWN))
830 success = FALSE;
831 }
832 } else {
833 if (!drop_uri (panel, pos, uri, PANEL_ICON_UNKNOWN))
834 success = FALSE;
835 }
836
837 g_clear_object (&info);
838 g_object_unref (file);
839 }
840
841 g_strfreev (uris);
842
843 return success;
844}

1211void
1212panel_receive_dnd_data (PanelWidget *panel,
1213 guint info,
1214 int pos,
1215 GtkSelectionData *selection_data,
1216 GdkDragContext *context,
1217 guint time_)
1218{
1219 const guchar *data;
1220 gboolean success = FALSE;
1221
1222 if (panel_lockdown_get_locked_down ()) {
1223 gtk_drag_finish (context, FALSE, FALSE, time_);
1224 return;
1225 }
1226
1227 data = gtk_selection_data_get_data (selection_data);
1228
1229 switch (info) {
1230 case TARGET_URL:
1231 success = drop_urilist (panel, pos, (char *)data);
1232 break;
1233 case TARGET_NETSCAPE_URL:
1234 success = drop_url (panel, pos, (char *)data);
1235 break;
1236 case TARGET_COLOR:
1237 gint n_bytes = gtk_selection_data_get_length (selection_data);
1238 gint length = n_bytes / sizeof (guint16);
1239 success = set_background_color (panel->toplevel, (guint16 *) data, length);
1240 break;
1241 case TARGET_BGIMAGE:
1242 success = set_background_image_from_uri (panel->toplevel, (char *) data);
1243 break;
1244 case TARGET_BACKGROUND_RESET:
1245 if (panel_profile_background_key_is_writable (panel->toplevel, "type")) {
1246 panel_profile_set_background_type (panel->toplevel, PANEL_BACK_NONE);
1247 success = TRUE;
1248 } else {
1249 success = FALSE;
1250 }
1251 break;
1252 case TARGET_DIRECTORY:
1253 success = drop_uri (panel, pos, (char *)data,
1254 PANEL_ICON_FOLDER);
1255 break;
1256 case TARGET_APPLET:
1257 if (!gtk_selection_data_get_data (selection_data)) {
1258 gtk_drag_finish (context, FALSE, FALSE, time_);
1259 return;
1260 }
1261 if (panel_profile_id_lists_are_writable ()) {
1262 mate_panel_applet_frame_create (panel->toplevel, pos, (char *) data);
1263 success = TRUE;
1264 } else {
1265 success = FALSE;
1266 }
1267 break;
1268 case TARGET_APPLET_INTERNAL:
1269 success = drop_internal_applet (panel, pos, (char *)data,
1270 gdk_drag_context_get_selected_action (context));
1271 break;
1272 case TARGET_ICON_INTERNAL:
1273 success = drop_internal_icon (panel, pos, (char *)data,
1274 gdk_drag_context_get_selected_action (context));
1275 break;
1276 default:
1277 gtk_drag_finish (context, FALSE, FALSE, time_);
1278 return;
1279 }
1280
1281 gtk_drag_finish (context, success, FALSE, time_);
1282}
