46static int utf8_fputs(const char* str, FILE* f)
47{
48 char* l;
49 int ret;
50
51 l = g_locale_from_utf8(str, -1, NULL, NULL, NULL);
52
53 if (l == NULL)
54 {
55 ret = fputs(str, f); /* just print it anyway, better than nothing */
56 }
57 else
58 {
59 ret = fputs(l, f);
60 }
61
62 g_free(l);
63
64 return ret;
65}

67void menu_verbose(const char* format, ...)
68{
69 va_list args;
70 char* str;
71

74 return;
75 }
76
77 va_start(args, format);
78 str = g_strdup_vprintf(format, args);
79 va_end(args);
80

83
84 g_free(str);

98static void append_children(MenuLayoutNode* node, int depth, GString* str)
99{
100 MenuLayoutNode* iter;

112static void append_simple_with_attr(MenuLayoutNode* node, int depth, const char* node_name, const char* attr_name, const char* attr_value, GString* str)
113{
114 const char* content;
115
116 append_spaces(str, depth);
117
118 if ((content = menu_layout_node_get_content(node)))
119 {
120 char* escaped;
121
122 escaped = g_markup_escape_text(content, -1);
123
124 if (attr_name && attr_value)
125 {
126 char* attr_escaped;
127
128 attr_escaped = g_markup_escape_text(attr_value, -1);
129
130 g_string_append_printf(str, "<%s %s=\"%s\">%s</%s>\n", node_name, attr_name, attr_escaped, escaped, node_name);
131
132 g_free(attr_escaped);
133 }
134 else
135 {
136 g_string_append_printf(str, "<%s>%s</%s>\n", node_name, escaped, node_name);
137 }
138
139 g_free(escaped);
140 }
141 else
142 {
143 if (attr_name && attr_value)
144 {
145 char* attr_escaped;
146
147 attr_escaped = g_markup_escape_text(attr_value, -1);
148
149 g_string_append_printf(str, "<%s %s=\"%s\"/>\n", node_name, attr_name, attr_escaped);
150
151 g_free(attr_escaped);
152 }
153 else
154 {
155 g_string_append_printf(str, "<%s/>\n", node_name);
156 }
157 }
158}

160static void append_layout(MenuLayoutNode* node, int depth, const char* node_name, MenuLayoutValues* layout_values, GString* str)
161{
162 const char* content;
163
164 append_spaces(str, depth);
165
166 if ((content = menu_layout_node_get_content(node)))
167 {
168 char* escaped;
169
170 escaped = g_markup_escape_text(content, -1);
171
172 g_string_append_printf(str,
173 "<%s show_empty=\"%s\" inline=\"%s\" inline_header=\"%s\""
174 " inline_alias=\"%s\" inline_limit=\"%d\">%s</%s>\n",
175 node_name,
176 layout_values->show_empty ? "true" : "false",
177 layout_values->inline_menus ? "true" : "false",
178 layout_values->inline_header ? "true" : "false",
179 layout_values->inline_alias ? "true" : "false",
180 (gint) layout_values->inline_limit,
181 escaped,
182 node_name);
183
184 g_free(escaped);
185 }
186 else
187 {
188 g_string_append_printf(str,
189 "<%s show_empty=\"%s\" inline=\"%s\" inline_header=\"%s\""
190 " inline_alias=\"%s\" inline_limit=\"%d\"/>\n",
191 node_name,
192 layout_values->show_empty ? "true" : "false",
193 layout_values->inline_menus ? "true" : "false",
194 layout_values->inline_header ? "true" : "false",
195 layout_values->inline_alias ? "true" : "false",
196 (gint) layout_values->inline_limit);
197 }
198}

200static void append_merge(MenuLayoutNode* node, int depth, const char* node_name, MenuLayoutMergeType merge_type, GString* str)
201{
202 const char* merge_type_str;
203
204 merge_type_str = NULL;
205
206 switch (merge_type)
207 {
208 case MENU_LAYOUT_MERGE_NONE:
209 merge_type_str = "none";
210 break;
211
212 case MENU_LAYOUT_MERGE_MENUS:
213 merge_type_str = "menus";
214 break;
215
216 case MENU_LAYOUT_MERGE_FILES:
217 merge_type_str = "files";
218 break;
219
220 case MENU_LAYOUT_MERGE_ALL:
221 merge_type_str = "all";
222 break;
223
224 default:
225 g_assert_not_reached();
226 break;
227 }
228
229 append_simple_with_attr(node, depth, node_name, "type", merge_type_str, str);
230}

251static void append_container(MenuLayoutNode* node, gboolean onelevel, int depth, const char* node_name, GString* str)
252{
253 append_start(node, depth, node_name, str);
254
255 if (!onelevel)
256 {

258 append_end(node, depth, node_name, str);

262static void append_to_string(MenuLayoutNode* node, gboolean onelevel, int depth, GString* str)
263{
264 MenuLayoutValues layout_values;
265
266 switch (menu_layout_node_get_type(node))
267 {
268 case MENU_LAYOUT_NODE_ROOT:
269 if (!onelevel)
270 append_children(node, depth - 1, str); /* -1 to ignore depth of root */
271 else
272 append_start(node, depth - 1, "Root", str);
273 break;
274
275 case MENU_LAYOUT_NODE_PASSTHROUGH:
276 g_string_append(str, menu_layout_node_get_content(node));
277 g_string_append_c(str, '\n');
278 break;
279
280 case MENU_LAYOUT_NODE_MENU:
281 append_container(node, onelevel, depth, "Menu", str);
282 break;
283
284 case MENU_LAYOUT_NODE_APP_DIR:
285 append_simple(node, depth, "AppDir", str);
286 break;
287
288 case MENU_LAYOUT_NODE_DEFAULT_APP_DIRS:
289 append_simple(node, depth, "DefaultAppDirs", str);
290 break;
291
292 case MENU_LAYOUT_NODE_DIRECTORY_DIR:
293 append_simple(node, depth, "DirectoryDir", str);
294 break;
295
296 case MENU_LAYOUT_NODE_DEFAULT_DIRECTORY_DIRS:
297 append_simple(node, depth, "DefaultDirectoryDirs", str);
298 break;
299
300 case MENU_LAYOUT_NODE_DEFAULT_MERGE_DIRS:
301 append_simple(node, depth, "DefaultMergeDirs", str);
302 break;
303
304 case MENU_LAYOUT_NODE_NAME:
305 append_simple(node, depth, "Name", str);
306 break;
307
308 case MENU_LAYOUT_NODE_DIRECTORY:
309 append_simple(node, depth, "Directory", str);
310 break;
311
312 case MENU_LAYOUT_NODE_ONLY_UNALLOCATED:
313 append_simple(node, depth, "OnlyUnallocated", str);
314 break;
315
316 case MENU_LAYOUT_NODE_NOT_ONLY_UNALLOCATED:
317 append_simple(node, depth, "NotOnlyUnallocated", str);
318 break;
319
320 case MENU_LAYOUT_NODE_INCLUDE:
321 append_container(node, onelevel, depth, "Include", str);
322 break;
323
324 case MENU_LAYOUT_NODE_EXCLUDE:
325 append_container(node, onelevel, depth, "Exclude", str);
326 break;
327
328 case MENU_LAYOUT_NODE_FILENAME:
329 append_simple(node, depth, "Filename", str);
330 break;
331
332 case MENU_LAYOUT_NODE_CATEGORY:
333 append_simple(node, depth, "Category", str);
334 break;
335
336 case MENU_LAYOUT_NODE_ALL:
337 append_simple(node, depth, "All", str);
338 break;
339
340 case MENU_LAYOUT_NODE_AND:
341 append_container(node, onelevel, depth, "And", str);
342 break;
343
344 case MENU_LAYOUT_NODE_OR:
345 append_container(node, onelevel, depth, "Or", str);
346 break;
347
348 case MENU_LAYOUT_NODE_NOT:
349 append_container(node, onelevel, depth, "Not", str);
350 break;
351
352 case MENU_LAYOUT_NODE_MERGE_FILE:
353 {
354 MenuMergeFileType type;
355
356 type = menu_layout_node_merge_file_get_type(node);
357
358 append_simple_with_attr(node, depth, "MergeFile", "type", type == MENU_MERGE_FILE_TYPE_PARENT ? "parent" : "path", str);
359 break;
360 }
361
362 case MENU_LAYOUT_NODE_MERGE_DIR:
363 append_simple(node, depth, "MergeDir", str);
364 break;
365
366 case MENU_LAYOUT_NODE_LEGACY_DIR:
367 append_simple_with_attr(node, depth, "LegacyDir", "prefix", menu_layout_node_legacy_dir_get_prefix (node), str);
368 break;
369
370 case MENU_LAYOUT_NODE_KDE_LEGACY_DIRS:
371 append_simple(node, depth, "KDELegacyDirs", str);
372 break;
373
374 case MENU_LAYOUT_NODE_MOVE:
375 append_container(node, onelevel, depth, "Move", str);
376 break;
377
378 case MENU_LAYOUT_NODE_OLD:
379 append_simple(node, depth, "Old", str);
380 break;
381
382 case MENU_LAYOUT_NODE_NEW:
383 append_simple(node, depth, "New", str);
384 break;
385
386 case MENU_LAYOUT_NODE_DELETED:
387 append_simple(node, depth, "Deleted", str);
388 break;
389
390 case MENU_LAYOUT_NODE_NOT_DELETED:
391 append_simple(node, depth, "NotDeleted", str);
392 break;
393
394 case MENU_LAYOUT_NODE_LAYOUT:
395 append_container(node, onelevel, depth, "Layout", str);
396 break;
397
398 case MENU_LAYOUT_NODE_DEFAULT_LAYOUT:
399 menu_layout_node_default_layout_get_values(node, &layout_values);
400 append_layout(node, depth, "DefaultLayout", &layout_values, str);
401 break;
402
403 case MENU_LAYOUT_NODE_MENUNAME:
404 menu_layout_node_menuname_get_values(node, &layout_values);
405 append_layout(node, depth, "MenuName", &layout_values, str);
406 break;
407
408 case MENU_LAYOUT_NODE_SEPARATOR:
409 append_simple(node, depth, "Name", str);
410 break;
411
412 case MENU_LAYOUT_NODE_MERGE:
413 append_merge(node, depth, "Merge", menu_layout_node_merge_get_type(node), str);
414 break;
415
416 default:
417 g_assert_not_reached();
418 break;
419 }
