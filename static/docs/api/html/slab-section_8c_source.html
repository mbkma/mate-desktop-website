48static void
49set_override_color (GtkWidget *widget,
50 GdkRGBA *rgba)
51{
52 gchar *css;
53 GtkCssProvider *provider;
54
55 provider = gtk_css_provider_new ();
56
57 css = g_strdup_printf ("* { color: %s;}",
58 gdk_rgba_to_string (rgba));
59 gtk_css_provider_load_from_data (provider, css, -1, NULL);
60 g_free (css);
61
62 gtk_style_context_add_provider (gtk_widget_get_style_context (widget),
63 GTK_STYLE_PROVIDER (provider),
64 GTK_STYLE_PROVIDER_PRIORITY_APPLICATION);
65 g_object_unref (provider);
66}

68static void
69slab_section_set_title_color (GtkWidget * widget)
70{
71 GtkStyleContext *context;
72 GdkRGBA *rgba = NULL;
73
74 context = gtk_widget_get_style_context (widget);
75
76 switch (SLAB_SECTION (widget)->style)
77 {
78 case Style1:
79 gtk_style_context_get (context,
80 GTK_STATE_FLAG_SELECTED,
81 "background-color", &rgba,
82 NULL);
83 set_override_color (SLAB_SECTION (widget)->title, rgba);
84 break;
85 case Style2:
86 if (SLAB_SECTION (widget)->selected)
87 {
88 gtk_style_context_get (context,
89 GTK_STATE_FLAG_SELECTED,
90 "background-color", &rgba,
91 NULL);
92 set_override_color (SLAB_SECTION (widget)->title, rgba);
93 }
94 else
95 {
96 gtk_style_context_get (context,
97 GTK_STATE_FLAG_INSENSITIVE,
98 "color", &rgba,
99 NULL);
100 set_override_color (SLAB_SECTION (widget)->title, rgba);
101 }
102 break;
103 default:
104 g_assert_not_reached ();
105 }
106}

158GtkWidget *
159slab_section_new_with_markup (const gchar * title_markup, SlabStyle style)
160{
161 SlabSection *section;
162 gchar * widget_theming_name;
163
164 section = g_object_new (SLAB_SECTION_TYPE, NULL);
165 gtk_orientable_set_orientation (GTK_ORIENTABLE (section), GTK_ORIENTATION_VERTICAL);
166 gtk_box_set_homogeneous (GTK_BOX (section), FALSE);
167 gtk_box_set_spacing (GTK_BOX (section), 0);
168 section->style = style;
169 section->selected = FALSE;
170
171 section->childbox = GTK_BOX (gtk_box_new (GTK_ORIENTATION_VERTICAL, 10));
172 switch (style)
173 {
174 case Style1:
175 widget_theming_name = "slab_section_style1";
176 break;
177 case Style2:
178 gtk_widget_set_margin_top (GTK_WIDGET (section->childbox), SLAB_TOP_PADDING);
179 gtk_widget_set_margin_bottom (GTK_WIDGET (section->childbox), SLAB_BOTTOM_PADDING);
180 gtk_widget_set_margin_start (GTK_WIDGET (section->childbox), SLAB_LEFT_PADDING);
181 gtk_widget_set_margin_end (GTK_WIDGET (section->childbox), 0);
182 widget_theming_name = "slab_section_style2";
183 break;
184 default:
185 g_assert_not_reached ();
186 }
187 gtk_box_pack_start (GTK_BOX (section), GTK_WIDGET (section->childbox), TRUE, TRUE, 0);
188
189 section->title = gtk_label_new (title_markup);
190 gtk_label_set_use_markup (GTK_LABEL (section->title), TRUE);
191 gtk_label_set_xalign (GTK_LABEL (section->title), 0.0);
192
193 gtk_widget_set_name (GTK_WIDGET (section), widget_theming_name);
194 g_signal_connect (section, "style-set",
195 G_CALLBACK (slab_section_style_set),
196 NULL);
197
198 gtk_box_pack_start (section->childbox, section->title, FALSE, FALSE, 0);
199
200 return GTK_WIDGET (section);
201}
