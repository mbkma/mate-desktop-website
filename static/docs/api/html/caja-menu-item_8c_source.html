136static void
137caja_menu_item_get_property (GObject *object,
138 guint param_id,
139 GValue *value,
140 GParamSpec *pspec)
141{
142 CajaMenuItem *item;
143
144 item = CAJA_MENU_ITEM (object);
145
146 switch (param_id) {
147 case PROP_NAME :
148 g_value_set_string (value, item->details->name);
149 break;
150 case PROP_LABEL :
151 g_value_set_string (value, item->details->label);
152 break;
153 case PROP_TIP :
154 g_value_set_string (value, item->details->tip);
155 break;
156 case PROP_ICON :
157 g_value_set_string (value, item->details->icon);
158 break;
159 case PROP_SENSITIVE :
160 g_value_set_boolean (value, item->details->sensitive);
161 break;
162 case PROP_PRIORITY :
163 g_value_set_boolean (value, item->details->priority);
164 break;
165 case PROP_MENU :
166 g_value_set_object (value, item->details->menu);
167 break;
168 default :
169 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, param_id, pspec);
170 break;
171 }

174static void
175caja_menu_item_set_property (GObject *object,
176 guint param_id,
177 const GValue *value,
178 GParamSpec *pspec)
179{
180 CajaMenuItem *item;
181
182 item = CAJA_MENU_ITEM (object);
183
184 switch (param_id) {
185 case PROP_NAME :
186 g_free (item->details->name);
187 item->details->name = g_strdup (g_value_get_string (value));
188 g_object_notify (object, "name");
189 break;
190 case PROP_LABEL :
191 g_free (item->details->label);
192 item->details->label = g_strdup (g_value_get_string (value));
193 g_object_notify (object, "label");
194 break;
195 case PROP_TIP :
196 g_free (item->details->tip);
197 item->details->tip = g_strdup (g_value_get_string (value));
198 g_object_notify (object, "tip");
199 break;
200 case PROP_ICON :
201 g_free (item->details->icon);
202 item->details->icon = g_strdup (g_value_get_string (value));
203 g_object_notify (object, "icon");
204 break;
205 case PROP_SENSITIVE :
206 item->details->sensitive = g_value_get_boolean (value);
207 g_object_notify (object, "sensitive");
208 break;
209 case PROP_PRIORITY :
210 item->details->priority = g_value_get_boolean (value);
211 g_object_notify (object, "priority");
212 break;
213 case PROP_MENU :
214 if (item->details->menu) {
215 g_object_unref (item->details->menu);
216 }
217 item->details->menu = g_object_ref (g_value_get_object (value));
218 g_object_notify (object, "menu");
219 break;
220 default :
221 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, param_id, pspec);
222 break;
223 }
224}
