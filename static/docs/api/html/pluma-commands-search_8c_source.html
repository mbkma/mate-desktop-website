105static void
106text_found (PlumaWindow *window,
107 gint occurrences)
108{
109 if (occurrences > 1)
110 {
111 pluma_statusbar_flash_message (PLUMA_STATUSBAR (window->priv->statusbar),
112 window->priv->generic_message_cid,
113 ngettext("Found and replaced %d occurrence",
114 "Found and replaced %d occurrences",
115 occurrences),
116 occurrences);
117 }
118 else
119 {
120 if (occurrences == 1)
121 pluma_statusbar_flash_message (PLUMA_STATUSBAR (window->priv->statusbar),
122 window->priv->generic_message_cid,
123 _("Found and replaced one occurrence"));
124 else
125 pluma_statusbar_flash_message (PLUMA_STATUSBAR (window->priv->statusbar),
126 window->priv->generic_message_cid,
127 " ");
128 }
129}

148static gboolean
149run_search (PlumaView *view,
150 gboolean wrap_around,
151 gboolean search_backwards)
152{
153 PlumaDocument *doc;
154 GtkTextIter start_iter;
155 GtkTextIter match_start;
156 GtkTextIter match_end;
157 gboolean found = FALSE;
158
159 doc = PLUMA_DOCUMENT (gtk_text_view_get_buffer (GTK_TEXT_VIEW (view)));
160
161 if (!search_backwards)
162 {
163 gtk_text_buffer_get_selection_bounds (GTK_TEXT_BUFFER (doc),
164 NULL,
165 &start_iter);
166
167 found = pluma_document_search_forward (doc,
168 &start_iter,
169 NULL,
170 &match_start,
171 &match_end);
172 }
173 else
174 {
175 gtk_text_buffer_get_selection_bounds (GTK_TEXT_BUFFER (doc),
176 &start_iter,
177 NULL);
178
179 found = pluma_document_search_backward (doc,
180 NULL,
181 &start_iter,
182 &match_start,
183 &match_end);
184 }
185
186 if (!found && wrap_around)
187 {
188 if (!search_backwards)
189 found = pluma_document_search_forward (doc,
190 NULL,
191 NULL, /* FIXME: set the end_inter */
192 &match_start,
193 &match_end);
194 else
195 found = pluma_document_search_backward (doc,
196 NULL, /* FIXME: set the start_inter */
197 NULL,
198 &match_start,
199 &match_end);
200 }
201
202 if (found)
203 {
204 gtk_text_buffer_place_cursor (GTK_TEXT_BUFFER (doc),
205 &match_start);
206
207 gtk_text_buffer_move_mark_by_name (GTK_TEXT_BUFFER (doc),
208 "selection_bound",
209 &match_end);
210
211 pluma_view_scroll_to_cursor (view);
212 }
213 else
214 {
215 gtk_text_buffer_place_cursor (GTK_TEXT_BUFFER (doc),
216 &start_iter);
217 }
218
219 return found;
220}

222static void
223do_find (PlumaSearchDialog *dialog,
224 PlumaWindow *window)
225{
226 PlumaView *active_view;
227 PlumaDocument *doc;
228 gchar *search_text;
229 const gchar *entry_text;
230 gboolean match_case;
231 gboolean match_regex;
232 gboolean entire_word;
233 gboolean wrap_around;
234 gboolean search_backwards;
235 gboolean parse_escapes;
236 guint flags = 0;
237 guint old_flags = 0;
238 gboolean found;
239
240 /* TODO: make the dialog insensitive when all the tabs are closed
241 * and assert here that the view is not NULL */
242 active_view = pluma_window_get_active_view (window);
243 if (active_view == NULL)
244 return;
245
246 doc = PLUMA_DOCUMENT (gtk_text_view_get_buffer (GTK_TEXT_VIEW (active_view)));
247
248 match_case = pluma_search_dialog_get_match_case (dialog);
249 match_regex = pluma_search_dialog_get_match_regex(dialog);
250 entire_word = pluma_search_dialog_get_entire_word (dialog);
251 search_backwards = pluma_search_dialog_get_backwards (dialog);
252 wrap_around = pluma_search_dialog_get_wrap_around (dialog);
253 parse_escapes = pluma_search_dialog_get_parse_escapes (dialog);
254
255 if (!parse_escapes) {
256 entry_text = pluma_utils_escape_search_text (pluma_search_dialog_get_search_text (dialog));
257 } else {
258 entry_text = pluma_search_dialog_get_search_text (dialog);
259 }
260
261 PLUMA_SEARCH_SET_CASE_SENSITIVE (flags, match_case);
262 PLUMA_SEARCH_SET_ENTIRE_WORD (flags, entire_word);
263 PLUMA_SEARCH_SET_MATCH_REGEX (flags, match_regex);
264
265 search_text = pluma_document_get_search_text (doc, &old_flags);
266
267 if ((search_text == NULL) ||
268 (strcmp (search_text, entry_text) != 0) ||
269 (flags != old_flags))
270 {
271 pluma_document_set_search_text (doc, entry_text, flags);
272 }
273
274 g_free (search_text);
275 if (match_regex)
276 {
277 pluma_document_set_last_replace_text (doc, pluma_search_dialog_get_replace_text (dialog));
278 }
279
280 found = run_search (active_view,
281 wrap_around,
282 search_backwards);
283
284 if (found)
285 text_found (window, 0);
286 else {
287 if (!parse_escapes) {
288 text_not_found (window, pluma_utils_unescape_search_text (entry_text));
289 } else {
290 text_not_found (window, entry_text);
291 }
292 }
293
294 gtk_dialog_set_response_sensitive (GTK_DIALOG (dialog),
295 PLUMA_SEARCH_DIALOG_REPLACE_RESPONSE,
296 found);
297}

300static gboolean
301get_selected_text (GtkTextBuffer *doc,
302 gchar **selected_text,
303 gint *len)
304{
305 GtkTextIter start, end;
306
307 g_return_val_if_fail (selected_text != NULL, FALSE);
308 g_return_val_if_fail (*selected_text == NULL, FALSE);
309
310 if (!gtk_text_buffer_get_selection_bounds (doc, &start, &end))
311 {
312 if (len != NULL)
313 *len = 0;
314
315 return FALSE;
316 }
317
318 *selected_text = gtk_text_buffer_get_slice (doc, &start, &end, TRUE);
319
320 if (len != NULL)
321 *len = g_utf8_strlen (*selected_text, -1);
322
323 return TRUE;
324}

326static void
327replace_selected_text (GtkTextBuffer *buffer,
328 const gchar *replace)
329{
330 g_return_if_fail (gtk_text_buffer_get_selection_bounds (buffer, NULL, NULL));
331 g_return_if_fail (replace != NULL);
332
333 gtk_text_buffer_begin_user_action (buffer);
334
335 gtk_text_buffer_delete_selection (buffer, FALSE, TRUE);
336
337 gtk_text_buffer_insert_at_cursor (buffer, replace, strlen (replace));
338
339 gtk_text_buffer_end_user_action (buffer);
340}

342static void
343do_replace (PlumaSearchDialog *dialog,
344 PlumaWindow *window)
345{
346 PlumaDocument *doc;
347 const gchar *search_entry_text;
348 const gchar *replace_entry_text;
349 gchar *unescaped_search_text;
350 gchar *unescaped_replace_text;
351 gchar *selected_text = NULL;
352 gboolean match_case;
353 gboolean match_regex;
354 gboolean parse_escapes;
355 gboolean need_refind;
356
357 doc = pluma_window_get_active_document (window);
358 if (doc == NULL)
359 return;
360
361 parse_escapes = pluma_search_dialog_get_parse_escapes (dialog);
362 if (!parse_escapes) {
363 search_entry_text = pluma_utils_escape_search_text (pluma_search_dialog_get_search_text (dialog));
364 } else {
365 search_entry_text = pluma_search_dialog_get_search_text (dialog);
366 }
367 g_return_if_fail ((search_entry_text) != NULL);
368 g_return_if_fail ((*search_entry_text) != '\0');
369
370 /* replace text may be "", we just delete */
371 if (!parse_escapes) {
372 replace_entry_text = pluma_utils_escape_search_text (pluma_search_dialog_get_replace_text (dialog));
373 } else {
374 replace_entry_text = pluma_search_dialog_get_replace_text (dialog);
375 }
376 g_return_if_fail ((replace_entry_text) != NULL);
377
378 unescaped_search_text = pluma_utils_unescape_search_text (search_entry_text);
379
380 get_selected_text (GTK_TEXT_BUFFER (doc),
381 &selected_text,
382 NULL);
383
384 match_case = pluma_search_dialog_get_match_case (dialog);
385 match_regex = pluma_search_dialog_get_match_regex(dialog);
386
387 if (selected_text != NULL)
388 {
389 if(!match_regex)
390 {
391 need_refind = (match_case && (strcmp (selected_text,unescaped_search_text) != 0))
392 || (!match_case && !g_utf8_caselessnmatch (selected_text,
393 unescaped_search_text,
394 strlen (selected_text),
395 strlen (unescaped_search_text)) != 0);
396 }
397 else
398 {
399 need_refind = !g_regex_match_simple(unescaped_search_text,
400 selected_text,
401 match_case ? 0 : G_REGEX_CASELESS,
402 0);
403 }
404 }
405 else
406 {
407 need_refind = TRUE;
408 }
409
410 if (need_refind)
411 {
412 pluma_document_set_last_replace_text (doc, replace_entry_text);
413 do_find (dialog, window);
414 g_free (unescaped_search_text);
415 g_free (selected_text);
416
417 return;
418 }
419
420 if(!match_regex)
421 unescaped_replace_text = pluma_utils_unescape_search_text (replace_entry_text);
422 else
423 unescaped_replace_text = g_strdup (pluma_document_get_last_replace_text (doc));
424
425 replace_selected_text (GTK_TEXT_BUFFER (doc), unescaped_replace_text);
426
427 g_free (unescaped_search_text);
428 g_free (selected_text);
429 g_free (unescaped_replace_text);
430
431 do_find (dialog, window);
432}

434static void
435do_replace_all (PlumaSearchDialog *dialog,
436 PlumaWindow *window)
437{
438 PlumaView *active_view;
439 PlumaDocument *doc;
440 const gchar *search_entry_text;
441 const gchar *replace_entry_text;
442 gboolean match_case;
443 gboolean match_regex;
444 gboolean entire_word;
445 gboolean parse_escapes;
446 guint flags = 0;
447 gint count;
448
449 active_view = pluma_window_get_active_view (window);
450 if (active_view == NULL)
451 return;
452
453 doc = PLUMA_DOCUMENT (gtk_text_view_get_buffer (GTK_TEXT_VIEW (active_view)));
454
455 parse_escapes = pluma_search_dialog_get_parse_escapes (dialog);
456 if (!parse_escapes) {
457 search_entry_text = pluma_utils_escape_search_text(pluma_search_dialog_get_search_text (dialog));
458 } else {
459 search_entry_text = pluma_search_dialog_get_search_text (dialog);
460 }
461 g_return_if_fail ((search_entry_text) != NULL);
462 g_return_if_fail ((*search_entry_text) != '\0');
463
464 /* replace text may be "", we just delete all occurrencies */
465 if (!parse_escapes) {
466 replace_entry_text = pluma_utils_escape_search_text (pluma_search_dialog_get_replace_text (dialog));
467 } else {
468 replace_entry_text = pluma_search_dialog_get_replace_text (dialog);
469 }
470 g_return_if_fail ((replace_entry_text) != NULL);
471
472 match_case = pluma_search_dialog_get_match_case (dialog);
473 match_regex = pluma_search_dialog_get_match_regex(dialog);
474 entire_word = pluma_search_dialog_get_entire_word (dialog);
475
476 PLUMA_SEARCH_SET_CASE_SENSITIVE (flags, match_case);
477 PLUMA_SEARCH_SET_MATCH_REGEX (flags, match_regex);
478 PLUMA_SEARCH_SET_ENTIRE_WORD (flags, entire_word);
479
480 count = pluma_document_replace_all (doc,
481 search_entry_text,
482 replace_entry_text,
483 flags);
484
485 if (count > 0)
486 {
487 text_found (window, count);
488 }
489 else
490 {
491 if (!parse_escapes) {
492 text_not_found (window, pluma_utils_unescape_search_text (search_entry_text));
493 } else {
494 text_not_found (window, search_entry_text);
495 }
496 }
497
498 gtk_dialog_set_response_sensitive (GTK_DIALOG (dialog),
499 PLUMA_SEARCH_DIALOG_REPLACE_RESPONSE,
500 FALSE);
501}

503static void
504search_dialog_response_cb (PlumaSearchDialog *dialog,
505 gint response_id,
506 PlumaWindow *window)
507{
508 pluma_debug (DEBUG_COMMANDS);
509
510 switch (response_id)
511 {
512 case PLUMA_SEARCH_DIALOG_FIND_RESPONSE:
513 do_find (dialog, window);
514 break;
515 case PLUMA_SEARCH_DIALOG_REPLACE_RESPONSE:
516 do_replace (dialog, window);
517 break;
518 case PLUMA_SEARCH_DIALOG_REPLACE_ALL_RESPONSE:
519 do_replace_all (dialog, window);
520 break;
521 default:
522 last_search_data_store_position (dialog);
523 gtk_widget_hide (GTK_WIDGET (dialog));
524 }
525}

579void
580_pluma_cmd_search_find (GtkAction *action,
581 PlumaWindow *window)
582{
583 gpointer data;
584 GtkWidget *search_dialog;
585 PlumaDocument *doc;
586 gboolean selection_exists;
587 gboolean parse_escapes;
588 gchar *find_text = NULL;
589 const gchar *search_text = NULL;
590 gint sel_len;
591
592 pluma_debug (DEBUG_COMMANDS);
593
594 data = g_object_get_data (G_OBJECT (window), PLUMA_SEARCH_DIALOG_KEY);
595
596 if (data == NULL)
597 {
598 search_dialog = create_dialog (window, FALSE);
599 }
600 else
601 {
602 g_return_if_fail (PLUMA_IS_SEARCH_DIALOG (data));
603
604 search_dialog = GTK_WIDGET (data);
605
606 /* turn the dialog into a find dialog if needed */
607 if (pluma_search_dialog_get_show_replace (PLUMA_SEARCH_DIALOG (search_dialog)))
608 pluma_search_dialog_set_show_replace (PLUMA_SEARCH_DIALOG (search_dialog),
609 FALSE);
610 }
611
612 doc = pluma_window_get_active_document (window);
613 g_return_if_fail (doc != NULL);
614
615 selection_exists = get_selected_text (GTK_TEXT_BUFFER (doc),
616 &find_text,
617 &sel_len);
618
619 if (selection_exists && find_text != NULL && sel_len < 80)
620 {
621 /*
622 * Special case: if the currently selected text
623 * is the same as the unescaped search text and
624 * escape sequence parsing is activated, use the
625 * same old search text. (Without this, if you e.g.
626 * search for '\n' in escaped mode and then open
627 * the search dialog again, you'll get an unprintable
628 * single-character literal '\n' in the "search for"
629 * box).
630 */
631 parse_escapes = pluma_search_dialog_get_parse_escapes (PLUMA_SEARCH_DIALOG (search_dialog));
632 search_text = pluma_search_dialog_get_search_text (PLUMA_SEARCH_DIALOG (search_dialog));
633 if (!(search_text != NULL
634 && !strcmp(pluma_utils_unescape_search_text(search_text), find_text)
635 && parse_escapes)) {
636 /* General case */
637 pluma_search_dialog_set_search_text (PLUMA_SEARCH_DIALOG (search_dialog),
638 find_text);
639 }
640 g_free (find_text);
641 }
642 else
643 {
644 g_free (find_text);
645 }
646
647 gtk_widget_show (search_dialog);
648 last_search_data_restore_position (PLUMA_SEARCH_DIALOG (search_dialog));
649 pluma_search_dialog_present_with_time (PLUMA_SEARCH_DIALOG (search_dialog),
650 GDK_CURRENT_TIME);
651}
