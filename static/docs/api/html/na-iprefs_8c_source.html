164/*
165 * na_iprefs_write_level_zero:
166 * @items: the #GList of items whose first level is to be written.
167 * @messages: a pointer to a #GSList in which we will add happening
168 * error messages;
169 * the pointer may be %NULL;
170 * if not %NULL, the #GSList must have been initialized by the
171 * caller.
172 *
173 * Rewrite the level-zero items in MateConf preferences.
174 *
175 * Returns: %TRUE if successfully written (i.e. writable, not locked,
176 * and so on), %FALSE else.
177 *
178 * @messages #GSList is only filled up in case of an error has occured.
179 * If there is no error (na_iprefs_write_level_zero() returns %TRUE), then
180 * the caller may safely assume that @messages is returned in the same
181 * state that it has been provided.
182 */
183gboolean
184na_iprefs_write_level_zero( const GList *items, GSList **messages )
185{
186 gboolean written;
187 const GList *it;
188 gchar *id;
189 GSList *content;
190
191 written = FALSE;
192 content = NULL;
193
194 for( it = items ; it ; it = it->next ){
195 id = na_object_get_id( it->data );
196 content = g_slist_prepend( content, id );
197 }
198 content = g_slist_reverse( content );
199
200 written = na_settings_set_string_list( NA_IPREFS_ITEMS_LEVEL_ZERO_ORDER, content );
201
202 na_core_utils_slist_free( content );
203
204 return( written );
205}

207static const gchar *
208enum_map_string_from_id( const EnumMap *map, guint id )
209{
210 const EnumMap *i = map;
211
212 while( i->id ){
213 if( i->id == id ){
214 return( i->str );
215 }
216 i++;
217 }
218 return( map->str );
219}

222enum_map_id_from_string( const EnumMap *map, const gchar *str )
223{
224 const EnumMap *i = map;
225
226 while( i->id ){

228 return( i->id );
229 }
230 i++;
231 }
232 return( map->id );
