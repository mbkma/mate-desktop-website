54static time_t
55mktime_from_string (char *date_s,
56 char *time_s)
57{
58 struct tm tm = {0, };
59 char **fields;
60
61 tm.tm_isdst = -1;
62
63 /* date */
64
65 fields = g_strsplit (date_s, "-", 3);
66 if (fields[0] != NULL) {
67 tm.tm_year = atoi (fields[0]) - 1900;
68 tm.tm_mon = atoi (fields[1]) - 1;
69 tm.tm_mday = atoi (fields[2]);
70 }
71 g_strfreev (fields);
72
73 /* time */
74
75 fields = g_strsplit (time_s, ":", 3);
76 if (fields[0] != NULL) {
77 tm.tm_hour = atoi (fields[0]);
78 if (fields[1] != NULL) {
79 tm.tm_min = atoi (fields[1]);
80 if (fields[2] != NULL)
81 tm.tm_sec = atoi (fields[2]);
82 }
83 }
84 g_strfreev (fields);
85
86 return mktime (&tm);
87}

89static void
90list__process_line (char *line,
91 gpointer data)
92{
93 FrCommand *comm = FR_COMMAND (data);
94 FrCommand7z *p7z_comm = FR_COMMAND_7Z (comm);
95 char **fields;
96 FileData *fdata;
97
98 g_return_if_fail (line != NULL);
99
100 if (! p7z_comm->list_started) {
101 if (strncmp (line, "p7zip Version ", 14) == 0) {
102 const char *ver_start;
103 int ver_len;
104 char version[256];
105
106 ver_start = eat_spaces (line + 14);
107 ver_len = strchr (ver_start, ' ') - ver_start;
108 strncpy (version, ver_start, ver_len);
109 version[ver_len] = 0;
110
111 if ((strcmp (version, "4.55") < 0) && (ver_len > 1) && (version[1] == '.'))
112 p7z_comm->old_style = TRUE;
113 else
114 p7z_comm->old_style = FALSE;
115 if ((strcmp (version, "9.38") < 0) && (ver_len > 1) && (version[1] == '.'))
116 spd_support = FALSE;
117 else
118 spd_support = TRUE;
119 }
120 else if (p7z_comm->old_style && (strncmp (line, "Listing archive: ", 17) == 0))
121 p7z_comm->list_started = TRUE;
122 else if (! p7z_comm->old_style && (strcmp (line, "----------") == 0))
123 p7z_comm->list_started = TRUE;
124 else if (strncmp (line, "Multivolume = ", 14) == 0) {
125 fields = g_strsplit (line, " = ", 2);
126 comm->multi_volume = (strcmp (fields[1], "+") == 0);
127 g_strfreev (fields);
128 }
129 else if (strncmp (line, "Unexpected end of archive", 25) == 0) {
130 unexpected_end_of_archive = TRUE;
131 }
132 return;
133 }
134
135 if (strcmp (line, "") == 0) {
136 if (p7z_comm->fdata != NULL) {
137 if (p7z_comm->fdata->original_path == NULL) {
138 file_data_free (p7z_comm->fdata);
139 p7z_comm->fdata = NULL;
140 }
141 else {
142 fdata = p7z_comm->fdata;
143 if (fdata->dir)
144 fdata->name = dir_name_from_path (fdata->full_path);
145 else
146 fdata->name = g_strdup (file_name_from_path (fdata->full_path));
147 fdata->path = remove_level_from_path (fdata->full_path);
148 fr_command_add_file (comm, fdata);
149 p7z_comm->fdata = NULL;
150 }
151 }
152 return;
153 }
154
155 if (p7z_comm->fdata == NULL)
156 p7z_comm->fdata = file_data_new ();
157
158 fields = g_strsplit (line, " = ", 2);
159
160 if (g_strv_length (fields) < 2) {
161 g_strfreev (fields);
162 return;
163 }
164
165 fdata = p7z_comm->fdata;
166
167 if (strcmp (fields[0], "Path") == 0) {
168 fdata->free_original_path = TRUE;
169 fdata->original_path = g_strdup (fields[1]);
170 fdata->full_path = g_strconcat ((fdata->original_path[0] != '/') ? "/" : "",
171 fdata->original_path,
172 (fdata->dir && (fdata->original_path[strlen (fdata->original_path) - 1] != '/')) ? "/" : "",
173 NULL);
174 }
175 else if (strcmp (fields[0], "Folder") == 0) {
176 fdata->dir = (strcmp (fields[1], "+") == 0);
177 }
178 else if (strcmp (fields[0], "Size") == 0) {
179 fdata->size = g_ascii_strtoull (fields[1], NULL, 10);
180 }
181 else if (strcmp (fields[0], "Modified") == 0) {
182 char **modified_fields;
183
184 modified_fields = g_strsplit (fields[1], " ", 2);
185 if (modified_fields[0] != NULL)
186 fdata->modified = mktime_from_string (modified_fields[0], modified_fields[1]);
187 g_strfreev (modified_fields);
188 }
189 else if (strcmp (fields[0], "Encrypted") == 0) {
190 if (strcmp (fields[1], "+") == 0)
191 fdata->encrypted = TRUE;
192 }
193 else if (strcmp (fields[0], "Method") == 0) {
194 if (strstr (fields[1], "AES") != NULL)
195 fdata->encrypted = TRUE;
196 }
197 else if (strcmp (fields[0], "Attributes") == 0) {
198 if (fields[1][0] == 'D')
199 fdata->dir = TRUE;
200 }
201 g_strfreev (fields);
202}

204static void
205fr_command_7z_begin_command (FrCommand *comm)
206{
207 // Modern 7-Zip by the original author.
208 // Statically linked from a binary distribution, almost guaranteed to work.
209 if (is_program_in_path ("7zzs"))
210 fr_process_begin_command (comm->process, "7zzs");
211 // Dynamically linked from either binary or source distribution.
212 else if (is_program_in_path ("7zz"))
213 fr_process_begin_command (comm->process, "7zz");
214 // Legacy p7zip project.
215 else if (is_program_in_path ("7z"))
216 fr_process_begin_command (comm->process, "7z");
217 else if (is_program_in_path ("7za"))
218 fr_process_begin_command (comm->process, "7za");
219 else if (is_program_in_path ("7zr"))
220 fr_process_begin_command (comm->process, "7zr");
221}

284static void
285process_line__add (char *line,
286 gpointer data)
287{
288 FrCommand *comm = FR_COMMAND (data);
289
290 if (strstr (line, "Enter password") != NULL)
291 password_required = TRUE;
292
293 if ((comm->volume_size > 0) && (strncmp (line, "Creating archive", 16) == 0)) {
294 char *volume_filename;
295 GFile *volume_file;
296
297 volume_filename = g_strconcat (comm->filename, ".001", NULL);
298 volume_file = g_file_new_for_path (volume_filename);
299 fr_command_set_multi_volume (comm, volume_file);
300
301 g_object_unref (volume_file);
302 g_free (volume_filename);
303 }
304
305 if (comm->n_files != 0)
306 parse_progress_line (comm, "Compressing ", _("Adding file: "), line);
307}

309static void
310fr_command_7z_add (FrCommand *comm,
311 const char *from_file,
312 GList *file_list,
313 const char *base_dir,
314 gboolean update,
315 gboolean recursive)
316{
317 GList *scan;
318
319 fr_process_use_standard_locale (comm->process, TRUE);
320 fr_process_set_out_line_func (comm->process,
321 process_line__add,
322 comm);
323
324 fr_command_7z_begin_command (comm);
325
326 if (update)
327 fr_process_add_arg (comm->process, "u");
328 else
329 fr_process_add_arg (comm->process, "a");
330
331 if (base_dir != NULL) {
332 fr_process_set_working_dir (comm->process, base_dir);
333 }
334
335 if (is_mime_type (comm->mime_type, "application/zip")
336 || is_mime_type (comm->mime_type, "application/x-cbz"))
337 {
338 fr_process_add_arg (comm->process, "-tzip");
339 fr_process_add_arg (comm->process, "-mem=AES128");
340 }
341
342 if (spd_support) fr_process_add_arg (comm->process, "-spd");
343 fr_process_add_arg (comm->process, "-bd");
344 fr_process_add_arg (comm->process, "-bb1");
345 fr_process_add_arg (comm->process, "-y");
346 if (recursive)
347 fr_process_add_arg (comm->process, "-l");
348 add_password_arg (comm, comm->password, FALSE);
349 if ((comm->password != NULL)
350 && (*comm->password != 0)
351 && comm->encrypt_header
352 && fr_command_is_capable_of (comm, FR_COMMAND_CAN_ENCRYPT_HEADER))
353 {
354 fr_process_add_arg (comm->process, "-mhe=on");
355 }
356
357 /* fr_process_add_arg (comm->process, "-ms=off"); FIXME: solid mode off? */
358
359 switch (comm->compression) {
360 case FR_COMPRESSION_VERY_FAST:
361 fr_process_add_arg (comm->process, "-mx=1");
362 break;
363 case FR_COMPRESSION_FAST:
364 fr_process_add_arg (comm->process, "-mx=5");
365 break;
366 case FR_COMPRESSION_NORMAL:
367 fr_process_add_arg (comm->process, "-mx=7");
368 break;
369 case FR_COMPRESSION_MAXIMUM:
370 fr_process_add_arg (comm->process, "-mx=9");
371 if (! is_mime_type (comm->mime_type, "application/zip")
372 && ! is_mime_type (comm->mime_type, "application/x-cbz"))
373 {
374 fr_process_add_arg (comm->process, "-m0=lzma2");;
375 }
376 break;
377 }
378
379 if (is_mime_type (comm->mime_type, "application/x-ms-dos-executable"))
380 fr_process_add_arg (comm->process, "-sfx");
381
382 if (comm->volume_size > 0)
383 fr_process_add_arg_printf (comm->process, "-v%ub", comm->volume_size);
384
385 if (from_file != NULL)
386 fr_process_add_arg_concat (comm->process, "-i@", from_file, NULL);
387
388 if (from_file == NULL)
389 for (scan = file_list; scan; scan = scan->next)
390 /* Files prefixed with '@' need to be handled specially */
391 if (g_str_has_prefix (scan->data, "@"))
392 fr_process_add_arg_concat (comm->process, "-i!", scan->data, NULL);
393
394 fr_process_add_arg (comm->process, "--");
395 fr_process_add_arg (comm->process, comm->filename);
396
397 if (from_file == NULL)
398 for (scan = file_list; scan; scan = scan->next)
399 /* Skip files prefixed with '@', already added */
400 if (!g_str_has_prefix (scan->data, "@"))
401 fr_process_add_arg (comm->process, scan->data);
402
403 fr_process_end_command (comm->process);
404}

536static void
537fr_command_7z_handle_error (FrCommand *comm,
538 FrProcError *error)
539{
540 if (error->type == FR_PROC_ERROR_NONE) {
541 FileData *first;
542 char *basename;
543 char *testname;
544
545 /* This is a way to fix bug #582712. */
546
547 if (comm->files->len != 1)
548 return;
549
550 if (! g_str_has_suffix (comm->filename, ".001"))
551 return;
552
553 first = g_ptr_array_index (comm->files, 0);
554 basename = g_path_get_basename (comm->filename);
555 testname = g_strconcat (first->original_path, ".001", NULL);
556
557 if (strcmp (basename, testname) == 0)
558 error->type = FR_PROC_ERROR_ASK_PASSWORD;
559
560 g_free (testname);
561 g_free (basename);
562
563 return;
564 }
565
566 if ((error->status <= 1) || (unexpected_end_of_archive)) {
567 error->type = FR_PROC_ERROR_NONE;
568 }
569 else {
570 if (password_required && (!password_handled))
571 {
572 error->type = FR_PROC_ERROR_ASK_PASSWORD;
573 return;
574 }
575
576 GList *scan;
577
578 for (scan = g_list_last (comm->process->out.raw); scan; scan = scan->prev) {
579 char *line = scan->data;
580
581 if ((strstr (line, "Wrong password?") != NULL)
582 || (strstr (line, "Enter password") != NULL))
583 {
584 error->type = FR_PROC_ERROR_ASK_PASSWORD;
585 break;
586 }
587 }
588 }
589}

621static FrCommandCaps
622fr_command_7z_get_capabilities (FrCommand *comm,
623 const char *mime_type,
624 gboolean check_command)
625{
626 FrCommandCaps capabilities;
627
628 capabilities = FR_COMMAND_CAN_ARCHIVE_MANY_FILES;
629 gboolean available_7zip = is_program_available ("7zz", check_command) || is_program_available ("7zzs", check_command);
630 if (! available_7zip\
631 && ! is_program_available ("7za", check_command) \
632 && ! is_program_available ("7zr", check_command) \
633 && ! is_program_available ("7z", check_command))
634 return capabilities;
635
636 if (is_mime_type (mime_type, "application/x-7z-compressed")) {
637 capabilities |= FR_COMMAND_CAN_READ_WRITE | FR_COMMAND_CAN_CREATE_VOLUMES;
638 if (is_program_available ("7z", check_command) || available_7zip)
639 capabilities |= FR_COMMAND_CAN_ENCRYPT | FR_COMMAND_CAN_ENCRYPT_HEADER;
640 }
641 else if (is_mime_type (mime_type, "application/x-7z-compressed-tar")) {
642 capabilities |= FR_COMMAND_CAN_READ_WRITE;
643 if (is_program_available ("7z", check_command) || available_7zip)
644 capabilities |= FR_COMMAND_CAN_ENCRYPT | FR_COMMAND_CAN_ENCRYPT_HEADER;
645 }
646 else if (is_program_available ("7z", check_command) || available_7zip) {
647 if (is_mime_type (mime_type, "application/x-rar")
648 || is_mime_type (mime_type, "application/x-cbr"))
649 {
650 if (! check_command || available_7zip || g_file_test ("/usr/lib/p7zip/Codecs/Rar29.so", G_FILE_TEST_EXISTS) || g_file_test ("/usr/lib/p7zip/Codecs/Rar.so", G_FILE_TEST_EXISTS)
651 || g_file_test ("/usr/libexec/p7zip/Codecs/Rar29.so", G_FILE_TEST_EXISTS) || g_file_test ("/usr/libexec/p7zip/Codecs/Rar.so", G_FILE_TEST_EXISTS))
652 capabilities |= FR_COMMAND_CAN_READ;
653 }
654 else
655 capabilities |= FR_COMMAND_CAN_READ;
656
657 if (is_mime_type (mime_type, "application/epub+zip")
658 || is_mime_type (mime_type, "application/x-cbz")
659 || is_mime_type (mime_type, "application/x-ms-dos-executable")
660 || is_mime_type (mime_type, "application/zip"))
661 {
662 capabilities |= FR_COMMAND_CAN_WRITE | FR_COMMAND_CAN_ENCRYPT;
663 }
664 }
