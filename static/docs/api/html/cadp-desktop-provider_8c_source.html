141static void
142class_init( CappDesktopProviderClass *klass )
143{
144 static const gchar *thisfn = "cadp_desktop_provider_class_init";
145 GObjectClass *object_class;
146
147 g_debug( "%s: klass=%p", thisfn, ( void * ) klass );
148
149 st_parent_class = g_type_class_peek_parent( klass );
150
151 object_class = G_OBJECT_CLASS( klass );
152 object_class->dispose = instance_dispose;
153 object_class->finalize = instance_finalize;
154
155 klass->private = g_new0( CappDesktopProviderClassPrivate, 1 );
156}

158static void
159instance_init( GTypeInstance *instance, gpointer klass )
160{
161 static const gchar *thisfn = "cadp_desktop_provider_instance_init";
162 CappDesktopProvider *self;
163
164 g_return_if_fail( CADP_IS_DESKTOP_PROVIDER( instance ));
165
166 g_debug( "%s: instance=%p (%s), klass=%p",
167 thisfn, ( void * ) instance, G_OBJECT_TYPE_NAME( instance ), ( void * ) klass );
168
169 self = CADP_DESKTOP_PROVIDER( instance );
170
171 self->private = g_new0( CappDesktopProviderPrivate, 1 );
172
173 self->private->dispose_has_run = FALSE;
174 self->private->monitors = NULL;
175 self->private->timeout.timeout = st_burst_timeout;
176 self->private->timeout.handler = ( NATimeoutFunc ) on_monitor_timeout;
177 self->private->timeout.user_data = self;
178 self->private->timeout.source_id = 0;
179}

181static void
182instance_dispose( GObject *object )
183{
184 static const gchar *thisfn = "cadp_desktop_provider_instance_dispose";
185 CappDesktopProvider *self;
186
187 g_return_if_fail( CADP_IS_DESKTOP_PROVIDER( object ));
188
189 self = CADP_DESKTOP_PROVIDER( object );
190
191 if( !self->private->dispose_has_run ){
192
193 g_debug( "%s: object=%p (%s)", thisfn, ( void * ) object, G_OBJECT_TYPE_NAME( object ));
194
195 self->private->dispose_has_run = TRUE;
196
197 cadp_desktop_provider_release_monitors( self );
198
199 /* chain up to the parent class */
200 if( G_OBJECT_CLASS( st_parent_class )->dispose ){
201 G_OBJECT_CLASS( st_parent_class )->dispose( object );
202 }
203 }
204}

347void
348cadp_desktop_provider_add_monitor( CappDesktopProvider *provider, const gchar *dir )
349{
350 CappMonitor *monitor;
351
352 g_return_if_fail( CADP_IS_DESKTOP_PROVIDER( provider ));
353
354 if( !provider->private->dispose_has_run ){
355
356 monitor = cadp_monitor_new( provider, dir );
357 provider->private->monitors = g_list_prepend( provider->private->monitors, monitor );
358 }
359}

369{

373

384void
385cadp_desktop_provider_release_monitors( CappDesktopProvider *provider )
386{
387 g_return_if_fail( CADP_IS_DESKTOP_PROVIDER( provider ));
388
389 if( provider->private->monitors ){
390
391 g_list_foreach( provider->private->monitors, ( GFunc ) g_object_unref, NULL );
392 g_list_free( provider->private->monitors );
393 provider->private->monitors = NULL;
394 }
