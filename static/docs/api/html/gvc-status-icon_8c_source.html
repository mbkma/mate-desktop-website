63static void
64update_icon_input (GvcStatusIcon *status_icon)
65{
66 MateMixerStreamControl *control = NULL;
67 gboolean show = FALSE;
68
69 /* Enable the input icon in case there is an input stream present and there
70 * is a non-mixer application using the input */
71 if (status_icon->priv->input != NULL) {
72 const gchar *app_id;
73 const GList *inputs =
74 mate_mixer_stream_list_controls (status_icon->priv->input);
75
76 control = mate_mixer_stream_get_default_control (status_icon->priv->input);
77
78 const gchar *stream_name =
79 mate_mixer_stream_get_name (status_icon->priv->input);
80 g_debug ("Got stream name %s", stream_name);
81 if (g_str_has_suffix (stream_name, ".monitor")) {
82 inputs = NULL;
83 g_debug ("Stream is a monitor, ignoring");
84 }
85
86 while (inputs != NULL) {
87 MateMixerStreamControl *input =
88 MATE_MIXER_STREAM_CONTROL (inputs->data);
89 MateMixerStreamControlRole role =
90 mate_mixer_stream_control_get_role (input);
91
92 if (role == MATE_MIXER_STREAM_CONTROL_ROLE_APPLICATION) {
93 MateMixerAppInfo *app_info =
94 mate_mixer_stream_control_get_app_info (input);
95
96 app_id = mate_mixer_app_info_get_id (app_info);
97 if (app_id == NULL) {
98 /* A recording application which has no
99 * identifier set */
100 g_debug ("Found a recording application control %s",
101 mate_mixer_stream_control_get_label (input));
102
103 if (G_UNLIKELY (control == NULL)) {
104 /* In the unlikely case when there is no
105 * default input control, use the application
106 * control for the icon */
107 control = input;
108 }
109 show = TRUE;
110 break;
111 }
112
113 if (strcmp (app_id, "org.mate.VolumeControl") != 0 &&
114 strcmp (app_id, "org.gnome.VolumeControl") != 0 &&
115 strcmp (app_id, "org.PulseAudio.pavucontrol") != 0) {
116 g_debug ("Found a recording application %s", app_id);
117
118 if (G_UNLIKELY (control == NULL))
119 control = input;
120
121 show = TRUE;
122 break;
123 }
124 }
125 inputs = inputs->next;
126 }
127
128 if (show == TRUE)
129 g_debug ("Input icon enabled");
130 else
131 g_debug ("There is no recording application, input icon disabled");
132 }
133
134 gvc_stream_status_icon_set_control (status_icon->priv->icon_input, control);
135
136 gtk_status_icon_set_visible (GTK_STATUS_ICON (status_icon->priv->icon_input), show);
137}

139static void
140update_icon_output (GvcStatusIcon *status_icon)
141{
142 MateMixerStream *stream;
143 MateMixerStreamControl *control = NULL;
144
145 stream = mate_mixer_context_get_default_output_stream (status_icon->priv->context);
146 if (stream != NULL)
147 control = mate_mixer_stream_get_default_control (stream);
148
149 gvc_stream_status_icon_set_control (status_icon->priv->icon_output, control);
150
151 if (control != NULL) {
152 g_debug ("Output icon enabled");
153 gtk_status_icon_set_visible (GTK_STATUS_ICON (status_icon->priv->icon_output),
154 TRUE);
155 }
156 else {
157 g_debug ("There is no output stream/control, output icon disabled");
158 gtk_status_icon_set_visible (GTK_STATUS_ICON (status_icon->priv->icon_output),
159 FALSE);
160 }
161}

163static void
164on_input_stream_control_added (MateMixerStream *stream,
165 const gchar *name,
166 GvcStatusIcon *status_icon)
167{
168 MateMixerStreamControl *control;
169
170 control = mate_mixer_stream_get_control (stream, name);
171 if (G_LIKELY (control != NULL)) {
172 MateMixerStreamControlRole role =
173 mate_mixer_stream_control_get_role (control);
174
175 /* Non-application input control doesn't affect the icon */
176 if (role != MATE_MIXER_STREAM_CONTROL_ROLE_APPLICATION)
177 return;
178 }
179
180 /* Either an application control has been added or we couldn't
181 * read the control, this shouldn't happen but let's revalidate the
182 * icon to be sure if it does */
183 update_icon_input (status_icon);
184}

186static void
187on_input_stream_control_removed (MateMixerStream *stream,
188 const gchar *name,
189 GvcStatusIcon *status_icon)
190{
191 /* The removed stream could be an application input, which may cause
192 * the input status icon to disappear */
193 update_icon_input (status_icon);

196static gboolean
197update_default_input_stream (GvcStatusIcon *status_icon)
198{
199 MateMixerStream *stream;
200
201 stream = mate_mixer_context_get_default_input_stream (status_icon->priv->context);
202 if (stream == status_icon->priv->input)
203 return FALSE;
204
205 /* The input stream has changed */
206 if (status_icon->priv->input != NULL) {
207 g_signal_handlers_disconnect_by_data (G_OBJECT (status_icon->priv->input),
208 status_icon);
209 g_object_unref (status_icon->priv->input);
210 }
211
212 status_icon->priv->input = (stream == NULL) ? NULL : g_object_ref (stream);
213 if (status_icon->priv->input != NULL) {
214 g_signal_connect (G_OBJECT (status_icon->priv->input),
215 "control-added",
216 G_CALLBACK (on_input_stream_control_added),
217 status_icon);
218 g_signal_connect (G_OBJECT (status_icon->priv->input),
219 "control-removed",
220 G_CALLBACK (on_input_stream_control_removed),
221 status_icon);
222 }
223
224 /* Return TRUE if the default input stream has changed */
225 return TRUE;
226}

228static void
229on_context_state_notify (MateMixerContext *context,
230 GParamSpec *pspec,
231 GvcStatusIcon *status_icon)
232{
233 MateMixerState state = mate_mixer_context_get_state (context);
234
235 switch (state) {
236 case MATE_MIXER_STATE_FAILED:
237 g_warning ("Failed to connect to a sound system");
238 break;
239
240 case MATE_MIXER_STATE_READY:
241 update_default_input_stream (status_icon);
242
243 /* Each status change may affect the visibility of the icons */
244 update_icon_output (status_icon);
245 update_icon_input (status_icon);
246 break;
247 default:
248 break;
249 }
250}

271void
272gvc_status_icon_start (GvcStatusIcon *status_icon)
273{
274 g_return_if_fail (GVC_IS_STATUS_ICON (status_icon));
275
276 if (G_UNLIKELY (status_icon->priv->running == TRUE))
277 return;
278
279 if (G_UNLIKELY (mate_mixer_context_open (status_icon->priv->context) == FALSE)) {
280 /* Normally this should never happen, in the worst case we
281 * should end up with the Null module */
282 g_warning ("Failed to connect to a sound system");
283 }
284
285 g_debug ("StatusIcon has been started");
286
287 status_icon->priv->running = TRUE;
288}

290static void
291gvc_status_icon_dispose (GObject *object)
292{
293 GvcStatusIcon *status_icon = GVC_STATUS_ICON (object);
294
295 if (status_icon->priv->input != NULL) {
296 g_signal_handlers_disconnect_by_data (G_OBJECT (status_icon->priv->input),
297 status_icon);
298 g_clear_object (&status_icon->priv->input);
299 }
300
301 g_clear_object (&status_icon->priv->context);
302 g_clear_object (&status_icon->priv->icon_input);
303 g_clear_object (&status_icon->priv->icon_output);
304
305 G_OBJECT_CLASS (gvc_status_icon_parent_class)->dispose (object);
306}

316static void
317gvc_status_icon_init (GvcStatusIcon *status_icon)
318{
319 status_icon->priv = gvc_status_icon_get_instance_private (status_icon);
320
321 status_icon->priv->icon_input = gvc_stream_status_icon_new (NULL, icon_names_input);
322 status_icon->priv->icon_output = gvc_stream_status_icon_new (NULL, icon_names_output);
323
324 gvc_stream_status_icon_set_display_name (status_icon->priv->icon_input, _("Input"));
325 gvc_stream_status_icon_set_display_name (status_icon->priv->icon_output, _("Output"));
326
327 gtk_status_icon_set_title (GTK_STATUS_ICON (status_icon->priv->icon_input),
328 _("Microphone Volume"));
329 gtk_status_icon_set_title (GTK_STATUS_ICON (status_icon->priv->icon_output),
330 _("Sound Output Volume"));
331
332 status_icon->priv->context = mate_mixer_context_new ();
333
334 mate_mixer_context_set_app_name (status_icon->priv->context,
335 _("MATE Volume Control StatusIcon"));
336
337 mate_mixer_context_set_app_id (status_icon->priv->context, GVC_STATUS_ICON_DBUS_NAME);
338 mate_mixer_context_set_app_version (status_icon->priv->context, VERSION);
339 mate_mixer_context_set_app_icon (status_icon->priv->context, "multimedia-volume-control");
340
341 g_signal_connect (G_OBJECT (status_icon->priv->context),
342 "notify::state",
343 G_CALLBACK (on_context_state_notify),
344 status_icon);
345 g_signal_connect (G_OBJECT (status_icon->priv->context),
346 "notify::default-input-stream",
347 G_CALLBACK (on_context_default_input_stream_notify),
348 status_icon);
349 g_signal_connect (G_OBJECT (status_icon->priv->context),
350 "notify::default-output-stream",
351 G_CALLBACK (on_context_default_output_stream_notify),
352 status_icon);
353}
