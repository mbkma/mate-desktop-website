268/* We do some weird things to avoid the constructors being optimized

270 * make a reference to the array from the wrapper to make sure its
271 * references. Then we use a pragma to make sure the wrapper function

273 * need to be extern (but not dllexport), even though they are not
274 * really used from another object file.

277/* We need to account for differences between the mangling of symbols
278 * for x86 and x64/ARM/ARM64 programs, as symbols on x86 are prefixed
279 * with an underscore but symbols on x64/ARM/ARM64 are not.

290#define G_MSVC_CTOR(_func,_sym_prefix) \
291 static void _func(void); \
292 extern int (* _array ## _func)(void); \
293 int _func ## _wrapper(void); \
294 int _func ## _wrapper(void) { _func(); g_slist_find (NULL, _array ## _func); return 0; } \
295 __pragma(comment(linker,"/include:" _sym_prefix # _func "_wrapper")) \
296 __pragma(section(".CRT$XCU",read)) \

299#define G_MSVC_DTOR(_func,_sym_prefix) \
300 static void _func(void); \
301 extern int (* _array ## _func)(void); \
302 int _func ## _constructor(void); \
303 int _func ## _constructor(void) { atexit (_func); g_slist_find (NULL, _array ## _func); return 0; } \
