128static char *
129get_encrypted_passwd (const char *user)
130{
131 char *result = NULL;
132
133#ifdef PWTYPE
134 if (user && *user && !result)
135 {
136 /* First check the shadow passwords. */
137 PWTYPE p = GETPW ((char *) user);
138 if (p && passwd_known (p->PWPSLOT))
139 {
140 result = g_strdup (p->PWPSLOT);
141 }
142 }
143#endif /* PWTYPE */
144
145 if (user && *user && !result)
146 {
147 /* Check non-shadow passwords too. */
148 struct passwd *p = getpwnam (user);
149 if (p && passwd_known (p->pw_passwd))
150 {
151 result = g_strdup (p->pw_passwd);
152 }
153 }
154
155 /* The manual for passwd(4) on HPUX 10.10 says:
156
157 Password aging is put in effect for a particular user if his
158 encrypted password in the password file is followed by a comma and
159 a nonnull string of characters from the above alphabet. This
160 string defines the "age" needed to implement password aging.
161
162 So this means that passwd->pw_passwd isn't simply a string of cyphertext,
163 it might have trailing junk. So, if there is a comma in the string, and
164 that comma is beyond position 13, terminate the string before the comma.
165 */
166 if (result && strlen (result) > 13)
167 {
168 char *s = strchr (result + 13, ',');
169 if (s)
170 {
171 *s = 0;
172 }
173 }
174
175#if !defined(HAVE_PAM) && !defined(HAVE_BSDAUTH)
176 /* We only issue this warning if not compiled with support for PAM,
177 or bsd_auth(3). If we're using PAM, it's not unheard of that
178 normal pwent passwords would be unavailable. */
179
180 if (!result)
181 {
182 g_warning ("Couldn't get password of \"%s\"",
183 (user ? user : "(null)"));
184 }
185
186#endif /* !HAVE_PAM */
187
188 return result;
189}

229static gboolean
230passwds_match (const char *cleartext,
231 const char *ciphertext)
232{
233 char *s = NULL; /* note that on some systems, crypt() may return null */
234
235 s = (char *) crypt (cleartext, ciphertext);
236 if (s && !strcmp (s, ciphertext))
237 {
238 return TRUE;
239 }
240
241#ifdef HAVE_BIGCRYPT
242 /* There seems to be no way to tell at runtime if an HP machine is in
243 "trusted" mode, and thereby, which of crypt() or bigcrypt() we should
244 be calling to compare passwords. So call them both, and see which
245 one works. */
246
247 s = (char *) bigcrypt (cleartext, ciphertext);
248 if (s && !strcmp (s, ciphertext))
249 {
250 return TRUE;
251 }
252
253#endif /* HAVE_BIGCRYPT */
254
255 return FALSE;
256}

259gs_auth_verify_user (const char *username,
260 const char *display,
261 GSAuthMessageFunc func,
262 gpointer data,
263 GError **error)
264{
265 char *password;
266
267 password = NULL;
268
269 /* ask for the password for user */
270 if (func != NULL)
271 {
272 func (GS_AUTH_MESSAGE_PROMPT_ECHO_OFF,
273 "Password: ",
274 &password,
275 data);
276 }
277
278 if (password == NULL)
279 {
280 return FALSE;
281 }
282
283 if (encrypted_user_passwd && passwds_match (password, encrypted_user_passwd))
284 {
285 return TRUE;
286 }
287 else
288 {
289 return FALSE;
290 }
