113static gboolean
114get_action_type_id (const gchar *name,
115 gint *id)
116{
117 gint i;
118
119 for (i = 0; i < PANEL_ACTION_LAST; i++)
120 {
121 if (g_ascii_strcasecmp (panel_action_type [i], name) == 0)
122 {
123 *id = i;
124 return TRUE;
125 }
126 }
127 *id = 0;
128 return FALSE;
129}

198static void
199panel_action_logout (GtkWidget *widget)
200{
201 PanelSessionManager *manager;
202 gboolean prompt = TRUE;
203
204 if (mate_gsettings_schema_exists (MATE_SESSION_SCHEMA)) {
205 GSettings *msm_settings;
206 msm_settings = g_settings_new (MATE_SESSION_SCHEMA);
207 prompt = g_settings_get_boolean (msm_settings, MATE_SESSION_LOGOUT_PROMPT_KEY);
208 g_object_unref (msm_settings);
209 }
210
211 manager = panel_session_manager_get ();
212
213 if (!prompt)
214 panel_session_manager_request_logout (manager,
215 PANEL_SESSION_MANAGER_LOGOUT_MODE_NO_CONFIRMATION);
216 else
217 /* FIXME: we need to use widget to get the screen for the
218 * confirmation dialog, see
219 * http://bugzilla.gnome.org/show_bug.cgi?id=536914 */
220 panel_session_manager_request_logout (manager,
221 PANEL_SESSION_MANAGER_LOGOUT_MODE_NORMAL);
222}

250static void wayland_shutdown_response_cb (GtkWidget *dialog, gint response_id)
251 {
252 int ret;
253
254 if (response_id == GTK_RESPONSE_CANCEL)
255 {
256 gtk_widget_destroy (dialog);
257 return;
258 }
259
260 if (response_id == GTK_RESPONSE_OK)
261 {
262 gtk_widget_destroy (dialog);
263 system ("shutdown now");
264 /*Try the system shutdown command first.
265 *This will fail if root is logged in
266 *Note that if -f --force was used this would block proper unmounting
267 *if this fails try systemd in case it's installed as it commonly is
268 *FIXME: we also need a logind equivalent for non-systemd users
269 *with session managers running as root
270 */
271 system ("systemctl poweroff -i");
272 return;
273 }
274
275 if (response_id == GTK_RESPONSE_ACCEPT)
276 {
277 gtk_widget_destroy (dialog);
278 system ("reboot now");
279 /*Same approach as shutdown command
280 *FIXME: we also need a logind equivalent for non-systemd users
281 *with session managers running as root
282 */
283 system ("systemctl reboot -i");
284 return;
285 }
286
287 if (response_id == GTK_RESPONSE_REJECT)
288 {
289 gtk_widget_destroy (dialog);
290 /*FIXME: we also need a logind equivalent for non-systemd users
291 *with session managers running as root
292 */
293 ret = system ("systemctl hibernate -i");
294 if (ret != 0)
295 {
296 GtkWidget *label;
297 label = gtk_label_new ("Hibernation not supported on this system" "\n" "\n"
298 "The \"resume = \" boot command line option must be set to a swap partition or file" "\n"
299 "Swapfile or partition must be large enough to support hibernation" "\n"
300 "System and hardware must support hibernation");
301 show_error_dialog (label);
302 }
303 return;
304 }
305
306 if (response_id == GTK_RESPONSE_APPLY)
307 {
308 gtk_widget_destroy (dialog);
309 /*FIXME: we also need a logind equivalent for non-systemd users
310 *with session managers running as root
311 */
312 ret = system ("systemctl suspend -i");
313 if (ret != 0)
314 {
315 GtkWidget *label;
316 label = gtk_label_new ("Suspend not supported on this system" "\n" "\n"
317 "Hardware and Firmware must support sleep / suspend");
318 show_error_dialog (label);
319 }
320
321 return;
322 }

345static void
346panel_action_shutdown (GtkWidget *widget)
347{
348#ifdef HAVE_WAYLAND
349 GdkDisplay *display = gdk_display_get_default ();
350 if (GDK_IS_WAYLAND_DISPLAY (display))
351 {
352 GtkWidget *dialog, *hbox, *buttonbox, *shutdown_btn, *label;
353 GtkStyleContext *context;
354
355 dialog = gtk_dialog_new_with_buttons ("System Shutdown",
356 NULL,
357 GTK_DIALOG_DESTROY_WITH_PARENT,
358 NULL,
359 NULL,
360 NULL);
361
362 /*Window icons in dialogs are currently broken or unsupported
363 *in many wayland compositors but this may not always be so
364 */
365 gtk_window_set_icon_name (GTK_WINDOW (dialog), "system-shutdown");
366 context = gtk_widget_get_style_context (GTK_WIDGET (dialog));
367 gtk_style_context_add_class (context, "logout-dialog");
368 context = NULL;
369
370 /*We use the inbuilt gtk response types for simplicity*/
371 wayland_shutdown_dialog_add_button (GTK_DIALOG (dialog),
372 _("S_uspend"), "battery",
373 GTK_RESPONSE_APPLY);
374
375 wayland_shutdown_dialog_add_button (GTK_DIALOG (dialog),
376 _("_Hibernate"), "drive-harddisk",
377 GTK_RESPONSE_REJECT);
378
379 wayland_shutdown_dialog_add_button (GTK_DIALOG (dialog),
380 _("_Restart"), "view-refresh",
381 GTK_RESPONSE_ACCEPT);
382
383 wayland_shutdown_dialog_add_button (GTK_DIALOG (dialog),
384 _("_Cancel"), "process-stop",
385 GTK_RESPONSE_CANCEL);
386
387 shutdown_btn = wayland_shutdown_dialog_add_button (GTK_DIALOG (dialog),
388 _("_Shut Down"), "system-shutdown",
389 GTK_RESPONSE_OK);
390
391 g_signal_connect_swapped (dialog, "response",
392 G_CALLBACK (wayland_shutdown_response_cb),
393 dialog);
394
395 hbox = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
396
397 label = gtk_label_new ("Shut this system down now?");
398 gtk_label_set_line_wrap (GTK_LABEL (label), TRUE);
399 gtk_label_set_use_markup (GTK_LABEL (label), TRUE);
400 gtk_box_pack_start (GTK_BOX (hbox), label, TRUE, TRUE, 6);
401 gtk_container_set_border_width (GTK_CONTAINER (hbox), 16);
402 gtk_container_add (GTK_CONTAINER (gtk_dialog_get_content_area (GTK_DIALOG (dialog))),
403 hbox);
404 gtk_dialog_set_default_response (GTK_DIALOG (dialog), GTK_RESPONSE_OK);
405
406 buttonbox = gtk_widget_get_parent (shutdown_btn);
407 gtk_widget_set_halign (buttonbox,GTK_ALIGN_CENTER);
408 context = gtk_widget_get_style_context (buttonbox);
409 gtk_style_context_add_class (context, "linked");
410 context = NULL;
411 gtk_widget_show_all (dialog);
412 }
413 else
414#endif
415 {
416 PanelSessionManager *manager;
417 manager = panel_session_manager_get ();
418 panel_session_manager_request_shutdown (manager);
419 }
420}

486/* Connect Server
487 */
488static void
489panel_action_connect_server (GtkWidget *widget)
490{
491 GdkScreen *screen;
492 char *command;
493 GError *error;
494
495 screen = gtk_widget_get_screen (GTK_WIDGET (widget));
496 error = NULL;
497
498 if (panel_is_program_in_path ("caja-connect-server"))
499 command = g_strdup ("caja-connect-server");
500 else if (panel_is_program_in_path ("nautilus-connect-server"))
501 command = g_strdup ("nautilus-connect-server");
502 else
503 command = g_strdup ("nemo-connect-server");
504
505 mate_gdk_spawn_command_line_on_screen (screen, command, &error);
506 g_free (command);
507
508 if (error) {
509 panel_error_dialog (NULL, screen,
510 "cannot_connect_server",
511 TRUE,
512 _("Could not connect to server"),
513 error->message);
514 g_clear_error (&error);
515 }

663static void
664panel_action_button_finalize (GObject *object)
665{
666 PanelActionButton *button = PANEL_ACTION_BUTTON (object);
667
668 if (button->priv->settings) {
669 g_signal_handlers_disconnect_by_func (button->priv->settings,
670 G_CALLBACK (panel_action_button_type_changed),
671 button);
672 g_clear_object (&button->priv->settings);
673 }
674
675 button->priv->info = NULL;
676 button->priv->type = PANEL_ACTION_NONE;
677
678 panel_lockdown_notify_remove (G_CALLBACK (panel_action_button_update_sensitivity),
679 button);
680
681 G_OBJECT_CLASS (panel_action_button_parent_class)->finalize (object);
682}

684static void
685panel_action_button_get_property (GObject *object,
686 guint prop_id,
687 GValue *value,
688 GParamSpec *pspec)
689{
690 PanelActionButton *button;
691
692 g_return_if_fail (PANEL_IS_ACTION_BUTTON (object));
693
694 button = PANEL_ACTION_BUTTON (object);
695
696 switch (prop_id) {
697 case PROP_ACTION_TYPE:
698 g_value_set_enum (value, button->priv->type);
699 break;
700 case PROP_DND_ENABLED:
701 g_value_set_boolean (value, button->priv->dnd_enabled);
702 break;
703 default:
704 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
705 break;
706 }
707}

715 PanelActionButton *button;
716
717 g_return_if_fail (PANEL_IS_ACTION_BUTTON (object));

729 break;

731 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
732 break;

1042 *old_applet_idx = strtol (elements [2], NULL, 10);
1043 retval = TRUE; /* Remove the old applet */
1044 }
1045
1046 g_strfreev (elements);
1047
1048 panel_action_button_create (toplevel, position, type);
