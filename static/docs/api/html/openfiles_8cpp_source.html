54static char *
55friendlier_hostname(const char *addr_str, int port)
56{
57 struct addrinfo hints = { };
58 struct addrinfo *res = NULL;
59 char hostname[NI_MAXHOST];
60 char service[NI_MAXSERV];
61 char port_str[6];
62
63 if (!addr_str[0]) return g_strdup("");
64
65 snprintf(port_str, sizeof port_str, "%d", port);
66
67 hints.ai_family = AF_UNSPEC;
68 hints.ai_socktype = SOCK_STREAM;
69
70 if (getaddrinfo(addr_str, port_str, &hints, &res))
71 goto failsafe;
72
73 if (getnameinfo(res->ai_addr, res->ai_addrlen, hostname,
74 sizeof hostname, service, sizeof service, NI_IDN))
75 goto failsafe;
76
77 if (res) freeaddrinfo(res);
78 return g_strdup_printf("%s, TCP port %d (%s)", hostname, port, service);
79
80 failsafe:
81 if (res) freeaddrinfo(res);
82 return g_strdup_printf("%s, TCP port %d", addr_str, port);
83}
