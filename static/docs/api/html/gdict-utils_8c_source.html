104static gboolean
105pre_parse_hook (GOptionContext *context,
106 GOptionGroup *group,
107 gpointer data,
108 GError **error)
109{
110 if (gdict_is_initialized)
111 return TRUE;
112
113#ifdef GDICT_ENABLE_DEBUG
114 const char *env_string = g_getenv ("GDICT_DEBUG");
115 if (env_string != NULL)

117 gdict_debug_flags =

119 gdict_debug_keys,

147GOptionGroup *
148gdict_get_option_group (void)
149{
150 GOptionGroup *group;
151
152 group = g_option_group_new ("gdict",
153 _("GDict Options"),
154 _("Show GDict Options"),

159 g_option_group_add_entries (group, gdict_args);

176void
177gdict_debug_init (gint *argc,
178 gchar ***argv)
179{
180 GOptionContext *option_context;
181 GOptionGroup *gdict_group;
182 GError *error = NULL;
183
184 if (gdict_is_initialized)
185 return;
186
187 option_context = g_option_context_new (NULL);
188 g_option_context_set_ignore_unknown_options (option_context, TRUE);
189 g_option_context_set_help_enabled (option_context, FALSE);
190
191 gdict_group = gdict_get_option_group ();
192 g_option_context_set_main_group (option_context, gdict_group);
193
194 if (!g_option_context_parse (option_context, argc, argv, &error))
195 {
196 g_warning ("%s", error->message);
197 g_error_free (error);
198 }
199
200 g_option_context_free (option_context);
201}

203gboolean
204gdict_check_version (guint required_major,
205 guint required_minor,
206 guint required_micro)
207{
208 gint gdict_effective_micro = 100 * GDICT_MINOR_VERSION + GDICT_MICRO_VERSION;
209 gint required_effective_micro = 100 * required_minor + required_micro;
210
211 if (required_major > GDICT_MAJOR_VERSION)
212 return FALSE;
213
214 if (required_major < GDICT_MAJOR_VERSION)
215 return FALSE;
216
217 if (required_effective_micro < gdict_effective_micro)
218 return FALSE;
219
220 if (required_effective_micro > gdict_effective_micro)
221 return FALSE;
222
223 return TRUE;
224}

246/* shows an error dialog making it transient for @parent */
247static void
248show_error_dialog (GtkWindow *parent,
249 const gchar *message,
250 const gchar *detail)
251{
252 GtkWidget *dialog;
253
254 dialog = gtk_message_dialog_new (parent,
255 GTK_DIALOG_DESTROY_WITH_PARENT,
256 GTK_MESSAGE_ERROR,
257 GTK_BUTTONS_OK,
258 "%s", message);
259 gtk_window_set_title (GTK_WINDOW (dialog), "");
260
261 if (detail)
262 gtk_message_dialog_format_secondary_text (GTK_MESSAGE_DIALOG (dialog),
263 "%s", detail);
264
265 if (parent && gtk_window_get_group (parent))
266 gtk_window_group_add_window (gtk_window_get_group (parent), GTK_WINDOW (dialog));
267
268 gtk_dialog_run (GTK_DIALOG (dialog));
269
270 gtk_widget_destroy (dialog);
271}
