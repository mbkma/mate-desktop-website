576static void
577handle_preference_init_enum (void)
578{
579 MetaEnumPreference *cursor = preferences_enum;
580
581 while (cursor->key!=NULL)
582 {
583 gint value;
584
585 if (cursor->target==NULL)
586 {
587 ++cursor;
588 continue;
589 }
590
591 value = g_settings_get_enum (SETTINGS (cursor->schema),
592 cursor->key);
593 *cursor->target = value;
594
595 ++cursor;
596 }
597}

615static void
616handle_preference_init_string (void)
617{
618 MetaStringPreference *cursor = preferences_string;
619
620 while (cursor->key!=NULL)
621 {
622 gchar *value;
623 gboolean dummy = TRUE;
624
625 /* the string "value" will be newly allocated */
626 value = g_settings_get_string (SETTINGS (cursor->schema),
627 cursor->key);
628
629 if (cursor->handler)
630 {
631 if (cursor->target)
632 meta_bug ("%s has both a target and a handler\n", cursor->key);
633
634 cursor->handler (cursor->pref, value, &dummy);
635
636 g_free (value);
637 }
638 else if (cursor->target)
639 {
640 if (*(cursor->target))
641 g_free (*(cursor->target));
642
643 *(cursor->target) = value;
644 }
645
646 ++cursor;
647 }
648}

650static void
651handle_preference_init_int (void)
652{
653 MetaIntPreference *cursor = preferences_int;
654
655 while (cursor->key!=NULL)
656 {
657 gint value;
658
659 value = g_settings_get_int (SETTINGS (cursor->schema),
660 cursor->key);
661
662 if (value < cursor->minimum || value > cursor->maximum)
663 {
664 /* FIXME: check if this can be avoided by GSettings */
665 meta_warning (_("%d stored in GSettings key %s is out of range %d to %d\n"),
666 value, cursor->key, cursor->minimum, cursor->maximum);
667 /* Former behaviour for out-of-range values was:
668 * - number of workspaces was clamped;
669 * - auto raise delay was always reset to zero even if too high!;
670 * - cursor size was ignored.
671 *
672 * These seem to be meaningless variations. If they did
673 * have meaning we could have put them into MetaIntPreference.
674 * The last of these is the closest to how we behave for
675 * other types, so I think we should standardise on that.
676 */
677 }
678 else if (cursor->target)
679 *cursor->target = value;
680
681 ++cursor;
682 }
683}

685static gboolean
686handle_preference_update_enum (const gchar *key, GSettings *settings)
687{
688 MetaEnumPreference *cursor = preferences_enum;
689 gint old_value;
690
691 while (cursor->key!=NULL && strcmp (key, cursor->key)!=0)
692 ++cursor;
693
694 if (cursor->key==NULL)
695 /* Didn't recognise that key. */
696 return FALSE;
697
698 /* We need to know whether the value changes, so
699 * store the current value away.
700 */
701
702 old_value = * ((gint *) cursor->target);
703
704 /* Now look it up... */
705 *cursor->target = g_settings_get_enum (settings, key);
706
707 /* Did it change? If so, tell the listeners about it. */
708
709 if (old_value != *((gint *) cursor->target))
710 queue_changed (cursor->pref);
711
712 return TRUE;
713}

715static gboolean
716handle_preference_update_bool (const gchar *key, GSettings *settings)
717{
718 MetaBoolPreference *cursor = preferences_bool;
719 gboolean old_value;
720
721 while (cursor->key!=NULL && strcmp (key, cursor->key)!=0)
722 ++cursor;
723
724 if (cursor->key==NULL)
725 /* Didn't recognise that key. */
726 return FALSE;
727
728 if (cursor->target==NULL)
729 /* No work for us to do. */
730 return TRUE;
731
732 /* We need to know whether the value changes, so
733 * store the current value away.
734 */
735
736 old_value = * ((gboolean *) cursor->target);
737
738 /* Now look it up... */
739
740 *((gboolean *) cursor->target) = g_settings_get_boolean (settings, key);
741
742 /* Did it change? If so, tell the listeners about it. */
743
744 if (old_value != *((gboolean *) cursor->target))
745 queue_changed (cursor->pref);
746
747 if (cursor->pref==META_PREF_DISABLE_WORKAROUNDS)
748 maybe_give_disable_workarounds_warning ();
749
750 return TRUE;
751}

753static gboolean
754handle_preference_update_string (const gchar *key, GSettings *settings)
755{
756 MetaStringPreference *cursor = preferences_string;
757 gchar *value;
758 gboolean inform_listeners = TRUE;
759
760 while (cursor->key!=NULL && strcmp (key, cursor->key)!=0)
761 ++cursor;
762
763 if (cursor->key==NULL)
764 /* Didn't recognise that key. */
765 return FALSE;
766
767 value = g_settings_get_string (settings, key);
768
769 if (cursor->handler)
770 cursor->handler (cursor->pref, value, &inform_listeners);
771 else if (cursor->target)
772 {
773 if (*(cursor->target))
774 g_free(*(cursor->target));
775
776 if (value!=NULL)
777 *(cursor->target) = g_strdup (value);
778 else
779 *(cursor->target) = NULL;
780
781 inform_listeners =
782 (value==NULL && *(cursor->target)==NULL) ||
783 (value!=NULL && *(cursor->target)!=NULL &&
784 strcmp (value, *(cursor->target))==0);
785 }
786
787 if (inform_listeners)
788 queue_changed (cursor->pref);
789
790 g_free (value);
791
792 return TRUE;
793}

796handle_preference_update_int (const gchar *key, GSettings *settings)
797{
798 MetaIntPreference *cursor = preferences_int;
799 gint value;
800
801 while (cursor->key!=NULL && strcmp (key, cursor->key)!=0)
802 ++cursor;
803
804 if (cursor->key==NULL)
805 /* Didn't recognise that key. */
806 return FALSE;
807
808 if (cursor->target==NULL)
809 /* No work for us to do. */
810 return TRUE;
811
812 value = g_settings_get_int (settings, key);
813
814 if (value < cursor->minimum || value > cursor->maximum)
815 {
816 /* FIXME! GSettings, instead of MateConf, has Minimum/Maximun in schema!
817 * But some preferences depends on costants for minimum/maximum values */
818 meta_warning (_("%d stored in GSettings key %s is out of range %d to %d\n"),
819 value, cursor->key,
820 cursor->minimum, cursor->maximum);
821 return TRUE;
822 }
823
824 /* Did it change? If so, tell the listeners about it. */
825
826 if (*cursor->target != value)
827 {
828 *cursor->target = value;
829 queue_changed (cursor->pref);
830 }
831
832 return TRUE;
833
834}

881{

904static gboolean
905changed_idle_handler (gpointer data)
906{
907 GList *tmp;
908 GList *copy;
909
910 changed_idle = 0;
911
912 copy = g_list_copy (changes); /* reentrancy paranoia */
913
914 g_list_free (changes);
915 changes = NULL;
916
917 tmp = copy;
918 while (tmp != NULL)
919 {
920 MetaPreference pref = GPOINTER_TO_INT (tmp->data);
921
922 emit_changed (pref);
923
924 tmp = tmp->next;
925 }
926
927 g_list_free (copy);
928
929 return FALSE;

932static void
933queue_changed (MetaPreference pref)
934{
935 meta_topic (META_DEBUG_PREFS, "Queueing change of pref %s\n",
936 meta_preference_to_string (pref));
937

942 meta_preference_to_string (pref));

946 if (changed_idle == 0)
947 changed_idle = g_idle_add_full (META_PRIORITY_PREFS_NOTIFY,
948 changed_idle_handler, NULL, NULL);

1058

1072 g_free(str);

1077 meta_topic (META_DEBUG_PREFS, "Key %s doesn't mean anything to Marco\n",
1078 key);
1079 }
1080 g_free (schema_name);
1081}

1393static void
1394button_layout_handler (MetaPreference pref,
1395 const gchar *string_value,
1396 gboolean *inform_listeners)
1397{
1398 MetaButtonLayout new_layout;
1399 char **sides = NULL;
1400 int i;
1401
1402 /* We need to ignore unknown button functions, for
1403 * compat with future versions
1404 */
1405
1406 if (string_value)
1407 sides = g_strsplit (string_value, ":", 2);
1408
1409 if (sides != NULL && sides[0] != NULL)
1410 {
1411 char **buttons;
1412 int b;
1413 gboolean used[META_BUTTON_FUNCTION_LAST];
1414
1415 i = 0;
1416 while (i < META_BUTTON_FUNCTION_LAST)
1417 {
1418 used[i] = FALSE;
1419 new_layout.left_buttons_has_spacer[i] = FALSE;
1420 ++i;
1421 }
1422
1423 buttons = g_strsplit (sides[0], ",", -1);
1424 i = 0;
1425 b = 0;
1426 while (buttons[b] != NULL)
1427 {
1428 MetaButtonFunction f = button_function_from_string (buttons[b]);
1429 if (i > 0 && strcmp("spacer", buttons[b]) == 0)
1430 {
1431 new_layout.left_buttons_has_spacer[i-1] = TRUE;
1432 f = button_opposite_function (f);
1433
1434 if (f != META_BUTTON_FUNCTION_LAST)
1435 {
1436 new_layout.left_buttons_has_spacer[i-2] = TRUE;
1437 }
1438 }
1439 else
1440 {
1441 if (f != META_BUTTON_FUNCTION_LAST && !used[f])
1442 {
1443 new_layout.left_buttons[i] = f;
1444 used[f] = TRUE;
1445 ++i;
1446
1447 f = button_opposite_function (f);
1448
1449 if (f != META_BUTTON_FUNCTION_LAST)
1450 new_layout.left_buttons[i++] = f;
1451
1452 }
1453 else
1454 {
1455 meta_topic (META_DEBUG_PREFS, "Ignoring unknown or already-used button name \"%s\"\n",
1456 buttons[b]);
1457 }
1458 }
1459
1460 ++b;
1461 }
1462
1463 new_layout.left_buttons[i] = META_BUTTON_FUNCTION_LAST;
1464 new_layout.left_buttons_has_spacer[i] = FALSE;
1465
1466 g_strfreev (buttons);
1467 }
1468
1469 if (sides != NULL && sides[0] != NULL && sides[1] != NULL)
1470 {
1471 char **buttons;
1472 int b;
1473 gboolean used[META_BUTTON_FUNCTION_LAST];
1474
1475 i = 0;
1476 while (i < META_BUTTON_FUNCTION_LAST)
1477 {
1478 used[i] = FALSE;
1479 new_layout.right_buttons_has_spacer[i] = FALSE;
1480 ++i;
1481 }
1482
1483 buttons = g_strsplit (sides[1], ",", -1);
1484 i = 0;
1485 b = 0;
1486 while (buttons[b] != NULL)
1487 {
1488 MetaButtonFunction f = button_function_from_string (buttons[b]);
1489 if (i > 0 && strcmp("spacer", buttons[b]) == 0)
1490 {
1491 new_layout.right_buttons_has_spacer[i-1] = TRUE;
1492 f = button_opposite_function (f);
1493 if (f != META_BUTTON_FUNCTION_LAST)
1494 {
1495 new_layout.right_buttons_has_spacer[i-2] = TRUE;
1496 }
1497 }
1498 else
1499 {
1500 if (f != META_BUTTON_FUNCTION_LAST && !used[f])
1501 {
1502 new_layout.right_buttons[i] = f;
1503 used[f] = TRUE;
1504 ++i;
1505
1506 f = button_opposite_function (f);
1507
1508 if (f != META_BUTTON_FUNCTION_LAST)
1509 new_layout.right_buttons[i++] = f;
1510
1511 }
1512 else
1513 {
1514 meta_topic (META_DEBUG_PREFS, "Ignoring unknown or already-used button name \"%s\"\n",
1515 buttons[b]);
1516 }
1517 }
1518
1519 ++b;
1520 }
1521
1522 new_layout.right_buttons[i] = META_BUTTON_FUNCTION_LAST;
1523 new_layout.right_buttons_has_spacer[i] = FALSE;
1524
1525 g_strfreev (buttons);
1526 }
1527
1528 g_strfreev (sides);
1529
1530 /* Invert the button layout for RTL languages */
1531 if (meta_ui_get_direction() == META_UI_DIRECTION_RTL)
1532 {
1533 MetaButtonLayout rtl_layout;
1534 int j;
1535
1536 for (i = 0; new_layout.left_buttons[i] != META_BUTTON_FUNCTION_LAST; i++);
1537 for (j = 0; j < i; j++)
1538 {
1539 rtl_layout.right_buttons[j] = new_layout.left_buttons[i - j - 1];
1540 if (j == 0)

1873static gboolean
1874update_binding (MetaKeyPref *binding,
1875 gchar *value)
1876{
1877 unsigned int keysym;
1878 unsigned int keycode;
1879 MetaVirtualModifier mods;
1880 MetaKeyCombo *combo;
1881 gboolean changed;
1882
1883 meta_topic (META_DEBUG_KEYBINDINGS,
1884 "Binding \"%s\" has new GSettings value \"%s\"\n",
1885 binding->name, value ? value : "none");
1886
1887 keysym = 0;
1888 keycode = 0;
1889 mods = 0;
1890 if (value)
1891 {
1892 if (!meta_ui_parse_accelerator (value, &keysym, &keycode, &mods))
1893 {
1894 meta_topic (META_DEBUG_KEYBINDINGS,
1895 "Failed to parse new GSettings value\n");
1896 meta_warning (_("\"%s\" found in configuration database is not a valid value for keybinding \"%s\"\n"),
1897 value, binding->name);
1898
1899 return FALSE;
1900 }
1901 }
1902
1903 /* If there isn't already a first element, make one. */
1904 if (!binding->bindings)
1905 {
1906 MetaKeyCombo *blank = g_malloc0 (sizeof (MetaKeyCombo));
1907 binding->bindings = g_slist_alloc();
1908 binding->bindings->data = blank;
1909 }
1910
1911 combo = binding->bindings->data;
1912
1913 /* Bug 329676: Bindings which can be shifted must not have no modifiers,
1914 * nor only SHIFT as a modifier.
1915 */
1916
1917 if (binding->add_shift &&
1918 0 != keysym &&
1919 (META_VIRTUAL_SHIFT_MASK == mods || 0 == mods))
1920 {
1921 gchar *old_setting;
1922
1923 meta_warning ("Cannot bind \"%s\" to %s: it needs a modifier "
1924 "such as Ctrl or Alt.\n",
1925 binding->name,
1926 value);
1927
1928 old_setting = meta_ui_accelerator_name (combo->keysym,
1929 combo->modifiers);
1930
1931 if (!strcmp(old_setting, value))
1932 {
1933 /* We were about to set it to the same value
1934 * that it had originally! This must be caused
1935 * by getting an invalid string back from
1936 * meta_ui_accelerator_name. Bail out now
1937 * so we don't get into an infinite loop.
1938 */
1939 g_free (old_setting);
1940 return TRUE;
1941 }
1942
1943 meta_warning ("Reverting \"%s\" to %s.\n",
1944 binding->name,
1945 old_setting);
1946
1947 /* FIXME: add_shift is currently screen_bindings only, but
1948 * there's no really good reason it should always be.
1949 * So we shouldn't blindly add KEY_SCREEN_BINDINGS_PREFIX
1950 * onto here.
1951 */
1952 g_settings_set_string(settings_screen_bindings,
1953 binding->name,
1954 old_setting);
1955
1956 g_free (old_setting);
1957
1958 /* The call to g_settings_set_string() will cause this function
1959 * to be called again with the new value, so there's no need to
1960 * carry on.
1961 */
1962 return TRUE;
1963 }
1964
1965 changed = FALSE;
1966 if (keysym != combo->keysym ||
1967 keycode != combo->keycode ||
1968 mods != combo->modifiers)
1969 {
1970 changed = TRUE;
1971
1972 combo->keysym = keysym;
1973 combo->keycode = keycode;
1974 combo->modifiers = mods;
1975
1976 meta_topic (META_DEBUG_KEYBINDINGS,
1977 "New keybinding for \"%s\" is keysym = 0x%x keycode = 0x%x mods = 0x%x\n",
1978 binding->name, combo->keysym, combo->keycode,
1979 combo->modifiers);
1980 }
1981 else
1982 {
1983 meta_topic (META_DEBUG_KEYBINDINGS,
1984 "Keybinding for \"%s\" is unchanged\n", binding->name);
1985 }
1986
1987 return changed;
1988}
