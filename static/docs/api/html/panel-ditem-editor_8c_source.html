202static PanelDItemEditorType
203map_type_from_desktop_item (const char *type,
204 gboolean terminal)
205{
206 if (type == NULL)
207 return PANEL_DITEM_EDITOR_TYPE_NULL;
208 else if (!strcmp (type, "Application")) {
209 if (terminal)
210 return PANEL_DITEM_EDITOR_TYPE_TERMINAL_APPLICATION;
211 else
212 return PANEL_DITEM_EDITOR_TYPE_APPLICATION;
213 } else if (!strcmp (type, "Link"))
214 return PANEL_DITEM_EDITOR_TYPE_LINK;
215 else if (!strcmp (type, "Directory"))
216 return PANEL_DITEM_EDITOR_TYPE_DIRECTORY;
217 else
218 return PANEL_DITEM_EDITOR_TYPE_NULL;
219}

221static GObject *
222panel_ditem_editor_constructor (GType type,
223 guint n_construct_properties,
224 GObjectConstructParam *construct_properties)
225{
226 GObject *obj;
227 PanelDItemEditor *dialog;
228 gboolean loaded;
229 char *desktop_type;
230
231 obj = G_OBJECT_CLASS (panel_ditem_editor_parent_class)->constructor (type,
232 n_construct_properties,
233 construct_properties);
234
235 dialog = PANEL_DITEM_EDITOR (obj);
236
237 /* Icon */
238 dialog->priv->icon_chooser = panel_icon_chooser_new (NULL);
239 panel_icon_chooser_set_fallback_icon_name (PANEL_ICON_CHOOSER (dialog->priv->icon_chooser),
240 PANEL_ICON_LAUNCHER);
241 gtk_box_pack_start (GTK_BOX (dialog->priv->icon_chooser_box), dialog->priv->icon_chooser, FALSE, FALSE, 0);
242 gtk_widget_set_valign (dialog->priv->icon_chooser, GTK_ALIGN_START);
243 gtk_widget_show (dialog->priv->icon_chooser);
244
245 if (dialog->priv->key_file) {
246 panel_ditem_editor_key_file_loaded (dialog);
247 dialog->priv->new_file = FALSE;
248 dialog->priv->free_key_file = FALSE;
249 loaded = TRUE;
250 } else {
251 dialog->priv->key_file = panel_key_file_new_desktop ();
252 if (dialog->priv->type_directory)
253 panel_key_file_set_string (dialog->priv->key_file,
254 "Type", "Directory");
255 dialog->priv->free_key_file = TRUE;
256 loaded = FALSE;
257 }
258
259 if (!loaded && dialog->priv->uri) {
260 GFile *file = g_file_new_for_uri (dialog->priv->uri);
261 if (g_file_query_exists (file, NULL)) {
262 /* FIXME what if there's an error? */
263 panel_ditem_editor_load_uri (dialog, NULL);
264 dialog->priv->new_file = FALSE;
265 } else {
266 dialog->priv->new_file = TRUE;
267 }
268 g_object_unref (file);
269 } else {
270 dialog->priv->new_file = !loaded;
271 }
272
273 dialog->priv->dirty = FALSE;
274
275 desktop_type = panel_key_file_get_string (dialog->priv->key_file,
276 "Type");
277 if (desktop_type && !strcmp (desktop_type, "Directory"))
278 dialog->priv->type_directory = TRUE;
279 g_free (desktop_type);
280
281 panel_ditem_editor_setup_ui (dialog);
282
283 if (dialog->priv->new_file)
284 setup_icon_chooser (dialog, NULL);
285
286 panel_ditem_editor_connect_signals (dialog);
287
288 return obj;
289}

350static void
351panel_ditem_editor_dispose (GObject *object)
352{
353 PanelDItemEditor *dialog;
354
355 dialog = PANEL_DITEM_EDITOR (object);
356
357 /* If there was a timeout, then something changed after last save,
358 * so we must save again now */
359 if (dialog->priv->save_timeout) {
360 g_source_remove (dialog->priv->save_timeout);
361 dialog->priv->save_timeout = 0;
362 panel_ditem_editor_save (dialog, FALSE);
363 }
364
365 /* remember, destroy can be run multiple times! */
366
367 if (dialog->priv->free_key_file && dialog->priv->key_file != NULL)
368 g_key_file_free (dialog->priv->key_file);
369 dialog->priv->key_file = NULL;
370
371 if (dialog->priv->revert_key_file != NULL)
372 g_key_file_free (dialog->priv->revert_key_file);
373 dialog->priv->revert_key_file = NULL;
374
375 g_clear_pointer (&dialog->priv->uri, g_free);
376
377 G_OBJECT_CLASS (panel_ditem_editor_parent_class)->dispose (object);
378}

521static void
522setup_combo (GtkWidget *combo_box,
523 ComboItem *items,
524 int nb_items,
525 const char *for_type)
526{
527 GtkListStore *model;
528 GtkTreeIter iter;
529 GtkCellRenderer *renderer;
530 int i;
531

534 G_TYPE_INT);
535
536 gtk_combo_box_set_model (GTK_COMBO_BOX (combo_box),
537 GTK_TREE_MODEL (model));
538
539 for (i = 0; i < nb_items; i++) {
540 if (for_type && strcmp (for_type, items [i].show_for))
541 continue;
542
543 gtk_list_store_append (model, &iter);
544 gtk_list_store_set (model, &iter,

547 -1);
548 }
549
550 renderer = gtk_cell_renderer_text_new ();
551 gtk_cell_layout_pack_start (GTK_CELL_LAYOUT (combo_box),
552 renderer, TRUE);
553 gtk_cell_layout_set_attributes (GTK_CELL_LAYOUT (combo_box),
554 renderer, "text", COLUMN_TEXT, NULL);
555
556 gtk_combo_box_set_active (GTK_COMBO_BOX (combo_box), 0);
557}

651/*
652 * Will save after SAVE_FREQUENCY milliseconds of no changes. If something is
653 * changed, the save is postponed to another SAVE_FREQUENCY seconds. This seems
654 * to be a saner behaviour than just saving every N seconds.
655 */
656static void
657panel_ditem_editor_changed (PanelDItemEditor *dialog)
658{
659 if (!dialog->priv->new_file) {
660 if (dialog->priv->save_timeout != 0)
661 g_source_remove (dialog->priv->save_timeout);
662
663 dialog->priv->save_timeout
664 = g_timeout_add_seconds (SAVE_FREQUENCY,
665 panel_ditem_editor_save_timeout,
666 dialog);
667
668 /* We can revert to the original state */
669 if (dialog->priv->revert_key_file != NULL)
670 gtk_dialog_set_response_sensitive (GTK_DIALOG (dialog),
671 REVERT_BUTTON,
672 TRUE);
673 }
674
675 dialog->priv->dirty = TRUE;
676 g_signal_emit (G_OBJECT (dialog), ditem_edit_signals[CHANGED], 0);
677}

1067/* Conform display to ditem */
1068void
1069panel_ditem_editor_sync_display (PanelDItemEditor *dialog)
1070{
1071 char *type;
1072 PanelDItemEditorType editor_type;
1073 gboolean run_in_terminal;
1074 GKeyFile *key_file;
1075 char *buffer;
1076 GtkTreeIter iter;
1077 GtkTreeModel *model;
1078 PanelDItemEditorType buf_type;
1079
1080 g_return_if_fail (PANEL_IS_DITEM_EDITOR (dialog));
1081
1082 key_file = dialog->priv->key_file;
1083
1084 /* Name */
1085 buffer = panel_key_file_get_locale_string (key_file, "X-MATE-FullName");
1086 if (!buffer)
1087 buffer = panel_key_file_get_locale_string (key_file, "Name");
1088 gtk_entry_set_text (GTK_ENTRY (dialog->priv->name_entry),
1089 buffer ? buffer : "");
1090 g_free (buffer);
1091
1092 /* Type */
1093 type = panel_key_file_get_string (key_file, "Type");
1094 if (!dialog->priv->combo_setuped) {
1095 setup_combo (dialog->priv->type_combo,
1096 type_items, G_N_ELEMENTS (type_items),
1097 type);
1098 dialog->priv->combo_setuped = TRUE;
1099 }
1100
1101 run_in_terminal = panel_key_file_get_boolean (key_file, "Terminal",
1102 FALSE);
1103 editor_type = map_type_from_desktop_item (type, run_in_terminal);
1104 g_free (type);
1105
1106 model = gtk_combo_box_get_model (GTK_COMBO_BOX (dialog->priv->type_combo));
1107 if (!gtk_tree_model_get_iter_first (model, &iter))
1108 g_assert_not_reached ();
1109 do {
1110 gtk_tree_model_get (model, &iter, COLUMN_TYPE, &buf_type, -1);
1111 if (editor_type == buf_type) {
1112 gtk_combo_box_set_active_iter (GTK_COMBO_BOX (dialog->priv->type_combo),
1113 &iter);
1114 break;
1115 }
1116 } while (gtk_tree_model_iter_next (model, &iter));
1117
1118 g_assert (editor_type == buf_type ||
1119 editor_type == PANEL_DITEM_EDITOR_TYPE_NULL);
1120
1121 /* Command */
1122 if (editor_type == PANEL_DITEM_EDITOR_TYPE_LINK)
1123 buffer = panel_key_file_get_string (key_file, "URL");
1124 else if (editor_type == PANEL_DITEM_EDITOR_TYPE_APPLICATION ||
1125 editor_type == PANEL_DITEM_EDITOR_TYPE_TERMINAL_APPLICATION)
1126 buffer = panel_key_file_get_string (key_file, "Exec");
1127 else
1128 buffer = NULL;
1129
1130 gtk_entry_set_text (GTK_ENTRY (dialog->priv->command_entry),
1131 buffer ? buffer : "");
1132 g_free (buffer);
1133
1134 /* Comment */
1135 buffer = panel_key_file_get_locale_string (key_file, "Comment");
1136 gtk_entry_set_text (GTK_ENTRY (dialog->priv->comment_entry),
1137 buffer ? buffer : "");
1138 g_free (buffer);
1139
1140 /* Icon */
1141 buffer = panel_key_file_get_locale_string (key_file, "Icon");
1142 setup_icon_chooser (dialog, buffer);
1143 g_free (buffer);
1144
1145 if (dialog->priv->save_timeout != 0) {
1146 g_source_remove (dialog->priv->save_timeout);
1147 dialog->priv->save_timeout = 0;
1148 }
1149}

1151static gboolean
1152panel_ditem_editor_save (PanelDItemEditor *dialog,
1153 gboolean report_errors)
1154{
1155 GKeyFile *key_file;
1156 const char *const_buf;
1157 GError *error;
1158
1159 g_return_val_if_fail (dialog != NULL, FALSE);
1160 g_return_val_if_fail (dialog->priv->save_uri != NULL ||
1161 dialog->priv->uri != NULL, FALSE);
1162
1163 if (dialog->priv->save_timeout != 0)
1164 g_source_remove (dialog->priv->save_timeout);
1165 dialog->priv->save_timeout = 0;
1166
1167 if (!dialog->priv->dirty)
1168 return TRUE;
1169
1170 /* Verify that the required informations are set */
1171 const_buf = gtk_entry_get_text (GTK_ENTRY (dialog->priv->name_entry));
1172 if (const_buf == NULL || const_buf [0] == '\0') {
1173 if (report_errors) {
1174 if (!dialog->priv->type_directory)
1175 g_signal_emit (G_OBJECT (dialog),
1176 ditem_edit_signals[ERROR_REPORTED], 0,
1177 _("Could not save launcher"),
1178 _("The name of the launcher is not set."));
1179 else
1180 g_signal_emit (G_OBJECT (dialog),
1181 ditem_edit_signals[ERROR_REPORTED], 0,
1182 _("Could not save directory properties"),
1183 _("The name of the directory is not set."));
1184 }
1185 return FALSE;
1186 }
1187
1188 const_buf = gtk_entry_get_text (GTK_ENTRY (dialog->priv->command_entry));
1189 if (!dialog->priv->type_directory &&
1190 (const_buf == NULL || const_buf [0] == '\0')) {
1191 PanelDItemEditorType type;
1192 char *err;
1193
1194 type = panel_ditem_editor_get_item_type (dialog);
1195
1196 switch (type) {
1197 case PANEL_DITEM_EDITOR_TYPE_APPLICATION:
1198 case PANEL_DITEM_EDITOR_TYPE_TERMINAL_APPLICATION:
1199 err = _("The command of the launcher is not set.");
1200 break;
1201 case PANEL_DITEM_EDITOR_TYPE_LINK:
1202 err = _("The location of the launcher is not set.");
1203 break;
1204 default:
1205 g_assert_not_reached ();
1206 }
1207
1208 if (report_errors)
1209 g_signal_emit (G_OBJECT (dialog),
1210 ditem_edit_signals[ERROR_REPORTED], 0,
1211 _("Could not save launcher"), err);
1212 return FALSE;
1213 }
1214
1215 key_file = dialog->priv->key_file;
1216
1217 panel_key_file_ensure_C_key (key_file, "Name");
1218 panel_key_file_ensure_C_key (key_file, "Comment");
1219 panel_key_file_ensure_C_key (key_file, "Icon");
1220
1221 if (dialog->priv->save_uri) {
1222 char *uri;
1223
1224 uri = dialog->priv->save_uri (dialog, dialog->priv->save_uri_data);
1225
1226 if (uri) {
1227 panel_ditem_editor_set_uri (dialog, uri);
1228 g_free (uri);
1229 }
1230 }
1231
1232 /* And now, try to save */
1233 error = NULL;
1234 panel_key_file_to_file (dialog->priv->key_file,
1235 dialog->priv->uri,
1236 &error);
1237 if (error != NULL) {
1238 if (report_errors)
1239 g_signal_emit (G_OBJECT (dialog),
1240 ditem_edit_signals[ERROR_REPORTED], 0,
1241 _("Could not save launcher"),
1242 error->message);
1243 g_error_free (error);
1244 return FALSE;
1245 } else {
1246 g_signal_emit (G_OBJECT (dialog),
1247 ditem_edit_signals[SAVED], 0);
1248 }
1249
1250 dialog->priv->dirty = FALSE;
1251
1252 return TRUE;
1253}

1308static void
1309panel_ditem_editor_revert (PanelDItemEditor *dialog)
1310{
1311 gsize i;
1312 char *string;
1313 gboolean boolean;
1314 GKeyFile *key_file;
1315 GKeyFile *revert_key_file;
1316
1317 g_return_if_fail (PANEL_IS_DITEM_EDITOR (dialog));
1318
1319 dialog->priv->reverting = TRUE;
1320
1321 key_file = dialog->priv->key_file;
1322 revert_key_file = dialog->priv->revert_key_file;
1323
1324 g_assert (revert_key_file != NULL);
1325
1326 for (i = 0; i < G_N_ELEMENTS (revert_keys); i++) {
1327 if (revert_keys [i].type == G_TYPE_STRING) {
1328 if (revert_keys [i].locale) {
1329 string = panel_key_file_get_locale_string (
1330 revert_key_file,
1331 revert_keys [i].key);
1332 if (string == NULL)
1333 panel_key_file_remove_all_locale_key (
1334 key_file,
1335 revert_keys [i].key);
1336 else
