36void
37mateweather_prefs_load (MateWeatherPrefs *prefs, GSettings *settings)
38{
39 g_return_if_fail (prefs != NULL);
40 g_return_if_fail (settings != NULL);
41
42 if (prefs->location) {
43 weather_location_free (prefs->location);
44 }
45 gchar *name, *code, *zone, *radar, *coordinates;
46 name = g_settings_get_string (settings, "location4");
47 code = g_settings_get_string (settings, "location1");
48 zone = g_settings_get_string (settings, "location2");
49 radar = g_settings_get_string (settings, "location3");
50 coordinates = g_settings_get_string (settings, "coordinates");
51 prefs->location = weather_location_new (name, code, zone, radar, coordinates,
52 NULL, NULL);
53
54 g_free (name);
55 g_free (code);
56 g_free (zone);
57 g_free (radar);
58 g_free (coordinates);
59
60 prefs->show_notifications =
61 g_settings_get_boolean (settings, "show-notifications");
62 prefs->update_interval =
63 g_settings_get_int (settings, "auto-update-interval");
64 prefs->update_interval = MAX (prefs->update_interval, 60);
65 prefs->update_enabled =
66 g_settings_get_boolean (settings, "auto-update");
67 prefs->detailed =
68 g_settings_get_boolean (settings, "enable-detailed-forecast");
69 prefs->radar_enabled =
70 g_settings_get_boolean (settings, "enable-radar-map");
71 prefs->use_custom_radar_url =
72 g_settings_get_boolean (settings, "use-custom-radar-url");
73
74 if (prefs->radar) {
75 g_free (prefs->radar);
76 prefs->radar = NULL;
77 }
78 prefs->radar = g_settings_get_string (settings, "radar");
79
80 prefs->temperature_unit = g_settings_get_enum (settings, GSETTINGS_TEMP_UNIT);
81 prefs->speed_unit = g_settings_get_enum (settings, GSETTINGS_SPEED_UNIT);
82 prefs->pressure_unit = g_settings_get_enum (settings, GSETTINGS_PRESSURE_UNIT);
83 prefs->distance_unit = g_settings_get_enum (settings, GSETTINGS_DISTANCE_UNIT);
84
85 return;
86}
