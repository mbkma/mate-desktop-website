165static void
166setup_stars (GSTEStarfield *sf)
167{
168 int i;
169
170 if (sf->priv->stars)
171 {
172 g_free (sf->priv->stars);
173 }
174 sf->priv->stars = g_new(point, sf->priv->count);
175
176 for (i = 0; i < sf->priv->count; ++i)
177 {
178 point *s = &sf->priv->stars[i];
179 s->z = -cbrt(g_random_double() - 1); /* icdf for beta(1, 3) distribution */
180 s->x = s->z * g_random_double_range(-XY_CLIP_LIMIT, XY_CLIP_LIMIT);
181 s->y = s->z * g_random_double_range(-XY_CLIP_LIMIT, XY_CLIP_LIMIT);
182 }
183 sf->priv->timestamp = g_get_monotonic_time ();
184 sf->priv->current_speed = 0.0;
185}

187static double
188get_elapsed_time (GSTEStarfield *sf)
189{
190 int64_t prev_timestamp = sf->priv->timestamp;
191
192 sf->priv->timestamp = g_get_monotonic_time ();
193
194 int64_t elapsed = sf->priv->timestamp - prev_timestamp;
195
196 return (double)elapsed / 1e6;
197}

206static void
207gste_starfield_real_show (GtkWidget *widget)
208{
209 GSTEStarfield *sf = GSTE_STARFIELD (widget);
210
211 /* start */
212 setup_stars (sf);
213
214 if (sf->priv->timeout_id > 0)
215 {
216 g_source_remove (sf->priv->timeout_id);
217 }
218 sf->priv->timeout_id = g_timeout_add (sf->priv->delay,
219 (GSourceFunc)draw_iter,
220 sf);
221
222 if (GTK_WIDGET_CLASS (parent_class)->show)
223 {
224 GTK_WIDGET_CLASS (parent_class)->show (widget);
225 }
226}

327static void
328draw_frame (GSTEStarfield *sf,
329 cairo_t *cr)
330{
331 int i;
332 int window_width;
333 int window_height;
334 GdkWindow *window;
335
336 window = gs_theme_engine_get_window (GS_THEME_ENGINE (sf));
337
338 if (window == NULL)
339 {
340 return;
341 }
342
343 gs_theme_engine_get_window_size (GS_THEME_ENGINE (sf),
344 &window_width,
345 &window_height);
346
347 double elapsed = get_elapsed_time(sf);
348 double speed = sf->priv->current_speed;
349 double step = SPEED_FACTOR * speed * elapsed;
350 double max_radius = fmax(sf->priv->size * SIZE_RATIO * window_height, 1);
351
352 cairo_set_line_cap (cr, CAIRO_LINE_CAP_ROUND);
353 cairo_set_antialias (cr, CAIRO_ANTIALIAS_FAST);
354
355 for (i = 0; i < sf->priv->count; ++i)
356 {
357 point *s = &sf->priv->stars[i];
358
359 s->z -= step;
360
361 if (s->z <= Z_NEAR ||
362 s->x > s->z * XY_CLIP_LIMIT ||
363 s->x < s->z * -XY_CLIP_LIMIT ||
364 s->y > s->z * XY_CLIP_LIMIT ||
365 s->y < s->z * -XY_CLIP_LIMIT)
366 {
367 s->z = Z_FAR - g_random_double_range(0, fmin(step, Z_FAR - Z_NEAR));
368 s->x = s->z * g_random_double_range(-XY_CLIP_LIMIT, XY_CLIP_LIMIT);
369 s->y = s->z * g_random_double_range(-XY_CLIP_LIMIT, XY_CLIP_LIMIT);
370 }
371
372 double x = (s->x / s->z + XY_CLIP_LIMIT) * window_width;
373 double y = (s->y / s->z + XY_CLIP_LIMIT) * window_height;
374 double illum = 1.0 / (1.0 + s->z);
375 illum *= illum;
376 double radius = max_radius * (1.0 - s->z);
377
378 cairo_set_source_rgb (cr, illum, illum, illum);
379 cairo_set_line_width (cr, radius);
380 cairo_move_to (cr, x, y);
381 cairo_line_to (cr, x, y);
382 cairo_stroke (cr);
383 }
384 if (speed != sf->priv->speed)
385 {
386 speed += sf->priv->acceleration * elapsed;
387 sf->priv->current_speed = fmin(speed, sf->priv->speed);
388 }
389}
