29void
30caja_view_factory_register (CajaViewInfo *view_info)
31{
32 g_return_if_fail (view_info != NULL);
33 g_return_if_fail (view_info->id != NULL);
34 g_return_if_fail (caja_view_factory_lookup (view_info->id) == NULL);
35
36 registered_views = g_list_append (registered_views, view_info);
37}

39const CajaViewInfo *
40caja_view_factory_lookup (const char *id)
41{
42 GList *l;
43 CajaViewInfo *view_info = NULL;
44
45 g_return_val_if_fail (id != NULL, NULL);
46
47 for (l = registered_views; l != NULL; l = l->next)
48 {
49 view_info = l->data;
50
51 if (strcmp (view_info->id, id) == 0)
52 {
53 return view_info;
54 }
55 }
56 return NULL;
57}

59CajaView *
60caja_view_factory_create (const char *id,
61 CajaWindowSlotInfo *slot)
62{
63 const CajaViewInfo *view_info;
64 CajaView *view;
65
66 view_info = caja_view_factory_lookup (id);
67 if (view_info == NULL)
68 {
69 return NULL;
70 }
71
72 view = view_info->create (slot);
73 if (g_object_is_floating (view))
74 {
75 g_object_ref_sink (view);
76 }
77 return view;

86 const CajaViewInfo *view_info;

102GList *
103caja_view_factory_get_views_for_uri (const char *uri,
104 GFileType file_type,
105 const char *mime_type)
106{
107 GList *l, *res;
108 const CajaViewInfo *view_info = NULL;
109
110 res = NULL;
111
112 for (l = registered_views; l != NULL; l = l->next)
113 {
114 view_info = l->data;
115
116 if (view_info->supports_uri (uri, file_type, mime_type))
117 {
118 if (view_info->single_view)
119 {
120 g_list_free_full (res, g_free);
121 res = g_list_prepend (NULL, g_strdup (view_info->id));
122 break;
123 } else {
124 res = g_list_prepend (res, g_strdup (view_info->id));
125 }
126 }
127 }
128
129 return g_list_reverse (res);
130}
