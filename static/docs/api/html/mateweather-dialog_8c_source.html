87static gchar*
88replace_multiple_new_lines(gchar* s)
89{
90 gchar *prev_s = s;
91 gint count;
92 gint i;
93
94 if (s == NULL) {
95 return s;
96 }
97
98 for (;*s != '\0';s++) {
99
100 count = 0;
101
102 if (*s == '\n') {
103 s++;
104 while (*s == '\n' || *s == ' ') {
105 count++;
106 s++;
107 }
108 }
109 for (i = count; i > 1; i--) {
110 *(s - i) = ' ';
111 }
112 }
113 return prev_s;
114}

116static PangoFontDescription* get_system_monospace_font(void)
117{
118 PangoFontDescription *desc = NULL;
119 GSettings *settings;
120 char *name;
121
122 settings = g_settings_new (MONOSPACE_FONT_SCHEMA);
123 name = g_settings_get_string (settings, MONOSPACE_FONT_KEY);
124
125 if (name) {
126 desc = pango_font_description_from_string (name);
127 g_free (name);
128 }
129
130 g_object_unref (settings);
131
132 return desc;
133}

135static void
136override_widget_font (GtkWidget *widget,
137 PangoFontDescription *font)
138{
139 static gboolean provider_added = FALSE;
140 static GtkCssProvider *provider;
141 gchar *css;
142 gchar *family;
143 gchar *weight;
144 const gchar *style;
145 gchar *size;
146
147 family = g_strdup_printf ("font-family: %s;", pango_font_description_get_family (font));
148
149 weight = g_strdup_printf ("font-weight: %d;", pango_font_description_get_weight (font));
150
151 if (pango_font_description_get_style (font) == PANGO_STYLE_NORMAL)
152 style = "font-style: normal;";
153 else if (pango_font_description_get_style (font) == PANGO_STYLE_ITALIC)
154 style = "font-style: italic;";
155 else
156 style = "font-style: oblique;";
157
158 size = g_strdup_printf ("font-size: %d%s;",
159 pango_font_description_get_size (font) / PANGO_SCALE,
160 pango_font_description_get_size_is_absolute (font) ? "px" : "pt");
161 if (!provider_added)
162 provider = gtk_css_provider_new ();
163
164 gtk_widget_set_name(GTK_WIDGET(widget), "MateWeatherAppletTextView");
165 css = g_strdup_printf ("#MateWeatherAppletTextView { %s %s %s %s }", family, weight, style, size);
166 gtk_css_provider_load_from_data (provider, css, -1, NULL);
167
168 if (!provider_added) {
169 gtk_style_context_add_provider_for_screen (gtk_widget_get_screen (widget),
170 GTK_STYLE_PROVIDER (provider),
171 GTK_STYLE_PROVIDER_PRIORITY_APPLICATION);
172 provider_added = TRUE;
173 }
174
175 g_free (css);
176 g_free (family);
177 g_free (weight);
178 g_free (size);
179}

181void
182mateweather_dialog_update (MateWeatherDialog *dialog)
183{
184 WeatherInfo *info;
185 gchar *forecast;
186 GtkTextBuffer *buffer;
187 PangoFontDescription *font_desc;
188 const gchar *icon_name;
189
190 /* Check for parallel network update in progress */
191 if (dialog->applet->mateweather_info == NULL)
192 return;
193
194 info = dialog->applet->mateweather_info;
195
196 /* Update icon */
197 icon_name = weather_info_get_icon_name (info);
198 gtk_image_set_from_icon_name (GTK_IMAGE (dialog->cond_image),
199 icon_name, GTK_ICON_SIZE_DIALOG);
200
201 /* Update current condition fields and forecast */
202 gtk_label_set_text (GTK_LABEL (dialog->cond_location), weather_info_get_location_name (info));
203 gtk_label_set_text (GTK_LABEL (dialog->cond_update), weather_info_get_update (info));
204 gtk_label_set_text (GTK_LABEL (dialog->cond_cond), weather_info_get_conditions (info));
205 gtk_label_set_text (GTK_LABEL (dialog->cond_sky), weather_info_get_sky (info));
206 gtk_label_set_text (GTK_LABEL (dialog->cond_temp), weather_info_get_temp (info));
207 gtk_label_set_text (GTK_LABEL (dialog->cond_apparent), weather_info_get_apparent (info));
208 gtk_label_set_text (GTK_LABEL (dialog->cond_dew), weather_info_get_dew (info));
209 gtk_label_set_text (GTK_LABEL (dialog->cond_humidity), weather_info_get_humidity (info));
210 gtk_label_set_text (GTK_LABEL (dialog->cond_wind), weather_info_get_wind (info));
211 gtk_label_set_text (GTK_LABEL (dialog->cond_pressure), weather_info_get_pressure (info));
212 gtk_label_set_text (GTK_LABEL (dialog->cond_vis), weather_info_get_visibility (info));
213 gtk_label_set_text (GTK_LABEL (dialog->cond_sunrise), weather_info_get_sunrise (info));
214 gtk_label_set_text (GTK_LABEL (dialog->cond_sunset), weather_info_get_sunset (info));
215
216 /* Update forecast */
217 if (dialog->applet->mateweather_pref.location->zone_valid) {
218 font_desc = get_system_monospace_font ();
219 if (font_desc) {
220 override_widget_font (dialog->forecast_text, font_desc);
221 pango_font_description_free (font_desc);
222 }
223
224 buffer = gtk_text_view_get_buffer (GTK_TEXT_VIEW (dialog->forecast_text));
225 forecast = g_strdup (weather_info_get_forecast (dialog->applet->mateweather_info));
226 if (forecast) {
227 forecast = g_strstrip (replace_multiple_new_lines (forecast));
228 gtk_text_buffer_set_text (buffer, forecast, -1);
229 g_free (forecast);
230 } else {
231 gtk_text_buffer_set_text (buffer, _("Forecast not currently available for this location."), -1);
232 }
233 gtk_widget_show (gtk_notebook_get_nth_page (GTK_NOTEBOOK (dialog->weather_notebook), 1));
234 } else {
235 gtk_widget_hide (gtk_notebook_get_nth_page (GTK_NOTEBOOK (dialog->weather_notebook), 1));
236 }
237
238 /* Update radar map */
239 if (dialog->applet->mateweather_pref.radar_enabled) {
240 GdkPixbufAnimation *radar;
241
242 radar = weather_info_get_radar (info);
243 if (radar) {
244 gtk_image_set_from_animation (GTK_IMAGE (dialog->radar_image), radar);
245 }
246 gtk_widget_show (gtk_notebook_get_nth_page (GTK_NOTEBOOK (dialog->weather_notebook), 2));
247 gtk_window_set_default_size (GTK_WINDOW (dialog), 570, 440);
248 } else {
249 gtk_widget_hide (gtk_notebook_get_nth_page (GTK_NOTEBOOK (dialog->weather_notebook), 2));
250 gtk_window_set_default_size (GTK_WINDOW (dialog), 590, 340);
251 }
252}
