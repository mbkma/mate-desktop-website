75static gboolean
76client_iochannel_watch (GIOChannel *channel,
77 GIOCondition condition,
78 GsmXSMPClient *client)
79{
80 gboolean keep_going;
81 GsmXSMPClientPrivate *priv;
82
83 g_object_ref (client);
84 priv = gsm_xsmp_client_get_instance_private (client);
85
86 switch (IceProcessMessages (priv->ice_connection, NULL, NULL)) {
87 case IceProcessMessagesSuccess:
88 keep_going = TRUE;
89 break;
90
91 case IceProcessMessagesIOError:
92 g_debug ("GsmXSMPClient: IceProcessMessagesIOError on '%s'", priv->description);
93 gsm_client_set_status (GSM_CLIENT (client), GSM_CLIENT_FAILED);
94 /* Emitting "disconnected" will eventually cause
95 * IceCloseConnection() to be called.
96 */
97 gsm_client_disconnected (GSM_CLIENT (client));
98 keep_going = FALSE;
99 break;
100
101 case IceProcessMessagesConnectionClosed:
102 g_debug ("GsmXSMPClient: IceProcessMessagesConnectionClosed on '%s'",
103 priv->description);
104 priv->ice_connection = NULL;
105 keep_going = FALSE;
106 break;
107
108 default:
109 g_assert_not_reached ();
110 }
111 g_object_unref (client);
112
113 return keep_going;
114}

116static SmProp *
117find_property (GsmXSMPClient *client,
118 const char *name,
119 int *index)
120{
121 SmProp *prop;
122 int i;
123 GsmXSMPClientPrivate *priv;
124
125 priv = gsm_xsmp_client_get_instance_private (client);
126
127 for (i = 0; i < priv->props->len; i++) {
128 prop = priv->props->pdata[i];
129
130 if (!strcmp (prop->name, name)) {
131 if (index) {
132 *index = i;
133 }
134 return prop;
135 }
136 }
137
138 return NULL;
139}

143{

157 (char *)prop->vals[0].value,
158 id);
159 } else if (id != NULL) {

161 } else {
162 priv->description = g_strdup_printf ("%p", client);

166static void
167setup_connection (GsmXSMPClient *client)
168{
169 GIOChannel *channel;
170 int fd;
171 GsmXSMPClientPrivate *priv;
172
173 priv = gsm_xsmp_client_get_instance_private (client);
174 g_debug ("GsmXSMPClient: Setting up new connection");
175
176 fd = IceConnectionNumber (priv->ice_connection);
177 fcntl (fd, F_SETFD, fcntl (fd, F_GETFD, 0) | FD_CLOEXEC);
178 channel = g_io_channel_unix_new (fd);
179 priv->watch_id = g_io_add_watch (channel,
180 G_IO_IN | G_IO_ERR,
181 (GIOFunc)client_iochannel_watch,
182 client);
183 g_io_channel_unref (channel);
184
185 set_description (client);
186
187 g_debug ("GsmXSMPClient: New client '%s'", priv->description);
188}

250static void
251debug_print_property (SmProp *prop)
252{
253 GString *tmp;
254 int i;
255
256 switch (prop->type[0]) {
257 case 'C': /* CARD8 */
258 g_debug ("GsmXSMPClient: %s = %d", prop->name, *(unsigned char *)prop->vals[0].value);
259 break;
260
261 case 'A': /* ARRAY8 */
262 g_debug ("GsmXSMPClient: %s = '%s'", prop->name, (char *)prop->vals[0].value);
263 break;
264
265 case 'L': /* LISTofARRAY8 */
266 tmp = g_string_new (NULL);
267 for (i = 0; i < prop->num_vals; i++) {
268 g_string_append_printf (tmp, "'%.*s' ", prop->vals[i].length,
269 (char *)prop->vals[i].value);
270 }
271 g_debug ("GsmXSMPClient: %s = %s", prop->name, tmp->str);
272 g_string_free (tmp, TRUE);
273 break;
274
275 default:
276 g_debug ("GsmXSMPClient: %s = ??? (%s)", prop->name, prop->type);
277 break;
278 }
279}

282set_properties_callback (SmsConn conn,
283 SmPointer manager_data,
284 int num_props,
285 SmProp **props)
286{
287 int i;
288 GsmXSMPClientPrivate *priv;

290

304

348static char *
349prop_to_command (SmProp *prop)
350{
351 GString *str;
352 int i, j;
353 gboolean need_quotes;
354
355 str = g_string_new (NULL);
356 for (i = 0; i < prop->num_vals; i++) {
357 char *val = prop->vals[i].value;
358
359 need_quotes = FALSE;
360 for (j = 0; j < prop->vals[i].length; j++) {
361 if (!g_ascii_isalnum (val[j]) && !strchr ("-_=:./", val[j])) {
362 need_quotes = TRUE;
363 break;
364 }
365 }
366
367 if (i > 0) {
368 g_string_append_c (str, ' ');
369 }
370
371 if (!need_quotes) {
372 g_string_append_printf (str,
373 "%.*s",
374 prop->vals[i].length,
375 (char *)prop->vals[i].value);
376 } else {
377 g_string_append_c (str, '\'');
378 while (val < (char *)prop->vals[i].value + prop->vals[i].length) {
379 if (*val == '\'') {
380 g_string_append (str, "'\''");
381 } else {
382 g_string_append_c (str, *val);
383 }
384 val++;
385 }
386 g_string_append_c (str, '\'');
387 }
388 }
389
390 return g_string_free (str, FALSE);
391}

421static void
422do_save_yourself (GsmXSMPClient *client,
423 int save_type,
424 gboolean allow_interact)
425{
426 GsmXSMPClientPrivate *priv;
427
428 priv = gsm_xsmp_client_get_instance_private (client);
429 g_assert (priv->conn != NULL);
430
431 if (priv->next_save_yourself != -1) {
432 /* Either we're currently doing a shutdown and there's a checkpoint
433 * queued after it, or vice versa. Either way, the new SaveYourself
434 * is redundant.
435 */
436 g_debug ("GsmXSMPClient: skipping redundant SaveYourself for '%s'",
437 priv->description);
438 } else if (priv->current_save_yourself != -1) {
439 g_debug ("GsmXSMPClient: queuing new SaveYourself for '%s'",
440 priv->description);
441 priv->next_save_yourself = save_type;
442 priv->next_save_yourself_allow_interact = (allow_interact != FALSE);
443 } else {
444 priv->current_save_yourself = save_type;
445 /* make sure we don't have anything queued */
446 priv->next_save_yourself = -1;
447 priv->next_save_yourself_allow_interact = FALSE;
448
449 switch (save_type) {
450 case SmSaveLocal:
451 /* Save state */
452 SmsSaveYourself (priv->conn,
453 SmSaveLocal,
454 FALSE,
455 SmInteractStyleNone,
456 FALSE);
457 break;
458
459 default:
460 /* Logout */
461 if (!allow_interact) {
462 SmsSaveYourself (priv->conn,
463 save_type, /* save type */
464 TRUE, /* shutdown */
465 SmInteractStyleNone, /* interact style */
466 TRUE); /* fast */
467 } else {
468 SmsSaveYourself (priv->conn,
469 save_type, /* save type */
470 TRUE, /* shutdown */
471 SmInteractStyleAny, /* interact style */
472 FALSE /* fast */);
473 }
474 break;
475 }
476 }
477}

531static char *
532get_desktop_file_path (GsmXSMPClient *client)
533{
534 SmProp *prop;
535 char *desktop_file_path = NULL;
536 char **dirs;
537 const char *program_name;
538
539 /* XSMP clients using eggsmclient defines a special property
540 * pointing to their respective desktop entry file */
541 prop = find_property (client, GsmDesktopFile, NULL);
542
543 if (prop) {
544 GFile *file = g_file_new_for_uri (prop->vals[0].value);
545 desktop_file_path = g_file_get_path (file);
546 g_object_unref (file);
547 goto out;
548 }
549
550 /* If we can't get desktop file from GsmDesktopFile then we
551 * try to find the desktop file from its program name */
552 prop = find_property (client, SmProgram, NULL);
553
554 if (!prop) {
555 goto out;
556 }
557
558 program_name = prop->vals[0].value;
559
560 dirs = gsm_util_get_autostart_dirs ();
561
562 desktop_file_path =
563 gsm_util_find_desktop_file_for_app_name (program_name,
564 dirs);
565
566 g_strfreev (dirs);
567
568out:
569 g_debug ("GsmXSMPClient: desktop file for client %s is %s",
570 gsm_client_peek_id (GSM_CLIENT (client)),
571 desktop_file_path ? desktop_file_path : "(null)");
572
573 return desktop_file_path;
574}

651static GKeyFile *
652xsmp_save (GsmClient *client,
653 GError **error)
654{
655 GsmClientRestartStyle restart_style;
656
657 GKeyFile *keyfile = NULL;
658 char *desktop_file_path = NULL;
659 char *exec_program = NULL;
660 char *exec_discard = NULL;
661 char *startup_id = NULL;
662 GError *local_error;
663
664 g_debug ("GsmXSMPClient: saving client with id %s",
665 gsm_client_peek_id (client));
666
667 local_error = NULL;
668
669 restart_style = xsmp_get_restart_style_hint (client);
670 if (restart_style == GSM_CLIENT_RESTART_NEVER) {
671 goto out;
672 }
673
674 exec_program = xsmp_get_restart_command (client);
675 if (!exec_program) {
676 goto out;
677 }
678
679 desktop_file_path = get_desktop_file_path (GSM_XSMP_CLIENT (client));
680
681 /* this can accept desktop_file_path == NULL */
682 keyfile = create_client_key_file (client,
683 desktop_file_path,
684 &local_error);
685
686 if (local_error) {
687 goto out;
688 }
689
690 g_object_get (client,
691 "startup-id", &startup_id,
692 NULL);
693
694 g_key_file_set_string (keyfile,
695 G_KEY_FILE_DESKTOP_GROUP,
696 GSM_AUTOSTART_APP_STARTUP_ID_KEY,
697 startup_id);
698
699 g_key_file_set_string (keyfile,
700 G_KEY_FILE_DESKTOP_GROUP,
701 G_KEY_FILE_DESKTOP_KEY_EXEC,
702 exec_program);
703
704 exec_discard = xsmp_get_discard_command (client);
705 if (exec_discard)
706 g_key_file_set_string (keyfile,
707 G_KEY_FILE_DESKTOP_GROUP,
708 GSM_AUTOSTART_APP_DISCARD_KEY,
709 exec_discard);
710
711out:
712 g_free (desktop_file_path);
713 g_free (exec_program);
714 g_free (exec_discard);
715 g_free (startup_id);
716
717 if (local_error != NULL) {
718 g_propagate_error (error, local_error);
719 g_key_file_free (keyfile);
720
721 return NULL;
722 }
723
724 return keyfile;
725}

750static gboolean
751xsmp_query_end_session (GsmClient *client,
752 guint flags,
753 GError **error)
754{
755 gboolean allow_interact;
756 int save_type;
757 GsmXSMPClientPrivate *priv;
758
759 priv = gsm_xsmp_client_get_instance_private (GSM_XSMP_CLIENT(client));
760
761 if (priv->conn == NULL) {
762 g_set_error (error,
763 GSM_CLIENT_ERROR,
764 GSM_CLIENT_ERROR_NOT_REGISTERED,
765 "Client is not registered");
766 return FALSE;
767 }
768
769 allow_interact = !(flags & GSM_CLIENT_END_SESSION_FLAG_FORCEFUL);
770
771 /* we don't want to save the session state, but we just want to know if
772 * there's user data the client has to save and we want to give the
773 * client a chance to tell the user about it. This is consistent with
774 * the manager not setting GSM_CLIENT_END_SESSION_FLAG_SAVE for this
775 * phase. */
776 save_type = SmSaveGlobal;
777
778 do_save_yourself (GSM_XSMP_CLIENT (client), save_type, allow_interact);
779 return TRUE;
780}

782static gboolean
783xsmp_end_session (GsmClient *client,
784 guint flags,
785 GError **error)
786{
787 gboolean phase2;
788 GsmXSMPClientPrivate *priv;
789
790 priv = gsm_xsmp_client_get_instance_private (GSM_XSMP_CLIENT(client));
791
792 if (priv->conn == NULL) {
793 g_set_error (error,
794 GSM_CLIENT_ERROR,
795 GSM_CLIENT_ERROR_NOT_REGISTERED,
796 "Client is not registered");
797 return FALSE;
798 }
799
800 phase2 = (flags & GSM_CLIENT_END_SESSION_FLAG_LAST);
801
802 if (phase2) {
803 xsmp_save_yourself_phase2 (client);
804 } else {
805 gboolean allow_interact;
806 int save_type;
807

980static gboolean

983{

992 return FALSE;
993 }
994

997 return FALSE;

1090static Status
1091register_client_callback (SmsConn conn,
1092 SmPointer manager_data,
1093 char *previous_id)
1094{
1095 gboolean handled;
1096 char *id;
1097 GsmXSMPClientPrivate *priv;
1098
1099 GsmXSMPClient *client = manager_data;
1100 priv = gsm_xsmp_client_get_instance_private (client);
1101
1102 g_debug ("GsmXSMPClient: Client '%s' received RegisterClient(%s)",
1103 priv->description,
1104 previous_id ? previous_id : "NULL");
1105
1106 /* There are three cases:
1107 * 1. id is NULL - we'll use a new one
1108 * 2. id is known - we'll use known one
1109 * 3. id is unknown - this is an error
1110 */
1111 id = g_strdup (previous_id);
1112
1113 handled = FALSE;
1114 g_signal_emit (client, signals[REGISTER_REQUEST], 0, &id, &handled);
1115 if (! handled) {
1116 g_debug ("GsmXSMPClient: RegisterClient not handled!");
1117 g_free (id);
1118 free (previous_id);
1119 g_assert_not_reached ();
1120 return FALSE;
1121 }
1122
1123 if (IS_STRING_EMPTY (id)) {
1124 g_debug ("GsmXSMPClient: rejected: invalid previous_id");
1125 free (previous_id);
1126 return FALSE;
1127 }
1128
1129 g_object_set (client, "startup-id", id, NULL);
1130
1131 set_description (client);
1132
1133 g_debug ("GsmXSMPClient: Sending RegisterClientReply to '%s'", priv->description);
1134
1135 SmsRegisterClientReply (conn, id);
1136
1137 if (IS_STRING_EMPTY (previous_id)) {
1138 /* Send the initial SaveYourself. */
1139 g_debug ("GsmXSMPClient: Sending initial SaveYourself");
1140 SmsSaveYourself (conn, SmSaveLocal, False, SmInteractStyleNone, False);
1141 priv->current_save_yourself = SmSaveLocal;
1142 }
1143
1144 gsm_client_set_status (GSM_CLIENT (client), GSM_CLIENT_REGISTERED);
1145
1146 g_free (id);
1147 free (previous_id);
1148
1149 return TRUE;
1150}

1152static void
1153save_yourself_request_callback (SmsConn conn,
1154 SmPointer manager_data,
1155 int save_type,
1156 Bool shutdown,
1157 int interact_style,
1158 Bool fast,
1159 Bool global)
1160{
1161 GsmXSMPClientPrivate *priv;
1162 GsmXSMPClient *client = manager_data;
1163
1164 priv = gsm_xsmp_client_get_instance_private (client);
1165
1166 g_debug ("GsmXSMPClient: Client '%s' received SaveYourselfRequest(%s, %s, %s, %s, %s)",
1167 priv->description,
1168 save_type == SmSaveLocal ? "SmSaveLocal" :
1169 save_type == SmSaveGlobal ? "SmSaveGlobal" : "SmSaveBoth",
1170 shutdown ? "Shutdown" : "!Shutdown",
1171 interact_style == SmInteractStyleAny ? "SmInteractStyleAny" :
1172 interact_style == SmInteractStyleErrors ? "SmInteractStyleErrors" :
1173 "SmInteractStyleNone", fast ? "Fast" : "!Fast",
1174 global ? "Global" : "!Global");
1175
1176 /* Examining the g_debug above, you can see that there are a total
1177 * of 72 different combinations of options that this could have been
1178 * called with. However, most of them are stupid.
1179 *
1180 * If @shutdown and @global are both TRUE, that means the caller is
1181 * requesting that a logout message be sent to all clients, so we do
1182 * that. We use @fast to decide whether or not to show a
1183 * confirmation dialog. (This isn't really what @fast is for, but
1184 * the old mate-session and ksmserver both interpret it that way,
1185 * so we do too.) We ignore @save_type because we pick the correct
1186 * save_type ourselves later based on user prefs, dialog choices,
1187 * etc, and we ignore @interact_style, because clients have not used
1188 * it correctly consistently enough to make it worth honoring.
1189 *
1190 * If @shutdown is TRUE and @global is FALSE, the caller is
1191 * confused, so we ignore the request.
1192 *
1193 * If @shutdown is FALSE and @save_type is SmSaveGlobal or
1194 * SmSaveBoth, then the client wants us to ask some or all open
1195 * applications to save open files to disk, but NOT quit. This is
1196 * silly and so we ignore the request.
1197 *
1198 * If @shutdown is FALSE and @save_type is SmSaveLocal, then the
1199 * client wants us to ask some or all open applications to update
1200 * their current saved state, but not log out. At the moment, the
1201 * code only supports this for the !global case (ie, a client
1202 * requesting that it be allowed to update *its own* saved state,
1203 * but not having everyone else update their saved state).
1204 */
1205
1206 if (shutdown && global) {
1207 g_debug ("GsmXSMPClient: initiating shutdown");
1208 g_signal_emit (client, signals[LOGOUT_REQUEST], 0, !fast);
1209 } else if (!shutdown && !global) {
1210 g_debug ("GsmXSMPClient: initiating checkpoint");
1211 do_save_yourself (client, SmSaveLocal, TRUE);
1212 } else {
1213 g_debug ("GsmXSMPClient: ignoring");
1214 }
1215}
