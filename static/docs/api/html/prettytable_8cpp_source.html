72PrettyTable::on_application_opened(WnckScreen* screen, WnckApplication* app, gpointer data)
73{
74 PrettyTable * const that = static_cast<PrettyTable*>(data);

96 icon = icon->scale_simple(APP_ICON_SIZE, APP_ICON_SIZE, Gdk::INTERP_HYPER);
97 }
98
99 if (not icon)
100 return;
101
102 that->register_application(pid, icon);

187Glib::RefPtr<Gdk::Pixbuf>
188PrettyTable::get_icon_from_default(const ProcInfo &info)
189{
190 Glib::RefPtr<Gdk::Pixbuf> pix;
191 string name;
192
193 if (this->get_default_icon_name(info.name, name)) {
194 IconCache::iterator it(this->defaults.find(name));
195
196 if (it == this->defaults.end()) {
197 pix = this->theme->load_icon(name, APP_ICON_SIZE);
198 if (pix)
199 this->defaults[name] = pix;
200 } else
201 pix = it->second;
202 }
203
204 return pix;
205}

288void
289PrettyTable::set_icon(ProcInfo &info)
290{
291 typedef Glib::RefPtr<Gdk::Pixbuf>
292 (PrettyTable::*Getter)(const ProcInfo &);
293
294 static std::vector<Getter> getters;
295
296 if (getters.empty())
297 {
298 getters.push_back(&PrettyTable::get_icon_from_gio);
299#ifdef HAVE_WNCK
300 getters.push_back(&PrettyTable::get_icon_from_wnck);
301#endif
302 getters.push_back(&PrettyTable::get_icon_from_theme);
303 getters.push_back(&PrettyTable::get_icon_from_default);
304 getters.push_back(&PrettyTable::get_icon_from_name);
305 if (has_kthreadd())
306 {
307 procman_debug("kthreadd is running with PID 2");
308 getters.push_back(&PrettyTable::get_icon_for_kernel);
309 }
310 getters.push_back(&PrettyTable::get_icon_dummy);
311 }
312
313 Glib::RefPtr<Gdk::Pixbuf> icon;
314
315 for (size_t i = 0; not icon and i < getters.size(); ++i) {
316 try {
317 icon = (this->*getters[i])(info);
318 }
319 catch (std::exception& e) {
320 g_warning("Failed to load icon for %s(%u) : %s", info.name, info.pid, e.what());
321 continue;
322 }
323 catch (Glib::Exception& e) {
324 g_warning("Failed to load icon for %s(%u) : %s", info.name, info.pid, e.what().c_str());
325 continue;
326 }
327 }
328
329 info.set_icon(icon);
330}
