120void
121pulse_device_switch_set_active_profile_by_name (PulseDeviceSwitch *swtch, const gchar *name)
122{
123 GList *item;
124
125 g_return_if_fail (PULSE_IS_DEVICE_SWITCH (swtch));
126 g_return_if_fail (name != NULL);
127
128 item = g_list_find_custom (swtch->priv->profiles, name, compare_profile_name);
129 if (G_UNLIKELY (item == NULL)) {
130 g_debug ("Invalid device switch profile name %s", name);
131 return;
132 }
133 pulse_device_switch_set_active_profile (swtch, PULSE_DEVICE_PROFILE (item->data));
134}

136static gboolean
137pulse_device_switch_set_active_option (MateMixerSwitch *mms, MateMixerSwitchOption *mmso)
138{
139 MateMixerDevice *device;
140 const gchar *device_name;
141 const gchar *profile_name;
142
143 g_return_val_if_fail (PULSE_IS_DEVICE_SWITCH (mms), FALSE);
144 g_return_val_if_fail (PULSE_IS_DEVICE_PROFILE (mmso), FALSE);
145
146 device = mate_mixer_device_switch_get_device (MATE_MIXER_DEVICE_SWITCH (mms));
147 if (G_UNLIKELY (device == NULL))
148 return FALSE;
149
150 device_name = mate_mixer_device_get_name (device);
151 profile_name = mate_mixer_switch_option_get_name (mmso);
152
153 return pulse_connection_set_card_profile (pulse_device_get_connection (PULSE_DEVICE (device)),
154 device_name,
155 profile_name);
156}

175{
176 PulseDeviceProfile *profile = PULSE_DEVICE_PROFILE (a);

179 return strcmp (pulse_device_profile_get_name (profile), name);
