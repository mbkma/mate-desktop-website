96static void
97parent_file_display_name_query_info_cb (GObject *source_object,
98 GAsyncResult *res,
99 gpointer user_data)
100{
101 EomMetadataSidebar *sidebar = EOM_METADATA_SIDEBAR (user_data);
102 GFile *parent_file = G_FILE (source_object);
103 GFileInfo *file_info;
104 gchar *baseuri;
105 gchar *display_name;
106 gchar *str;
107
108 file_info = g_file_query_info_finish (parent_file, res, NULL);
109 if (file_info == NULL) {
110 display_name = g_file_get_basename (parent_file);
111 } else {
112 display_name = g_strdup (
113 g_file_info_get_display_name (file_info));
114 g_object_unref (file_info);
115 }
116 baseuri = g_file_get_uri (parent_file);
117 str = g_markup_printf_escaped ("<a href=\"%s\">%s</a>",
118 baseuri,
119 display_name);
120 gtk_label_set_markup (GTK_LABEL (sidebar->priv->folder_label), str);
121
122 g_free (str);
123 g_free (baseuri);
124 g_free (display_name);
125
126 str = g_file_get_path (parent_file);
127 gtk_widget_set_tooltip_text (GTK_WIDGET (sidebar->priv->folder_label), str);
128 g_free (str);
129
130 g_object_unref (sidebar);
131}

133static void
134eom_metadata_sidebar_update_general_section (EomMetadataSidebar *sidebar)
135{
136 EomMetadataSidebarPrivate *priv = sidebar->priv;
137 EomImage *img = priv->image;
138 GFile *file, *parent_file;
139 GFileInfo *file_info;
140 gchar *str;
141 goffset bytes;
142 gint width, height;
143
144 if (G_UNLIKELY (img == NULL)) {
145 gtk_label_set_text (GTK_LABEL (priv->size_label), NULL);
146 gtk_label_set_text (GTK_LABEL (priv->type_label), NULL);
147 gtk_label_set_text (GTK_LABEL (priv->filesize_label), NULL);
148 gtk_label_set_text (GTK_LABEL (priv->folder_label), NULL);
149 return;
150 }
151
152 eom_image_get_size (img, &width, &height);
153 str = g_strdup_printf (ngettext("%i × %i pixel",
154 "%i × %i pixels", height),
155 width, height);
156 gtk_label_set_text (GTK_LABEL (priv->size_label), str);
157 g_free (str);
158
159 file = eom_image_get_file (img);
160 file_info = g_file_query_info (file,
161 G_FILE_ATTRIBUTE_STANDARD_CONTENT_TYPE","
162 G_FILE_ATTRIBUTE_STANDARD_FAST_CONTENT_TYPE,
163 0, NULL, NULL);
164 if (file_info == NULL) {
165 str = g_strdup (_("Unknown"));
166 } else {
167 const gchar *mime_str;
168
169 mime_str = eom_util_get_content_type_with_fallback (file_info);
170 str = g_content_type_get_description (mime_str);
171 g_object_unref (file_info);
172 }
173 gtk_label_set_text (GTK_LABEL (priv->type_label), str);
174 g_free (str);
175
176 bytes = eom_image_get_bytes (img);
177 str = g_format_size (bytes);
178 gtk_label_set_text (GTK_LABEL (priv->filesize_label), str);
179 g_free (str);
180
181 parent_file = g_file_get_parent (file);
182 if (parent_file == NULL) {
183 /* file is root directory itself */
184 parent_file = g_object_ref (file);
185 }
186 gtk_label_set_markup (GTK_LABEL (sidebar->priv->folder_label), NULL);
187 g_file_query_info_async (parent_file,
188 G_FILE_ATTRIBUTE_STANDARD_DISPLAY_NAME,
189 G_FILE_QUERY_INFO_NONE,
190 G_PRIORITY_DEFAULT,
191 NULL,
192 parent_file_display_name_query_info_cb,
193 g_object_ref (sidebar));
194
195 g_object_unref (parent_file);
196}

259static void
260eom_metadata_sidebar_set_image (EomMetadataSidebar *sidebar, EomImage *image)
261{
262 EomMetadataSidebarPrivate *priv = sidebar->priv;
263
264 if (image == priv->image)
265 return;
266
267#if GLIB_CHECK_VERSION(2,62,0)
268 g_clear_signal_handler (&priv->thumb_changed_id,
269 priv->image);
270#else
271 if (priv->thumb_changed_id != 0) {
272 g_signal_handler_disconnect (priv->image,
273 priv->thumb_changed_id);
274 priv->thumb_changed_id = 0;
275 }
276#endif
277
278 if (priv->image)
279 g_object_unref (priv->image);
280
281 priv->image = image;
282
283 if (priv->image) {
284 g_object_ref (priv->image);
285 priv->thumb_changed_id =
286 g_signal_connect (priv->image, "thumbnail-changed",
287 G_CALLBACK (_thumbnail_changed_cb),
288 sidebar);
289 eom_metadata_sidebar_update (sidebar);
290 }
291
292 g_object_notify (G_OBJECT (sidebar), "image");
293}

304

306 image);
307
308 if (image)
309 g_object_unref (image);

355static void
356eom_metadata_sidebar_set_parent_window (EomMetadataSidebar *sidebar,
357 EomWindow *window)
358{
359 EomMetadataSidebarPrivate *priv;
360 GtkWidget *view;
361
362 g_return_if_fail (EOM_IS_METADATA_SIDEBAR (sidebar));

365

367 eom_metadata_sidebar_update (sidebar);

371 sidebar);
372
373 g_object_notify (G_OBJECT (sidebar), "parent-window");

407static void
408eom_metadata_sidebar_get_property (GObject *object, guint property_id,
409 GValue *value, GParamSpec *pspec)
410{
411 EomMetadataSidebar *sidebar;
412
413 g_return_if_fail (EOM_IS_METADATA_SIDEBAR (object));
414
415 sidebar = EOM_METADATA_SIDEBAR (object);
416
417 switch (property_id) {
418 case PROP_IMAGE:
419 {
420 g_value_set_object (value, sidebar->priv->image);
421 break;
422 }
423 case PROP_PARENT_WINDOW:
424 g_value_set_object (value, sidebar->priv->parent_window);
425 break;
426 default:
427 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, property_id, pspec);
428 }
429}
