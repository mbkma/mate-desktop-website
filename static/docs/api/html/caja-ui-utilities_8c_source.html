94const char *
95caja_ui_string_get (const char *filename)
96{
97 static GHashTable *ui_cache = NULL;
98 char *ui;
99
100 if (ui_cache == NULL)
101 {
102 ui_cache = g_hash_table_new_full (g_str_hash, g_str_equal, g_free, g_free);
103 eel_debug_call_at_shutdown_with_data ((GFreeFunc)g_hash_table_destroy, ui_cache);
104 }
105
106 ui = g_hash_table_lookup (ui_cache, filename);
107 if (ui == NULL)
108 {
109 char *path;
110
111 path = caja_ui_file (filename);
112 if (path == NULL || !g_file_get_contents (path, &ui, NULL, NULL))
113 {
114 g_warning ("Unable to load ui file %s\n", filename);
115 }
116 g_free (path);
117 g_hash_table_insert (ui_cache,
118 g_strdup (filename),
119 ui);
120 }
121
122 return ui;
123}

148static cairo_surface_t *
149get_action_icon (const char *icon_name,
150 int size,
151 GtkWidget *parent_widget)
152{
153 CajaIconInfo *info;
154 cairo_surface_t *surface;
155 int scale;
156
157 scale = gtk_widget_get_scale_factor (parent_widget);
158
159 if (g_path_is_absolute (icon_name))
160 {
161 info = caja_icon_info_lookup_from_path (icon_name, size, scale);
162 }
163 else
164 {
165 info = caja_icon_info_lookup_from_name (icon_name, size, scale);
166 }
167 surface = caja_icon_info_get_surface_nodefault_at_size (info, size);
168 g_object_unref (info);
169
170 return surface;
171}

173GtkAction *
174caja_action_from_menu_item (CajaMenuItem *item,
175 GtkWidget *parent_widget)
176{
177 char *name, *label, *tip, *icon_name;
178 gboolean sensitive, priority;
179 GtkAction *action;
180
181 g_object_get (G_OBJECT (item),
182 "name", &name, "label", &label,
183 "tip", &tip, "icon", &icon_name,
184 "sensitive", &sensitive,
185 "priority", &priority,
186 NULL);
187
188 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
189 action = gtk_action_new (name,
190 label,
191 tip,
192 icon_name);
193 G_GNUC_END_IGNORE_DEPRECATIONS;
194
195 if (icon_name != NULL)
196 {
197 cairo_surface_t *surface;
198
199 surface = get_action_icon (icon_name,
200 caja_get_icon_size_for_stock_size (GTK_ICON_SIZE_MENU),
201 parent_widget);
202 if (surface != NULL)
203 {
204 g_object_set_data_full (G_OBJECT (action), "menu-icon",
205 surface,
206 (GDestroyNotify)cairo_surface_destroy);
207 }
208 }
209
210 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
211 gtk_action_set_sensitive (action, sensitive);
212 G_GNUC_END_IGNORE_DEPRECATIONS;
213 g_object_set (action, "is-important", priority, NULL);
214
215 g_signal_connect_data (action, "activate",
216 G_CALLBACK (extension_action_callback),
217 g_object_ref (item),
218 (GClosureNotify)g_object_unref, 0);
219
220 g_free (name);
221 g_free (label);
222 g_free (tip);
223 g_free (icon_name);
224
225 return action;
226}
