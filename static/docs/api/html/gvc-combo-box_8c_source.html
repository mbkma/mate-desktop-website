97static void
98on_switch_active_option_notify (MateMixerSwitch *swtch,
99 GParamSpec *pspec,
100 GvcComboBox *combobox)
101{
102 GtkTreeIter iter;
103 MateMixerSwitchOption *active;
104 gboolean cont;
105 const gchar *name;
106
107 active = mate_mixer_switch_get_active_option (swtch);
108 if (G_UNLIKELY (active == NULL)) {
109 g_warn_if_reached ();
110 return;
111 }
112
113 /* Select the newly activated switch option in the combo box */
114 name = mate_mixer_switch_option_get_name (active);
115 cont = gtk_tree_model_get_iter_first (combobox->priv->model, &iter);
116 while (cont == TRUE) {
117 gchar *current;
118
119 gtk_tree_model_get (combobox->priv->model, &iter,
120 COL_NAME, &current,
121 -1);
122 if (g_strcmp0 (name, current) == 0) {
123 gtk_combo_box_set_active_iter (GTK_COMBO_BOX (combobox->priv->combobox), &iter);
124 g_free (current);
125 return;
126 }
127 g_free (current);
128
129 cont = gtk_tree_model_iter_next (combobox->priv->model, &iter);
130 }
131 g_warning ("Could not find switch option '%s' in combo box", name);
132}

134static void
135gvc_combo_box_set_switch (GvcComboBox *combobox, MateMixerSwitch *swtch)
136{
137 MateMixerSwitchOption *active;
138 const GList *options;
139
140 g_return_if_fail (GVC_IS_COMBO_BOX (combobox));
141 g_return_if_fail (MATE_MIXER_IS_SWITCH (swtch));
142
143 combobox->priv->swtch = g_object_ref (swtch);
144
145 active = mate_mixer_switch_get_active_option (swtch);
146 options = mate_mixer_switch_list_options (swtch);
147 while (options != NULL) {
148 GtkTreeIter iter;
149 MateMixerSwitchOption *option = MATE_MIXER_SWITCH_OPTION (options->data);
150
151 gtk_list_store_insert_with_values (GTK_LIST_STORE (combobox->priv->model),
152 &iter,
153 G_MAXINT,
154 COL_NAME,
155 mate_mixer_switch_option_get_name (option),
156 COL_HUMAN_NAME,
157 mate_mixer_switch_option_get_label (option),
158 -1);
159
160 /* Select the currently active option of the switch */
161 if (option == active) {
162 gtk_combo_box_set_active_iter (GTK_COMBO_BOX (combobox->priv->combobox),
163 &iter);
164 }
165 options = options->next;
166 }
167
168 g_signal_connect (G_OBJECT (swtch),
169 "notify::active-option",
170 G_CALLBACK (on_switch_active_option_notify),
171 combobox);
172}

174static void
175gvc_combo_box_set_property (GObject *object,
176 guint prop_id,
177 const GValue *value,
178 GParamSpec *pspec)
179{
180 GvcComboBox *self = GVC_COMBO_BOX (object);
181
182 switch (prop_id) {
183 case PROP_SWITCH:
184 gvc_combo_box_set_switch (self, g_value_get_object (value));
185 break;
186 case PROP_LABEL:
187 gtk_label_set_text_with_mnemonic (GTK_LABEL (self->priv->label), g_value_get_string (value));
188 break;
189 case PROP_BUTTON_LABEL:
190 gtk_button_set_label (GTK_BUTTON (self->priv->button), g_value_get_string (value));
191 break;
192 case PROP_SHOW_BUTTON:
193 gtk_widget_set_visible (self->priv->button, g_value_get_boolean (value));
194 break;
195 default:
196 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
197 break;
198 }
199}

297static void
298on_combo_box_changed (GtkComboBox *widget, GvcComboBox *combobox)
299{
300 GtkTreeIter iter;
301 gchar *name;
302 MateMixerSwitchOption *option;
303
304 if (G_UNLIKELY (gtk_combo_box_get_active_iter (GTK_COMBO_BOX (widget), &iter) == FALSE))
305 return;
306
307 gtk_tree_model_get (combobox->priv->model, &iter,
308 COL_NAME, &name,
309 -1);
310
311 option = mate_mixer_switch_get_option (combobox->priv->swtch, name);
312 if (G_UNLIKELY (option == NULL)) {
313 g_warn_if_reached ();
314 g_free (name);
315 return;
316 }
317
318 /* Inform that we are about to change the active option of the switch */
319 g_signal_emit (combobox, signals[CHANGING], 0, option);
320
321 mate_mixer_switch_set_active_option (combobox->priv->swtch, option);
322 g_free (name);
323}

335{
336 GtkWidget *frame;

341
342 combobox->priv = gvc_combo_box_get_instance_private (combobox);
343
344 combobox->priv->model = GTK_TREE_MODEL (gtk_list_store_new (NUM_COLS,
345 G_TYPE_STRING,

349

353 gtk_label_set_yalign (GTK_LABEL (combobox->priv->label), 0.5);
354 gtk_label_set_mnemonic_widget (GTK_LABEL (combobox->priv->label),
355 combobox->priv->combobox);

357 renderer = gtk_cell_renderer_text_new ();
358 gtk_cell_layout_pack_start (GTK_CELL_LAYOUT (combobox->priv->combobox),

363 "text",

366 combobox->priv->drop_box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
367 combobox->priv->start_box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
368 combobox->priv->end_box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);

375
376 gtk_combo_box_set_popup_fixed_width (GTK_COMBO_BOX (combobox->priv->combobox), FALSE);
377
378 gtk_box_pack_start (GTK_BOX (combobox),

416static void
417gvc_combo_box_dispose (GObject *object)
418{
419 GvcComboBox *combobox;
420
421 combobox = GVC_COMBO_BOX (object);
422
423 if (G_LIKELY (combobox->priv->swtch != NULL)) {
424 g_signal_handlers_disconnect_by_func (G_OBJECT (combobox->priv->swtch),
425 G_CALLBACK (on_switch_active_option_notify),
426 combobox);
427 g_clear_object (&combobox->priv->swtch);
428 }
429
430 g_clear_object (&combobox->priv->model);
431
432 G_OBJECT_CLASS (gvc_combo_box_parent_class)->dispose (object);
433}
