30gboolean theme_is_writable (const gpointer theme)
31{
32 MateThemeCommonInfo *info = theme;
33 GFile *file;
34 GFileInfo *file_info;
35 gboolean writable;
36
37 if (info == NULL || info->path == NULL)
38 return FALSE;
39
40 file = g_file_new_for_path (info->path);
41 file_info = g_file_query_info (file,
42 G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE,
43 G_FILE_QUERY_INFO_NONE,
44 NULL, NULL);
45 g_object_unref (file);
46
47 if (file_info == NULL)
48 return FALSE;
49
50 writable = g_file_info_get_attribute_boolean (file_info,
51 G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE);
52 g_object_unref (file_info);
53
54 return writable;
55}

57gboolean theme_delete (const gchar *name, ThemeType type)
58{
59 gboolean rc;
60 GtkDialog *dialog;
61 gchar *theme_dir;
62 gint response;
63 MateThemeCommonInfo *theme;
64 GFile *dir;
65 gboolean del_empty_parent;
66
67 dialog = (GtkDialog *) gtk_message_dialog_new (NULL,
68 GTK_DIALOG_MODAL,
69 GTK_MESSAGE_QUESTION,
70 GTK_BUTTONS_CANCEL,
71 _("Would you like to delete this theme?"));
72 gtk_dialog_add_button (dialog, "gtk-delete", GTK_RESPONSE_ACCEPT);
73 response = gtk_dialog_run (dialog);
74 gtk_widget_destroy (GTK_WIDGET (dialog));
75 if (response != GTK_RESPONSE_ACCEPT)
76 return FALSE;
77
78 /* Most theme types are put into separate subdirectories. For those
79 we want to delete those directories as well. */
80 del_empty_parent = TRUE;
81
82 switch (type) {
83 case THEME_TYPE_GTK:
84 theme = (MateThemeCommonInfo *) mate_theme_info_find (name);
85 theme_dir = g_build_filename (theme->path, "gtk-2.0", NULL);
86 break;
87
88 case THEME_TYPE_ICON:
89 theme = (MateThemeCommonInfo *) mate_theme_icon_info_find (name);
90 theme_dir = g_path_get_dirname (theme->path);
91 del_empty_parent = FALSE;
92 break;
93
94 case THEME_TYPE_WINDOW:
95 theme = (MateThemeCommonInfo *) mate_theme_info_find (name);
96 theme_dir = g_build_filename (theme->path, "metacity-1", NULL);
97 break;
98
99 case THEME_TYPE_META:
100 theme = (MateThemeCommonInfo *) mate_theme_meta_info_find (name);
101 theme_dir = g_path_get_dirname (theme->path);
102 del_empty_parent = FALSE;
103 break;
104
105 case THEME_TYPE_CURSOR:
106 theme = (MateThemeCommonInfo *) mate_theme_cursor_info_find (name);
107 theme_dir = g_build_filename (theme->path, "cursors", NULL);
108 break;
109
110 default:
111 return FALSE;
112 }
113
114 dir = g_file_new_for_path (theme_dir);
115 g_free (theme_dir);
116
117 if (!capplet_file_delete_recursive (dir, NULL)) {
118 GtkWidget *info_dialog = gtk_message_dialog_new (NULL,
119 GTK_DIALOG_MODAL,
120 GTK_MESSAGE_ERROR,
121 GTK_BUTTONS_OK,
122 _("Theme cannot be deleted"));
123 gtk_dialog_run (GTK_DIALOG (info_dialog));
124 gtk_widget_destroy (info_dialog);
125 rc = FALSE;
126 } else {
127 if (del_empty_parent) {
128 /* also delete empty parent directories */
129 GFile *parent = g_file_get_parent (dir);
130 g_file_delete (parent, NULL, NULL);
131 g_object_unref (parent);
132 }
133 rc = TRUE;
134 }
135
136 g_object_unref (dir);
137 return rc;
138}

140gboolean theme_model_iter_last (GtkTreeModel *model, GtkTreeIter *iter)
141{
142 GtkTreeIter walk, prev;
143 gboolean valid;
144
145 valid = gtk_tree_model_get_iter_first (model, &walk);
146
147 if (valid) {
148 do {
149 prev = walk;
150 valid = gtk_tree_model_iter_next (model, &walk);
151 } while (valid);
152
153 *iter = prev;
154 return TRUE;
155 }
156 return FALSE;

159gboolean theme_find_in_model (GtkTreeModel *model, const gchar *name, GtkTreeIter *iter)
160{
161 GtkTreeIter walk;
162 gboolean valid;
163 gchar *test;
164
165 if (!name)
166 return FALSE;
167
168 for (valid = gtk_tree_model_get_iter_first (model, &walk); valid;
169 valid = gtk_tree_model_iter_next (model, &walk))
170 {
171 gtk_tree_model_get (model, &walk, COL_NAME, &test, -1);
172
173 if (test) {
174 gint cmp = strcmp (test, name);
175 g_free (test);
176
177 if (!cmp) {
178 if (iter)
179 *iter = walk;
180 return TRUE;
181 }
182 }
183 }
184
185 return FALSE;
186}

188gboolean packagekit_available (void)
189{
190 GDBusConnection *connection;
191 GDBusProxy *proxy;
192 gboolean available = FALSE;
193 GError *error = NULL;
194 GVariant *variant;
195
196 connection = g_bus_get_sync (G_BUS_TYPE_SESSION, NULL, NULL);
197 if (connection == NULL) {
198 return FALSE;
199 }
200
201 proxy = g_dbus_proxy_new_sync (connection,
202 G_DBUS_PROXY_FLAGS_NONE,
203 NULL,
204 "org.freedesktop.DBus",
205 "/org/freedesktop/DBus",
206 "org.freedesktop.DBus",
207 NULL,
208 NULL);
209 if (proxy == NULL) {
210 g_object_unref (connection);
211 return FALSE;
212 }
213
214 variant = g_dbus_proxy_call_sync (proxy, "NameHasOwner",
215 g_variant_new ("(s)", "org.freedesktop.PackageKit"),
216 G_DBUS_CALL_FLAGS_NONE,
217 -1,
218 NULL,
219 &error);
220 if (variant == NULL) {
221 g_warning ("Could not ask org.freedesktop.DBus if PackageKit is available: %s",
222 error->message);
223 g_error_free (error);
224 g_object_unref (proxy);
225 g_object_unref (connection);
226
227 return FALSE;
228 } else {
229 g_variant_get (variant, "(b)", &available);
230 g_variant_unref (variant);
231 }
232
233 g_object_unref (proxy);
234 g_object_unref (connection);
235
236 return available;
237}

239void theme_install_file(GtkWindow* parent, const gchar* path)
240{
241 GDBusConnection *connection;
242 GDBusProxy *proxy;
243 GError* error = NULL;
244 GVariant *variant;
245
246 connection = g_bus_get_sync (G_BUS_TYPE_SESSION, NULL, &error);
247
248 if (connection == NULL)
249 {
250 g_warning("Could not get session bus: %s", error->message);
251 g_error_free (error);
252 return;
253 }
254
255 proxy = g_dbus_proxy_new_sync (connection,
256 G_DBUS_PROXY_FLAGS_NONE,
257 NULL,
258 "org.freedesktop.PackageKit",
259 "/org/freedesktop/PackageKit",
260 "org.freedesktop.PackageKit",
261 NULL,
262 &error);
263 if (proxy == NULL) {
264 g_warning ("Failed to connect to PackageKit: %s\n", error->message);
265 g_error_free (error);
266 g_object_unref (connection);
267 return;
268 }
269
270 variant = g_dbus_proxy_call_sync (proxy, "InstallProvideFile",
271 g_variant_new ("(s)", path),
272 G_DBUS_CALL_FLAGS_NONE,
273 -1,
274 NULL,
275 &error);
276
277 if (variant == NULL) {
278 GtkWidget* dialog = gtk_message_dialog_new(NULL,
279 GTK_DIALOG_MODAL | GTK_DIALOG_DESTROY_WITH_PARENT,
280 GTK_MESSAGE_ERROR,
281 GTK_BUTTONS_OK,
282 _("Could not install theme engine"));
283 gtk_message_dialog_format_secondary_text(GTK_MESSAGE_DIALOG (dialog), "%s", error->message);
284
285 gtk_dialog_run(GTK_DIALOG(dialog));
286 gtk_widget_destroy(dialog);
287 g_error_free(error);
288 } else {
289 g_variant_unref (variant);
290 }
291
292 g_object_unref(proxy);
293 g_object_unref (connection);
294}
