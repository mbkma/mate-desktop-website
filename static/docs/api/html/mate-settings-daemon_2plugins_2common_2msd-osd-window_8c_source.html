67static gboolean
68fade_timeout (MsdOsdWindow *window)
69{
70 if (window->priv->fade_out_alpha <= 0.0) {
71 gtk_widget_hide (GTK_WIDGET (window));
72
73 /* Reset it for the next time */
74 window->priv->fade_out_alpha = 1.0;
75 window->priv->fade_timeout_id = 0;
76
77 return FALSE;
78 } else {
79 GdkRectangle rect;
80 GtkWidget *win = GTK_WIDGET (window);
81 GtkAllocation allocation;
82
83 window->priv->fade_out_alpha -= 0.10;
84
85 rect.x = 0;
86 rect.y = 0;
87 gtk_widget_get_allocation (win, &allocation);
88 rect.width = allocation.width;
89 rect.height = allocation.height;
90
91 gtk_widget_realize (win);
92 gdk_window_invalidate_rect (gtk_widget_get_window (win), &rect, FALSE);
93 }
94
95 return TRUE;
96}

98static gboolean
99hide_timeout (MsdOsdWindow *window)
100{
101 if (window->priv->is_composited) {
102 window->priv->hide_timeout_id = 0;
103 window->priv->fade_timeout_id = g_timeout_add (FADE_TIMEOUT,
104 (GSourceFunc) fade_timeout,
105 window);
106 } else {
107 gtk_widget_hide (GTK_WIDGET (window));
108 }
109
110 return FALSE;
111}

113static void
114remove_hide_timeout (MsdOsdWindow *window)
115{
116 if (window->priv->hide_timeout_id != 0) {

123 window->priv->fade_timeout_id = 0;

128static void
129add_hide_timeout (MsdOsdWindow *window)
130{
131 int timeout;
132
133 if (window->priv->is_composited) {
134 timeout = DIALOG_FADE_TIMEOUT;
135 } else {
136 timeout = DIALOG_TIMEOUT;
137 }
138 window->priv->hide_timeout_id = g_timeout_add (timeout,
139 (GSourceFunc) hide_timeout,

143/* This is our draw-event handler when the window is in a compositing manager.
144 * We draw everything by hand, using Cairo, so that we can have a nice
145 * transparent/rounded look.
146 */
147static void
148draw_when_composited (GtkWidget *widget, cairo_t *orig_cr)
149{
150 MsdOsdWindow *window;
151 cairo_t *cr;
152 cairo_surface_t *surface;
153 int width;
154 int height;
155 GtkStyleContext *context;
156
157 window = MSD_OSD_WINDOW (widget);
158
159 context = gtk_widget_get_style_context (widget);
160 cairo_set_operator (orig_cr, CAIRO_OPERATOR_SOURCE);
161 gtk_window_get_size (GTK_WINDOW (widget), &width, &height);
162
163 surface = cairo_surface_create_similar (cairo_get_target (orig_cr),
164 CAIRO_CONTENT_COLOR_ALPHA,
165 width,
166 height);
167
168 if (cairo_surface_status (surface) != CAIRO_STATUS_SUCCESS) {
169 goto done;
170 }
171
172 cr = cairo_create (surface);
173 if (cairo_status (cr) != CAIRO_STATUS_SUCCESS) {
174 goto done;
175 }
176
177 gtk_render_background (context, cr, 0, 0, width, height);
178 gtk_render_frame (context, cr, 0, 0, width, height);
179
180 g_signal_emit (window, signals[DRAW_WHEN_COMPOSITED], 0, cr);
181
182 cairo_destroy (cr);
183
184 /* Make sure we have a transparent background */
185 cairo_rectangle (orig_cr, 0, 0, width, height);
186 cairo_set_source_rgba (orig_cr, 0.0, 0.0, 0.0, 0.0);
187 cairo_fill (orig_cr);
188
189 cairo_set_source_surface (orig_cr, surface, 0, 0);
190 cairo_paint_with_alpha (orig_cr, window->priv->fade_out_alpha);
191
192done:
193 if (surface != NULL) {
194 cairo_surface_destroy (surface);
195 }
196}

198/* This is our draw-event handler when the window is *not* in a compositing manager.
199 * We just draw a rectangular frame by hand. We do this with hardcoded drawing code,
200 * instead of GtkFrame, to avoid changing the window's internal widget hierarchy: in
201 * either case (composited or non-composited), callers can assume that this works
202 * identically to a GtkWindow without any intermediate widgetry.
203 */
204static void
205draw_when_not_composited (GtkWidget *widget, cairo_t *cr)
206{
207 GtkStyleContext *context;
208 int width;
209 int height;
210
211 gtk_window_get_size (GTK_WINDOW (widget), &width, &height);
212 context = gtk_widget_get_style_context (widget);
213
214 gtk_style_context_set_state (context, GTK_STATE_FLAG_ACTIVE);
215 gtk_style_context_add_class(context,"msd-osd-window-solid");
216 gtk_render_frame (context,
217 cr,
218 0,
219 0,
220 width,
221 height);
222}

224static gboolean
225msd_osd_window_draw (GtkWidget *widget,
226 cairo_t *cr)
227{
228 MsdOsdWindow *window;
229 GtkWidget *child;
230
231 window = MSD_OSD_WINDOW (widget);
232
233 if (window->priv->is_composited)
234 draw_when_composited (widget, cr);
235 else
236 draw_when_not_composited (widget, cr);
237
238 child = gtk_bin_get_child (GTK_BIN (window));
239 if (child)
240 gtk_container_propagate_draw (GTK_CONTAINER (window), child, cr);
241
242 return FALSE;
243}

272static void
273msd_osd_window_real_realize (GtkWidget *widget)
274{
275 GdkScreen *screen;
276 GdkVisual *visual;
277 cairo_region_t *region;
278
279 screen = gtk_widget_get_screen (widget);
280 visual = gdk_screen_get_rgba_visual (screen);
281
282 if (visual == NULL) {
283 visual = gdk_screen_get_system_visual (screen);
284 }
285
286 gtk_widget_set_visual (widget, visual);
287
288 if (GTK_WIDGET_CLASS (msd_osd_window_parent_class)->realize) {
289 GTK_WIDGET_CLASS (msd_osd_window_parent_class)->realize (widget);
290 }
291
292 /* make the whole window ignore events */
293 region = cairo_region_create ();
294 gtk_widget_input_shape_combine_region (widget, region);
295 cairo_region_destroy (region);
296}

298static void
299msd_osd_window_style_updated (GtkWidget *widget)
300{
301 GtkStyleContext *context;
302 GtkBorder padding;
303
304 GTK_WIDGET_CLASS (msd_osd_window_parent_class)->style_updated (widget);
305
306 /* We set our border width to 12 (per the MATE standard), plus the
307 * padding of the frame that we draw in our expose/draw handler. This will
308 * make our child be 12 pixels away from the frame.
309 */
310
311 context = gtk_widget_get_style_context (widget);
312 gtk_style_context_get_padding (context, GTK_STATE_FLAG_NORMAL, &padding);
313 gtk_container_set_border_width (GTK_CONTAINER (widget), 12 + MAX (padding.left, padding.top));
314}

435static void
436msd_osd_window_init (MsdOsdWindow *window)
437{
438 GdkScreen *screen;
439
440 window->priv = msd_osd_window_get_instance_private (window);
441
442 screen = gtk_widget_get_screen (GTK_WIDGET (window));
443
444 window->priv->is_composited = (gdk_screen_is_composited (screen) != FALSE);
445 window->priv->scale_factor = gtk_widget_get_scale_factor (GTK_WIDGET (window));
446
447 if (window->priv->is_composited) {
448 gdouble scalew, scaleh, scale;
449 gint size;
450
451 gtk_window_set_decorated (GTK_WINDOW (window), FALSE);
452 gtk_widget_set_app_paintable (GTK_WIDGET (window), TRUE);
453
454 GtkStyleContext *style = gtk_widget_get_style_context (GTK_WIDGET (window));
455 gtk_style_context_add_class (style, "window-frame");
456
457 /* assume 110x110 on a 640x480 display and scale from there */
458 scalew = WidthOfScreen (gdk_x11_screen_get_xscreen (screen)) / (640.0 * window->priv->scale_factor);
459 scaleh = HeightOfScreen (gdk_x11_screen_get_xscreen (screen)) / (480.0 * window->priv->scale_factor);
460 scale = MIN (scalew, scaleh);
461 size = 110 * MAX (1, scale);
462
463 gtk_window_set_default_size (GTK_WINDOW (window), size, size);
464
465 window->priv->fade_out_alpha = 1.0;
466 } else {
467 gtk_container_set_border_width (GTK_CONTAINER (window), 12);
468 }
469}
