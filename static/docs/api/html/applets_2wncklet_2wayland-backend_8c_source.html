87static void
88wl_registry_handle_global (void *_data,
89 struct wl_registry *registry,
90 uint32_t id,
91 const char *interface,
92 uint32_t version)
93{
94 /* pull out needed globals */
95 if (strcmp (interface, zwlr_foreign_toplevel_manager_v1_interface.name) == 0)

97 g_warn_if_fail (zwlr_foreign_toplevel_manager_v1_interface.version == 2);
98 foreign_toplevel_manager_global_id = id;
99 foreign_toplevel_manager_global_version =

120static void
121wayland_tasklist_init_if_needed (void)
122{
123 if (has_initialized)
124 return;
125
126 GdkDisplay *gdk_display = gdk_display_get_default ();
127 g_return_if_fail (gdk_display);
128 g_return_if_fail (GDK_IS_WAYLAND_DISPLAY (gdk_display));
129
130 struct wl_display *wl_display = gdk_wayland_display_get_wl_display (gdk_display);
131 wl_registry_global = wl_display_get_registry (wl_display);
132 wl_registry_add_listener (wl_registry_global, &wl_registry_listener, NULL);
133 wl_display_roundtrip (wl_display);
134
135 if (!foreign_toplevel_manager_global_id)
136 g_warning ("%s not supported by Wayland compositor",
137 zwlr_foreign_toplevel_manager_v1_interface.name);
138
139 has_initialized = TRUE;
140}

174static void
175tasklist_manager_disconnected_from_widget (TasklistManager *tasklist)
176{
177 if (tasklist->list)
178 {
179 GList *children = gtk_container_get_children (GTK_CONTAINER (tasklist->list));
180 for (GList *iter = children; iter != NULL; iter = g_list_next (iter))
181 gtk_widget_destroy (GTK_WIDGET (iter->data));
182 g_list_free(children);
183 tasklist->list = NULL;
184 }
185
186 if (tasklist->outer_box)
187 tasklist->outer_box = NULL;
188
189 if (tasklist->manager)
190 zwlr_foreign_toplevel_manager_v1_stop (tasklist->manager);
191
192 if (tasklist->context_menu)
193 {
194 gtk_widget_destroy (tasklist->context_menu->menu);
195 g_free (tasklist->context_menu);
196 tasklist->context_menu = NULL;
197 }

235static ContextMenu *
236context_menu_new ()
237{
238 ContextMenu *menu = g_new0 (ContextMenu, 1);
239 menu->menu = gtk_menu_new ();
240 menu->maximize = gtk_menu_item_new ();
241 menu->minimize = gtk_menu_item_new ();
242 menu->on_top = gtk_check_menu_item_new_with_label ("Always On Top");
243 menu->close = gtk_menu_item_new_with_label ("Close");
244
245 gtk_menu_shell_append (GTK_MENU_SHELL (menu->menu), menu->maximize);
246 gtk_menu_shell_append (GTK_MENU_SHELL (menu->menu), menu->minimize);
247 gtk_menu_shell_append (GTK_MENU_SHELL (menu->menu), gtk_separator_menu_item_new ());
248 gtk_menu_shell_append (GTK_MENU_SHELL (menu->menu), menu->on_top);
249 gtk_menu_shell_append (GTK_MENU_SHELL (menu->menu), gtk_separator_menu_item_new ());
250 gtk_menu_shell_append (GTK_MENU_SHELL (menu->menu), menu->close);
251
252 gtk_widget_show_all (menu->menu);
253
254 g_signal_connect (menu->maximize, "activate", G_CALLBACK (menu_on_maximize), NULL);
255 g_signal_connect (menu->minimize, "activate", G_CALLBACK (menu_on_minimize), NULL);
256 g_signal_connect (menu->close, "activate", G_CALLBACK (menu_on_close), NULL);
257 gtk_widget_set_sensitive (menu->on_top, FALSE);
258 return menu;
259}

261static TasklistManager *
262tasklist_manager_new (void)
263{
264 if (!foreign_toplevel_manager_global_id)
265 return NULL;
266
267 TasklistManager *tasklist = g_new0 (TasklistManager, 1);
268 tasklist->list = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);
269 gtk_box_set_homogeneous (GTK_BOX (tasklist->list), TRUE);
270 tasklist->outer_box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);
271 gtk_box_pack_start (GTK_BOX (tasklist->outer_box), tasklist->list, FALSE, FALSE, 0);
272 gtk_widget_show (tasklist->list);
273 tasklist->manager = wl_registry_bind (wl_registry_global,
274 foreign_toplevel_manager_global_id,
275 &zwlr_foreign_toplevel_manager_v1_interface,
276 foreign_toplevel_manager_global_version);
277 zwlr_foreign_toplevel_manager_v1_add_listener (tasklist->manager,
278 &foreign_toplevel_manager_listener,
279 tasklist);
280 g_object_set_data_full (G_OBJECT (tasklist->outer_box),
281 tasklist_manager_key,
282 tasklist,
283 (GDestroyNotify)tasklist_manager_disconnected_from_widget);
284 tasklist->context_menu = context_menu_new ();
285 return tasklist;
286}

301static void
302foreign_toplevel_handle_app_id (void *data,
303 struct zwlr_foreign_toplevel_handle_v1 *toplevel,
304 const char *app_id)
305{
306 ToplevelTask *task = data;
307
308 gchar *app_id_lower = g_utf8_strdown (app_id, -1);
309 gchar *desktop_app_id = g_strdup_printf ("%s.desktop", app_id_lower);
310 GDesktopAppInfo *app_info = g_desktop_app_info_new (desktop_app_id);
311
312 if (app_info) {
313 GIcon *icon = g_app_info_get_icon (G_APP_INFO (app_info));
314 if (icon) {
315 gtk_image_set_from_gicon (GTK_IMAGE (task->icon), icon, GTK_ICON_SIZE_MENU);
316 goto cleanup;
317 }
318 }
319 gtk_image_set_from_icon_name (GTK_IMAGE (task->icon), app_id_lower, GTK_ICON_SIZE_MENU);
320
321cleanup:
322 if (app_info) {
323 g_object_unref (G_OBJECT (app_info));
324 }
325 g_free (app_id_lower);
326 g_free (desktop_app_id);
327}

345static void
346foreign_toplevel_handle_state (void *data,
347 struct zwlr_foreign_toplevel_handle_v1 *toplevel,
348 struct wl_array *state)
349{
350 ToplevelTask *task = data;
351
352 task->active = FALSE;
353 task->maximized = FALSE;
354 task->minimized = FALSE;
355 task->fullscreen = FALSE;
356
357 enum zwlr_foreign_toplevel_handle_v1_state *i;
358 wl_array_for_each (i, state)
359 {
360 switch (*i)
361 {
362 case ZWLR_FOREIGN_TOPLEVEL_HANDLE_V1_STATE_ACTIVATED:
363 task->active = TRUE;
364 break;
365 case ZWLR_FOREIGN_TOPLEVEL_HANDLE_V1_STATE_MAXIMIZED:
366 task->maximized = TRUE;
367 break;
368 case ZWLR_FOREIGN_TOPLEVEL_HANDLE_V1_STATE_MINIMIZED:
369 task->minimized = TRUE;
370 break;
371 case ZWLR_FOREIGN_TOPLEVEL_HANDLE_V1_STATE_FULLSCREEN:
372 task->fullscreen = TRUE;
373 break;
374 default:
375 break;
376 }
377 }
378
379 gtk_button_set_relief (GTK_BUTTON (task->button), task->active ? GTK_RELIEF_NORMAL : GTK_RELIEF_NONE);
380}

389static void
390adjust_buttons (GtkContainer *outer_box, int button_space, int real_buttons, ToplevelTask *task)
391{
392 GtkWidget *widget, *button, *box;
393
394 /*catch the case of an added button that can be missed
395 *Note that button space can come up zero on a first button
396 */
397 if (real_buttons < 2)
398 {
399 if(task)
400 {
401 gtk_widget_set_size_request (task->button, full_button_width, -1);
402 }
403 }
404
405 if ((task) && (button_space > 0) && (button_space < icon_size * 3))
406 {
407 gtk_widget_hide (task->icon);
408 }
409 else if (task)
410 {
411 gtk_widget_show (task->icon);
412 }
413
414 if ((task) && (button_space > 0) && (button_space < icon_size))
415 {
416 gtk_widget_hide (task->label);
417 }
418 else if (task)
419 {
420 gtk_widget_show (task->label);
421 }
422
423 GList* children = gtk_container_get_children (GTK_CONTAINER (outer_box));
424
425 while (children != NULL)
426 {
427 button = GTK_WIDGET (children->data);
428 box = gtk_bin_get_child (GTK_BIN (button));
429
430 if ((real_buttons < 2) || (real_buttons * full_button_width < tasklist_width * 0.75))
431 {
432 gtk_widget_set_size_request (button, full_button_width, -1);
433 gtk_widget_show_all (button);
434 return;
435 }
436 else
437 {
438 gtk_widget_set_size_request (button, MIN(button_space, full_button_width), -1);
439 }
440
441 /* if the number of buttons forces width to less than 3x the icon size, hide the icons
442 * if the number of buttons forces width to less than the icon size, hide the labels too.
443 * This is roughy the same behavior as on x11
444 * To find the icon and label we must iterate through the children of the box we packed
445 * into the button, there are two of them
446 */
447
448 GList* contents = gtk_container_get_children (GTK_CONTAINER (box));
449 while (contents != NULL)
450 {
451 widget = GTK_WIDGET (contents->data);
452 /*Show or hide the icon*/
453 if (GTK_IS_IMAGE (widget))
454 {
455 if ((button_space < icon_size * 3) && (button_space > 1))
456 gtk_widget_hide (widget);
457
458 else
459 gtk_widget_show (widget);
460
461 }
462
463 /*Show or hide the label*/
464 if (GTK_IS_LABEL (widget))
465 {
466 if ((button_space < icon_size) && (button_space > 1))
467 {
468 gtk_widget_hide (widget);
469 /*We can go a little wider for empty buttons*/
470 gtk_widget_set_size_request (button, tasklist_width / real_buttons * 0.9, -1);
471 if (task)
472 gtk_widget_hide (task->label);
473
474 }
475 else
476 {
477 gtk_widget_show (widget);
478 }
479 }
480 contents = contents->next;
481 }
482 children = children->next;
483 }
484 return;
485}

487static void
488foreign_toplevel_handle_closed (void *data,
489 struct zwlr_foreign_toplevel_handle_v1 *toplevel)
490{
491 ToplevelTask *task = data;
492
493 if (task->button)
494 {
495 GtkOrientation orient;
496 GtkWidget *outer_box, *parent_box;
497 int real_buttons, button_space;
498
499 outer_box = gtk_widget_get_parent (GTK_WIDGET (task->button));
500 gtk_widget_destroy (task->button);
501 buttons = buttons -1;
502
503 if (tasklist_invocations > 1)
504 real_buttons = buttons / 2;
505
506 else
507 real_buttons = buttons;
508
509 if (real_buttons == 0)
510 return;
511
512 /* We don't need to modify button size on a vertical panel*/
513 orient = gtk_orientable_get_orientation (GTK_ORIENTABLE (outer_box));
514 if (orient == GTK_ORIENTATION_VERTICAL)
515 return;
516
517 /*Get the box the tasklist outer box sits in
518 *and leave a little space so the buttons don't push other applets off the panel
519 */
520
521 parent_box = gtk_widget_get_ancestor ((outer_box), GTK_TYPE_BOX);
522 tasklist_width = MAX(gtk_widget_get_allocated_width (parent_box), tasklist_width) ;
523 button_space = (tasklist_width / real_buttons) * 0.75;
524 button_space = MIN(button_space, full_button_width);
525 adjust_buttons (GTK_CONTAINER(outer_box), button_space, real_buttons, NULL);
526 }
527}

602static gboolean on_toplevel_button_press (GtkWidget *button, GdkEvent *event, TasklistManager *tasklist)
603{
604 /* Assume event is a button press */
605 if (((GdkEventButton*)event)->button == GDK_BUTTON_SECONDARY)
606 {
607 ContextMenu *menu = tasklist->context_menu;
608 ToplevelTask *task = g_object_get_data (G_OBJECT (button), toplevel_task_key);
609
610 g_object_set_data (G_OBJECT (menu->maximize), toplevel_task_key, task);
611 g_object_set_data (G_OBJECT (menu->minimize), toplevel_task_key, task);
612 g_object_set_data (G_OBJECT (menu->close), toplevel_task_key, task);
613
614 gtk_menu_item_set_label (GTK_MENU_ITEM (menu->minimize),
615 task->minimized ? "Unminimize" : "Minimize");
616 gtk_menu_item_set_label (GTK_MENU_ITEM (menu->maximize),
617 task->maximized ? "Unmaximize" : "Maximize");
618
619 gtk_menu_popup_at_widget (GTK_MENU (menu->menu), button,
620 GDK_GRAVITY_NORTH_WEST, GDK_GRAVITY_SOUTH_WEST, event);
621 return TRUE;
622 }
623 else
624 {
625 return FALSE;
626 }
627}

629static ToplevelTask *
630toplevel_task_new (TasklistManager *tasklist, struct zwlr_foreign_toplevel_handle_v1 *toplevel)
631{
632 ToplevelTask *task = g_new0 (ToplevelTask, 1);
633 GtkOrientation orient;
634 GtkWidget *whole_panel_box, *parent_box;
635 int real_buttons, button_space, panel_width;
636
637 buttons = buttons + 1;
638 orient = gtk_orientable_get_orientation (GTK_ORIENTABLE (tasklist->outer_box));
639 task->button = gtk_button_new ();
640 g_signal_connect (task->button, "clicked", G_CALLBACK (toplevel_task_handle_clicked), task);
641 g_signal_connect (task->button, "activate", G_CALLBACK (toggle_window), task);
642
643 task->icon = gtk_image_new_from_icon_name ("unknown", icon_size);
644
645 task->label = gtk_label_new ("");
646 gtk_label_set_max_width_chars (GTK_LABEL (task->label), TASKLIST_TEXT_MAX_WIDTH);
647 gtk_label_set_ellipsize (GTK_LABEL (task->label), PANGO_ELLIPSIZE_END);
648 gtk_label_set_xalign (GTK_LABEL (task->label), 0.0);
649
650 GtkWidget *box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);
651 gtk_box_pack_start (GTK_BOX (box), task->icon, FALSE, FALSE, 6);
652 gtk_box_pack_start (GTK_BOX (box), task->label, TRUE, TRUE, 2);
653
654 gtk_container_add (GTK_CONTAINER (task->button), box);
655 gtk_widget_set_name (task->button , "tasklist-button");
656 gtk_widget_show_all (task->button);
657
658 task->toplevel = toplevel;
659 zwlr_foreign_toplevel_handle_v1_add_listener (toplevel,
660 &foreign_toplevel_handle_listener,
661 task);
662 g_object_set_data_full (G_OBJECT (task->button),
663 toplevel_task_key,
664 task,

670
671 /* Buttons on a vertical panel are not affected by how many are needed
672 * GTK handles compressing contents as needed as the window width tells
673 * GTK how much space to allocate the label and icon. Buttons will use
674 * the full width of a vertical panel without any special attention
675 * so break out here instead of breaking the vertical panel case
676 */
677
678 if (orient == GTK_ORIENTATION_VERTICAL)
679 return task;
680
681 /* On horizontal panels, GTK does not by default limit the width of the tasklist
682 * as it does not run out of space in the window until the entire panel is used,
683 * leaving buttons at full width until then and overflowing all other applets
684 *
685 * Thus we must get the tasklist's allocated width when extra space remains,
686 * which will be most of the distance between the handle and the next applet
687 * From there, we can expand buttons and/or hide elements as needed
688 * For some reason this function always gets called twice, so use half the value of buttons
689 * but do not attempt to adjust the global value as it would get adjusted twice
690 * Since we are adding a button here the true value cannot be zero
691 */
692 whole_panel_box = gtk_widget_get_toplevel(GTK_WIDGET (tasklist->outer_box));
693 parent_box = gtk_widget_get_ancestor(GTK_WIDGET (tasklist->outer_box), GTK_TYPE_BOX);
694 if (gtk_widget_get_allocated_width (parent_box) > 1)
695 {
696 tasklist_width = gtk_widget_get_allocated_width (parent_box);
697 }
698 else
699 {
700 tasklist_width = MAX(gtk_widget_get_allocated_width (parent_box), tasklist_width);
701 }
702
703 panel_width = gtk_widget_get_allocated_width (whole_panel_box);
704
705 /*on startup we get an allocated with of zero, so start with 1/3 the panel width
706 *as a sane default
707 *This may overflow on very crowded panels where the tasklist is less than 1/3ed the
708 *panel witth but will self-correct on opening or closing a few windows
709 */
710
711 if (tasklist_width <= 2)
712 tasklist_width = panel_width / 3;
713
714 if (tasklist_invocations > 1)
715 real_buttons = MAX ((buttons / 2), 1);
716
717 else
718 real_buttons = MAX ((buttons), 1);
719
720 /*always allow at least three buttons to fit without adjustment
721 *so short window lists don't overflow
722 */
723 if (tasklist_width > 0)
724 {
725 full_button_width = MIN(max_button_width, tasklist_width / 3);
726 }
727
728 /*Leave a little space so the buttons don't push other applets off the panel*/
729 button_space = (tasklist_width / real_buttons) * 0.75;
730 button_space = MIN(button_space, full_button_width);
731
732 /* iterate over all the buttons*/
733 adjust_buttons (GTK_CONTAINER (tasklist->list), button_space, real_buttons, task);
734
735 /*Reset the tasklist width after button adjustments*/
736 if (gtk_widget_get_allocated_width (parent_box) > 1)
737 {
738 tasklist_width = gtk_widget_get_allocated_width (parent_box);
739 }
740 else
741 {
742 tasklist_width = MAX(gtk_widget_get_allocated_width (parent_box), tasklist_width);
743 }
744 return task;
745}
