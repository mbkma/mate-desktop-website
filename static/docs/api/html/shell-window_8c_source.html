43GtkWidget *
44shell_window_new (AppShellData * app_data)
45{
46 ShellWindow *window = g_object_new (SHELL_WINDOW_TYPE, NULL);
47
48 gtk_widget_set_app_paintable (GTK_WIDGET (window), TRUE);
49 gtk_frame_set_shadow_type(GTK_FRAME(window), GTK_SHADOW_NONE);
50
51 window->_hbox = GTK_BOX (gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0));
52 gtk_container_add (GTK_CONTAINER (window), GTK_WIDGET (window->_hbox));
53
54 return GTK_WIDGET (window);
55}
