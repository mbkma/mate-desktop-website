199void
200open_file_cb (GtkMenuItem *pmenu, gpointer dummy)
201{
202 GFile *file;
203
204 g_assert (!dummy);
205 g_assert (baobab.selected_path);
206
207 file = g_file_parse_name (baobab.selected_path);
208
209 if (!g_file_query_exists (file, NULL)) {
210 message (_("The document does not exist."), "",
211 GTK_MESSAGE_INFO, baobab.window);
212 g_object_unref (file);
213 return;
214 }
215
216 open_file_with_application (file);
217 g_object_unref (file);
218}

220void
221trash_dir_cb (GtkMenuItem *pmenu, gpointer dummy)
222{
223 GFile *file;
224
225 g_assert (!dummy);
226 g_assert (baobab.selected_path);
227
228 file = g_file_parse_name (baobab.selected_path);
229
230 if (trash_file (file)) {
231 GtkTreeIter iter;
232 guint64 filesize;
233 GtkTreeSelection *selection;
234
235 selection =
236 gtk_tree_view_get_selection ((GtkTreeView *) baobab.
237 tree_view);
238 gtk_tree_selection_get_selected (selection, NULL, &iter);
239 gtk_tree_model_get ((GtkTreeModel *) baobab.model, &iter,
240 5, &filesize, -1);
241 gtk_tree_store_remove (GTK_TREE_STORE (baobab.model),
242 &iter);
243 }
244
245 g_object_unref (file);
246}

254void
255on_ck_allocated_activate (GtkToggleAction *action,
256 gpointer user_data)
257{
258 if (!baobab.is_local)
259 return;
260
261 baobab.show_allocated = gtk_toggle_action_get_active (action);
262
263 baobab_treeview_show_allocated_size (baobab.tree_view,
264 baobab.show_allocated);
265
266 baobab_set_busy (TRUE);
267 baobab_set_statusbar (_("Calculating percentage bars..."));
268 gtk_tree_model_foreach (GTK_TREE_MODEL (baobab.model),

270 baobab_set_busy (FALSE);
271 baobab_set_statusbar (_("Ready"));

298void
299on_tv_selection_changed (GtkTreeSelection *selection, gpointer user_data)
300{
301 GtkTreeIter iter;
302
303 if (gtk_tree_selection_get_selected (selection, NULL, &iter)) {
304 GtkTreePath *path;
305
306 path = gtk_tree_model_get_path (GTK_TREE_MODEL (baobab.model), &iter);
307
308 baobab_chart_set_root (baobab.rings_chart, path);
309 baobab_chart_set_root (baobab.treemap_chart, path);
310
311 gtk_tree_path_free (path);
312 }
