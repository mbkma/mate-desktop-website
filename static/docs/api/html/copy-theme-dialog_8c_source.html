87static void
88copy_theme_dialog_class_init (CopyThemeDialogClass *klass)
89{
90 GObjectClass *object_class = G_OBJECT_CLASS (klass);
91
92 klass->cancelled = copy_theme_dialog_cancel;

94
95 GTK_DIALOG_CLASS (klass)->response = copy_theme_dialog_response;
96
97 signals[CANCELLED] = g_signal_new ("cancelled",
98 G_TYPE_FROM_CLASS (object_class),

102 g_cclosure_marshal_VOID__VOID,
103 G_TYPE_NONE, 0);

109 NULL, NULL,
110 g_cclosure_marshal_VOID__VOID,
111 G_TYPE_NONE, 0);
112
113 parent_class = g_type_class_peek_parent (klass);
114}

231static void
232add_file_to_dialog (gpointer data, gpointer user_data)
233{
234 CopyThemeDialogPrivate *priv;
235 GFile *file;
236 gchar *basename = NULL, *raw_basename;
237
238 priv = COPY_THEME_DIALOG (user_data)->priv;
239 file = G_FILE (data);
240
241 raw_basename = g_file_get_basename (file);
242 if (g_str_has_suffix (raw_basename, ".desktop"))
243 {
244 /* FIXME: validate key file? */
245 basename = g_strndup (raw_basename,
246 /* 8 = strlen (".desktop") */
247 strlen (raw_basename) - 8);
248 }
249 g_free (raw_basename);
250
251 if (basename)
252 {
253 g_object_ref (file);
254 priv->all_files = g_slist_append (priv->all_files, file);
255 priv->all_basenames = g_slist_append (priv->all_basenames, basename);
256 priv->total_files++;
257 }
258
259 else
260 {
261 GtkWidget *dialog;
262 gchar *uri;
263
264 dialog = gtk_message_dialog_new (GTK_WINDOW (user_data),
265 GTK_DIALOG_MODAL,
266 GTK_MESSAGE_ERROR,
267 GTK_BUTTONS_OK,
268 _("Invalid screensaver theme"));
269 uri = g_file_get_uri (file);
270 gtk_message_dialog_format_secondary_text (GTK_MESSAGE_DIALOG (dialog),
271 _("%s does not appear to be a valid screensaver theme."),
272 uri);
273 g_free (uri);
274 gtk_window_set_title (GTK_WINDOW (dialog), "");
275 gtk_window_set_icon_name (GTK_WINDOW (dialog), "preferences-desktop-screensaver");
276
277 gtk_dialog_run (GTK_DIALOG (dialog));
278 gtk_widget_destroy (dialog);
279 }

282static void
283single_copy_complete (GObject *source_object, GAsyncResult *res,
284 gpointer user_data)
285{
286 GError *error = NULL;
287 gboolean should_continue = FALSE;
288 CopyThemeDialog *dialog = COPY_THEME_DIALOG (user_data);
289
290 if (g_file_copy_finish (G_FILE (source_object), res, &error))
291 {
292 should_continue = TRUE;
293 }
294
295 else
296 {
297 /* If the file already exists, generate a new random name
298 * and try again.
299 **/
300 if (g_error_matches (error, G_IO_ERROR, G_IO_ERROR_EXISTS))
301 {
302 GFile *file, *destination;
303 gchar *basename, *full_basename;
304 g_error_free (error);
305
306 file = G_FILE (dialog->priv->file->data);
307 basename = (gchar *) (dialog->priv->basename->data);
308
309 g_return_if_fail (file != NULL);
310 g_return_if_fail (basename != NULL);
311
312 full_basename = g_strdup_printf ("%s-%u.desktop",
313 basename,
314 g_random_int ());
315 destination = g_file_get_child (dialog->priv->theme_dir,
316 full_basename);
317 g_free (full_basename);
318
319 g_file_copy_async (file, destination, G_FILE_COPY_NONE,
320 G_PRIORITY_DEFAULT,
321 dialog->priv->cancellable,
322 NULL, NULL,
323 single_copy_complete, dialog);
324 }
325
326 else
327 {
328 if (g_error_matches (error, G_IO_ERROR,
329 G_IO_ERROR_CANCELLED))
330 {
331 /* User has cancelled the theme copy */
332 g_signal_emit (G_OBJECT (dialog),
333 signals[CANCELLED],
334 0, NULL);
335 }
336
337 else
338 {
339 /* Some other error occurred, ignore and
340 * try to copy remaining files
341 **/
342 should_continue = TRUE;
343 }
344
345 g_error_free (error);
346 }
347 }
348
349 /* Update informational widgets and, if needed, signal
350 * copy completion.
351 **/
352 if (should_continue)
353 {
354 dialog->priv->index++;
355 dialog->priv->file = dialog->priv->file->next;
356 dialog->priv->basename = dialog->priv->basename->next;
357 copy_theme_dialog_update_num_files (dialog);
358 copy_theme_dialog_copy_next (dialog);
359 }
360}

362/* Try to copy the theme file to the user's screensaver directory.
363 * If a file with the given name already exists, the error will be
364 * caught later and the copy re-attempted with a random value
365 * appended to the filename.
366**/
367static void
368copy_theme_dialog_copy_next (CopyThemeDialog *dialog)
369{
370 GFile *file, *destination;
371 gchar *basename, *full_basename;
372
373 if (copy_finished (dialog))
374 {
375 g_signal_emit (G_OBJECT (dialog), signals[COMPLETE],
376 0, NULL);
377 return;
378 }
379
380 file = G_FILE (dialog->priv->file->data);
381 basename = (gchar *) (dialog->priv->basename->data);
382
383 g_return_if_fail (file != NULL);
384 g_return_if_fail (basename != NULL);
385
386 full_basename = g_strdup_printf ("%s.desktop", basename);
387 destination = g_file_get_child (dialog->priv->theme_dir, full_basename);
388 g_free (full_basename);
389
390 g_file_copy_async (file, destination, G_FILE_COPY_NONE,
391 G_PRIORITY_DEFAULT, dialog->priv->cancellable,
392 NULL, NULL, single_copy_complete, dialog);
393}

395static gboolean
396timeout_display_dialog (gpointer data)
397{
398 if (IS_COPY_THEME_DIALOG (data))
399 {
400 CopyThemeDialog *dialog = COPY_THEME_DIALOG (data);
401 if (!copy_finished (dialog))
402 {
403 gtk_widget_show (GTK_WIDGET (dialog));
404
405 g_signal_connect (dialog, "response",
406 G_CALLBACK (copy_theme_dialog_response),
407 dialog);
408 }
409 }
