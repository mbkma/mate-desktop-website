99static void
100eel_wrap_table_set_property (GObject *object,
101 guint property_id,

104{
105 EelWrapTable *wrap_table;
106

123 break;

128
129 case PROP_HOMOGENEOUS:

233static int
234eel_wrap_table_draw (GtkWidget *widget,
235 cairo_t *cr)
236{
237 EelWrapTable *wrap_table;
238 GList *iterator;
239
240 g_assert (EEL_IS_WRAP_TABLE (widget));
241 g_assert (gtk_widget_get_realized (widget));
242
243 wrap_table = EEL_WRAP_TABLE (widget);
244
245 for (iterator = wrap_table->details->children; iterator; iterator = iterator->next)
246 {
247 g_assert (GTK_IS_WIDGET (iterator->data));
248 gtk_container_propagate_draw (GTK_CONTAINER (widget),
249 GTK_WIDGET (iterator->data),
250 cr);
251 }
252
253 /*Redraw the table once and only once to ensure it is displayed */
254 if (wrap_table->details->drawn == FALSE){
255 gtk_widget_queue_allocate (GTK_WIDGET(widget));
256 wrap_table->details->drawn = TRUE;
257 }
258 return FALSE;

321/* GtkContainerClass methods */
322static void
323eel_wrap_table_add (GtkContainer *container,
324 GtkWidget *child)
325{
326 EelWrapTable *wrap_table;
327 GtkWidget *widget;
328
329 g_assert (container != NULL);
330 g_assert (EEL_IS_WRAP_TABLE (container));
331 g_assert (GTK_IS_WIDGET (child));
332
333 widget = GTK_WIDGET (container);
334 wrap_table = EEL_WRAP_TABLE (container);
335
336 gtk_widget_set_parent (child, GTK_WIDGET (container));
337
338 wrap_table->details->children = g_list_append (wrap_table->details->children, child);
339
340 if (gtk_widget_get_realized (widget))
341 {
342 gtk_widget_realize (child);
343 }
344
345 if (gtk_widget_get_visible (widget) && gtk_widget_get_visible (child))
346 {
347 if (gtk_widget_get_mapped (widget))
348 {
349 gtk_widget_map (child);
350 }
351
352 gtk_widget_queue_resize (child);
353 }
354
355 if (wrap_table->details->is_scrolled)
356 {
357 g_signal_connect (child, "focus_in_event",
358 G_CALLBACK (wrap_table_child_focus_in),
359 wrap_table);
360 }
361}

363static void
364eel_wrap_table_remove (GtkContainer *container,
365 GtkWidget *child)
366{
367 EelWrapTable *wrap_table;
368 gboolean child_was_visible;
369
370 g_assert (EEL_IS_WRAP_TABLE (container));
371 g_assert (GTK_IS_WIDGET (child));
372
373 wrap_table = EEL_WRAP_TABLE (container);;
374
375 child_was_visible = gtk_widget_get_visible (child);
376 gtk_widget_unparent (child);
377 wrap_table->details->children = g_list_remove (wrap_table->details->children, child);
378
379 if (child_was_visible)
380 {
381 gtk_widget_queue_resize (GTK_WIDGET (container));
382 }
383
384 if (wrap_table->details->is_scrolled)
385 {
386 g_signal_handlers_disconnect_by_func (
387 child,
388 G_CALLBACK (wrap_table_child_focus_in),
389 wrap_table);
390 }

488static int
489wrap_table_get_num_fitting (int available,
490 int spacing,
491 int max_child_size)
492{
493 int num;
494
495 g_assert (max_child_size > 0);
496 g_assert (spacing >= 0);
497
498 available = MAX (available, 0);
499
500 num = (available + spacing) / (max_child_size + spacing);
501 num = MAX (num, 1);
502
503 return num;
504}

506static void
507wrap_table_layout (EelWrapTable *wrap_table)
508{
509 GList *iterator;
510 EelIPoint pos;
511 EelDimensions max_child_dimensions;
512 EelIRect content_bounds;
513 guint num_cols;
514 GtkAllocation allocation;
515
516 g_assert (EEL_IS_WRAP_TABLE (wrap_table));
517
518 max_child_dimensions = wrap_table_get_max_child_dimensions (wrap_table);
519 max_child_dimensions.width = MAX (max_child_dimensions.width, 1);
520 max_child_dimensions.height = MAX (max_child_dimensions.height, 1);
521
522 content_bounds = wrap_table_get_content_bounds (wrap_table);
523 pos.x = content_bounds.x0;
524 pos.y = content_bounds.y0;
525
526 gtk_widget_get_allocation (GTK_WIDGET (wrap_table), &allocation);
527 num_cols = wrap_table_get_num_fitting (allocation.width -
528 gtk_container_get_border_width (GTK_CONTAINER (wrap_table)) * 2,
529 wrap_table->details->x_spacing,
530 max_child_dimensions.width);
531 if (num_cols != wrap_table->details->cols)
532 {
533 wrap_table->details->cols = num_cols;
534 gtk_widget_queue_resize (GTK_WIDGET (wrap_table));
535 return;
536 }
537
538 for (iterator = wrap_table->details->children; iterator; iterator = iterator->next)
539 {
540 GtkWidget *item;
541
542 item = iterator->data;
543
544 if (gtk_widget_get_visible (item))
545 {
546 GtkAllocation item_allocation;
547
548 if (wrap_table->details->homogeneous)
549 {
550 item_allocation.x = pos.x;
551 item_allocation.y = pos.y;
552 item_allocation.width = max_child_dimensions.width;
553 item_allocation.height = max_child_dimensions.height;
554
555 if ((pos.x + max_child_dimensions.width) > content_bounds.x1)
556 {
557 pos.x = content_bounds.x0 + wrap_table->details->x_spacing + max_child_dimensions.width;
558 pos.y += (max_child_dimensions.height + wrap_table->details->y_spacing);
559 item_allocation.x = content_bounds.x0;
560 item_allocation.y = pos.y;
561 }
562 else
563 {
564 pos.x += (wrap_table->details->x_spacing + max_child_dimensions.width);
565 }
566 }
567 else
568 {
569 GtkRequisition item_requisition;
570
571 gtk_widget_get_preferred_size (item, &item_requisition, NULL);
572
573 item_allocation.x = pos.x;
574 item_allocation.y = pos.y;
575 item_allocation.width = item_requisition.width;
576 item_allocation.height = item_requisition.height;
577
578 g_assert (item_allocation.width <= max_child_dimensions.width);
579 g_assert (item_allocation.height <= max_child_dimensions.height);
580
581 if ((pos.x + max_child_dimensions.width) > content_bounds.x1)
582 {
583 pos.x = content_bounds.x0 + wrap_table->details->x_spacing + max_child_dimensions.width;
584 pos.y += (max_child_dimensions.height + wrap_table->details->y_spacing);
585 item_allocation.x = content_bounds.x0;
586 item_allocation.y = pos.y;
587 }
588 else
589 {
590 pos.x += (wrap_table->details->x_spacing + max_child_dimensions.width);
591 }
592
593 switch (wrap_table->details->x_justification)
594 {
595 case EEL_JUSTIFICATION_MIDDLE:
596 item_allocation.x += (max_child_dimensions.width - (int) item_allocation.width) / 2;
597 break;
598 case EEL_JUSTIFICATION_END:
599 item_allocation.x += (max_child_dimensions.width - (int) item_allocation.width);
600 break;
601 default:
602 break;
603 }
604
605 switch (wrap_table->details->y_justification)
606 {
607 case EEL_JUSTIFICATION_MIDDLE:
608 item_allocation.y += (max_child_dimensions.height - (int) item_allocation.height) / 2;
609 break;
610 case EEL_JUSTIFICATION_END:
611 item_allocation.y += (max_child_dimensions.height - (int) item_allocation.height);
612 break;
613 default:
614 break;
615 }
616 }
617
618 gtk_widget_size_allocate (item, &item_allocation);
619 }
620 }

638static EelDimensions
639wrap_table_get_max_child_dimensions (const EelWrapTable *wrap_table)
640{
641 EelDimensions max_dimensions;
642 GList *iterator;
643
644 g_assert (EEL_IS_WRAP_TABLE (wrap_table));
645
646 max_dimensions = eel_dimensions_empty;
647
648 for (iterator = wrap_table->details->children; iterator; iterator = iterator->next)
649 {
650 GtkWidget *child;
651
652 child = iterator->data;
653
654 if (gtk_widget_get_visible (child))
655 {
656 GtkRequisition child_requisition;
657 EelDimensions child_dimensions;
658
659 gtk_widget_get_preferred_size (child, &child_requisition, NULL);
660
661 child_dimensions.width = (int) child_requisition.width;
662 child_dimensions.height = (int) child_requisition.height;
663
664 max_dimensions = wrap_table_irect_max_dimensions (&child_dimensions, &max_dimensions);
665 }
666 }
667
668 return max_dimensions;
669}

671static EelDimensions
672wrap_table_get_content_dimensions (const EelWrapTable *wrap_table)
673{
674 EelDimensions content_dimensions;
675 guint num_children;
676
677 g_assert (EEL_IS_WRAP_TABLE (wrap_table));
678
679 content_dimensions = eel_dimensions_empty;
680
681 num_children = g_list_length (wrap_table->details->children);
682
683 if (num_children > 0)
684 {
685 EelDimensions max_child_dimensions;
686 EelDimensions dimensions;
687 int num_cols;
688 int num_rows;
689
690 dimensions = eel_gtk_widget_get_dimensions (GTK_WIDGET (wrap_table));
691 max_child_dimensions = wrap_table_get_max_child_dimensions (wrap_table);
692
693 max_child_dimensions.width = MAX (max_child_dimensions.width, 1);
694 max_child_dimensions.height = MAX (max_child_dimensions.height, 1);
695
696 num_cols = wrap_table_get_num_fitting (dimensions.width -
697 gtk_container_get_border_width (GTK_CONTAINER (wrap_table)) * 2,
698 wrap_table->details->x_spacing,
699 max_child_dimensions.width);
700 num_rows = num_children / num_cols;
701 num_rows = MAX (num_rows, 1);
702
703 if ((num_children % num_rows) > 0)
704 {
705 num_rows++;
706 }
707
708 content_dimensions.width = dimensions.width;
709 content_dimensions.height = num_rows * max_child_dimensions.height;
710
711 content_dimensions.width += (num_cols - 1) * wrap_table->details->x_spacing;
712 content_dimensions.height += (num_rows - 1) * wrap_table->details->y_spacing;
713 }
714
715 return content_dimensions;
716}

743static gboolean
744wrap_table_child_visible_in (GtkWidget *child, GtkWidget *scrolled)
745{
746 gint x, y;
