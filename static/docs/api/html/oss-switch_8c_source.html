98OssSwitch *
99oss_switch_new (OssStream *stream,
100 const gchar *name,
101 const gchar *label,
102 gint fd,
103 GList *options)
104{
105 OssSwitch *swtch;
106 gint newfd;
107
108 g_return_val_if_fail (OSS_IS_STREAM (stream), NULL);
109 g_return_val_if_fail (name != NULL, NULL);
110 g_return_val_if_fail (label != NULL, NULL);
111 g_return_val_if_fail (options != NULL, NULL);
112
113 newfd = dup (fd);
114 if (newfd == -1) {
115 g_warning ("Failed to duplicate file descriptor: %s",
116 g_strerror (errno));
117 return NULL;
118 }
119
120 swtch = g_object_new (OSS_TYPE_SWITCH,
121 "name", name,
122 "label", label,
123 "role", MATE_MIXER_STREAM_SWITCH_ROLE_PORT,
124 "stream", stream,
125 NULL);
126
127 /* Takes ownership of options */
128 swtch->priv->fd = newfd;
129 swtch->priv->options = options;
130
131 return swtch;
132}

134void
135oss_switch_load (OssSwitch *swtch)
136{
137 OssSwitchOption *option;
138 gint recsrc;
139 gint ret;
140
141 g_return_if_fail (OSS_IS_SWITCH (swtch));
142
143 if (G_UNLIKELY (swtch->priv->fd == -1))
144 return;
145
146 /* Recsrc contains a bitmask of currently enabled recording sources */
147 ret = ioctl (swtch->priv->fd, MIXER_READ (SOUND_MIXER_RECSRC), &recsrc);
148 if (ret == -1)
149 return;
150
151 if (recsrc != 0) {
152 GList *list = swtch->priv->options;
153
154 /* Find out which option is currently selected */
155 while (list != NULL) {
156 gint devnum;
157
158 option = OSS_SWITCH_OPTION (list->data);
159 devnum = oss_switch_option_get_devnum (option);
160
161 if (recsrc & (1 << devnum)) {
162 /* It is possible that some hardware might allow and have more recording
163 * sources active at the same time, but we only support one active
164 * source at a time */
165 _mate_mixer_switch_set_active_option (MATE_MIXER_SWITCH (swtch),
166 MATE_MIXER_SWITCH_OPTION (option));
167 return;
168 }
169 list = list->next;
170 }
171
172 g_debug ("Switch %s has an unknown device as the active option",
173 mate_mixer_switch_get_name (MATE_MIXER_SWITCH (swtch)));
174
175 /* OSS shouldn't let a non-record device be selected, let's step in and select
176 * something reasonable instead... */
177 } else {
178 g_debug ("Switch %s has no active device",
179 mate_mixer_switch_get_name (MATE_MIXER_SWITCH (swtch)));
180
181 /* According to the OSS Programmer's Guide, if the recsrc value is 0, the
182 * microphone will be selected implicitly.
183 * Let's not assume that's true everywhere and select something explicitly... */
184 }
185
186 option = choose_default_option (swtch);
187
188 g_debug ("Selecting default device %s as active for switch %s",
189 mate_mixer_switch_option_get_name (MATE_MIXER_SWITCH_OPTION (option)),
190 mate_mixer_switch_get_name (MATE_MIXER_SWITCH (swtch)));
191
192 if (mate_mixer_switch_set_active_option (MATE_MIXER_SWITCH (swtch),
193 MATE_MIXER_SWITCH_OPTION (option)) == FALSE) {
194 g_debug ("Failed to set the default device, assuming it is selected anyway");
195
196 _mate_mixer_switch_set_active_option (MATE_MIXER_SWITCH (swtch),
197 MATE_MIXER_SWITCH_OPTION (option));
198 }
199}

213static gboolean
214oss_switch_set_active_option (MateMixerSwitch *mms, MateMixerSwitchOption *mmso)
215{
216 OssSwitch *swtch;
217 gint ret;
218 gint recsrc;
219
220 g_return_val_if_fail (OSS_IS_SWITCH (mms), FALSE);
221 g_return_val_if_fail (OSS_IS_SWITCH_OPTION (mmso), FALSE);
222
223 swtch = OSS_SWITCH (mms);
224
225 if (G_UNLIKELY (swtch->priv->fd == -1))
226 return FALSE;
227
228 recsrc = 1 << oss_switch_option_get_devnum (OSS_SWITCH_OPTION (mmso));
229
230 ret = ioctl (swtch->priv->fd, MIXER_WRITE (SOUND_MIXER_RECSRC), &recsrc);
231 if (ret == -1)
232 return FALSE;
233
234 return TRUE;
235}

247static OssSwitchOption *
248choose_default_option (OssSwitch *swtch)
249{
250 GList *list = swtch->priv->options;
251
252 /* Search for the preferred device */
253 while (list != NULL) {
254 OssSwitchOption *option;
255 gint devnum;
256
257 option = OSS_SWITCH_OPTION (list->data);
258 devnum = oss_switch_option_get_devnum (option);
259
260 if (devnum == OSS_SWITCH_PREFERRED_DEFAULT_DEVNUM)
261 return option;
262
263 list = list->next;
264 }
265 /* If the preferred device is not present, use the first available one */
266 return OSS_SWITCH_OPTION (swtch->priv->options->data);
267}
