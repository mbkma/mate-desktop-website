33/* when a user closes the prefs-dialog we assume that applet is now
34 setup, so store all values in gsettings */
35void prefs_dialog_close(SensorsApplet *sensors_applet) {
36
37 if (sensors_applet->sensors != NULL) {
38 sensors_applet_settings_save_sensors(sensors_applet);
39 }
40
41 if (sensors_applet->prefs_dialog) {
42 gtk_widget_destroy(GTK_WIDGET(sensors_applet->prefs_dialog->dialog));
43 g_free(sensors_applet->prefs_dialog);
44 sensors_applet->prefs_dialog = NULL;
45 }
46
47 if (sensors_applet->timeout_id == 0) {
48 sensors_applet->timeout_id = g_timeout_add_seconds(g_settings_get_int (sensors_applet->settings, TIMEOUT) / 1000, (GSourceFunc)sensors_applet_update_active_sensors, sensors_applet);
49 }
50
51}

53static
54void prefs_dialog_response(GtkDialog *prefs_dialog,
55 gint response,
56 gpointer data) {
57
58 SensorsApplet *sensors_applet;
59 GError *error = NULL;
60 gint current_page;
61 gchar *uri;
62
63 sensors_applet = (SensorsApplet *)data;
64
65 switch (response) {
66 case GTK_RESPONSE_HELP:
67 g_debug("loading help in prefs");
68 current_page = gtk_notebook_get_current_page(sensors_applet->prefs_dialog->notebook);
69 uri = g_strdup_printf("help:mate-sensors-applet/%s",
70 ((current_page == 0) ?
71 "sensors-applet-general-options" :
72 ((current_page == 1) ?
73 "sensors-applet-sensors" :
74 NULL)));
75 gtk_show_uri_on_window(NULL, uri, gtk_get_current_event_time(), &error);
76 g_free(uri);
77
78 if (error) {
79 g_debug("Could not open help document: %s ",error->message);
80 g_error_free (error);
81 }
82 break;
83 default:
84 g_debug("closing prefs dialog");
85 prefs_dialog_close(sensors_applet);
86 }
87}

89static gboolean prefs_dialog_convert_low_and_high_values(GtkTreeModel *model,
90 GtkTreePath *path,
91 GtkTreeIter *iter,
92 TemperatureScale scales[2]) {
93
94 SensorType sensor_type;
95 gdouble low_value, high_value;
96
97 gtk_tree_model_get(model,
98 iter,
99 SENSOR_TYPE_COLUMN, &sensor_type,
100 LOW_VALUE_COLUMN, &low_value,
101 HIGH_VALUE_COLUMN, &high_value,
102 -1);
103
104 if (sensor_type == TEMP_SENSOR) {
105 low_value = sensors_applet_convert_temperature(low_value,
106 scales[OLD_TEMP_SCALE],
107 scales[NEW_TEMP_SCALE]);
108
109 high_value = sensors_applet_convert_temperature(high_value,
110 scales[OLD_TEMP_SCALE],
111 scales[NEW_TEMP_SCALE]);
112
113 gtk_tree_store_set(GTK_TREE_STORE(model),
114 iter,
115 LOW_VALUE_COLUMN, low_value,
116 HIGH_VALUE_COLUMN, high_value,
117 -1);
118 }
119 return FALSE;
120}

169static void prefs_dialog_temperature_scale_changed(GtkComboBox *temperature_scale_combo_box,
170 PrefsDialog *prefs_dialog) {
171
172 /* get old temp scale value */
173 TemperatureScale scales[2];
174 GtkTreeModel *model;
175
176 scales[OLD_TEMP_SCALE] = (TemperatureScale) g_settings_get_int (prefs_dialog->sensors_applet->settings, TEMPERATURE_SCALE);
177
178 scales[NEW_TEMP_SCALE] = (TemperatureScale)gtk_combo_box_get_active(temperature_scale_combo_box);
179
180 g_settings_set_int (prefs_dialog->sensors_applet->settings,
181 TEMPERATURE_SCALE,
182 scales[NEW_TEMP_SCALE]);
183
184 /* now go thru and convert all low and high sensor values in
185 * the tree to either celcius or Fahrenheit */
186 model = gtk_tree_view_get_model(prefs_dialog->view);
187 gtk_tree_model_foreach(model,
188 (GtkTreeModelForeachFunc)prefs_dialog_convert_low_and_high_values,
189 scales);
190
191 /* finally update display of active sensors */
192 sensors_applet_update_active_sensors(prefs_dialog->sensors_applet);
193}

235/* callbacks for the tree of sensors */
236static void prefs_dialog_sensor_toggled(GtkCellRenderer *renderer, gchar *path_str, PrefsDialog *prefs_dialog) {
237 GtkTreeIter iter;
238 GtkTreePath *path;
239
240 gboolean old_value;
241
242 path = gtk_tree_path_new_from_string(path_str);
243
244 gtk_tree_model_get_iter(GTK_TREE_MODEL(prefs_dialog->sensors_applet->sensors), &iter, path);
245 gtk_tree_model_get(GTK_TREE_MODEL(prefs_dialog->sensors_applet->sensors),
246 &iter,
247 ENABLE_COLUMN, &old_value,
248 -1);
249
250 if (old_value) {
251 sensors_applet_sensor_disabled(prefs_dialog->sensors_applet, path);
252 } else {
253 sensors_applet_sensor_enabled(prefs_dialog->sensors_applet, path);
254 }
255
256 gtk_tree_store_set(prefs_dialog->sensors_applet->sensors, &iter,
257 ENABLE_COLUMN, !old_value,
258 -1);
259
260 gtk_tree_path_free(path);
261}

288static void prefs_dialog_sensor_up_button_clicked(GtkButton *button, PrefsDialog *prefs_dialog) {
289 GtkTreeModel *model;
290 GtkTreeIter iter;
291
292 if (gtk_tree_selection_get_selected(prefs_dialog->sensors_applet->selection, &model, &iter)) {
293 /* if has no prev node set up button insentive */
294 GtkTreePath *path;
295
296 path = gtk_tree_model_get_path(model, &iter);
297 if (gtk_tree_path_prev(path)) {
298
299 GtkTreeIter prev_iter;
300 /* check is a valid node in out model */
301 if (gtk_tree_model_get_iter(model, &prev_iter, path)) {
302
303 gtk_tree_store_move_before(GTK_TREE_STORE(model),
304 &iter,
305 &prev_iter);
306 g_signal_emit_by_name(prefs_dialog->sensors_applet->selection, "changed");
307 sensors_applet_reorder_sensors(prefs_dialog->sensors_applet);
308
309 }
310 }
311
312 gtk_tree_path_free(path);
313
314 }

341/* if a sensor is selected, make config sure button is able to be
342 * clicked and also set the sensitivities properly for the up and down
343 * buttons */
344static void prefs_dialog_selection_changed(GtkTreeSelection *selection,
345 PrefsDialog *prefs_dialog) {
346
347 GtkTreeIter iter;
348 GtkTreeModel *model;
349
350 /* if there is a selection with no children make config button sensitive */
351 if (gtk_tree_selection_get_selected(selection, &model, &iter)) {
352 GtkTreePath *path;
353
354 if (!gtk_tree_model_iter_has_child(model, &iter)) {
355 gtk_widget_set_sensitive(GTK_WIDGET(prefs_dialog->sensor_config_button), TRUE);
356 } else {
357 /* otherwise make insensitive */
358 gtk_widget_set_sensitive(GTK_WIDGET(prefs_dialog->sensor_config_button), FALSE);
359 }
360
361 /* if has no prev node set up button insentive */
362 path = gtk_tree_model_get_path(model, &iter);
363 if (gtk_tree_path_prev(path)) {
364 GtkTreeIter prev_iter;
365 /* check is a valid node in out model */
366 if (gtk_tree_model_get_iter(model, &prev_iter, path)) {
367 gtk_widget_set_sensitive(GTK_WIDGET(prefs_dialog->sensor_up_button), TRUE);
368 } else {
369 gtk_widget_set_sensitive(GTK_WIDGET(prefs_dialog->sensor_up_button), FALSE);
370 }
371 } else {
372 gtk_widget_set_sensitive(GTK_WIDGET(prefs_dialog->sensor_up_button), FALSE);
373 }
374
375 gtk_tree_path_free(path);
376
377 /* if has no next node set down button insentive */
378 if (gtk_tree_model_iter_next(model, &iter)) {
379 gtk_widget_set_sensitive(GTK_WIDGET(prefs_dialog->sensor_down_button), TRUE);
380 } else {
381 gtk_widget_set_sensitive(GTK_WIDGET(prefs_dialog->sensor_down_button), FALSE);
382 }
383
384 } else {
385 /* otherwise make all insensitive */
386 gtk_widget_set_sensitive(GTK_WIDGET(prefs_dialog->sensor_config_button), FALSE);
387 gtk_widget_set_sensitive(GTK_WIDGET(prefs_dialog->sensor_up_button), FALSE);
388 gtk_widget_set_sensitive(GTK_WIDGET(prefs_dialog->sensor_down_button), FALSE);
389
390 }
