156caja_launch_application (GAppInfo *application,
157 GList *files,

159{

169 parent_window);
170 g_list_free_full (uris, g_free);

192void
193caja_launch_application_by_uri (GAppInfo *application,
194 GList *uris,
195 GtkWindow *parent_window)
196{
197 CajaFile *file;
198 gboolean result;
199 GError *error;
200 GdkDisplay *display;
201 GdkAppLaunchContext *launch_context;
202 CajaIconInfo *icon;
203 int count;
204 GList *locations, *l;
205 GFile *location = NULL;
206
207 g_assert (uris != NULL);
208
209 /* count the number of uris with local paths */
210 count = 0;
211 locations = NULL;
212 for (l = uris; l != NULL; l = l->next)
213 {
214 char *uri;
215
216 uri = l->data;
217
218 location = g_file_new_for_uri (uri);
219 if (g_file_is_native (location))
220 {
221 count++;
222 }
223 locations = g_list_prepend (locations, location);
224 }
225 locations = g_list_reverse (locations);
226
227 if (parent_window != NULL) {
228 display = gtk_widget_get_display (GTK_WIDGET (parent_window));
229 } else {
230 display = gdk_display_get_default ();
231 }
232
233 launch_context = gdk_display_get_app_launch_context (display);
234
235 if (parent_window != NULL) {
236 gdk_app_launch_context_set_screen (launch_context,
237 gtk_window_get_screen (parent_window));
238 }
239
240 file = caja_file_get_by_uri (uris->data);
241 icon = caja_file_get_icon (file,
242 48, gtk_widget_get_scale_factor (GTK_WIDGET (parent_window)),
243 0);
244 caja_file_unref (file);
245 if (icon)
246 {
247 gdk_app_launch_context_set_icon_name (launch_context,
248 caja_icon_info_get_used_name (icon));
249 g_object_unref (icon);
250 }
251
252 error = NULL;
253
254 result = g_desktop_app_info_launch_uris_as_manager (G_DESKTOP_APP_INFO (application),
255 uris,
256 G_APP_LAUNCH_CONTEXT (launch_context),
257 G_SPAWN_SEARCH_PATH | G_SPAWN_DO_NOT_REAP_CHILD,
258 NULL, NULL,
259 gather_pid_callback, application,
260 &error);
261
262 g_object_unref (launch_context);
263
264 if (!result)
265 {
266 if (error->domain == G_IO_ERROR &&
267 error->code == G_IO_ERROR_NOT_SUPPORTED)
268 {
269 char *uri_scheme;
270
271 uri_scheme = g_uri_parse_scheme (uris->data);
272 application_cannot_open_location (application,
273 file,
274 uri_scheme,
275 parent_window);
276 g_free (uri_scheme);
277 }
278 else
279 {
280#ifdef NEW_MIME_COMPLETE
281 caja_program_chooser_show_invalid_message
282 (MATE_VFS_MIME_ACTION_TYPE_APPLICATION, file, parent_window);
283#else
284 g_warning ("Cannot open app: %s\n", error->message);
285#endif
286 }
287 g_error_free (error);
288 }
289 else
290 {
291 for (l = uris; l != NULL; l = l->next)
292 {
293 file = caja_file_get_by_uri (l->data);
294 caja_recent_add_file (file, application);
295 caja_file_unref (file);
296 }
297 }
298
299 g_list_free_full (locations, g_object_unref);
300}

312void
313caja_launch_application_from_command (GdkScreen *screen,
314 const char *name,
315 const char *command_string,
316 gboolean use_terminal,
317 ...)
318{
319 char *full_command, *tmp;
320 char *parameter;
321 va_list ap;
322
323 full_command = g_strdup (command_string);
324
325 va_start (ap, use_terminal);
326
327 while ((parameter = va_arg (ap, char *)) != NULL)
328 {
329 char *quoted_parameter;
330
331 quoted_parameter = g_shell_quote (parameter);
332 tmp = g_strconcat (full_command, " ", quoted_parameter, NULL);
333 g_free (quoted_parameter);
334
335 g_free (full_command);
336 full_command = tmp;
337
338 }
339
340 va_end (ap);
341
342 if (use_terminal)
343 {
344 eel_mate_open_terminal_on_screen (full_command, screen);
345 }
346 else
347 {
348 GAppInfo *app_info = NULL;
349 app_info = g_app_info_create_from_commandline (full_command,
350 NULL,
351 G_APP_INFO_CREATE_NONE,
352 NULL);
353 if (app_info != NULL)
354 {
355 GdkAppLaunchContext *launch_context;
356 GdkDisplay *display;
357
358 display = gdk_screen_get_display (screen);
359 launch_context = gdk_display_get_app_launch_context (display);
360 gdk_app_launch_context_set_screen (launch_context, screen);
361 g_app_info_launch (app_info, NULL, G_APP_LAUNCH_CONTEXT (launch_context), NULL);
362 g_object_unref (launch_context);
363 g_object_unref (app_info);
364 }
365 }
366
367 g_free (full_command);
368}

436void
437caja_launch_desktop_file (GdkScreen *screen,
438 const char *desktop_file_uri,
439 const GList *parameter_uris,
440 GtkWindow *parent_window)
441{
442 GError *error;
443 char *desktop_file_path;
444 const GList *p;
445 GList *files;
446 int total, count;
447 GDesktopAppInfo *app_info;
448 GdkAppLaunchContext *context;
449 GFile *desktop_file;
450 GFile *file = NULL;
451
452 /* Don't allow command execution from remote locations
453 * to partially mitigate the security
454 * risk of executing arbitrary commands.
455 */
456 desktop_file = g_file_new_for_uri (desktop_file_uri);
457 desktop_file_path = g_file_get_path (desktop_file);
458 if (!g_file_is_native (desktop_file))
459 {
460 g_free (desktop_file_path);
461 g_object_unref (desktop_file);
462 eel_show_error_dialog
463 (_("Sorry, but you cannot execute commands from "
464 "a remote site."),
465 _("This is disabled due to security considerations."),
466 parent_window);
467
468 return;
469 }
470 g_object_unref (desktop_file);
471
472 app_info = g_desktop_app_info_new_from_filename (desktop_file_path);
473 g_free (desktop_file_path);
474 if (app_info == NULL)
475 {
476 eel_show_error_dialog
477 (_("There was an error launching the application."),
478 NULL,
479 parent_window);
480 return;
481 }
482
483 /* count the number of uris with local paths */
484 count = 0;
485 total = g_list_length ((GList *) parameter_uris);
486 files = NULL;
487 for (p = parameter_uris; p != NULL; p = p->next)
488 {
489 file = g_file_new_for_uri ((const char *) p->data);
490 if (g_file_is_native (file))
491 {
492 count++;
493 }
494 files = g_list_prepend (files, file);
495 }
496
497 /* check if this app only supports local files */
498 if (g_app_info_supports_files (G_APP_INFO (app_info)) &&
499 !g_app_info_supports_uris (G_APP_INFO (app_info)) &&
500 parameter_uris != NULL)
501 {
502 if (count == 0)
503 {
504 /* all files are non-local */
505 eel_show_error_dialog
506 (_("This drop target only supports local files."),
507 _("To open non-local files copy them to a local folder and then"
508 " drop them again."),
509 parent_window);
510
511 g_list_free_full (files, g_object_unref);
512 g_object_unref (app_info);
513 return;
514 }
515 else if (count != total)
516 {
517 /* some files are non-local */
518 eel_show_warning_dialog
519 (_("This drop target only supports local files."),
520 _("To open non-local files copy them to a local folder and then"
521 " drop them again. The local files you dropped have already been opened."),
522 parent_window);
523 }
524 }
525
526 error = NULL;
527
528 context = gdk_display_get_app_launch_context (gtk_widget_get_display (GTK_WIDGET (parent_window)));
529
530 /* TODO: Ideally we should accept a timestamp here instead of using GDK_CURRENT_TIME */
531 gdk_app_launch_context_set_timestamp (context, GDK_CURRENT_TIME);
532 gdk_app_launch_context_set_screen (context,
533 gtk_window_get_screen (parent_window));
534 g_desktop_app_info_launch_uris_as_manager (app_info,
535 (GList *) parameter_uris,
536 G_APP_LAUNCH_CONTEXT (context),
537 G_SPAWN_SEARCH_PATH | G_SPAWN_DO_NOT_REAP_CHILD,
538 NULL, NULL,
539 gather_pid_callback, app_info,
540 &error);
541 if (error != NULL)
542 {
543 char *message;
544
545 message = g_strconcat (_("Details: "), error->message, NULL);
546 eel_show_error_dialog
547 (_("There was an error launching the application."),
548 message,
549 parent_window);
550
551 g_error_free (error);
552 g_free (message);
553 }
554
555 g_list_free_full (files, g_object_unref);
556 g_object_unref (context);
557 g_object_unref (app_info);
558}
