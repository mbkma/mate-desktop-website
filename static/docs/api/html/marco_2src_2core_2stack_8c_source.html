106void
107meta_stack_add (MetaStack *stack,
108 MetaWindow *window)
109{
110 meta_topic (META_DEBUG_STACK, "Adding window %s to the stack\n", window->desc);
111
112 if (window->stack_position >= 0)
113 meta_bug ("Window %s had stack position already\n", window->desc);
114
115 stack->added = g_list_prepend (stack->added, window);
116
117 window->stack_position = stack->n_positions;
118 stack->n_positions += 1;
119 meta_topic (META_DEBUG_STACK,
120 "Window %s has stack_position initialized to %d\n",
121 window->desc, window->stack_position);
122
123 stack_sync_to_server (stack);
124}

126void
127meta_stack_remove (MetaStack *stack,
128 MetaWindow *window)
129{
130 meta_topic (META_DEBUG_STACK, "Removing window %s from the stack\n", window->desc);
131
132 if (window->stack_position < 0)
133 meta_bug ("Window %s removed from stack but had no stack position\n",
134 window->desc);
135
136 /* Set window to top position, so removing it will not leave gaps
137 * in the set of positions
138 */
139 meta_window_set_stack_position_no_sync (window,
140 stack->n_positions - 1);
141 window->stack_position = -1;
142 stack->n_positions -= 1;
143
144 /* We don't know if it's been moved from "added" to "stack" yet */
145 stack->added = g_list_remove (stack->added, window);
146 stack->sorted = g_list_remove (stack->sorted, window);
147
148 /* Remember the window ID to remove it from the stack array.
149 * The macro is safe to use: Window is guaranteed to be 32 bits, and
150 * GUINT_TO_POINTER says it only works on 32 bits.
151 */
152 stack->removed = g_list_prepend (stack->removed,
153 GUINT_TO_POINTER (window->xwindow));
154 if (window->frame)
155 stack->removed = g_list_prepend (stack->removed,
156 GUINT_TO_POINTER (window->frame->xwindow));
157
158 stack_sync_to_server (stack);
159}

179/* raise/lower within a layer */
180void
181meta_stack_raise (MetaStack *stack,
182 MetaWindow *window)
183{
184 GList *l;
185 int max_stack_position = window->stack_position;
186 MetaWorkspace *workspace;
187
188 g_assert (stack->added == NULL);
189
190 workspace = meta_window_get_workspace (window);
191 for (l = stack->sorted; l; l = l->next)
192 {
193 MetaWindow *w = (MetaWindow *) l->data;
194 if (meta_window_located_on_workspace (w, workspace) &&
195 w->stack_position > max_stack_position)
196 max_stack_position = w->stack_position;
197 }
198
199 if (max_stack_position == window->stack_position)
200 return;
201
202 meta_window_set_stack_position_no_sync (window, max_stack_position);
203
204 stack_sync_to_server (stack);
205}

272/* Get layer ignoring any transient or group relationships */
273static MetaStackLayer
274get_standalone_layer (MetaWindow *window)
275{
276 MetaStackLayer layer;
277 gboolean focused_transient = FALSE;
278
279 switch (window->type)
280 {
281 case META_WINDOW_DESKTOP:
282 layer = META_LAYER_DESKTOP;
283 break;
284
285 case META_WINDOW_DOCK:
286 /* still experimenting here */
287 if (window->wm_state_below)
288 layer = META_LAYER_BOTTOM;
289 else
290 layer = META_LAYER_DOCK;
291 break;
292
293 default:
294 meta_window_foreach_transient (window,
295 is_focused_foreach,
296 &focused_transient);
297
298 if (window->wm_state_below)
299 layer = META_LAYER_BOTTOM;
300 else if (window->fullscreen &&
301 (focused_transient ||
302 window == window->display->expected_focus_window ||
303 window->display->expected_focus_window == NULL ||
304 (window->display->expected_focus_window != NULL &&
305 windows_on_different_xinerama (window,
306 window->display->expected_focus_window))))
307 layer = META_LAYER_FULLSCREEN;
308 else if (window->wm_state_above)
309 layer = META_LAYER_TOP;
310 else
311 layer = META_LAYER_NORMAL;
312 break;
313 }
314
315 return layer;
316}

318/* Note that this function can never use window->layer only
319 * get_standalone_layer, or we'd have issues.
320 */
321static MetaStackLayer
322get_maximum_layer_in_group (MetaWindow *window)
323{
324 GSList *members;
325 MetaGroup *group;
326 GSList *tmp;
327 MetaStackLayer max;
328 MetaStackLayer layer;
329
330 max = META_LAYER_DESKTOP;
331
332 group = meta_window_get_group (window);
333
334 if (group != NULL)
335 members = meta_group_list_windows (group);
336 else
337 members = NULL;
338
339 tmp = members;
340 while (tmp != NULL)
341 {
342 MetaWindow *w = tmp->data;
343
344 layer = get_standalone_layer (w);
345 if (layer > max)
346 max = layer;
347
348 tmp = tmp->next;
349 }
350
351 g_slist_free (members);
352
353 return max;
354}

356static void
357compute_layer (MetaWindow *window)
358{
359 window->layer = get_standalone_layer (window);
360
361 /* We can only do promotion-due-to-group for dialogs and other
362 * transients, or weird stuff happens like the desktop window and
363 * caja windows getting in the same layer, or all mate-terminal
364 * windows getting in fullscreen layer if any terminal is
365 * fullscreen.
366 */
367 if (WINDOW_HAS_TRANSIENT_TYPE(window) &&
368 (window->xtransient_for == None ||
369 window->transient_parent_is_root_window))
370 {
371 /* We only do the group thing if the dialog is NOT transient for
372 * a particular window. Imagine a group with a normal window, a dock,
373 * and a dialog transient for the normal window; you don't want the dialog
374 * above the dock if it wouldn't normally be.
375 */
376
377 MetaStackLayer group_max;
378
379 group_max = get_maximum_layer_in_group (window);
380
381 if (group_max > window->layer)
382 {
383 meta_topic (META_DEBUG_STACK,
384 "Promoting window %s from layer %u to %u due to group membership\n",
385 window->desc, window->layer, group_max);
386 window->layer = group_max;
387 }
388 }
389
390 meta_topic (META_DEBUG_STACK, "Window %s on layer %u type = %u has_focus = %d\n",
391 window->desc, window->layer,
392 window->type, window->has_focus);
393}

395/* Front of the layer list is the topmost window,
396 * so the lower stack position is later in the list
397 */
398static int
399compare_window_position (void *a,
400 void *b)
401{
402 MetaWindow *window_a = a;
403 MetaWindow *window_b = b;
404
405 /* Go by layer, then stack_position */
406 if (window_a->layer < window_b->layer)
407 return 1; /* move window_a later in list */
408 else if (window_a->layer > window_b->layer)
409 return -1;
410 else if (window_a->stack_position < window_b->stack_position)
411 return 1; /* move window_a later in list */
412 else if (window_a->stack_position > window_b->stack_position)
413 return -1;
414 else
415 return 0; /* not reached */
416}

418/*
419 * Stacking constraints
420 *
421 * Assume constraints of the form "AB" meaning "window A must be
422 * below window B"
423 *
424 * If we have windows stacked from bottom to top
425 * "ABC" then raise A we get "BCA". Say C is
426 * transient for B is transient for A. So
427 * we have constraints AB and BC.
428 *
429 * After raising A, we need to reapply the constraints.
430 * If we do this by raising one window at a time -
431 *
432 * start: BCA
433 * apply AB: CAB
434 * apply BC: ABC
435 *
436 * but apply constraints in the wrong order and it breaks:
437 *
438 * start: BCA
439 * apply BC: BCA
440 * apply AB: CAB
441 *
442 * We make a directed graph of the constraints by linking
443 * from "above windows" to "below windows as follows:
444 *
445 * AB -> BC -> CD
446 * \
447 * CE
448 *
449 * If we then walk that graph and apply the constraints in the order
450 * that they appear, we will apply them correctly. Note that the
451 * graph MAY have cycles, so we have to guard against that.
452 *
453 */
454
455typedef struct Constraint Constraint;
456
457struct Constraint
458{
459 MetaWindow *above;
460 MetaWindow *below;
461
462 /* used to keep the constraint in the
463 * list of constraints for window "below"
464 */
465 Constraint *next;
466
467 /* used to create the graph. */
468 GSList *next_nodes;
469
470 /* constraint has been applied, used
471 * to detect cycles.
472 */
473 unsigned int applied : 1;
474
475 /* constraint has a previous node in the graph,
476 * used to find places to start in the graph.
477 * (I think this also has the side effect
478 * of preventing cycles, since cycles will
479 * have no starting point - so maybe
480 * the "applied" flag isn't needed.)
481 */
482 unsigned int has_prev : 1;
483};
484
485/* We index the array of constraints by window
486 * stack positions, just because the stack
487 * positions are a convenient index.
488 */
489static void
490add_constraint (Constraint **constraints,
491 MetaWindow *above,
492 MetaWindow *below)
493{
494 Constraint *c;
495
496 g_assert (above->screen == below->screen);
497
498 /* check if constraint is a duplicate */
499 c = constraints[below->stack_position];
500 while (c != NULL)
501 {
502 if (c->above == above)
503 return;
504 c = c->next;
505 }
506
507 /* if not, add the constraint */
508 c = g_new (Constraint, 1);
509 c->above = above;
510 c->below = below;
511 c->next = constraints[below->stack_position];
512 c->next_nodes = NULL;
513 c->applied = FALSE;
514 c->has_prev = FALSE;
515
516 constraints[below->stack_position] = c;
517}

519static void
520create_constraints (Constraint **constraints,
521 GList *windows)
522{
523 GList *tmp;
524
525 tmp = windows;
526 while (tmp != NULL)
527 {
528 MetaWindow *w = tmp->data;
529
530 if (!WINDOW_IN_STACK (w))
531 {
532 meta_topic (META_DEBUG_STACK, "Window %s not in the stack, not constraining it\n",
533 w->desc);
534 tmp = tmp->next;
535 continue;
536 }
537
538 if (WINDOW_TRANSIENT_FOR_WHOLE_GROUP (w))
539 {
540 GSList *group_windows;
541 GSList *tmp2;
542 MetaGroup *group;
543
544 group = meta_window_get_group (w);
545
546 if (group != NULL)
547 group_windows = meta_group_list_windows (group);
548 else
549 group_windows = NULL;
550
551 tmp2 = group_windows;
552
553 while (tmp2 != NULL)
554 {
555 MetaWindow *group_window = tmp2->data;
556
557 if (!WINDOW_IN_STACK (group_window) ||
558 w->screen != group_window->screen)
559 {
560 tmp2 = tmp2->next;
561 continue;
562 }
563
564#if 0
565 /* old way of doing it */
566 if (!(meta_window_is_ancestor_of_transient (w, group_window)) &&
567 !WINDOW_TRANSIENT_FOR_WHOLE_GROUP (group_window)) /* note */;/*note*/
568#else
569 /* better way I think, so transient-for-group are constrained
570 * only above non-transient-type windows in their group
571 */
572 if (!WINDOW_HAS_TRANSIENT_TYPE (group_window))
573#endif
574 {
575 meta_topic (META_DEBUG_STACK, "Constraining %s above %s as it's transient for its group\n",
576 w->desc, group_window->desc);
577 add_constraint (constraints, w, group_window);
578 }
579
580 tmp2 = tmp2->next;
581 }
582
583 g_slist_free (group_windows);
584 }
585 else if (w->xtransient_for != None &&
586 !w->transient_parent_is_root_window)
587 {
588 MetaWindow *parent;
589
590 parent =
591 meta_display_lookup_x_window (w->display, w->xtransient_for);
592
593 if (parent && WINDOW_IN_STACK (parent) &&
594 parent->screen == w->screen)
595 {
596 meta_topic (META_DEBUG_STACK, "Constraining %s above %s due to transiency\n",
597 w->desc, parent->desc);
598 add_constraint (constraints, w, parent);
599 }
600 }
601
602 tmp = tmp->next;
603 }
604}

606static void
607graph_constraints (Constraint **constraints,
608 int n_constraints)
609{
610 int i;
611
612 i = 0;
613 while (i < n_constraints)
614 {
615 Constraint *c;
616
617 /* If we have "A below B" and "B below C" then AB -> BC so we
618 * add BC to next_nodes in AB.
619 */
620
621 c = constraints[i];
622 while (c != NULL)
623 {
624 Constraint *n;
625
626 g_assert (c->below->stack_position == i);
627
628 /* Constraints where ->above is below are our
629 * next_nodes and we are their previous
630 */
631 n = constraints[c->above->stack_position];
632 while (n != NULL)
633 {
634 c->next_nodes = g_slist_prepend (c->next_nodes,
635 n);
636 /* c is a previous node of n */
637 n->has_prev = TRUE;
638
639 n = n->next;
640 }
641
642 c = c->next;
643 }
644
645 ++i;
646 }
647}

652{
653 int i;
654
655 i = 0;

676static void
677ensure_above (MetaWindow *above,
678 MetaWindow *below)
679{
680 if (WINDOW_HAS_TRANSIENT_TYPE(above) &&
681 above->layer < below->layer)
682 {
683 meta_topic (META_DEBUG_STACK,
684 "Promoting window %s from layer %u to %u due to contraint\n",
685 above->desc, above->layer, below->layer);
686 above->layer = below->layer;
687 }
688
689 if (above->stack_position < below->stack_position)
690 {
691 /* move above to below->stack_position bumping below down the stack */
692 meta_window_set_stack_position_no_sync (above, below->stack_position);
693 g_assert (below->stack_position + 1 == above->stack_position);
694 }
695 meta_topic (META_DEBUG_STACK, "%s above at %d > %s below at %d\n",
696 above->desc, above->stack_position,
697 below->desc, below->stack_position);
698}

700static void
701traverse_constraint (Constraint *c)
702{
703 GSList *tmp;
704
705 if (c->applied)
706 return;
707
708 ensure_above (c->above, c->below);
709 c->applied = TRUE;
710
711 tmp = c->next_nodes;
712 while (tmp != NULL)
713 {
714 traverse_constraint (tmp->data);
715
716 tmp = tmp->next;
717 }
718}

720static void
721apply_constraints (Constraint **constraints,
722 int n_constraints)
723{
724 GSList *heads;
725 GSList *tmp;
726 int i;
727
728 /* List all heads in an ordered constraint chain */
729 heads = NULL;
730 i = 0;
731 while (i < n_constraints)
732 {
733 Constraint *c;
734
735 c = constraints[i];
736 while (c != NULL)
737 {
738 if (!c->has_prev)
739 heads = g_slist_prepend (heads, c);
740
741 c = c->next;
742 }
743
744 ++i;
745 }
746
747 /* Now traverse the chain and apply constraints */
748 tmp = heads;
749 while (tmp != NULL)
750 {
751 Constraint *c = tmp->data;
752
753 traverse_constraint (c);
754
755 tmp = tmp->next;
756 }
757
758 g_slist_free (heads);
759}

765static void
766stack_do_window_deletions (MetaStack *stack)
767{
768 /* Do removals before adds, with paranoid idea that we might re-add
769 * the same window IDs.
770 */
771 GList *tmp;
772 int i;
773
774 tmp = stack->removed;
775 while (tmp != NULL)
776 {
777 Window xwindow;
778 xwindow = GPOINTER_TO_UINT (tmp->data);
779
780 /* We go from the end figuring removals are more
781 * likely to be recent.
782 */
783 i = stack->windows->len;
784 while (i > 0)
785 {
786 --i;
787
788 /* there's no guarantee we'll actually find windows to
789 * remove, e.g. the same xwindow could have been
790 * added/removed before we ever synced, and we put
791 * both the window->xwindow and window->frame->xwindow
792 * in the removal list.
793 */
794 if (xwindow == g_array_index (stack->windows, Window, i))
795 {
796 g_array_remove_index (stack->windows, i);
797 goto next;
798 }
799 }
800
801 next:
802 tmp = tmp->next;
803 }
804
805 g_list_free (stack->removed);
806 stack->removed = NULL;
807}

809static void
810stack_do_window_additions (MetaStack *stack)
811{
812 GList *tmp;
813 gint i, n_added;
814
815 n_added = g_list_length (stack->added);
816 if (n_added > 0)
817 {
818 Window *end;
819 int old_size;
820
821 meta_topic (META_DEBUG_STACK,
822 "Adding %d windows to sorted list\n",
823 n_added);
824
825 old_size = stack->windows->len;
826 g_array_set_size (stack->windows, old_size + n_added);
827
828 end = &g_array_index (stack->windows, Window, old_size);
829
830 /* stack->added has the most recent additions at the
831 * front of the list, so we need to reverse it
832 */
833 stack->added = g_list_reverse (stack->added);
834
835 i = 0;
836 tmp = stack->added;
837 while (tmp != NULL)
838 {
839 MetaWindow *w;
840
841 w = tmp->data;
842
843 end[i] = w->xwindow;
844
845 /* add to the main list */
846 stack->sorted = g_list_prepend (stack->sorted, w);
847
848 ++i;
849 tmp = tmp->next;
850 }
851
852 stack->need_resort = TRUE; /* may not be needed as we add to top */
853 stack->need_constrain = TRUE;
854 stack->need_relayer = TRUE;
855 }
856
857 g_list_free (stack->added);
858 stack->added = NULL;
859}

864static void
865stack_do_relayer (MetaStack *stack)
866{
867 GList *tmp;
868
869 if (!stack->need_relayer)
870 return;
871
872 meta_topic (META_DEBUG_STACK,
873 "Recomputing layers\n");
874
875 tmp = stack->sorted;
876
877 while (tmp != NULL)
878 {
879 MetaWindow *w;
880 MetaStackLayer old_layer;
881
882 w = tmp->data;
883 old_layer = w->layer;
884
885 compute_layer (w);
886
887 if (w->layer != old_layer)
888 {
889 meta_topic (META_DEBUG_STACK,
890 "Window %s moved from layer %u to %u\n",
891 w->desc, old_layer, w->layer);
892
893 stack->need_resort = TRUE;
894 stack->need_constrain = TRUE;
895 /* don't need to constrain as constraining
896 * purely operates in terms of stack_position
897 * not layer
898 */
899 }
900
901 tmp = tmp->next;
902 }
903
904 stack->need_relayer = FALSE;
905}

913{

915
916 /* It'd be nice if this were all faster, probably */
917

987static void
988raise_window_relative_to_managed_windows (MetaScreen *screen,
989 Window xwindow)
990{
991
992 Window ignored1, ignored2;
993 Window *children;
994 unsigned int n_children;
995 int i;
996
997 /* Normally XQueryTree() means "must grab server" but here
998 * we don't, since we know we won't manage any new windows
999 * or restack any windows before using the XQueryTree results.
1000 */
1001
1002 meta_error_trap_push (screen->display);
1003
1004 XQueryTree (screen->display->xdisplay,
1005 screen->xroot,
1006 &ignored1, &ignored2, &children, &n_children);
1007
1008 if (meta_error_trap_pop_with_return (screen->display, TRUE) != Success)
1009 {
1010 meta_topic (META_DEBUG_STACK,
1011 "Error querying root children to raise window 0x%lx\n",
1012 xwindow);
1013 return;
1014 }
1015
1016 /* Children are in order from bottom to top. We want to
1017 * find the topmost managed child, then configure
1018 * our window to be above it.
1019 */
1020 i = n_children - 1;
1021 while (i >= 0)
1022 {
1023 if (children[i] == xwindow)
1024 {
1025 /* Do nothing. This means we're already the topmost managed
1026 * window, but it DOES NOT mean we are already just above
1027 * the topmost managed window. This is important because if
1028 * an override redirect window is up, and we map a new
1029 * managed window, the new window is probably above the old
1030 * popup by default, and we want to push it below that
1031 * popup. So keep looking for a sibling managed window
1032 * to be moved below.
1033 */
1034 }
1035 else if (meta_display_lookup_x_window (screen->display,
1036 children[i]) != NULL)
1037 {
1038 XWindowChanges changes;
1039
1040 /* children[i] is the topmost managed child */
1041 meta_topic (META_DEBUG_STACK,
1042 "Moving 0x%lx above topmost managed child window 0x%lx\n",
1043 xwindow, children[i]);
1044
1045 changes.sibling = children[i];
1046 changes.stack_mode = Above;
1047
1048 meta_error_trap_push (screen->display);
1049 XConfigureWindow (screen->display->xdisplay,
1050 xwindow,
1051 CWSibling | CWStackMode,
1052 &changes);
1053 meta_error_trap_pop (screen->display, FALSE);
1054
1055 break;
1056 }
1057
1058 --i;
1059 }
1060
1061 if (i < 0)
1062 {
1063 /* No sibling to use, just lower ourselves to the bottom
1064 * to be sure we're below any override redirect windows.
1065 */
1066 meta_error_trap_push (screen->display);
1067 XLowerWindow (screen->display->xdisplay,
1068 xwindow);
1069 meta_error_trap_pop (screen->display, FALSE);
1070 }
1071
1072 if (children)
1073 XFree (children);
1074}

1083static void
1084stack_sync_to_server (MetaStack *stack)
1085{
1086 GArray *stacked;
1087 GArray *root_children_stacked;
1088 GList *tmp;
1089
1090 /* Bail out if frozen */
1091 if (stack->freeze_count > 0)
1092 return;
1093
1094 meta_topic (META_DEBUG_STACK, "Syncing window stack to server\n");
1095
1096 stack_ensure_sorted (stack);
1097
1098 /* Create stacked xwindow arrays.
1099 * Painfully, "stacked" is in bottom-to-top order for the
1100 * _NET hints, and "root_children_stacked" is in top-to-bottom
1101 * order for XRestackWindows()
1102 */
1103 stacked = g_array_new (FALSE, FALSE, sizeof (Window));
1104 root_children_stacked = g_array_new (FALSE, FALSE, sizeof (Window));
1105
1106 meta_topic (META_DEBUG_STACK, "Top to bottom: ");
1107 meta_push_no_msg_prefix ();
1108
1109 tmp = stack->sorted;
1110 while (tmp != NULL)
1111 {
1112 MetaWindow *w;
1113
1114 w = tmp->data;
1115
1116 /* remember, stacked is in reverse order (bottom to top) */
1117 g_array_prepend_val (stacked, w->xwindow);
1118
1119 /* build XRestackWindows() array from top to bottom */
1120 if (w->frame)
1121 g_array_append_val (root_children_stacked, w->frame->xwindow);
1122 else
1123 g_array_append_val (root_children_stacked, w->xwindow);
1124
1125 meta_topic (META_DEBUG_STACK, "%u:%d - %s ", w->layer, w->stack_position, w->desc);
1126
1127 tmp = tmp->next;
1128 }
1129
1130 meta_topic (META_DEBUG_STACK, "\n");
1131 meta_pop_no_msg_prefix ();
1132
1133 /* All windows should be in some stacking order */
1134 if (stacked->len != stack->windows->len)
1135 meta_bug ("%u windows stacked, %u windows exist in stack\n",
1136 stacked->len, stack->windows->len);
1137
1138 /* Sync to server */
1139
1140 meta_topic (META_DEBUG_STACK, "Restacking %u windows\n",
1141 root_children_stacked->len);
1142
1143 meta_error_trap_push (stack->screen->display);
1144
1145 if (stack->last_root_children_stacked == NULL)
1146 {
1147 /* Just impose our stack, we don't know the previous state.
1148 * This involves a ton of circulate requests and may flicker.
1149 */
1150 meta_topic (META_DEBUG_STACK, "Don't know last stack state, restacking everything\n");
1151
1152 if (root_children_stacked->len > 0)
1153 XRestackWindows (stack->screen->display->xdisplay,
1154 (Window *) root_children_stacked->data,
1155 root_children_stacked->len);
1156 }
1157 else if (root_children_stacked->len > 0)
1158 {
1159 /* Try to do minimal window moves to get the stack in order */
1160 /* A point of note: these arrays include frames not client windows,
1161 * so if a client window has changed frame since last_root_children_stacked
1162 * was saved, then we may have inefficiency, but I don't think things
1163 * break...
1164 */
1165 const Window *old_stack = (Window *) stack->last_root_children_stacked->data;
1166 const Window *new_stack = (Window *) root_children_stacked->data;
1167 const int old_len = stack->last_root_children_stacked->len;
1168 const int new_len = root_children_stacked->len;
1169 const Window *oldp = old_stack;
1170 const Window *newp = new_stack;
1171 const Window *old_end = old_stack + old_len;
1172 const Window *new_end = new_stack + new_len;
1173 Window last_window = None;
1174
1175 while (oldp != old_end &&
1176 newp != new_end)
1177 {
1178 if (*oldp == *newp)
1179 {
1180 /* Stacks are the same here, move on */
1181 ++oldp;
1182 last_window = *newp;
1183 ++newp;
1184 }
1185 else if (meta_display_lookup_x_window (stack->screen->display,
1186 *oldp) == NULL)
1187 {
1188 /* *oldp is no longer known to us (probably destroyed),
1189 * so we can just skip it
1190 */
1191 ++oldp;
1192 }
1193 else
1194 {
1195 /* Move *newp below last_window */
1196 if (last_window == None)
1197 {
1198 meta_topic (META_DEBUG_STACK, "Using window 0x%lx as topmost (but leaving it in-place)\n", *newp);
1199
1200 raise_window_relative_to_managed_windows (stack->screen,
1201 *newp);
1202 }
1203 else
1204 {
1205 /* This means that if last_window is dead, but not
1206 * *newp, then we fail to restack *newp; but on
1207 * unmanaging last_window, we'll fix it up.
1208 */
1209
1210 XWindowChanges changes;
1211
1212 changes.sibling = last_window;
1213 changes.stack_mode = Below;
1214
1215 meta_topic (META_DEBUG_STACK, "Placing window 0x%lx below 0x%lx\n",
1216 *newp, last_window);
1217
1218 XConfigureWindow (stack->screen->display->xdisplay,
1219 *newp,
1220 CWSibling | CWStackMode,
1221 &changes);
1222 }
1223
1224 last_window = *newp;
1225 ++newp;
1226 }
1227 }
1228
1229 if (newp != new_end)
1230 {
1231 /* Restack remaining windows */
1232 meta_topic (META_DEBUG_STACK, "Restacking remaining %d windows\n",
1233 (int) (new_end - newp));
1234 /* We need to include an already-stacked window
1235 * in the restack call, so we get in the proper position
1236 * with respect to it.
1237 */
1238 if (newp != new_stack)
1239 --newp;
1240 XRestackWindows (stack->screen->display->xdisplay,
1241 (Window *) newp, new_end - newp);
1242 }
1243 }
1244
1245 meta_error_trap_pop (stack->screen->display, FALSE);
1246 /* on error, a window was destroyed; it should eventually
1247 * get removed from the stacking list when we unmanage it
1248 * and we'll fix stacking at that time.
1249 */
1250
1251 /* Sync _NET_CLIENT_LIST and _NET_CLIENT_LIST_STACKING */
1252
1253 XChangeProperty (stack->screen->display->xdisplay,
1254 stack->screen->xroot,
1255 stack->screen->display->atom__NET_CLIENT_LIST,
1256 XA_WINDOW,
1257 32, PropModeReplace,
1258 (unsigned char *)stack->windows->data,
1259 stack->windows->len);
1260 XChangeProperty (stack->screen->display->xdisplay,
1261 stack->screen->xroot,
1262 stack->screen->display->atom__NET_CLIENT_LIST_STACKING,
1263 XA_WINDOW,
1264 32, PropModeReplace,
1265 (unsigned char *)stacked->data,
1266 stacked->len);
1267
1268 g_array_free (stacked, TRUE);
1269
1270 if (stack->last_root_children_stacked)
1271 g_array_free (stack->last_root_children_stacked, TRUE);
1272 stack->last_root_children_stacked = root_children_stacked;
1273
1274 /* That was scary... */
1275}

1365static MetaWindow*
1366get_default_focus_window (MetaStack *stack,
1367 MetaWorkspace *workspace,
1368 MetaWindow *not_this_one,
1369 gboolean must_be_at_point,
1370 int root_x,
1371 int root_y)
1372{
1373 /* Find the topmost, focusable, mapped, window.
1374 * not_this_one is being unfocused or going away, so exclude it.
1375 * Also, prefer to focus transient parent of not_this_one,
1376 * or top window in same group as not_this_one.
1377 */
1378
1379 MetaWindow *topmost_dock;
1380 MetaWindow *transient_parent;
1381 MetaWindow *topmost_in_group;
1382 MetaWindow *topmost_overall;
1383 MetaGroup *not_this_one_group;
1384 GList *link;
1385
1386 topmost_dock = NULL;
1387 transient_parent = NULL;
1388 topmost_in_group = NULL;
1389 topmost_overall = NULL;
1390 if (not_this_one)
1391 not_this_one_group = meta_window_get_group (not_this_one);
1392 else
1393 not_this_one_group = NULL;
1394
1395 stack_ensure_sorted (stack);
1396
1397 /* top of this layer is at the front of the list */
1398 link = stack->sorted;
1399
1400 while (link)
1401 {
1402 MetaWindow *window = link->data;
1403
1404 if (window &&
1405 window != not_this_one &&
1406 (window->unmaps_pending == 0) &&
1407 !window->minimized &&
1408 (window->input || window->take_focus) &&
1409 (workspace == NULL ||
1410 meta_window_located_on_workspace (window, workspace)))
1411 {
1412 if (topmost_dock == NULL &&
1413 window->type == META_WINDOW_DOCK)
1414 topmost_dock = window;
1415
1416 if (not_this_one != NULL)
1417 {
1418 if (transient_parent == NULL &&
1419 not_this_one->xtransient_for != None &&
1420 not_this_one->xtransient_for == window->xwindow &&
1421 (!must_be_at_point ||
1422 window_contains_point (window, root_x, root_y)))
1423 transient_parent = window;
1424
1425 if (topmost_in_group == NULL &&
1426 not_this_one_group != NULL &&
1427 not_this_one_group == meta_window_get_group (window) &&
1428 (!must_be_at_point ||
1429 window_contains_point (window, root_x, root_y)))
1430 topmost_in_group = window;
1431 }
1432
1433 /* Note that DESKTOP windows can be topmost_overall so
1434 * we prefer focusing desktop or other windows over
1435 * focusing dock, even though docks are stacked higher.
1436 */
1437 if (topmost_overall == NULL &&
1438 window->type != META_WINDOW_DOCK &&
1439 (!must_be_at_point ||
1440 window_contains_point (window, root_x, root_y)))
1441 topmost_overall = window;
1442
1443 /* We could try to bail out early here for efficiency in
1444 * some cases, but it's just not worth the code.
1445 */
1446 }
1447
1448 link = link->next;
1449 }
1450
1451 if (transient_parent)
1452 return transient_parent;
1453 else if (topmost_in_group)
1454 return topmost_in_group;
1455 else if (topmost_overall)
1456 return topmost_overall;
1457 else
1458 return topmost_dock;
1459}

1573static gboolean
1574lists_contain_same_windows (GList *a,
1575 GList *b)
1576{
1577 GList *copy1, *copy2;
1578 GList *tmp1, *tmp2;
1579
1580 if (g_list_length (a) != g_list_length (b))
1581 return FALSE;
1582
1583 tmp1 = copy1 = g_list_sort (g_list_copy (a), compare_pointers);
1584 tmp2 = copy2 = g_list_sort (g_list_copy (b), compare_pointers);
1585
1586 while (tmp1 && tmp1->data == tmp2->data) /* tmp2 is non-NULL if tmp1 is */
1587 {
1588 tmp1 = tmp1->next;
1589 tmp2 = tmp2->next;
1590 }
1591
1592 g_list_free (copy1);
1593 g_list_free (copy2);
1594
1595 return (tmp1 == NULL); /* tmp2 is non-NULL if tmp1 is */
1596}

1598void
1599meta_stack_set_positions (MetaStack *stack,
1600 GList *windows)
1601{
1602 int i;
1603 GList *tmp;
1604
1605 /* Make sure any adds or removes aren't in limbo -- is this needed? */
1606 stack_ensure_sorted (stack);
1607
1608 if (!lists_contain_same_windows (windows, stack->sorted))
1609 {
1610 meta_warning ("This list of windows has somehow changed; not resetting "
1611 "positions of the windows.\n");
1612 return;
1613 }
1614
1615 g_list_free (stack->sorted);
1616 stack->sorted = g_list_copy (windows);
1617
1618 stack->need_resort = TRUE;
1619 stack->need_constrain = TRUE;
1620
1621 i = 0;
1622 tmp = windows;
1623 while (tmp != NULL)
1624 {
1625 MetaWindow *w = tmp->data;
1626 w->stack_position = i++;
1627 tmp = tmp->next;
1628 }
1629
1630 meta_topic (META_DEBUG_STACK,
1631 "Reset the stack positions of (nearly) all windows\n");
1632
1633 stack_sync_to_server (stack);
1634}

1636void
1637meta_window_set_stack_position_no_sync (MetaWindow *window,
1638 int position)
1639{
1640 int low, high, delta;
1641 GList *tmp;
1642
1643 g_return_if_fail (window->screen->stack != NULL);
1644 g_return_if_fail (window->stack_position >= 0);
1645 g_return_if_fail (position >= 0);
1646 g_return_if_fail (position < window->screen->stack->n_positions);
1647
1648 if (position == window->stack_position)
1649 {
1650 meta_topic (META_DEBUG_STACK, "Window %s already has position %d\n",
1651 window->desc, position);
1652 return;
1653 }
1654
1655 window->screen->stack->need_resort = TRUE;
1656 window->screen->stack->need_constrain = TRUE;
1657
1658 if (position < window->stack_position)
1659 {
1660 low = position;
1661 high = window->stack_position - 1;
1662 delta = 1;
1663 }
1664 else
1665 {
1666 low = window->stack_position + 1;
1667 high = position;
1668 delta = -1;
1669 }
1670
1671 tmp = window->screen->stack->sorted;
1672 while (tmp != NULL)
1673 {
1674 MetaWindow *w = tmp->data;
1675
1676 if (w->stack_position >= low &&
1677 w->stack_position <= high)
1678 w->stack_position += delta;
1679
1680 tmp = tmp->next;
1681 }
1682
1683 window->stack_position = position;
1684
1685 meta_topic (META_DEBUG_STACK,
1686 "Window %s had stack_position set to %d\n",
1687 window->desc, window->stack_position);
1688}
