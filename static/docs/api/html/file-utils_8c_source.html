65static gboolean
66uri_is_filetype (const char *uri,
67 GFileType file_type)
68{
69 gboolean result = FALSE;
70 GFile *file;
71 GFileInfo *info;
72 GError *error = NULL;
73
74 file = g_file_new_for_uri (uri);
75
76 if (! g_file_query_exists (file, NULL)) {
77 g_object_unref (file);
78 return FALSE;

80
81 info = g_file_query_info (file, G_FILE_ATTRIBUTE_STANDARD_TYPE, 0, NULL, &error);

86 g_warning ("Failed to get file type for uri %s: %s", uri, error->message);
87 g_error_free (error);
88 }
89
90 g_object_unref (info);
91 g_object_unref (file);
92
93 return result;

127char *
128get_dir_content_if_unique (const char *uri)
129{
130 GFile *file;
131 GFileEnumerator *file_enum;
132 GFileInfo *info;
133 GError *err = NULL;
134 char *content_uri = NULL;
135
136 file = g_file_new_for_uri (uri);
137
138 if (! g_file_query_exists (file, NULL)) {
139 g_object_unref (file);
140 return NULL;
141 }
142
143 file_enum = g_file_enumerate_children (file, G_FILE_ATTRIBUTE_STANDARD_NAME, 0, NULL, &err);
144 if (err != NULL) {
145 g_warning ("Failed to enumerate children of %s: %s", uri, err->message);
146 g_error_free (err);
147 return NULL;
148 }
149
150 while ((info = g_file_enumerator_next_file (file_enum, NULL, &err)) != NULL) {
151 const char *name;
152 char *new_name;
153
154 if (err != NULL) {
155 g_warning ("Failed to get info while enumerating children: %s", err->message);
156 g_clear_error (&err);
157 g_object_unref (info);
158 continue;
159 }
160
161 name = g_file_info_get_name (info);
162 if ((strcmp (name, ".") == 0) || (strcmp (name, "..") == 0)) {
163 g_object_unref (info);
164 continue;
165 }
166
167 if (content_uri != NULL) {
168 g_free (content_uri);
169 g_object_unref (info);
170 content_uri = NULL;
171 break;
172 }
173
174 new_name = g_uri_escape_string (name, NULL, TRUE);
175 content_uri = build_uri (uri, new_name, NULL);
176 g_object_unref (info);
177 g_free (new_name);
178 }
179
180 if (err != NULL) {
181 g_warning ("Failed to get info after enumerating children: %s", err->message);
182 g_clear_error (&err);
183 }
184
185 g_object_unref (file_enum);
186 g_object_unref (file);
187
188 return content_uri;
189}

191/* Check whether the dirname is contained in filename */
192gboolean
193path_in_path (const char *dirname,
194 const char *filename)
195{
196 int dirname_l, filename_l, separator_position;
197
198 if ((dirname == NULL) || (filename == NULL))
199 return FALSE;
200
201 dirname_l = strlen (dirname);
202 filename_l = strlen (filename);
203
204 if ((dirname_l == filename_l + 1)
205 && (dirname[dirname_l - 1] == '/'))
206 return FALSE;
207
208 if ((filename_l == dirname_l + 1)
209 && (filename[filename_l - 1] == '/'))
210 return FALSE;
211
212 if (dirname[dirname_l - 1] == '/')
213 separator_position = dirname_l - 1;
214 else
215 separator_position = dirname_l;
216
217 return ((filename_l > dirname_l)
218 && (strncmp (dirname, filename, dirname_l) == 0)
219 && (filename[separator_position] == '/'));
220}

310/* like g_path_get_basename but does not warn about NULL and does not
311 * alloc a new string. */
312const gchar* file_name_from_path(const gchar *file_name)
313{
314 register char *base;
315 register gssize last_char;
316
317 if (file_name == NULL)
318 return NULL;
319
320 if ((file_name[0] == '\0') || (strlen (file_name) == 0))
321 return "";
322
323 last_char = strlen (file_name) - 1;
324
325 if (file_name [last_char] == G_DIR_SEPARATOR)
326 return "";
327
328 base = g_utf8_strrchr (file_name, -1, G_DIR_SEPARATOR);
329 if (! base)
330 return file_name;
331
332 return base + 1;
333}

335char *
336dir_name_from_path (const gchar *path)
337{
338 register gssize base;
339 register gssize last_char;
340
341 if (path == NULL)
342 return NULL;
343
344 if (path[0] == '\0')
345 return g_strdup ("");
346
347 last_char = strlen (path) - 1;
348 if (path[last_char] == G_DIR_SEPARATOR)
349 last_char--;
350
351 base = last_char;
352 while ((base >= 0) && (path[base] != G_DIR_SEPARATOR))
353 base--;
354
355 return g_strndup (path + base + 1, last_char - base);
356}

360{

396char *
397build_uri (const char *base, ...)
398{
399 va_list args;
400 const char *child;
401 GString *uri;
402
403 uri = g_string_new (base);
404
405 va_start (args, base);
406 while ((child = va_arg (args, const char *)) != NULL) {
407 if (! g_str_has_suffix (uri->str, "/") && ! g_str_has_prefix (child, "/"))
408 g_string_append (uri, "/");
409 g_string_append (uri, child);
410 }
411 va_end (args);
412
413 return g_string_free (uri, FALSE);
414}

431gboolean
432make_directory_tree (GFile *dir,
433 mode_t mode,
434 GError **error)
435{
436 gboolean success = TRUE;
437 GFile *parent;
438
439 if ((dir == NULL) || g_file_query_exists (dir, NULL))
440 return TRUE;
441
442 parent = g_file_get_parent (dir);
443 if (parent != NULL) {
444 success = make_directory_tree (parent, mode, error);
445 g_object_unref (parent);
446 if (! success)
447 return FALSE;
448 }
449
450 success = g_file_make_directory (dir, NULL, error);
451 if ((error != NULL) && (*error != NULL) && g_error_matches (*error, G_IO_ERROR, G_IO_ERROR_EXISTS)) {
452 g_clear_error (error);
453 success = TRUE;
454 }
455
456 if (success)
457 g_file_set_attribute_uint32 (dir,
458 G_FILE_ATTRIBUTE_UNIX_MODE,
459 mode,
460 0,
461 NULL,
462 NULL);
463
464 return success;
465}

507const char *
508get_file_extension (const char *filename)
509{
510 const char *ptr = filename;
511 int len;
512 int p;
513 const char *ext;
514 const char *tar_exts[] = {".7z", ".br", ".bz", ".bz2", ".gz", ".lrz", ".lz", ".lzma", ".lzo", ".xz", ".Z", ".xst", NULL};
515
516 if (filename == NULL)
517 return NULL;
518
519 len = strlen (filename);
520 if (len <= 1)
521 return NULL;
522
523 p = len - 1;
524 while ((p >= 0) && (ptr[p] != '.'))
525 p--;
526 if (p < 0)
527 return NULL;
528
529 ext = filename + p;
530 p = 0;
531 if (ext - 4 > filename) {
532 const char *test = ext - 4;
533 if (strncmp (test, ".tar", 4) == 0) {
534 while (tar_exts[p] != NULL) {
535 if (strcmp (ext, tar_exts[p]) == 0)
536 ext = ext - 4;
537 p++;
538 }
539 }
540 }
541 return ext;
542}

651static gboolean
652delete_directory_recursive (GFile *dir,
653 GError **error)
654{
655 char *uri;
656 GFileEnumerator *file_enum;
657 GFileInfo *info;
658 gboolean error_occurred = FALSE;
659
660 if (error != NULL)
661 *error = NULL;
662
663 file_enum = g_file_enumerate_children (dir,
664 G_FILE_ATTRIBUTE_STANDARD_NAME ","
665 G_FILE_ATTRIBUTE_STANDARD_TYPE,
666 0, NULL, error);
667
668 uri = g_file_get_uri (dir);
669 while (! error_occurred && (info = g_file_enumerator_next_file (file_enum, NULL, error)) != NULL) {
670 char *child_uri;
671 GFile *child;
672
673 child_uri = build_uri (uri, g_file_info_get_name (info), NULL);
674 child = g_file_new_for_uri (child_uri);
675
676 switch (g_file_info_get_file_type (info)) {
677 case G_FILE_TYPE_DIRECTORY:
678 if (! delete_directory_recursive (child, error))
679 error_occurred = TRUE;
680 break;
681 default:
682 if (! g_file_delete (child, NULL, error))
683 error_occurred = TRUE;
684 break;
685 }
686
687 g_object_unref (child);
688 g_free (child_uri);
689 g_object_unref (info);
690 }
691 g_free (uri);
692
693 if (! error_occurred && ! g_file_delete (dir, NULL, error))
694 error_occurred = TRUE;
695
696 g_object_unref (file_enum);
697
698 return ! error_occurred;
699}

753char *
754get_temp_work_dir (const char *parent_folder)
755{
756 guint64 max_size = 0;
757 char *best_folder = NULL;
758 char *template;
759 char *result = NULL;
760
761 if (parent_folder == NULL) {
762 /* find the folder with more free space. */
763 int i;
764
765 for (i = 0; try_folder[i] != NULL; i++) {
766 char *folder;
767 guint64 size;
768
769 folder = ith_temp_folder_to_try (i);
770 size = get_dest_free_space (folder);
771 if (max_size < size) {
772 max_size = size;
773 g_free (best_folder);
774 best_folder = folder;
775 }
776 else
777 g_free (folder);
778 }
779 }
780 else
781 best_folder = g_strdup (parent_folder);
782
783 if (best_folder == NULL)
784 return NULL;
785
786 template = g_strconcat (best_folder, "/.fr-XXXXXX", NULL);
787 result = mkdtemp (template);
788
789 if ((result == NULL) || (*result == '\0')) {
790 g_free (template);
791 result = NULL;
792 }
793
794 return result;
795}

836gboolean
837file_list__match_pattern (const char *line,
838 const char *pattern)
839{
840 const char *l = line, *p = pattern;
841
842 for (; (*p != 0) && (*l != 0); p++, l++) {
843 if (*p != '%') {
844 if (*p != *l)
845 return FALSE;
846 }
847 else {
848 p++;
849 switch (*p) {
850 case 'a':
851 break;
852 case 'n':
853 if (!isdigit (*l))
854 return FALSE;
855 break;
856 case 'c':
857 if (!isalpha (*l))
858 return FALSE;
859 break;
860 default:
861 return FALSE;
862 }
863 }
864 }
865
866 return (*p == 0);
867}
