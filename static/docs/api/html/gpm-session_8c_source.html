89gpm_session_logout (GpmSession *session)
90{
91 g_return_val_if_fail (GPM_IS_SESSION (session), FALSE);
92
93 /* no mate-session */
94 if (session->priv->proxy == NULL) {
95 g_warning ("no mate-session");

97 }
98
99 /* we have to use no reply, as the SM calls into g-p-m to get the can_suspend property */
100 dbus_g_proxy_call_no_reply (session->priv->proxy, "Shutdown", G_TYPE_INVALID);
101 return TRUE;

139{
140 gboolean is_idle;

152static gboolean
153gpm_session_is_idle (GpmSession *session)
154{
155 gboolean ret;
156 gboolean is_idle = FALSE;
157 GError *error = NULL;
158 GValue *value;
159
160 /* no mate-session */
161 if (session->priv->proxy_prop == NULL) {
162 g_warning ("no mate-session");
163 goto out;
164 }
165
166 value = g_new0(GValue, 1);
167 /* find out if this change altered the inhibited state */
168 ret = dbus_g_proxy_call (session->priv->proxy_prop, "Get", &error,
169 G_TYPE_STRING, GPM_SESSION_MANAGER_PRESENCE_INTERFACE,
170 G_TYPE_STRING, "status",
171 G_TYPE_INVALID,
172 G_TYPE_VALUE, value,
173 G_TYPE_INVALID);
174 if (!ret) {
175 g_warning ("failed to get idle status: %s", error->message);
176 g_error_free (error);
177 is_idle = FALSE;
178 goto out;
179 }
180 is_idle = (g_value_get_uint (value) == GPM_SESSION_STATUS_ENUM_IDLE);
181 g_free (value);
182out:
183 return is_idle;
184}

314gboolean
315gpm_session_register_client (GpmSession *session, const gchar *app_id, const gchar *client_startup_id)
316{
317 gboolean ret = FALSE;
318 gchar *client_id = NULL;
319 GError *error = NULL;
320 DBusGConnection *connection;
321
322 g_return_val_if_fail (GPM_IS_SESSION (session), FALSE);
323
324 /* no mate-session */
325 if (session->priv->proxy == NULL) {
326 g_warning ("no mate-session");
327 goto out;
328 }
329
330 /* find out if this change altered the inhibited state */
331 ret = dbus_g_proxy_call (session->priv->proxy, "RegisterClient", &error,
332 G_TYPE_STRING, app_id,
333 G_TYPE_STRING, client_startup_id,
334 G_TYPE_INVALID,
335 DBUS_TYPE_G_OBJECT_PATH, &client_id,
336 G_TYPE_INVALID);
337 if (!ret) {
338 g_warning ("failed to register client '%s': %s", client_startup_id, error->message);
339 g_error_free (error);
340 goto out;
341 }
342
343 /* get org.gnome.SessionManager.ClientPrivate interface */
344 connection = dbus_g_bus_get (DBUS_BUS_SESSION, NULL);
345 session->priv->proxy_client_private = dbus_g_proxy_new_for_name_owner (connection, GPM_SESSION_MANAGER_SERVICE,
346 client_id, GPM_SESSION_MANAGER_CLIENT_PRIVATE_INTERFACE, &error);
347 if (session->priv->proxy_client_private == NULL) {
348 g_warning ("DBUS error: %s", error->message);
349 g_error_free (error);
350 goto out;
351 }
352
353 /* get Stop */
354 dbus_g_proxy_add_signal (session->priv->proxy_client_private, "Stop", G_TYPE_INVALID);
355 dbus_g_proxy_connect_signal (session->priv->proxy_client_private, "Stop", G_CALLBACK (gpm_session_stop_cb), session, NULL);
356
357 /* get QueryEndSession */
358 dbus_g_proxy_add_signal (session->priv->proxy_client_private, "QueryEndSession", G_TYPE_UINT, G_TYPE_INVALID);
359 dbus_g_proxy_connect_signal (session->priv->proxy_client_private, "QueryEndSession", G_CALLBACK (gpm_session_query_end_session_cb), session, NULL);
360
361 /* get EndSession */
362 dbus_g_proxy_add_signal (session->priv->proxy_client_private, "EndSession", G_TYPE_UINT, G_TYPE_INVALID);
363 dbus_g_proxy_connect_signal (session->priv->proxy_client_private, "EndSession", G_CALLBACK (gpm_session_end_session_cb), session, NULL);
364
365 g_debug ("registered startup '%s' to client id '%s'", client_startup_id, client_id);
366out:
367 g_free (client_id);
368 return ret;
369}

374static void
375gpm_session_inhibit_changed_cb (DBusGProxy *proxy, const gchar *id, GpmSession *session)
376{
377 gboolean is_idle_inhibited;
378 gboolean is_suspend_inhibited;
379
380 is_idle_inhibited = gpm_session_is_idle_inhibited (session);
381 is_suspend_inhibited = gpm_session_is_suspend_inhibited (session);
382 if (is_idle_inhibited != session->priv->is_idle_inhibited_old || is_suspend_inhibited != session->priv->is_suspend_inhibited_old) {
383 g_debug ("emitting inhibited-changed : idle=(%i), suspend=(%i)", is_idle_inhibited, is_suspend_inhibited);
384 session->priv->is_idle_inhibited_old = is_idle_inhibited;
385 session->priv->is_suspend_inhibited_old = is_suspend_inhibited;
386 g_signal_emit (session, signals [INHIBITED_CHANGED], 0, is_idle_inhibited, is_suspend_inhibited);
387 }
388}

448static void
449gpm_session_init (GpmSession *session)
450{
451 DBusGConnection *connection;
452 GError *error = NULL;
453
454 session->priv = gpm_session_get_instance_private (session);
455 session->priv->is_idle_old = FALSE;
456 session->priv->is_idle_inhibited_old = FALSE;
457 session->priv->is_suspend_inhibited_old = FALSE;
458 session->priv->proxy_client_private = NULL;
459
460 connection = dbus_g_bus_get (DBUS_BUS_SESSION, NULL);
461
462 /* get org.gnome.SessionManager interface */
463 session->priv->proxy = dbus_g_proxy_new_for_name_owner (connection, GPM_SESSION_MANAGER_SERVICE,
464 GPM_SESSION_MANAGER_PATH,
465 GPM_SESSION_MANAGER_INTERFACE, &error);
466 if (session->priv->proxy == NULL) {
467 g_warning ("DBUS error: %s", error->message);
468 g_error_free (error);
469 return;
470 }
471
472 /* get org.gnome.SessionManager.Presence interface */
473 session->priv->proxy_presence = dbus_g_proxy_new_for_name_owner (connection, GPM_SESSION_MANAGER_SERVICE,
474 GPM_SESSION_MANAGER_PRESENCE_PATH,
475 GPM_SESSION_MANAGER_PRESENCE_INTERFACE, &error);
476 if (session->priv->proxy_presence == NULL) {
477 g_warning ("DBUS error: %s", error->message);
478 g_error_free (error);
479 return;
480 }
481
482 /* get properties interface */
483 session->priv->proxy_prop = dbus_g_proxy_new_for_name_owner (connection, GPM_SESSION_MANAGER_SERVICE,
484 GPM_SESSION_MANAGER_PRESENCE_PATH,
485 GPM_DBUS_PROPERTIES_INTERFACE, &error);
486 if (session->priv->proxy_prop == NULL) {
487 g_warning ("DBUS error: %s", error->message);
488 g_error_free (error);
489 return;
490 }
491
492 /* get StatusChanged */
493 dbus_g_proxy_add_signal (session->priv->proxy_presence, "StatusChanged", G_TYPE_UINT, G_TYPE_INVALID);
494 dbus_g_proxy_connect_signal (session->priv->proxy_presence, "StatusChanged", G_CALLBACK (gpm_session_presence_status_changed_cb), session, NULL);
495
496 /* get InhibitorAdded */
497 dbus_g_proxy_add_signal (session->priv->proxy, "InhibitorAdded", DBUS_TYPE_G_OBJECT_PATH, G_TYPE_INVALID);
498 dbus_g_proxy_connect_signal (session->priv->proxy, "InhibitorAdded", G_CALLBACK (gpm_session_inhibit_changed_cb), session, NULL);
499
500 /* get InhibitorRemoved */
501 dbus_g_proxy_add_signal (session->priv->proxy, "InhibitorRemoved", DBUS_TYPE_G_OBJECT_PATH, G_TYPE_INVALID);
502 dbus_g_proxy_connect_signal (session->priv->proxy, "InhibitorRemoved", G_CALLBACK (gpm_session_inhibit_changed_cb), session, NULL);
503
504 /* coldplug */
505 session->priv->is_idle_inhibited_old = gpm_session_is_idle_inhibited (session);
506 session->priv->is_suspend_inhibited_old = gpm_session_is_suspend_inhibited (session);
507 session->priv->is_idle_old = gpm_session_is_idle (session);
508 g_debug ("idle: %i, idle_inhibited: %i, suspend_inhibited: %i", session->priv->is_idle_old, session->priv->is_idle_inhibited_old, session->priv->is_suspend_inhibited_old);
509}

515static void
516gpm_session_finalize (GObject *object)
517{
518 GpmSession *session;
519 g_return_if_fail (object != NULL);
520 g_return_if_fail (GPM_IS_SESSION (object));
521
522 session = GPM_SESSION (object);
523 session->priv = gpm_session_get_instance_private (session);
524
525 g_object_unref (session->priv->proxy);
526 g_object_unref (session->priv->proxy_presence);
527 if (session->priv->proxy_client_private != NULL)
528 g_object_unref (session->priv->proxy_client_private);
529 g_object_unref (session->priv->proxy_prop);
530
531 G_OBJECT_CLASS (gpm_session_parent_class)->finalize (object);
532}
