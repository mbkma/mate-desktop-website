182static GSList *
183get_history_list (GsearchHistoryEntry *entry)
184{
185 GtkListStore *store;
186 GtkTreeIter iter;
187 gboolean valid;
188 GSList *list = NULL;
189
190 store = get_history_store (entry);
191
192 valid = gtk_tree_model_get_iter_first (GTK_TREE_MODEL (store),
193 &iter);
194
195 while (valid)
196 {
197 gchar *str;
198
199 gtk_tree_model_get (GTK_TREE_MODEL (store),
200 &iter,
201 0, &str,
202 -1);
203
204 list = g_slist_prepend (list, str);
205
206 valid = gtk_tree_model_iter_next (GTK_TREE_MODEL (store),
207 &iter);
208 }
209
210 return g_slist_reverse (list);
211}

213static void
214gsearch_history_entry_save_history (GsearchHistoryEntry *entry)
215{
216 GVariant *history;
217 GSList *items;
218 gchar *key;
219 GVariantBuilder item_builder;
220 GVariantBuilder history_builder;
221 GVariantIter *iter;
222 GVariant *item;
223 GVariant *history_list;
224 GSList *list_iter;
225 gchar *history_key;
226
227 g_return_if_fail (GSEARCH_IS_HISTORY_ENTRY (entry));
228
229 items = get_history_list (entry);
230 key = get_history_key (entry);
231
232 history = g_settings_get_value (entry->priv->settings,
233 "search-history");
234
235 g_variant_builder_init (&item_builder, G_VARIANT_TYPE ("as"));
236 for (list_iter = items; list_iter; list_iter = list_iter->next)
237 g_variant_builder_add (&item_builder, "s", (gchar *) list_iter->data);
238
239 g_variant_builder_init (&history_builder, G_VARIANT_TYPE ("a{sas}"));
240 g_variant_builder_add (&history_builder, "{sas}", key, &item_builder);
241
242 if (history) {
243 g_variant_get (history, "a{sas}", &iter);
244 while ((item = g_variant_iter_next_value (iter))) {
245 g_variant_get (item, "{s@as}", &history_key, &history_list);
246 if (g_strcmp0 (history_key, key) != 0)
247 g_variant_builder_add (&history_builder, "{s@as}", history_key, history_list);
248 g_free (history_key);
249 g_variant_unref (history_list);
250 g_variant_unref (item);
251 }
252 g_variant_iter_free (iter);
253 g_variant_unref (history);
254 }
255
256 g_settings_set_value (entry->priv->settings,
257 "search-history",
258 g_variant_new ("a{sas}", &history_builder));
259
260 g_slist_free_full (items, g_free);
261 g_free (key);
262}

264static gboolean
265remove_item (GtkListStore *store,
266 const gchar *text)
267{
268 GtkTreeIter iter;
269
270 g_return_val_if_fail (text != NULL, FALSE);
271
272 if (!gtk_tree_model_get_iter_first (GTK_TREE_MODEL (store), &iter))
273 return FALSE;
274
275 do
276 {
277 gchar *item_text;
278
279 gtk_tree_model_get (GTK_TREE_MODEL (store),
280 &iter,
281 0,
282 &item_text,
283 -1);
284
285 if (item_text != NULL &&
286 strcmp (item_text, text) == 0)
287 {
288 gtk_list_store_remove (store, &iter);
289 g_free (item_text);
290 return TRUE;
291 }
292
293 g_free (item_text);
294
295 } while (gtk_tree_model_iter_next (GTK_TREE_MODEL (store), &iter));
296
297 return FALSE;
298}

300static void
301clamp_list_store (GtkListStore *store,
302 guint max)
303{
304 GtkTreePath *path;
305 GtkTreeIter iter;
306
307 /* -1 because TreePath counts from 0 */
308 path = gtk_tree_path_new_from_indices (max - 1, -1);
309
310 if (gtk_tree_model_get_iter (GTK_TREE_MODEL (store), &iter, path))
311 {
312 while (1)
313 {
314 if (!gtk_list_store_remove (store, &iter))
315 break;
316 }
317 }
318
319 gtk_tree_path_free (path);
320}

322static void
323insert_history_item (GsearchHistoryEntry *entry,
324 const gchar *text,
325 gboolean prepend)
326{
327 GtkListStore *store;
328 GtkTreeIter iter;
329
330 if (g_utf8_strlen (text, -1) <= MIN_ITEM_LEN)
331 return;
332
333 store = get_history_store (entry);
334
335 /* remove the text from the store if it was already
336 * present. If it wasn't, clamp to max history - 1
337 * before inserting the new row, otherwise appending
338 * would not work */
339
340 if (!remove_item (store, text))
341 clamp_list_store (store,
342 entry->priv->history_length - 1);
343
344 if (prepend)
345 gtk_list_store_insert (store, &iter, 0);
346 else
347 gtk_list_store_append (store, &iter);
348
349 gtk_list_store_set (store,
350 &iter,
351 0,
352 text,
353 -1);
354
355 gsearch_history_entry_save_history (entry);
356}

378static void
379gsearch_history_entry_load_history (GsearchHistoryEntry *entry)
380{
381 GtkListStore *store;
382 GtkTreeIter iter;
383 GVariant *history;
384 gchar *key;
385 gint i;
386
387 g_return_if_fail (GSEARCH_IS_HISTORY_ENTRY (entry));
388
389 store = get_history_store (entry);
390 key = get_history_key (entry);
391
392 history = g_settings_get_value (entry->priv->settings,
393 "search-history");
394
395 gtk_list_store_clear (store);
396
397 if (history) {
398 GVariantIter *history_iter, *history_subiter;
399 GVariant *history_item, *history_subitem;
400 gchar *history_key;
401 gchar *text;
402
403 g_variant_get (history, "a{sas}", &history_iter);
404
405 while ((history_item = g_variant_iter_next_value (history_iter))) {
406 i = 0;
407 g_variant_get (history_item, "{sas}", &history_key, &history_subiter);
408
409 if (g_strcmp0 (history_key, key) == 0) {
410 while ((history_subitem = g_variant_iter_next_value (history_subiter)) &&
411 i < entry->priv->history_length) {
412 g_variant_get (history_subitem, "s", &text);
413 gtk_list_store_append (store, &iter);
414 gtk_list_store_set (store,
415 &iter,
416 0,
417 text,
418 -1);
419 g_free (text);
420 g_variant_unref (history_subitem);
421 i++;
422 }
423 }
424 g_free (history_key);
425 g_variant_iter_free (history_subiter);
426 g_variant_unref (history_item);
427 }
428 g_variant_iter_free (history_iter);
429 g_variant_unref (history);
430 }
431 g_free (key);
432}

491void
492gsearch_history_entry_set_enable_completion (GsearchHistoryEntry *entry,
493 gboolean enable)
494{
495 g_return_if_fail (GSEARCH_IS_HISTORY_ENTRY (entry));
496
497 if (enable)
498 {
499 if (entry->priv->completion != NULL)
500 return;
501
502 entry->priv->completion = gtk_entry_completion_new ();
503 gtk_entry_completion_set_model (entry->priv->completion,
504 GTK_TREE_MODEL (get_history_store (entry)));
505
506 /* Use model column 0 as the text column */
507 gtk_entry_completion_set_text_column (entry->priv->completion, 0);
508
509 gtk_entry_completion_set_minimum_key_length (entry->priv->completion,
510 MIN_ITEM_LEN);
511
512 gtk_entry_completion_set_popup_completion (entry->priv->completion, FALSE);
513 gtk_entry_completion_set_inline_completion (entry->priv->completion, TRUE);
514
515 /* Assign the completion to the entry */
516 gtk_entry_set_completion (GTK_ENTRY (gsearch_history_entry_get_entry(entry)),
517 entry->priv->completion);
518 }
519 else
520 {
521 if (entry->priv->completion == NULL)
522 return;
523
524 gtk_entry_set_completion (GTK_ENTRY (gsearch_history_entry_get_entry (entry)),
525 NULL);
526
527 g_object_unref (entry->priv->completion);
528
529 entry->priv->completion = NULL;
530 }

541GtkWidget *
542gsearch_history_entry_new (const gchar *history_id,
543 gboolean enable_completion)
544{
545 GtkWidget *ret;
546 GtkListStore *store;
547
548 g_return_val_if_fail(history_id != NULL, NULL);
549
550 /* Note that we are setting the model, so
551 * user must be careful to always manipulate
552 * data in the history through gsearch_history_entry_
553 * functions.
554 */
555
556 store = gtk_list_store_new(1, G_TYPE_STRING);
557
558 ret = g_object_new(GSEARCH_TYPE_HISTORY_ENTRY,
559 "has-entry", TRUE,
560 "history-id", history_id,
561 "model", store,
562 "entry-text-column", 0,
563 NULL);
564
565 g_object_unref (store);
566
567 /* loading has to happen after the model
568 * has been set. However the model is not a
569 * G_PARAM_CONSTRUCT property of GtkComboBox
570 * so we cannot do this in the constructor.
571 * For now we simply do here since this widget is
572 * not bound to other programming languages.
573 * A maybe better alternative is to override the
574 * model property of combobox and mark CONTRUCT_ONLY.
575 * This would also ensure that the model cannot be
576 * set explicitely at a later time.
577 */
578 gsearch_history_entry_load_history (GSEARCH_HISTORY_ENTRY (ret));
579
580 gsearch_history_entry_set_enable_completion (GSEARCH_HISTORY_ENTRY (ret),
581 enable_completion);
582
583 return ret;
584}
