38/*
39 * na_factory_provider_read_data:
40 * @reader: the instance which implements this #NAIFactoryProvider interface.
41 * @reader_data: instance data.
42 * @object: the #NAIFactoryobject being unserialized.
43 * @def: a #NADataDef structure which identifies the data to be unserialized.
44 * @messages: a pointer to a #GSList list of strings; the implementation
45 * may append messages to this list, but shouldn't reinitialize it.
46 *
47 * Reads the specified data and set it up into the @boxed.
48 *
49 * Returns: a new #NADataBoxed object which contains the data.
50 */
51NADataBoxed *
52na_factory_provider_read_data( const NAIFactoryProvider *reader, void *reader_data,
53 const NAIFactoryObject *object, const NADataDef *def,
54 GSList **messages )
55{
56 NADataBoxed *boxed;
57
58 g_return_val_if_fail( NA_IS_IFACTORY_PROVIDER( reader ), NULL );
59 g_return_val_if_fail( NA_IS_IFACTORY_OBJECT( object ), NULL );
60
61 boxed = NULL;
62
63 if( NA_IFACTORY_PROVIDER_GET_INTERFACE( reader )->read_data ){
64 boxed = NA_IFACTORY_PROVIDER_GET_INTERFACE( reader )->read_data( reader, reader_data, object, def, messages );
65 }
66
67 return( boxed );
68}

84 GSList **messages )
85{
86 guint code;
87
88 g_return_val_if_fail( NA_IS_IFACTORY_PROVIDER( writer ), NA_IIO_PROVIDER_CODE_PROGRAM_ERROR );
89 g_return_val_if_fail( NA_IS_IFACTORY_OBJECT( object ), NA_IIO_PROVIDER_CODE_PROGRAM_ERROR );
90
91 code = NA_IIO_PROVIDER_CODE_NOT_WILLING_TO_RUN;
92
93 if( NA_IFACTORY_PROVIDER_GET_INTERFACE( writer )->write_data ){
