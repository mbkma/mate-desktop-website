112/* This is cut out of gconvert.c from glib (and mildly modified). Not all
113 backends give valid UTF-8 for properties, so we make sure that is.
114 */
115static gchar *
116make_valid_utf8 (const gchar *name)
117{
118 GString *string;
119 const gchar *remainder, *invalid;
120 gint remaining_bytes, valid_bytes;
121
122 string = NULL;
123 remainder = name;
124 remaining_bytes = strlen (name);
125
126 while (remaining_bytes != 0)
127 {
128 if (g_utf8_validate (remainder, remaining_bytes, &invalid))
129 break;
130 valid_bytes = invalid - remainder;
131
132 if (string == NULL)
133 string = g_string_sized_new (remaining_bytes);
134
135 g_string_append_len (string, remainder, valid_bytes);
136 g_string_append_c (string, '?');
137
138 remaining_bytes -= valid_bytes + 1;
139 remainder = invalid + 1;
140 }
141
142 if (string == NULL)
143 return g_strdup (name);
144
145 g_string_append (string, remainder);
146
147 g_assert (g_utf8_validate (string->str, -1, NULL));
148
149 return g_string_free (string, FALSE);
150}

152static void
153set_property (EvPropertiesView *properties,
154 GtkGrid *grid,
155 Property property,
156 const gchar *text,
157 gint *row)
158{
159 GtkWidget *label;
160 gchar *markup;
161 gchar *valid_text;
162
163 if (!properties->labels[property]) {
164 label = gtk_label_new (NULL);
165 g_object_set (G_OBJECT (label), "xalign", 0.0, NULL);
166 markup = g_strdup_printf ("<b>%s</b>", _(properties_info[property].label));
167 gtk_label_set_markup (GTK_LABEL (label), markup);
168 g_free (markup);
169
170 gtk_grid_attach (grid, label, 0, *row, 1, 1);
171 gtk_widget_show (label);
172 }
173
174 if (!properties->labels[property]) {
175 label = gtk_label_new (NULL);
176
177 g_object_set (G_OBJECT (label),
178 "xalign", 0.0,
179 "width_chars", 25,
180 "selectable", TRUE,
181 "ellipsize", PANGO_ELLIPSIZE_END,
182 NULL);
183 } else {
184 label = properties->labels[property];
185 }
186
187 if (text == NULL || text[0] == '\000') {
188 /* translators: This is used when a document property does
189 not have a value. Examples:
190 Author: None
191 Keywords: None
192 */
193 markup = g_markup_printf_escaped ("<i>%s</i>", _("None"));
194 gtk_label_set_markup (GTK_LABEL (label), markup);
195 g_free (markup);
196 } else {
197 valid_text = make_valid_utf8 (text);
198 gtk_label_set_text (GTK_LABEL (label), valid_text);
199 g_free (valid_text);
200 }
201
202 if (!properties->labels[property]) {
203 gtk_grid_attach (grid, label, 1, *row, 1, 1);
204 properties->labels[property] = label;
205 }
206
207 gtk_widget_show (label);
208
209 *row += 1;
210}

212static GtkUnit
213get_default_user_units (void)
214{
215 /* Translate to the default units to use for presenting
216 * lengths to the user. Translate to default:inch if you
217 * want inches, otherwise translate to default:mm.
218 * Do *not* translate it to "predefinito:mm", if it
219 * it isn't default:mm or default:inch it will not work
220 */
221 gchar *e = _("default:mm");
222
223#ifdef HAVE__NL_MEASUREMENT_MEASUREMENT
224 gchar *imperial = NULL;
225
226 imperial = nl_langinfo (_NL_MEASUREMENT_MEASUREMENT);
227 if (imperial && imperial[0] == 2)
228 return GTK_UNIT_INCH; /* imperial */
229 if (imperial && imperial[0] == 1)
230 return GTK_UNIT_MM; /* metric */
231#endif
232
233 if (strcmp (e, "default:mm") == 0)
234 return GTK_UNIT_MM;
235 if (strcmp (e, "default:inch") == 0)
236 return GTK_UNIT_INCH;
237
238 g_warning ("Whoever translated default:mm did so wrongly.\n");
239
240 return GTK_UNIT_MM;
241}

243static gdouble
244get_tolerance (gdouble size)
245{
246 if (size < 150.0f)
247 return 1.5f;
248 else if (size >= 150.0f && size <= 600.0f)
249 return 2.0f;
250 else
251 return 3.0f;
252}

254static char *
255ev_regular_paper_size (const EvDocumentInfo *info)
256{
257 GList *paper_sizes, *l;
258 gchar *exact_size;
259 gchar *str = NULL;
260 GtkUnit units;
261
262 units = get_default_user_units ();
263
264 if (units == GTK_UNIT_MM) {
265 exact_size = g_strdup_printf(_("%.0f %.0f mm"),
266 info->paper_width,
267 info->paper_height);
268 } else {
269 exact_size = g_strdup_printf (_("%.2f %.2f inch"),
270 info->paper_width / 25.4f,
271 info->paper_height / 25.4f);
272 }
273
274 paper_sizes = gtk_paper_size_get_paper_sizes (FALSE);
275
276 for (l = paper_sizes; l && l->data; l = g_list_next (l)) {
277 GtkPaperSize *size = (GtkPaperSize *) l->data;
278 gdouble paper_width;
279 gdouble paper_height;
280 gdouble width_tolerance;
281 gdouble height_tolerance;
282
283 paper_width = gtk_paper_size_get_width (size, GTK_UNIT_MM);
284 paper_height = gtk_paper_size_get_height (size, GTK_UNIT_MM);
285
286 width_tolerance = get_tolerance (paper_width);
287 height_tolerance = get_tolerance (paper_height);
288
289 if (ABS (info->paper_height - paper_height) <= height_tolerance &&
290 ABS (info->paper_width - paper_width) <= width_tolerance) {
291 /* Note to translators: first placeholder is the paper name (eg.
292 * A4), second placeholder is the paper size (eg. 297x210 mm) */
293 str = g_strdup_printf (_("%s, Portrait (%s)"),
294 gtk_paper_size_get_display_name (size),
295 exact_size);
296 } else if (ABS (info->paper_width - paper_height) <= height_tolerance &&
297 ABS (info->paper_height - paper_width) <= width_tolerance) {
298 /* Note to translators: first placeholder is the paper name (eg.
299 * A4), second placeholder is the paper size (eg. 297x210 mm) */
300 str = g_strdup_printf ( _("%s, Landscape (%s)"),
301 gtk_paper_size_get_display_name (size),
302 exact_size);
303 }
304 }
305
306 g_list_foreach (paper_sizes, (GFunc) gtk_paper_size_free, NULL);
307 g_list_free (paper_sizes);
308
309 if (str != NULL) {
310 g_free (exact_size);
311 return str;
312 }
313
314 return exact_size;
315}

317void
318ev_properties_view_set_info (EvPropertiesView *properties, const EvDocumentInfo *info)
319{
320 GtkWidget *grid;
321 gchar *text;
322 gint row = 0;
323
324 grid = properties->grid;
325
326 if (info->fields_mask & EV_DOCUMENT_INFO_TITLE) {
327 set_property (properties, GTK_GRID (grid), TITLE_PROPERTY, info->title, &row);
328 }
329 set_property (properties, GTK_GRID (grid), URI_PROPERTY, properties->uri, &row);
330 if (info->fields_mask & EV_DOCUMENT_INFO_SUBJECT) {
331 set_property (properties, GTK_GRID (grid), SUBJECT_PROPERTY, info->subject, &row);
332 }
333 if (info->fields_mask & EV_DOCUMENT_INFO_AUTHOR) {
334 set_property (properties, GTK_GRID (grid), AUTHOR_PROPERTY, info->author, &row);
335 }
336 if (info->fields_mask & EV_DOCUMENT_INFO_KEYWORDS) {
337 set_property (properties, GTK_GRID (grid), KEYWORDS_PROPERTY, info->keywords, &row);
338 }
339 if (info->fields_mask & EV_DOCUMENT_INFO_PRODUCER) {
340 set_property (properties, GTK_GRID (grid), PRODUCER_PROPERTY, info->producer, &row);
341 }
342 if (info->fields_mask & EV_DOCUMENT_INFO_CREATOR) {
343 set_property (properties, GTK_GRID (grid), CREATOR_PROPERTY, info->creator, &row);
344 }
345 if ((info->fields_mask & EV_DOCUMENT_INFO_CREATION_DATE) && (info->creation_date > 0)) {
346 text = ev_document_misc_format_date (info->creation_date);
347 set_property (properties, GTK_GRID (grid), CREATION_DATE_PROPERTY, text, &row);
348 g_free (text);
349 }
350 if ((info->fields_mask & EV_DOCUMENT_INFO_MOD_DATE) && (info->modified_date > 0)) {
351 text = ev_document_misc_format_date (info->modified_date);
352 set_property (properties, GTK_GRID (grid), MOD_DATE_PROPERTY, text, &row);
353 g_free (text);
354 }
355 if (info->fields_mask & EV_DOCUMENT_INFO_FORMAT) {
356 set_property (properties, GTK_GRID (grid), FORMAT_PROPERTY, info->format, &row);
357 }
358 if (info->fields_mask & EV_DOCUMENT_INFO_N_PAGES) {
359 text = g_strdup_printf ("%d", info->n_pages);
360 set_property (properties, GTK_GRID (grid), N_PAGES_PROPERTY, text, &row);
361 g_free (text);
362 }
363 if (info->fields_mask & EV_DOCUMENT_INFO_LINEARIZED) {
364 set_property (properties, GTK_GRID (grid), LINEARIZED_PROPERTY, info->linearized, &row);
365 }
366 if (info->fields_mask & EV_DOCUMENT_INFO_SECURITY) {
367 set_property (properties, GTK_GRID (grid), SECURITY_PROPERTY, info->security, &row);
368 }
369 if (info->fields_mask & EV_DOCUMENT_INFO_PAPER_SIZE) {
370 text = ev_regular_paper_size (info);
371 set_property (properties, GTK_GRID (grid), PAPER_SIZE_PROPERTY, text, &row);
372 g_free (text);
373 }
374}
