52static void
53write_out_password (const char *password)
54{
55 char *to_hash;
56 char *ascii_digest;
57 char *line;
58 char *filename;
59 FILE *file;
60
61 to_hash = g_strdup_printf ("%s:%s:%s", USER, REALM, password);
62 ascii_digest = g_compute_checksum_for_string (G_CHECKSUM_MD5, to_hash, strlen (to_hash));
63 g_free (to_hash);
64 line = g_strdup_printf ("%s:%s:%s\n", USER, REALM, ascii_digest);
65 g_free (ascii_digest);
66
67 filename = g_build_filename (g_get_user_config_dir (), "user-share", "passwd", NULL);
68
69 file = fopen (filename, "w");
70 if (file != NULL) {
71 fwrite (line, strlen (line), 1, file);
72 fclose (file);
73 }
74
75 g_free (filename);
76 g_free (line);
77}

79static void
80flush_password (void)
81{
82 GtkWidget *password_entry;
83 const char *password;
84
85 password_entry = GTK_WIDGET (gtk_builder_get_object (builder, "password_entry"));
86
87 if (g_object_get_data (G_OBJECT( password_entry), "user_edited")) {
88 password = gtk_entry_get_text (GTK_ENTRY (password_entry));
89 if (password != NULL && password[0] != 0)
90 write_out_password (password);
91 }
92}

95static void
96update_ui (void)
97{
98 GSettings *settings;
99 gboolean enabled;
100#ifdef HAVE_BLUETOOTH
101 gboolean bluetooth_enabled, bluetooth_write_enabled, require_pairing_enabled;
102 gboolean bluetooth_obexpush_enabled, bluetooth_obexpush_notify;
103#endif /* HAVE_BLUETOOTH */
104 char *str;
105 PasswordSetting password_setting;
106 AcceptSetting accept_setting;
107 GtkWidget *check;
108 GtkWidget *password_combo;
109 GtkWidget *password_entry;
110#ifdef HAVE_BLUETOOTH
111 GtkWidget *vbox_share_bluetooth;
112 GtkWidget *vbox_receive_bluetooth;
113 GtkWidget *bluetooth_check;
114 GtkWidget *allow_write_bluetooth_check;
115 GtkWidget *require_pairing_check;
116 GtkWidget *enable_obexpush_check;
117 GtkWidget *accept_obexpush_combo;
118 GtkWidget *notify_received_obexpush_check;
119#endif /* HAVE_BLUETOOTH */
120
121 settings = g_settings_new (GSETTINGS_SCHEMA);
122
123 enabled = g_settings_get_boolean (settings,
124 FILE_SHARING_ENABLED);
125
126 str = g_settings_get_string (settings, FILE_SHARING_REQUIRE_PASSWORD);
127 password_setting = password_setting_from_string (str);
128 g_free (str);
129
130#ifdef HAVE_BLUETOOTH
131 bluetooth_enabled = g_settings_get_boolean (settings,
132 FILE_SHARING_BLUETOOTH_ENABLED);
133 bluetooth_write_enabled = g_settings_get_boolean (settings,
134 FILE_SHARING_BLUETOOTH_ALLOW_WRITE);
135 require_pairing_enabled = g_settings_get_boolean (settings,
136 FILE_SHARING_BLUETOOTH_REQUIRE_PAIRING);
137 bluetooth_obexpush_enabled = g_settings_get_boolean (settings,
138 FILE_SHARING_BLUETOOTH_OBEXPUSH_ENABLED);
139 bluetooth_obexpush_notify = g_settings_get_boolean (settings,
140 FILE_SHARING_BLUETOOTH_OBEXPUSH_NOTIFY);
141
142 str = g_settings_get_string (settings, FILE_SHARING_BLUETOOTH_OBEXPUSH_ACCEPT_FILES);
143 accept_setting = accept_setting_from_string (str);
144 g_free (str);
145#endif /* HAVE_BLUETOOTH */
146
147 check = GTK_WIDGET (gtk_builder_get_object (builder, "enable_check"));
148 password_combo = GTK_WIDGET (gtk_builder_get_object (builder, "password_combo"));
149 password_entry = GTK_WIDGET (gtk_builder_get_object (builder, "password_entry"));
150#ifdef HAVE_BLUETOOTH
151 vbox_share_bluetooth = GTK_WIDGET (gtk_builder_get_object (builder, "vbox_share_bluetooth"));
152 vbox_receive_bluetooth = GTK_WIDGET (gtk_builder_get_object (builder, "vbox_receive_bluetooth"));
153 bluetooth_check = GTK_WIDGET (gtk_builder_get_object (builder, "enable_bluetooth_check"));
154 allow_write_bluetooth_check = GTK_WIDGET (gtk_builder_get_object (builder, "allow_write_bluetooth_check"));
155 require_pairing_check = GTK_WIDGET (gtk_builder_get_object (builder, "require_pairing_check"));
156 enable_obexpush_check = GTK_WIDGET (gtk_builder_get_object (builder, "enable_obexpush_check"));
157 accept_obexpush_combo = GTK_WIDGET (gtk_builder_get_object (builder, "accept_obexpush_combo"));
158 notify_received_obexpush_check = GTK_WIDGET (gtk_builder_get_object (builder, "notify_received_obexpush_check"));
159#endif /* HAVE_BLUETOOTH */
160
161 /* Network */
162 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (check), enabled);
163 gtk_widget_set_sensitive (password_combo, enabled);
164 gtk_widget_set_sensitive (password_entry, enabled && password_setting != PASSWORD_NEVER);
165
166 gtk_combo_box_set_active (GTK_COMBO_BOX (password_combo),
167 password_setting);
168
169#ifdef HAVE_BLUETOOTH
170 /* Bluetooth ObexFTP */
171 gtk_widget_set_visible (vbox_share_bluetooth, TRUE);
172 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (bluetooth_check), bluetooth_enabled);
173 gtk_widget_set_sensitive (allow_write_bluetooth_check, bluetooth_enabled);
174 gtk_widget_set_sensitive (require_pairing_check, bluetooth_enabled);
175
176 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (allow_write_bluetooth_check),
177 bluetooth_write_enabled);
178 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (require_pairing_check),
179 require_pairing_enabled);
180
181 /* Bluetooth ObexPush */
182 gtk_widget_set_visible (vbox_receive_bluetooth, TRUE);
183 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (enable_obexpush_check), bluetooth_obexpush_enabled);
184 gtk_widget_set_sensitive (accept_obexpush_combo, bluetooth_obexpush_enabled);
185 gtk_widget_set_sensitive (notify_received_obexpush_check, bluetooth_obexpush_enabled);
186
187 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (notify_received_obexpush_check),
188 bluetooth_obexpush_notify);
189
190 gtk_combo_box_set_active (GTK_COMBO_BOX (accept_obexpush_combo),
191 accept_setting);
192#endif /* HAVE_BLUETOOTH */
193
194 g_object_unref (settings);
195}

409static GtkWidget *
410error_dialog (const char *title,
411 const char *reason,
412 GtkWindow *parent)
413{
414 GtkWidget *error_dialog;
415
416 if (reason == NULL)
417 reason = _("No reason");
418
419 error_dialog =
420 gtk_message_dialog_new (parent,
421 GTK_DIALOG_MODAL,
422 GTK_MESSAGE_ERROR,
423 GTK_BUTTONS_OK,
424 "%s", title);
425 gtk_message_dialog_format_secondary_text
426 (GTK_MESSAGE_DIALOG (error_dialog), "%s", reason);
427
428 gtk_window_set_title (GTK_WINDOW (error_dialog), ""); /* as per HIG */
429 gtk_container_set_border_width (GTK_CONTAINER (error_dialog), 5);
430 gtk_dialog_set_default_response (GTK_DIALOG (error_dialog),
431 GTK_RESPONSE_OK);
432 gtk_window_set_modal (GTK_WINDOW (error_dialog), TRUE);
433
434 return error_dialog;
435}

437static void
438help_button_clicked (GtkButton *button, GtkWidget *window)
439{
440 GError *error = NULL;
441
442 if (gtk_show_uri_on_window (GTK_WINDOW (window), "help:mate-user-share", gtk_get_current_event_time (), &error) == FALSE) {
443 GtkWidget *dialog;
444
445 dialog = error_dialog (_("Could not display the help contents."), error->message, GTK_WINDOW (window));
446 g_signal_connect (G_OBJECT (dialog), "response", G_CALLBACK
447 (gtk_widget_destroy), error_dialog);
448 gtk_window_present (GTK_WINDOW (dialog));
449
450 g_error_free (error);
451 }
