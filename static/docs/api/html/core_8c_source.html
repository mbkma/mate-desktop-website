35/* Looks up the MetaWindow representing the frame of the given X window.
36 * Used as a helper function by a bunch of the functions below.
37 *
38 * FIXME: The functions that use this function throw the result away
39 * after use. Many of these functions tend to be called in small groups,
40 * which results in get_window() getting called several times in succession
41 * with the same parameters. We should profile to see whether this wastes
42 * much time, and if it does we should look into a generalised
43 * meta_core_get_window_info() which takes a bunch of pointers to variables
44 * to put its results in, and only fills in the non-null ones.
45 */
46static MetaWindow *
47get_window (Display *xdisplay,
48 Window frame_xwindow)
49{
50 MetaDisplay *display;
51 MetaWindow *window;
52
53 display = meta_display_for_x_display (xdisplay);
54 window = meta_display_lookup_x_window (display, frame_xwindow);
55
56 if (window == NULL || window->frame == NULL)
57 {
58 meta_bug ("No such frame window 0x%lx!\n", frame_xwindow);
59 return NULL;
60 }
61
62 return window;
63}

65void
66meta_core_get (Display *xdisplay,
67 Window xwindow,
68 ...)
69{
70 va_list args;
71 MetaCoreGetType request;
72
73 MetaDisplay *display = meta_display_for_x_display (xdisplay);
74 MetaWindow *window = meta_display_lookup_x_window (display, xwindow);
75
76 va_start (args, xwindow);
77
78 request = va_arg (args, MetaCoreGetType);
79
80 /* Now, we special-case the first request slightly. Mostly, requests
81 * for information on windows which have no frame are errors.
82 * But sometimes we may want to know *whether* a window has a frame.
83 * In this case, pass the key META_CORE_WINDOW_HAS_FRAME
84 * as the *first* request, with a pointer to a boolean; if the window
85 * has no frame, this will be set to False and meta_core_get will
86 * exit immediately (so the values of any other requests will be
87 * undefined). Otherwise it will be set to True and meta_core_get will
88 * continue happily on its way.
89 */
90
91 if (request != META_CORE_WINDOW_HAS_FRAME &&
92 (window == NULL || window->frame == NULL)) {
93 meta_bug ("No such frame window 0x%lx!\n", xwindow);
94 goto out;
95 }
96
97 while (request != META_CORE_GET_END) {
98
99 gpointer answer = va_arg (args, gpointer);
100
101 switch (request) {
102 case META_CORE_WINDOW_HAS_FRAME:
103 *((gboolean*)answer) = window != NULL && window->frame != NULL;
104 if (!*((gboolean*)answer)) goto out; /* see above */
105 break;
106 case META_CORE_GET_CLIENT_WIDTH:
107 *((gint*)answer) = window->rect.width;
108 break;
109 case META_CORE_GET_CLIENT_HEIGHT:
110 *((gint*)answer) = window->rect.height;
111 break;
112 case META_CORE_IS_TITLEBAR_ONSCREEN:
113 *((gboolean*)answer) = meta_window_titlebar_is_onscreen (window);
114 break;
115 case META_CORE_GET_CLIENT_XWINDOW:
116 *((Window*)answer) = window->xwindow;
117 break;
118 case META_CORE_GET_FRAME_FLAGS:
119 *((MetaFrameFlags*)answer) = meta_frame_get_flags (window->frame);
120 break;
121 case META_CORE_GET_FRAME_TYPE:
122 {
123 MetaFrameType base_type = META_FRAME_TYPE_LAST;
124
125 switch (window->type)
126 {
127 case META_WINDOW_NORMAL:
128 base_type = META_FRAME_TYPE_NORMAL;
129 break;
130
131 case META_WINDOW_DIALOG:
132 base_type = META_FRAME_TYPE_DIALOG;
133 break;
134
135 case META_WINDOW_MODAL_DIALOG:
136 if (meta_prefs_get_attach_modal_dialogs () &&
137 meta_window_get_transient_for (window) != NULL)
138 base_type = META_FRAME_TYPE_ATTACHED;
139 else
140 base_type = META_FRAME_TYPE_MODAL_DIALOG;
141 break;
142
143 case META_WINDOW_MENU:
144 base_type = META_FRAME_TYPE_MENU;
145 break;
146
147 case META_WINDOW_UTILITY:
148 base_type = META_FRAME_TYPE_UTILITY;
149 break;
150
151 case META_WINDOW_DESKTOP:
152 case META_WINDOW_DOCK:
153 case META_WINDOW_TOOLBAR:
154 case META_WINDOW_SPLASHSCREEN:
155 /* No frame */
156 base_type = META_FRAME_TYPE_LAST;
157 break;
158
159 }
160
161 if (base_type == META_FRAME_TYPE_LAST)
162 {
163 /* can't add border if undecorated */
164 *((MetaFrameType*)answer) = META_FRAME_TYPE_LAST;
165 }
166 else if (window->border_only && base_type != META_FRAME_TYPE_ATTACHED)
167 {
168 /* override base frame type */
169 *((MetaFrameType*)answer) = META_FRAME_TYPE_BORDER;
170 }
171 else
172 {
173 *((MetaFrameType*)answer) = base_type;
174 }
175
176 break;
177 }
178 case META_CORE_GET_MINI_ICON:
179 *((GdkPixbuf**)answer) = window->mini_icon;
180 break;
181 case META_CORE_GET_ICON:
182 *((GdkPixbuf**)answer) = window->icon;
183 break;
184 case META_CORE_GET_X:
185 meta_window_get_position (window, (int*)answer, NULL);
186 break;
187 case META_CORE_GET_Y:
188 meta_window_get_position (window, NULL, (int*)answer);
189 break;
190 case META_CORE_GET_FRAME_WORKSPACE:
191 *((gint*)answer) = meta_window_get_net_wm_desktop (window);
192 break;
193 case META_CORE_GET_FRAME_X:
194 *((gint*)answer) = window->frame->rect.x;
195 break;
196 case META_CORE_GET_FRAME_Y:
197 *((gint*)answer) = window->frame->rect.y;
198 break;
199 case META_CORE_GET_FRAME_WIDTH:
200 *((gint*)answer) = window->frame->rect.width;
201 break;
202 case META_CORE_GET_FRAME_HEIGHT:
203 *((gint*)answer) = window->frame->rect.height;
204 break;
205 case META_CORE_GET_THEME_VARIANT:
206 *((char**)answer) = window->gtk_theme_variant;
207 break;
208 case META_CORE_GET_SCREEN_WIDTH:
209 *((gint*)answer) = window->screen->rect.width;
210 break;
211 case META_CORE_GET_SCREEN_HEIGHT:
212 *((gint*)answer) = window->screen->rect.height;
213 break;
214
215 default:
216 meta_warning(_("Unknown window information request: %d"), request);
217 }
218
219 request = va_arg (args, MetaCoreGetType);
220 }
221
222 out:
223 va_end (args);
224}

267void
268meta_core_user_lower_and_unfocus (Display *xdisplay,
269 Window frame_xwindow,
270 guint32 timestamp)
271{
272 MetaWindow *window = get_window (xdisplay, frame_xwindow);
273
274 meta_window_lower (window);
275
276 if (meta_prefs_get_raise_on_click ())
277 {
278 /* Move window to the back of the focusing workspace's MRU list.
279 * Do extra sanity checks to avoid possible race conditions.
280 * (Borrowed from window.c.)
281 */
282 if (window->screen->active_workspace &&
283 meta_window_located_on_workspace (window,
284 window->screen->active_workspace))
285 {
286 GList* link;
287 link = g_list_find (window->screen->active_workspace->mru_list,
288 window);
289 g_assert (link);
290
291 window->screen->active_workspace->mru_list =
292 g_list_remove_link (window->screen->active_workspace->mru_list,
293 link);
294 g_list_free (link);
295
296 window->screen->active_workspace->mru_list =
297 g_list_append (window->screen->active_workspace->mru_list,
298 window);
299 }
300 }
301
302 /* focus the default window, if needed */
303 if (window->has_focus)
304 meta_workspace_focus_default_window (window->screen->active_workspace,
305 NULL,
306 timestamp);
307}

309void
310meta_core_lower_beneath_focus_window (Display *xdisplay,
311 Window xwindow,
312 guint32 timestamp)
313{
314 XWindowChanges changes;
315 MetaDisplay *display;
316 MetaScreen *screen;
317 MetaWindow *focus_window;
318
319 display = meta_display_for_x_display (xdisplay);
320 screen = meta_display_screen_for_xwindow (display, xwindow);
321 focus_window = meta_stack_get_top (screen->stack);
322
323 if (focus_window == NULL)
324 return;
325
326 changes.stack_mode = Below;
327 changes.sibling = focus_window->frame ? focus_window->frame->xwindow
328 : focus_window->xwindow;
329
330 meta_error_trap_push (display);
331 XConfigureWindow (xdisplay,
332 xwindow,
333 CWSibling | CWStackMode,
334 &changes);
335 meta_error_trap_pop (display, FALSE);
336}

549void
550meta_core_get_menu_accelerator (MetaMenuOp menu_op,
551 int workspace,
552 unsigned int *keysym,
553 MetaVirtualModifier *modifiers)
554{
555 const char *name;
556
557 name = NULL;
558
559 switch (menu_op)
560 {
561 case META_MENU_OP_NONE:
562 /* No keybinding for this one */
563 break;
564 case META_MENU_OP_DELETE:
565 name = "close";
566 break;
567 case META_MENU_OP_MINIMIZE:
568 name = "minimize";
569 break;
570 case META_MENU_OP_UNMAXIMIZE:
571 name = "unmaximize";
572 break;
573 case META_MENU_OP_MAXIMIZE:
574 name = "maximize";
575 break;
576 case META_MENU_OP_UNSHADE:
577 case META_MENU_OP_SHADE:
578 name = "toggle-shaded";
579 break;
580 case META_MENU_OP_UNSTICK:
581 case META_MENU_OP_STICK:
582 name = "toggle-on-all-workspaces";
583 break;
584 case META_MENU_OP_ABOVE:
585 case META_MENU_OP_UNABOVE:
586 name = "toggle-above";
587 break;
588 case META_MENU_OP_WORKSPACES:
589 switch (workspace)
590 {
591 case 1:
592 name = "move-to-workspace-1";
593 break;
594 case 2:
595 name = "move-to-workspace-2";
596 break;
597 case 3:
598 name = "move-to-workspace-3";
599 break;
600 case 4:
601 name = "move-to-workspace-4";
602 break;
603 case 5:
604 name = "move-to-workspace-5";
605 break;
606 case 6:
607 name = "move-to-workspace-6";
608 break;
609 case 7:
610 name = "move-to-workspace-7";
611 break;
612 case 8:
613 name = "move-to-workspace-8";
614 break;
615 case 9:
616 name = "move-to-workspace-9";
617 break;
618 case 10:
619 name = "move-to-workspace-10";
620 break;
621 case 11:
622 name = "move-to-workspace-11";
623 break;
624 case 12:
625 name = "move-to-workspace-12";
626 break;
627 }
628 break;
629 case META_MENU_OP_MOVE:
630 name = "begin-move";
631 break;
632 case META_MENU_OP_RESIZE:
633 name = "begin-resize";
634 break;
635 case META_MENU_OP_MOVE_LEFT:
636 name = "move-to-workspace-left";
637 break;
638 case META_MENU_OP_MOVE_RIGHT:
639 name = "move-to-workspace-right";
640 break;
641 case META_MENU_OP_MOVE_UP:
642 name = "move-to-workspace-up";
643 break;
644 case META_MENU_OP_MOVE_DOWN:
645 name = "move-to-workspace-down";
646 break;
647 case META_MENU_OP_RECOVER:
648 /* No keybinding for this one */
649 break;
650 }
651
652 if (name)
653 {
654 meta_prefs_get_window_binding (name, keysym, modifiers);
655 }
656 else
657 {
658 *keysym = 0;
659 *modifiers = 0;
660 }
661}

794void
795meta_invalidate_default_icons (void)
796{
797 MetaDisplay *display = meta_get_display ();
798 GSList *windows;
799 GSList *l;
800
801 if (display == NULL)
802 return; /* We can validly be called before the display is opened. */
803
804 windows = meta_display_list_windows (display);
805 for (l = windows; l != NULL; l = l->next)
806 {
807 MetaWindow *window = (MetaWindow*)l->data;
808
809 if (window->icon_cache.origin == USING_FALLBACK_ICON)
810 {
