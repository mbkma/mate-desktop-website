271static gsize
272read_line (PlumaDocumentInputStream *stream,
273 gchar *outbuf,
274 gsize space_left)
275{
276 GtkTextIter start, next, end;
277 gchar *buf;
278 gint bytes; /* int since it's what iter_get_offset returns */
279 gsize bytes_to_write, newline_size, read;
280 const gchar *newline;
281 gboolean is_last;
282
283 gtk_text_buffer_get_iter_at_mark (stream->priv->buffer,
284 &start,
285 stream->priv->pos);
286
287 if (gtk_text_iter_is_end (&start))
288 return 0;
289
290 end = next = start;
291 newline = get_new_line (stream);
292
293 /* Check needed for empty lines */
294 if (!gtk_text_iter_ends_line (&end))
295 gtk_text_iter_forward_to_line_end (&end);
296
297 gtk_text_iter_forward_line (&next);
298
299 buf = gtk_text_iter_get_slice (&start, &end);
300
301 /* the bytes of a line includes also the newline, so with the
302 offsets we remove the newline and we add the new newline size */
303 bytes = gtk_text_iter_get_bytes_in_line (&start) - stream->priv->bytes_partial;
304
305 /* bytes_in_line includes the newlines, so we remove that assuming that
306 they are single byte characters */
307 bytes = bytes - (gtk_text_iter_get_offset (&next) - gtk_text_iter_get_offset (&end));
308 is_last = gtk_text_iter_is_end (&end);
309
310 /* bytes_to_write contains the amount of bytes we would like to write.
311 This means its the amount of bytes in the line (without the newline
312 in the buffer) + the amount of bytes for the newline we want to
313 write (newline_size) */
314 bytes_to_write = bytes;
315
316 /* do not add the new newline_size for the last line */
317 newline_size = get_new_line_size (stream);
318 if (!is_last)
319 bytes_to_write += newline_size;
320
321 if (bytes_to_write > space_left)
322 {
323 gchar *ptr;
324 gint char_offset;
325 gint written;
326 gsize to_write;
327
328 /* Here the line does not fit in the buffer, we thus write
329 the amount of bytes we can still fit, storing the position
330 for the next read with the mark. Do not try to write the
331 new newline in this case, it will be handled in the next
332 iteration */
333 to_write = MIN (space_left, bytes);
334 ptr = buf;
335 written = 0;
336 char_offset = 0;
337
338 while (written < to_write)
339 {
340 gint w;
341
342 ptr = g_utf8_next_char (ptr);
343 w = (ptr - buf);
344 if (w > to_write)
345 {
346 break;
347 }
348 else
349 {
350 written = w;
351 ++char_offset;
352 }
353 }
354
355 memcpy (outbuf, buf, written);
356
357 /* Note: offset is one past what we wrote */
358 gtk_text_iter_forward_chars (&start, char_offset);
359 stream->priv->bytes_partial += written;
360 read = written;
361 }
362 else
363 {
364 /* First just copy the bytes without the newline */
365 memcpy (outbuf, buf, bytes);
366
367 /* Then add the newline, but not for the last line */
368 if (!is_last)
369 {
370 memcpy (outbuf + bytes, newline, newline_size);
371 }
372
373 start = next;
374 stream->priv->bytes_partial = 0;
375 read = bytes_to_write;
376 }
377
378 gtk_text_buffer_move_mark (stream->priv->buffer,
379 stream->priv->pos,
380 &start);
381
382 g_free (buf);
383 return read;
384}

386static gssize
387pluma_document_input_stream_read (GInputStream *stream,
388 void *buffer,
389 gsize count,
390 GCancellable *cancellable,
391 GError **error)
392{
393 PlumaDocumentInputStream *dstream;
394 GtkTextIter iter;
395 gssize space_left, read, n;
396
397 dstream = PLUMA_DOCUMENT_INPUT_STREAM (stream);
398
399 if (count < 6)
400 {
401 g_set_error_literal (error, G_IO_ERROR, G_IO_ERROR_NO_SPACE,
402 "Not enougth space in destination");
403 return -1;
404 }
405
406 if (g_cancellable_set_error_if_cancelled (cancellable, error))
407 return -1;
408
409 /* Initialize the mark to the first char in the text buffer */
410 if (!dstream->priv->is_initialized)
411 {
412 gtk_text_buffer_get_start_iter (dstream->priv->buffer, &iter);
413 dstream->priv->pos = gtk_text_buffer_create_mark (dstream->priv->buffer,
414 NULL,
415 &iter,
416 FALSE);
417
418 dstream->priv->is_initialized = TRUE;
419 }
420
421 space_left = count;
422 read = 0;
423
424 do
425 {
426 n = read_line (dstream, (void *) ((gsize) buffer + read), space_left);
427 read += n;
428 space_left -= n;
429 } while (space_left > 0 && n != 0 && dstream->priv->bytes_partial == 0);
430
431 /* Make sure that non-empty files are always terminated with \n (see bug #95676).
432 * Note that we strip the trailing \n when loading the file */
433 gtk_text_buffer_get_iter_at_mark (dstream->priv->buffer,
434 &iter,
435 dstream->priv->pos);
436
437 if (gtk_text_iter_is_end (&iter) &&
438 !gtk_text_iter_is_start (&iter))
439 {
440 gssize newline_size;
441
442 newline_size = get_new_line_size (dstream);
443
444 if (space_left >= newline_size &&
445 !dstream->priv->newline_added)
446 {
447 const gchar *newline;
448
449 newline = get_new_line (dstream);
450
451 memcpy ((void *) ((gsize) buffer + read), newline, newline_size);
452
453 read += newline_size;
454 dstream->priv->newline_added = TRUE;
455 }
456 }
457
458 return read;
459}
