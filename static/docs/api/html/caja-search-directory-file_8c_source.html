111static CajaRequestStatus
112search_directory_file_get_deep_counts (CajaFile *file,
113 guint *directory_count,
114 guint *file_count,
115 guint *unreadable_directory_count,
116 goffset *total_size,
117 goffset *total_size_on_disk)
118{
119 GList *file_list, *l;
120 guint dirs, files;
121 GFileType type;
122 CajaFile *dir_file = NULL;
123
124 file_list = caja_directory_get_file_list (file->details->directory);
125
126 dirs = files = 0;
127 for (l = file_list; l != NULL; l = l->next)
128 {
129 dir_file = CAJA_FILE (l->data);
130 type = caja_file_get_file_type (dir_file);
131 if (type == G_FILE_TYPE_DIRECTORY)
132 {
133 dirs++;
134 }
135 else
136 {
137 files++;
138 }
139 }
140
141 if (directory_count != NULL)
142 {
143 *directory_count = dirs;
144 }
145 if (file_count != NULL)
146 {
147 *file_count = files;
148 }
149 if (unreadable_directory_count != NULL)
150 {
151 *unreadable_directory_count = 0;
152 }
153 if (total_size != NULL)
154 {
155 /* FIXME: Maybe we want to calculate this? */
156 *total_size = 0;
157 }
158 if (total_size_on_disk != NULL)
159 {
160 /* FIXME: Maybe we want to calculate this? */
161 *total_size_on_disk = 0;
162 }
163
164 caja_file_list_free (file_list);
165
166 return CAJA_REQUEST_DONE;
167}

175void
176caja_search_directory_file_update_display_name (CajaSearchDirectoryFile *search_file)
177{
178 CajaFile *file;
179 char *display_name;
180 gboolean changed;
181
182 display_name = NULL;
183 file = CAJA_FILE (search_file);
184 if (file->details->directory)
185 {
186 CajaSearchDirectory *search_dir;
187 CajaQuery *query;
188
189 search_dir = CAJA_SEARCH_DIRECTORY (file->details->directory);
190 query = caja_search_directory_get_query (search_dir);
191
192 if (query != NULL)
193 {
194 display_name = caja_query_to_readable_string (query);
195 g_object_unref (query);
196 }
197 }
198
199 if (display_name == NULL)
200 {
201 display_name = g_strdup (_("Search"));
202 }
203
204 changed = caja_file_set_display_name (file, display_name, NULL, TRUE);
205 if (changed)
206 {
207 caja_file_emit_changed (file);
208 }
209}
