85void
86matekbd_indicator_load_images ()
87{
88 int i;
89 GSList *image_filename;
90
91 globals.images = NULL;
92 matekbd_indicator_config_load_image_filenames (&globals.ind_cfg,
93 &globals.kbd_cfg);
94
95 if (!globals.ind_cfg.show_flags)
96 return;
97
98 image_filename = globals.ind_cfg.image_filenames;
99
100 for (i = xkl_engine_get_max_num_groups (globals.engine);
101 --i >= 0; image_filename = image_filename->next) {
102 GdkPixbuf *image = NULL;
103 char *image_file = (char *) image_filename->data;
104
105 if (image_file != NULL) {
106 GError *gerror = NULL;
107 image =
108 gdk_pixbuf_new_from_file (image_file, &gerror);
109 if (image == NULL) {
110 GtkWidget *dialog =
111 gtk_message_dialog_new (NULL,
112 GTK_DIALOG_DESTROY_WITH_PARENT,
113 GTK_MESSAGE_ERROR,
114 GTK_BUTTONS_OK,
115 _
116 ("There was an error loading an image: %s"),
117 gerror->
118 message);
119 g_signal_connect (G_OBJECT (dialog),
120 "response",
121 G_CALLBACK
122 (gtk_widget_destroy),
123 NULL);
124
125 gtk_window_set_resizable (GTK_WINDOW
126 (dialog), FALSE);
127
128 gtk_widget_show (dialog);
129 g_error_free (gerror);
130 }
131 xkl_debug (150,
132 "Image %d[%s] loaded -> %p[%dx%d]\n",
133 i, image_file, image,
134 gdk_pixbuf_get_width (image),
135 gdk_pixbuf_get_height (image));
136 }
137 /* We append the image anyway - even if it is NULL! */
138 globals.images = g_slist_append (globals.images, image);
139 }
140}

142static void
143matekbd_indicator_free_images ()
144{
145 GdkPixbuf *pi;
146 GSList *img_node;
147
148 matekbd_indicator_config_free_image_filenames (&globals.ind_cfg);
149
150 while ((img_node = globals.images) != NULL) {
151 pi = GDK_PIXBUF (img_node->data);
152 /* It can be NULL - some images may be missing */
153 if (pi != NULL) {
154 g_object_unref (pi);
155 }
156 globals.images =
157 g_slist_remove_link (globals.images, img_node);
158 g_slist_free_1 (img_node);
159 }

216static gboolean matekbd_indicator_key_pressed(GtkWidget* widget, GdkEventKey* event, MatekbdIndicator* gki)
217{
218 switch (event->keyval)
219 {
220 case GDK_KEY_KP_Enter:
221 case GDK_KEY_ISO_Enter:
222 case GDK_KEY_3270_Enter:
223 case GDK_KEY_Return:
224 case GDK_KEY_space:
225 case GDK_KEY_KP_Space:
226 matekbd_desktop_config_lock_next_group(&globals.cfg);
227 return TRUE;
228 default:
229 break;
230 }
231
232 return FALSE;
233}

253static void
254draw_flag (GtkWidget * flag, cairo_t * cr, GdkPixbuf * image)
255{
256 /* Image width and height */
257 int iw = gdk_pixbuf_get_width (image);
258 int ih = gdk_pixbuf_get_height (image);
259 GtkAllocation allocation;
260 double xwiratio, ywiratio, wiratio;
261
262 gtk_widget_get_allocation (flag, &allocation);
263
264 /* widget-to-image scales, X and Y */
265 xwiratio = 1.0 * allocation.width / iw;
266 ywiratio = 1.0 * allocation.height / ih;
267 wiratio = xwiratio < ywiratio ? xwiratio : ywiratio;
268
269 /* transform cairo context */
270 cairo_translate (cr, allocation.width / 2.0, allocation.height / 2.0);
271 cairo_scale (cr, wiratio, wiratio);
272 cairo_translate (cr, - iw / 2.0, - ih / 2.0);
273
274 gdk_cairo_set_source_pixbuf (cr, image, 0, 0);
275 cairo_paint (cr);
276}

278gchar *
279matekbd_indicator_extract_layout_name (int group, XklEngine * engine,
280 MatekbdKeyboardConfig * kbd_cfg,
281 gchar ** short_group_names,
282 gchar ** full_group_names)
283{
284 char *layout_name = NULL;
285 if (group < g_strv_length (short_group_names)) {
286 if (xkl_engine_get_features (engine) &
287 XKLF_MULTIPLE_LAYOUTS_SUPPORTED) {
288 char *full_layout_name =
289 kbd_cfg->layouts_variants[group];
290 char *variant_name;
291 if (!matekbd_keyboard_config_split_items
292 (full_layout_name, &layout_name,
293 &variant_name))
294 /* just in case */
295 layout_name = full_layout_name;
296
297 /* make it freeable */
298 layout_name = g_strdup (layout_name);
299
300 if (short_group_names != NULL) {
301 char *short_group_name =
302 short_group_names[group];
303 if (short_group_name != NULL
304 && *short_group_name != '\0') {
305 /* drop the long name */
306 g_free (layout_name);
307 layout_name =
308 g_strdup (short_group_name);
309 }
310 }
311 } else {
312 layout_name = g_strdup (full_group_names[group]);
313 }
314 }
315
316 if (layout_name == NULL)
317 layout_name = g_strdup ("");
318
319 return layout_name;
320}

322gchar *
323matekbd_indicator_create_label_title (int group, GHashTable ** ln2cnt_map,
324 gchar * layout_name)
325{
326 gpointer pcounter = NULL;
327 char *prev_layout_name = NULL;
328 char *lbl_title = NULL;
329 int counter = 0;
330
331 if (group == 0) {
332 *ln2cnt_map =
333 g_hash_table_new_full (g_str_hash, g_str_equal,
334 g_free, NULL);
335 }
336
337 /* Process layouts with repeating description */
338 if (g_hash_table_lookup_extended
339 (*ln2cnt_map, layout_name, (gpointer *) & prev_layout_name,
340 &pcounter)) {
341 /* "next" same description */
342 gchar appendix[10] = "";
343 gint utf8length;
344 gunichar cidx;
345 counter = GPOINTER_TO_INT (pcounter);
346 /* Unicode subscript 2, 3, 4 */
347 cidx = 0x2081 + counter;
348 utf8length = g_unichar_to_utf8 (cidx, appendix);
349 appendix[utf8length] = '\0';
350 lbl_title = g_strconcat (layout_name, appendix, NULL);
351 } else {
352 /* "first" time this description */
353 lbl_title = g_strdup (layout_name);
354 }
355 g_hash_table_insert (*ln2cnt_map, layout_name,
356 GINT_TO_POINTER (counter + 1));
357 return lbl_title;
358}

360static GtkWidget *
361matekbd_indicator_prepare_drawing (MatekbdIndicator * gki, int group)
362{
363 gpointer pimage;
364 GdkPixbuf *image;
365 GtkWidget *ebox;
366
367 pimage = g_slist_nth_data (globals.images, group);
368 ebox = gtk_event_box_new ();
369 gtk_event_box_set_visible_window (GTK_EVENT_BOX (ebox), FALSE);
370 if (globals.ind_cfg.show_flags) {
371 GtkWidget *flag;
372 if (pimage == NULL)
373 return NULL;
374 image = GDK_PIXBUF (pimage);
375 flag = gtk_drawing_area_new ();
376 gtk_widget_add_events (GTK_WIDGET (flag),
377 GDK_BUTTON_PRESS_MASK);
378 g_signal_connect (G_OBJECT (flag), "draw",
379 G_CALLBACK (draw_flag), image);
380 gtk_container_add (GTK_CONTAINER (ebox), flag);
381 } else {
382 char *lbl_title = NULL;
383 char *layout_name = NULL;
384 GtkWidget *label;
385 static GHashTable *ln2cnt_map = NULL;
386
387 layout_name =
388 matekbd_indicator_extract_layout_name (group,
389 globals.engine,
390 &globals.kbd_cfg,
391 globals.short_group_names,
392 globals.full_group_names);
393
394 lbl_title =
395 matekbd_indicator_create_label_title (group,
396 &ln2cnt_map,
397 layout_name);
398
399 label = gtk_label_new (lbl_title);
400 gtk_widget_set_halign (label, GTK_ALIGN_CENTER);
401 gtk_widget_set_valign (label, GTK_ALIGN_CENTER);
402 gtk_widget_set_margin_start (label, 2);
403 gtk_widget_set_margin_end (label, 2);
404 gtk_widget_set_margin_top (label, 2);
405 gtk_widget_set_margin_bottom (label, 2);
406 g_free (lbl_title);
407 gtk_label_set_angle (GTK_LABEL (label), gki->priv->angle);
408
409 if (group + 1 ==
410 xkl_engine_get_num_groups (globals.engine)) {
411 g_hash_table_destroy (ln2cnt_map);
412 ln2cnt_map = NULL;
413 }
414
415 gtk_container_add (GTK_CONTAINER (ebox), label);
416 }
417
418 g_signal_connect (G_OBJECT (ebox),
419 "button_press_event",
420 G_CALLBACK (matekbd_indicator_button_pressed), gki);
421
422 g_signal_connect (G_OBJECT (gki),
423 "key_press_event",
424 G_CALLBACK (matekbd_indicator_key_pressed), gki);
425
426 /* We have everything prepared for that size */
427
428 return ebox;
429}

431static void
432matekbd_indicator_update_tooltips (MatekbdIndicator * gki)
433{
434 XklState *state = xkl_engine_get_current_state (globals.engine);
435 gchar *buf;
436 if (state == NULL || state->group < 0
437 || state->group >= g_strv_length (globals.full_group_names))
438 return;

440 buf = g_strdup_printf (globals.tooltips_format,
441 globals.full_group_names[state->group]);
442

444 g_free (buf);
445}

496static void
497matekbd_indicator_load_group_names (const gchar ** layout_ids,
498 const gchar ** variant_ids)
499{
500 if (!matekbd_desktop_config_load_group_descriptions
501 (&globals.cfg, globals.registry, layout_ids, variant_ids,
502 &globals.short_group_names, &globals.full_group_names)) {
503 /* We just populate no short names (remain NULL) -
504 * full names are going to be used anyway */
505 gint i, total_groups =
506 xkl_engine_get_num_groups (globals.engine);
507 globals.full_group_names =
508 g_new0 (gchar *, total_groups + 1);
509
510 if (xkl_engine_get_features (globals.engine) &
511 XKLF_MULTIPLE_LAYOUTS_SUPPORTED) {
512 gchar **lst = globals.kbd_cfg.layouts_variants;
513 for (i = 0; *lst; lst++, i++) {
514 globals.full_group_names[i] =
515 g_strdup ((char *) *lst);
516 }
517 } else {
518 for (i = total_groups; --i >= 0;) {
519 globals.full_group_names[i] =
520 g_strdup_printf ("Group %d", i);
521 }
522 }
523 }
524}

595/* Should be called once for all widgets */
596static GdkFilterReturn
597matekbd_indicator_filter_x_evt (GdkXEvent * xev, GdkEvent * event)
598{
599 XEvent *xevent = (XEvent *) xev;
600
601 xkl_engine_filter_events (globals.engine, xevent);
602 switch (xevent->type) {
603 case ReparentNotify:
604 {
605 XReparentEvent *rne = (XReparentEvent *) xev;
606
607 ForAllIndicators () {
608 GdkWindow *w =
609 gtk_widget_get_parent_window
610 (GTK_WIDGET (gki));
611
612 /* compare the indicator's parent window with the even window */
613 if (w != NULL
614 && GDK_WINDOW_XID (w) == rne->window) {
615 /* if so - make it transparent... */
616 xkl_engine_set_window_transparent
617 (globals.engine, rne->window,
618 TRUE);
619 }
620 }
621 NextIndicator ()
622 }
623 break;
624 }
625 return GDK_FILTER_CONTINUE;
626}

662static void matekbd_indicator_init(MatekbdIndicator* gki)
663{
664 GtkWidget *def_drawing;
665 GtkNotebook *notebook;
666
667 if (!g_slist_length(globals.widget_instances))
668 {
669 matekbd_indicator_global_init();
670 }
671
672 gki->priv = g_new0 (MatekbdIndicatorPrivate, 1);
673
674 notebook = GTK_NOTEBOOK (gki);
675
676 xkl_debug (100, "Initiating the widget startup process for %p\n", gki);
677
678 gtk_notebook_set_show_tabs (notebook, FALSE);
679 gtk_notebook_set_show_border (notebook, FALSE);
680
681 def_drawing =
682 gtk_image_new_from_icon_name ("process-stop",
683 GTK_ICON_SIZE_BUTTON);
684
685 gtk_notebook_append_page (notebook, def_drawing,
686 gtk_label_new (""));
687
688 if (globals.engine == NULL) {
689 matekbd_indicator_set_tooltips (gki,
690 _
691 ("XKB initialization error"));
692 return;
693 }
694
695 matekbd_indicator_set_tooltips (gki, NULL);
696
697 matekbd_indicator_fill (gki);
698 matekbd_indicator_set_current_page (gki);
699
700 gtk_widget_add_events (GTK_WIDGET (gki), GDK_BUTTON_PRESS_MASK);
701
702 /* append AFTER all initialization work is finished */
703 globals.widget_instances =
704 g_slist_append (globals.widget_instances, gki);
705}

707static void
708matekbd_indicator_finalize (GObject * obj)
709{
