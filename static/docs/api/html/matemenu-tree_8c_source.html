264static void
265matemenu_tree_add_menu_file_monitor (MateMenuTree *tree,
266 const char *path,
267 MenuFileMonitorType type)
268{
269 MenuFileMonitor *monitor;
270
271 monitor = g_slice_new0 (MenuFileMonitor);
272
273 monitor->type = type;
274
275 switch (type)
276 {
277 case MENU_FILE_MONITOR_FILE:
278 menu_verbose ("Adding a menu file monitor for \"%s\"\n", path);
279
280 monitor->monitor = menu_get_file_monitor (path);
281 menu_monitor_add_notify (monitor->monitor,
282 (MenuMonitorNotifyFunc) handle_menu_file_changed,
283 tree);
284 break;
285
286 case MENU_FILE_MONITOR_NONEXISTENT_FILE:
287 menu_verbose ("Adding a menu file monitor for non-existent \"%s\"\n", path);
288
289 monitor->monitor = menu_get_file_monitor (path);
290 menu_monitor_add_notify (monitor->monitor,
291 (MenuMonitorNotifyFunc) handle_nonexistent_menu_file_changed,
292 tree);
293 break;
294
295 case MENU_FILE_MONITOR_DIRECTORY:
296 menu_verbose ("Adding a menu directory monitor for \"%s\"\n", path);
297
298 monitor->monitor = menu_get_directory_monitor (path);
299 menu_monitor_add_notify (monitor->monitor,
300 (MenuMonitorNotifyFunc) handle_menu_file_directory_changed,
301 tree);
302 break;
303
304 default:
305 g_assert_not_reached ();
306 break;
307 }
308
309 tree->menu_file_monitors = g_slist_prepend (tree->menu_file_monitors, monitor);
310}

312static void
313remove_menu_file_monitor (MenuFileMonitor *monitor,
314 MateMenuTree *tree)
315{
316 switch (monitor->type)
317 {
318 case MENU_FILE_MONITOR_FILE:
319 menu_monitor_remove_notify (monitor->monitor,
320 (MenuMonitorNotifyFunc) handle_menu_file_changed,
321 tree);
322 break;
323
324 case MENU_FILE_MONITOR_NONEXISTENT_FILE:
325 menu_monitor_remove_notify (monitor->monitor,
326 (MenuMonitorNotifyFunc) handle_nonexistent_menu_file_changed,
327 tree);
328 break;
329
330 case MENU_FILE_MONITOR_DIRECTORY:
331 menu_monitor_remove_notify (monitor->monitor,
332 (MenuMonitorNotifyFunc) handle_menu_file_directory_changed,
333 tree);
334 break;
335
336 default:
337 g_assert_not_reached ();
338 break;
339 }
340
341 menu_monitor_unref (monitor->monitor);
342 monitor->monitor = NULL;
343
344 monitor->type = MENU_FILE_MONITOR_INVALID;
345
346 g_slice_free (MenuFileMonitor, monitor);
347}

361static gboolean
362canonicalize_path (MateMenuTree *tree,
363 const char *path)
364{
365 tree->canonical_path = realpath (path, NULL);
366 if (tree->canonical_path)
367 {
368 tree->canonical = TRUE;
369 matemenu_tree_add_menu_file_monitor (tree,
370 tree->canonical_path,
371 MENU_FILE_MONITOR_FILE);
372 }
373 else
374 {
375 matemenu_tree_add_menu_file_monitor (tree,
376 path,
377 MENU_FILE_MONITOR_NONEXISTENT_FILE);
378 }
379
380 return tree->canonical;
381}

398static void
399canonicalize_basename (MateMenuTree *tree,
400 const char *basename)
401{
402 if (!canonicalize_basename_with_config_dir (tree,
403 basename,
404 g_get_user_config_dir ()))
405 {
406 const char * const *system_config_dirs;
407 int i;
408
409 system_config_dirs = g_get_system_config_dirs ();
410
411 i = 0;
412 while (system_config_dirs[i] != NULL)
413 {
414 if (canonicalize_basename_with_config_dir (tree,
415 basename,
416 system_config_dirs[i]))
417 break;
418
419 ++i;
420 }
421 }

424static gboolean matemenu_tree_canonicalize_path(MateMenuTree* tree,
425 GError **error)
426{
427 const char *menu_file = NULL;
428
429 if (tree->canonical)
430 return TRUE;
431
432 g_assert(tree->canonical_path == NULL);
433
434 matemenu_tree_remove_menu_file_monitors (tree);
435
436 if (tree->path)
437 {
438 menu_file = tree->path;
439 canonicalize_path (tree, tree->path);
440 }
441 else
442 {
443 const gchar *xdg_menu_prefix;
444
445 menu_file = tree->basename;
446 xdg_menu_prefix = g_getenv ("XDG_MENU_PREFIX");
447
448 if (xdg_menu_prefix != NULL)
449 {
450 gchar *prefixed_basename;
451
452 prefixed_basename = g_strdup_printf ("%sapplications.menu",
453 xdg_menu_prefix);
454
455 /* Some gnome-menus using applications just use "applications.menu"
456 * as the basename and expect gnome-menus to prefix it. Others (e.g.
457 * Alacarte) explicitly use "${XDG_MENU_PREFIX}applications.menu" as
458 * the basename, because they want to save changes to the right files
459 * in ~. In both cases, we want to use "applications-merged" as the
460 * merge directory (as required by the fd.o menu spec), so we save
461 * the non-prefixed basename and use it later when calling
462 * menu_layout_load().
463 */
464 if (!g_strcmp0 (tree->basename, "mate-applications.menu") ||
465 !g_strcmp0 (tree->basename, prefixed_basename))
466 {
467 canonicalize_basename (tree, prefixed_basename);
468 g_free (tree->non_prefixed_basename);
469 tree->non_prefixed_basename = g_strdup ("mate-applications.menu");
470 }
471 g_free (prefixed_basename);
472 }
473
474 if (!tree->canonical)
475 canonicalize_basename (tree, tree->basename);
476 }
477
478 if (tree->canonical)
479 {
480 menu_verbose ("Successfully looked up menu_file for \"%s\": %s\n",
481 menu_file, tree->canonical_path);
482 return TRUE;
483 }
484 else
485 {
486 g_set_error (error,
487 G_IO_ERROR,
488 G_IO_ERROR_FAILED,
489 "Failed to look up menu_file for \"%s\"\n",
490 menu_file);
491 return FALSE;
492 }
493}

549static GObject *
550matemenu_tree_constructor (GType type,
551 guint n_construct_properties,
552 GObjectConstructParam *construct_properties)
553{
554 GObject *obj;
555 MateMenuTree *self;
556
557 obj = G_OBJECT_CLASS (matemenu_tree_parent_class)->constructor (type,
558 n_construct_properties,
559 construct_properties);
560
561 /* If MateMenuTree:menu-path is set, then we should make sure that
562 * MateMenuTree:menu-basename is unset (especially as it has a default
563 * value). This has to be done here, in the constructor, since the
564 * properties are construct-only. */
565
566 self = MATEMENU_TREE (obj);
567
568 if (self->path != NULL)
569 g_object_set (self, "menu-basename", NULL, NULL);
570
571 return obj;
572}

629{

650 tree->entries_by_id = NULL;

652#ifdef WITH_COLLECTION

662
663 g_object_unref (tree->settings);

665 g_source_remove (tree->emit_changed_signal);
666 tree->emit_changed_signal = 0;
667#endif /* WITH_COLLECTION */
668
669 G_OBJECT_CLASS (matemenu_tree_parent_class)->finalize (object);

676{
677 ObjectType object_type;
678 char *object_path;
679 GSettings *settings;
680
681 object_path = g_strdup_printf ("/org/mate/panel/objects/%s/", id);
682 settings = g_settings_new_with_path ("org.mate.panel.object", object_path);
683
684 object_type = g_settings_get_enum (settings, "object-type");
685 if (object_type == OBJECT_LAUNCHER)
686 {
687 char *location;
688 char *desktop_name;
689
690 if (self->collection_applet == NULL)
691 self->collection_applet = g_ptr_array_new_with_free_func ((GDestroyNotify) g_free);
692
693 location = g_settings_get_string (settings, "launcher-location");
694 desktop_name = g_path_get_basename (location);
695 if (strstr (desktop_name, "-1.") != NULL )
696 {
697 char **str;
698
699 str = g_strsplit (desktop_name, "-1.", -1);
700 g_free (desktop_name);
701 desktop_name = g_strdup_printf ("%s.%s", str[0], str[1]);
702 g_strfreev (str);
703 }
704 g_ptr_array_add (self->collection_applet, desktop_name);
705 g_free (location);
706 }
707 g_free (object_path);
708 g_object_unref (settings);
709}

898static MateMenuTreeDirectory *
899find_path (MateMenuTreeDirectory *directory,
900 const char *path)
901{
902 const char *name;
903 char *slash;
904 char *freeme;
905 GSList *tmp;
906
907 while (path[0] == G_DIR_SEPARATOR) path++;
908
909 if (path[0] == '\0')
910 return directory;
911
912 freeme = NULL;
913 slash = strchr (path, G_DIR_SEPARATOR);
914 if (slash)
915 {
916 name = freeme = g_strndup (path, (gsize)(slash - path));
917 path = slash + 1;
918 }
919 else
920 {
921 name = path;
922 path = NULL;
923 }
924
925 tmp = directory->contents;
926 while (tmp != NULL)
927 {
928 MateMenuTreeItem *item = tmp->data;
929
930 if (item->type != MATEMENU_TREE_ITEM_DIRECTORY)
931 {
932 tmp = tmp->next;
933 continue;
934 }
935
936 if (!strcmp (name, MATEMENU_TREE_DIRECTORY (item)->name))
937 {
938 g_free (freeme);
939
940 if (path)
941 return find_path (MATEMENU_TREE_DIRECTORY (item), path);
942 else
943 return MATEMENU_TREE_DIRECTORY (item);
944 }
945
946 tmp = tmp->next;
947 }
948
949 g_free (freeme);
950
951 return NULL;
952}

2882static void
2883matemenu_tree_strip_duplicate_children (MateMenuTree *tree,
2884 MenuLayoutNode *layout)
2885{
2886 MenuLayoutNode *child;
2887 GSList *simple_nodes;
2888 GSList *menu_layout_nodes;
2889 GSList *prev;
2890 GSList *tmp;
2891
2892 /* to strip dups, we find all the child nodes where
2893 * we want to kill dups, sort them,
2894 * then nuke the adjacent nodes that are equal
2895 */
2896
2897 simple_nodes = NULL;
2898 menu_layout_nodes = NULL;
2899
2900 child = menu_layout_node_get_children (layout);
2901 while (child != NULL)
2902 {
2903 switch (menu_layout_node_get_type (child))
2904 {
2905 /* These are dups if their content is the same */
2906 case MENU_LAYOUT_NODE_APP_DIR:
2907 case MENU_LAYOUT_NODE_DIRECTORY_DIR:
2908 case MENU_LAYOUT_NODE_DIRECTORY:
2909 simple_nodes = g_slist_prepend (simple_nodes, child);
2910 break;
2911
2912 /* These have to be merged in a more complicated way,
2913 * and then recursed
2914 */
2915 case MENU_LAYOUT_NODE_MENU:
2916 menu_layout_nodes = g_slist_prepend (menu_layout_nodes, child);
2917 break;
2918
2919 default:
2920 break;
2921 }
2922
2923 child = menu_layout_node_get_next (child);
2924 }
2925
2926 /* Note that the lists are all backward. So we want to keep
2927 * the items that are earlier in the list, because they were
2928 * later in the file
2929 */
2930
2931 /* stable sort the simple nodes */
2932 simple_nodes = g_slist_sort (simple_nodes,
2933 node_compare_func);
2934
2935 prev = NULL;
2936 tmp = simple_nodes;
2937 while (tmp != NULL)
2938 {
2939 GSList *next = tmp->next;
2940
2941 if (prev)
2942 {
2943 MenuLayoutNode *p = prev->data;
2944 MenuLayoutNode *n = tmp->data;
2945
2946 if (node_compare_func (p, n) == 0)
2947 {
2948 /* nuke it! */
2949 menu_layout_node_unlink (n);
2950 simple_nodes = g_slist_delete_link (simple_nodes, tmp);
2951 tmp = prev;
2952 }
2953 }
2954
2955 prev = tmp;
2956 tmp = next;
2957 }
2958
2959 g_slist_free (simple_nodes);
2960 simple_nodes = NULL;
2961
2962 /* stable sort the menu nodes (the sort includes the
2963 * parents of the nodes in the comparison). Remember
2964 * the list is backward.
2965 */
2966 menu_layout_nodes = g_slist_sort (menu_layout_nodes,
2967 node_menu_compare_func);
2968
2969 prev = NULL;
2970 tmp = menu_layout_nodes;
2971 while (tmp != NULL)
2972 {
2973 GSList *next = tmp->next;
2974
2975 if (prev)
2976 {
2977 MenuLayoutNode *p = prev->data;
2978 MenuLayoutNode *n = tmp->data;
2979
2980 if (node_menu_compare_func (p, n) == 0)
2981 {
2982 /* Move children of first menu to the start of second
2983 * menu and nuke the first menu
2984 */
2985 move_children (n, p);
2986 menu_layout_node_unlink (n);
2987 menu_layout_nodes = g_slist_delete_link (menu_layout_nodes, tmp);
2988 tmp = prev;
2989 }
2990 }
2991
2992 prev = tmp;
2993 tmp = next;
2994 }
2995
2996 g_slist_free (menu_layout_nodes);
2997 menu_layout_nodes = NULL;
2998
2999 /* Recursively clean up all children */
3000 child = menu_layout_node_get_children (layout);
3001 while (child != NULL)
3002 {
3003 if (menu_layout_node_get_type (child) == MENU_LAYOUT_NODE_MENU)
3004 matemenu_tree_strip_duplicate_children (tree, child);
3005
3006 child = menu_layout_node_get_next (child);
3007 }

3311

3313
3314 child = menu_layout_node_get_children (layout);
3315 while (child != NULL)
3316 {
3317 DesktopEntrySet *child_set;
3318
3319 child_set = process_include_rules (child, entry_pool);
3320
3321 if (set == NULL)
3322 {
3323 set = child_set;
3324 }
3325 else
3326 {
3327 desktop_entry_set_intersection (set, child_set);
3328 desktop_entry_set_unref (child_set);

3332 * because it's an AND

3336
3337 child = menu_layout_node_get_next (child);
3338 }
3339 menu_verbose ("Processed <And>\n");

3345 MenuLayoutNode *child;

3363 desktop_entry_set_unref (child_set);

3365
3366 child = menu_layout_node_get_next (child);
3367 }
3368 menu_verbose ("Processed <Or>\n");
3369 }

3372 case MENU_LAYOUT_NODE_NOT:
3373 {
3374 /* First get the OR of all the rules */
3375 MenuLayoutNode *child;
3376
3377 menu_verbose ("Processing <Not>\n");
3378

3380 while (child != NULL)

3383

3400 {

3402

3406 desktop_entry_set_subtract (inverted, set);

4169static void
4170preprocess_layout_info (MateMenuTree *tree,
4171 MateMenuTreeDirectory *directory)
4172{
4173 GSList *tmp;
4174 GSList *layout_info;
4175 gboolean using_default_layout;
4176 GSList *last_subdir;
4177 gboolean strip_duplicates;
4178 gboolean contents_added;
4179 gboolean should_remove;
4180 GSList *subdirs_sentinel;
4181
4182 /* Note: we need to preprocess all menus, even if the layout mask for a menu
4183 * is MENU_LAYOUT_VALUES_NONE: in this case, we need to remove empty menus;
4184 * and the layout mask can be different for a submenu anyway */
4185
4186 menu_verbose ("Processing menu layout inline hints for %s\n", directory->name);
4187 g_assert (!directory->preprocessed);
4188
4189 strip_duplicates = FALSE;
4190 /* we use last_subdir to track the last non-inlined subdirectory */
4191 last_subdir = g_slist_last (directory->subdirs);
4192
4193 /*
4194 * First process subdirectories with explicit layout
4195 */
4196 layout_info = get_layout_info (directory, &using_default_layout);
4197 tmp = layout_info;
4198 /* see comment below about Menuname to understand why we leave the loop if
4199 * last_subdir is NULL */
4200 while (tmp != NULL && last_subdir != NULL)
4201 {
4202 MenuLayoutNode *node = tmp->data;
4203 MenuLayoutValues layout_values;
4204 const char *name;
4205 MateMenuTreeDirectory *subdir;
4206 GSList *subdir_l;
4207
4208 tmp = tmp->next;
4209
4210 /* only Menuname nodes are relevant here */
4211 if (menu_layout_node_get_type (node) != MENU_LAYOUT_NODE_MENUNAME)
4212 continue;
4213
4214 get_values_with_defaults (node,
4215 &layout_values,
4216 &directory->default_layout_values);
4217
4218 /* find the subdirectory that is affected by those attributes */
4219 name = menu_layout_node_get_content (node);
4220 subdir = NULL;
4221 subdir_l = directory->subdirs;
4222 while (subdir_l != NULL)
4223 {
4224 subdir = subdir_l->data;
4225
4226 if (!strcmp (subdir->name, name))
4227 break;
4228
4229 subdir = NULL;
4230 subdir_l = subdir_l->next;
4231
4232 /* We do not want to use Menuname on a menu that appeared via
4233 * inlining: without inlining, the Menuname wouldn't have matched
4234 * anything, and we want to keep the same behavior.
4235 * Unless the layout is a default layout, in which case the Menuname
4236 * does match the subdirectory. */
4237 if (!using_default_layout && subdir_l == last_subdir)
4238 {
4239 subdir_l = NULL;
4240 break;
4241 }
4242 }
4243
4244 if (subdir == NULL)
4245 continue;
4246
4247 preprocess_layout_info_subdir_helper (tree, directory,
4248 subdir, &layout_values,
4249 &contents_added, &should_remove);
4250 strip_duplicates = strip_duplicates || contents_added;
4251 if (should_remove)
4252 {
4253 if (last_subdir == subdir_l)
4254 {
4255 /* we need to recompute last_subdir since we'll remove it from
4256 * the list */
4257 GSList *buf;
4258
4259 if (subdir_l == directory->subdirs)
4260 last_subdir = NULL;
4261 else
4262 {
4263 buf = directory->subdirs;
4264 while (buf != NULL && buf->next != subdir_l)
4265 buf = buf->next;
4266 last_subdir = buf;
4267 }
4268 }
4269
4270 directory->subdirs = g_slist_remove (directory->subdirs, subdir);
4271 matemenu_tree_item_unref_and_unset_parent (MATEMENU_TREE_ITEM (subdir));
4272 }
4273 }
4274
4275 /*
4276 * Now process the subdirectories with no explicit layout
4277 */
4278 /* this is bogus data, but we just need the pointer anyway */
4279 subdirs_sentinel = g_slist_prepend (directory->subdirs, PACKAGE);
4280 directory->subdirs = subdirs_sentinel;
4281
4282 tmp = directory->subdirs;
4283 while (tmp->next != NULL)
4284 {
4285 MateMenuTreeDirectory *subdir = tmp->next->data;
4286
4287 if (subdir->preprocessed)
4288 {
4289 tmp = tmp->next;
4290 continue;
4291 }
4292
4293 preprocess_layout_info_subdir_helper (tree, directory,
4294 subdir, &directory->default_layout_values,
4295 &contents_added, &should_remove);
4296 strip_duplicates = strip_duplicates || contents_added;
4297 if (should_remove)
4298 {
4299 tmp = g_slist_delete_link (tmp, tmp->next);
4300 matemenu_tree_item_unref_and_unset_parent (MATEMENU_TREE_ITEM (subdir));
4301 }
4302 else
4303 tmp = tmp->next;
4304 }
4305
4306 /* remove the sentinel */
4307 directory->subdirs = g_slist_delete_link (directory->subdirs,
4308 directory->subdirs);
4309
4310 /*
4311 * Finally, remove duplicates if needed
4312 */
4313 if (strip_duplicates)
4314 {
4315 /* strip duplicate entries; there should be no duplicate directories */
4316 directory->entries = g_slist_sort (directory->entries,
4317 (GCompareFunc) matemenu_tree_entry_compare_by_id);
4318 tmp = directory->entries;
4319 while (tmp != NULL && tmp->next != NULL)
4320 {
4321 MateMenuTreeItem *a = tmp->data;
4322 MateMenuTreeItem *b = tmp->next->data;
4323
4324 if (a->type == MATEMENU_TREE_ITEM_ALIAS)
4325 a = MATEMENU_TREE_ALIAS (a)->aliased_item;
4326
4327 if (b->type == MATEMENU_TREE_ITEM_ALIAS)
4328 b = MATEMENU_TREE_ALIAS (b)->aliased_item;
4329
4330 if (strcmp (MATEMENU_TREE_ENTRY (a)->desktop_file_id,
4331 MATEMENU_TREE_ENTRY (b)->desktop_file_id) == 0)
4332 {
4333 tmp = g_slist_delete_link (tmp, tmp->next);
4334 matemenu_tree_item_unref (b);
4335 }
4336 else
4337 tmp = tmp->next;
4338 }
4339 }
4340
4341 directory->preprocessed = TRUE;
4342}
