42gchar *
43gpm_get_timestring (guint time_secs)
44{
45 char* timestring = NULL;
46 gint hours;
47 gint minutes;
48
49 /* Add 0.5 to do rounding */
50 minutes = (int) ( ( time_secs / 60.0 ) + 0.5 );
51
52 if (minutes == 0) {
53 timestring = g_strdup (_("Unknown time"));
54 return timestring;
55 }
56
57 if (minutes < 60) {
58 timestring = g_strdup_printf (ngettext ("%i minute",
59 "%i minutes",
60 minutes), minutes);
61 return timestring;
62 }
63
64 hours = minutes / 60;
65 minutes = minutes % 60;
66
67 if (minutes == 0)
68 timestring = g_strdup_printf (ngettext (
69 "%i hour",
70 "%i hours",
71 hours), hours);
72 else
73 /* TRANSLATOR: "%i %s %i %s" are "%i hours %i minutes"
74 * Swap order with "%2$s %2$i %1$s %1$i if needed */
75 timestring = g_strdup_printf (_("%i %s %i %s"),
76 hours, ngettext ("hour", "hours", hours),
77 minutes, ngettext ("minute", "minutes", minutes));
78 return timestring;
79}

91guint
92gpm_discrete_from_percent (guint percentage, guint levels)
93{
94 /* for levels < 10 min value is 0 */
95 gint factor;
96 factor = levels < 10 ? 0 : 1;
97 /* check we are in range */
98 if (percentage > 100)
99 return levels;
100 if (levels == 0) {
101 g_warning ("levels is 0!");
102 return 0;
103 }
104 return (guint) ((((gfloat) percentage * (gfloat) (levels - factor)) / 100.0f) + 0.5f);
105}

116guint
117gpm_discrete_to_percent (guint discrete, guint levels)
118{
119 /* for levels < 10 min value is 0 */
120 gint factor;
121 factor = levels < 10 ? 0 : 1;
122 /* check we are in range */
123 if (discrete > levels)
124 return 100;
125 if (levels == 0) {
126 g_warning ("levels is 0!");
127 return 0;
128 }
129 return (guint) (((gfloat) discrete * (100.0f / (gfloat) (levels - factor))) + 0.5f);
130}

136void

138{
139 GError *error = NULL;
140 gchar *uri;
141
142 if (link_id != NULL)
143 uri = g_strconcat ("help:mate-power-manager/", link_id, NULL);
144 else
145 uri = g_strdup ("help:mate-power-manager");
146
147 gtk_show_uri_on_window (NULL, uri, GDK_CURRENT_TIME, &error);
148
149 if (error != NULL) {
150 GtkWidget *d;
151 d = gtk_message_dialog_new (NULL, GTK_DIALOG_MODAL | GTK_DIALOG_DESTROY_WITH_PARENT,
152 GTK_MESSAGE_ERROR, GTK_BUTTONS_OK, "%s", error->message);
153 gtk_dialog_run (GTK_DIALOG(d));
154 gtk_widget_destroy (d);
155 g_error_free (error);
156 }
157 g_free (uri);
158}

160gboolean
161gpm_notebook_scroll_event_cb (GtkWidget *widget,
162 GdkEventScroll *event)
163{
164 GtkNotebook *notebook = GTK_NOTEBOOK (widget);
165 GtkWidget *child, *event_widget, *action_widget;
166
167 child = gtk_notebook_get_nth_page (notebook, gtk_notebook_get_current_page (notebook));
168 if (child == NULL)
169 return FALSE;
170
171 event_widget = gtk_get_event_widget ((GdkEvent *) event);
172
173 /* Ignore scroll events from the content of the page */
174 if (event_widget == NULL || event_widget == child || gtk_widget_is_ancestor (event_widget, child))
175 return FALSE;
176
177 /* And also from the action widgets */
178 action_widget = gtk_notebook_get_action_widget (notebook, GTK_PACK_START);
179 if (event_widget == action_widget || (action_widget != NULL && gtk_widget_is_ancestor (event_widget, action_widget)))
180 return FALSE;
181
182 action_widget = gtk_notebook_get_action_widget (notebook, GTK_PACK_END);
183 if (event_widget == action_widget || (action_widget != NULL && gtk_widget_is_ancestor (event_widget, action_widget)))
184 return FALSE;
185
186 switch (event->direction) {
187 case GDK_SCROLL_RIGHT:
188 case GDK_SCROLL_DOWN:
189 gtk_notebook_next_page (notebook);
190 break;
191 case GDK_SCROLL_LEFT:
192 case GDK_SCROLL_UP:
193 gtk_notebook_prev_page (notebook);
194 break;
195 case GDK_SCROLL_SMOOTH:
196 switch (gtk_notebook_get_tab_pos (notebook)) {
197 case GTK_POS_LEFT:
198 case GTK_POS_RIGHT:
199 if (event->delta_y > 0)
200 gtk_notebook_next_page (notebook);
201 else if (event->delta_y < 0)
202 gtk_notebook_prev_page (notebook);
203 break;
204 case GTK_POS_TOP:
205 case GTK_POS_BOTTOM:
206 if (event->delta_x > 0)
207 gtk_notebook_next_page (notebook);
208 else if (event->delta_x < 0)
209 gtk_notebook_prev_page (notebook);
210 break;
211 default:
212 break;
213 }
214 break;
215 default:
216 break;
217 }
218
219 return TRUE;
220}
