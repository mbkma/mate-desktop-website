85static GType
86register_type( void )
87{
88 static const gchar *thisfn = "na_ioption_register_type";
89 GType type;
90
91 static const GTypeInfo info = {
92 sizeof( NAIOptionInterface ),
93 ( GBaseInitFunc ) interface_base_init,
94 ( GBaseFinalizeFunc ) interface_base_finalize,
95 NULL,
96 NULL,
97 NULL,
98 0,
99 0,
100 NULL
101 };
102
103 g_debug( "%s", thisfn );
104
105 type = g_type_register_static( G_TYPE_INTERFACE, "NAIOption", &info, 0 );
106
107 g_type_interface_add_prerequisite( type, G_TYPE_OBJECT );
108
109 return( type );
110}

112static void
113interface_base_init( NAIOptionInterface *iface )
114{
115 static const gchar *thisfn = "na_ioption_interface_base_init";
116
117 if( !st_initializations ){
118
119 g_debug( "%s: iface=%p (%s)", thisfn, ( void * ) iface, G_OBJECT_CLASS_NAME( iface ));
120
121 iface->private = g_new0( NAIOptionInterfacePrivate, 1 );
122
123 iface->get_version = ioption_get_version;
124 }
125
126 st_initializations += 1;
127}
128
129static void
130interface_base_finalize( NAIOptionInterface *iface )
131{
132 static const gchar *thisfn = "na_ioption_interface_base_finalize";
133
134 st_initializations -= 1;
135
136 if( !st_initializations ){
137
138 g_debug( "%s: iface=%p", thisfn, ( void * ) iface );
139
140 g_free( iface->private );
141 }
142}

150static IOptionData *
151get_ioption_data( NAIOption *instance )
152{
153 IOptionData *data;
154
155 data = ( IOptionData * ) g_object_get_data( G_OBJECT( instance ), IOPTION_PROP_DATA );
156
157 if( !data ){
158 data = g_new0( IOptionData, 1 );
159 g_object_set_data( G_OBJECT( instance ), IOPTION_PROP_DATA, data );
160 g_object_weak_ref( G_OBJECT( instance ), ( GWeakNotify ) on_instance_finalized, NULL );
161
162 data->initialized = TRUE;
163 }
164
165 return( data );
166}

168static void
169on_instance_finalized( gpointer user_data, NAIOption *instance )

171 static const gchar *thisfn = "na_ioption_on_instance_finalized";
172 IOptionData *data;
173
174 g_debug( "%s: user_data=%p, instance=%p", thisfn, ( void * ) user_data, ( void * ) instance );
175
176 data = get_ioption_data( instance );
177
178 g_free( data );
179}

181/*
182 * na_ioption_get_id:
183 * @option: this #NAIOption instance.
184 *
185 * Returns: the string identifier of the format, as a newly
186 * allocated string which should be g_free() by the caller.
187 */
188gchar *
189na_ioption_get_id( const NAIOption *option )
190{
191 gchar *id;
192
193 g_return_val_if_fail( NA_IS_IOPTION( option ), NULL );
194
195 get_ioption_data( NA_IOPTION( option ));
196 id = NULL;
197
198 if( NA_IOPTION_GET_INTERFACE( option )->get_id ){
199 id = NA_IOPTION_GET_INTERFACE( option )->get_id( option );
200 }
201
202 return( id );
203}
