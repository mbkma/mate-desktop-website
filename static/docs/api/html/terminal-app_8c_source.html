52/* Settings storage works as follows:
53 * /apps/mate-terminal/global/
54 * /apps/mate-terminal/profiles/Foo/
55 *
56 * It's somewhat tricky to manage the profiles/ dir since we need to track
57 * the list of profiles, but GSettings doesn't have a concept of notifying that
58 * a directory has appeared or disappeared.
59 *
60 * Session state is stored entirely in the RestartCommand command line.
61 *
62 * The number one rule: all stored information is EITHER per-session,
63 * per-profile, or set from a command line option. THERE CAN BE NO
64 * OVERLAP. The UI and implementation totally break if you overlap
65 * these categories. See mate-terminal 1.x for why.
66 *
67 * Don't use this code as an example of how to use GSettings - it's hugely
68 * overcomplicated due to the profiles stuff. Most apps should not
69 * have to do scary things of this nature, and should not have
70 * a profiles feature.
71 *
72 */

164static gboolean
165gsettings_append_strv (GSettings *settings,
166 const gchar *key,
167 const gchar *value)
168{
169 gchar **old;
170 gchar **new;
171 gint size;
172 gboolean retval;
173
174 old = g_settings_get_strv (settings, key);
175
176 for (size = 0; old[size] != NULL; size++);
177
178 size += 1; /* appended value */
179 size += 1; /* NULL */
180
181 new = g_realloc_n (old, size, sizeof (gchar *));
182
183 new[size - 2] = g_strdup (value);
184 new[size - 1] = NULL;
185
186 retval = g_settings_set_strv (settings, key,
187 (const gchar **) new);
188
189 g_strfreev (new);
190
191 return retval;
192}

194static gboolean
195gsettings_remove_all_from_strv (GSettings *settings,
196 const gchar *key,
197 const gchar *value)
198{
199 GArray *array;
200 gchar **old;
201 gint i;
202 gboolean retval;
203
204 old = g_settings_get_strv (settings, key);
205 array = g_array_new (TRUE, TRUE, sizeof (gchar *));
206
207 for (i = 0; old[i] != NULL; i++) {
208 if (g_strcmp0 (old[i], value) != 0)
209 array = g_array_append_val (array, old[i]);
210 }
211
212 retval = g_settings_set_strv (settings, key,
213 (const gchar **) array->data);
214
215 g_strfreev (old);
216 g_array_free (array, TRUE);
217
218 return retval;
219}

223static GdkScreen*
224terminal_app_get_screen_by_display_name (const char *display_name)
225{
226 GdkDisplay *display = NULL;
227 GdkScreen *screen = NULL;
228
229 if (display_name == NULL)
230 display = gdk_display_get_default ();
231 else
232 {
233 GSList *displays, *l;
234 const char *period;
235
236 period = strrchr (display_name, '.');
237 displays = gdk_display_manager_list_displays (gdk_display_manager_get ());
238 for (l = displays; l != NULL; l = l->next)
239 {
240 GdkDisplay *disp = l->data;
241
242 /* compare without the screen number part, if present */
243 if ((period && strncmp (gdk_display_get_name (disp), display_name, period - display_name) == 0) ||
244 (period == NULL && strcmp (gdk_display_get_name (disp), display_name) == 0))
245 {
246 display = disp;
247 break;
248 }
249 }
250 g_slist_free (displays);
251
252 if (display == NULL)
253 display = gdk_display_open (display_name); /* FIXME we never close displays */
254 }
255
256 if (display == NULL)
257 return NULL;
258 else
259 screen = gdk_display_get_default_screen (display);
260
261 return screen;
262}

288static int
289profiles_alphabetic_cmp (gconstpointer pa,
290 gconstpointer pb)
291{
292 TerminalProfile *a = (TerminalProfile *) pa;
293 TerminalProfile *b = (TerminalProfile *) pb;
294 int result;
295
296 result = g_utf8_collate (terminal_profile_get_property_string (a, TERMINAL_PROFILE_VISIBLE_NAME),
297 terminal_profile_get_property_string (b, TERMINAL_PROFILE_VISIBLE_NAME));
298 if (result == 0)
299 result = strcmp (terminal_profile_get_property_string (a, TERMINAL_PROFILE_NAME),
300 terminal_profile_get_property_string (b, TERMINAL_PROFILE_NAME));
301
302 return result;
303}

346 }
347

351static void
352terminal_app_delete_profile (TerminalApp *app, TerminalProfile *profile)
353{
354 const char *profile_name;
355 char *profile_dir;
356 GError *error = NULL;
357
358 profile_name = terminal_profile_get_property_string (profile, TERMINAL_PROFILE_NAME);
359 profile_dir = g_strconcat (CONF_PROFILE_PREFIX, profile_name, "/", NULL);
360
361 gsettings_remove_all_from_strv (app->settings_global, PROFILE_LIST_KEY, profile_name);
362
363 /* And remove the profile directory */
364 DConfClient *client = dconf_client_new ();
365 if (!dconf_client_write_sync (client, profile_dir, NULL, NULL, NULL, &error))
366 {
367 g_warning ("Failed to recursively unset %s: %s\n", profile_dir, error->message);
368 g_error_free (error);
369 }
370
371 g_object_unref (client);
372 g_free (profile_dir);
373}

413static /* ref */ GtkTreeModel *
414terminal_app_get_profile_liststore (TerminalApp *app,
415 TerminalProfile *selected_profile,
416 GtkTreeIter *selected_profile_iter,
417 gboolean *selected_profile_iter_set)
418{
419 GtkListStore *store;
420 GtkTreeIter iter;
421 GList *profiles, *l;
422
423 store = gtk_list_store_new (NUM_COLUMNS, TERMINAL_TYPE_PROFILE);
424
425 *selected_profile_iter_set = FALSE;
426
427 if (selected_profile &&
428 _terminal_profile_get_forgotten (selected_profile))
429 selected_profile = NULL;
430
431 profiles = terminal_app_get_profile_list (app);
432
433 for (l = profiles; l != NULL; l = l->next)
434 {
435 TerminalProfile *profile = TERMINAL_PROFILE (l->data);
436
437 gtk_list_store_insert_with_values (store, &iter, 0,
438 (int) COL_PROFILE, profile,
439 (int) -1);
440
441 if (selected_profile_iter && profile == selected_profile)
442 {
443 *selected_profile_iter = iter;
444 *selected_profile_iter_set = TRUE;
445 }
446 }
447 g_list_free (profiles);
448
449 /* Now turn on sorting */
450 gtk_tree_sortable_set_sort_func (GTK_TREE_SORTABLE (store),
451 COL_PROFILE,
452 terminal_app_profile_sort_func,
453 NULL, NULL);
454 gtk_tree_sortable_set_sort_column_id (GTK_TREE_SORTABLE (store),
455 COL_PROFILE, GTK_SORT_ASCENDING);
456
457 return GTK_TREE_MODEL (store);
458}

787static void
788terminal_app_profile_list_notify_cb (GSettings *settings,
789 const gchar *key,
790 gpointer user_data)
791{
792 TerminalApp *app = TERMINAL_APP (user_data);
793 GObject *object = G_OBJECT (app);
794 GVariant *val;
795 const gchar **value_list;
796 int i;
797 GList *profiles_to_delete, *l;
798 gboolean need_new_default;
799 TerminalProfile *fallback;
800 guint count;
801
802 g_object_freeze_notify (object);
803
804 profiles_to_delete = terminal_app_get_profile_list (app);
805
806 val = g_settings_get_value (settings, key);
807 if (val == NULL ||
808 (!g_variant_is_of_type (val, G_VARIANT_TYPE_STRING_ARRAY) &&
809 !g_variant_is_of_type (val, G_VARIANT_TYPE_STRING)))
810 goto ensure_one_profile;
811
812 value_list = g_variant_get_strv (val, NULL);
813 if (value_list == NULL)
814 goto ensure_one_profile;
815
816 /* Add any new ones */
817 for (i = 0; value_list[i] != NULL; ++i)
818 {
819 const char *profile_name = value_list[i];
820 GList *link;
821
822 if (!profile_name)
823 continue;
824
825 link = find_profile_link (profiles_to_delete, profile_name);
826 if (link)
827 /* make profiles_to_delete point to profiles we didn't find in the list */
828 profiles_to_delete = g_list_delete_link (profiles_to_delete, link);
829 else
830 terminal_app_create_profile (app, profile_name);
831 }
832
833 g_free (value_list);
834
835ensure_one_profile:
836
837 if (val != NULL)
838 g_variant_unref (val);
839
840 fallback = NULL;
841 count = g_hash_table_size (app->profiles);
842 if (count == 0 || count <= g_list_length (profiles_to_delete))
843 {
844 /* We are going to run out, so create the fallback
845 * to be sure we always have one. Must be done
846 * here before we emit "forgotten" signals so that
847 * screens have a profile to fall back to.
848 *
849 * If the profile with the FALLBACK_ID already exists,
850 * we aren't allowed to delete it, unless at least one
851 * other profile will still exist. And if you delete
852 * all profiles, the FALLBACK_ID profile returns as
853 * the living dead.
854 */
855 fallback = terminal_app_get_profile_by_name (app, FALLBACK_PROFILE_ID);
856 if (fallback == NULL)
857 fallback = terminal_app_create_profile (app, FALLBACK_PROFILE_ID);
858 g_assert (fallback != NULL);
859 }
860
861 /* Forget no-longer-existing profiles */
862 need_new_default = FALSE;
863 for (l = profiles_to_delete; l != NULL; l = l->next)
864 {
865 TerminalProfile *profile = TERMINAL_PROFILE (l->data);
866 const char *name;
867
868 name = terminal_profile_get_property_string (profile, TERMINAL_PROFILE_NAME);
869 if (strcmp (name, FALLBACK_PROFILE_ID) == 0)
870 continue;
871
872 if (profile == app->default_profile)
873 {
874 app->default_profile = NULL;
875 need_new_default = TRUE;
876 }
877
878 _terminal_profile_forget (profile);
879 g_hash_table_remove (app->profiles, name);
880
881 /* |profile| possibly isn't dead yet since the profiles dialogue's tree model holds a ref too... */
882 }
883 g_list_free (profiles_to_delete);
884
885 if (need_new_default)
886 {
887 TerminalProfile *new_default;
888 TerminalProfile **new_default_ptr = &new_default;
889
890 new_default = terminal_app_get_profile_by_name (app, FALLBACK_PROFILE_ID);
891 if (new_default == NULL)
892 {
893 GHashTableIter iter;
894
895 g_hash_table_iter_init (&iter, app->profiles);
896 if (!g_hash_table_iter_next (&iter, NULL, (gpointer *) new_default_ptr))
897 /* shouldn't really happen ever, but just to be safe */
898 new_default = terminal_app_create_profile (app, FALLBACK_PROFILE_ID);
899 }
900 g_assert (new_default != NULL);
901
902 app->default_profile = new_default;
903
904 g_object_notify (object, TERMINAL_APP_DEFAULT_PROFILE);
905 }
906
907 g_assert (g_hash_table_size (app->profiles) > 0);
908
909 g_signal_emit (app, signals[PROFILE_LIST_CHANGED], 0);
910
911 g_object_thaw_notify (object);
912}

1001 if (!charset)

1003

1005 if (!terminal_encoding_is_valid (encoding))
1006 continue;
1007

1009 }
1010
1011 g_free (strings);

1547 &argc, &argv,
1548 &error,
1549 NULL);

1551 if (options == NULL)
1552 {
1553 g_application_command_line_printerr (command_line,
1554 _("Failed to parse arguments: %s\n"),
1555 error->message);
1556 g_error_free (error);
1557 g_strfreev (argv);

1807gboolean
1808terminal_app_handle_options (TerminalApp *app,
1809 TerminalOptions *options,
1810 gboolean allow_resume,
1811 GError **error)
1812{
1813 GList *lw;
1814 GdkScreen *gdk_screen;
1815
1816 gdk_screen = terminal_app_get_screen_by_display_name (options->display_name);
1817
1818 if (options->save_config)
1819 {
1820 if (options->remote_arguments)
1821 return terminal_app_save_config_file (app, options->config_file, error);
1822
1823 g_set_error_literal (error, TERMINAL_OPTION_ERROR, TERMINAL_OPTION_ERROR_NOT_IN_FACTORY,
1824 "Cannot use \"--save-config\" when starting the factory process");
1825 return FALSE;
1826 }
1827
1828 if (options->load_config)
1829 {
1830 GKeyFile *key_file;
1831 gboolean result;
1832
1833 key_file = g_key_file_new ();
1834 result = g_key_file_load_from_file (key_file, options->config_file, 0, error) &&
1835 terminal_options_merge_config (options, key_file, SOURCE_DEFAULT, error);
1836 g_key_file_free (key_file);
1837
1838 if (!result)
1839 return FALSE;
1840
1841 /* fall-through on success */
1842 }
1843
1844#ifdef HAVE_SMCLIENT
1845 EggSMClient *sm_client;
1846
1847 sm_client = egg_sm_client_get ();
1848
1849 if (allow_resume && egg_sm_client_is_resumed (sm_client))
1850 {
1851 GKeyFile *key_file;
1852
1853 key_file = egg_sm_client_get_state_file (sm_client);
1854 if (key_file != NULL &&
1855 !terminal_options_merge_config (options, key_file, SOURCE_SESSION, error))
1856 return FALSE;
1857 }
1858#endif /* HAVE_SMCLIENT */
1859
1860 /* Make sure we open at least one window */
1861 terminal_options_ensure_window (options);
1862
1863 if (options->startup_id != NULL)
1864 _terminal_debug_print (TERMINAL_DEBUG_FACTORY,
1865 "Startup ID is '%s'\n",
1866 options->startup_id);
1867
1868 for (lw = options->initial_windows; lw != NULL; lw = lw->next)
1869 {
1870 InitialWindow *iw = lw->data;
1871 TerminalWindow *window = NULL;
1872 GList *lt;
1873
1874 g_assert (iw->tabs);
1875
1876 if ( lw == options->initial_windows && ((InitialTab *)iw->tabs->data)->attach_window )
1877 window = terminal_app_get_current_window(app, gdk_screen, options->initial_workspace);
1878
1879 if (!window)
1880 {
1881 /* Create & setup new window */
1882 window = terminal_app_new_window (app, gdk_screen);
1883
1884 /* Restored windows shouldn't demand attention; see bug #586308. */
1885 if (iw->source_tag == SOURCE_SESSION)
1886 terminal_window_set_is_restored (window);
1887
1888 if (options->startup_id != NULL)
1889 gtk_window_set_startup_id (GTK_WINDOW (window), options->startup_id);
1890
1891 /* Overwrite the default, unique window role set in terminal_window_init */
1892 if (iw->role)
1893 gtk_window_set_role (GTK_WINDOW (window), iw->role);
1894
1895 if (iw->icon)
1896 terminal_window_set_icon (window, iw->icon);
1897
1898 if (iw->force_menubar_state)
1899 terminal_window_set_menubar_visible (window, iw->menubar_state);
1900
1901 if (iw->start_fullscreen)
1902 gtk_window_fullscreen (GTK_WINDOW (window));
1903 if (iw->start_maximized)
1904 gtk_window_maximize (GTK_WINDOW (window));
1905 }
1906
1907 /* Now add the tabs */
1908 for (lt = iw->tabs; lt != NULL; lt = lt->next)
1909 {
1910 InitialTab *it = lt->data;
1911 TerminalProfile *profile = NULL;
1912 TerminalScreen *screen;
1913 const char *profile_name;
1914 gboolean profile_is_id;
1915
1916 if (it->profile)
1917 {
1918 profile_name = it->profile;
1919 profile_is_id = it->profile_is_id;
1920 }
1921 else
1922 {
1923 profile_name = options->default_profile;
1924 profile_is_id = options->default_profile_is_id;
1925 }
1926
1927 if (profile_name)
1928 {
1929 if (profile_is_id)
1930 profile = terminal_app_get_profile_by_name (app, profile_name);
1931 else
1932 profile = terminal_app_get_profile_by_visible_name (app, profile_name);
1933
1934 if (profile == NULL)
1935 g_printerr (_("No such profile \"%s\", using default profile\n"), it->profile);
1936 }
1937 if (profile == NULL)
1938 profile = terminal_app_get_profile_for_new_term (app);
1939 g_assert (profile);
1940
1941 screen = terminal_app_new_terminal (app, window, profile,
1942 it->exec_argv ? it->exec_argv : options->exec_argv,
1943 it->title ? it->title : options->default_title,
1944 it->working_dir ? it->working_dir : options->default_working_dir,
1945 options->env,
1946 it->zoom_set ? it->zoom : options->zoom);
1947
1948 if (it->active)
1949 terminal_window_switch_screen (window, screen);
1950 }
1951
1952 if (iw->geometry)
1953 {
1954 _terminal_debug_print (TERMINAL_DEBUG_GEOMETRY,
1955 "[window %p] applying geometry %s\n",
1956 window, iw->geometry);
1957
1958 if (!terminal_window_update_size_set_geometry (window,
1959 terminal_window_get_active (window),
1960 FALSE,
1961 iw->geometry))
1962 g_printerr (_("Invalid geometry string \"%s\"\n"), iw->geometry);
1963 }
1964
1965 gtk_window_present (GTK_WINDOW (window));
1966 }
1967
1968 return TRUE;
1969}

2035/*
2036* Get the window in the given screen and workspace. If nothing is found,
2037* a NULL is returned.
2038*/
2039TerminalWindow *
2040terminal_app_get_current_window (TerminalApp *app,
2041 GdkScreen *from_screen,
2042 int workspace)
2043{
2044 GList *windows;
2045 GList *res;
2046 TerminalWindow *ret = NULL;
2047
2048 windows = gtk_application_get_windows (GTK_APPLICATION (app));
2049 if (windows == NULL)
2050 return NULL;
2051
2052 res = g_list_last (windows);
2053
2054 g_assert (from_screen != NULL);
2055
2056 while (res)
2057 {
2058 int win_workspace;
2059
2060 if (!TERMINAL_IS_WINDOW (res->data))
2061 {
2062 res = g_list_previous (res);
2063 continue;
2064 }
2065
2066 if (gtk_window_get_screen(GTK_WINDOW(res->data)) != from_screen)
2067 {
2068 res = g_list_previous (res);
2069 continue;
2070 }
2071
2072 win_workspace = terminal_app_get_workspace_for_window(TERMINAL_WINDOW(res->data));
2073
2074 /* Same workspace or if the window is set to show up on all workspaces */
