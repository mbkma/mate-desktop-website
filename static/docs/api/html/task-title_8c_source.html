58static gboolean
59start_logout_dialog (TaskTitle *title)
60{
61 GError *error = NULL;
62 GAppInfo *app_info;
63 GdkScreen *gdkscreen;
64
65 g_return_val_if_fail (TASK_IS_TITLE (title), FALSE);
66
67 gdkscreen = gtk_widget_get_screen (GTK_WIDGET (title));
68 app_info = g_app_info_create_from_commandline (
69 LOGOUT, _("Log off, switch user, lock screen or power down the computer"),
70 G_APP_INFO_CREATE_NONE, &error);
71
72 if (!error) {
73 GdkDisplay *display;
74 GdkAppLaunchContext *launch_context;
75 display = gdk_screen_get_display (gdkscreen);
76 launch_context = gdk_display_get_app_launch_context (display);
77 gdk_app_launch_context_set_screen (launch_context, gdkscreen);
78 g_app_info_launch (app_info, NULL, G_APP_LAUNCH_CONTEXT (launch_context), &error);
79 g_object_unref (launch_context);
80 return TRUE;
81 }
82
83 GtkWidget *dialog;
84 dialog = gtk_message_dialog_new (NULL, GTK_DIALOG_DESTROY_WITH_PARENT,
85 GTK_MESSAGE_ERROR, GTK_BUTTONS_OK,
86 _("There was an error executing '%s': %s"),
87 LOGOUT, error->message);
88
89 g_signal_connect (dialog, "response", G_CALLBACK (gtk_widget_destroy), NULL);
90
91 gtk_window_set_resizable (GTK_WINDOW (dialog), FALSE);
92 gtk_window_set_screen (GTK_WINDOW (dialog), gdkscreen);
93
94 gtk_widget_show (dialog);
95 g_error_free (error);
96 return FALSE;
97}

99static gboolean
100on_close_clicked (GtkButton *button,
101 GdkEventButton *event,
102 TaskTitle *title)
103{
104 TaskTitlePrivate *priv;
105 WnckWindow *window;
106 gboolean retval;
107
108 g_return_val_if_fail (TASK_IS_TITLE (title), FALSE);
109 priv = title->priv;
110 retval = FALSE;
111
112 if (event->button != 1 || !priv->mouse_in_close_button)
113 return retval;
114
115 window = wnck_screen_get_active_window (priv->screen);
116
117 if (WNCK_IS_WINDOW (window)
118 && wnck_window_get_window_type (window) != WNCK_WINDOW_DESKTOP)
119 {
120 if (priv->window == window)
121 disconnect_window (title);
122 // event->time is buggy, thus we have to workaroud this
123 GdkScreen *gdkscreen;
124 GdkDisplay *display;
125 gdkscreen = gtk_widget_get_screen (GTK_WIDGET (title));
126 display = gdk_screen_get_display (gdkscreen);
127 wnck_window_close (window, gdk_x11_display_get_user_time (display));
128 retval = TRUE;
129 }
130 else if (priv->show_home_title)
131 {
132 // This is a logout click
133 retval = start_logout_dialog (title);
134 }
135 gtk_widget_queue_draw (GTK_WIDGET (title));
136
137 return retval;
138}

166static gboolean
167on_button_draw (GtkWidget *widget,
168 cairo_t *cr,
169 TaskTitle *title)
170{
171 g_return_val_if_fail (TASK_IS_TITLE (title), FALSE);
172
173 TaskTitlePrivate *priv;
174 priv = title->priv;
175
176 if (priv->mouse_in_close_button)
177 {
178 GtkStyle *style = gtk_widget_get_style (widget);
179 GdkRectangle area;
180 gdouble x1, y1, x2, y2;
181 cairo_clip_extents (cr, &x1, &y1, &x2, &y2);
182 area.x = floor (x1);
183 area.y = floor (y1);
184 area.width = ceil (x2) - area.x;
185 area.height = ceil (y2) - area.y;
186
187 gtk_paint_box (style,
188 cr,
189 GTK_STATE_PRELIGHT,
190 GTK_SHADOW_NONE,
191 NULL,
192 NULL,
193 area.x,
194 area.y + 2,
195 area.width,
196 area.height - 4);
197 }
198 return FALSE;
199}

276static void
277on_active_window_changed (WnckScreen *screen,
278 WnckWindow *old_window,
279 TaskTitle *title)
280{
281 TaskTitlePrivate *priv;
282 WnckWindow *act_window;
283 WnckWindowType type = WNCK_WINDOW_NORMAL;
284
285 g_return_if_fail (TASK_IS_TITLE (title));
286 priv = title->priv;
287
288 act_window = wnck_screen_get_active_window (screen);
289 if (act_window)
290 type = wnck_window_get_window_type (act_window);
291
292 if (WNCK_IS_WINDOW (act_window)
293 && wnck_window_is_skip_tasklist (act_window)
294 && type != WNCK_WINDOW_DESKTOP)
295 return;
296
297 if (type == WNCK_WINDOW_DOCK
298 || type == WNCK_WINDOW_SPLASHSCREEN
299 || type == WNCK_WINDOW_MENU)
300 return;
301
302 disconnect_window (title);
303
304 if (!WNCK_IS_WINDOW (act_window)
305 || wnck_window_get_window_type (act_window) == WNCK_WINDOW_DESKTOP)
306 {
307 if (priv->show_home_title)
308 {
309 gtk_label_set_text (GTK_LABEL (priv->label), _("Desktop"));
310 gtk_image_set_from_pixbuf (GTK_IMAGE (priv->button_image),
311 priv->quit_icon);
312
313 gtk_widget_set_tooltip_text (GTK_WIDGET (title), NULL);
314 gtk_widget_set_tooltip_text (priv->button,
315 _("Log off, switch user, lock screen or "
316 "power down the computer"));
317
318 gtk_widget_set_state_flags (GTK_WIDGET (title), GTK_STATE_FLAG_ACTIVE,
319 TRUE);
320 gtk_widget_show (priv->box);
321 }
322 else
323 {
324 gtk_widget_set_tooltip_text (priv->button, NULL);
325 gtk_widget_set_tooltip_text (GTK_WIDGET (title), NULL);
326 gtk_widget_set_state_flags (GTK_WIDGET (title), GTK_STATE_FLAG_NORMAL,
327 TRUE);
328 gtk_widget_hide (priv->box);
329 }
330 }
331 else
332 {
333 gtk_label_set_text (GTK_LABEL (priv->label),
334 wnck_window_get_name (act_window));
335 gtk_image_set_from_icon_name (GTK_IMAGE (priv->button_image),
336 "window-close", GTK_ICON_SIZE_MENU);
337
338 gtk_widget_set_tooltip_text (GTK_WIDGET (title),
339 wnck_window_get_name (act_window));
340 gtk_widget_set_tooltip_text (priv->button, _("Close window"));
341
342 g_signal_connect (act_window, "name-changed",
343 G_CALLBACK (on_name_changed), title);
344 g_signal_connect (act_window, "icon-changed",
345 G_CALLBACK (on_icon_changed), title);
346 g_signal_connect_after (act_window, "state-changed",
347 G_CALLBACK (on_state_changed), title);
348 priv->window = act_window;
349 if (wnck_window_is_maximized (act_window))
350 {
351 gtk_widget_set_state_flags (GTK_WIDGET (title), GTK_STATE_FLAG_ACTIVE,
352 TRUE);
353 gtk_widget_show (priv->box);
354 }
355 else
356 {
357 gtk_widget_set_state_flags (GTK_WIDGET (title), GTK_STATE_FLAG_NORMAL,
358 TRUE);
359 gtk_widget_hide (priv->box);
360 }
361 }
362
363 gtk_widget_queue_draw (GTK_WIDGET (title));
364}

366static gboolean
367on_button_release (GtkWidget *title, GdkEventButton *event)
368{
369 TaskTitlePrivate *priv;
370 WnckWindow *window;
371
372 g_return_val_if_fail (TASK_IS_TITLE (title), FALSE);
373 priv = TASK_TITLE (title)->priv;
374
375 window = wnck_screen_get_active_window (priv->screen);
376
377 g_return_val_if_fail (WNCK_IS_WINDOW (window), FALSE);
378
379 if (event->button == 3)
380 {
381 if (wnck_window_get_window_type (window) != WNCK_WINDOW_DESKTOP)
382 {
383 GtkWidget *menu;
384 menu = wnck_action_menu_new (window);
385 gtk_menu_popup_at_pointer (GTK_MENU (menu), (GdkEvent*)event);
386 return TRUE;
387 }
388 }
389 else if (event->button == 1)
390 {
391 if (event->type == GDK_2BUTTON_PRESS && wnck_window_is_maximized (window))
392 {
393 wnck_window_unmaximize (window);
394 }
395 }
396
397 return FALSE;
398}

454static void
455task_title_get_property (GObject *object,
456 guint prop_id,
457 GValue *value,
458 GParamSpec *pspec)
459{
460 TaskTitle *title = TASK_TITLE (object);
461 TaskTitlePrivate *priv;
462
463 g_return_if_fail (TASK_IS_TITLE (title));
464 priv = title->priv;
465
466 switch (prop_id)
467 {
468 case PROP_SHOW_HOME_TITLE:
469 g_value_set_boolean (value, priv->show_home_title);
470 break;
471 case PROP_BOLD_WINDOW_TITLE:
472 g_value_set_boolean (value, priv->bold_window_title);
473 break;
474 default:
475 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
476 }
477}

544static void
545task_title_init (TaskTitle *title)
546{
547 TaskTitlePrivate *priv;
548 GdkScreen *gdkscreen;
549 GtkIconTheme *theme;
550 AtkObject *atk;
551 int width, height;
552
553 priv = title->priv = task_title_get_instance_private (title);
554
555 priv->screen = wnck_screen_get_default ();
556 priv->window = NULL;
557
558 // Default values
559 priv->show_home_title = FALSE;
560 priv->bold_window_title = TRUE;
561
562 gtk_widget_add_events (GTK_WIDGET (title), GDK_ALL_EVENTS_MASK);
563
564 priv->align = gtk_alignment_new (0.0, 0.5, 1.0, 1.0);
565 gtk_alignment_set_padding (GTK_ALIGNMENT (priv->align),
566 0, 0, 6, 6);
567 gtk_container_add (GTK_CONTAINER (title), priv->align);
568
569 priv->box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 2);
570 gtk_container_add (GTK_CONTAINER (priv->align), priv->box);
571 gtk_widget_set_no_show_all (priv->box, TRUE);
572 gtk_widget_show (priv->box);
573
574 priv->label = gtk_label_new (_("Desktop"));
575 gtk_label_set_ellipsize (GTK_LABEL (priv->label), PANGO_ELLIPSIZE_END);
576 gtk_widget_set_halign (priv->label, GTK_ALIGN_START);
577 gtk_widget_set_valign (priv->label, GTK_ALIGN_CENTER);
578
579 if (priv->bold_window_title)
580 {
581 PangoAttrList *attr_list = pango_attr_list_new ();
582 PangoAttribute *attr = pango_attr_weight_new (PANGO_WEIGHT_BOLD);
583 pango_attr_list_insert (attr_list, attr);
584 gtk_label_set_attributes (GTK_LABEL (priv->label), attr_list);
585 }
586
587 gtk_box_pack_start (GTK_BOX (priv->box), priv->label, TRUE, TRUE, 0);
588 gtk_widget_show (priv->label);
589
590 priv->button = g_object_new (GTK_TYPE_EVENT_BOX,
591 "visible-window", FALSE,
592 "above-child", TRUE,
593 NULL);
594 gtk_box_pack_start (GTK_BOX (priv->box), priv->button, FALSE, FALSE, 0);
595 gtk_widget_show (priv->button);
596
597 atk = gtk_widget_get_accessible (priv->button);
598 atk_object_set_name (atk, _("Close"));
599 atk_object_set_description (atk, _("Close current window."));
600 atk_object_set_role (atk, ATK_ROLE_PUSH_BUTTON);
601
602 g_signal_connect (priv->button, "button-release-event",
603 G_CALLBACK (on_close_clicked), title);
604 g_signal_connect (priv->button, "enter-notify-event",
605 G_CALLBACK (on_enter_notify), title);
606 g_signal_connect (priv->button, "leave-notify-event",
607 G_CALLBACK (on_leave_notify), title);
608 g_signal_connect (priv->button, "draw",
609 G_CALLBACK (on_button_draw), title);
610
611 // Initialize button_image to allow easy setting by icon name after that.
612 gdkscreen = gtk_widget_get_screen (GTK_WIDGET (title));
613 theme = gtk_icon_theme_get_for_screen (gdkscreen);
614 gtk_icon_size_lookup (GTK_ICON_SIZE_MENU, &width, &height);
615 priv->quit_icon = gtk_icon_theme_load_icon (theme, "system-log-out", width, 0, NULL);
616 priv->button_image = gtk_image_new_from_pixbuf (priv->quit_icon);
617
618 gtk_container_add (GTK_CONTAINER (priv->button), priv->button_image);
619 gtk_widget_show (priv->button_image);
620
621 gtk_widget_hide (priv->box);
622
623 gtk_widget_add_events (GTK_WIDGET (title), GDK_ALL_EVENTS_MASK);
624
625 g_signal_connect (priv->screen, "active-window-changed",
626 G_CALLBACK (on_active_window_changed), title);
627 g_signal_connect (title, "button-press-event",
628 G_CALLBACK (on_button_release), NULL);
629}
