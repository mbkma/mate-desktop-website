107static const char *
108panel_menu_path_root_to_filename (MenuPathRoot path_root)
109{
110 const char *retval;
111 gsize i;
112
113 retval = NULL;
114
115 for (i = 0; i < G_N_ELEMENTS (root_items); i++) {
116 if (root_items [i].root_id == path_root) {
117 retval = root_items [i].filename;
118 break;
119 }
120 }
121
122 return retval;
123}

125static const char *
126panel_menu_filename_to_scheme (const char *filename)
127{
128 const char *retval;
129 gsize i;
130 gchar *basename;
131
132 retval = NULL;
133
134 if (!filename)
135 return retval;
136
137 /* Note:
138 * filename in root_items dose not have the parent directory,
139 * if filename passed is absolute path, this function will return NULL.
140 * so here use the basename for compare.
141 */
142 if (g_path_is_absolute (filename)) {
143 basename = g_path_get_basename (filename);
144 } else {
145 basename = g_strdup(filename);
146 }
147
148 for (i = 0; i < G_N_ELEMENTS (root_items); i++) {
149 if (root_items [i].filename &&
150 !strncmp (basename, root_items [i].filename,
151 strlen (root_items [i].filename))) {
152 retval = root_items [i].scheme;
153 break;
154 }
155 }
156 g_free(basename);
157
158 return retval;
159}

161static MenuPathRoot
162panel_menu_scheme_to_path_root (const char *scheme)
163{
164 MenuPathRoot retval;
165 gsize i;
166
167 retval = LAST_MENU;
168
169 if (!scheme)
170 return retval;
171
172 for (i = 0; i < G_N_ELEMENTS (root_items); i++) {
173 if (root_items [i].scheme &&
174 !strncmp (scheme, root_items [i].scheme,
175 strlen (root_items [i].scheme))) {
176 retval = root_items [i].root_id;
177 break;
178 }
179 }
180
181 return retval;
182}

202static void

204{
205 PanelMenuButton *button = PANEL_MENU_BUTTON (object);

222
223 G_OBJECT_CLASS (panel_menu_button_parent_class)->finalize (object);

390static GtkWidget *
391panel_menu_button_create_menu (PanelMenuButton *button)
392{
393 PanelWidget *panel_widget;
394
395 if (button->priv->menu)
396 return button->priv->menu;
397
398 if (!button->priv->toplevel)
399 return NULL;
400
401 panel_widget = panel_toplevel_get_panel_widget (button->priv->toplevel);
402
403 if (button->priv->use_menu_path &&
404 button->priv->path_root > FIRST_MENU &&
405 button->priv->path_root < LAST_MENU) {
406 const char *filename;
407
408 filename = panel_menu_path_root_to_filename (button->priv->path_root);
409 button->priv->menu = create_applications_menu (filename,
410 button->priv->menu_path,
411 TRUE);
412 } else
413 button->priv->menu = create_main_menu (panel_widget);
414
415 gtk_menu_attach_to_widget (GTK_MENU (button->priv->menu),
416 GTK_WIDGET (button),
417 (GtkMenuDetachFunc) panel_menu_button_menu_detacher);
418
419 panel_menu_button_associate_panel (button);
420
421 g_signal_connect_swapped (button->priv->menu, "deactivate",
422 G_CALLBACK (panel_menu_button_menu_deactivated),
423 button);
424
425 return button->priv->menu;
426}

436void
437panel_menu_button_popup_menu (PanelMenuButton *button,
438 guint n_button,
439 guint32 activate_time)
440{
441 GdkScreen *screen;
442
443 g_return_if_fail (PANEL_IS_MENU_BUTTON (button));
444
445 panel_menu_button_create_menu (button);
446
447 panel_toplevel_push_autohide_disabler (button->priv->toplevel);
448
449 button_widget_set_ignore_leave (BUTTON_WIDGET (button), TRUE);
450
451 screen = gtk_window_get_screen (GTK_WINDOW (button->priv->toplevel));
452 gtk_menu_set_screen (GTK_MENU (button->priv->menu), screen);
453
454 gtk_window_set_attached_to (GTK_WINDOW (gtk_widget_get_toplevel (button->priv->menu)),
455 GTK_WIDGET (button));
456
457 GdkGravity widget_anchor = GDK_GRAVITY_NORTH_WEST;
458 GdkGravity menu_anchor = GDK_GRAVITY_NORTH_WEST;
459 switch (panel_toplevel_get_orientation (button->priv->toplevel)) {
460 case PANEL_ORIENTATION_TOP:
461 widget_anchor = GDK_GRAVITY_SOUTH_WEST;
462 /*g_message ("PANEL_ORIENTATION_TOP"); comment this out by default,
463 *it is useful in debugging menu issues in wayland so don't remove it
464 */
465 break;
466 case PANEL_ORIENTATION_BOTTOM:
467 menu_anchor = GDK_GRAVITY_SOUTH_WEST;
468 /*g_message ("PANEL_ORIENTATION_BOTTOM");*/
469 break;
470 case PANEL_ORIENTATION_LEFT:
471 widget_anchor = GDK_GRAVITY_NORTH_EAST;
472 /*g_message ("PANEL_ORIENTATION_LEFT");*/
473 break;
474 case PANEL_ORIENTATION_RIGHT:
475 menu_anchor = GDK_GRAVITY_NORTH_EAST;
476 /*g_message ("PANEL_ORIENTATION_RIGHT");*/
477 break;
478 }
479
480 gtk_menu_popup_at_widget (GTK_MENU (button->priv->menu),
481 GTK_WIDGET (button),
482 widget_anchor,
483 menu_anchor,
484 NULL);
485}

607static void
608panel_menu_button_gsettings_notify (GSettings *settings,
609 gchar *key,
610 PanelMenuButton *button)
611{
612 if (!strcmp (key, PANEL_OBJECT_MENU_PATH_KEY)) {
613 gchar *value = g_settings_get_string (settings, key);
614 panel_menu_button_set_menu_path (button, value);
615 g_free (value);
616 } else if (!strcmp (key, PANEL_OBJECT_CUSTOM_ICON_KEY)) {
617 gchar *value = g_settings_get_string (settings, key);
618 panel_menu_button_set_custom_icon (button, value);
619 g_free (value);
620 } else if (!strcmp (key, PANEL_OBJECT_TOOLTIP_KEY)) {
621 gchar *value = g_settings_get_string (settings, key);
622 panel_menu_button_set_tooltip (button, value);
623 g_free (value);
624 } else if (!strcmp (key, PANEL_OBJECT_USE_MENU_PATH_KEY)) {
625 panel_menu_button_set_use_menu_path (button,
626 g_settings_get_boolean (settings, key));
627 } else if (!strcmp (key, PANEL_OBJECT_USE_CUSTOM_ICON_KEY)) {
628 panel_menu_button_set_use_custom_icon (button,
629 g_settings_get_boolean (settings, key));
630 } else if (!strcmp (key, PANEL_OBJECT_HAS_ARROW_KEY)) {
631 panel_menu_button_set_has_arrow (button,
632 g_settings_get_boolean (settings, key));
633 }
634}

662static void
663panel_menu_button_load (const char *menu_path,
664 gboolean use_menu_path,
665 const char *custom_icon,
666 gboolean use_custom_icon,
667 const char *tooltip,
668 PanelWidget *panel,
669 gboolean locked,
670 int position,
671 gboolean exactpos,
672 gboolean has_arrow,
673 const char *id)
674{
675 PanelMenuButton *button;
676 AppletInfo *info;
677
678 g_return_if_fail (panel != NULL);
679
680 button = g_object_new (PANEL_TYPE_MENU_BUTTON,
681 "menu-path", menu_path,
682 "custom-icon", custom_icon,
683 "tooltip", tooltip,
684 "use-menu-path", use_menu_path,
685 "use-custom-icon", use_custom_icon,
686 "has-arrow", has_arrow,
687 NULL);
688
689 info = mate_panel_applet_register (GTK_WIDGET (button), NULL, NULL,
690 panel, locked, position, exactpos,
691 PANEL_OBJECT_MENU, id);
692 if (!info) {
693 gtk_widget_destroy (GTK_WIDGET (button));
694 return;
695 }
696
697 gtk_widget_set_name (GTK_WIDGET (button), "mate-panel-main-menu-button");
698
699 button->priv->applet_id = g_strdup (info->id);
700
701 mate_panel_applet_add_callback (info, "help", "help-browser", _("_Help"), NULL);
702
703 if (!panel_lockdown_get_locked_down () &&
704 (panel_is_program_in_path ("mozo") ||
705 panel_is_program_in_path ("menulibre")))
706 mate_panel_applet_add_callback (info, "edit", "document-properties",
707 _("_Edit Menus"), NULL);
708
709 panel_widget_set_applet_expandable (panel, GTK_WIDGET (button), FALSE, TRUE);
710 panel_widget_set_applet_size_constrained (panel, GTK_WIDGET (button), TRUE);
711
712 panel_menu_button_connect_to_gsettings (button);
713
714 panel_lockdown_notify_add (G_CALLBACK (panel_menu_button_recreate_menu),
715 button);
716}

718static char *
719panel_menu_button_get_icon (PanelMenuButton *button)
720{
721 char *retval = NULL;
722

773static const char *
774split_menu_uri (const char *menu_uri,
775 char **menu_scheme)
776{
777 char *p;
778
779 if (!menu_uri)
780 return NULL;
781
782 p = strchr (menu_uri, ':');
783
784 if (!p || p == menu_uri)
785 return NULL;
786
787 if (menu_scheme)
788 *menu_scheme = g_strndup (menu_uri, p - menu_uri);
789
790 if (*(++p) != '/')
791 return NULL;
792
793 while (*p != '\0' && *(p + 1) == '/') p++;
794
795 return p;
796}
