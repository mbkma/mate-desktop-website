89#define EEL_ACCESSIBLE_FACTORY(type, factory_name, type_as_function, opt_create_accessible) \
90 \
91static GType \
92type_as_function ## _factory_get_accessible_type (void) \
93{ \
94 return type; \
95} \
96 \
97static AtkObject* \
98type_as_function ## _factory_create_accessible (GObject *obj) \
99{ \
100 AtkObject *accessible; \
101 \
102 g_assert (G_IS_OBJECT (obj)); \
103 \
104 accessible = opt_create_accessible (obj); \
105 \
106 return accessible; \
107} \
108 \
109static void \
110type_as_function ## _factory_class_init (AtkObjectFactoryClass *klass) \
111{ \
112 klass->create_accessible = type_as_function ## _factory_create_accessible; \
113 klass->get_accessible_type = type_as_function ## _factory_get_accessible_type;\
114} \
115 \
116static GType \
117type_as_function ## _factory_get_type (void) \
118{ \
119 static GType t = 0; \
120 \
121 if (!t) \
122 { \
123 static const GTypeInfo tinfo = \
124 { \
125 sizeof (AtkObjectFactoryClass), \
126 NULL, NULL, (GClassInitFunc) type_as_function ## _factory_class_init, \
127 NULL, NULL, sizeof (AtkObjectFactory), 0, NULL, NULL \
128 }; \
129 \
130 t = g_type_register_static ( \
131 ATK_TYPE_OBJECT_FACTORY, factory_name, &tinfo, 0); \
132 } \
133 \
134 return t; \
135}
