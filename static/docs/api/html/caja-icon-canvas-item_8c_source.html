71/* special text height handling
72 * each item has three text height variables:
73 * + text_height: actual height of the displayed (i.e. on-screen) PangoLayout.
74 * + text_height_for_layout: height used in icon grid layout algorithms.
75 * “sane amount” of text.
76 * “sane amount“ as of
77 * + hard-coded to three lines in text-below-icon mode.
78 * + unlimited in text-besides-icon mode (see VOODOO-TODO)
79 *
80 * This layout height is used by grid layout algorithms, even
81 * though the actually displayed and/or requested text size may be larger
82 * and overlap adjacent icons, if an icon is selected.
83 *
84 * + text_height_for_entire_text: height needed to display the entire PangoLayout,
85 * if it wasn't ellipsized.
86 */

314/* Currently we require pixbufs in this format (for hit testing).
315 * Perhaps gdk-pixbuf will be changed so it can do the hit testing
316 * and we won't have this requirement any more.
317 */
318static gboolean
319pixbuf_is_acceptable (GdkPixbuf *pixbuf)
320{
321 return gdk_pixbuf_get_colorspace (pixbuf) == GDK_COLORSPACE_RGB
322 && ((!gdk_pixbuf_get_has_alpha (pixbuf)
323 && gdk_pixbuf_get_n_channels (pixbuf) == 3)
324 || (gdk_pixbuf_get_has_alpha (pixbuf)
325 && gdk_pixbuf_get_n_channels (pixbuf) == 4))
326 && gdk_pixbuf_get_bits_per_sample (pixbuf) == 8;
327}

359/* Set property handler for the icon item. */
360static void
361caja_icon_canvas_item_set_property (GObject *object,
362 guint property_id,
363 const GValue *value,
364 GParamSpec *pspec)
365{
366 CajaIconCanvasItem *item;
367 CajaIconCanvasItemPrivate *details;
368 AtkObject *accessible;
369
370 item = CAJA_ICON_CANVAS_ITEM (object);
371 details = item->details;
372 accessible = atk_gobject_accessible_for_object (G_OBJECT (item));
373
374 switch (property_id)
375 {
376
377 case PROP_EDITABLE_TEXT:
378 if (g_strcmp0 (details->editable_text,
379 g_value_get_string (value)) == 0)
380 {
381 return;
382 }
383
384 g_free (details->editable_text);
385 details->editable_text = g_strdup (g_value_get_string (value));
386 if (details->text_util)
387 {
388 gail_text_util_text_setup (details->text_util,
389 details->editable_text);
390 }
391
392 caja_icon_canvas_item_invalidate_label_size (item);
393 if (details->editable_text_layout)
394 {
395 g_object_unref (details->editable_text_layout);
396 details->editable_text_layout = NULL;
397 }
398 break;
399
400 case PROP_ADDITIONAL_TEXT:
401 if (g_strcmp0 (details->additional_text,
402 g_value_get_string (value)) == 0)
403 {
404 return;
405 }
406
407 g_free (details->additional_text);
408 details->additional_text = g_strdup (g_value_get_string (value));
409
410 caja_icon_canvas_item_invalidate_label_size (item);
411 if (details->additional_text_layout)
412 {
413 g_object_unref (details->additional_text_layout);
414 details->additional_text_layout = NULL;
415 }
416 break;
417
418 case PROP_HIGHLIGHTED_FOR_SELECTION:
419 if (!details->is_highlighted_for_selection == !g_value_get_boolean (value))
420 {
421 return;
422 }
423 details->is_highlighted_for_selection = (g_value_get_boolean (value) != FALSE);
424 caja_icon_canvas_item_invalidate_label_size (item);
425
426 atk_object_notify_state_change (accessible, ATK_STATE_SELECTED,
427 details->is_highlighted_for_selection);
428 break;
429
430 case PROP_HIGHLIGHTED_AS_KEYBOARD_FOCUS:
431 if (!details->is_highlighted_as_keyboard_focus == !g_value_get_boolean (value))
432 {
433 return;
434 }
435 details->is_highlighted_as_keyboard_focus = (g_value_get_boolean (value) != FALSE);
436 atk_object_notify_state_change (accessible, ATK_STATE_FOCUSED,
437 details->is_highlighted_as_keyboard_focus);
438 break;
439
440 case PROP_HIGHLIGHTED_FOR_DROP:
441 if (!details->is_highlighted_for_drop == !g_value_get_boolean (value))
442 {
443 return;
444 }
445 details->is_highlighted_for_drop = (g_value_get_boolean (value) != FALSE);
446 break;
447
448 case PROP_HIGHLIGHTED_FOR_CLIPBOARD:
449 if (!details->is_highlighted_for_clipboard == !g_value_get_boolean (value))
450 {
451 return;
452 }
453 details->is_highlighted_for_clipboard = (g_value_get_boolean (value) != FALSE);
454 break;
455
456 default:
457 g_warning ("caja_icons_view_item_item_set_arg on unknown argument");
458 return;
459 }
460
461 eel_canvas_item_request_update (EEL_CANVAS_ITEM (object));
462}

530cairo_surface_t *
531caja_icon_canvas_item_get_drag_surface (CajaIconCanvasItem *item)
532{
533 cairo_surface_t *surface;
534
535 EelCanvas *canvas;
536 int width, height;
537 int pix_width, pix_height;
538 int item_offset_x, item_offset_y;
539 EelIRect icon_rect;
540 EelIRect emblem_rect;
541 GdkPixbuf *emblem_pixbuf;
542 EmblemLayout emblem_layout;
543 double item_x, item_y;
544 gboolean is_rtl;
545 cairo_t *cr;
546 GtkStyleContext *context;
547 cairo_surface_t *drag_surface;
548
549 g_return_val_if_fail (CAJA_IS_ICON_CANVAS_ITEM (item), NULL);
550
551 canvas = EEL_CANVAS_ITEM (item)->canvas;
552 context = gtk_widget_get_style_context (GTK_WIDGET (canvas));
553
554 gtk_style_context_save (context);
555 gtk_style_context_add_class (context, "caja-canvas-item");
556
557 /* Assume we're updated so canvas item data is right */
558
559 /* Calculate the offset from the top-left corner of the
560 new image to the item position (where the pixmap is placed) */
561 eel_canvas_world_to_window (canvas,
562 item->details->x, item->details->y,
563 &item_x, &item_y);
564
565 item_offset_x = item_x - EEL_CANVAS_ITEM (item)->x1;
566 item_offset_y = item_y - EEL_CANVAS_ITEM (item)->y1;
567
568 /* Calculate the width of the item */
569 width = EEL_CANVAS_ITEM (item)->x2 - EEL_CANVAS_ITEM (item)->x1;
570 height = EEL_CANVAS_ITEM (item)->y2 - EEL_CANVAS_ITEM (item)->y1;
571
572 surface = gdk_window_create_similar_surface (gtk_widget_get_window (GTK_WIDGET (canvas)),
573 CAIRO_CONTENT_COLOR_ALPHA,
574 width, height);
575
576 cr = cairo_create (surface);
577
578 drag_surface = gdk_cairo_surface_create_from_pixbuf (item->details->pixbuf,
579 gtk_widget_get_scale_factor (GTK_WIDGET (canvas)),
580 gtk_widget_get_window (GTK_WIDGET (canvas)));
581 gtk_render_icon_surface (context, cr, drag_surface,
582 item_offset_x, item_offset_y);
583 cairo_surface_destroy (drag_surface);
584
585 get_scaled_icon_size (item, &pix_width, &pix_height);
586
587 icon_rect.x0 = item_offset_x;
588 icon_rect.y0 = item_offset_y;
589 icon_rect.x1 = item_offset_x + pix_width;
590 icon_rect.y1 = item_offset_y + pix_height;
591
592 is_rtl = caja_icon_container_is_layout_rtl (CAJA_ICON_CONTAINER (canvas));
593
594 emblem_layout_reset (&emblem_layout, item, icon_rect, is_rtl);
595
596 while (emblem_layout_next (&emblem_layout, &emblem_pixbuf, &emblem_rect, is_rtl))
597 {
598 gdk_cairo_set_source_pixbuf (cr, emblem_pixbuf, emblem_rect.x0, emblem_rect.y0);
599 cairo_rectangle (cr, emblem_rect.x0, emblem_rect.y0,
600 gdk_pixbuf_get_width (emblem_pixbuf),
601 gdk_pixbuf_get_height (emblem_pixbuf));
602 cairo_fill (cr);
603 }
604
605 draw_embedded_text (item, cr,
606 item_offset_x, item_offset_y);
607 draw_label_text (item, cr, FALSE, icon_rect);
608 cairo_destroy (cr);
609
610 gtk_style_context_restore (context);
611
612 return surface;
613
614}

616void

619{
620 CajaIconCanvasItemPrivate *details;
621

624

645 details->pixbuf = image;
646
647 caja_icon_canvas_item_invalidate_bounds_cache (item);
648 eel_canvas_item_request_update (EEL_CANVAS_ITEM (item));
649}

651void
652caja_icon_canvas_item_set_emblems (CajaIconCanvasItem *item,
653 GList *emblem_pixbufs)
654{
655 GList *p;
656
657 g_return_if_fail (CAJA_IS_ICON_CANVAS_ITEM (item));
658
659 g_assert (item->details->emblem_pixbufs != emblem_pixbufs || emblem_pixbufs == NULL);
660
661 /* The case where the emblems are identical is fairly common,
662 * so lets take the time to check for it.
663 */
664 if (eel_g_list_equal (item->details->emblem_pixbufs, emblem_pixbufs))
665 {
666 return;
667 }
668
669 /* Check if they are acceptable. */
670 for (p = emblem_pixbufs; p != NULL; p = p->next)
671 {
672 g_return_if_fail (pixbuf_is_acceptable (p->data));
673 }
674
675 /* Take in the new list of emblems. */
676 g_list_foreach (emblem_pixbufs, (GFunc) g_object_ref, NULL);
677 g_list_free_full (item->details->emblem_pixbufs, g_object_unref);
678 item->details->emblem_pixbufs = g_list_copy (emblem_pixbufs);
679
680 caja_icon_canvas_item_invalidate_bounds_cache (item);
681 eel_canvas_item_request_update (EEL_CANVAS_ITEM (item));
682}

877void
878caja_icon_canvas_item_update_bounds (CajaIconCanvasItem *item,
879 double i2w_dx, double i2w_dy)
880{
881 EelIRect before, after, emblem_rect;
882 EmblemLayout emblem_layout;
883 EelCanvasItem *canvas_item;
884 GdkPixbuf *emblem_pixbuf;
885 gboolean is_rtl;
886
887 canvas_item = EEL_CANVAS_ITEM (item);
888
889 /* Compute new bounds. */
890 before = get_current_canvas_bounds (canvas_item);
891 recompute_bounding_box (item, i2w_dx, i2w_dy);
892 after = get_current_canvas_bounds (canvas_item);
893
894 /* If the bounds didn't change, we are done. */
895 if (eel_irect_equal (before, after))
896 {
897 return;
898 }
899
900 is_rtl = caja_icon_container_is_layout_rtl (CAJA_ICON_CONTAINER (canvas_item->canvas));
901
902 /* Update canvas and text rect cache */
903 get_icon_canvas_rectangle (item, &item->details->canvas_rect);
904 item->details->text_rect = compute_text_rectangle (item, item->details->canvas_rect,
905 TRUE, BOUNDS_USAGE_FOR_DISPLAY);
906
907 /* Update emblem rect cache */
908 item->details->emblem_rect.x0 = 0;
909 item->details->emblem_rect.x1 = 0;
910 item->details->emblem_rect.y0 = 0;
911 item->details->emblem_rect.y1 = 0;
912 emblem_layout_reset (&emblem_layout, item, item->details->canvas_rect, is_rtl);
913 while (emblem_layout_next (&emblem_layout, &emblem_pixbuf, &emblem_rect, is_rtl))
914 {
915 eel_irect_union (&item->details->emblem_rect, &item->details->emblem_rect, &emblem_rect);
916 }
917
918 /* queue a redraw. */
919 eel_canvas_request_redraw (canvas_item->canvas,
920 before.x0, before.y0,
921 before.x1 + 1, before.y1 + 1);
922}

1102static void
1103measure_label_text (CajaIconCanvasItem *item)
1104{

1106 CajaIconContainer *container;
1107 gint editable_height, editable_height_for_layout, editable_height_for_entire_text, editable_width, editable_dx;
1108 gint additional_height, additional_width, additional_dx;
1109 PangoLayout *editable_layout;
1110 PangoLayout *additional_layout;
1111 gboolean have_editable, have_additional;
1112
1113 /* check to see if the cached values are still valid; if so, there's
1114 * no work necessary
1115 */
1116
1117 if (item->details->text_width >= 0 && item->details->text_height >= 0)
1118 {
1119 return;
1120 }
1121
1122 details = item->details;
1123
1124 have_editable = details->editable_text != NULL && details->editable_text[0] != '\0';
1125 have_additional = details->additional_text != NULL && details->additional_text[0] != '\0';
1126
1127 /* No font or no text, then do no work. */
1128 if (!have_editable && !have_additional)
1129 {
1130 details->text_height = 0;
1131 details->text_height_for_layout = 0;
1132 details->text_height_for_entire_text = 0;
1133 details->text_width = 0;

1137#ifdef PERFORMANCE_TEST_MEASURE_DISABLE

1146 editable_width = 0;
1147 editable_height = 0;
1148 editable_height_for_layout = 0;

1213 details->text_height = editable_height;

1215 details->text_height_for_entire_text = editable_height_for_entire_text;
1216 }

1547static gboolean
1548emblem_layout_next (EmblemLayout *layout,
1549 GdkPixbuf **emblem_pixbuf,
1550 EelIRect *emblem_rect,
1551 gboolean is_rtl)
1552{
1553 GdkPixbuf *pixbuf;
1554 int width, height, x, y;
1555 GdkPoint *attach_points;
1556
1557 /* Check if we have layed out all of the pixbufs. */
1558 if (layout->emblem == NULL)
1559 {
1560 return FALSE;
1561 }
1562
1563 /* Get the pixbuf. */
1564 pixbuf = layout->emblem->data;
1565 width = gdk_pixbuf_get_width (pixbuf);
1566 height = gdk_pixbuf_get_height (pixbuf);
1567
1568 /* Advance to the next emblem. */
1569 layout->emblem = layout->emblem->next;
1570
1571 attach_points = layout->icon_item->details->attach_points;
1572 if (attach_points != NULL)
1573 {
1574 if (layout->index >= layout->icon_item->details->n_attach_points)
1575 {
1576 return FALSE;
1577 }
1578
1579 x = layout->icon_rect.x0 + attach_points[layout->index].x;
1580 y = layout->icon_rect.y0 + attach_points[layout->index].y;
1581
1582 layout->index += 1;
1583
1584 /* Return the rectangle and pixbuf. */
1585 *emblem_pixbuf = pixbuf;
1586 emblem_rect->x0 = x - width / 2;
1587 emblem_rect->y0 = y - height / 2;
1588 emblem_rect->x1 = emblem_rect->x0 + width;
1589 emblem_rect->y1 = emblem_rect->y0 + height;
1590
1591 return TRUE;
1592
1593 }
1594
1595 for (;;)
1596 {
1597
1598 /* Find the side to lay out along. */
1599 switch (layout->side)
1600 {
1601 case RIGHT_SIDE:
1602 x = layout->icon_rect.x1;
1603 y = is_rtl ? layout->icon_rect.y1 : layout->icon_rect.y0;
1604 break;
1605 case BOTTOM_SIDE:
1606 x = is_rtl ? layout->icon_rect.x0 : layout->icon_rect.x1;
1607 y = layout->icon_rect.y1;
1608 break;
1609 case LEFT_SIDE:
1610 x = layout->icon_rect.x0;
1611 y = is_rtl ? layout->icon_rect.y0 : layout->icon_rect.y1;
1612 break;
1613 case TOP_SIDE:
1614 x = is_rtl ? layout->icon_rect.x1 : layout->icon_rect.x0;
1615 y = layout->icon_rect.y0;
1616 break;
1617 default:
1618 g_assert_not_reached ();
1619 x = 0;
1620 y = 0;
1621 break;
1622 }
1623 if (layout->position != 0)
1624 {
1625 switch (layout->side)
1626 {
1627 case RIGHT_SIDE:
1628 y += (is_rtl ? -1 : 1) * (layout->position + height / 2);
1629 break;
1630 case BOTTOM_SIDE:
1631 x += (is_rtl ? 1 : -1 ) * (layout->position + width / 2);
1632 break;
1633 case LEFT_SIDE:
1634 y += (is_rtl ? 1 : -1) * (layout->position + height / 2);
1635 break;
1636 case TOP_SIDE:
1637 x += (is_rtl ? -1 : 1) * (layout->position + width / 2);
1638 break;
1639 }
1640 }
1641
1642 /* Check to see if emblem fits in current side. */
1643 if (x >= layout->icon_rect.x0 && x <= layout->icon_rect.x1
1644 && y >= layout->icon_rect.y0 && y <= layout->icon_rect.y1)
1645 {
1646
1647 /* It fits. */
1648
1649 /* Advance along the side. */
1650 switch (layout->side)
1651 {
1652 case RIGHT_SIDE:
1653 case LEFT_SIDE:
1654 layout->position += height + EMBLEM_SPACING;
1655 break;
1656 case BOTTOM_SIDE:
1657 case TOP_SIDE:
1658 layout->position += width + EMBLEM_SPACING;
1659 break;
1660 }
1661
1662 /* Return the rectangle and pixbuf. */
1663 *emblem_pixbuf = pixbuf;
1664 emblem_rect->x0 = x - width / 2;
1665 emblem_rect->y0 = y - height / 2;
1666 emblem_rect->x1 = emblem_rect->x0 + width;
1667 emblem_rect->y1 = emblem_rect->y0 + height;
1668
1669 return TRUE;
1670 }
1671
1672 /* It doesn't fit, so move to the next side. */
1673 switch (layout->side)
1674 {
1675 case RIGHT_SIDE:
1676 layout->side = is_rtl ? TOP_SIDE : BOTTOM_SIDE;
1677 break;
1678 case BOTTOM_SIDE:
1679 layout->side = is_rtl ? RIGHT_SIDE : LEFT_SIDE;
1680 break;
1681 case LEFT_SIDE:
1682 layout->side = is_rtl ? BOTTOM_SIDE : TOP_SIDE;
1683 break;
1684 case TOP_SIDE:
1685 default:
1686 return FALSE;
1687 }
1688 layout->position = 0;
1689 }
1690}
