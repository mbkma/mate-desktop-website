56gboolean
57double_click_detector_is_double_click (DoubleClickDetector *this, guint32 event_time,
58 gboolean auto_update)
59{
60 gint32 delta;
61
62 if (event_time == 0)
63 event_time = (guint32) (g_get_monotonic_time () / 1000); /* milliseconds */
64
65 if (this->last_click_time == 0) {
66 if (auto_update)
67 double_click_detector_update_click_time (this, event_time);
68
69 return FALSE;
70 }
71
72 delta = (gint32) event_time - (gint32) this->last_click_time;
73
74 if (auto_update)
75 double_click_detector_update_click_time (this, event_time);
76
77 return delta < this->double_click_time;
78}
