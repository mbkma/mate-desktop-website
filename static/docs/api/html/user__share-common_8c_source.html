30static char *
31lookup_special_dir (GUserDirectory directory,
32 const char *name,
33 gboolean create_dir)
34{
35 const char *special_dir;
36 char *dir;
37
38 special_dir = g_get_user_special_dir (directory);
39 if (special_dir != NULL && strcmp (special_dir, g_get_home_dir ()) != 0) {
40 if (create_dir != FALSE)
41 g_mkdir_with_parents (special_dir, 0755);
42 return g_strdup (special_dir);
43 }
44
45 dir = g_build_filename (g_get_home_dir (), name, NULL);
46 if (create_dir != FALSE)
47 g_mkdir_with_parents (dir, 0755);
48 return dir;
49}
