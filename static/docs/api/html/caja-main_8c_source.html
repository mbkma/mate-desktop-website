149static void
150setup_debug_log_signals (void)
151{
152 struct sigaction sa;
153 GIOChannel *io;
154
155 if (pipe (debug_log_pipes) == -1)
156 g_error ("Could not create pipe() for debug log");
157
158 io = g_io_channel_unix_new (debug_log_pipes[0]);
159 g_io_add_watch (io, G_IO_IN, debug_log_io_cb, NULL);
160
161 sa.sa_handler = sigusr1_handler;
162 sigemptyset (&sa.sa_mask);
163 sa.sa_flags = 0;
164 sigaction (SIGUSR1, &sa, NULL);
165
166 /* This is totally broken as we're using non-signal safe
167 * calls in sigfatal_handler. Disable by default. */
168#ifdef USE_SEGV_HANDLER
169 sa.sa_handler = sigfatal_handler;
170 sigemptyset (&sa.sa_mask);
171 sa.sa_flags = 0;
172
173 sigaction(SIGSEGV, &sa, &old_segv_sa);
174 sigaction(SIGABRT, &sa, &old_abrt_sa);
175 sigaction(SIGTRAP, &sa, &old_trap_sa);
176 sigaction(SIGFPE, &sa, &old_fpe_sa);
177 sigaction(SIGBUS, &sa, &old_bus_sa);
178#endif
179}

183static void
184log_override_cb (const gchar *log_domain,
185 GLogLevelFlags log_level,
186 const gchar *message,
187 gpointer user_data)
188{
189 gboolean is_debug;
190 gboolean is_milestone;
191
192 is_debug = ((log_level & G_LOG_LEVEL_DEBUG) != 0);
193 is_milestone = !is_debug;
194
195 caja_debug_log (is_milestone, CAJA_DEBUG_LOG_DOMAIN_GLOG, "%s", message);
196
197 if (!is_debug)
198 (* default_log_handler) (log_domain, log_level, message, user_data);
199}

220int
221main (int argc, char *argv[])
222{
223 gint retval;
224 CajaApplication *application;
225
226#if defined (HAVE_MALLOPT) && defined(M_MMAP_THRESHOLD)
227 /* Caja uses lots and lots of small and medium size allocations,
228 * and then a few large ones for the desktop background. By default
229 * glibc uses a dynamic treshold for how large allocations should
230 * be mmaped. Unfortunately this triggers quickly for caja when
231 * it does the desktop background allocations, raising the limit
232 * such that a lot of temporary large allocations end up on the
233 * heap and are thus not returned to the OS. To fix this we set
234 * a hardcoded limit. I don't know what a good value is, but 128K
235 * was the old glibc static limit, lets use that.
236 */
237 mallopt (M_MMAP_THRESHOLD, 128 *1024);
238#endif
239
240 if (g_getenv ("CAJA_DEBUG") != NULL) {
241 eel_make_warnings_and_criticals_stop_in_debugger ();
242 }
243
244#ifdef ENABLE_NLS
245 /* Initialize gettext support */
246 bindtextdomain (GETTEXT_PACKAGE, MATELOCALEDIR);
247 bind_textdomain_codeset (GETTEXT_PACKAGE, "UTF-8");
248 textdomain (GETTEXT_PACKAGE);
249#endif /* ENABLE_NLS */
250
251 g_set_prgname ("caja");
252
253 if (g_file_test (DATADIR "/applications/caja.desktop", G_FILE_TEST_EXISTS)) {
254 egg_set_desktop_file (DATADIR "/applications/caja.desktop");
255 }
256
257#ifdef HAVE_EXEMPI
258 xmp_init();
259#endif
260
261 setup_debug_log ();
262
263 /* Initialize the services that we use. */
264 LIBXML_TEST_VERSION
265
266 /* Run the caja application. */
267 application = caja_application_new();
268
269 retval = g_application_run (G_APPLICATION (application),
270 argc, argv);
271
272 g_object_unref (application);
273
274 eel_debug_shut_down ();
275
276 return retval;
277}
