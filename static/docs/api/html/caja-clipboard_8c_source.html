323static void
324merge_in_clipboard_menu_items (GObject *widget_as_object,
325 TargetCallbackData *target_data)
326{
327 gboolean add_selection_callback;
328
329 g_assert (target_data != NULL);
330
331 add_selection_callback = target_data->shares_selection_changes;
332
333 gtk_ui_manager_insert_action_group (target_data->ui_manager,
334 target_data->action_group, 0);
335
336 set_paste_sensitive_if_clipboard_contains_data (target_data->action_group);
337
338 g_signal_connect (gtk_clipboard_get (GDK_SELECTION_CLIPBOARD), "owner_change",
339 G_CALLBACK (owner_change_callback), target_data);
340
341 if (add_selection_callback)
342 {
343 target_data->connect_callbacks (widget_as_object, target_data);
344 }
345 else
346 {
347 /* If we don't use sensitivity, everything should be on */
348 set_clipboard_menu_items_sensitive (target_data->action_group);
349 }
350 set_clipboard_items_are_merged_in (widget_as_object, TRUE);
351}

353static void
354merge_out_clipboard_menu_items (GObject *widget_as_object,
355 TargetCallbackData *target_data)
356
357{
358 gboolean selection_callback_was_added;
359
360 g_assert (target_data != NULL);
361
362 gtk_ui_manager_remove_action_group (target_data->ui_manager,
363 target_data->action_group);
364
365 g_signal_handlers_disconnect_matched (gtk_clipboard_get (GDK_SELECTION_CLIPBOARD),
366 G_SIGNAL_MATCH_FUNC | G_SIGNAL_MATCH_DATA,
367 0, 0, NULL,
368 G_CALLBACK (owner_change_callback),
369 target_data);
370
371 selection_callback_was_added = target_data->shares_selection_changes;
372
373 if (selection_callback_was_added)
374 {
375 target_data->disconnect_callbacks (widget_as_object, target_data);
376 }
377 set_clipboard_items_are_merged_in (widget_as_object, FALSE);
378}

380static gboolean
381focus_changed_callback (GtkWidget *widget,
382 GdkEventAny *event,
383 gpointer callback_data)
384{
385 /* Connect the component to the container if the widget has focus. */
386 if (gtk_widget_has_focus (widget))
387 {
388 if (!clipboard_items_are_merged_in (widget))
389 {
390 merge_in_clipboard_menu_items (G_OBJECT (widget), callback_data);
391 }
392 }
393 else
394 {
395 if (clipboard_items_are_merged_in (widget))
396 {
397 merge_out_clipboard_menu_items (G_OBJECT (widget), callback_data);
398 }
399 }
400
401 return FALSE;
402}

405selection_changed_callback (GtkWidget *widget,
406 gpointer callback_data)
407{

419 {
420 set_clipboard_menu_items_sensitive (target_data->action_group);

484static TargetCallbackData *
485initialize_clipboard_component_with_callback_data (GtkEditable *target,
486 GtkUIManager *ui_manager,
487 gboolean shares_selection_changes,
488 SelectAllCallback select_all_callback,
489 ConnectCallbacksFunc connect_callbacks,
490 ConnectCallbacksFunc disconnect_callbacks)
491{
492 GtkActionGroup *action_group;
493 TargetCallbackData *target_data;
494
495 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
496 action_group = gtk_action_group_new ("ClipboardActions");
497#ifdef ENABLE_NLS
498 gtk_action_group_set_translation_domain (action_group, GETTEXT_PACKAGE);
499#endif /* ENABLE_NLS */
500 gtk_action_group_add_actions (action_group,
501 clipboard_entries, G_N_ELEMENTS (clipboard_entries),
502 target);
503 G_GNUC_END_IGNORE_DEPRECATIONS;
504
505 /* Do the actual connection of the UI to the container at
506 * focus time, and disconnect at both focus and destroy
507 * time.
508 */
509 target_data = g_new (TargetCallbackData, 1);
510 target_data->ui_manager = ui_manager;
511 target_data->action_group = action_group;
512 target_data->shares_selection_changes = shares_selection_changes;
513 target_data->select_all_callback = select_all_callback;
514 target_data->connect_callbacks = connect_callbacks;
515 target_data->disconnect_callbacks = disconnect_callbacks;
516
517 return target_data;
518}

520static void
521caja_clipboard_real_set_up (gpointer target,
522 GtkUIManager *ui_manager,
523 gboolean shares_selection_changes,
524 SelectAllCallback select_all_callback,
525 ConnectCallbacksFunc connect_callbacks,
526 ConnectCallbacksFunc disconnect_callbacks)
527{
528 TargetCallbackData *target_data;
529
530 if (g_object_get_data (G_OBJECT (target), "Caja:clipboard_target_data") != NULL)
531 {
532 return;
533 }
534
535 target_data = initialize_clipboard_component_with_callback_data
536 (target,
537 ui_manager,
538 shares_selection_changes,
539 select_all_callback,
540 connect_callbacks,
541 disconnect_callbacks);
542
543 g_signal_connect (target, "focus_in_event",
544 G_CALLBACK (focus_changed_callback), target_data);
545 g_signal_connect (target, "focus_out_event",
546 G_CALLBACK (focus_changed_callback), target_data);
547 g_signal_connect (target, "destroy",
548 G_CALLBACK (target_destroy_callback), target_data);
549
550 g_object_set_data_full (G_OBJECT (target), "Caja:clipboard_target_data",
551 target_data, (GDestroyNotify) target_data_free);
552
553 /* Call the focus changed callback once to merge if the window is
554 * already in focus.
555 */
556 focus_changed_callback (GTK_WIDGET (target), NULL, target_data);
557}

587static GList *
588convert_lines_to_str_list (char **lines, gboolean *cut)
589{
590 int i;
591 GList *result;
592
593 if (cut)
594 {
595 *cut = FALSE;
596 }
597
598 if (lines[0] == NULL)
599 {
600 return NULL;
601 }
602
603 if (strcmp (lines[0], "cut") == 0)
604 {
605 if (cut)
606 {
607 *cut = TRUE;
608 }
609 }
610 else if (strcmp (lines[0], "copy") != 0)
611 {
612 return NULL;
613 }
614
615 result = NULL;
616 for (i = 1; lines[i] != NULL; i++)
617 {
618 result = g_list_prepend (result, g_strdup (lines[i]));
619 }
620 return g_list_reverse (result);
621}

623GList*
624caja_clipboard_get_uri_list_from_selection_data (GtkSelectionData *selection_data,
625 gboolean *cut,
626 GdkAtom copied_files_atom)
627{
628 GList *items;
629
630 if (gtk_selection_data_get_data_type (selection_data) != copied_files_atom
631 || gtk_selection_data_get_length (selection_data) <= 0)
632 {
633 items = NULL;
634 }
635 else
636 {
637 char **lines;
638 guchar *data;
639 /* Not sure why it's legal to assume there's an extra byte
640 * past the end of the selection data that it's safe to write
641 * to. But gtk_editable_selection_received does this, so I
642 * think it is OK.
643 */
644 data = (guchar *) gtk_selection_data_get_data (selection_data);
645 data[gtk_selection_data_get_length (selection_data)] = '\0';
646 lines = g_strsplit (data, "\n", 0);
647 items = convert_lines_to_str_list (lines, cut);
648 g_strfreev (lines);
649 }
650
651 return items;
652}

661void
662caja_clipboard_clear_if_colliding_uris (GtkWidget *widget,
663 const GList *item_uris,
664 GdkAtom copied_files_atom)
665{
666 GtkSelectionData *data;
667 GList *clipboard_item_uris, *l;
668 gboolean collision;
669
670 collision = FALSE;
671 data = gtk_clipboard_wait_for_contents (caja_clipboard_get (widget),
672 copied_files_atom);
673 if (data == NULL) {
674 return;
675 }
676
677 clipboard_item_uris = caja_clipboard_get_uri_list_from_selection_data (data, NULL,
678 copied_files_atom);
679
680 for (l = (GList *) item_uris; l; l = l->next) {
681 if (g_list_find_custom ((GList *) item_uris, l->data,
682 (GCompareFunc) g_strcmp0)) {
683 collision = TRUE;
684 break;
685 }
686 }
687
688 if (collision) {
689 gtk_clipboard_clear (caja_clipboard_get (widget));
690 }
691
692 if (clipboard_item_uris) {
693 g_list_free_full (clipboard_item_uris, g_free);
694 }
