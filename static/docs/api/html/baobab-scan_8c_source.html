34/*
35 Hardlinks handling.
36
37 As long as we're optimistic about hardlinks count
38 over the whole system (250 files with st_nlink > 1 here),
39 we keep linear search. If it turns out to be an bottleneck
40 we can switch to an hash table or tree.
41
42 TODO: get real timings about this code. find out the average
43 number of files with st_nlink > 1 on average computer.
44
45 To save memory, we store only { inode, dev } instead of full
46 GFileInfo.
47
48 EDIT: /me stupid. I realize that this code was not called that often
49 1 call per file with st_nlink > 1. BUT, i'm using pdumpfs to backup
50 my /etc. pdumpfs massively uses hard links. So there are more than
51 5000 files with st_nlink > 1. I believe this is the worst case.
52*/

67static gboolean
68baobab_hardlinks_array_has (BaobabHardLinkArray *a,
69 BaobabHardLink *s)
70{
71 guint i;
72
73 for (i = 0; i < a->len; ++i) {
74 BaobabHardLink *cur = &g_array_index (a, BaobabHardLink, i);
75
76 /*
77 * cur->st_dev == s->st_dev is the common case and may be more
78 * expansive than cur->st_ino == s->st_ino
79 * so keep this order */
80 if (cur->inode == s->inode && cur->device == s->device)
81 return TRUE;
82 }
83
84 return FALSE;
85}

87/* return FALSE if the element was already in the array */
88static gboolean
89baobab_hardlinks_array_add (BaobabHardLinkArray *a,
90 GFileInfo *s)
91{
92
93 if (g_file_info_has_attribute (s, G_FILE_ATTRIBUTE_UNIX_INODE) &&
94 g_file_info_has_attribute (s, G_FILE_ATTRIBUTE_UNIX_DEVICE))
95 {
96 BaobabHardLink hl;
97
98 hl.inode = g_file_info_get_attribute_uint64 (s,
99 G_FILE_ATTRIBUTE_UNIX_INODE);
100 hl.device = g_file_info_get_attribute_uint32 (s,
101 G_FILE_ATTRIBUTE_UNIX_DEVICE);
102
103 if (baobab_hardlinks_array_has (a, &hl))
104 return FALSE;
105
106 g_array_append_val (a, hl);
107
108 return TRUE;
109 }
110 else
111 {
112 g_warning ("Could not obtain inode and device for hardlink");
113 }
114
115 return FALSE;
116}

145static gboolean
146is_in_dot_gvfs (GFile *file)
147{
148 static GFile *dot_gvfs_dir = NULL;
149 GFile *parent;
150 gboolean res = FALSE;
151
152 if (dot_gvfs_dir == NULL)
153 {
154 gchar *dot_gvfs;
155
156 dot_gvfs = g_build_filename (g_get_home_dir (), ".gvfs", NULL);
157
158 dot_gvfs_dir = g_file_new_for_path (dot_gvfs);
159
160 g_free (dot_gvfs);
161 }
162
163 parent = g_file_get_parent (file);
164
165 if (parent != NULL)
166 {
167 res = g_file_equal (parent, dot_gvfs_dir);
168 g_object_unref (parent);
169 }
170
171 return res;
172}

174static struct allsizes
175loopdir (GFile *file,
176 GFileInfo *info,
177 guint count,
178 BaobabHardLinkArray *hla,
179 gint current_depth)
180{
181 guint64 tempHLsize = 0;
182 gint elements = 0;
183 struct chan_data data;
184 struct allsizes retloop, temp;
185 GFileInfo *temp_info;
186 GFileEnumerator *file_enum;
187 gchar *dir_uri = NULL;
188 gchar *display_name = NULL;
189 gchar *parse_name = NULL;
190 GError *err = NULL;
191
192 count++;
193 retloop.size = 0;
194 retloop.alloc_size = 0;
195 retloop.depth = 0;
196
197 /* Skip the user excluded folders */
198 if (baobab_is_excluded_location (file))
199 goto exit;
200
201 /* Skip the virtual file systems */
202 if (is_virtual_filesystem (file))
203 goto exit;
204
205 /* FIXME: skip dirs in ~/.gvfs. It would be better to have a way
206 * to check if a file is a FUSE mountpoint instead of just
207 * hardcoding .gvfs */
208 if (is_in_dot_gvfs (file))
209 goto exit;
210
211 parse_name = g_file_get_parse_name (file);
212
213 if (g_file_info_has_attribute (info, G_FILE_ATTRIBUTE_STANDARD_SIZE)) {
214 goffset file_size = g_file_info_get_size (info);
215 retloop.size = (guint64) file_size;
216 }
217
218 if (g_file_info_has_attribute (info, G_FILE_ATTRIBUTE_UNIX_BLOCKS))
219 retloop.alloc_size = BLOCK_SIZE *
220 g_file_info_get_attribute_uint64 (info,
221 G_FILE_ATTRIBUTE_UNIX_BLOCKS);
222
223 if (g_file_info_has_attribute (info, G_FILE_ATTRIBUTE_STANDARD_DISPLAY_NAME))
224 display_name = g_strdup (g_file_info_get_display_name (info));
225 else
226 /* paranoid fallback */
227 display_name = g_filename_display_basename (g_file_info_get_name (info));
228
229 /* load up the file enumerator */
230 file_enum = g_file_enumerate_children (file,
231 dir_attributes,
232 G_FILE_QUERY_INFO_NOFOLLOW_SYMLINKS,
233 NULL,
234 &err);
235
236 if (file_enum == NULL) {
237 if (!g_error_matches (err, G_IO_ERROR, G_IO_ERROR_PERMISSION_DENIED)) {
238 g_warning ("couldn't get dir enum for dir %s: %s\n",
239 parse_name, err->message);
240 }
241 goto exit;
242 }
243
244 /* All skipped folders (i.e. bad type, excluded, /proc) must be
245 skept *before* this point. Everything passes the prefill-model
246 will be part of the GUI. */
247
248 /* prefill the model */
249 data.size = 1UL;
250 data.alloc_size = 1UL;
251 data.depth = (gint) count - 1;
252 data.elements = -1;
253 data.display_name = display_name;
254 data.parse_name = parse_name;
255 data.tempHLsize = tempHLsize;
256 baobab_fill_model (&data);
257
258 g_clear_error (&err);
259 while ((temp_info = g_file_enumerator_next_file (file_enum,
260 NULL,
261 &err)) != NULL) {
262 GFileType temp_type = g_file_info_get_file_type (temp_info);
263 if (baobab.STOP_SCANNING) {
264 g_object_unref (temp_info);
265 g_object_unref (file_enum);
266 goto exit;
267 }
268
269 /* is a directory? */
270 if (temp_type == G_FILE_TYPE_DIRECTORY) {
271 GFile *child_dir = g_file_get_child (file,
272 g_file_info_get_name (temp_info));
273 temp = loopdir (child_dir, temp_info, count, hla, current_depth+1);
274 retloop.size += temp.size;
275 retloop.alloc_size += temp.alloc_size;
276 retloop.depth = ((temp.depth + 1) > retloop.depth) ? temp.depth + 1 : retloop.depth;
277 elements++;
278 g_object_unref (child_dir);
279 }
280
281 /* is it a regular file? */
282 else if (temp_type == G_FILE_TYPE_REGULAR) {
283
284 /* check for hard links only on local files */
285 if (g_file_info_has_attribute (temp_info,
286 G_FILE_ATTRIBUTE_UNIX_NLINK) &&
287 g_file_info_get_attribute_uint32 (temp_info,
288 G_FILE_ATTRIBUTE_UNIX_NLINK) > 1) {
289
290 if (!baobab_hardlinks_array_add (hla, temp_info)) {
291 /* we already acconted for it */
292 goffset file_size = g_file_info_get_size (temp_info);
293 tempHLsize += (guint64) file_size;
294 g_object_unref (temp_info);
295 continue;
296 }
297 }
298
299 if (g_file_info_has_attribute (temp_info, G_FILE_ATTRIBUTE_UNIX_BLOCKS)) {
300 retloop.alloc_size += BLOCK_SIZE *
301 g_file_info_get_attribute_uint64 (temp_info,
302 G_FILE_ATTRIBUTE_UNIX_BLOCKS);
303 }
304 retloop.size += g_file_info_get_size (temp_info);
305 elements++;
306 }
307
308 /* ignore other types (symlinks, sockets, devices, etc) */
309
310 g_object_unref (temp_info);
311 }
312
313 /* won't be an error if we've finished normally */
314 if (err != NULL) {
315 g_warning ("error in dir %s: %s\n",
316 parse_name, err->message);
317 }
318
319 data.display_name = display_name;
320 data.parse_name = parse_name;
321 data.size = retloop.size;
322 data.alloc_size = retloop.alloc_size;
323 data.depth = (gint) count - 1;
324 data.elements = elements;
325 data.tempHLsize = tempHLsize;
326 baobab_fill_model (&data);
327 g_object_unref (file_enum);
328
329 exit:
330 g_free (dir_uri);
331 g_free (display_name);
332 g_free (parse_name);
333 if (err)
334 g_error_free (err);
335
336 return retloop;
337}

339void
340baobab_scan_execute (GFile *location)
341{
342 BaobabHardLinkArray *hla;
343 GFileInfo *info;
344 GError *err = NULL;
345 GFileType ftype;
346 struct allsizes sizes;
347
348 g_return_if_fail (location != NULL);
349
350 /* NOTE: for the root of the scan we follow symlinks */
351 info = g_file_query_info (location,
352 dir_attributes,
353 G_FILE_QUERY_INFO_NONE,
354 NULL,
355 &err);
356
357 if (info == NULL) {
358 char *parse_name = g_file_get_parse_name (location);
359 g_warning ("couldn't get info for dir %s: %s\n",
360 parse_name, err->message);
361 g_free (parse_name);
362 g_error_free (err);
363
364 return;
365 }
366
367 ftype = g_file_info_get_file_type (info);
368
369 if (ftype == G_FILE_TYPE_DIRECTORY) {
370 hla = baobab_hardlinks_array_create ();
371
372 sizes = loopdir (location, info, 0, hla, 0);
373 baobab.model_max_depth = sizes.depth;
374
375 baobab_hardlinks_array_free (hla);
376 }
377
378 g_object_unref (info);
379}
