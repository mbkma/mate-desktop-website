73static void
74ev_progress_message_area_init (EvProgressMessageArea *area)
75{
76 GtkWidget *contents;
77 GtkWidget *vbox;
78
79 area->priv = ev_progress_message_area_get_instance_private (area);
80
81 contents = _ev_message_area_get_main_box (EV_MESSAGE_AREA (area));
82
83 vbox = gtk_box_new (GTK_ORIENTATION_VERTICAL, 6);
84
85 area->priv->label = gtk_label_new (NULL);
86 gtk_label_set_use_markup (GTK_LABEL (area->priv->label), TRUE);
87 gtk_label_set_ellipsize (GTK_LABEL (area->priv->label),
88 PANGO_ELLIPSIZE_END);
89 gtk_label_set_xalign (GTK_LABEL (area->priv->label), 0.0);
90 gtk_box_pack_start (GTK_BOX (vbox), area->priv->label, TRUE, TRUE, 0);
91 gtk_widget_show (area->priv->label);
92
93 area->priv->progress_bar = gtk_progress_bar_new ();
94 gtk_widget_set_size_request (area->priv->progress_bar, -1, 15);
95 gtk_box_pack_start (GTK_BOX (vbox), area->priv->progress_bar, TRUE, FALSE, 0);
96 gtk_widget_show (area->priv->progress_bar);
97
98 gtk_box_pack_start (GTK_BOX (contents), vbox, TRUE, TRUE, 0);
99 gtk_widget_show (vbox);
100}

102static void
103ev_progress_message_area_set_property (GObject *object,
104 guint prop_id,
105 const GValue *value,
106 GParamSpec *pspec)
107{
108 EvProgressMessageArea *area = EV_PROGRESS_MESSAGE_AREA (object);
109
110 switch (prop_id) {
111 case PROP_STATUS:
112 ev_progress_message_area_set_status (area, g_value_get_string (value));
113 break;
114 case PROP_FRACTION:
115 ev_progress_message_area_set_fraction (area, g_value_get_double (value));
116 break;
117 default:
118 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
119 }
120}

122static void
123ev_progress_message_area_get_property (GObject *object,
124 guint prop_id,
125 GValue *value,
126 GParamSpec *pspec)
127{
128 EvProgressMessageArea *area = EV_PROGRESS_MESSAGE_AREA (object);
129
130 switch (prop_id) {
131 case PROP_STATUS:
132 g_value_set_string (value, gtk_label_get_label (GTK_LABEL (area->priv->label)));
133 break;
134 case PROP_FRACTION: {
135 gdouble fraction;
136
137 fraction = gtk_progress_bar_get_fraction (GTK_PROGRESS_BAR (area->priv->progress_bar));
138 g_value_set_double (value, fraction);
139 }
140 break;
141 default:
142 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
143 }
144}
