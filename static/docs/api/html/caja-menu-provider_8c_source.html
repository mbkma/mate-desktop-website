41static void
42caja_menu_provider_base_init (gpointer g_class)
43{
44 static gboolean initialized = FALSE;
45
46 if (!initialized) {
47 /* This signal should be emited each time the extension modify the list of menu items */
48 g_signal_new ("items_updated",
49 CAJA_TYPE_MENU_PROVIDER,
50 G_SIGNAL_RUN_LAST,
51 0,
52 NULL, NULL,
53 g_cclosure_marshal_VOID__VOID,
54 G_TYPE_NONE, 0);
55 initialized = TRUE;
56 }
57}
