123static char *
124caja_location_bar_get_location (CajaLocationBar *bar)
125{
126 char *user_location, *uri;
127 GFile *location;
128
129 user_location = gtk_editable_get_chars (GTK_EDITABLE (bar->details->entry), 0, -1);
130 location = g_file_parse_name (user_location);
131 g_free (user_location);
132 uri = g_file_get_uri (location);
133 g_object_unref (location);
134 return uri;
135}

149static void
150drag_data_received_callback (GtkWidget *widget,
151 GdkDragContext *context,
152 int x,
153 int y,
154 GtkSelectionData *data,
155 guint info,
156 guint32 time,
157 gpointer callback_data)
158{
159 char **names;
160 int name_count;
161 CajaNavigationWindow *window;
162 gboolean new_windows_for_extras;
163 CajaLocationBar *self = CAJA_LOCATION_BAR (widget);
164
165 g_assert (data != NULL);
166 g_assert (callback_data == NULL);
167
168 names = g_uri_list_extract_uris (gtk_selection_data_get_data (data));
169
170 if (names == NULL || *names == NULL)
171 {
172 g_warning ("No D&D URI's");
173 g_strfreev (names);
174 gtk_drag_finish (context, FALSE, FALSE, time);
175 return;
176 }
177
178 window = caja_location_bar_get_window (widget);
179 new_windows_for_extras = FALSE;
180 /* Ask user if they really want to open multiple windows
181 * for multiple dropped URIs. This is likely to have been
182 * a mistake.
183 */
184 name_count = g_strv_length (names);
185 if (name_count > 1)
186 {
187 char *prompt;
188 char *detail;
189
190 prompt = g_strdup_printf (ngettext("Do you want to view %d location?",
191 "Do you want to view %d locations?",
192 name_count),
193 name_count);
194 detail = g_strdup_printf (ngettext("This will open %d separate window.",
195 "This will open %d separate windows.",
196 name_count),
197 name_count);
198 /* eel_run_simple_dialog should really take in pairs
199 * like gtk_dialog_new_with_buttons() does. */
200 new_windows_for_extras = eel_run_simple_dialog
201 (GTK_WIDGET (window),
202 TRUE,
203 GTK_MESSAGE_QUESTION,
204 prompt,
205 detail,
206 "process-stop", "gtk-ok",
207 NULL) != 0 /* MATE_OK */;
208
209 g_free (prompt);
210 g_free (detail);
211
212 if (!new_windows_for_extras)
213 {
214 g_strfreev (names);
215 gtk_drag_finish (context, FALSE, FALSE, time);
216 return;
217 }
218 }
219
220 caja_location_bar_set_location (self, names[0]);
221 emit_location_changed (self);
222
223 if (new_windows_for_extras)
224 {
225 CajaApplication *application;
226 GdkScreen *screen;
227 int i;
228 CajaWindow *new_window = NULL;
229 GFile *location = NULL;
230
231 application = CAJA_WINDOW (window)->application;
232 screen = gtk_window_get_screen (GTK_WINDOW (window));
233
234 for (i = 1; names[i] != NULL; ++i)
235 {
236 new_window = caja_application_create_navigation_window (application, screen);
237
238 location = g_file_new_for_uri (names[i]);
239 caja_window_go_to (new_window, location);
240 g_object_unref (location);
241 }
242 }
243
244 g_strfreev (names);
245
246 gtk_drag_finish (context, TRUE, FALSE, time);
247}

249static void
250drag_data_get_callback (GtkWidget *widget,
251 GdkDragContext *context,
252 GtkSelectionData *selection_data,
253 guint info,
254 guint32 time,
255 gpointer callback_data)
256{
257 CajaLocationBar *self;
258 char *entry_text;
259
260 g_assert (selection_data != NULL);
261 self = callback_data;
262
263 entry_text = caja_location_bar_get_location (self);
264
265 switch (info)
266 {
267 case CAJA_DND_URI_LIST:
268 case CAJA_DND_TEXT_PLAIN:
269 gtk_selection_data_set (selection_data,
270 gtk_selection_data_get_target (selection_data),
271 8, (guchar *) entry_text,
272 eel_strlen (entry_text));
273 break;
274 default:
275 g_assert_not_reached ();
276 }
277 g_free (entry_text);
278}

280/* routine that determines the usize for the label widget as larger
281 then the size of the largest string and then sets it to that so
282 that we don't have localization problems. see
283 gtk_label_finalize_lines in gtklabel.c (line 618) for the code that

286static void
287style_set_handler (GtkWidget *widget, GtkStyleContext *previous_style)
288{
289 PangoLayout *layout;
290 int width, width2;
291 int xpad;
292 gint margin_start, margin_end;
293
294 layout = gtk_label_get_layout (GTK_LABEL(widget));
295
296 layout = pango_layout_copy (layout);
297
298 pango_layout_set_text (layout, LOCATION_LABEL, -1);
299 pango_layout_get_pixel_size (layout, &width, NULL);
300
301 pango_layout_set_text (layout, GO_TO_LABEL, -1);
302 pango_layout_get_pixel_size (layout, &width2, NULL);
303 width = MAX (width, width2);
304
305 margin_start = gtk_widget_get_margin_start (widget);
306 margin_end = gtk_widget_get_margin_end (widget);
307 xpad = margin_start + margin_end;
308
309 width += 2 * xpad;
310
311 gtk_widget_set_size_request (widget, width, -1);
312
313 g_object_unref (layout);
314}

316static gboolean
317label_button_pressed_callback (GtkWidget *widget,
318 GdkEventButton *event)
319{
320 CajaNavigationWindow *window;
321 CajaWindowSlot *slot;
322 CajaView *view;
323 GtkWidget *label;
324
325 if (event->button != 3)
326 {
327 return FALSE;
328 }
329
330 window = caja_location_bar_get_window (gtk_widget_get_parent (widget));
331 slot = CAJA_WINDOW (window)->details->active_pane->active_slot;
332 view = slot->content_view;
333 label = gtk_bin_get_child (GTK_BIN (widget));
334 /* only pop-up if the URI in the entry matches the displayed location */
335 if (view == NULL ||
336 strcmp (gtk_label_get_text (GTK_LABEL (label)), LOCATION_LABEL))
337 {
338 return FALSE;
339 }
340
341 caja_view_pop_up_location_context_menu (view, event, NULL);
342
343 return FALSE;
344}

366static void
367caja_location_bar_update_label (CajaLocationBar *bar)
368{
369 const char *current_text;
370 GFile *location;
371 GFile *last_location;
372
373 if (bar->details->last_location == NULL){
374 gtk_label_set_text (GTK_LABEL (bar->details->label), GO_TO_LABEL);
375 caja_location_entry_set_secondary_action (CAJA_LOCATION_ENTRY (bar->details->entry),
376 CAJA_LOCATION_ENTRY_ACTION_GOTO);
377 return;
378 }
379
380 current_text = gtk_entry_get_text (GTK_ENTRY (bar->details->entry));
381 location = g_file_parse_name (current_text);
382 last_location = g_file_parse_name (bar->details->last_location);
383
384 if (g_file_equal (last_location, location)) {
385 gtk_label_set_text (GTK_LABEL (bar->details->label), LOCATION_LABEL);
386 caja_location_entry_set_secondary_action (CAJA_LOCATION_ENTRY (bar->details->entry),
387 CAJA_LOCATION_ENTRY_ACTION_CLEAR);
388 } else {
389 gtk_label_set_text (GTK_LABEL (bar->details->label), GO_TO_LABEL);
390 caja_location_entry_set_secondary_action (CAJA_LOCATION_ENTRY (bar->details->entry),
391 CAJA_LOCATION_ENTRY_ACTION_GOTO);
392 }
393
394 g_object_unref (location);
395 g_object_unref (last_location);
396}

477static void
478caja_location_bar_init (CajaLocationBar *bar)
479{
480 GtkWidget *label;
481 GtkWidget *entry;
482 GtkWidget *event_box;
483
484 bar->details = caja_location_bar_get_instance_private (bar);
485
486 gtk_orientable_set_orientation (GTK_ORIENTABLE (bar),
487 GTK_ORIENTATION_HORIZONTAL);
488
489 event_box = gtk_event_box_new ();
490 gtk_event_box_set_visible_window (GTK_EVENT_BOX (event_box), FALSE);
491
492 gtk_container_set_border_width (GTK_CONTAINER (event_box), 4);
493 label = gtk_label_new (LOCATION_LABEL);
494 gtk_container_add (GTK_CONTAINER (event_box), label);
495 gtk_label_set_justify (GTK_LABEL (label), GTK_JUSTIFY_RIGHT);
496 gtk_label_set_xalign (GTK_LABEL (label), 1.0);
497 gtk_label_set_yalign (GTK_LABEL (label), 0.5);
498 g_signal_connect (label, "style_set",
499 G_CALLBACK (style_set_handler), NULL);
500
501 gtk_box_pack_start (GTK_BOX (bar), event_box, FALSE, TRUE, 4);
502
503 entry = caja_location_entry_new ();
504
505 g_signal_connect_object (entry, "activate",
506 G_CALLBACK (editable_activate_callback), bar, G_CONNECT_AFTER);
507 g_signal_connect_object (entry, "changed",
508 G_CALLBACK (editable_changed_callback), bar, 0);
509
510 gtk_box_pack_start (GTK_BOX (bar), entry, TRUE, TRUE, 0);
511
512 eel_accessibility_set_up_label_widget_relation (label, entry);
513
514 /* Label context menu */
515 g_signal_connect (event_box, "button-press-event",
516 G_CALLBACK (label_button_pressed_callback), NULL);
517
518 /* Drag source */
519 gtk_drag_source_set (GTK_WIDGET (event_box),
520 GDK_BUTTON1_MASK | GDK_BUTTON3_MASK,
521 drag_types, G_N_ELEMENTS (drag_types),
522 GDK_ACTION_COPY | GDK_ACTION_LINK);
523 g_signal_connect_object (event_box, "drag_data_get",
524 G_CALLBACK (drag_data_get_callback), bar, 0);
525
526 /* Drag dest. */
527 gtk_drag_dest_set (GTK_WIDGET (bar),
528 GTK_DEST_DEFAULT_ALL,
529 drop_types, G_N_ELEMENTS (drop_types),
530 GDK_ACTION_COPY | GDK_ACTION_MOVE | GDK_ACTION_LINK);
531 g_signal_connect (bar, "drag_data_received",
532 G_CALLBACK (drag_data_received_callback), NULL);
533
534 bar->details->label = GTK_LABEL (label);
535 bar->details->entry = CAJA_ENTRY (entry);
536
537 gtk_widget_show_all (GTK_WIDGET (bar));
538}

558void
559caja_location_bar_set_location (CajaLocationBar *bar,
560 const char *location)
561{
562 g_assert (location != NULL);
563
564 /* Note: This is called in reaction to external changes, and
565 * thus should not emit the LOCATION_CHANGED signal. */
566
567 if (eel_uri_is_search (location))
568 {
569 caja_location_entry_set_special_text (CAJA_LOCATION_ENTRY (bar->details->entry),
570 "");
571 }
572 else
573 {
574 char *formatted_location;
575 GFile *file;
576
577 file = g_file_new_for_uri (location);
578 formatted_location = g_file_get_parse_name (file);
579 g_object_unref (file);
580 caja_location_entry_update_current_location (CAJA_LOCATION_ENTRY (bar->details->entry),
581 formatted_location);
582 g_free (formatted_location);
583 }
584
585 /* remember the original location for later comparison */
586
587 if (bar->details->last_location != location)
588 {
589 g_free (bar->details->last_location);
590 bar->details->last_location = g_strdup (location);
591 }
592
593 caja_location_bar_update_label (bar);
594}

616/* change background color based on activity state */
617void
618caja_location_bar_set_active (CajaLocationBar *location_bar, gboolean is_active)
619{
620 GtkStyleContext *style;
621 GdkRGBA color;
622 GdkRGBA *c;
623 static GdkRGBA bg_active;
624 static GdkRGBA bg_inactive;
625
626 style = gtk_widget_get_style_context (GTK_WIDGET (location_bar->details->entry));
627
628 if (is_active)
629 gtk_style_context_get (style, GTK_STATE_FLAG_NORMAL,
630 GTK_STYLE_PROPERTY_BACKGROUND_COLOR,
631 &c, NULL);
632 else
633 gtk_style_context_get (style, GTK_STATE_FLAG_INSENSITIVE,
634 GTK_STYLE_PROPERTY_BACKGROUND_COLOR,
635 &c, NULL);
636
637 color = *c;
638 gdk_rgba_free (c);
639
640 if (is_active)
641 {
642 if (gdk_rgba_equal (&bg_active, &bg_inactive))
643 bg_active = color;
644
645 override_background_color (GTK_WIDGET (location_bar->details->entry), &bg_active);
646 }
647 else
648 {
649 if (gdk_rgba_equal (&bg_active, &bg_inactive))
650 bg_inactive = color;
651
652 override_background_color(GTK_WIDGET (location_bar->details->entry), &bg_inactive);
653 }
654}
