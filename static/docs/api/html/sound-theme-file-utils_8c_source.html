61static gboolean
62directory_delete_recursive (GFile *directory, GError **error)
63{
64 GFileEnumerator *enumerator;
65 GFileInfo *info;
66 gboolean success = TRUE;
67
68 enumerator = g_file_enumerate_children (directory,
69 G_FILE_ATTRIBUTE_STANDARD_NAME ","
70 G_FILE_ATTRIBUTE_STANDARD_TYPE,
71 G_FILE_QUERY_INFO_NONE,
72 NULL, error);
73 if (enumerator == NULL)
74 return FALSE;
75
76 while (success &&
77 (info = g_file_enumerator_next_file (enumerator, NULL, NULL))) {
78 GFile *child;
79
80 child = g_file_get_child (directory, g_file_info_get_name (info));
81
82 if (g_file_info_get_file_type (info) == G_FILE_TYPE_DIRECTORY) {
83 success = directory_delete_recursive (child, error);
84 }
85 g_object_unref (info);
86
87 if (success)
88 success = g_file_delete (child, NULL, error);
89 }
90 g_file_enumerator_close (enumerator, NULL, NULL);
91
92 if (success)
93 success = g_file_delete (directory, NULL, error);
94
95 return success;
96}

106static gboolean
107capplet_file_delete_recursive (GFile *file, GError **error)
108{
109 GFileInfo *info;
110 GFileType type;
111
112 g_return_val_if_fail (error == NULL || *error == NULL, FALSE);
113
114 info = g_file_query_info (file,
115 G_FILE_ATTRIBUTE_STANDARD_TYPE,
116 G_FILE_QUERY_INFO_NONE,
117 NULL, error);
118 if (info == NULL)
119 return FALSE;
120
121 type = g_file_info_get_file_type (info);
122 g_object_unref (info);
123
124 if (type == G_FILE_TYPE_DIRECTORY)
125 return directory_delete_recursive (file, error);
126 else
127 return g_file_delete (file, NULL, error);
128}

145gboolean
146custom_theme_dir_is_empty (void)
147{
148 char *dir;
149 GFile *file;
150 gboolean is_empty;
151 GFileEnumerator *enumerator;
152 GFileInfo *info;
153 GError *error = NULL;
154
155 dir = custom_theme_dir_path (NULL);
156 file = g_file_new_for_path (dir);
157 g_free (dir);
158
159 is_empty = TRUE;
160
161 enumerator = g_file_enumerate_children (file,
162 G_FILE_ATTRIBUTE_STANDARD_NAME ","
163 G_FILE_ATTRIBUTE_STANDARD_TYPE,
164 G_FILE_QUERY_INFO_NONE,
165 NULL, &error);
166 if (enumerator == NULL) {
167 g_warning ("Unable to enumerate files: %s", error->message);
168 g_error_free (error);
169 goto out;
170 }
171
172 while (is_empty &&
173 (info = g_file_enumerator_next_file (enumerator, NULL, NULL))) {
174
175 if (strcmp ("index.theme", g_file_info_get_name (info)) != 0) {
176 is_empty = FALSE;
177 }
178
179 g_object_unref (info);
180 }
181 g_file_enumerator_close (enumerator, NULL, NULL);
182
183 out:
184 g_object_unref (file);
185
186 return is_empty;
187}

255void
256add_custom_file (const char **sounds, const char *filename)
257{
258 guint i;
259
260 for (i = 0; sounds[i] != NULL; i++) {
261 GFile *file;
262 char *name, *path;
263
264 /* We use *.ogg because it's the first type of file that
265 * libcanberra looks at */
266 name = g_strdup_printf ("%s.ogg", sounds[i]);
267 path = custom_theme_dir_path (name);
268 g_free (name);
269 /* In case there's already a link there, delete it */
270 g_unlink (path);
271 file = g_file_new_for_path (path);
272 g_free (path);
273
274 /* Create the link */
275 g_file_make_symbolic_link (file, filename, NULL, NULL);
276 g_object_unref (file);
277 }

306 custom_theme_update_time ();
