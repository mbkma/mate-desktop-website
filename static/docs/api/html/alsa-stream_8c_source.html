68static void
69alsa_stream_dispose (GObject *object)
70{
71 AlsaStream *stream;
72
73 stream = ALSA_STREAM (object);
74
75 if (stream->priv->controls != NULL) {
76 g_list_free_full (stream->priv->controls, g_object_unref);
77 stream->priv->controls = NULL;
78 }
79 if (stream->priv->switches != NULL) {
80 g_list_free_full (stream->priv->switches, g_object_unref);
81 stream->priv->switches = NULL;
82 }
83
84 G_OBJECT_CLASS (alsa_stream_parent_class)->dispose (object);
85}

107void
108alsa_stream_add_control (AlsaStream *stream, AlsaStreamControl *control)
109{
110 const gchar *name;
111
112 g_return_if_fail (ALSA_IS_STREAM (stream));
113 g_return_if_fail (ALSA_IS_STREAM_CONTROL (control));
114
115 name = mate_mixer_stream_control_get_name (MATE_MIXER_STREAM_CONTROL (control));
116
117 stream->priv->controls =
118 g_list_append (stream->priv->controls, g_object_ref (control));
119
120 g_signal_emit_by_name (G_OBJECT (stream),
121 "control-added",
122 name);
123
124 if (alsa_stream_has_default_control (stream) == FALSE)
125 alsa_stream_set_default_control (stream, control);
126}

237void
238alsa_stream_load_elements (AlsaStream *stream, const gchar *name)
239{
240 GList *item;
241
242 g_return_if_fail (ALSA_IS_STREAM (stream));
243 g_return_if_fail (name != NULL);
244
245 item = g_list_find_custom (stream->priv->controls, name, compare_control_name);
246 if (item != NULL)
247 alsa_element_load (ALSA_ELEMENT (item->data));
248
249 item = g_list_find_custom (stream->priv->switches, name, compare_switch_name);
250 if (item != NULL)
251 alsa_element_load (ALSA_ELEMENT (item->data));
252}

254gboolean
255alsa_stream_remove_elements (AlsaStream *stream, const gchar *name)
256{
257 GList *item;
258 gboolean removed = FALSE;
259
260 g_return_val_if_fail (ALSA_IS_STREAM (stream), FALSE);
261 g_return_val_if_fail (name != NULL, FALSE);
262
263 item = g_list_find_custom (stream->priv->controls, name, compare_control_name);
264 if (item != NULL) {
265 MateMixerStreamControl *control = MATE_MIXER_STREAM_CONTROL (item->data);
266
267 alsa_element_close (ALSA_ELEMENT (control));
268 stream->priv->controls = g_list_delete_link (stream->priv->controls, item);
269
270 /* Change the default control if we have just removed it */
271 if (control == mate_mixer_stream_get_default_control (MATE_MIXER_STREAM (stream))) {
272 AlsaStreamControl *first = NULL;
273
274 if (stream->priv->controls != NULL)
275 first = ALSA_STREAM_CONTROL (stream->priv->controls->data);
276
277 alsa_stream_set_default_control (stream, first);
278 }
279
280 g_signal_emit_by_name (G_OBJECT (stream),
281 "control-removed",
282 mate_mixer_stream_control_get_name (control));
283
284 g_object_unref (control);
285 removed = TRUE;
286 }
287
288 item = g_list_find_custom (stream->priv->switches, name, compare_switch_name);
289 if (item != NULL) {
290 MateMixerSwitch *swtch = MATE_MIXER_SWITCH (item->data);
291
292 alsa_element_close (ALSA_ELEMENT (swtch));
293
294 stream->priv->switches = g_list_delete_link (stream->priv->switches, item);
295 g_signal_emit_by_name (G_OBJECT (stream),
296 "switch-removed",
297 mate_mixer_switch_get_name (swtch));
298
299 g_object_unref (swtch);
300 removed = TRUE;
301 }
302
303 return removed;
304}

306void
307alsa_stream_remove_all (AlsaStream *stream)
308{
309 GList *list;
310
311 g_return_if_fail (ALSA_IS_STREAM (stream));
312
313 /* Remove all stream controls */
314 list = stream->priv->controls;
315 while (list != NULL) {
316 MateMixerStreamControl *control = MATE_MIXER_STREAM_CONTROL (list->data);
317 GList *next = list->next;
318
319 alsa_element_close (ALSA_ELEMENT (control));
320
321 stream->priv->controls = g_list_delete_link (stream->priv->controls, list);
322 g_signal_emit_by_name (G_OBJECT (stream),
323 "control-removed",
324 mate_mixer_stream_control_get_name (control));
325
326 g_object_unref (control);
327 list = next;
328 }
329
330 /* Unset the default stream control */
331 alsa_stream_set_default_control (stream, NULL);
332
333 /* Remove all stream switches */
334 list = stream->priv->switches;
335 while (list != NULL) {
336 MateMixerSwitch *swtch = MATE_MIXER_SWITCH (list->data);
337 GList *next = list->next;
338
339 alsa_element_close (ALSA_ELEMENT (swtch));
340
341 stream->priv->switches = g_list_delete_link (stream->priv->switches, list);
342 g_signal_emit_by_name (G_OBJECT (stream),
343 "switch-removed",
344 mate_mixer_switch_get_name (swtch));
345
346 g_object_unref (swtch);
347 list = next;
348 }
349}
