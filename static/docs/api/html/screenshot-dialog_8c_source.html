72static void
73on_preview_draw (GtkWidget *drawing_area,
74 cairo_t *cr,
75 gpointer data)
76{
77 ScreenshotDialog *dialog = data;
78 GtkStyleContext *context;
79 int width, height;
80
81 width = gtk_widget_get_allocated_width (drawing_area);
82 height = gtk_widget_get_allocated_height (drawing_area);
83
84 if (!dialog->preview_image ||
85 gdk_pixbuf_get_width (dialog->preview_image) != width ||
86 gdk_pixbuf_get_height (dialog->preview_image) != height)
87 {
88 g_clear_object (&dialog->preview_image);
89 dialog->preview_image = gdk_pixbuf_scale_simple (dialog->screenshot,
90 width,
91 height,
92 GDK_INTERP_BILINEAR);
93 }
94
95 context = gtk_widget_get_style_context (drawing_area);
96 gtk_style_context_save (context);
97
98 gtk_style_context_set_state (context, gtk_widget_get_state_flags (drawing_area));
99 gtk_render_icon (context, cr, dialog->preview_image, 0, 0);
100
101 gtk_style_context_restore (context);
102}

130static void
131drag_data_get (GtkWidget *widget,
132 GdkDragContext *context,
133 GtkSelectionData *selection_data,
134 guint info,
135 guint time,
136 ScreenshotDialog *dialog)
137{
138 if (info == TYPE_TEXT_URI_LIST)
139 {
140 gchar **uris;
141
142 uris = g_new (gchar *, 2);
143 uris[0] = g_strconcat ("file://",
144 screenshot_save_get_filename (),
145 NULL);
146 uris[1] = NULL;
147
148 gtk_selection_data_set_uris (selection_data, uris);
149 g_strfreev (uris);
150 }
151 else if (info == TYPE_IMAGE_PNG)
152 {
153 gtk_selection_data_set_pixbuf (selection_data, dialog->screenshot);
154 }
155 else
156 {
157 g_warning ("Unknown type %d", info);
158 }
159}

170ScreenshotDialog *
171screenshot_dialog_new (GdkPixbuf *screenshot,
172 char *initial_uri,
173 gboolean take_window_shot)
174{
175 ScreenshotDialog *dialog;
176 GtkWidget *toplevel;
177 GtkWidget *preview_darea;
178 GtkWidget *aspect_frame;
179 GtkWidget *file_chooser_box;
180 gint width, height;
181 char *current_folder;
182 char *current_name;
183 char *ext;
184 gint pos;
185 GFile *tmp_file;
186 GFile *parent_file;
187 GError *error = NULL;
188
189 tmp_file = g_file_new_for_uri (initial_uri);
190 parent_file = g_file_get_parent (tmp_file);
191
192 current_name = g_file_get_basename (tmp_file);
193 current_folder = g_file_get_uri (parent_file);
194 g_object_unref (tmp_file);
195 g_object_unref (parent_file);
196
197 dialog = g_new0 (ScreenshotDialog, 1);
198
199 dialog->builder = gtk_builder_new ();
200
201 dialog->screenshot = screenshot;
202
203 if (gtk_builder_add_from_resource (dialog->builder, "/org/mate/screenshot/mate-screenshot.ui", &error) == 0)
204 {
205 GtkWidget *dialog;
206 dialog = gtk_message_dialog_new (NULL, 0,
207 GTK_MESSAGE_ERROR,
208 GTK_BUTTONS_OK,
209 _("Error loading UI definition file for the screenshot program: \n%s\n\n"
210 "Please check your installation of mate-utils."), error->message);
211 gtk_dialog_run (GTK_DIALOG (dialog));
212 gtk_widget_destroy (dialog);
213
214 g_error_free (error);
215 exit (1);
216 }
217
218 gtk_builder_set_translation_domain (dialog->builder, GETTEXT_PACKAGE);
219
220 width = gdk_pixbuf_get_width (screenshot);
221 height = gdk_pixbuf_get_height (screenshot);
222
223 width /= 5;
224 height /= 5;
225
226 toplevel = GET_WIDGET ("toplevel");
227 aspect_frame = GET_WIDGET ("aspect_frame");
228 preview_darea = GET_WIDGET ("preview_darea");
229 dialog->filename_entry = GET_WIDGET ("filename_entry");
230 file_chooser_box = GET_WIDGET ("file_chooser_box");
231
232 dialog->save_widget = gtk_file_chooser_button_new (_("Select a folder"), GTK_FILE_CHOOSER_ACTION_SELECT_FOLDER);
233 gtk_file_chooser_set_local_only (GTK_FILE_CHOOSER (dialog->save_widget), FALSE);
234 gtk_file_chooser_set_current_folder_uri (GTK_FILE_CHOOSER (dialog->save_widget), current_folder);
235 gtk_entry_set_text (GTK_ENTRY (dialog->filename_entry), current_name);
236
237 gtk_box_pack_start (GTK_BOX (file_chooser_box), dialog->save_widget, TRUE, TRUE, 0);
238 g_free (current_folder);
239
240 gtk_widget_set_size_request (preview_darea, width, height);
241 height = gdk_pixbuf_get_height (screenshot);
242 width = gdk_pixbuf_get_width (screenshot);
243 gtk_aspect_frame_set (GTK_ASPECT_FRAME (aspect_frame), 0.0, 0.5,
244 (gfloat) width / (gfloat) height, FALSE);
245 g_signal_connect (toplevel, "key_press_event", G_CALLBACK (on_toplevel_key_press_event), dialog);
246 g_signal_connect (preview_darea, "draw", G_CALLBACK (on_preview_draw), dialog);
247 g_signal_connect (preview_darea, "button_press_event", G_CALLBACK (on_preview_button_press_event), dialog);
248 g_signal_connect (preview_darea, "button_release_event", G_CALLBACK (on_preview_button_release_event), dialog);
249
250 if (take_window_shot)
251 gtk_frame_set_shadow_type (GTK_FRAME (aspect_frame), GTK_SHADOW_NONE);
252 else
253 gtk_frame_set_shadow_type (GTK_FRAME (aspect_frame), GTK_SHADOW_IN);
254
255 /* setup dnd */
256 g_signal_connect (preview_darea, "drag_begin",
257 G_CALLBACK (drag_begin),
258 dialog);
259 g_signal_connect (preview_darea, "drag_data_get",
260 G_CALLBACK (drag_data_get),
261 dialog);
262
263 gtk_widget_show_all (toplevel);
264
265 /* select the name of the file but leave out the extension if there's any;
266 * the dialog must be realized for select_region to work
267 */
268 ext = g_utf8_strrchr (current_name, -1, '.');
269 if (ext)
270 pos = g_utf8_strlen (current_name, -1) - g_utf8_strlen (ext, -1);
271 else
272 pos = -1;
273
274 gtk_editable_select_region (GTK_EDITABLE (dialog->filename_entry),
275 0,
276 pos);
277
278 g_free (current_name);
279
280 return dialog;
281}

309char *
310screenshot_dialog_get_uri (ScreenshotDialog *dialog)
311{
312 gchar *folder;
313 const gchar *file_name;
314 gchar *uri, *file, *tmp;
315 GError *error;
316
317 folder = gtk_file_chooser_get_uri (GTK_FILE_CHOOSER (dialog->save_widget));
318 file_name = gtk_entry_get_text (GTK_ENTRY (dialog->filename_entry));
319
320 error = NULL;
321 tmp = g_filename_from_utf8 (file_name, -1, NULL, NULL, &error);
322 if (error)
323 {
324 g_warning ("Unable to convert `%s' to valid UTF-8: %s\n"
325 "Falling back to default file.",
326 file_name,
327 error->message);
328 g_error_free (error);
329 tmp = g_strdup (_("Screenshot.png"));
330 }
331
332 file = g_uri_escape_string (tmp, NULL, FALSE);
333 uri = g_build_filename (folder, file, NULL);
334
335 g_free (folder);
336 g_free (tmp);
337 g_free (file);
338
339 return uri;
340}

354void
355screenshot_dialog_set_busy (ScreenshotDialog *dialog,
356 gboolean busy)
357{
358 GtkWidget *toplevel;
359 GdkDisplay *display;
360
361 toplevel = screenshot_dialog_get_toplevel (dialog);
362 display = gtk_widget_get_display (toplevel);
363
364 if (busy)
365 {
366 GdkCursor *cursor;
367 /* Change cursor to busy */
368 cursor = gdk_cursor_new_for_display (display, GDK_WATCH);
369 gdk_window_set_cursor (gtk_widget_get_window (toplevel), cursor);
370 g_object_unref (cursor);
371 }
372 else
373 {
374 gdk_window_set_cursor (gtk_widget_get_window (toplevel), NULL);
375 }
376
377 gtk_widget_set_sensitive (toplevel, ! busy);
378
379 gdk_display_flush (display);
380}
