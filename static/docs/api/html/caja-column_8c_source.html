101static void
102caja_column_get_property (GObject *object,
103 guint param_id,
104 GValue *value,
105 GParamSpec *pspec)
106{
107 CajaColumn *column;
108
109 column = CAJA_COLUMN (object);
110
111 switch (param_id) {
112 case PROP_NAME :
113 g_value_set_string (value, column->details->name);
114 break;
115 case PROP_ATTRIBUTE :
116 g_value_set_string (value, g_quark_to_string (column->details->attribute));
117 break;
118 case PROP_ATTRIBUTE_Q :
119 g_value_set_uint (value, column->details->attribute);
120 break;
121 case PROP_LABEL :
122 g_value_set_string (value, column->details->label);
123 break;
124 case PROP_DESCRIPTION :
125 g_value_set_string (value, column->details->description);
126 break;
127 case PROP_XALIGN :
128 g_value_set_float (value, column->details->xalign);
129 break;
130 default :
131 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, param_id, pspec);
132 break;
133 }

136static void
137caja_column_set_property (GObject *object,
138 guint param_id,
139 const GValue *value,
140 GParamSpec *pspec)
141{
142 CajaColumn *column;
143
144 column = CAJA_COLUMN (object);
145
146 switch (param_id) {
147 case PROP_NAME :
148 g_free (column->details->name);
149 column->details->name = g_strdup (g_value_get_string (value));
150 g_object_notify (object, "name");
151 break;
152 case PROP_ATTRIBUTE :
153 column->details->attribute = g_quark_from_string (g_value_get_string (value));
154 g_object_notify (object, "attribute");
155 g_object_notify (object, "attribute_q");
156 break;
157 case PROP_LABEL :
158 g_free (column->details->label);
159 column->details->label = g_strdup (g_value_get_string (value));
160 g_object_notify (object, "label");
161 break;
162 case PROP_DESCRIPTION :
163 g_free (column->details->description);
164 column->details->description = g_strdup (g_value_get_string (value));
165 g_object_notify (object, "description");
166 break;
167 case PROP_XALIGN :
168 column->details->xalign = g_value_get_float (value);
169 g_object_notify (object, "xalign");
170 break;
171 default :
172 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, param_id, pspec);
173 break;
174 }
