45static void
46solve(const char *equation)
47{
48 MPEquationOptions options;
49 MPErrorCode error;
50 MPNumber result = mp_new();
51 char *result_str;
52
53 memset(&options, 0, sizeof(options));
54 options.base = 10;
55 options.wordlen = 32;
56 options.angle_units = MP_DEGREES;
57 options.convert = do_convert;
58
59 error = mp_equation_parse(equation, &options, &result, NULL);
60 if(error == PARSER_ERR_MP) {
61 fprintf(stderr, "Error: %s\n", mp_get_error());
62 mp_clear(&result);
63 exit(1);
64 }
65 else if(error != 0) {
66 fprintf(stderr, "Error: %s\n", mp_error_code_to_string(error));
67 mp_clear(&result);
68 exit(1);
69 }
70 else {
71 result_str = mp_serializer_to_string(mp_serializer_new(MP_DISPLAY_FORMAT_AUTOMATIC, 10, 9), &result);
72 printf("%s\n", result_str);
73 mp_clear(&result);
74 exit(0);
75 }
76}

123static void
124get_options(int argc, char *argv[])
125{
126 int i;
127 char *progname, *arg;
128
129 progname = g_path_get_basename(argv[0]);
130
131 for (i = 1; i < argc; i++) {
132 arg = argv[i];
133
134 if (strcmp(arg, "-v") == 0 ||
135 strcmp(arg, "--version") == 0) {
136 version(progname);
137 g_free(progname);
138 exit(0);
139 }
140 else if (strcmp(arg, "-h") == 0 ||
141 strcmp(arg, "-?") == 0 ||
142 strcmp(arg, "--help") == 0) {
143 usage(progname, TRUE, FALSE);
144 g_free(progname);
145 exit(0);
146 }
147 else if (strcmp(arg, "--help-all") == 0) {
148 usage(progname, TRUE, TRUE);
149 g_free(progname);
150 exit(0);
151 }
152 else if (strcmp(arg, "--help-gtk") == 0) {
153 usage(progname, FALSE, TRUE);
154 g_free(progname);
155 exit(0);
156 }
157 else if (strcmp(arg, "-s") == 0 ||
158 strcmp(arg, "--solve") == 0) {
159 i++;
160 if (i >= argc) {
161 fprintf(stderr,
162 /* Error printed to stderr when user uses --solve argument without an equation */
163 _("Argument --solve requires an equation to solve"));
164 fprintf(stderr, "\n");
165 g_free(progname);
166 exit(1);
167 }
168 else
169 solve(argv[i]);
170 }
171 else {
172 fprintf(stderr,
173 /* Error printed to stderr when user provides an unknown command-line argument */
174 _("Unknown argument '%s'"), arg);
175 fprintf(stderr, "\n");
176 usage(progname, TRUE, FALSE);
177 g_free(progname);
178 exit(1);
179 }
180 }
181
182 g_free(progname);
183}
