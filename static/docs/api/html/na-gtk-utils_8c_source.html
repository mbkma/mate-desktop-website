45/*
46 * na_gtk_utils_find_widget_by_name:
47 * @container: a #GtkContainer, usually the #GtkWindow toplevel.
48 * @name: the name of the searched widget.
49 *
50 * Returns: the searched widget.
51 */
52GtkWidget *
53na_gtk_utils_find_widget_by_name( GtkContainer *container, const gchar *name )
54{
55 GList *children = gtk_container_get_children( container );
56 GList *ic;
57 GtkWidget *found = NULL;
58 GtkWidget *child;
59 const gchar *child_name;
60
61 for( ic = children ; ic && !found ; ic = ic->next ){
62
63 if( GTK_IS_WIDGET( ic->data )){
64 child = GTK_WIDGET( ic->data );
65 child_name = gtk_buildable_get_name( GTK_BUILDABLE( child ));
66 if( child_name && strlen( child_name ) && !g_ascii_strcasecmp( name, child_name )){
67 found = child;
68 break;
69 }
70 if( GTK_IS_CONTAINER( child )){
71 found = na_gtk_utils_find_widget_by_name( GTK_CONTAINER( child ), name );
72 }
73 }
74 }
75
76 g_list_free( children );
77 return( found );
78}

81static void
82dump_children( const gchar *thisfn, GtkContainer *container, int level )
83{
84 GList *children = gtk_container_get_children( container );
85 GList *ic;
86 GtkWidget *child;
87 const gchar *child_name;
88 GString *prefix;
89 int i;
90
91 prefix = g_string_new( "" );
92 for( i = 0 ; i <= level ; ++i ){
93 g_string_append_printf( prefix, "%s", "| " );
94 }
95
96 for( ic = children ; ic ; ic = ic->next ){
97
98 if( GTK_IS_WIDGET( ic->data )){
99 child = GTK_WIDGET( ic->data );
100 child_name = gtk_buildable_get_name( GTK_BUILDABLE( child ));
101 g_debug( "%s: %s%s\t%p %s",
102 thisfn, prefix->str, G_OBJECT_TYPE_NAME( child ), ( void * ) child, child_name );
103
104 if( GTK_IS_CONTAINER( child )){
105 dump_children( thisfn, GTK_CONTAINER( child ), level+1 );
106 }
107 }
108 }
109
110 g_list_free( children );
111 g_string_free( prefix, TRUE );
112}

134void
135na_gtk_utils_restore_window_position( GtkWindow *toplevel, const gchar *wsp_name )
136{
137 static const gchar *thisfn = "na_gtk_utils_restore_window_position";
138 GList *list;
139 gint x=0, y=0, width=0, height=0;
140 GdkDisplay *display;
141 GdkScreen *screen;
142 gint screen_width, screen_height;
143
144 g_return_if_fail( GTK_IS_WINDOW( toplevel ));
145 g_return_if_fail( wsp_name && strlen( wsp_name ));
146
147 g_debug( "%s: toplevel=%p (%s), wsp_name=%s",
148 thisfn, ( void * ) toplevel, G_OBJECT_TYPE_NAME( toplevel ), wsp_name );
149
150 list = na_settings_get_uint_list( wsp_name, NULL, NULL );
151
152 if( list ){
153 int_list_to_position( list, &x, &y, &width, &height );
154 g_debug( "%s: wsp_name=%s, x=%d, y=%d, width=%d, height=%d", thisfn, wsp_name, x, y, width, height );
155 free_int_list( list );
156 }
157
158 x = MAX( 1, x );
159 y = MAX( 1, y );
160 width = MAX( 1, width );
161 height = MAX( 1, height );
162
163 /* bad hack for the first time we open the main window
164 * try to target an ideal size and position
165 */
166 if( !strcmp( wsp_name, NA_IPREFS_MAIN_WINDOW_WSP )){
167 if( x == 1 && y == 1 && width == 1 && height == 1 ){
168 x = 50;
169 y = 70;
170 width = 1030;
171 height = 560;
172
173 } else {
174 display = gdk_display_get_default();
175 screen = gdk_display_get_default_screen( display );
176 screen_width = WidthOfScreen (gdk_x11_screen_get_xscreen (screen));
177 screen_height = HeightOfScreen (gdk_x11_screen_get_xscreen (screen));
178 g_debug( "%s: screen=(%d,%d), DEFAULT_HEIGHT=%d",
179 thisfn, screen_width, screen_height, DEFAULT_HEIGHT );
180
181 screen_height -= 2*DEFAULT_HEIGHT;
182 width = MIN( width, screen_width-x );
183 height = MIN( height, screen_height-y );
184 }
185 }
186
187 g_debug( "%s: wsp_name=%s, x=%d, y=%d, width=%d, height=%d",
188 thisfn, wsp_name, x, y, width, height );
189
190 gtk_window_move( toplevel, x, y );
191 gtk_window_resize( toplevel, width, height );
192}

220/*
221 * extract the position of the window from the list of unsigned integers
222 */
223static void
224int_list_to_position( GList *list, gint *x, gint *y, gint *width, gint *height )
225{
226 GList *it;
227 int i;
228
229 g_assert( x );
230 g_assert( y );
231 g_assert( width );
232 g_assert( height );
233
234 for( it=list, i=0 ; it ; it=it->next, i+=1 ){
235 switch( i ){
236 case 0:
237 *x = GPOINTER_TO_UINT( it->data );
238 break;
239 case 1:
240 *y = GPOINTER_TO_UINT( it->data );
241 break;
242 case 2:
243 *width = GPOINTER_TO_UINT( it->data );
244 break;
245 case 3:
246 *height = GPOINTER_TO_UINT( it->data );
247 break;
248 }
249 }
250}

290void
291na_gtk_utils_set_editable( GObject *widget, gboolean editable )
292{
293 GList *renderers, *irender;
294
295/* GtkComboBoxEntry is deprecated from Gtk+3
296 * see. http://git.gnome.org/browse/gtk+/commit/?id=9612c648176378bf237ad0e1a8c6c995b0ca7c61
297 * while 'has_entry' property exists since 2.24
298 */
299 if( GTK_IS_COMBO_BOX( widget ) && gtk_combo_box_get_has_entry( GTK_COMBO_BOX( widget ))){
300 /* idem as GtkEntry */
301 gtk_editable_set_editable( GTK_EDITABLE( gtk_bin_get_child( GTK_BIN( widget ))), editable );
302 g_object_set( G_OBJECT( gtk_bin_get_child( GTK_BIN( widget ))), "can-focus", editable, NULL );
303 /* disable the listbox button itself */
304 gtk_combo_box_set_button_sensitivity( GTK_COMBO_BOX( widget ), editable ? GTK_SENSITIVITY_ON : GTK_SENSITIVITY_OFF );
305
306 } else if( GTK_IS_COMBO_BOX( widget )){
307 /* disable the listbox button itself */
308 gtk_combo_box_set_button_sensitivity( GTK_COMBO_BOX( widget ), editable ? GTK_SENSITIVITY_ON : GTK_SENSITIVITY_OFF );
309
310 } else if( GTK_IS_ENTRY( widget )){
311 gtk_editable_set_editable( GTK_EDITABLE( widget ), editable );
312 /* removing the frame leads to a disturbing modification of the
313 * height of the control */
314 /*g_object_set( G_OBJECT( widget ), "has-frame", editable, NULL );*/
315 /* this prevents the caret to be displayed when we click in the entry */
316 g_object_set( G_OBJECT( widget ), "can-focus", editable, NULL );
317
318 } else if( GTK_IS_TEXT_VIEW( widget )){
319 g_object_set( G_OBJECT( widget ), "can-focus", editable, NULL );
320 gtk_text_view_set_editable( GTK_TEXT_VIEW( widget ), editable );
321
322 } else if( GTK_IS_TOGGLE_BUTTON( widget )){
323 /* transforms to a quasi standard GtkButton */
324 /*g_object_set( G_OBJECT( widget ), "draw-indicator", editable, NULL );*/
325 /* this at least prevent the keyboard focus to go to the button
326 * (which is better than nothing) */
327 g_object_set( G_OBJECT( widget ), "can-focus", editable, NULL );
328
329 } else if( GTK_IS_TREE_VIEW_COLUMN( widget )){
330 renderers = gtk_cell_layout_get_cells( GTK_CELL_LAYOUT( GTK_TREE_VIEW_COLUMN( widget )));
331 for( irender = renderers ; irender ; irender = irender->next ){
332 if( GTK_IS_CELL_RENDERER_TEXT( irender->data )){
333 g_object_set( G_OBJECT( irender->data ), "editable", editable, "editable-set", TRUE, NULL );
334 }
335 }
336 g_list_free( renderers );
337
338 } else if( GTK_IS_BUTTON( widget )){
339 gtk_widget_set_sensitive( GTK_WIDGET( widget ), editable );
340 }
341}

383 gtk_widget_set_sensitive( GTK_WIDGET( other ), sensitive );
384 }

423 gtk_toggle_button_set_active( initial_button, TRUE );

426 if( handler ){
