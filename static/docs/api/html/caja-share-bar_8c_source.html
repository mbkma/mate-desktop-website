43static void
44caja_share_bar_set_property (GObject *object,
45 guint prop_id,
46 const GValue *value,
47 GParamSpec *pspec)
48{
49 CajaShareBar *self;

51 self = CAJA_SHARE_BAR (object);
52
53 switch (prop_id) {
54 case PROP_LABEL: {
55 gtk_label_set_text (GTK_LABEL (self->priv->label),
56 g_value_get_string (value));

58 }

62 }

106 gtk_widget_show (label);
