311static void
312clock_set_timeout (ClockData *cd,
313 time_t now)
314{
315 int timeouttime;
316
317 if (cd->format == CLOCK_FORMAT_INTERNET) {
318 int itime_ms;
319
320 itime_ms = ((unsigned int) (get_itime (now) * 1000));
321
322 if (!cd->showseconds)
323 timeouttime = (999 - itime_ms % 1000) * 86.4 + 1;
324 else {
325 struct timeval tv;
326 gettimeofday (&tv, NULL);
327 itime_ms += (tv.tv_usec * 86.4) / 1000;
328 timeouttime = ((999 - itime_ms % 1000) * 86.4) / 100 + 1;
329 }
330 } else {
331 struct timeval tv;
332
333 gettimeofday (&tv, NULL);
334 timeouttime = (G_USEC_PER_SEC - tv.tv_usec)/1000+20;
335
336 /* timeout of one minute if we don't care about the seconds */
337 if (cd->format != CLOCK_FORMAT_UNIX &&
338 !cd->showseconds &&
339 (!cd->set_time_window || !gtk_widget_get_visible (cd->set_time_window)))
340 timeouttime += 1000 * (59 - now % 60);
341 }
342
343 cd->timeout = g_timeout_add (timeouttime,
344 clock_timeout_callback,
345 cd);
346}

348static int
349clock_timeout_callback (gpointer data)
350{
351 ClockData *cd = data;
352 time_t new_time;
353
354 time (&new_time);
355
356 if (!cd->showseconds &&
357 (!cd->set_time_window || !gtk_widget_get_visible (cd->set_time_window)) &&
358 cd->format != CLOCK_FORMAT_UNIX &&
359 cd->format != CLOCK_FORMAT_CUSTOM) {
360 if (cd->format == CLOCK_FORMAT_INTERNET &&
361 (unsigned int)get_itime (new_time) !=
362 (unsigned int)get_itime (cd->current_time)) {
363 update_clock (cd);
364 } else if ((cd->format == CLOCK_FORMAT_12 ||
365 cd->format == CLOCK_FORMAT_24) &&
366 new_time / 60 != cd->current_time / 60) {
367 update_clock (cd);
368 }
369 } else {
370 update_clock (cd);
371 }
372
373 clock_set_timeout (cd, new_time);
374
375 return FALSE;
376}

378static float
379get_itime (time_t current_time)
380{
381 struct tm *tm;
382 float itime;
383 time_t bmt;
384
385 /* BMT (Biel Mean Time) is GMT+1 */
386 bmt = current_time + 3600;
387 tm = gmtime (&bmt);
388 itime = (tm->tm_hour*3600.0 + tm->tm_min*60.0 + tm->tm_sec)/86.4;
389
390 return itime;
391}

444static char *
445get_updated_timeformat (ClockData *cd)
446{
447 /* Show date in another line if panel is vertical, or
448 * horizontal but large enough to hold two lines of text
449 */
450 char *result;
451 const char *time_format;
452 char *clock_format;
453 const gchar *env_language;
454 const gchar *env_lc_time;
455 gboolean use_lctime;
456
457 /* Override LANGUAGE with the LC_TIME environment variable
458 * This is needed for gettext to fetch our clock format
459 * according to LC_TIME, and not according to the DE LANGUAGE.
460 */
461 env_language = g_getenv("LANGUAGE");
462 env_lc_time = g_getenv("LC_TIME");
463 use_lctime = (env_language != NULL) && (env_lc_time != NULL) && (g_strcmp0 (env_language, env_lc_time) != 0);
464
465 if (use_lctime) {
466 g_setenv("LANGUAGE", env_lc_time, TRUE);
467 }
468
469 if (cd->format == CLOCK_FORMAT_12)
470 /* Translators: This is a strftime format string.
471 * It is used to display the time in 12-hours format (eg, like
472 * in the US: 8:10 am). The %p expands to am/pm. */
473 time_format = cd->showseconds ? _("%_I:%M:%S %p") : _("%_I:%M %p");
474 else
475 /* Translators: This is a strftime format string.
476 * It is used to display the time in 24-hours format (eg, like
477 * in France: 20:10). */
478 time_format = cd->showseconds ? _("%H:%M:%S") : _("%H:%M");
479
480 if (!cd->showdate)
481 clock_format = g_strdup (time_format);
482
483 else {
484 /* Translators: This is a strftime format string.
485 * It is used to display the date. Replace %e with %d if, when
486 * the day of the month as a decimal number is a single digit,
487 * it should begin with a 0 in your locale (e.g. "May 01"
488 * instead of "May 1"). */
489 const char *date_format = _("%a %b %e");
490
491 if (use_two_line_format (cd))
492 /* translators: reverse the order of these arguments
493 * if the time should come before the
494 * date on a clock in your locale.
495 */
496 clock_format = g_strdup_printf (_("%1$s\n%2$s"),
497 date_format,
498 time_format);
499 else
500 /* translators: reverse the order of these arguments
501 * if the time should come before the
502 * date on a clock in your locale.
503 */
504 clock_format = g_strdup_printf (_("%1$s, %2$s"),
505 date_format,
506 time_format);
507 }
508
509 /* Set back LANGUAGE the way it was before */
510 if (use_lctime) {
511 g_setenv("LANGUAGE", env_language, TRUE);
512 }
513
514 result = g_locale_from_utf8 (clock_format, -1, NULL, NULL, NULL);
515 g_free (clock_format);
516
517 /* let's be paranoid */
518 if (!result)
519 result = g_strdup ("???");
520
521 return result;
522}

563static char *
564format_time (ClockData *cd)
565{
566 struct tm *tm;
567 char hour[256];
568 char *utf8;
569
570 utf8 = NULL;
571
572 tm = localtime (&cd->current_time);
573
574 if (cd->format == CLOCK_FORMAT_UNIX) {
575 if (use_two_line_format (cd)) {
576 utf8 = g_strdup_printf ("%lu\n%05lu",
577 (unsigned long)(cd->current_time / 100000L),
578 (unsigned long)(cd->current_time % 100000L));
579 } else {
580 utf8 = g_strdup_printf ("%lu",
581 (unsigned long)cd->current_time);
582 }
583 } else if (cd->format == CLOCK_FORMAT_INTERNET) {
584 float itime = get_itime (cd->current_time);
585 if (cd->showseconds)
586 utf8 = g_strdup_printf ("@%3.2f", itime);
587 else
588 utf8 = g_strdup_printf ("@%3d", (unsigned int) itime);
589 } else if (cd->format == CLOCK_FORMAT_CUSTOM) {
590 char *timeformat = g_locale_from_utf8 (cd->custom_format, -1,
591 NULL, NULL, NULL);
592 if (!timeformat)
593 strcpy (hour, "???");
594 else if (strftime (hour, sizeof (hour), timeformat, tm)== 0)
595 strcpy (hour, "???");
596 g_free (timeformat);
597
598 utf8 = g_locale_to_utf8 (hour, -1, NULL, NULL, NULL);
599 } else {
600 if (strftime (hour, sizeof (hour), cd->timeformat, tm) == 0)
601 strcpy (hour, "???");
602
603 utf8 = g_locale_to_utf8 (hour, -1, NULL, NULL, NULL);
604 }
605
606 if (!utf8)
607 utf8 = g_strdup (hour);
608
609 return utf8;
610}

625{
626 gboolean use_markup;

645 g_free (utf8);

648 update_orient (cd);

653

666update_tooltip (ClockData * cd)
667{
668 char *tip;
669 char *old_tip;
670 if (!cd->showdate) {
671 struct tm *tm;
672 char date[256];

681

685 * it will be used to insert the timezone name later. */
686 loc = g_locale_from_utf8 (_("%A %B %d (%%s)"), -1, NULL, NULL, NULL);
687 if (!loc)
688 strcpy (date, "???");
689 else if (strftime (date, sizeof (date), loc, tm) == 0)
690 strcpy (date, "???");
691 g_free (loc);
692
693 utf8 = g_locale_to_utf8 (date, -1, NULL, NULL, NULL);
694
695 /* Add the timezone name */
696
697 tzset ();
698 time (&now_t);
699 localtime_r (&now_t, &now);
700
701 if (now.tm_isdst > 0) {
702 zone = tzname[1];
703 } else {
704 zone = tzname[0];
705 }
706
707 tip = g_strdup_printf (utf8, zone);

1217static gint
1218sort_locations_by_time (gconstpointer a, gconstpointer b)
1219{
1220 ClockLocation *loc_a = (ClockLocation *) a;
1221 ClockLocation *loc_b = (ClockLocation *) b;
1222
1223 struct tm tm_a;
1224 struct tm tm_b;
1225 gint ret;
1226
1227 clock_location_localtime (loc_a, &tm_a);
1228 clock_location_localtime (loc_b, &tm_b);
1229
1230 ret = tm_a.tm_year - tm_b.tm_year;
1231 if (! ret)
1232 ret = tm_a.tm_mon - tm_b.tm_mon;
1233 if (! ret)
1234 ret = tm_a.tm_mday - tm_b.tm_mday;
1235 if (! ret)
1236 ret = tm_a.tm_hour - tm_b.tm_hour;
1237 if (! ret)
1238 ret = tm_a.tm_min - tm_b.tm_min;
1239 if (! ret)
1240 ret = tm_a.tm_sec - tm_b.tm_sec;
1241
1242 return ret;
1243}

1946static void
1947wrap_cb (GtkSpinButton *spin, ClockData *cd)
1948{
1949 gdouble value;
1950 gdouble min, max;
1951 GtkSpinType direction;
1952
1953 value = gtk_spin_button_get_value (spin);
1954 gtk_spin_button_get_range (spin, &min, &max);
1955
1956 if (value == min)
1957 direction = GTK_SPIN_STEP_FORWARD;
1958 else
1959 direction = GTK_SPIN_STEP_BACKWARD;
1960
1961 if (spin == (GtkSpinButton *) cd->seconds_spin)
1962 gtk_spin_button_spin (GTK_SPIN_BUTTON (cd->minutes_spin), direction, 1.0);
1963 else if (spin == (GtkSpinButton *) cd->minutes_spin)
1964 gtk_spin_button_spin (GTK_SPIN_BUTTON (cd->hours_spin), direction, 1.0);
1965 else {
1966 guint year, month, day;
1967 GDate *date;
1968
1969 gtk_calendar_get_date (GTK_CALENDAR (cd->calendar), &year, &month, &day);
1970
1971 date = g_date_new_dmy (day, month + 1, year);
1972
1973 if (direction == GTK_SPIN_STEP_FORWARD)
1974 g_date_add_days (date, 1);
1975 else
1976 g_date_subtract_days (date, 1);
1977
1978 year = g_date_get_year (date);
1979 month = g_date_get_month (date) - 1;
1980 day = g_date_get_day (date);
1981
1982 gtk_calendar_select_month (GTK_CALENDAR (cd->calendar), month, year);
1983 gtk_calendar_select_day (GTK_CALENDAR (cd->calendar), day);
1984
1985 g_date_free (date);
1986 }

2640/* When the system manager (ConsoleKit or systemd-logind) reports that
2641 * the system has just resumed from sleep mode (e.g. suspend/hibernate),
2642 * update the clock and the weather/temperature readings. That way, if
2643 * the user suspended the system for an hour and then wakes the system up,
2644 * the user will immediately see the current time and (if possible)
2645 * updated weather data. Without this extra code, the user would most
2646 * likely wake the system up and see the weather from an hour ago, and if
2647 * the clock is set to not display seconds (only minutes), the clock may
2648 * show an inaccurate time for up to a minute after resume.
2649 */
2650static void
2651system_manager_signal_cb (GDBusProxy *proxy,
2652 gchar *sender_name,
2653 gchar *signal_name,
2654 GVariant *parameters,
2655 ClockData *cd)
2656{
2657 if (g_strcmp0 (signal_name, "PrepareForSleep") == 0)
2658 {
2659 GVariant *variant;
2660 gboolean active;
2661
2662 variant = g_variant_get_child_value (parameters, 0);
2663 active = g_variant_get_boolean (variant);
2664 g_variant_unref (variant);
2665
2666 /* The PrepareForSleep signal is emitted by ConsoleKit2 and
2667 * logind, both before sleep mode is entered, and
2668 * immediately after the system has resumed from sleep mode.
2669 * Listeners of this signal distinguish between the two
2670 * states by checking the parameter to this signal. If the
2671 * parameter is TRUE, the system is about to enter sleep
2672 * mode; if FALSE, the system has resumed from sleep mode.
2673 * We only care about updating the clock after resumption,
2674 * so test if the parameter is FALSE.
2675 */
2676 if (active == FALSE)
2677 {
2678 update_clock (cd);
2679 update_weather_locations (cd);
2680 }
2681 }

3557on_notebook_scroll_event (GtkWidget *widget,
3558 GdkEventScroll *event)
3559{
3560 GtkNotebook *notebook = GTK_NOTEBOOK (widget);
3561 GtkWidget *child, *event_widget, *action_widget;
3562
3563 child = gtk_notebook_get_nth_page (notebook, gtk_notebook_get_current_page (notebook));
3564 if (child == NULL)
3565 return FALSE;
3566
3567 event_widget = gtk_get_event_widget ((GdkEvent*) event);
3568
3569 /* Ignore scroll events from the content of the page */
3570 if (event_widget == NULL || event_widget == child || gtk_widget_is_ancestor (event_widget, child))
3571 return FALSE;
3572
3573 /* And also from the action widgets */
3574 action_widget = gtk_notebook_get_action_widget (notebook, GTK_PACK_START);
3575 if (event_widget == action_widget || (action_widget != NULL && gtk_widget_is_ancestor (event_widget, action_widget)))
3576 return FALSE;
3577
3578 action_widget = gtk_notebook_get_action_widget (notebook, GTK_PACK_END);
3579 if (event_widget == action_widget || (action_widget != NULL && gtk_widget_is_ancestor (event_widget, action_widget)))
3580 return FALSE;
3581
3582 switch (event->direction) {
3583 case GDK_SCROLL_RIGHT:
3584 case GDK_SCROLL_DOWN:
3585 gtk_notebook_next_page (notebook);
3586 break;
3587 case GDK_SCROLL_LEFT:
3588 case GDK_SCROLL_UP:
3589 gtk_notebook_prev_page (notebook);
3590 break;
3591 case GDK_SCROLL_SMOOTH:
3592 switch (gtk_notebook_get_tab_pos (notebook)) {
3593 case GTK_POS_LEFT:
3594 case GTK_POS_RIGHT:
3595 if (event->delta_y > 0)
3596 gtk_notebook_next_page (notebook);
3597 else if (event->delta_y < 0)
3598 gtk_notebook_prev_page (notebook);
3599 break;
3600 case GTK_POS_TOP:
3601 case GTK_POS_BOTTOM:
