89static void
90panel_global_config_set_entry (GSettings *settings, gchar *key)
91{
92 g_return_if_fail (settings != NULL);
93 g_return_if_fail (key != NULL);
94
95 if (strcmp (key, "tooltips-enabled") == 0)
96 global_config.tooltips_enabled =
97 (g_settings_get_boolean (settings, key) != FALSE);
98
99 else if (strcmp (key, "enable-animations") == 0)
100 global_config.enable_animations =
101 (g_settings_get_boolean (settings, key) != FALSE);
102
103 else if (strcmp (key, "drawer-autoclose") == 0)
104 global_config.drawer_auto_close =
105 (g_settings_get_boolean (settings, key) != FALSE);
106
107 else if (strcmp (key, "confirm-panel-remove") == 0)
108 global_config.confirm_panel_remove =
109 (g_settings_get_boolean (settings, key) != FALSE);
110
111 else if (strcmp (key, "highlight-launchers-on-mouseover") == 0)
112 global_config.highlight_when_over =
113 (g_settings_get_boolean (settings, key) != FALSE);
114}

124void
125panel_global_config_load (void)
126{
127 GSettingsSchema *schema;
128 gchar **keys;
129 gint i;
130
131 panel_global_settings = g_settings_new ("org.mate.panel");
132
133 g_object_get (panel_global_settings, "settings-schema", &schema, NULL);
134 keys = g_settings_schema_list_keys (schema);
135 g_settings_schema_unref (schema);
136
137 for (i = 0; keys[i]; i++) {
138 panel_global_config_set_entry (panel_global_settings, keys[i]);
139 }
140 g_strfreev (keys);
141
142 g_signal_connect (panel_global_settings, "changed", G_CALLBACK (panel_global_config_notify), NULL);
143
144 global_config_initialised = TRUE;
145}
