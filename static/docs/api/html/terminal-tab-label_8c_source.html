100static GObject *
101terminal_tab_label_constructor (GType type,
102 guint n_construct_properties,
103 GObjectConstructParam *construct_params)
104{
105 GObject *object;
106 TerminalTabLabel *tab_label;
107 TerminalTabLabelPrivate *priv;
108 GtkWidget *hbox, *label, *close_button;
109
110 object = G_OBJECT_CLASS (terminal_tab_label_parent_class)->constructor
111 (type, n_construct_properties, construct_params);
112
113 tab_label = TERMINAL_TAB_LABEL (object);
114 hbox = GTK_WIDGET (tab_label);
115 priv = tab_label->priv;
116
117 g_assert (priv->screen != NULL);
118
119 gtk_box_set_spacing (GTK_BOX (hbox), SPACING);
120
121 priv->label = label = gtk_label_new (NULL);
122
123 gtk_label_set_xalign (GTK_LABEL (label), 0.0);
124 gtk_label_set_yalign (GTK_LABEL (label), 0.5);
125 gtk_label_set_ellipsize (GTK_LABEL (label), PANGO_ELLIPSIZE_END);
126 gtk_label_set_single_line_mode (GTK_LABEL (label), TRUE);
127
128 gtk_box_pack_start (GTK_BOX (hbox), label, TRUE, TRUE, 0);
129
130 priv->close_button = close_button = terminal_close_button_new ();
131 gtk_widget_set_tooltip_text (close_button, _("Close tab"));
132
133 gtk_box_pack_end (GTK_BOX (hbox), close_button, FALSE, FALSE, 0);
134
135 sync_tab_label (priv->screen, NULL, label);
136 g_signal_connect (priv->screen, "notify::title",
137 G_CALLBACK (sync_tab_label), label);
138
139 g_signal_connect (close_button, "clicked",
140 G_CALLBACK (close_button_clicked_cb), tab_label);
141
142 gtk_widget_show_all (hbox);
143
144 return object;
145}

247void
248terminal_tab_label_set_bold (TerminalTabLabel *tab_label,
249 gboolean bold)
250{
251 TerminalTabLabelPrivate *priv = tab_label->priv;
252 PangoAttrList *attr_list;
253 PangoAttribute *weight_attr;
254 gboolean free_list = FALSE;
255
256 bold = bold != FALSE;
257 if (priv->bold == bold)
258 return;
259
260 priv->bold = bold;
261
262 attr_list = gtk_label_get_attributes (GTK_LABEL (priv->label));
263 if (!attr_list)
264 {
265 attr_list = pango_attr_list_new ();
266 free_list = TRUE;
267 }
268
269 if (bold)
270 weight_attr = pango_attr_weight_new (PANGO_WEIGHT_BOLD);
271 else
272 weight_attr = pango_attr_weight_new (PANGO_WEIGHT_NORMAL);
273
274 /* gtk_label_get_attributes() returns the label's internal list,
275 * which we're probably not supposed to modify directly.
276 * It seems to work ok however.
277 */
278 pango_attr_list_change (attr_list, weight_attr);
279
280 gtk_label_set_attributes (GTK_LABEL (priv->label), attr_list);
281
282 if (free_list)
283 pango_attr_list_unref (attr_list);
284}
