88static gboolean
89find_by_app (GtkTreeModel *model,
90 GtkTreeIter *iter,
91 GspApp *app)
92{
93 GspApp *iter_app = NULL;
94
95 if (!gtk_tree_model_get_iter_first (model, iter)) {
96 return FALSE;
97 }
98
99 do {
100 gtk_tree_model_get (model, iter,
101 STORE_COL_APP, &iter_app,
102 -1);
103
104 if (iter_app == app) {
105 g_object_unref (iter_app);
106 return TRUE;
107 }
108
109 g_object_unref (iter_app);
110 } while (gtk_tree_model_iter_next (model, iter));
111
112 return FALSE;
113}

115static void
116_fill_iter_from_app (GtkListStore *list_store,
117 GtkTreeIter *iter,
118 GspApp *app)
119{
120 gboolean enabled;
121 GIcon *icon;
122 const char *description;
123 const char *app_name;
124
125 enabled = !gsp_app_get_hidden (app);
126 icon = gsp_app_get_icon (app);
127 description = gsp_app_get_description (app);
128 app_name = gsp_app_get_name (app);
129
130 if (G_IS_THEMED_ICON (icon)) {
131 GtkIconTheme *theme;
132 const char * const *icon_names;
133
134 theme = gtk_icon_theme_get_default ();
135 icon_names = g_themed_icon_get_names (G_THEMED_ICON (icon));
136 if (icon_names[0] == NULL ||
137 !gtk_icon_theme_has_icon (theme, icon_names[0])) {
138 g_object_unref (icon);
139 icon = NULL;
140 }
141 } else if (G_IS_FILE_ICON (icon)) {
142 GFile *iconfile;
143
144 iconfile = g_file_icon_get_file (G_FILE_ICON (icon));
145 if (!g_file_query_exists (iconfile, NULL)) {
146 g_object_unref (icon);
147 icon = NULL;
148 }
149 }
150
151 if (icon == NULL) {
152 icon = g_themed_icon_new (STARTUP_APP_ICON);
153 }
154
155 gtk_list_store_set (list_store, iter,
156 STORE_COL_ENABLED, enabled,
157 STORE_COL_GICON, icon,
158 STORE_COL_DESCRIPTION, description,
159 STORE_COL_APP, app,
160 STORE_COL_SEARCH, app_name,
161 -1);
162 g_object_unref (icon);
163}

165static void
166_app_changed (GsmPropertiesDialog *dialog,
167 GspApp *app)
168{
169 GtkTreeIter iter;
170
171 if (!find_by_app (GTK_TREE_MODEL (dialog->list_store),
172 &iter, app)) {
173 return;
174 }
175
176 _fill_iter_from_app (dialog->list_store, &iter, app);
177}

179static void
180append_app (GsmPropertiesDialog *dialog,
181 GspApp *app)
182{
183 GtkTreeIter iter;
184 if (find_by_app (GTK_TREE_MODEL (dialog->list_store),
185 &iter, app)) {
186 return;
187 }
188
189 gtk_list_store_append (dialog->list_store, &iter);
190 _fill_iter_from_app (dialog->list_store, &iter, app);
191
192 g_signal_connect_swapped (app, "changed",
193 G_CALLBACK (_app_changed), dialog);
194}

284static void
285on_drag_data_received (GtkWidget *widget,
286 GdkDragContext *drag_context,
287 gint x,
288 gint y,
289 GtkSelectionData *data,
290 guint info,
291 guint time,
292 GsmPropertiesDialog *dialog)
293{
294 gboolean dnd_success;
295
296 dnd_success = FALSE;
297
298 if (data != NULL) {
299 char **filenames;
300 int i;
301
302 filenames = gtk_selection_data_get_uris (data);
303
304 for (i = 0; filenames[i] && filenames[i][0]; i++) {
305 /* Return success if at least one file succeeded */
306 gboolean file_success;
307 file_success = gsp_app_copy_desktop_file (filenames[i]);
308 dnd_success = dnd_success || file_success;
309 }
310
311 g_strfreev (filenames);
312 }
313
314 gtk_drag_finish (drag_context, dnd_success, FALSE, time);
315 g_signal_stop_emission_by_name (widget, "drag_data_received");
316}

318static void
319on_drag_begin (GtkWidget *widget,
320 GdkDragContext *context,
321 GsmPropertiesDialog *dialog)
322{
323 GtkTreePath *path;
324 GtkTreeIter iter;
325 GspApp *app;
326
327 gtk_tree_view_get_cursor (GTK_TREE_VIEW (widget), &path, NULL);
328 gtk_tree_model_get_iter (GTK_TREE_MODEL (dialog->tree_filter),
329 &iter, path);
330 gtk_tree_path_free (path);
331
332 gtk_tree_model_get (GTK_TREE_MODEL (dialog->tree_filter),
333 &iter,
334 STORE_COL_APP, &app,
335 -1);
336
337 if (app) {
338 g_object_set_data_full (G_OBJECT (context), "gsp-app",
339 g_object_ref (app), g_object_unref);
340 g_object_unref (app);
341 }

345static void
346on_drag_data_get (GtkWidget *widget,
347 GdkDragContext *context,
348 GtkSelectionData *selection_data,
349 guint info,
350 guint time,
351 GsmPropertiesDialog *dialog)
352{
353 GspApp *app;
354
355 app = g_object_get_data (G_OBJECT (context), "gsp-app");
356 if (app) {
357 const char *uris[2];
358 char *uri;
359
360 uri = g_filename_to_uri (gsp_app_get_path (app), NULL, NULL);
361
362 uris[0] = uri;
363 uris[1] = NULL;
364 gtk_selection_data_set_uris (selection_data, (char **) uris);
365
366 g_free (uri);
367 }

516static gboolean
517on_main_notebook_scroll_event (GtkWidget *widget,
518 GdkEventScroll *event)
519{
520 GtkNotebook *notebook = GTK_NOTEBOOK (widget);
521 GtkWidget *child, *event_widget, *action_widget;
522
523 child = gtk_notebook_get_nth_page (notebook, gtk_notebook_get_current_page (notebook));
524 if (child == NULL)
525 return FALSE;
526
527 event_widget = gtk_get_event_widget ((GdkEvent*) event);
528
529 /* Ignore scroll events from the content of the page */
530 if (event_widget == NULL || event_widget == child || gtk_widget_is_ancestor (event_widget, child))
531 return FALSE;
532
533 /* And also from the action widgets */
534 action_widget = gtk_notebook_get_action_widget (notebook, GTK_PACK_START);
535 if (event_widget == action_widget || (action_widget != NULL && gtk_widget_is_ancestor (event_widget, action_widget)))
536 return FALSE;
537
538 action_widget = gtk_notebook_get_action_widget (notebook, GTK_PACK_END);
539 if (event_widget == action_widget || (action_widget != NULL && gtk_widget_is_ancestor (event_widget, action_widget)))
540 return FALSE;
541
542 switch (event->direction) {
543 case GDK_SCROLL_RIGHT:
544 case GDK_SCROLL_DOWN:
545 gtk_notebook_next_page (notebook);
546 break;
547 case GDK_SCROLL_LEFT:
548 case GDK_SCROLL_UP:
549 gtk_notebook_prev_page (notebook);
550 break;
551 case GDK_SCROLL_SMOOTH:
552 switch (gtk_notebook_get_tab_pos (notebook)) {
553 case GTK_POS_LEFT:
554 case GTK_POS_RIGHT:
555 if (event->delta_y > 0)
556 gtk_notebook_next_page (notebook);
557 else if (event->delta_y < 0)
558 gtk_notebook_prev_page (notebook);
559 break;
560 case GTK_POS_TOP:
561 case GTK_POS_BOTTOM:
562 if (event->delta_x > 0)
563 gtk_notebook_next_page (notebook);
564 else if (event->delta_x < 0)
565 gtk_notebook_prev_page (notebook);
566 break;
567 }
568 break;
569 }
570
571 return TRUE;
572}

574static gboolean
575visible_func (GtkTreeModel *model,
576 GtkTreeIter *iter,
577 gpointer data)
578{
579 gboolean visible = FALSE;
580 GtkToggleButton *toggle_button = data;;
581 GspApp *app;
582 gboolean show_hidden;
583
584 show_hidden = gtk_toggle_button_get_active (toggle_button);
585
586 gtk_tree_model_get (model, iter,
587 STORE_COL_APP, &app,
588 -1);
589
590 if (app) {
591 gboolean nodisplay;
592
593 nodisplay = gsp_app_get_nodisplay (app);
594 visible = show_hidden || !nodisplay;
595 g_object_unref (app);
596 } else {
597 visible = show_hidden;
598 }
599
600 return visible;
601}

666 dialog);

670 column = gtk_tree_view_column_new_with_attributes (_("Enabled"),
671 renderer,
672 "active", STORE_COL_ENABLED,
673 NULL);
674 gtk_tree_view_append_column (treeview, column);
675 g_signal_connect (renderer,
676 "toggled",
677 G_CALLBACK (on_startup_enabled_toggled),
678 dialog);
679
680 /* ICON COLUMN */
681 renderer = gtk_cell_renderer_pixbuf_new ();
682 column = gtk_tree_view_column_new_with_attributes (_("Icon"),
683 renderer,
684 "gicon", STORE_COL_GICON,
685 "sensitive", STORE_COL_ENABLED,
686 NULL);
687 g_object_set (renderer,
688 "stock-size", GSM_PROPERTIES_ICON_SIZE,
689 NULL);
690 gtk_tree_view_append_column (treeview, column);
691
692 /* NAME COLUMN */
693 renderer = gtk_cell_renderer_text_new ();
694 column = gtk_tree_view_column_new_with_attributes (_("Program"),
695 renderer,
696 "markup", STORE_COL_DESCRIPTION,
697 "sensitive", STORE_COL_ENABLED,
698 NULL);
699 g_object_set (renderer,
700 "ellipsize", PANGO_ELLIPSIZE_END,
701 NULL);
702 gtk_tree_view_append_column (treeview, column);
703
704 gtk_tree_view_column_set_sort_column_id (column, STORE_COL_DESCRIPTION);
705 gtk_tree_view_set_search_column (treeview, STORE_COL_SEARCH);
706
707 gtk_tree_view_enable_model_drag_source (treeview,
708 GDK_BUTTON1_MASK|GDK_BUTTON2_MASK,
709 NULL, 0,
710 GDK_ACTION_COPY);
711 gtk_drag_source_add_uri_targets (GTK_WIDGET (treeview));
712
713 gtk_drag_dest_set (GTK_WIDGET (treeview),
714 GTK_DEST_DEFAULT_ALL,
715 NULL, 0,
716 GDK_ACTION_COPY);
717 gtk_drag_dest_add_uri_targets (GTK_WIDGET (treeview));
718
719 /* we don't want to accept drags coming from this widget */
720 targetlist = gtk_drag_dest_get_target_list (GTK_WIDGET (treeview));
721 if (targetlist != NULL) {
722 GtkTargetEntry *targets;
723 gint n_targets;
724 gint i;
725 targets = gtk_target_table_new_from_list (targetlist, &n_targets);
726 for (i = 0; i < n_targets; i++)
727 targets[i].flags = GTK_TARGET_OTHER_WIDGET;
728 targetlist = gtk_target_list_new (targets, n_targets);
729 gtk_drag_dest_set_target_list (GTK_WIDGET (treeview), targetlist);
730 gtk_target_list_unref (targetlist);
731 gtk_target_table_free (targets, n_targets);
732 }

735 G_CALLBACK (on_drag_begin),
736 dialog);
