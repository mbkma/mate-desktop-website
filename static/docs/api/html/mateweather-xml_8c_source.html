40static gboolean
41mateweather_xml_parse_node (MateWeatherLocation *gloc,
42 GtkTreeStore *store, GtkTreeIter *parent)
43{
44 GtkTreeIter iter, *self = &iter;
45 MateWeatherLocation **children, *parent_loc;
46 MateWeatherLocationLevel level;
47 WeatherLocation *wloc;
48 const char *name;
49 int i;
50
51 name = mateweather_location_get_name (gloc);
52 children = mateweather_location_get_children (gloc);
53 level = mateweather_location_get_level (gloc);
54
55 if (!children[0] && level < MATEWEATHER_LOCATION_WEATHER_STATION) {
56 mateweather_location_free_children (gloc, children);
57 return TRUE;
58 }
59
60 switch (mateweather_location_get_level (gloc)) {
61 case MATEWEATHER_LOCATION_WORLD:
62 case MATEWEATHER_LOCATION_ADM2:
63 self = parent;
64 break;
65
66 case MATEWEATHER_LOCATION_REGION:
67 case MATEWEATHER_LOCATION_COUNTRY:
68 case MATEWEATHER_LOCATION_ADM1:
69 /* Create a row with a name but no WeatherLocation */
70 gtk_tree_store_append (store, &iter, parent);
71 gtk_tree_store_set (store, &iter,
72 MATEWEATHER_XML_COL_LOC, name,
73 -1);
74 break;
75
76 case MATEWEATHER_LOCATION_CITY:
77 /* If multiple children, treat this like a
78 * region/country/adm1. If a single child, merge with that
79 * location.
80 */
81 gtk_tree_store_append (store, &iter, parent);
82 gtk_tree_store_set (store, &iter,
83 MATEWEATHER_XML_COL_LOC, name,
84 -1);
85 if (children[0] && !children[1]) {
86 wloc = mateweather_location_to_weather_location (children[0], name);
87 gtk_tree_store_set (store, &iter,
88 MATEWEATHER_XML_COL_POINTER, wloc,
89 -1);
90 }
91 break;
92
93 case MATEWEATHER_LOCATION_WEATHER_STATION:
94 gtk_tree_store_append (store, &iter, parent);
95 gtk_tree_store_set (store, &iter,
96 MATEWEATHER_XML_COL_LOC, name,
97 -1);
98
99 parent_loc = mateweather_location_get_parent (gloc);
100 if (parent_loc && mateweather_location_get_level (parent_loc) == MATEWEATHER_LOCATION_CITY)
101 name = mateweather_location_get_name (parent_loc);
102 wloc = mateweather_location_to_weather_location (gloc, name);
103 gtk_tree_store_set (store, &iter,
104 MATEWEATHER_XML_COL_POINTER, wloc,
105 -1);
106 break;
107 }
108
109 for (i = 0; children[i]; i++) {
110 if (!mateweather_xml_parse_node (children[i], store, self)) {
111 mateweather_location_free_children (gloc, children);
112 return FALSE;
113 }
114 }
115
116 mateweather_location_free_children (gloc, children);
117 return TRUE;
118}

120GtkTreeModel *
121mateweather_xml_load_locations (void)
122{
123 MateWeatherLocation *world;
124 GtkTreeStore *store;
125
126 world = mateweather_location_new_world (TRUE);
127 if (!world)
128 return NULL;
129
130 store = gtk_tree_store_new (2, G_TYPE_STRING, G_TYPE_POINTER);
131
132 if (!mateweather_xml_parse_node (world, store, NULL)) {
133 mateweather_xml_free_locations ((GtkTreeModel *)store);
134 store = NULL;
135 }
136
137 mateweather_location_unref (world);
138
139 return (GtkTreeModel *)store;
140}

142static gboolean
143free_locations (GtkTreeModel *model, GtkTreePath *path, GtkTreeIter *iter, gpointer data)
144{
145 WeatherLocation *loc = NULL;
146
147 gtk_tree_model_get (model, iter,
148 MATEWEATHER_XML_COL_POINTER, &loc,
149 -1);
150
151 if (loc) {
152 weather_location_free (loc);
153 gtk_tree_store_set ((GtkTreeStore *)model, iter,
154 MATEWEATHER_XML_COL_POINTER, NULL,
155 -1);
156 }
157
158 return FALSE;
159}
