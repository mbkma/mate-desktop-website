59static CajaBookmark *
60new_bookmark_from_uri (const char *uri, const char *label)
61{
62 CajaBookmark *new_bookmark;
63 char *name;
64 gboolean has_label;
65 GFile *location;
66 gboolean native;
67
68 location = NULL;
69 if (uri)
70 {
71 location = g_file_new_for_uri (uri);
72 }
73
74 has_label = FALSE;
75 if (!label)
76 {
77 name = caja_compute_title_for_location (location);
78 }
79 else
80 {
81 name = g_strdup (label);
82 has_label = TRUE;
83 }
84
85 new_bookmark = NULL;
86
87 if (uri)
88 {
89 CajaFile *file;
90 GIcon *icon;
91
92 native = g_file_is_native (location);
93 file = caja_file_get (location);
94
95 icon = NULL;
96
97 if (caja_file_check_if_ready (file,
98 CAJA_FILE_ATTRIBUTES_FOR_ICON))
99 {
100 icon = caja_file_get_gicon (file, 0);
101 }
102 caja_file_unref (file);
103
104 if (icon == NULL)
105 {
106 icon = native ? g_themed_icon_new (CAJA_ICON_FOLDER) :
107 g_themed_icon_new (CAJA_ICON_FOLDER_REMOTE);
108 }
109
110 new_bookmark = caja_bookmark_new (location, name, has_label, icon);
111
112 g_object_unref (icon);
113
114 }
115 g_free (name);
116 g_object_unref (location);
117 return new_bookmark;
118}

344void
345caja_bookmark_list_delete_item_at (CajaBookmarkList *bookmarks,
346 guint index)
347{
348 GList *doomed;
349
350 g_return_if_fail (CAJA_IS_BOOKMARK_LIST (bookmarks));
351 g_return_if_fail (index < g_list_length (bookmarks->list));
352
353 doomed = g_list_nth (bookmarks->list, index);
354 g_return_if_fail (doomed != NULL);
355
356 bookmarks->list = g_list_remove_link (bookmarks->list, doomed);
357
358 g_assert (CAJA_IS_BOOKMARK (doomed->data));
359 stop_monitoring_bookmark (bookmarks, CAJA_BOOKMARK (doomed->data));
360 g_object_unref (doomed->data);
361
362 g_list_free_1 (doomed);
363
364 caja_bookmark_list_save_file (bookmarks);
365}

374void
375caja_bookmark_list_move_item (CajaBookmarkList *bookmarks,
376 guint index,
377 guint destination)
378{
379 GList *bookmark_item;
380
381 if (index == destination)
382 {
383 return;
384 }
385
386 bookmark_item = g_list_nth (bookmarks->list, index);
387 g_return_if_fail (bookmark_item != NULL);
388
389 bookmarks->list = g_list_remove_link (bookmarks->list,
390 bookmark_item);
391
392 if (index < destination)
393 {
394 bookmarks->list = g_list_insert (bookmarks->list,
395 bookmark_item->data,
396 destination - 1);
397 }
398 else
399 {
400 bookmarks->list = g_list_insert (bookmarks->list,
401 bookmark_item->data,
402 destination);
403 }
404
405 caja_bookmark_list_save_file (bookmarks);
406}

415void
416caja_bookmark_list_delete_items_with_uri (CajaBookmarkList *bookmarks,
417 const char *uri)
418{
419 GList *node, *next;
420 gboolean list_changed;
421
422 g_return_if_fail (CAJA_IS_BOOKMARK_LIST (bookmarks));
423 g_return_if_fail (uri != NULL);
424
425 list_changed = FALSE;
426 for (node = bookmarks->list; node != NULL; node = next)
427 {
428 char *bookmark_uri;
429
430 next = node->next;
431
432 bookmark_uri = caja_bookmark_get_uri (CAJA_BOOKMARK (node->data));
433 if (g_strcmp0 (bookmark_uri, uri) == 0)
434 {
435 bookmarks->list = g_list_remove_link (bookmarks->list, node);
436 stop_monitoring_bookmark (bookmarks, CAJA_BOOKMARK (node->data));
437 g_object_unref (node->data);
438 g_list_free_1 (node);
439 list_changed = TRUE;
440 }
441 g_free (bookmark_uri);
442 }
443
444 if (list_changed)
445 {
446 caja_bookmark_list_save_file (bookmarks);
447 }

523static void
524load_file_finish (CajaBookmarkList *bookmarks,
525 GObject *source,
526 GAsyncResult *res)
527{
528 GError *error = NULL;
529 gchar *contents = NULL;
530
531 g_file_load_contents_finish (G_FILE (source),
532 res, &contents, NULL, NULL, &error);
533
534 if (error == NULL)
535 {
536 char **lines;
537 int i;
538
539 lines = g_strsplit (contents, "\n", -1);
540 for (i = 0; lines[i]; i++)
541 {
542 /* Ignore empty or invalid lines that cannot be parsed properly */
543 if (lines[i][0] != '\0' && lines[i][0] != ' ')
544 {
545 /* gtk 2.7/2.8 might have labels appended to bookmarks which are separated by a space */
546 /* we must seperate the bookmark uri and the potential label */
547 char *space, *label;
548
549 label = NULL;
550 space = strchr (lines[i], ' ');
551 if (space)
552 {
553 *space = '\0';
554 label = g_strdup (space + 1);
555 }
556 insert_bookmark_internal (bookmarks,
557 new_bookmark_from_uri (lines[i], label),
558 -1);
559
560 g_free (label);
561 }
562 }
563 g_free (contents);
564 g_strfreev (lines);
565
566 g_signal_emit (bookmarks, signals[CONTENTS_CHANGED], 0);
567 }
568 else if (!g_error_matches (error, G_IO_ERROR, G_IO_ERROR_NOT_FOUND))
569 {
570 g_warning ("Could not load bookmark file: %s\n", error->message);
571 g_error_free (error);
572 }
573}

575static void
576load_file_async (CajaBookmarkList *self,
577 GAsyncReadyCallback callback)
578{
579 GFile *file;
580
581 file = caja_bookmark_list_get_file ();
582 if (!g_file_query_exists (file, NULL)) {
583 file = caja_bookmark_list_get_legacy_file ();
584 }
585
586 /* Wipe out old list. */
587 clear (self);
588
589 /* keep the bookmark list alive */
590 g_object_ref (self);
591 g_file_load_contents_async (file, NULL, callback, self);
592
593 g_object_unref (file);
594}

596static void
597save_file_finish (CajaBookmarkList *bookmarks,
598 GObject *source,
599 GAsyncResult *res)
600{
601 GError *error = NULL;
602 GFile *file;
603
604 g_file_replace_contents_finish (G_FILE (source),
605 res, NULL, &error);
606
607 if (error != NULL)
608 {
609 g_warning ("Unable to replace contents of the bookmarks file: %s",
610 error->message);
611 g_error_free (error);
612 }
613
614 file = caja_bookmark_list_get_file ();
615
616 /* re-enable bookmark file monitoring */
617 bookmarks->monitor = g_file_monitor_file (file, 0, NULL, NULL);
618 g_file_monitor_set_rate_limit (bookmarks->monitor, 1000);
619 g_signal_connect (bookmarks->monitor, "changed",
620 G_CALLBACK (bookmark_monitor_changed_cb), bookmarks);
621
622 g_object_unref (file);
623}

625static void
626save_file_async (CajaBookmarkList *bookmarks,
627 GAsyncReadyCallback callback)
628{
629 GFile *file;
630 GList *l;
631 GString *bookmark_string;
632 GFile *parent;
633 char *path;
634
635 /* temporarily disable bookmark file monitoring when writing file */
636 if (bookmarks->monitor != NULL)
637 {
638 g_file_monitor_cancel (bookmarks->monitor);
639 bookmarks->monitor = NULL;
640 }
641
642 file = caja_bookmark_list_get_file ();
643 bookmark_string = g_string_new (NULL);
644
645 for (l = bookmarks->list; l; l = l->next)
646 {
647 CajaBookmark *bookmark;
648
649 bookmark = CAJA_BOOKMARK (l->data);
650
651 /* make sure we save label if it has one for compatibility with GTK 2.7 and 2.8 */
652 if (caja_bookmark_get_has_custom_name (bookmark))
653 {
654 char *label, *uri;
655 label = caja_bookmark_get_name (bookmark);
656 uri = caja_bookmark_get_uri (bookmark);
657 g_string_append_printf (bookmark_string,
658 "%s %s\n", uri, label);
659 g_free (uri);
660 g_free (label);
661 }
662 else
663 {
664 char *uri;
665 uri = caja_bookmark_get_uri (bookmark);
666 g_string_append_printf (bookmark_string, "%s\n", uri);
667 g_free (uri);
668 }
669 }
670
671 /* keep the bookmark list alive */
672 g_object_ref (bookmarks);
673
674 parent = g_file_get_parent (file);
675 path = g_file_get_path (parent);
676 g_mkdir_with_parents (path, 0700);
677 g_free (path);
678 g_object_unref (parent);
679
680 g_file_replace_contents_async (file, bookmark_string->str,
681 bookmark_string->len, NULL,
682 FALSE, 0, NULL, callback,
683 bookmarks);
684
685 g_object_unref (file);
686}

688static void
689process_next_op (CajaBookmarkList *bookmarks);
690
691static void
692op_processed_cb (GObject *source,
693 GAsyncResult *res,
694 gpointer user_data)
695{
696 CajaBookmarkList *self = user_data;
697 int op;
698
699 op = GPOINTER_TO_INT (g_queue_pop_tail (self->pending_ops));
700
701 if (op == LOAD_JOB)
702 {
703 load_file_finish (self, source, res);
704 }
705 else
706 {
707 save_file_finish (self, source, res);
708 }
709
710 if (!g_queue_is_empty (self->pending_ops))
711 {
712 process_next_op (self);
713 }
714
715 /* release the reference acquired during the _async method */
716 g_object_unref (self);
717}
718
719static void
720process_next_op (CajaBookmarkList *bookmarks)
721{
722 gint op;
723
724 op = GPOINTER_TO_INT (g_queue_peek_tail (bookmarks->pending_ops));
725
726 if (op == LOAD_JOB)
727 {
728 load_file_async (bookmarks, op_processed_cb);
729 }
730 else
731 {
732 save_file_async (bookmarks, op_processed_cb);
733 }
734}

742static void
743caja_bookmark_list_load_file (CajaBookmarkList *bookmarks)
744{
745 g_queue_push_head (bookmarks->pending_ops, GINT_TO_POINTER (LOAD_JOB));
746
747 if (g_queue_get_length (bookmarks->pending_ops) == 1)
748 {
749 process_next_op (bookmarks);
750 }
751}

769 }
