151static GtkTreeModel *
152_panel_run_get_recent_programs_list (PanelRunDialog *dialog)
153{
154 GtkListStore *list;
155 gchar **items;
156 guint history_max_size;
157 gboolean history_reverse = FALSE;
158 guint i = 0;
159
160 list = gtk_list_store_new (1, G_TYPE_STRING);
161
162 history_max_size = g_settings_get_uint (dialog->settings, PANEL_RUN_HISTORY_MAX_SIZE_KEY);
163 history_reverse = g_settings_get_boolean (dialog->settings, PANEL_RUN_HISTORY_REVERSE_KEY);
164 items = g_settings_get_strv (dialog->settings, PANEL_RUN_HISTORY_KEY);
165 for (i = 0; i < history_max_size && items[i]; i++) {
166 GtkTreeIter iter;
167 /* add history in reverse */
168 if (history_reverse)
169 gtk_list_store_prepend (list, &iter);
170 else
171 gtk_list_store_append (list, &iter);
172 gtk_list_store_set (list, &iter, 0, items[i], -1);
173 }
174
175 g_strfreev (items);
176
177 return GTK_TREE_MODEL (list);
178}

180static void
181_panel_run_save_recent_programs_list (PanelRunDialog *dialog,
182 GtkComboBox *entry,
183 char *lastcommand)
184{
185 gboolean history_reverse =
186 g_settings_get_boolean (dialog->settings, PANEL_RUN_HISTORY_REVERSE_KEY);
187 guint history_max_size =
188 g_settings_get_uint (dialog->settings, PANEL_RUN_HISTORY_MAX_SIZE_KEY);
189
190 if (history_max_size == 0)
191 g_settings_set_strv (dialog->settings, PANEL_RUN_HISTORY_KEY, NULL);
192 else {
193 GtkTreeIter iter;
194 GtkTreeModel *model = gtk_combo_box_get_model (GTK_COMBO_BOX (entry));
195 /* reasonable upper bound for zero-terminated array with new command */
196 gchar *items[gtk_tree_model_iter_n_children (model, NULL) + 2];
197 guint items_added = 0;
198
199 items[0] = lastcommand;
200
201 if (gtk_tree_model_get_iter_first (model, &iter)) {
202 char *command;
203 do {
204 gtk_tree_model_get (model, &iter, 0, &command, -1);
205 if (g_strcmp0 (command, lastcommand) == 0) {
206 g_free (command);
207 continue;
208 }
209 items[items_added + 1] = command;
210 items_added++;
211 } while (gtk_tree_model_iter_next (model, &iter));
212 }
213
214 if (history_reverse) {
215 guint pos;
216 for (pos = 0; pos < items_added / 2; pos++)
217 {
218 gchar *tmp = items[pos + 1];
219 items[pos + 1] = items[items_added - pos];
220 items[items_added - pos] = tmp;
221 }
222 }
223
224 if (history_max_size < items_added + 1) {
225 g_clear_pointer (&items[history_max_size], g_free);
226 } else {
227 items[items_added + 1] = NULL;
228 }
229 g_settings_set_strv (dialog->settings, PANEL_RUN_HISTORY_KEY, (const gchar **)items);
230
231 /* free the items */
232 for (; items_added; items_added--)
233 g_free (items[items_added]);
234 }
235}

308static void
309panel_run_dialog_set_icon (PanelRunDialog *dialog,
310 GIcon *icon,
311 gboolean force)
312{
313 GdkPixbuf *pixbuf = NULL;
314
315 if (!force && g_icon_equal(icon, dialog->icon))
316 return;
317
318 g_clear_object(&(dialog->icon));
319
320 if (icon) {
321 int size;
322

345 g_object_unref (pixbuf);
346 } else {
347 panel_run_dialog_set_default_icon (dialog, TRUE);
348 }

351static gboolean
352command_is_executable (const char *command,
353 int *argcp,
354 char ***argvp)
355{
356 gboolean result;
357 GRegex *regex = NULL;
358 gsize argv_len;
359 g_autofree gchar *home_path = NULL;
360 char **argv;
361 char *path;
362 int argc;
363
364 result = g_shell_parse_argv (command, &argc, &argv, NULL);
365
366 if (!result)
367 return FALSE;
368
369 regex = g_regex_new ("^~/", 0, 0, NULL);
370 argv_len = g_strv_length (argv);
371 home_path = g_build_filename (g_get_home_dir (), "/", NULL);
372 for (gsize i = 0; i < argv_len; i++)
373 {
374 gchar *tmp_argv = NULL;
375
376 tmp_argv = g_regex_replace_literal (regex, argv[i], -1, 0, home_path, 0, NULL);
377 g_free (argv[i]);
378 argv[i] = tmp_argv;
379 }
380 g_regex_unref (regex);
381 path = g_find_program_in_path (argv[0]);
382
383 if (!path) {
384 g_strfreev (argv);
385 return FALSE;
386 }
387
388 /* If we pass an absolute path to g_find_program it just returns
389 * that absolute path without checking if it is executable. Also
390 * make sure its a regular file so we don't try to launch
391 * directories or device nodes.
392 */
393 if (!g_file_test (path, G_FILE_TEST_IS_EXECUTABLE) ||
394 !g_file_test (path, G_FILE_TEST_IS_REGULAR)) {
395 g_free (path);
396 g_strfreev (argv);
397 return FALSE;
398 }
399
400 g_free (path);
401
402 if (argcp)
403 *argcp = argc;
404 if (argvp)
405 *argvp = argv;
406
407 return TRUE;
408}

421static gboolean
422panel_run_dialog_launch_command (PanelRunDialog *dialog,
423 const char *command,
424 const char *locale_command)
425{
426 gboolean result;
427 GError *error = NULL;
428 char **argv;
429 int argc;
430 GPid pid;
431
432 if (!command_is_executable (locale_command, &argc, &argv))
433 return FALSE;
434
435 if (gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (dialog->terminal_checkbox)))
436 mate_desktop_prepend_terminal_to_vector (&argc, &argv);
437
438 result = g_spawn_async (NULL, /* working directory */
439 argv,
440 NULL, /* envp */
441 G_SPAWN_SEARCH_PATH | G_SPAWN_DO_NOT_REAP_CHILD,
442 NULL,
443 NULL,
444 &pid,
445 &error);
446
447 if (!result) {
448 char *primary;
449
450 primary = g_markup_printf_escaped (_("Could not run command '%s'"),
451 command);
452 panel_error_dialog (GTK_WINDOW (dialog->run_dialog), NULL,
453 "cannot_spawn_command", TRUE,
454 primary, error->message);
455 g_free (primary);
456
457 g_error_free (error);
458 } else {
459 g_child_watch_add (pid, dummy_child_watch, NULL);
460 }
461
462 g_strfreev (argv);
463
464 return result;
465}

467static void
468panel_run_dialog_execute (PanelRunDialog *dialog)
469{
470 GError *error;
471 gboolean result;
472 char *command;
473 char *disk;
474 char *scheme;
475
476 command = g_strdup (panel_run_dialog_get_combo_text (dialog));
477 command = g_strchug (command);
478
479 if (!command || !command [0]) {
480 g_free (command);
481 return;
482 }
483
484 error = NULL;
485 disk = g_locale_from_utf8 (command, -1, NULL, NULL, &error);
486
487 if (!disk || error) {
488 char *primary;
489
490 primary = g_strdup_printf (_("Could not convert '%s' from UTF-8"),
491 command);
492 panel_error_dialog (GTK_WINDOW (dialog->run_dialog), NULL,
493 "cannot_convert_command_from_utf8", TRUE,
494 primary, error->message);
495 g_free (primary);
496
497 g_error_free (error);
498 return;
499 }
500
501 result = FALSE;
502
503 scheme = g_uri_parse_scheme (disk);
504 /* if it's an absolute path or not a URI, it's possibly an executable,
505 * so try it before displaying it */
506 if (g_path_is_absolute (disk) || !scheme)
507 result = panel_run_dialog_launch_command (dialog, command, disk);
508
509 if (!result) {
510 GFile *file;
511 char *uri;
512 GdkScreen *screen;
513
514 file = panel_util_get_file_optional_homedir (command);
515 uri = g_file_get_uri (file);
516 g_object_unref (file);
517
518 screen = gtk_window_get_screen (GTK_WINDOW (dialog->run_dialog));
519 result = panel_show_uri (screen, uri,
520 gtk_get_current_event_time (), NULL);
521
522 g_free (uri);
523 }
524
525 if (result) {
526 /* only save working commands in history */
527 _panel_run_save_recent_programs_list
528 (dialog, GTK_COMBO_BOX (dialog->combobox), command);
529
530 /* only close the dialog if we successfully showed or launched
531 * something */
532 gtk_widget_destroy (dialog->run_dialog);
533 }
534
535 g_free (command);
536 g_free (disk);
537 g_free (scheme);
538}

564/* only quote the string if really needed */
565static char *
566quote_string (const char *s)
567{
568 const char *p;
569
570 for (p = s; *p != '\0'; p++) {
571 if ((*p >= 'a' && *p <= 'z') ||
572 (*p >= 'A' && *p <= 'Z') ||
573 (*p >= '0' && *p <= '9') ||
574 strchr ("-_./=:", *p) != NULL)
575 ;
576 else
577 return g_shell_quote (s);
578 }
579
580 return g_strdup (s);
581}

625static gboolean
626fuzzy_command_match (const char *cmd1,
627 const char *cmd2,
628 gboolean *fuzzy)
629{
630 char **tokens;
631 char *word1, *word2;
632
633 g_return_val_if_fail (cmd1 && cmd2, TRUE);
634
635 *fuzzy = FALSE;
636
637 if (!strcmp (cmd1, cmd2))
638 return TRUE;
639
640 /* find basename of exec from desktop item.
641 strip of all arguments after the initial command */
642 tokens = g_strsplit (cmd1, " ", -1);
643 if (!tokens || !tokens [0]) {
644 g_strfreev (tokens);
645 return FALSE;
646 }
647
648 word1 = g_path_get_basename (tokens [0]);
649 g_strfreev (tokens);
650
651 /* same for the user command */
652 tokens = g_strsplit (cmd2, " ", -1);
653 if (!tokens || !tokens [0]) {
654 g_free (word1);
655 g_strfreev (tokens);
656 return FALSE;
657 }
658
659 word2 = g_path_get_basename (tokens [0]);
660 g_strfreev (tokens);
661
662 if (!strcmp (word1, word2)) {
663 g_free (word1);
664 g_free (word2);
665 *fuzzy = TRUE;
666 return TRUE;
667 }
668
669 g_free (word1);
670 g_free (word2);
671
672 return FALSE;
673}

675static gboolean
676panel_run_dialog_find_command_idle (PanelRunDialog *dialog)
677{
678 GtkTreeIter iter;
679 GtkTreeModel *model;
680 GtkTreePath *path;
681 char *text;
682 GIcon *found_icon;
683 char *found_name;
684 gboolean fuzzy;
685 gint visible_program_idx = 0;
686
687 model = GTK_TREE_MODEL (dialog->program_list_store);
688 path = gtk_tree_path_new_first ();
689
690 if (!path || !gtk_tree_model_get_iter (model, &iter, path)) {
691 if (path)
692 gtk_tree_path_free (path);
693
694 panel_run_dialog_set_icon (dialog, NULL, FALSE);
695
696 dialog->find_command_idle_id = 0;
697 return FALSE;
698 }
699
700 text = g_strdup (panel_run_dialog_get_combo_text (dialog));
701 found_icon = NULL;
702 found_name = NULL;
703 fuzzy = FALSE;
704 g_hash_table_remove_all (accelerator_keys_to_tree_iter_map);
705
706 do {
707 char *exec = NULL;
708 GIcon *icon = NULL;
709 char *name = NULL;
710 char *comment = NULL;
711 gboolean visible = FALSE;
712
713 gtk_tree_model_get (model, &iter,
714 COLUMN_EXEC, &exec,
715 COLUMN_GICON, &icon,
716 COLUMN_NAME, &name,
717 COLUMN_COMMENT, &comment,
718 -1);
719
720 if (!fuzzy && exec && icon &&
721 fuzzy_command_match (text, exec, &fuzzy)) {
722 g_clear_object (&found_icon);
723 g_free (found_name);
724
725 found_icon = g_object_ref (icon);
726 found_name = g_strdup (name);
727
728 gtk_list_store_set (dialog->program_list_store,
729 &iter,
730 COLUMN_VISIBLE, TRUE,
731 -1);
732 } else if (panel_g_utf8_strstrcase (exec, text) != NULL ||
733 panel_g_utf8_strstrcase (name, text) != NULL ||
734 panel_g_utf8_strstrcase (comment, text) != NULL) {
735 gtk_list_store_set (dialog->program_list_store,
736 &iter,
737 COLUMN_VISIBLE, TRUE,
738 -1);
739 } else {
740 gtk_list_store_set (dialog->program_list_store,
741 &iter,
742 COLUMN_VISIBLE, FALSE,
743 -1);
744 }
745
746 gtk_tree_model_get (model, &iter, COLUMN_VISIBLE, &visible, -1);
747 if (visible && visible_program_idx < G_N_ELEMENTS (accelerator_key_mapping)) {
748 gtk_list_store_set (dialog->program_list_store,
749 &iter,
750 COLUMN_ACCELERATOR_MASK, (gint)accelerator_key_mapping[visible_program_idx].modifier,
751 COLUMN_ACCELERATOR_KEY_VALUE, accelerator_key_mapping[visible_program_idx].key_id,
752 -1);

835{
836 MateMenuTreeIter *iter;

839 iter = matemenu_tree_directory_iter (directory);
840 while ((type = matemenu_tree_iter_next (iter)) != MATEMENU_TREE_ITEM_INVALID) {

857 matemenu_tree_item_unref(item);

1023static char *
1024remove_parameters (const char *exec)
1025{
1026 GString *str;
1027 char *retval, *p;
1028
1029 str = g_string_new (exec);
1030
1031 while ((p = strstr (str->str, "%"))) {
1032 switch (p [1]) {
1033 case '%':
1034 g_string_erase (str, p - str->str, 1);
1035 break;
1036 case 'U':
1037 case 'F':
1038 case 'N':
1039 case 'D':
1040 case 'f':
1041 case 'u':
1042 case 'd':
1043 case 'n':
1044 case 'm':
1045 case 'i':
1046 case 'c':
1047 case 'k':
1048 case 'v':
1049 g_string_erase (str, p - str->str, 2);
1050 break;
1051 default:
1052 break;
1053 }
1054 }
1055
1056 retval = str->str;
1057 g_string_free (str, FALSE);
1058
1059 return retval;

1345static GList *
1346fill_files_from (const char *dirname,
1347 const char *dirprefix,
1348 char prefix,
1349 GList *existing_items)
1350{
1351 GList *list;
1352 DIR *dir;
1353 struct dirent *dent;
1354
1355 list = NULL;
1356 dir = opendir (dirname);
1357
1358 if (!dir)
1359 return list;
1360
1361 while ((dent = readdir (dir))) {
1362 char *file;
1363 char *item;
1364 const char *suffix;
1365
1366 if (dent->d_name [0] != prefix)
1367 continue;
1368
1369 file = g_build_filename (dirname, dent->d_name, NULL);
1370
1371 suffix = NULL;
1372 if (
1373#ifdef HAVE_STRUCT_DIRENT_D_TYPE
1374 /* don't use g_file_test at first so we don't stat() */
1375 dent->d_type == DT_DIR ||
1376 (dent->d_type == DT_LNK &&
1377 g_file_test (file, G_FILE_TEST_IS_DIR))
1378#else
1379 g_file_test (file, G_FILE_TEST_IS_DIR)
1380#endif
1381 )
1382 suffix = "/";
1383
1384 g_free (file);
1385
1386 item = g_build_filename (dirprefix, dent->d_name, suffix, NULL);
1387
1388 list = g_list_prepend (list, item);
1389 }

1396static GList *
1397fill_possible_executables (void)
1398{
1399 GList *list;
1400 const char *path;
1401 char **pathv;
1402 int i;
1403
1404 list = NULL;
1405 path = g_getenv ("PATH");
1406
1407 if (!path || !path [0])
1408 return list;
1409
1410 pathv = g_strsplit (path, ":", 0);
1411
1412 for (i = 0; pathv [i]; i++) {
1413 const char *file;
1414 GDir *dir;
1415
1416 dir = g_dir_open (pathv [i], 0, NULL);
1417
1418 if (!dir)
1419 continue;
1420
1421 while ((file = g_dir_read_name (dir))) {
1422 list = g_list_prepend (list, g_build_filename (pathv [i], file, NULL));
1423 }
1424
1425 g_dir_close (dir);
1426 }
1427
1428 g_strfreev (pathv);
1429
1430 return list;
1431}

1437{

1441 list = NULL;
1442
1443 for (l = possible_executables; l; l = l->next) {
1444 const char *filename;

1447 filename = l->data;
1448 basename = g_path_get_basename (filename);

1450 if (basename [0] == prefix &&
1451 g_file_test (filename, G_FILE_TEST_IS_REGULAR) &&
1452 g_file_test (filename, G_FILE_TEST_IS_EXECUTABLE)) {

1455 (GCompareFunc) strcmp)) {

1458 }
1459

1461 } else {
1462 g_free (basename);
1463 }
1464 }
1465
1466 return list;

1496static void
1497panel_run_dialog_update_completion (PanelRunDialog *dialog,
1498 const char *text)
1499{
1500 GList *list;
1501 GList *executables;
1502 char prefix;
1503 char *buf;
1504 char *dirname;
1505 char *dirprefix;
1506 char *key;
1507
1508 g_assert (text != NULL && *text != '\0' && !g_ascii_isspace (*text));
1509
1510 list = NULL;
1511 executables = NULL;
1512
1513 buf = g_path_get_basename (text);
1514 prefix = buf[0];
1515 g_free (buf);
1516 if (prefix == '/' || prefix == '.')
1517 return;
1518
1519 if (text [0] == '/') {
1520 /* complete against absolute path */
1521 dirname = g_path_get_dirname (text);
1522 dirprefix = g_strdup (dirname);
1523 } else {
1524 /* complete against relative path and executable name */
1525 if (!strchr (text, '/')) {
1526 executables = fill_executables (dialog->possible_executables,
1527 dialog->completion_items,
1528 text [0]);
1529 dirprefix = g_strdup ("");
1530 } else {
1531 dirprefix = g_path_get_dirname (text);
1532 }
1533
1534 dirname = g_build_filename (g_get_home_dir (), dirprefix, NULL);
1535 }
1536
1537 key = g_strdup_printf ("%s%c%c", dirprefix, G_DIR_SEPARATOR, prefix);
1538
1539 if (!g_hash_table_lookup (dialog->dir_hash, key)) {
1540 g_hash_table_insert (dialog->dir_hash, key, dialog);
1541
1542 list = fill_files_from (dirname, dirprefix, prefix,
1543 dialog->completion_items);
1544 } else {
1545 g_free (key);
1546 }
1547
1548 list = g_list_concat (list, executables);
1549
1550 g_free (dirname);
1551 g_free (dirprefix);
1552
1553 if (list == NULL)
1554 return;
1555
1556 completion_add_items (dialog->completion, list);
1557 dialog->completion_items = g_list_concat (dialog->completion_items,
1558 list);
1559}

1561static gboolean
1562entry_event (GtkEditable *entry,
1563 GdkEventKey *event,
1564 PanelRunDialog *dialog)
1565{
1566 char *prefix;
1567 char *nospace_prefix;
1568 char *temp;
1569 int pos, tmp;
1570
1571 if (event->type != GDK_KEY_PRESS)
1572 return FALSE;
1573
1574 /* if user typed something we're not using the list anymore
1575 (but only if it is enabled anyway)
1576 */
1577 dialog->use_program_list = FALSE;
1578 if (panel_profile_get_enable_program_list ()) {
1579 GtkTreeSelection *selection = gtk_tree_view_get_selection (GTK_TREE_VIEW (dialog->program_list));
1580 gtk_tree_selection_unselect_all (selection);
1581 }
1582
1583 if (!panel_profile_get_enable_autocompletion ())
1584 return FALSE;
1585
1586 /* tab completion */
1587 if (event->keyval == GDK_KEY_Tab) {
1588 gtk_editable_get_selection_bounds (entry, &pos, &tmp);
1589
1590 if (dialog->completion_started &&
1591 pos != tmp &&
1592 pos != 1 &&
1593 ((size_t) tmp) == strlen (gtk_entry_get_text (GTK_ENTRY (entry)))) {
1594 gtk_editable_select_region (entry, 0, 0);
1595 gtk_editable_set_position (entry, -1);
1596
1597 return TRUE;
1598 }
1599 } else if (event->length > 0) {
1600
1601 gtk_editable_get_selection_bounds (entry, &pos, &tmp);
1602
1603 if (dialog->completion_started &&
1604 pos != tmp &&
1605 pos != 0 &&
1606 ((size_t) tmp) == strlen (gtk_entry_get_text (GTK_ENTRY (entry)))) {
1607 temp = gtk_editable_get_chars (entry, 0, pos);
1608 prefix = g_strconcat (temp, event->string, NULL);
1609 g_free (temp);
1610 } else if (pos == tmp &&
1611 ((size_t) tmp) == strlen (gtk_entry_get_text (GTK_ENTRY (entry)))) {
1612 prefix = g_strconcat (gtk_entry_get_text (GTK_ENTRY (entry)),
1613 event->string, NULL);
1614 } else {
1615 return FALSE;
1616 }
1617
1618 nospace_prefix = prefix;
1619 while (*nospace_prefix != '\0' &&
1620 g_ascii_isspace (*nospace_prefix))
1621 nospace_prefix++;
1622 if (*nospace_prefix == '\0')
1623 return FALSE;
1624
1625 panel_run_dialog_update_completion (dialog, nospace_prefix);
1626
1627 if (!dialog->completion) {
1628 g_free (prefix);
1629 return FALSE;
1630 }
1631
1632 g_free (prefix);
1633 }
1634
1635 return FALSE;
1636}
