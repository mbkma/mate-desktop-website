257static void
258on_main_selection_changed( CactIBasenamesTab *instance, GList *selected_items, gpointer user_data )
259{
260 NAIContext *context;
261 gboolean editable;
262 gboolean enable_tab;
263 GtkToggleButton *matchcase_button;
264 gboolean matchcase;
265 IBasenamesData *data;
266
267 g_object_get( G_OBJECT( instance ),
268 MAIN_PROP_CONTEXT, &context, MAIN_PROP_EDITABLE, &editable,
269 NULL );
270
271 enable_tab = ( context != NULL );
272 cact_main_tab_enable_page( CACT_MAIN_WINDOW( instance ), TAB_BASENAMES, enable_tab );
273
274 data = get_ibasenames_data( CACT_IBASENAMES_TAB( instance ));
275 data->on_selection_change = TRUE;
276
277 matchcase_button = GTK_TOGGLE_BUTTON( base_window_get_widget( BASE_WINDOW( instance ), "BasenamesMatchcaseButton" ));
278 matchcase = context ? na_object_get_matchcase( context ) : FALSE;
279 gtk_toggle_button_set_active( GTK_TOGGLE_BUTTON( matchcase_button ), matchcase );
280 base_gtk_utils_set_editable( G_OBJECT( matchcase_button ), editable );
281
282 data->on_selection_change = FALSE;
283}

285static void
286on_matchcase_toggled( GtkToggleButton *button, BaseWindow *window )
287{
288 NAIContext *context;
289 gboolean editable;
290 gboolean matchcase;
291 IBasenamesData *data;
292
293 data = get_ibasenames_data( CACT_IBASENAMES_TAB( window ));
294
295 if( !data->on_selection_change ){
296 g_object_get( G_OBJECT( window ),
297 MAIN_PROP_CONTEXT, &context, MAIN_PROP_EDITABLE, &editable,
298 NULL );
299
300 if( context ){
301 matchcase = gtk_toggle_button_get_active( button );
302
303 if( editable ){
304 na_object_set_matchcase( context, matchcase );
305 g_signal_emit_by_name( G_OBJECT( window ), TAB_UPDATABLE_SIGNAL_ITEM_UPDATED, context, 0 );
306
307 } else {
308 g_signal_handlers_block_by_func(( gpointer ) button, on_matchcase_toggled, window );
309 gtk_toggle_button_set_active( button, !matchcase );
310 g_signal_handlers_unblock_by_func(( gpointer ) button, on_matchcase_toggled, window );
311 }
312 }
313 }
