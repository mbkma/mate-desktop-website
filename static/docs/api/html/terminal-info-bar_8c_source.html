61GtkWidget *
62terminal_info_bar_new (GtkMessageType type,
63 const char *first_button_text,
64 ...)
65{
66 GtkWidget *info_bar;
67 va_list args;
68
69 info_bar = g_object_new (TERMINAL_TYPE_INFO_BAR,
70 "message-type", type,
71 NULL);
72
73 va_start (args, first_button_text);
74 while (first_button_text != NULL)
75 {
76 int response_id;
77
78 response_id = va_arg (args, int);
79 gtk_info_bar_add_button (GTK_INFO_BAR (info_bar),
80 first_button_text, response_id);
81
82 first_button_text = va_arg (args, const char *);
83 }
84 va_end (args);
85
86 return info_bar;
87}

93{
94 TerminalInfoBarPrivate *priv;

102

105 va_end (args);
106
107 label = gtk_label_new (text);
108 g_free (text);
109
110 gtk_label_set_line_wrap (GTK_LABEL (label), TRUE);

112 gtk_label_set_xalign (GTK_LABEL (label), 0.0);
113 gtk_label_set_yalign (GTK_LABEL (label), 0.0);
114
115 gtk_box_pack_start (GTK_BOX (priv->content_box), label, FALSE, FALSE, 0);
116 gtk_widget_show_all (priv->content_box);
