69char *
70gsm_util_find_desktop_file_for_app_name (const char *name,
71 char **autostart_dirs)
72{
73 char *app_path;
74 char **app_dirs;
75 GKeyFile *key_file;
76 char *desktop_file;
77 int i;
78
79 app_path = NULL;
80
81 app_dirs = gsm_util_get_app_dirs ();
82
83 key_file = g_key_file_new ();
84
85 desktop_file = g_strdup_printf ("%s.desktop", name);
86
87 g_debug ("GsmUtil: Looking for file '%s'", desktop_file);
88
89 for (i = 0; app_dirs[i] != NULL; i++) {
90 g_debug ("GsmUtil: Looking in '%s'", app_dirs[i]);
91 }
92
93 g_key_file_load_from_dirs (key_file,
94 desktop_file,
95 (const char **) app_dirs,
96 &app_path,
97 G_KEY_FILE_NONE,
98 NULL);
99
100 if (app_path != NULL) {
101 g_debug ("GsmUtil: found in XDG app dirs: '%s'", app_path);
102 }
103
104 if (app_path == NULL && autostart_dirs != NULL) {
105 g_key_file_load_from_dirs (key_file,
106 desktop_file,
107 (const char **) autostart_dirs,
108 &app_path,
109 G_KEY_FILE_NONE,
110 NULL);
111 if (app_path != NULL) {
112 g_debug ("GsmUtil: found in autostart dirs: '%s'", app_path);
113 }
114
115 }
116
117 /* look for mate vender prefix */
118 if (app_path == NULL) {
119 g_free (desktop_file);
120 desktop_file = g_strdup_printf ("mate-%s.desktop", name);
121
122 g_key_file_load_from_dirs (key_file,
123 desktop_file,
124 (const char **) app_dirs,
125 &app_path,
126 G_KEY_FILE_NONE,
127 NULL);
128 if (app_path != NULL) {
129 g_debug ("GsmUtil: found in XDG app dirs: '%s'", app_path);
130 }
131 }
132
133 if (app_path == NULL && autostart_dirs != NULL) {
134 g_key_file_load_from_dirs (key_file,
135 desktop_file,
136 (const char **) autostart_dirs,
137 &app_path,
138 G_KEY_FILE_NONE,
139 NULL);
140 if (app_path != NULL) {
141 g_debug ("GsmUtil: found in autostart dirs: '%s'", app_path);
142 }
143 }
144
145 g_free (desktop_file);
146 g_key_file_free (key_file);
147
148 g_strfreev (app_dirs);
149
150 return app_path;
151}

153static gboolean
154ensure_dir_exists (const char *dir)
155{
156 if (g_file_test (dir, G_FILE_TEST_IS_DIR))
157 return TRUE;
158
159 if (g_mkdir_with_parents (dir, 0755) == 0)
160 return TRUE;
161
162 if (errno == EEXIST)
163 return g_file_test (dir, G_FILE_TEST_IS_DIR);
164
165 g_warning ("GsmSessionSave: Failed to create directory %s: %s", dir, strerror (errno));
166
167 return FALSE;
168}

170gchar *
171gsm_util_get_empty_tmp_session_dir (void)
172{
173 char *tmp;
174 gboolean exists;
175
176 tmp = g_build_filename (g_get_user_config_dir (),
177 "mate-session",
178 "saved-session.new",
179 NULL);
180
181 exists = ensure_dir_exists (tmp);
182
183 if (G_UNLIKELY (!exists)) {
184 g_warning ("GsmSessionSave: could not create directory for saved session: %s", tmp);
185 g_free (tmp);
186 return NULL;
187 } else {
188 /* make sure it's empty */
189 GDir *dir;
190 const char *filename;
191
192 dir = g_dir_open (tmp, 0, NULL);
193 if (dir) {
194 while ((filename = g_dir_read_name (dir))) {
195 gchar *path = g_build_filename (tmp, filename,
196 NULL);
197 g_unlink (path);
198 g_free (path);
199 }
200 g_dir_close (dir);
201 }
202 }
203
204 return tmp;
205}

298char **
299gsm_util_get_desktop_dirs ()
300{
301 char **apps;
302 char **autostart;
303 char **result;
304 int size;
305 int i;
306
307 apps = gsm_util_get_app_dirs ();
308 autostart = gsm_util_get_autostart_dirs ();
309
310 size = 0;
311 for (i = 0; apps[i] != NULL; i++) { size++; }
312 for (i = 0; autostart[i] != NULL; i++) { size++; }
313 size += 2; /* saved session + last NULL */
314
315 result = g_new (char *, size + 1);
316
317 for (i = 0; apps[i] != NULL; i++) {
318 result[i] = apps[i];
319 }
320 g_free (apps);
321 size = i;
322
323 for (i = 0; autostart[i] != NULL; i++) {
324 result[size + i] = autostart[i];
325 }
326 g_free (autostart);
327 size = size + i;
328
329 result[size] = g_strdup (gsm_util_get_saved_session_dir ());
330 result[size + 1] = NULL;
331
332 return result;
333}

410char *
411gsm_util_generate_startup_id (void)
412{
413 static int sequence = -1;
414 static guint rand1 = 0;
415 static guint rand2 = 0;
416 static pid_t pid = 0;
417 struct timeval tv;
418
419 /* The XSMP spec defines the ID as:
420 *
421 * Version: "1"
422 * Address type and address:
423 * "1" + an IPv4 address as 8 hex digits
424 * "2" + a DECNET address as 12 hex digits
425 * "6" + an IPv6 address as 32 hex digits
426 * Time stamp: milliseconds since UNIX epoch as 13 decimal digits
427 * Process-ID type and process-ID:
428 * "1" + POSIX PID as 10 decimal digits
429 * Sequence number as 4 decimal digits
430 *
431 * XSMP client IDs are supposed to be globally unique: if
432 * SmsGenerateClientID() is unable to determine a network
433 * address for the machine, it gives up and returns %NULL.
434 * MATE and KDE have traditionally used a fourth address
435 * format in this case:
436 * "0" + 16 random hex digits
437 *
438 * We don't even bother trying SmsGenerateClientID(), since the
439 * user's IP address is probably "192.168.1.*" anyway, so a random
440 * number is actually more likely to be globally unique.
441 */
442
443 if (!rand1) {
444 rand1 = g_random_int ();
445 rand2 = g_random_int ();
446 pid = getpid ();
447 }
448
449 sequence = (sequence + 1) % 10000;
450 gettimeofday (&tv, NULL);
451 return g_strdup_printf ("10%.04x%.04x%.10lu%.3u%.10lu%.4d",
452 rand1,
453 rand2,
454 (unsigned long) tv.tv_sec,
455 (unsigned) tv.tv_usec,
456 (unsigned long) pid,
457 sequence);
458}

460static gboolean
461gsm_util_update_activation_environment (const char *variable,
462 const char *value,
463 GError **error)
464{
465 GDBusConnection *connection;
466 gboolean environment_updated;
467 GVariantBuilder builder;
468 GVariant *reply;
469 GError *bus_error = NULL;
470
471 environment_updated = FALSE;
472 connection = g_bus_get_sync (G_BUS_TYPE_SESSION, NULL, error);
473
474 if (connection == NULL) {
475 return FALSE;
476 }
477
478 g_variant_builder_init (&builder, G_VARIANT_TYPE ("a{ss}"));
479 g_variant_builder_add (&builder, "{ss}", variable, value);
480
481 reply = g_dbus_connection_call_sync (connection,
482 "org.freedesktop.DBus",
483 "/org/freedesktop/DBus",
484 "org.freedesktop.DBus",
485 "UpdateActivationEnvironment",
486 g_variant_new ("(@a{ss})",
487 g_variant_builder_end (&builder)),
488 NULL,
489 G_DBUS_CALL_FLAGS_NONE,
490 -1, NULL, &bus_error);
491
492 if (bus_error != NULL) {
493 g_propagate_error (error, bus_error);
494 } else {
495 environment_updated = TRUE;
496 g_variant_unref (reply);
497 }
498
499 g_clear_object (&connection);
500
501 return environment_updated;
502}

511gboolean
512gsm_util_export_activation_environment (GError **error)
513{
514 GDBusConnection *connection;
515 gboolean environment_updated = FALSE;
516 char **entry_names;
517 int i = 0;
518 GVariantBuilder builder;
519 GRegex *name_regex, *value_regex;
520 GVariant *reply;
521 GError *bus_error = NULL;
522
523 connection = g_bus_get_sync (G_BUS_TYPE_SESSION, NULL, error);
524
525 if (connection == NULL) {
526 return FALSE;
527 }
528
529 name_regex = g_regex_new ("^" ENV_NAME_PATTERN "$", G_REGEX_OPTIMIZE, 0, error);
530
531 if (name_regex == NULL) {
532 return FALSE;
533 }
534
535
536 value_regex = g_regex_new ("^" ENV_VALUE_PATTERN "$", G_REGEX_OPTIMIZE, 0, error);
537
538 if (value_regex == NULL) {
539 return FALSE;
540 }
541
542 g_variant_builder_init (&builder, G_VARIANT_TYPE ("a{ss}"));
543 for (entry_names = g_listenv (); entry_names[i] != NULL; i++) {
544 const char *entry_name = entry_names[i];
545 const char *entry_value = g_getenv (entry_name);
546
547 if (!g_utf8_validate (entry_name, -1, NULL))
548 continue;
549
550 if (!g_regex_match (name_regex, entry_name, 0, NULL))
551 continue;
552
553 if (!g_utf8_validate (entry_value, -1, NULL))
554 continue;
555
556 if (!g_regex_match (value_regex, entry_value, 0, NULL))
557 continue;
558
559 g_variant_builder_add (&builder, "{ss}", entry_name, entry_value);
560 }
561 g_regex_unref (name_regex);
562 g_regex_unref (value_regex);
563
564 g_strfreev (entry_names);
565
566 reply = g_dbus_connection_call_sync (connection,
567 "org.freedesktop.DBus",
568 "/org/freedesktop/DBus",
569 "org.freedesktop.DBus",
570 "UpdateActivationEnvironment",
571 g_variant_new ("(@a{ss})",
572 g_variant_builder_end (&builder)),
573 NULL,
574 G_DBUS_CALL_FLAGS_NONE,
575 -1, NULL, &bus_error);
576
577 if (bus_error != NULL) {
578 g_propagate_error (error, bus_error);
579 } else {
580 environment_updated = TRUE;
581 g_variant_unref (reply);
582 }
583
584 g_clear_object (&connection);
585
586 return environment_updated;
587}

701void
702gsm_util_setenv (const char *variable,
703 const char *value)
704{
705 GError *error = NULL;
706
707 g_setenv (variable, value, TRUE);
708
709 /* If this fails it isn't fatal, it means some things like session
710 * management and keyring won't work in activated clients.
711 */
712 if (!gsm_util_update_activation_environment (variable, value, &error)) {
713 g_warning ("Could not make bus activated clients aware of %s=%s environment variable: %s", variable, value, error->message);
714 g_clear_error (&error);
715 }
716
717#ifdef HAVE_SYSTEMD
718 /* If this fails, the system user session won't get the updated environment
719 */
720 if (!gsm_util_update_user_environment (variable, value, &error)) {
721 g_debug ("Could not make systemd aware of %s=%s environment variable: %s", variable, value, error->message);
722 g_clear_error (&error);
723 }
724#endif
725}
