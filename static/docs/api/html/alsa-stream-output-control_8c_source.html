95static gboolean
96alsa_stream_output_control_load (AlsaStreamControl *control)
97{
98 AlsaControlData data;
99 snd_mixer_elem_t *el;
100
101 g_return_val_if_fail (ALSA_IS_STREAM_OUTPUT_CONTROL (control), FALSE);
102
103 el = alsa_element_get_snd_element (ALSA_ELEMENT (control));
104 if (G_UNLIKELY (el == NULL))
105 return FALSE;
106
107 if (G_UNLIKELY (snd_mixer_selem_has_playback_volume (el) == 0 &&
108 snd_mixer_selem_has_common_volume (el) == 0)) {
109 g_warn_if_reached ();
110 return FALSE;
111 }
112
113 memset (&data, 0, sizeof (AlsaControlData));
114
115 /* We model any control switch as mute */
116 if (snd_mixer_selem_has_playback_switch (el) == 1 ||
117 snd_mixer_selem_has_common_switch (el) == 1)
118 data.switch_usable = TRUE;
119
120 data.active = snd_mixer_selem_is_active (el);
121
122 /* Read the volume data but do not error out if it fails, since ALSA reports
123 * the control to have a volume, expect the control to match what we need - slider
124 * with an optional mute toggle.
125 * If it fails to read the volume data, just treat it as a volumeless control */
126 read_volume_data (el, &data);
127
128 alsa_stream_control_set_data (control, &data);
129 return TRUE;
130}

139
140 el = alsa_element_get_snd_element (ALSA_ELEMENT (control));
141 if (G_UNLIKELY (el == NULL))
142 return FALSE;
143

148 return FALSE;

255static void
256read_volume_data (snd_mixer_elem_t *el, AlsaControlData *data)
257{
258 glong volume;
259 glong min, max;
260 gint ret;
261 gint i;
262
263 /* Read volume ranges, this call should never fail on valid input */
264#if SND_LIB_VERSION >= ALSA_PACK_VERSION (1, 0, 10)
265 ret = snd_mixer_selem_get_playback_volume_range (el, &min, &max);
266 if (G_UNLIKELY (ret < 0)) {
267 g_warning ("Failed to read playback volume range: %s", snd_strerror (ret));
268 return;
269 }
270#else
271 snd_mixer_selem_get_playback_volume_range (el, &min, &max);
272#endif
273 data->min = (guint) min;
274 data->max = (guint) max;
275
276#if SND_LIB_VERSION >= ALSA_PACK_VERSION (1, 0, 10)
277 /* This fails when decibels are not supported */
278 ret = snd_mixer_selem_get_playback_dB_range (el, &min, &max);
279 if (ret == 0) {
280 data->min_decibel = min / 100.0;
281 data->max_decibel = max / 100.0;
282 } else
283 data->min_decibel = data->max_decibel = -MATE_MIXER_INFINITY;
284#else
285 data->min_decibel = data->max_decibel = -MATE_MIXER_INFINITY;
286#endif
287
288 for (i = 0; i < MATE_MIXER_CHANNEL_MAX; i++)
289 data->v[i] = data->min;
290
291 data->volume = data->min;
292 data->volume_joined = snd_mixer_selem_has_playback_volume_joined (el);
293
294 if (data->switch_usable == TRUE)
295 data->switch_joined = snd_mixer_selem_has_playback_switch_joined (el);
296
297 if (snd_mixer_selem_is_playback_mono (el) == 1) {
298 /* Special handling for single channel controls */
299 ret = snd_mixer_selem_get_playback_volume (el, SND_MIXER_SCHN_MONO, &volume);
300 if (ret == 0) {
301 data->channels = 1;
302
303 data->c[0] = MATE_MIXER_CHANNEL_MONO;
304 data->v[0] = data->volume = (guint) volume;
305 } else {
306 g_warning ("Failed to read playback volume: %s", snd_strerror (ret));
307 }
308
309 if (data->switch_usable == TRUE) {
310 gint value;
311
312 ret = snd_mixer_selem_get_playback_switch (el, SND_MIXER_SCHN_MONO, &value);
313 if (G_LIKELY (ret == 0))
314 data->m[0] = !value;
315 }
316 } else {
317 snd_mixer_selem_channel_id_t channel;
318
319 /* We use numeric channel indices, but ALSA only works with channel
320 * positions, go over all the positions supported by ALSA and create
321 * a list of channels */
322 for (channel = 0; channel < SND_MIXER_SCHN_LAST; channel++) {
323 if (snd_mixer_selem_has_playback_channel (el, channel) == 0)
324 continue;
325
326 if (data->switch_usable == TRUE) {
327 gint value;
328
329 ret = snd_mixer_selem_get_playback_switch (el, channel, &value);
330 if (ret == 0)
331 data->m[channel] = !value;
332 }
333
334 ret = snd_mixer_selem_get_playback_volume (el, channel, &volume);
335 if (ret < 0) {
336 g_warning ("Failed to read playback volume: %s", snd_strerror (ret));
337 continue;
338 }
339 data->channels++;
340
341 /* The single value volume is the highest channel volume */
342 if (data->volume < volume)
343 data->volume = volume;
344
345 data->c[channel] = alsa_channel_map_from[channel];
346 data->v[channel] = (guint) volume;
347 }
348 }
