95static EelCanvasItem *
96create_selection_shadow (CajaIconContainer *container,
97 GList *list)
98{
99 EelCanvasGroup *group;
100 EelCanvas *canvas;
101 int max_x, max_y;
102 int min_x, min_y;
103 GList *p;
104 GtkAllocation allocation;
105
106 if (list == NULL)
107 {
108 return NULL;
109 }
110
111 /* if we're only dragging a single item, don't worry about the shadow */
112 if (list->next == NULL)
113 {
114 return NULL;
115 }
116
117 canvas = EEL_CANVAS (container);
118 gtk_widget_get_allocation (GTK_WIDGET (container), &allocation);
119
120 /* Creating a big set of rectangles in the canvas can be expensive, so
121 we try to be smart and only create the maximum number of rectangles
122 that we will need, in the vertical/horizontal directions. */
123
124 max_x = allocation.width;
125 min_x = -max_x;
126
127 max_y = allocation.height;
128 min_y = -max_y;
129
130 /* Create a group, so that it's easier to move all the items around at
131 once. */
132 group = EEL_CANVAS_GROUP
133 (eel_canvas_item_new (EEL_CANVAS_GROUP (canvas->root),
134 eel_canvas_group_get_type (),
135 NULL));
136
137 for (p = list; p != NULL; p = p->next)
138 {
139 CajaDragSelectionItem *item;
140 int x1, y1, x2, y2;
141 GdkRGBA black = { 0, 0, 0, 1 };
142
143 item = p->data;
144
145 if (!item->got_icon_position)
146 {
147 continue;
148 }
149
150 x1 = item->icon_x;
151 y1 = item->icon_y;
152 x2 = x1 + item->icon_width;
153 y2 = y1 + item->icon_height;
154
155 if (x2 >= min_x && x1 <= max_x && y2 >= min_y && y1 <= max_y)
156 eel_canvas_item_new
157 (group,
158 eel_canvas_rect_get_type (),
159 "x1", (double) x1,
160 "y1", (double) y1,
161 "x2", (double) x2,
162 "y2", (double) y2,
163 "outline-color-rgba", &black,
164 "outline-stippling", TRUE,
165 "width_pixels", 1,
166 NULL);
167 }
168
169 return EEL_CANVAS_ITEM (group);
170}

194static void
195canvas_rect_world_to_widget (EelCanvas *canvas,
196 EelDRect *world_rect,
197 EelIRect *widget_rect)
198{
199 EelDRect window_rect;
200 GtkAdjustment *hadj, *vadj;
201
202 hadj = gtk_scrollable_get_hadjustment (GTK_SCROLLABLE (canvas));
203 vadj = gtk_scrollable_get_vadjustment (GTK_SCROLLABLE (canvas));
204
205 eel_canvas_world_to_window (canvas,
206 world_rect->x0, world_rect->y0,
207 &window_rect.x0, &window_rect.y0);
208 eel_canvas_world_to_window (canvas,
209 world_rect->x1, world_rect->y1,
210 &window_rect.x1, &window_rect.y1);
211 widget_rect->x0 = (int) window_rect.x0 - gtk_adjustment_get_value (hadj);
212 widget_rect->y0 = (int) window_rect.y0 - gtk_adjustment_get_value (vadj);
213 widget_rect->x1 = (int) window_rect.x1 - gtk_adjustment_get_value (hadj);
214 widget_rect->y1 = (int) window_rect.y1 - gtk_adjustment_get_value (vadj);
215}

228static gboolean
229icon_get_data_binder (CajaIcon *icon, gpointer data)
230{
231 IconGetDataBinderContext *context;
232 EelDRect world_rect;
233 EelIRect widget_rect;
234 char *uri;
235 CajaIconContainer *container;
236
237 context = (IconGetDataBinderContext *)data;
238
239 g_assert (CAJA_IS_ICON_CONTAINER (context->iterator_context));
240
241 container = CAJA_ICON_CONTAINER (context->iterator_context);
242
243 world_rect = caja_icon_canvas_item_get_icon_rectangle (icon->item);
244
245 canvas_rect_world_to_widget (EEL_CANVAS (container), &world_rect, &widget_rect);
246
247 uri = caja_icon_container_get_icon_uri (container, icon);
248 if (uri == NULL)
249 {
250 g_warning ("no URI for one of the iterated icons");
251 return TRUE;
252 }
253
254 widget_rect = eel_irect_offset_by (widget_rect,
255 - container->details->dnd_info->drag_info.start_x,
256 - container->details->dnd_info->drag_info.start_y);
257
258 widget_rect = eel_irect_scale_by (widget_rect,
259 1 / EEL_CANVAS (container)->pixels_per_unit);
260
261 /* pass the uri, mouse-relative x/y and icon width/height */
262 context->iteratee (uri,
263 (int) widget_rect.x0,
264 (int) widget_rect.y0,
265 widget_rect.x1 - widget_rect.x0,
266 widget_rect.y1 - widget_rect.y0,
267 context->iteratee_data);
268
269 g_free (uri);
270
271 return TRUE;
272}

390static char *
391get_direct_save_filename (GdkDragContext *context)
392{
393 guchar *prop_text;
394 gint prop_len;
395
396 if (!gdk_property_get (gdk_drag_context_get_source_window (context), gdk_atom_intern (CAJA_ICON_DND_XDNDDIRECTSAVE_TYPE, FALSE),
397 gdk_atom_intern ("text/plain", FALSE), 0, 1024, FALSE, NULL, NULL,
398 &prop_len, &prop_text))
399 {
400 return NULL;
401 }
402
403 /* Zero-terminate the string */
404 prop_text = g_realloc (prop_text, prop_len + 1);
405 prop_text[prop_len] = '\0';
406
407 /* Verify that the file name provided by the source is valid */
408 if (*prop_text == '\0' ||
409 strchr ((const gchar *) prop_text, G_DIR_SEPARATOR) != NULL)
410 {
411 caja_debug_log (FALSE, CAJA_DEBUG_LOG_DOMAIN_USER,
412 "Invalid filename provided by XDS drag site");
413 g_free (prop_text);
414 return NULL;
415 }
416
417 return prop_text;
418}

420static void
421set_direct_save_uri (GtkWidget *widget, GdkDragContext *context, CajaDragInfo *drag_info, int x, int y)
422{
423 char *filename, *drop_target;
424 gchar *uri;
425
426 drag_info->got_drop_data_type = TRUE;
427 drag_info->data_type = CAJA_ICON_DND_XDNDDIRECTSAVE;
428
429 uri = NULL;
430
431 filename = get_direct_save_filename (context);
432 drop_target = caja_icon_container_find_drop_target (CAJA_ICON_CONTAINER (widget),
433 context, x, y, NULL, TRUE);
434
435 if (drop_target && eel_uri_is_trash (drop_target))
436 {
437 g_free (drop_target);
438 drop_target = NULL; /* Cannot save to trash ...*/
439 }
440
441 if (filename != NULL && drop_target != NULL)
442 {
443 GFile *base, *child;
444
445 /* Resolve relative path */
446 base = g_file_new_for_uri (drop_target);
447 child = g_file_get_child (base, filename);
448 uri = g_file_get_uri (child);
449 g_object_unref (base);
450 g_object_unref (child);
451
452 /* Change the uri property */
453 gdk_property_change (gdk_drag_context_get_source_window (context),
454 gdk_atom_intern (CAJA_ICON_DND_XDNDDIRECTSAVE_TYPE, FALSE),
455 gdk_atom_intern ("text/plain", FALSE), 8,
456 GDK_PROP_MODE_REPLACE, (const guchar *) uri,
457 strlen (uri));
458
459 drag_info->direct_save_uri = uri;
460 }
461
462 g_free (filename);
463 g_free (drop_target);
464}

467static void
468get_data_on_first_target_we_support (GtkWidget *widget, GdkDragContext *context, guint32 time, int x, int y)
469{
470 GtkTargetList *list;
471 GdkAtom target;
472
473 if (drop_types_list == NULL)
474 {
475 drop_types_list = gtk_target_list_new (drop_types,
476 G_N_ELEMENTS (drop_types) - 1);
477 gtk_target_list_add_text_targets (drop_types_list, CAJA_ICON_DND_TEXT);
478 }
479 if (drop_types_list_root == NULL)
480 {
481 drop_types_list_root = gtk_target_list_new (drop_types,
482 G_N_ELEMENTS (drop_types));
483 gtk_target_list_add_text_targets (drop_types_list_root, CAJA_ICON_DND_TEXT);
484 }
485
486 if (caja_icon_container_get_is_desktop (CAJA_ICON_CONTAINER (widget)))
487 {
488 list = drop_types_list_root;
489 }
490 else
491 {
492 list = drop_types_list;
493 }
494
495 target = gtk_drag_dest_find_target (widget, context, list);
496 if (target != GDK_NONE)
497 {
498 guint info;
499 CajaDragInfo *drag_info;
500 gboolean found;
501
502 drag_info = &(CAJA_ICON_CONTAINER (widget)->details->dnd_info->drag_info);
503
504 found = gtk_target_list_find (list, target, &info);
505 g_assert (found);
506
507 /* Don't get_data for destructive ops */
508 if ((info == CAJA_ICON_DND_ROOTWINDOW_DROP ||
509 info == CAJA_ICON_DND_XDNDDIRECTSAVE) &&
510 !drag_info->drop_occured)
511 {
512 /* We can't call get_data here, because that would
513 make the source execute the rootwin action or the direct save */
514 drag_info->got_drop_data_type = TRUE;
515 drag_info->data_type = info;
516 }
517 else
518 {
519 if (info == CAJA_ICON_DND_XDNDDIRECTSAVE)
520 {
521 set_direct_save_uri (widget, context, drag_info, x, y);
522 }
523 gtk_drag_get_data (GTK_WIDGET (widget), context,
524 target, time);
525 }
526 }

559static CajaIcon *
560caja_icon_container_item_at (CajaIconContainer *container,
561 int x, int y)
562{
563 GList *p;
564 int size;
565 EelDRect point;
566 EelIRect canvas_point;
567
568 /* build the hit-test rectangle. Base the size on the scale factor to ensure that it is
569 * non-empty even at the smallest scale factor
570 */
571
572 size = MAX (1, 1 + (1 / EEL_CANVAS (container)->pixels_per_unit));
573 point.x0 = x;
574 point.y0 = y;
575 point.x1 = x + size;
576 point.y1 = y + size;
577
578 for (p = container->details->icons; p != NULL; p = p->next)
579 {
580 CajaIcon *icon;
581 icon = p->data;
582
583 eel_canvas_w2c (EEL_CANVAS (container),
584 point.x0,
585 point.y0,
586 &canvas_point.x0,
587 &canvas_point.y0);
588 eel_canvas_w2c (EEL_CANVAS (container),
589 point.x1,
590 point.y1,
591 &canvas_point.x1,
592 &canvas_point.y1);
593 if (caja_icon_canvas_item_hit_test_rectangle (icon->item, canvas_point))
594 {
595 return icon;
596 }
597 }
598
599 return NULL;
600}

839static int
840auto_scroll_timeout_callback (gpointer data)
841{
842 CajaIconContainer *container;
843 GtkWidget *widget;
844 float x_scroll_delta, y_scroll_delta;
845 GdkRectangle exposed_area;
846 GtkAllocation allocation;
847
848 g_assert (CAJA_IS_ICON_CONTAINER (data));
849 widget = GTK_WIDGET (data);
850 container = CAJA_ICON_CONTAINER (widget);
851
852 if (container->details->dnd_info->drag_info.waiting_to_autoscroll
853 && container->details->dnd_info->drag_info.start_auto_scroll_in > g_get_monotonic_time())
854 {
855 /* not yet */
856 return TRUE;
857 }
858
859 container->details->dnd_info->drag_info.waiting_to_autoscroll = FALSE;
860
861 caja_drag_autoscroll_calculate_delta (widget, &x_scroll_delta, &y_scroll_delta);
862 if (x_scroll_delta == 0 && y_scroll_delta == 0)
863 {
864 /* no work */
865 return TRUE;
866 }
867
868 /* Clear the old dnd highlight frame */
869 dnd_highlight_queue_redraw (widget);
870
871 if (!caja_icon_container_scroll (container, (int)x_scroll_delta, (int)y_scroll_delta))
872 {
873 /* the scroll value got pinned to a min or max adjustment value,
874 * we ended up not scrolling
875 */
876 return TRUE;
877 }
878
879 /* Make sure the dnd highlight frame is redrawn */
880 dnd_highlight_queue_redraw (widget);
881
882 /* update cached drag start offsets */
883 container->details->dnd_info->drag_info.start_x -= x_scroll_delta;
884 container->details->dnd_info->drag_info.start_y -= y_scroll_delta;
885
886 /* Due to a glitch in GtkLayout, whe need to do an explicit draw of the exposed
887 * area.
888 * Calculate the size of the area we need to draw
889 */
890 gtk_widget_get_allocation (widget, &allocation);
891 exposed_area.x = allocation.x;
892 exposed_area.y = allocation.y;
893 exposed_area.width = allocation.width;
894 exposed_area.height = allocation.height;
895
896 if (x_scroll_delta > 0)
897 {
898 exposed_area.x = exposed_area.width - x_scroll_delta;
899 }
900 else if (x_scroll_delta < 0)
901 {
902 exposed_area.width = -x_scroll_delta;
903 }
904
905 if (y_scroll_delta > 0)
906 {
907 exposed_area.y = exposed_area.height - y_scroll_delta;
908 }
909 else if (y_scroll_delta < 0)
910 {
911 exposed_area.height = -y_scroll_delta;
912 }
913
914 /* offset it to 0, 0 */
915 exposed_area.x -= allocation.x;
916 exposed_area.y -= allocation.y;
917
918 gtk_widget_queue_draw_area (widget,
919 exposed_area.x,
920 exposed_area.y,
921 exposed_area.width,
922 exposed_area.height);
923
924 return TRUE;
925}

942static void
943handle_local_move (CajaIconContainer *container,
944 double world_x, double world_y)
945{
946 GList *moved_icons, *p;
947 CajaFile *file;
948 char screen_string[32];
949 time_t now;
950 CajaDragSelectionItem *item = NULL;
951 CajaIcon *icon = NULL;
952
953 if (container->details->auto_layout)
954 {
955 return;
956 }
957
958 time (&now);
959
960 /* Move and select the icons. */
961 moved_icons = NULL;
962 for (p = container->details->dnd_info->drag_info.selection_list; p != NULL; p = p->next)
963 {
964 item = p->data;
965
966 icon = caja_icon_container_get_icon_by_uri
967 (container, item->uri);
968
969 if (icon == NULL)
970 {
971 /* probably dragged from another screen. Add it to
972 * this screen
973 * Update: since GTK 3.22 there is only ONE screen
974 * even with multiple monitors
975 */
976
977 file = caja_file_get_by_uri (item->uri);
978
979 g_snprintf (screen_string, sizeof (screen_string), "%d",
980 /*gdk_x11_screen_get_screen_number (screen)); This is ALWAYS 0 since GTK 3.22*/
981 0);
982 caja_file_set_metadata (file,
983 CAJA_METADATA_KEY_SCREEN,
984 NULL, screen_string);
985 caja_file_set_time_metadata (file,
986 CAJA_METADATA_KEY_ICON_POSITION_TIMESTAMP, now);
987
988 caja_icon_container_add (container, CAJA_ICON_CONTAINER_ICON_DATA (file));
989
990 icon = caja_icon_container_get_icon_by_uri
991 (container, item->uri);
992 }
993
994 if (item->got_icon_position)
995 {
996 caja_icon_container_move_icon
997 (container, icon,
998 world_x + item->icon_x, world_y + item->icon_y,
999 icon->scale,
1000 TRUE, TRUE, TRUE);
1001 }
1002 moved_icons = g_list_prepend (moved_icons, icon);
1003 }
1004 caja_icon_container_select_list_unselect_others
1005 (container, moved_icons);
1006 /* Might have been moved in a way that requires adjusting scroll region. */
1007 caja_icon_container_update_scroll_region (container);
1008 g_list_free (moved_icons);
1009}

1011static void
1012handle_nonlocal_move (CajaIconContainer *container,
1013 GdkDragAction action,
1014 int x, int y,
1015 const char *target_uri,
1016 gboolean icon_hit)
1017{
1018 GList *source_uris, *p;
1019 GArray *source_item_locations;
1020 gboolean free_target_uri, is_rtl;
1021 GtkAllocation allocation;
1022
1023 if (container->details->dnd_info->drag_info.selection_list == NULL)
1024 {
1025 return;
1026 }
1027
1028 source_uris = NULL;
1029 for (p = container->details->dnd_info->drag_info.selection_list; p != NULL; p = p->next)
1030 {
1031 /* do a shallow copy of all the uri strings of the copied files */
1032 source_uris = g_list_prepend (source_uris, ((CajaDragSelectionItem *)p->data)->uri);
1033 }
1034 source_uris = g_list_reverse (source_uris);
1035
1036 is_rtl = caja_icon_container_is_layout_rtl (container);
1037
1038 source_item_locations = g_array_new (FALSE, TRUE, sizeof (GdkPoint));
1039 if (!icon_hit)
1040 {
1041 int index;
1042
1043 /* Drop onto a container. Pass along the item points to allow placing
1044 * the items in their same relative positions in the new container.
1045 */
1046 source_item_locations = g_array_set_size (source_item_locations,
1047 g_list_length (container->details->dnd_info->drag_info.selection_list));
1048
1049 for (index = 0, p = container->details->dnd_info->drag_info.selection_list;
1050 p != NULL; index++, p = p->next)
1051 {
1052 int item_x;
1053
1054 item_x = ((CajaDragSelectionItem *)p->data)->icon_x;
1055 if (is_rtl)
1056 item_x = -item_x - ((CajaDragSelectionItem *)p->data)->icon_width;
1057 g_array_index (source_item_locations, GdkPoint, index).x = item_x;
1058 g_array_index (source_item_locations, GdkPoint, index).y =
1059 ((CajaDragSelectionItem *)p->data)->icon_y;
1060 }
1061 }
1062
1063 free_target_uri = FALSE;
1064 /* Rewrite internal desktop URIs to the normal target uri */
1065 if (eel_uri_is_desktop (target_uri))
1066 {
1067 target_uri = caja_get_desktop_directory_uri ();
1068 free_target_uri = TRUE;
1069 }
1070
1071 if (is_rtl)
1072 {
1073 gtk_widget_get_allocation (GTK_WIDGET (container), &allocation);
1074 x = CANVAS_WIDTH (container, allocation) - x;
1075 }
1076
1077 /* start the copy */
1078 g_signal_emit_by_name (container, "move_copy_items",
1079 source_uris,
1080 source_item_locations,
1081 target_uri,
1082 action,
1083 x, y);
1084
1085 if (free_target_uri)
1086 {
1087 g_free ((char *)target_uri);
1088 }
1089
1090 g_list_free (source_uris);
1091 g_array_free (source_item_locations, TRUE);
1092}

1094static char *
1095caja_icon_container_find_drop_target (CajaIconContainer *container,
1096 GdkDragContext *context,
1097 int x, int y,
1098 gboolean *icon_hit,
1099 gboolean rewrite_desktop)
1100{
1101 CajaIcon *drop_target_icon;
1102 double world_x, world_y;
1103
1104 if (icon_hit)
1105 {
1106 *icon_hit = FALSE;
1107 }
1108
1109 if (!container->details->dnd_info->drag_info.got_drop_data_type)
1110 {
1111 return NULL;
1112 }
1113
1114 canvas_widget_to_world (EEL_CANVAS (container), x, y, &world_x, &world_y);
1115
1116 /* FIXME bugzilla.gnome.org 42485:
1117 * These "can_accept_items" tests need to be done by
1118 * the icon view, not here. This file is not supposed to know
1119 * that the target is a file.
1120 */
1121
1122 /* Find the item we hit with our drop, if any */
1123 drop_target_icon = caja_icon_container_item_at (container, world_x, world_y);
1124 if (drop_target_icon != NULL)
1125 {
1126 char *icon_uri;
1127
1128 icon_uri = caja_icon_container_get_icon_uri (container, drop_target_icon);
1129 if (icon_uri != NULL)
1130 {
1131 CajaFile *file;
1132
1133 file = caja_file_get_by_uri (icon_uri);
1134
1135 if (!caja_drag_can_accept_info (file,
1136 container->details->dnd_info->drag_info.data_type,
1137 container->details->dnd_info->drag_info.selection_list))
1138 {
1139 /* the item we dropped our selection on cannot accept the items,
1140 * do the same thing as if we just dropped the items on the canvas
1141 */
1142 drop_target_icon = NULL;
1143 }
1144
1145 g_free (icon_uri);
1146 caja_file_unref (file);
1147 }
1148 }
1149
1150 if (drop_target_icon == NULL)
1151 {
1152 char *container_uri;
1153
1154 if (icon_hit)
1155 {
1156 *icon_hit = FALSE;
1157 }
1158
1159 container_uri = get_container_uri (container);
1160
1161 if (rewrite_desktop &&
1162 container_uri != NULL &&
1163 eel_uri_is_desktop (container_uri))
1164 {
1165 g_free (container_uri);
1166 container_uri = caja_get_desktop_directory_uri ();
1167 }
1168
1169 return container_uri;
1170 }
1171
1172 if (icon_hit)
1173 {
1174 *icon_hit = TRUE;
1175 }
1176 return caja_icon_container_get_icon_drop_target_uri (container, drop_target_icon);
1177}

1219static void
1220caja_icon_container_receive_dropped_icons (CajaIconContainer *container,
1221 GdkDragContext *context,
1222 int x, int y)
1223{
1224 char *drop_target;
1225 gboolean local_move_only;
1226 double world_x, world_y;
1227 gboolean icon_hit;
1228 GdkDragAction action, real_action;
1229
1230 drop_target = NULL;
1231
1232 if (container->details->dnd_info->drag_info.selection_list == NULL)
1233 {
1234 return;
1235 }
1236
1237 real_action = gdk_drag_context_get_selected_action (context);
1238
1239 if (real_action == GDK_ACTION_ASK)
1240 {
1241 /* FIXME bugzilla.gnome.org 42485: This belongs in FMDirectoryView, not here. */
1242 /* Check for special case items in selection list */
1243 if (caja_drag_selection_includes_special_link (container->details->dnd_info->drag_info.selection_list))
1244 {
1245 /* We only want to move the trash */
1246 action = GDK_ACTION_MOVE;
1247 }
1248 else
1249 {
1250 action = GDK_ACTION_MOVE | GDK_ACTION_COPY | GDK_ACTION_LINK;
1251
1252 if (selection_is_image_file (container->details->dnd_info->drag_info.selection_list))
1253 {
1254 action |= CAJA_DND_ACTION_SET_AS_BACKGROUND;
1255 }
1256 }
1257 real_action = caja_drag_drop_action_ask (GTK_WIDGET (container), action);
1258 }
1259
1260 if (real_action == (GdkDragAction) CAJA_DND_ACTION_SET_AS_BACKGROUND)
1261 {
1262 CajaDragSelectionItem *selected_item;
1263
1264 selected_item = container->details->dnd_info->drag_info.selection_list->data;
1265 eel_background_set_dropped_image (eel_get_widget_background (GTK_WIDGET (container)),
1266 real_action, selected_item->uri);
1267 return;
1268 }
1269
1270 if (real_action > 0)
1271 {
1272 eel_canvas_window_to_world (EEL_CANVAS (container),
1273 x + gtk_adjustment_get_value (gtk_scrollable_get_hadjustment (GTK_SCROLLABLE (container))),
1274 y + gtk_adjustment_get_value (gtk_scrollable_get_vadjustment (GTK_SCROLLABLE (container))),
1275 &world_x, &world_y);
1276
1277 drop_target = caja_icon_container_find_drop_target (container,
1278 context, x, y, &icon_hit, FALSE);
1279
1280 local_move_only = FALSE;
1281 if (!icon_hit && real_action == GDK_ACTION_MOVE)
1282 {
1283 /* we can just move the icon positions if the move ended up in
1284 * the item's parent container
1285 */
1286 local_move_only = caja_icon_container_selection_items_local
1287 (container, container->details->dnd_info->drag_info.selection_list);
1288 }
1289
1290 if (local_move_only)
1291 {
1292 handle_local_move (container, world_x, world_y);
1293 }
1294 else
1295 {
1296 handle_nonlocal_move (container, real_action, world_x, world_y, drop_target, icon_hit);
1297 }
1298 }
1299
1300 g_free (drop_target);
1301 caja_drag_destroy_selection_list (container->details->dnd_info->drag_info.selection_list);
1302 container->details->dnd_info->drag_info.selection_list = NULL;
1303}

1305static void
1306caja_icon_container_get_drop_action (CajaIconContainer *container,
1307 GdkDragContext *context,
1308 int x, int y,
1309 int *action)
1310{
1311 char *drop_target;
1312 gboolean icon_hit;
1313 CajaIcon *icon;
1314 double world_x, world_y;
1315
1316 icon_hit = FALSE;
1317 if (!container->details->dnd_info->drag_info.got_drop_data_type)
1318 {
1319 /* drag_data_received_callback didn't get called yet */
1320 return;
1321 }
1322
1323 /* find out if we're over an icon */
1324 canvas_widget_to_world (EEL_CANVAS (container), x, y, &world_x, &world_y);
1325
1326 icon = caja_icon_container_item_at (container, world_x, world_y);
1327
1328 *action = 0;
1329
1330 /* case out on the type of object being dragged */
1331 switch (container->details->dnd_info->drag_info.data_type)
1332 {

1334 if (container->details->dnd_info->drag_info.selection_list == NULL)
1335 {
1336 return;
1337 }

1339 context, x, y, &icon_hit, FALSE);
1340 if (!drop_target)
1341 {
1342 return;
1343 }
1344 caja_drag_default_drop_action_for_icons (context, drop_target,
1345 container->details->dnd_info->drag_info.selection_list,
1346 action);
1347 g_free (drop_target);
1348 break;
1349 case CAJA_ICON_DND_URI_LIST:
1350 drop_target = caja_icon_container_find_drop_target (container,
1351 context, x, y, &icon_hit, FALSE);
1352 *action = caja_drag_default_drop_action_for_uri_list (context, drop_target);
1353
1354 g_free (drop_target);
1355 break;
1356
1357 /* handle emblems by setting the action if we're over an object */
1358 case CAJA_ICON_DND_KEYWORD:
1359 if (icon != NULL)
1360 {
1361 *action = gdk_drag_context_get_suggested_action (context);
1362 }
1363 break;
1364
1365 case CAJA_ICON_DND_NETSCAPE_URL:
1366 *action = caja_drag_default_drop_action_for_netscape_url (context);
1367 break;
1368
1369 case CAJA_ICON_DND_COLOR:
1370 case CAJA_ICON_DND_BGIMAGE:
1371 case CAJA_ICON_DND_RESET_BACKGROUND:
1372 case CAJA_ICON_DND_ROOTWINDOW_DROP:
1373 *action = gdk_drag_context_get_suggested_action (context);
1374 break;
1375
1376 case CAJA_ICON_DND_TEXT:
1377 case CAJA_ICON_DND_XDNDDIRECTSAVE:
1378 case CAJA_ICON_DND_RAW:
1379 *action = GDK_ACTION_COPY;
1380 break;
1381 }
1382}
