79static gboolean
80gpm_load_get_cpu_values (long unsigned *cpu_idle, long unsigned *cpu_total)
81{
82 long unsigned cpu_user = 0;
83 long unsigned cpu_kernel = 0;
84 long unsigned cpu_wait = 0;
85 kstat_ctl_t *kc = NULL;
86 kstat_named_t *kn = NULL;
87 kstat_t *ks = NULL;
88 cpu_stat_t data;
89 int ncpus;
90 int count;
91
92 kc = kstat_open();
93 if (!kc) {
94 g_warning ("Cannot open kstat!\n");
95 return FALSE;
96 }
97
98 ks = kstat_lookup(kc, "unix", 0, "system_misc");
99 if (kstat_read(kc, ks, NULL) == -1) {
100 g_warning ("Cannot read kstat on module unix!\n");
101 goto out;
102 }
103 kn = kstat_data_lookup (ks, "ncpus");
104 if (!kn) {
105 g_warning ("Cannot get number of cpus in current system!\n");
106 goto out;
107 }
108 ncpus = kn->value.ui32;
109
110 /*
111 * To aggresive ticks used of all cpus,
112 * traverse kstat chain to access very cpu_stat instane.
113 */
114 for(count = 0, *cpu_idle =0, *cpu_total = 0; count < ncpus; count++){
115
116 ks = kstat_lookup(kc, "cpu_stat", count, NULL);
117 if (ks == NULL) {
118 g_warning ("Null output for kstat on cpu%d\n", count);
119 goto out;
120 }
121
122 if (kstat_read(kc, ks, &data) == -1) {
123 g_warning ("Cannot read kstat entry on cpu%d\n", count);
124 goto out;
125 }
126
127 g_debug ("cpu%d:\t%lu\t%lu\t%lu\t%lu\n", count,
128 data.cpu_sysinfo.cpu[CPU_IDLE],
129 data.cpu_sysinfo.cpu[CPU_USER],
130 data.cpu_sysinfo.cpu[CPU_KERNEL],
131 data.cpu_sysinfo.cpu[CPU_WAIT]);
132
133 *cpu_idle += data.cpu_sysinfo.cpu[CPU_IDLE];
134 cpu_user += data.cpu_sysinfo.cpu[CPU_USER];
135 cpu_kernel += data.cpu_sysinfo.cpu[CPU_KERNEL];
136 cpu_wait += data.cpu_sysinfo.cpu[CPU_WAIT];
137 }
138 kstat_close(kc);
139 /*
140 * Summing up all these times gives you the system uptime.
141 * This is what the uptime command does.
142 */
143 *cpu_total = cpu_user + cpu_kernel + cpu_wait + *cpu_idle;
144 return TRUE;
145
146out:
147 kstat_close(kc);
148 return FALSE;
149}

159static gboolean
160gpm_load_get_cpu_values (long unsigned *cpu_idle, long unsigned *cpu_total)
161{
162 long unsigned cpu_user;
163 long unsigned cpu_nice;
164 long unsigned cpu_system;
165 int len;
166 char tmp[5];
167 char str[80];
168 FILE *fd;
169 char *suc;
170 gboolean ret = FALSE;
171
172 /* open file */
173 fd = fopen("/proc/stat", "r");
174 if (!fd)
175 goto out;
176
177 /* get data */
178 suc = fgets (str, 80, fd);
179 if (suc == NULL)
180 goto out;
181
182 /* parse */
183 len = sscanf (str, "%s %lu %lu %lu %lu", tmp,
184 &cpu_user, &cpu_nice, &cpu_system, cpu_idle);
185 if (len != 5)
186 goto out;
187
188 /* summing up all these times gives you the system uptime in jiffies */
189 *cpu_total = cpu_user + cpu_nice + cpu_system + *cpu_idle;
190 ret = TRUE;
191out:
192 if (fd)
193 fclose (fd);
194 return ret;
195}

203gdouble
204gpm_load_get_current (GpmLoad *load)
205{
206 double percentage_load;
207 long unsigned cpu_idle;
208 long unsigned cpu_total;
209 long unsigned diff_idle;
210 long unsigned diff_total;
211 gboolean ret;
212
213 /* work out the differences */
214 ret = gpm_load_get_cpu_values (&cpu_idle, &cpu_total);
215 if (!ret)
216 return 0.0;
217
218 diff_idle = cpu_idle - load->priv->old_idle;
219 diff_total = cpu_total - load->priv->old_total;
220
221 /* If we divide the total time by idle time we get the load. */
222 if (diff_idle > 0)
223 percentage_load = (double) diff_total / (double) diff_idle;
224 else
225 percentage_load = 100;
226
227 load->priv->old_idle = cpu_idle;
228 load->priv->old_total = cpu_total;
229
230 return percentage_load;
231}
