61static gint
62find_selected_layout_idx (GtkBuilder * dialog)
63{
64 GtkTreeSelection *selection =
65 gtk_tree_view_get_selection (GTK_TREE_VIEW
66 (WID ("xkb_layouts_selected")));
67 GtkTreeIter selected_iter;
68 GtkTreeModel *model;
69 GtkTreePath *path;
70 gint *indices;
71 gint rv;
72
73 if (!gtk_tree_selection_get_selected
74 (selection, &model, &selected_iter))
75 return -1;
76
77 path = gtk_tree_model_get_path (model, &selected_iter);
78 if (path == NULL)
79 return -1;
80
81 indices = gtk_tree_path_get_indices (path);
82 rv = indices[0];
83 gtk_tree_path_free (path);
84 return rv;
85}

87GSList *
88xkb_layouts_get_selected_list (void)
89{
90 gchar **array;
91 GSList *retval = NULL;
92 gint i;
93 array = g_settings_get_strv (xkb_kbd_settings, "layouts");
94 if (array != NULL) {
95 for (i = 0; array[i]; i++) {
96 retval = g_slist_append (retval, g_strdup (array[i]));
97 }
98 }
99 g_strfreev (array);
100
101 if (retval == NULL) {
102 if (initial_config.layouts_variants != NULL) {
103 for (i = 0; initial_config.layouts_variants[i] != NULL; i++)
104 retval =
105 g_slist_prepend (retval, g_strdup (initial_config.layouts_variants[i]));
106 }
107 retval = g_slist_reverse (retval);
108 }
109
110 return retval;
111}

126static void
127xkb_layouts_enable_disable_buttons (GtkBuilder * dialog)
128{
129 GtkWidget *add_layout_btn = WID ("xkb_layouts_add");
130 GtkWidget *show_layout_btn = WID ("xkb_layouts_show");
131 GtkWidget *del_layout_btn = WID ("xkb_layouts_remove");
132 GtkWidget *selected_layouts_tree = WID ("xkb_layouts_selected");
133 GtkWidget *move_up_layout_btn = WID ("xkb_layouts_move_up");
134 GtkWidget *move_down_layout_btn = WID ("xkb_layouts_move_down");
135
136 GtkTreeSelection *s_selection =
137 gtk_tree_view_get_selection (GTK_TREE_VIEW
138 (selected_layouts_tree));
139 const int n_selected_selected_layouts =
140 gtk_tree_selection_count_selected_rows (s_selection);
141 GtkTreeModel *selected_layouts_model = gtk_tree_view_get_model
142 (GTK_TREE_VIEW (selected_layouts_tree));
143 const int n_selected_layouts =
144 gtk_tree_model_iter_n_children (selected_layouts_model,
145 NULL);
146 gint sidx = find_selected_layout_idx (dialog);
147
148 if (disable_buttons_sensibility_update)
149 return;
150
151 gtk_widget_set_sensitive (add_layout_btn,
152 (n_selected_layouts <
153 max_selected_layouts
154 || max_selected_layouts == 0));
155 gtk_widget_set_sensitive (del_layout_btn, (n_selected_layouts > 1)
156 && (n_selected_selected_layouts > 0));
157 gtk_widget_set_sensitive (show_layout_btn,
158 (n_selected_selected_layouts > 0));
159 gtk_widget_set_sensitive (move_up_layout_btn, sidx > 0);
160 gtk_widget_set_sensitive (move_down_layout_btn, sidx >= 0
161 && sidx < (n_selected_layouts - 1));
162}

178static void
179xkb_layouts_dnd_data_received (GtkWidget * widget, GdkDragContext * dc,
180 gint x, gint y,
181 GtkSelectionData * selection_data,
182 guint info, guint t, GtkBuilder * dialog)
183{
184 gint sidx = find_selected_layout_idx (dialog);
185 GtkWidget *tree_view = WID ("xkb_layouts_selected");
186 GtkTreePath *path = NULL;
187 GtkTreeViewDropPosition pos;
188 gint didx;
189 gchar *id;
190 GSList *layouts_list;
191 GSList *node2Remove;
192
193 if (sidx == -1)
194 return;
195
196 layouts_list = xkb_layouts_get_selected_list ();
197 node2Remove = g_slist_nth (layouts_list, sidx);
198
199 id = (gchar *) node2Remove->data;
200 layouts_list = g_slist_delete_link (layouts_list, node2Remove);
201
202 if (!gtk_tree_view_get_dest_row_at_pos
203 (GTK_TREE_VIEW (tree_view), x, y, &path, &pos)) {
204 /* Move to the very end */
205 layouts_list =
206 g_slist_append (layouts_list, g_strdup (id));
207 xkb_layouts_set_selected_list (layouts_list);
208 } else if (path != NULL) {
209 gint *indices = gtk_tree_path_get_indices (path);
210 didx = indices[0];
211 gtk_tree_path_free (path);
212 /* Move to the new position */
213 if (sidx != didx) {
214 layouts_list =
215 g_slist_insert (layouts_list, g_strdup (id),
216 didx);
217 xkb_layouts_set_selected_list (layouts_list);
218 }
219 }
220 g_free (id);
221 clear_xkb_elements_list (layouts_list);
222}

224void
225xkb_layouts_prepare_selected_tree (GtkBuilder * dialog)
226{
227 GtkListStore *list_store =
228 gtk_list_store_new (3, G_TYPE_STRING, G_TYPE_STRING,
229 G_TYPE_BOOLEAN);
230 GtkWidget *tree_view = WID ("xkb_layouts_selected");
231 GtkTreeSelection *selection;
232 GtkTargetEntry self_drag_target =
233 { "xkb_layouts_selected", GTK_TARGET_SAME_WIDGET, 0 };
234 GtkTreeViewColumn *desc_column;
235
236 text_renderer = GTK_CELL_RENDERER (gtk_cell_renderer_text_new ());
237
238 desc_column =
239 gtk_tree_view_column_new_with_attributes (_("Layout"),
240 text_renderer,
241 "text",
242 SEL_LAYOUT_TREE_COL_DESCRIPTION,
243 "sensitive",
244 SEL_LAYOUT_TREE_COL_ENABLED,
245 NULL);
246 selection =
247 gtk_tree_view_get_selection (GTK_TREE_VIEW (tree_view));
248
249 gtk_tree_view_set_model (GTK_TREE_VIEW (tree_view),
250 GTK_TREE_MODEL (list_store));
251
252 gtk_tree_view_column_set_sizing (desc_column,
253 GTK_TREE_VIEW_COLUMN_AUTOSIZE);
254 gtk_tree_view_column_set_resizable (desc_column, TRUE);
255 gtk_tree_view_column_set_expand (desc_column, TRUE);
256
257 gtk_tree_view_append_column (GTK_TREE_VIEW (tree_view),
258 desc_column);
259
260 g_signal_connect_swapped (selection, "changed",
261 G_CALLBACK(xkb_layouts_enable_disable_buttons),
262 dialog);
263 max_selected_layouts = xkl_engine_get_max_num_groups (engine);
264
265 /* Setting up DnD */
266 gtk_drag_source_set (tree_view, GDK_BUTTON1_MASK,
267 &self_drag_target, 1, GDK_ACTION_MOVE);
268 gtk_drag_source_set_icon_name (tree_view, "input-keyboard");
269 gtk_drag_dest_set (tree_view, GTK_DEST_DEFAULT_ALL,
270 &self_drag_target, 1, GDK_ACTION_MOVE);
271
272 g_signal_connect (tree_view, "drag_data_get",
273 G_CALLBACK (xkb_layouts_dnd_data_get),
274 dialog);
275 g_signal_connect (tree_view, "drag_data_received",
276 G_CALLBACK (xkb_layouts_dnd_data_received),
277 dialog);
278}

290void
291xkb_layouts_fill_selected_tree (GtkBuilder * dialog)
292{
293 GSList *layouts = xkb_layouts_get_selected_list ();
294 GSList *cur_layout;
295 GtkListStore *list_store =
296 GTK_LIST_STORE (gtk_tree_view_get_model
297 (GTK_TREE_VIEW
298 (WID ("xkb_layouts_selected"))));
299 int counter = 0;
300
301 /* temporarily disable the buttons' status update */
302 disable_buttons_sensibility_update = TRUE;
303
304 gtk_list_store_clear (list_store);
305
306 for (cur_layout = layouts; cur_layout != NULL;
307 cur_layout = cur_layout->next, counter++) {
308 GtkTreeIter iter;
309 const char *visible = (char *) cur_layout->data;
310 gchar *utf_visible = xkb_layout_description_utf8 (visible);
311 gtk_list_store_append (list_store, &iter);
312 gtk_list_store_set (list_store, &iter,
313 SEL_LAYOUT_TREE_COL_DESCRIPTION,
314 utf_visible,
315 SEL_LAYOUT_TREE_COL_ID,
316 cur_layout->data,
317 SEL_LAYOUT_TREE_COL_ENABLED,
318 counter < max_selected_layouts, -1);
319 g_free (utf_visible);
320 }
321
322 clear_xkb_elements_list (layouts);
323
324 /* enable the buttons' status update */
325 disable_buttons_sensibility_update = FALSE;
326
327 if (idx2select != -1) {
328 GtkTreeSelection *selection =
329 gtk_tree_view_get_selection ((GTK_TREE_VIEW
330 (WID
331 ("xkb_layouts_selected"))));
332 GtkTreePath *path =
333 gtk_tree_path_new_from_indices (idx2select, -1);
334 gtk_tree_selection_select_path (selection, path);
335 gtk_tree_path_free (path);
336 idx2select = -1;
337 } else {
338 /* if there is nothing to select - just enable/disable the buttons,
339 otherwise it would be done by the selection change */
340 xkb_layouts_enable_disable_buttons (dialog);
341 }
342}

365static void
366remove_selected_layout (GtkWidget * button, GtkBuilder * dialog)
367{
368 gint idx = find_selected_layout_idx (dialog);
369
370 if (idx != -1) {
371 GSList *layouts_list = xkb_layouts_get_selected_list ();
372 char *id = NULL;
373 GSList *node2Remove = g_slist_nth (layouts_list, idx);
374
375 layouts_list =
376 g_slist_remove_link (layouts_list, node2Remove);
377
378 id = (char *) node2Remove->data;
379 g_slist_free_1 (node2Remove);
380 g_free (id);
381
382 if (default_group > idx)
383 xkb_save_default_group (default_group - 1);
384 else if (default_group == idx)
385 xkb_save_default_group (-1);
386
387 xkb_layouts_set_selected_list (layouts_list);
388 clear_xkb_elements_list (layouts_list);
389 }
