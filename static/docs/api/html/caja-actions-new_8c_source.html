170int
171main( int argc, char** argv )
172{
173 int status = EXIT_SUCCESS;
174 GOptionContext *context;
175 GError *error = NULL;
176 NAObjectAction *action;
177 GSList *msg = NULL;
178 GSList *im;
179 gchar *help;
180 gint errors;
181
182 setlocale( LC_ALL, "" );
183 console_init_log_handler();
184
185 context = init_options();
186
187 if( argc == 1 ){
188 g_set_prgname( argv[0] );
189 help = g_option_context_get_help( context, FALSE, NULL );
190 g_print( "\n%s", help );
191 g_free( help );
192 exit( status );
193 }
194
195 if( !g_option_context_parse( context, &argc, &argv, &error )){
196 g_printerr( _( "Syntax error: %s\n" ), error->message );
197 g_error_free (error);
198 exit_with_usage();
199 }
200
201 if( version ){
202 na_core_utils_print_version();
203 exit( status );
204 }
205
206 errors = 0;
207
208 if( !label || !g_utf8_strlen( label, -1 )){
209 g_printerr( _( "Error: an action label is mandatory.\n" ));
210 errors += 1;
211 }
212
213 if( enabled && disabled ){
214 g_printerr( CANNOT_BOTH, "--enabled", "--disabled" );
215 errors += 1;
216 } else if( !disabled ){
217 enabled = TRUE;
218 }
219
220 if( target_selection && nocontext ){
221 g_printerr( CANNOT_BOTH, "--context", "--nocontext" );
222 errors += 1;
223 } else if( !nocontext ){
224 target_selection = TRUE;
225 }
226
227 if( target_location && nolocation ){
228 g_printerr( CANNOT_BOTH, "--location", "--nolocation" );
229 errors += 1;
230 }
231
232 if( target_toolbar && notoolbar ){
233 g_printerr( CANNOT_BOTH, "--toolbar", "--notoolbar" );
234 errors += 1;
235 }
236
237 if( matchcase && nocase ){
238 g_printerr( CANNOT_BOTH, "--match-case", "--nocase" );
239 errors += 1;
240 } else if( !nocase ){
241 matchcase = TRUE;
242 }
243
244 if( accept_multiple && strlen( selection_count )){
245 g_printerr( CANNOT_BOTH, "--accept-multiple", "--selection-count" );
246 errors += 1;
247 }
248
249 if( onlyshow_array && notshow_array ){
250 g_printerr( CANNOT_BOTH, "--only-show-in", "--not-show-in" );
251 errors += 1;
252 }
253
254 if( output_stdout && output_desktop ){
255 g_printerr( _( "Error: only one output option may be specified.\n" ));
256 errors += 1;
257 }
258
259 if( errors ){
260 exit_with_usage();
261 }
262
263 action = get_action_from_cmdline();
264
265 if( output_desktop ){
266 output_to_desktop( action, &msg );
267 } else {
268 output_to_stdout( action, &msg );
269 }
270
271 if( msg ){
272 for( im = msg ; im ; im = im->next ){
273 g_printerr( "%s\n", ( gchar * ) im->data );
274 }
275 na_core_utils_slist_free( msg );
276 status = EXIT_FAILURE;
277 }
278
279 g_object_unref( action );
280 g_option_context_free( context );
281
282 exit( status );
283}

285static GOptionEntry *
286build_option_entries( const ArgFromDataDef *defs, guint nbdefs, const GOptionEntry *adds, guint nbadds )
287{
288 static const gchar *thisfn = "caja_actions_new_build_option_entries";
289 GOptionEntry *entries;
290 GOptionEntry *ient;
291 const GOptionEntry *iadd;
292 const ArgFromDataDef *idef;
293 const NADataDef *data_def;
294
295 entries = g_new0( GOptionEntry, 1+nbdefs+nbadds );
296 ient = entries;
297
298 idef = defs;
299 while( idef->group ){
300 data_def = na_data_def_get_data_def( idef->group, idef->group_name, idef->data_name );
301
302 if( data_def ){
303 ient->long_name = data_def->option_long;
304 ient->short_name = data_def->option_short;
305 ient->flags = data_def->option_flags;
306 ient->arg = data_def->option_arg;
307 ient->arg_data = idef->arg_data;
308 ient->description = data_def->option_label ? data_def->option_label : data_def->short_label;
309 ient->arg_description = data_def->option_arg_label;
310
311 } else {
312 g_warning( "%s: group=%s, name=%s: unable to find NADataDef structure", thisfn, idef->group_name, idef->data_name );
313 }
314
315 idef++;
316 ient++;
317 }
318
319 iadd = adds;
320 while( iadd->long_name ){
321 memcpy( ient, iadd, sizeof( GOptionEntry ));
322 iadd++;
323 ient++;
324 }
325
326 return( entries );
327}

332static GOptionContext *
333init_options( void )
334{
335 GOptionContext *context;
336 gchar* description;
337 GOptionGroup *output_group;
338 GOptionGroup *misc_group;
339 GOptionEntry *entries;
340
341 context = g_option_context_new( _( "Define a new action." ));
342
343 g_option_context_set_summary( context, _(
344 "The created action defaults to be written to stdout.\n"
345 "It can also be written to an output folder, in a file later suitable for an import in CACT.\n"
346 "Or you may choose to directly write the action into your Caja-Actions configuration." ));
347
348 g_option_context_set_translation_domain( context, GETTEXT_PACKAGE );
349
350 entries = build_option_entries( st_arg_from_data_def, G_N_ELEMENTS( st_arg_from_data_def ), st_added_entries, G_N_ELEMENTS( st_added_entries ) );
351
352#ifdef ENABLE_NLS
353 bindtextdomain( GETTEXT_PACKAGE, MATELOCALEDIR );
354# ifdef HAVE_BIND_TEXTDOMAIN_CODESET
355 bind_textdomain_codeset( GETTEXT_PACKAGE, "UTF-8" );
356# endif
357 textdomain( GETTEXT_PACKAGE );
358 g_option_context_add_main_entries( context, entries, GETTEXT_PACKAGE );
359#else
360 g_option_context_add_main_entries( context, entries, NULL );
361#endif
362
363 g_free( entries );
364
365 description = console_cmdline_get_description();
366 g_option_context_set_description( context, description );
367 g_free( description );
368
369 output_group = g_option_group_new(
370 "output", _( "Output of the program" ), _( "Choose where the program creates the action" ), NULL, NULL );
371 g_option_group_add_entries( output_group, output_entries );
372 g_option_group_set_translation_domain( output_group, GETTEXT_PACKAGE );
373 g_option_context_add_group( context, output_group );
374
375 misc_group = g_option_group_new(
376 "misc", _( "Miscellaneous options" ), _( "Miscellaneous options" ), NULL, NULL );
377 g_option_group_add_entries( misc_group, misc_entries );
378 g_option_group_set_translation_domain( misc_group, GETTEXT_PACKAGE );
379 g_option_context_add_group( context, misc_group );
380
381 return( context );
382}

387static NAObjectAction *
388get_action_from_cmdline( void )
389{
390 NAObjectAction *action;
391 NAObjectProfile *profile;
392 int i;
393 GSList *basenames;
394 GSList *mimetypes;
395 GSList *schemes;
396 GSList *folders;
397 gboolean toolbar_same_label;
398 gchar *msg;
399 GSList *only_show_in;
400 GSList *not_show_in;
401 GSList *capabilities;
402
403 action = na_object_action_new_with_defaults();
404 profile = NA_OBJECT_PROFILE(( GList * ) na_object_get_items( action )->data );
405
406 na_object_set_label( action, label );
407 if( tooltip && g_utf8_strlen( tooltip, -1 )){
408 na_object_set_tooltip( action, tooltip );
409 }
410 if( icon && g_utf8_strlen( icon, -1 )){
411 na_object_set_icon( action, icon );
412 }
413 na_object_set_enabled( action, enabled );
414 na_object_set_target_selection( action, target_selection );
415 na_object_set_target_location( action, target_location );
416 na_object_set_target_toolbar( action, target_toolbar );
417
418 toolbar_same_label = FALSE;
419 if( !label_toolbar || !g_utf8_strlen( label_toolbar, -1 )){
420 label_toolbar = g_strdup( label );
421 toolbar_same_label = TRUE;
422 }
423 na_object_set_toolbar_same_label( action, toolbar_same_label );
424 na_object_set_toolbar_label( action, label_toolbar );
425
426 na_object_set_path( profile, command );
427 na_object_set_parameters( profile, parameters );
428
429 i = 0;
430 basenames = NULL;
431 while( basenames_array != NULL && basenames_array[i] != NULL ){
432 basenames = g_slist_append( basenames, g_strdup( basenames_array[i] ));
433 i++;
434 }
435 if( basenames && g_slist_length( basenames )){
436 na_object_set_basenames( profile, basenames );
437 na_core_utils_slist_free( basenames );
438 }
439
440 na_object_set_matchcase( profile, matchcase );
441
442 mimetypes = NULL;
443 if( isfile ){
444 msg = g_strdup_printf( DEPRECATED, "accept-files", "mimetype" );
445 g_warning( "%s", msg );
446 g_free( msg );
447 }
448 if( isdir ){
449 msg = g_strdup_printf( DEPRECATED, "accept-dirs", "mimetype" );
450 g_warning( "%s", msg );
451 g_free( msg );
452 }
453 if( isfile && !isdir ){
454 mimetypes = g_slist_prepend( mimetypes, g_strdup( "all/allfiles" ));
455 } else if( isdir && !isfile ){
456 mimetypes = g_slist_prepend( mimetypes, g_strdup( "inode/directory" ));
457 }
458 i = 0;
459 while( mimetypes_array != NULL && mimetypes_array[i] != NULL ){
460 mimetypes = g_slist_append( mimetypes, g_strdup( mimetypes_array[i] ));
461 i++;
462 }
463 if( mimetypes && g_slist_length( mimetypes )){
464 na_object_set_mimetypes( profile, mimetypes );
465 na_core_utils_slist_free( mimetypes );
466 }
467
468 if( accept_multiple ){
469 msg = g_strdup_printf( DEPRECATED, "accept-multiple", "selection-count" );
470 g_warning( "%s", msg );
471 g_free( msg );
472 selection_count = g_strdup( ">0" );
473 }
474 if( strlen( selection_count )){
475 na_object_set_selection_count( profile, selection_count );
476 }
477
478 i = 0;
479 schemes = NULL;
480 while( schemes_array != NULL && schemes_array[i] != NULL ){
481 schemes = g_slist_append( schemes, g_strdup( schemes_array[i] ));
482 i++;
483 }
484 if( schemes && g_slist_length( schemes )){
485 na_object_set_schemes( profile, schemes );
486 na_core_utils_slist_free( schemes );
487 }
488
489 i = 0;
490 folders = NULL;
491 while( folders_array != NULL && folders_array[i] != NULL ){
492 folders = g_slist_append( folders, g_strdup( folders_array[i] ));
493 i++;
494 }
495 if( folders && g_slist_length( folders )){
496 na_object_set_folders( profile, folders );
497 na_core_utils_slist_free( folders );
498 }
499
500 if( onlyshow_array ){
501 only_show_in = NULL;
502 for( i = 0 ; onlyshow_array[i] && strlen( onlyshow_array[i] ) ; ++i ){
503 only_show_in = g_slist_append( only_show_in, g_strdup( onlyshow_array[i] ));
504 }
505 if( only_show_in && g_slist_length( only_show_in )){
506 na_object_set_only_show_in( profile, only_show_in );
507 na_core_utils_slist_free( only_show_in );
508 }
509 }
510
511 if( notshow_array ){
512 not_show_in = NULL;
513 for( i = 0 ; notshow_array[i] && strlen( notshow_array[i] ) ; ++i ){
514 not_show_in = g_slist_append( not_show_in, g_strdup( notshow_array[i] ));
515 }
516 if( not_show_in && g_slist_length( not_show_in )){
517 na_object_set_not_show_in( profile, not_show_in );
518 na_core_utils_slist_free( not_show_in );
519 }
520 }
521
522 if( try_exec && strlen( try_exec )){
523 na_object_set_try_exec( profile, try_exec );
524 }
525
526 if( show_registered && strlen( show_registered )){
527 na_object_set_show_if_registered( profile, show_registered );
528 }
529
530 if( show_true && strlen( show_true )){
531 na_object_set_show_if_true( profile, show_true );
532 }
533
534 if( show_running && strlen( show_running )){
535 na_object_set_show_if_running( profile, show_running );
536 }
537
538 if( capability_array ){
539 capabilities = NULL;
540 for( i = 0 ; capability_array[i] && strlen( capability_array[i] ) ; ++i ){
541 const gchar *cap = ( const gchar * ) capability_array[i];
542 /* 'Owner', 'Readable', 'Writable', 'Executable' and 'Local' */
543 if( strcmp( cap, "Owner" ) &&
544 strcmp( cap, "Readable" ) &&
545 strcmp( cap, "Writable" ) &&
546 strcmp( cap, "Executable" ) &&
547 strcmp( cap, "Local" )){
548 g_warning( "%s: unknown capability", cap );
549 } else {
550 capabilities = g_slist_append( capabilities, g_strdup( capability_array[i] ));
551 }
552 }
553 if( capabilities && g_slist_length( capabilities )){
554 na_object_set_capabilities( profile, capabilities );
555 na_core_utils_slist_free( capabilities );
556 }
557 }
558
559 return( action );
560}

562/*
563 * write the new item as a .desktop file
564 */
565static gboolean
566output_to_desktop( NAObjectAction *action, GSList **msgs )
567{
568 NAUpdater *updater;
569 NAIOProvider *provider;
570 guint ret;
571 gboolean code;
572
573 updater = na_updater_new();
574 provider = na_io_provider_find_io_provider_by_id( NA_PIVOT( updater ), "na-desktop" );
575
576 if( provider ){
577 na_object_set_provider( action, provider );
578 ret = na_updater_write_item( updater, NA_OBJECT_ITEM( action ), msgs );
579 code = ( ret == NA_IIO_PROVIDER_CODE_OK );
580
581 } else {
582 /* Translators: 'na-desktop' is a plugin identifier - do not translate */
583 *msgs = g_slist_append( *msgs, _( "Error: unable to find 'na-desktop' i/o provider." ));
584 code = FALSE;
585 }
586
587 g_object_unref( updater );
588
589 return( code );
590}

592static gboolean
593output_to_stdout( const NAObjectAction *action, GSList **msgs )
594{
595 gboolean ret;
596 NAUpdater *updater;
597 gchar *buffer;
598
599 updater = na_updater_new();
600 buffer = na_exporter_to_buffer( NA_PIVOT( updater ), NA_OBJECT_ITEM( action ), "Desktop1", msgs );
601 ret = ( buffer != NULL );
602
603 if( buffer ){
604 g_printf( "%s", buffer );
605 g_free( buffer );
606 }
607
608 g_object_unref( updater );
609
610 return( ret );
611}
