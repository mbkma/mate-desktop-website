93/* colorshift a surface */
94static void
95stickynotes_make_prelight_icon (cairo_surface_t *dest,
96 cairo_surface_t *src,
97 int shift)
98{
99 gint i, j;
100 gint width, height, has_alpha, srcrowstride, destrowstride;
101 guchar *target_pixels;
102 guchar *original_pixels;
103 guchar *pixsrc;
104 guchar *pixdest;
105 int val;
106 guchar r,g,b;
107
108 has_alpha = cairo_surface_get_content (src) != CAIRO_CONTENT_COLOR;
109 width = cairo_image_surface_get_width (src);
110 height = cairo_image_surface_get_height (src);
111 srcrowstride = cairo_image_surface_get_stride (src);
112 destrowstride = cairo_image_surface_get_stride (dest);
113 original_pixels = cairo_image_surface_get_data (src);
114 target_pixels = cairo_image_surface_get_data (dest);
115
116 for (i = 0; i < height; i++) {
117 pixdest = target_pixels + i*destrowstride;
118 pixsrc = original_pixels + i*srcrowstride;
119 for (j = 0; j < width; j++) {
120 r = *(pixsrc++);
121 g = *(pixsrc++);
122 b = *(pixsrc++);
123 val = r + shift;
124 *(pixdest++) = CLAMP (val, 0, 255);
125 val = g + shift;
126 *(pixdest++) = CLAMP (val, 0, 255);
127 val = b + shift;
128 *(pixdest++) = CLAMP (val, 0, 255);
129 if (has_alpha)
130 *(pixdest++) = *(pixsrc++);
131 }
132 }
133}

163/* Create and initalize global sticky notes instance */
164void
165stickynotes_applet_init (MatePanelApplet *mate_panel_applet)
166{
167 cairo_t *cr;
168 gint size, scale;
169 int screen_height;
170 GdkScreen *screen;
171
172 stickynotes = g_new (StickyNotes, 1);
173
174 stickynotes->notes = NULL;
175 stickynotes->applets = NULL;
176 stickynotes->last_timeout_data = 0;
177
178 size = mate_panel_applet_get_size (mate_panel_applet);
179 scale = gtk_widget_get_scale_factor (GTK_WIDGET (mate_panel_applet));
180
181#ifndef ENABLE_IN_PROCESS
182 g_set_application_name (_("Sticky Notes"));
183#endif
184 gtk_window_set_default_icon_name ("mate-sticky-notes-applet");
185
186 stickynotes->icon_normal =
187 gtk_icon_theme_load_surface (gtk_icon_theme_get_default (),
188 "mate-sticky-notes-applet",
189 size, scale, NULL, 0, NULL);
190
191 stickynotes->icon_prelight =
192 cairo_surface_create_similar (stickynotes->icon_normal,
193 cairo_surface_get_content (stickynotes->icon_normal),
194 cairo_image_surface_get_width (stickynotes->icon_normal),
195 cairo_image_surface_get_height (stickynotes->icon_normal));
196
197 stickynotes_make_prelight_icon (stickynotes->icon_prelight,
198 stickynotes->icon_normal, 30);
199
200 cr = cairo_create (stickynotes->icon_prelight);
201 cairo_set_operator (cr, CAIRO_OPERATOR_DEST_IN);
202 cairo_mask_surface (cr, stickynotes->icon_normal, 0, 0);
203
204 stickynotes->settings = g_settings_new (STICKYNOTES_SCHEMA);
205 stickynotes->visible = TRUE;
206
207 stickynotes_applet_init_icons ();
208 stickynotes_applet_init_prefs ();
209
210 /* Watch GSettings values */
211 g_signal_connect (stickynotes->settings, "changed",
212 G_CALLBACK (preferences_apply_cb), NULL);
213
214 /* Max height for large notes in x11 only
215 * In wayland we do not have access to global screen dimensions
216 */
217
218 screen = gtk_widget_get_screen (GTK_WIDGET (mate_panel_applet));
219#ifdef GDK_WINDOWING_X11
220 if (GDK_IS_X11_DISPLAY (gdk_screen_get_display (screen)))
221 screen_height = HeightOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale;
222
223 else
224 {
225 /*No root window or primary monitor in wayland unless compositors add it back*/
226 GdkRectangle geometry = {0};
227 GdkMonitor *monitor;
228 monitor = gdk_display_get_monitor (gdk_screen_get_display (screen), 0);
229 gdk_monitor_get_geometry (monitor, &geometry);
230 screen_height = geometry.height;
231 }
232#endif
233
234 stickynotes->max_height = (int) (0.8 * (double) screen_height);
235
236 /* Load sticky notes */
237 stickynotes_load (screen);
238
239 install_check_click_on_desktop (screen);
240}

374/* Create a Sticky Notes applet */
375StickyNotesApplet *
376stickynotes_applet_new (MatePanelApplet *mate_panel_applet)
377{
378 AtkObject *atk_obj;
379
380 /* Create Sticky Notes Applet */
381 StickyNotesApplet *applet = g_new (StickyNotesApplet, 1);
382
383 /* Initialize Sticky Notes Applet */
384 applet->w_applet = GTK_WIDGET (mate_panel_applet);
385 applet->w_image = gtk_image_new ();
386 applet->destroy_all_dialog = NULL;
387 applet->prelighted = FALSE;
388
389 applet->menu_tip = NULL;
390
391 /* Expand the applet for Fitts' law complience. */
392 mate_panel_applet_set_flags (mate_panel_applet,
393 MATE_PANEL_APPLET_EXPAND_MINOR);
394
395 /* Add the applet icon */
396 gtk_container_add (GTK_CONTAINER (mate_panel_applet),
397 applet->w_image);
398 applet->panel_size = mate_panel_applet_get_size (mate_panel_applet);
399 applet->panel_orient = mate_panel_applet_get_orient (mate_panel_applet);
400 stickynotes_applet_update_icon (applet);
401
402 /* Add the popup menu */
403 applet->action_group = gtk_action_group_new ("StickyNotes Applet Actions");
404 gtk_action_group_set_translation_domain (applet->action_group,
405 GETTEXT_PACKAGE);
406 gtk_action_group_add_actions (applet->action_group,
407 stickynotes_applet_menu_actions,
408 G_N_ELEMENTS (stickynotes_applet_menu_actions),
409 applet);
410 gtk_action_group_add_toggle_actions (applet->action_group,
411 stickynotes_applet_menu_toggle_actions,
412 G_N_ELEMENTS (stickynotes_applet_menu_toggle_actions),
413 applet);
414 mate_panel_applet_setup_menu_from_resource (mate_panel_applet,
415 GRESOURCE "menu.xml",
416 applet->action_group);
417
418 if (mate_panel_applet_get_locked_down (mate_panel_applet)) {
419 GtkAction *action;
420
421 action = gtk_action_group_get_action (applet->action_group,
422 "preferences");
423 gtk_action_set_visible (action, FALSE);
424 }
425
426 /* Connect all signals for applet management */
427 g_signal_connect (applet->w_applet, "button-press-event",
428 G_CALLBACK (applet_button_cb), applet);
429 g_signal_connect (applet->w_applet, "key-press-event",
430 G_CALLBACK (applet_key_cb), applet);
431 g_signal_connect (applet->w_applet, "focus-in-event",
432 G_CALLBACK (applet_focus_cb), applet);
433 g_signal_connect (applet->w_applet, "focus-out-event",
434 G_CALLBACK (applet_focus_cb), applet);
435 g_signal_connect (applet->w_applet, "enter-notify-event",
436 G_CALLBACK (applet_cross_cb), applet);
437 g_signal_connect (applet->w_applet, "leave-notify-event",
438 G_CALLBACK (applet_cross_cb), applet);
439 g_signal_connect (applet->w_applet, "size-allocate",
440 G_CALLBACK (applet_size_allocate_cb), applet);
441 g_signal_connect (applet->w_applet, "change-orient",
442 G_CALLBACK (applet_change_orient_cb), applet);
443 g_signal_connect (applet->w_applet, "destroy",
444 G_CALLBACK (applet_destroy_cb), applet);
445 g_signal_connect (applet->w_applet, "destroy",
446 G_CALLBACK (stickynotes_destroy), NULL);
447
448 atk_obj = gtk_widget_get_accessible (applet->w_applet);
449 atk_object_set_name (atk_obj, _("Sticky Notes"));
450
451 /* Show the applet */
452 gtk_widget_show_all (applet->w_applet);
453
454 return applet;
455}

457void
458stickynotes_applet_update_icon (StickyNotesApplet *applet)
459{
460 cairo_t *cr;
461 cairo_surface_t *surface;
462
463 gint size = applet->panel_size;
464
465 if (size > 3)
466 size = size - 3;
467
468 /* Choose appropriate icon and size it */
469 if (applet->prelighted)
470 surface = cairo_surface_create_similar (stickynotes->icon_prelight,
471 cairo_surface_get_content (stickynotes->icon_prelight),
472 size, size);
473 else
474 surface = cairo_surface_create_similar (stickynotes->icon_normal,
475 cairo_surface_get_content (stickynotes->icon_normal),
476 size, size);
477
478 cr = cairo_create (surface);
479 cairo_set_source_surface (cr, applet->prelighted ? stickynotes->icon_prelight : stickynotes->icon_normal, 0, 0);
480 cairo_paint (cr);
481
482 /* Apply the finished surface to the applet image */
483 gtk_image_set_from_surface (GTK_IMAGE (applet->w_image), surface);
484
485 cairo_surface_destroy (surface);
486}

488void
489stickynotes_applet_update_prefs (void)
490{
491 gint width, height;
492 gboolean sys_color, sys_font, sticky, force_default, desktop_hide;
493 char *font_str;
494 char *color_str, *font_color_str;
495 GdkRGBA color, font_color;
496
497 width = g_settings_get_int (stickynotes->settings,
498 "default-width");
499 width = MAX (width, 1);
500 height = g_settings_get_int (stickynotes->settings,
501 "default-height");
502 height = MAX (height, 1);
503
504 sys_color = g_settings_get_boolean (stickynotes->settings,
505 "use-system-color");
506 sys_font = g_settings_get_boolean (stickynotes->settings,
507 "use-system-font");
508 sticky = g_settings_get_boolean (stickynotes->settings,
509 "sticky");
510 force_default = g_settings_get_boolean (stickynotes->settings,
511 "force-default");
512 desktop_hide = g_settings_get_boolean (stickynotes->settings,
513 "desktop-hide");
514
515 font_str = g_settings_get_string (stickynotes->settings,
516 "default-font");
517 if (!font_str)
518 font_str = g_strdup ("Sans 10");
519
520 color_str = g_settings_get_string (stickynotes->settings,
521 "default-color");
522 if (!color_str)
523 color_str = g_strdup ("#ECF833");
524
525 font_color_str = g_settings_get_string (stickynotes->settings,
526 "default-font-color");
527 if (!font_color_str)
528 font_color_str = g_strdup ("#000000");
529
530 gdk_rgba_parse (&color, color_str);
531 gdk_rgba_parse (&font_color, font_color_str);
532
533 g_free (color_str);
534 g_free (font_color_str);
535
536 gtk_adjustment_set_value (stickynotes->w_prefs_width, width);
537 gtk_adjustment_set_value (stickynotes->w_prefs_height, height);
538 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (stickynotes->w_prefs_sys_color),
539 sys_color);
540 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (stickynotes->w_prefs_sys_font),
541 sys_font);
542 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (stickynotes->w_prefs_sticky),
543 sticky);
544 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (stickynotes->w_prefs_force),
545 force_default);
546 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (stickynotes->w_prefs_desktop),
547 desktop_hide);
548
549 gtk_color_chooser_set_rgba (GTK_COLOR_CHOOSER (stickynotes->w_prefs_color),
550 &color);
551 gtk_color_chooser_set_rgba (GTK_COLOR_CHOOSER (stickynotes->w_prefs_font_color),
552 &font_color);
553
554 gtk_font_button_set_font_name (GTK_FONT_BUTTON (stickynotes->w_prefs_font),
555 font_str);
556 g_free (font_str);
557
558 if (g_settings_is_writable (stickynotes->settings, "default-color")) {
559 gtk_widget_set_sensitive (GTK_WIDGET (gtk_builder_get_object (stickynotes->builder,
560 "prefs_color_label")),
561 !sys_color);
562 gtk_widget_set_sensitive (stickynotes->w_prefs_color, !sys_color);
563 }
564 if (g_settings_is_writable (stickynotes->settings, "default-font-color")) {
565 gtk_widget_set_sensitive (GTK_WIDGET (gtk_builder_get_object (stickynotes->builder,
566 "prefs_font_color_label")),
567 !sys_color);
568 gtk_widget_set_sensitive (stickynotes->w_prefs_font_color, !sys_color);
569 }
570 if (g_settings_is_writable (stickynotes->settings, "default-font")) {
571 gtk_widget_set_sensitive (GTK_WIDGET (gtk_builder_get_object (stickynotes->builder,
572 "prefs_font_label")),
573 !sys_font);
574 gtk_widget_set_sensitive (stickynotes->w_prefs_font, !sys_font);
575 }
576}

578void
579stickynotes_applet_update_menus (void)
580{
581 GList *l;
582 gboolean inconsistent = FALSE;
583
