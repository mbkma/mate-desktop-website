173void
174cact_tree_ieditable_initialize( CactTreeIEditable *instance, GtkTreeView *treeview, BaseWindow *window )
175{
176 static const gchar *thisfn = "cact_tree_ieditable_initialize";
177 IEditableData *ied;
178 CactApplication *application;
179 GtkTreeViewColumn *column;
180 GList *renderers;
181
182 g_return_if_fail( CACT_IS_TREE_IEDITABLE( instance ));
183
184 g_debug( "%s: instance=%p (%s), treeview=%p, window=%p",
185 thisfn,
186 ( void * ) instance, G_OBJECT_TYPE_NAME( instance ),
187 ( void * ) treeview, ( void * ) window );
188
189 ied = get_instance_data( instance );
190 ied->window = window;
191 ied->treeview = treeview;
192 ied->model = CACT_TREE_MODEL( gtk_tree_view_get_model( treeview ));
193
194 application = CACT_APPLICATION( base_window_get_application( window ));
195 ied->updater = cact_application_get_updater( application );
196
197 /* inline label edition with F2 */
198 base_window_signal_connect(
199 window,
200 G_OBJECT( treeview ),
201 "key-press-event",
202 G_CALLBACK( on_key_pressed_event ));
203
204 /* label edition: inform the corresponding tab */
205 column = gtk_tree_view_get_column( treeview, TREE_COLUMN_LABEL );
206 renderers = gtk_cell_layout_get_cells( GTK_CELL_LAYOUT( column ));
207 base_window_signal_connect(
208 window,
209 G_OBJECT( renderers->data ),
210 "edited",
211 G_CALLBACK( on_label_edited ));
212
213 /* monitors status changes to refresh the display */
214 na_iduplicable_register_consumer( G_OBJECT( instance ));
215 ied->modified_handler_id = base_window_signal_connect(
216 window,
217 G_OBJECT( instance ),
218 IDUPLICABLE_SIGNAL_MODIFIED_CHANGED,
219 G_CALLBACK( on_object_modified_status_changed ));
220 ied->valid_handler_id = base_window_signal_connect(
221 window,
222 G_OBJECT( instance ),
223 IDUPLICABLE_SIGNAL_VALID_CHANGED,
224 G_CALLBACK( on_object_valid_status_changed ));
225
226 /* monitors the reloading of the tree */
227 base_window_signal_connect(
228 window,
229 G_OBJECT( window ),
230 TREE_SIGNAL_MODIFIED_STATUS_CHANGED,
231 G_CALLBACK( on_tree_view_modified_status_changed ));
232
233 /* monitors the level zero of the tree */
234 base_window_signal_connect(
235 window,
236 G_OBJECT( window ),
237 TREE_SIGNAL_LEVEL_ZERO_CHANGED,
238 G_CALLBACK( on_tree_view_level_zero_changed ));
239
240 /* monitors the main selection */
241 base_window_signal_connect(
242 window,
243 G_OBJECT( window ),
244 MAIN_SIGNAL_SELECTION_CHANGED,
245 G_CALLBACK( on_main_selection_changed ));
246}

335/*
336 * modification status of an edited object has changed
337 * - refresh the display
338 */
339static void
340on_object_modified_status_changed( CactTreeIEditable *instance, NAObject *object, gboolean is_modified, BaseWindow *window )
341{
342 static const gchar *thisfn = "cact_tree_ieditable_on_object_modified_status_changed";
343 IEditableData *ied;
344 gboolean prev_status, status;
345
346 g_debug( "%s: instance=%p, object=%p (%s), is_modified=%s, window=%p",
347 thisfn, ( void * ) instance,
348 ( void * ) object, G_OBJECT_TYPE_NAME( object ), is_modified ? "True":"False",
349 ( void * ) window );
350
351 ied = ( IEditableData * ) g_object_get_data( G_OBJECT( instance ), VIEW_DATA_IEDITABLE );
352 prev_status = get_modification_status( ied );
353 gtk_tree_model_filter_refilter( GTK_TREE_MODEL_FILTER( ied->model ));
354
355 if( NA_IS_OBJECT_ITEM( object )){
356 if( is_modified ){
357 ied->count_modified += 1;
358 } else {
359 ied->count_modified -= 1;
360 }
361 }
362
363 if( cact_tree_view_are_notify_allowed( CACT_TREE_VIEW( instance ))){
364 status = get_modification_status( ied );
365 if( status != prev_status ){
366 g_signal_emit_by_name( window, TREE_SIGNAL_MODIFIED_STATUS_CHANGED, status );
367 }
368 }
369}

450void
451cact_tree_ieditable_delete( CactTreeIEditable *instance, GList *items, TreeIEditableDeleteOpe ope )
452{
453 static const gchar *thisfn = "cact_tree_ieditable_delete";
454 IEditableData *ied;
455 gboolean prev_status, status;
456 GtkTreePath *path;
457 GList *it;
458 NAObjectItem *parent;
459
460 g_return_if_fail( CACT_IS_TREE_IEDITABLE( instance ));
461
462 g_debug( "%s: instance=%p, items=%p (count=%d), ope=%u",
463 thisfn, ( void * ) instance, ( void * ) items, g_list_length( items ), ope );
464
465 cact_tree_view_set_notify_allowed( CACT_TREE_VIEW( instance ), FALSE );
466 ied = get_instance_data( instance );
467 prev_status = get_modification_status( ied );
468
469 decrement_counters( instance, ied, items );
470 path = NULL;
471
472 for( it = items ; it ; it = it->next ){
473 if( path ){
474 gtk_tree_path_free( path );
475 }
476
477 /* check the status of the parent of the deleted object
478 * so potentially update the modification status
479 */
480 parent = na_object_get_parent( it->data );
481 path = cact_tree_model_delete( ied->model, NA_OBJECT( it->data ));
482 if( parent ){
483 na_object_check_status( parent );
484 } else {
485 check_level_zero_status( instance );
486 }
487
488 /* record the deleted item in the 'deleted' list,
489 * incrementing the 'modified' count if the deleted item was not yet modified
490 */
491 if( ope == TREE_OPE_DELETE ){
492 add_to_deleted_rec( ied, NA_OBJECT( it->data ));
493 }
494
495 g_debug( "%s: object=%p (%s, ref_count=%d)", thisfn,
496 ( void * ) it->data, G_OBJECT_TYPE_NAME( it->data ), G_OBJECT( it->data )->ref_count );
497 }
498
499 gtk_tree_model_filter_refilter( GTK_TREE_MODEL_FILTER( ied->model ));
500
501 cact_tree_view_set_notify_allowed( CACT_TREE_VIEW( instance ), TRUE );
502
503 if( path ){
504 if( ope == TREE_OPE_DELETE ){
505 cact_tree_view_select_row_at_path( CACT_TREE_VIEW( instance ), path );
506 }
507 gtk_tree_path_free( path );
508 }
509
510 status = get_modification_status( ied );
511 if( status != prev_status ){
512 g_signal_emit_by_name( G_OBJECT( ied->window ), TREE_SIGNAL_MODIFIED_STATUS_CHANGED, status );
513
514 }
515}

529static void
530add_to_deleted_rec( IEditableData *ied, NAObject *object )
531{
532 GList *subitems, *it;
533
534 if( NA_IS_OBJECT_ITEM( object )){
535
536 if( !g_list_find( ied->deleted, object )){
537 ied->deleted = g_list_prepend( ied->deleted, object );
538
539 if( na_object_is_modified( object )){
540 ied->count_modified -= 1;
541 }
542
543 if( na_object_get_provider( object )){
544 ied->count_deleted += 1;
545 }
546 }
547
548 if( NA_IS_OBJECT_MENU( object )){
549 subitems = na_object_get_items( object );
550 for( it = subitems ; it ; it = it->next ){
551 add_to_deleted_rec( ied, it->data );
552 }
553 }
554 }
555}

591gboolean
592cact_tree_ieditable_remove_deleted( CactTreeIEditable *instance, GSList **messages )
593{
594 static const gchar *thisfn = "cact_tree_ieditable_remove_deleted";
595 gboolean delete_ok;
596 IEditableData *ied;
597 GList *it;
598 NAObjectItem *item;
599 GList *not_deleted;
600
601 g_return_val_if_fail( CACT_IS_TREE_IEDITABLE( instance ), TRUE );
602
603 delete_ok = TRUE;
604 ied = get_instance_data( instance );
605 not_deleted = NULL;
606
607 for( it = ied->deleted ; it ; it = it->next ){
608 item = NA_OBJECT_ITEM( it->data );
609 g_debug( "%s: item=%p (%s)", thisfn, ( void * ) item, G_OBJECT_TYPE_NAME( item ));
610 na_object_dump_norec( item );
611
612 if( na_updater_delete_item( ied->updater, item, messages ) != NA_IIO_PROVIDER_CODE_OK ){
613 not_deleted = g_list_prepend( not_deleted, na_object_ref( item ));
614 delete_ok = FALSE;
615 }
616 }
617
618 ied->deleted = na_object_free_items( ied->deleted );
619
620 /* items that we cannot delete are reinserted in the tree view
621 * in the state they were when they were deleted
622 * (i.e. possibly modified)
623 */
624 if( not_deleted ){
625 cact_tree_ieditable_insert_items( instance, not_deleted, NULL );
626 na_object_free_items( not_deleted );
627 }
628
629 return( delete_ok );
630}

676void
677cact_tree_ieditable_insert_items( CactTreeIEditable *instance, GList *items, NAObject *sibling )
678{
679 static const gchar *thisfn = "cact_tree_ieditable_insert_items";
680 IEditableData *ied;
681 GtkTreePath *insert_path;
682 NAObject *object, *parent;
683
684 g_return_if_fail( CACT_IS_TREE_IEDITABLE( instance ));
685 g_return_if_fail( items );
686
687 g_debug( "%s: instance=%p, items=%p (count=%d), sibling=%p",
688 thisfn, ( void * ) instance, ( void * ) items, g_list_length( items ), ( void * ) sibling );
689
690 insert_path = NULL;
691 ied = get_instance_data( instance );
692
693 if( sibling ){
694 insert_path = cact_tree_model_object_to_path( ied->model, sibling );
695
696 } else {
697 insert_path = get_selection_first_path( ied->treeview );
698 object = cact_tree_model_object_at_path( ied->model, insert_path );
699 g_debug( "%s: current object at insertion path is %p", thisfn, ( void * ) object );
700
701 /* as a particular case, insert a new profile _into_ current action
702 */
703 if( NA_IS_OBJECT_ACTION( object ) && NA_IS_OBJECT_PROFILE( items->data )){
704 cact_tree_ieditable_insert_into( instance, items );
705 gtk_tree_path_free( insert_path );
706 return;
707 }
708
709 /* insert a new NAObjectItem before current NAObjectItem
710 */
711 if( NA_IS_OBJECT_PROFILE( object ) && NA_IS_OBJECT_ITEM( items->data )){
712 parent = ( NAObject * ) na_object_get_parent( object );
713 gtk_tree_path_free( insert_path );
714 insert_path = cact_tree_model_object_to_path( ied->model, parent );
715 }
716 }
717
718 cact_tree_ieditable_insert_at_path( instance, items, insert_path );
719 gtk_tree_path_free( insert_path );
720}

828/*
829 * inserting a list of objects at path is:
830 * - for each item starting from the end, do
831 * > insert the item at path
832 * > recursively insert item children into the insertion path
833 *
834 * Returns the actual insertion path suitable for the next selection.
835 * This path should be gtk_tree_path_free() by the caller.
836 */
837static GtkTreePath *
838do_insert_before( IEditableData *ied, GList *items, GtkTreePath *asked_path )
839{
840 static const gchar *thisfn = "cact_tree_ieditable_do_insert_before";
841 GList *reversed;
842 GList *it;
843 GtkTreePath *path;
844 GtkTreePath *actual_path;
845
846 reversed = g_list_reverse( items );
847 actual_path = NULL;
848
849 for( it = reversed ; it ; it = it->next ){
850 if( actual_path ){
851 path = gtk_tree_path_copy( actual_path );
852 gtk_tree_path_free( actual_path );
853 } else {
854 path = gtk_tree_path_copy( asked_path );
855 }
856 actual_path = cact_tree_model_insert_before( ied->model, NA_OBJECT( it->data ), path );
857 gtk_tree_path_free( path );
858 g_debug( "%s: object=%p (%s, ref_count=%d)", thisfn,
859 ( void * ) it->data, G_OBJECT_TYPE_NAME( it->data ), G_OBJECT( it->data )->ref_count );
860
861 /* recursively insert subitems
862 */
863 if( NA_IS_OBJECT_ITEM( it->data )){
864 path = do_insert_into( ied, na_object_get_items( it->data ), actual_path );
865 gtk_tree_path_free( path );
866 }
867 }
868
869 /* an l-value here is useless, but makes gcc happy */
870 items = g_list_reverse( reversed );
871
872 return( actual_path );
873}

875/*
876 * inserting a list of objects into an object is as:
877 * - inserting the last child into the object
878 * - then inserting other children just before the last insertion path
879 *
880 * Returns the actual insertion path suitable for the next selection.
881 * This path should be gtk_tree_path_free() by the caller.
882 */
883static GtkTreePath *
884do_insert_into( IEditableData *ied, GList *items, GtkTreePath *asked_path )
885{
886 GList *copy;
887 GList *last;
888 GtkTreePath *actual_path;
889 GtkTreePath *insertion_path;
890
891 actual_path = NULL;
892
893 if( items ){
894 copy = g_list_copy( items );
895 last = g_list_last( copy );
896 copy = g_list_remove_link( copy, last );
897
898 /* insert the last child into the wished path
899 * and recursively all its children
900 */
901 actual_path = cact_tree_model_insert_into( ied->model, NA_OBJECT( last->data ), asked_path );
902 gtk_tree_view_expand_to_path( ied->treeview, actual_path );
903
904 if( NA_IS_OBJECT_ITEM( last->data )){
905 insertion_path = do_insert_into( ied, na_object_get_items( last->data ), actual_path );
906 gtk_tree_path_free( insertion_path );
907 }
908
909 /* then insert other objects
910 */
911 if( copy ){
912 insertion_path = do_insert_before( ied, copy, actual_path );
913 gtk_tree_path_free( actual_path );
914 actual_path = insertion_path;
915 g_list_free( copy );
916 }
917 }
918
919 return( actual_path );
920}

938/*
939 * cact_tree_ieditable_set_items:
940 * @instance: this #CactTreeIEditable *instance.
941 * @items: a #GList of items to be set in the tree view.
942 *
943 * The provided list of @items will override the items already in the list
944 * which have the same identifier. But:
945 * - menus keep their children
946 * - entering actions fully override existing ones.
947 *
948 * Items which would not been found are just ignored.
949 * Items which are not of the same type are ignored.
950 *
951 * This feature is typically used when importing an item whose identifier
952 * already exists, and the user have chosen to override existing item.
953 * So we should be almost sure that each item actually exists in the view.
954 */
955void
956cact_tree_ieditable_set_items( CactTreeIEditable *instance, GList *items )
957{
958 static const gchar *thisfn = "cact_tree_ieditable_set_items";
959 IEditableData *ied;
960 GList *it;
961 NAObjectItem *new_item, *old_item;
962 gchar *id;
963 GtkTreePath *path, *insert_path;
964
965 g_return_if_fail( CACT_IS_TREE_IEDITABLE( instance ));
966
967 g_debug( "%s: instance=%p, items=%p (count=%d)",
968 thisfn, ( void * ) instance, ( void * ) items, g_list_length( items ));
969
970 ied = get_instance_data( instance );
971
972 for( it = items ; it ; it = it->next ){
973 new_item = NA_OBJECT_ITEM( it->data );
974
975 id = na_object_get_id( new_item );
976 old_item = cact_tree_view_get_item_by_id( CACT_TREE_VIEW( instance ), id );
977
978 if( !old_item ){
979 g_warning( "%s: id=%s: item not found - ignored", thisfn, id );
980
981 } else if( G_OBJECT_TYPE( old_item ) != G_OBJECT_TYPE( new_item )){
982 g_warning( "%s: id=%s: old is a %s while new is a %s - ignored", thisfn, id,
983 G_OBJECT_TYPE_NAME( old_item ), G_OBJECT_TYPE_NAME( new_item ));
984
985 } else if( NA_IS_OBJECT_MENU( old_item )){
986 /* hopefully, na_factory_object_copy only copy valuable properties
987 * keeping dynamic variables as parent pointer, provider and provider
988 * data, read-only status - notably children are not impacted by this
989 * copy
990 */
991 na_factory_object_copy( NA_IFACTORY_OBJECT( old_item ), NA_IFACTORY_OBJECT( new_item ));
992
993 } else if( NA_IS_OBJECT_ACTION( old_item )){
994 /* na_factory_object is not a deep copy, which is fine for the menu
995 * but not for the action - it appears more easier to just substitute
996 * the old item with the new one
997 *
998 * only children of the old item are its own profiles, which are to be
999 * replaced with the profiles provided by the new item
1000 */
1001 path = cact_tree_model_delete( ied->model, NA_OBJECT( old_item ));
1002 insert_path = cact_tree_model_insert_before( ied->model, NA_OBJECT( new_item ), path );
1003 gtk_tree_path_free( path );
1004 gtk_tree_path_free( insert_path );
1005
1006 } else {
1007 g_warning( "%s: should not come here!", thisfn );
1008 }
1009
1010 g_free( id );
1011 }
