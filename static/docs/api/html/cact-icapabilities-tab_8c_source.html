236static void
237on_add_clicked( GtkButton *button, BaseWindow *window )
238{
239 NAIContext *context;
240 GSList *capabilities;
241 gchar *new_cap;
242
243 g_object_get( G_OBJECT( window ), MAIN_PROP_CONTEXT, &context, NULL );
244
245 if( context ){
246 capabilities = cact_match_list_get_rows( window, ITAB_NAME );
247 new_cap = cact_add_capability_dialog_run( window, capabilities );
248
249 if( new_cap ){
250 cact_match_list_insert_row( window, ITAB_NAME, new_cap, FALSE, FALSE );
251 g_free( new_cap );
252 }
253
254 na_core_utils_slist_free( capabilities );
255 }
