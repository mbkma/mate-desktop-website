42static gboolean
43draw_handler (GtkWidget *widget,
44 cairo_t *cr)
45{
46 GtkStyleContext *context;
47 gint width;
48 gint height;
49
50 if (widget == NULL)
51 return FALSE;
52
53 context = gtk_widget_get_style_context (widget);
54 gtk_style_context_add_class (context, "tooltip");
55 width = gtk_widget_get_allocated_width (widget);
56 height = gtk_widget_get_allocated_height (widget);
57
58 gtk_render_background (context, cr, 0, 0, width, height);
59 gtk_render_frame (context, cr, 0, 0, width, height);
60
61 return FALSE;
62}

64void
65meta_fixed_tip_show (int root_x, int root_y,
66 const char *markup_text)
67{
68 gint w;
69 gint h;
70 GdkMonitor *mon_num;
71 GdkRectangle monitor;
72 gint screen_right_edge;
73
74 if (tip == NULL)
75 {
76 GdkVisual *visual;
77
78 tip = gtk_window_new (GTK_WINDOW_POPUP);
79
80 gtk_window_set_type_hint (GTK_WINDOW(tip), GDK_WINDOW_TYPE_HINT_TOOLTIP);
81 gtk_style_context_add_class (gtk_widget_get_style_context (tip),
82 GTK_STYLE_CLASS_TOOLTIP);
83
84 screen = gdk_display_get_default_screen (gdk_display_get_default ());
85 visual = gdk_screen_get_rgba_visual (screen);
86
87 gtk_window_set_screen (GTK_WINDOW (tip), screen);
88
89 if (visual != NULL)
90 gtk_widget_set_visual (tip, visual);
91
92 gtk_widget_set_app_paintable (tip, TRUE);
93 gtk_window_set_resizable (GTK_WINDOW (tip), FALSE);
94 g_signal_connect (tip, "draw", G_CALLBACK (draw_handler), NULL);
95
96 label = gtk_label_new (NULL);
97 gtk_label_set_line_wrap (GTK_LABEL (label), TRUE);
98 gtk_label_set_xalign (GTK_LABEL (label), 0.5);
99 gtk_label_set_yalign (GTK_LABEL (label), 0.5);
100 gtk_widget_show (label);
101
102 gtk_container_set_border_width (GTK_CONTAINER (tip), 4);
103 gtk_container_add (GTK_CONTAINER (tip), label);
104
105 g_signal_connect (tip, "destroy",
106 G_CALLBACK (gtk_widget_destroyed), &tip);
107 }
108
109 mon_num = gdk_display_get_monitor_at_point (gdk_screen_get_display (screen), root_x, root_y);
110 gdk_monitor_get_geometry (mon_num, &monitor);
111 screen_right_edge = monitor.x + monitor.width;
112
113 gtk_label_set_markup (GTK_LABEL (label), markup_text);
114
115 gtk_window_get_size (GTK_WINDOW (tip), &w, &h);
116
117 if (meta_ui_get_direction() == META_UI_DIRECTION_RTL)
118 root_x = MAX(0, root_x - w);
119
120 if ((root_x + w) > screen_right_edge)
121 root_x -= (root_x + w) - screen_right_edge;
122
123 gtk_window_move (GTK_WINDOW (tip), root_x, root_y);
124
125 gtk_widget_show (tip);
126}
