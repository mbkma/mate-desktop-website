55static void
56zoom_changed_callback (CajaWindow *window,
57 CajaZoomLevel zoom_level,
58 gboolean supports_zooming,
59 gboolean can_zoom,
60 gboolean can_zoom_in,
61 gboolean can_zoom_out,
62 GtkWidget *zoom_control)
63{
64 if (supports_zooming)
65 {
66 gtk_widget_set_sensitive (zoom_control, can_zoom);
67 gtk_widget_show (zoom_control);
68 if (can_zoom)
69 {
70 caja_zoom_control_set_zoom_level (CAJA_ZOOM_CONTROL (zoom_control),
71 zoom_level);
72 }
73 }
74 else
75 {
76 gtk_widget_hide (zoom_control);
77 }
78}

80static void
81connect_proxy (GtkAction *action,
82 GtkWidget *proxy)
83{
84 if (GTK_IS_TOOL_ITEM (proxy))
85 {
86 GtkToolItem *item = GTK_TOOL_ITEM (proxy);
87 CajaZoomAction *zaction = CAJA_ZOOM_ACTION (action);
88 CajaNavigationWindow *window = zaction->priv->window;
89 GtkWidget *zoom_control;
90
91 zoom_control = caja_zoom_control_new ();
92 gtk_container_add (GTK_CONTAINER (item), zoom_control);
93 gtk_widget_show (zoom_control);
94
95 g_signal_connect_object (zoom_control, "zoom_in",
96 G_CALLBACK (caja_window_zoom_in),
97 window, G_CONNECT_SWAPPED);
98 g_signal_connect_object (zoom_control, "zoom_out",
99 G_CALLBACK (caja_window_zoom_out),
100 window, G_CONNECT_SWAPPED);
101 g_signal_connect_object (zoom_control, "zoom_to_level",
102 G_CALLBACK (caja_window_zoom_to_level),
103 window, G_CONNECT_SWAPPED);
104 g_signal_connect_object (zoom_control, "zoom_to_default",
105 G_CALLBACK (caja_window_zoom_to_default),
106 window, G_CONNECT_SWAPPED);
107
108 g_signal_connect (window, "zoom-changed",
109 G_CALLBACK (zoom_changed_callback),
110 zoom_control);
111 }
112
113 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
114 (* GTK_ACTION_CLASS (parent_class)->connect_proxy) (action, proxy);
115 G_GNUC_END_IGNORE_DEPRECATIONS;
116}

118static void
119disconnect_proxy (GtkAction *action,
120 GtkWidget *proxy)
121{

124 GtkToolItem *item = GTK_TOOL_ITEM (proxy);
125 CajaZoomAction *zaction = CAJA_ZOOM_ACTION (action);
126 CajaNavigationWindow *window = zaction->priv->window;
127 GtkWidget *child;
128
129 child = gtk_bin_get_child (GTK_BIN (item));

131 g_signal_handlers_disconnect_by_func (window, G_CALLBACK (zoom_changed_callback), child);

135 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
136 (* GTK_ACTION_CLASS (parent_class)->disconnect_proxy) (action, proxy);
