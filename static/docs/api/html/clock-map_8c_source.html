166void
167clock_map_refresh (ClockMap *this)
168{
169 ClockMapPrivate *priv = clock_map_get_instance_private (this);
170 GtkWidget *widget = GTK_WIDGET (this);
171 GtkAllocation allocation;
172
173 gtk_widget_get_allocation (widget, &allocation);
174
175 /* Only do something if we have some space allocated.
176 * Note that 1x1 is not really some space... */
177 if (allocation.width <= 1 || allocation.height <= 1)
178 return;
179
180 /* Allocation changed => we reload the map */
181 if (priv->width != allocation.width ||
182 priv->height != allocation.height) {
183 if (priv->stock_map_pixbuf) {
184 g_object_unref (priv->stock_map_pixbuf);
185 priv->stock_map_pixbuf = NULL;
186 }
187
188 priv->width = allocation.width;
189 priv->height = allocation.height;
190 }
191
192 if (!priv->stock_map_pixbuf) {
193 priv->stock_map_pixbuf = gdk_pixbuf_new_from_resource_at_scale (CLOCK_RESOURCE_PATH "icons/clock-map.png",
194 priv->width, priv->height,
195 FALSE,
196 NULL);
197 }
198
199 clock_map_place_locations (this);
200
201 clock_map_display (this);
202}

266static void
267clock_map_mark (ClockMap *this, gfloat latitude, gfloat longitude, gint mark)
268{
269 ClockMapPrivate *priv = clock_map_get_instance_private (this);
270 GdkPixbuf *marker = priv->location_marker_pixbuf[mark];
271 GdkPixbuf *partial = NULL;
272
273 int x, y;
274 int width, height;
275 int marker_width, marker_height;
276 int dest_x, dest_y, dest_width, dest_height;
277
278 width = gdk_pixbuf_get_width (priv->location_map_pixbuf);
279 height = gdk_pixbuf_get_height (priv->location_map_pixbuf);
280
281 x = (width / 2.0 + (width / 2.0) * longitude / 180.0);
282 y = (height / 2.0 - (height / 2.0) * latitude / 90.0);
283
284 marker_width = gdk_pixbuf_get_width (marker);
285 marker_height = gdk_pixbuf_get_height (marker);
286
287 dest_x = x - marker_width / 2;
288 dest_y = y - marker_height / 2;
289 dest_width = marker_width;
290 dest_height = marker_height;
291
292 /* create a small partial pixbuf if the mark is too close to
293 the north or south pole */
294 if (dest_y < 0) {
295 partial = gdk_pixbuf_new_subpixbuf
296 (marker, 0, dest_y + marker_height,
297 marker_width, -dest_y);
298
299 dest_y = 0.0;
300 marker_height = gdk_pixbuf_get_height (partial);
301 } else if (dest_y + dest_height > height) {
302 partial = gdk_pixbuf_new_subpixbuf
303 (marker, 0, 0, marker_width, height - dest_y);
304 marker_height = gdk_pixbuf_get_height (partial);
305 }
306
307 if (partial) {
308 marker = partial;
309 }
310
311 /* handle the cases where the marker needs to be placed across
312 the 180 degree longitude line */
313 if (dest_x < 0) {
314 /* split into our two pixbufs for the left and right edges */
315 GdkPixbuf *lhs = NULL;
316 GdkPixbuf *rhs = NULL;
317
318 lhs = gdk_pixbuf_new_subpixbuf
319 (marker, -dest_x, 0, marker_width + dest_x, marker_height);
320
321 gdk_pixbuf_composite (lhs, priv->location_map_pixbuf,
322 0, dest_y,
323 gdk_pixbuf_get_width (lhs),
324 gdk_pixbuf_get_height (lhs),
325 0, dest_y,
326 1.0, 1.0, GDK_INTERP_NEAREST, 0xFF);
327
328 rhs = gdk_pixbuf_new_subpixbuf
329 (marker, 0, 0, -dest_x, marker_height);
330
331 gdk_pixbuf_composite (rhs, priv->location_map_pixbuf,
332 width - gdk_pixbuf_get_width (rhs) - 1,
333 dest_y,
334 gdk_pixbuf_get_width (rhs),
335 gdk_pixbuf_get_height (rhs),
336 width - gdk_pixbuf_get_width (rhs) - 1,
337 dest_y,
338 1.0, 1.0, GDK_INTERP_NEAREST, 0xFF);
339
340 g_object_unref (lhs);
341 g_object_unref (rhs);
342 } else if (dest_x + dest_width > width) {
343 /* split into our two pixbufs for the left and right edges */
344 GdkPixbuf *lhs = NULL;
345 GdkPixbuf *rhs = NULL;
346
347 lhs = gdk_pixbuf_new_subpixbuf
348 (marker, width - dest_x, 0, marker_width - width + dest_x, marker_height);
349
350 gdk_pixbuf_composite (lhs, priv->location_map_pixbuf,
351 0, dest_y,
352 gdk_pixbuf_get_width (lhs),
353 gdk_pixbuf_get_height (lhs),
354 0, dest_y,
355 1.0, 1.0, GDK_INTERP_NEAREST, 0xFF);
356
357 rhs = gdk_pixbuf_new_subpixbuf
358 (marker, 0, 0, width - dest_x, marker_height);
359
360 gdk_pixbuf_composite (rhs, priv->location_map_pixbuf,
361 width - gdk_pixbuf_get_width (rhs) - 1,
362 dest_y,
363 gdk_pixbuf_get_width (rhs),
364 gdk_pixbuf_get_height (rhs),
365 width - gdk_pixbuf_get_width (rhs) - 1,
366 dest_y,
367 1.0, 1.0, GDK_INTERP_NEAREST, 0xFF);
368
369 g_object_unref (lhs);
370 g_object_unref (rhs);
371 } else {
372 gdk_pixbuf_composite (marker, priv->location_map_pixbuf,
373 dest_x, dest_y,
374 gdk_pixbuf_get_width (marker),
375 gdk_pixbuf_get_height (marker),
376 dest_x, dest_y,
377 1.0, 1.0, GDK_INTERP_NEAREST, 0xFF);
378 }
379
380 if (partial != NULL) {
381 g_object_unref (partial);
382 }

390{

458static void
459clock_map_compute_vector (gdouble lat, gdouble lon, gdouble *vec)
460{
461 gdouble lat_rad, lon_rad;
462 lat_rad = lat * (M_PI/180.0);
463 lon_rad = lon * (M_PI/180.0);
464
465 vec[0] = sin(lon_rad) * cos(lat_rad);
466 vec[1] = sin(lat_rad);
467 vec[2] = cos(lon_rad) * cos(lat_rad);
468}

470static guchar
471clock_map_is_sunlit (gdouble pos_lat, gdouble pos_long,
472 gdouble sun_lat, gdouble sun_long)
473{
474 gdouble pos_vec[3];
475 gdouble sun_vec[3];
476 gdouble dot;
477
478 /* twilight */
479 gdouble epsilon = 0.01;
480
481 clock_map_compute_vector (pos_lat, pos_long, pos_vec);
482 clock_map_compute_vector (sun_lat, sun_long, sun_vec);
483
484 /* compute the dot product of the two */
485 dot = pos_vec[0]*sun_vec[0] + pos_vec[1]*sun_vec[1]
486 + pos_vec[2]*sun_vec[2];
487
488 if (dot > epsilon) {
489 return 0x00;
490 }
491
492 if (dot < -epsilon) {
493 return 0xFF;
494 }
495
496 return (guchar)(-128 * ((dot / epsilon) - 1));
497}

499static void
500clock_map_render_shadow_pixbuf (GdkPixbuf *pixbuf)
501{
502 int x, y;
503 int height, width;
504 int n_channels, rowstride;
505 guchar *pixels, *p;
506 gdouble sun_lat, sun_lon;
507 time_t now = time (NULL);
508
509 n_channels = gdk_pixbuf_get_n_channels (pixbuf);
510 rowstride = gdk_pixbuf_get_rowstride (pixbuf);
511 pixels = gdk_pixbuf_get_pixels (pixbuf);
512
513 width = gdk_pixbuf_get_width (pixbuf);
514 height = gdk_pixbuf_get_height (pixbuf);
515
516 sun_position (now, &sun_lat, &sun_lon);
517
518 for (y = 0; y < height; y++) {
519 gdouble lat = (height / 2.0 - y) / (height / 2.0) * 90.0;
520
521 for (x = 0; x < width; x++) {
522 guchar shade;
523 gdouble lon =
524 (x - width / 2.0) / (width / 2.0) * 180.0;
525
526 shade = clock_map_is_sunlit (lat, lon,
527 sun_lat, sun_lon);
528
529 p = pixels + y * rowstride + x * n_channels;
530 p[3] = shade;
531 }
532 }
533}
