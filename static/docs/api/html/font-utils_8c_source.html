26gchar *
27font_utils_get_font_name (FT_Face face)
28{
29 gchar *name;
30
31 if (g_strcmp0 (face->style_name, "Regular") == 0)
32 name = g_strdup (face->family_name);
33 else
34 name = g_strconcat (face->family_name, ", ", face->style_name, NULL);
35
36 return name;
37}

39gchar *
40font_utils_get_font_name_for_file (FT_Library library,
41 const gchar *path,
42 gint face_index)
43{
44 GFile *file;
45 gchar *uri, *contents = NULL, *name = NULL;
46 GError *error = NULL;
47 FT_Face face;
48
49 file = g_file_new_for_path (path);
50 uri = g_file_get_uri (file);
51
52 face = sushi_new_ft_face_from_uri (library, uri, face_index, &contents,
53 &error);
54 if (face != NULL) {
55 name = font_utils_get_font_name (face);
56 FT_Done_Face (face);
57 } else if (error != NULL) {
58 g_warning ("Can't get font name: %s\n", error->message);
59 g_error_free (error);
60 }
61
62 g_free (uri);
63 g_object_unref (file);
64 g_free (contents);
65
66 return name;
67}
