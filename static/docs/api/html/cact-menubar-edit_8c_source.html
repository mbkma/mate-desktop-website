63void
64cact_menubar_edit_on_update_sensitivities( const CactMenubar *bar )
65{
66 gboolean cut_enabled;
67 gboolean copy_enabled;
68 gboolean paste_enabled;
69 gboolean paste_into_enabled;
70 gboolean duplicate_enabled;
71 gboolean delete_enabled;
72 NAObject *parent_item;
73 NAObject *selected_action;
74 NAObject *selected_item;
75 gboolean is_clipboard_empty;
76
77 is_clipboard_empty = ( bar->private->clipboard_menus + bar->private->clipboard_actions + bar->private->clipboard_profiles == 0 );
78
79 /* cut requires a non-empty selection
80 * and that the selection is writable (can be modified, i.e. is not read-only)
81 * and that all parents are writable (as implies a delete operation)
82 */
83 duplicate_enabled = bar->private->treeview_has_focus || bar->private->popup_handler;
84 duplicate_enabled &= bar->private->count_selected > 0;
85 duplicate_enabled &= bar->private->are_parents_writable;
86 cut_enabled = duplicate_enabled;
87 cut_enabled &= bar->private->are_items_writable;
88 cact_menubar_enable_item( bar, "CutItem", cut_enabled );
89
90 /* copy only requires a non-empty selection */
91 copy_enabled = bar->private->treeview_has_focus || bar->private->popup_handler;
92 copy_enabled &= bar->private->count_selected > 0;
93 cact_menubar_enable_item( bar, "CopyItem", copy_enabled );
94
95 /* paste enabled if
96 * - clipboard is not empty
97 * - current selection is not multiple
98 * - if clipboard contains only profiles,
99 * then current selection must be a profile or an action
100 * and the action must be writable
101 * - if clipboard contains actions or menus,
102 * then current selection (if any) must be a menu or an action
103 * and its parent must be writable
104 */
105 paste_enabled = bar->private->treeview_has_focus || bar->private->popup_handler;
106 paste_enabled &= !is_clipboard_empty;
107 paste_enabled &= bar->private->count_selected <= 1;
108 if( bar->private->clipboard_profiles ){
109 paste_enabled &= bar->private->count_selected == 1;
110 paste_enabled &= bar->private->is_action_writable;
111 } else {
112 paste_enabled &= bar->private->has_writable_providers;
113 if( bar->private->count_selected ){
114 paste_enabled &= bar->private->is_parent_writable;
115 } else {
116 paste_enabled &= bar->private->is_level_zero_writable;
117 }
118 }
119 cact_menubar_enable_item( bar, "PasteItem", paste_enabled );
120
121 /* paste into enabled if
122 * - clipboard is not empty
123 * - current selection is not multiple
124 * - if clipboard contains only profiles,
125 * then current selection must be an action
126 * and the action must be writable
127 * - if clipboard contains actions or menus,
128 * then current selection (if any) must be a menu
129 * and its parent must be writable
130 */
131 paste_into_enabled = bar->private->treeview_has_focus || bar->private->popup_handler;
132 paste_into_enabled &= !is_clipboard_empty;
133 paste_into_enabled &= bar->private->count_selected <= 1;
134 if( bar->private->clipboard_profiles ){
135 paste_into_enabled &= bar->private->count_selected == 1;
136 if( paste_into_enabled ){
137 selected_action = NA_OBJECT( bar->private->selected_items->data );
138 paste_into_enabled &= NA_IS_OBJECT_ACTION( selected_action );
139 if( paste_into_enabled ){
140 paste_into_enabled &= na_object_is_finally_writable( selected_action, NULL );
141 }
142 }
143 } else {
144 paste_into_enabled &= bar->private->has_writable_providers;
145 if( bar->private->count_selected ){
146 selected_item = NA_OBJECT( bar->private->selected_items->data );
147 paste_into_enabled &= NA_IS_OBJECT_MENU( selected_item );
148 if( paste_into_enabled ){
149 parent_item = ( NAObject * ) na_object_get_parent( selected_item );
150 paste_into_enabled &= parent_item
151 ? na_object_is_finally_writable( parent_item, NULL )
152 : bar->private->is_level_zero_writable;
153 }
154 } else {
155 paste_into_enabled &= bar->private->is_level_zero_writable;
156 }
157 }
158 cact_menubar_enable_item( bar, "PasteIntoItem", paste_into_enabled );
159
160 /* duplicate items will be duplicated besides each one
161 * selection must be non-empty
162 * each parent must be writable
163 */
164 cact_menubar_enable_item( bar, "DuplicateItem", duplicate_enabled );
165
166 /* delete is same that cut
167 * but items themselves must be writable (because physically deleted)
168 * this will be checked on delete activated
169 */
170 delete_enabled = cut_enabled;
171 cact_menubar_enable_item( bar, "DeleteItem", delete_enabled );
172
173 /* reload items always enabled */
174
175 /* preferences always enabled */
176}

199void
200cact_menubar_edit_on_cut( GtkAction *gtk_action, BaseWindow *window )
201{
202 static const gchar *thisfn = "cact_menubar_edit_on_cut";
203 GList *items;
204 CactClipboard *clipboard;
205 GList *to_delete;
206 GSList *non_deletables;
207 CactTreeView *view;
208
209 g_debug( "%s: gtk_action=%p, window=%p", thisfn, ( void * ) gtk_action, ( void * ) window );
210 g_return_if_fail( GTK_IS_ACTION( gtk_action ));
211
212 BAR_WINDOW_VOID( window );
213
214 items = na_object_copyref_items( bar->private->selected_items );
215 non_deletables = NULL;
216 to_delete = get_deletables( bar->private->updater, items, &non_deletables );
217
218 if( non_deletables ){
219 gchar *second = na_core_utils_slist_join_at_end( non_deletables, "\n" );
220 base_window_display_error_dlg(
221 BASE_WINDOW( window ),
222 _( "Not all items have been cut as following ones are not modifiable:" ),
223 second );
224 g_free( second );
225 na_core_utils_slist_free( non_deletables );
226 }
227
228 if( to_delete ){
229 clipboard = cact_main_window_get_clipboard( CACT_MAIN_WINDOW( window ));
230 cact_clipboard_primary_set( clipboard, to_delete, CLIPBOARD_MODE_CUT );
231 update_clipboard_counters( window );
232 view = cact_main_window_get_items_view( CACT_MAIN_WINDOW( window ));
233 cact_tree_ieditable_delete( CACT_TREE_IEDITABLE( view ), to_delete, TREE_OPE_DELETE );
234 }
235
236 na_object_free_items( items );
237}

337static GList *
338prepare_for_paste( BaseWindow *window )
339{
340 static const gchar *thisfn = "cact_menubar_edit_prepare_for_paste";
341 GList *items, *it;
342 CactClipboard *clipboard;
343 NAObjectAction *action;
344 gboolean relabel;
345 gboolean renumber;
346
347 BAR_WINDOW_VALUE( window, NULL );
348
349 clipboard = cact_main_window_get_clipboard( CACT_MAIN_WINDOW( window ));
350 items = cact_clipboard_primary_get( clipboard, &renumber );
351 action = NULL;
352
353 /* if pasted items are profiles, then setup the target action
354 */
355 for( it = items ; it ; it = it->next ){
356
357 if( NA_IS_OBJECT_PROFILE( it->data )){
358 if( !action ){
359 g_object_get( G_OBJECT( window ), MAIN_PROP_ITEM, &action, NULL );
360 g_return_val_if_fail( NA_IS_OBJECT_ACTION( action ), NULL );
361 }
362 }
363
364 relabel = na_updater_should_pasted_be_relabeled( bar->private->updater, NA_OBJECT( it->data ));
365 na_object_prepare_for_paste( it->data, relabel, renumber, action );
366 na_object_check_status( it->data );
367 }
368
369 g_debug( "%s: action=%p (%s)",
370 thisfn, ( void * ) action, action ? G_OBJECT_TYPE_NAME( action ): "(null)" );
371
372 return( items );
373}

388cact_menubar_edit_on_duplicate( GtkAction *gtk_action, BaseWindow *window )

391 NAObjectAction *action;

419 na_object_set_origin( obj, NULL );
420 na_object_check_status( obj );
421 dup = g_list_prepend( NULL, obj );
422 view = cact_main_window_get_items_view( CACT_MAIN_WINDOW( window ));

484static GList *
485get_deletables( NAUpdater *updater, GList *selected, GSList **non_deletables )
486{
487 GList *to_delete;
488 GList *it;
489 GList *subitems;
490 GSList *sub_deletables;
491 guint reason;
492 NAObjectItem *item;
493
494 to_delete = NULL;
495 for( it = selected ; it ; it = it->next ){
496
497 if( NA_IS_OBJECT_PROFILE( it->data )){
498 item = na_object_get_parent( it->data );
499 } else {
500 item = NA_OBJECT_ITEM( it->data );
501 }
502
503 if( !na_object_is_finally_writable( item, &reason )){
504 *non_deletables = g_slist_prepend(
505 *non_deletables, add_non_deletable_msg( NA_OBJECT_ITEM( it->data ), reason ));
506 continue;
507 }
508
509 if( NA_IS_OBJECT_MENU( it->data )){
510 subitems = na_object_get_items( it->data );
511 sub_deletables = get_deletables_rec( updater, subitems );
512
513 if( sub_deletables ){
514 *non_deletables = g_slist_concat( *non_deletables, sub_deletables );
515 continue;
516 }
517 }
518
519 to_delete = g_list_prepend( to_delete, na_object_ref( it->data ));
520 }
521
522 return( to_delete );
523}

525static GSList *
526get_deletables_rec( NAUpdater *updater, GList *tree )
527{
528 GSList *msgs;
529 GList *it;
530 GList *subitems;
531 guint reason;
532
533 msgs = NULL;
534 for( it = tree ; it ; it = it->next ){
535
536 if( !na_object_is_finally_writable( it->data, &reason )){
537 msgs = g_slist_prepend(
538 msgs, add_non_deletable_msg( NA_OBJECT_ITEM( it->data ), reason ));
539 continue;
540 }
541
542 if( NA_IS_OBJECT_MENU( it->data )){
543 subitems = na_object_get_items( it->data );
544 msgs = g_slist_concat( msgs, get_deletables_rec( updater, subitems ));
545 }
546 }
547
548 return( msgs );
549}

551static gchar *
552add_non_deletable_msg( const NAObjectItem *item, gint reason )
553{
554 gchar *msg;
555 gchar *label;
556 gchar *reason_str;
557
558 label = na_object_get_label( item );
559 reason_str = na_io_provider_get_readonly_tooltip( reason );
560
561 msg = g_strdup_printf( "%s: %s", label, reason_str );
562
563 g_free( reason_str );
564 g_free( label );
565
566 return( msg );
