58 gtk_recent_chooser_set_sort_type (GTK_RECENT_CHOOSER (toolbar_recent_menu), GTK_RECENT_SORT_MRU);

66 gtk_recent_chooser_set_filter (GTK_RECENT_CHOOSER (toolbar_recent_menu), filter);
67
68 GtkWidget *icon = gtk_image_new_from_icon_name ("document-open", GTK_ICON_SIZE_SMALL_TOOLBAR);
69 tool_item = GTK_WIDGET (gtk_menu_tool_button_new (icon, NULL));
70
71 gtk_menu_tool_button_set_arrow_tooltip_text (GTK_MENU_TOOL_BUTTON (tool_item),
72 _("Open a recently used document"));
73 gtk_menu_tool_button_set_menu (GTK_MENU_TOOL_BUTTON (tool_item),
74 GTK_WIDGET (toolbar_recent_menu));
