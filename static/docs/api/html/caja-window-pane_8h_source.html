50/* A CajaWindowPane is a layer between a slot and a window.
51 * Each slot is contained in one pane, and each pane can contain
52 * one or more slots. It also supports the notion of an "active slot".
53 * On the other hand, each pane is contained in a window, while each
54 * window can contain one or multiple panes. Likewise, the window has
55 * the notion of an "active pane".
56 *
57 * A spatial window has only one pane, which contains a single slot.
58 * A navigation window may have one or more panes.
59 */
60struct _CajaWindowPane
61{
62 GObject parent;
63
64 /* hosting window */
65 CajaWindow *window;
66 gboolean visible;
67
68 /* available slots, and active slot.
69 * Both of them may never be NULL. */
70 GList *slots;
71 GList *active_slots;
72 CajaWindowSlot *active_slot;
73
74 /* whether or not this pane is active */
75 gboolean is_active;
76};
