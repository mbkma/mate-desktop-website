134OssStreamControl *
135oss_stream_control_new (const gchar *name,
136 const gchar *label,
137 MateMixerStreamControlRole role,
138 OssStream *stream,
139 gint fd,
140 gint devnum,
141 gboolean stereo)
142{
143 OssStreamControl *control;
144 gint newfd;
145 MateMixerStreamControlFlags flags;
146
147 g_return_val_if_fail (name != NULL, NULL);
148 g_return_val_if_fail (label != NULL, NULL);
149 g_return_val_if_fail (OSS_IS_STREAM (stream), NULL);
150
151 newfd = dup (fd);
152 if (newfd == -1) {
153 g_warning ("Failed to duplicate file descriptor: %s",
154 g_strerror (errno));
155 return NULL;
156 }
157
158 flags = MATE_MIXER_STREAM_CONTROL_VOLUME_READABLE |
159 MATE_MIXER_STREAM_CONTROL_VOLUME_WRITABLE;
160 if (stereo == TRUE)
161 flags |= MATE_MIXER_STREAM_CONTROL_CAN_BALANCE;
162
163 control = g_object_new (OSS_TYPE_STREAM_CONTROL,
164 "name", name,
165 "label", label,
166 "flags", flags,
167 "role", role,
168 "stream", stream,
169 NULL);
170
171 control->priv->fd = newfd;
172 control->priv->devnum = devnum;
173 control->priv->stereo = stereo;
174 return control;
175}

185void
186oss_stream_control_load (OssStreamControl *control)
187{
188 gint v, ret;
189
190 g_return_if_fail (OSS_IS_STREAM_CONTROL (control));
191
192 if (G_UNLIKELY (control->priv->fd == -1))
193 return;
194
195 ret = ioctl (control->priv->fd, MIXER_READ (control->priv->devnum), &v);
196 if (ret == -1)
197 return;
198
199 store_volume (control, v);
200}

254oss_stream_control_get_channel_volume (MateMixerStreamControl *mmsc, guint channel)

272static gboolean
273oss_stream_control_set_channel_volume (MateMixerStreamControl *mmsc,
274 guint channel,
275 guint volume)
276{
277 OssStreamControl *control;
278 gint v;
279
280 g_return_val_if_fail (OSS_IS_STREAM_CONTROL (mmsc), FALSE);
281
282 control = OSS_STREAM_CONTROL (mmsc);
283
284 if (G_UNLIKELY (control->priv->fd == -1))
285 return FALSE;
286
287 if (channel != LEFT_CHANNEL &&
288 (control->priv->stereo == FALSE || channel != RIGHT_CHANNEL))
289 return FALSE;
290
291 volume = CLAMP (volume, 0, 100);
292 if (channel == LEFT_CHANNEL)
293 v = OSS_VOLUME_JOIN (volume, control->priv->volume[RIGHT_CHANNEL]);
294 else
295 v = OSS_VOLUME_JOIN (control->priv->volume[LEFT_CHANNEL], volume);
296
297 return write_and_store_volume (control, v);
298}

337static gboolean
338oss_stream_control_set_balance (MateMixerStreamControl *mmsc, gfloat balance)
339{
340 OssStreamControl *control;
341 guint max;
342 gint volume[2];
343
344 g_return_val_if_fail (OSS_IS_STREAM_CONTROL (mmsc), FALSE);
345
346 control = OSS_STREAM_CONTROL (mmsc);
347
348 if (G_UNLIKELY (control->priv->fd == -1))
349 return FALSE;
350
351 max = MAX (control->priv->volume[LEFT_CHANNEL],
352 control->priv->volume[RIGHT_CHANNEL]);
353 if (balance <= 0) {
354 volume[RIGHT_CHANNEL] = (balance + 1.0f) * max;
355 volume[LEFT_CHANNEL] = max;
356 } else {
357 volume[LEFT_CHANNEL] = (1.0f - balance) * max;
358 volume[RIGHT_CHANNEL] = max;
359 }
360 return write_and_store_volume (control, OSS_VOLUME_JOIN_ARRAY (volume));
361}

393static void
394store_volume (OssStreamControl *control, gint volume)
395{
396 if (control->priv->stereo == TRUE) {
397 if (volume == OSS_VOLUME_JOIN_ARRAY (control->priv->volume))
398 return;
399
400 control->priv->volume[LEFT_CHANNEL] = OSS_VOLUME_TAKE_LEFT (volume);
401 control->priv->volume[RIGHT_CHANNEL] = OSS_VOLUME_TAKE_RIGHT (volume);
402
403 g_object_freeze_notify (G_OBJECT (control));
404
405 g_object_notify (G_OBJECT (control), "volume");
406
407 /* Emits signal if balance has changed */
408 update_balance (control);
409
410 g_object_thaw_notify (G_OBJECT (control));
411 } else {
412 volume = OSS_VOLUME_TAKE_LEFT (volume);
413 if (volume == control->priv->volume[LEFT_CHANNEL])
414 return;
415
416 control->priv->volume[LEFT_CHANNEL] = volume;
417
418 g_object_notify (G_OBJECT (control), "volume");
419 }
420}

422static void
423update_balance (OssStreamControl *control)
424{
425 gfloat balance;
426 guint left;
427 guint right;
428
429 left = control->priv->volume[LEFT_CHANNEL];
430 right = control->priv->volume[RIGHT_CHANNEL];
431
432 if (left == right)
433 balance = 0.0f;
434 else if (left > right)
435 balance = -1.0f + ((gfloat) right / (gfloat) left);
436 else
437 balance = +1.0f - ((gfloat) left / (gfloat) right);
438
439 _mate_mixer_stream_control_set_balance (MATE_MIXER_STREAM_CONTROL (control),
440 balance);
441}

443static gboolean
444write_and_store_volume (OssStreamControl *control, gint volume)
445{
446 gint ret;
447
448 /* Nothing to do? */
449 if (volume == OSS_VOLUME_JOIN_ARRAY (control->priv->volume))
450 return TRUE;
451
452 /* The ioctl might also change the passed volume */
453 ret = ioctl (control->priv->fd, MIXER_WRITE (control->priv->devnum), &volume);
454 if (ret == -1)
455 return FALSE;
456
457 store_volume (control, volume & 0xFFFF);
458 return TRUE;
