81static gboolean
82iter_for_face_foreach (GtkTreeModel *model,
83 GtkTreePath *path,
84 GtkTreeIter *iter,
85 gpointer user_data)
86{
87 IterForFaceData *data = user_data;
88 gchar *font_name, *match_name;
89 gboolean retval;
90
91 gtk_tree_model_get (GTK_TREE_MODEL (model), iter,
92 COLUMN_NAME, &font_name,
93 -1);
94
95 match_name = font_utils_get_font_name (data->face);
96 retval = (g_strcmp0 (font_name, match_name) == 0);
97
98 g_free (match_name);
99 g_free (font_name);
100
101 if (retval) {
102 data->iter = *iter;
103 data->found = TRUE;
104 }
105
106 return retval;

111 FT_Face face,

113{

120 gtk_tree_model_foreach (GTK_TREE_MODEL (self),

172static GdkPixbuf *
173create_thumbnail (ThumbInfoData *thumb_info)
174{
175 GFile *file = thumb_info->font_file;
176 MateDesktopThumbnailFactory *factory;
177 guint64 mtime;
178
179 GdkPixbuf *pixbuf = NULL;
180 GFileInfo *info = NULL;
181
182 info = g_file_query_info (file, ATTRIBUTES_FOR_CREATING_THUMBNAIL,
183 G_FILE_QUERY_INFO_NONE,
184 NULL, NULL);
185
186 /* we don't care about reporting errors here, just fail the
187 * thumbnail.
188 */
189 if (info == NULL)
190 goto out;
191
192 mtime = g_file_info_get_attribute_uint64 (info, G_FILE_ATTRIBUTE_TIME_MODIFIED);
193
194 factory = mate_desktop_thumbnail_factory_new (MATE_DESKTOP_THUMBNAIL_SIZE_NORMAL);
195 pixbuf = mate_desktop_thumbnail_factory_generate_thumbnail
196 (factory,
197 thumb_info->uri, g_file_info_get_content_type (info));
198
199 if (pixbuf != NULL)
200 mate_desktop_thumbnail_factory_save_thumbnail (factory, pixbuf,
201 thumb_info->uri, (time_t) mtime);
202 else
203 mate_desktop_thumbnail_factory_create_failed_thumbnail (factory,
204 thumb_info->uri, (time_t) mtime);
205
206 g_object_unref (factory);
207
208 out:
209 g_clear_object (&info);
210
211 return pixbuf;
212}

214static void
215ensure_thumbnails_job (GTask *task,
216 gpointer source_object,
217 gpointer user_data,
218 GCancellable *cancellable)
219{
220 GList *thumb_infos = user_data, *l;
221
222 for (l = thumb_infos; l != NULL; l = l->next) {
223 gboolean thumb_failed;
224 gchar *thumb_path = NULL;
225 ThumbInfoData *thumb_info = l->data;
226
227 GError *error = NULL;
228 GFile *thumb_file = NULL;
229 GFileInputStream *is = NULL;
230 GFileInfo *info = NULL;
231
232 if (thumb_info->face_index == 0) {
233 thumb_info->uri = g_file_get_uri (thumb_info->font_file);
234 info = g_file_query_info (thumb_info->font_file,
235 ATTRIBUTES_FOR_EXISTING_THUMBNAIL,
236 G_FILE_QUERY_INFO_NONE,
237 NULL, &error);
238
239 if (error != NULL) {
240 gchar *font_path;
241
242 font_path = g_file_get_path (thumb_info->font_file);
243 g_debug ("Can't query info for file %s: %s\n", font_path, error->message);
244 g_free (font_path);
245
246 goto next;
247 }
248
249 thumb_failed = g_file_info_get_attribute_boolean (info, G_FILE_ATTRIBUTE_THUMBNAILING_FAILED);
250 if (thumb_failed)
251 goto next;
252
253 thumb_path = g_strdup (g_file_info_get_attribute_byte_string (info, G_FILE_ATTRIBUTE_THUMBNAIL_PATH));
254 } else {
255 gchar *file_uri;
256 gchar *checksum;
257 gchar *filename;
258
259 file_uri = g_file_get_uri (thumb_info->font_file);
260 thumb_info->uri = g_strdup_printf ("%s#0x%08X", file_uri, thumb_info->face_index);
261 g_free (file_uri);
262
263 checksum = g_compute_checksum_for_data (G_CHECKSUM_MD5,
264 (const guchar *) thumb_info->uri,
265 strlen (thumb_info->uri));
266 filename = g_strdup_printf ("%s.png", checksum);
267 g_free (checksum);
268
269 thumb_path = g_build_filename (g_get_user_cache_dir (),
270 "thumbnails",
271 "large",
272 filename,
273 NULL);
274 g_free (filename);
275
276 if (!g_file_test (thumb_path, G_FILE_TEST_IS_REGULAR)) {
277 g_clear_pointer (&thumb_path, g_free);
278 }
279 }
280
281 if (thumb_path != NULL) {
282 thumb_file = g_file_new_for_path (thumb_path);
283 is = g_file_read (thumb_file, NULL, &error);
284
285 if (error != NULL) {
286 g_debug ("Can't read file %s: %s\n", thumb_path, error->message);
287 goto next;
288 }
289
290 thumb_info->pixbuf = gdk_pixbuf_new_from_stream_at_scale (G_INPUT_STREAM (is),
291 128, 128, TRUE,
292 NULL, &error);
293
294 if (error != NULL) {
295 g_debug ("Can't read thumbnail pixbuf %s: %s\n", thumb_path, error->message);
296 goto next;
297 }
298 } else {
299 thumb_info->pixbuf = create_thumbnail (thumb_info);
300 }
301
302 next:
303 g_clear_error (&error);
304 g_clear_object (&is);
305 g_clear_object (&thumb_file);
306 g_clear_object (&info);
307 g_clear_pointer (&thumb_path, g_free);
308
309 g_main_context_invoke (NULL, one_thumbnail_done,
310 thumb_info);
311 }
312
313 g_list_free (thumb_infos);
314}

332static void
333font_infos_loaded (GObject *source_object,
334 GAsyncResult *result,
335 gpointer user_data)
336{
337 FontViewModel *self = FONT_VIEW_MODEL (source_object);
338 GTask *task = NULL;
339 GList *l, *thumb_infos = NULL;
340 GList *font_infos = g_task_propagate_pointer (G_TASK (result), NULL);
341
342 for (l = font_infos; l != NULL; l = l->next) {
343 FontInfoData *font_info = l->data;
344 gchar *collation_key;
345 GtkTreeIter iter;
346 ThumbInfoData *thumb_info;
347
348 collation_key = g_utf8_collate_key (font_info->font_name, -1);
349 gtk_list_store_insert_with_values (GTK_LIST_STORE (self), &iter, -1,
350 COLUMN_NAME, font_info->font_name,
351 COLUMN_PATH, font_info->font_path,
352 COLUMN_FACE_INDEX, font_info->face_index,
353 COLUMN_ICON, self->priv->fallback_icon,
354 COLUMN_COLLATION_KEY, collation_key,
355 -1);
356 g_free (collation_key);
357
358 thumb_info = g_slice_new0 (ThumbInfoData);
359 thumb_info->font_file = g_file_new_for_path (font_info->font_path);
360 thumb_info->face_index = font_info->face_index;
361 thumb_info->iter = iter;
362 thumb_info->self = g_object_ref (self);
363
364 font_info_data_free (font_info);
365 thumb_infos = g_list_prepend (thumb_infos, thumb_info);
366 }
367
368 g_signal_emit (self, signals[CONFIG_CHANGED], 0);
369 g_list_free (font_infos);
370
371 task = g_task_new (NULL, NULL, NULL, NULL);
372 g_task_set_task_data (task, thumb_infos, NULL);
373 g_task_run_in_thread (task, ensure_thumbnails_job);
374 g_object_unref (task);
375}

377static void
378load_font_infos (GTask *task,
379 gpointer source_object,
380 gpointer user_data,
381 GCancellable *cancellable)
382{
383 FontViewModel *self = FONT_VIEW_MODEL (source_object);
384 gint i, n_fonts;
385 GList *font_infos = NULL;
386
387 n_fonts = self->priv->font_list->nfont;
388
389 for (i = 0; i < n_fonts; i++) {
390 FontInfoData *font_info;
391 FcChar8 *file;
392 int index;
393 gchar *font_name;
394
395 if (g_cancellable_is_cancelled (cancellable))
396 break;
397
398 g_mutex_lock (&self->priv->font_list_mutex);
399 FcPatternGetString (self->priv->font_list->fonts[i], FC_FILE, 0, &file);
400 FcPatternGetInteger (self->priv->font_list->fonts[i], FC_INDEX, 0, &index);
401 g_mutex_unlock (&self->priv->font_list_mutex);
402
403 font_name = font_utils_get_font_name_for_file (self->priv->library,
404 (const gchar *) file,
405 index);
406
407 if (!font_name)
408 continue;
409
410 font_info = g_slice_new0 (FontInfoData);
411 font_info->font_name = font_name;
412 font_info->font_path = g_strdup ((const gchar *) file);
413 font_info->face_index = index;
414
415 font_infos = g_list_prepend (font_infos, font_info);
416 }
417
418 g_task_return_pointer (task, font_infos, NULL);
419}

421/* make sure the font list is valid */
422static void
423ensure_font_list (FontViewModel *self)
424{
425 FcPattern *pat;
426 FcObjectSet *os;
427 GTask *task;
428
429 /* always reinitialize the font database */
430 if (!FcInitReinitialize())
431 return;
432
433 if (self->priv->cancellable) {
434 g_cancellable_cancel (self->priv->cancellable);
435 g_clear_object (&self->priv->cancellable);
436 }
437
438 gtk_list_store_clear (GTK_LIST_STORE (self));
439
440 pat = FcPatternCreate ();
441 os = FcObjectSetBuild (FC_FILE, FC_INDEX, FC_FAMILY, FC_WEIGHT, FC_SLANT, NULL);
442
443 g_mutex_lock (&self->priv->font_list_mutex);
444
445 if (self->priv->font_list) {
446 FcFontSetDestroy (self->priv->font_list);
447 self->priv->font_list = NULL;
448 }
449
450 self->priv->font_list = FcFontList (NULL, pat, os);
451
452 g_mutex_unlock (&self->priv->font_list_mutex);
453
454 FcPatternDestroy (pat);
455 FcObjectSetDestroy (os);
456
457 if (!self->priv->font_list)
458 return;
459
460 self->priv->cancellable = g_cancellable_new ();
461
462 task = g_task_new (self, self->priv->cancellable, font_infos_loaded, NULL);
463 g_task_set_return_on_cancel (task, TRUE);
464 g_task_run_in_thread (task, load_font_infos);
465}

517{
518 FcConfig *config;
519 FcStrList *str_list;

521 GFile *file;
522 GFileMonitor *monitor;
523
524 config = FcConfigGetCurrent ();
525 str_list = FcConfigGetFontDirs (config);
526
527 while ((path = FcStrListNext (str_list)) != NULL) {
528 file = g_file_new_for_path ((const gchar *) path);
529 monitor = g_file_monitor (file, G_FILE_MONITOR_NONE,
530 NULL, NULL);
531
532 if (monitor != NULL) {
533 self->priv->monitors = g_list_prepend (self->priv->monitors, monitor);
534 g_signal_connect (monitor, "changed",
535 G_CALLBACK (file_monitor_changed_cb), self);
536 }
537
538 g_object_unref (file);
539 }
540
541 FcStrListDone (str_list);

567static void
568font_view_model_init (FontViewModel *self)
569{
570 GType types[NUM_COLUMNS] =
