261void
262gdict_source_loader_add_search_path (GdictSourceLoader *loader,
263 const gchar *path)
264{
265 GSList *l;
266
267 g_return_if_fail (GDICT_IS_SOURCE_LOADER (loader));
268 g_return_if_fail (path != NULL);
269
270 /* avoid duplications */
271 for (l = loader->priv->paths; l != NULL; l = l->next)
272 if (strcmp (path, (gchar *) l->data) == 0)
273 return;
274
275 loader->priv->paths = g_slist_append (loader->priv->paths, g_strdup (path));
276 loader->priv->paths_dirty = TRUE;
277}

299/* create the list of dictionary source files, by scanning the search path
300 * directories for .desktop files; we disavow symlinks and sub-directories
301 * for the time being.
302 */
303static GSList *
304build_source_filenames (GdictSourceLoader *loader)
305{
306 GSList *retval, *d;
307
308 g_assert (GDICT_IS_SOURCE_LOADER (loader));
309
310 if (!loader->priv->paths)
311 return NULL;
312
313 retval = NULL;
314 for (d = loader->priv->paths; d != NULL; d = d->next)
315 {
316 gchar *path = (gchar *) d->data;
317 const gchar *filename;
318 GDir *dir;
319
320 dir = g_dir_open (path, 0, NULL);
321 if (!dir)
322 continue;
323
324 do
325 {
326 filename = g_dir_read_name (dir);
327 if (filename)
328 {
329 gchar *full_path;
330
331 if (!g_str_has_suffix (filename, GDICT_SOURCE_FILE_SUFFIX))
332 break;
333
334 full_path = g_build_filename (path, filename, NULL);
335 if (g_file_test (full_path, G_FILE_TEST_IS_REGULAR))
336 {
337 retval = g_slist_prepend (retval, full_path);
338 }
339 }
340 }
341 while (filename != NULL);
342
343 g_dir_close (dir);
344 }
345
346 return g_slist_reverse (retval);
347}

349static void
350gdict_source_loader_update_sources (GdictSourceLoader *loader)
351{
352 GSList *filenames, *f;
353
354 g_assert (GDICT_IS_SOURCE_LOADER (loader));
355
356 g_slist_free_full (loader->priv->sources,
357 g_object_unref);
358 loader->priv->sources = NULL;
359
360 filenames = build_source_filenames (loader);
361 for (f = filenames; f != NULL; f = f->next)
362 {
363 GdictSource *source;
364 GError *load_err;
365 gchar *path = (gchar *) f->data;
366
367 g_assert (path != NULL);
368
369 source = gdict_source_new ();
370
371 load_err = NULL;
372 gdict_source_load_from_file (source, path, &load_err);
373 if (load_err)
374 {
375 g_warning ("Unable to load dictionary source at '%s': %s\n",
376 path,
377 load_err->message);
378 g_error_free (load_err);
379
380 continue;
381 }
382
383 loader->priv->sources = g_slist_append (loader->priv->sources,
384 source);
385 g_hash_table_replace (loader->priv->sources_by_name,
386 g_strdup (gdict_source_get_name (source)),
387 source);
388
389 g_signal_emit (loader, loader_signals[SOURCE_LOADED], 0, source);
390 }
391
392 g_slist_free_full (filenames, g_free);
393
394 loader->priv->paths_dirty = FALSE;
395}

410gchar **
411gdict_source_loader_get_names (GdictSourceLoader *loader,
412 gsize *length)
413{
414 GSList *l;
415 gchar **names;
416 gsize i;
417
418 g_return_val_if_fail (GDICT_IS_SOURCE_LOADER (loader), NULL);
419
420 if (loader->priv->paths_dirty)
421 gdict_source_loader_update_sources (loader);
422
423 names = g_new0 (gchar *, g_slist_length (loader->priv->sources) + 1);
424
425 i = 0;
426 for (l = loader->priv->sources; l != NULL; l = l->next)
427 {
428 GdictSource *s = GDICT_SOURCE (l->data);
429
430 g_assert (s != NULL);
431
432 names[i++] = g_strdup (gdict_source_get_name (s));
433 }
434 names[i] = NULL;
435
436 if (length)
437 *length = i;
438
439 return names;
440}

511gboolean
512gdict_source_loader_remove_source (GdictSourceLoader *loader,
513 const gchar *name)
514{
515 GdictSourceLoaderPrivate *priv;
516 GSList *l;
517
518 g_return_val_if_fail (GDICT_IS_SOURCE_LOADER (loader), FALSE);
519 g_return_val_if_fail (name != NULL, FALSE);
520
521 priv = loader->priv;
522
523 if (priv->paths_dirty)
524 gdict_source_loader_update_sources (loader);
525
526 for (l = priv->sources; l != NULL; l = l->next)
527 {
528 GdictSource *s = GDICT_SOURCE (l->data);
529
530 if (strcmp (gdict_source_get_name (s), name) == 0)
531 {
532 gchar *filename;
533
534 g_object_get (G_OBJECT (s), "filename", &filename, NULL);
535
536 if (g_unlink (filename) == -1)
537 {
538 g_warning ("Unable to remove filename '%s' for the "
539 "dictionary source '%s'\n",
540 filename,
541 name);
542
543 return FALSE;
544 }
545
546 g_hash_table_remove (priv->sources_by_name, name);
547
548 priv->sources = g_slist_remove_link (priv->sources, l);
549
550 g_object_unref (s);
551 g_slist_free (l);
552
553 return TRUE;
554 }
555 }
556
557 return FALSE;
558}
