64 class InodeDevices
65 {
66 typedef std::map<guint16, string> Map;
67 Map devices;
68
69 public:
70
71 void update()
72 {
73 this->devices.clear();
74
75 glibtop_mountlist list;
76 glibtop_mountentry *entries = glibtop_get_mountlist(&list, 1);
77
78 for (unsigned i = 0; i != list.number; ++i) {
79 struct stat buf;
80
81 if (stat(entries[i].devname, &buf) != -1)
82 this->devices[buf.st_rdev] = entries[i].devname;
83 }
84
85 g_free(entries);
86 }
87
88 string get(guint64 dev64)
89 {
90 if (dev64 == 0)
91 return "";
92
93 guint16 dev = dev64 & 0xffff;
94
95 if (dev != dev64)
96 g_warning("weird device %" G_GINT64_MODIFIER "x", dev64);
97
98 Map::iterator it(this->devices.find(dev));
99
100 if (it != this->devices.end())
101 return it->second;
102
103 guint8 major, minor;
104 major = dev >> 8;
105 minor = dev;
106
107 std::ostringstream out;
108 out << std::hex
109 << std::setfill('0')
110 << std::setw(2) << unsigned(major)
111 << ':'
112 << std::setw(2) << unsigned(minor);
113
114 this->devices[dev] = out.str();
115 return out.str();
116 }
117 };

162static void
163update_row(GtkTreeModel *model, GtkTreeIter &row, const MemMapsData &mm, const glibtop_map_entry *memmaps)
164{
165 guint64 size;
166 string filename, device;
167 string vmstart, vmend, vmoffset;
168 char flags[5] = "----";
169
170 size = memmaps->end - memmaps->start;
171
172 if(memmaps->perm & GLIBTOP_MAP_PERM_READ) flags [0] = 'r';
173 if(memmaps->perm & GLIBTOP_MAP_PERM_WRITE) flags [1] = 'w';
174 if(memmaps->perm & GLIBTOP_MAP_PERM_EXECUTE) flags [2] = 'x';
175 if(memmaps->perm & GLIBTOP_MAP_PERM_SHARED) flags [3] = 's';
176 if(memmaps->perm & GLIBTOP_MAP_PERM_PRIVATE) flags [3] = 'p';
177
178 if (memmaps->flags & (1 << GLIBTOP_MAP_ENTRY_FILENAME))
179 filename = memmaps->filename;
180
181 vmstart = mm.format(memmaps->start);
182 vmend = mm.format(memmaps->end);
183 vmoffset = mm.format(memmaps->offset);
184 device = mm.devices.get(memmaps->device);
185
186 gtk_list_store_set (GTK_LIST_STORE (model), &row,
187 MMAP_COL_FILENAME, filename.c_str(),
188 MMAP_COL_VMSTART, vmstart.c_str(),
189 MMAP_COL_VMEND, vmend.c_str(),
190 MMAP_COL_VMSZ, size,
191 MMAP_COL_FLAGS, flags,
192 MMAP_COL_VMOFFSET, vmoffset.c_str(),
193 MMAP_COL_PRIVATE_CLEAN, memmaps->private_clean,
194 MMAP_COL_PRIVATE_DIRTY, memmaps->private_dirty,
195 MMAP_COL_SHARED_CLEAN, memmaps->shared_clean,
196 MMAP_COL_SHARED_DIRTY, memmaps->shared_dirty,
197 MMAP_COL_DEVICE, device.c_str(),
198 MMAP_COL_INODE, memmaps->inode,
199 -1);
200}

205static void
206update_memmaps_dialog (MemMapsData *mmdata)
207{
208 GtkTreeModel *model;
209 glibtop_map_entry *memmaps;
210 glibtop_proc_map procmap;
211
212 memmaps = glibtop_get_proc_map (&procmap, mmdata->info->pid);
213 /* process has disappeared */
214 if(!memmaps or procmap.number == 0) return;
215
216 mmdata->format.set(memmaps[procmap.number - 1]);
217
218 model = gtk_tree_view_get_model (GTK_TREE_VIEW (mmdata->tree));
219
220 GtkTreeIter iter;
221
222 typedef std::map<guint64, GtkTreeIter> IterCache;
223 IterCache iter_cache;
224
225 /*
226 removes the old maps and
227 also fills a cache of start -> iter in order to speed
228 up add
229 */
230
231 if (gtk_tree_model_get_iter_first(model, &iter)) {
232 while (true) {
233 char *vmstart = 0;
234 guint64 start;
235 gtk_tree_model_get(model, &iter,
236 MMAP_COL_VMSTART, &vmstart,
237 -1);
238
239 try {
240 std::istringstream(vmstart) >> std::hex >> start;
241 } catch (std::logic_error &e) {
242 g_warning("Could not parse %s", vmstart);
243 start = 0;
244 }
245
246 g_free(vmstart);
247
248 bool found = std::binary_search(memmaps, memmaps + procmap.number,
249 start, glibtop_map_entry_cmp());
250
251 if (found) {
252 iter_cache[start] = iter;
253 if (!gtk_tree_model_iter_next(model, &iter))
254 break;
255 } else {
256 if (!gtk_list_store_remove(GTK_LIST_STORE(model), &iter))
257 break;
258 }
259 }
260 }
261
262 mmdata->devices.update();
263
264 /*
265 add the new maps
266 */
267
268 for (guint i = 0; i != procmap.number; i++) {
269 GtkTreeIter iter;
270 IterCache::iterator it(iter_cache.find(memmaps[i].start));
271
272 if (it != iter_cache.end())
273 iter = it->second;
274 else
275 gtk_list_store_prepend(GTK_LIST_STORE(model), &iter);
276
277 update_row(model, iter, *mmdata, &memmaps[i]);
278 }
279
280 g_free (memmaps);
281}
