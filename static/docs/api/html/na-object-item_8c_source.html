226static void
227object_copy( NAObject *target, const NAObject *source, guint mode )
228{
229 static const gchar *thisfn = "na_object_item_object_copy";
230 void *provider;
231 NAObjectItem *dest, *src;
232
233 g_return_if_fail( NA_IS_OBJECT_ITEM( target ));
234 g_return_if_fail( NA_IS_OBJECT_ITEM( source ));
235
236 dest = NA_OBJECT_ITEM( target );
237 src = NA_OBJECT_ITEM( source );
238
239 if( !dest->private->dispose_has_run && !src->private->dispose_has_run ){
240
241 if( mode == DUPLICATE_REC ||
242 ( mode == DUPLICATE_OBJECT && G_OBJECT_TYPE( source ) == NA_TYPE_OBJECT_ACTION )){
243
244 copy_children( dest, src, mode );
245 }
246
247 provider = na_object_get_provider( source );
248
249 if( provider ){
250 if( !NA_IS_IO_PROVIDER( provider )){
251 g_warning( "%s: source=%p (%s), provider=%p is not a NAIOProvider",
252 thisfn,
253 ( void * ) source, G_OBJECT_TYPE_NAME( source ),
254 ( void * ) provider );
255
256 } else {
257 na_io_provider_duplicate_data( NA_IO_PROVIDER( provider ), NA_OBJECT_ITEM( target ), NA_OBJECT_ITEM( source ), NULL );
258 }
259 }
260
261 dest->private->writable = src->private->writable;
262 dest->private->reason = src->private->reason;
263
264 /* chain up to the parent class */
265 if( NA_OBJECT_CLASS( st_parent_class )->copy ){
266 NA_OBJECT_CLASS( st_parent_class )->copy( target, source, mode );
267 }
268 }

271/*
272 * object_are_equal:
273 * @a: the first (original) #NAObjectItem instance.
274 * @b: the second #NAObjectItem instance.
275 *
276 * This function participates to the #na_iduplicable_check_status() stack,
277 * and is triggered after all comparable elementary data (in #NAIFactoryObject
278 * sense) have already been successfully compared.
279 *
280 * We have to deal here with the subitems: comparing children by their ids
281 * between @a and @b.
282 *
283 * Returns: %TRUE if @a is equal to @b.
284 */
285static gboolean
286object_are_equal( const NAObject *a, const NAObject *b )
287{
288 static const gchar *thisfn = "na_object_item_object_are_equal";
289 gboolean are_equal;
290 NAObjectItem *origin, *duplicate;
291 GSList *a_slist, *b_slist;
292 gchar *a_list, *b_list;
293
294 g_return_val_if_fail( NA_IS_OBJECT_ITEM( a ), FALSE );
295 g_return_val_if_fail( NA_IS_OBJECT_ITEM( b ), FALSE );
296
297 are_equal = FALSE;
298 origin = NA_OBJECT_ITEM( a );
299 duplicate = NA_OBJECT_ITEM( b );
300
301 if( !origin->private->dispose_has_run &&
302 !duplicate->private->dispose_has_run ){
303
304 g_debug( "%s: a=%p, b=%p", thisfn, ( void * ) a, ( void * ) b );
305
306 a_slist = get_children_slist( origin );
307 a_list = na_core_utils_slist_join_at_end( a_slist, ";" );
308 na_core_utils_slist_free( a_slist );
309
310 b_slist = get_children_slist( duplicate );
311 b_list = na_core_utils_slist_join_at_end( b_slist, ";" );
312 na_core_utils_slist_free( b_slist );
313
314 are_equal = ( strcmp( a_list, b_list ) == 0 );
315
316 g_free( a_list );
317 g_free( b_list );
318 }
319
320 /* chain call to parent class */
321 if( NA_OBJECT_CLASS( st_parent_class )->are_equal ){
322 are_equal &= NA_OBJECT_CLASS( st_parent_class )->are_equal( a, b );
323 }
324
325 return( are_equal );
326}

328/*
329 * must have at least one valid subitem
330 */
331static gboolean
332object_is_valid( const NAObject *object )
333{
334 static const gchar *thisfn = "na_object_item_object_is_valid";
335 gboolean is_valid;
336 NAObjectItem *item;
337 GList *children, *ic;
338 gint valid_children;
339
340 g_return_val_if_fail( NA_IS_OBJECT_ITEM( object ), FALSE );
341
342 is_valid = FALSE;
343 item = NA_OBJECT_ITEM( object );
344
345 if( !item->private->dispose_has_run ){
346 g_debug( "%s: item=%p (%s)", thisfn, ( void * ) item, G_OBJECT_TYPE_NAME( item ));
347
348 is_valid = TRUE;
349
350 valid_children = 0;
351 children = na_object_get_items( item );
352 for( ic = children ; ic && !valid_children ; ic = ic->next ){
353 if( na_object_is_valid( ic->data )){
354 valid_children += 1;
355 }
356 }
357
358 is_valid &= ( valid_children > 0 );
359
360 if( !is_valid ){
361 na_object_debug_invalid( item, "no valid child" );
362 }
363 }
364
365 /* chain up to the parent class */
366 if( NA_OBJECT_CLASS( st_parent_class )->is_valid ){
367 is_valid &= NA_OBJECT_CLASS( st_parent_class )->is_valid( object );
368 }
369
370 return( is_valid );
371}

374 * new_parent is not relevant when allocating a new identifier for an
375 * action or a menu ; it may safely be left as NULL though there is no

378static gchar *
379object_id_new_id( const NAObjectId *item, const NAObjectId *new_parent )
380{
381 GList *children, *it;
382 uuid_t uuid;
383 gchar uuid_str[64];
384 gchar *new_uuid = NULL;
385
386 g_return_val_if_fail( NA_IS_OBJECT_ITEM( item ), NULL );
387
388 if( !NA_OBJECT_ITEM( item )->private->dispose_has_run ){
389
390 /* recurse into NAObjectItems children
391 * i.e., if a menu, recurse into embedded actions
392 */
393 children = na_object_get_items( item );
394 for( it = children ; it ; it = it->next ){
395 na_object_set_new_id( it->data, new_parent );
396 }
397
398 uuid_generate( uuid );
399 uuid_unparse_lower( uuid, uuid_str );
400 new_uuid = g_strdup( uuid_str );
401 }
402
403 return( new_uuid );
404}

544 na_object_set_items( item, children );
545 }

680static void
681count_items_rec( GList *items, gint *menus, gint *actions, gint *profiles, gboolean recurse )
682{
683 GList *it;
684
685 for( it = items ; it ; it = it->next ){
686
687 if( recurse ){
688 if( NA_IS_OBJECT_ITEM( it->data )){

690 count_items_rec( subitems, menus, actions, profiles, recurse );
691 }
692 }
693
694 if( NA_IS_OBJECT_MENU( it->data )){
695 *menus += 1;
696
697 } else if( NA_IS_OBJECT_ACTION( it->data )){
698 *actions += 1;
699
700 } else if( NA_IS_OBJECT_PROFILE( it->data )){
701 *profiles += 1;
702 }
703 }

842/*
843 * only copy children if mode is 'full recursive'
844 * or mode is 'duplicate object' and this is an action with profiles
845 */
846static void
847copy_children( NAObjectItem *target, const NAObjectItem *source, guint mode )
848{
849 static const gchar *thisfn = "na_object_item_copy_children";
850 GList *tgt_children, *src_children, *ic;
851 NAObject *dup;
852
853 tgt_children = na_object_get_items( target );
854 if( tgt_children ){
855 g_warning( "%s: target_children=%p (count=%d)",
856 thisfn,
857 ( void * ) tgt_children, g_list_length( tgt_children ));
858 g_return_if_fail( tgt_children == NULL );
859 }
860
861 src_children = na_object_get_items( source );
862 for( ic = src_children ; ic ; ic = ic->next ){
863
864 dup = ( NAObject * ) na_object_duplicate( ic->data, mode );
865 na_object_set_parent( dup, target );
866 tgt_children = g_list_prepend( tgt_children, dup );
867 }
868
869 tgt_children = g_list_reverse( tgt_children );
870 na_object_set_items( target, tgt_children );
871}

882gboolean
883na_object_item_is_finally_writable( const NAObjectItem *item, guint *reason )
884{
885 gboolean writable;
886
