99 * return the default layout file path, making it overridable by
100 * distributions

102static gchar *
103panel_layout_filename (void)
104{
105 GSettings *settings;
106 gchar *layout;
107 gchar *filename;
108
109 settings = g_settings_new (PANEL_SCHEMA);
110 layout = g_settings_get_string (settings, PANEL_DEFAULT_LAYOUT);
111 if (g_str_has_prefix (layout, "/")) {
112 filename = g_strdup (layout);
113 }
114 else {
115 filename = g_strdup_printf (PANEL_LAYOUTS_DIR "%s.layout", layout);
116 }
117 g_free (layout);
118 g_object_unref (settings);
119
120 if (g_file_test (filename, G_FILE_TEST_IS_REGULAR)) {
121 return filename;
122 }
123 else {
124 g_free (filename);
125 return NULL;
126 }
127}

129static gboolean
130panel_layout_append_group_helper (GKeyFile *keyfile,
131 const char *group,
132 int set_screen_to,
133 const char *group_prefix,
134 char *id_list_key,
135 const char *schema,
136 const char *path_prefix,
137 const char *default_prefix,
138 PanelLayoutKeyDefinition *key_definitions,
139 int key_definitions_len,
140 const char *type_for_error_message)
141{
142 gboolean retval = FALSE;
143 GError *error = NULL;
144 gboolean existing_id = FALSE;
145 const gchar *dconf_path;
146 const char *id;
147 gchar **existing_ids;
148 char *unique_id;
149 char **keyfile_keys;
150 GSettings *settings;
151
152 PanelGSettingsKeyType type;
153
154 /* Try to extract an id from the group, by stripping the prefix,
155 * and create a unique id out of that */
156 id = group + strlen (group_prefix);
157 while (g_ascii_isspace (*id))
158 id++;
159
160 if (!*id)
161 id = NULL;
162
163 if (id && !mate_gsettings_is_valid_keyname (id, &error)) {
164 g_warning ("Invalid id name in layout '%s' (%s)", id, error->message);
165 g_error_free (error);
166 return FALSE;
167 }
168
169 if (g_strcmp0 (id_list_key, PANEL_TOPLEVEL_ID_LIST_KEY) == 0) {
170 dconf_path = PANEL_RESOURCE_PATH "/toplevels";
171 type = PANEL_GSETTINGS_TOPLEVELS;
172 }
173 else if (g_strcmp0 (id_list_key, PANEL_OBJECT_ID_LIST_KEY) == 0) {
174 dconf_path = PANEL_RESOURCE_PATH "/objects";
175 type = PANEL_GSETTINGS_OBJECTS;
176 }
177 else {
178 g_critical ("Unknown key \"%s\"", id_list_key);
179 return FALSE;
180 }
181
182 existing_ids = mate_dconf_list_subdirs (dconf_path, TRUE);
183
184 if (id) {
185 int i;
186
187 if (set_screen_to > 0)
188 id = g_strdup_printf ("%s-screen%d", id, set_screen_to);
189
190 for (i = 0; existing_ids[i]; i++) {
191 if (!strcmp (existing_ids[i], id)) {
192 existing_id = TRUE;
193 break;
194 }
195 }
196 }
197 g_strfreev (existing_ids);
198
199 if (existing_id || !id)
200 unique_id = panel_profile_find_new_id (type);
201 else
202 unique_id = g_strdup (id);
203
204 char *path = g_strdup_printf ("%s%s/", path_prefix, unique_id);
205 settings = g_settings_new_with_path (schema, path);
206 g_free (path);
207
208 keyfile_keys = g_key_file_get_keys (keyfile, group, NULL, NULL);
209
210 if (keyfile_keys) {
211 int i;
212
213 /* validate/add keys from the keyfile */
214 for (i = 0; keyfile_keys[i] != NULL; i++) {
215 gboolean found = FALSE;
216 int j;
217
218 for (j = 0; j < key_definitions_len; j++) {
219 if (g_strcmp0 (keyfile_keys[i],
220 key_definitions[j].name) == 0) {
221 found = TRUE;
222 break;
223 }
224 }
225
226 if (!found) {
227 g_warning ("Unknown key '%s' for %s",
228 keyfile_keys[i],
229 unique_id);
230 return FALSE;
231 }
232
233 switch (key_definitions[j].type) {
234 case G_TYPE_STRING: {
235 char *value_str =
236 g_key_file_get_string (keyfile,
237 group, keyfile_keys[i],
238 NULL);
239 if (value_str)
240 g_settings_set_string (settings,
241 key_definitions[j].name,
242 value_str);
243 g_free (value_str);
244 break;
245 }
246 case G_TYPE_INT: {
247 gint value_int =
248 g_key_file_get_integer (keyfile,
249 group, keyfile_keys[i],
250 NULL);
251 g_settings_set_int (settings,
252 key_definitions[j].name,
253 value_int);
254 break;
255 }
256 case G_TYPE_BOOLEAN: {
257 gboolean value_boolean =
258 g_key_file_get_boolean (keyfile,
259 group, keyfile_keys[i],
260 NULL);
261 g_settings_set_boolean (settings,
262 key_definitions[j].name,
263 value_boolean);
264 break;
265 }
266 default:
267 g_assert_not_reached ();
268 break;
269 }
270 }
271
272 if (set_screen_to != -1 &&
273 g_strcmp0 (schema, PANEL_TOPLEVEL_SCHEMA) == 0)
274 g_settings_set_int (settings,
275 PANEL_TOPLEVEL_SCREEN_KEY,
276 set_screen_to);
277
278 GSettings *panel_settings = g_settings_new (PANEL_SCHEMA);
279 mate_gsettings_append_strv (panel_settings,
280 id_list_key,
281 unique_id);
282 g_object_unref (panel_settings);
283
284 retval = TRUE;
285 }
286
287 if (keyfile_keys)
288 g_strfreev (keyfile_keys);
289
290 if (settings)
291 g_object_unref (settings);
292
293 if (unique_id)
294 g_free (unique_id);
295
296 return retval;
297}

299static void
300panel_layout_apply_minimal_default (int set_screen_to,
301 const char *schema,
302 const char *path_prefix)
303{
304 const char *unique_id = "bottom";
305 char *path = NULL;
306 GSettings *settings = NULL;
307
308 path = g_strdup_printf ("%s%s/", path_prefix, unique_id);
309 settings = g_settings_new_with_path (schema, path);
310 g_return_if_fail(settings);
311 g_free (path);
312
313 g_settings_set_boolean(settings, "expand", TRUE);
314 g_settings_set_string(settings, "orientation", "bottom");
315 g_settings_set_int(settings, "size", 24);
316 g_settings_set_int(settings, "screen", 0);
317
318 GSettings *panel_settings;
319 panel_settings = g_settings_new (PANEL_SCHEMA);
320 g_return_if_fail(panel_settings);
321 mate_gsettings_append_strv (panel_settings,
322 PANEL_TOPLEVEL_ID_LIST_KEY,
323 unique_id);
324
325 g_object_unref (panel_settings);
326 g_object_unref (settings);
327}

329void
330panel_layout_apply_default_from_gkeyfile (GdkScreen *screen)
331{
332 int screen_n;
333 gchar *layout_file = NULL;
334 GKeyFile *keyfile = NULL;
335 gchar **groups = NULL;
336 GError *error = NULL;
337
338 screen_n = 0;
339#ifdef HAVE_X11
340 if (GDK_IS_X11_SCREEN (screen))
341 screen_n = gdk_x11_screen_get_screen_number (screen);
342#endif /* HAVE_11 */
343
344 layout_file = panel_layout_filename();
345
346 if (layout_file)
347 {
348 keyfile = g_key_file_new ();
349 if (g_key_file_load_from_file (keyfile,
350 layout_file,
351 G_KEY_FILE_NONE,
352 &error))
353 {
354 int i;
355
356 groups = g_key_file_get_groups (keyfile, NULL);
357
358 for (i = 0; groups[i] != NULL; i++) {
359
360 if (g_strcmp0 (groups[i], "Toplevel") == 0 ||
361 g_str_has_prefix (groups[i], "Toplevel "))
362
363 panel_layout_append_group_helper (
364 keyfile, groups[i],
365 screen_n,
366 "Toplevel",
367 PANEL_TOPLEVEL_ID_LIST_KEY,
368 PANEL_TOPLEVEL_SCHEMA,
369 PANEL_TOPLEVEL_PATH,
370 PANEL_TOPLEVEL_DEFAULT_PREFIX,
371 panel_layout_toplevel_keys,
372 G_N_ELEMENTS (panel_layout_toplevel_keys),
373 "toplevel");
374
375 else if (g_strcmp0 (groups[i], "Object") == 0 ||
376 g_str_has_prefix (groups[i], "Object "))
377
378 panel_layout_append_group_helper (
379 keyfile, groups[i],
380 -1,
381 "Object",
382 PANEL_OBJECT_ID_LIST_KEY,
383 PANEL_OBJECT_SCHEMA,
384 PANEL_OBJECT_PATH,
385 PANEL_OBJECT_DEFAULT_PREFIX,
386 panel_layout_object_keys,
387 G_N_ELEMENTS (panel_layout_object_keys),
388 "object");
389
390 else
391
392 g_warning ("Unknown group in default layout: '%s'",
393 groups[i]);
394
395 }
396
397 }
398 else
399 {
400 g_warning ("Error while parsing default layout from '%s': %s\n",
401 layout_file, error->message);
402 g_error_free (error);
403 }
404
405 }
406 else {
407 g_warning ("Cant find the layout file!");
408 panel_layout_apply_minimal_default(screen_n,
409 PANEL_TOPLEVEL_SCHEMA,
410 PANEL_TOPLEVEL_PATH);
411 }
412
413 if (groups)
414 g_strfreev (groups);
415
416 if (keyfile)
417 g_key_file_free (keyfile);
418
419 if (layout_file)
420 g_free (layout_file);
421}
