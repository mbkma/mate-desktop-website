93static void
94caja_desktop_window_init (CajaDesktopWindow *window)
95{
96 GtkAction *action;
97 AtkObject *accessible;
98
99 window->details = caja_desktop_window_get_instance_private (window);
100
101 GtkStyleContext *context;
102
103 context = gtk_widget_get_style_context (GTK_WIDGET (window));
104 gtk_style_context_add_class (context, "caja-desktop-window");
105
106 gtk_window_move (GTK_WINDOW (window), 0, 0);
107
108 /* shouldn't really be needed given our semantic type
109 * of _NET_WM_TYPE_DESKTOP, but why not
110 */
111 gtk_window_set_resizable (GTK_WINDOW (window),
112 FALSE);
113
114 g_object_set_data (G_OBJECT (window), "is_desktop_window",
115 GINT_TO_POINTER (1));
116
117 gtk_widget_hide (CAJA_WINDOW (window)->details->statusbar);
118 gtk_widget_hide (CAJA_WINDOW (window)->details->menubar);
119
120 /* Don't allow close action on desktop */
121 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
122 action = gtk_action_group_get_action (CAJA_WINDOW (window)->details->main_action_group,
123 CAJA_ACTION_CLOSE);
124 gtk_action_set_sensitive (action, FALSE);
125 G_GNUC_END_IGNORE_DEPRECATIONS;
126
127 /* Set the accessible name so that it doesn't inherit the cryptic desktop URI. */
128 accessible = gtk_widget_get_accessible (GTK_WIDGET (window));
129
130 if (accessible) {
131 atk_object_set_name (accessible, _("Desktop"));
132 }
133}

156static void
157caja_desktop_window_screen_size_changed (GdkScreen *screen,
158 CajaDesktopWindow *window)
159{
160 int width_request, height_request;
161
162 GdkDisplay *display = gdk_screen_get_display (screen);
163 if (GDK_IS_X11_DISPLAY (display))
164 {
165 GdkWindow *root_window;
166 root_window = gdk_screen_get_root_window (screen);
167 gdk_window_get_geometry (root_window, NULL, NULL, &width_request, &height_request);
168 }
169 else
170 {
171 /*No root window or primary monitor in wayland unless compositors add it back*/
172 GdkRectangle geometry = {0};
173 GdkMonitor *monitor;
174 monitor = gdk_display_get_monitor (display, 0);
175 gdk_monitor_get_geometry (monitor, &geometry);
176 width_request = geometry.width;
177 height_request = geometry.height;
178 }
179
180 g_object_set (window,
181 "width_request", width_request,
182 "height_request", height_request,
183 NULL);
184}

186CajaDesktopWindow *
187caja_desktop_window_new (CajaApplication *application,
188 GdkScreen *screen)
189{
190 CajaDesktopWindow *window;
191 int width_request, height_request;
192 int scale;
193
194 GdkDisplay *display = gdk_screen_get_display (screen);
195 if (GDK_IS_X11_DISPLAY (display))
196 {
197 scale = gdk_window_get_scale_factor (gdk_screen_get_root_window (screen));
198 width_request = WidthOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale;
199 height_request = HeightOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale;
200 }
201 else
202 {
203 /*FIXME: There is no primary monitor in wayland itself
204 *compositors can implement this but as this is written
205 *only a few wayland compositors allow setting a primary monitor
206 *and they all do it differently. For now, use the first monitor
207 */
208 GdkRectangle geometry = {0};
209 GdkMonitor *monitor;
210 monitor = gdk_display_get_monitor (display, 0);
211 gdk_monitor_get_geometry (monitor, &geometry);
212 width_request = geometry.width;
213 height_request = geometry.height;
214 }
215
216 window = CAJA_DESKTOP_WINDOW
217 (gtk_widget_new (caja_desktop_window_get_type(),
218 "app", application,
219 "width_request", width_request,
220 "height_request", height_request,
221 "screen", screen,
222 "decorated", FALSE,
223 NULL));
224 /* Stop wrong desktop window size in GTK 3.20*/
225 /* We don't want to set a default size, which the parent does, since this */
226 /* will cause the desktop window to open at the wrong size in gtk 3.20 */
227 gtk_window_set_default_size (GTK_WINDOW (window), -1, -1);
228
229 /*For wayland only
230 *Code taken from gtk-layer-shell simple-example.c
231 */
232#ifdef HAVE_WAYLAND
233 if (GDK_IS_WAYLAND_DISPLAY (display))
234 {
235 GtkWindow *gtkwin;
236 gtkwin = (GTK_WINDOW(window));
237
238 /* Before the window is first realized, set it up to be a layer surface */
239 gtk_layer_init_for_window (gtkwin);
240
241 /* Order below normal windows */
242 gtk_layer_set_layer (gtkwin, GTK_LAYER_SHELL_LAYER_BACKGROUND);
243
244 gtk_layer_set_namespace (gtkwin, "desktop");
245
246 /*Anchor the desktop to all four corners
247 *This is much simpler than on x11 and
248 *should always render the desktop across
249 *all of the screen
250 */
251 gtk_layer_set_anchor (gtkwin, GTK_LAYER_SHELL_EDGE_TOP, TRUE);
252 gtk_layer_set_anchor (gtkwin, GTK_LAYER_SHELL_EDGE_BOTTOM, TRUE);
253 gtk_layer_set_anchor (gtkwin, GTK_LAYER_SHELL_EDGE_LEFT, TRUE);
254 gtk_layer_set_anchor (gtkwin, GTK_LAYER_SHELL_EDGE_RIGHT, TRUE);
255
256 /*Enable keyboard use on the desktop*/
257 gtk_layer_set_keyboard_mode (gtkwin, GTK_LAYER_SHELL_KEYBOARD_MODE_ON_DEMAND);
258 }
259#endif
260 /* Special sawmill setting*/
261 GdkWindow *gdkwin;
262 if ((GDK_IS_X11_DISPLAY (display)))
263 gtk_widget_realize (GTK_WIDGET (window));
264 else
265 gtk_widget_show (GTK_WIDGET (window));
266
267 gdkwin = gtk_widget_get_window (GTK_WIDGET (window));
268 if ((GDK_IS_X11_DISPLAY (display)) && (gdk_window_ensure_native (gdkwin)))
269 {
270 Display *disp = GDK_DISPLAY_XDISPLAY (gdk_window_get_display (gdkwin));
271 XClassHint *xch = XAllocClassHint ();
272 xch->res_name = "desktop_window";
273 xch->res_class = "Caja";
274 XSetClassHint (disp, GDK_WINDOW_XID(gdkwin), xch);
275 XFree(xch);
276 }
277
278 gdk_window_set_title (gdkwin, _("Desktop"));
279
280 g_signal_connect (window, "delete_event", G_CALLBACK (caja_desktop_window_delete_event), NULL);
281
282 /* Point window at the desktop folder.
283 * Note that caja_desktop_window_init is too early to do this.
284 */
285 caja_desktop_window_update_directory (window);
286
287 return window;
288}

298static void
299unrealize (GtkWidget *widget)
300{
301 CajaDesktopWindow *window;
302 CajaDesktopWindowPrivate *details;
303 GdkWindow *root_window;
304 GdkDisplay *display;
305
306 window = CAJA_DESKTOP_WINDOW (widget);
307 details = window->details;
308 display = gtk_widget_get_display (widget);
309
310 /*Avoid root window on wayland-it's not supposed to work*/
311 if (GDK_IS_X11_DISPLAY (display))
312 {
313 root_window = gdk_screen_get_root_window (
314 gtk_window_get_screen (GTK_WINDOW (window)));
315
316 gdk_property_delete (root_window,
317 gdk_atom_intern ("CAJA_DESKTOP_WINDOW_ID", TRUE));
318 }
319
320 if (details->size_changed_id != 0) {
321 g_signal_handler_disconnect (gtk_window_get_screen (GTK_WINDOW (window)),
322 details->size_changed_id);
323 details->size_changed_id = 0;
324 }
325
326 GTK_WIDGET_CLASS (caja_desktop_window_parent_class)->unrealize (widget);
327}

344static void
345set_desktop_window_id (CajaDesktopWindow *window,
346 GdkWindow *gdkwindow)
347{
348 /* Tuck the desktop windows xid in the root to indicate we own the desktop in on x11
349 */
350 Window window_xid;
351 GdkWindow *root_window;
352
353 root_window = gdk_screen_get_root_window (
354 gtk_window_get_screen (GTK_WINDOW (window)));
355 window_xid = GDK_WINDOW_XID (gdkwindow);
356
357 gdk_property_change (root_window,
358 gdk_atom_intern ("CAJA_DESKTOP_WINDOW_ID", FALSE),
359 gdk_x11_xatom_to_atom (XA_WINDOW), 32,
360 GDK_PROP_MODE_REPLACE, (guchar *) &window_xid, 1);
361}

363static void
364realize (GtkWidget *widget)
365{
366 CajaDesktopWindow *window;
367 CajaDesktopWindowPrivate *details;
368 window = CAJA_DESKTOP_WINDOW (widget);
369 details = window->details;
370 GdkDisplay *display;
371
372 /* Make sure we get keyboard events */
373 display = gtk_widget_get_display (widget);
374 if (GDK_IS_X11_DISPLAY (display))
375 gtk_widget_set_events (widget, gtk_widget_get_events (widget)
376 | GDK_KEY_PRESS_MASK | GDK_KEY_RELEASE_MASK);
377
378 /*Do the work of realizing. */
379 GTK_WIDGET_CLASS (caja_desktop_window_parent_class)->realize (widget);
380
381 /* This is the new way to set up the desktop window in x11 but not for wayland */
382 display = gtk_widget_get_display (widget);
383 if (GDK_IS_X11_DISPLAY (display))
384 {
385 set_wmspec_desktop_hint (gtk_widget_get_window (widget));
386 set_desktop_window_id (window, gtk_widget_get_window (widget));
387 }
388
389 details->size_changed_id =
390 g_signal_connect (gtk_window_get_screen (GTK_WINDOW (window)), "size_changed",
391 G_CALLBACK (caja_desktop_window_screen_size_changed), window);
392}

413caja_desktop_window_class_init (CajaDesktopWindowClass *klass)
414{
415 GtkWidgetClass *wclass = GTK_WIDGET_CLASS (klass);
416 CajaWindowClass *nclass = CAJA_WINDOW_CLASS (klass);
417

428
429 nclass->window_type = CAJA_WINDOW_DESKTOP;
430 nclass->get_icon = real_get_icon;
431}
