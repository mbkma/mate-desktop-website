43static void
44log_level_to_priority_and_prefix (GLogLevelFlags log_level,
45 int *priorityp,
46 const char **prefixp)
47{
48 int priority;
49 const char *prefix;
50
51 /* Process the message prefix and priority */
52 switch (log_level & G_LOG_LEVEL_MASK) {
53 case G_LOG_FLAG_FATAL:
54 priority = LOG_EMERG;
55 prefix = "FATAL";
56 break;
57 case G_LOG_LEVEL_ERROR:
58 priority = LOG_ERR;
59 prefix = "ERROR";
60 break;
61 case G_LOG_LEVEL_CRITICAL:
62 priority = LOG_CRIT;
63 prefix = "CRITICAL";
64 break;
65 case G_LOG_LEVEL_WARNING:
66 priority = LOG_WARNING;
67 prefix = "WARNING";
68 break;
69 case G_LOG_LEVEL_MESSAGE:
70 priority = LOG_NOTICE;
71 prefix = "MESSAGE";
72 break;
73 case G_LOG_LEVEL_INFO:
74 priority = LOG_INFO;
75 prefix = "INFO";
76 break;
77 case G_LOG_LEVEL_DEBUG:
78 /* if debug was requested then bump this up to ERROR
79 * to ensure it is seen in a log */
80 if (syslog_levels & G_LOG_LEVEL_DEBUG) {
81 priority = LOG_WARNING;
82 prefix = "DEBUG(+)";
83 } else {
84 priority = LOG_DEBUG;
85 prefix = "DEBUG";
86 }
87 break;
88 default:
89 priority = LOG_DEBUG;
90 prefix = "UNKNOWN";
91 break;
92 }
93
94 if (priorityp != NULL) {
95 *priorityp = priority;
96 }
97 if (prefixp != NULL) {
98 *prefixp = prefix;
99 }
100}

102void
103mdm_log_default_handler (const gchar *log_domain,
104 GLogLevelFlags log_level,
105 const gchar *message,
106 gpointer unused_data)
107{
108 GString *gstring;
109 int priority;
110 const char *level_prefix;
111 char *string;
112 gboolean do_log;
113 gboolean is_fatal;
114
115 is_fatal = (log_level & G_LOG_FLAG_FATAL) != 0;
116
117 do_log = (log_level & syslog_levels);
118 if (! do_log) {
119 return;
120 }
121
122 if (! initialized) {
123 mdm_log_init ();
124 }
125
126 log_level_to_priority_and_prefix (log_level,
127 &priority,
128 &level_prefix);
129
130 gstring = g_string_new (NULL);
131
132 if (log_domain != NULL) {
133 g_string_append (gstring, log_domain);
134 g_string_append_c (gstring, '-');
135 }
136 g_string_append (gstring, level_prefix);
137
138 g_string_append (gstring, ": ");
139 if (message == NULL) {
140 g_string_append (gstring, "(NULL) message");
141 } else {
142 g_string_append (gstring, message);
143 }
144 if (is_fatal) {
145 g_string_append (gstring, "\naborting...\n");
146 } else {
147 g_string_append (gstring, "\n");
148 }
149
150 string = g_string_free (gstring, FALSE);
151
152 syslog (priority, "%s", string);
153
154 g_free (string);
155}

157void
158mdm_log_toggle_debug (void)
159{
160 if (syslog_levels & G_LOG_LEVEL_DEBUG) {
161 g_debug ("Debugging disabled");
162 syslog_levels &= ~G_LOG_LEVEL_DEBUG;
163 } else {
164 syslog_levels |= G_LOG_LEVEL_DEBUG;
165 g_debug ("Debugging enabled");
166 }
167}

169void
170mdm_log_set_debug (gboolean debug)
171{
172 if (debug) {
173 syslog_levels |= G_LOG_LEVEL_DEBUG;
174 g_debug ("Enabling debugging");
175 } else {
176 g_debug ("Disabling debugging");
177 syslog_levels &= ~G_LOG_LEVEL_DEBUG;
178 }
179}

181void
182mdm_log_init (void)
183{
184 const char *prg_name;
185 int options;
186
187 g_log_set_default_handler (mdm_log_default_handler, NULL);
188
189 prg_name = g_get_prgname ();
190
191 options = LOG_PID;
192#ifdef LOG_PERROR
193 options |= LOG_PERROR;
194#endif
195
196 openlog (prg_name, options, LOG_DAEMON);
197
198 initialized = TRUE;
199}
