143static gboolean
144emit_finished (EvJob *job)
145{
146 ev_debug_message (DEBUG_JOBS, "%s (%p)", EV_GET_TYPE_NAME (job), job);
147
148 job->idle_finished_id = 0;
149
150 if (job->cancelled) {
151 ev_debug_message (DEBUG_JOBS, "%s (%p) job was cancelled, do not emit finished", EV_GET_TYPE_NAME (job), job);
152 } else {
153 ev_profiler_stop (EV_PROFILE_JOBS, "%s (%p)", EV_GET_TYPE_NAME (job), job);
154 g_signal_emit (job, job_signals[FINISHED], 0);
155 }
156
157 return FALSE;
158}

160static void
161ev_job_emit_finished (EvJob *job)
162{
163 ev_debug_message (DEBUG_JOBS, "%s (%p)", EV_GET_TYPE_NAME (job), job);
164
165 if (g_cancellable_is_cancelled (job->cancellable)) {
166 ev_debug_message (DEBUG_JOBS, "%s (%p) job was cancelled, returning", EV_GET_TYPE_NAME (job), job);
167 return;
168 }
169
170 job->finished = TRUE;
171
172 if (job->run_mode == EV_JOB_RUN_THREAD) {
173 job->idle_finished_id =
174 g_idle_add_full (G_PRIORITY_DEFAULT_IDLE,
175 (GSourceFunc)emit_finished,
176 g_object_ref (job),
177 (GDestroyNotify)g_object_unref);
178 } else {
179 ev_profiler_stop (EV_PROFILE_JOBS, "%s (%p)", EV_GET_TYPE_NAME (job), job);
180 g_signal_emit (job, job_signals[FINISHED], 0);
181 }
182}

192void
193ev_job_cancel (EvJob *job)
194{
195 if (job->cancelled)
196 return;
197
198 ev_debug_message (DEBUG_JOBS, "job %s (%p) cancelled", EV_GET_TYPE_NAME (job), job);
199 ev_profiler_stop (EV_PROFILE_JOBS, "%s (%p)", EV_GET_TYPE_NAME (job), job);
200
201 /* This should never be called from a thread */
202 job->cancelled = TRUE;
203 g_cancellable_cancel (job->cancellable);
204
205 if (job->finished && job->idle_finished_id == 0)
206 return;
207
208 g_signal_emit (job, job_signals[CANCELLED], 0);
209}

211void
212ev_job_failed (EvJob *job,
213 GQuark domain,
214 gint code,
215 const gchar *format,
216 ...)
217{
218 va_list args;
219 gchar *message;
220
221 if (job->failed || job->finished)
222 return;
223
224 ev_debug_message (DEBUG_JOBS, "job %s (%p) failed", EV_GET_TYPE_NAME (job), job);
225
226 job->failed = TRUE;
227
228 va_start (args, format);
229 message = g_strdup_vprintf (format, args);
230 va_end (args);
231
232 job->error = g_error_new_literal (domain, code, message);
233 g_free (message);
234
235 ev_job_emit_finished (job);
236}

486static gboolean
487ev_job_annots_run (EvJob *job)
488{
489 EvJobAnnots *job_annots = EV_JOB_ANNOTS (job);
490 gint i;
491
492 ev_debug_message (DEBUG_JOBS, NULL);
493 ev_profiler_start (EV_PROFILE_JOBS, "%s (%p)", EV_GET_TYPE_NAME (job), job);
494
495 ev_document_doc_mutex_lock ();
496 for (i = 0; i < ev_document_get_n_pages (job->document); i++) {
497 EvMappingList *mapping_list;
498 EvPage *page;
499
500 page = ev_document_get_page (job->document, i);
501 mapping_list = ev_document_annotations_get_annotations (EV_DOCUMENT_ANNOTATIONS (job->document),
502 page);
503 g_object_unref (page);
504
505 if (mapping_list)
506 job_annots->annots = g_list_prepend (job_annots->annots, mapping_list);
507 }
508 ev_document_doc_mutex_unlock ();
509
510 job_annots->annots = g_list_reverse (job_annots->annots);
511
512 ev_job_succeeded (job);
513
514 return FALSE;
515}

574static gboolean
575ev_job_render_run (EvJob *job)
576{
577 EvJobRender *job_render = EV_JOB_RENDER (job);
578 EvPage *ev_page;
579 EvRenderContext *rc;
580
581 ev_debug_message (DEBUG_JOBS, "page: %d (%p)", job_render->page, job);
582 ev_profiler_start (EV_PROFILE_JOBS, "%s (%p)", EV_GET_TYPE_NAME (job), job);
583
584 ev_document_doc_mutex_lock ();
585
586 ev_profiler_start (EV_PROFILE_JOBS, "Rendering page %d", job_render->page);
587
588 ev_document_fc_mutex_lock ();
589
590 ev_page = ev_document_get_page (job->document, job_render->page);
591
592 if ( job->document->iswebdocument == TRUE )
593 {
594 return TRUE;
595
596 if (g_cancellable_is_cancelled (job->cancellable)) {
597 ev_document_fc_mutex_unlock ();
598 ev_document_doc_mutex_unlock ();
599 g_object_unref (rc);
600
601 return FALSE;
602 }
603
604 ev_document_fc_mutex_unlock ();
605 ev_document_doc_mutex_unlock ();
606 ev_job_succeeded (job);
607 return FALSE;
608 }
609 rc = ev_render_context_new (ev_page, job_render->rotation, job_render->scale);
610 g_object_unref (ev_page);
611
612 if ((job_render->surface = ev_document_render (job->document, rc)) == NULL) {
613 ev_document_fc_mutex_unlock ();
614 ev_document_doc_mutex_unlock ();
615 g_object_unref (rc);
616 ev_job_failed (job,
617 EV_DOCUMENT_ERROR,
618 EV_DOCUMENT_ERROR_INVALID,
619 _("Failed to render page %d"),
620 job_render->page);
621 return FALSE;
622 }
623
624 /* If job was cancelled during the page rendering,
625 * we return now, so that the thread is finished ASAP
626 */
627 if (g_cancellable_is_cancelled (job->cancellable)) {
628 ev_document_fc_mutex_unlock ();
629 ev_document_doc_mutex_unlock ();
630 g_object_unref (rc);
631
632 return FALSE;
633 }
634
635 if (job_render->include_selection && EV_IS_SELECTION (job->document)) {
636 ev_selection_render_selection (EV_SELECTION (job->document),
637 rc,
638 &(job_render->selection),
639 &(job_render->selection_points),
640 NULL,
641 job_render->selection_style,
642 &(job_render->text), &(job_render->base));
643 job_render->selection_region =
644 ev_selection_get_selection_region (EV_SELECTION (job->document),
645 rc,
646 job_render->selection_style,
647 &(job_render->selection_points));
648 }
649
650 g_object_unref (rc);
651
652 ev_document_fc_mutex_unlock ();
653 ev_document_doc_mutex_unlock ();
654
655 ev_job_succeeded (job);
656
657 return FALSE;
658}

716static gboolean
717ev_job_page_data_run (EvJob *job)
718{
719 EvJobPageData *job_pd = EV_JOB_PAGE_DATA (job);
720 EvPage *ev_page;
721
722 ev_debug_message (DEBUG_JOBS, "page: %d (%p)", job_pd->page, job);
723 ev_profiler_start (EV_PROFILE_JOBS, "%s (%p)", EV_GET_TYPE_NAME (job), job);
724
725 ev_document_doc_mutex_lock ();
726 ev_page = ev_document_get_page (job->document, job_pd->page);
727
728 if ((job_pd->flags & EV_PAGE_DATA_INCLUDE_TEXT_MAPPING) && EV_IS_DOCUMENT_TEXT (job->document))
729 job_pd->text_mapping =
730 ev_document_text_get_text_mapping (EV_DOCUMENT_TEXT (job->document), ev_page);
731 if ((job_pd->flags & EV_PAGE_DATA_INCLUDE_TEXT) && EV_IS_DOCUMENT_TEXT (job->document))
732 job_pd->text =
733 ev_document_text_get_text (EV_DOCUMENT_TEXT (job->document), ev_page);
734 if ((job_pd->flags & EV_PAGE_DATA_INCLUDE_TEXT_LAYOUT) && EV_IS_DOCUMENT_TEXT (job->document))
735 ev_document_text_get_text_layout (EV_DOCUMENT_TEXT (job->document),
736 ev_page,
737 &(job_pd->text_layout),
738 &(job_pd->text_layout_length));
739 if ((job_pd->flags & EV_PAGE_DATA_INCLUDE_TEXT_ATTRS) && EV_IS_DOCUMENT_TEXT (job->document))
740 job_pd ->text_attrs =
741 ev_document_text_get_text_attrs (EV_DOCUMENT_TEXT (job->document),
742 ev_page);
743 if ((job_pd->flags & EV_PAGE_DATA_INCLUDE_TEXT_LOG_ATTRS) && job_pd->text) {
744 job_pd->text_log_attrs_length = g_utf8_strlen (job_pd->text, -1);
745 job_pd->text_log_attrs = g_new0 (PangoLogAttr, job_pd->text_log_attrs_length + 1);

1100static gboolean
1101ev_job_save_run (EvJob *job)
1102{
1103 EvJobSave *job_save = EV_JOB_SAVE (job);
1104 gint fd;
1105 gchar *tmp_filename = NULL;
1106 gchar *local_uri;
1107 GError *error = NULL;
1108
1109 ev_debug_message (DEBUG_JOBS, "uri: %s, document_uri: %s", job_save->uri, job_save->document_uri);
1110 ev_profiler_start (EV_PROFILE_JOBS, "%s (%p)", EV_GET_TYPE_NAME (job), job);
1111
1112 fd = ev_mkstemp ("saveacopy.XXXXXX", &tmp_filename, &error);
1113 if (fd == -1) {
1114 ev_job_failed_from_error (job, error);
1115 g_error_free (error);
1116
1117 return FALSE;
1118 }
1119
1120 ev_document_doc_mutex_lock ();
1121
1122 /* Save document to temp filename */
1123 local_uri = g_filename_to_uri (tmp_filename, NULL, &error);
1124 if (local_uri != NULL) {
1125 ev_document_save (job->document, local_uri, &error);
1126 }
1127
1128 close (fd);
1129
1130 ev_document_doc_mutex_unlock ();
1131
1132 if (error) {
1133 g_free (local_uri);
1134 ev_job_failed_from_error (job, error);
1135 g_error_free (error);
1136
1137 return FALSE;
1138 }
1139
1140 /* If original document was compressed,
1141 * compress it again before saving
1142 */
1143 if (g_object_get_data (G_OBJECT (job->document), "uri-uncompressed")) {
1144 EvCompressionType ctype = EV_COMPRESSION_NONE;
1145 const gchar *ext;
1146 gchar *uri_comp;
1147
1148 ext = g_strrstr (job_save->document_uri, ".gz");
1149 if (ext && g_ascii_strcasecmp (ext, ".gz") == 0)
1150 ctype = EV_COMPRESSION_GZIP;
1151
1152 ext = g_strrstr (job_save->document_uri, ".bz2");
1153 if (ext && g_ascii_strcasecmp (ext, ".bz2") == 0)
1154 ctype = EV_COMPRESSION_BZIP2;
1155
1156 uri_comp = ev_file_compress (local_uri, ctype, &error);
1157 g_free (local_uri);
1158 g_unlink (tmp_filename);
1159
1160 if (!uri_comp || error) {
1161 local_uri = NULL;
1162 } else {
1163 local_uri = uri_comp;
1164 }
1165 }
1166
1167 g_free (tmp_filename);
1168
1169 if (error) {
1170 g_free (local_uri);
1171 ev_job_failed_from_error (job, error);
1172 g_error_free (error);
1173
1174 return FALSE;
1175 }
1176
1177 if (!local_uri)
1178 return FALSE;
1179
1180 ev_xfer_uri_simple (local_uri, job_save->uri, &error);
1181 ev_tmp_uri_unlink (local_uri);
1182
1183 if (error) {
1184 ev_job_failed_from_error (job, error);
1185 g_error_free (error);
1186 } else {
1187 ev_job_succeeded (job);
1188 }
1189
1190 return FALSE;
1191}

1281 job_find->results[job_find->current_page] = ev_document_find_check_for_hits(find, ev_page, job_find->text,

1296 job_find->has_results = (job_find->results[job_find->current_page] > 0);
1297 }

1299 if (job->document->iswebdocument == FALSE) {
1300 job_find->pages[job_find->current_page] = matches;

1303 g_signal_emit (job_find, job_find_signals[FIND_UPDATED], 0, job_find->current_page);
