137EvFormField *
138ev_form_field_text_new (gint id,
139 EvFormFieldTextType type)
140{
141 EvFormField *field;
142
143 g_return_val_if_fail (id >= 0, NULL);
144 g_return_val_if_fail (type >= EV_FORM_FIELD_TEXT_NORMAL &&
145 type <= EV_FORM_FIELD_TEXT_FILE_SELECT, NULL);
146
147 field = EV_FORM_FIELD (g_object_new (EV_TYPE_FORM_FIELD_TEXT, NULL));
148 field->id = id;
149 EV_FORM_FIELD_TEXT (field)->type = type;
150
151 return field;
