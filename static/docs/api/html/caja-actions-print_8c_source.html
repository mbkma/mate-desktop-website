83int
84main( int argc, char** argv )
85{
86 int status = EXIT_SUCCESS;
87 GOptionContext *context;
88 GError *error = NULL;
89 gchar *help;
90 gint errors;
91 NAObjectItem *item;
92 NAIExporter *exporter;
93
94 setlocale( LC_ALL, "" );
95 console_init_log_handler();
96
97 context = init_options();
98
99 if( argc == 1 ){
100 g_set_prgname( argv[0] );
101 help = g_option_context_get_help( context, FALSE, NULL );
102 g_print( "\n%s", help );
103 g_free( help );
104 exit( status );
105 }
106
107 if( !g_option_context_parse( context, &argc, &argv, &error )){
108 g_printerr( _( "Syntax error: %s\n" ), error->message );
109 g_error_free (error);
110 exit_with_usage();
111 }
112
113 g_option_context_free( context );
114
115 if( version ){
116 na_core_utils_print_version();
117 exit( status );
118 }
119
120 errors = 0;
121
122 if( !id || !strlen( id )){
123 g_printerr( _( "Error: a menu or action id is mandatory.\n" ));
124 errors += 1;
125 }
126
127 item = get_item( id );
128 if( !item ){
129 errors += 1;
130 }
131
132 if( !format || !strlen( format )){
133 format = "Desktop1";
134 }
135
136 exporter = na_exporter_find_for_format( pivot, format );
137 if( !exporter ){
138 /* Translators: %s stands for the id of the export format, and is not translatable */
139 g_printerr( _( "Error: %s: unknown export format.\n" ), format );
140 errors += 1;
141 }
142
143 if( errors ){
144 exit_with_usage();
145 }
146
147 export_item( item, format );
148
149 exit( status );
150}

155static GOptionContext *
156init_options( void )
157{
158 GOptionContext *context;
159 gchar* description;
160 GOptionGroup *misc_group;
161
162 context = g_option_context_new( program_summary );
163 g_option_context_set_translation_domain( context, GETTEXT_PACKAGE );
164
165#ifdef ENABLE_NLS
166 bindtextdomain( GETTEXT_PACKAGE, MATELOCALEDIR );
167# ifdef HAVE_BIND_TEXTDOMAIN_CODESET
168 bind_textdomain_codeset( GETTEXT_PACKAGE, "UTF-8" );
169# endif
170 textdomain( GETTEXT_PACKAGE );
171 g_option_context_add_main_entries( context, entries, GETTEXT_PACKAGE );
172#else
173 g_option_context_add_main_entries( context, entries, NULL );
174#endif
175
176 description = console_cmdline_get_description();
177 g_option_context_set_description( context, description );
178 g_free( description );
179
180 misc_group = g_option_group_new(
181 "misc", _( "Miscellaneous options" ), _( "Miscellaneous options" ), NULL, NULL );
182 g_option_group_add_entries( misc_group, misc_entries );
183 g_option_group_set_translation_domain( misc_group, GETTEXT_PACKAGE );
184 g_option_context_add_group( context, misc_group );
185
186 return( context );
187}

189/*
190 * search for the action in the repository
191 */
192static NAObjectItem *
193get_item (const gchar *item_id)
194{
195 NAObjectItem *item = NULL;
196
197 pivot = na_pivot_new ();
198 na_pivot_set_loadable (pivot, PIVOT_LOAD_ALL);
199 na_pivot_load_items (pivot);
200
201 if ((item = na_pivot_get_item (pivot, item_id)) == NULL) {
202 g_printerr (_( "Error: item '%s' doesn't exist.\n" ), item_id);
203 }
204
205 return item;
206}

208/*
209 * displays the specified item on stdout, in the specified export format
210 */
211static void
212export_item (const NAObjectItem *item,
213 const gchar *format_str)
214{
215 GSList *messages = NULL;
216 GSList *it;
217
218 gchar *buffer = na_exporter_to_buffer( pivot, item, format_str, &messages );
219
220 for( it = messages ; it ; it = it->next ){
221 g_printerr( "%s\n", ( const gchar * ) it->data );
222 }
223 na_core_utils_slist_free( messages );
224
225 if( buffer ){
226 g_printf( "%s\n", buffer );
227 g_free( buffer );
228 }
229}
