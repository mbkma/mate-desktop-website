131static void
132update_app_indicator (DrWright *dr)
133{
134 AppIndicatorStatus new_status;
135
136 if (!dr->enabled) {
137 app_indicator_set_status (dr->indicator,
138 APP_INDICATOR_STATUS_PASSIVE);
139 return;
140 }
141
142 switch (dr->state) {
143 case STATE_WARN:
144 case STATE_BREAK_SETUP:
145 case STATE_BREAK:
146 new_status = APP_INDICATOR_STATUS_ATTENTION;
147 break;
148 default:
149 new_status = APP_INDICATOR_STATUS_ACTIVE;
150 }
151
152 app_indicator_set_status (dr->indicator, new_status);
153}

192static gboolean
193maybe_change_state (DrWright *dr)
194{
195 gint elapsed_time;
196 gint elapsed_idle_time;
197
198 if (debug) {
199 drw_timer_start (dr->idle_timer);
200 }
201
202 elapsed_time = drw_timer_elapsed (dr->timer) + dr->save_last_time;
203 elapsed_idle_time = drw_timer_elapsed (dr->idle_timer);
204
205 if (elapsed_time > dr->last_elapsed_time + dr->warn_time) {
206 /* If the timeout is delayed by the amount of warning time, then
207 * we must have been suspended or stopped, so we just start
208 * over.
209 */
210 dr->state = STATE_START;
211 }
212
213 switch (dr->state) {
214 case STATE_START:
215 if (dr->break_window) {
216 gtk_widget_destroy (dr->break_window);
217 dr->break_window = NULL;
218 }
219
220 dr->save_last_time = 0;
221
222 drw_timer_start (dr->timer);
223 drw_timer_start (dr->idle_timer);
224
225 if (dr->enabled) {
226 dr->state = STATE_RUNNING;
227 }
228
229 update_status (dr);
230 break;
231
232 case STATE_RUNNING:
233 case STATE_WARN:
234 if (elapsed_idle_time >= dr->break_time) {
235 dr->state = STATE_BREAK_DONE_SETUP;
236 } else if (elapsed_time >= dr->type_time) {
237 dr->state = STATE_BREAK_SETUP;
238 } else if (dr->state != STATE_WARN
239 && elapsed_time >= dr->type_time - dr->warn_time) {
240 dr->state = STATE_WARN;
241 }
242 break;
243
244 case STATE_BREAK_SETUP:
245 /* Don't allow more than one break window to coexist, can happen
246 * if a break is manually enforced.
247 */
248 if (dr->break_window) {
249 dr->state = STATE_BREAK;
250 break;
251 }
252
253 drw_timer_start (dr->timer);
254
255 dr->break_window = drw_break_window_new ();
256
257 g_signal_connect (dr->break_window, "map_event",
258 G_CALLBACK (break_window_map_event_cb),
259 dr);
260
261 g_signal_connect (dr->break_window,
262 "done",
263 G_CALLBACK (break_window_done_cb),
264 dr);
265
266 g_signal_connect (dr->break_window,
267 "postpone",
268 G_CALLBACK (break_window_postpone_cb),
269 dr);
270
271 g_signal_connect (dr->break_window,
272 "destroy",
273 G_CALLBACK (break_window_destroy_cb),
274 dr);
275
276 dr->secondary_break_windows = create_secondary_break_windows ();
277
278 gtk_widget_show (dr->break_window);
279
280 dr->save_last_time = elapsed_time;
281 dr->state = STATE_BREAK;
282 break;
283
284 case STATE_BREAK:
285 if (elapsed_time - dr->save_last_time >= dr->break_time) {
286 dr->state = STATE_BREAK_DONE_SETUP;
287 }
288 break;
289
290 case STATE_BREAK_DONE_SETUP:
291
292 dr->state = STATE_BREAK_DONE;
293 break;
294
295 case STATE_BREAK_DONE:
296 dr->state = STATE_START;
297 if (dr->break_window) {
298 gtk_widget_destroy (dr->break_window);
299 dr->break_window = NULL;
300 }
301 break;
302 }
303
304 dr->last_elapsed_time = elapsed_time;
305
306 update_app_indicator (dr);
307
308 return TRUE;
309}

311static gboolean
312update_status (DrWright *dr)
313{
314 gint min;
315 gchar *str;
316
317 if (!dr->enabled) {
318 app_indicator_set_status (dr->indicator,
319 APP_INDICATOR_STATUS_PASSIVE);
320 return TRUE;
321 }
322
323 min = get_time_left (dr);
324
325 if (min >= 1) {
326 str = g_strdup_printf (_("Take a break now (next in %dm)"), min);
327 } else {
328 str = g_strdup_printf (_("Take a break now (next in less than one minute)"));
329 }
330
331 gtk_menu_item_set_label (GTK_MENU_ITEM (dr->break_item), str);
332
333 g_free (str);
334
335 return TRUE;
336}

338static gint
339get_time_left (DrWright *dr)
340{
341 gint elapsed_time;
342
343 elapsed_time = drw_timer_elapsed (dr->timer);
344
345 return floor (0.5 + (dr->type_time - elapsed_time - dr->save_last_time) / 60.0);
346}

355static void
356gsettings_notify_cb (GSettings *settings,
357 gchar *key,
358 gpointer user_data)
359{
360 DrWright *dr = user_data;
361
362 if (!strcmp (key, "type-time")) {
363 dr->type_time = 60 * g_settings_get_int (settings, key);
364 dr->warn_time = MIN (dr->type_time / 10, 5*60);
365
366 dr->state = STATE_START;
367 }
368 else if (!strcmp (key, "break-time")) {
369 dr->break_time = 60 * g_settings_get_int (settings, key);
370 dr->state = STATE_START;
371 }
372 else if (!strcmp (key, "enabled")) {
373 dr->enabled = g_settings_get_boolean (settings, key);
374 dr->state = STATE_START;
375
376 gtk_widget_set_sensitive (dr->break_item, dr->enabled);
377
378 update_status (dr);
379 }
380
381 maybe_change_state (dr);
382}

462static void
463break_window_postpone_cb (GtkWidget *window,
464 DrWright *dr)
465{
466 gint elapsed_time;
467
468 gtk_widget_destroy (dr->break_window);
469
470 dr->state = STATE_RUNNING;
471 dr->break_window = NULL;
472
473 elapsed_time = drw_timer_elapsed (dr->timer);
474
475 if (elapsed_time + dr->save_last_time >= dr->type_time) {
476 /* Typing time has expired, but break was postponed.
477 * We'll warn again in (elapsed * sqrt (typing_time))^2 */
478 gfloat postpone_time = (((float) elapsed_time) / dr->break_time)
479 * sqrt (dr->type_time);
480 postpone_time *= postpone_time;
481 dr->save_last_time = dr->type_time - MAX (dr->warn_time, (gint) postpone_time);
482 }
483
484 drw_timer_start (dr->timer);
485 maybe_change_state (dr);
486 update_status (dr);
487 update_app_indicator (dr);
488}

527static GList *
528create_secondary_break_windows (void)
529{
530 GdkDisplay *display;
531 GdkScreen *screen;
532 GtkWidget *window;
533 GList *windows = NULL;
534 gint scale;
535
536 display = gdk_display_get_default ();
537
538 screen = gdk_display_get_default_screen (display);
539
540 if (screen != gdk_screen_get_default ()) {
541 /* Handled by DrwBreakWindow. */
542
543 window = gtk_window_new (GTK_WINDOW_POPUP);
544
545 windows = g_list_prepend (windows, window);
546 scale = gtk_widget_get_scale_factor (GTK_WIDGET (window));
547
548 gtk_window_set_screen (GTK_WINDOW (window), screen);
549
550 gtk_window_set_default_size (GTK_WINDOW (window),
551 WidthOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale,
552 HeightOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale);
553
554 gtk_widget_set_app_paintable (GTK_WIDGET (window), TRUE);
555 drw_setup_background (GTK_WIDGET (window));
556 gtk_window_stick (GTK_WINDOW (window));
557 gtk_widget_show (window);
558 }
559
560 return windows;
561}
