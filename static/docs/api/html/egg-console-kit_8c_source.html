66gboolean
67egg_console_kit_restart (EggConsoleKit *console, GError **error)
68{
69 gboolean ret;
70 GError *error_local = NULL;
71
72 g_return_val_if_fail (EGG_IS_CONSOLE_KIT (console), FALSE);
73 g_return_val_if_fail (console->priv->proxy_manager != NULL, FALSE);
74
75 ret = dbus_g_proxy_call (console->priv->proxy_manager, "Restart", &error_local,
76 G_TYPE_INVALID, G_TYPE_INVALID);
77 if (!ret) {
78 g_warning ("Couldn't restart: %s", error_local->message);
79 if (error != NULL)
80 *error = g_error_new (1, 0, "%s", error_local->message);
81 g_error_free (error_local);
82 }
83 return ret;
84}

275gboolean
276egg_console_kit_is_local (EggConsoleKit *console)
277{
278 gboolean ret = FALSE;
279 gboolean value = FALSE;
280 GError *error = NULL;
281
282 g_return_val_if_fail (EGG_IS_CONSOLE_KIT (console), FALSE);
283
284 /* maybe console kit does not know about our session */
285 if (console->priv->proxy_session == NULL) {
286 g_warning ("no ConsoleKit session");
287 goto out;
288 }
289
290 /* is our session local */
291 ret = dbus_g_proxy_call (console->priv->proxy_session, "IsLocal", &error, G_TYPE_INVALID,
292 G_TYPE_BOOLEAN, &value, G_TYPE_INVALID);
293 if (!ret) {
294 g_warning ("IsLocal failed: %s", error->message);
295 g_error_free (error);
296 goto out;
297 }
298
299 /* return value only if we successed */
300 ret = value;
301out:
302 return ret;
303}

371static void
372egg_console_kit_init (EggConsoleKit *console)
373{
374 gboolean ret;
375 GError *error = NULL;
376 guint32 pid;
377
378 console->priv = egg_console_kit_get_instance_private (console);
379 console->priv->proxy_manager = NULL;
380 console->priv->session_id = NULL;
381
382 /* connect to D-Bus */
383 console->priv->connection = dbus_g_bus_get (DBUS_BUS_SYSTEM, &error);
384 if (console->priv->connection == NULL) {
385 g_warning ("Failed to connect to the D-Bus daemon: %s", error->message);
386 g_error_free (error);
387 goto out;
388 }
389
390 /* connect to ConsoleKit */
391 console->priv->proxy_manager =
392 dbus_g_proxy_new_for_name (console->priv->connection, CONSOLEKIT_NAME,
393 CONSOLEKIT_MANAGER_PATH, CONSOLEKIT_MANAGER_INTERFACE);
394 if (console->priv->proxy_manager == NULL) {
395 g_warning ("cannot connect to ConsoleKit");
396 goto out;
397 }
398
399 /* get the session we are running in */
400 pid = getpid ();
401 ret = dbus_g_proxy_call (console->priv->proxy_manager, "GetSessionForUnixProcess", &error,
402 G_TYPE_UINT, pid,
403 G_TYPE_INVALID,
404 DBUS_TYPE_G_OBJECT_PATH, &console->priv->session_id,
405 G_TYPE_INVALID);
406 if (!ret) {
407 g_warning ("Failed to get session for pid %u: %s", pid, error->message);
408 g_error_free (error);
409 goto out;
410 }
411 g_debug ("ConsoleKit session ID: %s", console->priv->session_id);
412
413 /* connect to session */
414 console->priv->proxy_session =
415 dbus_g_proxy_new_for_name (console->priv->connection, CONSOLEKIT_NAME,
416 console->priv->session_id, CONSOLEKIT_SESSION_INTERFACE);
417 if (console->priv->proxy_session == NULL) {
418 g_warning ("cannot connect to: %s", console->priv->session_id);
419 goto out;
420 }
421 dbus_g_proxy_add_signal (console->priv->proxy_session, "ActiveChanged", G_TYPE_BOOLEAN, G_TYPE_INVALID);
422 dbus_g_proxy_connect_signal (console->priv->proxy_session, "ActiveChanged",
423 G_CALLBACK (egg_console_kit_active_changed_cb), console, NULL);
424
425out:
426 return;
427}

433static void
434egg_console_kit_finalize (GObject *object)
435{
436 EggConsoleKit *console;
437
438 g_return_if_fail (EGG_IS_CONSOLE_KIT (object));
439
440 console = EGG_CONSOLE_KIT (object);
441
442 g_return_if_fail (console->priv != NULL);
443 if (console->priv->proxy_manager != NULL)
444 g_object_unref (console->priv->proxy_manager);
445 if (console->priv->proxy_session != NULL)
446 g_object_unref (console->priv->proxy_session);
447 g_free (console->priv->session_id);
448
449 G_OBJECT_CLASS (egg_console_kit_parent_class)->finalize (object);
450}
