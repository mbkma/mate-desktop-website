55static time_t
56mktime_from_string (const char *time_s)
57{
58 struct tm tm = {0, };
59 tm.tm_isdst = -1;
60 strptime (time_s, LSTAR_DATE_FORMAT, &tm);
61 return mktime (&tm);
62}

64static char*
65tar_get_last_field (const char *line,
66 int start_from,
67 int field_n)
68{
69 const char *f_start, *f_end;
70
71 line = line + start_from;
72
73 f_start = line;
74 while ((*f_start == ' ') && (*f_start != *line))
75 f_start++;
76 f_end = f_start;
77
78 while ((field_n > 0) && (*f_end != 0)) {
79 if (*f_end == ' ') {
80 field_n--;
81 if (field_n == 1)
82 f_start = f_end;
83 }
84 f_end++;
85 }
86
87 if (*f_start == ' ')
88 f_start++;
89
90 return g_strdup (f_start);
91}

93static void
94process_line (char *line,
95 gpointer data)
96{
97 FileData *fdata;
98 FrCommand *comm = FR_COMMAND (data);
99 char **fields;
100 int date_idx;
101 char *field_date, *field_size, *field_name;
102 char *name;
103
104 g_return_if_fail (line != NULL);
105
106 date_idx = file_list__get_index_from_pattern (line, "%n%n%n%n-%n%n-%n%n %n%n:%n%n");
107 if (date_idx < 0)
108 return;
109
110 fdata = file_data_new ();
111
112 field_size = file_list__get_prev_field (line, date_idx, 1);
113 fdata->size = g_ascii_strtoull (field_size, NULL, 10);
114 g_free (field_size);
115
116 field_date = g_strndup (line + date_idx, 16);
117 fdata->modified = mktime_from_string (field_date);
118 g_free (field_date);
119
120 /* Full path */
121
122 field_name = tar_get_last_field (line, date_idx, 3);
123 fields = g_strsplit (field_name, " -> ", 2);
124
125 if (fields[1] == NULL) {
126 g_strfreev (fields);
127 fields = g_strsplit (field_name, " link to ", 2);
128 }
129
130 name = g_strcompress (fields[0]);
131 if (*name == '/') {
132 fdata->full_path = g_strdup (name);
133 fdata->original_path = fdata->full_path;
134 } else {
135 fdata->full_path = g_strconcat ("/", name, NULL);
136 fdata->original_path = fdata->full_path + 1;
137 }
138 g_free (name);
139 name = g_filename_from_utf8 (fdata->original_path, -1, NULL, NULL, NULL);
140 if (name) {
141 fdata->original_path = g_strdup (name);
142 fdata->free_original_path = TRUE;
143 }
144 g_free (name);
145
146 if (fields[1] != NULL)
147 fdata->link = g_strdup (fields[1]);
148 g_strfreev (fields);
149 g_free (field_name);
150
151 fdata->dir = line[0] == 'd';
152 if (fdata->dir)
153 fdata->name = dir_name_from_path (fdata->full_path);
154 else
155 fdata->name = g_strdup (file_name_from_path (fdata->full_path));
156 fdata->path = remove_level_from_path (fdata->full_path);
157
158 if (*fdata->name == 0)
159 file_data_free (fdata);
160 else
161 fr_command_add_file (comm, fdata);
162}

164static void
165add_compress_arg (FrCommand *comm)
166{
167 if (is_mime_type (comm->mime_type, "application/x-compressed-tar"))
168 fr_process_add_arg (comm->process, "-z");
169
170 else if (is_mime_type (comm->mime_type, "application/x-brotli-compressed-tar"))
171 fr_process_add_arg (comm->process, "--use-compress-program=brotli");
172
173 else if (is_mime_type (comm->mime_type, "application/x-bzip-compressed-tar"))
174 if (is_program_in_path ("lbzip2"))
175 fr_process_add_arg (comm->process, "--use-compress-program=lbzip2");
176 else
177 fr_process_add_arg (comm->process, "--use-compress-program=bzip2");
178
179 else if (is_mime_type (comm->mime_type, "application/x-tarz")) {
180 if (is_program_in_path ("gzip"))
181 fr_process_add_arg (comm->process, "-z");
182 else
183 fr_process_add_arg (comm->process, "-Z");
184 }
185 else if (is_mime_type (comm->mime_type, "application/x-lrzip-compressed-tar"))
186 fr_process_add_arg (comm->process, "--use-compress-program=lrzip");
187
188 else if (is_mime_type (comm->mime_type, "application/x-lzip-compressed-tar"))
189 fr_process_add_arg (comm->process, "--use-compress-program=lzip");
190
191 else if (is_mime_type (comm->mime_type, "application/x-lzma-compressed-tar"))
192 fr_process_add_arg (comm->process, "--use-compress-program=lzma");
193
194 else if (is_mime_type (comm->mime_type, "application/x-xz-compressed-tar"))
195 if (is_program_in_path ("pixz"))
196 fr_process_add_arg (comm->process, "--use-compress-program=pixz");
197 else
198 fr_process_add_arg (comm->process, "--use-compress-program=xz");
199
200 else if (is_mime_type (comm->mime_type, "application/x-lzop-compressed-tar"))
201 fr_process_add_arg (comm->process, "--use-compress-program=lzop");
202
203 else if (is_mime_type (comm->mime_type, "application/x-7z-compressed-tar")) {
204 FrCommandTar *comm_tar = (FrCommandTar*) comm;
205 char *option;
206
207 option = g_strdup_printf ("--use-compress-program=%s", comm_tar->compress_command);
208 fr_process_add_arg (comm->process, option);
209 g_free (option);
210 }
211}

213static void
214begin_tar_command (FrCommand *comm)
215{
216 char *command = NULL;
217
218 /* In solaris gtar is present under /usr/sfw/bin */
219
220 command = g_find_program_in_path ("gtar");
221#if defined (__SVR4) && defined (__sun)
222 if (g_file_test ("/usr/sfw/bin/gtar", G_FILE_TEST_IS_EXECUTABLE)) {
223 g_free (command);
224 command = g_strdup ("/usr/sfw/bin/gtar");
225 }
226#endif
227 if (command != NULL)
228 fr_process_begin_command (comm->process, command);
229 else
230 fr_process_begin_command (comm->process, "tar");
231 g_free (command);
232}

255static void
256process_line__generic (char *line,
257 gpointer data,
258 char *action_msg)
259{
260 FrCommand *comm = FR_COMMAND (data);
261
262 if (line == NULL)
263 return;
264
265 if (line[strlen (line) - 1] == '/') /* ignore directories */
266 return;
267
268 if (comm->n_files != 0) {
269 double fraction = (double) ++comm->n_file / (comm->n_files + 1);
270 fr_command_progress (comm, fraction);
271 }
272 else {
273 char *msg = g_strconcat (action_msg, file_name_from_path (line), NULL);
274 fr_command_message (comm, msg);
275 g_free (msg);
276 }
277}

287static void
288fr_command_tar_add (FrCommand *comm,
289 const char *from_file,
290 GList *file_list,
291 const char *base_dir,
292 gboolean update,
293 gboolean recursive)
294{
295 FrCommandTar *c_tar = FR_COMMAND_TAR (comm);
296 GList *scan;
297
298 fr_process_set_out_line_func (FR_COMMAND (comm)->process,
299 process_line__add,
300 comm);
301
302 begin_tar_command (comm);
303 fr_process_add_arg (comm->process, "--force-local");
304 if (! recursive)
305 fr_process_add_arg (comm->process, "--no-recursion");
306 fr_process_add_arg (comm->process, "--no-wildcards");
307 fr_process_add_arg (comm->process, "--no-unquote");
308 fr_process_add_arg (comm->process, "-v");
309 fr_process_add_arg (comm->process, "-p");
310
311 if (base_dir != NULL) {
312 fr_process_add_arg (comm->process, "-C");
313 fr_process_add_arg (comm->process, base_dir);
314 }
315
316 if (can_create_a_compressed_archive (comm)) {
317 fr_process_add_arg (comm->process, "-cf");
318 fr_process_add_arg (comm->process, comm->filename);
319 add_compress_arg (comm);
320 }
321 else {
322 if (comm->creating_archive)
323 fr_process_add_arg (comm->process, "-cf");
324 else
325 fr_process_add_arg (comm->process, "-rf");
326 fr_process_add_arg (comm->process, c_tar->uncomp_filename);
327 }
328
329 if (from_file != NULL) {
330 fr_process_add_arg (comm->process, "-T");
331 fr_process_add_arg (comm->process, from_file);
332 }
333
334 fr_process_add_arg (comm->process, "--");
335
336 if (from_file == NULL)
337 for (scan = file_list; scan; scan = scan->next)
338 fr_process_add_arg (comm->process, scan->data);
339
340 fr_process_end_command (comm->process);
341}

461static gboolean
462gzip_continue_func (gpointer user_data)
463{
464 FrCommand *comm = user_data;
465
466 /* ignore gzip warnings */
467
468 if (comm->process->error.status == 2) {
469 comm->process->error.type = FR_PROC_ERROR_NONE;
470 comm->process->error.status = 0;
471 g_clear_error (&comm->process->error.gerror);
472 }
473
474 return comm->process->error.status == 0;
475}

477static void
478fr_command_tar_recompress (FrCommand *comm)
479{
480 FrCommandTar *c_tar = FR_COMMAND_TAR (comm);
481 char *new_name = NULL;
482
483 if (can_create_a_compressed_archive (comm))
484 return;
485
486 if (is_mime_type (comm->mime_type, "application/x-compressed-tar")) {
487 fr_process_begin_command (comm->process, "gzip");
488 fr_process_set_begin_func (comm->process, begin_func__recompress, comm);
489 fr_process_set_continue_func (comm->process, gzip_continue_func, comm);
490 switch (comm->compression) {
491 case FR_COMPRESSION_VERY_FAST:

501 fr_process_add_arg (comm->process, c_tar->uncomp_filename);

638 fr_process_add_arg (comm->process, "-6"); break;
639 case FR_COMPRESSION_MAXIMUM:
640 fr_process_add_arg (comm->process, "-9"); break;
641 }
642 fr_process_add_arg (comm->process, "-fU");
643 fr_process_add_arg (comm->process, "--no-stdin");
644 fr_process_add_arg (comm->process, c_tar->uncomp_filename);
645 fr_process_end_command (comm->process);
646
647 new_name = g_strconcat (c_tar->uncomp_filename, ".lzo", NULL);
648 }
649 else if (is_mime_type (comm->mime_type, "application/x-7z-compressed-tar")) {
650 FrCommandTar *comm_tar = (FrCommandTar*) comm;
651
652 fr_process_begin_command (comm->process, comm_tar->compress_command);
653 fr_process_set_begin_func (comm->process, begin_func__recompress, comm);
654 switch (comm->compression) {
655 case FR_COMPRESSION_VERY_FAST:
656 fr_process_add_arg (comm->process, "-mx=1"); break;
657 case FR_COMPRESSION_FAST:
658 fr_process_add_arg (comm->process, "-mx=5"); break;
659 case FR_COMPRESSION_NORMAL:
660 fr_process_add_arg (comm->process, "-mx=5"); break;
661 case FR_COMPRESSION_MAXIMUM:
662 fr_process_add_arg (comm->process, "-mx=7"); break;
663 }
664 fr_process_add_arg (comm->process, "a");
665 fr_process_add_arg (comm->process, "-bd");
666 fr_process_add_arg (comm->process, "-y");
667 if (g_strcmp0 (comm_tar->compress_command, "7zzs") != 0 && g_strcmp0 (comm_tar->compress_command, "7zz") != 0)
668 fr_process_add_arg (comm->process, "-l");
669
670 new_name = g_strconcat (c_tar->uncomp_filename, ".7z", NULL);
671 fr_process_add_arg_concat (comm->process, new_name, NULL);
672
673 fr_process_add_arg (comm->process, c_tar->uncomp_filename);
674 fr_process_end_command (comm->process);
675
676 /* remove the uncompressed tar */
677
678 fr_process_begin_command (comm->process, "rm");
679 fr_process_add_arg (comm->process, "-f");
680 fr_process_add_arg (comm->process, c_tar->uncomp_filename);
681 fr_process_end_command (comm->process);
682 }
683
684 if (c_tar->name_modified) {
685 char *tmp_dir;
686
687 /* Restore original name. */
688
689 fr_process_begin_command (comm->process, "mv");
690 fr_process_add_arg (comm->process, "-f");
691 fr_process_add_arg (comm->process, new_name);
692 fr_process_add_arg (comm->process, comm->filename);
693 fr_process_end_command (comm->process);
694
695 tmp_dir = remove_level_from_path (new_name);
696
697 fr_process_begin_command (comm->process, "rm");
698 fr_process_set_sticky (comm->process, TRUE);
699 fr_process_add_arg (comm->process, "-fr");
700 fr_process_add_arg (comm->process, tmp_dir);
701 fr_process_end_command (comm->process);
702
703 g_free (tmp_dir);
704 }
705
706 g_free (new_name);
707 g_free (c_tar->uncomp_filename);
708 c_tar->uncomp_filename = NULL;
709}

719static char *
720get_uncompressed_name (FrCommandTar *c_tar,
721 const char *e_filename)
722{
723 FrCommand *comm = FR_COMMAND (c_tar);
724 char *new_name = g_strdup (e_filename);
725 int l = strlen (new_name);
726
727 if (is_mime_type (comm->mime_type, "application/x-compressed-tar")) {
728 /* X.tgz --> X.tar
729 * X.tar.gz --> X.tar */
730 if (file_extension_is (e_filename, ".tgz")) {
731 new_name[l - 2] = 'a';
732 new_name[l - 1] = 'r';
733 }
734 else if (file_extension_is (e_filename, ".tar.gz"))
735 new_name[l - 3] = 0;
736 }
737 else if (is_mime_type (comm->mime_type, "application/x-brotli-compressed-tar")) {
738 /* X.tar.br --> X.tar */
739 if (file_extension_is (e_filename, ".tar.br"))
740 new_name[l - 3] = 0;
741 }
742 else if (is_mime_type (comm->mime_type, "application/x-bzip-compressed-tar")) {
743 /* X.tbz2 --> X.tar
744 * X.tar.bz2 --> X.tar */
745 if (file_extension_is (e_filename, ".tbz2")) {
746 new_name[l - 3] = 'a';
747 new_name[l - 2] = 'r';
748 new_name[l - 1] = 0;
749 }
750 else if (file_extension_is (e_filename, ".tar.bz2"))
751 new_name[l - 4] = 0;
752 }
753 else if (is_mime_type (comm->mime_type, "application/x-tarz")) {
754 /* X.taz --> X.tar
755 * X.tar.Z --> X.tar */
756 if (file_extension_is (e_filename, ".taz"))
757 new_name[l - 1] = 'r';
758 else if (file_extension_is (e_filename, ".tar.Z"))
759 new_name[l - 2] = 0;
760 }
761 else if (is_mime_type (comm->mime_type, "application/x-lrzip-compressed-tar")) {
762 /* X.tlrz --> X.tar
763 * X.tar.lrz --> X.tar */
764 if (file_extension_is (e_filename, ".tlrz")) {
765 new_name[l - 3] = 'a';
766 new_name[l - 2] = 'r';
767 new_name[l - 1] = 0;
768 }
769 else if (file_extension_is (e_filename, ".tar.lrz"))
770 new_name[l - 4] = 0;
771 }
772 else if (is_mime_type (comm->mime_type, "application/x-lzip-compressed-tar")) {
773 /* X.tlz --> X.tar
774 * X.tar.lz --> X.tar */
775 if (file_extension_is (e_filename, ".tlz")) {
776 new_name[l - 2] = 'a';
777 new_name[l - 1] = 'r';
778 }
779 else if (file_extension_is (e_filename, ".tar.lz"))
780 new_name[l - 3] = 0;
781 }
782 else if (is_mime_type (comm->mime_type, "application/x-lzma-compressed-tar")) {
783 /* X.tar.lzma --> X.tar
784 * (There doesn't seem to be a shorthand suffix) */
785 if (file_extension_is (e_filename, ".tar.lzma"))
786 new_name[l - 5] = 0;
787 }
788 else if (is_mime_type (comm->mime_type, "application/x-xz-compressed-tar")) {
789 /* X.txz --> X.tar
790 * X.tar.xz --> X.tar */
791 if (file_extension_is (e_filename, ".txz")) {
792 new_name[l - 2] = 'a';
793 new_name[l - 1] = 'r';
794 }
795 else if (file_extension_is (e_filename, ".tar.xz"))
796 new_name[l - 3] = 0;
797 }
798 else if (is_mime_type (comm->mime_type, "application/x-lzop-compressed-tar")) {
799 /* X.tzo --> X.tar
800 * X.tar.lzo --> X.tar */
801 if (file_extension_is (e_filename, ".tzo")) {
802 new_name[l - 2] = 'a';
803 new_name[l - 1] = 'r';
804 }
805 else if (file_extension_is (e_filename, ".tar.lzo"))
806 new_name[l - 4] = 0;
807 }
808 else if (is_mime_type (comm->mime_type, "application/x-7z-compressed-tar")) {
809 /* X.tar.7z --> X.tar */
810 if (file_extension_is (e_filename, ".tar.7z"))
811 new_name[l - 3] = 0;
812 }
813 else if (is_mime_type (comm->mime_type, "application/x-zstd-compressed-tar")) {
814 /* X.tar.zst --> X.tar */
815 if (file_extension_is (e_filename, ".tar.zst"))
816 new_name[l - 4] = 0;
817 }
818
819 return new_name;
820}

839static void
840fr_command_tar_uncompress (FrCommand *comm)
841{
842 FrCommandTar *c_tar = FR_COMMAND_TAR (comm);
843 char *tmp_name;
844 gboolean archive_exists;
845
846 if (can_create_a_compressed_archive (comm))
847 return;
848
849 if (c_tar->uncomp_filename != NULL) {
850 g_free (c_tar->uncomp_filename);
851 c_tar->uncomp_filename = NULL;
852 }
853
854 archive_exists = ! comm->creating_archive;
855
856 c_tar->name_modified = ! is_mime_type (comm->mime_type, "application/x-tar");
857 if (c_tar->name_modified) {
858 tmp_name = get_temp_name (c_tar, comm->filename);
859 if (archive_exists) {
860 fr_process_begin_command (comm->process, "mv");
861 fr_process_add_arg (comm->process, "-f");
862 fr_process_add_arg (comm->process, comm->filename);
863 fr_process_add_arg (comm->process, tmp_name);
864 fr_process_end_command (comm->process);
865 }
866 }
867 else
868 tmp_name = g_strdup (comm->filename);
869
870 if (archive_exists) {
871 if (is_mime_type (comm->mime_type, "application/x-compressed-tar")) {
872 fr_process_begin_command (comm->process, "gzip");
873 fr_process_set_begin_func (comm->process, begin_func__uncompress, comm);
874 fr_process_set_continue_func (comm->process, gzip_continue_func, comm);
875 fr_process_add_arg (comm->process, "-f");
876 fr_process_add_arg (comm->process, "-d");
877 fr_process_add_arg (comm->process, tmp_name);
878 fr_process_end_command (comm->process);
879 }
880 else if (is_mime_type (comm->mime_type, "application/x-brotli-compressed-tar")) {
881 fr_process_begin_command (comm->process, "brotli");
882 fr_process_set_begin_func (comm->process, begin_func__uncompress, comm);
883 fr_process_add_arg (comm->process, "-f");
884 fr_process_add_arg (comm->process, "-d");
885 fr_process_add_arg (comm->process, tmp_name);
886 fr_process_end_command (comm->process);
887 }
888 else if (is_mime_type (comm->mime_type, "application/x-bzip-compressed-tar")) {
889 fr_process_begin_command (comm->process, "bzip2");
890 fr_process_set_begin_func (comm->process, begin_func__uncompress, comm);
891 fr_process_add_arg (comm->process, "-f");
892 fr_process_add_arg (comm->process, "-d");
893 fr_process_add_arg (comm->process, tmp_name);
894 fr_process_end_command (comm->process);
895 }
896 else if (is_mime_type (comm->mime_type, "application/x-tarz")) {
897 if (is_program_in_path ("gzip")) {
898 fr_process_begin_command (comm->process, "gzip");
899 fr_process_set_continue_func (comm->process, gzip_continue_func, comm);
900 }
901 else
902 fr_process_begin_command (comm->process, "uncompress");
903 fr_process_set_begin_func (comm->process, begin_func__uncompress, comm);
904 fr_process_add_arg (comm->process, "-f");
905 fr_process_add_arg (comm->process, tmp_name);
906 fr_process_end_command (comm->process);
907 }
908 else if (is_mime_type (comm->mime_type, "application/x-lrzip-compressed-tar")) {
909 fr_process_begin_command (comm->process, "lrzip");
910 fr_process_set_begin_func (comm->process, begin_func__uncompress, comm);
911 fr_process_add_arg (comm->process, "-f");
912 fr_process_add_arg (comm->process, "-d");
913 fr_process_add_arg (comm->process, tmp_name);
914 fr_process_end_command (comm->process);
915 }
916 else if (is_mime_type (comm->mime_type, "application/x-lzip-compressed-tar")) {
917 fr_process_begin_command (comm->process, "lzip");
918 fr_process_set_begin_func (comm->process, begin_func__uncompress, comm);
919 fr_process_add_arg (comm->process, "-f");
920 fr_process_add_arg (comm->process, "-d");
921 fr_process_add_arg (comm->process, tmp_name);
922 fr_process_end_command (comm->process);
923 }
924 else if (is_mime_type (comm->mime_type, "application/x-lzma-compressed-tar")) {
925 fr_process_begin_command (comm->process, "lzma");
926 fr_process_set_begin_func (comm->process, begin_func__uncompress, comm);
927 fr_process_add_arg (comm->process, "-f");
928 fr_process_add_arg (comm->process, "-d");
929 fr_process_add_arg (comm->process, tmp_name);
930 fr_process_end_command (comm->process);
931 }
932 else if (is_mime_type (comm->mime_type, "application/x-xz-compressed-tar")) {
933 fr_process_begin_command (comm->process, "xz");
934 fr_process_set_begin_func (comm->process, begin_func__uncompress, comm);
935 fr_process_add_arg (comm->process, "-f");
936 fr_process_add_arg (comm->process, "-d");
937 fr_process_add_arg (comm->process, tmp_name);
938 fr_process_end_command (comm->process);
939 }
940 else if (is_mime_type (comm->mime_type, "application/x-lzop-compressed-tar")) {
941 fr_process_begin_command (comm->process, "lzop");
942 fr_process_set_begin_func (comm->process, begin_func__uncompress, comm);
943 fr_process_add_arg (comm->process, "-dfU");
944 fr_process_add_arg (comm->process, "--no-stdin");
945 fr_process_add_arg (comm->process, tmp_name);
946 fr_process_end_command (comm->process);
947 }
948 else if (is_mime_type (comm->mime_type, "application/x-7z-compressed-tar")) {
949 FrCommandTar *comm_tar = (FrCommandTar*) comm;
950
951 fr_process_begin_command (comm->process, comm_tar->compress_command);
952 fr_process_set_begin_func (comm->process, begin_func__uncompress, comm);
953 fr_process_add_arg (comm->process, "e");
954 fr_process_add_arg (comm->process, "-bd");
955 fr_process_add_arg (comm->process, "-y");
956 fr_process_add_arg (comm->process, tmp_name);
957 fr_process_end_command (comm->process);
958
959 /* remove the compressed tar */
960
961 fr_process_begin_command (comm->process, "rm");
962 fr_process_add_arg (comm->process, "-f");
963 fr_process_add_arg (comm->process, tmp_name);
964 fr_process_end_command (comm->process);
965 }
966 else if (is_mime_type (comm->mime_type, "application/x-zstd-compressed-tar")) {
967 fr_process_begin_command (comm->process, "zstd");
968 fr_process_set_begin_func (comm->process, begin_func__uncompress, comm);
969 fr_process_add_arg (comm->process, "-f");
970 fr_process_add_arg (comm->process, "-d");
971 fr_process_add_arg (comm->process, tmp_name);
972 fr_process_end_command (comm->process);
973 }
974 }
975
976 c_tar->uncomp_filename = get_uncompressed_name (c_tar, tmp_name);
977 g_free (tmp_name);
978}

1010static FrCommandCaps
1011fr_command_tar_get_capabilities (FrCommand *comm,
1012 const char *mime_type,
1013 gboolean check_command)
1014{
1015 FrCommandCaps capabilities;
1016
1017 capabilities = FR_COMMAND_CAN_ARCHIVE_MANY_FILES;
1018
1019 /* In solaris gtar is present under /usr/sfw/bin */
1020 if (! is_program_available ("tar", check_command) && ! is_program_available ("/usr/sfw/bin/gtar", check_command))
1021 return capabilities;
1022
1023 if (is_mime_type (mime_type, "application/x-tar")) {
1024 capabilities |= FR_COMMAND_CAN_READ_WRITE;
1025 }
1026 else if (is_mime_type (mime_type, "application/x-compressed-tar")) {
1027 if (is_program_available ("gzip", check_command))
1028 capabilities |= FR_COMMAND_CAN_READ_WRITE;
1029 }
1030 else if (is_mime_type (mime_type, "application/x-brotli-compressed-tar")) {
1031 if (is_program_available ("brotli", check_command))
1032 capabilities |= FR_COMMAND_CAN_READ_WRITE;
1033 }
1034 else if (is_mime_type (mime_type, "application/x-bzip-compressed-tar")) {
1035 if (is_program_available ("bzip2", check_command))
1036 capabilities |= FR_COMMAND_CAN_READ_WRITE;
1037 }
1038 else if (is_mime_type (mime_type, "application/x-tarz")) {
1039 if (is_program_available ("compress", check_command) && is_program_available ("uncompress", check_command))
1040 capabilities |= FR_COMMAND_CAN_READ_WRITE;
1041 else if (is_program_available ("gzip", check_command))
1042 capabilities |= FR_COMMAND_CAN_READ;
1043 }
1044 else if (is_mime_type (mime_type, "application/x-lrzip-compressed-tar")) {
1045 if (is_program_available ("lrzip", check_command))
1046 capabilities |= FR_COMMAND_CAN_READ_WRITE;
1047 }
1048 else if (is_mime_type (mime_type, "application/x-lzip-compressed-tar")) {
1049 if (is_program_available ("lzip", check_command))
1050 capabilities |= FR_COMMAND_CAN_READ_WRITE;
1051 }
1052 else if (is_mime_type (mime_type, "application/x-lzma-compressed-tar")) {
1053 if (is_program_available ("lzma", check_command))
1054 capabilities |= FR_COMMAND_CAN_READ_WRITE;
1055 }
1056 else if (is_mime_type (mime_type, "application/x-xz-compressed-tar")) {
1057 if (is_program_available ("xz", check_command))
1058 capabilities |= FR_COMMAND_CAN_READ_WRITE;
1059 }
1060 else if (is_mime_type (mime_type, "application/x-lzop-compressed-tar")) {
1061 if (is_program_available ("lzop", check_command))
1062 capabilities |= FR_COMMAND_CAN_READ_WRITE;
1063 }
1064 else if (is_mime_type (mime_type, "application/x-7z-compressed-tar")) {
1065 const char *try_command[5] = { "7zzs", "7zz", "7za", "7zr", "7z" };
1066 size_t i;
1067
1068 for (i = 0; i < G_N_ELEMENTS (try_command); i++) {
1069 if (is_program_available (try_command[i], check_command)) {
1070 capabilities |= FR_COMMAND_CAN_WRITE;
1071 break;
1072 }
1073 }
1074 }
1075 else if (is_mime_type (mime_type, "application/x-zstd-compressed-tar")) {
1076 if (is_program_available ("zstd", check_command))
1077 capabilities |= FR_COMMAND_CAN_READ;
1078 }
1079
1080 return capabilities;
1081}
