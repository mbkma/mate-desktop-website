52cadp_utils_gslist_remove_from( GSList *list, const gchar *string )
53{
54 GSList *is;
55
56 for( is = list ; is ; is = is->next ){

71gboolean
72cadp_utils_uri_delete( const gchar *uri )
73{
74 gboolean deleted;
75 gchar *scheme;
76 gchar *path;
77
78 deleted = FALSE;
79 scheme = g_uri_parse_scheme( uri );
80
81 if( !strcmp( scheme, "file" )){
82 path = g_filename_from_uri( uri, NULL, NULL );
83
84 if( path ){
85 deleted = na_core_utils_file_delete( path );
86 g_free( path );
87 }
88 }
89
90 g_free( scheme );
91
92 return( deleted );
93}

107gboolean
108cadp_utils_uri_is_writable( const gchar *uri )
109{
110 static const gchar *thisfn = "cadp_utils_uri_is_writable";
111 GFile *file;
112 GError *error = NULL;
113 GFileInfo *info;
114 gboolean writable;
115
116 if( !uri || !g_utf8_strlen( uri, -1 )){
117 return( FALSE );
118 }
119
120 file = g_file_new_for_uri( uri );
121 info = g_file_query_info( file,
122 G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE "," G_FILE_ATTRIBUTE_STANDARD_TYPE,
123 G_FILE_QUERY_INFO_NONE, NULL, &error );
124
125 if( error ){
126 g_warning( "%s: g_file_query_info error: %s", thisfn, error->message );
127 g_error_free( error );
128 g_object_unref( file );
129 return( FALSE );
130 }
131
132 writable = g_file_info_get_attribute_boolean( info, G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE );
133 if( !writable ){
134 g_debug( "%s: %s is not writable", thisfn, uri );
135 }
136
137 g_object_unref( info );
138
139 return( writable );
140}
