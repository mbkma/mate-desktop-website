46void
47_gs_theme_engine_profile_log (const char *func,
48 const char *note,
49 const char *format,
50 ...)
51{
52 va_list args;
53 char *str;
54 char *formatted;
55
56 va_start (args, format);
57 formatted = g_strdup_vprintf (format, args);
58 va_end (args);
59
60 if (func != NULL)
61 {
62 str = g_strdup_printf ("MARK: %s %s: %s %s", g_get_prgname(), func, note ? note : "", formatted);
63 }
64 else
65 {
66 str = g_strdup_printf ("MARK: %s: %s %s", g_get_prgname(), note ? note : "", formatted);
67 }
68
69 g_free (formatted);
70
71 access (str, F_OK);
72 g_free (str);
73}
