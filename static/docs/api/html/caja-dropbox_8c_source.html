56/*
57 Simplifies a path by removing navigation elements such as '.' and '..'
58
59 Arguments:
60 - path: input path to be canonicalized
61
62 Returns:
63 Canonicalized path if input path is valid.
64 NULL otherwise.
65*/
66static gchar *
67canonicalize_path(gchar *path) {
68 int i, j = 0;
69 gchar *toret = NULL;
70 gchar **cpy, **elts;
71
72 g_assert(path != NULL);
73 g_assert(path[0] == '/');
74
75 elts = g_strsplit(path, "/", 0);
76 cpy = g_new(gchar *, g_strv_length(elts)+1);
77 cpy[j++] = "/";
78 for (i = 0; elts[i] != NULL; i++) {
79 if (strcmp(elts[i], "..") == 0) {
80 if (j > 0) {
81 j--;
82 }
83 else {
84 // Input path has too many parent directory references and is invalid
85 toret = NULL;
86 goto exit;
87 }
88 }
89 else if (strcmp(elts[i], ".") != 0 && elts[i][0] != '\0') {
90 cpy[j++] = elts[i];
91 }
92 }
93
94 cpy[j] = NULL;
95 toret = g_build_filenamev(cpy);
96
97exit:
98 g_free(cpy);
99 g_strfreev(elts);
100
101 return toret;
102}

121static void
122when_file_dies(CajaDropbox *cvs, CajaFileInfo *address) {
123 gchar *filename;
124
125 filename = g_hash_table_lookup(cvs->obj2filename, address);
126
127 /* we never got a change to view this file */
128 if (filename == NULL) {
129 return;
130 }
131
132 /* too chatty */
133 /* debug("removing %s <-> 0x%p", filename, address); */
134
135 g_hash_table_remove(cvs->filename2obj, filename);
136 g_hash_table_remove(cvs->obj2filename, address);

139static void
140changed_cb(CajaFileInfo *file, CajaDropbox *cvs) {
141 /* check if this file's path has changed, if so update the hash and invalidate
142 the file */
143 gchar *filename, *pfilename;
144 gchar *filename2;
145 gchar *uri;
146
147 uri = caja_file_info_get_uri(file);
148 pfilename = g_filename_from_uri(uri, NULL, NULL);
149 filename = pfilename ? canonicalize_path(pfilename) : NULL;
150
151 /* Canonicalization will only null-out a non-null filename if it is invalid */
152 g_assert((pfilename == NULL && filename == NULL) || (pfilename != NULL && filename != NULL));
153
154 filename2 = g_hash_table_lookup(cvs->obj2filename, file);
155
156 g_free(pfilename);
157 g_free(uri);
158
159 /* if filename2 is NULL we've never seen this file in update_file_info */
160 if (filename2 == NULL) {
161 g_free(filename);
162 return;
163 }
164
165 if (filename == NULL) {
166 /* A file has moved to offline storage. Lets remove it from our tables. */
167 g_object_weak_unref(G_OBJECT(file), (GWeakNotify) when_file_dies, cvs);
168 g_hash_table_remove(cvs->filename2obj, filename2);
169 g_hash_table_remove(cvs->obj2filename, file);
170 g_signal_handlers_disconnect_by_func(file, G_CALLBACK(changed_cb), cvs);
171 reset_file(file);
172 return;
173 }
174
175 /* this is a hack, because caja doesn't do this for us, for some reason
176 the file's path has changed */
177 if (strcmp(filename, filename2) != 0) {
178 debug("shifty old: %s, new %s", filename2, filename);
179
180 /* gotta do this first, the call after this frees filename2 */
181 g_hash_table_remove(cvs->filename2obj, filename2);
182
183 g_hash_table_replace(cvs->obj2filename, file, g_strdup(filename));
184
185 {
186 CajaFileInfo *f2;
187 /* we shouldn't have another mapping from filename to an object */
188 f2 = g_hash_table_lookup(cvs->filename2obj, filename);
189 if (f2 != NULL) {
190 /* lets fix it if it's true, just remove the mapping */
191 g_hash_table_remove(cvs->filename2obj, filename);
192 g_hash_table_remove(cvs->obj2filename, f2);
193 }
194 }
195
196 g_hash_table_insert(cvs->filename2obj, g_strdup(filename), file);
197 reset_file(file);
198 }
199
200 g_free(filename);
201}

203static CajaOperationResult
204caja_dropbox_update_file_info(CajaInfoProvider *provider,
205 CajaFileInfo *file,
206 GClosure *update_complete,
207 CajaOperationHandle **handle) {
208 CajaDropbox *cvs;
209
210 cvs = CAJA_DROPBOX(provider);
211
212 /* this code adds this file object to our two-way hash of file objects
213 so we can shell touch these files later */
214 {
215 gchar *pfilename, *uri;
216
217 uri = caja_file_info_get_uri(file);
218 pfilename = g_filename_from_uri(uri, NULL, NULL);
219 g_free(uri);
220 if (pfilename == NULL) {
221 return CAJA_OPERATION_COMPLETE;
222 }
223 else {
224 int cmp = 0;
225 gchar *stored_filename;
226 gchar *filename;
227
228 filename = canonicalize_path(pfilename);
229 g_free(pfilename);
230 if (filename == NULL) {
231 /* pfilename path was invalid if canonicalize operation nulled it out */
232 return CAJA_OPERATION_FAILED;
233 }
234 stored_filename = g_hash_table_lookup(cvs->obj2filename, file);
235
236 /* don't worry about the dup checks, gcc is smart enough to optimize this
237 GCSE ftw */
238 if ((stored_filename != NULL && (cmp = strcmp(stored_filename, filename)) != 0) ||
239 stored_filename == NULL) {
240
241 if (stored_filename != NULL && cmp != 0) {
242 /* this happens when the filename changes name on a file obj
243 but changed_cb isn't called */
244 g_object_weak_unref(G_OBJECT(file), (GWeakNotify) when_file_dies, cvs);
245 g_hash_table_remove(cvs->obj2filename, file);
246 g_hash_table_remove(cvs->filename2obj, stored_filename);
247 g_signal_handlers_disconnect_by_func(file, G_CALLBACK(changed_cb), cvs);
248 }
249 else if (stored_filename == NULL) {
250 CajaFileInfo *f2;
251
252 if ((f2 = g_hash_table_lookup(cvs->filename2obj, filename)) != NULL) {
253 /* if the filename exists in the filename2obj hash
254 but the file obj doesn't exist in the obj2filename hash:
255
256 this happens when caja allocates another file object
257 for a filename without first deleting the original file object
258
259 just remove the association to the older file object, it's obsolete
260 */
261 g_object_weak_unref(G_OBJECT(f2), (GWeakNotify) when_file_dies, cvs);
262 g_signal_handlers_disconnect_by_func(f2, G_CALLBACK(changed_cb), cvs);
263 g_hash_table_remove(cvs->filename2obj, filename);
264 g_hash_table_remove(cvs->obj2filename, f2);
265 }
266 }
267
268 /* too chatty */
269 /* debug("adding %s <-> 0x%p", filename, file);*/
270 g_object_weak_ref(G_OBJECT(file), (GWeakNotify) when_file_dies, cvs);
271 g_hash_table_insert(cvs->filename2obj, g_strdup(filename), file);
272 g_hash_table_insert(cvs->obj2filename, file, g_strdup(filename));
273 g_signal_connect(file, "changed", G_CALLBACK(changed_cb), cvs);
274 }
275
276 g_free(filename);
277 }
278 }
279
280 if (dropbox_client_is_connected(&(cvs->dc)) == FALSE ||
281 caja_file_info_is_gone(file)) {
282 return CAJA_OPERATION_COMPLETE;
283 }
284
285 {
286 DropboxFileInfoCommand *dfic = g_new0(DropboxFileInfoCommand, 1);
287
288 dfic->cancelled = FALSE;
289 dfic->provider = provider;
290 dfic->dc.request_type = GET_FILE_INFO;
291 dfic->update_complete = g_closure_ref(update_complete);
292 dfic->file = g_object_ref(file);
293
294 dropbox_command_client_request(&(cvs->dc.dcc), (DropboxCommand *) dfic);
295
296 *handle = (CajaOperationHandle *) dfic;
297
298 return dropbox_use_operation_in_progress_workaround
299 ? CAJA_OPERATION_COMPLETE
300 : CAJA_OPERATION_IN_PROGRESS;
301 }
302}

304static void
305handle_shell_touch(GHashTable *args, CajaDropbox *cvs) {
306 gchar **path;
307
308 // debug_enter();
309
310 if ((path = g_hash_table_lookup(args, "path")) != NULL &&
311 path[0][0] == '/') {
312 CajaFileInfo *file;
313 gchar *filename;
314
315 filename = canonicalize_path(path[0]);
316 if (filename != NULL) {
317 debug("shell touch for %s", filename);
318
319 file = g_hash_table_lookup(cvs->filename2obj, filename);
320
321 if (file != NULL) {
322 debug("gonna reset %s", filename);
323 reset_file(file);
324 }
325 g_free(filename);
326 }
327 }
328
329 return;

332gboolean
333caja_dropbox_finish_file_info_command(DropboxFileInfoCommandResponse *dficr) {
334
335 //debug_enter();
336 CajaOperationResult result = CAJA_OPERATION_FAILED;
337
338 if (!dficr->dfic->cancelled) {
339 gchar **status = NULL;
340 gboolean isdir;
341
342 isdir = caja_file_info_is_directory(dficr->dfic->file) ;
343
344 /* if we have emblems just use them. */
345 if (dficr->emblems_response != NULL &&
346 (status = g_hash_table_lookup(dficr->emblems_response, "emblems")) != NULL) {
347 int i;
348 for ( i = 0; status[i] != NULL; i++) {
349 if (status[i][0])
350 caja_file_info_add_emblem(dficr->dfic->file, status[i]);
351 }
352 result = CAJA_OPERATION_COMPLETE;
353 }
354 /* if the file status command went okay */
355 else if ((dficr->file_status_response != NULL &&
356 (status =
357 g_hash_table_lookup(dficr->file_status_response, "status")) != NULL) &&
358 ((isdir == TRUE &&
359 dficr->folder_tag_response != NULL) || isdir == FALSE)) {
360 gchar **tag = NULL;
361
362 /* set the tag emblem */
363 if (isdir &&
364 (tag = g_hash_table_lookup(dficr->folder_tag_response, "tag")) != NULL) {
365 if (strcmp("public", tag[0]) == 0) {
366 caja_file_info_add_emblem(dficr->dfic->file, "web");
367 }
368 else if (strcmp("shared", tag[0]) == 0) {
369 caja_file_info_add_emblem(dficr->dfic->file, "people");
370 }
371 else if (strcmp("photos", tag[0]) == 0) {
372 caja_file_info_add_emblem(dficr->dfic->file, "photos");
373 }
374 else if (strcmp("sandbox", tag[0]) == 0) {
375 caja_file_info_add_emblem(dficr->dfic->file, "star");
376 }
377 }
378
379 /* set the status emblem */
380 {
381 int emblem_code = 0;
382
383 if (strcmp("up to date", status[0]) == 0) {
384 emblem_code = 1;
385 }
386 else if (strcmp("syncing", status[0]) == 0) {
387 emblem_code = 2;
388 }
389 else if (strcmp("unsyncable", status[0]) == 0) {
390 emblem_code = 3;
391 }
392
393 if (emblem_code > 0) {
394 /*
395 debug("%s to %s", emblems[emblem_code-1],
396 g_filename_from_uri(caja_file_info_get_uri(dficr->dfic->file),
397 NULL, NULL));
398 */
399 caja_file_info_add_emblem(dficr->dfic->file, emblems[emblem_code-1]);
400 }
401 }
402 result = CAJA_OPERATION_COMPLETE;
403 }
404 }
405
406 /* complete the info request */
407 if (!dropbox_use_operation_in_progress_workaround) {
408 caja_info_provider_update_complete_invoke(dficr->dfic->update_complete,
409 dficr->dfic->provider,
410 (CajaOperationHandle*) dficr->dfic,
411 result);
412 }
413
414 /* destroy the objects we created */
415 if (dficr->file_status_response != NULL)
416 g_hash_table_unref(dficr->file_status_response);
417 if (dficr->folder_tag_response != NULL)
418 g_hash_table_unref(dficr->folder_tag_response);
419 if (dficr->emblems_response != NULL)
420 g_hash_table_unref(dficr->emblems_response);
421
422 /* unref the objects we didn't create */
423 g_closure_unref(dficr->dfic->update_complete);
424 g_object_unref(dficr->dfic->file);
425
426 /* now free the structs */
427 g_free(dficr->dfic);
428 g_free(dficr);
429
430 return FALSE;
431}

444 gchar *verb;

500static char from_hex(gchar ch) {
501 return isdigit(ch) ? ch - '0' : tolower(ch) - 'a' + 10;
502}

504// decode in --> out, but dont fill more than n chars into out
505// returns len of out if thing went well, -1 if n wasn't big enough
506// can be used in place (whoa!)
507int GhettoURLDecode(gchar* out, gchar* in, int n) {
508 char *out_initial;
509
510 for(out_initial = out; out-out_initial < n && *in != '\0'; out++) {
511 if (*in == '%') {
512 if ((in[1] != '\0') && (in[2] != '\0')) {
513 *out = from_hex(in[1]) << 4 | from_hex(in[2]);
514 in += 3;
515 }
516 else {
517 // Input string isn't well-formed
518 return -1;
519 }
520 }
521 else {
522 *out = *in;
523 in++;
524 }
525 }
526
527 if (out-out_initial < n) {
528 *out = '\0';
529 return out-out_initial;
530 }
531 return -1;
532}

534static int
535caja_dropbox_parse_menu(gchar **options,
536 CajaMenu *menu,
537 GString *old_action_string,
538 GList *toret,
539 CajaMenuProvider *provider,
540 GList *files)
541{
542 int ret = 0;
543 int i;
544
545 for ( i = 0; options[i] != NULL; i++) {
546 gchar **option_info = g_strsplit(options[i], "~", 3);
547 /* if this is a valid string */
548 if (option_info[0] == NULL || option_info[1] == NULL ||
549 option_info[2] == NULL || option_info[3] != NULL) {
550 g_strfreev(option_info);
551 continue;
552 }
553
554 gchar* item_name = option_info[0];
555 gchar* item_inner = option_info[1];
556 gchar* verb = option_info[2];
557
558 GhettoURLDecode(item_name, item_name, strlen(item_name));
559 GhettoURLDecode(verb, verb, strlen(verb));
560 GhettoURLDecode(item_inner, item_inner, strlen(item_inner));
561
562 // If the inner section has a menu in it then we create a submenu. The verb will be ignored.
563 // Otherwise add the verb to our map and add the menu item to the list.
564 if (strchr(item_inner, '~') != NULL) {
565 GString *new_action_string = g_string_new(old_action_string->str);
566 gchar **suboptions = g_strsplit(item_inner, "|", -1);
567 CajaMenuItem *item;
568 CajaMenu *submenu = caja_menu_new();
569
570 g_string_append(new_action_string, item_name);
571 g_string_append(new_action_string, "::");
572
573 ret += caja_dropbox_parse_menu(suboptions, submenu, new_action_string,
574 toret, provider, files);
575
576 item = caja_menu_item_new(new_action_string->str,
577 item_name, "", NULL);
578 caja_menu_item_set_submenu(item, submenu);
579 caja_menu_append_item(menu, item);
580
581 g_strfreev(suboptions);
582 g_object_unref(item);
583 g_object_unref(submenu);
584 g_string_free(new_action_string, TRUE);
585 } else {
586 CajaMenuItem *item;
587 GString *new_action_string = g_string_new(old_action_string->str);
588 gboolean grayed_out = FALSE;
589
590 g_string_append(new_action_string, verb);
591
592 if (item_name[0] == '!') {
593 item_name++;
594 grayed_out = TRUE;
595 }
596
597 item = caja_menu_item_new(new_action_string->str, item_name, item_inner, NULL);
598
599 caja_menu_append_item(menu, item);
600 /* add the file metadata to this item */
601 g_object_set_data_full (G_OBJECT(item), "caja_dropbox_files",
602 caja_file_info_list_copy (files),
603 (GDestroyNotify) caja_file_info_list_free);
604 /* add the verb metadata */
605 g_object_set_data_full (G_OBJECT(item), "caja_dropbox_verb",
606 g_strdup(verb),
607 (GDestroyNotify) g_free);
608 g_signal_connect (item, "activate", G_CALLBACK (menu_item_cb), provider);
609
610 if (grayed_out) {
611 GValue sensitive = { 0 };
612 g_value_init (&sensitive, G_TYPE_BOOLEAN);
613 g_value_set_boolean (&sensitive, FALSE);
614 g_object_set_property (G_OBJECT(item), "sensitive", &sensitive);
615 }
616
617 g_object_unref(item);
618 g_string_free(new_action_string, TRUE);
619 ret++;
620 }
621 g_strfreev(option_info);
622 }
623 return ret;
624}

626static void
627get_file_items_callback(GHashTable *response, gpointer ud)
628{
629 GAsyncQueue *reply_queue = ud;
630
631 /* queue_push doesn't accept NULL as a value so we create an empty hash table
632 * if we got no response. */
633 g_async_queue_push(reply_queue, response ? g_hash_table_ref(response) :
634 g_hash_table_new((GHashFunc) g_str_hash, (GEqualFunc) g_str_equal));
635 g_async_queue_unref(reply_queue);
636}

639static GList *
640caja_dropbox_get_file_items(CajaMenuProvider *provider,
641 GtkWidget *window,
642 GList *files)
643{
644 /*
645 * 1. Convert files to filenames.
646 */
647 int file_count = g_list_length(files);
648
649 if (file_count < 1)
650 return NULL;
651
652 gchar **paths = g_new0(gchar *, file_count + 1);
653 int i = 0;
654 GList* elem;
655
656 for (elem = files; elem; elem = elem->next, i++) {
657 gchar *uri = caja_file_info_get_uri(elem->data);
658 gchar *filename_un = uri ? g_filename_from_uri(uri, NULL, NULL) : NULL;
659 gchar *filename = filename_un ? g_filename_to_utf8(filename_un, -1, NULL, NULL, NULL) : NULL;
660
661 g_free(uri);
662 g_free(filename_un);
663
664 if (filename == NULL) {
665 /* oooh, filename wasn't correctly encoded, or isn't a local file. */
666 g_strfreev(paths);
667 return NULL;
668 }
669
670 paths[i] = filename;
671 }
672
673 GAsyncQueue *reply_queue = g_async_queue_new_full((GDestroyNotify)g_hash_table_unref);
674
675 /*
676 * 2. Create a DropboxGeneralCommand to call "icon_overlay_context_options"
677 */
678
679 DropboxGeneralCommand *dgc = g_new0(DropboxGeneralCommand, 1);
680 dgc->dc.request_type = GENERAL_COMMAND;
681 dgc->command_name = g_strdup("icon_overlay_context_options");
682 dgc->command_args = g_hash_table_new_full((GHashFunc) g_str_hash,
683 (GEqualFunc) g_str_equal,
684 (GDestroyNotify) g_free,
685 (GDestroyNotify) g_strfreev);
686 g_hash_table_insert(dgc->command_args, g_strdup("paths"), paths);
687 dgc->handler = get_file_items_callback;
688 dgc->handler_ud = g_async_queue_ref(reply_queue);
689
690 /*
691 * 3. Queue it up for the helper thread to run it.
692 */
693 CajaDropbox *cvs = CAJA_DROPBOX(provider);
694 dropbox_command_client_request(&(cvs->dc.dcc), (DropboxCommand *) dgc);
695
696 /*
697 * 4. We have to block until it's done because caja expects a reply. But we will
698 * only block for 50 ms for a reply.
699 */
700
701 GHashTable *context_options_response = g_async_queue_timeout_pop(reply_queue, 50000);
702 g_async_queue_unref(reply_queue);
703
704 if (!context_options_response) {
705 return NULL;
706 }
707
708 /*
709 * 5. Parse the reply.
710 */
711
712 char **options = g_hash_table_lookup(context_options_response, "options");
713 GList *toret = NULL;
714
715 if (options && *options && **options) {
716 /* build the menu */
717 CajaMenuItem *root_item;
718 CajaMenu *root_menu;
719
720 root_menu = caja_menu_new();
721 root_item = caja_menu_item_new("CajaDropbox::root_item",
722 "Dropbox", "Dropbox Options", "dropbox");
723
724 toret = g_list_append(toret, root_item);
725 GString *action_string = g_string_new("CajaDropbox::");
726
727 if (!caja_dropbox_parse_menu(options, root_menu, action_string,
728 toret, provider, files)) {
729 g_object_unref(toret);
730 toret = NULL;
731 }
732
733 caja_menu_item_set_submenu(root_item, root_menu);
734
735 g_string_free(action_string, TRUE);
736 g_object_unref(root_menu);
737 }
738
739 g_hash_table_unref(context_options_response);
740
741 return toret;
742}

744gboolean
745add_emblem_paths(GHashTable* emblem_paths_response)
746{
747 /* Only run this on the main loop or you'll cause problems. */
748 if (!emblem_paths_response)
749 return FALSE;
750
751 gchar **emblem_paths_list;
752 int i;
753
754 GtkIconTheme *theme = gtk_icon_theme_get_default();
755
756 if (emblem_paths_response &&
757 (emblem_paths_list = g_hash_table_lookup(emblem_paths_response, "path"))) {
758 for (i = 0; emblem_paths_list[i] != NULL; i++) {
759 if (emblem_paths_list[i][0])
760 gtk_icon_theme_append_search_path(theme, emblem_paths_list[i]);
761 }
762 }
763 g_hash_table_unref(emblem_paths_response);
764 return FALSE;

767gboolean
768remove_emblem_paths(GHashTable* emblem_paths_response)
769{
770 /* Only run this on the main loop or you'll cause problems. */
771 if (!emblem_paths_response)
772 return FALSE;
773
774 gchar **emblem_paths_list = g_hash_table_lookup(emblem_paths_response, "path");
775 if (!emblem_paths_list)
776 goto exit;
777
778 // We need to remove the old paths.
779 GtkIconTheme * icon_theme = gtk_icon_theme_get_default();
780 gchar ** paths;
781 gint path_count;
782
783 gtk_icon_theme_get_search_path(icon_theme, &paths, &path_count);
784
785 gint i, j, out = 0;
786 gboolean found = FALSE;
787 for (i = 0; i < path_count; i++) {
788 gboolean keep = TRUE;
789 for (j = 0; emblem_paths_list[j] != NULL; j++) {
790 if (emblem_paths_list[j][0]) {
791 if (!g_strcmp0(paths[i], emblem_paths_list[j])) {
792 found = TRUE;
793 keep = FALSE;
794 g_free(paths[i]);
795 break;
796 }
797 }
798 }
799 if (keep) {
800 paths[out] = paths[i];
801 out++;
802 }
803 }
804
805 /* If we found one we need to reset the path to
806 accomodate the changes */
807 if (found) {
808 paths[out] = NULL; /* Clear the last one */
809 gtk_icon_theme_set_search_path(icon_theme, (const gchar **)paths, out);
810 }
811
812 g_strfreev(paths);
813exit:
814 g_hash_table_unref(emblem_paths_response);
815 return FALSE;
816}

818void get_emblem_paths_cb(GHashTable *emblem_paths_response, CajaDropbox *cvs)
819{
820 if (!emblem_paths_response) {
821 emblem_paths_response = g_hash_table_new((GHashFunc) g_str_hash,
822 (GEqualFunc) g_str_equal);
823 g_hash_table_insert(emblem_paths_response, "path", DEFAULT_EMBLEM_PATHS);
824 } else {
825 /* Increase the ref so that finish_general_command doesn't delete it. */
826 g_hash_table_ref(emblem_paths_response);
827 }
828
829 g_mutex_lock(&(cvs->emblem_paths_mutex));
830 if (cvs->emblem_paths) {
831 g_idle_add((GSourceFunc) remove_emblem_paths, cvs->emblem_paths);
832 cvs->emblem_paths = NULL;
833 }
834 cvs->emblem_paths = emblem_paths_response;
835 g_mutex_unlock(&(cvs->emblem_paths_mutex));
836
837 g_idle_add((GSourceFunc) add_emblem_paths, g_hash_table_ref(emblem_paths_response));
838 g_idle_add((GSourceFunc) reset_all_files, cvs);
839}
