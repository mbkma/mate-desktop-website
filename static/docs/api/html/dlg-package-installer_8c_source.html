47static void
48package_installer_terminated (InstallerData *idata,
49 FrProcErrorType error_type,
50 const char *error_message)
51{
52 GdkWindow *window;
53
54 window = gtk_widget_get_window (GTK_WIDGET (idata->window));
55 if (window != NULL)
56 gdk_window_set_cursor (window, NULL);
57
58 if (error_type != FR_PROC_ERROR_NONE) {
59 fr_archive_action_completed (idata->archive,
60 idata->action,
61 error_type,
62 error_message);
63 }
64 else {
65 update_registered_commands_capabilities ();
66 if (fr_window_is_batch_mode (idata->window))
67 fr_window_resume_batch (idata->window);
68 else
69 fr_window_restart_current_batch_action (idata->window);
70 }
71
72 installer_data_free (idata);
73}

77static void
78packagekit_install_package_names_ready_cb (GObject *source_object,
79 GAsyncResult *res,
80 gpointer user_data)
81{
82 InstallerData *idata = user_data;
83 GDBusProxy *proxy;
84 GVariant *values;
85 GError *error = NULL;
86 FrProcErrorType error_type = FR_PROC_ERROR_NONE;
87 char *error_message = NULL;
88
89 proxy = G_DBUS_PROXY (source_object);
90 values = g_dbus_proxy_call_finish (proxy, res, &error);
91 if (values == NULL) {
92 if (g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED)
93 || (g_error_matches (error, G_IO_ERROR, G_IO_ERROR_DBUS_ERROR)
94 && (error->message != NULL)
95 && (strstr (error->message, "org.freedesktop.Packagekit.Modify.Cancelled") != NULL)))
96 {
97 error_type = FR_PROC_ERROR_STOPPED;
98 error_message = NULL;
99 }
100 else {
101 error_type = FR_PROC_ERROR_GENERIC;
102 error_message = g_strdup_printf ("%s\n%s",
103 _("There was an internal error trying to search for applications:"),
104 error->message);
105 }
106 g_clear_error (&error);
107 }
108
109 package_installer_terminated (idata, error_type, error_message);
110
111 g_free (error_message);
112 if (values != NULL)
113 g_variant_unref (values);
114 g_object_unref (proxy);
115}

117static char **
118get_packages_real_names (char **names)
119{
120 char **real_names;
121 GKeyFile *key_file;
122 int i;
123
124 real_names = g_new0 (char *, g_strv_length (names));
125 key_file = g_key_file_new ();
126 g_key_file_load_from_file (key_file,
127 PRIVDATADIR "/packages.match",
128 G_KEY_FILE_NONE, NULL);
129
130 for (i = 0; names[i] != NULL; i++) {
131 char *real_name;
132
133 real_name = g_key_file_get_string (key_file, "Package Matches", names[i], NULL);
134 if (real_name != NULL)
135 real_name = g_strstrip (real_name);
136 if ((real_name == NULL) || (strncmp (real_name, "", 1) == 0)) {
137 g_free (real_name);
138 real_name = g_strdup (names[i]);
139 }
140 real_names[i] = real_name;
141 real_name = NULL;
142 }
143
144 g_key_file_free (key_file);
145
146 return real_names;
147}

149static void
150install_packages (InstallerData *idata)
151{
152 GDBusConnection *connection;
153 GError *error = NULL;
154
155 connection = g_bus_get_sync (G_BUS_TYPE_SESSION, NULL, &error);
156 if (connection != NULL) {
157 GdkWindow *window;
158 GDBusProxy *proxy;
159
160 window = gtk_widget_get_window (GTK_WIDGET (idata->window));
161 if (window != NULL) {
162 GdkCursor *cursor;
163 GdkDisplay *display;
164
165 display = gtk_widget_get_display (GTK_WIDGET (idata));
166 cursor = gdk_cursor_new_for_display (display, GDK_WATCH);
167 gdk_window_set_cursor (window, cursor);
168 g_object_unref (cursor);
169 }
170
171 proxy = g_dbus_proxy_new_sync (connection,
172 G_DBUS_PROXY_FLAGS_NONE,
173 NULL,
174 "org.freedesktop.PackageKit",
175 "/org/freedesktop/PackageKit",
176 "org.freedesktop.PackageKit.Modify",
177 NULL,
178 &error);
179
180 if (proxy != NULL) {
181 guint xid;
182 char **names;
183 char **real_names;
184
185 if (window != NULL)
186 xid = GDK_WINDOW_XID (window);
187 else
188 xid = 0;
189
190 names = g_strsplit (idata->packages, ",", -1);
191 real_names = get_packages_real_names (names);
192
193 g_dbus_proxy_call (proxy,
194 "InstallPackageNames",
195 g_variant_new ("(u^ass)",
196 xid,
197 real_names,
198 "hide-confirm-search,hide-finished,hide-warning"),
199 G_DBUS_CALL_FLAGS_NONE,
200 G_MAXINT,
201 NULL,
202 packagekit_install_package_names_ready_cb,
203 idata);
204
205 g_strfreev (real_names);
206 g_strfreev (names);
207 }
208 }
209
210 if (error != NULL) {
211 char *message;
212
213 message = g_strdup_printf ("%s\n%s",
214 _("There was an internal error trying to search for applications:"),
215 error->message);
216 package_installer_terminated (idata, FR_PROC_ERROR_GENERIC, message);
217
218 g_clear_error (&error);
219 }
220}

265 FrArchive *archive,
266 FrAction action)
267{
268 InstallerData *idata;
269 GType command_type;
270 FrCommand *command;
271
272 idata = g_new0 (InstallerData, 1);
273 idata->window = g_object_ref (window);
274 idata->archive = g_object_ref (archive);
275 idata->action = action;
276
277 command_type = get_preferred_command_for_mime_type (idata->archive->content_type, FR_COMMAND_CAN_READ_WRITE);
278 if (command_type == 0)
279 command_type = get_preferred_command_for_mime_type (idata->archive->content_type, FR_COMMAND_CAN_READ);
280 if (command_type == 0) {
281 package_installer_terminated (idata, FR_PROC_ERROR_GENERIC, _("Archive type not supported."));
282 return;
283 }
284
285 command = g_object_new (command_type, 0);
286 idata->packages = fr_command_get_packages (command, idata->archive->content_type);
287 g_object_unref (command);
288
289 if (idata->packages == NULL) {
290 package_installer_terminated (idata, FR_PROC_ERROR_GENERIC, _("Archive type not supported."));
291 return;
292 }
