153static void
154cpufreq_monitor_finalize (GObject *object)
155{
156 CPUFreqMonitor *monitor = CPUFREQ_MONITOR (object);
157
158 monitor->priv->online = FALSE;
159
160 if (monitor->priv->timeout_handler > 0) {
161 g_source_remove (monitor->priv->timeout_handler);
162 monitor->priv->timeout_handler = 0;
163 }
164
165 if (monitor->priv->governor) {
166 g_free (monitor->priv->governor);
167 monitor->priv->governor = NULL;
168 }
169
170 if (monitor->priv->available_freqs) {
171 g_list_free_full (monitor->priv->available_freqs, g_free);
172 monitor->priv->available_freqs = NULL;
173 }
174
175 if (monitor->priv->available_govs) {
176 g_list_free_full (monitor->priv->available_govs, g_free);
177 monitor->priv->available_govs = NULL;
178 }
179
180 G_OBJECT_CLASS (cpufreq_monitor_parent_class)->finalize (object);
181}

183static void
184cpufreq_monitor_set_property (GObject *object,
185 guint prop_id,
186 const GValue *value,
187 GParamSpec *spec)
188{
189 CPUFreqMonitor *monitor;
190
191 monitor = CPUFREQ_MONITOR (object);
192
193 switch (prop_id) {
194 case PROP_CPU: {
195 guint cpu = g_value_get_uint (value);
196
197 if (cpu != monitor->priv->cpu) {
198 monitor->priv->cpu = cpu;
199 monitor->priv->changed = TRUE;
200 }
201 break;
202 }
203 case PROP_ONLINE:
204 monitor->priv->online = g_value_get_boolean (value);
205
206 break;
207 case PROP_FREQUENCY: {
208 gint freq = g_value_get_int (value);
209
210 if (freq != monitor->priv->cur_freq) {
211 monitor->priv->cur_freq = freq;
212 monitor->priv->changed = TRUE;
213 }
214 break;
215 }
216 case PROP_MAX_FREQUENCY: {
217 gint freq = g_value_get_int (value);
218
219 if (freq != monitor->priv->max_freq) {
220 monitor->priv->max_freq = freq;
221 monitor->priv->changed = TRUE;
222 }
223 break;
224 }
225 case PROP_GOVERNOR: {
226 const gchar *gov = g_value_get_string (value);
227
228 if (monitor->priv->governor) {
229 if (g_ascii_strcasecmp (gov, monitor->priv->governor) != 0) {
230 g_free (monitor->priv->governor);
231 monitor->priv->governor = gov ? g_strdup (gov) : NULL;
232 monitor->priv->changed = TRUE;
233 }
234 } else {
235 monitor->priv->governor = gov ? g_strdup (gov) : NULL;
236 monitor->priv->changed = TRUE;
237 }
238 break;
239 }
240 default:
241 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, spec);
242 }
243}

276static gboolean
277cpufreq_monitor_run_cb (CPUFreqMonitor *monitor)
278{
279 CPUFreqMonitorClass *class;
280 gboolean retval = FALSE;
281
282 class = CPUFREQ_MONITOR_GET_CLASS (monitor);
283
284 if (class->run)
285 retval = class->run (monitor);
286
287 if (monitor->priv->changed) {
288 g_signal_emit (monitor, signals[SIGNAL_CHANGED], 0);
289 monitor->priv->changed = FALSE;
290 }
291
292 return retval;
293}

295void
296cpufreq_monitor_run (CPUFreqMonitor *monitor)
297{
298 g_return_if_fail (CPUFREQ_IS_MONITOR (monitor));
299
300 if (monitor->priv->timeout_handler > 0)
301 return;
302
303 monitor->priv->timeout_handler =
304 g_timeout_add_seconds (CPUFREQ_MONITOR_INTERVAL,
305 (GSourceFunc) cpufreq_monitor_run_cb,
306 (gpointer) monitor);
