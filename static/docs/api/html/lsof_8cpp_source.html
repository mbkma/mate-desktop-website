50 template<typename OutputIterator>
51 void search(const ProcInfo &info, OutputIterator out) const
52 {
53 glibtop_open_files_entry *entries;
54 glibtop_proc_open_files buf;
55
56 entries = glibtop_get_proc_open_files(&buf, info.pid);
57
58 for (unsigned i = 0; i != buf.number; ++i) {
59 if (entries[i].type & GLIBTOP_FILE_TYPE_FILE) {
60 const string filename(entries[i].info.file.name);
61 if (this->matches(filename))
62 *out++ = filename;
63 }
64 }
65
66 g_free(entries);
67 }

153 void search()
154 {
155 typedef std::set<string> MatchSet;
156 typedef MatchSet::const_iterator iterator;
157
158 this->clear_results();
159
160
161 try {
162 Lsof lsof(this->pattern(), this->case_insensitive);
163
164 unsigned count = 0;
165
166 for (ProcInfo::Iterator it(ProcInfo::begin()); it != ProcInfo::end(); ++it) {
167 const ProcInfo &info(*it->second);
168
169 MatchSet matches;
170 lsof.search(info, std::inserter(matches, matches.begin()));
171 count += matches.size();
172
173 for (iterator it(matches.begin()), end(matches.end()); it != end; ++it) {
174 GtkTreeIter file;
175 gtk_list_store_append(this->model, &file);
176 gtk_list_store_set(this->model, &file,
177 PROCMAN_LSOF_COL_SURFACE, info.surface,
178 PROCMAN_LSOF_COL_PROCESS, info.name,
179 PROCMAN_LSOF_COL_PID, info.pid,
180 PROCMAN_LSOF_COL_FILENAME, it->c_str(),
181 -1);
182 }
183 }
184
185 this->update_count(count);
186 }
187 catch (Glib::RegexError& error) {
188 this->display_regex_error(error);
189 }
190 }
