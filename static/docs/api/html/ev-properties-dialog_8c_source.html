55static gboolean
56on_notebook_scroll_event (GtkWidget *widget,
57 GdkEventScroll *event)
58{
59 GtkNotebook *notebook = GTK_NOTEBOOK (widget);
60 GtkWidget *child, *event_widget, *action_widget;
61
62 child = gtk_notebook_get_nth_page (notebook, gtk_notebook_get_current_page (notebook));
63 if (child == NULL)
64 return FALSE;
65
66 event_widget = gtk_get_event_widget ((GdkEvent*) event);
67
68 /* Ignore scroll events from the content of the page */
69 if (event_widget == NULL || event_widget == child || gtk_widget_is_ancestor (event_widget, child))
70 return FALSE;
71
72 /* And also from the action widgets */
73 action_widget = gtk_notebook_get_action_widget (notebook, GTK_PACK_START);
74 if (event_widget == action_widget || (action_widget != NULL && gtk_widget_is_ancestor (event_widget, action_widget)))
75 return FALSE;
76
77 action_widget = gtk_notebook_get_action_widget (notebook, GTK_PACK_END);
78 if (event_widget == action_widget || (action_widget != NULL && gtk_widget_is_ancestor (event_widget, action_widget)))
79 return FALSE;
80
81 switch (event->direction) {
82 case GDK_SCROLL_RIGHT:
83 case GDK_SCROLL_DOWN:
84 gtk_notebook_next_page (notebook);
85 break;
86 case GDK_SCROLL_LEFT:
87 case GDK_SCROLL_UP:
88 gtk_notebook_prev_page (notebook);
89 break;
90 case GDK_SCROLL_SMOOTH:
91 switch (gtk_notebook_get_tab_pos (notebook)) {
92 case GTK_POS_LEFT:
93 case GTK_POS_RIGHT:
94 if (event->delta_y > 0)
95 gtk_notebook_next_page (notebook);
96 else if (event->delta_y < 0)
97 gtk_notebook_prev_page (notebook);
98 break;
99 case GTK_POS_TOP:
100 case GTK_POS_BOTTOM:
101 if (event->delta_x > 0)
102 gtk_notebook_next_page (notebook);
103 else if (event->delta_x < 0)
104 gtk_notebook_prev_page (notebook);
105 break;
106 }
107 break;
108 }
109
110 return TRUE;
111}

113static void
114ev_properties_dialog_init (EvPropertiesDialog *properties)
115{
116 GtkBox *content_area;
117
118 content_area = GTK_BOX (gtk_dialog_get_content_area (GTK_DIALOG (properties)));
119
120 gtk_window_set_title (GTK_WINDOW (properties), _("Properties"));
121 gtk_window_set_destroy_with_parent (GTK_WINDOW (properties), TRUE);
122 gtk_container_set_border_width (GTK_CONTAINER (properties), 5);
123 gtk_box_set_spacing (content_area, 2);
124
125 gtk_dialog_add_button (GTK_DIALOG (properties), "gtk-close",
126 GTK_RESPONSE_CANCEL);
127 gtk_dialog_set_default_response (GTK_DIALOG (properties),
128 GTK_RESPONSE_CANCEL);
129
130 properties->notebook = gtk_notebook_new ();
131 gtk_widget_add_events (properties->notebook, GDK_SCROLL_MASK);
132 g_signal_connect (properties->notebook,
133 "scroll-event",
134 G_CALLBACK (on_notebook_scroll_event),
135 NULL);
136
137 gtk_container_set_border_width (GTK_CONTAINER (properties->notebook), 5);
138 gtk_box_pack_start (content_area, properties->notebook, TRUE, TRUE, 0);
139 gtk_widget_show (properties->notebook);
140
141 g_signal_connect (properties, "response",
142 G_CALLBACK (gtk_widget_destroy), NULL);
143}

145void
146ev_properties_dialog_set_document (EvPropertiesDialog *properties,
147 const gchar *uri,
148 EvDocument *document)
149{
150 GtkWidget *label;
151 const EvDocumentInfo *info;
152
153 properties->document = document;
154
155 info = ev_document_get_info (document);
156
157 if (properties->general_page == NULL) {
158 label = gtk_label_new (_("General"));
159 properties->general_page = ev_properties_view_new (uri);
160 gtk_notebook_append_page (GTK_NOTEBOOK (properties->notebook),
161 properties->general_page, label);
162 gtk_widget_show (properties->general_page);
163 }
164 ev_properties_view_set_info (EV_PROPERTIES_VIEW (properties->general_page), info);
165
166 if (EV_IS_DOCUMENT_FONTS (document)) {
167 if (properties->fonts_page == NULL) {
168 label = gtk_label_new (_("Fonts"));
169 properties->fonts_page = ev_properties_fonts_new ();
170 gtk_notebook_append_page (GTK_NOTEBOOK (properties->notebook),
171 properties->fonts_page, label);
172 gtk_widget_show (properties->fonts_page);
173 }
174
175 ev_properties_fonts_set_document
176 (EV_PROPERTIES_FONTS (properties->fonts_page), document);
177 }
178
179 if (info->fields_mask & EV_DOCUMENT_INFO_LICENSE && info->license) {
180 if (properties->license_page == NULL) {
181 label = gtk_label_new (_("Document License"));
182 properties->license_page = ev_properties_license_new ();
183 gtk_notebook_append_page (GTK_NOTEBOOK (properties->notebook),
184 properties->license_page, label);
185 gtk_widget_show (properties->license_page);
186 }
187
188 ev_properties_license_set_license
189 (EV_PROPERTIES_LICENSE (properties->license_page), info->license);
190 }
191}
