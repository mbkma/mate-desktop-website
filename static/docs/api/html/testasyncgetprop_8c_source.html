87static int
88x_error_handler (Display *xdisplay,
89 XErrorEvent *error)
90{
91 char buf[64];
92
93 XGetErrorText (xdisplay, error->error_code, buf, 63);
94
95 if (error_trap_depth == 0)
96 {
97 print_backtrace ();
98
99 fprintf (stderr, "Unexpected X error: %s serial %ld error_code %d request_code %d minor_code %d)\n",
100 buf,
101 error->serial,
102 error->error_code,
103 error->request_code,
104 error->minor_code);
105
106 exit (1);
107 }
108
109 return 1; /* return value is meaningless */
110}

112static void
113error_trap_push (Display *xdisplay)
114{
115 ++error_trap_depth;

118static void
119error_trap_pop (Display *xdisplay)
120{
121 if (error_trap_depth == 0)
122 {
123 fprintf (stderr, "Error trap underflow!\n");
124 exit (1);
125 }
126
127 XSync (xdisplay, False); /* get all errors out of the queue */
128 --error_trap_depth;
129}

147static char*
148atom_name (Display *display,
149 Atom atom)
150{
151 if (atom == None)
152 {
153 return my_strdup ("None");
154 }
155 else
156 {
157 char *xname;
158 char *ret;
159
160 error_trap_push (display);
161 xname = XGetAtomName (display, atom);
162 error_trap_pop (display);
163 if (xname == NULL)
164 return my_strdup ("[unknown atom]");
165
166 ret = my_strdup (xname);
167 XFree (xname);
168
169 return ret;
170 }
171}

179static Bool
180try_get_reply (Display *xdisplay,
181 AgGetPropertyTask *task)
182{
183 if (ag_task_have_reply (task))
184 {
185 int result;
186 Atom actual_type;
187 int actual_format;
188 unsigned long n_items;
189 unsigned long bytes_after;
190 unsigned char *data;
191 char *name;
192 struct timeval current_time;
193
194 gettimeofday (&current_time, NULL);
195
196 printf (" %gms (we have a reply for property %lu)\n",
197 ELAPSED (program_start_time, current_time),
198 ag_task_get_property (task));
199
200 data = NULL;
201
202 name = atom_name (xdisplay,
203 ag_task_get_property (task));
204 printf (" %s on 0x%lx:\n", name,
205 ag_task_get_window (task));
206 free (name);
207
208 result = ag_task_get_reply_and_free (task,
209 &actual_type,
210 &actual_format,
211 &n_items,
212 &bytes_after,
213 &data);
214 task = NULL;
215
216 if (result != Success)
217 {
218 fprintf (stderr, " error code %d getting reply\n", result);
219 }
220 else
221 {
222 name = atom_name (xdisplay, actual_type);
223 printf (" actual_type = %s\n", name);
224 free (name);
225
226 printf (" actual_format = %d\n", actual_format);
227
228 printf (" n_items = %lu\n", n_items);
229 printf (" bytes_after = %lu\n", bytes_after);
230
231 printf (" data = \"%s\"\n", data ? (char*) data : "NULL");
232 }
233
234 return True;
235 }
236
237 return False;
238}

245{
246 Display *xdisplay;
247 int i;
248 int n_left;
249 int n_props;
250 Window window;
251 const char *window_str;
252 char *end;
253 Atom *props;
254 struct timeval current_time;
255
256 if (argc < 2)
257 {
258 fprintf (stderr, "specify window ID\n");
259 return 1;
260 }
261
262 window_str = argv[1];
263
264 end = NULL;
265 window = strtoul (window_str, &end, 0);
266 if (end == NULL || *end != '\0')
267 {
268 fprintf (stderr, "\"%s\" does not parse as a window ID\n", window_str);
269 return 1;
270 }
271
272 xdisplay = XOpenDisplay (NULL);
273 if (xdisplay == NULL)
274 {
275 fprintf (stderr, "Could not open display\n");
276 return 1;
277 }
278
279 if (getenv ("MARCO_SYNC") != NULL)
280 XSynchronize (xdisplay, True);
281
282 XSetErrorHandler (x_error_handler);
283
284 n_props = 0;
285 props = XListProperties (xdisplay, window, &n_props);
286 if (n_props == 0 || props == NULL)
287 {
288 fprintf (stderr, "Window has no properties\n");
289 return 1;
290 }
291
292 gettimeofday (&program_start_time, NULL);
293
294 i = 0;
295 while (i < n_props)
296 {
297 gettimeofday (&current_time, NULL);
298 printf (" %gms (sending request for property %lu)\n",
299 ELAPSED (program_start_time, current_time),
300 props[i]);
301 if (ag_task_create (xdisplay,
302 window, props[i],
303 0, 0xffffffff,
304 False,
305 AnyPropertyType) == NULL)
306 {
307 fprintf (stderr, "Failed to send request\n");
308 return 1;
309 }
310
311 ++i;
312 }
313
314 XFree (props);
315 props = NULL;
316
317 n_left = n_props;
318
319 while (TRUE)
320 {
321 XEvent xevent;
322 int connection;
323 fd_set set;
324 AgGetPropertyTask *task;
325
326 /* Mop up event queue */
327 while (XPending (xdisplay) > 0)
328 {
329 XNextEvent (xdisplay, &xevent);
330 gettimeofday (&current_time, NULL);
331 printf (" %gms (processing event type %d)\n",
332 ELAPSED (program_start_time, current_time),
333 xevent.xany.type);
334 }
335
336 while ((task = ag_get_next_completed_task (xdisplay)))
337 {
338 try_get_reply (xdisplay, task);
339 n_left -= 1;
340 }
341
342 if (n_left == 0)
343 {
344 printf ("All %d replies received.\n", n_props);
345 break;
346 }
347
348 /* Wake up if we may have a reply */
349 connection = ConnectionNumber (xdisplay);
350
351 FD_ZERO (&set);
352 FD_SET (connection, &set);
353
354 gettimeofday (&current_time, NULL);
355 printf (" %gms (blocking for data %d left)\n",
356 ELAPSED (program_start_time, current_time), n_left);
357 select (connection + 1, &set, NULL, NULL, NULL);
358 }
359
360 run_speed_comparison (xdisplay, window);
361
362 return 0;
363}
