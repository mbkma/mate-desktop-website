70static char *
71make_exec_uri (const char *exec)
72{
73 GString *str;
74 const char *c;
75
76 if (exec == NULL) {
77 return g_strdup ("");
78 }
79
80 if (strchr (exec, ' ') == NULL) {
81 return g_strdup (exec);
82 }
83
84 str = g_string_new_len (NULL, strlen (exec));
85
86 str = g_string_append_c (str, '"');
87 for (c = exec; *c != '\0'; c++) {
88 /* FIXME: GKeyFile will add an additional backslach so we'll
89 * end up with toto\\" instead of toto\"
90 * We could use g_key_file_set_value(), but then we don't
91 * benefit from the other escaping that glib is doing...
92 */
93 if (*c == '"') {
94 str = g_string_append (str, "\\\"");
95 } else {
96 str = g_string_append_c (str, *c);
97 }
98 }
99 str = g_string_append_c (str, '"');
100
101 return g_string_free (str, FALSE);
102}

104static void
105on_browse_button_clicked (GtkWidget *widget,
106 GsmAppDialog *dialog)
107{
108 GtkWidget *chooser;
109 int response;
110
111 chooser = gtk_file_chooser_dialog_new ("",
112 GTK_WINDOW (dialog),
113 GTK_FILE_CHOOSER_ACTION_OPEN,
114 "gtk-cancel",
115 GTK_RESPONSE_CANCEL,
116 "gtk-open",
117 GTK_RESPONSE_ACCEPT,
118 NULL);
119
120 gtk_window_set_transient_for (GTK_WINDOW (chooser),
121 GTK_WINDOW (dialog));
122
123 gtk_window_set_destroy_with_parent (GTK_WINDOW (chooser), TRUE);
124
125 gtk_window_set_title (GTK_WINDOW (chooser), _("Select Command"));
126
127 gtk_widget_show (chooser);
128
129 response = gtk_dialog_run (GTK_DIALOG (chooser));
130
131 if (response == GTK_RESPONSE_ACCEPT) {
132 char *text;
133 char *uri;
134
135 text = gtk_file_chooser_get_filename (GTK_FILE_CHOOSER (chooser));
136
137 uri = make_exec_uri (text);
138
139 g_free (text);
140
141 gtk_entry_set_text (GTK_ENTRY (dialog->command_entry), uri);
142
143 g_free (uri);
144 }
145
146 gtk_widget_destroy (chooser);
147}

156static gboolean
157on_spin_output (GtkSpinButton *spin, GsmAppDialog *dialog)
158{
159 GtkAdjustment *adjustment;
160 gchar *text;
161 int value;
162
163 adjustment = gtk_spin_button_get_adjustment (spin);
164 value = gtk_adjustment_get_value (adjustment);
165 dialog->delay = value;
166
167 if (value == 1)
168 text = g_strdup_printf ("%d %s", value, _("second"));
169 else if (value > 1)
170 text = g_strdup_printf ("%d %s", value, _("seconds"));
171 else
172 text = g_strdup_printf ("%d", value);
173
174 gtk_entry_set_text (GTK_ENTRY (spin), text);
175 g_free (text);
176
177 return TRUE;
178}

180static void
181setup_dialog (GsmAppDialog *dialog)
182{
183 GtkWidget *content_area;
184 GtkWidget *widget;
185 GtkBuilder *xml;
186 GError *error;
187
188 xml = gtk_builder_new ();
189#ifdef ENABLE_NLS
190 gtk_builder_set_translation_domain (xml, GETTEXT_PACKAGE);
191#endif /* ENABLE_NLS */
192
193 error = NULL;
194 if (!gtk_builder_add_from_file (xml,
195 GTKBUILDER_DIR "/" GTKBUILDER_FILE,
196 &error)) {
197 if (error) {
198 g_warning ("Could not load capplet UI file: %s",
199 error->message);
200 g_error_free (error);
201 } else {
202 g_warning ("Could not load capplet UI file.");
203 }
204 }
205
206 content_area = gtk_dialog_get_content_area (GTK_DIALOG (dialog));
207 widget = GTK_WIDGET (gtk_builder_get_object (xml, "main-table"));
208 gtk_container_add (GTK_CONTAINER (content_area), widget);
209
210 gtk_container_set_border_width (GTK_CONTAINER (dialog), 6);
211 gtk_window_set_icon_name (GTK_WINDOW (dialog), "mate-session-properties");
212
213 g_object_set (dialog,
214 "resizable", FALSE,
215 NULL);
216
217 gsm_util_dialog_add_button (GTK_DIALOG (dialog),
218 _("_Cancel"), "process-stop",
219 GTK_RESPONSE_CANCEL);
220
221 if (dialog->name == NULL
222 && dialog->command == NULL
223 && dialog->comment == NULL) {
224 gtk_window_set_title (GTK_WINDOW (dialog), _("Add Startup Program"));
225 gsm_util_dialog_add_button (GTK_DIALOG (dialog),
226 _("_Add"), "list-add",
227 GTK_RESPONSE_OK);
228 } else {
229 gtk_window_set_title (GTK_WINDOW (dialog), _("Edit Startup Program"));
230 gsm_util_dialog_add_button (GTK_DIALOG (dialog),
231 _("_Save"), "document-save",
232 GTK_RESPONSE_OK);
233 }
234
235 dialog->name_entry = GTK_WIDGET (gtk_builder_get_object (xml, CAPPLET_NAME_ENTRY_WIDGET_NAME));
236 g_signal_connect (dialog->name_entry,
237 "activate",
238 G_CALLBACK (on_entry_activate),
239 dialog);
240 if (dialog->name != NULL) {
241 gtk_entry_set_text (GTK_ENTRY (dialog->name_entry), dialog->name);
242 }
243
244 dialog->browse_button = GTK_WIDGET (gtk_builder_get_object (xml, CAPPLET_BROWSE_WIDGET_NAME));
245 g_signal_connect (dialog->browse_button,
246 "clicked",
247 G_CALLBACK (on_browse_button_clicked),
248 dialog);
249
250 dialog->command_entry = GTK_WIDGET (gtk_builder_get_object (xml, CAPPLET_COMMAND_ENTRY_WIDGET_NAME));
251 g_signal_connect (dialog->command_entry,
252 "activate",
253 G_CALLBACK (on_entry_activate),
254 dialog);
255 if (dialog->command != NULL) {
256 gtk_entry_set_text (GTK_ENTRY (dialog->command_entry), dialog->command);
257 }
258
259 dialog->comment_entry = GTK_WIDGET (gtk_builder_get_object (xml, CAPPLET_COMMENT_ENTRY_WIDGET_NAME));
260 g_signal_connect (dialog->comment_entry,
261 "activate",
262 G_CALLBACK (on_entry_activate),
263 dialog);
264 if (dialog->comment != NULL) {
265 gtk_entry_set_text (GTK_ENTRY (dialog->comment_entry), dialog->comment);
266 }
267
268 dialog->delay_spin = GTK_WIDGET(gtk_builder_get_object (xml, CAPPLET_DELAY_SPIN_WIDGET_NAME));
269 g_signal_connect (dialog->delay_spin,
270 "output",
271 G_CALLBACK (on_spin_output),
272 dialog);
273 if (dialog->delay > 0) {
274 GtkAdjustment *adjustment;
275 adjustment = gtk_spin_button_get_adjustment (GTK_SPIN_BUTTON(dialog->delay_spin));
276 gtk_adjustment_set_value (adjustment, (gdouble) dialog->delay);
277 }
278
279 if (xml != NULL) {
280 g_object_unref (xml);
281 }
282}

514gboolean
515gsm_app_dialog_run (GsmAppDialog *dialog,
516 char **name_p,
517 char **command_p,
518 char **comment_p,
519 guint *delay_p)
520{
521 gboolean retval;
522
523 retval = FALSE;
524
525 while (gtk_dialog_run (GTK_DIALOG (dialog)) == GTK_RESPONSE_OK) {
526 const char *name;
527 const char *exec;
528 const char *comment;
529 guint delay;
530 const char *error_msg;
531 GError *error;
532 char **argv;
533 int argc;
534
535 name = gsm_app_dialog_get_name (GSM_APP_DIALOG (dialog));
536 exec = gsm_app_dialog_get_command (GSM_APP_DIALOG (dialog));
537 comment = gsm_app_dialog_get_comment (GSM_APP_DIALOG (dialog));
538 delay = gsm_app_dialog_get_delay (GSM_APP_DIALOG (dialog));
539
540 error = NULL;
541 error_msg = NULL;
542
543 if (gsm_util_text_is_blank (exec)) {
544 error_msg = _("The startup command cannot be empty");
545 } else {
546 if (!g_shell_parse_argv (exec, &argc, &argv, &error)) {
547 if (error != NULL) {
548 error_msg = error->message;
549 } else {
550 error_msg = _("The startup command is not valid");
551 }
552 }
553 }
554
555 if (error_msg != NULL) {
556 GtkWidget *msgbox;
557
558 msgbox = gtk_message_dialog_new (GTK_WINDOW (dialog),
559 GTK_DIALOG_MODAL,
560 GTK_MESSAGE_ERROR,
561 GTK_BUTTONS_CLOSE,
562 "%s", error_msg);
563
564 if (error != NULL) {
565 g_error_free (error);
566 }
567
568 gtk_dialog_run (GTK_DIALOG (msgbox));
569
570 gtk_widget_destroy (msgbox);
571
572 continue;
573 }
574
575 if (gsm_util_text_is_blank (name)) {
576 name = argv[0];
577 }
578
579 if (name_p) {
580 *name_p = g_strdup (name);
581 }
582
583 g_strfreev (argv);
584
585 if (command_p) {
586 *command_p = g_strdup (exec);
587 }
588
589 if (comment_p) {
590 *comment_p = g_strdup (comment);
591 }
592
593 if (delay_p) {
594 *delay_p = delay;
595 }
596
597 retval = TRUE;
598 break;
599 }
600
601 gtk_widget_destroy (GTK_WIDGET (dialog));
602
603 return retval;
