82static void
83update_history (CajaHistorySidebar *sidebar)
84{
85 GtkListStore *store;
86 GtkTreeSelection *selection;
87 GtkTreeIter iter;
88 GList *l, *history;
89 CajaBookmark *bookmark = NULL;
90 cairo_surface_t *surface = NULL;
91
92 store = GTK_LIST_STORE (gtk_tree_view_get_model (sidebar->tree_view));
93
94 gtk_list_store_clear (store);
95
96 history = caja_window_info_get_history (sidebar->window);
97 for (l = history; l != NULL; l = l->next)
98 {
99 char *name;
100
101 bookmark = caja_bookmark_copy (l->data);
102
103 surface = caja_bookmark_get_surface (bookmark, GTK_ICON_SIZE_MENU);
104 name = caja_bookmark_get_name (bookmark);
105 gtk_list_store_append (store, &iter);
106 gtk_list_store_set (store, &iter,
107 HISTORY_SIDEBAR_COLUMN_ICON, surface,
108 HISTORY_SIDEBAR_COLUMN_NAME, name,
109 HISTORY_SIDEBAR_COLUMN_BOOKMARK, bookmark,
110 -1);
111 g_object_unref (bookmark);
112
113 if (surface != NULL)
114 {
115 cairo_surface_destroy (surface);
116 }
117 g_free (name);
118 }
119 g_list_free_full (history, g_object_unref);
120
121 selection = GTK_TREE_SELECTION (gtk_tree_view_get_selection (sidebar->tree_view));
122
123 if (gtk_tree_model_get_iter_first (GTK_TREE_MODEL (store), &iter))
124 {
125 gtk_tree_selection_select_iter (selection, &iter);
126 }
127}

136static void
137open_selected_item (CajaHistorySidebar *sidebar,
138 GtkTreePath *path,
139 CajaWindowOpenFlags flags)
140{
141 CajaWindowSlotInfo *slot;
142 GtkTreeModel *model;
143 GtkTreeIter iter;
144 CajaBookmark *bookmark;
145 GFile *location;
146
147 model = gtk_tree_view_get_model (sidebar->tree_view);
148
149 if (!gtk_tree_model_get_iter (model, &iter, path))
150 {
151 return;
152 }
153
154 gtk_tree_model_get
155 (model, &iter, HISTORY_SIDEBAR_COLUMN_BOOKMARK, &bookmark, -1);
156
157 /* Navigate to the clicked location. */
158 location = caja_bookmark_get_location (CAJA_BOOKMARK (bookmark));
159 slot = caja_window_info_get_active_slot (sidebar->window);
160 caja_window_slot_info_open_location
161 (slot,
162 location, CAJA_WINDOW_OPEN_ACCORDING_TO_MODE,
163 flags, NULL);
164 g_object_unref (location);
165}

181static gboolean
182button_press_event_callback (GtkWidget *widget,
183 GdkEventButton *event,
184 gpointer user_data)
185{
186 if (event->button == 2 && event->type == GDK_BUTTON_PRESS)
187 {
188 /* Open new tab on middle click. */
189 CajaHistorySidebar *sidebar;
190 GtkTreePath *path;
191
192 sidebar = CAJA_HISTORY_SIDEBAR (user_data);
193 g_assert (sidebar->tree_view == GTK_TREE_VIEW (widget));
194
195 if (gtk_tree_view_get_path_at_pos (sidebar->tree_view,
196 event->x, event->y,
197 &path, NULL, NULL, NULL))
198 {
199 open_selected_item (sidebar,
200 path,
201 CAJA_WINDOW_OPEN_FLAG_NEW_TAB);
202 gtk_tree_path_free (path);
203 }
204 }
205
206 return FALSE;
207}

230static void
231caja_history_sidebar_init (CajaHistorySidebar *sidebar)
232{
233 GtkTreeView *tree_view;
234 GtkTreeViewColumn *col;
235 GtkCellRenderer *cell;
236 GtkListStore *store;
237 GtkTreeSelection *selection;
238
239 tree_view = GTK_TREE_VIEW (gtk_tree_view_new ());
240 gtk_tree_view_set_headers_visible (tree_view, FALSE);
241 gtk_widget_show (GTK_WIDGET (tree_view));
242
243 col = GTK_TREE_VIEW_COLUMN (gtk_tree_view_column_new ());
244
245 cell = gtk_cell_renderer_pixbuf_new ();
246 gtk_tree_view_column_pack_start (col, cell, FALSE);
247 gtk_tree_view_column_set_attributes (col, cell,
248 "surface", HISTORY_SIDEBAR_COLUMN_ICON,
249 NULL);
250
251 cell = gtk_cell_renderer_text_new ();
252 gtk_tree_view_column_pack_start (col, cell, TRUE);
253 gtk_tree_view_column_set_attributes (col, cell,
254 "text", HISTORY_SIDEBAR_COLUMN_NAME,
255 NULL);
256
257 gtk_tree_view_column_set_fixed_width (col, CAJA_ICON_SIZE_SMALLER);
258 gtk_tree_view_append_column (tree_view, col);
259
260 store = gtk_list_store_new (HISTORY_SIDEBAR_COLUMN_COUNT,
261 CAIRO_GOBJECT_TYPE_SURFACE,
262 G_TYPE_STRING,
263 CAJA_TYPE_BOOKMARK);
264
265 gtk_tree_view_set_model (tree_view, GTK_TREE_MODEL (store));
266 g_object_unref (store);
267
268 gtk_scrolled_window_set_policy (GTK_SCROLLED_WINDOW (sidebar),
269 GTK_POLICY_AUTOMATIC,
270 GTK_POLICY_AUTOMATIC);
271 gtk_scrolled_window_set_hadjustment (GTK_SCROLLED_WINDOW (sidebar), NULL);
272 gtk_scrolled_window_set_vadjustment (GTK_SCROLLED_WINDOW (sidebar), NULL);
273 gtk_scrolled_window_set_shadow_type (GTK_SCROLLED_WINDOW (sidebar), GTK_SHADOW_IN);
274 gtk_scrolled_window_set_overlay_scrolling (GTK_SCROLLED_WINDOW (sidebar), FALSE);
275
276 gtk_container_add (GTK_CONTAINER (sidebar), GTK_WIDGET (tree_view));
277 gtk_widget_show (GTK_WIDGET (sidebar));
278
279 sidebar->tree_view = tree_view;
280
281 selection = gtk_tree_view_get_selection (tree_view);
282 gtk_tree_selection_set_mode (selection, GTK_SELECTION_SINGLE);
283
284 g_signal_connect_object
285 (tree_view, "row_activated",
286 G_CALLBACK (row_activated_callback), sidebar, 0);
287
288 g_signal_connect_object (caja_signaller_get_current (),
289 "history_list_changed",
290 G_CALLBACK (history_changed_callback), sidebar, 0);
291
292 g_signal_connect (tree_view, "button-press-event",
293 G_CALLBACK (button_press_event_callback), sidebar);
294
295 g_signal_connect_swapped (caja_preferences,
296 "changed::" CAJA_PREFERENCES_CLICK_POLICY,
297 G_CALLBACK(click_policy_changed_callback),
298 sidebar);
299 update_click_policy (sidebar);
300}
