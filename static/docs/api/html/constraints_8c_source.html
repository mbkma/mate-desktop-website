246static gboolean
247do_all_constraints (MetaWindow *window,
248 ConstraintInfo *info,
249 ConstraintPriority priority,
250 gboolean check_only)
251{
252 const Constraint *constraint;
253 gboolean satisfied;
254
255 constraint = &all_constraints[0];
256 satisfied = TRUE;
257 while (constraint->func != NULL)
258 {
259 satisfied = satisfied &&
260 (*constraint->func) (window, info, priority, check_only);
261
262 if (!check_only)
263 {
264 /* Log how the constraint modified the position */
265 meta_topic (META_DEBUG_GEOMETRY,
266 "info->current is %d,%d +%d,%d after %s\n",
267 info->current.x, info->current.y,
268 info->current.width, info->current.height,
269 constraint->name);
270 }
271 else if (!satisfied)
272 {
273 /* Log which constraint was not satisfied */
274 meta_topic (META_DEBUG_GEOMETRY,
275 "constraint %s not satisfied.\n",
276 constraint->name);
277 return FALSE;
278 }
279 ++constraint;
280 }
281
282 return TRUE;
283}

285void
286meta_window_constrain (MetaWindow *window,
287 MetaFrameBorders *orig_borders,
288 MetaMoveResizeFlags flags,
289 int resize_gravity,
290 const MetaRectangle *orig,
291 MetaRectangle *new)
292{
293 ConstraintInfo info;
294 ConstraintPriority priority = PRIORITY_MINIMUM;
295 gboolean satisfied = FALSE;
296
297 /* WARNING: orig and new specify positions and sizes of the inner window,
298 * not the outer. This is a common gotcha since half the constraints
299 * deal with inner window position/size and half deal with outer. See
300 * doc/how-constraints-works.txt for more information.
301 */
302 meta_topic (META_DEBUG_GEOMETRY,
303 "Constraining %s in move from %d,%d %dx%d to %d,%d %dx%d\n",
304 window->desc,
305 orig->x, orig->y, orig->width, orig->height,
306 new->x, new->y, new->width, new->height);
307
308 setup_constraint_info (&info,
309 window,
310 orig_borders,
311 flags,
312 resize_gravity,
313 orig,
314 new);
315 place_window_if_needed (window, &info);
316
317 while (!satisfied && priority <= PRIORITY_MAXIMUM) {
318 gboolean check_only = TRUE;
319
320 /* Individually enforce all the high-enough priority constraints */
321 do_all_constraints (window, &info, priority, !check_only);
322
323 /* Check if all high-enough priority constraints are simultaneously
324 * satisfied
325 */
326 satisfied = do_all_constraints (window, &info, priority, check_only);
327
328 /* Drop the least important constraints if we can't satisfy them all */
329 priority++;
330 }
331
332 /* Make sure we use the constrained position */
333 *new = info.current;
334
335 /* We may need to update window->require_fully_onscreen,
336 * window->require_on_single_xinerama, and perhaps other quantities
337 * if this was a user move or user move-and-resize operation.
338 */
339 update_onscreen_requirements (window, &info);
340
341 /* Ew, what an ugly way to do things. Destructors (in a real OOP language,
342 * not gobject-style--gobject would be more pain than it's worth) or
343 * smart pointers would be so much nicer here. *shrug*
344 */
345 if (info.must_free_borders)
346 g_free (info.borders);
347}

349static void
350setup_constraint_info (ConstraintInfo *info,
351 MetaWindow *window,
352 MetaFrameBorders *orig_borders,
353 MetaMoveResizeFlags flags,
354 int resize_gravity,
355 const MetaRectangle *orig,
356 MetaRectangle *new)
357{
358 const MetaXineramaScreenInfo *xinerama_info;
359 MetaWorkspace *cur_workspace;
360
361 info->orig = *orig;
362 info->current = *new;
363
364 /* Create a fake frame geometry if none really exists */
365 if (orig_borders && !window->fullscreen)
366 {
367 info->borders = orig_borders;
368 info->must_free_borders = FALSE;
369 }
370 else
371 {
372 info->borders = g_new0 (MetaFrameBorders, 1);
373 info->must_free_borders = TRUE;
374 }
375
376 if (flags & META_IS_MOVE_ACTION && flags & META_IS_RESIZE_ACTION)
377 info->action_type = ACTION_MOVE_AND_RESIZE;
378 else if (flags & META_IS_RESIZE_ACTION)
379 info->action_type = ACTION_RESIZE;
380 else if (flags & META_IS_MOVE_ACTION)
381 info->action_type = ACTION_MOVE;
382 else
383 g_error ("BAD, BAD developer! No treat for you! (Fix your calls to "
384 "meta_window_move_resize_internal()).\n");
385
386 info->is_user_action = (flags & META_IS_USER_ACTION);
387
388 info->resize_gravity = resize_gravity;
389
390 /* FIXME: fixed_directions might be more sane if we (a) made it
391 * depend on the grab_op type instead of current amount of movement
392 * (thus implying that it only has effect when user_action is true,
393 * and (b) ignored it for aspect ratio windows -- at least in those
394 * cases where both directions do actually change size.
395 */
396 info->fixed_directions = FIXED_DIRECTION_NONE;
397 /* If x directions don't change but either y direction does */
398 if ( orig->x == new->x && orig->x + orig->width == new->x + new->width &&
399 (orig->y != new->y || orig->y + orig->height != new->y + new->height))
400 {
401 info->fixed_directions = FIXED_DIRECTION_X;
402 }
403 /* If y directions don't change but either x direction does */
404 if ( orig->y == new->y && orig->y + orig->height == new->y + new->height &&
405 (orig->x != new->x || orig->x + orig->width != new->x + new->width ))
406 {
407 info->fixed_directions = FIXED_DIRECTION_Y;
408 }
409 /* The point of fixed directions is just that "move to nearest valid
410 * position" is sometimes a poorer choice than "move to nearest
411 * valid position but only change this coordinate" for windows the
412 * user is explicitly moving. This isn't ever true for things that
413 * aren't explicit user interaction, though, so just clear it out.
414 */
415 if (!info->is_user_action)
416 info->fixed_directions = FIXED_DIRECTION_NONE;
417
418 xinerama_info =
419 meta_screen_get_xinerama_for_rect (window->screen, &info->current);
420 meta_window_get_work_area_for_xinerama (window,
421 xinerama_info->number,
422 &info->work_area_xinerama);
423
424 if (!window->fullscreen || window->fullscreen_monitors[0] == -1)
425 {
426 info->entire_xinerama = xinerama_info->rect;
427 }
428 else
429 {
430 int i = 0;
431 long monitor;
432
433 monitor = window->fullscreen_monitors[i];
434 info->entire_xinerama =
435 window->screen->xinerama_infos[monitor].rect;
436 for (i = 1; i <= 3; i++)
437 {
438 monitor = window->fullscreen_monitors[i];
439 meta_rectangle_union (&info->entire_xinerama,
440 &window->screen->xinerama_infos[monitor].rect,
441 &info->entire_xinerama);
442 }
443 }
444
445 cur_workspace = window->screen->active_workspace;
446 info->usable_screen_region =
447 meta_workspace_get_onscreen_region (cur_workspace);
448 info->usable_xinerama_region =
449 meta_workspace_get_onxinerama_region (cur_workspace,
450 xinerama_info->number);
451
452 /* Log all this information for debugging */
453 meta_topic (META_DEBUG_GEOMETRY,
454 "Setting up constraint info:\n"
455 " orig: %d,%d +%d,%d\n"
456 " new : %d,%d +%d,%d\n"
457 " action_type : %s\n"
458 " is_user_action : %s\n"
459 " resize_gravity : %s\n"
460 " fixed_directions: %s\n"
461 " work_area_xinerama: %d,%d +%d,%d\n"
462 " entire_xinerama : %d,%d +%d,%d\n",
463 info->orig.x, info->orig.y, info->orig.width, info->orig.height,
464 info->current.x, info->current.y,
465 info->current.width, info->current.height,
466 (info->action_type == ACTION_MOVE) ? "Move" :
467 (info->action_type == ACTION_RESIZE) ? "Resize" :
468 (info->action_type == ACTION_MOVE_AND_RESIZE) ? "Move&Resize" :
469 "Freakin' Invalid Stupid",
470 (info->is_user_action) ? "true" : "false",
471 meta_gravity_to_string (info->resize_gravity),
472 (info->fixed_directions == FIXED_DIRECTION_NONE) ? "None" :
473 (info->fixed_directions == FIXED_DIRECTION_X) ? "X fixed" :
474 (info->fixed_directions == FIXED_DIRECTION_Y) ? "Y fixed" :
475 "Freakin' Invalid Stupid",
476 info->work_area_xinerama.x, info->work_area_xinerama.y,
477 info->work_area_xinerama.width,
478 info->work_area_xinerama.height,
479 info->entire_xinerama.x, info->entire_xinerama.y,
480 info->entire_xinerama.width, info->entire_xinerama.height);
481}

483static void
484place_window_if_needed(MetaWindow *window,
485 ConstraintInfo *info)
486{
487 gboolean did_placement;
488
489 /* Do placement if any, so we go ahead and apply position
490 * constraints in a move-only context. Don't place
491 * maximized/minimized/fullscreen windows until they are
492 * unmaximized, unminimized and unfullscreened.
493 */
494 did_placement = FALSE;
495 if (!window->placed &&
496 window->calc_placement &&
497 !(window->maximized_horizontally ||
498 window->maximized_vertically) &&
499 !window->minimized &&
500 !window->fullscreen)
501 {
502 MetaRectangle placed_rect = info->orig;
503 MetaWorkspace *cur_workspace;
504 const MetaXineramaScreenInfo *xinerama_info;
505
506 meta_window_place (window, info->borders, info->orig.x, info->orig.y,
507 &placed_rect.x, &placed_rect.y);
508 did_placement = TRUE;
509
510 /* placing the window may have changed the xinerama. Find the
511 * new xinerama and update the ConstraintInfo
512 */
513 xinerama_info =
514 meta_screen_get_xinerama_for_rect (window->screen, &placed_rect);
515 info->entire_xinerama = xinerama_info->rect;
516 meta_window_get_work_area_for_xinerama (window,
517 xinerama_info->number,
518 &info->work_area_xinerama);
519 cur_workspace = window->screen->active_workspace;
520 info->usable_xinerama_region =
521 meta_workspace_get_onxinerama_region (cur_workspace,
522 xinerama_info->number);
523
524 info->current.x = placed_rect.x;
525 info->current.y = placed_rect.y;
526
527 /* Since we just barely placed the window, there's no reason to
528 * consider any of the directions fixed.
529 */
530 info->fixed_directions = FIXED_DIRECTION_NONE;
531 }
532
533 if (window->placed || did_placement)
534 {
535 if (window->maximize_horizontally_after_placement ||
536 window->maximize_vertically_after_placement ||
537 window->fullscreen_after_placement)
538 {
539 /* define a sane saved_rect so that the user can unmaximize or
540 * make unfullscreen to something reasonable.
541 */
542 if (info->current.width >= info->work_area_xinerama.width)
543 {
544 info->current.width = (int) (.75 * (double) info->work_area_xinerama.width);
545 info->current.x = info->work_area_xinerama.x +
546 (int) (.125 * (double) info->work_area_xinerama.width);
547 }
548 if (info->current.height >= info->work_area_xinerama.height)
549 {
550 info->current.height = (int) (.75 * (double) info->work_area_xinerama.height);
551 info->current.y = info->work_area_xinerama.y +
552 (int) (.083 * (double) info->work_area_xinerama.height);
553 }
554
555 if (window->maximize_horizontally_after_placement ||
556 window->maximize_vertically_after_placement)
557 meta_window_maximize_internal (window,
558 (window->maximize_horizontally_after_placement ?
559 META_MAXIMIZE_HORIZONTAL : 0 ) |
560 (window->maximize_vertically_after_placement ?
561 META_MAXIMIZE_VERTICAL : 0), &info->current);
562
563 /* maximization may have changed frame geometry */
564 if (window->frame && !window->fullscreen)
565 meta_frame_calc_borders (window->frame, info->borders);
566
567 if (window->fullscreen_after_placement)
568 {
569 window->saved_rect = info->current;
570 window->fullscreen = TRUE;
571 window->fullscreen_after_placement = FALSE;
572 }
573
574 window->maximize_horizontally_after_placement = FALSE;
575 window->maximize_vertically_after_placement = FALSE;
576 }
577 window->user_rect = info->current;
578 if (window->minimize_after_placement)
579 meta_window_minimize (window);
580 }
581
582 window->minimize_after_placement = FALSE;
583}

585static void
586update_onscreen_requirements (MetaWindow *window,
587 ConstraintInfo *info)
588{
589 gboolean old;
590
591 /* We only apply the various onscreen requirements to normal windows */
592 if (window->type == META_WINDOW_DESKTOP ||
593 window->type == META_WINDOW_DOCK)
594 return;
595
596 /* We don't want to update the requirements for fullscreen windows;
597 * fullscreen windows are specially handled anyway, and it updating
598 * the requirements when windows enter fullscreen mode mess up the
599 * handling of the window when it leaves that mode (especially when
600 * the application sends a bunch of configurerequest events). See
601 * #353699.
602 */
603 if (window->fullscreen)
604 return;
605
606 /* USABILITY NOTE: Naturally, I only want the require_fully_onscreen,
607 * require_on_single_xinerama, and require_titlebar_visible flags to
608 * *become false* due to user interactions (which is allowed since
609 * certain constraints are ignored for user interactions regardless of
610 * the setting of these flags). However, whether to make these flags
611 * *become true* due to just an application interaction is a little
612 * trickier. It's possible that users may find not doing that strange
613 * since two application interactions that resize in opposite ways don't
614 * necessarily end up cancelling--but it may also be strange for the user
615 * to have an application resize the window so that it's onscreen, the
616 * user forgets about it, and then later the app is able to resize itself
617 * off the screen. Anyway, for now, I think the latter is the more
618 * problematic case but this may need to be revisited.
619 */
620
621 /* The require onscreen/on-single-xinerama and titlebar_visible
622 * stuff is relative to the outer window, not the inner
623 */
624 extend_by_frame (window, &info->current, info->borders);
625
626 /* Update whether we want future constraint runs to require the
627 * window to be on fully onscreen.
628 */
629 old = window->require_fully_onscreen;
630 window->require_fully_onscreen =
631 (meta_rectangle_contained_in_region (info->usable_screen_region,
632 &info->current) != FALSE);
633 if (old ^ window->require_fully_onscreen)
634 meta_topic (META_DEBUG_GEOMETRY,
635 "require_fully_onscreen for %s toggled to %s\n",
636 window->desc,
637 window->require_fully_onscreen ? "TRUE" : "FALSE");
638
639 /* Update whether we want future constraint runs to require the
640 * window to be on a single xinerama.
641 */
642 old = window->require_on_single_xinerama;
643 window->require_on_single_xinerama =
644 (meta_rectangle_contained_in_region (info->usable_xinerama_region,
645 &info->current) != FALSE);
646 if (old ^ window->require_on_single_xinerama)
647 meta_topic (META_DEBUG_GEOMETRY,
648 "require_on_single_xinerama for %s toggled to %s\n",
649 window->desc,
650 window->require_on_single_xinerama ? "TRUE" : "FALSE");
651
652 /* Update whether we want future constraint runs to require the
653 * titlebar to be visible.
654 */
655 if (window->frame && window->decorated)
656 {
657 MetaRectangle titlebar_rect;
658
659 titlebar_rect = info->current;
660 titlebar_rect.height = info->borders->visible.top;
661 old = window->require_titlebar_visible;
662 window->require_titlebar_visible =
663 (meta_rectangle_overlaps_with_region (info->usable_screen_region,
664 &titlebar_rect) != FALSE);
665 if (old ^ window->require_titlebar_visible)
666 meta_topic (META_DEBUG_GEOMETRY,
667 "require_titlebar_visible for %s toggled to %s\n",
668 window->desc,
669 window->require_titlebar_visible ? "TRUE" : "FALSE");
670 }
671
672 /* Don't forget to restore the position of the window */

718static inline void
719get_size_limits (const MetaWindow *window,
720 const MetaFrameBorders *borders,
721 gboolean include_frame,
722 MetaRectangle *min_size,
723 MetaRectangle *max_size)
724{
725 /* We pack the results into MetaRectangle structs just for convienience; we
726 * don't actually use the position of those rects.
727 */
728 min_size->width = window->size_hints.min_width;
729 min_size->height = window->size_hints.min_height;
730 max_size->width = window->size_hints.max_width;
731 max_size->height = window->size_hints.max_height;
732
733 if (include_frame)
734 {
735 int fw;
736 int fh;
737
738 if (window->frame)
739 {
740 fw = borders->visible.left + borders->visible.right;
741 fh = borders->visible.top + borders->visible.bottom;
742
743 min_size->width += fw;
744 min_size->height += fh;
745
746 /* Do check to avoid overflow (e.g. max_size->width & max_size->height
747 * may be set to G_MAXINT by meta_set_normal_hints()).
748 */
749 if (max_size->width < (G_MAXINT - fw))
750 max_size->width += fw;
751 else
752 max_size->width = G_MAXINT;
753
754 if (max_size->height < (G_MAXINT - fh))
755 max_size->height += fh;
756 else
757 max_size->height = G_MAXINT;
758 }
759 else
760 {
761 fw = window->custom_frame_extents.left + window->custom_frame_extents.right;
762 fh = window->custom_frame_extents.top + window->custom_frame_extents.bottom;
763
764 min_size->width -= fw;
765 min_size->height -= fh;
766
767 max_size->width -= fw;
768 max_size->height -= fh;
769 }
770 }

815static gboolean
816constrain_maximization (MetaWindow *window,
817 ConstraintInfo *info,
818 ConstraintPriority priority,
819 gboolean check_only)
820{
821 MetaRectangle target_size;
822 MetaRectangle min_size, max_size;
823 gboolean hminbad, vminbad;
824 gboolean horiz_equal, vert_equal;
825 gboolean constraint_already_satisfied;
826
827 if (priority > PRIORITY_MAXIMIZATION)
828 return TRUE;
829
830 /* Determine whether constraint applies; exit if it doesn't */
831 if ((!window->maximized_horizontally && !window->maximized_vertically) ||
832 META_WINDOW_TILED (window))
833 return TRUE;
834
835 /* Calculate target_size = maximized size of (window + frame) */
836 if (META_WINDOW_MAXIMIZED (window))
837 {
838 target_size = info->work_area_xinerama;
839 }
840 else
841 {
842 /* Amount of maximization possible in a single direction depends
843 * on which struts could occlude the window given its current
844 * position. For example, a vertical partial strut on the right
845 * is only relevant for a horizontally maximized window when the
846 * window is at a vertical position where it could be occluded
847 * by that partial strut.
848 */
849 MetaDirection direction;
850 GSList *active_workspace_struts;
851
852 if (window->maximized_horizontally)
853 direction = META_DIRECTION_HORIZONTAL;
854 else
855 direction = META_DIRECTION_VERTICAL;
856 active_workspace_struts = window->screen->active_workspace->all_struts;
857
858 target_size = info->current;
859 extend_by_frame (window, &target_size, info->borders);
860 meta_rectangle_expand_to_avoiding_struts (&target_size,
861 &info->entire_xinerama,
862 direction,
863 active_workspace_struts);
864 }
865 /* Now make target_size = maximized size of client window */
866 unextend_by_frame (window, &target_size, info->borders);
867
868 /* Check min size constraints; max size constraints are ignored for maximized
869 * windows, as per bug 327543.
870 */
871 get_size_limits (window, info->borders, FALSE, &min_size, &max_size);
872 hminbad = target_size.width < min_size.width && window->maximized_horizontally;
873 vminbad = target_size.height < min_size.height && window->maximized_vertically;
874 if (hminbad || vminbad)
875 return TRUE;
876
877 /* Determine whether constraint is already satisfied; exit if it is */
878 horiz_equal = target_size.x == info->current.x &&
879 target_size.width == info->current.width;
880 vert_equal = target_size.y == info->current.y &&
881 target_size.height == info->current.height;
882 constraint_already_satisfied =
883 (horiz_equal || !window->maximized_horizontally) &&
884 (vert_equal || !window->maximized_vertically);
885 if (check_only || constraint_already_satisfied)
886 return constraint_already_satisfied;
887
888 /*** Enforce constraint ***/
889 if (window->maximized_horizontally)
890 {
891 info->current.x = target_size.x;
892 info->current.width = target_size.width;
893 }
894 if (window->maximized_vertically)
895 {
896 info->current.y = target_size.y;
897 info->current.height = target_size.height;
898 }
899 return TRUE;
900}

902static gboolean
903constrain_tiling (MetaWindow *window,
904 ConstraintInfo *info,
905 ConstraintPriority priority,
906 gboolean check_only)
907{
908 MetaRectangle target_size;
909 MetaRectangle min_size, max_size;
910 gboolean hminbad, vminbad;
911 gboolean horiz_equal, vert_equal;
912 gboolean constraint_already_satisfied;
913 gboolean allow_resize_horizontally = FALSE;
914 gboolean allow_resize_vertically = FALSE;
915
916 if (priority > PRIORITY_TILING)
917 return TRUE;
918
919 /* Determine whether constraint applies; exit if it doesn't */
920 if (!META_WINDOW_TILED(window))
921 return TRUE;
922
923 /* Calculate target_size - as the tile previews need this as well, we
924 * use an external function for the actual calculation
925 */
926 meta_window_get_current_tile_area (window, &target_size);
927 unextend_by_frame (window, &target_size, info->borders);
928
929 /* Check min size constraints; max size constraints are ignored as for
930 * maximized windows.
931 */
932 get_size_limits (window, info->borders, FALSE, &min_size, &max_size);
933 hminbad = target_size.width < min_size.width;
934 vminbad = target_size.height < min_size.height;
935 if (hminbad || vminbad)
936 return TRUE;
937
938 /* Determine whether constraint is already satisfied; exit if it is */
939 horiz_equal = target_size.x == info->current.x &&
940 target_size.width == info->current.width;
941 vert_equal = target_size.y == info->current.y &&
942 target_size.height == info->current.height;
943 constraint_already_satisfied = horiz_equal && vert_equal;
944 if (check_only || constraint_already_satisfied)
945 return constraint_already_satisfied;
946
947 /* Allow the user to resize horizontally when tiled */
948 if (info->is_user_action)
949 {
950 can_resize_tiled(window->tile_mode,
951 info->resize_gravity,
952 &allow_resize_horizontally,
953 &allow_resize_vertically);
954
955 window->tile_resized = TRUE;
956 }
957
958 if (window->tile_resized)
959 {
960 /* Maintain current tile size for user-resized windows */
961 target_size.y = info->orig.y;
962 target_size.height = info->orig.height;
963 target_size.x = info->orig.x;
964 target_size.width = info->orig.width;
965 }
966
967 /*** Enforce constraint ***/
968 if (!allow_resize_vertically)
969 {
970 info->current.y = target_size.y;
971 info->current.height = target_size.height;
972 }
973
974 if(!allow_resize_horizontally)
975 {
976 info->current.x = target_size.x;
977 info->current.width = target_size.width;
978 }
979
980 return TRUE;
981}

1097static gboolean
1098constrain_size_increments (MetaWindow *window,
1099 ConstraintInfo *info,
1100 ConstraintPriority priority,
1101 gboolean check_only)
1102{
1103 int bh, hi, bw, wi, extra_height, extra_width;
1104 int new_width, new_height;
1105 gboolean constraint_already_satisfied;
1106 MetaRectangle *start_rect;
1107
1108 if (priority > PRIORITY_SIZE_HINTS_INCREMENTS)
1109 return TRUE;
1110
1111 /* Determine whether constraint applies; exit if it doesn't */
1112 if (META_WINDOW_MAXIMIZED (window) || window->fullscreen ||
1113 META_WINDOW_TILED (window) || info->action_type == ACTION_MOVE)
1114 return TRUE;
1115
1116 /* Determine whether constraint is already satisfied; exit if it is */
1117 bh = window->size_hints.base_height;
1118 hi = window->size_hints.height_inc;
1119 bw = window->size_hints.base_width;
1120 wi = window->size_hints.width_inc;
1121 extra_height = (info->current.height - bh) % hi;
1122 extra_width = (info->current.width - bw) % wi;
1123 /* ignore size increments for maximized windows */
1124 if (window->maximized_horizontally)
1125 extra_width *= 0;
1126 if (window->maximized_vertically)
1127 extra_height *= 0;
1128 /* constraint is satisfied iff there is no extra height or width */
1129 constraint_already_satisfied =
1130 (extra_height == 0 && extra_width == 0);
1131
1132 if (check_only || constraint_already_satisfied)
1133 return constraint_already_satisfied;
1134
1135 /*** Enforce constraint ***/
1136 new_width = info->current.width - extra_width;
1137 new_height = info->current.height - extra_height;
1138
1139 /* Adjusting down instead of up (as done in the above two lines) may
1140 * violate minimum size constraints; fix the adjustment if this
1141 * happens.
1142 */
1143 if (new_width < window->size_hints.min_width)
1144 new_width += ((window->size_hints.min_width - new_width)/wi + 1)*wi;
1145 if (new_height < window->size_hints.min_height)
1146 new_height += ((window->size_hints.min_height - new_height)/hi + 1)*hi;
1147
1148 /* Figure out what original rect to pass to meta_rectangle_resize_with_gravity
1149 * See bug 448183
1150 */
1151 if (info->action_type == ACTION_MOVE_AND_RESIZE)
1152 start_rect = &info->current;
1153 else
1154 start_rect = &info->orig;
1155
1156 /* Resize to the new size */
1157 meta_rectangle_resize_with_gravity (start_rect,
1158 &info->current,
1159 info->resize_gravity,
1160 new_width,
1161 new_height);
1162 return TRUE;
1163}

1220static gboolean
1221constrain_aspect_ratio (MetaWindow *window,
1222 ConstraintInfo *info,
1223 ConstraintPriority priority,
1224 gboolean check_only)
1225{
1226 double minr, maxr;
1227 gboolean constraints_are_inconsistent, constraint_already_satisfied;
1228 int fudge;
1229 double new_width, new_height;
1230 double best_width, best_height;
1231 double alt_width, alt_height;
1232 MetaRectangle *start_rect;
1233
1234 if (priority > PRIORITY_ASPECT_RATIO)
1235 return TRUE;
1236
1237 /* Determine whether constraint applies; exit if it doesn't. */
1238 minr = window->size_hints.min_aspect.x /
1239 (double)window->size_hints.min_aspect.y;
1240 maxr = window->size_hints.max_aspect.x /
1241 (double)window->size_hints.max_aspect.y;
1242 constraints_are_inconsistent = minr > maxr;
1243 if (constraints_are_inconsistent ||
1244 META_WINDOW_MAXIMIZED (window) || window->fullscreen ||
1245 META_WINDOW_SIDE_TILED (window) || info->action_type == ACTION_MOVE)
1246 return TRUE;
1247
1248 /* Determine whether constraint is already satisfied; exit if it is. We
1249 * need the following to hold:
1250 *
1251 * width
1252 * minr <= ------ <= maxr
1253 * height
1254 *
1255 * But we need to allow for some slight fudging since width and height
1256 * are integers instead of floating point numbers (this is particularly
1257 * important when minr == maxr), so we allow width and height to be off
1258 * a little bit from strictly satisfying these equations. For just one
1259 * sided resizing, we have to make the fudge factor a little bigger
1260 * because of how meta_rectangle_resize_with_gravity treats those as
1261 * being a resize increment (FIXME: I should handle real resize
1262 * increments better here...)
1263 */
1264 switch (info->resize_gravity)
1265 {
1266 case WestGravity:
1267 case NorthGravity:
1268 case SouthGravity:
1269 case EastGravity:
1270 fudge = 2;
1271 break;
1272
1273 case NorthWestGravity:
1274 case SouthWestGravity:
1275 case CenterGravity:
1276 case NorthEastGravity:
1277 case SouthEastGravity:
1278 case StaticGravity:
1279 default:
1280 fudge = 1;
1281 break;
1282 }
1283 constraint_already_satisfied =
1284 info->current.width - (info->current.height * minr ) > -minr*fudge &&
1285 info->current.width - (info->current.height * maxr ) < maxr*fudge;
1286 if (check_only || constraint_already_satisfied)
1287 return constraint_already_satisfied;
1288
1289 /*** Enforce constraint ***/
1290 new_width = (double) info->current.width;
1291 new_height = (double) info->current.height;
1292
1293 switch (info->resize_gravity)
1294 {
1295 case WestGravity:
1296 case EastGravity:
1297 new_height = CLAMP (new_height, new_width / maxr, new_width / minr);
1298 break;
1299
1300 case NorthGravity:
1301 case SouthGravity:
1302 new_width = CLAMP (new_width, new_height * minr, new_height * maxr);
1303 break;
1304
1305 case NorthWestGravity:
1306 case SouthWestGravity:
1307 case CenterGravity:
1308 case NorthEastGravity:
1309 case SouthEastGravity:
1310 case StaticGravity:
1311 default:
1312 /* Find what width would correspond to new_height, and what height would
1313 * correspond to new_width */
1314 alt_width = CLAMP (new_width, new_height * minr, new_height * maxr);
1315 alt_height = CLAMP (new_height, new_width / maxr, new_width / minr);
1316
1317 /* The line connecting the points (alt_width, new_height) and
1318 * (new_width, alt_height) provide a range of
1319 * valid-for-the-aspect-ratio-constraint sizes. We want the
1320 * size in that range closest to the value requested, i.e. the
1321 * point on the line which is closest to the point (new_width,
1322 * new_height)
1323 */
1324 meta_rectangle_find_linepoint_closest_to_point (alt_width, new_height,
1325 new_width, alt_height,
1326 new_width, new_height,
1327 &best_width, &best_height);
1328
1329 new_width = round (best_width);
1330 new_height = round (best_height);
1331
1332 break;
1333 }
1334
1335 /* Figure out what original rect to pass to meta_rectangle_resize_with_gravity
1336 * See bug 448183
1337 */
1338 if (info->action_type == ACTION_MOVE_AND_RESIZE)
1339 start_rect = &info->current;
1340 else
1341 start_rect = &info->orig;
1342
1343 meta_rectangle_resize_with_gravity (start_rect,
1344 &info->current,
1345 info->resize_gravity,
1346 (int) new_width,
1347 (int) new_height);
1348
1349 return TRUE;
1350}
