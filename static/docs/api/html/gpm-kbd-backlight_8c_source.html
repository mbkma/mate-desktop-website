105static gboolean
106gpm_kbd_backlight_set (GpmKbdBacklight *backlight,
107 guint percentage,
108 gboolean save)
109{
110 gint scale;
111 guint goal;
112
113 g_return_val_if_fail (GPM_IS_KBD_BACKLIGHT (backlight), FALSE);
114 /* avoid warnings if no keyboard brightness is available */
115 if (backlight->priv->max_brightness < 1)
116 return FALSE;
117 /* if we're setting the same we are, don't bother */
118 //g_return_val_if_fail (backlight->priv->brightness_percent != percentage, FALSE);
119
120 goal = gpm_discrete_from_percent (percentage, backlight->priv->max_brightness);
121 scale = percentage > backlight->priv->brightness_percent ? 1 : -1;
122
123 /* if percentage change too small force next value */
124 if (goal == backlight->priv->brightness) {
125 goal += percentage == backlight->priv->brightness_percent ? 0 : scale;
126 }
127
128 /* step loop down by 1 for a dimming effect */
129 while (backlight->priv->brightness != goal) {
130 backlight->priv->brightness += scale;
131 backlight->priv->brightness_percent = gpm_discrete_to_percent (backlight->priv->brightness, backlight->priv->max_brightness);
132
133 g_dbus_proxy_call (backlight->priv->upower_proxy,
134 "SetBrightness",
135 g_variant_new ("(i)", (gint) backlight->priv->brightness),
136 G_DBUS_CALL_FLAGS_NONE,
137 -1,
138 NULL,
139 NULL,
140 NULL);
141 }
142
143 /* On user interaction, save the target brightness in the only setting we
144 * have, to be able to restore it next time. */
145 if (save) {
146 gint ac_value = backlight->priv->brightness_percent;
147
148 /* If on battery, try and scale the saved value up so applying it with
149 * the scaling would yield the current value. It's however not possible
150 * to make it accurate as any value that is higher than the dim percentage
151 * cannot be saved (as the saved value is 0-100). */
152 if (g_settings_get_boolean (backlight->priv->settings, GPM_SETTINGS_KBD_BACKLIGHT_BATT_REDUCE) &&
153 up_client_get_on_battery (backlight->priv->client)) {
154 gint dim_pct;
155
156 dim_pct = g_settings_get_int (backlight->priv->settings, GPM_SETTINGS_KBD_BRIGHTNESS_DIM_BY_ON_BATT);
157
158 if (dim_pct > 100) {
159 g_warning ("Cannot scale brightness down by more than 100%%. Scaling by 50%%");
160 dim_pct = 50;
161 }
162
163 ac_value = (gint) (((gdouble) ac_value) / ((100 - dim_pct) / 100.0));
164 ac_value = CLAMP (ac_value, 0, 100);
165 }
166
167 g_settings_set_int (backlight->priv->settings, GPM_SETTINGS_KBD_BRIGHTNESS_ON_AC, ac_value);
168 }
169
170 g_debug("Set brightness to %u", backlight->priv->brightness);
171 return TRUE;
172}

200static void
201gpm_kbd_backlight_dialog_show (GpmKbdBacklight *backlight)
202{
203 int orig_w;
204 int orig_h;
205 int screen_w;
206 int screen_h;
207 int x;
208 int y;
209 int pointer_x;
210 int pointer_y;
211 GtkRequisition win_req;
212 GdkScreen *pointer_screen;
213 GdkRectangle geometry;
214 GdkMonitor *monitor;
215 GdkDisplay *display;
216 GdkSeat *seat;
217 GdkDevice *device;
218
219 /*
220 * get the window size
221 * if the window hasn't been mapped, it doesn't necessarily
222 * know its true size, yet, so we need to jump through hoops
223 */
224 gtk_window_get_default_size (GTK_WINDOW (backlight->priv->popup), &orig_w, &orig_h);
225 gtk_widget_get_preferred_size (backlight->priv->popup, NULL, &win_req);
226
227 if (win_req.width > orig_w) {
228 orig_w = win_req.width;
229 }
230 if (win_req.height > orig_h) {
231 orig_h = win_req.height;
232 }
233
234 pointer_screen = NULL;
235 display = gtk_widget_get_display (backlight->priv->popup);
236 seat = gdk_display_get_default_seat (display);
237 device = gdk_seat_get_pointer (seat);
238 gdk_device_get_position (device,
239 &pointer_screen,
240 &pointer_x,
241 &pointer_y);
242
243 monitor = gdk_display_get_monitor_at_point (gdk_screen_get_display (pointer_screen),
244 pointer_x,
245 pointer_y);
246
247 gdk_monitor_get_geometry (monitor, &geometry);
248
249 screen_w = geometry.width;
250 screen_h = geometry.height;
251
252 x = ((screen_w - orig_w) / 2) + geometry.x;
253 y = geometry.y + (screen_h / 2) + (screen_h / 2 - orig_h) / 2;
254
255 gtk_window_move (GTK_WINDOW (backlight->priv->popup), x, y);
256
257 gtk_widget_show (backlight->priv->popup);
258
259 gdk_display_sync (gtk_widget_get_display (backlight->priv->popup));
260}

265static gboolean
266gpm_kbd_backlight_brightness_up (GpmKbdBacklight *backlight)
267{
268 guint new;
269
270 new = MIN (backlight->priv->brightness_percent + GPM_KBD_BACKLIGHT_STEP, 100u);
271 return gpm_kbd_backlight_set (backlight, new, TRUE);
272}

277static gboolean
278gpm_kbd_backlight_brightness_down (GpmKbdBacklight *backlight)
279{
280 guint new;
281
282 // we can possibly go below 0 here, so by converting to a gint we avoid underflow errors.
283 new = MAX ((gint) backlight->priv->brightness_percent - GPM_KBD_BACKLIGHT_STEP, 0);
284 return gpm_kbd_backlight_set (backlight, new, TRUE);
285}

353static guint
354gpm_kbd_backlight_get_ac_percentage_dimmed (GpmKbdBacklight *backlight,
355 guint dim_by)
356{
357 guint value;
358
359 value = g_settings_get_int (backlight->priv->settings, GPM_SETTINGS_KBD_BRIGHTNESS_ON_AC);
360 if (value > 100) {
361 value = 100;
362 }
363
364 if (dim_by) {
365 gfloat scale;
366
367 if (dim_by > 100) {
368 g_warning ("Cannot scale brightness down by more than 100%%. Scaling by 50%%");
369 dim_by = 50;
370 }
371
372 scale = (100 - dim_by) / 100.0f;
373
374 value = (guint) ((gfloat) value * scale);
375 }
376
377 return value;
378}

380static gboolean
381gpm_kbd_backlight_evaluate_power_source_and_set (GpmKbdBacklight *backlight)
382{
383 guint value;
384 guint dim_by = 0;
385
386 if (g_settings_get_boolean (backlight->priv->settings, GPM_SETTINGS_KBD_BACKLIGHT_ENABLE) == FALSE) {
387 g_debug ("policy is no dimming");
388 return TRUE;
389 }
390
391 if (up_client_get_on_battery (backlight->priv->client) &&
392 g_settings_get_boolean (backlight->priv->settings, GPM_SETTINGS_KBD_BACKLIGHT_BATT_REDUCE)) {
393 dim_by = g_settings_get_int (backlight->priv->settings, GPM_SETTINGS_KBD_BRIGHTNESS_DIM_BY_ON_BATT);
394 }
395
396 value = gpm_kbd_backlight_get_ac_percentage_dimmed (backlight, dim_by);
397
398 return gpm_kbd_backlight_set (backlight, value, FALSE);
399}

443static void
444gpm_kbd_backlight_button_pressed_cb (GpmButton *button,
445 const gchar *type,
446 GpmKbdBacklight *backlight)
447{
448 static guint saved_brightness;
449 gboolean ret;
450
451 if (g_strcmp0 (type, GPM_BUTTON_KBD_BRIGHT_UP) == 0) {
452 ret = gpm_kbd_backlight_brightness_up (backlight);
453
454 if (ret) {
455 g_debug("Going to display OSD");
456 gpm_kbd_backlight_dialog_init (backlight);
457 msd_media_keys_window_set_volume_level (MSD_MEDIA_KEYS_WINDOW (backlight->priv->popup), backlight->priv->brightness_percent);
458 gpm_kbd_backlight_dialog_show (backlight);
459 }
460
461 } else if (g_strcmp0 (type, GPM_BUTTON_KBD_BRIGHT_DOWN) == 0) {
462 ret = gpm_kbd_backlight_brightness_down (backlight);
463
464 if (ret) {
465 g_debug("Going to display OSD");
466 gpm_kbd_backlight_dialog_init (backlight);
467 msd_media_keys_window_set_volume_level (MSD_MEDIA_KEYS_WINDOW (backlight->priv->popup), backlight->priv->brightness_percent);
468 gpm_kbd_backlight_dialog_show (backlight);
469 }
470
471 } else if (g_strcmp0 (type, GPM_BUTTON_KBD_BRIGHT_TOGGLE) == 0) {
472 if (backlight->priv->brightness_percent == 0) {
473 /* backlight is off turn it back on.
474 * If the initial value is 0, which probably means we saved on_ac=0, we
475 * try and restore some arbitrary value for the toggle not to seem
476 * broken. */
477 if (saved_brightness == 0) {
478 saved_brightness = 100u;
479 }
480 gpm_kbd_backlight_set (backlight, saved_brightness, TRUE);
481 } else {
482 /* backlight is on, turn it off and save current value */
483 saved_brightness = backlight->priv->brightness_percent;
484 gpm_kbd_backlight_set (backlight, 0, TRUE);
485 }
486 }

500static void
501gpm_kbd_backlight_idle_changed_cb (GpmIdle *idle,
502 GpmIdleMode mode,
503 GpmKbdBacklight *backlight)
504{
505 guint value;
506 gboolean enable_action;
507
508 g_debug("Idle changed");
509
510 if (gpm_button_is_lid_closed (backlight->priv->button))
511 return;
512
513 enable_action = up_client_get_on_battery (backlight->priv->client)
514 ? g_settings_get_boolean (backlight->priv->settings, GPM_SETTINGS_IDLE_DIM_BATT)
515 : g_settings_get_boolean (backlight->priv->settings, GPM_SETTINGS_IDLE_DIM_AC);
516
517 if (!enable_action)
518 return;
519
520 if (mode == GPM_IDLE_MODE_NORMAL) {
521 g_debug ("GPM_IDLE_MODE_NORMAL");
522 gpm_kbd_backlight_evaluate_power_source_and_set (backlight);
523 } else if (mode == GPM_IDLE_MODE_DIM) {
524 g_debug ("GPM_IDLE_MODE_DIM");
525 value = g_settings_get_int (backlight->priv->settings, GPM_SETTINGS_KBD_BRIGHTNESS_DIM_BY_ON_IDLE);
526 value = gpm_kbd_backlight_get_ac_percentage_dimmed (backlight, value);
527 gpm_kbd_backlight_set (backlight, value, FALSE);
528 } else if (mode == GPM_IDLE_MODE_BLANK) {
529 gpm_kbd_backlight_set (backlight, 0u, FALSE);
530 }
531}

537static void
538gpm_kbd_backlight_finalize (GObject *object)
539{
540 GpmKbdBacklight *backlight;
541
542 g_return_if_fail (object != NULL);
543 g_return_if_fail (GPM_IS_KBD_BACKLIGHT (object));
544
545 backlight = GPM_KBD_BACKLIGHT (object);
546
547 if (backlight->priv->upower_proxy != NULL) {
548 g_object_unref (backlight->priv->upower_proxy);
549 }
550 if (backlight->priv->bus_connection != NULL) {
551 g_dbus_connection_unregister_object (backlight->priv->bus_connection,
552 backlight->priv->bus_object_id);
553 g_object_unref (backlight->priv->bus_connection);
554 }
555
556 g_timer_destroy (backlight->priv->idle_timer);
557
558 g_object_unref (backlight->priv->control);
559 g_object_unref (backlight->priv->settings);
560 g_object_unref (backlight->priv->client);
561 g_object_unref (backlight->priv->button);
562 g_object_unref (backlight->priv->idle);
563
564 g_return_if_fail (backlight->priv != NULL);
565 G_OBJECT_CLASS (gpm_kbd_backlight_parent_class)->finalize (object);
566}

597static void
598gpm_kbd_backlight_init (GpmKbdBacklight *backlight)
599{
600 GVariant *u_brightness;
601 GVariant *u_max_brightness;
602 GError *error = NULL;
603
604 backlight->priv = gpm_kbd_backlight_get_instance_private (backlight);
605
606 backlight->priv->upower_proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SYSTEM,
607 G_DBUS_PROXY_FLAGS_DO_NOT_LOAD_PROPERTIES,
608 NULL,
609 "org.freedesktop.UPower",
610 "/org/freedesktop/UPower/KbdBacklight",
611 "org.freedesktop.UPower.KbdBacklight",
612 NULL,
613 &error);
614 if (backlight->priv->upower_proxy == NULL) {
615 g_printerr ("Could not connect to UPower system bus: %s", error->message);
616 g_error_free (error);
617 goto err;
618 }
619
620 g_signal_connect (backlight->priv->upower_proxy,
621 "g-signal",
622 G_CALLBACK (gpm_kbd_backlight_on_dbus_signal),
623 backlight);
624
625 u_brightness = g_dbus_proxy_call_sync (backlight->priv->upower_proxy,
626 "GetBrightness",
627 NULL,
628 G_DBUS_CALL_FLAGS_NONE,
629 -1,
630 NULL,
631 &error);
632 if (u_brightness == NULL) {
633 if (error->domain != G_DBUS_ERROR || error->code != G_DBUS_ERROR_UNKNOWN_METHOD)
634 g_warning ("Failed to get brightness: %s", error->message);
635 g_error_free (error);
636 goto err;
637 }
638
639 error = NULL;
640 u_max_brightness = g_dbus_proxy_call_sync (backlight->priv->upower_proxy,
641 "GetMaxBrightness",
642 NULL,
643 G_DBUS_CALL_FLAGS_NONE,
644 -1,
645 NULL,
646 &error);
647 if (u_max_brightness == NULL) {
648 g_warning ("Failed to get max brightness: %s", error->message);
649 g_error_free (error);
650 g_variant_unref (u_brightness);
651 goto err;
652 }
653
654 g_variant_get (u_brightness, "(i)", &backlight->priv->brightness);
655 g_variant_get (u_max_brightness, "(i)", &backlight->priv->max_brightness);
656
657 backlight->priv->brightness_percent = gpm_discrete_to_percent (backlight->priv->brightness,
658 backlight->priv->max_brightness);
659
660 g_variant_unref (u_brightness);
661 g_variant_unref (u_max_brightness);
662 goto noerr;
663
664err:
665 backlight->priv->brightness = 0;
666 backlight->priv->brightness_percent = 100;
667 backlight->priv->max_brightness = 0;
668
669noerr:
670 backlight->priv->idle_timer = g_timer_new ();
671 backlight->priv->can_dim = backlight->priv->max_brightness > 1;
672
673 /* Use upower for ac changed signal */
674 backlight->priv->client = up_client_new ();
675 g_signal_connect (backlight->priv->client, "notify",
676 G_CALLBACK (gpm_kbd_backlight_client_changed_cb), backlight);
677
678 backlight->priv->settings = g_settings_new (GPM_SETTINGS_SCHEMA);
679
680 /* watch for kbd brightness up and down button presses */
681 backlight->priv->button = gpm_button_new ();
682 g_signal_connect (backlight->priv->button, "button-pressed",
683 G_CALLBACK (gpm_kbd_backlight_button_pressed_cb), backlight);
684
685 backlight->priv->idle = gpm_idle_new ();
686 g_signal_connect (backlight->priv->idle, "idle-changed",
687 G_CALLBACK (gpm_kbd_backlight_idle_changed_cb), backlight);
688
689 /* use a visual widget */
690 backlight->priv->popup = msd_media_keys_window_new ();
691 msd_media_keys_window_set_action_custom (MSD_MEDIA_KEYS_WINDOW (backlight->priv->popup),
692 "gpm-brightness-kbd", TRUE);
693 gtk_window_set_position (GTK_WINDOW (backlight->priv->popup), GTK_WIN_POS_NONE);
694
695 /* since gpm is just starting we can pretty safely assume that we're not idle */
696 backlight->priv->system_is_idle = FALSE;
697 backlight->priv->idle_dim_timeout = g_settings_get_int (backlight->priv->settings, GPM_SETTINGS_IDLE_DIM_TIME);
698 gpm_idle_set_timeout_dim (backlight->priv->idle, backlight->priv->idle_dim_timeout);
699
700 /* make sure we turn the keyboard backlight back on after resuming */
701 backlight->priv->control = gpm_control_new ();
702 g_signal_connect (backlight->priv->control, "resume",
703 G_CALLBACK (gpm_kbd_backlight_control_resume_cb), backlight);
704
705 /* set initial values for whether we're on AC or battery*/
706 gpm_kbd_backlight_evaluate_power_source_and_set (backlight);
707}
