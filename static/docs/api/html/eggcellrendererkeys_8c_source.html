233{
234 gboolean visible;
235
236 g_object_get (keys, "visible", &visible, NULL);
237
238 /* For screen reader's sake, make the text empty when hidden, otherwise a
239 * more or less random value is presented. This could also be fixed in the

327static gboolean is_modifier(guint keycode)
328{
329 gint i;
330 gint map_size;
331 XModifierKeymap* mod_keymap;
332 gboolean retval = FALSE;
333
334 mod_keymap = XGetModifierMapping(gdk_x11_display_get_xdisplay(gdk_display_get_default()));
335
336 map_size = 8 * mod_keymap->max_keypermod;
337 i = 0;
338
339 while (i < map_size)
340 {
341 if (keycode == mod_keymap->modifiermap[i])
342 {
343 retval = TRUE;
344 break;
345 }
346
347 ++i;
348 }
349
350 XFreeModifiermap(mod_keymap);
351
352 return retval;
353}

383static gboolean grab_key_callback(GtkWidget* widget, GdkEventKey* event, void* data)
384{
385 GdkModifierType accel_mods = 0;
386 guint accel_keyval;
387 EggCellRendererKeys *keys;
388 char *path;
389 gboolean edited;
390 gboolean cleared;
391 GdkModifierType consumed_modifiers;
392 guint upper;
393 GdkModifierType ignored_modifiers;
394 GdkDisplay *display;
395 GdkSeat *seat;
396
397 keys = EGG_CELL_RENDERER_KEYS(data);
398
399 if (is_modifier(event->hardware_keycode))
400 {
401 return TRUE;
402 }
403
404 edited = FALSE;
405 cleared = FALSE;
406
407 consumed_modifiers = 0;
408 gdk_keymap_translate_keyboard_state (gdk_keymap_get_for_display (gdk_display_get_default ()),
409 event->hardware_keycode,
410 event->state,
411 event->group,
412 NULL, NULL, NULL, &consumed_modifiers);
413
414 upper = event->keyval;
415 accel_keyval = gdk_keyval_to_lower(upper);
416
417 if (accel_keyval == GDK_KEY_ISO_Left_Tab)
418 {
419 accel_keyval = GDK_KEY_Tab;
420 }
421
422 /* Put shift back if it changed the case of the key, not otherwise. */
423 if (upper != accel_keyval && (consumed_modifiers & GDK_SHIFT_MASK))
424 {
425 consumed_modifiers &= ~(GDK_SHIFT_MASK);
426 }
427
428 egg_keymap_resolve_virtual_modifiers (gdk_keymap_get_for_display (gdk_display_get_default ()),
429 EGG_VIRTUAL_NUM_LOCK_MASK |
430 EGG_VIRTUAL_SCROLL_LOCK_MASK |
431 EGG_VIRTUAL_LOCK_MASK,
432 &ignored_modifiers);
433
434 /* http://bugzilla.gnome.org/show_bug.cgi?id=139605
435 * mouse keys should effect keybindings */
436 ignored_modifiers |= GDK_BUTTON1_MASK |
437 GDK_BUTTON2_MASK |
438 GDK_BUTTON3_MASK |
439 GDK_BUTTON4_MASK |
440 GDK_BUTTON5_MASK;
441
442 /* filter consumed/ignored modifiers */
443 if (keys->accel_mode == EGG_CELL_RENDERER_KEYS_MODE_GTK)
444 {
445 accel_mods = event->state & GDK_MODIFIER_MASK & ~(consumed_modifiers | ignored_modifiers);
446 }
447 else if (keys->accel_mode == EGG_CELL_RENDERER_KEYS_MODE_X)
448 {
449 accel_mods = event->state & GDK_MODIFIER_MASK & ~(ignored_modifiers);
450 }
451 else
452 {
453 g_assert_not_reached();
454 }
455
456 if (accel_mods == 0 && accel_keyval == GDK_KEY_Escape)
457 {
458 goto out; /* cancel */
459 }
460
461 /* clear the accelerator on Backspace */
462 if (accel_mods == 0 && accel_keyval == GDK_KEY_BackSpace)
463 {
464 cleared = TRUE;
465 goto out;
466 }
467
468 if (keys->accel_mode == EGG_CELL_RENDERER_KEYS_MODE_GTK)
469 {
470 if (!gtk_accelerator_valid (accel_keyval, accel_mods))
471 {
472 accel_keyval = 0;
473 accel_mods = 0;
474 }
475 }
476
477 edited = TRUE;
478
479 out:
480
481 display = gtk_widget_get_display (widget);
482 seat = gdk_display_get_default_seat (display);
483
484 gdk_seat_ungrab (seat);
485
486 path = g_strdup(g_object_get_data(G_OBJECT(keys->edit_widget), EGG_CELL_RENDERER_TEXT_PATH));
487
488 gtk_cell_editable_editing_done(GTK_CELL_EDITABLE(keys->edit_widget));
489 gtk_cell_editable_remove_widget(GTK_CELL_EDITABLE(keys->edit_widget));
490 keys->edit_widget = NULL;
491 keys->grab_widget = NULL;
492
493 if (edited)
494 {
495 g_signal_emit_by_name(G_OBJECT(keys), "accel_edited", path, accel_keyval, accel_mods, event->hardware_keycode);
496 }
497 else if (cleared)
498 {
499 g_signal_emit_by_name(G_OBJECT(keys), "accel_cleared", path);
500 }
501
502 g_free (path);
503 return TRUE;
504}

698void egg_cell_renderer_keys_set_accelerator(EggCellRendererKeys* keys, guint keyval, guint keycode, EggVirtualModifierType mask)
699{
700 gboolean changed;
701
702 g_return_if_fail (EGG_IS_CELL_RENDERER_KEYS (keys));
703
704 g_object_freeze_notify (G_OBJECT (keys));
705
706 changed = FALSE;
707
708 if (keyval != keys->accel_key)
709 {
710 keys->accel_key = keyval;
711 g_object_notify (G_OBJECT (keys), "accel_key");
712 changed = TRUE;
713 }
714
715 if (mask != keys->accel_mask)
