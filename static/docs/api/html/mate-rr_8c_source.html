242static void
243screen_info_free (ScreenInfo *info)
244{
245 MateRROutput **output;
246 MateRRCrtc **crtc;
247 MateRRMode **mode;
248
249 g_assert (info != NULL);
250
251#ifdef HAVE_RANDR
252 if (info->resources)
253 {
254 XRRFreeScreenResources (info->resources);
255
256 info->resources = NULL;
257 }
258#endif
259
260 if (info->outputs)
261 {
262 for (output = info->outputs; *output; ++output)
263 output_free (*output);
264 g_free (info->outputs);
265 }
266
267 if (info->crtcs)
268 {
269 for (crtc = info->crtcs; *crtc; ++crtc)
270 crtc_free (*crtc);
271 g_free (info->crtcs);
272 }
273
274 if (info->modes)
275 {
276 for (mode = info->modes; *mode; ++mode)
277 mode_free (*mode);
278 g_free (info->modes);
279 }
280
281 if (info->clone_modes)
282 {
283 /* The modes themselves were freed above */
284 g_free (info->clone_modes);
285 }
286
287 g_free (info);
288}

290static gboolean
291has_similar_mode (MateRROutput *output, MateRRMode *mode)
292{
293 int i;
294 MateRRMode **modes = mate_rr_output_list_modes (output);
295 guint width = mate_rr_mode_get_width (mode);
296 guint height = mate_rr_mode_get_height (mode);
297
298 for (i = 0; modes[i] != NULL; ++i)
299 {
300 MateRRMode *m = modes[i];
301
302 if (mate_rr_mode_get_width (m) == width &&
303 mate_rr_mode_get_height (m) == height)
304 {
305 return TRUE;
306 }
307 }
308
309 return FALSE;
310}

312static void
313gather_clone_modes (ScreenInfo *info)
314{
315 int i;
316 GPtrArray *result = g_ptr_array_new ();
317
318 for (i = 0; info->outputs[i] != NULL; ++i)
319 {
320 int j;
321 MateRROutput *output1, *output2;
322
323 output1 = info->outputs[i];
324
325 if (!output1->connected)
326 continue;
327
328 for (j = 0; output1->modes[j] != NULL; ++j)
329 {
330 MateRRMode *mode = output1->modes[j];
331 gboolean valid;
332 int k;
333
334 valid = TRUE;
335 for (k = 0; info->outputs[k] != NULL; ++k)
336 {
337 output2 = info->outputs[k];
338
339 if (!output2->connected)
340 continue;
341
342 if (!has_similar_mode (output2, mode))
343 {
344 valid = FALSE;
345 break;
346 }
347 }
348
349 if (valid)
350 g_ptr_array_add (result, mode);
351 }
352 }
353
354 g_ptr_array_add (result, NULL);
355
356 info->clone_modes = (MateRRMode **)g_ptr_array_free (result, FALSE);
357}

359#ifdef HAVE_RANDR
360static gboolean
361fill_screen_info_from_resources (ScreenInfo *info,
362 XRRScreenResources *resources,
363 GError **error)
364{
365 int i;
366 GPtrArray *a;
367 MateRRCrtc **crtc;
368 MateRROutput **output;
369
370 info->resources = resources;
371
372 /* We create all the structures before initializing them, so
373 * that they can refer to each other.
374 */
375 a = g_ptr_array_new ();
376 for (i = 0; i < resources->ncrtc; ++i)
377 {
378 g_ptr_array_add (a, crtc_new (info, resources->crtcs[i]));
379 }
380 g_ptr_array_add (a, NULL);
381 info->crtcs = (MateRRCrtc **)g_ptr_array_free (a, FALSE);
382
383 a = g_ptr_array_new ();
384 for (i = 0; i < resources->noutput; ++i)
385 {
386 g_ptr_array_add (a, output_new (info, resources->outputs[i]));
387 }
388 g_ptr_array_add (a, NULL);
389 info->outputs = (MateRROutput **)g_ptr_array_free (a, FALSE);
390
391 a = g_ptr_array_new ();
392 for (i = 0; i < resources->nmode; ++i)
393 {
394 MateRRMode *mode = mode_new (info, resources->modes[i].id);
395
396 g_ptr_array_add (a, mode);
397 }
398 g_ptr_array_add (a, NULL);
399 info->modes = (MateRRMode **)g_ptr_array_free (a, FALSE);
400
401 /* Initialize */
402 for (crtc = info->crtcs; *crtc; ++crtc)
403 {
404 if (!crtc_initialize (*crtc, resources, error))
405 return FALSE;
406 }
407
408 for (output = info->outputs; *output; ++output)
409 {
410 if (!output_initialize (*output, resources, error))
411 return FALSE;
412 }
413
414 for (i = 0; i < resources->nmode; ++i)
415 {
416 MateRRMode *mode = mode_by_id (info, resources->modes[i].id);
417
418 mode_initialize (mode, &(resources->modes[i]));
419 }
420
421 gather_clone_modes (info);
422
423 return TRUE;
424}
425#endif /* HAVE_RANDR */

427static gboolean
428fill_out_screen_info (Display *xdisplay,
429 Window xroot,
430 ScreenInfo *info,
431 gboolean needs_reprobe,
432 GError **error)
433{
434#ifdef HAVE_RANDR
435 XRRScreenResources *resources;
436 GdkDisplay *display;
437
438 g_assert (xdisplay != NULL);
439 g_assert (info != NULL);
440
441 /* First update the screen resources */
442
443 if (needs_reprobe)
444 resources = XRRGetScreenResources (xdisplay, xroot);
445 else
446 resources = XRRGetScreenResourcesCurrent (xdisplay, xroot);
447
448 if (resources)
449 {
450 if (!fill_screen_info_from_resources (info, resources, error))
451 return FALSE;
452 }
453 else
454 {
455 g_set_error (error, MATE_RR_ERROR, MATE_RR_ERROR_RANDR_ERROR,
456 /* Translators: a CRTC is a CRT Controller (this is X terminology). */
457 _("could not get the screen resources (CRTCs, outputs, modes)"));
458 return FALSE;
459 }
460
461 /* Then update the screen size range. We do this after XRRGetScreenResources() so that
462 * the X server will already have an updated view of the outputs.
463 */
464
465 if (needs_reprobe) {
466 gboolean success;
467
468 display = gdk_display_get_default ();
469 gdk_x11_display_error_trap_push (display);
470 success = XRRGetScreenSizeRange (xdisplay, xroot,
471 &(info->min_width),
472 &(info->min_height),
473 &(info->max_width),
474 &(info->max_height));
475 gdk_display_flush (display);
476 if (gdk_x11_display_error_trap_pop (display)) {
477 g_set_error (error, MATE_RR_ERROR, MATE_RR_ERROR_UNKNOWN,
478 _("unhandled X error while getting the range of screen sizes"));
479 return FALSE;
480 }
481
482 if (!success) {
483 g_set_error (error, MATE_RR_ERROR, MATE_RR_ERROR_RANDR_ERROR,
484 _("could not get the range of screen sizes"));
485 return FALSE;
486 }
487 }
488 else
489 {
490 mate_rr_screen_get_ranges (info->screen,
491 &(info->min_width),
492 &(info->max_width),
493 &(info->min_height),
494 &(info->max_height));
495 }
496
497 info->primary = None;
498 display = gdk_display_get_default ();
499 gdk_x11_display_error_trap_push (display);
500 info->primary = XRRGetOutputPrimary (xdisplay, xroot);
501 gdk_x11_display_error_trap_pop_ignored (display);
502
503 return TRUE;
504#else
505 return FALSE;
506#endif /* HAVE_RANDR */
507}

509static ScreenInfo *
510screen_info_new (MateRRScreen *screen, gboolean needs_reprobe, GError **error)
511{
512 ScreenInfo *info = g_new0 (ScreenInfo, 1);
513 MateRRScreenPrivate *priv;
514
515 g_assert (screen != NULL);
516
517 priv = screen->priv;
518
519 info->outputs = NULL;
520 info->crtcs = NULL;
521 info->modes = NULL;
522 info->screen = screen;
523
524 if (fill_out_screen_info (priv->xdisplay, priv->xroot, info, needs_reprobe, error))
525 {
526 return info;
527 }
528 else
529 {
530 screen_info_free (info);
531 return NULL;
532 }
533}

535static gboolean
536screen_update (MateRRScreen *screen, gboolean force_callback, gboolean needs_reprobe, GError **error)
537{
538 ScreenInfo *info;
539 gboolean changed = FALSE;
540
541 g_assert (screen != NULL);
542
543 info = screen_info_new (screen, needs_reprobe, error);
544 if (!info)
545 return FALSE;
546
547#ifdef HAVE_RANDR
548 if (info->resources->configTimestamp != screen->priv->info->resources->configTimestamp)
549 changed = TRUE;
550#endif
551
552 screen_info_free (screen->priv->info);
553
554 screen->priv->info = info;
555
556 if (changed || force_callback)
557 g_signal_emit (G_OBJECT (screen), screen_signals[SCREEN_CHANGED], 0);
558
559 return changed;
560}

562static GdkFilterReturn
563screen_on_event (GdkXEvent *xevent,
564 GdkEvent *event,
565 gpointer data)
566{
567#ifdef HAVE_RANDR
568 MateRRScreen *screen = data;
569 MateRRScreenPrivate *priv = screen->priv;
570 XEvent *e = xevent;
571 int event_num;
572
573 if (!e)
574 return GDK_FILTER_CONTINUE;
575
576 event_num = e->type - priv->randr_event_base;
577
578 if (event_num == RRScreenChangeNotify) {
579 /* We don't reprobe the hardware; we just fetch the X server's latest
580 * state. The server already knows the new state of the outputs; that's
581 * why it sent us an event!
582 */
583 screen_update (screen, TRUE, FALSE, NULL); /* NULL-GError */
584#if 0
585 /* Enable this code to get a dialog showing the RANDR timestamps, for debugging purposes */
586 {
587 GtkWidget *dialog;
588 XRRScreenChangeNotifyEvent *rr_event;
589 static int dialog_num;

593 dialog = gtk_message_dialog_new (NULL,
594 0,
595 GTK_MESSAGE_INFO,
596 GTK_BUTTONS_CLOSE,
597 "RRScreenChangeNotify timestamps (%d):\n"
598 "event change: %u\n"
599 "event config: %u\n"
600 "event serial: %lu\n"
601 "----------------------"
602 "screen change: %u\n"
603 "screen config: %u\n",
604 dialog_num++,
605 (guint32) rr_event->timestamp,
606 (guint32) rr_event->config_timestamp,
607 rr_event->serial,
608 (guint32) priv->info->resources->timestamp,
609 (guint32) priv->info->resources->configTimestamp);
610 g_signal_connect (dialog, "response",
611 G_CALLBACK (gtk_widget_destroy), NULL);
612 gtk_widget_show (dialog);
613 }
614#endif
615 }
616#if 0
617 /* WHY THIS CODE IS DISABLED:
618 *
619 * Note that in mate_rr_screen_new(), we only select for
620 * RRScreenChangeNotifyMask. We used to select for other values in
621 * RR*NotifyMask, but we weren't really doing anything useful with those
622 * events. We only care about "the screens changed in some way or another"
623 * for now.
624 *
625 * If we ever run into a situtation that could benefit from processing more
626 * detailed events, we can enable this code again.
627 *
628 * Note that the X server sends RRScreenChangeNotify in conjunction with the
629 * more detailed events from RANDR 1.2 - see xserver/randr/randr.c:TellChanged().
630 */
631 else if (event_num == RRNotify)
632 {
633 /* Other RandR events */
634
635 XRRNotifyEvent *event = (XRRNotifyEvent *)e;

637 /* Here we can distinguish between RRNotify events supported

640 * we leave this as an empty switch().
641 */
642 switch (event->subtype)
643 {
644 default:
645 break;
646 }
647
648 /* No need to reprobe hardware here */
649 screen_update (screen, TRUE, FALSE, NULL); /* NULL-GError */
650 }
651#endif
652
653#endif /* HAVE_RANDR */
654
655 /* Pass the event on to GTK+ */
656 return GDK_FILTER_CONTINUE;

659static gboolean
660mate_rr_screen_initable_init (GInitable *initable, GCancellable *canc, GError **error)
661
662{
663 MateRRScreen *self = MATE_RR_SCREEN (initable);
664 MateRRScreenPrivate *priv = self->priv;
665 Display *dpy = GDK_SCREEN_XDISPLAY (self->priv->gdk_screen);
666 int event_base;
667 int ignore;
668
669 priv->connector_type_atom = XInternAtom (dpy, "ConnectorType", FALSE);
670
671#ifdef HAVE_RANDR
672 if (XRRQueryExtension (dpy, &event_base, &ignore))
673 {
674 priv->randr_event_base = event_base;
675
676 XRRQueryVersion (dpy, &priv->rr_major_version, &priv->rr_minor_version);
677 if (priv->rr_major_version < 1 || (priv->rr_major_version == 1 && priv->rr_minor_version < 3)) {
678 g_set_error (error, MATE_RR_ERROR, MATE_RR_ERROR_NO_RANDR_EXTENSION,
679 "RANDR extension is too old (must be at least 1.3)");
680 return FALSE;
681 }
682
683 priv->info = screen_info_new (self, TRUE, error);
684
685 if (!priv->info) {
686 return FALSE;
687 }
688
689 XRRSelectInput (priv->xdisplay,
690 priv->xroot,
691 RRScreenChangeNotifyMask);
692 gdk_x11_register_standard_event_type (gdk_screen_get_display (priv->gdk_screen),
693 event_base,
694 RRNotify + 1);
695 gdk_window_add_filter (priv->gdk_root, screen_on_event, self);
696
697 return TRUE;
698 }
699 else
700 {
701#endif /* HAVE_RANDR */
702 g_set_error (error, MATE_RR_ERROR, MATE_RR_ERROR_NO_RANDR_EXTENSION,
703 _("RANDR extension is not present"));
704
705 return FALSE;
706
707#ifdef HAVE_RANDR
708 }
709#endif
710}

800void
801mate_rr_screen_init (MateRRScreen *self)
802{
803 MateRRScreenPrivate *priv = mate_rr_screen_get_instance_private (self);
804 self->priv = priv;
