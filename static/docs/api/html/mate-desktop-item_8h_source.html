109typedef enum {
110 /* Never launch more instances even if the app can only
111 * handle one file and we have passed many */
112 MATE_DESKTOP_ITEM_LAUNCH_ONLY_ONE = 1<<0,
113 /* Use current directory instead of home directory */
114 MATE_DESKTOP_ITEM_LAUNCH_USE_CURRENT_DIR = 1<<1,
115 /* Append the list of URIs to the command if no Exec
116 * parameter is specified, instead of launching the
117 * app without parameters. */
118 MATE_DESKTOP_ITEM_LAUNCH_APPEND_URIS = 1<<2,
119 /* Same as above but instead append local paths */
120 MATE_DESKTOP_ITEM_LAUNCH_APPEND_PATHS = 1<<3,
121 /* Don't automatically reap child process. */
122 MATE_DESKTOP_ITEM_LAUNCH_DO_NOT_REAP_CHILD = 1<<4
123} MateDesktopItemLaunchFlags;
