126static gboolean
127alsa_switch_set_active_option (MateMixerSwitch *mms, MateMixerSwitchOption *mmso)
128{
129 AlsaSwitch *swtch;
130 guint index;
131 gboolean set_item = FALSE;
132 snd_mixer_selem_channel_id_t channel;
133
134 g_return_val_if_fail (ALSA_IS_SWITCH (mms), FALSE);
135 g_return_val_if_fail (ALSA_IS_SWITCH_OPTION (mmso), FALSE);
136
137 swtch = ALSA_SWITCH (mms);
138
139 if (G_UNLIKELY (swtch->priv->element == NULL))
140 return FALSE;
141
142 /* The channel mask is created when reading the active option the first
143 * time, so a successful load must be done before changing the option */
144 if (G_UNLIKELY (swtch->priv->channel_mask == 0)) {
145 g_debug ("Not setting active switch option, channel mask unknown");
146 return FALSE;
147 }
148
149 index = alsa_switch_option_get_id (ALSA_SWITCH_OPTION (mmso));
150
151 for (channel = 0; channel < SND_MIXER_SCHN_LAST; channel++) {
152 /* The option is set per-channel, make sure to set it only for channels
153 * we successfully read the value from */
154 if (swtch->priv->channel_mask & (1 << channel)) {
155 gint ret = snd_mixer_selem_set_enum_item (swtch->priv->element,
156 channel,
157 index);
158 if (ret == 0)
159 set_item = TRUE;
160 else
161 g_warning ("Failed to set active option of switch %s: %s",
162 snd_mixer_selem_get_name (swtch->priv->element),
163 snd_strerror (ret));
164 }
165 }
166 return set_item;
167}

193static gboolean
194alsa_switch_load (AlsaElement *element)
195{
196 AlsaSwitch *swtch;
197 GList *list;
198 guint item;
199 gint ret;
200 snd_mixer_selem_channel_id_t c;
201
202 g_return_val_if_fail (ALSA_IS_SWITCH (element), FALSE);
203
204 swtch = ALSA_SWITCH (element);
205
206 if (G_UNLIKELY (swtch->priv->element == NULL))
207 return FALSE;
208
209 /* When reading the first time we try all the channels, otherwise only the
210 * ones which returned success before */
211 if (swtch->priv->channel_mask == 0) {
212 for (c = 0; c < SND_MIXER_SCHN_LAST; c++) {
213 ret = snd_mixer_selem_get_enum_item (swtch->priv->element, c, &item);
214
215 /* The active enum option is set per-channel, so when reading it the
216 * first time, create a mask of all channels for which we read the
217 * value successfully */
218 if (ret == 0)
219 swtch->priv->channel_mask |= 1 << c;
220 }
221
222 /* The last ALSA call might have failed, but it doesn't matter if we have
223 * a channel mask */
224 if (swtch->priv->channel_mask > 0)
225 ret = 0;
226 } else {
227 for (c = 0; !(swtch->priv->channel_mask & (1 << c)); c++)
228 ;
229
230 /* When not reading the mask, the first usable channel is enough, we don't
231 * support per-channel selections anyway */
232 ret = snd_mixer_selem_get_enum_item (swtch->priv->element, c, &item);
233 }
234
235 if (ret < 0) {
236 g_warning ("Failed to read active option of switch %s: %s",
237 snd_mixer_selem_get_name (swtch->priv->element),
238 snd_strerror (ret));
239 return FALSE;
240 }
241
242 list = swtch->priv->options;
243 while (list != NULL) {
244 AlsaSwitchOption *option = ALSA_SWITCH_OPTION (list->data);
245
246 /* Mark the selected option when we find it, ALSA indentifies them
247 * by numeric indices */
248 if (alsa_switch_option_get_id (option) == item) {
249 _mate_mixer_switch_set_active_option (MATE_MIXER_SWITCH (swtch),
250 MATE_MIXER_SWITCH_OPTION (option));
251 return TRUE;
252 }
253 list = list->next;
254 }
255
256 g_warning ("Unknown active option of switch %s: %d",
257 mate_mixer_switch_get_name (MATE_MIXER_SWITCH (swtch)),
258 item);
259
260 return FALSE;
261}
