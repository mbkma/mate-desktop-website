62static ThemeEngine* load_theme_engine(const char *name)
63{
64 ThemeEngine* engine;
65 char* path;
66
67 path = g_module_build_path (ENGINES_DIR, name);
68
69 engine = g_new0(ThemeEngine, 1);
70 engine->ref_count = 1;
71 engine->module = g_module_open(path, G_MODULE_BIND_LAZY);
72
73 g_free(path);
74
75 if (engine->module == NULL)
76 goto error;
77
78 #define BIND_REQUIRED_FUNC(name) \
79 if (!g_module_symbol(engine->module, #name, (gpointer*) &engine->name)) \
80 { \
81 /* Too harsh! Fall back to default. */ \
82 g_warning("Theme doesn't provide the required function '%s'", #name); \
83 goto error; \
84 }
85
86 #define BIND_OPTIONAL_FUNC(name) \
87 g_module_symbol(engine->module, #name, (gpointer*) &engine->name);
88
89 BIND_REQUIRED_FUNC(theme_check_init);
90 BIND_REQUIRED_FUNC(get_theme_info);
91 BIND_REQUIRED_FUNC(create_notification);
92 BIND_REQUIRED_FUNC(set_notification_text);
93 BIND_REQUIRED_FUNC(set_notification_icon);
94 BIND_REQUIRED_FUNC(set_notification_arrow);
95 BIND_REQUIRED_FUNC(add_notification_action);
96 BIND_REQUIRED_FUNC(clear_notification_actions);
97 BIND_REQUIRED_FUNC(move_notification);
98
99 BIND_OPTIONAL_FUNC(destroy_notification);
100 BIND_OPTIONAL_FUNC(show_notification);
101 BIND_OPTIONAL_FUNC(hide_notification);
102 BIND_OPTIONAL_FUNC(set_notification_timeout);
103 BIND_OPTIONAL_FUNC(set_notification_hints);
104 BIND_OPTIONAL_FUNC(notification_tick);
105 BIND_OPTIONAL_FUNC(get_always_stack);
106
107 if (!engine->theme_check_init(NOTIFICATION_DAEMON_MAJOR_VERSION, NOTIFICATION_DAEMON_MINOR_VERSION, NOTIFICATION_DAEMON_MICRO_VERSION))
108 {
109 g_warning ("Theme doesn't work with this version of mate-notification-daemon");
110 goto error;
111 }
112
113 return engine;
114
115 error:
116
117 if (engine->module != NULL && !g_module_close (engine->module))
118 {
119 g_warning("%s: %s", name, g_module_error());
120 }
121
122 g_free(engine);
123 return NULL;
124}

126static void destroy_engine(ThemeEngine* engine)
127{
128 g_assert(engine->ref_count == 0);
129
130 if (active_engine == engine)
131 active_engine = NULL;
132
133 g_module_close(engine->module);
134 g_free(engine);
135}

137static gboolean theme_engine_destroy(ThemeEngine* engine)
138{
139 destroy_engine(engine);
140 return FALSE;

143static void theme_engine_unref(ThemeEngine* engine)
144{
145 engine->ref_count--;
146
147 if (engine->ref_count == 0)
148 {
149 /*
150 * Destroy the engine in an idle loop since the last reference
151 * might have been the one of a notification which is being
152 * destroyed and that still has references to the engine
153 * module. This way, we're sure the notification is completely
154 * destroyed before the engine is.
155 */
156 g_idle_add((GSourceFunc) theme_engine_destroy, engine);
157 }
158}

160static void theme_changed_cb(GSettings *settings, gchar *key, gpointer user_data)
161{
162 if (active_engine == NULL)
163 {
164 return;
165 }
166
167 theme_engine_unref(active_engine);
168
169 /* This is no longer the true active engine, so reset this. */
170 active_engine = NULL;
171}

173static ThemeEngine* get_theme_engine(void)
174{
175 if (active_engine == NULL)
176 {
177 GSettings* gsettings = g_settings_new (GSETTINGS_SCHEMA);
178
179 if (theme_prop_notify_id == 0)
180 {
181 theme_prop_notify_id = g_signal_connect (gsettings, "changed::" GSETTINGS_KEY_THEME, G_CALLBACK (theme_changed_cb), NULL);
182 }
183
184 char* enginename = g_settings_get_string(gsettings, GSETTINGS_KEY_THEME);
185 if (enginename == NULL || *enginename == '\0')
186 {
187 active_engine = load_theme_engine("standard");
188 g_assert(active_engine != NULL);
189 }
190 else
191 {
192 active_engine = load_theme_engine(enginename);
193
194 if (active_engine == NULL)
195 {
196 g_warning("Unable to load theme engine '%s'", enginename);
197 active_engine = load_theme_engine("standard");
198 }
199
200 g_assert(active_engine != NULL);
201 }
202
203 g_free(enginename);
204 }
205
206 return active_engine;
207}

209GtkWindow* theme_create_notification(UrlClickedCb url_clicked_cb)
210{
211 ThemeEngine* engine = get_theme_engine();
212 GtkWindow* nw = engine->create_notification(url_clicked_cb);
213 g_object_set_data_full(G_OBJECT(nw), "_theme_engine", engine, (GDestroyNotify) theme_engine_unref);
214 engine->ref_count++;
215#if HAVE_WAYLAND
216 if (GDK_IS_WAYLAND_DISPLAY (gdk_display_get_default ()))
217 {
218 wayland_init_notification (nw);
219 }
220#endif /* HAVE_WAYLAND */
221 return nw;
222}
