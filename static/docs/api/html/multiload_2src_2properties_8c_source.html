24#define NEVER_SENSITIVE "never_sensitive"

26/* set sensitive and setup NEVER_SENSITIVE appropriately */
27static void
28hard_set_sensitive (GtkWidget *w, gboolean sensitivity)
29{
30 gtk_widget_set_sensitive (w, sensitivity);
31 g_object_set_data (G_OBJECT (w), NEVER_SENSITIVE,
32 GINT_TO_POINTER ( ! sensitivity));
33}
34
35/* set sensitive, but always insensitive if NEVER_SENSITIVE is set */
36static void
37soft_set_sensitive (GtkWidget *w, gboolean sensitivity)
38{
39 if (g_object_get_data (G_OBJECT (w), NEVER_SENSITIVE))
40 gtk_widget_set_sensitive (w, FALSE);
41 else
42 gtk_widget_set_sensitive (w, sensitivity);
43}

45static void
46properties_set_insensitive(MultiloadApplet *ma)
47{
48 guint total_graphs = 0;
49 guint last_graph = 0;
50 guint i;
51
52 for (i = 0; i < graph_n; i++)
53 if (ma->graphs[i]->visible)
54 {
55 last_graph = i;
56 total_graphs++;
57 }
58
59 if (total_graphs < 2)
60 soft_set_sensitive (ma->check_boxes[last_graph], FALSE);
61}

63static void
64on_properties_dialog_response (GtkWidget *widget,
65 gint arg,
66 MultiloadApplet *ma)
67{
68 GError *error = NULL;
69
70 switch (arg) {
71 case GTK_RESPONSE_HELP:
72
73 gtk_show_uri_on_window (NULL,
74 "help:mate-multiload/multiload-prefs",
75 gtk_get_current_event_time (),
76 &error);
77
78 if (error) { /* FIXME: the user needs to see this */
79 g_warning ("help error: %s\n", error->message);
80 g_error_free (error);
81 }
82 break;
83
84 case GTK_RESPONSE_CLOSE:
85 default:
86 gtk_widget_destroy (widget);
87 ma->prop_dialog = NULL;
88 }
89}

91static void
92on_speed_spin_button_value_changed (GtkSpinButton *spin_button,
93 gpointer user_data)
94{
95 MultiloadApplet *ma = user_data;
96 gint value;
97 guint i;
98
99 value = gtk_spin_button_get_value_as_int (spin_button);
100 g_settings_set_uint (ma->settings, REFRESH_RATE_KEY, (guint) value);
101 for (i = 0; i < graph_n; i++) {
102 load_graph_stop (ma->graphs[i]);
103 ma->graphs[i]->speed = (guint) value;
104 if (ma->graphs[i]->visible)
105 load_graph_start (ma->graphs[i]);
106 }
107}

414static void
415graph_set_active (MultiloadApplet *ma,
416 LoadGraph *graph,
417 gboolean active)
418{
419 graph->visible = active;
420 if (active) {
421 guint i;
422
423 for (i = 0; i < graph_n; i++)
424 soft_set_sensitive(ma->check_boxes[i], TRUE);
425 gtk_widget_show_all (graph->main_widget);
426 load_graph_start (graph);
427 } else {
428 load_graph_stop (graph);
429 gtk_widget_hide (graph->main_widget);
430 properties_set_insensitive (ma);
431 }
432}

518static void
519read_color_button (GtkWidget *widget,
520 GSettings *settings,
521 const char *key)
522{
523 GdkRGBA color;
524 gchar *color_string;
525
526 color_string = g_settings_get_string (settings, key);
527 if (*color_string != '\0') {
528 gdk_rgba_parse (&color, color_string);
529 } else {
530 gdk_rgba_parse (&color, "#000000");
531 }
532 g_free (color_string);
533 gtk_color_chooser_set_rgba (GTK_COLOR_CHOOSER (widget), &color);
534
535 if (!g_settings_is_writable (settings, key))
536 hard_set_sensitive (widget, FALSE);
537}

543{
544 GtkBuilder *builder;
545 GtkWidget *dialog = NULL;
546 GtkWidget *graph_size_spin_button_label;
