77static GdkColor*
78color_shade (GdkColor *a,
79 gdouble shade,
80 GdkColor *b)
81{
82 guint16 red, green, blue;
83
84 red = CLAMP ((a->red) * shade, 0, 0xFFFF);
85 green = CLAMP ((a->green) * shade, 0, 0xFFFF);
86 blue = CLAMP ((a->blue) * shade, 0, 0xFFFF);
87
88 b->red = red;
89 b->green = green;
90 b->blue = blue;
91
92 return b;
93}

148static GSList*
149scan_ad_directory (const char *path,
150 GError **error)
151{
152 GSList *list;
153 GDir *dir;
154 const char *entry;
155 GError *local_error;
156
157 list = NULL;
158
159 g_return_val_if_fail (path != NULL, NULL);
160
161 local_error = NULL;
162 dir = g_dir_open (path, 0, &local_error);
163 if (local_error != NULL) {
164 g_propagate_error (error, local_error);
165 return NULL;
166 }
167
168 while ((entry = g_dir_read_name (dir)) != NULL) {
169 if (g_str_has_suffix (entry, ".ad")) {
170 list = g_slist_prepend (list, g_strdup_printf ("%s/%s", path, entry));
171 }
172 }
173
174 g_dir_close (dir);
175
176 /* TODO: sort still? */
177 return g_slist_sort (list, (GCompareFunc)strcmp);
178}

183static gint
184compare_basenames (gconstpointer a,
185 gconstpointer b)
186{
187 char *base_a;
188 char *base_b;
189 int res;
190
191 base_a = g_path_get_basename (a);
192 base_b = g_path_get_basename (b);
193 res = strcmp (base_a, base_b);
194 g_free (base_a);
195 g_free (base_b);
196
197 return res;
198}

204static GSList*
205scan_for_files (MsdXrdbManager *manager,
206 GError **error)
207{
208 const char *home_dir;
209 GSList *user_list;
210 GSList *system_list;
211 GSList *list;
212 GSList *p;
213 GError *local_error;
214
215 list = NULL;
216 user_list = NULL;
217 system_list = NULL;
218
219 local_error = NULL;
220 system_list = scan_ad_directory (SYSTEM_AD_DIR, &local_error);
221 if (local_error != NULL) {
222 g_propagate_error (error, local_error);
223 return NULL;
224 }
225
226 home_dir = g_get_home_dir ();
227 if (home_dir != NULL) {
228 char *user_ad;
229
230 user_ad = g_build_filename (home_dir, USER_AD_DIR, NULL);
231
232 if (g_file_test (user_ad, G_FILE_TEST_IS_DIR)) {
233
234 local_error = NULL;
235 user_list = scan_ad_directory (user_ad, &local_error);
236 if (local_error != NULL) {
237 g_propagate_error (error, local_error);
238
239 g_slist_free_full (system_list, g_free);
240 g_free (user_ad);
241 return NULL;
242 }
243 }
244
245 g_free (user_ad);
246
247 } else {
248 g_warning (_("Cannot determine user's home directory"));
249 }
250
251 /* An alternative approach would be to strdup() the strings
252 and free the entire contents of these lists, but that is a
253 little inefficient for my liking - RB */
254 for (p = system_list; p != NULL; p = g_slist_next (p)) {
255 if (strcmp (p->data, GENERAL_AD) == 0) {
256 /* We ignore this, free the data now */
257 g_free (p->data);
258 continue;
259 }
260 if (g_slist_find_custom (user_list, p->data, compare_basenames)) {
261 /* Ditto */
262 g_free (p->data);
263 continue;
264 }
265 list = g_slist_prepend (list, p->data);
266 }
267 g_slist_free (system_list);
268
269 for (p = user_list; p != NULL; p = g_slist_next (p)) {
270 list = g_slist_prepend (list, p->data);
271 }
272 g_slist_free (user_list);
273
274 /* Reverse the order so it is the correct way */
275 list = g_slist_reverse (list);
276
277 /* Add the initial file */
278 list = g_slist_prepend (list, g_strdup (GENERAL_AD));
279
280 return list;
281}

305static void
306append_xresource_file (const char *filename,
307 GString *string,
308 GError **error)
309{
310 const char *home_path;
311 char *xresources;
312
313 g_return_if_fail (string != NULL);
314
315 home_path = g_get_home_dir ();
316 if (home_path == NULL) {
317 g_warning (_("Cannot determine user's home directory"));
318 return;
319 }
320
321 xresources = g_build_filename (home_path, filename, NULL);
322 if (g_file_test (xresources, G_FILE_TEST_EXISTS)) {
323 GError *local_error;
324
325 local_error = NULL;
326
327 append_file (xresources, string, &local_error);
328 if (local_error != NULL) {
329 g_warning ("%s", local_error->message);
330 g_propagate_error (error, local_error);
331 }
332 }
333 g_free (xresources);
334}

336static gboolean
337write_all (int fd,
338 const char *buf,
339 gsize to_write)
340{
341 while (to_write > 0) {
342 gssize count = write (fd, buf, to_write);
343 if (count < 0) {
344 if (errno != EINTR)
345 return FALSE;
346 } else {
347 to_write -= count;
348 buf += count;
349 }
350 }
351
352 return TRUE;
353}

367static void
368spawn_with_input (const char *command,
369 const char *input)
370{
371 char **argv;
372 int child_pid;
373 int inpipe;
374 GError *error;
375 gboolean res;
376
377 argv = NULL;
378 res = g_shell_parse_argv (command, NULL, &argv, NULL);
379 if (! res) {
380 g_warning ("Unable to parse command: %s", command);
381 return;
382 }
383
384 error = NULL;
385 res = g_spawn_async_with_pipes (NULL,
386 argv,
387 NULL,
388 G_SPAWN_SEARCH_PATH | G_SPAWN_DO_NOT_REAP_CHILD,
389 NULL,
390 NULL,
391 &child_pid,
392 &inpipe,
393 NULL,
394 NULL,
395 &error);
396 g_strfreev (argv);
397
398 if (! res) {
399 g_warning ("Could not execute %s: %s", command, error->message);
400 g_error_free (error);
401
402 return;
403 }
404
405 if (input != NULL) {
406 if (! write_all (inpipe, input, strlen (input))) {
407 g_warning ("Could not write input to %s", command);
408 }
409
410 close (inpipe);
411 }
412
413 g_child_watch_add (child_pid, (GChildWatchFunc) child_watch_cb, (gpointer)command);
414}

416static void
417apply_settings (MsdXrdbManager *manager,
418 GtkStyle *style)
419{
420 const char *command;
421 GString *string;
422 GSList *list;
423 GSList *p;
424 GError *error;
425
426 mate_settings_profile_start (NULL);
427
428 command = "xrdb -merge -quiet";
429
430 string = g_string_sized_new (256);
431 append_theme_colors (style, string);
432
433 error = NULL;
434 list = scan_for_files (manager, &error);
435 if (error != NULL) {
436 g_warning ("%s", error->message);
437 g_error_free (error);
438 }
439
440 for (p = list; p != NULL; p = p->next) {
441 error = NULL;
442 append_file (p->data, string, &error);
443 if (error != NULL) {
444 g_warning ("%s", error->message);
445 g_error_free (error);
446 }
447 }
448
449 g_slist_free_full (list, g_free);
450
451 error = NULL;
452 append_xresource_file (USER_X_RESOURCES, string, &error);
453 if (error != NULL) {
454 g_warning ("%s", error->message);
455 g_error_free (error);
456 }
457
458 error = NULL;
459 append_xresource_file (USER_X_DEFAULTS, string, &error);
460 if (error != NULL) {
461 g_warning ("%s", error->message);
462 g_error_free (error);
463 }
464
465 spawn_with_input (command, string->str);
466 g_string_free (string, TRUE);
467
468 mate_settings_profile_end (NULL);
469
470 return;
471}

484{
