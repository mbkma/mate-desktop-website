58static void
59mate_panel_applet_factory_finalize (GObject *object)
60{
61 MatePanelAppletFactory *factory = MATE_PANEL_APPLET_FACTORY (object);
62
63 if (factory->registration_id) {
64 g_dbus_connection_unregister_object (factory->connection,
65 factory->registration_id);
66 factory->registration_id = 0;
67 }
68
69 if (factory->owner_id) {
70 g_bus_unown_name (factory->owner_id);
71 factory->owner_id = 0;
72 }
73
74 g_hash_table_remove (factories, factory->factory_id);
75
76 if (g_hash_table_size (factories) == 0) {
77 g_hash_table_unref (factories);
78 factories = NULL;
79 }
80
81 g_clear_pointer (&factory->factory_id, g_free);
82
83 if (factory->applets) {
84 g_hash_table_unref (factory->applets);
85 factory->applets = NULL;
86 }
87
88 if (factory->closure) {
89 g_closure_unref (factory->closure);
90 factory->closure = NULL;
91 }
92
93 G_OBJECT_CLASS (mate_panel_applet_factory_parent_class)
94 ->finalize (object);
95}

127MatePanelAppletFactory *
128mate_panel_applet_factory_new (const gchar *factory_id,
129 gboolean out_of_process,
130 GType applet_type,
131 GClosure *closure)
132{
133 MatePanelAppletFactory *factory;
134
135 factory = MATE_PANEL_APPLET_FACTORY (g_object_new (PANEL_TYPE_APPLET_FACTORY, NULL));
136 factory->factory_id = g_strdup (factory_id);
137 factory->out_of_process = out_of_process;
138 factory->applet_type = applet_type;
139 factory->closure = g_closure_ref (closure);
140
141 if (factories == NULL)
142 factories = g_hash_table_new (g_str_hash, g_str_equal);
143
144 g_hash_table_insert (factories, factory->factory_id, factory);
145
146 return factory;
147}

149static void
150set_applet_constructor_properties (GObject *applet,
151 GVariant *props)
152{
153 GVariantIter iter;
154 GVariant *value;
155 gchar *key;
156
157 g_variant_iter_init (&iter, props);
158 while (g_variant_iter_loop (&iter, "{sv}", &key, &value)) {
159 switch (g_variant_classify (value)) {
160 case G_VARIANT_CLASS_UINT32: {
161 guint32 v = g_variant_get_uint32 (value);
162 g_object_set (applet, key, v, NULL);
163 break;
164 } case G_VARIANT_CLASS_STRING: {
165 const gchar *v = g_variant_get_string (value, NULL);
166 g_object_set (applet, key, v, NULL);
167 break;
168 } case G_VARIANT_CLASS_BOOLEAN: {
169 gboolean v = g_variant_get_boolean (value);
170 g_object_set (applet, key, v, NULL);
171 break;
172 } default:
173 g_assert_not_reached ();
174 }
175 }

178static void
179mate_panel_applet_factory_get_applet (MatePanelAppletFactory *factory,
180 GDBusConnection *connection,
181 GVariant *parameters,
182 GDBusMethodInvocation *invocation)
183{
184 GObject *applet;
185 const gchar *applet_id;
186 gint screen_num;
187 GVariant *props;
188 GVariant *return_value;
189 guint32 xid;
190 guint32 uid;
191 const gchar *object_path;
192
193 g_variant_get (parameters, "(&si@a{sv})", &applet_id, &screen_num, &props);
194
195 applet = g_object_new (factory->applet_type,
196 "out-of-process", factory->out_of_process,
197 "id", applet_id,
198 "connection", connection,
199 "closure", factory->closure,
200 NULL);
201 factory->n_applets++;
202 g_object_weak_ref (applet,
203 (GWeakNotify) mate_panel_applet_factory_applet_removed,
204 factory);
205
206 set_applet_constructor_properties (applet, props);
207 g_variant_unref (props);
208
209#ifdef HAVE_X11
210 if (GDK_IS_X11_DISPLAY (gdk_display_get_default ())) {
211 GdkScreen *screen;
212
213 screen = screen_num != -1 ?
214 gdk_display_get_default_screen (gdk_display_get_default ()) :
215 gdk_screen_get_default ();
216 xid = mate_panel_applet_get_xid (MATE_PANEL_APPLET (applet), screen);
217 } else
218#endif /* HAVE_X11 */
219 { /* Not using X11 */
220 xid = 0;
221 }
222
223 uid = factory->next_uid++;
224 object_path = mate_panel_applet_get_object_path (MATE_PANEL_APPLET (applet));
225 g_hash_table_insert (factory->applets, GUINT_TO_POINTER (uid), applet);
226 g_object_set_data (applet, "uid", GUINT_TO_POINTER (uid));
227
228 return_value = g_variant_new ("(obuu)",
229 object_path,
230 factory->out_of_process,
231 xid,
232 uid);
233
234 g_dbus_method_invocation_return_value (invocation, return_value);
235}

278static void
279on_bus_acquired (GDBusConnection *connection,
280 const gchar *name,
281 MatePanelAppletFactory *factory)
282{
283 gchar *object_path;
284 GError *error = NULL;
285
286 if (!introspection_data)
287 introspection_data
288 = g_dbus_node_info_new_for_xml (introspection_xml, NULL);
289
290 object_path = g_strdup_printf (MATE_PANEL_APPLET_FACTORY_OBJECT_PATH,
291 factory->factory_id);
292
293 factory->connection = connection;
294 factory->registration_id
295 = g_dbus_connection_register_object (connection,
296 object_path,
297 introspection_data->interfaces[0],
298 &interface_vtable,
299 factory, NULL,
300 &error);
301 if (error) {
302 g_printerr ("Failed to register object %s: %s\n",
303 object_path, error->message);
304 g_error_free (error);
305 }
306
307 g_free (object_path);
308}

318gboolean
319mate_panel_applet_factory_register_service (MatePanelAppletFactory *factory)
320{
321 gchar *service_name;
322
323 if (!factory)
324 return FALSE;
325
326 service_name = g_strdup_printf (MATE_PANEL_APPLET_FACTORY_SERVICE_NAME,
327 factory->factory_id);
328
329 factory->owner_id = g_bus_own_name (G_BUS_TYPE_SESSION,
330 service_name,
331 G_BUS_NAME_OWNER_FLAGS_NONE,
332 (GBusAcquiredCallback) on_bus_acquired,
333 NULL,
334 (GBusNameLostCallback) on_name_lost,
335 factory, NULL);
336 g_free (service_name);
337
338 return TRUE;
339}

341GtkWidget *
342mate_panel_applet_factory_get_applet_widget (const gchar *id,
343 guint uid)
344{
345 MatePanelAppletFactory *factory;
346 GObject *object;
347
348 if (!factories)
349 return NULL;
350
351 factory = g_hash_table_lookup (factories, id);
352 if (!factory)
353 return NULL;
354
355 object = g_hash_table_lookup (factory->applets, GUINT_TO_POINTER (uid));
356 if (!object || !GTK_IS_WIDGET (object))
357 return NULL;
358
359 return GTK_WIDGET (object);
360}
