48gboolean
49dbus_context_connect (MateNotificationDBusContext *context)
50{
51 GError *error = NULL;
52
53 if (!context)
54 return FALSE;
55
56 /* Clean up existing connection */
57 dbus_context_disconnect (context);
58
59 /* Create D-Bus proxy for notification daemon */
60 context->daemon_proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SESSION,
61 G_DBUS_PROXY_FLAGS_NONE,
62 NULL, /* GDBusInterfaceInfo */
63 NOTIFICATION_BUS_NAME,
64 NOTIFICATION_BUS_PATH,
65 "org.freedesktop.Notifications",
66 NULL, /* GCancellable */
67 &error);
68
69 if (error) {
70 g_warning ("Failed to connect to notification daemon: %s", error->message);
71 g_error_free (error);
72 context->daemon_available = FALSE;
73 return FALSE;
74 }
75
76 context->daemon_available = TRUE;
77 return TRUE;
78}

80void
81dbus_context_disconnect (MateNotificationDBusContext *context)
82{
83 if (!context)
84 return;
85
86 if (context->daemon_proxy) {
87 g_object_unref (context->daemon_proxy);
88 context->daemon_proxy = NULL;
89 }
90
91 context->daemon_available = FALSE;
92}

100guint
101dbus_context_get_notification_count (MateNotificationDBusContext *context)
102{
103 GVariant *result;
104 GError *error = NULL;
105 guint count = 0;
106
107 if (!dbus_context_is_available (context))
108 return 0;
109
110 result = g_dbus_proxy_call_sync (context->daemon_proxy,
111 "GetNotificationCount",
112 NULL,
113 G_DBUS_CALL_FLAGS_NONE,
114 -1, /* timeout */
115 NULL, /* GCancellable */
116 &error);
117
118 if (error) {
119 g_warning ("Failed to get notification count: %s", error->message);
120 g_error_free (error);
121 context->daemon_available = FALSE;
122 } else if (result) {
123 g_variant_get (result, "(u)", &count);
124 g_variant_unref (result);
125 }
126
127 return count;
128}
