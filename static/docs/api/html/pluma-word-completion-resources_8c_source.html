286#define G_MSVC_CTOR(_func,_sym_prefix) \
287 static void _func(void); \
288 extern int (* _array ## _func)(void); \
289 int _func ## _wrapper(void); \
290 int _func ## _wrapper(void) { _func(); g_slist_find (NULL, _array ## _func); return 0; } \

299 int _func ## _constructor(void) { atexit (_func); g_slist_find (NULL, _array ## _func); return 0; } \
