115PulseSource *
116pulse_source_new (PulseConnection *connection,
117 const pa_source_info *info,
118 PulseDevice *device)
119{
120 PulseSource *source;
121
122 g_return_val_if_fail (PULSE_IS_CONNECTION (connection), NULL);
123 g_return_val_if_fail (info != NULL, NULL);
124 g_return_val_if_fail (device == NULL || PULSE_IS_DEVICE (device), NULL);
125
126 source = g_object_new (PULSE_TYPE_SOURCE,
127 "name", info->name,
128 "label", info->description,
129 "device", device,
130 "direction", MATE_MIXER_DIRECTION_INPUT,
131 "connection", connection,
132 "index", info->index,
133 NULL);
134
135 source->priv->control = pulse_source_control_new (connection, info, source);
136
137 if (info->n_ports > 0) {
138 pa_source_port_info **ports = info->ports;
139
140 /* Create the port switch */
141 source->priv->pswitch = pulse_source_switch_new ("port", _("Connector"), source);
142
143 while (*ports != NULL) {
144 pa_source_port_info *p = *ports++;
145 PulsePort *port;
146 const gchar *icon = NULL;
147
148 /* A port may include an icon but in PulseAudio sink and source ports
149 * the property is not included, for this reason ports are also read from
150 * devices where the icons may be present */
151 if (device != NULL) {
152 port = pulse_device_get_port (PULSE_DEVICE (device), p->name);
153 if (port != NULL)
154 icon = mate_mixer_switch_option_get_icon (MATE_MIXER_SWITCH_OPTION (port));
155 }
156
157 port = pulse_port_new (p->name,
158 p->description,
159 icon,
160 p->priority);
161
162 pulse_port_switch_add_port (source->priv->pswitch, port);
163
164 if (p == info->active_port)
165 pulse_port_switch_set_active_port (source->priv->pswitch, port);
166 }
167 source->priv->pswitch_list = g_list_prepend (NULL, source->priv->pswitch);
168
169 g_debug ("Created port list for source %s", info->name);
170 }
171
172 pulse_source_update (source, info);
173
174 _mate_mixer_stream_set_default_control (MATE_MIXER_STREAM (source),
175 MATE_MIXER_STREAM_CONTROL (source->priv->control));
176 return source;
177}

179gboolean
180pulse_source_add_output (PulseSource *source, const pa_source_output_info *info)
181{
182 PulseSourceOutput *output;
183
184 g_return_val_if_fail (PULSE_IS_SOURCE (source), FALSE);
185 g_return_val_if_fail (info != NULL, FALSE);
186
187 /* This function is used for both creating and refreshing source outputs */
188 output = g_hash_table_lookup (source->priv->outputs, GUINT_TO_POINTER (info->index));
189 if (output == NULL) {
190 const gchar *name;
191 PulseConnection *connection;
192
193 connection = pulse_stream_get_connection (PULSE_STREAM (source));
194 output = pulse_source_output_new (connection,
195 info,
196 source);
197 g_hash_table_insert (source->priv->outputs,
198 GUINT_TO_POINTER (info->index),
199 output);
200
201 free_list_controls (source);
202
203 name = mate_mixer_stream_control_get_name (MATE_MIXER_STREAM_CONTROL (output));
204 g_signal_emit_by_name (G_OBJECT (source),
205 "control-added",
206 name);
207 return TRUE;
208 }
209
210 pulse_source_output_update (output, info);
211 return FALSE;
212}

214void
215pulse_source_remove_output (PulseSource *source, guint32 index)
216{
217 PulseSourceOutput *output;
218 gchar *name;
219
220 g_return_if_fail (PULSE_IS_SOURCE (source));
221
222 output = g_hash_table_lookup (source->priv->outputs, GUINT_TO_POINTER (index));
223 if (G_UNLIKELY (output == NULL))
224 return;
225
226 name = g_strdup (mate_mixer_stream_control_get_name (MATE_MIXER_STREAM_CONTROL (output)));
227
228 g_hash_table_remove (source->priv->outputs, GUINT_TO_POINTER (index));
229
230 free_list_controls (source);
231 g_signal_emit_by_name (G_OBJECT (source),
232 "control-removed",
233 name);
234 g_free (name);
235}

237void
238pulse_source_update (PulseSource *source,
239 const pa_source_info *info)
240{
241 g_return_if_fail (PULSE_IS_SOURCE (source));
242 g_return_if_fail (info != NULL);
243
244 /* The switch doesn't allow being unset, PulseAudio should always include
245 * the active port name if the are any ports available */
246 if (info->active_port != NULL)
247 pulse_port_switch_set_active_port_by_name (source->priv->pswitch,
248 info->active_port->name);
249
250 pulse_source_control_update (source->priv->control, info);
251}

253static const GList *
254pulse_source_list_controls (MateMixerStream *mms)
255{
256 PulseSource *source;
257
258 g_return_val_if_fail (PULSE_IS_SOURCE (mms), NULL);
259
260 source = PULSE_SOURCE (mms);
261
262 if (source->priv->outputs_list == NULL) {
263 source->priv->outputs_list = g_hash_table_get_values (source->priv->outputs);
264 if (source->priv->outputs_list != NULL)
265 g_list_foreach (source->priv->outputs_list, (GFunc) g_object_ref, NULL);
266
267 source->priv->outputs_list = g_list_prepend (source->priv->outputs_list,
268 g_object_ref (source->priv->control));
269 }
270 return source->priv->outputs_list;
271}
