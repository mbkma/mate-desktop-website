33/*
34 * NOTES:
35 *
36 * OSS has a predefined list of channels (or "devices"), which historically used
37 * to be mapped to individual sound card pins. At this time, the channels are
38 * chosen somehow arbitrarily by drivers.
39 *
40 * Each of the channels may have a record switch, which toggles between playback
41 * and capture direction. OSS doesn't have mute switches and we can't really use
42 * the record switch as one. For this reason all channels are modelled as
43 * muteless stream controls and the record switch is modelled as a port switch.
44 *
45 * Also, we avoid modelling capturable channels as both input and output channels,
46 * because the ones which allow capture are normally capture-only channels
47 * (OSS just doesn't have the ability to make the distinction), so each channel in
48 * the list contains a flag of whether it can be used as a capture source, given
49 * that it's reported as capturable. Capturable channels are therefore modelled
50 * as input controls and this approach avoids for example putting PCM in an input
51 * stream (which makes no sense).
52 *
53 * OSS also has an indicator of whether the record switch is exclusive (only
54 * allows one capture source at a time), to simplify the lives of applications
55 * we always create a port switch and therefore assume the exclusivity is always
56 * true. Ideally, we should probably model a bunch of toggles, one for each channel
57 * with capture capability if they are known not to be exclusive.
58 */

266OssDevice *
267oss_device_new (const gchar *name,
268 const gchar *label,
269 const gchar *path,
270 gint fd)
271{
272 OssDevice *device;
273 gint newfd;
274
275 g_return_val_if_fail (name != NULL, NULL);
276 g_return_val_if_fail (label != NULL, NULL);
277 g_return_val_if_fail (path != NULL, NULL);
278
279 newfd = dup (fd);
280 if (newfd == -1) {
281 g_warning ("Failed to duplicate file descriptor: %s",
282 g_strerror (errno));
283 return NULL;
284 }
285
286 device = g_object_new (OSS_TYPE_DEVICE,
287 "name", name,
288 "label", label,
289 "icon", OSS_DEVICE_ICON,
290 NULL);
291
292 device->priv->fd = newfd;
293 device->priv->path = g_strdup (path);
294
295 return device;
296}

298gboolean
299oss_device_open (OssDevice *device)
300{
301 gint ret;
302
303 g_return_val_if_fail (OSS_IS_DEVICE (device), FALSE);
304
305 g_debug ("Opening device %s (%s)",
306 device->priv->path,
307 mate_mixer_device_get_label (MATE_MIXER_DEVICE (device)));
308
309 /* Read the essential information about the device, these values are not
310 * expected to change and will not be queried */
311 ret = ioctl (device->priv->fd, MIXER_READ (SOUND_MIXER_DEVMASK),
312 &device->priv->devmask);
313 if (ret == -1)
314 goto fail;
315
316 ret = ioctl (device->priv->fd, MIXER_READ (SOUND_MIXER_STEREODEVS),
317 &device->priv->stereodevs);
318 if (ret == -1)
319 goto fail;
320
321 ret = ioctl (device->priv->fd, MIXER_READ (SOUND_MIXER_RECMASK),
322 &device->priv->recmask);
323 if (ret == -1)
324 goto fail;
325
326 /* NOTE: Linux also supports SOUND_MIXER_OUTSRC and SOUND_MIXER_OUTMASK which
327 * inform about/enable input->output, we could potentially create toggles
328 * for these, but these constants are not defined on any BSD. */
329
330 return TRUE;
331
332fail:
333 g_warning ("Failed to read device %s: %s",
334 device->priv->path,
335 g_strerror (errno));
336
337 return FALSE;
338}

351void
352oss_device_close (OssDevice *device)
353{
354 g_return_if_fail (OSS_IS_DEVICE (device));
355
356 if (device->priv->fd == -1)
357 return;
358
359 /* Make each stream remove its controls and switch */
360 if (device->priv->input != NULL) {
361 const gchar *name =
362 mate_mixer_stream_get_name (MATE_MIXER_STREAM (device->priv->input));
363
364 oss_stream_remove_all (device->priv->input);
365 free_stream_list (device);
366
367 g_signal_emit_by_name (G_OBJECT (device),
368 "stream-removed",
369 name);
370
371 g_clear_object (&device->priv->input);
372 }
373
374 if (device->priv->output != NULL) {
375 const gchar *name =
376 mate_mixer_stream_get_name (MATE_MIXER_STREAM (device->priv->output));
377
378 oss_stream_remove_all (device->priv->output);
379 free_stream_list (device);
380
381 g_signal_emit_by_name (G_OBJECT (device),
382 "stream-removed",
383 name);
384
385 g_clear_object (&device->priv->output);
386 }
387
388 if (device->priv->poll_tag != 0)
389 g_source_remove (device->priv->poll_tag);
390
391 if (device->priv->poll_tag_restore != 0)
392 g_source_remove (device->priv->poll_tag_restore);
393
394 close (device->priv->fd);
395 device->priv->fd = -1;
396
397 g_signal_emit (G_OBJECT (device), signals[CLOSED], 0);
398}

400void
401oss_device_load (OssDevice *device)
402{
403 const GList *controls;
404 const gchar *name;
405 gchar *stream_name;
406 guint i;
407
408 g_return_if_fail (OSS_IS_DEVICE (device));
409
410 name = mate_mixer_device_get_name (MATE_MIXER_DEVICE (device));
411
412 stream_name = g_strdup_printf ("oss-input-%s", name);
413 device->priv->input = oss_stream_new (stream_name,
414 MATE_MIXER_DEVICE (device),
415 MATE_MIXER_DIRECTION_INPUT);
416 g_free (stream_name);
417
418 stream_name = g_strdup_printf ("oss-output-%s", name);
419 device->priv->output = oss_stream_new (stream_name,
420 MATE_MIXER_DEVICE (device),
421 MATE_MIXER_DIRECTION_OUTPUT);
422 g_free (stream_name);
423
424 read_mixer_devices (device);
425
426 /* Set default input control */
427 if (oss_stream_has_controls (device->priv->input) == TRUE) {
428 controls = mate_mixer_stream_list_controls (MATE_MIXER_STREAM (device->priv->input));
429
430 for (i = 0; i < G_N_ELEMENTS (oss_input_priority); i++) {
431 GList *item = g_list_find_custom ((GList *) controls,
432 GINT_TO_POINTER (oss_input_priority[i]),
433 compare_stream_control_devnum);
434 if (item == NULL)
435 continue;
436
437 oss_stream_set_default_control (device->priv->input,
438 OSS_STREAM_CONTROL (item->data));
439 break;
440 }
441
442 /* Create an input switch for recording sources */
443 if (device->priv->recmask != 0)
444 read_mixer_switch (device);
445 } else
446 g_clear_object (&device->priv->input);
447
448 /* Set default output control */
449 if (oss_stream_has_controls (device->priv->output) == TRUE) {
450 controls = mate_mixer_stream_list_controls (MATE_MIXER_STREAM (device->priv->output));
451
452 for (i = 0; i < G_N_ELEMENTS (oss_output_priority); i++) {
453 GList *item = g_list_find_custom ((GList *) controls,
454 GINT_TO_POINTER (oss_output_priority[i]),
455 compare_stream_control_devnum);
456 if (item == NULL)
457 continue;
458
459 oss_stream_set_default_control (device->priv->output,
460 OSS_STREAM_CONTROL (item->data));
461 break;
462 }
463 } else
464 g_clear_object (&device->priv->output);
465
466 /*
467 * See if we can use the modify_counter field to optimize polling.
468 *
469 * Only do this on Linux for now as the counter doesn't update on BSDs.
470 */
471#if defined(SOUND_MIXER_INFO) && defined(__linux__)
472 do {
473 struct mixer_info info;
474 gint ret;
475
476 ret = ioctl (device->priv->fd, SOUND_MIXER_INFO, &info);
477 if (ret == 0) {
478 device->priv->poll_counter = info.modify_counter;
479 device->priv->poll_use_counter = TRUE;
480 }
481 } while (0);
482#endif
483
484 /*
485 * Use a polling strategy inspired by KMix:
486 *
487 * Poll for changes with the OSS_POLL_TIMEOUT_NORMAL interval. When we
488 * encounter a change in modify_counter, decrease the interval to
489 * OSS_POLL_TIMEOUT_RAPID for a few seconds to allow for smoother
490 * adjustments, for example when user drags a slider.
491 *
492 * This is not used on systems which don't support the modify_counter
493 * field, because there is no way to find out whether anything has
494 * changed and therefore when to start the rapid polling.
495 */
496 device->priv->poll_tag = create_poll_source (device, OSS_POLL_NORMAL);
497}

544#define OSS_MASK_HAS_DEVICE(mask,i) ((gboolean) (((mask) & (1 << (i))) > 0))
545
546static void
547read_mixer_devices (OssDevice *device)
548{
549 OssStreamControl *control;
550 const gchar *name;
551 guint i;
552
553 name = mate_mixer_device_get_name (MATE_MIXER_DEVICE (device));
554
555 for (i = 0; i < OSS_N_DEVICES; i++) {
556 OssStream *stream;
557 gboolean stereo;
558
559 /* Skip unavailable controls */
560 if (OSS_MASK_HAS_DEVICE (device->priv->devmask, i) == FALSE)
561 continue;
562
563 /* The control is assigned to a stream according to the predefined type.
564 *
565 * OSS may allow some controls to be both input and output, but the API
566 * is too simple to tell what exactly is a control capable of. Here we
567 * simplify things a bit and assign each control to exactly one stream. */
568 switch (oss_devices[i].type) {
569 case OSS_DEV_INPUT:
570 stream = device->priv->input;
571 break;
572 case OSS_DEV_OUTPUT:
573 stream = device->priv->output;
574 break;
575 default:
576 if (OSS_MASK_HAS_DEVICE (device->priv->recmask, i) == TRUE)
577 stream = device->priv->input;
578 else
579 stream = device->priv->output;
580 break;
581 }
582
583 stereo = OSS_MASK_HAS_DEVICE (device->priv->stereodevs, i);
584 control = oss_stream_control_new (oss_devices[i].name,
585 gettext (oss_devices[i].label),
586 oss_devices[i].role,
587 stream,
588 device->priv->fd,
589 i,
590 stereo);
591 if (G_UNLIKELY (control == NULL))
592 continue;
593
594 if (oss_stream_has_controls (stream) == FALSE) {
595 const gchar *name =
596 mate_mixer_stream_get_name (MATE_MIXER_STREAM (stream));
597
598 free_stream_list (device);
599
600 /* Pretend the stream has just been created now that we are adding
601 * the first control */
602 g_signal_emit_by_name (G_OBJECT (device),
603 "stream-added",
604 name);
605 }
606
607 g_debug ("Adding device %s control %s",
608 name,
609 mate_mixer_stream_control_get_name (MATE_MIXER_STREAM_CONTROL (control)));
610
611 oss_stream_add_control (stream, control);
612 oss_stream_control_load (control);
613
614 g_object_unref (control);
615 }
616}

618static void
619read_mixer_switch (OssDevice *device)
620{
621 GList *options = NULL;
622 guint i;
623
624 /* This is always an input recording source switch */
625 for (i = 0; i < G_N_ELEMENTS (oss_input_priority); i++) {
626 gint devnum = oss_input_priority[i];
627 OssSwitchOption *option;
628
629 /* Avoid devices that are not present or not recordable */
630 if (OSS_MASK_HAS_DEVICE (device->priv->devmask, devnum) == FALSE ||
631 OSS_MASK_HAS_DEVICE (device->priv->recmask, devnum) == FALSE)
632 continue;
633
634 option = oss_switch_option_new (oss_devices[devnum].name,
635 gettext (oss_devices[devnum].label),
636 oss_devices[devnum].icon,
637 devnum);
638 options = g_list_prepend (options, option);
639 }
640
641 if (G_LIKELY (options != NULL))
642 oss_stream_set_switch_data (device->priv->input,
643 device->priv->fd,
644 g_list_reverse (options));
645}

647static gboolean
648poll_mixer (OssDevice *device)
649{
650 gboolean load = TRUE;
651
652 if (G_UNLIKELY (device->priv->fd == -1))
653 return G_SOURCE_REMOVE;
654
655#ifdef SOUND_MIXER_INFO
656 if (device->priv->poll_use_counter == TRUE) {
657 gint ret;
658 struct mixer_info info;
659
660 /*
661 * The modify_counter field increases each time a change occurs on
662 * the device.
663 *
664 * If this ioctl() works, we use the field to only poll the controls
665 * if a change actually occured and we can also adjust the poll interval.
666 *
667 * The call is also used to detect unplugged devices early.
668 */
669 ret = ioctl (device->priv->fd, SOUND_MIXER_INFO, &info);
670 if (ret == -1) {
671 if (errno == EINTR)
672 return G_SOURCE_CONTINUE;
673
674 oss_device_close (device);
675 return G_SOURCE_REMOVE;
676 }
677
678 if (device->priv->poll_counter < info.modify_counter)
679 device->priv->poll_counter = info.modify_counter;
680 else
681 load = FALSE;
682 }
683#endif
684
685 if (load == TRUE) {
686 if (device->priv->input != NULL)
687 oss_stream_load (device->priv->input);
688 if (device->priv->output != NULL)
689 oss_stream_load (device->priv->output);
690
691 if (device->priv->poll_use_counter == TRUE &&
692 device->priv->poll_mode == OSS_POLL_NORMAL) {
693 /* Create a new rapid source */
694 device->priv->poll_tag = create_poll_source (device, OSS_POLL_RAPID);
695
696 /* Also create another source to restore the poll interval to the
697 * original state */
698 device->priv->poll_tag_restore = create_poll_restore_source (device);
699

707static gboolean
708poll_mixer_restore (OssDevice *device)
709{
710 if (G_LIKELY (device->priv->poll_mode == OSS_POLL_RAPID)) {
711 /* Remove the current rapid source */
712 g_source_remove (device->priv->poll_tag);
713
714 device->priv->poll_tag = create_poll_source (device, OSS_POLL_NORMAL);
715 device->priv->poll_mode = OSS_POLL_NORMAL;
716 }
717
718 /* Remove the tag for this function as it is only called once, the tag
719 * is only kept so we can remove it in the case the device is closed */
720 device->priv->poll_tag_restore = 0;
721
722 return G_SOURCE_REMOVE;
723}

725static guint
726create_poll_source (OssDevice *device, OssPollMode mode)
727{
728 GSource *source;
729 guint timeout;
730 guint tag;
731
732 switch (mode) {
733 case OSS_POLL_NORMAL:
734 timeout = OSS_POLL_TIMEOUT_NORMAL;
735 break;
736 case OSS_POLL_RAPID:
737 timeout = OSS_POLL_TIMEOUT_RAPID;
738 break;
739 default:
740 g_warn_if_reached ();
741 return 0;
742 }
743
744 source = g_timeout_source_new (timeout);
745 g_source_set_callback (source,
746 (GSourceFunc) poll_mixer,
747 device,
748 NULL);
749
750 tag = g_source_attach (source, g_main_context_get_thread_default ());
751 g_source_unref (source);
752
753 return tag;
754}
