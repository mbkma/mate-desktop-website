44static GsmInhibitorFlags parse_flags (const gchar *arg)
45{
46 GsmInhibitorFlags flags;
47 gchar **args;
48 gint i;
49
50 flags = 0;
51
52 args = g_strsplit (arg, ":", 0);
53 for (i = 0; args[i]; i++)
54 {
55 if (strcmp (args[i], "logout") == 0)
56 flags |= GSM_INHIBITOR_FLAG_LOGOUT;
57 else if (strcmp (args[i], "switch-user") == 0)
58 flags |= GSM_INHIBITOR_FLAG_SWITCH_USER;
59 else if (strcmp (args[i], "suspend") == 0)
60 flags |= GSM_INHIBITOR_FLAG_SUSPEND;
61 else if (strcmp (args[i], "idle") == 0)
62 flags |= GSM_INHIBITOR_FLAG_IDLE;
63 else if (strcmp (args[i], "automount") == 0)
64 flags |= GSM_INHIBITOR_FLAG_AUTOMOUNT;
65 else
66 g_print ("Ignoring inhibit argument: %s\n", args[i]);
67 }
68
69 g_strfreev (args);
70
71 return flags;
72}

74static gboolean inhibit (const gchar *app_id,
75 const gchar *reason,
76 GsmInhibitorFlags flags)
77{
78 GDBusConnection *bus;
79 GVariant *ret;
80 GError *error = NULL;
81
82 bus = g_bus_get_sync (G_BUS_TYPE_SESSION, NULL, &error);
83
84 if (bus == NULL)
85 {
86 g_warning ("Failed to connect to session bus: %s", error->message);
87 g_error_free (error);
88 return FALSE;
89 }
90
91 ret = g_dbus_connection_call_sync (bus,
92 "org.gnome.SessionManager",
93 "/org/gnome/SessionManager",
94 "org.gnome.SessionManager",
95 "Inhibit",
96 g_variant_new ("(susu)",
97 app_id, 0, reason, flags),
98 G_VARIANT_TYPE ("(u)"),
99 0,
100 G_MAXINT,
101 NULL,
102 &error);
103
104 if (ret == NULL)
105 {
106 g_warning ("Failed to call Inhibit: %s\n", error->message);
107 g_error_free (error);
108 return FALSE;
109 }
110
111 g_variant_unref (ret);
112
113 return TRUE;
114}
