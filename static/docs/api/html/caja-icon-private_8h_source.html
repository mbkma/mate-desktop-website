34/* An Icon. */
35
36typedef struct
37{
38 /* Object represented by this icon. */
39 CajaIconData *data;
40
41 /* Canvas item for the icon. */
42 CajaIconCanvasItem *item;
43
44 /* X/Y coordinates. */
45 double x, y;
46
47 /*
48 * In RTL mode x is RTL x position, we use saved_ltr_x for
49 * keeping track of x value before it gets converted into
50 * RTL value, this is used for saving the icon position
51 * to the caja metafile.
52 */
53 double saved_ltr_x;
54
55 /* Scale factor (stretches icon). */
56 double scale;
57
58 /* Whether this item is selected. */
59 eel_boolean_bit is_selected : 1;
60
61 /* Whether this item was selected before rubberbanding. */
62 eel_boolean_bit was_selected_before_rubberband : 1;
63
64 /* Whether this item is visible in the view. */
65 eel_boolean_bit is_visible : 1;
66
67 /* Whether a monitor was set on this icon. */
68 eel_boolean_bit is_monitored : 1;
69
70 eel_boolean_bit has_lazy_position : 1;
71} CajaIcon;
