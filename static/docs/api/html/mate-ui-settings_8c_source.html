181/* Helper for combo box string binding */
182typedef struct
183{
184 gint column;
185} ComboBoxBindingData;
186
187static gboolean
188combo_box_get_mapping(GValue *value,
189 GVariant *variant,
190 gpointer user_data)
191{
192 ComboBoxBindingData *data = user_data;
193 const gchar *str = g_variant_get_string(variant, NULL);
194
195 GtkComboBox *combo = GTK_COMBO_BOX(g_value_get_object(value));
196 GtkTreeModel *model = gtk_combo_box_get_model(combo);
197 GtkTreeIter iter;
198 gboolean valid;
199
200 valid = gtk_tree_model_get_iter_first(model, &iter);
201 gint index = 0;
202
203 while (valid)
204 {
205 gchar *item_str = NULL;
206 gtk_tree_model_get(model, &iter, data->column, &item_str, -1);
207
208 if (g_strcmp0(item_str, str) == 0)
209 {
210 g_free(item_str);
211 g_value_set_int(value, index);
212 return TRUE;
213 }
214
215 g_free(item_str);
216 valid = gtk_tree_model_iter_next(model, &iter);
217 index++;
218 }
219
220 return FALSE;
221}
222
223static GVariant *
224combo_box_set_mapping(const GValue *value,
225 const GVariantType *expected_type G_GNUC_UNUSED,
226 gpointer user_data)
227{
228 ComboBoxBindingData *data = user_data;
229 GtkComboBox *combo = GTK_COMBO_BOX(g_value_get_object(value));
230 GtkTreeModel *model = gtk_combo_box_get_model(combo);
231 GtkTreeIter iter;
232
233 if (gtk_combo_box_get_active_iter(combo, &iter))
234 {
235 gchar *str = NULL;
236 gtk_tree_model_get(model, &iter, data->column, &str, -1);
237 GVariant *result = g_variant_new_string(str ? str : "");
238 g_free(str);
239 return result;
240 }
241
242 return g_variant_new_string("");
243}

254void
255mate_ui_settings_bind_combo_box(GSettings *settings,
256 const gchar *key,
257 GtkComboBox *combo_box,
258 gint model_column)
259{
260 g_return_if_fail(G_IS_SETTINGS(settings));
261 g_return_if_fail(key != NULL);
262 g_return_if_fail(GTK_IS_COMBO_BOX(combo_box));
263
264 /* Check if this is a simple integer index binding */
265 GVariant *value = g_settings_get_value(settings, key);
266 if (g_variant_is_of_type(value, G_VARIANT_TYPE_INT32) ||
267 g_variant_is_of_type(value, G_VARIANT_TYPE_UINT32))
268 {
269 g_variant_unref(value);
270 g_settings_bind(settings, key, combo_box, "active",
271 G_SETTINGS_BIND_DEFAULT);
272 return;
273 }
274 g_variant_unref(value);
275
276 /* For string values, use mapping */
277 ComboBoxBindingData *data = g_new0(ComboBoxBindingData, 1);
278 data->column = model_column;
279
280 g_settings_bind_with_mapping(settings, key, combo_box, "active",
281 G_SETTINGS_BIND_DEFAULT,
282 combo_box_get_mapping,
283 combo_box_set_mapping,
284 data, g_free);
285}

308/* Color button binding helpers */
309static gboolean
310color_get_mapping(GValue *value,
311 GVariant *variant,
312 gpointer user_data G_GNUC_UNUSED)
313{
314 const gchar *str = g_variant_get_string(variant, NULL);
315 GdkRGBA color;
316
317 if (gdk_rgba_parse(&color, str))
318 {
319 g_value_set_boxed(value, &color);
320 return TRUE;
321 }
322
323 return FALSE;
324}
325
326static GVariant *
327color_set_mapping(const GValue *value,
328 const GVariantType *expected_type G_GNUC_UNUSED,
329 gpointer user_data G_GNUC_UNUSED)
330{
331 GdkRGBA *color = g_value_get_boxed(value);
332 if (color != NULL)
333 {
334 gchar *str = gdk_rgba_to_string(color);
335 GVariant *result = g_variant_new_string(str);
336 g_free(str);
337 return result;
338 }
339
340 return g_variant_new_string("#000000");
341}

368static gboolean
369file_chooser_get_mapping(GValue *value,
370 GVariant *variant,
371 gpointer user_data G_GNUC_UNUSED)
372{
373 const gchar *path = g_variant_get_string(variant, NULL);
374 if (path != NULL && *path != '\0')
375 {
376 g_value_take_object(value, g_file_new_for_path(path));
377 return TRUE;
378 }
379 return FALSE;
380}
