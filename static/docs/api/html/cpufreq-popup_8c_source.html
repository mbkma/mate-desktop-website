176static void
177cpufreq_popup_frequencies_menu_activate (GtkAction *action,
178 CPUFreqPopup *popup)
179{
180 CPUFreqSelector *selector;
181 const gchar *name;
182 guint cpu;
183 guint freq;
184
185 if (!gtk_toggle_action_get_active (GTK_TOGGLE_ACTION (action)))
186 return;
187
188 selector = cpufreq_selector_get_default ();
189
190 cpu = cpufreq_monitor_get_cpu (popup->priv->monitor);
191 name = gtk_action_get_name (action);
192 freq = (guint) atoi (name + strlen ("Frequency"));
193
194 cpufreq_selector_set_frequency_async (selector, cpu, freq);
195}

218static void
219cpufreq_popup_menu_add_action (CPUFreqPopup *popup,
220 const gchar *menu,
221 GtkActionGroup *action_group,
222 const gchar *action_name,
223 const gchar *label,
224 gboolean sensitive)
225{
226 GtkToggleAction *action;
227 gchar *name;
228
229 name = g_strdup_printf ("%s%s", menu, action_name);
230
231 action = g_object_new (GTK_TYPE_RADIO_ACTION,
232 "name", name,
233 "label", label,
234 NULL);
235
236 gtk_action_set_sensitive (GTK_ACTION (action), sensitive);
237
238 gtk_radio_action_set_group (GTK_RADIO_ACTION (action), popup->priv->radio_group);
239 popup->priv->radio_group = gtk_radio_action_get_group (GTK_RADIO_ACTION (action));
240
241 if (g_ascii_strcasecmp (menu, "Frequency") == 0) {
242 popup->priv->freqs_actions = g_slist_prepend (popup->priv->freqs_actions,
243 (gpointer) action);
244
245 g_signal_connect (action, "activate",
246 G_CALLBACK (cpufreq_popup_frequencies_menu_activate),
247 popup);
248 } else if (g_ascii_strcasecmp (menu, "Governor") == 0) {
249 popup->priv->govs_actions = g_slist_prepend (popup->priv->govs_actions,
250 (gpointer) action);
251
252 g_signal_connect (action, "activate",
253 G_CALLBACK (cpufreq_popup_governors_menu_activate),
254 popup);
255 }
256
257 gtk_action_group_add_action (action_group, GTK_ACTION (action));
258 g_object_unref (action);
259
260 g_free (name);
261}

263static void
264frequencies_menu_create_actions (CPUFreqPopup *popup)
265{
266 GList *available_freqs;
267
268 available_freqs = cpufreq_monitor_get_available_frequencies (popup->priv->monitor);
269
270 while (available_freqs) {
271 const gchar *text;
272 gchar *freq_text;
273 gchar *label;
274 gchar *unit;
275 gint freq;
276
277 text = (const gchar *) available_freqs->data;
278 freq = atoi (text);
279
280 freq_text = cpufreq_utils_get_frequency_label (freq);
281 unit = cpufreq_utils_get_frequency_unit (freq);
282
283 label = g_strdup_printf ("%s %s", freq_text, unit);
284 g_free (freq_text);
285 g_free (unit);
286
287 cpufreq_popup_menu_add_action (popup,
288 "Frequency",
289 popup->priv->freqs_group,
290 text, label, TRUE);
291 g_free (label);
292
293 available_freqs = g_list_next (available_freqs);
294 }
295}

297static void
298governors_menu_create_actions (CPUFreqPopup *popup)
299{
300 GList *available_govs;
301
302 available_govs = cpufreq_monitor_get_available_governors (popup->priv->monitor);
303 available_govs = g_list_sort (available_govs, (GCompareFunc)g_ascii_strcasecmp);
304
305 while (available_govs) {
306 const gchar *governor;
307 gchar *label;
308
309 governor = (const gchar *) available_govs->data;
310 if (g_ascii_strcasecmp (governor, "userspace") == 0) {
311 popup->priv->show_freqs = TRUE;
312 available_govs = g_list_next (available_govs);
313 continue;
314 }
315
316 label = g_strdup (governor);
317 label[0] = g_ascii_toupper (label[0]);
318
319 cpufreq_popup_menu_add_action (popup,
320 "Governor",
321 popup->priv->govs_group,
322 governor, label, TRUE);
323 g_free (label);
324
325 available_govs = g_list_next (available_govs);
326 }
327}

329static void
330cpufreq_popup_build_ui (CPUFreqPopup *popup,
331 GSList *actions,
332 const gchar *menu_path)
333{
334 GSList *l = NULL;
335
336 for (l = actions; l && l->data; l = g_slist_next (l)) {
337 GtkAction *action;
338 gchar *name = NULL;
339 gchar *label = NULL;
340
341 action = (GtkAction *) l->data;
342
343 g_object_get (G_OBJECT (action),
344 "name", &name,
345 "label", &label,
346 NULL);
347
348 gtk_ui_manager_add_ui (popup->priv->ui_manager,

350 menu_path,

352 GTK_UI_MANAGER_MENUITEM,

360static void
361cpufreq_popup_build_frequencies_menu (CPUFreqPopup *popup,
362 const gchar *path)
363{
364 if (!popup->priv->freqs_group) {

373 gtk_ui_manager_insert_action_group (popup->priv->ui_manager,
374 action_group, 0);
375 }
376
377 cpufreq_popup_build_ui (popup,
378 popup->priv->freqs_actions,
379 path);

422static void
423cpufreq_popup_menu_set_active_action (CPUFreqPopup *popup,
424 GtkActionGroup *action_group,
425 const gchar *prefix,
426 const gchar *item)
427{
428 gchar name[128];
429 GtkAction *action;
430
431 g_snprintf (name, sizeof (name), "%s%s", prefix, item);
432 action = gtk_action_group_get_action (action_group, name);
433
434 /* gtk_action_group_get_action can return NULL with frequencies (userspace governor)
435 * when the CPU does not actually stay locked to that exact frequency but rather to a
436 * frequency range. Since cpufreq_monitor_get_frequency gets the realtime frequency, this
437 * may not match any named frequency and then returns NULL. Return when this happens to
438 * avoid segfaults
439 */
440 if (action == NULL)
441 return;
442
443 g_signal_handlers_block_by_func (action,
444 cpufreq_popup_frequencies_menu_activate,
445 popup);
446 g_signal_handlers_block_by_func (action,
447 cpufreq_popup_governors_menu_activate,
448 popup);
449
450 gtk_toggle_action_set_active (GTK_TOGGLE_ACTION (action), TRUE);
451
452 g_signal_handlers_unblock_by_func (action,
453 cpufreq_popup_frequencies_menu_activate,
454 popup);
455 g_signal_handlers_unblock_by_func (action,
456 cpufreq_popup_governors_menu_activate,
457 popup);
458}

460static void
461cpufreq_popup_menu_set_active (CPUFreqPopup *popup)
462{
463 const gchar *governor;
464
465 governor = cpufreq_monitor_get_governor (popup->priv->monitor);
466
467 if (g_ascii_strcasecmp (governor, "userspace") == 0) {
468 gchar *active;
469 guint freq;
470
471 freq = cpufreq_monitor_get_frequency (popup->priv->monitor);
472 active = g_strdup_printf ("%d", freq);
473 cpufreq_popup_menu_set_active_action (popup,
474 popup->priv->freqs_group,
475 "Frequency", active);
476 g_free (active);
477 } else {
478 cpufreq_popup_menu_set_active_action (popup,
479 popup->priv->govs_group,
480 "Governor", governor);
481 }
482}
