172void
173base_gtk_utils_toggle_set_initial_state( BaseWindow *window,
174 const gchar *button_name, GCallback handler,
175 gboolean active, gboolean editable, gboolean sensitive )
176{
177 typedef void ( *toggle_handler )( GtkToggleButton *, BaseWindow * );
178 GtkToggleButton *button;
179
180 button = GTK_TOGGLE_BUTTON( base_window_get_widget( window, button_name ));
181
182 if( button ){
183 base_window_signal_connect( window, G_OBJECT( button ), "toggled", handler );
184
185 g_object_set_data( G_OBJECT( button ), NA_TOGGLE_DATA_HANDLER, handler );
186 g_object_set_data( G_OBJECT( button ), NA_TOGGLE_DATA_USER_DATA, window );
187 g_object_set_data( G_OBJECT( button ), NA_TOGGLE_DATA_EDITABLE, GUINT_TO_POINTER( editable ));
188
189 base_gtk_utils_set_editable( G_OBJECT( button ), editable );
190 gtk_widget_set_sensitive( GTK_WIDGET( button ), sensitive );
191 gtk_toggle_button_set_active( button, active );
192
193 ( *( toggle_handler ) handler )( button, window );
194 }
195}

205void
206base_gtk_utils_toggle_reset_initial_state( GtkToggleButton *button )
207{
208 gboolean editable;
209 GCallback handler;
210 gpointer user_data;
211 gboolean active;
212
213 editable = ( gboolean ) GPOINTER_TO_UINT( g_object_get_data( G_OBJECT( button ), NA_TOGGLE_DATA_EDITABLE ));
214
215 if( !editable ){
216 active = gtk_toggle_button_get_active( button );
217 handler = G_CALLBACK( g_object_get_data( G_OBJECT( button ), NA_TOGGLE_DATA_HANDLER ));
218 user_data = g_object_get_data( G_OBJECT( button ), NA_TOGGLE_DATA_USER_DATA );
219
220 g_signal_handlers_block_by_func(( gpointer ) button, handler, user_data );
221 gtk_toggle_button_set_active( button, !active );
222 g_signal_handlers_unblock_by_func(( gpointer ) button, handler, user_data );
223 }
224}

234GdkPixbuf *
235base_gtk_utils_get_pixbuf( const gchar *name, GtkWidget *widget, GtkIconSize size )
236{
237 static const gchar *thisfn = "base_gtk_utils_get_pixbuf";
238 GdkPixbuf* pixbuf;
239 GError *error;
240 gint width, height;
241 GtkIconTheme *icon_theme;
242
243 error = NULL;
244 pixbuf = NULL;
245
246 if( !gtk_icon_size_lookup( size, &width, &height )){
247 width = DEFAULT_WIDTH;
248 height = DEFAULT_HEIGHT;
249 }
250
251 if( name && strlen( name )){
252 if( g_path_is_absolute( name )){
253 pixbuf = gdk_pixbuf_new_from_file_at_size( name, width, height, &error );
254 if( error ){
255 if( error->code != G_FILE_ERROR_NOENT ){
256 g_warning( "%s: gdk_pixbuf_new_from_file_at_size: name=%s, error=%s (%d)", thisfn, name, error->message, error->code );
257 }
258 g_error_free( error );
259 error = NULL;
260 pixbuf = NULL;
261 }
262
263 } else {
264 icon_theme = gtk_icon_theme_get_default();
265 pixbuf = gtk_icon_theme_load_icon(
266 icon_theme, name, width, GTK_ICON_LOOKUP_GENERIC_FALLBACK, &error );
267 if( error ){
268 /* it happens that the message "Icon 'xxxx' not present in theme"
269 * is generated with a domain of 'gtk-icon-theme-error-quark' and
270 * an error code of zero - it seems difficult to just test zero
271 * so does not display warning, but just debug
272 */
273 g_debug( "%s: %s (%s:%d)",
274 thisfn, error->message, g_quark_to_string( error->domain ), error->code );
275 g_error_free( error );
276 }
277 }
278 }
279
280 if( !pixbuf ){
281 g_debug( "%s: null pixbuf, loading transparent image", thisfn );
282 pixbuf = gdk_pixbuf_new_from_file_at_size( PKGUIDIR "/transparent.png", width, height, NULL );
283 }
284
285 return( pixbuf );
286}

296void
297base_gtk_utils_render( const gchar *name, GtkImage *widget, GtkIconSize size )
298{
299 static const gchar *thisfn = "base_gtk_utils_render";
300 GdkPixbuf* pixbuf;
301 gint width, height;
302
303 g_debug( "%s: name=%s, widget=%p, size=%d", thisfn, name, ( void * ) widget, size );
304
305 if( name ){
306 pixbuf = base_gtk_utils_get_pixbuf( name, GTK_WIDGET( widget ), size );
307
308 } else {
309 if( !gtk_icon_size_lookup( size, &width, &height )){
310 width = DEFAULT_WIDTH;
311 height = DEFAULT_HEIGHT;
312 }
313 pixbuf = gdk_pixbuf_new_from_file_at_size( PKGUIDIR "/transparent.png", width, height, NULL );
314 }
315
316 if( pixbuf ){
317 gtk_image_set_from_pixbuf( widget, pixbuf );
318 g_object_unref( pixbuf );
319 }
320}

348static GtkWidget*
349base_gtk_utils_dialog_add_button (GtkDialog *dialog,
350 const gchar *button_text,
351 const gchar *icon_name,
352 gint response_id)
353{
354 GtkWidget *button;
355
356 button = gtk_button_new_with_mnemonic (button_text);
357 gtk_button_set_image (GTK_BUTTON (button), gtk_image_new_from_icon_name (icon_name, GTK_ICON_SIZE_BUTTON));
358
359 gtk_button_set_use_underline (GTK_BUTTON (button), TRUE);
360 gtk_style_context_add_class (gtk_widget_get_style_context (button), "text-button");
361 gtk_widget_set_can_default (button, TRUE);
362 gtk_widget_show (button);
363 gtk_dialog_add_action_widget (GTK_DIALOG (dialog), button, response_id);
364
365 return button;
366}

368static GtkWidget *
369base_gtk_utils_file_chooser_dialog_new_valist (const gchar *title,
370 GtkWindow *parent,
371 GtkFileChooserAction action,
372 const gchar *first_button_text,
373 va_list varargs)
374{
375 GtkWidget *result;
376 const char *button_text = first_button_text;
377 gint response_id;
378
379 result = g_object_new (GTK_TYPE_FILE_CHOOSER_DIALOG,
380 "title", title,
381 "action", action,
382 NULL);
383
384 if (parent)
385 gtk_window_set_transient_for (GTK_WINDOW (result), parent);
386
387 while (button_text)
388 {
389 response_id = va_arg (varargs, gint);
390
391 if (g_strcmp0 (button_text, "process-stop") == 0)
392 base_gtk_utils_dialog_add_button (GTK_DIALOG (result), _("_Cancel"), button_text, response_id);
393 else if (g_strcmp0 (button_text, "document-open") == 0)
394 base_gtk_utils_dialog_add_button (GTK_DIALOG (result), _("_Open"), button_text, response_id);
395 else
396 gtk_dialog_add_button (GTK_DIALOG (result), button_text, response_id);
397
398 button_text = va_arg (varargs, const gchar *);
399 }
400
401 return result;
402}

442void
443base_gtk_utils_select_file_with_preview( BaseWindow *window,
444 const gchar *title, const gchar *wsp_name,
445 GtkWidget *entry, const gchar *entry_name,
446 GCallback update_preview_cb )
447{
448 GtkWindow *toplevel;
449 GtkWidget *dialog;
450 const gchar *text;
451 gchar *filename, *uri;
452 GtkWidget *preview;
453
454 toplevel = base_window_get_gtk_toplevel( window );
455
456 dialog = base_gtk_utils_file_chooser_dialog_new(
457 title,
458 toplevel,
459 GTK_FILE_CHOOSER_ACTION_OPEN,
460 "process-stop", GTK_RESPONSE_CANCEL,
461 "document-open", GTK_RESPONSE_ACCEPT,
462 NULL
463 );
464
465 if( update_preview_cb ){
466 preview = gtk_image_new();
467 gtk_file_chooser_set_preview_widget( GTK_FILE_CHOOSER( dialog ), preview );
468 g_signal_connect( dialog, "update-preview", update_preview_cb, preview );
469 }
470
471 base_gtk_utils_restore_window_position( window, wsp_name );
472
473 text = gtk_entry_get_text( GTK_ENTRY( entry ));
474
475 if( text && strlen( text )){
476 gtk_file_chooser_set_filename( GTK_FILE_CHOOSER( dialog ), text );
477
478 } else {
479 uri = na_settings_get_string( entry_name, NULL, NULL );
480 if( uri ){
481 gtk_file_chooser_set_current_folder_uri( GTK_FILE_CHOOSER( dialog ), uri );
482 g_free( uri );
483 }
484 }
485
486 if( gtk_dialog_run( GTK_DIALOG( dialog )) == GTK_RESPONSE_ACCEPT ){
487 filename = gtk_file_chooser_get_filename( GTK_FILE_CHOOSER( dialog ));
488 gtk_entry_set_text( GTK_ENTRY( entry ), filename );
489 g_free( filename );
490 }
491
492 uri = gtk_file_chooser_get_current_folder_uri( GTK_FILE_CHOOSER( dialog ));
493 na_settings_set_string( entry_name, uri );
494 g_free( uri );
495
496 base_gtk_utils_save_window_position( window, wsp_name );
497
498 gtk_widget_destroy( dialog );
499}
