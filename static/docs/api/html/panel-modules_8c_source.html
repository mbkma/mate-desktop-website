46void
47panel_modules_ensure_loaded (void)
48{
49 static gboolean loaded_dirs = FALSE;
50
51 panel_modules_ensure_extension_points_registered ();
52
53 if (!loaded_dirs) {
54 const char *module_path;
55 GList *modules;
56 loaded_dirs = TRUE;
57
58 /* We load the modules explicitly instead of using scan_all
59 * so that we can leak a reference to them. This prevents them
60 * from getting unloaded later (something they aren't designed
61 * to cope with) */
62 modules = g_io_modules_load_all_in_directory (PANEL_MODULES_DIR);
63 g_list_free (modules);
64
65 module_path = g_getenv ("MATE_PANEL_EXTRA_MODULES");
66
67 if (module_path) {
68 gchar **paths;
69 int i;
70
71 paths = g_strsplit (module_path, ":", 0);
72
73 for (i = 0; paths[i] != NULL; i++) {
74 modules = g_io_modules_load_all_in_directory (paths[i]);
75 g_list_free (modules);
76 }
77
78 g_strfreev (paths);
79 }
