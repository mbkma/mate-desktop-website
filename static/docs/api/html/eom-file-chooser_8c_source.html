131 gtk_widget_destroy (msg_dialog);
132
133 g_signal_stop_emission_by_name (dlg, "response");

139static void
140eom_file_chooser_add_filter (EomFileChooser *chooser)
141{
142 GSList *it;
143 GSList *formats;
144 GtkFileFilter *all_file_filter;
145 GtkFileFilter *all_img_filter;
146 GtkFileFilter *filter;
147 GSList *filters = NULL;
148 gchar **mime_types, **pattern, *tmp;
149 int i;
150 GtkFileChooserAction action;
151
152 action = gtk_file_chooser_get_action (GTK_FILE_CHOOSER (chooser));
153
154 if (action != GTK_FILE_CHOOSER_ACTION_SAVE && action != GTK_FILE_CHOOSER_ACTION_OPEN) {
155 return;
156 }
157
158 /* All Files Filter */
159 all_file_filter = gtk_file_filter_new ();
160 gtk_file_filter_set_name (all_file_filter, _("All Files"));
161 gtk_file_filter_add_pattern (all_file_filter, "*");
162
163 /* All Image Filter */
164 all_img_filter = gtk_file_filter_new ();
165 gtk_file_filter_set_name (all_img_filter, _("All Images"));
166
167 if (action == GTK_FILE_CHOOSER_ACTION_SAVE) {
168 formats = eom_pixbuf_get_savable_formats ();
169 }
170 else {
171 formats = gdk_pixbuf_get_formats ();
172 }
173
174 /* Image filters */
175 for (it = formats; it != NULL; it = it->next) {
176 char *filter_name;
177 char *description, *extension;
178 GdkPixbufFormat *format;
179 filter = gtk_file_filter_new ();
180
181 format = (GdkPixbufFormat*) it->data;
182 description = gdk_pixbuf_format_get_description (format);
183 extension = gdk_pixbuf_format_get_name (format);
184
185 /* Filter name: First description then file extension, eg. "The PNG-Format (*.png)".*/
186 filter_name = g_strdup_printf (_("%s (*.%s)"), description, extension);
187 g_free (description);
188 g_free (extension);
189
190 gtk_file_filter_set_name (filter, filter_name);
191 g_free (filter_name);
192
193 mime_types = gdk_pixbuf_format_get_mime_types ((GdkPixbufFormat *) it->data);
194 for (i = 0; mime_types[i] != NULL; i++) {
195 gtk_file_filter_add_mime_type (filter, mime_types[i]);
196 gtk_file_filter_add_mime_type (all_img_filter, mime_types[i]);
197 }
198 g_strfreev (mime_types);
199
200 pattern = gdk_pixbuf_format_get_extensions ((GdkPixbufFormat *) it->data);
201 for (i = 0; pattern[i] != NULL; i++) {
202 tmp = g_strconcat ("*.", pattern[i], NULL);
203 gtk_file_filter_add_pattern (filter, tmp);
204 gtk_file_filter_add_pattern (all_img_filter, tmp);
205 g_free (tmp);
206 }
207 g_strfreev (pattern);
208
209 /* attach GdkPixbufFormat to filter, see also
210 * eom_file_chooser_get_format. */
211 g_object_set_data (G_OBJECT (filter),
212 FILE_FORMAT_KEY,
213 format);
214
215 filters = g_slist_prepend (filters, filter);
216 }
217 g_slist_free (formats);
218
219 /* Add filter to filechooser */
220 gtk_file_chooser_add_filter (GTK_FILE_CHOOSER (chooser), all_file_filter);
221 gtk_file_chooser_add_filter (GTK_FILE_CHOOSER (chooser), all_img_filter);
222 gtk_file_chooser_set_filter (GTK_FILE_CHOOSER (chooser), all_img_filter);
223
224 for (it = filters; it != NULL; it = it->next) {
225 gtk_file_chooser_add_filter (GTK_FILE_CHOOSER (chooser), GTK_FILE_FILTER (it->data));
226 }
227 g_slist_free (filters);
228}

242/* Sets the pixbuf as preview thumbnail and tries to read and display
243 * further information according to the thumbnail spec.
244 */
245static void
246set_preview_pixbuf (EomFileChooser *chooser, GdkPixbuf *pixbuf, goffset size)
247{
248 EomFileChooserPrivate *priv;
249 int bytes;
250 int pixels;
251 const char *bytes_str;
252 const char *width;
253 const char *height;
254 const char *creator = NULL;
255 char *size_str = NULL;
256 char *dim_str = NULL;
257
258 g_return_if_fail (EOM_IS_FILE_CHOOSER (chooser));
259
260 priv = chooser->priv;
261
262 gtk_image_set_from_pixbuf (GTK_IMAGE (priv->image), pixbuf);
263
264 if (pixbuf != NULL) {
265 /* try to read file size */
266 bytes_str = gdk_pixbuf_get_option (pixbuf, "tEXt::Thumb::Size");
267 if (bytes_str != NULL) {
268 bytes = atoi (bytes_str);
269 size_str = g_format_size (bytes);
270 }
271 else {
272 size_str = g_format_size (size);
273 }
274
275 /* try to read image dimensions */
276 width = gdk_pixbuf_get_option (pixbuf, "tEXt::Thumb::Image::Width");
277 height = gdk_pixbuf_get_option (pixbuf, "tEXt::Thumb::Image::Height");
278
279 if ((width != NULL) && (height != NULL)) {
280 pixels = atoi (height);
281 /* Pixel size of image: width x height in pixel */
282 dim_str = g_strdup_printf ("%s x %s %s", width, height, ngettext ("pixel", "pixels", pixels));
283 }
284
285#if 0
286 /* Not sure, if this is really useful, therefore its commented out for now. */
287
288 /* try to read creator of the thumbnail */
289 creator = gdk_pixbuf_get_option (pixbuf, "tEXt::Software");
290
291 /* stupid workaround to display nicer string if the
292 * thumbnail is created through the mate libraries.
293 */
294 if (g_ascii_strcasecmp (creator, "Mate::ThumbnailFactory") == 0) {
295 creator = "MATE Libs";
296 }
297#endif
298 }
299
300 set_preview_label (priv->size_label, size_str);
301 set_preview_label (priv->dim_label, dim_str);
302 set_preview_label (priv->creator_label, creator);
303
304 if (size_str != NULL) {
305 g_free (size_str);
306 }
307
308 if (dim_str != NULL) {
309 g_free (dim_str);
310 }
311}

313static void
314update_preview_cb (GtkFileChooser *file_chooser, gpointer data)
315{
316 EomFileChooserPrivate *priv;
317 char *uri;
318 char *thumb_path = NULL;
319 GFile *file;
320 GFileInfo *file_info;
321 GdkPixbuf *pixbuf = NULL;
322 gboolean have_preview = FALSE;
323
324 priv = EOM_FILE_CHOOSER (file_chooser)->priv;
325
326 uri = gtk_file_chooser_get_preview_uri (file_chooser);
327 if (uri == NULL) {
328 gtk_file_chooser_set_preview_widget_active (file_chooser, FALSE);
329 return;
330 }
331
332 file = g_file_new_for_uri (uri);
333 file_info = g_file_query_info (file,
334 G_FILE_ATTRIBUTE_TIME_MODIFIED ","
335 G_FILE_ATTRIBUTE_STANDARD_TYPE ","
336 G_FILE_ATTRIBUTE_STANDARD_SIZE ","
337 G_FILE_ATTRIBUTE_STANDARD_CONTENT_TYPE","
338 G_FILE_ATTRIBUTE_STANDARD_FAST_CONTENT_TYPE,
339 0, NULL, NULL);
340 g_object_unref (file);
341
342 if ((file_info != NULL) && (priv->thumb_factory != NULL)
343 && g_file_info_get_file_type (file_info) != G_FILE_TYPE_SPECIAL) {
344 guint64 mtime;
345
346 mtime = g_file_info_get_attribute_uint64 (file_info,
347 G_FILE_ATTRIBUTE_TIME_MODIFIED);
348 thumb_path = mate_desktop_thumbnail_factory_lookup (priv->thumb_factory, uri, mtime);
349
350 if (thumb_path != NULL && g_file_test (thumb_path, G_FILE_TEST_EXISTS)) {
351 /* try to load and display preview thumbnail */
352 pixbuf = gdk_pixbuf_new_from_file (thumb_path, NULL);
353 } else if (g_file_info_get_size (file_info) <= 100000) {
354 /* read files smaller than 100kb directly */
355
356 gchar *mime_type = g_content_type_get_mime_type (
357 eom_util_get_content_type_with_fallback (file_info));
358
359 if (G_LIKELY (mime_type)) {
360 gboolean can_thumbnail, has_failed;
361
362 can_thumbnail = mate_desktop_thumbnail_factory_can_thumbnail (
363 priv->thumb_factory,
364 uri, mime_type, mtime);
365 has_failed = mate_desktop_thumbnail_factory_has_valid_failed_thumbnail (
366 priv->thumb_factory,
367 uri, mtime);
368
369 if (G_LIKELY (can_thumbnail && !has_failed))
370 pixbuf = mate_desktop_thumbnail_factory_generate_thumbnail (
371 priv->thumb_factory, uri, mime_type);
372
373 g_free (mime_type);
374 }
375 }
376
377 if (pixbuf != NULL) {
378 have_preview = TRUE;
379
380 set_preview_pixbuf (EOM_FILE_CHOOSER (file_chooser), pixbuf,
381 g_file_info_get_size (file_info));
382
383 g_object_unref (pixbuf);
384 }
385 }
386
387 if (thumb_path != NULL) {
388 g_free (thumb_path);
389 }
390
391 g_free (uri);
392 g_object_unref (file_info);
393
394 gtk_file_chooser_set_preview_widget_active (file_chooser, have_preview);
395}

432GtkWidget *
433eom_file_chooser_new (GtkFileChooserAction action)
434{
435 GtkWidget *chooser;
436 gchar *title = NULL;
437
438 chooser = g_object_new (EOM_TYPE_FILE_CHOOSER,
439 "action", action,
440 "select-multiple", (action == GTK_FILE_CHOOSER_ACTION_OPEN),
441 "local-only", FALSE,
442 NULL);
443
444 switch (action) {
445 case GTK_FILE_CHOOSER_ACTION_OPEN:
446 gtk_dialog_add_buttons (GTK_DIALOG (chooser),
447 "gtk-cancel", GTK_RESPONSE_CANCEL,
448 "gtk-open", GTK_RESPONSE_OK,
449 NULL);
450 title = _("Open Image");
451 break;
452
453 case GTK_FILE_CHOOSER_ACTION_SAVE:
454 gtk_dialog_add_buttons (GTK_DIALOG (chooser),
455 "gtk-cancel", GTK_RESPONSE_CANCEL,
456 "gtk-save", GTK_RESPONSE_OK,
457 NULL);
458 title = _("Save Image");
459 break;
460
461 case GTK_FILE_CHOOSER_ACTION_SELECT_FOLDER:
462 gtk_dialog_add_buttons (GTK_DIALOG (chooser),
463 "gtk-cancel", GTK_RESPONSE_CANCEL,
464 "gtk-open", GTK_RESPONSE_OK,
465 NULL);
466 title = _("Open Folder");
467 break;
468
469 default:
470 g_assert_not_reached ();
471 }
472
473 if (action != GTK_FILE_CHOOSER_ACTION_SELECT_FOLDER) {
474 eom_file_chooser_add_filter (EOM_FILE_CHOOSER (chooser));
475 eom_file_chooser_add_preview (chooser);
476 }
477
478 if (last_dir[action] != NULL) {
479 gtk_file_chooser_set_current_folder (GTK_FILE_CHOOSER (chooser), last_dir [action]);
480 }
481
482 g_signal_connect (chooser, "response",
483 G_CALLBACK ((action == GTK_FILE_CHOOSER_ACTION_SAVE) ?
484 save_response_cb : response_cb),
485 NULL);
486
487 gtk_window_set_title (GTK_WINDOW (chooser), title);
488 gtk_dialog_set_default_response (GTK_DIALOG (chooser), GTK_RESPONSE_OK);
489
490 gtk_file_chooser_set_do_overwrite_confirmation (GTK_FILE_CHOOSER (chooser), TRUE);
491
492 return chooser;
493}
