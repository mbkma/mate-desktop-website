127static gboolean
128try_convert (GCharsetConverter *converter,
129 const void *inbuf,
130 gsize inbuf_size)
131{
132 GError *err;
133 gsize bytes_read, nread;
134 gsize bytes_written, nwritten;
135 GConverterResult res;
136 gchar *out;
137 gboolean ret;
138 gsize out_size;
139
140 if (inbuf == NULL || inbuf_size == 0)
141 {
142 return FALSE;
143 }
144
145 err = NULL;
146 nread = 0;
147 nwritten = 0;
148 out_size = inbuf_size * 4;
149 out = g_malloc (out_size);
150
151 do
152 {
153 res = g_converter_convert (G_CONVERTER (converter),
154 (void *) ((gsize) inbuf + nread),
155 inbuf_size - nread,
156 out + nwritten,
157 out_size - nwritten,
158 G_CONVERTER_INPUT_AT_END,
159 &bytes_read,
160 &bytes_written,
161 &err);
162
163 nread += bytes_read;
164 nwritten += bytes_written;
165 } while (res != G_CONVERTER_FINISHED && res != G_CONVERTER_ERROR && err == NULL);
166
167 if (err != NULL)
168 {
169 if (err->code == G_CONVERT_ERROR_PARTIAL_INPUT)
170 {
171 /* FIXME We can get partial input while guessing the
172 encoding because we just take some amount of text
173 to guess from. */
174 ret = TRUE;
175 }
176 else
177 {
178 ret = FALSE;
179 }
180
181 g_error_free (err);
182 }
183 else
184 {
185 ret = TRUE;
186 }
187
188 /* FIXME: Check the remainder? */
189 if (ret == TRUE && !g_utf8_validate (out, nwritten, NULL))
190 {
191 ret = FALSE;
192 }
193
194 g_free (out);
195
196 return ret;
197}

199static GCharsetConverter *
200guess_encoding (PlumaSmartCharsetConverter *smart,
201 const void *inbuf,
202 gsize inbuf_size)
203{
204 GCharsetConverter *conv = NULL;
205
206 if (inbuf == NULL || inbuf_size == 0)
207 {
208 smart->priv->is_utf8 = TRUE;
209 return NULL;
210 }
211
212 if (smart->priv->encodings != NULL &&
213 smart->priv->encodings->next == NULL)
214 smart->priv->use_first = TRUE;
215
216 /* We just check the first block */
217 while (TRUE)
218 {
219 const PlumaEncoding *enc;
220
221 if (conv != NULL)
222 {
223 g_object_unref (conv);
224 conv = NULL;
225 }
226
227 /* We get an encoding from the list */
228 enc = get_encoding (smart);
229
230 /* if it is NULL we didn't guess anything */
231 if (enc == NULL)
232 {
233 break;
234 }
235
236 pluma_debug_message (DEBUG_UTILS, "trying charset: %s",
237 pluma_encoding_get_charset (smart->priv->current_encoding->data));
238
239 if (enc == pluma_encoding_get_utf8 ())
240 {
241 gsize remainder;
242 const gchar *end;
243
244 if (g_utf8_validate (inbuf, inbuf_size, &end) ||
245 smart->priv->use_first)
246 {
247 smart->priv->is_utf8 = TRUE;
248 break;
249 }
250
251 /* Check if the end is less than one char */
252 remainder = inbuf_size - (end - (gchar *)inbuf);
253 if (remainder < 6)
254 {
255 smart->priv->is_utf8 = TRUE;
256 break;
257 }
258
259 continue;
260 }
261
262 conv = g_charset_converter_new ("UTF-8",
263 pluma_encoding_get_charset (enc),
264 NULL);
265
266 /* If we tried all encodings we use the first one */
267 if (smart->priv->use_first)
268 {
269 break;
270 }
271
272 /* Try to convert */
273 if (try_convert (conv, inbuf, inbuf_size))
274 {
275 break;
276 }
277 }
278
279 if (conv != NULL)
280 {
281 g_converter_reset (G_CONVERTER (conv));
282
283 /* FIXME: uncomment this when we want to use the fallback
284 g_charset_converter_set_use_fallback (conv, TRUE);*/
285 }
286
287 return conv;
288}

290static GConverterResult
291pluma_smart_charset_converter_convert (GConverter *converter,
292 const void *inbuf,
293 gsize inbuf_size,
294 void *outbuf,
295 gsize outbuf_size,
296 GConverterFlags flags,
297 gsize *bytes_read,
298 gsize *bytes_written,
299 GError **error)
300{
301 PlumaSmartCharsetConverter *smart = PLUMA_SMART_CHARSET_CONVERTER (converter);
302
303 /* Guess the encoding if we didn't make it yet */
304 if (smart->priv->charset_conv == NULL &&
305 !smart->priv->is_utf8)
306 {
307 smart->priv->charset_conv = guess_encoding (smart, inbuf, inbuf_size);
308
309 /* If we still have the previous case is that we didn't guess
310 anything */
311 if (smart->priv->charset_conv == NULL &&
312 !smart->priv->is_utf8)
313 {
314 /* FIXME: Add a different domain when we kill pluma_convert */
315 g_set_error_literal (error, PLUMA_DOCUMENT_ERROR,
316 PLUMA_DOCUMENT_ERROR_ENCODING_AUTO_DETECTION_FAILED,
317 _("It is not possible to detect the encoding automatically"));
318 return G_CONVERTER_ERROR;
319 }
320 }
321
322 /* Now if the encoding is utf8 just redirect the input to the output */
323 if (smart->priv->is_utf8)
324 {
325 gsize size;
326 GConverterResult ret;
327
328 size = MIN (inbuf_size, outbuf_size);
329
330 memcpy (outbuf, inbuf, size);
331 *bytes_read = size;
332 *bytes_written = size;
333
334 ret = G_CONVERTER_CONVERTED;
335
336 if (flags & G_CONVERTER_INPUT_AT_END)
337 ret = G_CONVERTER_FINISHED;
338 else if (flags & G_CONVERTER_FLUSH)
339 ret = G_CONVERTER_FLUSHED;
340
341 return ret;
342 }
343
344 /* If we reached here is because we need to convert the text so, we
345 convert it with the charset converter */
346 return g_converter_convert (G_CONVERTER (smart->priv->charset_conv),
347 inbuf,
348 inbuf_size,
349 outbuf,
350 outbuf_size,
351 flags,
352 bytes_read,
353 bytes_written,
354 error);
355}
