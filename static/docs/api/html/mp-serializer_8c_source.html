56static void
57mp_to_string_real(MpSerializer *serializer, const MPNumber *x, int base, gboolean force_sign, int *n_digits, GString *string)
58{
59 static gchar digits[] = "0123456789ABCDEF";
60 MPNumber number = mp_new();
61 MPNumber integer_component = mp_new();
62 MPNumber fractional_component = mp_new();
63 MPNumber temp = mp_new();
64 int i;
65 gsize last_non_zero;
66
67 if (mp_is_negative(x))
68 mp_abs(x, &number);
69 else
70 mp_set_from_mp(x, &number);
71
72 /* Add rounding factor */
73 mp_set_from_integer(base, &temp);
74 mp_xpowy_integer(&temp, -(serializer->priv->trailing_digits+1), &temp);
75 mp_multiply_integer(&temp, base, &temp);
76 mp_divide_integer(&temp, 2, &temp);
77 mp_add(&number, &temp, &temp);
78
79 /* If trying to add rounding factor causes overflow, don't add it */
80 if (!mp_get_error())
81 mp_set_from_mp(&temp, &number);
82
83 /* Split into integer and fractional component */
84 mp_floor(&number, &integer_component);
85 mp_fractional_component(&number, &fractional_component);
86
87 /* Write out the integer component least significant digit to most */
88 mp_set_from_mp(&integer_component, &temp);
89 i = 0;
90 MPNumber t = mp_new();
91 MPNumber t2 = mp_new();
92 MPNumber t3 = mp_new();
93 do {
94 long d;
95
96 if (serializer->priv->base == 10 && serializer->priv->show_tsep && i == serializer->priv->tsep_count) {
97 g_string_prepend_unichar(string, serializer->priv->tsep);
98 i = 0;
99 }
100 i++;
101
102 mp_divide_integer(&temp, base, &t);
103 mp_floor(&t, &t);
104 mp_multiply_integer(&t, base, &t2);
105
106 mp_subtract(&temp, &t2, &t3);
107
108 d = mp_to_integer(&t3);
109 if (d < 16 && d >= 0)
110 g_string_prepend_c(string, digits[d]);
111 else
112 { /* FIXME: Should show an overflow error message in fixed mode */
113 g_string_assign(string, "0");
114 *n_digits = serializer->priv->leading_digits+1;
115 break;
116 }
117 (*n_digits)++;
118
119 mp_set_from_mp(&t, &temp);
120 } while (!mp_is_zero(&temp));
121 mp_clear(&t);
122 mp_clear(&t2);
123 mp_clear(&t3);
124
125 last_non_zero = string->len;
126
127 g_string_append_unichar(string, serializer->priv->radix);
128
129 /* Write out the fractional component */
130 mp_set_from_mp(&fractional_component, &temp);
131 MPNumber digit = mp_new();
132 for (i = serializer->priv->trailing_digits; i > 0 && !mp_is_zero(&temp); i--) {
133 int d;
134
135 mp_multiply_integer(&temp, base, &temp);
136 mp_floor(&temp, &digit);
137 d = mp_to_integer(&digit);
138
139 g_string_append_c(string, digits[d]);
140
141 if(d != 0)
142 last_non_zero = string->len;
143 mp_subtract(&temp, &digit, &temp);
144 }
145 mp_clear(&digit);
146
147 /* Strip trailing zeroes */
148 if (!serializer->priv->show_zeroes || serializer->priv->trailing_digits == 0)
149 g_string_truncate(string, last_non_zero);
150
151 /* Add sign on non-zero values */
152 if (strcmp(string->str, "0") != 0 || force_sign) {
153 if (mp_is_negative(x))
154 g_string_prepend(string, "−");
155 else if (force_sign)
156 g_string_prepend(string, "+");
157 }
158
159 /* Append base suffix if not in default base */
160 if (base != serializer->priv->base) {
161 const gchar *base_digits[] = {"₀", "₁", "₂", "₃", "₄", "₅", "₆", "₇", "₈", "₉"};
162 int multiplier = 1;
163 int b = base;
164
165 while (base / multiplier != 0)
166 multiplier *= 10;
167 while (multiplier != 1) {
168 int d;
169 multiplier /= 10;
170 d = b / multiplier;
171 g_string_append(string, base_digits[d]);
172 b -= d * multiplier;
173 }
174 }
175 mp_clear(&number);
176 mp_clear(&integer_component);
177 mp_clear(&fractional_component);
178 mp_clear(&temp);
179}

181static gchar *
182mp_to_string(MpSerializer *serializer, const MPNumber *x, int *n_digits)
183{
184 GString *string;
185 MPNumber x_real = mp_new();
186 gchar *result;
187
188 string = g_string_sized_new(1024);
189
190 mp_real_component(x, &x_real);
191 mp_to_string_real(serializer, &x_real, serializer->priv->base, FALSE, n_digits, string);
192 if (mp_is_complex(x)) {
193 GString *s;
194 gboolean force_sign = TRUE;
195 MPNumber x_im = mp_new();
196 int n_complex_digits = 0;
197
198 mp_imaginary_component(x, &x_im);
199
200 if (strcmp(string->str, "0") == 0) {
201 g_string_assign(string, "");
202 force_sign = FALSE;
203 }
204
205 s = g_string_sized_new(1024);
206 mp_to_string_real(serializer, &x_im, 10, force_sign, &n_complex_digits, s);
207 if (n_complex_digits > *n_digits)
208 *n_digits = n_complex_digits;
209 if (strcmp(s->str, "0") == 0 || strcmp(s->str, "+0") == 0 || strcmp(s->str, "−0") == 0) {
210 /* Ignore unless the real part was also zero and string was cleared*/
211 if (string->len == 0) {
212 g_string_append(string, "0");
213 }
214 }
215 else if (strcmp(s->str, "1") == 0) {
216 g_string_append(string, "i");
217 }
218 else if (strcmp(s->str, "+1") == 0) {
219 g_string_append(string, "+i");
220 }
221 else if (strcmp(s->str, "−1") == 0) {
222 g_string_append(string, "−i");
223 }
224 else {
225 if (strcmp(s->str, "+0") == 0)
226 g_string_append(string, "+");
227 else if (strcmp(s->str, "0") != 0)
228 g_string_append(string, s->str);
229
230 g_string_append(string, "i");
231 }
232 g_string_free(s, TRUE);
233 mp_clear(&x_im);
234 }
235
236 result = g_strndup(string->str, string->len + 1);
237 g_string_free(string, TRUE);
238 mp_clear(&x_real);
239
240 return result;
241}

243static int
244mp_to_exponential_string_real(MpSerializer *serializer, const MPNumber *x, GString *string, gboolean eng_format, int *n_digits)
245{
246 gchar *fixed;
247 MPNumber t = mp_new();
248 MPNumber z = mp_new();
249 MPNumber base = mp_new();
250 MPNumber base3 = mp_new();
251 MPNumber base10 = mp_new();
252 MPNumber base10inv = mp_new();
253 MPNumber mantissa = mp_new();
254 int exponent = 0;
255
256 mp_abs(x, &z);
257 if (mp_is_negative(x))
258 g_string_append(string, "−");
259 mp_set_from_mp(&z, &mantissa);
260
261 mp_set_from_integer(serializer->priv->base, &base);
262 mp_xpowy_integer(&base, 3, &base3);
263 mp_xpowy_integer(&base, 10, &base10);
264 mp_set_from_integer(1, &t);
265 mp_divide(&t, &base10, &base10inv);
266
267 if (!mp_is_zero(&mantissa)) {
268 while (!eng_format && mp_is_greater_equal(&mantissa, &base10)) {
269 exponent += 10;
270 mp_multiply(&mantissa, &base10inv, &mantissa);
271 }
272
273 while ((!eng_format && mp_is_greater_equal(&mantissa, &base)) ||
274 (eng_format && (mp_is_greater_equal(&mantissa, &base3) || exponent % 3 != 0))) {
275 exponent += 1;
276 mp_divide(&mantissa, &base, &mantissa);
277 }
278
279 while (!eng_format && mp_is_less_than(&mantissa, &base10inv)) {
280 exponent -= 10;
281 mp_multiply(&mantissa, &base10, &mantissa);
282 }
283
284 mp_set_from_integer(1, &t);
285 while (mp_is_less_than(&mantissa, &t) || (eng_format && exponent % 3 != 0)) {
286 exponent -= 1;
287 mp_multiply(&mantissa, &base, &mantissa);
288 }
289 }
290
291 fixed = mp_to_string(serializer, &mantissa, n_digits);
292 g_string_append(string, fixed);
293 g_free(fixed);
294
295 mp_clear(&t);
296 mp_clear(&z);
297 mp_clear(&base);
298 mp_clear(&base3);
299 mp_clear(&base10);
300 mp_clear(&base10inv);
301 mp_clear(&mantissa);
302
303 return exponent;
304}

306static void
307append_exponent(GString *string, int exponent)
308{
309 const gchar *super_digits[] = {"⁰", "¹", "²", "³", "⁴", "⁵", "⁶", "⁷", "⁸", "⁹"};
310 gchar *super_value, *c;
311
312 if (exponent == 0)
313 return;
314
315 g_string_append_printf(string, "×10"); // FIXME: Use the current base
316 if (exponent < 0) {
317 exponent = -exponent;
318 g_string_append(string, "⁻");
319 }
320
321 super_value = g_strdup_printf("%d", exponent);
322 for (c = super_value; *c; c++)
323 g_string_append(string, super_digits[*c - '0']);
324 g_free (super_value);
325}

327static gchar *
328mp_to_exponential_string(MpSerializer *serializer, const MPNumber *x, gboolean eng_format, int *n_digits)
329{
330 GString *string;
331 MPNumber x_real = mp_new();
332 gchar *result;
333 int exponent;
334
335 string = g_string_sized_new(1024);
336
337 mp_real_component(x, &x_real);
338 exponent = mp_to_exponential_string_real(serializer, &x_real, string, eng_format, n_digits);
339 append_exponent(string, exponent);
340
341 if (mp_is_complex(x)) {
342 GString *s;
343 MPNumber x_im = mp_new();
344 int n_complex_digits = 0;
345
346 mp_imaginary_component(x, &x_im);
347
348 if (strcmp(string->str, "0") == 0)
349 g_string_assign(string, "");
350 else if (!mp_is_negative(&x_im))
351 g_string_append(string, "+");
352
353 s = g_string_sized_new(1024);
354 exponent = mp_to_exponential_string_real(serializer, &x_im, s, eng_format, &n_complex_digits);
355 if (n_complex_digits > *n_digits)
356 *n_digits = n_complex_digits;
357 if (strcmp(s->str, "0") == 0 || strcmp(s->str, "+0") == 0 || strcmp(s->str, "−0") == 0) {
358 /* Ignore */
359 }
360 else if (strcmp(s->str, "1") == 0) {
361 g_string_append(string, "i");
362 }
363 else if (strcmp(s->str, "+1") == 0) {
364 g_string_append(string, "+i");
365 }
366 else if (strcmp(s->str, "−1") == 0) {
367 g_string_append(string, "−i");
368 }
369 else {
370 if (strcmp(s->str, "+0") == 0)
371 g_string_append(string, "+");
372 else if (strcmp(s->str, "0") != 0)
373 g_string_append(string, s->str);
374
375 g_string_append(string, "i");
376 }
377 g_string_free(s, TRUE);
378 mp_clear(&x_im);
379 append_exponent(string, exponent);
380 }
381
382 result = g_strndup(string->str, string->len + 1);
383 g_string_free(string, TRUE);
384 mp_clear(&x_real);
385
386 return result;
387}

389gchar *
390mp_serializer_to_string(MpSerializer *serializer, const MPNumber *x)
391{
392 gchar *s0;
393 int n_digits = 0;
394 switch(serializer->priv->format) {
395 default:
396 case MP_DISPLAY_FORMAT_AUTOMATIC:
397 s0 = mp_to_string(serializer, x, &n_digits);
398 if (n_digits <= serializer->priv->leading_digits)
399 return s0;
400 else {
401 g_free (s0);
402 return mp_to_exponential_string(serializer, x, FALSE, &n_digits);
403 }
404 break;
405 case MP_DISPLAY_FORMAT_FIXED:
406 return mp_to_string(serializer, x, &n_digits);
407 case MP_DISPLAY_FORMAT_SCIENTIFIC:
408 return mp_to_exponential_string(serializer, x, FALSE, &n_digits);
409 case MP_DISPLAY_FORMAT_ENGINEERING:
410 return mp_to_exponential_string(serializer, x, TRUE, &n_digits);
411 }
412}

628
629 serializer->priv->base = 10;
630 serializer->priv->leading_digits = 12;
631 serializer->priv->trailing_digits = 9;
632 serializer->priv->show_zeroes = FALSE;
633 serializer->priv->show_tsep = FALSE;
634 serializer->priv->format = MP_DISPLAY_FORMAT_AUTOMATIC;
