140static gboolean
141drw_monitor_timeout (DrwMonitor *monitor)
142{
143 DrwMonitorPriv *priv;
144 time_t now;
145
146 priv = monitor->priv;
147
148 if (XScreenSaverQueryInfo (GDK_DISPLAY_XDISPLAY(gdk_display_get_default()), DefaultRootWindow (GDK_DISPLAY_XDISPLAY(gdk_display_get_default())), priv->ss_info) != 0) {
149 if (priv->ss_info->idle < priv->last_idle) {
150 now = time (NULL);
151
152 if (now - priv->last_activity < 25) {
153 g_signal_emit (monitor, signals[ACTIVITY], 0, NULL);
154 }
155
156 priv->last_activity = now;
157 }
158
159 priv->last_idle = priv->ss_info->idle;
160 }
161
162 return TRUE;
