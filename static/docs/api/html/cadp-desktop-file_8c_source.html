218CappDesktopFile *
219cadp_desktop_file_new_from_path( const gchar *path )
220{
221 static const gchar *thisfn = "cadp_desktop_file_new_from_path";
222 CappDesktopFile *ndf;
223 GError *error;
224 gchar *uri;
225
226 ndf = NULL;
227 g_debug( "%s: path=%s", thisfn, path );
228 g_return_val_if_fail( path && g_utf8_strlen( path, -1 ) && g_path_is_absolute( path ), ndf );
229
230 error = NULL;
231 uri = g_filename_to_uri( path, NULL, &error );
232 if( !uri || error ){
233 g_warning( "%s: %s: %s", thisfn, path, error->message );
234 g_error_free( error );
235 g_free( uri );
236 return( NULL );
237 }
238
239 ndf = ndf_new( uri );
240
241 g_free( uri );
242
243 g_key_file_load_from_file( ndf->private->key_file, path, G_KEY_FILE_KEEP_COMMENTS | G_KEY_FILE_KEEP_TRANSLATIONS, &error );
244 if( error ){
245 g_warning( "%s: %s: %s", thisfn, path, error->message );
246 g_error_free( error );
247 g_object_unref( ndf );
248 return( NULL );
249 }
250
251 if( !check_key_file( ndf )){
252 g_object_unref( ndf );
253 return( NULL );
254 }
255
256 return( ndf );
257}

271CappDesktopFile *
272cadp_desktop_file_new_from_uri( const gchar *uri )
273{
274 static const gchar *thisfn = "cadp_desktop_file_new_from_uri";
275 CappDesktopFile *ndf;
276 GError *error;
277 gchar *data;
278 gsize length;
279
280 ndf = NULL;
281 data = NULL;
282 length = 0;
283
284 g_debug( "%s: uri=%s", thisfn, uri );
285 g_return_val_if_fail( uri && g_utf8_strlen( uri, -1 ), ndf );
286
287 data = na_core_utils_file_load_from_uri( uri, &length );
288 g_debug( "%s: length=%lu", thisfn, ( unsigned long ) length );
289
290 /* normally, length and data should be both NULL or both not NULL
291 */
292 if( !length || !data ){
293 return( NULL );
294 }
295
296 error = NULL;
297 ndf = ndf_new( uri );
298 g_key_file_load_from_data( ndf->private->key_file, data, length, G_KEY_FILE_KEEP_COMMENTS | G_KEY_FILE_KEEP_TRANSLATIONS, &error );
299 g_free( data );
300
301 if( error ){
302 if( error->code != G_KEY_FILE_ERROR_GROUP_NOT_FOUND ){
303 g_debug( "%s: %s", thisfn, error->message );
304 }
305 g_error_free( error );
306 g_object_unref( ndf );
307 return( NULL );
308 }
309
310 if( !check_key_file( ndf )){
311 g_object_unref( ndf );
312 return( NULL );
313 }
314
315 return( ndf );
316}

445static gboolean
446check_key_file( CappDesktopFile *ndf )
447{
448 static const gchar *thisfn = "cadp_desktop_file_check_key_file";
449 gboolean ret;
450 gchar *start_group;
451 gboolean has_key;
452 gboolean hidden;
453 gchar *type;
454 GError *error;
455
456 ret = TRUE;
457 error = NULL;
458
459 /* start group must be [Desktop Entry] */
460 start_group = g_key_file_get_start_group( ndf->private->key_file );
461 if( strcmp( start_group, CADP_GROUP_DESKTOP )){
462 g_debug( "%s: %s: invalid start group, found %s, waited for %s",
463 thisfn, ndf->private->uri, start_group, CADP_GROUP_DESKTOP );
464 ret = FALSE;
465 }
466
467 /* must not have Hidden=true value */
468 if( ret ){
469 has_key = g_key_file_has_key( ndf->private->key_file, start_group, CADP_KEY_HIDDEN, &error );
470 if( error ){
471 g_debug( "%s: %s: %s", thisfn, ndf->private->uri, error->message );
472 ret = FALSE;
473
474 } else if( has_key ){
475 hidden = g_key_file_get_boolean( ndf->private->key_file, start_group, CADP_KEY_HIDDEN, &error );
476 if( error ){
477 g_debug( "%s: %s: %s", thisfn, ndf->private->uri, error->message );
478 ret = FALSE;
479
480 } else if( hidden ){
481 g_debug( "%s: %s: Hidden=true", thisfn, ndf->private->uri );
482 ret = FALSE;
483 }
484 }
485 }
486
487 /* must have no Type (which defaults to Action)
488 * or a known one (Action or Menu)
489 */
490 if( ret ){
491 type = NULL;
492 has_key = g_key_file_has_key( ndf->private->key_file, start_group, CADP_KEY_TYPE, &error );
493 if( error ){
494 g_debug( "%s: %s", thisfn, error->message );
495 g_error_free( error );
496 ret = FALSE;
497
498 } else if( has_key ){
499 type = g_key_file_get_string( ndf->private->key_file, start_group, CADP_KEY_TYPE, &error );
500 if( error ){
501 g_debug( "%s: %s", thisfn, error->message );
502 g_free( type );
503 g_error_free( error );
504 ret = FALSE;
505 }
506 }
507 if( ret ){
508 if( !type || !strlen( type )){
509 type = g_strdup( CADP_VALUE_TYPE_ACTION );
510
511 } else if( strcmp( type, CADP_VALUE_TYPE_MENU ) && strcmp( type, CADP_VALUE_TYPE_ACTION )){
512 g_debug( "%s: unmanaged type: %s", thisfn, type );
513 g_free( type );
514 ret = FALSE;
515 }
516 }
517 if( ret ){
518 g_return_val_if_fail( type && strlen( type ), FALSE );
519 ndf->private->type = type;
520 }
521 }
522
523 g_free( start_group );
524
525 return( ret );
526}

585GSList *
586cadp_desktop_file_get_profiles( const CappDesktopFile *ndf )
587{
588 GSList *list;
589 gchar **groups, **ig;
590 gchar *profile_pfx;
591 gchar *profile_id;
592 guint pfx_len;
593
594 g_return_val_if_fail( CADP_IS_DESKTOP_FILE( ndf ), NULL );
595
596 list = NULL;
597
598 if( !ndf->private->dispose_has_run ){
599
600 groups = g_key_file_get_groups( ndf->private->key_file, NULL );
601 if( groups ){
602 ig = groups;
603 profile_pfx = g_strdup_printf( "%s ", CADP_GROUP_PROFILE );
604 pfx_len = strlen( profile_pfx );
605
606 while( *ig ){
607 if( !strncmp( *ig, profile_pfx, pfx_len )){
608 profile_id = g_strdup( *ig+pfx_len );
609 list = g_slist_prepend( list, profile_id );
610 }
611
612 ig++;
613 }
614
615 g_strfreev( groups );
616 g_free( profile_pfx );
617 }
618 }
619
620 return( list );
621}

1012void
1013cadp_desktop_file_set_locale_string( const CappDesktopFile *ndf, const gchar *group, const gchar *key, const gchar *value )
1014{
1015 char **locales;
1016 guint i;
1017 gchar *prefix;
1018 gboolean write;
1019
1020 g_return_if_fail( CADP_IS_DESKTOP_FILE( ndf ));
1021
1022 if( !ndf->private->dispose_has_run ){
1023
1024 locales = ( char ** ) g_get_language_names();
1025 /*
1026 en_US.UTF-8
1027 en_US
1028 en.UTF-8
1029 en
1030 C
1031 */
1032
1033 prefix = g_strdup( locales[0] );
1034 for( i = 0 ; prefix[i] ; ++i ){
1035 if( prefix[i] == '_' || prefix[i] == '@' || prefix[i] == '.' ){
1036 prefix[i] = '\0';
1037 break;
1038 }
1039 }
1040
1041 /* using locales[0] writes a string with, e.g. Label[en_US.UTF-8]
1042 * after that trying to read the same key with another locale, even en_US.utf-8,
1043 * fails ans returns an empty string.
1044 * so write all available locales for the string, so that there is a chance at
1045 * least one of these will be used as default
1046 *
1047 * pwi 2010-12-30 v 3.0.4
1048 * no more write encoding part of the locale as desktop files are supposed to
1049 * be UTF-8 encoded.
1050 */
1051 for( i = 0 ; i < g_strv_length( locales ) ; ++i ){
1052 write = FALSE;
1053
1054 if( g_strstr_len( locales[i], -1, "." )){
1055 continue;
1056 }
1057
1058 /* write the locale string for all locales derived from the first one */
1059 if( !strncmp( locales[i], prefix, strlen( prefix ))){
1060 write = TRUE;
1061
1062 /* also write other locales if first was a 'en'-derivative */
1063 } else if( !strcmp( prefix, "en" )){
1064 write = TRUE;
1065 }
1066
1067 if( write ){
1068 g_key_file_set_locale_string( ndf->private->key_file, group, key, locales[i], value );
1069 }
1070 }
1071
1072 g_free( prefix );
1073 }
1074}
