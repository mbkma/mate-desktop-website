33/* @title: NAIImporter
34 * @short_description: The #NAIImporter Internal Functions
35 * @include: core/na-importer.h
36 *
37 * Internal Caja-Actions code should never directly call a
38 * #NAIImporter interface method, but rather should call the
39 * corresponding na_importer_xxx() functions.
40 *
41 * Importing items is a three-phase operation:
42 *
43 * - first, just try to find an i/o provider which is willing to import
44 * the item;
45 * at this time, only some uris have been successfully imported
46 *
47 * - check then for existence of each imported item;
48 * depending of the preferred import mode, this may be an interactive
49 * process;
50 * at this time, the importation of some objects may have been cancelled
51 * by the user
52 *
53 * - last, and depending of the exact individual import mode of each item,
54 * insert new items or override existing ones in the referential of the
55 * import context.
56 */

69/*

94/*
95 * NAImporterCheckFn:
96 * @imported: the currently imported #NAObjectItem -derived object.
97 * @fn_data: some data to be passed to the function.
98 *
99 * The library takes care of checking for duplicates inside of the imported
100 * population.
101 *
102 * The caller may provide this function in order to check for duplicates
103 * in its own import context, e.g. (and typically) by checking for a same
104 * identifier in the main window items tree view.
105 *
106 * The function should return the already existing item which has the same id
107 * than the currently being imported one, or %NULL if the imported id will be
108 * unique.
109 *
110 * If the caller does not provide its own check function, then each imported
111 * item will be systematically renumbered (allocated a new identifier).
112 *
113 * Returns: the already existing #NAObjectItem with same id, or %NULL.
114 *
115 * Since: 3.2
116 */
117typedef NAObjectItem * ( *NAImporterCheckFn )( const NAObjectItem *, void * );
