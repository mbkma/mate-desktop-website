123/* An application is valid if:
124 *
125 * 1) The file exists
126 * 2) The user has permissions to run the file
127 */
128static gboolean
129check_application (CajaOpenWithDialog *dialog)
130{
131 char *command;
132 char *path = NULL;
133 char **argv = NULL;
134 int argc;
135 GError *error = NULL;
136 gint retval = TRUE;
137
138 command = NULL;
139 if (dialog->details->selected_app_info != NULL)
140 {
141 command = g_strdup (g_app_info_get_executable (dialog->details->selected_app_info));
142 }
143
144 if (command == NULL)
145 {
146 command = g_strdup (gtk_entry_get_text (GTK_ENTRY (dialog->details->entry)));
147 }
148
149 g_shell_parse_argv (command, &argc, &argv, &error);
150 if (error)
151 {
152 eel_show_error_dialog (_("Could not run application"),
153 error->message,
154 GTK_WINDOW (dialog));
155 g_error_free (error);
156 retval = FALSE;
157 goto cleanup;
158 }
159
160 path = g_find_program_in_path (argv[0]);
161 if (!path)
162 {
163 char *error_message;
164
165 error_message = g_strdup_printf (_("Could not find '%s'"),
166 argv[0]);
167
168 eel_show_error_dialog (_("Could not find application"),
169 error_message,
170 GTK_WINDOW (dialog));
171 g_free (error_message);
172 retval = FALSE;
173 goto cleanup;
174 }
175
176cleanup:
177 g_strfreev (argv);
178 g_free (path);
179 g_free (command);
180
181 return retval;
182}

184/* Only called for non-desktop files */
185static char *
186get_app_name (const char *commandline, GError **error)
187{
188 char *basename;
189 char *unquoted;
190 char **argv;
191 int argc;
192
193 if (!g_shell_parse_argv (commandline,
194 &argc, &argv, error))
195 {
196 return NULL;
197 }
198
199 unquoted = g_shell_unquote (argv[0], NULL);
200 if (unquoted)
201 {
202 basename = g_path_get_basename (unquoted);
203 }
204 else
205 {
206 basename = g_strdup (argv[0]);
207 }
208
209 g_free (unquoted);
210 g_strfreev (argv);
211
212 return basename;
213}

215/* This will check if the application the user wanted exists will return that
216 * application. If it doesn't exist, it will create one and return that.
217 * It also sets the app info as the default for this type.
218 */
219static GAppInfo *
220add_or_find_application (CajaOpenWithDialog *dialog)
221{
222 GAppInfo *app;
223 GError *error;
224 gboolean success, should_set_default;
225 char *message;
226
227 error = NULL;
228 app = NULL;
229 if (dialog->details->selected_app_info)
230 {
231 app = g_object_ref (dialog->details->selected_app_info);
232 }
233 else
234 {
235 char *app_name;
236 const char *commandline;
237
238 commandline = gtk_entry_get_text (GTK_ENTRY (dialog->details->entry));
239 app_name = get_app_name (commandline, &error);
240 if (app_name != NULL)
241 {
242 app = g_app_info_create_from_commandline (commandline,
243 app_name,
244 G_APP_INFO_CREATE_NONE,
245 &error);
246 g_free (app_name);
247 }
248 }
249
250 if (app == NULL)
251 {
252 message = g_strdup_printf (_("Could not add application to the application database: %s"), error ? error->message : _("Unknown error"));
253 eel_show_error_dialog (_("Could not add application"),
254 message,
255 GTK_WINDOW (dialog));
256 g_free (message);
257
258 if (error)
259 g_error_free (error);
260
261 return NULL;
262 }
263
264 should_set_default = (dialog->details->add_mode) ||
265 gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (dialog->details->checkbox));
266 success = TRUE;
267
268 if (should_set_default)
269 {
270 if (dialog->details->content_type)
271 {
272 success = g_app_info_set_as_default_for_type (app,
273 dialog->details->content_type,
274 &error);
275 }
276 else
277 {
278 success = g_app_info_set_as_default_for_extension (app,
279 dialog->details->extension,
280 &error);
281 }
282 }
283 else
284 {
285 GList *applications;
286
287 applications = g_app_info_get_all_for_type (dialog->details->content_type);
288 if (dialog->details->content_type && applications != NULL)
289 {
290 /* we don't care about reporting errors here */
291 g_app_info_add_supports_type (app,
292 dialog->details->content_type,
293 NULL);
294 }
295
296 if (applications != NULL)
297 {
298 g_list_free_full (applications, g_object_unref);
299 }
300 }
301
302 if (!success && should_set_default)
303 {
304 message = g_strdup_printf (_("Could not set application as the default: %s"), error->message);
305 eel_show_error_dialog (_("Could not set as default application"),
306 message,
307 GTK_WINDOW (dialog));
308 g_free (message);
309 g_error_free (error);
310 }
311
312 g_signal_emit_by_name (caja_signaller_get_current (),
313 "mime_data_changed");
314 return app;
315}

325static void
326response_cb (CajaOpenWithDialog *dialog,
327 int response_id,
328 gpointer data)
329{
330 switch (response_id)
331 {
332 case RESPONSE_OPEN:
333 if (check_application (dialog))
334 {
335 GAppInfo *application;
336
337 application = add_or_find_application (dialog);
338
339 if (application)
340 {
341 emit_application_selected (dialog, application);
342 g_object_unref (application);
343
344 gtk_widget_destroy (GTK_WIDGET (dialog));
345 }
346 }
347
348 break;
349 case RESPONSE_REMOVE:
350 if (dialog->details->selected_app_info != NULL)
351 {
352 if (g_app_info_delete (dialog->details->selected_app_info))
353 {
354 GtkTreeModel *model;
355 GtkTreeIter iter;
356 GAppInfo *info, *selected;
357
358 selected = dialog->details->selected_app_info;
359 dialog->details->selected_app_info = NULL;
360
361 model = GTK_TREE_MODEL (dialog->details->program_list_store);
362 if (gtk_tree_model_get_iter_first (model, &iter))
363 {
364 do
365 {
366 gtk_tree_model_get (model, &iter,
367 COLUMN_APP_INFO, &info,
368 -1);
369 if (g_app_info_equal (selected, info))
370 {
371 gtk_list_store_remove (dialog->details->program_list_store, &iter);
372 break;
373 }
374 }
375 while (gtk_tree_model_iter_next (model, &iter));
376 }
377
378 g_object_unref (selected);
379 }
380 }
381 break;
382 case GTK_RESPONSE_NONE:
383 case GTK_RESPONSE_DELETE_EVENT:
384 case GTK_RESPONSE_CANCEL:
385 gtk_widget_destroy (GTK_WIDGET (dialog));
386 break;
387 default :
388 g_assert_not_reached ();
389 }
390
391}

497#define CAJA_OPEN_WITH_DIALOG_ICON_SIZE 24
498static cairo_surface_t *
499get_surface_for_icon (GIcon *icon)
500{
501 cairo_surface_t *surface;
502 gint icon_scale;
503
504 surface = NULL;
505 icon_scale = gdk_window_get_scale_factor (gdk_get_default_root_window ());
506
507 if (G_IS_FILE_ICON (icon))
508 {
509 char *filename;
510
511 filename = g_file_get_path (g_file_icon_get_file (G_FILE_ICON (icon)));
512 if (filename)
513 {
514 GdkPixbuf *pixbuf;
515 pixbuf = gdk_pixbuf_new_from_file_at_size (filename,
516 CAJA_OPEN_WITH_DIALOG_ICON_SIZE * icon_scale,
517 CAJA_OPEN_WITH_DIALOG_ICON_SIZE * icon_scale,
518 NULL);
519 surface = gdk_cairo_surface_create_from_pixbuf (pixbuf, icon_scale, NULL);
520 g_object_unref (pixbuf);
521 }
522 g_free (filename);
523 }
524 else if (G_IS_THEMED_ICON (icon))
525 {
526 const char * const *names;
527
528 names = g_themed_icon_get_names (G_THEMED_ICON (icon));
529
530 if (names != NULL && names[0] != NULL)
531 {
532 char *icon_no_extension;
533 char *p;
534
535 icon_no_extension = g_strdup (names[0]);
536 p = strrchr (icon_no_extension, '.');
537 if (p &&
538 (strcmp (p, ".png") == 0 ||
539 strcmp (p, ".xpm") == 0 ||
540 strcmp (p, ".svg") == 0))
541 {
542 *p = 0;
543 }
544 surface = gtk_icon_theme_load_surface (gtk_icon_theme_get_default (),
545 icon_no_extension,
546 CAJA_OPEN_WITH_DIALOG_ICON_SIZE,
547 icon_scale,
548 NULL,
549 GTK_ICON_LOOKUP_FORCE_SIZE,
550 NULL);
551 g_free (icon_no_extension);
552 }
553 }
554 return surface;
555}

557static gboolean
558caja_open_with_dialog_add_icon_idle (CajaOpenWithDialog *dialog)
559{
560 cairo_surface_t *surface;
561 GIcon *icon;
562 gboolean long_operation;
563 GtkTreeIter iter;
564 GtkTreePath *path = NULL;
565
566 long_operation = FALSE;
567 do
568 {
569 if (!dialog->details->add_icon_paths)
570 {
571 dialog->details->add_icons_idle_id = 0;
572 return FALSE;
573 }
574
575 path = dialog->details->add_icon_paths->data;
576 dialog->details->add_icon_paths->data = NULL;
577 dialog->details->add_icon_paths = g_slist_delete_link (dialog->details->add_icon_paths,
578 dialog->details->add_icon_paths);
579
580 if (!gtk_tree_model_get_iter (GTK_TREE_MODEL (dialog->details->program_list_store),
581 &iter, path))
582 {
583 gtk_tree_path_free (path);
584 continue;
585 }
586
587 gtk_tree_path_free (path);
588
589 gtk_tree_model_get (GTK_TREE_MODEL (dialog->details->program_list_store), &iter,
590 COLUMN_GICON, &icon, -1);
591
592 if (icon == NULL)
593 {
594 continue;
595 }
596
597 surface = get_surface_for_icon (icon);
598 if (surface)
599 {
600 long_operation = TRUE;
601 gtk_list_store_set (dialog->details->program_list_store, &iter, COLUMN_ICON, surface, -1);
602 cairo_surface_destroy (surface);
603 }
604
605 /* don't go back into the main loop if this wasn't very hard to do */
606 }
607 while (!long_operation);
608
609 return TRUE;
610}

612static gboolean
613caja_open_with_search_equal_func (GtkTreeModel *model,
614 int column,
615 const char *key,
616 GtkTreeIter *iter,
617 gpointer user_data)
618{
619 char *name;
620 char *path;
621 gboolean ret;
622
623 if (key != NULL)
624 {
625 char *normalized_key;
626
627 normalized_key = g_utf8_casefold (key, -1);
628 g_assert (normalized_key != NULL);
629
630 ret = TRUE;
631
632 gtk_tree_model_get (model, iter,
633 COLUMN_NAME, &name,
634 COLUMN_EXEC, &path,
635 -1);
636
637 if (name != NULL)
638 {
639 char *normalized_name;
640
641 normalized_name = g_utf8_casefold (name, -1);
642 g_assert (normalized_name != NULL);
643
644 if (strncmp (normalized_name, normalized_key, strlen (normalized_key)) == 0)
645 {
646 ret = FALSE;
647 }
648
649 g_free (normalized_name);
650 }
651
652 if (ret && path != NULL)
653 {
654 char *normalized_path;
655 char *basename, *normalized_basename;
656
657 normalized_path = g_utf8_casefold (path, -1);
658 g_assert (normalized_path != NULL);
659
660 basename = g_path_get_basename (path);
661 g_assert (basename != NULL);
662
663 normalized_basename = g_utf8_casefold (basename, -1);
664 g_assert (normalized_basename != NULL);
665
666 if (strncmp (normalized_path, normalized_key, strlen (normalized_key)) == 0 ||
667 strncmp (normalized_basename, normalized_key, strlen (normalized_key)) == 0)
668 {
669 ret = FALSE;
670 }
671
672 g_free (basename);
673 g_free (normalized_basename);
674 g_free (normalized_path);
675 }
676
677 g_free (name);
678 g_free (path);
679 g_free (normalized_key);
680
681 return ret;
682 }
683 else
684 {
685 return TRUE;
686 }

689static gboolean
690caja_open_with_dialog_add_items_idle (CajaOpenWithDialog *dialog)
691{
692 GtkCellRenderer *renderer;
693 GtkTreeViewColumn *column;
694 GtkTreeModel *sort;
695 GList *all_applications;
696 GList *l;
697
698 /* create list store */
699 dialog->details->program_list_store = gtk_list_store_new (NUM_COLUMNS,
700 G_TYPE_APP_INFO,
701 CAIRO_GOBJECT_TYPE_SURFACE,
702 G_TYPE_ICON,
703 G_TYPE_STRING,
704 G_TYPE_STRING,
705 G_TYPE_STRING);
706 sort = gtk_tree_model_sort_new_with_model (GTK_TREE_MODEL (dialog->details->program_list_store));
707 all_applications = g_app_info_get_all ();
708
709 for (l = all_applications; l; l = l->next)
710 {
711 GAppInfo *app = l->data;
712 GtkTreeIter iter;
713 GtkTreePath *path;
714
715 if (!g_app_info_supports_uris (app) &&
716 !g_app_info_supports_files (app))
717 continue;
718
719 gtk_list_store_append (dialog->details->program_list_store, &iter);
720 gtk_list_store_set (dialog->details->program_list_store, &iter,
721 COLUMN_APP_INFO, app,
722 COLUMN_ICON, NULL,
723 COLUMN_GICON, g_app_info_get_icon (app),
724 COLUMN_NAME, g_app_info_get_display_name (app),
725 COLUMN_COMMENT, g_app_info_get_description (app),
726 COLUMN_EXEC, g_app_info_get_executable,
727 -1);
728
729 path = gtk_tree_model_get_path (GTK_TREE_MODEL (dialog->details->program_list_store), &iter);
730 if (path != NULL)
731 {
732 dialog->details->add_icon_paths = g_slist_prepend (dialog->details->add_icon_paths, path);
733 }
734 }
735 g_list_free (all_applications);
736
737 gtk_tree_view_set_model (GTK_TREE_VIEW (dialog->details->program_list),
738 GTK_TREE_MODEL (sort));
739 gtk_tree_sortable_set_sort_column_id (GTK_TREE_SORTABLE (sort),
740 COLUMN_NAME, GTK_SORT_ASCENDING);
741 gtk_tree_view_set_search_equal_func (GTK_TREE_VIEW (dialog->details->program_list),
742 caja_open_with_search_equal_func,
743 NULL, NULL);
744
745 renderer = gtk_cell_renderer_pixbuf_new ();
746 column = gtk_tree_view_column_new ();
747 gtk_tree_view_column_pack_start (column, renderer, FALSE);
748 gtk_tree_view_column_set_attributes (column, renderer,
749 "surface", COLUMN_ICON,
750 NULL);
751
752 renderer = gtk_cell_renderer_text_new ();
753 gtk_tree_view_column_pack_start (column, renderer, TRUE);
754 gtk_tree_view_column_set_attributes (column, renderer,
755 "text", COLUMN_NAME,
756 NULL);
757 gtk_tree_view_column_set_sort_column_id (column, COLUMN_NAME);
758 gtk_tree_view_append_column (GTK_TREE_VIEW (dialog->details->program_list), column);
759
760 dialog->details->add_icon_paths = g_slist_reverse (dialog->details->add_icon_paths);
761
762 if (!dialog->details->add_icons_idle_id)
763 {
764 dialog->details->add_icons_idle_id =
765 g_idle_add_full (G_PRIORITY_DEFAULT_IDLE, (GSourceFunc) caja_open_with_dialog_add_icon_idle,
766 dialog, NULL);
767 }
768
769 dialog->details->add_items_idle_id = 0;
770 return FALSE;
771}

773static void
774program_list_selection_changed (GtkTreeSelection *selection,
775 CajaOpenWithDialog *dialog)
776{
777 GtkTreeModel *model;
778 GtkTreeIter iter;
779 GAppInfo *info;
780
781 if (!gtk_tree_selection_get_selected (selection, &model, &iter))
782 {
783 gtk_widget_set_sensitive (dialog->details->button, FALSE);
784 gtk_dialog_set_response_sensitive (GTK_DIALOG (dialog),
785 RESPONSE_REMOVE,
786 FALSE);
787 return;
788 }
789
790 info = NULL;
791 gtk_tree_model_get (model, &iter,
792 COLUMN_APP_INFO, &info,
793 -1);
794
795 if (info == NULL)
796 {
797 return;

1019static void
1020set_uri_and_type (CajaOpenWithDialog *dialog,
1021 const char *uri,
1022 const char *mime_type,
1023 const char *passed_extension,
1024 gboolean add_mode)
1025{
1026 char *label;
1027 char *emname;
1028 char *name, *extension;
1029 char *checkbox_text;
1030
1031 name = NULL;
1032 extension = NULL;
1033
1034 if (uri != NULL)
1035 {
1036 GFile *file;
1037
1038 file = g_file_new_for_uri (uri);
1039 name = g_file_get_basename (file);
1040 g_object_unref (file);
1041 }
1042 if (passed_extension == NULL && name != NULL)
1043 {
1044 extension = get_extension (name);
1045 }
1046 else
1047 {
1048 extension = g_strdup (passed_extension);
1049 }
1050
1051 if (extension != NULL &&
1052 g_content_type_is_unknown (mime_type))
1053 {
1054 dialog->details->extension = g_strdup (extension);
1055
1056 if (name != NULL)
1057 {
1058 emname = g_strdup_printf ("<i>%s</i>", name);
1059 if (add_mode)
1060 {
1061 /* Translators: first %s is a filename and second %s is a file extension */
1062 label = g_strdup_printf (_("Open %s and other %s document with:"),
1063 emname, dialog->details->extension);
1064 }
1065 else
1066 {
1067 /* Translators: the %s here is a file name */
1068 label = g_strdup_printf (_("Open %s with:"), emname);
1069 checkbox_text = g_strdup_printf (_("_Remember this application for %s documents"),
1070 dialog->details->extension);
1071
1072 gtk_button_set_label (GTK_BUTTON (dialog->details->checkbox), checkbox_text);
1073 g_free (checkbox_text);
1074 }
1075 g_free (emname);
1076 }
1077 else
1078 {
1079 /* Only in add mode - the %s here is a file extension */
1080 label = g_strdup_printf (_("Open all %s documents with:"),
1081 dialog->details->extension);
1082 }
1083 g_free (extension);
1084 }
1085 else
1086 {
1087 char *description;
1088
1089 dialog->details->content_type = g_strdup (mime_type);
1090 description = g_content_type_get_description (mime_type);
1091
1092 if (description == NULL)
1093 {
1094 description = g_strdup (_("Unknown"));
1095 }
1096
1097 if (name != NULL)
1098 {
1099 emname = g_strdup_printf ("<i>%s</i>", name);
1100 if (add_mode)
1101 {
1102 /* Translators: First %s is a filename, second is a description
1103 * of the type, eg "plain text document" */
1104 label = g_strdup_printf (_("Open %s and other \"%s\" files with:"),
1105 emname, description);
1106 }
1107 else
1108 {
1109 /* Translators: %s is a filename */
1110 label = g_strdup_printf (_("Open %s with:"), emname);
1111 /* Translators: %s is a file type description */
1112 checkbox_text = g_strdup_printf (_("_Remember this application for \"%s\" files"),
1113 description);
1114
1115 gtk_button_set_label (GTK_BUTTON (dialog->details->checkbox), checkbox_text);
1116 g_free (checkbox_text);
1117 }
1118 g_free (emname);
1119 }
1120 else
1121 {
1122 /* Only in add mode */
1123 label = g_strdup_printf (_("Open all \"%s\" files with:"), description);
1124 }
1125
1126 g_free (description);
1127 }
1128
1129 dialog->details->add_mode = add_mode;
1130 if (add_mode)
1131 {
1132 gtk_widget_hide (dialog->details->checkbox);
1133
1134 gtk_label_set_text_with_mnemonic (GTK_LABEL (dialog->details->open_label),
1135 _("_Add"));
1136 gtk_window_set_title (GTK_WINDOW (dialog), _("Add Application"));
1137 }
1138
1139 gtk_label_set_markup (GTK_LABEL (dialog->details->label), label);
1140
1141 g_free (label);
1142 g_free (name);
1143}
