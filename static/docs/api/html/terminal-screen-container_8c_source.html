58static void
59terminal_screen_container_set_placement_internal (TerminalScreenContainer *container,
60 GtkCornerType corner)
61{
62 TerminalScreenContainerPrivate *priv = container->priv;
63
64#ifdef USE_SCROLLED_WINDOW
65 gtk_scrolled_window_set_placement (GTK_SCROLLED_WINDOW (priv->scrolled_window), corner);
66#else
67 switch (corner)
68 {
69 case GTK_CORNER_TOP_LEFT:
70 case GTK_CORNER_BOTTOM_LEFT:
71 gtk_box_reorder_child (GTK_BOX (priv->hbox), priv->vscrollbar, 1);
72 break;
73 case GTK_CORNER_TOP_RIGHT:
74 case GTK_CORNER_BOTTOM_RIGHT:
75 gtk_box_reorder_child (GTK_BOX (priv->hbox), priv->vscrollbar, 0);
76 break;
77 default:
78 g_assert_not_reached ();
79 }
80#endif
81
82 priv->window_placement = corner;
83 g_object_notify (G_OBJECT (container), "window-placement");
84}

129static GObject *
130terminal_screen_container_constructor (GType type,
131 guint n_construct_properties,
132 GObjectConstructParam *construct_params)
133{
134 GObject *object;
135 TerminalScreenContainer *container;
136 TerminalScreenContainerPrivate *priv;
137
138 object = G_OBJECT_CLASS (terminal_screen_container_parent_class)->constructor
139 (type, n_construct_properties, construct_params);
140
141 container = TERMINAL_SCREEN_CONTAINER (object);
142 priv = container->priv;
143
144 g_assert (priv->screen != NULL);
145
146#ifdef USE_SCROLLED_WINDOW
147 priv->scrolled_window = gtk_scrolled_window_new (NULL, gtk_scrollable_get_vadjustment (GTK_SCROLLABLE (priv->screen)));
148
149 gtk_scrolled_window_set_policy (GTK_SCROLLED_WINDOW (priv->scrolled_window),
150 priv->hscrollbar_policy,
151 priv->vscrollbar_policy);
152 gtk_scrolled_window_set_shadow_type (GTK_SCROLLED_WINDOW (priv->scrolled_window),
153 GTK_SHADOW_NONE);
154 gtk_container_add (GTK_CONTAINER (priv->scrolled_window), GTK_WIDGET (priv->screen));
155 gtk_widget_show (GTK_WIDGET (priv->screen));
156 gtk_box_pack_end (GTK_BOX (container), priv->scrolled_window, TRUE, TRUE, 0);
157 gtk_widget_show (priv->scrolled_window);
158
159#ifdef MATE_ENABLE_DEBUG
160 g_signal_connect (priv->scrolled_window, "size-allocate", G_CALLBACK (size_allocate_cb), container);
161#endif
162
163#else
164
165 priv->hbox = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);
166
167 priv->vscrollbar = gtk_scrollbar_new (GTK_ORIENTATION_VERTICAL, gtk_scrollable_get_vadjustment (GTK_SCROLLABLE (priv->screen)));
168
169 gtk_box_pack_start (GTK_BOX (priv->hbox), GTK_WIDGET (priv->screen), TRUE, TRUE, 0);
170 gtk_box_pack_start (GTK_BOX (priv->hbox), priv->vscrollbar, FALSE, FALSE, 0);
171
172 gtk_box_pack_end (GTK_BOX (container), priv->hbox, TRUE, TRUE, 0);
173 gtk_widget_show_all (priv->hbox);
174#endif /* USE_SCROLLED_WINDOW */
175
176 _terminal_screen_update_scrollbar (priv->screen);
177
178 return object;
179}

181static void
182terminal_screen_container_get_property (GObject *object,
183 guint prop_id,
184 GValue *value,
185 GParamSpec *pspec)
186{
187 TerminalScreenContainer *container = TERMINAL_SCREEN_CONTAINER (object);
188 TerminalScreenContainerPrivate *priv = container->priv;
189
190 switch (prop_id)
191 {
192 case PROP_SCREEN:
193 break;
194 case PROP_HSCROLLBAR_POLICY:
195 g_value_set_enum (value, priv->hscrollbar_policy);
196 break;
197 case PROP_VSCROLLBAR_POLICY:
198 g_value_set_enum (value, priv->vscrollbar_policy);
199 break;
200 case PROP_WINDOW_PLACEMENT:
201 g_value_set_enum (value, priv->window_placement);
202 break;
203 case PROP_WINDOW_PLACEMENT_SET:
204 g_value_set_boolean (value, priv->window_placement_set);
205 break;
206 default:
207 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
208 break;
209 }

212static void
213terminal_screen_container_set_property (GObject *object,
214 guint prop_id,
215 const GValue *value,
216 GParamSpec *pspec)
217{
218 TerminalScreenContainer *container = TERMINAL_SCREEN_CONTAINER (object);
219 TerminalScreenContainerPrivate *priv = container->priv;
220
221 switch (prop_id)
222 {
223 case PROP_SCREEN:
224 priv->screen = g_value_get_object (value);
225 break;
226 case PROP_HSCROLLBAR_POLICY:
227 terminal_screen_container_set_policy (container,
228 g_value_get_enum (value),
229 priv->vscrollbar_policy);
230 break;
231 case PROP_VSCROLLBAR_POLICY:
232 terminal_screen_container_set_policy (container,
233 priv->hscrollbar_policy,
234 g_value_get_enum (value));
235 break;
236 case PROP_WINDOW_PLACEMENT:
237 terminal_screen_container_set_placement_internal (container, g_value_get_enum (value));
238 break;
239 case PROP_WINDOW_PLACEMENT_SET:
240 terminal_screen_container_set_placement_set (container, g_value_get_boolean (value));
241 break;
242 default:
243 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
244 break;
245 }

353void
354terminal_screen_container_set_policy (TerminalScreenContainer *container,
355 GtkPolicyType hpolicy G_GNUC_UNUSED,
356 GtkPolicyType vpolicy)
357{
358 TerminalScreenContainerPrivate *priv;
359 GObject *object;
360
361 g_return_if_fail (TERMINAL_IS_SCREEN_CONTAINER (container));
362
363 object = G_OBJECT (container);
364 priv = container->priv;
365
366 g_object_freeze_notify (object);
367
368 if (priv->hscrollbar_policy != hpolicy)
369 {
370 priv->hscrollbar_policy = hpolicy;
371 g_object_notify (object, "hscrollbar-policy");
372 }
373 if (priv->vscrollbar_policy != vpolicy)
374 {
375 priv->vscrollbar_policy = vpolicy;
376 g_object_notify (object, "vscrollbar-policy");
377 }
378
379#ifdef USE_SCROLLED_WINDOW
380 gtk_scrolled_window_set_policy (GTK_SCROLLED_WINDOW (priv->scrolled_window), hpolicy, vpolicy);
381#else
382 switch (vpolicy)
383 {
384 case GTK_POLICY_NEVER:
385 gtk_widget_hide (priv->vscrollbar);
386 break;
387 case GTK_POLICY_AUTOMATIC:
388 case GTK_POLICY_ALWAYS:
389 gtk_widget_show (priv->vscrollbar);
390 break;
391 default:
392 g_assert_not_reached ();
393 }
394#endif
395
396 g_object_thaw_notify (object);
397}
