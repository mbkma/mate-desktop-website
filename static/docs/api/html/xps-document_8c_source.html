89static gboolean
90xps_document_load (EvDocument *document,
91 const char *uri,
92 GError **error)
93{
94 XPSDocument *xps = XPS_DOCUMENT (document);
95
96 xps->file = g_file_new_for_uri (uri);
97 xps->xps = gxps_file_new (xps->file, error);
98
99 if (!xps->xps)
100 return FALSE;
101
102 /* FIXME: what if there are multiple docs? */
103 xps->doc = gxps_file_get_document (xps->xps, 0, error);
104 if (!xps->doc) {
105 g_object_unref (xps->xps);
106 xps->xps = NULL;
107
108 return FALSE;
109 }
110
111 return TRUE;
112}

202static cairo_surface_t *
203xps_document_render (EvDocument *document,
204 EvRenderContext *rc)
205{
206 GXPSPage *xps_page;
207 gdouble page_width, page_height;
208 guint width, height;
209 cairo_surface_t *surface;
210 cairo_t *cr;
211 GError *error = NULL;
212
213 xps_page = GXPS_PAGE (rc->page->backend_page);
214
215 gxps_page_get_size (xps_page, &page_width, &page_height);
216 if (rc->rotation == 90 || rc->rotation == 270) {
217 width = (guint) ((page_height * rc->scale) + 0.5);
218 height = (guint) ((page_width * rc->scale) + 0.5);
219 } else {
220 width = (guint) ((page_width * rc->scale) + 0.5);
221 height = (guint) ((page_height * rc->scale) + 0.5);
222 }
223
224 surface = cairo_image_surface_create (CAIRO_FORMAT_ARGB32,
225 width, height);
226 cr = cairo_create (surface);
227
228 cairo_set_source_rgb (cr, 1., 1., 1.);
229 cairo_paint (cr);
230
231 switch (rc->rotation) {
232 case 90:
233 cairo_translate (cr, width, 0);
234 break;
235 case 180:
236 cairo_translate (cr, width, height);
237 break;
238 case 270:
239 cairo_translate (cr, 0, height);
240 break;
241 default:
242 cairo_translate (cr, 0, 0);
243 }
244
245 cairo_scale (cr, rc->scale, rc->scale);
246 cairo_rotate (cr, rc->rotation * G_PI / 180.0);
247 gxps_page_render (xps_page, cr, &error);
248 cairo_destroy (cr);
249
250 if (error) {
251 g_warning ("Error rendering page %d: %s\n",
252 rc->page->index, error->message);
253 g_error_free (error);
254 }
255
256 return surface;
257}

295static EvLink *
296ev_link_from_target (XPSDocument *xps_document,
297 GXPSLinkTarget *target)
298{
299 EvLinkAction *ev_action;
300
301 if (gxps_link_target_is_internal (target)) {
302 EvLinkDest *dest = NULL;
303 guint doc;
304 const gchar *anchor;
305
306 anchor = gxps_link_target_get_anchor (target);
307
308 /* FIXME: multidoc */
309 doc = gxps_file_get_document_for_link_target (xps_document->xps, target);
310 if (doc == 0) {
311 if (!anchor)
312 return NULL;
313
314 dest = ev_link_dest_new_named (anchor);
315 ev_action = ev_link_action_new_dest (dest);
316 } else if (doc == -1 && anchor &&
317 gxps_document_get_page_for_anchor (xps_document->doc, anchor) >= 0) {
318 /* Internal, but source is not a doc,
319 * let's try with doc = 0
320 */
321 dest = ev_link_dest_new_named (anchor);
322 ev_action = ev_link_action_new_dest (dest);
323 } else {
324 gchar *filename;
325
326 /* FIXME: remote uri? */
327 filename = g_file_get_path (xps_document->file);
328
329 if (anchor)
330 dest = ev_link_dest_new_named (anchor);
331 ev_action = ev_link_action_new_remote (dest, filename);
332 g_free (filename);
333 }
334 } else {
335 const gchar *uri;
336
337 uri = gxps_link_target_get_uri (target);
338 ev_action = ev_link_action_new_external_uri (uri);
339 }
340
341 return ev_link_new (NULL, ev_action);
342}

344static void
345build_tree (XPSDocument *xps_document,
346 GtkTreeModel *model,
347 GtkTreeIter *parent,
348 GXPSOutlineIter *iter)
349{
350 do {
351 GtkTreeIter tree_iter;
352 GXPSOutlineIter child_iter;
353 EvLink *link;
354 GXPSLinkTarget *target;
355 gchar *title;
356
357 target = gxps_outline_iter_get_target (iter);
358 title = g_markup_escape_text (gxps_outline_iter_get_description (iter), -1);
359 link = ev_link_from_target (xps_document, target);
360 gxps_link_target_free (target);
361
362 gtk_tree_store_append (GTK_TREE_STORE (model), &tree_iter, parent);
363 gtk_tree_store_set (GTK_TREE_STORE (model), &tree_iter,
364 EV_DOCUMENT_LINKS_COLUMN_MARKUP, title,
365 EV_DOCUMENT_LINKS_COLUMN_LINK, link,
366 EV_DOCUMENT_LINKS_COLUMN_EXPAND, FALSE,
367 -1);
368 g_object_unref (link);
369 g_free (title);
370
371 if (gxps_outline_iter_children (&child_iter, iter))
372 build_tree (xps_document, model, &tree_iter, &child_iter);
373 } while (gxps_outline_iter_next (iter));
374}

376static GtkTreeModel *
377xps_document_links_get_links_model (EvDocumentLinks *document_links)
378{
379 XPSDocument *xps_document = XPS_DOCUMENT (document_links);
380 GXPSDocumentStructure *structure;
381 GXPSOutlineIter iter;
382 GtkTreeModel *model = NULL;
383
384 structure = gxps_document_get_structure (xps_document->doc);
385 if (!structure)
386 return NULL;
387
388 if (gxps_document_structure_outline_iter_init (&iter, structure)) {
389 model = (GtkTreeModel *) gtk_tree_store_new (EV_DOCUMENT_LINKS_COLUMN_NUM_COLUMNS,
390 G_TYPE_STRING,
391 G_TYPE_OBJECT,
392 G_TYPE_BOOLEAN,
393 G_TYPE_STRING);
394 build_tree (xps_document, model, NULL, &iter);
395 }
396
397 g_object_unref (structure);
398
399 return model;
400}

402static EvMappingList *
403xps_document_links_get_links (EvDocumentLinks *document_links,
404 EvPage *page)
405{
406 XPSDocument *xps_document = XPS_DOCUMENT (document_links);
407 GXPSPage *xps_page;
408 GList *retval = NULL;
409 GList *mapping_list;
410 GList *list;
411
412 xps_page = GXPS_PAGE (page->backend_page);
413 mapping_list = gxps_page_get_links (xps_page, NULL);
414
415 for (list = mapping_list; list; list = list->next) {
416 GXPSLink *xps_link;
417 GXPSLinkTarget *target;
418 EvMapping *ev_link_mapping;
419 cairo_rectangle_t area;
420
421 xps_link = (GXPSLink *)list->data;
422 ev_link_mapping = g_new (EvMapping, 1);
423 gxps_link_get_area (xps_link, &area);
424 target = gxps_link_get_target (xps_link);
425 ev_link_mapping->data = ev_link_from_target (xps_document, target);
426
427 ev_link_mapping->area.x1 = area.x;
428 ev_link_mapping->area.x2 = area.x + area.width;
429 ev_link_mapping->area.y1 = area.y;
430 ev_link_mapping->area.y2 = area.y + area.height;
431
432 retval = g_list_prepend (retval, ev_link_mapping);
433 gxps_link_free (xps_link);
434 }
435
436 g_list_free (mapping_list);
437
438 return ev_mapping_list_new (page->index, g_list_reverse (retval), (GDestroyNotify)g_object_unref);
439}

441static EvLinkDest *
442xps_document_links_find_link_dest (EvDocumentLinks *document_links, const gchar *link_name)
443{
444 XPSDocument *xps_document = XPS_DOCUMENT (document_links);

447 cairo_rectangle_t area;
448 EvLinkDest *dest = NULL;
449

457
458 if (gxps_page_get_anchor_destination (xps_page, link_name, &area, NULL))
459 dest = ev_link_dest_new_xyz (page, area.x, area.y, 1., TRUE, TRUE, FALSE);
460
461 g_object_unref (xps_page);
462
463 return dest;
464}
