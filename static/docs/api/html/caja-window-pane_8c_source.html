39static inline CajaWindowSlot *
40get_first_inactive_slot (CajaWindowPane *pane)
41{
42 GList *l;
43 CajaWindowSlot *slot = NULL;
44
45 for (l = pane->slots; l != NULL; l = l->next)
46 {
47 slot = CAJA_WINDOW_SLOT (l->data);
48 if (slot != pane->active_slot)
49 {
50 return slot;
51 }
52 }
53
54 return NULL;
55}

130void
131caja_window_pane_slot_close (CajaWindowPane *pane, CajaWindowSlot *slot)
132{
133 if (pane->window)
134 {
135 CajaWindow *window;
136 window = pane->window;
137 if (pane->active_slot == slot)
138 {
139 g_assert (pane->active_slots != NULL);
140 g_assert (pane->active_slots->data == slot);
141
142 CajaWindowSlot *next_slot;
143
144 next_slot = NULL;
145 if (pane->active_slots->next != NULL)
146 {
147 next_slot = CAJA_WINDOW_SLOT (pane->active_slots->next->data);
148 }
149
150 if (next_slot == NULL)
151 {
152 next_slot = get_first_inactive_slot (CAJA_WINDOW_PANE (pane));
153 }
154
155 caja_window_set_active_slot (window, next_slot);
156 }
157 caja_window_close_slot (slot);
158
159 /* If that was the last slot in the active pane, close the pane or even the whole window. */
160 if (window->details->active_pane->slots == NULL)
161 {
162 CajaWindowPane *next_pane;
163 next_pane = caja_window_get_next_pane (window);
164
165 /* If next_pane is non-NULL, we have more than one pane available. In this
166 * case, close the current pane and switch to the next one. If there is
167 * no next pane, close the window. */
168 if(next_pane)
169 {
170 caja_window_close_pane (pane);
171 caja_window_pane_switch_to (next_pane);
172 if (CAJA_IS_NAVIGATION_WINDOW (window))
173 {
174 caja_navigation_window_update_show_hide_menu_items (CAJA_NAVIGATION_WINDOW (window));
175 }
176 }
177 else
178 {
179 caja_window_close (window);
180 }
181 }
182 }

240void
241caja_window_pane_set_active (CajaWindowPane *pane, gboolean is_active)
242{
243 if (is_active == pane->is_active)
244 {
245 return;
246 }
247
248 pane->is_active = is_active;
249
250 /* notify the current slot about its activity state (so that it can e.g. modify the bg color) */
251 caja_window_slot_is_in_active_pane (pane->active_slot, is_active);
252
253 EEL_CALL_METHOD (CAJA_WINDOW_PANE_CLASS, pane,
254 set_active, (pane, is_active));
255}
