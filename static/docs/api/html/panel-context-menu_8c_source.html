163static void
164panel_context_menu_setup_delete_panel_item (GtkWidget *menu,
165 GtkWidget *menuitem)
166{
167 PanelWidget *panel_widget;
168 gboolean sensitive;
169
170 panel_widget = menu_get_panel (menu);
171
172 g_assert (PANEL_IS_TOPLEVEL (panel_widget->toplevel));
173
174 sensitive =
175 !panel_toplevel_is_last_unattached (panel_widget->toplevel) &&
176 !panel_lockdown_get_locked_down () &&
177 panel_profile_id_lists_are_writable ();
178
179 gtk_widget_set_sensitive (menuitem, sensitive);

193static void
194query_panel_reset (PanelToplevel *toplevel)
195{
196 GtkWidget *dialog;
197 char *text1;
198 char *text2;
199
200 text1 = _("Reset all panels?");
201 text2 = _("When the panels are reset, all \n"
202 "custom settings are lost.");
203
204 dialog = gtk_message_dialog_new (
205 GTK_WINDOW (toplevel),
206 GTK_DIALOG_MODAL,
207 GTK_MESSAGE_WARNING,
208 GTK_BUTTONS_NONE,
209 "%s", text1);
210
211 gtk_message_dialog_format_secondary_text (GTK_MESSAGE_DIALOG (dialog),
212 "%s", text2);
213 gtk_dialog_add_buttons (GTK_DIALOG (dialog),
214 _("_Cancel"), GTK_RESPONSE_CANCEL,
215 _("_Reset Panels"), GTK_RESPONSE_OK,
216 NULL);
217
218 gtk_dialog_set_default_response (GTK_DIALOG (dialog), GTK_RESPONSE_CANCEL);
219
220 gtk_window_set_position (GTK_WINDOW (dialog), GTK_WIN_POS_CENTER);
221
222 g_signal_connect (dialog, "response",
223 G_CALLBACK (panel_reset_response),
224 NULL);
225
226 gtk_widget_show_all (dialog);

229static void
230panel_context_menu_build_edition (PanelWidget *panel_widget,
231 GtkWidget *menu)
232{
233 GtkWidget *menuitem;
234
235 gtk_menu_set_reserve_toggle_size (GTK_MENU (menu), FALSE);
236
237 menuitem = panel_image_menu_item_new_from_icon ("list-add", _("_Add to Panel…"));
238
239 gtk_widget_show (menuitem);
240 gtk_menu_shell_append (GTK_MENU_SHELL (menu), menuitem);
241 g_signal_connect (menuitem, "activate",
242 G_CALLBACK (panel_addto_present),
243 panel_widget);
244
245 if (!panel_profile_id_lists_are_writable ())
246 gtk_widget_set_sensitive (menuitem, FALSE);
247
248 menuitem = panel_image_menu_item_new_from_icon ("document-properties", _("_Properties"));
249
250 gtk_widget_show (menuitem);
251 gtk_menu_shell_append (GTK_MENU_SHELL (menu), menuitem);
252 g_signal_connect_swapped (menuitem, "activate",
253 G_CALLBACK (panel_properties_dialog_present),
254 panel_widget->toplevel);
255
256 add_menu_separator (menu);
257
258 menuitem = panel_image_menu_item_new_from_icon ("document-revert", _("_Reset All Panels"));
259
260 gtk_widget_show (menuitem);
261 gtk_menu_shell_append (GTK_MENU_SHELL (menu), menuitem);
262 g_signal_connect_swapped (menuitem, "activate",
263 G_CALLBACK (query_panel_reset), panel_widget->toplevel);
264
265 menuitem = panel_image_menu_item_new_from_icon ("edit-delete", _("_Delete This Panel"));
266
267 gtk_widget_show (menuitem);
268 gtk_menu_shell_append (GTK_MENU_SHELL (menu), menuitem);
269 g_signal_connect_swapped (menuitem, "activate",
270 G_CALLBACK (panel_context_menu_delete_panel),
271 panel_widget->toplevel);
272 g_signal_connect (menu, "show",
273 G_CALLBACK (panel_context_menu_setup_delete_panel_item),
274 menuitem);
275
276 add_menu_separator (menu);
277
278 menuitem = panel_image_menu_item_new_from_icon ("document-new", _("_New Panel"));
279
280 gtk_widget_show (menuitem);
281 gtk_menu_shell_append (GTK_MENU_SHELL (menu), menuitem);
282 g_signal_connect (menuitem, "activate",
283 G_CALLBACK (panel_context_menu_create_new_panel),
284 NULL);
285 gtk_widget_set_sensitive (menuitem,
286 panel_profile_id_lists_are_writable ());
287
288 add_menu_separator (menu);
289}

291GtkWidget *
292panel_context_menu_create (PanelWidget *panel)
293{
294 GtkWidget *retval;
295 GtkWidget *menuitem;
296
297 if (panel->master_widget) {
298 gpointer *pointer;
299 AppletInfo *info;
300
301 pointer = g_object_get_data (G_OBJECT (panel->master_widget),
302 "applet_info");
303
304 g_assert (pointer != NULL);
305 info = (AppletInfo *) pointer;
306
307 if (info->menu == NULL) {
308 info->menu = mate_panel_applet_create_menu (info);
309 }
310
311 return info->menu;
312 }
313
314 retval = create_empty_menu ();
315 gtk_widget_set_name (retval, "mate-panel-context-menu");
316
317 if (!panel_lockdown_get_locked_down ())
318 panel_context_menu_build_edition (panel, retval);
319
320 menuitem = panel_image_menu_item_new_from_icon ("help-browser", _("_Help"));
321
322 gtk_widget_show (menuitem);
323 gtk_menu_shell_append (GTK_MENU_SHELL (retval), menuitem);
324 g_signal_connect (menuitem, "activate",
325 G_CALLBACK (panel_context_menu_show_help), NULL);
326
327 menuitem = panel_image_menu_item_new_from_icon ("help-about", _("A_bout Panels"));
328
329 gtk_widget_show (menuitem);
330 gtk_menu_shell_append (GTK_MENU_SHELL (retval), menuitem);
331 g_signal_connect (menuitem, "activate",
332 G_CALLBACK (panel_context_menu_show_about_dialog),
333 NULL);
334
335 /* FIXME: can we get rid of this? (needed by menu_get_panel()) */
336 g_object_set_data (G_OBJECT (retval), "menu_panel", panel);
337
338/* Set up theme and transparency support */
339 GtkWidget *toplevel = gtk_widget_get_toplevel (retval);
340/* Fix any failures of compiz/other wm's to communicate with gtk for transparency */
341 GdkScreen *screen = gtk_widget_get_screen(GTK_WIDGET(toplevel));
342 GdkVisual *visual = gdk_screen_get_rgba_visual(screen);
343 gtk_widget_set_visual(GTK_WIDGET(toplevel), visual);
344/* Set menu and it's toplevel window to follow panel theme */
345 GtkStyleContext *context;
346 context = gtk_widget_get_style_context (GTK_WIDGET(toplevel));
347 gtk_style_context_add_class(context,"gnome-panel-menu-bar");
348 gtk_style_context_add_class(context,"mate-panel-menu-bar");
349
350 return retval;
351}
