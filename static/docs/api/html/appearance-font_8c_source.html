36/* X servers sometimes lie about the screen's physical dimensions, so we cannot
37 * compute an accurate DPI value. When this happens, the user gets fonts that
38 * are too huge or too tiny. So, we see what the server returns: if it reports
39 * something outside of the range [DPI_LOW_REASONABLE_VALUE,
40 * DPI_HIGH_REASONABLE_VALUE], then we assume that it is lying and we use
41 * DPI_FALLBACK instead.
42 *
43 * See get_dpi_from_mate_conf_or_server() below, and also
44 * https://bugzilla.novell.com/show_bug.cgi?id=217790
45 */
46#define DPI_FALLBACK 96
47#define DPI_LOW_REASONABLE_VALUE 50
48#define DPI_HIGH_REASONABLE_VALUE 500

52static void sample_draw(GtkWidget* darea, cairo_t* cr)
53{
54 cairo_surface_t* surface = g_object_get_data(G_OBJECT(darea), "sample-surface");
55 GtkAllocation allocation;
56 int x, y, w, h;
57
58 gtk_widget_get_allocation (darea, &allocation);
59 x = allocation.width;
60 y = allocation.height;
61 w = cairo_image_surface_get_width (surface);
62 h = cairo_image_surface_get_height (surface);
63
64 cairo_set_line_width (cr, 1);
65 cairo_set_line_cap (cr, CAIRO_LINE_CAP_SQUARE);
66
67 cairo_set_source_rgb (cr, 1.0, 1.0, 1.0);
68 cairo_rectangle (cr, 0, 0, x, y);
69 cairo_fill_preserve (cr);
70 cairo_set_source_rgb (cr, 0.0, 0.0, 0.0);
71 cairo_stroke (cr);
72
73 cairo_set_source_surface (cr, surface, (x - w) / 2, (y - h) / 2);
74
75 cairo_paint(cr);
76}

98static void set_fontoptions(PangoContext *context, Antialiasing antialiasing, Hinting hinting)
99{
100 cairo_font_options_t *opt;
101 cairo_antialias_t aa;
102 cairo_hint_style_t hs;
103
104 switch (antialiasing) {
105 case ANTIALIAS_NONE:
106 aa = CAIRO_ANTIALIAS_NONE;
107 break;
108 case ANTIALIAS_GRAYSCALE:
109 aa = CAIRO_ANTIALIAS_GRAY;
110 break;
111 case ANTIALIAS_RGBA:
112 aa = CAIRO_ANTIALIAS_SUBPIXEL;
113 break;
114 default:
115 aa = CAIRO_ANTIALIAS_DEFAULT;
116 break;
117 }
118
119 switch (hinting) {
120 case HINT_NONE:
121 hs = CAIRO_HINT_STYLE_NONE;
122 break;
123 case HINT_SLIGHT:
124 hs = CAIRO_HINT_STYLE_SLIGHT;
125 break;
126 case HINT_MEDIUM:
127 hs = CAIRO_HINT_STYLE_MEDIUM;
128 break;
129 case HINT_FULL:
130 hs = CAIRO_HINT_STYLE_FULL;
131 break;
132 default:
133 hs = CAIRO_HINT_STYLE_DEFAULT;
134 break;
135 }
136
137 opt = cairo_font_options_create ();
138 cairo_font_options_set_antialias (opt, aa);
139 cairo_font_options_set_hint_style (opt, hs);
140 pango_cairo_context_set_font_options (context, opt);
141 cairo_font_options_destroy (opt);
142}

144static void setup_font_sample(GtkWidget* darea, Antialiasing antialiasing, Hinting hinting)
145{
146 const char *str = "<span font=\"18\" style=\"normal\">abcfgop AO </span>"
147 "<span font=\"20\" style=\"italic\">abcfgop</span>";
148
149 PangoContext *context;
150 PangoLayout *layout;
151 PangoFontDescription *fd;
152 PangoRectangle extents;
153 cairo_surface_t *surface;
154 cairo_t *cr;
155 int width, height;
156
157 context = gtk_widget_get_pango_context (darea);
158 set_fontoptions (context, antialiasing, hinting);
159 layout = pango_layout_new (context);
160
161 fd = pango_font_description_from_string ("Serif");
162 pango_layout_set_font_description (layout, fd);
163 pango_font_description_free (fd);
164
165 pango_layout_set_markup (layout, str, -1);
166
167 pango_layout_get_extents (layout, NULL, &extents);
168 width = PANGO_PIXELS(extents.width) + 4;
169 height = PANGO_PIXELS(extents.height) + 2;
170
171 surface = cairo_image_surface_create (CAIRO_FORMAT_A8, width, height);
172 cr = cairo_create (surface);
173
174 cairo_move_to (cr, 2, 1);
175 pango_cairo_show_layout (cr, layout);
176 g_object_unref (layout);
177 cairo_destroy (cr);
178
179 g_object_set_data_full(G_OBJECT(darea), "sample-surface", surface, (GDestroyNotify) cairo_surface_destroy);
180
181 gtk_widget_set_size_request (GTK_WIDGET(darea), width + 2, height + 2);
182 g_signal_connect(darea, "draw", G_CALLBACK(sample_draw), NULL);
183}

186 * Code implementing a group of radio buttons with different cairo option combinations.
187 * If one of the buttons is matched by the GSettings key, we pick it. Otherwise we
188 * show the group as inconsistent.

191typedef struct {
192 Antialiasing antialiasing;
193 Hinting hinting;
194 GtkToggleButton *radio;
195} FontPair;
196
197static GSList *font_pairs = NULL;
198
199static void
200font_render_load (GSettings *settings)
201{
202 Antialiasing antialiasing;
203 Hinting hinting;
204 gboolean inconsistent = TRUE;
205 GSList *tmp_list;
206
207 antialiasing = g_settings_get_enum (settings, FONT_ANTIALIASING_KEY);
208 hinting = g_settings_get_enum (settings, FONT_HINTING_KEY);
209
210 in_change = TRUE;
211
212 for (tmp_list = font_pairs; tmp_list; tmp_list = tmp_list->next) {
213 FontPair *pair = tmp_list->data;
214
215 if (antialiasing == pair->antialiasing && hinting == pair->hinting) {
216 gtk_toggle_button_set_active (pair->radio, TRUE);
217 inconsistent = FALSE;
218 break;
219 }
220 }
221
222 for (tmp_list = font_pairs; tmp_list; tmp_list = tmp_list->next) {
223 FontPair *pair = tmp_list->data;
224
225 gtk_toggle_button_set_inconsistent (pair->radio, inconsistent);
226 }
227
228 in_change = FALSE;
229}

231static void
232font_render_changed (GSettings *settings,
233 gchar *key,
234 gpointer user_data)
235{
236 font_render_load (settings);
237}

239static void
240font_radio_toggled (GtkToggleButton *toggle_button,
241 FontPair *pair)
242{
243 if (!in_change) {
244 GSettings *settings = g_settings_new (FONT_RENDER_SCHEMA);
245
246 g_settings_set_enum (settings, FONT_ANTIALIASING_KEY, pair->antialiasing);
247 g_settings_set_enum (settings, FONT_HINTING_KEY, pair->hinting);
248
249 /* Restore back to the previous state until we get notification */
250 font_render_load (settings);
251 g_object_unref (settings);
252 }
253}

290/*
291 * EnumGroup - a group of radio buttons for a gsettings enum
292 */
293typedef struct
294{
295 GSettings *settings;
296 GSList *items;
297 gchar *settings_key;
298 gulong settings_signal_id;
299} EnumGroup;
300
301typedef struct
302{
303 EnumGroup *group;
304 GtkToggleButton *widget;
305 int value;
306} EnumItem;
307
308static void
309enum_group_load (EnumGroup *group)
310{
311 gint val = g_settings_get_enum (group->settings, group->settings_key);
312 GSList *tmp_list;
313
314 in_change = TRUE;
315
316 for (tmp_list = group->items; tmp_list; tmp_list = tmp_list->next) {
317 EnumItem *item = tmp_list->data;
318
319 if (val == item->value)
320 gtk_toggle_button_set_active (item->widget, TRUE);
321 }
322
323 in_change = FALSE;
324}
325
326static void
327enum_group_changed (GSettings *settings,
328 gchar *key,
329 gpointer user_data)
330{
331 enum_group_load (user_data);
332}
333
334static void
335enum_item_toggled (GtkToggleButton *toggle_button,
336 EnumItem *item)
337{
338 EnumGroup *group = item->group;
339
340 if (!in_change) {
341 g_settings_set_enum (group->settings, group->settings_key, item->value);
342 }
343
344 /* Restore back to the previous state until we get notification */
345 enum_group_load (group);
346}
347
348static EnumGroup *
349enum_group_create (GSettings *settings,
350 const gchar *settings_key,
351 GtkWidget *first_widget,
352 ...)
353{
354 EnumGroup *group;
355 GtkWidget *widget;
356 va_list args;
357
358 group = g_new (EnumGroup, 1);
359
360 group->settings = g_object_ref (settings);
361 group->settings_key = g_strdup (settings_key);
362 group->items = NULL;
363
364 va_start (args, first_widget);
365
366 widget = first_widget;
367 while (widget) {
368 EnumItem *item;
369
370 item = g_new (EnumItem, 1);
371 item->group = group;
372 item->widget = GTK_TOGGLE_BUTTON (widget);
373 item->value = va_arg (args, int);
374
375 g_signal_connect (item->widget, "toggled",
376 G_CALLBACK (enum_item_toggled), item);
377
378 group->items = g_slist_prepend (group->items, item);
379
380 widget = va_arg (args, GtkWidget *);
381 }
382
383 va_end (args);
384
385 enum_group_load (group);
386
387 gchar *signal_name = g_strdup_printf("changed::%s", settings_key);
388 group->settings_signal_id = g_signal_connect (settings, signal_name,
389 G_CALLBACK (enum_group_changed), group);
390 g_free (signal_name);
391
392 return group;
393}
394
395static void
396enum_group_destroy (gpointer data)
397{
398 EnumGroup *group = data;
399
400 g_signal_handler_disconnect (group->settings, group->settings_signal_id);
401 g_clear_object (&group->settings);
402 group->settings_signal_id = 0;
403 g_free (group->settings_key);
404
405 g_slist_free_full (group->items, g_free);
406
407 g_free (group);
408}

410static double
411dpi_from_pixels_and_mm (int pixels, int mm)
412{
413 double dpi;
414
415 if (mm >= 1)
416 dpi = pixels / (mm / 25.4);
417 else
418 dpi = 0;
419
420 return dpi;
421}

423static double
424get_dpi_from_x_server_or_monitor (void)
425{
426 GdkScreen *screen;
427 GdkDisplay *display;
428 double dpi;
429
430 screen = gdk_screen_get_default ();
431 display = gdk_display_get_default();
432
433 if (screen) {
434 double width_dpi, height_dpi;
435
436 if (GDK_IS_X11_DISPLAY (display))
437 {
438 Screen *xscreen = gdk_x11_screen_get_xscreen (screen);
439
440 width_dpi = dpi_from_pixels_and_mm (WidthOfScreen (xscreen), WidthMMOfScreen (xscreen));
441 height_dpi = dpi_from_pixels_and_mm (HeightOfScreen (xscreen), HeightMMOfScreen (xscreen));
442 }
443 else
444 {
445 GdkMonitor *monitor;
446 GdkRectangle geometry = {0};
447
448 /*FIXME: we have to use the main monitor for this
449 *which may not always be the leftmost monitor
450 *Separate per-monitor settings for this would be ideal
451 */
452 monitor = gdk_display_get_monitor (display, 0);
453 gdk_monitor_get_geometry (monitor, &geometry);
454
455 width_dpi = dpi_from_pixels_and_mm (geometry.width, gdk_monitor_get_width_mm(monitor));
456 height_dpi = dpi_from_pixels_and_mm (geometry.height, gdk_monitor_get_height_mm(monitor));
457 }
458
459 if (width_dpi < DPI_LOW_REASONABLE_VALUE || width_dpi > DPI_HIGH_REASONABLE_VALUE ||
460 height_dpi < DPI_LOW_REASONABLE_VALUE || height_dpi > DPI_HIGH_REASONABLE_VALUE)
461 dpi = DPI_FALLBACK;
462 else
463 dpi = (width_dpi + height_dpi) / 2.0;
464 } else {
465 /* Huh!? No screen? */
466 dpi = DPI_FALLBACK;
467 }
468
469 return dpi;
470}

473 * The font rendering details dialog
474 */
475static void
476dpi_load (GSettings *settings,
477 GtkSpinButton *spinner)
478{
479 GdkScreen *screen;
480 gint scale;
481 gdouble dpi;
482
483 screen = gdk_screen_get_default ();
484 scale = gdk_window_get_scale_factor (gdk_screen_get_root_window (screen));
485 dpi = g_settings_get_double (settings, FONT_DPI_KEY);
486
487 if (dpi == 0)
488 dpi = get_dpi_from_x_server_or_monitor ();
489
490 dpi *= (double)scale;
491 dpi = CLAMP(dpi, DPI_LOW_REASONABLE_VALUE, DPI_HIGH_REASONABLE_VALUE);
492
493 in_change = TRUE;
494 gtk_spin_button_set_value (spinner, dpi);
495 in_change = FALSE;
496}

526static void
527dpi_value_changed (GtkSpinButton *spinner,
528 AppearanceData *data)
529{
530 /* Like any time when using a spin button with GSettings, there is
531 * a race condition here. When we change, we send the new
532 * value to GSettings, then restore to the old value until
533 * we get a response to emulate the proper model/view behavior.
534 *
535 * If the user changes the value faster than responses are
536 * received from GSettings, this may cause mildly strange effects.
537 */
538 if (!in_change) {
539 GdkScreen *screen;
540 GtkWidget *toggle;
541 gint scale;
542 gdouble new_dpi;
543
544 screen = gdk_screen_get_default ();
545 scale = gdk_window_get_scale_factor (gdk_screen_get_root_window (screen));
546 new_dpi = gtk_spin_button_get_value (spinner) / (double)scale;
547
548 g_settings_set_double (data->font_settings, FONT_DPI_KEY, new_dpi);
549
550 dpi_load (data->font_settings, spinner);
551
552 toggle = appearance_capplet_get_widget (data, "dpi_reset_switch");
553 gtk_switch_set_active (GTK_SWITCH (toggle), FALSE);
554 }
555}

557static gboolean
558dpi_value_reset (GtkSwitch *toggle,
559 gboolean state,
560 AppearanceData *data)
561{
562 GtkWidget *spinner;
563 spinner = appearance_capplet_get_widget (data, "dpi_spinner");
564
565 if (state)
566 g_settings_set_double (data->font_settings, FONT_DPI_KEY, 0);
567 else
568 dpi_value_changed (GTK_SPIN_BUTTON (spinner), data);
569
570 gtk_switch_set_state(toggle, state);
571 gtk_widget_set_sensitive (spinner, !state);
572
573 return TRUE;
574}

586static void install_new_font (const gchar *filepath)
587{
588 GFile *src, *dst;
589 gchar *fontdir, *fontpath;
590 char *basename;
591 GError *error = NULL;
592
593 fontdir = g_build_filename (g_get_home_dir(), ".fonts", NULL);
594 if (!g_file_test (fontdir, G_FILE_TEST_IS_DIR))
595 {
596 if(g_mkdir (fontdir, 0755) != 0) {
597 g_free (fontdir);
598 return;
599 }
600 }
601 g_free (fontdir);
602
603 basename = g_path_get_basename (filepath);
604 fontpath = g_build_filename (g_get_home_dir(), ".fonts", basename, NULL);
605 g_free (basename);
606 src = g_file_new_for_path (filepath);
607 dst = g_file_new_for_path (fontpath);
608 g_free (fontpath);
609
610 if (!g_file_copy (src,
611 dst,
612 G_FILE_COPY_NONE,
613 NULL,
614 NULL,
615 NULL,
616 &error)) {
617 g_warning ("install new font failed: %s\n", error->message);
618 g_error_free (error);
619 }
620 g_object_unref (src);
621 g_object_unref (dst);
622}

625cb_add_new_font (GtkWidget *button,
626 AppearanceData *data)
627{
628 GtkWidget *dialog;
629 GtkFileFilter *filter;
630 GtkFileChooser *chooser;
631 GtkFileChooserAction action = GTK_FILE_CHOOSER_ACTION_OPEN;

655 }
656 gtk_widget_destroy (dialog);
657}

659static void
660cb_show_details (GtkWidget *button,
661 AppearanceData *data)
662{
663 if (!data->font_details) {
664 GtkAdjustment *adjustment;
665 GtkWidget *spinner;
666 GtkWidget *toggle;
667 EnumGroup *group;
668 gdouble dpi;
669
670 data->font_details = appearance_capplet_get_widget (data, "render_details");
671
672 gtk_window_set_transient_for (GTK_WINDOW (data->font_details),
673 GET_WINDOW ("appearance_window"));
674
675 spinner = appearance_capplet_get_widget (data, "dpi_spinner");
676 toggle = appearance_capplet_get_widget (data, "dpi_reset_switch");
677
678 /* Set initial state for widgets */
679 dpi = g_settings_get_double (data->font_settings, FONT_DPI_KEY);
680 gtk_switch_set_active (GTK_SWITCH (toggle), dpi == 0);
681 gtk_widget_set_sensitive (GTK_WIDGET (spinner), dpi != 0);
682
683 /* pick a sensible maximum dpi */
684 adjustment = gtk_spin_button_get_adjustment (GTK_SPIN_BUTTON (spinner));
685 gtk_adjustment_set_lower (adjustment, DPI_LOW_REASONABLE_VALUE);
686 gtk_adjustment_set_upper (adjustment, DPI_HIGH_REASONABLE_VALUE);
687 gtk_adjustment_set_step_increment (adjustment, 1);
688
689 dpi_load (data->font_settings, GTK_SPIN_BUTTON (spinner));
690 g_signal_connect (spinner, "value-changed",
691 G_CALLBACK (dpi_value_changed), data);
692 g_signal_connect (toggle, "state-set",
693 G_CALLBACK (dpi_value_reset), data);
694
695 g_signal_connect (data->font_settings, "changed::" FONT_DPI_KEY, G_CALLBACK (dpi_changed), data);
696
697 /* Update font DPI when window scaling factor is changed */
698 g_signal_connect (gdk_screen_get_default (), "monitors-changed", G_CALLBACK (monitors_changed), data);
699
700 setup_font_sample (appearance_capplet_get_widget (data, "antialias_none_sample"), ANTIALIAS_NONE, HINT_SLIGHT);
701 setup_font_sample (appearance_capplet_get_widget (data, "antialias_grayscale_sample"), ANTIALIAS_GRAYSCALE, HINT_SLIGHT);
702 setup_font_sample (appearance_capplet_get_widget (data, "antialias_subpixel_sample"), ANTIALIAS_RGBA, HINT_SLIGHT);
703
704 group = enum_group_create (
705 data->font_settings, FONT_ANTIALIASING_KEY,
706 appearance_capplet_get_widget (data, "antialias_none_radio"), ANTIALIAS_NONE,
707 appearance_capplet_get_widget (data, "antialias_grayscale_radio"), ANTIALIAS_GRAYSCALE,
708 appearance_capplet_get_widget (data, "antialias_subpixel_radio"), ANTIALIAS_RGBA,
709 NULL);
