89static GtkTreeModel *
90ev_sidebar_annotations_create_simple_model (const gchar *message)
91{
92 GtkTreeModel *retval;
93 GtkTreeIter iter;
94 gchar *markup;
95
96 /* Creates a fake model to indicate that we're loading */
97 retval = (GtkTreeModel *)gtk_list_store_new (N_COLUMNS,
98 G_TYPE_STRING,
99 GDK_TYPE_PIXBUF,
100 G_TYPE_POINTER);
101
102 gtk_list_store_append (GTK_LIST_STORE (retval), &iter);
103 markup = g_strdup_printf ("<span size=\"larger\" style=\"italic\">%s</span>",
104 message);
105 gtk_list_store_set (GTK_LIST_STORE (retval), &iter,
106 COLUMN_MARKUP, markup,
107 -1);
108 g_free (markup);
109
110 return retval;

113static void
114ev_sidebar_annotations_text_annot_button_toggled (GtkWidget *button,
115 EvSidebarAnnotations *sidebar_annots)
116{
117 EvAnnotationType annot_type;
118
119 if (!gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (button))) {
120 g_signal_emit (sidebar_annots, signals[ANNOT_ADD_CANCELLED], 0, NULL);
121 return;
122 }
123
124 if (button == sidebar_annots->priv->annot_text_item)
125 annot_type = EV_ANNOTATION_TYPE_TEXT;
126 else
127 annot_type = EV_ANNOTATION_TYPE_UNKNOWN;
128
129 g_signal_emit (sidebar_annots, signals[BEGIN_ANNOT_ADD], 0, annot_type);
130}

326static void
327job_finished_callback (EvJobAnnots *job,
328 EvSidebarAnnotations *sidebar_annots)
329{
330 EvSidebarAnnotationsPrivate *priv;
331 GtkTreeStore *model;
332 GtkTreeSelection *selection;
333 GList *l;
334 GdkPixbuf *text_icon = NULL;
335 GdkPixbuf *attachment_icon = NULL;
336
337 priv = sidebar_annots->priv;
338
339 if (!job->annots) {
340 GtkTreeModel *list;
341
342 list = ev_sidebar_annotations_create_simple_model (_("Document contains no annotations"));
343 gtk_tree_view_set_model (GTK_TREE_VIEW (priv->tree_view), list);
344 g_object_unref (list);
345
346 g_object_unref (job);
347 priv->job = NULL;
348
349 return;
350 }
351
352 selection = gtk_tree_view_get_selection (GTK_TREE_VIEW (priv->tree_view));
353 gtk_tree_selection_set_mode (selection, GTK_SELECTION_SINGLE);
354 if (priv->selection_changed_id == 0) {
355 priv->selection_changed_id =
356 g_signal_connect (selection, "changed",
357 G_CALLBACK (selection_changed_cb),
358 sidebar_annots);
359 }
360
361 model = gtk_tree_store_new (N_COLUMNS,
362 G_TYPE_STRING,
363 GDK_TYPE_PIXBUF,
364 G_TYPE_POINTER);
365
366 for (l = job->annots; l; l = g_list_next (l)) {
367 EvMappingList *mapping_list;
368 GList *ll;
369 gchar *page_label;
370 GtkTreeIter iter;
371 gboolean found = FALSE;
372
373 mapping_list = (EvMappingList *)l->data;
374 page_label = g_strdup_printf (_("Page %d"),
375 ev_mapping_list_get_page (mapping_list) + 1);
376 gtk_tree_store_append (model, &iter, NULL);
377 gtk_tree_store_set (model, &iter,
378 COLUMN_MARKUP, page_label,
379 -1);
380 g_free (page_label);
381
382 for (ll = ev_mapping_list_get_list (mapping_list); ll; ll = g_list_next (ll)) {
383 EvAnnotation *annot;
384 const gchar *label;
385 const gchar *modified;
386 gchar *markup;
387 GtkTreeIter child_iter;
388 GdkPixbuf *pixbuf = NULL;
389 GtkIconTheme *icon_theme;
390
391 annot = ((EvMapping *)(ll->data))->data;
392 if (!EV_IS_ANNOTATION_MARKUP (annot))
393 continue;
394
395 label = ev_annotation_markup_get_label (EV_ANNOTATION_MARKUP (annot));
396 modified = ev_annotation_get_modified (annot);
397 if (modified) {
398 markup = g_strdup_printf ("<span weight=\"bold\">%s</span>\n%s",
399 label, modified);
400 } else {
401 markup = g_strdup_printf ("<span weight=\"bold\">%s</span>", label);
402 }
403
404 icon_theme = gtk_icon_theme_get_default();
405
406 if (EV_IS_ANNOTATION_TEXT (annot)) {
407 if (!text_icon) {
408 text_icon = gtk_icon_theme_load_icon (icon_theme,
409 "starred",
410 GTK_ICON_SIZE_BUTTON,
411 GTK_ICON_LOOKUP_FORCE_REGULAR,
412 NULL);
413 }
414 pixbuf = text_icon;
415 } else if (EV_IS_ANNOTATION_ATTACHMENT (annot)) {
416 if (!attachment_icon) {
417 attachment_icon = gtk_icon_theme_load_icon (icon_theme,
418 "mail-attachment",
419 GTK_ICON_SIZE_BUTTON,
420 GTK_ICON_LOOKUP_FORCE_REGULAR,
421 NULL);
422 }
423 pixbuf = attachment_icon;
424 }
425
426 gtk_tree_store_append (model, &child_iter, &iter);
427 gtk_tree_store_set (model, &child_iter,
428 COLUMN_MARKUP, markup,
429 COLUMN_ICON, pixbuf,
430 COLUMN_ANNOT_MAPPING, ll->data,
431 -1);
432 g_free (markup);
433 found = TRUE;
434 }
435
436 if (!found)
437 gtk_tree_store_remove (model, &iter);
438 }
439
440 gtk_tree_view_set_model (GTK_TREE_VIEW (priv->tree_view),
441 GTK_TREE_MODEL (model));
442 g_object_unref (model);
443
444 if (text_icon)
445 g_object_unref (text_icon);
446 if (attachment_icon)
447 g_object_unref (attachment_icon);
448
449 g_object_unref (job);
450 priv->job = NULL;
451}

453static void
454ev_sidebar_annotations_load (EvSidebarAnnotations *sidebar_annots)
455{
456 EvSidebarAnnotationsPrivate *priv = sidebar_annots->priv;
457
458 if (priv->job) {
459 g_signal_handlers_disconnect_by_func (priv->job,
460 job_finished_callback,
461 sidebar_annots);
462 g_object_unref (priv->job);
463 }
464
465 priv->job = ev_job_annots_new (priv->document);
466 g_signal_connect (priv->job, "finished",
467 G_CALLBACK (job_finished_callback),
468 sidebar_annots);
469 /* The priority doesn't matter for this job */
470 ev_job_scheduler_push_job (priv->job, EV_JOB_PRIORITY_NONE);
471}

473static void
474ev_sidebar_annotations_document_changed_cb (EvDocumentModel *model,
475 GParamSpec *pspec,
476 EvSidebarAnnotations *sidebar_annots)
477{
478 EvDocument *document = ev_document_model_get_document (model);
479 EvSidebarAnnotationsPrivate *priv = sidebar_annots->priv;
480 gboolean enable_add;
481
482 if (!EV_IS_DOCUMENT_ANNOTATIONS (document))
483 return;
484
485 if (priv->document)
486 g_object_unref (priv->document);
487 priv->document = g_object_ref (document);
488
489 enable_add = ev_document_annotations_can_add_annotation (EV_DOCUMENT_ANNOTATIONS (document));
490 gtk_widget_set_sensitive (priv->annot_text_item, enable_add);
491
492 ev_sidebar_annotations_load (sidebar_annots);
493}
