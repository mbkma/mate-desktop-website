191static gboolean
192oss_backend_open (MateMixerBackend *backend)
193{
194 OssBackend *oss;
195
196 g_return_val_if_fail (OSS_IS_BACKEND (backend), FALSE);
197
198 oss = OSS_BACKEND (backend);
199
200 /* Discover added or removed OSS devices every second */
201 oss->priv->timeout_source = g_timeout_source_new_seconds (1);
202 g_source_set_callback (oss->priv->timeout_source,
203 (GSourceFunc) read_devices,
204 oss,
205 NULL);
206 g_source_attach (oss->priv->timeout_source,
207 g_main_context_get_thread_default ());
208
209 /* Read the initial list of devices so we have some starting point, there
210 * isn't really a way to detect errors here, failing to add a device may
211 * be a device-related problem so make the backend always open successfully */
212 read_devices (oss);
213
214 _mate_mixer_backend_set_state (backend, MATE_MIXER_STATE_READY);
215 return TRUE;
216}

218void
219oss_backend_close (MateMixerBackend *backend)
220{
221 OssBackend *oss;
222
223 g_return_if_fail (OSS_IS_BACKEND (backend));
224
225 oss = OSS_BACKEND (backend);
226
227 g_source_destroy (oss->priv->timeout_source);
228
229 if (oss->priv->devices != NULL) {
230 g_list_free_full (oss->priv->devices, g_object_unref);
231 oss->priv->devices = NULL;
232 }
233 if (oss->priv->default_device != NULL) {
234 g_free (oss->priv->default_device);
235 oss->priv->default_device = NULL;
236 }
237
238 free_stream_list (oss);
239
240 g_hash_table_remove_all (oss->priv->devices_paths);
241
242 _mate_mixer_backend_set_state (backend, MATE_MIXER_STATE_IDLE);
243}

253static const GList *
254oss_backend_list_streams (MateMixerBackend *backend)
255{
256 OssBackend *oss;
257
258 g_return_val_if_fail (OSS_IS_BACKEND (backend), NULL);
259
260 oss = OSS_BACKEND (backend);
261
262 if (oss->priv->streams == NULL) {
263 GList *list;
264
265 /* Walk through the list of devices and create the stream list, each
266 * device has at most one input and one output stream */
267 list = g_list_last (oss->priv->devices);
268
269 while (list != NULL) {
270 OssDevice *device = OSS_DEVICE (list->data);
271 OssStream *stream;
272
273 stream = oss_device_get_output_stream (device);
274 if (stream != NULL)
275 oss->priv->streams =
276 g_list_prepend (oss->priv->streams, g_object_ref (stream));
277
278 stream = oss_device_get_input_stream (device);
279 if (stream != NULL)
280 oss->priv->streams =
281 g_list_prepend (oss->priv->streams, g_object_ref (stream));
282
283 list = list->prev;
284 }
285 }
286 return oss->priv->streams;
287}

289static gboolean
290read_devices (OssBackend *oss)
291{
292 gint i;
293 gboolean added = FALSE;
294
295 for (i = 0; i < OSS_MAX_DEVICES; i++) {
296 gchar *path;
297 gboolean added_current;
298
299 path = g_strdup_printf ("/dev/mixer%i", i);
300
301 /* On recent FreeBSD both /dev/mixer and /dev/mixer0 point to the same
302 * mixer device, on NetBSD and OpenBSD /dev/mixer is a symlink to one
303 * of the real mixer device nodes, on Linux /dev/mixer is the first
304 * device and /dev/mixer1 is the second device.
305 * Handle all of these cases by trying /dev/mixer if /dev/mixer0 fails */
306 if (read_device (oss, path, &added_current) == FALSE && i == 0)
307 read_device (oss, "/dev/mixer", &added_current);
308
309 if (added_current == TRUE)
310 added = TRUE;
311
312 g_free (path);
313 }
314
315 /* If any card has been added, make sure we have the most suitable default
316 * input and output streams */
317 if (added == TRUE) {
318 select_default_input_stream (oss);
319 select_default_output_stream (oss);
320 }
321 return G_SOURCE_CONTINUE;
322}

324static gboolean
325read_device (OssBackend *oss, const gchar *path, gboolean *added)
326{
327 OssDevice *device;
328 gint fd;
329 gchar *bname;
330 gchar *label;
331
332 *added = FALSE;
333
334 fd = g_open (path, O_RDWR, 0);
335 if (fd == -1) {
336 if (errno != ENOENT && errno != ENXIO)
337 g_debug ("%s: %s", path, g_strerror (errno));
338
339 remove_device_by_path (oss, path);
340 return FALSE;
341 }
342
343 /* Don't proceed if the device is already known, opening the device was
344 * still tested to be absolutely sure that the device is removed it case
345 * it has disappeared, but normally the device's polling facility should
346 * handle this by itself */
347 if (g_hash_table_contains (oss->priv->devices_paths, path) == TRUE) {
348 close (fd);
349 return TRUE;
350 }
351
352 bname = g_path_get_basename (path);
353 label = read_device_label (oss, path, fd);
354
355 device = oss_device_new (bname, label, path, fd);
356 g_free (bname);
357 g_free (label);
358
359 close (fd);
360
361 if (G_LIKELY (device != NULL)) {
362 *added = oss_device_open (device);
363 if (*added == TRUE)
364 add_device (oss, device);
365 else
366 g_object_unref (device);
367 }
368 return *added;
369}

371static gchar *
372read_device_label (OssBackend *oss, const gchar *path, gint fd)
373{
374 guint index;
375
376#ifdef SOUND_MIXER_INFO
377 do {
378 struct mixer_info info;
379
380 /* Prefer device name supplied by the system, but this calls fails
381 * with EINVAL on FreeBSD */
382 if (ioctl (fd, SOUND_MIXER_INFO, &info) == 0)
383 return g_strdup (info.name);
384 } while (0);
385#endif
386
387 index = (guint) g_ascii_strtoull (path + sizeof ("/dev/mixer") - 1,
388 NULL,
389 10);
390#ifdef OSS_PATH_SNDSTAT
391 /* If the ioctl doesn't succeed, assume that the mixer device number
392 * matches the pcm number in the sndstat file, this is a bit desperate, but
393 * it should be correct on FreeBSD */
394 do {
395 gchar *label;
396
397 label = read_device_label_sndstat (oss, OSS_PATH_SNDSTAT, path, index);
398 if (label != NULL)
399 return label;
400 } while (0);
401#endif
402
403 return g_strdup_printf (_("OSS Mixer %d"), index);
404}

406static gchar *
407read_device_label_sndstat (OssBackend *oss,
408 const gchar *sndstat,
409 const gchar *path,
410 guint index)
411{
412 FILE *fp;
413 gchar *label = NULL;
414 gchar *prefix;
415 gchar line[512];
416
417 fp = g_fopen (sndstat, "r");
418 if (fp == NULL) {
419 g_debug ("Failed to open %s: %s", sndstat, g_strerror (errno));
420 return NULL;
421 }
422
423 /* Example line:
424 * pcm0: <ATI R6xx (HDMI)> (play) default */
425 prefix = g_strdup_printf ("pcm%u: ", index);
426
427 while (fgets (line, sizeof (line), fp) != NULL) {
428 gchar *p;
429
430 if (g_str_has_prefix (line, prefix) == FALSE)
431 continue;
432
433 p = strchr (line, '<');
434 if (p != NULL && *p && *(++p)) {
435 gchar *end = strchr (p, '>');
436
437 if (end != NULL) {
438 label = g_strndup (p, end - p);
439
440 /* Normally the default OSS device is /dev/dsp, but on FreeBSD
441 * /dev/dsp doesn't physically exist on the filesystem, but is
442 * managed by the kernel according to the user-settable default
443 * device, in sndstat the default card definition ends with the
444 * word "default" */
445 if (g_str_has_suffix (line, "default")) {
446 g_free (oss->priv->default_device);
447
448 oss->priv->default_device = g_strdup (path);
449 }
450 } else {
451 g_debug ("Failed to read sndstat line: %s", line);
452 }
453 break;
454 }
455 }
456
457 g_free (prefix);
458 fclose (fp);
459
460 return label;
461}

463static void
464add_device (OssBackend *oss, OssDevice *device)
465{
466 /* Takes reference of device */
467 oss->priv->devices =
468 g_list_insert_sorted_with_data (oss->priv->devices,
469 device,
470 (GCompareDataFunc) compare_devices,
471 NULL);
472
473 /* Keep track of added device paths */
474 g_hash_table_add (oss->priv->devices_paths,
475 g_strdup (oss_device_get_path (device)));
476
477 g_signal_connect_swapped (G_OBJECT (device),
478 "closed",
479 G_CALLBACK (remove_device),
480 oss);
481 g_signal_connect_swapped (G_OBJECT (device),
482 "stream-removed",
483 G_CALLBACK (remove_stream),
484 oss);
485
486 g_signal_connect_swapped (G_OBJECT (device),
487 "closed",
488 G_CALLBACK (free_stream_list),
489 oss);
490 g_signal_connect_swapped (G_OBJECT (device),
491 "stream-added",
492 G_CALLBACK (free_stream_list),
493 oss);
494 g_signal_connect_swapped (G_OBJECT (device),
495 "stream-removed",
496 G_CALLBACK (free_stream_list),
497 oss);
498
499 g_signal_emit_by_name (G_OBJECT (oss),
500 "device-added",
501 mate_mixer_device_get_name (MATE_MIXER_DEVICE (device)));
502
503 /* Load the device elements after emitting device-added, because the load
504 * function will most likely emit stream-added on the device and backend */
505 oss_device_load (device);
506}

528static void
529remove_device_by_list_item (OssBackend *oss, GList *item)
530{
531 OssDevice *device;
532 const gchar *path;
533
534 device = OSS_DEVICE (item->data);
535
536 g_signal_handlers_disconnect_by_func (G_OBJECT (device),
537 G_CALLBACK (remove_device),
538 oss);
539
540 /* May emit removed signals */
541 if (oss_device_is_open (device) == TRUE)
542 oss_device_close (device);
543
544 g_signal_handlers_disconnect_by_data (G_OBJECT (device),
545 oss);
546
547 oss->priv->devices = g_list_delete_link (oss->priv->devices, item);
548
549 path = oss_device_get_path (device);
550 g_hash_table_remove (oss->priv->devices_paths, path);
551
552 if (g_strcmp0 (oss->priv->default_device, path) == 0) {
553 g_free (oss->priv->default_device);
554 oss->priv->default_device = NULL;
555 }
556
557 /* The list may have been invalidated by device signals */
558 free_stream_list (oss);
559
560 g_signal_emit_by_name (G_OBJECT (oss),
561 "device-removed",
562 mate_mixer_device_get_name (MATE_MIXER_DEVICE (device)));
563
564 g_object_unref (device);
565}

567static void
568remove_stream (OssBackend *oss, const gchar *name)
569{
570 MateMixerStream *stream;
571
572 stream = mate_mixer_backend_get_default_input_stream (MATE_MIXER_BACKEND (oss));
573
574 if (stream != NULL && strcmp (mate_mixer_stream_get_name (stream), name) == 0)
575 select_default_input_stream (oss);
576
577 stream = mate_mixer_backend_get_default_output_stream (MATE_MIXER_BACKEND (oss));
578
579 if (stream != NULL && strcmp (mate_mixer_stream_get_name (stream), name) == 0)
580 select_default_output_stream (oss);
581}

600static void
601select_default_input_stream (OssBackend *oss)
602{
603 OssDevice *device;
604 OssStream *stream;
605 GList *list;
606
607 device = get_default_device (oss);
608 if (device != NULL) {
609 stream = oss_device_get_input_stream (device);
610 if (stream != NULL) {
611 _mate_mixer_backend_set_default_input_stream (MATE_MIXER_BACKEND (oss),
612 MATE_MIXER_STREAM (stream));
613 return;
614 }
615 }
616
617 list = oss->priv->devices;
618 while (list != NULL) {
619 device = OSS_DEVICE (list->data);
620 stream = oss_device_get_input_stream (device);
621
622 if (stream != NULL) {
623 _mate_mixer_backend_set_default_input_stream (MATE_MIXER_BACKEND (oss),
624 MATE_MIXER_STREAM (stream));
625 return;
626 }
627 list = list->next;
628 }
629
630 /* In the worst case unset the default stream */
631 _mate_mixer_backend_set_default_input_stream (MATE_MIXER_BACKEND (oss), NULL);
632}
