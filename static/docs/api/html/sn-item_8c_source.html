158static void
159sn_item_get_action_coordinates (SnItem *item,
160 gint *x,
161 gint *y)
162{
163 GtkWidget *widget;
164 SnItemPrivate *priv;
165 GdkWindow *window;
166 GtkWidget *toplevel;
167 gint width;
168 gint height;
169
170 priv = SN_ITEM (item)->priv;
171 widget = GTK_WIDGET (item);
172 window = gtk_widget_get_window (widget);
173 toplevel = gtk_widget_get_toplevel (widget);
174
175 gdk_window_get_geometry (window, x, y, &width, &height);
176 gtk_widget_translate_coordinates (widget, toplevel, *x, *y, x, y);
177
178 if (priv->orientation == GTK_ORIENTATION_HORIZONTAL)
179 *y += height;
180 else
181 *x += width;
182}

184static gboolean
185sn_item_button_press_event (GtkWidget *widget,
186 GdkEventButton *event)
187{
188 SnItem *item;
189 SnItemPrivate *priv;
190 gint x;
191 gint y;
192
193 if (event->button < 2 || event->button > 3)
194 return GTK_WIDGET_CLASS (sn_item_parent_class)->button_press_event (widget, event);
195
196 item = SN_ITEM (widget);
197 priv = SN_ITEM (item)->priv;
198
199 sn_item_get_action_coordinates (item, &x, &y);
200
201 if (event->button == 2)
202 {
203 gdk_seat_ungrab (gdk_device_get_seat (event->device));
204 SN_ITEM_GET_CLASS (item)->secondary_activate (item, x, y);
205 }
206 else if (event->button == 3)
207 {
208 if (priv->menu != NULL)
209 {
210 gtk_menu_popup_at_widget (priv->menu, widget,
211 GDK_GRAVITY_SOUTH_WEST,
212 GDK_GRAVITY_NORTH_WEST,
213 (GdkEvent *) event);
214 /*Fix positioning if size changed since last shown*/
215 gtk_menu_reposition(priv->menu);
216 }
217 else
218 {
219 gdk_seat_ungrab (gdk_device_get_seat (event->device));
220 SN_ITEM_GET_CLASS (item)->context_menu (item, x, y);
221 }
222 }
223 else
224 {
225 g_assert_not_reached ();
226 }
227
228 return GTK_WIDGET_CLASS (sn_item_parent_class)->button_press_event (widget, event);
229}

276static gboolean
277sn_item_scroll_event (GtkWidget *widget,
278 GdkEventScroll *event)
279{
280 SnItem *item;
281 GdkScrollDirection direction;
282 SnItemOrientation orientation;
283 gdouble dx;
284 gdouble dy;
285 gint delta;
286
287 item = SN_ITEM (widget);
288
289 if (!gdk_event_get_scroll_direction ((GdkEvent *) event, &direction))
290 {
291 g_assert_not_reached ();
292 }
293 else
294 {
295 switch (direction)
296 {
297 case GDK_SCROLL_UP:
298 case GDK_SCROLL_DOWN:
299 orientation = SN_ITEM_ORIENTATION_VERTICAL;
300 break;
301
302 case GDK_SCROLL_LEFT:
303 case GDK_SCROLL_RIGHT:
304 orientation = SN_ITEM_ORIENTATION_HORIZONTAL;
305 break;
306
307 case GDK_SCROLL_SMOOTH:
308 default:
309 g_assert_not_reached ();
310 break;
311 }
312 }
313
314 if (!gdk_event_get_scroll_deltas ((GdkEvent *) event, &dx, &dy))
315 {
316 switch (direction)
317 {
318 case GDK_SCROLL_UP:
319 case GDK_SCROLL_LEFT:
320 delta = 1;
321 break;
322
323 case GDK_SCROLL_DOWN:
324 case GDK_SCROLL_RIGHT:
325 delta = -1;
326 break;
327
328 case GDK_SCROLL_SMOOTH:
329 default:
330 g_assert_not_reached ();
331 break;
332 }
333 }
334 else
335 {
336 if (dy != 0)
337 delta = (gint) dy;
338 else
339 delta = (gint) dx;
340 }
341
342 SN_ITEM_GET_CLASS (item)->scroll (item, delta, orientation);
343
344 return GDK_EVENT_STOP;
345}

347static void
348sn_item_ready (SnItem *item)
349{
350 const gchar *menu;
351 SnItemPrivate *priv;
352
353 menu = SN_ITEM_GET_CLASS (item)->get_menu (item);
354
355 if (menu == NULL || *menu == '\0' || g_strcmp0 (menu, "/") == 0)
356 return;
357
358 priv = SN_ITEM (item)->priv;
359 priv->menu = sn_dbus_menu_new (priv->bus_name, menu);
360 g_object_ref_sink (priv->menu);
361 gtk_button_set_always_show_image (GTK_BUTTON (item),TRUE);
362}

370static NaItemCategory
371sn_item_get_category (NaItem *item)
372{
373 const gchar *string;
374 NaItemCategory category;
375
376 string = SN_ITEM_GET_CLASS (item)->get_category (SN_ITEM (item));
377
378 if (g_strcmp0 (string, "Hardware") == 0)
379 category = NA_ITEM_CATEGORY_HARDWARE;
380 else if (g_strcmp0 (string, "SystemServices") == 0)
381 category = NA_ITEM_CATEGORY_SYSTEM_SERVICES;
382 else if (g_strcmp0 (string, "Communications") == 0)
383 category = NA_ITEM_CATEGORY_COMMUNICATIONS;
384 else
385 category = NA_ITEM_CATEGORY_APPLICATION_STATUS;
386
387 return category;
388}
