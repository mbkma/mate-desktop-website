98GType
99caja_window_info_get_type (void)
100{
101 static GType type = 0;
102
103 if (!type)
104 {
105 const GTypeInfo info =
106 {
107 sizeof (CajaWindowInfoIface),
108 caja_window_info_base_init,
109 NULL,
110 NULL,
111 NULL,
112 NULL,
113 0,
114 0,
115 NULL,
116 NULL
117 };
118
119 type = g_type_register_static (G_TYPE_INTERFACE,
120 "CajaWindowInfo",
121 &info, 0);
122 g_type_interface_add_prerequisite (type, G_TYPE_OBJECT);
123 }
124
125 return type;
