59static GdkFilterReturn
60property_notify_filter (GdkXEvent *xevent,
61 GdkEvent *event,
62 gpointer data)
63{
64 XPropertyEvent *ev = xevent;
65
66 if (ev->type == PropertyNotify && ev->atom == is_accelerated_atom) {
67 property_changed = TRUE;
68 gtk_main_quit ();
69 }
70
71 return GDK_FILTER_CONTINUE;
72}

74static gboolean
75wait_for_property_notify (void)
76{
77 GdkDisplay *display;
78 GdkScreen *screen;
79 GdkWindow *root;
80 Window rootwin;
81
82 property_changed = FALSE;
83
84 display = gdk_display_get_default ();
85 screen = gdk_display_get_default_screen (display);
86 root = gdk_screen_get_root_window (screen);
87 rootwin = gdk_x11_window_get_xid (root);
88
89 XSelectInput (GDK_DISPLAY_XDISPLAY (display), rootwin, PropertyChangeMask);
90 gdk_window_add_filter (root, property_notify_filter, NULL);
91 g_timeout_add (PROPERTY_CHANGE_TIMEOUT, on_property_notify_timeout, NULL);
92
93 gtk_main ();
94
95 return property_changed;
96}

98static char *
99get_gtk_gles_renderer (void)
100{
101 GtkWidget *win;
102 GdkGLContext *context;
103 char *renderer = NULL;
104
105 win = gtk_window_new (GTK_WINDOW_TOPLEVEL);
106 gtk_widget_realize (win);
107 context = gdk_window_create_gl_context (gtk_widget_get_window (win), NULL);
108 if (!context)
109 return NULL;
110 gdk_gl_context_make_current (context);
111 renderer = g_strdup ((char *) glGetString (GL_RENDERER));
112 gdk_gl_context_clear_current ();
113 g_object_unref (context);
114
115 return renderer;
116}

131int
132main (int argc, char **argv)
133{
134 GdkDisplay *display = NULL;
135 int estatus;
136 char *gl_helper_argv[] = { LIBEXECDIR "/mate-session-check-accelerated-gl-helper", "--print-renderer", NULL };
137#ifdef HAVE_GLESV2
138 char *gles_helper_argv[] = { LIBEXECDIR "/mate-session-check-accelerated-gles-helper", "--print-renderer", NULL };
139 GError *gles_error = NULL;
140 char *gles_renderer_string = NULL;
141#endif
142 char *renderer_string = NULL;
143 char *gl_renderer_string = NULL;
144 gboolean gl_software_rendering = FALSE, gles_software_rendering = FALSE;
145 Window rootwin;
146 glong is_accelerated, is_software_rendering;
147 GError *gl_error = NULL;
148
149 gtk_init (NULL, NULL);
150
151 /* mate-session-check-accelerated gets run before X is started in the wayland
152 * case, and it currently requires X. Until we have that working, just always
153 * assume wayland will work.
154 * Also make sure that we don't read cached information about the first GPU
155 * when requesting information about the second.
156 */
157 if (is_discrete_gpu_check () || g_strcmp0 (g_getenv ("XDG_SESSION_TYPE"), "x11") != 0) {
158 renderer_string = get_gtk_gles_renderer ();
159 if (renderer_string) {
160 g_print ("%s", renderer_string);
161 g_free (renderer_string);
162 return 0;
163 }
164 return 1;
165 }
166
167 display = gdk_display_get_default ();
168 /* when running on X11 with a nested wayland GDK will default to wayland
169 * so looking for X11 atoms will not work (and crash).
170 */
171 if (!GDK_IS_X11_DISPLAY (display)) {
172 g_printerr ("mate-session-check-accelerated: no X11 display found\n");
173 return 1;
174 }
175
176 rootwin = gdk_x11_get_default_root_xwindow ();
177
178 is_accelerated_atom = gdk_x11_get_xatom_by_name_for_display (display, "_GNOME_SESSION_ACCELERATED");
179 is_software_rendering_atom = gdk_x11_get_xatom_by_name_for_display (display, "_GNOME_IS_SOFTWARE_RENDERING");
180 renderer_atom = gdk_x11_get_xatom_by_name_for_display (display, "_GNOME_SESSION_RENDERER");
181
182 {
183 Atom type;
184 gint format;
185 gulong nitems;
186 gulong bytes_after;
187 guchar *data;
188
189 read:
190 gdk_x11_display_error_trap_push (display);
191 XGetWindowProperty (GDK_DISPLAY_XDISPLAY (display), rootwin,
192 is_accelerated_atom,
193 0, G_MAXLONG, False, XA_CARDINAL, &type, &format, &nitems,
194 &bytes_after, &data);
195 gdk_x11_display_error_trap_pop_ignored (display);
196
197 if (type == XA_CARDINAL) {
198 glong *is_accelerated_ptr = (glong*) data;
199
200 if (*is_accelerated_ptr == ACCEL_CHECK_RUNNING) {
201 /* Test in progress, wait */
202 if (wait_for_property_notify ())
203 goto read;
204 /* else fall through and do the check ourselves */
205
206 } else {
207 gdk_x11_display_error_trap_push (display);
208 XGetWindowProperty (GDK_DISPLAY_XDISPLAY (display), rootwin,
209 renderer_atom,
210 0, G_MAXLONG, False, XA_STRING, &type, &format, &nitems,
211 &bytes_after, &data);
212 gdk_x11_display_error_trap_pop_ignored (display);
213
214 if (type == XA_STRING) {
215 g_print ("%s", data);
216 }
217
218 return (*is_accelerated_ptr == 0 ? 1 : 0);
219 }
220 }
221 }
222
223 /* We don't have the property or it's the wrong type.
224 * Try to compute it now.
225 */
226
227 /* First indicate that a test is in progress */
228 is_accelerated = ACCEL_CHECK_RUNNING;
229 is_software_rendering = FALSE;
230 estatus = 1;
231
232 XChangeProperty (GDK_DISPLAY_XDISPLAY (display),
233 rootwin,
234 is_accelerated_atom,
235 XA_CARDINAL, 32, PropModeReplace, (guchar *) &is_accelerated, 1);
236
237 gdk_display_sync (display);
238
239 /* First, try the GL helper */
240 if (g_spawn_sync (NULL, (char **) gl_helper_argv, NULL, 0,
241 NULL, NULL, &gl_renderer_string, NULL, &estatus, &gl_error)) {
242 is_accelerated = (WEXITSTATUS(estatus) == HELPER_ACCEL);
243 gl_software_rendering = (WEXITSTATUS(estatus) == HELPER_SOFTWARE_RENDERING);
244 if (is_accelerated) {
245 renderer_string = gl_renderer_string;
246 goto finish;
247 }
248
249 g_printerr ("mate-session-check-accelerated: GL Helper exited with code %d\n", estatus);
250 }
251
252#ifdef HAVE_GLESV2
253 /* Then, try the GLES helper */
254 if (g_spawn_sync (NULL, (char **) gles_helper_argv, NULL, 0,
255 NULL, NULL, &gles_renderer_string, NULL, &estatus, &gles_error)) {
256 is_accelerated = (WEXITSTATUS(estatus) == HELPER_ACCEL);
257 gles_software_rendering = (WEXITSTATUS(estatus) == HELPER_SOFTWARE_RENDERING);
258 if (is_accelerated) {
259 renderer_string = gles_renderer_string;
260 goto finish;
261 }
262
263 g_printerr ("mate-session-check-accelerated: GLES Helper exited with code %d\n", estatus);
264 }
265#endif
266
267 /* If we got here, GL software rendering is our best bet */
268 if (gl_software_rendering || gles_software_rendering) {
269 is_software_rendering = TRUE;
270 is_accelerated = TRUE;
271
272 if (gl_software_rendering)
273 renderer_string = gl_renderer_string;
274#ifdef HAVE_GLESV2
275 else if (gles_software_rendering)
276 renderer_string = gles_renderer_string;
277#endif
278
279 goto finish;
280 }
281
282 /* Both helpers failed; print their error messages */
283 if (gl_error != NULL) {
284 g_printerr ("mate-session-check-accelerated: Failed to run GL helper: %s\n", gl_error->message);
285 g_clear_error (&gl_error);
286 }
287
288#ifdef HAVE_GLESV2
289 if (gles_error != NULL) {
290 g_printerr ("mate-session-check-accelerated: Failed to run GLES helper: %s\n", gles_error->message);
291 g_clear_error (&gles_error);
292 }
293#endif
294
295 finish:
296 if (is_accelerated) {
297 XChangeProperty (GDK_DISPLAY_XDISPLAY (display),
298 rootwin,
299 is_accelerated_atom,
300 XA_CARDINAL, 32, PropModeReplace, (guchar *) &is_accelerated, 1);
301 }
302
303 if (is_software_rendering) {
304 XChangeProperty (GDK_DISPLAY_XDISPLAY (display),
305 rootwin,
306 is_software_rendering_atom,
307 XA_CARDINAL, 32, PropModeReplace, (guchar *) &is_software_rendering, 1);
308 }
309
310 if (renderer_string != NULL) {
311 XChangeProperty (GDK_DISPLAY_XDISPLAY (display),
312 rootwin,
313 renderer_atom,
314 XA_STRING, 8, PropModeReplace, (guchar *) renderer_string, strlen (renderer_string));
315
316 /* Print the renderer */
317 g_print ("%s", renderer_string);
318 }
319
320 gdk_display_sync (display);
321
322 g_free (gl_renderer_string);
323#ifdef HAVE_GLESV2
324 g_free (gles_renderer_string);
325#endif
326
327 return is_accelerated ? 0 : 1;
328}
