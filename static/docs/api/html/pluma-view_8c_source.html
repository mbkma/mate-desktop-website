210static gboolean
211pluma_view_scroll_event (GtkWidget *widget,
212 GdkEventScroll *event)
213{
214 if (event->direction == GDK_SCROLL_UP)
215 {
216 event->delta_x = 0;
217 event->delta_y = -1;
218 }
219 else if (event->direction == GDK_SCROLL_DOWN)
220 {
221 event->delta_x = 0;
222 event->delta_y = 1;
223 }
224 else if (event->direction == GDK_SCROLL_LEFT)
225 {
226 event->delta_x = -1;
227 event->delta_y = 0;
228 }
229 else if (event->direction == GDK_SCROLL_RIGHT)
230 {
231 event->delta_x = 1;
232 event->delta_y = 0;
233 }
234
235 event->direction = GDK_SCROLL_SMOOTH;
236
237 return FALSE;
238}

240static void
241pluma_view_class_init (PlumaViewClass *klass)
242{
243 GObjectClass *object_class = G_OBJECT_CLASS (klass);
244 GtkWidgetClass *widget_class = GTK_WIDGET_CLASS (klass);
245 GtkTextViewClass *text_view_class = GTK_TEXT_VIEW_CLASS (klass);
246
247 GtkBindingSet *binding_set;
248
249 object_class->dispose = pluma_view_dispose;
250 object_class->finalize = pluma_view_finalize;
251
252 widget_class->focus_out_event = pluma_view_focus_out;
253 widget_class->draw = pluma_view_draw;
254 widget_class->scroll_event = pluma_view_scroll_event;
255
256 /*
257 * Override the gtk_text_view_drag_motion and drag_drop
258 * functions to get URIs
259 *
260 * If the mime type is text/uri-list, then we will accept
261 * the potential drop, or request the data (depending on the
262 * function).
263 *
264 * If the drag context has any other mime type, then pass the
265 * information onto the GtkTextView's standard handlers.
266 * (widget_class->function_name).
267 *
268 * See bug #89881 for details
269 */
270 widget_class->drag_motion = pluma_view_drag_motion;
271 widget_class->drag_data_received = pluma_view_drag_data_received;
272 widget_class->drag_drop = pluma_view_drag_drop;
273 widget_class->button_press_event = pluma_view_button_press_event;
274 widget_class->button_release_event = pluma_view_button_release_event;
275
276 widget_class->realize = pluma_view_realize;
277 widget_class->unrealize = pluma_view_unrealize;
278
279 text_view_class->populate_popup = pluma_view_populate_popup;
280 klass->start_interactive_search = start_interactive_search;
281 klass->start_interactive_goto_line = start_interactive_goto_line;
282 klass->reset_searched_text = reset_searched_text;
283
284 text_view_class->delete_from_cursor = pluma_view_delete_from_cursor;
285
286 view_signals[START_INTERACTIVE_SEARCH] =
287 g_signal_new ("start_interactive_search",
288 G_TYPE_FROM_CLASS (object_class),
289 G_SIGNAL_RUN_LAST | G_SIGNAL_ACTION,
290 G_STRUCT_OFFSET (PlumaViewClass, start_interactive_search),

292 G_TYPE_BOOLEAN, 0);

294 view_signals[START_INTERACTIVE_GOTO_LINE] =
295 g_signal_new ("start_interactive_goto_line",

297 G_SIGNAL_RUN_LAST | G_SIGNAL_ACTION,
298 G_STRUCT_OFFSET (PlumaViewClass, start_interactive_goto_line),

301
302 view_signals[RESET_SEARCHED_TEXT] =
303 g_signal_new ("reset_searched_text",
304 G_TYPE_FROM_CLASS (object_class),
305 G_SIGNAL_RUN_LAST | G_SIGNAL_ACTION,
306 G_STRUCT_OFFSET (PlumaViewClass, reset_searched_text),
307 NULL, NULL, NULL,
308 G_TYPE_BOOLEAN, 0);
309
310 /* A new signal DROP_URIS has been added to allow plugins to intercept
311 * the default dnd behaviour of 'text/uri-list'. PlumaView now handles
312 * dnd in the default handlers of drag_drop, drag_motion and
313 * drag_data_received. The view emits drop_uris from drag_data_received
314 * if valid uris have been dropped. Plugins should connect to
315 * drag_motion, drag_drop and drag_data_received to change this
316 * default behaviour. They should _NOT_ use this signal because this
317 * will not prevent pluma from loading the uri
318 */
319 view_signals[DROP_URIS] =
320 g_signal_new ("drop_uris",
321 G_TYPE_FROM_CLASS (object_class),
322 G_SIGNAL_RUN_LAST | G_SIGNAL_ACTION,
323 G_STRUCT_OFFSET (PlumaViewClass, drop_uris),
324 NULL, NULL, NULL,
325 G_TYPE_NONE, 1, G_TYPE_STRV);
326
327 binding_set = gtk_binding_set_by_class (klass);
328
329 gtk_binding_entry_add_signal (binding_set,
330 GDK_KEY_k,
331 GDK_CONTROL_MASK,
332 "start_interactive_search", 0);
333
334 gtk_binding_entry_add_signal (binding_set,
335 GDK_KEY_i,
336 GDK_CONTROL_MASK,
337 "start_interactive_goto_line", 0);
338

342 "reset_searched_text", 0);

349 G_TYPE_INT, 1);
350

352 GDK_KEY_u,
353 GDK_CONTROL_MASK,
354 "change_case", 1,
355 G_TYPE_ENUM, GTK_SOURCE_CHANGE_CASE_UPPER);
356 gtk_binding_entry_add_signal (binding_set,
357 GDK_KEY_l,
358 GDK_CONTROL_MASK,
359 "change_case", 1,

390static void
391on_notify_buffer_cb (PlumaView *view,
392 GParamSpec *arg1,
393 gpointer userdata)
394{
395 GtkTextBuffer *buffer;
396
397 current_buffer_removed (view);
398 buffer = gtk_text_view_get_buffer (GTK_TEXT_VIEW (view));
399
400 if (buffer == NULL || !PLUMA_IS_DOCUMENT (buffer))
401 return;
402
403 view->priv->current_buffer = g_object_ref (buffer);
404 g_signal_connect (buffer,
405 "notify::read-only",
406 G_CALLBACK (document_read_only_notify_handler),
407 view);
408
409 gtk_text_view_set_editable (GTK_TEXT_VIEW (view),
410 !pluma_document_get_readonly (PLUMA_DOCUMENT (buffer)));
411
412 g_signal_connect (buffer,
413 "search_highlight_updated",
414 G_CALLBACK (search_highlight_updated_cb),
415 view);
416}

419void
420pluma_set_source_space_drawer_by_level (GtkSourceView *view,
421 gint level,
422 GtkSourceSpaceTypeFlags type)
423{
424 GtkSourceSpaceLocationFlags locs[] = {GTK_SOURCE_SPACE_LOCATION_LEADING,
425 GTK_SOURCE_SPACE_LOCATION_INSIDE_TEXT,
426 GTK_SOURCE_SPACE_LOCATION_TRAILING};
427 /* this array links the level to the location */
428 GtkSourceSpaceLocationFlags levels[] =
429 {0,
430 GTK_SOURCE_SPACE_LOCATION_TRAILING,
431 GTK_SOURCE_SPACE_LOCATION_INSIDE_TEXT |
432 GTK_SOURCE_SPACE_LOCATION_TRAILING |
433 GTK_SOURCE_SPACE_LOCATION_LEADING
434 };
435
436 gint i;
437
438 GtkSourceSpaceDrawer *drawer = gtk_source_view_get_space_drawer (view);
439
440 if (level >= G_N_ELEMENTS(levels) || level < 0)
441 level = 0;
442
443 for (i = 0 ; i < G_N_ELEMENTS(locs) ; i++ ) {
444 GtkSourceSpaceTypeFlags f;
445 f = gtk_source_space_drawer_get_types_for_locations (drawer,
446 locs[i]);
447 if (locs[i] & levels[level])
448 f |= type;
449 else
450 f &= ~type;
451 gtk_source_space_drawer_set_types_for_locations (drawer, locs[i], f);
452 }
453}
454#endif

499static void
500pluma_view_init (PlumaView *view)
501{
502 PlumaSettings *settings;
503 GtkTargetList *tl;
504 gboolean use_default_font;
505 gboolean display_line_numbers;
506 gboolean auto_indent;
507 gboolean insert_spaces;
508 gboolean display_right_margin;
509 gboolean hl_current_line;
510 guint tabs_size;
511 guint right_margin_position;
512 GtkWrapMode wrap_mode;
513 GtkSourceSmartHomeEndType smart_home_end;
514 GtkStyleContext *context;
515
516 pluma_debug (DEBUG_VIEW);
517
518 view->priv = pluma_view_get_instance_private (view);
519
520 settings = _pluma_settings_get_singleton ();
521 view->priv->editor_settings = g_settings_new (PLUMA_SCHEMA_ID);
522 view->priv->interface_settings = g_settings_new ("org.mate.interface");
523
524 view->priv->css_provider = gtk_css_provider_new ();
525 context = gtk_widget_get_style_context (GTK_WIDGET (view));
526 gtk_style_context_add_class (context, "pluma-view");
527
528 gtk_style_context_add_provider (context,
529 GTK_STYLE_PROVIDER (view->priv->css_provider),
530 GTK_STYLE_PROVIDER_PRIORITY_APPLICATION);
531
532
533 /* Get setting values */
534 use_default_font = g_settings_get_boolean (view->priv->editor_settings,
535 PLUMA_SETTINGS_USE_DEFAULT_FONT);
536
537 /*
538 * Set tab, fonts, wrap mode, colors, etc. according
539 * to preferences
540 */
541 if (!use_default_font)
542 {
543 gchar *editor_font;
544
545 editor_font = g_settings_get_string (view->priv->editor_settings,
546 PLUMA_SETTINGS_EDITOR_FONT);
547
548 pluma_view_set_font (view, FALSE, editor_font);
549
550 g_free (editor_font);
551 }
552 else
553 {
554 pluma_view_set_font (view, TRUE, NULL);
555 }
556
557 display_line_numbers = g_settings_get_boolean (view->priv->editor_settings, PLUMA_SETTINGS_DISPLAY_LINE_NUMBERS);
558 auto_indent = g_settings_get_boolean (view->priv->editor_settings, PLUMA_SETTINGS_AUTO_INDENT);
559 tabs_size = g_settings_get_uint (view->priv->editor_settings, PLUMA_SETTINGS_TABS_SIZE);
560 insert_spaces = g_settings_get_boolean (view->priv->editor_settings, PLUMA_SETTINGS_INSERT_SPACES);
561 display_right_margin = g_settings_get_boolean (view->priv->editor_settings, PLUMA_SETTINGS_DISPLAY_RIGHT_MARGIN);
562 right_margin_position = g_settings_get_uint (view->priv->editor_settings, PLUMA_SETTINGS_RIGHT_MARGIN_POSITION);
563 hl_current_line = g_settings_get_boolean (view->priv->editor_settings, PLUMA_SETTINGS_HIGHLIGHT_CURRENT_LINE);
564
565 wrap_mode = pluma_settings_get_wrap_mode (view->priv->editor_settings, PLUMA_SETTINGS_WRAP_MODE);
566
567 smart_home_end = pluma_settings_get_smart_home_end (settings);
568
569 g_object_set (G_OBJECT (view),
570 "wrap_mode", wrap_mode,
571 "show_line_numbers", display_line_numbers,
572 "auto_indent", auto_indent,
573 "tab_width", tabs_size,
574 "insert_spaces_instead_of_tabs", insert_spaces,
575 "show_right_margin", display_right_margin,
576 "right_margin_position", right_margin_position,
577 "highlight_current_line", hl_current_line,
578 "smart_home_end", smart_home_end,
579 "indent_on_tab", TRUE,
580 NULL);
581
582 g_settings_bind (view->priv->editor_settings,
583 PLUMA_SETTINGS_BACKGROUND_PATTERN,
584 view,
585 "background-pattern",
586 G_SETTINGS_BIND_DEFAULT);
587
588 pluma_set_source_space_drawer (view->priv->editor_settings, GTK_SOURCE_VIEW (view));
589
590 view->priv->typeselect_flush_timeout = 0;
591 view->priv->wrap_around = TRUE;
592
593 /* Drag and drop support */
594 tl = gtk_drag_dest_get_target_list (GTK_WIDGET (view));
595
596 if (tl != NULL)
597 gtk_target_list_add_uri_targets (tl, TARGET_URI_LIST);
598
599 view->priv->extensions =
600 peas_extension_set_new (PEAS_ENGINE (pluma_plugins_engine_get_default ()),
601 PLUMA_TYPE_VIEW_ACTIVATABLE,
602 "view", view,
603 NULL);
604
605
606 /* Act on buffer change */
607 g_signal_connect (view,
608 "notify::buffer",
609 G_CALLBACK (on_notify_buffer_cb),
610 NULL);
611}

613static void
614pluma_view_dispose (GObject *object)
615{
616 PlumaView *view;
617
618 view = PLUMA_VIEW (object);
619
620 if (view->priv->extensions != NULL)
621 {
622 g_object_unref (view->priv->extensions);
623 view->priv->extensions = NULL;
624 }
625
626 if (view->priv->search_window != NULL)
627 {
628 gtk_widget_destroy (view->priv->search_window);
629 view->priv->search_window = NULL;
630 view->priv->search_entry = NULL;
631
632 if (view->priv->typeselect_flush_timeout != 0)
633 {
634 g_source_remove (view->priv->typeselect_flush_timeout);
635 view->priv->typeselect_flush_timeout = 0;
636 }
637 }
638
639 g_clear_object (&view->priv->editor_settings);
640 g_clear_object (&view->priv->interface_settings);
641
642 /* Disconnect notify buffer because the destroy of the textview will
643 set the buffer to NULL, and we call get_buffer in the notify which
644 would reinstate a GtkTextBuffer which we don't want */
645 current_buffer_removed (view);
646 g_signal_handlers_disconnect_by_func (view, on_notify_buffer_cb, NULL);
647
648 g_clear_object (&view->priv->css_provider);
649 g_clear_pointer (&view->priv->font_desc, pango_font_description_free);
650
651 (* G_OBJECT_CLASS (pluma_view_parent_class)->dispose) (object);
652}

1006static void
1007add_search_completion_entry (const gchar *str)
1008{
1009 gchar *text;
1010 gboolean valid;
1011 GtkTreeModel *model;
1012 GtkTreeIter iter;
1013
1014 if (str == NULL)
1015 return;
1016
1017 text = pluma_utils_unescape_search_text (str);
1018
1019 if (g_utf8_strlen (text, -1) < MIN_SEARCH_COMPLETION_KEY_LEN)
1020 {
1021 g_free (text);
1022 return;
1023 }
1024
1025 g_return_if_fail (GTK_IS_TREE_MODEL (search_completion_model));
1026
1027 model = GTK_TREE_MODEL (search_completion_model);
1028
1029 /* Get the first iter in the list */
1030 valid = gtk_tree_model_get_iter_first (model, &iter);
1031
1032 while (valid)
1033 {
1034 /* Walk through the list, reading each row */
1035 gchar *str_data;
1036
1037 gtk_tree_model_get (model,
1038 &iter,
1039 0,
1040 &str_data,
1041 -1);
1042
1043 if (strcmp (text, str_data) == 0)
1044 {
1045 g_free (text);
1046 g_free (str_data);
1047 gtk_list_store_move_after (GTK_LIST_STORE (model),
1048 &iter,
1049 NULL);
1050
1051 return;
1052 }
1053
1054 g_free (str_data);
1055
1056 valid = gtk_tree_model_iter_next (model, &iter);
1057 }
1058
1059 gtk_list_store_prepend (GTK_LIST_STORE (model), &iter);
1060 gtk_list_store_set (GTK_LIST_STORE (model),
1061 &iter,
1062 0,
1063 text,
1064 -1);
1065
1066 g_free (text);
1067}

1085static gboolean
1086run_search (PlumaView *view,
1087 const gchar *entry_text,
1088 gboolean search_backward,
1089 gboolean wrap_around,
1090 gboolean typing)
1091{
1092 GtkTextIter start_iter;
1093 GtkTextIter match_start;
1094 GtkTextIter match_end;
1095 gboolean found = FALSE;
1096 PlumaDocument *doc;
1097
1098 g_return_val_if_fail (view->priv->search_mode == SEARCH, FALSE);
1099
1100 doc = PLUMA_DOCUMENT (gtk_text_view_get_buffer (GTK_TEXT_VIEW (view)));
1101
1102 start_iter = view->priv->start_search_iter;
1103
1104 if (*entry_text != '\0')
1105 {
1106 if (!search_backward)
1107 {
1108 if (!typing)
1109 {
1110 /* forward and _NOT_ typing */
1111 gtk_text_buffer_get_selection_bounds (GTK_TEXT_BUFFER (doc),
1112 &start_iter,
1113 &match_end);
1114
1115 gtk_text_iter_order (&match_end, &start_iter);
1116 }
1117
1118 /* run search */
1119 found = pluma_document_search_forward (doc,
1120 &start_iter,
1121 NULL,
1122 &match_start,
1123 &match_end);
1124 }
1125 else if (!typing)
1126 {
1127 /* backward and not typing */
1128 gtk_text_buffer_get_selection_bounds (GTK_TEXT_BUFFER (doc),
1129 &start_iter,
1130 &match_end);
1131
1132 /* run search */
1133 found = pluma_document_search_backward (doc,
1134 NULL,
1135 &start_iter,
1136 &match_start,
1137 &match_end);
1138 }
1139 else
1140 {
1141 /* backward (while typing) */
1142 g_return_val_if_reached (FALSE);
1143
1144 }
1145
1146 if (!found && wrap_around)
1147 {
1148 if (!search_backward)
1149 found = pluma_document_search_forward (doc,
1150 NULL,
1151 NULL, /* FIXME: set the end_inter */
1152 &match_start,
1153 &match_end);
1154 else
1155 found = pluma_document_search_backward (doc,
1156 NULL, /* FIXME: set the start_inter */
1157 NULL,
1158 &match_start,
1159 &match_end);
1160 }
1161 }
1162 else
1163 {
1164 gtk_text_buffer_get_selection_bounds (GTK_TEXT_BUFFER (doc),
1165 &start_iter,
1166 NULL);
1167 }
1168
1169 if (found)
1170 {
1171 gtk_text_buffer_place_cursor (GTK_TEXT_BUFFER (doc), &match_start);
1172
1173 gtk_text_buffer_move_mark_by_name (GTK_TEXT_BUFFER (doc),
1174 "selection_bound",
1175 &match_end);
1176 }
1177 else
1178 {
1179 if (typing)
1180 {
1181 gtk_text_buffer_place_cursor (GTK_TEXT_BUFFER (doc),
1182 &view->priv->start_search_iter);
1183 }
1184 }
1185
1186 if (found || (*entry_text == '\0'))
1187 {
1188 pluma_view_scroll_to_cursor (view);
1189
1190 set_entry_state (view->priv->search_entry,
1191 PLUMA_SEARCH_ENTRY_NORMAL);
1192 }
1193 else
1194 {
1195 set_entry_state (view->priv->search_entry,
1196 PLUMA_SEARCH_ENTRY_NOT_FOUND);
1197 }
1198
1199 return found;
1200}

1369static gboolean
1370search_window_key_press_event (GtkWidget *widget,
1371 GdkEventKey *event,
1372 PlumaView *view)
1373{
1374 gboolean retval = FALSE;
1375 guint modifiers;
1376
1377 modifiers = gtk_accelerator_get_default_mod_mask ();
1378
1379 /* Close window */
1380 if (event->keyval == GDK_KEY_Tab)
1381 {
1382 hide_search_window (view, FALSE);
1383 retval = TRUE;
1384 }
1385
1386 /* Close window and cancel the search */
1387 if (event->keyval == GDK_KEY_Escape)
1388 {
1389 if (view->priv->search_mode == SEARCH)
1390 {
1391 PlumaDocument *doc;
1392
1393 /* restore document search so that Find Next does the right thing */
1394 doc = PLUMA_DOCUMENT (gtk_text_view_get_buffer (GTK_TEXT_VIEW (view)));
1395 pluma_document_set_search_text (doc,
1396 view->priv->old_search_text,
1397 view->priv->old_search_flags);
1398
1399 }
1400
1401 hide_search_window (view, TRUE);
1402 retval = TRUE;
1403 }
1404
1405 if (view->priv->search_mode == GOTO_LINE)
1406 return retval;
1407
1408 /* SEARCH mode */
1409
1410 /* select previous matching iter */
1411 if (event->keyval == GDK_KEY_Up || event->keyval == GDK_KEY_KP_Up)
1412 {
1413 search_again (view, TRUE);
1414 retval = TRUE;
1415 }
1416
1417 if (((event->state & modifiers) == (GDK_CONTROL_MASK | GDK_SHIFT_MASK)) &&
1418 (event->keyval == GDK_KEY_g || event->keyval == GDK_KEY_G))
1419 {
1420 search_again (view, TRUE);
1421 retval = TRUE;
1422 }
1423
1424 /* select next matching iter */
1425 if (event->keyval == GDK_KEY_Down || event->keyval == GDK_KEY_KP_Down)
1426 {
1427 search_again (view, FALSE);
1428 retval = TRUE;
1429 }
1430
1431 if (((event->state & modifiers) == GDK_CONTROL_MASK) &&
1432 (event->keyval == GDK_KEY_g || event->keyval == GDK_KEY_G))
1433 {
1434 search_again (view, FALSE);
1435 retval = TRUE;
1436 }
1437
1438 return retval;
1439}

1565static void
1566search_entry_insert_text (GtkEditable *editable,
1567 const gchar *text,
1568 gint length,
1569 gint *position,
1570 PlumaView *view)
1571{
1572 if (view->priv->search_mode == GOTO_LINE)
1573 {
1574 gunichar c;
1575 const gchar *p;
1576 const gchar *end;
1577 const gchar *next;
1578
1579 p = text;
1580 end = text + length;
1581
1582 if (p == end)
1583 return;
1584
1585 c = g_utf8_get_char (p);
1586
1587 if (((c == '-' || c == '+') && *position == 0) ||
1588 (c == ':' && *position != 0))
1589 {
1590 gchar *s = NULL;
1591
1592 if (c == ':')
1593 {
1594 s = gtk_editable_get_chars (editable, 0, -1);
1595 s = g_utf8_strchr (s, -1, ':');
1596 }
1597
1598 if (s == NULL || s == p)
1599 {
1600 next = g_utf8_next_char (p);
1601 p = next;
1602 }
1603
1604 g_free (s);
1605 }
1606
1607 while (p != end)
1608 {
1609 next = g_utf8_next_char (p);
1610
1611 c = g_utf8_get_char (p);
1612
1613 if (!g_unichar_isdigit (c)) {
1614 g_signal_stop_emission_by_name (editable, "insert_text");
1615 gtk_widget_error_bell (view->priv->search_entry);
1616 break;
1617 }
1618
1619 p = next;
1620 }
1621 }
1622 else
1623 {
1624 /* SEARCH mode */
1625 static gboolean insert_text = FALSE;
1626 gchar *escaped_text;
1627 gint new_len;
1628
1629 pluma_debug_message (DEBUG_SEARCH, "Text: %s", text);
1630
1631 /* To avoid recursive behavior */
1632 if (insert_text)
1633 return;
1634
1635 escaped_text = pluma_utils_escape_search_text (text);
1636
1637 pluma_debug_message (DEBUG_SEARCH, "Escaped Text: %s", escaped_text);
1638
1639 new_len = strlen (escaped_text);
1640
1641 if (new_len == length)
1642 {
1643 g_free (escaped_text);
1644 return;
1645 }
1646
1647 insert_text = TRUE;
1648
1649 g_signal_stop_emission_by_name (editable, "insert_text");
1650
1651 gtk_editable_insert_text (editable, escaped_text, new_len, position);
1652
1653 insert_text = FALSE;
1654
1655 g_free (escaped_text);
1656 }
1657}

1682static gboolean
1683completion_func (GtkEntryCompletion *completion,
1684 const char *key,
1685 GtkTreeIter *iter,
1686 gpointer data)
1687{
1688 gchar *item = NULL;
1689 gboolean ret = FALSE;
1690 GtkTreeModel *model;
1691 PlumaViewPrivate *priv = (PlumaViewPrivate *)data;
1692 const gchar *real_key;
1693
1694 if (priv->search_mode == GOTO_LINE)
1695 return FALSE;
1696
1697 real_key = gtk_entry_get_text (GTK_ENTRY (gtk_entry_completion_get_entry (completion)));
1698
1699 if (g_utf8_strlen (real_key, -1) <= MIN_SEARCH_COMPLETION_KEY_LEN)
1700 return FALSE;
1701
1702 model = gtk_entry_completion_get_model (completion);
1703 g_return_val_if_fail (gtk_tree_model_get_column_type (model, 0) == G_TYPE_STRING,
1704 FALSE);
1705
1706 gtk_tree_model_get (model,
1707 iter,
1708 0,
1709 &item,
1710 -1);
1711
1712 if (item == NULL)
1713 return FALSE;
1714
1715 if (PLUMA_SEARCH_IS_CASE_SENSITIVE (priv->search_flags))
1716 {
1717 if (!strncmp (real_key, item, strlen (real_key)))
1718 ret = TRUE;
1719 }
1720 else
1721 {
1722 gchar *normalized_string;
1723 gchar *case_normalized_string;
1724
1725 normalized_string = g_utf8_normalize (item, -1, G_NORMALIZE_ALL);
1726 case_normalized_string = g_utf8_casefold (normalized_string, -1);
1727
1728 if (!strncmp (key, case_normalized_string, strlen (key)))
1729 ret = TRUE;
1730
1731 g_free (normalized_string);
1732 g_free (case_normalized_string);
1733
1734 }
1735
1736 g_free (item);
1737
1738 return ret;

1941static void
1942search_init (GtkWidget *entry,
1943 PlumaView *view)
1944{
1945 PlumaDocument *doc;
1946 const gchar *entry_text;
1947
1948 /* renew the flush timeout */
1949 if (view->priv->typeselect_flush_timeout != 0)
1950 {
1951 g_source_remove (view->priv->typeselect_flush_timeout);
1952 view->priv->typeselect_flush_timeout =
1953 g_timeout_add (PLUMA_VIEW_SEARCH_DIALOG_TIMEOUT,
1954 (GSourceFunc)search_entry_flush_timeout,
1955 view);
1956 }
1957
1958 doc = PLUMA_DOCUMENT (gtk_text_view_get_buffer (GTK_TEXT_VIEW (view)));
1959
1960 entry_text = gtk_entry_get_text (GTK_ENTRY (entry));
1961
1962 if (view->priv->search_mode == SEARCH)
1963 {
1964 gchar *search_text;
1965 guint search_flags;
1966
1967 search_text = pluma_document_get_search_text (doc, &search_flags);
1968
1969 if ((search_text == NULL) ||
1970 (strcmp (search_text, entry_text) != 0) ||
1971 search_flags != view->priv->search_flags)
1972 {
1973 pluma_document_set_search_text (doc,
1974 entry_text,
1975 view->priv->search_flags);
1976 }
1977
1978 g_free (search_text);
1979
1980 run_search (view,
1981 entry_text,
1982 FALSE,
1983 view->priv->wrap_around,
1984 TRUE);
1985 }
1986 else
1987 {
1988 if (*entry_text != '\0')
1989 {
1990 gboolean moved, moved_offset;
1991 gint line;
1992 gint offset_line = 0;
1993 gint line_offset = 0;
1994 gchar **split_text = NULL;
1995 const gchar *text;
1996
1997 split_text = g_strsplit (entry_text, ":", -1);
1998
1999 if (g_strv_length (split_text) > 1)
2000 {
2001 text = split_text[0];
2002 }
2003 else
2004 {
2005 text = entry_text;
2006 }
2007
2008 if (*text == '-')
2009 {
2010 gint cur_line = gtk_text_iter_get_line (&view->priv->start_search_iter);
2011
2012 if (*(text + 1) != '\0')
2013 offset_line = MAX (atoi (text + 1), 0);
2014
2015 line = MAX (cur_line - offset_line, 0);
2016 }
2017 else if (*entry_text == '+')
2018 {
2019 gint cur_line = gtk_text_iter_get_line (&view->priv->start_search_iter);
2020
2021 if (*(text + 1) != '\0')
2022 offset_line = MAX (atoi (text + 1), 0);
2023
2024 line = cur_line + offset_line;
2025 }
2026 else
2027 {
2028 line = MAX (atoi (text) - 1, 0);
2029 }
2030
2031 if (split_text[1] != NULL)
2032 {
2033 line_offset = atoi (split_text[1]);
2034 }
2035
2036 g_strfreev (split_text);
2037
2038 moved = pluma_document_goto_line (doc, line);
2039 moved_offset = pluma_document_goto_line_offset (doc, line, line_offset);
2040
2041 pluma_view_scroll_to_cursor (view);
2042
2043 if (!moved || !moved_offset)
2044 set_entry_state (view->priv->search_entry,
2045 PLUMA_SEARCH_ENTRY_NOT_FOUND);
2046 else
2047 set_entry_state (view->priv->search_entry,
2048 PLUMA_SEARCH_ENTRY_NORMAL);
2049 }
2050 }
