191static void
192add_or_remove (PlumaEncodingsComboBox *menu,
193 GtkTreeModel *model)
194{
195 GtkTreeIter iter;
196 gboolean add_item = FALSE;
197
198 if (gtk_combo_box_get_active_iter (GTK_COMBO_BOX (menu), &iter))
199 {
200 gtk_tree_model_get (model, &iter,
201 ADD_COLUMN, &add_item,
202 -1);
203 }
204
205 if (!add_item)
206 {
207 menu->priv->activated_item = gtk_combo_box_get_active (GTK_COMBO_BOX (menu));
208 }
209 else
210 {
211 GtkWidget *dialog;
212
213 GtkWidget *toplevel = gtk_widget_get_toplevel (GTK_WIDGET (menu));
214
215 if (!gtk_widget_is_toplevel (toplevel))
216 toplevel = NULL;
217
218 g_signal_handler_block (menu, menu->priv->changed_id);
219 gtk_combo_box_set_active (GTK_COMBO_BOX (menu),
220 menu->priv->activated_item);
221 g_signal_handler_unblock (menu, menu->priv->changed_id);
222
223 dialog = pluma_encodings_dialog_new();
224
225 if (toplevel != NULL)
226 {
227 GtkWindowGroup *wg;
228
229 gtk_window_set_transient_for (GTK_WINDOW (dialog),
230 GTK_WINDOW (toplevel));
231
232 wg = gtk_window_get_group (GTK_WINDOW (toplevel));
233 if (wg == NULL)
234 {
235 wg = gtk_window_group_new ();
236 gtk_window_group_add_window (wg,
237 GTK_WINDOW (toplevel));
238 }
239
240 gtk_window_group_add_window (wg,
241 GTK_WINDOW (dialog));
242 }
243
244 gtk_window_set_modal (GTK_WINDOW (dialog), TRUE);
245
246 g_signal_connect (dialog,
247 "response",
248 G_CALLBACK (dialog_response_cb),
249 menu);
250
251 gtk_widget_show (dialog);
252 }

268static void
269update_menu (PlumaEncodingsComboBox *menu)
270{
271 GtkListStore *store;
272 GtkTreeIter iter;
273 GSList *encodings, *l;
274 gchar *str;
275 const PlumaEncoding *utf8_encoding;
276 const PlumaEncoding *current_encoding;
277 gchar **enc_strv;
278
279 store = menu->priv->store;
280
281 /* Unset the previous model */
282 g_signal_handler_block (menu, menu->priv->changed_id);
283 gtk_list_store_clear (store);
284 gtk_combo_box_set_model (GTK_COMBO_BOX (menu),
285 NULL);
286
287 utf8_encoding = pluma_encoding_get_utf8 ();
288 current_encoding = pluma_encoding_get_current ();
289
290 if (!menu->priv->save_mode)
291 {
292 gtk_list_store_append (store, &iter);
293 gtk_list_store_set (store, &iter,
294 NAME_COLUMN, _("Automatically Detected"),
295 ENCODING_COLUMN, NULL,
296 ADD_COLUMN, FALSE,
297 -1);
298
299 gtk_list_store_append (store, &iter);
300 gtk_list_store_set (store, &iter,
301 NAME_COLUMN, "",
302 ENCODING_COLUMN, NULL,
303 ADD_COLUMN, FALSE,
304 -1);
305 }
306
307 if (current_encoding != utf8_encoding)
308 str = pluma_encoding_to_string (utf8_encoding);
309 else
310 str = g_strdup_printf (_("Current Locale (%s)"),
311 pluma_encoding_get_charset (utf8_encoding));
312
313 gtk_list_store_append (store, &iter);
314 gtk_list_store_set (store, &iter,
315 NAME_COLUMN, str,
316 ENCODING_COLUMN, utf8_encoding,
317 ADD_COLUMN, FALSE,
318 -1);
319
320 g_free (str);
321
322 if ((utf8_encoding != current_encoding) &&
323 (current_encoding != NULL))
324 {
325 str = g_strdup_printf (_("Current Locale (%s)"),
326 pluma_encoding_get_charset (current_encoding));
327
328 gtk_list_store_append (store, &iter);
329 gtk_list_store_set (store, &iter,
330 NAME_COLUMN, str,
331 ENCODING_COLUMN, current_encoding,
332 ADD_COLUMN, FALSE,
333 -1);
334
335 g_free (str);
336 }
337
338 enc_strv = g_settings_get_strv (menu->priv->enc_settings,
339 PLUMA_SETTINGS_ENCODING_SHOWN_IN_MENU);
340
341 encodings = _pluma_encoding_strv_to_list ((const gchar * const *)enc_strv);
342 g_strfreev (enc_strv);
343
344 for (l = encodings; l != NULL; l = g_slist_next (l))
345 {
346 const PlumaEncoding *enc = (const PlumaEncoding *)l->data;
347
348 if ((enc != current_encoding) &&
349 (enc != utf8_encoding) &&
350 (enc != NULL))
351 {
352 str = pluma_encoding_to_string (enc);
353
354 gtk_list_store_append (store, &iter);
355 gtk_list_store_set (store, &iter,
356 NAME_COLUMN, str,
357 ENCODING_COLUMN, enc,
358 ADD_COLUMN, FALSE,
359 -1);
360
361 g_free (str);
362 }
363 }
364
365 g_slist_free (encodings);
366
367 gtk_list_store_append (store, &iter);
368 /* separator */
369 gtk_list_store_set (store, &iter,
370 NAME_COLUMN, "",
371 ENCODING_COLUMN, NULL,
372 ADD_COLUMN, FALSE,
373 -1);
374
375 gtk_list_store_append (store, &iter);
376 gtk_list_store_set (store, &iter,
377 NAME_COLUMN, _("Add or Remove..."),
378 ENCODING_COLUMN, NULL,
379 ADD_COLUMN, TRUE,
380 -1);
381
382 /* set the model back */
383 gtk_combo_box_set_model (GTK_COMBO_BOX (menu),
384 GTK_TREE_MODEL (menu->priv->store));
385 gtk_combo_box_set_active (GTK_COMBO_BOX (menu), 0);
386
387 g_signal_handler_unblock (menu, menu->priv->changed_id);
388}

390static void
391pluma_encodings_combo_box_init (PlumaEncodingsComboBox *menu)
392{
393 menu->priv = pluma_encodings_combo_box_get_instance_private (menu);
394
395 menu->priv->enc_settings = g_settings_new (PLUMA_SCHEMA_ID);
396
397 menu->priv->store = gtk_list_store_new (N_COLUMNS,
398 G_TYPE_STRING,
399 G_TYPE_POINTER,
400 G_TYPE_BOOLEAN);
401
402 gtk_combo_box_set_row_separator_func (GTK_COMBO_BOX (menu),
403 separator_func, NULL,
404 NULL);
405
406 menu->priv->changed_id = g_signal_connect (menu, "changed",
407 G_CALLBACK (add_or_remove),
408 menu->priv->store);
409
410 update_menu (menu);
411}

453{

468 ENCODING_COLUMN, &enc,

472 {
473 gtk_combo_box_set_active_iter (GTK_COMBO_BOX (menu),
474 &iter);
475
476 return;
477 }
478
479 b = gtk_tree_model_iter_next (model, &iter);
480 }
