78static void
79context_ready (MateMixerContext *context, GtkApplication *application)
80{
81 /* The dialog might be already created, e.g. when reconnected
82 * to a sound server */
83 GtkApplication *app;
84 GError *error = NULL;
85
86 app = gtk_application_new ("context.ready", G_APPLICATION_FLAGS_NONE);
87 g_application_register (G_APPLICATION (app), NULL, &error);
88 if (error != NULL)
89 {
90 g_warning ("%s", error->message);
91 g_error_free (error);
92 error = NULL;
93 }
94
95 if (g_application_get_is_remote (G_APPLICATION (app)))
96 {
97 g_application_activate (G_APPLICATION (app));
98 g_object_unref (app);
99 app_dialog = GTK_WIDGET (gvc_mixer_dialog_new (context));
100 gtk_main_quit ();
101 }
102
103 if (app_dialog != NULL)
104 return;
105
106 app_dialog = GTK_WIDGET (gvc_mixer_dialog_new (context));
107
108 g_signal_connect (G_OBJECT (app_dialog),
109 "response",
110 G_CALLBACK (on_dialog_response),
111 GINT_TO_POINTER (FALSE));
112 g_signal_connect (G_OBJECT (app_dialog),
113 "close",
114 G_CALLBACK (on_dialog_close),
115 GINT_TO_POINTER (FALSE));
116
117 gvc_mixer_dialog_set_page (GVC_MIXER_DIALOG (app_dialog), page);
118
119 gtk_widget_show (app_dialog);
120
121 g_signal_connect_swapped (app, "activate", G_CALLBACK (gtk_window_present), app_dialog);
122}

124static void
125on_context_state_notify (MateMixerContext *context,
126 GParamSpec *pspec,
127 GtkApplication *app)
128{
129 MateMixerState state = mate_mixer_context_get_state (context);
130
131 if (state == MATE_MIXER_STATE_READY) {
132 remove_warning_dialog ();
133 context_ready (context, app);
134 }
135 else if (state == MATE_MIXER_STATE_FAILED) {
136 GtkWidget *dialog;
137
138 remove_warning_dialog ();
139
140 dialog = gtk_message_dialog_new (GTK_WINDOW (app_dialog),
141 0,
142 GTK_MESSAGE_ERROR,
143 GTK_BUTTONS_CLOSE,
144 _("Sound system is not available"));
145
146 g_signal_connect (G_OBJECT (dialog),
147 "response",
148 G_CALLBACK (on_dialog_response),
149 GINT_TO_POINTER (TRUE));
150 g_signal_connect (G_OBJECT (dialog),
151 "close",
152 G_CALLBACK (on_dialog_close),
153 GINT_TO_POINTER (TRUE));
154
155 gtk_widget_show (dialog);
156 }
157}

173 "close",
174 G_CALLBACK (on_dialog_close),

178
179 return FALSE;
180}

182int
183main (int argc, char **argv)
184{
185 GError *error = NULL;
186 gchar *backend = NULL;
187 MateMixerContext *context;
188 GApplication *app;
189
190 GOptionEntry entries[] = {
191 { "backend", 'b', 0, G_OPTION_ARG_STRING, &backend, N_("Sound system backend"), "pulse|alsa|oss|null" },
192 { "debug", 'd', 0, G_OPTION_ARG_NONE, &debug, N_("Enable debug"), NULL },
193 { "page", 'p', 0, G_OPTION_ARG_STRING, &page, N_("Startup page"), "effects|hardware|input|output|applications" },
194 { "version", 'v', 0, G_OPTION_ARG_NONE, &show_version, N_("Version of this application"), NULL },
195 { NULL, 0, 0, G_OPTION_ARG_NONE, NULL, NULL, NULL }
196 };
197
198 bindtextdomain (GETTEXT_PACKAGE, LOCALE_DIR);
199 bind_textdomain_codeset (GETTEXT_PACKAGE, "UTF-8");
200 textdomain (GETTEXT_PACKAGE);
201
202 gtk_init_with_args (&argc, &argv,
203 _(" â€” MATE Volume Control"),
204 entries, GETTEXT_PACKAGE,
205 &error);
206
207 if (error != NULL) {
208 g_warning ("%s", error->message);
209 g_error_free (error);
210 return 1;
211 }
212 if (show_version == TRUE) {
213 g_print ("%s %s\n", argv[0], VERSION);
214 return 0;
215 }
216 if (debug == TRUE) {
217 g_setenv ("G_MESSAGES_DEBUG", "all", FALSE);
218 }
219
220 app = g_application_new (GVC_DIALOG_DBUS_NAME, G_APPLICATION_FLAGS_NONE);
221
222 if (!g_application_register (app, NULL, &error))
223 {
224 g_warning ("%s", error->message);
225 g_error_free (error);
226 return 1;
227 }
228
229 if (mate_mixer_init () == FALSE) {
230 g_warning ("libmatemixer initialization failed, exiting");
231 return 1;
232 }
233
234 context = mate_mixer_context_new ();
235
236 if (backend != NULL) {
237 if (strcmp (backend, "pulse") == 0)
238 mate_mixer_context_set_backend_type (context, MATE_MIXER_BACKEND_PULSEAUDIO);
239 else if (strcmp (backend, "alsa") == 0)
240 mate_mixer_context_set_backend_type (context, MATE_MIXER_BACKEND_ALSA);
241 else if (strcmp (backend, "oss") == 0)
242 mate_mixer_context_set_backend_type (context, MATE_MIXER_BACKEND_OSS);
243 else if (strcmp (backend, "null") == 0)
244 mate_mixer_context_set_backend_type (context, MATE_MIXER_BACKEND_NULL);
245 else {
246 g_warning ("Invalid backend: %s", backend);
247 g_object_unref (context);
248 g_object_unref (app);
249 g_free (backend);
250 return 1;
251 }
252
253 g_free (backend);
254 }
255
256 mate_mixer_context_set_app_name (context, _("Volume Control"));
257 mate_mixer_context_set_app_id (context, GVC_DIALOG_DBUS_NAME);
258 mate_mixer_context_set_app_version (context, VERSION);
259 mate_mixer_context_set_app_icon (context, "multimedia-volume-control");
260
261 g_signal_connect (G_OBJECT (context),
262 "notify::state",
263 G_CALLBACK (on_context_state_notify),
264 app);
265
266 mate_mixer_context_open (context);
267
268 if (mate_mixer_context_get_state (context) == MATE_MIXER_STATE_CONNECTING) {
269 popup_id = g_timeout_add_seconds (DIALOG_POPUP_TIMEOUT,
270 dialog_popup_timeout,
271 NULL);
272 }
273
274 gtk_icon_theme_append_search_path (gtk_icon_theme_get_default (),
275 ICON_DATA_DIR);
276
277 gtk_window_set_default_icon_name ("multimedia-volume-control");
278
279 gtk_main ();
280
281 g_object_unref (context);
282 g_object_unref (app);
283
284 return 0;
