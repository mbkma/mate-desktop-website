99 "off-state-option", off,

106static gboolean
107alsa_toggle_set_active_option (MateMixerSwitch *mms, MateMixerSwitchOption *mmso)
108{
109 AlsaToggle *toggle;
110 gint value;
111 gint ret;
112
113 g_return_val_if_fail (ALSA_IS_TOGGLE (mms), FALSE);
114 g_return_val_if_fail (ALSA_IS_SWITCH_OPTION (mmso), FALSE);
115
116 toggle = ALSA_TOGGLE (mms);
117
118 if (G_UNLIKELY (toggle->priv->element == NULL))
119 return FALSE;
120
121 /* For toggles the 0/1 value is stored as the switch option id, there is not really
122 * a need to validate that the option belong to the switch, just make sure it
123 * contains the value 0 or 1 */
124 value = alsa_switch_option_get_id (ALSA_SWITCH_OPTION (mmso));
125 if (G_UNLIKELY (value != 0 && value != 1)) {
126 g_warn_if_reached ();
127 return FALSE;
128 }
129
130 if (toggle->priv->type == ALSA_TOGGLE_CAPTURE)
131 ret = snd_mixer_selem_set_capture_switch_all (toggle->priv->element, value);
132 else
133 ret = snd_mixer_selem_set_playback_switch_all (toggle->priv->element, value);
134
135 if (ret < 0) {
136 g_warning ("Failed to set value of toggle %s: %s",
137 snd_mixer_selem_get_name (toggle->priv->element),
138 snd_strerror (ret));
139 return FALSE;
140 }
141
142 return TRUE;
143}

161static gboolean
162alsa_toggle_load (AlsaElement *element)
163{
164 AlsaToggle *toggle;
165 gint value;
166 gint ret;
167 snd_mixer_selem_channel_id_t c;
168
169 g_return_val_if_fail (ALSA_IS_TOGGLE (element), FALSE);
170
171 toggle = ALSA_TOGGLE (element);
172
173 if (G_UNLIKELY (toggle->priv->element == NULL))
174 return FALSE;
175
176 /* When reading the first time we try all the channels, otherwise only the
177 * ones which returned success before */
178 if (toggle->priv->channel_mask == 0) {
179 for (c = 0; c < SND_MIXER_SCHN_LAST; c++) {
180 if (toggle->priv->type == ALSA_TOGGLE_CAPTURE)
181 ret = snd_mixer_selem_get_capture_switch (toggle->priv->element,
182 c,
183 &value);
184 else
185 ret = snd_mixer_selem_get_playback_switch (toggle->priv->element,
186 c,
187 &value);
188
189 /* The active enum option is set per-channel, so when reading it the
190 * first time, create a mask of all channels for which we read the
191 * value successfully */
192 if (ret == 0)
193 toggle->priv->channel_mask |= 1 << c;
194 }
195
196 /* The last ALSA call might have failed, but it doesn't matter if we have
197 * a channel mask */
198 if (toggle->priv->channel_mask > 0)
199 ret = 0;
200 } else {
201 for (c = 0; !(toggle->priv->channel_mask & (1 << c)); c++)
202 ;
203
204 /* When not reading the mask, the first usable channel is enough, we don't
205 * support per-channel selections anyway */
206 if (toggle->priv->type == ALSA_TOGGLE_CAPTURE)
207 ret = snd_mixer_selem_get_capture_switch (toggle->priv->element,
208 c,
209 &value);
210 else
211 ret = snd_mixer_selem_get_playback_switch (toggle->priv->element,
212 c,
213 &value);
214 }
215
216 if (ret == 0) {
217 MateMixerSwitchOption *active;
218
219 if (value > 0)
220 active = mate_mixer_stream_toggle_get_state_option (MATE_MIXER_STREAM_TOGGLE (toggle),
221 TRUE);
222 else
223 active = mate_mixer_stream_toggle_get_state_option (MATE_MIXER_STREAM_TOGGLE (toggle),
224 FALSE);
225
226 _mate_mixer_switch_set_active_option (MATE_MIXER_SWITCH (toggle), active);
227 return TRUE;
228 }
229
230 g_warning ("Failed to read state of toggle %s: %s",
231 snd_mixer_selem_get_name (toggle->priv->element),
232 snd_strerror (ret));
233
234 return FALSE;
235}
