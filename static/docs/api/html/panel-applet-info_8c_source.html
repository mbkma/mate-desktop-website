40MatePanelAppletInfo *
41mate_panel_applet_info_new (const gchar *iid,
42 const gchar *name,
43 const gchar *comment,
44 const gchar *icon,
45 const gchar **old_ids,
46 gboolean x11_supported,
47 gboolean wayland_supported)
48{
49 MatePanelAppletInfo *info = g_slice_new0 (MatePanelAppletInfo);
50
51 info->iid = g_strdup (iid);
52 info->name = g_strdup (name);
53 info->comment = g_strdup (comment);
54 info->icon = g_strdup (icon);
55
56 /* MateComponent compatibility */
57 if (old_ids != NULL) {
58 int len;
59 if ((len = g_strv_length ((gchar **) old_ids)) > 0) {
60 int i;
61 info->old_ids = g_new0 (gchar *, len + 1);
62 for (i = 0; i < len; i++)
63 info->old_ids[i] = g_strdup (old_ids[i]);
64 }
65 }
66
67 info->x11_supported = x11_supported;
68 info->wayland_supported = wayland_supported;
69
70 if (!x11_supported && !wayland_supported)
71 g_warning ("Applet %s has no supported platforms", name);
72
73 return info;
74}
