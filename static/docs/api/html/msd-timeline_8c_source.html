228static void

230 guint prop_id,
231 const GValue *value,
232 GParamSpec *pspec)
233{
234 MsdTimeline *timeline;

237

250 msd_timeline_set_direction (timeline, g_value_get_enum (value));
251 break;
252 case PROP_SCREEN:
253 msd_timeline_set_screen (timeline,

255 break;
256 case PROP_PROGRESS_TYPE:
257 msd_timeline_set_progress_type (timeline, g_value_get_enum (value));

260 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
261 }

321static gdouble
322sinusoidal_progress (gdouble progress)
323{
324 return (sinf ((progress * G_PI) / 2));

327static gdouble
328exponential_progress (gdouble progress)
329{
330 return progress * progress;

333static MsdTimelineProgressFunc
334progress_type_to_func (MsdTimelineProgressType type)
335{
336 if (type == MSD_TIMELINE_PROGRESS_SINUSOIDAL)
337 return sinusoidal_progress;
338 else if (type == MSD_TIMELINE_PROGRESS_EXPONENTIAL)
339 return exponential_progress;
340
341 return NULL;

344static gboolean
345msd_timeline_run_frame (MsdTimeline *timeline,
346 gboolean enable_animations)
347{
348 MsdTimelinePrivate *priv;
349 gdouble linear_progress, progress;
350 guint elapsed_time;
351 MsdTimelineProgressFunc progress_func = NULL;
352
353 priv = msd_timeline_get_instance_private (timeline);
354
355 if (enable_animations)
356 {
357 elapsed_time = (guint) (g_timer_elapsed (priv->timer, NULL) * 1000);
358
359 linear_progress = (gdouble) elapsed_time / priv->duration;
360
361 if (priv->direction == MSD_TIMELINE_DIRECTION_BACKWARD)
362 linear_progress = 1 - linear_progress;
363
364 linear_progress = CLAMP (linear_progress, 0., 1.);
365
366 if (priv->progress_func)
367 progress_func = priv->progress_func;
368 else if (priv->progress_type)
369 progress_func = progress_type_to_func (priv->progress_type);
370
371 if (progress_func)
372 progress = (progress_func) (linear_progress);
373 else
374 progress = linear_progress;
375 }
376 else
377 progress = (priv->direction == MSD_TIMELINE_DIRECTION_FORWARD) ? 1.0 : 0.0;
378
379 g_signal_emit (timeline, signals [FRAME], 0,
380 CLAMP (progress, 0.0, 1.0));
381
382 if ((priv->direction == MSD_TIMELINE_DIRECTION_FORWARD && progress >= 1.0) ||
383 (priv->direction == MSD_TIMELINE_DIRECTION_BACKWARD && progress <= 0.0))
384 {
385 if (!priv->loop)
386 {
387 if (priv->source_id)
388 {
389 g_source_remove (priv->source_id);
390 priv->source_id = 0;
391 }
392
393 g_signal_emit (timeline, signals [FINISHED], 0);
394 return FALSE;
395 }
396 else
397 msd_timeline_rewind (timeline);
398 }
399
400 return TRUE;
401}

441void
442msd_timeline_start (MsdTimeline *timeline)
443{
444 MsdTimelinePrivate *priv;
445 GtkSettings *settings;
446 gboolean enable_animations = FALSE;
447
448 g_return_if_fail (MSD_IS_TIMELINE (timeline));
449
450 priv = msd_timeline_get_instance_private (timeline);
451
452 if (priv->screen)
453 {
454 settings = gtk_settings_get_for_screen (priv->screen);
455 g_object_get (settings, "gtk-enable-animations", &enable_animations, NULL);
456 }
457
458 if (enable_animations)
459 {
460 if (!priv->source_id)
461 {
462 if (priv->timer)
463 g_timer_continue (priv->timer);
464 else
465 priv->timer = g_timer_new ();
466
467 /* sanity check */
468 g_assert (priv->fps > 0);
469
470 g_signal_emit (timeline, signals [STARTED], 0);
471
472 priv->source_id = gdk_threads_add_timeout (FRAME_INTERVAL (priv->fps),
473 (GSourceFunc) msd_timeline_frame_idle_func,
474 timeline);
475 }
476 }
477 else
478 {
479 /* If animations are not enabled, only run the last frame,
480 * it take us instantaneously to the last state of the animation.
481 * The only potential flaw happens when people use the ::finished
482 * signal to trigger another animation, or even worse, finally
483 * loop into this animation again.
484 */
485 g_signal_emit (timeline, signals [STARTED], 0);
486 msd_timeline_run_frame (timeline, FALSE);
487 }
