126static void
127panel_properties_dialog_orientation_changed (PanelPropertiesDialog *dialog,
128 GtkComboBox *combo_box)
129{
130 GtkTreeIter iter;
131 GtkTreeModel *model;
132 OrientationComboItem *item;
133
134 g_assert (dialog->orientation_combo == GTK_WIDGET (combo_box));
135
136 if (!gtk_combo_box_get_active_iter (combo_box, &iter))
137 return;
138
139 model = gtk_combo_box_get_model (combo_box);
140 gtk_tree_model_get (model, &iter, COLUMN_ITEM, &item, -1);
141 if (item == NULL)
142 return;
143

145 item->orientation);

166static void
167panel_properties_dialog_setup_orientation_combo (PanelPropertiesDialog *dialog,
168 GtkBuilder *gui)
169{
170 PanelOrientation orientation;
171 GtkListStore *model;
172 GtkTreeIter iter;
173 GtkCellRenderer *renderer;
174 gsize i;
175
176 dialog->orientation_combo = PANEL_GTK_BUILDER_GET (gui, "orientation_combo");
177 g_return_if_fail (dialog->orientation_combo != NULL);
178 dialog->orientation_label = PANEL_GTK_BUILDER_GET (gui, "orientation_label");
179 g_return_if_fail (dialog->orientation_label != NULL);
180
181 orientation = panel_profile_get_toplevel_orientation (dialog->toplevel);
182
183 model = gtk_list_store_new (NUMBER_COLUMNS,
184 G_TYPE_STRING,
185 G_TYPE_POINTER);
186
187 gtk_combo_box_set_model (GTK_COMBO_BOX (dialog->orientation_combo),
188 GTK_TREE_MODEL (model));
189
190 for (i = 0; i < G_N_ELEMENTS (orientation_items); i++) {
191 gtk_list_store_append (model, &iter);
192 gtk_list_store_set (model, &iter,
193 COLUMN_TEXT, g_dpgettext2 (NULL, "Orientation", orientation_items [i].name),
194 COLUMN_ITEM, &(orientation_items [i]),
195 -1);
196 if (orientation == orientation_items [i].orientation)
197 gtk_combo_box_set_active_iter (GTK_COMBO_BOX (dialog->orientation_combo),
198 &iter);
199 }
200
201 renderer = gtk_cell_renderer_text_new ();
202 gtk_cell_layout_pack_start (GTK_CELL_LAYOUT (dialog->orientation_combo),
203 renderer, TRUE);
204 gtk_cell_layout_set_attributes (GTK_CELL_LAYOUT (dialog->orientation_combo),
205 renderer, "text", COLUMN_TEXT, NULL);
206
207 g_signal_connect_swapped (dialog->orientation_combo, "changed",
208 G_CALLBACK (panel_properties_dialog_orientation_changed),
209 dialog);
210
211 panel_properties_dialog_setup_orientation_combo_sensitivty (dialog, NULL);
212}

514 dialog->default_radio = PANEL_GTK_BUILDER_GET (gui, "default_radio");

537
538 panel_properties_dialog_upd_sensitivity (dialog, background_type);
539
540 g_signal_connect_swapped (dialog->default_radio, "toggled",
541 G_CALLBACK (panel_properties_dialog_background_toggled),
542 dialog);
543 g_signal_connect_swapped (dialog->color_radio, "toggled",

546 g_signal_connect_swapped (dialog->image_radio, "toggled",
547 G_CALLBACK (panel_properties_dialog_background_toggled),
548 dialog);
549
550 if ( ! panel_profile_background_key_is_writable (dialog->toplevel, "type")) {
551 gtk_widget_set_sensitive (dialog->default_radio, FALSE);
552 gtk_widget_set_sensitive (dialog->color_radio, FALSE);

603static void
604panel_properties_dialog_update_orientation (PanelPropertiesDialog *dialog,
605 PanelOrientation orientation)
606{
607 GtkTreeModel *model;
608 GtkTreeIter iter;
609 OrientationComboItem *item;
610 int max_size;
611 int spin_size;
612 int profile_size;
613
614 /* change the maximum size of the panel */
615 /* TODO: we should also do this when the monitor size changes */
616 max_size = panel_toplevel_get_maximum_size (dialog->toplevel);
617 spin_size = gtk_spin_button_get_value_as_int (GTK_SPIN_BUTTON (dialog->size_spin));
618 profile_size = panel_profile_get_toplevel_size (dialog->toplevel);
619
620 gtk_spin_button_set_range (GTK_SPIN_BUTTON (dialog->size_spin),
621 panel_toplevel_get_minimum_size (dialog->toplevel),
622 max_size);
623
624 if (spin_size > max_size)
625 gtk_spin_button_set_value (GTK_SPIN_BUTTON (dialog->size_spin),
626 max_size);
627 else if (spin_size != profile_size)
628 gtk_spin_button_set_value (GTK_SPIN_BUTTON (dialog->size_spin),
629 MIN (profile_size, max_size));
630
631 /* update the orientation combo box */
632 model = gtk_combo_box_get_model (GTK_COMBO_BOX (dialog->orientation_combo));
633
634 if (!gtk_tree_model_get_iter_first (model, &iter))
635 return;
636
637 do {
638 gtk_tree_model_get (model, &iter, COLUMN_ITEM, &item, -1);
639 if (item != NULL && item->orientation == orientation) {
640 gtk_combo_box_set_active_iter (GTK_COMBO_BOX (dialog->orientation_combo),
641 &iter);
642 return;
643 }
644 } while (gtk_tree_model_iter_next (model, &iter));
645}
