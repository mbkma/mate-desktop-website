65static int
66extract_cb (GtkWidget *w,
67 DialogData *data)
68{
69 FrWindow *window = data->window;
70 gboolean do_not_extract = FALSE;
71 char *extract_to_dir;
72 char *sub_dir;
73 gboolean extract_sub_dir;
74 gboolean overwrite;
75 gboolean skip_newer;
76 gboolean selected_files;
77 gboolean pattern_files;
78 gboolean junk_paths;
79 GList *file_list;
80 char *base_dir = NULL;
81 GError *error = NULL;
82
83 data->extract_clicked = TRUE;
84
85 /* collect extraction options. */
86
87 extract_to_dir = gtk_file_chooser_get_uri (GTK_FILE_CHOOSER (data->dialog));
88
89 /* check directory existence. */
90
91 if (! uri_is_dir (extract_to_dir)) {
92 if (! ForceDirectoryCreation) {
93 GtkWidget *d;
94 int r;
95 char *folder_name;
96 char *msg;
97
98 folder_name = g_filename_display_name (extract_to_dir);
99 msg = g_strdup_printf (_("Destination folder \"%s\" does not exist.\n\nDo you want to create it?"), folder_name);
100 g_free (folder_name);
101
102 d = _gtk_message_dialog_new (GTK_WINDOW (data->dialog),
103 GTK_DIALOG_MODAL,
104 "dialog-question",
105 msg,
106 NULL,
107 "gtk-cancel", GTK_RESPONSE_CANCEL,
108 _("Create _Folder"), GTK_RESPONSE_YES,
109 NULL);
110
111 gtk_dialog_set_default_response (GTK_DIALOG (d), GTK_RESPONSE_YES);
112 r = gtk_dialog_run (GTK_DIALOG (d));
113 gtk_widget_destroy (GTK_WIDGET (d));
114
115 g_free (msg);
116
117 if (r != GTK_RESPONSE_YES)
118 do_not_extract = TRUE;
119 }
120
121 if (! do_not_extract && ! ensure_dir_exists (extract_to_dir, 0755, &error)) {
122 GtkWidget *d;
123
124 d = _gtk_error_dialog_new (GTK_WINDOW (window),
125 GTK_DIALOG_DESTROY_WITH_PARENT,
126 NULL,
127 _("Extraction not performed"),
128 _("Could not create the destination folder: %s."),
129 error->message);
130 gtk_dialog_run (GTK_DIALOG (d));
131 gtk_widget_destroy (GTK_WIDGET (d));
132
133 g_error_free (error);
134
135 return FALSE;
136 }
137 }
138
139 if (do_not_extract) {
140 GtkWidget *d;
141
142 d = _gtk_message_dialog_new (GTK_WINDOW (window),
143 GTK_DIALOG_DESTROY_WITH_PARENT,
144 "dialog-warning",
145 _("Extraction not performed"),
146 NULL,
147 "gtk-ok", GTK_RESPONSE_OK,
148 NULL);
149 gtk_dialog_set_default_response (GTK_DIALOG (d), GTK_RESPONSE_OK);
150 gtk_dialog_run (GTK_DIALOG (d));
151 gtk_widget_destroy (GTK_WIDGET (d));
152
153 if (fr_window_is_batch_mode (data->window))
154 gtk_widget_destroy (data->dialog);
155
156 return FALSE;
157 }
158
159 /* check extraction directory permissions. */
160
161 if (uri_is_dir (extract_to_dir)
162 && ! check_permissions (extract_to_dir, R_OK | W_OK))
163 {
164 GtkWidget *d;
165 char *utf8_path;
166
167 utf8_path = g_filename_display_name (extract_to_dir);
168
169 d = _gtk_error_dialog_new (GTK_WINDOW (window),
170 GTK_DIALOG_DESTROY_WITH_PARENT,
171 NULL,
172 _("Extraction not performed"),
173 _("You don't have the right permissions to extract archives in the folder \"%s\""),
174 utf8_path);
175 gtk_dialog_run (GTK_DIALOG (d));
176 gtk_widget_destroy (GTK_WIDGET (d));
177
178 g_free (utf8_path);
179 g_free (extract_to_dir);
180
181 return FALSE;
182 }
183
184 overwrite = gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (GET_WIDGET ("overwrite_checkbutton")));
185 skip_newer = ! gtk_toggle_button_get_inconsistent (GTK_TOGGLE_BUTTON (GET_WIDGET ("not_newer_checkbutton"))) && gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (GET_WIDGET ("not_newer_checkbutton")));
186 junk_paths = ! gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (GET_WIDGET ("recreate_dir_checkbutton")));
187 extract_sub_dir = gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (GET_WIDGET ("create_subdir_checkbutton")));
188
189 g_settings_set_boolean (data->settings, PREF_EXTRACT_OVERWRITE, overwrite);
190 if (! gtk_toggle_button_get_inconsistent (GTK_TOGGLE_BUTTON (GET_WIDGET ("not_newer_checkbutton"))))
191 g_settings_set_boolean (data->settings, PREF_EXTRACT_SKIP_NEWER, skip_newer);
192 g_settings_set_boolean (data->settings, PREF_EXTRACT_RECREATE_FOLDERS, ! junk_paths);
193 g_settings_set_boolean (data->settings, PREF_EXTRACT_CREATE_SUBDIR, extract_sub_dir);
194
195 fr_window_set_extract_default_dir (window, extract_to_dir, !extract_sub_dir);
196
197 selected_files = gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (GET_WIDGET ("selected_files_radiobutton")));
198 pattern_files = gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (GET_WIDGET ("file_pattern_radiobutton")));
199
200 /* create the file list. */
201
202 file_list = NULL;
203
204 if (selected_files) {
205 file_list = data->selected_files;
206 data->selected_files = NULL; /* do not the list when destroying the dialog. */
207 }
208 else if (pattern_files) {
209 const char *pattern;
210
211 pattern = gtk_entry_get_text (GTK_ENTRY (GET_WIDGET ("file_pattern_entry")));
212 file_list = fr_window_get_file_list_pattern (window, pattern);
213 if (file_list == NULL) {
214 gtk_widget_destroy (data->dialog);
215 g_free (extract_to_dir);
216 return FALSE;
217 }
218 }
219
220 if (selected_files) {
221 base_dir = data->base_dir_for_selection;
222 data->base_dir_for_selection = NULL;
223 }
224 else
225 base_dir = NULL;
226
227 if (extract_sub_dir)
228 sub_dir = remove_extension_from_path (g_filename_display_basename (fr_window_get_archive_uri (window)));
229 else
230 sub_dir = NULL;
231
232 /* close the dialog. */
233
234 gtk_widget_destroy (data->dialog);
235
236 /* extract ! */
237
238 fr_window_archive_extract (window,
239 file_list,
240 extract_to_dir,
241 sub_dir,
242 base_dir,
243 skip_newer,
244 overwrite ? FR_OVERWRITE_YES : FR_OVERWRITE_NO,
245 junk_paths,
246 TRUE);
247
248 path_list_free (file_list);
249 g_free (extract_to_dir);
250 g_free (sub_dir);
251 g_free (base_dir);
252
253 return TRUE;
254}

256static int
257file_sel_response_cb (GtkWidget *widget,
258 int response,
259 DialogData *data)
260{
261 if ((response == GTK_RESPONSE_CANCEL) || (response == GTK_RESPONSE_DELETE_EVENT)) {
262 gtk_widget_destroy (data->dialog);
263 return TRUE;
264 }
265
266 if (response == GTK_RESPONSE_HELP) {
267 show_help_dialog (GTK_WINDOW (data->dialog), "engrampa-extract-options");
268 return TRUE;
269 }
270
271 if (response == GTK_RESPONSE_OK)
272 return extract_cb (widget, data);
273
274 return FALSE;
275}
