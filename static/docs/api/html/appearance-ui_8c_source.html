27static void
28set_have_icons (AppearanceData *data, gboolean value)
29{
30 static const char *menu_item_names[] = {
31 "menu_item_1",
32 "menu_item_2",
33 "menu_item_3",
34 "menu_item_4",
35 "menu_item_5",
36 "cut",
37 "copy",
38 "paste",
39 NULL
40 };
41
42 const char **name;
43
44 for (name = menu_item_names; *name != NULL; name++) {
45 MateImageMenuItem *item = MATE_IMAGE_MENU_ITEM (appearance_capplet_get_widget (data, *name));
46 GtkWidget *image;
47
48 if (value) {
49 image = g_object_get_data (G_OBJECT (item), "image");
50 if (image) {
51 mate_image_menu_item_set_image (item, image);
52 g_object_unref (image);
53 }
54 }
55 else
56 {
57 image = mate_image_menu_item_get_image (item);
58 g_object_set_data (G_OBJECT (item), "image", image);
59 g_object_ref (image);
60 mate_image_menu_item_set_image (item, NULL);
61 }
62 }
63}
