68static void
69impl_activate (MateSettingsPlugin *plugin)
70{
71 gboolean res;
72 GError *error;
73
74 g_debug ("Activating a11y-settings plugin");
75
76 error = NULL;
77 res = msd_a11y_settings_manager_start (MSD_A11Y_SETTINGS_PLUGIN (plugin)->priv->manager, &error);
78 if (! res) {
79 g_warning ("Unable to start a11y-settings manager: %s", error->message);
80 g_error_free (error);
81 }
