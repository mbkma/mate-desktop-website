73static gboolean
74popup_dock (GvcStreamAppletIcon *icon, guint time)
75{
76 GtkAllocation allocation;
77 GdkDisplay *display;
78 GdkScreen *screen;
79 int x, y;
80 GdkMonitor *monitor_num;
81 GdkRectangle monitor;
82 GtkRequisition dock_req;
83
84 screen = gtk_widget_get_screen (GTK_WIDGET (icon));
85 gtk_widget_get_allocation (GTK_WIDGET (icon), &allocation);
86 gdk_window_get_origin (gtk_widget_get_window (GTK_WIDGET (icon)), &allocation.x, &allocation.y);
87 gtk_widget_set_state_flags (GTK_WIDGET (icon), GTK_STATE_FLAG_CHECKED, FALSE);
88
89 /* position roughly */
90 gtk_window_set_screen (GTK_WINDOW (icon->priv->dock), screen);
91 switch (icon->priv->orient) {
92 case MATE_PANEL_APPLET_ORIENT_LEFT:
93 case MATE_PANEL_APPLET_ORIENT_RIGHT:
94 gvc_channel_bar_set_orientation (GVC_CHANNEL_BAR (icon->priv->bar), GTK_ORIENTATION_HORIZONTAL);
95 break;
96 case MATE_PANEL_APPLET_ORIENT_UP:
97 case MATE_PANEL_APPLET_ORIENT_DOWN:
98 default:
99 gvc_channel_bar_set_orientation (GVC_CHANNEL_BAR (icon->priv->bar), GTK_ORIENTATION_VERTICAL);
100 }
101
102 display = gdk_screen_get_display (screen);
103 monitor_num = gdk_display_get_monitor_at_point (display, allocation.x, allocation.y);
104 gdk_monitor_get_geometry (monitor_num, &monitor);
105
106 gtk_container_foreach (GTK_CONTAINER (icon->priv->dock), (GtkCallback) gtk_widget_show_all, NULL);
107 gtk_widget_get_preferred_size (icon->priv->dock, &dock_req, NULL);
108
109#if defined(ENABLE_WAYLAND)
110 if (GDK_IS_WAYLAND_DISPLAY (display))
111 {
112 gboolean top, bottom, left, right;
113 GtkWidget *toplevel;
114 toplevel = gtk_widget_get_toplevel (GTK_WIDGET (icon));
115
116 if (!gtk_layer_is_layer_window (GTK_WINDOW (icon->priv->dock)))
117 {
118 gtk_layer_init_for_window (GTK_WINDOW (icon->priv->dock));
119 gtk_layer_set_layer (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_LAYER_TOP);
120 gtk_layer_set_keyboard_mode (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_KEYBOARD_MODE_ON_DEMAND);
121 }
122
123 top = gtk_layer_get_anchor (GTK_WINDOW (toplevel), GTK_LAYER_SHELL_EDGE_TOP);
124 bottom = gtk_layer_get_anchor (GTK_WINDOW (toplevel), GTK_LAYER_SHELL_EDGE_BOTTOM);
125 left = gtk_layer_get_anchor (GTK_WINDOW (toplevel), GTK_LAYER_SHELL_EDGE_LEFT);
126 right = gtk_layer_get_anchor (GTK_WINDOW (toplevel), GTK_LAYER_SHELL_EDGE_RIGHT);
127
128 /*Set anchors to the edges (will hold to panel edge) and position along the panel
129 *Unset margins and anchors from any other position so as to avoid rendering issues
130 *when orientation changes as when the panel is moved
131 */
132
133 if (top && left && right)
134 {
135 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_TOP, TRUE);
136 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_LEFT, TRUE);
137 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_BOTTOM, FALSE);
138 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_RIGHT, FALSE);
139 gtk_layer_set_margin (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_LEFT, allocation.x);
140 gtk_layer_set_margin (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_TOP, 0);
141 }
142 if (bottom && left && right)
143 {
144 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_BOTTOM, TRUE);
145 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_LEFT, TRUE);
146 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_TOP, FALSE);
147 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_RIGHT, FALSE);
148 gtk_layer_set_margin (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_LEFT, allocation.x);
149 gtk_layer_set_margin (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_TOP, 0);
150 }
151 if (left && bottom && top && !right)
152 {
153 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_LEFT, TRUE);
154 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_TOP, TRUE);
155 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_RIGHT, FALSE);
156 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_BOTTOM, FALSE);
157 gtk_layer_set_margin (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_TOP, allocation.y);
158 gtk_layer_set_margin (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_LEFT, 0);
159 }
160 if (right && bottom && top && !left)
161 {
162 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_RIGHT, TRUE);
163 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_TOP, TRUE);
164 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_LEFT, FALSE);
165 gtk_layer_set_anchor (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_BOTTOM, FALSE);
166 gtk_layer_set_margin (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_TOP, allocation.y);
167 gtk_layer_set_margin (GTK_WINDOW (icon->priv->dock), GTK_LAYER_SHELL_EDGE_LEFT, 0);
168 }
169 gtk_widget_show_all (icon->priv->dock);
170
171 /* Grab focus */
172 gtk_grab_add (icon->priv->dock);
173 gtk_widget_grab_focus (icon->priv->dock);
174
175 return TRUE;
176 }
177#endif /* wayland support */
178
179 if (icon->priv->orient == MATE_PANEL_APPLET_ORIENT_LEFT || icon->priv->orient == MATE_PANEL_APPLET_ORIENT_RIGHT) {
180 if (allocation.x + allocation.width + dock_req.width <= monitor.x + monitor.width)
181 x = allocation.x + allocation.width;
182 else
183 x = allocation.x - dock_req.width;
184
185 if (allocation.y + dock_req.height <= monitor.y + monitor.height)
186 y = allocation.y;
187 else
188 y = monitor.y + monitor.height - dock_req.height;
189 } else {
190 if (allocation.y + allocation.height + dock_req.height <= monitor.y + monitor.height)
191 y = allocation.y + allocation.height;
192 else
193 y = allocation.y - dock_req.height;
194
195 if (allocation.x + dock_req.width <= monitor.x + monitor.width)
196 x = allocation.x;
197 else
198 x = monitor.x + monitor.width - dock_req.width;
199 }
200
201 gtk_window_move (GTK_WINDOW (icon->priv->dock), x, y);
202
203 /* Without this, the popup window appears as a square after changing
204 * the orientation */
205 gtk_window_resize (GTK_WINDOW (icon->priv->dock), 1, 1);
206
207 gtk_widget_show_all (icon->priv->dock);
208
209 /* Grab focus */
210 gtk_grab_add (icon->priv->dock);
211
212 display = gtk_widget_get_display (icon->priv->dock);
213
214 do {
215 GdkSeat *seat = gdk_display_get_default_seat (display);
216 GdkWindow *window = gtk_widget_get_window (icon->priv->dock);
217
218 if (gdk_seat_grab (seat,
219 window,
220 GDK_SEAT_CAPABILITY_ALL,
221 TRUE,
222 NULL,
223 NULL,
224 NULL,
225 NULL) != GDK_GRAB_SUCCESS) {
226 gtk_grab_remove (icon->priv->dock);
227 gtk_widget_hide (icon->priv->dock);
228 break;
229 }
230 } while (0);
231
232 gtk_widget_grab_focus (icon->priv->dock);
233
234 return TRUE;
235}

237static gboolean
238on_applet_icon_button_press (GtkWidget *applet_icon,
239 GdkEventButton *event,
240 GvcStreamAppletIcon *icon)
241{
242 if (event->button == 1) {
243 popup_dock (icon, GDK_CURRENT_TIME);
244 return TRUE;
245 }
246
247 /* Middle click acts as mute/unmute */
248 if (event->button == 2) {
249 gboolean is_muted = mate_mixer_stream_control_get_mute (icon->priv->control);
250
251 mate_mixer_stream_control_set_mute (icon->priv->control, !is_muted);
252 return TRUE;
253 }
254 return FALSE;
255}

269void
270gvc_stream_applet_icon_volume_control (GvcStreamAppletIcon *icon)
271{
272 GError *error = NULL;
273
274 mate_gdk_spawn_command_line_on_screen (gtk_widget_get_screen (icon->priv->dock),
275 "mate-volume-control",
276 &error);
277
278 if (error != NULL) {
279 GtkWidget *dialog;
280
281 dialog = gtk_message_dialog_new (NULL,
282 0,
283 GTK_MESSAGE_ERROR,
284 GTK_BUTTONS_CLOSE,
285 _("Failed to start Sound Preferences: %s"),
286 error->message);
287 g_signal_connect (G_OBJECT (dialog),
288 "response",
289 G_CALLBACK (gtk_widget_destroy),
290 NULL);
291 gtk_widget_show (dialog);
292 g_error_free (error);
293 }
294}

417static void
418update_icon (GvcStreamAppletIcon *icon)
419{
420 guint volume = 0;
421 gdouble decibel = 0;
422 guint normal = 0;
423 gboolean muted = FALSE;
424 guint n = 0;
425 gchar *markup;
426 const gchar *description;
427 MateMixerStreamControlFlags flags;
428
429 if (icon->priv->control == NULL) {
430 /* Do not bother creating a tooltip for an unusable icon as it
431 * has no practical use */
432 gtk_widget_set_has_tooltip (GTK_WIDGET (icon), FALSE);
433 return;
434 } else
435 gtk_widget_set_has_tooltip (GTK_WIDGET (icon), TRUE);
436
437 flags = mate_mixer_stream_control_get_flags (icon->priv->control);
438
439 if (flags & MATE_MIXER_STREAM_CONTROL_MUTE_READABLE)
440 muted = mate_mixer_stream_control_get_mute (icon->priv->control);
441
442 if (flags & MATE_MIXER_STREAM_CONTROL_VOLUME_READABLE) {
443 volume = mate_mixer_stream_control_get_volume (icon->priv->control);
444 normal = mate_mixer_stream_control_get_normal_volume (icon->priv->control);
445
446 /* Select an icon, they are expected to be sorted, the lowest index being
447 * the mute icon and the rest being volume increments */
448 if (volume <= 0 || muted)
449 n = 0;
450 else
451 n = CLAMP (3 * volume / normal + 1, 1, 3);
452 }
453 if (flags & MATE_MIXER_STREAM_CONTROL_HAS_DECIBEL)
454 decibel = mate_mixer_stream_control_get_decibel (icon->priv->control);
455
456 /* Apparently applet icon will reset icon even if it doesn't change */
457 if (icon->priv->current_icon != n) {
458 gvc_stream_applet_icon_set_icon_from_name (icon, icon->priv->icon_names[n]);
459 icon->priv->current_icon = n;
460 }
461
462 description = mate_mixer_stream_control_get_label (icon->priv->control);
463
464 guint volume_percent = (guint) round (100.0 * volume / normal);
465 if (muted) {
466 markup = g_strdup_printf ("<b>%s: %s %u%%</b>\n<small>%s</small>",
467 icon->priv->display_name,
468 _("Muted at"),
469 volume_percent,
470 description);
471 } else if (flags & MATE_MIXER_STREAM_CONTROL_VOLUME_READABLE) {
472 if (flags & MATE_MIXER_STREAM_CONTROL_HAS_DECIBEL) {
473 if (decibel > -MATE_MIXER_INFINITY) {
474 markup = g_strdup_printf ("<b>%s: %u%%</b>\n"
475 "<small>%0.2f dB\n%s</small>",
476 icon->priv->display_name,
477 volume_percent,
478 decibel,
479 description);
480 } else {
481 markup = g_strdup_printf ("<b>%s: %u%%</b>\n"
482 "<small>-&#8734; dB\n%s</small>",
483 icon->priv->display_name,
484 volume_percent,
485 description);
486 }
487 } else {
488 markup = g_strdup_printf ("<b>%s: %u%%</b>\n<small>%s</small>",
489 icon->priv->display_name,
490 volume_percent,
491 description);
492 }
493 } else {
494 markup = g_strdup_printf ("<b>%s</b>\n<small>%s</small>",
495 icon->priv->display_name,
496 description);
497 }
498
499 gtk_widget_set_tooltip_markup (GTK_WIDGET (icon), markup);
500
501 g_free (markup);
502}

504void
505gvc_stream_applet_icon_set_size (GvcStreamAppletIcon *icon,
506 guint size)
507{
508
509 /*Iterate through the icon sizes so they can be kept sharp*/
510 if (size < 22)
511 size = 16;
512 else if (size < 24)
513 size = 22;
514 else if (size < 32)
515 size = 24;
516 else if (size < 48)
517 size = 32;
518
519 icon->priv->size = size;
520 gvc_stream_applet_icon_set_icon_from_name (icon, icon->priv->icon_names[icon->priv->current_icon]);
521}

537void
538gvc_stream_applet_icon_set_icon_names (GvcStreamAppletIcon *icon,
539 const gchar **names)
540{
541 g_return_if_fail (GVC_IS_STREAM_APPLET_ICON (icon));
542 g_return_if_fail (names != NULL && *names != NULL);
543
544 if (G_UNLIKELY (g_strv_length ((gchar **) names) != 4)) {
545 g_warn_if_reached ();
546 return;
547 }
548
549 g_strfreev (icon->priv->icon_names);
550
551 icon->priv->icon_names = g_strdupv ((gchar **) names);
552
553 /* Set the first icon as the initial one, the icon may be immediately
554 * updated or not depending on whether a stream is available */
555 gvc_stream_applet_icon_set_icon_from_name (icon, names[0]);
556 update_icon (icon);
557
558 g_object_notify_by_pspec (G_OBJECT (icon), properties[PROP_ICON_NAMES]);
559}

591void
592gvc_stream_applet_icon_set_control (GvcStreamAppletIcon *icon,
593 MateMixerStreamControl *control)
594{
595 g_return_if_fail (GVC_IS_STREAM_APPLET_ICON (icon));
596
597 if (icon->priv->control == control)
598 return;
599
600 if (control != NULL)
601 g_object_ref (control);
602
603 if (icon->priv->control != NULL) {
604 g_signal_handlers_disconnect_by_func (G_OBJECT (icon->priv->control),
605 G_CALLBACK (on_stream_control_volume_notify),
606 icon);
607 g_signal_handlers_disconnect_by_func (G_OBJECT (icon->priv->control),
608 G_CALLBACK (on_stream_control_mute_notify),
609 icon);
610
611 g_object_unref (icon->priv->control);
612 }
613
614 icon->priv->control = control;
615
616 if (icon->priv->control != NULL) {
617 g_signal_connect (G_OBJECT (icon->priv->control),
618 "notify::volume",
619 G_CALLBACK (on_stream_control_volume_notify),
620 icon);
621 g_signal_connect (G_OBJECT (icon->priv->control),
622 "notify::mute",
623 G_CALLBACK (on_stream_control_mute_notify),
624 icon);
625
626 // XXX when no stream set some default icon and "unset" dock
627 update_icon (icon);
