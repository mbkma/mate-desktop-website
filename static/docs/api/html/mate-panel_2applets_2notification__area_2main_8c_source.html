78static GfStatusNotifierWatcher *

80{
81 GSettings *settings;
82 settings = g_settings_new ("org.mate.panel");
83
84 if (g_settings_get_boolean (settings, "enable-sni-support") == TRUE)

86 if (sn_watcher_service != NULL)

94 }
95
96 g_object_unref (settings);
97 return sn_watcher_service;

173static void
174ensure_prefs_window_is_created (NaTrayApplet *applet)
175{
176 if (applet->priv->dialog)
177 return;
178
179 applet->priv->dialog = g_new0 (NAPreferencesDialog, 1);
180
181 applet->priv->dialog->preferences_dialog = GTK_WIDGET (gtk_builder_get_object (applet->priv->builder, "notification_area_preferences_dialog"));
182
183 gtk_window_set_icon_name (GTK_WINDOW (applet->priv->dialog->preferences_dialog), NOTIFICATION_AREA_ICON);
184
185 applet->priv->dialog->min_icon_size_spin = GTK_WIDGET (gtk_builder_get_object (applet->priv->builder, "min_icon_size_spin"));
186 g_return_if_fail (applet->priv->dialog->min_icon_size_spin != NULL);
187
188 gtk_spin_button_set_range (GTK_SPIN_BUTTON (applet->priv->dialog->min_icon_size_spin), 7, 130);
189 gtk_spin_button_set_value (GTK_SPIN_BUTTON (applet->priv->dialog->min_icon_size_spin), applet->priv->min_icon_size);
190
191 g_signal_connect_swapped (applet->priv->dialog->min_icon_size_spin, "value-changed",
192 G_CALLBACK (na_preferences_dialog_min_icon_size_changed),
193 applet);
194
195 g_signal_connect_swapped (applet->priv->dialog->preferences_dialog, "response",
196 G_CALLBACK (na_preferences_dialog_response), applet);
197
198 g_signal_connect (applet->priv->dialog->preferences_dialog, "delete-event",
199 G_CALLBACK (na_preferences_dialog_hide_event), applet);

213static void help_cb(GtkAction* action, NaTrayApplet* applet)
214{
215 GError* error = NULL;
216 char* uri;
217 #define NA_HELP_DOC "mate-user-guide"
218
219 uri = g_strdup_printf("help:%s/%s", NA_HELP_DOC, "panels-notification-area");
220 gtk_show_uri_on_window (NULL, uri, gtk_get_current_event_time (), &error);
221 g_free(uri);
222
223 if (error && g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
224 {
225 g_error_free(error);
226 }
227 else if(error)
228 {
229 GtkWidget* dialog;
230 char* primary;
231
232 primary = g_markup_printf_escaped (_("Could not display help document '%s'"), NA_HELP_DOC);
233 dialog = gtk_message_dialog_new (NULL, GTK_DIALOG_DESTROY_WITH_PARENT, GTK_MESSAGE_ERROR, GTK_BUTTONS_CLOSE, "%s", primary);
234
235 gtk_message_dialog_format_secondary_text(GTK_MESSAGE_DIALOG(dialog), "%s", error->message);
236
237 g_error_free(error);
238 g_free(primary);
239
240 g_signal_connect(dialog, "response", G_CALLBACK (gtk_widget_destroy), NULL);
241
242 gtk_window_set_icon_name (GTK_WINDOW (dialog), NOTIFICATION_AREA_ICON);
243 gtk_window_set_screen (GTK_WINDOW (dialog), gtk_widget_get_screen (GTK_WIDGET (applet)));
244 /* we have no parent window */
245 gtk_window_set_skip_taskbar_hint (GTK_WINDOW (dialog), FALSE);
246 gtk_window_set_title (GTK_WINDOW (dialog), _("Error displaying help document"));
247
248 gtk_widget_show (dialog);
249 }
250}

297static void
298na_tray_applet_realize (GtkWidget *widget)
299{
300 NaTrayApplet *applet = NA_TRAY_APPLET (widget);
301 if (parent_class_realize)
302 parent_class_realize (widget);
303
304 GtkActionGroup* action_group;
305 action_group = gtk_action_group_new("NA Applet Menu Actions");
306 gtk_action_group_set_translation_domain(action_group, GETTEXT_PACKAGE);
307 gtk_action_group_add_actions(action_group, menu_actions, G_N_ELEMENTS(menu_actions), applet);
308 mate_panel_applet_setup_menu_from_resource (MATE_PANEL_APPLET (applet),
309 NA_RESOURCE_PATH "notification-area-menu.xml",
310 action_group);
311 g_object_unref(action_group);
312
313 setup_gsettings (applet);
314
315 /* load min icon size */
316 gsettings_changed_min_icon_size (applet->priv->settings, KEY_MIN_ICON_SIZE, applet);
317
318 applet->priv->builder = gtk_builder_new ();
319 gtk_builder_set_translation_domain (applet->priv->builder, GETTEXT_PACKAGE);
320 gtk_builder_add_from_resource (applet->priv->builder, NA_RESOURCE_PATH "notification-area-preferences-dialog.ui", NULL);
321}

402na_tray_applet_focus (GtkWidget *widget,
403 GtkDirectionType direction)
404{
405 NaTrayApplet *applet = NA_TRAY_APPLET (widget);
406
407 /* We let the grid handle the focus movement because we behave more like a
408 * container than a single applet. But if focus didn't move, we let the

411 return TRUE;
412
413 return GTK_WIDGET_CLASS (na_tray_applet_parent_class)->focus (widget, direction);
414}
