60static gboolean
61read_cb (GIOChannel *source,
62 GIOCondition condition,
63 gpointer user_data)
64{
65 GTask *task;
66 CommandData *data;
67 gchar buffer[BUFFER_SIZE];
68 gsize bytes_read;
69 GError *error;
70 GIOStatus status;
71
72 task = (GTask *) user_data;
73 data = g_task_get_task_data (task);
74
75 if (g_task_return_error_if_cancelled (task))
76 {
77 g_object_unref (task);
78
79 data->io_watch_id = 0;
80
81 return G_SOURCE_REMOVE;
82 }
83
84 error = NULL;
85 status = g_io_channel_read_chars (source, buffer, BUFFER_SIZE,
86 &bytes_read, &error);
87
88 if (status == G_IO_STATUS_AGAIN)
89 {
90 g_clear_error (&error);
91
92 return G_SOURCE_CONTINUE;
93 }
94 else if (status != G_IO_STATUS_NORMAL)
95 {
96 if (error != NULL)
97 {
98 g_task_return_error (task, error);
99 g_object_unref (task);
100 }
101
102 data->io_watch_id = 0;
103
104 return G_SOURCE_REMOVE;
105 }
106
107 g_string_append_len (data->input, buffer, bytes_read);
108
109 return G_SOURCE_CONTINUE;
110}

320void
321ma_command_run_async (MaCommand *command,
322 GCancellable *cancellable,
323 GAsyncReadyCallback callback,
324 gpointer user_data)
325{
326 GTask *task;
327 CommandData *data;
328 GSpawnFlags spawn_flags;
329 gint command_stdout;
330 GError *error;
331 GIOChannel *channel;
332 GIOStatus status;
333 GIOCondition condition;
334
335 g_return_if_fail (MA_IS_COMMAND (command));
336 g_return_if_fail (callback != NULL);
337
338 task = g_task_new (command, cancellable, callback, user_data);
339 g_task_set_source_tag (task, ma_command_run_async);
340
341 if (cancellable)
342 {
343 g_signal_connect_object (cancellable, "cancelled",
344 G_CALLBACK (cancelled_cb), task,
345 G_CONNECT_AFTER);
346 }
347
348 data = g_new0 (CommandData, 1);
349 g_task_set_task_data (task, data, command_data_free);
350
351 spawn_flags = G_SPAWN_SEARCH_PATH | G_SPAWN_DO_NOT_REAP_CHILD;
352 error = NULL;
353
354 if (!g_spawn_async_with_pipes (NULL, command->argv, NULL, spawn_flags,
355 NULL, NULL, &data->pid, NULL, &command_stdout,
356 NULL, &error))
357 {
358 g_task_return_error (task, error);
359 g_object_unref (task);
360
361 return;
362 }
363
364 channel = data->channel = g_io_channel_unix_new (command_stdout);
365 g_io_channel_set_close_on_unref (channel, TRUE);
366
367 g_assert (error == NULL);
368 status = g_io_channel_set_encoding (channel, NULL, &error);
369
370 if (status != G_IO_STATUS_NORMAL)
371 {
372 g_task_return_error (task, error);
373 g_object_unref (task);
374
375 return;
376 }
377
378 g_assert (error == NULL);
379 status = g_io_channel_set_flags (channel, G_IO_FLAG_NONBLOCK, &error);
380
381 if (status != G_IO_STATUS_NORMAL)
382 {
383 g_task_return_error (task, error);
384 g_object_unref (task);
385
386 return;
387 }
388
389 data->input = g_string_new (NULL);
390
391 condition = G_IO_IN | G_IO_PRI | G_IO_ERR | G_IO_HUP;
392 data->io_watch_id = g_io_add_watch (channel, condition, read_cb, task);
393
394 data->child_watch_id = g_child_watch_add (data->pid, child_watch_cb, task);
395}
