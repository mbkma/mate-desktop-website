91static void
92init_toolbar( BaseWindow *window, GtkActionGroup *group, int toolbar_id )
93{
94 ToolbarProps *props;
95 gboolean is_active;
96 GtkToggleAction *action;
97
98 props = get_toolbar_properties( toolbar_id );
99 if( props ){
100 is_active = na_settings_get_boolean( props->prefs_key, NULL, NULL );
101 if( is_active ){
102 action = GTK_TOGGLE_ACTION( gtk_action_group_get_action( group, props->ui_item ));
103 gtk_toggle_action_set_active( action, TRUE );
104 }
105 }

122void
123cact_main_toolbar_activate( CactMainWindow *window, int toolbar_id, GtkUIManager *ui_manager, gboolean is_active )
124{
125 static const gchar *thisfn = "cact_main_toolbar_activate";
126 ToolbarProps *props;
127 GtkWidget *toolbar, *hbox;
128
129 props = get_toolbar_properties( toolbar_id );
130 if( !props ){
131 return;
132 }
133
134 toolbar = gtk_ui_manager_get_widget( ui_manager, props->ui_path );
135 g_debug( "%s: toolbar=%p, path=%s, ref_count=%d", thisfn, ( void * ) toolbar, props->ui_path, G_OBJECT( toolbar )->ref_count );
136 hbox = base_window_get_widget( BASE_WINDOW( window ), "ToolbarHBox" );
137
138 if( is_active ){
139 gtk_container_add( GTK_CONTAINER( hbox ), toolbar );
140 gtk_widget_set_hexpand (toolbar, TRUE);
141 reorder_toolbars( hbox, toolbar_id, toolbar );
142 gtk_widget_show_all( toolbar );
143
144 } else {
145 gtk_container_remove( GTK_CONTAINER( hbox ), toolbar );
146 }
147 na_settings_set_boolean( props->prefs_key, is_active );
148}

150/*
151 * @hbox: the GtkHBox container
152 * @toolbar_id: toolbar identifier
153 * @handle: hbox child, which used to be a GtkHandleBox (for moveable
154 * toolbars), and becomes the GtkToolbar itself starting with Gtk 3.4
155 *
156 * reposition the newly activated toolbar in handle
157 * so that the relative positions of toolbars are respected in hbox
158 */
159static void
160reorder_toolbars( GtkWidget *hbox, int toolbar_id, GtkWidget *handle )
161{
162 int this_canonic_rel_pos;
163 int i;
164 GList *children, *ic;
165 int pos;
166 int canonic_pos;
167
168 this_canonic_rel_pos = 0;
169 for( i = 0 ; i < G_N_ELEMENTS( toolbar_pos ); ++ i ){
170 if( toolbar_pos[i] == toolbar_id ){
171 this_canonic_rel_pos = i;
172 break;
173 }
174 }
175 g_object_set_data( G_OBJECT( handle ), "toolbar-canonic-pos", GINT_TO_POINTER( this_canonic_rel_pos ));
176
177 pos = 0;
178 children = gtk_container_get_children( GTK_CONTAINER( hbox ));
179 for( ic = children ; ic ; ic = ic->next ){
180 canonic_pos = GPOINTER_TO_INT( g_object_get_data( G_OBJECT( ic->data ), "toolbar-canonic-pos" ));
181 if( canonic_pos >= this_canonic_rel_pos ){
182 break;
183 }
184 pos += 1;
185 }
186
187 gtk_box_reorder_child( GTK_BOX( hbox ), handle, pos );
188}

190static ToolbarProps *
191get_toolbar_properties( int toolbar_id )
192{
193 static const gchar *thisfn = "cact_main_toolbar_get_toolbar_properties";
194 ToolbarProps *props;
195 int i;
196
197 props = NULL;
198
199 for( i = 0 ; i < G_N_ELEMENTS( toolbar_props ) && props == NULL ; ++i ){
200 if( toolbar_props[i].id == toolbar_id ){
201 props = &toolbar_props[i];
202 }
203 }
204
205 if( !props ){
206 g_warning( "%s: unable to find toolbar properties for id=%d", thisfn, toolbar_id );
207 }
208
209 return( props );
