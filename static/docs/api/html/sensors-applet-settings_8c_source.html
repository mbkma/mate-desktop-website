41gchar* sensors_applet_settings_get_unique_id (const gchar *interface, const gchar *id, const gchar *path) {
42 gchar *unique_id = NULL;
43 gchar *unique_id_hash = NULL;
44 GChecksum *checksum = NULL;
45 guint8 digest[16];
46 gsize digest_len = sizeof (digest);
47
48 unique_id = g_strdup_printf ("%s/%s/%s", interface, id, path);
49
50 checksum = g_checksum_new (G_CHECKSUM_MD5);
51 g_checksum_update (checksum, (const guchar *) unique_id, strlen (unique_id));
52 g_checksum_get_digest (checksum, digest, &digest_len);
53 g_assert (digest_len == 16);
54
55 unique_id_hash = g_strdup (g_checksum_get_string (checksum));
56
57 g_checksum_free (checksum);
58 g_free (unique_id);
59
60 return unique_id_hash;
61}

63/* load sensors from gsettings with sorting */
64gboolean sensors_applet_settings_load_sensors (SensorsApplet *sensors_applet) {
65 /* everything gets stored except alarm timeout indexes, which
66 we set to -1, and visible which we set to false for all
67 parent nodes and true for all child nodes */
68
69 gchar *applet_path = NULL;
70 /* not sure about pointer, it is unclear if it is freed by loop, probably yes */
71 GVariantIter *iter = NULL;
72 gchar *gsuid = NULL;
73
74 /* string variables, to help free up memory in loop */
75 gchar *current_path = NULL;
76 gchar *current_id = NULL;
77 gchar *current_label = NULL;
78 gchar *current_interface = NULL;
79 gchar *current_low_alarm_command = NULL;
80 gchar *current_high_alarm_command = NULL;
81 gchar *current_graph_color = NULL;
82
83 /* get gsettings path for applet */
84 applet_path = mate_panel_applet_get_preferences_path (sensors_applet->applet);
85
86 /* get sensors-list array from gsettings */
87 g_settings_get (sensors_applet->settings, "sensors-list", GSGVTSA, &iter);
88
89 /* load sensor data into applet one by one
90 * first get unique id for the sensor
91 * then load sensor data from gsettings
92 * then add sensor
93 * gsuid is freed by g_variant_iter_loop() */
94 while (g_variant_iter_loop (iter, GSGVTS, &gsuid)) {
95
96 /* load sensor data from gsettings individually
97 * g_strdup_printf doesn't free args!
98 * applet_path is freed at the end */
99 gchar *path = g_strdup_printf ("%s%s/", applet_path, gsuid);
100 /*g_free (gsuid); // freed by loop */
101
102 /* make a schema which points to one sensor data */
103 GSettings *settings;
104 settings = g_settings_new_with_path ("org.mate.sensors-applet.sensor", path);
105 g_free (path);
106
107 /* laod sensor data to temp variables
108 * g_settings_get_string allocates memory!
109 * sensors_applet_add_sensor() gtk_tree_store_set() copies strings, so free them */
110 sensors_applet_add_sensor(sensors_applet,
111 current_path = g_settings_get_string (settings, PATH),
112 current_id = g_settings_get_string (settings, ID),
113 current_label = g_settings_get_string (settings, LABEL),
114 current_interface = g_settings_get_string (settings, INTERFACE),
115 g_settings_get_int (settings, SENSOR_TYPE),
116 g_settings_get_boolean (settings, ENABLED),
117 g_settings_get_double (settings, LOW_VALUE),
118 g_settings_get_double (settings, HIGH_VALUE),
119 g_settings_get_boolean (settings, ALARM_ENABLED),
120 current_low_alarm_command = g_settings_get_string (settings, LOW_ALARM_COMMAND),
121 current_high_alarm_command = g_settings_get_string (settings, HIGH_ALARM_COMMAND),
122 g_settings_get_int (settings, ALARM_TIMEOUT),
123 g_settings_get_double (settings, MULTIPLIER),
124 g_settings_get_double (settings, OFFSET),
125 g_settings_get_int (settings, ICON_TYPE),
126 current_graph_color = g_settings_get_string (settings, GRAPH_COLOR));
127
128 g_free (current_path);
129 g_free (current_id);
130 g_free (current_label);
131 g_free (current_interface);
132 g_free (current_low_alarm_command);
133 g_free (current_high_alarm_command);
134 g_free (current_graph_color);
135
136 g_object_unref (settings);
137
138 }
139
140 g_free (applet_path);
141
142 return TRUE;
143}

207/* compare two iters using their paths
208 * kinda same as active_sensor_compare () */
209static gint sensors_applet_settings_sort_sensors_iter_compare (SensorsApplet *sensors_applet, GtkTreeIter ti_a, GtkTreeIter ti_b) {
210
211 GtkTreePath *tp_a;
212 GtkTreePath *tp_b;
213 gint ret_val;
214
215 tp_a = gtk_tree_model_get_path (GTK_TREE_MODEL(sensors_applet->sensors), &ti_a);
216 tp_b = gtk_tree_model_get_path (GTK_TREE_MODEL(sensors_applet->sensors), &ti_b);
217
218 ret_val = gtk_tree_path_compare(tp_a, tp_b);
219
220 gtk_tree_path_free(tp_a);
221 gtk_tree_path_free(tp_b);
222
223 return ret_val;

226/* find two sensors in the sensors tree based on their hash and sort them
227 * Return codes:
228 * 0: something went very wrong...
229 * 1: normal run, a and b are in place
230 * 2: hash_b has already been removed from the system
231 * 3: hash_a is not found */
232static gint sensors_applet_settings_sort_sensors_sort (SensorsApplet *sensors_applet,
233 const gchar *hash_a,
234 const gchar *hash_b,
235 gboolean a_is_first) {
236
237 g_assert(sensors_applet);
238 g_assert(sensors_applet->sensors);
239
240 GtkTreeIter interfaces_iter;
241 GtkTreeIter sensors_iter;
242 gboolean not_end_of_interfaces = TRUE;
243 gboolean not_end_of_sensors = TRUE;
244
245 gchar *interface_name = NULL;
246 gchar *sensor_id = NULL;
247 gchar *sensor_path = NULL;
248 gchar *sensor_hash = NULL;
249
250 GtkTreeIter interface_iter_a;
251 GtkTreeIter interface_iter_b;
252 GtkTreeIter sensor_iter_a;
253 GtkTreeIter sensor_iter_b;
254 gboolean found_a = FALSE;
255 gboolean found_b = FALSE;
256
257 /* iterate through the sensor tree
258 * code from sensors-applet.c sensors_applet_add_sensor()
259 * first go through the interfaces */
260 for (not_end_of_interfaces = gtk_tree_model_get_iter_first(GTK_TREE_MODEL(sensors_applet->sensors), &interfaces_iter);
261 not_end_of_interfaces;
262 not_end_of_interfaces = gtk_tree_model_iter_next(GTK_TREE_MODEL(sensors_applet->sensors), &interfaces_iter)) {
263
264 /* get interface name */
265 gtk_tree_model_get(GTK_TREE_MODEL(sensors_applet->sensors), &interfaces_iter, INTERFACE_COLUMN, &interface_name, -1);
266
267 /* then go through the individual sensors under one interface */
268 for (not_end_of_sensors = gtk_tree_model_iter_children(GTK_TREE_MODEL(sensors_applet->sensors), &sensors_iter, &interfaces_iter);
269 not_end_of_sensors;
270 not_end_of_sensors = gtk_tree_model_iter_next(GTK_TREE_MODEL(sensors_applet->sensors), &sensors_iter)) {
271
272 gtk_tree_model_get(GTK_TREE_MODEL(sensors_applet->sensors), &sensors_iter,
273 PATH_COLUMN, &sensor_path,
274 ID_COLUMN, &sensor_id,
275 -1);
276
277 sensor_hash = sensors_applet_settings_get_unique_id (interface_name, sensor_id, sensor_path);
278
279 /* save interface_name and iters for swap */
280 if (g_ascii_strcasecmp(sensor_hash, hash_a) == 0) {
281 /* can copy by value, don't free */
282 interface_iter_a = interfaces_iter;
283 sensor_iter_a = sensors_iter;
284 found_a = TRUE;
285 }
286
287 if (g_ascii_strcasecmp(sensor_hash, hash_b) == 0) {
288 interface_iter_b = interfaces_iter;
289 sensor_iter_b = sensors_iter;
290 found_b = TRUE;
291 }
292
293 g_free(sensor_id);
294 g_free(sensor_path);
295 g_free(sensor_hash);
296
297 /* break after freeing loop variables */
298 if (found_a && found_b) {
299 break;
300 }
301 }
302
303 g_free(interface_name);
304
305 /* break after freeing loop variables */
306 if (found_a && found_b) {
307 break;
308 }
309 }
310
311 /* make the switch */
312 if (found_a && found_b) {
313
314 GtkTreeIter first_iter;
315 GtkTreeIter iter_next;
316
317 gint ret_val;
318
319 /* set a's interface to be the first interface in the sensors tree */
320 if (a_is_first) {
321
322 /* this 'fails' if the tree is empty, but this is already checked with the first loop */
323 gtk_tree_model_get_iter_first (GTK_TREE_MODEL(sensors_applet->sensors), &first_iter);
324
325 /* we are only interested in the case, where the two are not equal
326 * that means the a's interface is not the first interface */
327 if ( 0 != sensors_applet_settings_sort_sensors_iter_compare (sensors_applet, first_iter, interface_iter_a)) {
328
329 /* it should be, so it needs to be moved to the first position */
330 gtk_tree_store_move_after (GTK_TREE_STORE(sensors_applet->sensors), &interface_iter_a, NULL);
331 }
332
333 }
334
335 /* check b's interface position
336 * (1) if it is before a's if. - not possible, bc I have just set a's if. as first
337 * afterwards every new if. will be set right after (one) a's if.
338 * (0) if it is the same as a's, noop
339 * (-1) if it is after a, check if it is right after a, if not move it there */
340 ret_val = sensors_applet_settings_sort_sensors_iter_compare (sensors_applet, interface_iter_a, interface_iter_b);
341 if (-1 == ret_val) {
342
343 /* set iter_next to the iter after a's if. iter, can copy like this */
344 iter_next = interface_iter_a;
345 /* this 'fails' if there is no node after a's if. iter, but we already know, that at least b's if. iter is */
346 gtk_tree_model_iter_next (GTK_TREE_MODEL(sensors_applet->sensors), &iter_next);
347
348 /* the node right after a's if. iter is not b's if. iter, so move b's if. iter */
349 if (0 != sensors_applet_settings_sort_sensors_iter_compare (sensors_applet, iter_next, interface_iter_b)) {
350
351 gtk_tree_store_move_after (GTK_TREE_STORE(sensors_applet->sensors), &interface_iter_b, &interface_iter_a);
352 }
353 }
354
355 /* at this point the interfaces are sorted, the sensors are next */
356 /* set a to be the first sensor in the sensors tree, under the first if. iter */
357 if (a_is_first) {
358
359 /* this 'fails' if the tree is empty, but at least a is in it */
360 gtk_tree_model_iter_children(GTK_TREE_MODEL(sensors_applet->sensors), &first_iter, &interface_iter_a);
361
362 /* we are only interested in the case, where the two are not equal
363 * that means the a is not the first sensor */
364 if ( 0 != sensors_applet_settings_sort_sensors_iter_compare (sensors_applet, first_iter, sensor_iter_a)) {
365
366 /* it should be, so it needs to be moved to the first position */
367 gtk_tree_store_move_after (GTK_TREE_STORE(sensors_applet->sensors), &sensor_iter_a, NULL);
368 }
369
370 }
371
372 /* check b's position
373 * if a's if. and b's if is the same
374 * (1) if it is before a's if. - not possible, bc I have just set a as first
375 * afterwards every new sensor will be set right after (one) a
376 * (0) if it is the same as a's, not possible, a and b would be equal
377 * (-1) if it is after a, check if it is right after a, if not move it there */
378 if (0 == ret_val && -1 == sensors_applet_settings_sort_sensors_iter_compare (sensors_applet, sensor_iter_a, sensor_iter_b)) {
379
380 /* set iter_next to the iter after a's iter, can copy like this */
381 iter_next = sensor_iter_a;
382 /* this 'fails' if there is no node after a's iter, but we already know, that at least b's iter is */
383 gtk_tree_model_iter_next (GTK_TREE_MODEL(sensors_applet->sensors), &iter_next);
384
385 /* the node right after a's iter is not b's iter, so move b's iter */
386 if (0 != sensors_applet_settings_sort_sensors_iter_compare (sensors_applet, iter_next, sensor_iter_b)) {
387
388 gtk_tree_store_move_after (GTK_TREE_STORE(sensors_applet->sensors), &sensor_iter_b, &sensor_iter_a);
389 }
390 }
391
392 /* if a's if. and b's if is not the same
393 * as b comes after a, b must be under a new if.
394 * and so b must be the first node under that if. */
395 /* set b to be the first sensor in the sensors tree, under the b's if. iter */
396 if (-1 == ret_val) {
397
398 /* this 'fails' if the tree is empty, but at least b is in it */
399 gtk_tree_model_iter_children(GTK_TREE_MODEL(sensors_applet->sensors), &first_iter, &interface_iter_b);
400
401 /* we are only interested in the case, where the two are not equal
402 * that means the b is not the first sensor */
403 if ( 0 != sensors_applet_settings_sort_sensors_iter_compare (sensors_applet, first_iter, sensor_iter_b)) {
404
405 /* it should be, so it needs to be moved to the first position */
406 gtk_tree_store_move_after (GTK_TREE_STORE(sensors_applet->sensors), &sensor_iter_b, NULL);
407 }
408
409 }
410
411 return 1;
412
413 }
414
415 /* hash_b has already been removed from the system */
416 if (found_a && !found_b) {
417 return 2;
418 }
419
420 /* this is the least likely case
421 * if hash_a is not found, then the sensors-list starts with a sensor
422 * that has already been removed from the system */
423 if (!found_a) {
424 return 3;
425 }
426
427 /* we should never get here */
428 return 0;
429}

431/* sort sensors based on sensors-list array in gsettings */
432gboolean sensors_applet_settings_sort_sensors (SensorsApplet *sensors_applet) {
433
434 gchar **sensors_list = NULL;
435
436 gchar *hash_a = NULL;
437 gchar *hash_b = NULL;
438
439 gint ret_val = -1;
440 /* marks the first iteration */
441 gint first_counter = 1;
442
443 /* get sensors-list array from gsettings
444 * newly allocated, has to be freed fully
445 * for an empty array a pointer to a NULL pointer will be returned (g_variant_dup_strv ()) */
446 sensors_list = g_settings_get_strv (sensors_applet->settings, "sensors-list");
447
448 /* list is empty (on the first run) */
449 if (NULL == *sensors_list) {
450 /* still need to free the array, even if it is empty */
451 g_strfreev (sensors_list);
452 return FALSE;
453 }
454
455 gint i;
456 /* hash lists are equal */
457 gboolean hle = TRUE;
458 /* compare saved sorted list to newly loaded sensors' list */
459 for (i = 0; (NULL != sensors_applet->sensors_hash_array[i]) && (NULL != sensors_list[i]); i++) {
460
461 if (g_ascii_strcasecmp(sensors_applet->sensors_hash_array[i], sensors_list[i]) != 0) {
462 hle = FALSE;
463 break;
464 }
465 }
466
467 /* lists are the same -> nothing to do */
468 if (hle) {
469#ifdef SORT_DEBUG
470 syslog(LOG_ERR, "sensor sort: saved list is the same as the new one, returning");
471#endif
472
473 /* still need to free the array */
474 g_strfreev (sensors_list);
475 return TRUE;
476 }
477
478#ifdef SORT_DEBUG
479 sensors_applet_settings_print_sensors_tree (sensors_applet);
480#endif
481
482 for (i = 0; NULL != sensors_list[i]; i++) {
483
484 /* first pass */
485 if (i == 0) {
486 /* make a copy */
487 hash_a = g_strdup (sensors_list[i]);
488 continue;
489 }
490
491 hash_b = g_strdup (sensors_list[i]);
492
493 /* now that we have two hashes, find the two corresponding sensors and sort them
494 * if i == 1, we have both a and b set, this is the first time calling this function */
495 ret_val = sensors_applet_settings_sort_sensors_sort (sensors_applet, hash_a, hash_b, (i == first_counter));
496
497 /* hash_b has already been removed from the system
498 * don't free hash_a, don't reassign hash_b
499 * as we already know that b is no longer in the system */
500 if (2 == ret_val) {
501 continue;
502 }
503
504 /* after sorting free hash_a (it should be in place) and reassign hash_b to hash_a */
505 g_free (hash_a);
506 hash_a = hash_b;
507
508 /* hash_a not found
509 * set hash_b as hash_a and set the first element counter
510 * (essentially skip the first element in the sensors-list array) */
511 if (3 == ret_val) {
512 first_counter++;
513 }
514 }
515
516#ifdef SORT_DEBUG
517 sensors_applet_settings_print_sensors_tree (sensors_applet);
518#endif
519
520 /* hash_a already freed
521 * only in the following case do we need to free it */
522 if (2 == ret_val) {
523 g_free (hash_a);
524 }
525 g_free (hash_b);
526
527 g_strfreev (sensors_list);
528
529 /* reorder active sensors based on reordered sensors tree */
530 sensors_applet_reorder_sensors (sensors_applet);
531
532 return TRUE;
533}

535/* save sensor data under a unique hash

537gboolean sensors_applet_settings_save_sensors (SensorsApplet *sensors_applet) {
538
539 /* write everything to GSettings except VISIBLE and ALARM_TIMEOUT_INDEX
540 * for stepping through GtkTreeStore data structure */
541 GtkTreeIter interfaces_iter;
542 GtkTreeIter sensors_iter;
543 gboolean not_end_of_interfaces = TRUE;
544 gboolean not_end_of_sensors = TRUE;
545 gchar *applet_path = NULL;
546
547 gchar *current_path = NULL;
548 gchar *current_id = NULL;
549 gchar *current_label = NULL;
550 gchar *current_interface = NULL;
551 gchar *current_low_alarm_command = NULL;
552 gchar *current_high_alarm_command = NULL;
553 gchar *current_graph_color = NULL;
554 gboolean current_enable,
555 current_alarm_enable;
556 gdouble current_low_value,
557 current_high_value,
558 current_multiplier,
559 current_offset;
560 guint current_alarm_timeout,
561 current_sensor_type,
562 current_icon_type;
563
564 /* data structure (array) to be able to save sensors list to gsettings */
565 GVariantBuilder builder;
566 g_variant_builder_init (&builder, G_VARIANT_TYPE (GSGVTSA));
567
568 applet_path = mate_panel_applet_get_preferences_path (sensors_applet->applet);
569
570 /* now step through the GtkTreeStore sensors to find which sensors are available / loaded */
571 for (not_end_of_interfaces = gtk_tree_model_get_iter_first(GTK_TREE_MODEL(sensors_applet->sensors), &interfaces_iter);
572 not_end_of_interfaces;
573 not_end_of_interfaces = gtk_tree_model_iter_next(GTK_TREE_MODEL(sensors_applet->sensors), &interfaces_iter)) {
574
575 for (not_end_of_sensors = gtk_tree_model_iter_children(GTK_TREE_MODEL(sensors_applet->sensors), &sensors_iter, &interfaces_iter);
576 not_end_of_sensors;
577 not_end_of_sensors = gtk_tree_model_iter_next(GTK_TREE_MODEL(sensors_applet->sensors), &sensors_iter)) {
578
579 gtk_tree_model_get(GTK_TREE_MODEL(sensors_applet->sensors),
580 &sensors_iter,
581 PATH_COLUMN, &current_path,
582 ID_COLUMN, &current_id,
583 LABEL_COLUMN, &current_label,
584 INTERFACE_COLUMN, &current_interface,
585 SENSOR_TYPE_COLUMN, &current_sensor_type,

589 ALARM_ENABLE_COLUMN, &current_alarm_enable,
590 LOW_ALARM_COMMAND_COLUMN, &current_low_alarm_command,
591 HIGH_ALARM_COMMAND_COLUMN, &current_high_alarm_command,
592 ALARM_TIMEOUT_COLUMN, &current_alarm_timeout,

595 ICON_TYPE_COLUMN, &current_icon_type,

597 -1);
