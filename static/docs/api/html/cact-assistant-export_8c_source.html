141{
142 static const gchar *thisfn = "cact_assistant_export_register_type";
143 GType type;
144
145 static GTypeInfo info = {
146 sizeof( CactAssistantExportClass ),
147 ( GBaseInitFunc ) NULL,
148 ( GBaseFinalizeFunc ) NULL,

294static void
295instance_dispose( GObject *window )
296{
297 static const gchar *thisfn = "cact_assistant_export_instance_dispose";
298 CactAssistantExport *self;
299 GtkAssistant *assistant;
300 GtkWidget *page;
301 guint pos;
302 GtkWidget *pane;
303
304 g_return_if_fail( CACT_IS_ASSISTANT_EXPORT( window ));
305
306 self = CACT_ASSISTANT_EXPORT( window );
307
308 if( !self->private->dispose_has_run ){
309 g_debug( "%s: window=%p (%s)", thisfn, ( void * ) window, G_OBJECT_TYPE_NAME( window ));
310
311 self->private->dispose_has_run = TRUE;
312
313 g_object_unref( self->private->items_view );
314
315 if( self->private->selected_items ){
316 self->private->selected_items = na_object_free_items( self->private->selected_items );
317 }
318
319 assistant = GTK_ASSISTANT( base_window_get_gtk_toplevel( BASE_WINDOW( window )));
320 page = gtk_assistant_get_nth_page( assistant, ASSIST_PAGE_ACTIONS_SELECTION );
321 pane = na_gtk_utils_find_widget_by_name( GTK_CONTAINER( page ), "p1-HPaned" );
322 pos = gtk_paned_get_position( GTK_PANED( pane ));
323 na_settings_set_uint( NA_IPREFS_EXPORT_ASSISTANT_PANED, pos );
324
325 /* chain up to the parent class */
326 if( G_OBJECT_CLASS( st_parent_class )->dispose ){
327 G_OBJECT_CLASS( st_parent_class )->dispose( window );
328 }
329 }

472static void
473on_base_initialize_base_window( CactAssistantExport *window, gpointer user_data )
474{
475 static const gchar *thisfn = "cact_assistant_export_on_base_initialize_base_window";
476 GtkAssistant *assistant;
477 GtkWidget *page;
478 guint pos;
479 GtkWidget *pane;
480 gchar *format;
481 gboolean mandatory;
482 GtkWidget *tree_view;
483
484 g_return_if_fail( CACT_IS_ASSISTANT_EXPORT( window ));
485
486 if( !window->private->dispose_has_run ){
487 g_debug( "%s: window=%p, user_data=%p", thisfn, ( void * ) window, ( void * ) user_data );
488
489 assistant = GTK_ASSISTANT( base_window_get_gtk_toplevel( BASE_WINDOW( window )));
490
491 /* intro page is always true
492 */
493 page = gtk_assistant_get_nth_page( assistant, ASSIST_PAGE_INTRO );
494 gtk_assistant_set_page_complete( assistant, page, TRUE );
495
496 /* set the slider position of the item selection page
497 */
498 pos = na_settings_get_uint( NA_IPREFS_EXPORT_ASSISTANT_PANED, NULL, NULL );
499 if( pos ){
500 page = gtk_assistant_get_nth_page( assistant, ASSIST_PAGE_ACTIONS_SELECTION );
501 pane = na_gtk_utils_find_widget_by_name( GTK_CONTAINER( page ), "p1-HPaned" );
502 gtk_paned_set_position( GTK_PANED( pane ), pos );
503 }
504
505 /* initialize the export format tree view
506 */
507 page = gtk_assistant_get_nth_page( assistant, ASSIST_PAGE_FORMAT_SELECTION );
508 tree_view = na_gtk_utils_find_widget_by_name( GTK_CONTAINER( page ), "p3-ExportFormatTreeView" );
509 format = na_settings_get_string( NA_IPREFS_EXPORT_PREFERRED_FORMAT, NULL, &mandatory );
510 na_ioptions_list_set_editable(
511 NA_IOPTIONS_LIST( window ), tree_view,
512 !mandatory && !window->private->preferences_locked );
513 na_ioptions_list_set_default(
514 NA_IOPTIONS_LIST( window ), tree_view, format );
515 g_free( format );
516 }
