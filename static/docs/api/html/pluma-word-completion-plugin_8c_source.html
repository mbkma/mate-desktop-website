132static void
133update_activation (GtkSourceCompletionWords *provider,
134 GSettings *settings)
135{
136 GtkSourceCompletionActivation activation;
137
138 g_object_get (provider, "activation", &activation, NULL);
139
140 if (g_settings_get_boolean (settings, SETTINGS_KEY_INTERACTIVE_COMPLETION))
141 {
142 activation |= (unsigned int)GTK_SOURCE_COMPLETION_ACTIVATION_INTERACTIVE;
143 }
144 else
145 {
146 activation &= ~(unsigned int)GTK_SOURCE_COMPLETION_ACTIVATION_INTERACTIVE;
147 }
148
149 g_object_set (provider, "activation", activation, NULL);
150}

160static GtkSourceCompletionWords *
161create_provider (void)
162{
163 GtkSourceCompletionWords *provider;
164 GSettings *settings;
165
166 provider = gtk_source_completion_words_new (_("Document Words"), NULL);
167
168 settings = g_settings_new (WORD_COMPLETION_SETTINGS_BASE);
169
170 g_settings_bind (settings, SETTINGS_KEY_MINIMUM_WORD_SIZE,
171 provider, "minimum-word-size",
172 G_SETTINGS_BIND_GET);
173
174 update_activation (provider, settings);
175
176 g_signal_connect_object (settings,
177 "changed::" SETTINGS_KEY_INTERACTIVE_COMPLETION,
178 G_CALLBACK (on_interactive_completion_changed_cb),
179 provider,
180 0);
181
182 g_object_unref (settings);
183
184 return provider;
185}

217static void
218pluma_word_completion_view_activate (PlumaViewActivatable *activatable)
219{
220 PlumaWordCompletionPluginPrivate *priv;
221 GtkSourceCompletion *completion;
222 GtkSourceCompletionProvider *provider;
223 GtkTextBuffer *buf;
224
225 pluma_debug (DEBUG_PLUGINS);
226
227 priv = PLUMA_WORD_COMPLETION_PLUGIN (activatable)->priv;
228
229 priv->window = gtk_widget_get_toplevel (GTK_WIDGET (priv->view));
230
231 /* We are disposing the window */
232 g_object_ref (priv->window);
233
234 completion = gtk_source_view_get_completion (GTK_SOURCE_VIEW (priv->view));
235 buf = gtk_text_view_get_buffer (GTK_TEXT_VIEW (priv->view));
236
237 provider = g_object_get_data (G_OBJECT (priv->window), WINDOW_PROVIDER);
238
239 if (provider == NULL)
240 {
241 /* Standalone provider */
242 provider = GTK_SOURCE_COMPLETION_PROVIDER (create_provider ());
243 }
244
245 priv->provider = g_object_ref (provider);
246
247 gtk_source_completion_add_provider (completion, provider, NULL);
248 gtk_source_completion_words_register (GTK_SOURCE_COMPLETION_WORDS (provider),
249 buf);
250}

262

264 buf = gtk_text_view_get_buffer (GTK_TEXT_VIEW (priv->view));

267 priv->provider,
268 NULL);
269
270 gtk_source_completion_words_unregister (GTK_SOURCE_COMPLETION_WORDS (priv->provider),
271 buf);
272}

274static GtkWidget *
275pluma_word_completion_create_configure_widget (PeasGtkConfigurable *configurable)
276{
277 GtkBuilder *builder;
278 GtkWidget *content;
279 GtkWidget *interactive_completion;
280 GtkWidget *min_word_size;
281 GSettings *settings;
282 gchar *data_dir;
283 gchar *ui_file;
284
285 data_dir = peas_extension_base_get_data_dir (PEAS_EXTENSION_BASE (configurable));
286 ui_file = g_build_filename (data_dir, "pluma-word-completion-configure.ui", NULL);
287
288 builder = gtk_builder_new_from_resource ("/org/mate/pluma/plugins/wordcompletion/ui/pluma-word-completion-configure.ui");
289 gtk_builder_set_translation_domain (builder, GETTEXT_PACKAGE);
290
291 content = GTK_WIDGET (gtk_builder_get_object (builder, "content"));
292 g_object_ref (content);
293
294 interactive_completion = GTK_WIDGET (gtk_builder_get_object (builder, "check_button_interactive_completion"));
295 min_word_size = GTK_WIDGET (gtk_builder_get_object (builder, "spin_button_min_word_size"));
296
297 g_free(data_dir);
298 g_free(ui_file);
299 g_object_unref (builder);
300
301 settings = g_settings_new (WORD_COMPLETION_SETTINGS_BASE);
302
303 g_settings_bind (settings, SETTINGS_KEY_INTERACTIVE_COMPLETION,
304 interactive_completion, "active",
305 G_SETTINGS_BIND_DEFAULT | G_SETTINGS_BIND_GET_NO_CHANGES);
306
307 g_settings_bind (settings, SETTINGS_KEY_MINIMUM_WORD_SIZE,
308 min_word_size, "value",
309 G_SETTINGS_BIND_DEFAULT | G_SETTINGS_BIND_GET_NO_CHANGES);
310
311 g_object_unref (settings);
312
313 return content;
314}
