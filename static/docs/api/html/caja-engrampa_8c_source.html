35static void
36extract_to_callback (CajaMenuItem *item,
37 gpointer user_data)
38{
39 GList *files, *scan;
40 char *default_dir;
41 char *quoted_default_dir;
42 GString *cmd;
43
44 files = g_object_get_data (G_OBJECT (item), "files");
45 default_dir = caja_file_info_get_parent_uri (files->data);
46 quoted_default_dir = g_shell_quote (default_dir);
47
48 cmd = g_string_new ("engrampa");
49 g_string_append_printf(cmd," --default-dir=%s --extract", quoted_default_dir);
50
51 for (scan = files; scan; scan = scan->next) {
52 char *uri, *quoted_uri;
53
54 uri = caja_file_info_get_uri (scan->data);
55 quoted_uri = g_shell_quote (uri);
56 g_string_append_printf (cmd, " %s", quoted_uri);
57 g_free (uri);
58 g_free (quoted_uri);
59 }
60
61#ifdef MATE_ENABLE_DEBUG
62 g_print ("EXEC: %s\n", cmd->str);
63#endif
64
65 g_spawn_command_line_async (cmd->str, NULL);
66
67 g_string_free (cmd, TRUE);
68 g_free (default_dir);
69 g_free (quoted_default_dir);
70}

207static FileMimeInfo
208get_file_mime_info (CajaFileInfo *file)
209{
210 FileMimeInfo file_mime_info;
211 int i;
212
213 file_mime_info.is_archive = FALSE;
214 file_mime_info.is_derived_archive = FALSE;
215 file_mime_info.is_compressed_archive = FALSE;
216
217 for (i = 0; archive_mime_types[i].mime_type != NULL; i++)
218 if (caja_file_info_is_mime_type (file, archive_mime_types[i].mime_type)) {
219 char *mime_type;
220 char *content_type_mime_file;
221 char *content_type_mime_compare;
222
223 mime_type = caja_file_info_get_mime_type (file);
224
225 content_type_mime_file = g_content_type_from_mime_type (mime_type);
226 content_type_mime_compare = g_content_type_from_mime_type (archive_mime_types[i].mime_type);
227
228 file_mime_info.is_archive = TRUE;
229 file_mime_info.is_compressed_archive = archive_mime_types[i].is_compressed;
230 if ((content_type_mime_file != NULL) && (content_type_mime_compare != NULL))
231 file_mime_info.is_derived_archive = ! g_content_type_equals (content_type_mime_file, content_type_mime_compare);
232
233 g_free (mime_type);
234 g_free (content_type_mime_file);
235 g_free (content_type_mime_compare);
236
237 return file_mime_info;
238 }
239
240 return file_mime_info;
241}

243static gboolean
244unsupported_scheme (CajaFileInfo *file)
245{
246 gboolean result = FALSE;
247 GFile *location;
248 char *scheme;
249
250 location = caja_file_info_get_location (file);
251 scheme = g_file_get_uri_scheme (location);
252
253 if (scheme != NULL) {
254 const char *unsupported[] = { "trash", "computer", NULL };
255 int i;
256
257 for (i = 0; unsupported[i] != NULL; i++)
258 if (strcmp (scheme, unsupported[i]) == 0)
259 result = TRUE;
260 }
261
262 g_free (scheme);
263 g_object_unref (location);
264
265 return result;
266}

268static GList *
269caja_fr_get_file_items (CajaMenuProvider *provider,
270 GtkWidget *window,
271 GList *files)
272{
273 GList *items = NULL;
274 GList *scan;
275 gboolean can_write = TRUE;
276 gboolean one_item;
277 gboolean one_archive = FALSE;
278 gboolean one_derived_archive = FALSE;
279 gboolean one_compressed_archive = FALSE;
280 gboolean all_archives = TRUE;
281 gboolean all_archives_derived = TRUE;
282 gboolean all_archives_compressed = TRUE;
283
284 if (files == NULL)
285 return NULL;
286
287 if (unsupported_scheme ((CajaFileInfo *) files->data))
288 return NULL;
289
290 for (scan = files; scan; scan = scan->next) {
291 CajaFileInfo *file = scan->data;
292 FileMimeInfo file_mime_info;
293
294 file_mime_info = get_file_mime_info (file);
295
296 if (all_archives && ! file_mime_info.is_archive)
297 all_archives = FALSE;
298
299 if (all_archives_compressed && file_mime_info.is_archive && ! file_mime_info.is_compressed_archive)
300 all_archives_compressed = FALSE;
301
302 if (all_archives_derived && file_mime_info.is_archive && ! file_mime_info.is_derived_archive)
303 all_archives_derived = FALSE;
304
305 if (can_write) {
306 CajaFileInfo *parent;
307
308 parent = caja_file_info_get_parent_info (file);
309 can_write = caja_file_info_can_write (parent);
310 g_object_unref (parent);
311 }
312 }
313
314 /**/
315
316 one_item = (files != NULL) && (files->next == NULL);
317 one_archive = one_item && all_archives;
318 one_derived_archive = one_archive && all_archives_derived;
319 one_compressed_archive = one_archive && all_archives_compressed;
320
321 if (all_archives && can_write) {
322 CajaMenuItem *item;
323
324 item = caja_menu_item_new ("CajaFr::extract_here",
325 _("Extract Here"),
326 /* Translators: the current position is the current folder */
327 _("Extract the selected archive to the current position"),
328 "drive-harddisk");
329 g_signal_connect (item,
330 "activate",
331 G_CALLBACK (extract_here_callback),
332 provider);
333 g_object_set_data_full (G_OBJECT (item),
334 "files",
335 caja_file_info_list_copy (files),
336 (GDestroyNotify) caja_file_info_list_free);
337
338 items = g_list_append (items, item);
339 }
340 if (all_archives) {
341 CajaMenuItem *item;
342
343 item = caja_menu_item_new ("CajaFr::extract_to",
344 _("Extract To..."),
345 _("Extract the selected archive"),
346 "drive-harddisk");
347 g_signal_connect (item,
348 "activate",
349 G_CALLBACK (extract_to_callback),
350 provider);
351 g_object_set_data_full (G_OBJECT (item),
352 "files",
353 caja_file_info_list_copy (files),
354 (GDestroyNotify) caja_file_info_list_free);
355
356 items = g_list_append (items, item);
357
358 }
359
360 if (! one_compressed_archive || one_derived_archive) {
361 CajaMenuItem *item;
362
363 item = caja_menu_item_new ("CajaFr::add",
364 _("Compress..."),
365 _("Create a compressed archive with the selected objects"),
366 "package-x-generic");
367 g_signal_connect (item,
368 "activate",
369 G_CALLBACK (add_callback),
370 provider);
371 g_object_set_data_full (G_OBJECT (item),
372 "files",
373 caja_file_info_list_copy (files),
374 (GDestroyNotify) caja_file_info_list_free);
375
376 items = g_list_append (items, item);
377 }
378
379 return items;
380}
