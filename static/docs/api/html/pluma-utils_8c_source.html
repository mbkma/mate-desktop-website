235/*
236 * n: len of the string in bytes
237 */
238gboolean
239g_utf8_caselessnmatch (const char *s1, const char *s2, gssize n1, gssize n2)
240{
241 gchar *casefold;
242 gchar *normalized_s1;
243 gchar *normalized_s2;
244 gint len_s1;
245 gint len_s2;
246 gboolean ret = FALSE;
247
248 g_return_val_if_fail (s1 != NULL, FALSE);
249 g_return_val_if_fail (s2 != NULL, FALSE);
250 g_return_val_if_fail (n1 > 0, FALSE);
251 g_return_val_if_fail (n2 > 0, FALSE);
252
253 casefold = g_utf8_casefold (s1, n1);
254 normalized_s1 = g_utf8_normalize (casefold, -1, G_NORMALIZE_NFD);
255 g_free (casefold);
256
257 casefold = g_utf8_casefold (s2, n2);
258 normalized_s2 = g_utf8_normalize (casefold, -1, G_NORMALIZE_NFD);
259 g_free (casefold);
260
261 len_s1 = strlen (normalized_s1);
262 len_s2 = strlen (normalized_s2);
263
264 if (len_s1 < len_s2)
265 goto finally_2;
266
267 ret = (strncmp (normalized_s1, normalized_s2, len_s2) == 0);
268
269finally_2:
270 g_free (normalized_s1);
271 g_free (normalized_s2);
272
273 return ret;
274}

358gchar *
359pluma_utils_escape_search_text (const gchar* text)
360{
361 GString *str;
362 gint length;
363 const gchar *p;
364 const gchar *end;
365
366 if (text == NULL)
367 return NULL;
368
369 pluma_debug_message (DEBUG_SEARCH, "Text: %s", text);
370
371 length = strlen (text);
372
373 /* no escape when typing.
374 * The short circuit works only for ascii, but we only
375 * care about not escaping a single '\' */
376 if (length == 1)
377 return g_strdup (text);
378
379 str = g_string_new ("");
380
381 p = text;
382 end = text + length;
383
384 while (p != end)
385 {
386 const gchar *next;
387 next = g_utf8_next_char (p);
388
389 switch (*p)
390 {
391 case '\n':
392 g_string_append (str, "\\n");
393 break;
394 case '\r':
395 g_string_append (str, "\\r");
396 break;
397 case '\t':
398 g_string_append (str, "\\t");
399 break;
400 case '\\':
401 g_string_append (str, "\\\\");
402 break;
403 default:
404 g_string_append_len (str, p, next - p);
405 break;
406 }
407
408 p = next;
409 }
410
411 return g_string_free (str, FALSE);
412}

414gchar *
415pluma_utils_unescape_search_text (const gchar *text)
416{
417 GString *str;
418 gint length;
419 gboolean drop_prev = FALSE;
420 const gchar *cur;
421 const gchar *end;
422 const gchar *prev;
423
424 if (text == NULL)
425 return NULL;
426
427 length = strlen (text);
428
429 str = g_string_new ("");
430
431 cur = text;
432 end = text + length;
433 prev = NULL;
434
435 while (cur != end)
436 {
437 const gchar *next;
438 next = g_utf8_next_char (cur);
439
440 if (prev && (*prev == '\\'))
441 {
442 switch (*cur)
443 {
444 case 'n':
445 str = g_string_append (str, "\n");
446 break;
447 case 'r':
448 str = g_string_append (str, "\r");
449 break;
450 case 't':
451 str = g_string_append (str, "\t");
452 break;
453 case '\\':
454 str = g_string_append (str, "\\");
455 drop_prev = TRUE;
456 break;
457 default:
458 str = g_string_append (str, "\\");
459 str = g_string_append_len (str, cur, next - cur);
460 break;
461 }
462 }
463 else if (*cur != '\\')
464 {
465 str = g_string_append_len (str, cur, next - cur);
466 }
467 else if ((next == end) && (*cur == '\\'))
468 {
469 str = g_string_append (str, "\\");
470 }
471
472 if (!drop_prev)
473 {
474 prev = cur;
475 }
476 else
477 {
478 prev = NULL;
479 drop_prev = FALSE;
480 }
481
482 cur = next;
483 }
484
485 return g_string_free (str, FALSE);
486}

572static gchar *
573pluma_utils_str_truncate (const gchar *string,
574 guint truncate_length,
575 gboolean middle)
576{
577 GString *truncated;
578 guint length;
579 guint n_chars;
580 guint num_left_chars;
581 guint right_offset;
582 guint delimiter_length;
583 const gchar *delimiter = "\342\200\246";
584
585 g_return_val_if_fail (string != NULL, NULL);
586
587 length = strlen (string);
588
589 g_return_val_if_fail (g_utf8_validate (string, length, NULL), NULL);
590
591 /* It doesnt make sense to truncate strings to less than
592 * the size of the delimiter plus 2 characters (one on each
593 * side)
594 */
595 delimiter_length = g_utf8_strlen (delimiter, -1);
596 if (truncate_length < (delimiter_length + 2)) {
597 return g_strdup (string);
598 }
599
600 n_chars = g_utf8_strlen (string, length);
601
602 /* Make sure the string is not already small enough. */
603 if (n_chars <= truncate_length) {
604 return g_strdup (string);
605 }
606
607 /* Find the 'middle' where the truncation will occur. */
608 if (middle)
609 {
610 num_left_chars = (truncate_length - delimiter_length) / 2;
611 right_offset = n_chars - truncate_length + num_left_chars + delimiter_length;
612
613 truncated = g_string_new_len (string,
614 g_utf8_offset_to_pointer (string, num_left_chars) - string);
615 g_string_append (truncated, delimiter);
616 g_string_append (truncated, g_utf8_offset_to_pointer (string, right_offset));
617 }
618 else
619 {
620 num_left_chars = truncate_length - delimiter_length;
621 truncated = g_string_new_len (string,
622 g_utf8_offset_to_pointer (string, num_left_chars) - string);
623 g_string_append (truncated, delimiter);
624 }
625
626 return g_string_free (truncated, FALSE);
627}

643gchar *
644pluma_utils_make_valid_utf8 (const char *name)
645{
646 GString *string;
647 const char *remainder, *invalid;
648 int remaining_bytes, valid_bytes;
649
650 g_return_val_if_fail (name != NULL, NULL);
651
652 string = NULL;
653 remainder = name;
654 remaining_bytes = strlen (name);
655
656 while (remaining_bytes != 0) {
657 if (g_utf8_validate (remainder, remaining_bytes, &invalid)) {
658 break;
659 }
660 valid_bytes = invalid - remainder;
661
662 if (string == NULL) {
663 string = g_string_sized_new (remaining_bytes);
664 }
665 g_string_append_len (string, remainder, valid_bytes);
666 /* append U+FFFD REPLACEMENT CHARACTER */
667 g_string_append (string, "\357\277\275");
668
669 remaining_bytes -= valid_bytes + 1;
670 remainder = invalid + 1;
671 }
672
673 if (string == NULL) {
674 return g_strdup (name);
675 }
676
677 g_string_append (string, remainder);
678
679 g_assert (g_utf8_validate (string->str, -1, NULL));
680
681 return g_string_free (string, FALSE);
682}

726gchar *
727pluma_utils_location_get_dirname_for_display (GFile *location)
728{
729 gchar *uri;
730 gchar *res;
731 GMount *mount;
732
733 g_return_val_if_fail (location != NULL, NULL);
734
735 /* we use the parse name, that is either the local path
736 * or an uri but which is utf8 safe */
737 uri = g_file_get_parse_name (location);
738
739 /* FIXME: this is sync... is it a problem? */
740 mount = g_file_find_enclosing_mount (location, NULL, NULL);
741 if (mount != NULL)
742 {
743 gchar *mount_name;
744 gchar *path = NULL;
745 gchar *dirname;
746
747 mount_name = g_mount_get_name (mount);
748 g_object_unref (mount);
749
750 /* obtain the "path" part of the uri */
751 pluma_utils_decode_uri (uri,
752 NULL, NULL,
753 NULL, NULL,
754 &path);
755
756 if (path == NULL)
757 {
758 dirname = pluma_utils_uri_get_dirname (uri);
759 }
760 else
761 {
762 dirname = pluma_utils_uri_get_dirname (path);
763 }
764
765 if (dirname == NULL || strcmp (dirname, ".") == 0)
766 {
767 res = mount_name;
768 }
769 else
770 {
771 res = g_strdup_printf ("%s %s", mount_name, dirname);
772 g_free (mount_name);
773 }
774
775 g_free (path);
776 g_free (dirname);
777 }
778 else
779 {
780 /* fallback for local files or uris without mounts */
781 res = pluma_utils_uri_get_dirname (uri);
782 }
783
784 g_free (uri);
785
786 return res;
787}

789gchar *
790pluma_utils_replace_home_dir_with_tilde (const gchar *uri)
791{
792 gchar *tmp;
793 gchar *home;
794
795 g_return_val_if_fail (uri != NULL, NULL);
796
797 /* Note that g_get_home_dir returns a const string */
798 tmp = (gchar *)g_get_home_dir ();
799
800 if (tmp == NULL)
801 return g_strdup (uri);
802
803 home = g_filename_to_utf8 (tmp, -1, NULL, NULL, NULL);
804 if (home == NULL)
805 return g_strdup (uri);
806
807 if (strcmp (uri, home) == 0)
808 {
809 g_free (home);
810
811 return g_strdup ("~");
812 }
813
814 tmp = home;
815 home = g_strdup_printf ("%s/", tmp);
816 g_free (tmp);
817
818 if (g_str_has_prefix (uri, home))
819 {
820 gchar *res;
821
822 res = g_strdup_printf ("~/%s", uri + strlen (home));
823
824 g_free (home);
825
826 return res;
827 }
828
829 g_free (home);
830
831 return g_strdup (uri);
832}

1434gboolean
1435pluma_utils_decode_uri (const gchar *uri,
1436 gchar **scheme,
1437 gchar **user,
1438 gchar **host,
1439 gchar **port,
1440 gchar **path
1441)
1442{
1443 /* Largely copied from glib/gio/gdummyfile.c:_g_decode_uri. This
1444 * functionality should be in glib/gio, but for now we implement it
1445 * ourselves (see bug #546182) */
1446
1447 const char *p, *in, *hier_part_start, *hier_part_end;
1448 char *out;
1449 char c;
1450
1451 /* From RFC 3986 Decodes:
1452 * URI = scheme ":" hier-part [ "?" query ] [ "#" fragment ]
1453 */
1454
1455 p = uri;
1456
1457 null_ptr (scheme);
1458 null_ptr (user);
1459 null_ptr (port);
1460 null_ptr (host);
1461 null_ptr (path);
1462
1463 /* Decode scheme:
1464 * scheme = ALPHA *( ALPHA / DIGIT / "+" / "-" / "." )
1465 */
1466
1467 if (!g_ascii_isalpha (*p))
1468 return FALSE;
1469
1470 while (1)
1471 {
1472 c = *p++;
1473
1474 if (c == ':')
1475 break;
1476
1477 if (!(g_ascii_isalnum(c) ||
1478 c == '+' ||
1479 c == '-' ||
1480 c == '.'))
1481 return FALSE;
1482 }
1483
1484 if (scheme)
1485 {
1486 *scheme = g_malloc (p - uri);
1487 out = *scheme;
1488
1489 for (in = uri; in < p - 1; in++)
1490 *out++ = g_ascii_tolower (*in);
1491
1492 *out = '\0';
1493 }
1494
1495 hier_part_start = p;
1496 hier_part_end = p + strlen (p);
1497
1498 if (hier_part_start[0] == '/' && hier_part_start[1] == '/')
1499 {
1500 const char *authority_start, *authority_end;
1501 const char *userinfo_start, *userinfo_end;
1502 const char *host_start, *host_end;
1503 const char *port_start;
1504
1505 authority_start = hier_part_start + 2;
1506 /* authority is always followed by / or nothing */
1507 authority_end = memchr (authority_start, '/', hier_part_end - authority_start);
1508
1509 if (authority_end == NULL)
1510 authority_end = hier_part_end;
1511
1512 /* 3.2:
1513 * authority = [ userinfo "@" ] host [ ":" port ]
1514 */
1515
1516 userinfo_end = memchr (authority_start, '@', authority_end - authority_start);
1517
1518 if (userinfo_end)
1519 {
1520 userinfo_start = authority_start;
1521
1522 if (user)
1523 *user = g_uri_unescape_segment (userinfo_start, userinfo_end, NULL);
1524
1525 if (user && *user == NULL)
1526 {
1527 if (scheme)
1528 g_free (*scheme);
1529
1530 return FALSE;
1531 }
1532
1533 host_start = userinfo_end + 1;
1534 }
1535 else
1536 host_start = authority_start;
1537
1538 port_start = memchr (host_start, ':', authority_end - host_start);
1539
1540 if (port_start)
1541 {
1542 host_end = port_start++;
1543
1544 if (port)
1545 *port = g_strndup (port_start, authority_end - port_start);
1546 }
1547 else
1548 host_end = authority_end;
1549
1550 if (host)
1551 *host = g_strndup (host_start, host_end - host_start);
1552
1553 hier_part_start = authority_end;
1554 }
1555
1556 if (path)
1557 *path = g_uri_unescape_segment (hier_part_start, hier_part_end, "/");
1558
1559 return TRUE;
1560}
