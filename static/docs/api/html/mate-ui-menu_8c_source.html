29 GtkAccelGroup *accel_group,

31{
32 g_return_val_if_fail(label != NULL, NULL);

38 if (accel != NULL && accel_group != NULL)

47 }
48 }
49
50 return item;
51}

86 gtk_widget_show_all(box);

101GtkWidget *
102mate_ui_menu_new_from_entries(const MateUiMenuEntry *entries,
103 gsize n_entries,
104 GtkAccelGroup *accel_group)
105{
106 g_return_val_if_fail(entries != NULL || n_entries == 0, NULL);
107
108 GtkWidget *menu = gtk_menu_new();
109
110 for (gsize i = 0; i < n_entries; i++)
111 {
112 const MateUiMenuEntry *entry = &entries[i];
113 GtkWidget *item;
114
115 /* Check for separator */
116 if (entry->label == NULL && entry->action_name == NULL)
117 {
118 item = gtk_separator_menu_item_new();
119 }
120 else if (entry->icon_name != NULL)
121 {
122 item = mate_ui_menu_item_new_with_icon(entry->label,
123 entry->icon_name,
124 entry->action_name);
125 if (entry->accel != NULL && accel_group != NULL)
126 {
127 guint key;
128 GdkModifierType mods;
129 gtk_accelerator_parse(entry->accel, &key, &mods);
130 if (key != 0)
131 {
132 gtk_widget_add_accelerator(item, "activate", accel_group,
133 key, mods, GTK_ACCEL_VISIBLE);
134 }
135 }
136 }
137 else
138 {
139 item = mate_ui_menu_item_new_with_action(entry->label,
140 entry->action_name,
141 accel_group,
142 entry->accel);
143 }
144
145 gtk_menu_shell_append(GTK_MENU_SHELL(menu), item);
146 gtk_widget_show(item);
147 }
148
149 return menu;
150}

197GMenuModel *
198mate_ui_menu_model_new_from_entries(const MateUiSubmenu *submenus,
199 gsize n_submenus)
200{
201 g_return_val_if_fail(submenus != NULL || n_submenus == 0, NULL);
202
203 GMenu *menubar = g_menu_new();
204
205 for (gsize i = 0; i < n_submenus; i++)
206 {
207 const MateUiSubmenu *submenu = &submenus[i];
208 GMenu *menu = g_menu_new();
209
210 for (gsize j = 0; j < submenu->n_entries; j++)
211 {
212 const MateUiMenuEntry *entry = &submenu->entries[j];
213
214 /* Check for separator */
215 if (entry->label == NULL && entry->action_name == NULL)
216 {
217 /* GMenu doesn't have explicit separators; create a section break */
218 GMenu *section = g_menu_new();
219 g_menu_append_section(menu, NULL, G_MENU_MODEL(section));
220 g_object_unref(section);
221 }
222 else
223 {
224 /* Remove mnemonic underscore for GMenu */
225 gchar *label = g_strdup(entry->label);
226 gchar *p = label;
227 gchar *q = label;
228 while (*p)
229 {
230 if (*p != '_')
231 *q++ = *p;
232 p++;
233 }
234 *q = '\0';
235
236 GMenuItem *item = g_menu_item_new(label, entry->action_name);
237 g_free(label);
238
239 if (entry->icon_name != NULL)
240 {
241 GIcon *icon = g_themed_icon_new(entry->icon_name);
242 g_menu_item_set_icon(item, icon);
243 g_object_unref(icon);
244 }
245
246 if (entry->accel != NULL)
247 {
248 g_menu_item_set_attribute(item, "accel", "s", entry->accel);
249 }
250
251 g_menu_append_item(menu, item);
252 g_object_unref(item);
253 }
254 }
255
256 /* Remove mnemonic underscore from submenu label */
257 gchar *sublabel = g_strdup(submenu->label);
258 gchar *p = sublabel;
259 gchar *q = sublabel;
260 while (*p)
261 {
262 if (*p != '_')
263 *q++ = *p;
264 p++;
265 }
266 *q = '\0';
267
268 g_menu_append_submenu(menubar, sublabel, G_MENU_MODEL(menu));
269 g_free(sublabel);
270 g_object_unref(menu);
271 }
272
273 return G_MENU_MODEL(menubar);
274}

288GtkWidget *
289mate_ui_menu_add_recent_chooser(GtkMenu *menu,
290 const gchar *label,
291 GtkRecentFilter *filter,
292 GCallback callback,
293 gpointer user_data)
294{
295 g_return_val_if_fail(GTK_IS_MENU(menu), NULL);
296 g_return_val_if_fail(label != NULL, NULL);
297
298 GtkWidget *item = gtk_menu_item_new_with_mnemonic(label);
299
300 GtkRecentManager *manager = gtk_recent_manager_get_default();
301 GtkWidget *recent_menu = gtk_recent_chooser_menu_new_for_manager(manager);
302
303 gtk_recent_chooser_set_show_tips(GTK_RECENT_CHOOSER(recent_menu), TRUE);
304 gtk_recent_chooser_set_sort_type(GTK_RECENT_CHOOSER(recent_menu),
305 GTK_RECENT_SORT_MRU);
306 gtk_recent_chooser_set_limit(GTK_RECENT_CHOOSER(recent_menu), 10);
307 gtk_recent_chooser_set_local_only(GTK_RECENT_CHOOSER(recent_menu), FALSE);
308
309 if (filter != NULL)
310 {
311 gtk_recent_chooser_add_filter(GTK_RECENT_CHOOSER(recent_menu), filter);
312 gtk_recent_chooser_set_filter(GTK_RECENT_CHOOSER(recent_menu), filter);
313 }
314
315 if (callback != NULL)
316 {
317 g_signal_connect(recent_menu, "item-activated", callback, user_data);
318 }
319
320 gtk_menu_item_set_submenu(GTK_MENU_ITEM(item), recent_menu);
321 gtk_menu_shell_append(GTK_MENU_SHELL(menu), item);
322 gtk_widget_show(item);
323
324 return item;
325}
