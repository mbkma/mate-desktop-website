48/* art_affine_flip modified to work with cairo_matrix_t */
49static void
50_eom_cairo_matrix_flip (cairo_matrix_t *dst, const cairo_matrix_t *src, gboolean horiz, gboolean vert)
51{
52 dst->xx = horiz ? -src->xx : src->xx;
53 dst->yx = horiz ? -src->yx : src->yx;
54 dst->xy = vert ? -src->xy : src->xy;
55 dst->yy = vert ? -src->yy : src->yy;
56 dst->x0 = horiz ? -src->x0 : src->x0;
57 dst->y0 = vert ? -src->y0 : src->y0;
58}

60static gboolean
61_cairo_ctx_supports_jpg_metadata (cairo_t *cr)
62{
63 cairo_surface_t *surface = cairo_get_target (cr);
64 cairo_surface_type_t type = cairo_surface_get_type (surface);

66 /* Based on cairo-1.10 */
67 return (type == CAIRO_SURFACE_TYPE_PDF || type == CAIRO_SURFACE_TYPE_PS ||
68 type == CAIRO_SURFACE_TYPE_SVG || type == CAIRO_SURFACE_TYPE_WIN32_PRINTING);

71static void
72eom_print_draw_page (GtkPrintOperation *operation,
73 GtkPrintContext *context,
74 gint page_nr,
75 gpointer user_data)
76{
77 cairo_t *cr;
78 gdouble dpi_x, dpi_y;
79 gdouble x0, y0;
80 gdouble scale_factor;
81 gdouble p_width, p_height;
82 gint width, height;
83 EomPrintData *data;
84 GtkPageSetup *page_setup;
85
86 eom_debug (DEBUG_PRINTING);
87
88 data = (EomPrintData *) user_data;
89
90 scale_factor = data->scale_factor/100;
91
92 dpi_x = gtk_print_context_get_dpi_x (context);
93 dpi_y = gtk_print_context_get_dpi_y (context);
94
95 switch (data->unit) {
96 case GTK_UNIT_INCH:
97 x0 = data->left_margin * dpi_x;
98 y0 = data->top_margin * dpi_y;
99 break;
100 case GTK_UNIT_MM:
101 x0 = data->left_margin * dpi_x/25.4;
102 y0 = data->top_margin * dpi_y/25.4;
103 break;
104 default:
105 g_assert_not_reached ();
106 }
107
108 cr = gtk_print_context_get_cairo_context (context);
109
110 cairo_translate (cr, x0, y0);
111
112 page_setup = gtk_print_context_get_page_setup (context);
113 p_width = gtk_page_setup_get_page_width (page_setup, GTK_UNIT_POINTS);
114 p_height = gtk_page_setup_get_page_height (page_setup, GTK_UNIT_POINTS);
115
116 eom_image_get_size (data->image, &width, &height);
117
118 /* this is both a workaround for a bug in cairo's PDF backend, and
119 a way to ensure we are not printing outside the page margins */
120 cairo_rectangle (cr, 0, 0, MIN (width*scale_factor, p_width), MIN (height*scale_factor, p_height));
121 cairo_clip (cr);
122
123 cairo_scale (cr, scale_factor, scale_factor);
124
125#ifdef HAVE_RSVG
126 if (eom_image_is_svg (data->image))
127 {
128 RsvgHandle *svg = eom_image_get_svg (data->image);
129
130 rsvg_handle_render_cairo (svg, cr);
131 return;
132 } else
133#endif
134 /* JPEGs can be attached to the cairo surface which simply embeds the JPEG file into the
135 * destination PDF skipping (PNG-)recompression. This should reduce PDF sizes enormously. */
136 if (eom_image_is_jpeg (data->image) && _cairo_ctx_supports_jpg_metadata (cr))
137 {
138 GFile *file;
139 char *img_data;
140 gsize data_len;
141 cairo_surface_t *surface = NULL;
142
143 eom_debug_message (DEBUG_PRINTING, "Attaching image to cairo surface");
144
145 file = eom_image_get_file (data->image);
146 if (g_file_load_contents (file, NULL, &img_data, &data_len, NULL, NULL))
147 {
148 EomTransform *tf = eom_image_get_transform (data->image);
149 EomTransform *auto_tf = eom_image_get_autorotate_transform (data->image);
150 cairo_matrix_t mx, mx2;
151
152 if (!tf && auto_tf) {
153 /* If only autorotation data present,
154 * make it the normal rotation. */
155 tf = auto_tf;
156 auto_tf = NULL;
157 }
158
159 /* Care must be taken with height and width values. They are not the original
160 * values but were affected by the transformation. As the surface needs to be
161 * generated using the original dimensions they might need to be flipped. */
162 if (tf) {
163 if (auto_tf) {
164 /* If we have an autorotation apply
165 * it before the others */
166 tf = eom_transform_compose (auto_tf, tf);
167 }
168
169 switch (eom_transform_get_transform_type (tf)) {
170 case EOM_TRANSFORM_ROT_90:
171 surface = cairo_image_surface_create (
172 CAIRO_FORMAT_RGB24, height, width);
173 cairo_rotate (cr, 90.0 * (G_PI/180.0));
174 cairo_translate (cr, 0.0, -width);
175 break;
176 case EOM_TRANSFORM_ROT_180:
177 surface = cairo_image_surface_create (
178 CAIRO_FORMAT_RGB24, width, height);
179 cairo_rotate (cr, 180.0 * (G_PI/180.0));
180 cairo_translate (cr, -width, -height);
181 break;
182 case EOM_TRANSFORM_ROT_270:
183 surface = cairo_image_surface_create (
184 CAIRO_FORMAT_RGB24, height, width);
185 cairo_rotate (cr, 270.0 * (G_PI/180.0));
186 cairo_translate (cr, -height, 0.0);
187 break;
188 case EOM_TRANSFORM_FLIP_HORIZONTAL:
189 surface = cairo_image_surface_create (
190 CAIRO_FORMAT_RGB24, width, height);
191 cairo_matrix_init_identity (&mx);
192 _eom_cairo_matrix_flip (&mx2, &mx, TRUE, FALSE);
193 cairo_transform (cr, &mx2);
194 cairo_translate (cr, -width, 0.0);
195 break;
196 case EOM_TRANSFORM_FLIP_VERTICAL:
197 surface = cairo_image_surface_create (
198 CAIRO_FORMAT_RGB24, width, height);
199 cairo_matrix_init_identity (&mx);
200 _eom_cairo_matrix_flip (&mx2, &mx, FALSE, TRUE);
201 cairo_transform (cr, &mx2);
202 cairo_translate (cr, 0.0, -height);
203 break;
204 case EOM_TRANSFORM_TRANSPOSE:
205 surface = cairo_image_surface_create (
206 CAIRO_FORMAT_RGB24, height, width);
207 cairo_matrix_init_rotate (&mx, 90.0 * (G_PI/180.0));
208 cairo_matrix_init_identity (&mx2);
209 _eom_cairo_matrix_flip (&mx2, &mx2, TRUE, FALSE);
210 cairo_matrix_multiply (&mx2, &mx, &mx2);
211 cairo_transform (cr, &mx2);
212 break;
213 case EOM_TRANSFORM_TRANSVERSE:
214 surface = cairo_image_surface_create (
215 CAIRO_FORMAT_RGB24, height, width);
216 cairo_matrix_init_rotate (&mx, 90.0 * (G_PI/180.0));
217 cairo_matrix_init_identity (&mx2);
218 _eom_cairo_matrix_flip (&mx2, &mx2, FALSE, TRUE);
219 cairo_matrix_multiply (&mx2, &mx, &mx2);
220 cairo_transform (cr, &mx2);
221 cairo_translate (cr, -height , -width);
222 break;
223 case EOM_TRANSFORM_NONE:
224 default:
225 surface = cairo_image_surface_create (
226 CAIRO_FORMAT_RGB24, width, height);
227 break;
228 }
229 }
230
231 if (!surface)
232 surface = cairo_image_surface_create (CAIRO_FORMAT_RGB24,
233 width, height);
234 cairo_surface_set_mime_data (surface,
235 CAIRO_MIME_TYPE_JPEG,
236 (unsigned char*)img_data, data_len,
237 g_free, img_data);
238 cairo_set_source_surface (cr, surface, 0, 0);
239 cairo_paint (cr);
240 cairo_surface_destroy (surface);
241 g_object_unref (file);
242 return;
243 }
244 g_object_unref (file);
245
246 }
247
248 {
249 GdkPixbuf *pixbuf;
250
251 pixbuf = eom_image_get_pixbuf (data->image);
252 gdk_cairo_set_source_pixbuf (cr, pixbuf, 0, 0);
253 cairo_paint (cr);
254 g_object_unref (pixbuf);
255 }
256}

313GtkPrintOperation *
314eom_print_operation_new (EomImage *image,
315 GtkPrintSettings *print_settings,
316 GtkPageSetup *page_setup)
317{
318 GtkPrintOperation *print;
319 EomPrintData *data;
320 gint width, height;
321
322 eom_debug (DEBUG_PRINTING);
323
324 print = gtk_print_operation_new ();
325
326 data = g_slice_new0 (EomPrintData);
327
328 data->left_margin = 0;
329 data->top_margin = 0;
330 data->scale_factor = 100;
331 data->image = g_object_ref (image);
332 data->unit = GTK_UNIT_INCH;
333
334 eom_image_get_size (image, &width, &height);
335
336 if (page_setup == NULL)
337 page_setup = gtk_page_setup_new ();
338
339 if (height >= width) {
340 gtk_page_setup_set_orientation (page_setup,
341 GTK_PAGE_ORIENTATION_PORTRAIT);
342 } else {
343 gtk_page_setup_set_orientation (page_setup,
344 GTK_PAGE_ORIENTATION_LANDSCAPE);
345 }
346
347 gtk_print_operation_set_print_settings (print, print_settings);
348 gtk_print_operation_set_default_page_setup (print,
349 page_setup);
350 gtk_print_operation_set_n_pages (print, 1);
351 gtk_print_operation_set_job_name (print,
352 eom_image_get_caption (image));
353 gtk_print_operation_set_embed_page_setup (print, TRUE);
354
355 g_signal_connect (print, "draw_page",
356 G_CALLBACK (eom_print_draw_page),
357 data);
358 g_signal_connect (print, "create-custom-widget",
359 G_CALLBACK (eom_print_create_custom_widget),
360 data);
361 g_signal_connect (print, "custom-widget-apply",
362 G_CALLBACK (eom_print_custom_widget_apply),
363 data);
364 g_signal_connect (print, "end-print",
365 G_CALLBACK (eom_print_end_print),
366 data);
367 g_signal_connect (print, "update-custom-widget",
368 G_CALLBACK (eom_print_image_setup_update),
369 data);
370
371 gtk_print_operation_set_custom_tab_label (print, _("Image Settings"));
372
373 return print;
374}

376static GKeyFile *
377eom_print_get_key_file (void)
378{
379 GKeyFile *key_file;
380 GError *error = NULL;
381 gchar *filename;
382 GFile *file;
383 const gchar *dot_dir = eom_util_dot_dir ();
384
385 filename = g_build_filename (dot_dir, EOM_PRINT_SETTINGS_FILE, NULL);
386
387 file = g_file_new_for_path (filename);
388 key_file = g_key_file_new ();
389
390 if (g_file_query_exists (file, NULL)) {
391 g_key_file_load_from_file (key_file, filename,
392 G_KEY_FILE_KEEP_COMMENTS | G_KEY_FILE_KEEP_TRANSLATIONS,
393 &error);
394 if (error) {
395 g_warning ("Error loading print settings file: %s", error->message);
396 g_error_free (error);
397 g_object_unref (file);
398 g_free (filename);
399 g_key_file_free (key_file);
400 return NULL;
401 }
402 }
403
404 g_object_unref (file);
405 g_free (filename);
406
407 return key_file;
408}
