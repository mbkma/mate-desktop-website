110void
111math_window_set_show_history(MathWindow *window, gboolean visible)
112{
113 g_return_if_fail(window != NULL);
114
115 if (math_window_get_show_history(window) == visible)
116 return;
117
118 window->priv->show_history = visible;
119
120 if (visible)
121 {
122 gtk_widget_show(GTK_WIDGET(window->priv->history));
123 gtk_window_set_resizable(GTK_WINDOW(window), TRUE);
124 }
125 else
126 {
127 gtk_widget_hide(GTK_WIDGET(window->priv->history));
128 gtk_window_resize(GTK_WINDOW(window), 1, 1); //FIXME: is there a better way to shrink window size to its children?
129 gtk_window_set_resizable(GTK_WINDOW(window), FALSE);
130 }
131 g_object_notify(G_OBJECT(window), "show-history");
132}

201static gboolean
202key_press_cb(MathWindow *window, GdkEventKey *event)
203{
204 gboolean result;
205 g_signal_emit_by_name(window->priv->display, "key-press-event", event, &result);
206
207 /* Keyboard navigation for history */
208 if ((event->state & GDK_MOD1_MASK) == GDK_MOD1_MASK && (event->keyval == GDK_KEY_Up || event->keyval == GDK_KEY_Down))
209 {
210 switch (event->keyval)
211 {
212 case GDK_KEY_Up:
213 math_history_set_current(window->priv->history, -1);
214 break;
215 case GDK_KEY_Down:
216 math_history_set_current(window->priv->history, 1);
217 break;
218 }
219
220 MathHistoryEntry *entry = math_history_get_entry_at(window->priv->history, math_history_get_current(window->priv->history));
221 if (entry)
222 {
223 gchar *equation_string = math_history_entry_get_equation(entry);
224 math_equation_set(window->priv->equation, equation_string);
225 g_free(equation_string);
226 }
227 return TRUE;
228 }
229 else if (math_buttons_get_mode (window->priv->buttons) == PROGRAMMING && (event->state & GDK_CONTROL_MASK) == GDK_CONTROL_MASK) {
230 switch(event->keyval)
231 {
232 /* Binary */
233 case GDK_KEY_b:
234 math_equation_set_base (window->priv->equation, 2);
235 return TRUE;
236 /* Octal */
237 case GDK_KEY_o:
238 math_equation_set_base (window->priv->equation, 8);
239 return TRUE;
240 /* Decimal */
241 case GDK_KEY_d:
242 math_equation_set_base (window->priv->equation, 10);
243 return TRUE;
244 /* Hexdecimal */
245 case GDK_KEY_h:
246 math_equation_set_base (window->priv->equation, 16);
247 return TRUE;
248 }
249 }
250
251 return result;
252}

280{

282

284 "help:mate-calc",

287
288 if (error != NULL)
289 {
290 GtkWidget *d;
291 /* Translators: Error message displayed when unable to launch help browser */
292 const char *message = _("Unable to open help file");
293
294 d = gtk_message_dialog_new(GTK_WINDOW(window), GTK_DIALOG_MODAL | GTK_DIALOG_DESTROY_WITH_PARENT,
295 GTK_MESSAGE_ERROR, GTK_BUTTONS_CLOSE, "%s", message);
296 gtk_message_dialog_format_secondary_text(GTK_MESSAGE_DIALOG(d), "%s", error->message);
297 g_signal_connect(d, "response", G_CALLBACK(gtk_widget_destroy), NULL);
298 gtk_window_present(GTK_WINDOW (d));
299
300 g_error_free(error);
301 }

307static void about_cb(GtkWidget* widget, MathWindow* window)
308{
309 const char* documenters[] = {
310 N_("Sun Microsystems"),
311 N_("MATE Documentation Team"),
312 NULL
313 };
314
315 /* The license this software is under (GPL2+) */
316 const char* license[] = {
317 N_("MATE Calculator is free software; you can redistribute it and/or modify "
318 "it under the terms of the GNU General Public License as published by "
319 "the Free Software Foundation; either version 2 of the License, or "
320 "(at your option) any later version."),
321 N_("MATE Calculator is distributed in the hope that it will be useful, "
322 "but WITHOUT ANY WARRANTY; without even the implied warranty of "
323 "MERCHANTABILITY or FITNESS FOR A PARTICULAR PURPOSE. See the "
324 "GNU General Public License for more details."),
325 N_("You should have received a copy of the GNU General Public License "
326 "along with MATE Calculator; if not, write to the Free Software Foundation, Inc., "
327 "51 Franklin Street, Fifth Floor, Boston, MA 02110-1301, USA.")
328 };
329
330 char *license_trans = g_strjoin ("\n\n", _(license[0]), _(license[1]), _(license[2]), NULL);
331
332 GKeyFile *key_file;
333 GBytes *bytes;
334 const guint8 *data;
335 gsize data_len;
336 GError *error = NULL;
337 char **authors;
338 gsize n_authors = 0, i;
339 gchar *comments = NULL;
340
341 bytes = g_resources_lookup_data ("/org/mate/calculator/ui/mate-calc.about", G_RESOURCE_LOOKUP_FLAGS_NONE, &error);
342 g_assert_no_error (error);
343
344 data = g_bytes_get_data (bytes, &data_len);
345 key_file = g_key_file_new ();
346 g_key_file_load_from_data (key_file, (const char *) data, data_len, 0, &error);
347 g_assert_no_error (error);
348
349 authors = g_key_file_get_string_list (key_file, ABOUT_GROUP, "Authors", &n_authors, NULL);
350
351 g_key_file_free (key_file);
352 g_bytes_unref (bytes);
353
354 for (i = 0; i < n_authors; ++i)
355 authors[i] = EMAILIFY (authors[i]);
356
357 const char **p;
358
359 for (p = documenters; *p; ++p)
360 *p = _(*p);
361
362 comments = g_strdup_printf (_("Calculator with financial and scientific modes.\nUsing GNU MPFR %s and GNU MPC %s"),
363 mpfr_get_version (), mpc_get_version ());
364
365 gtk_show_about_dialog(GTK_WINDOW(window),
366 "program-name", _("MATE Calculator"),
367 "version", VERSION,
368 "title", _("About MATE Calculator"),
369 "copyright", _("Copyright \xc2\xa9 19862010 The GCalctool authors\n"

371 "license", license_trans,
372 "comments", comments,
373 "authors", authors,
374 "documenters", documenters,
375 "translator_credits", _("translator-credits"),
376 "wrap-license", TRUE,
377 "website", PACKAGE_URL,
378 "icon-name", "accessories-calculator",
379 "logo-icon-name", "accessories-calculator",
380 NULL);
381
382 g_free (comments);
383 g_strfreev (authors);
384 g_free (license_trans);
385}

484static GtkWidget *gtk_image_menu_item_new_from_icon (const gchar *icon_name,
485 const gchar *label_name,
486 GtkAccelGroup *accel_group)
487{
488 gchar *concat = g_strconcat (label_name, " ", NULL);
489 GtkWidget *box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
490 GtkWidget *icon = gtk_image_new_from_icon_name (icon_name, GTK_ICON_SIZE_MENU);

492 GtkWidget *menu_item = gtk_menu_item_new ();
493
494 g_free (concat);
495
496 gtk_container_add (GTK_CONTAINER (box), icon);
497
498 gtk_label_set_use_underline (GTK_LABEL (label), TRUE);
499 gtk_label_set_xalign (GTK_LABEL (label), 0.0);
500
501 gtk_accel_label_set_accel_widget (GTK_ACCEL_LABEL (label), menu_item);
502
503 gtk_box_pack_end (GTK_BOX (box), label, TRUE, TRUE, 0);
504
505 gtk_container_add (GTK_CONTAINER (menu_item), box);
506
507 gtk_widget_show_all (menu_item);
508
509 return menu_item;
510}

512static void create_menu(MathWindow* window)
513{
514 GtkAccelGroup* accel_group;
515 GtkWidget* menu;
516 GtkWidget* menu_item;
517 GSList* group = NULL;
518
519 accel_group = gtk_accel_group_new();
520 gtk_window_add_accel_group(GTK_WINDOW(window), accel_group);
521
522 /* Calculator menu */
523 #define CALCULATOR_MENU_LABEL _("_Calculator")
524 /* Mode menu */
525 #define MODE_MENU_LABEL _("_Mode")
526 /* View menu */
527 #define VIEW_MENU_LABEL _("_View")
528 /* Help menu label */
529 #define HELP_MENU_LABEL _("_Help")
530 /* Basic menu label */
531 #define MODE_BASIC_LABEL _("_Basic")
532 /* Advanced menu label */
533 #define MODE_ADVANCED_LABEL _("_Advanced")
534 /* Financial menu label */
535 #define MODE_FINANCIAL_LABEL _("_Financial")
536 /* Programming menu label */
537 #define MODE_PROGRAMMING_LABEL _("_Programming")
538 /* History menu label */
539 #define VIEW_HISTORY_LABEL _("_History")
540 /* Help>Contents menu label */
541 #define HELP_CONTENTS_LABEL _("_Contents")
542
543 menu = add_menu(window->priv->menu_bar, CALCULATOR_MENU_LABEL);
544 menu_item = add_menu_item(menu, gtk_image_menu_item_new_from_icon("edit-copy",_("_Copy"), accel_group), G_CALLBACK(copy_cb), window);
545 gtk_widget_add_accelerator(menu_item, "activate", accel_group, GDK_KEY_C, GDK_CONTROL_MASK, GTK_ACCEL_VISIBLE);
546 menu_item = add_menu_item(menu, gtk_image_menu_item_new_from_icon("edit-paste",_("_Paste"), accel_group), G_CALLBACK(paste_cb), window);
547 gtk_widget_add_accelerator(menu_item, "activate", accel_group, GDK_KEY_V, GDK_CONTROL_MASK, GTK_ACCEL_VISIBLE);
548 menu_item = add_menu_item(menu, gtk_image_menu_item_new_from_icon("edit-undo",_("_Undo"), accel_group), G_CALLBACK(undo_cb), window);
549 gtk_widget_add_accelerator(menu_item, "activate", accel_group, GDK_KEY_Z, GDK_CONTROL_MASK, GTK_ACCEL_VISIBLE);
550 menu_item = add_menu_item(menu, gtk_image_menu_item_new_from_icon("edit-redo",_("_Redo"), accel_group), G_CALLBACK(redo_cb), window);
551 gtk_widget_add_accelerator(menu_item, "activate", accel_group, GDK_KEY_Z, GDK_CONTROL_MASK | GDK_SHIFT_MASK, GTK_ACCEL_VISIBLE);
552 menu_item = add_menu_item(menu, gtk_image_menu_item_new_from_icon("edit-clear",_("_Clear History"), accel_group), G_CALLBACK(clear_history_cb), window);
553 gtk_widget_add_accelerator(menu_item, "activate", accel_group, GDK_KEY_Delete, GDK_CONTROL_MASK | GDK_SHIFT_MASK, GTK_ACCEL_VISIBLE);
554 add_menu_item(menu, gtk_separator_menu_item_new(), NULL, NULL);
555 add_menu_item(menu, gtk_image_menu_item_new_from_icon("preferences-desktop",_("_Preferences"), accel_group), G_CALLBACK(show_preferences_cb), window);
556 add_menu_item(menu, gtk_separator_menu_item_new(), NULL, NULL);
557 menu_item = add_menu_item(menu, gtk_image_menu_item_new_from_icon("application-exit",_("_Quit"), accel_group), G_CALLBACK(quit_cb), window);
558 gtk_widget_add_accelerator(menu_item, "activate", accel_group, GDK_KEY_Q, GDK_CONTROL_MASK, GTK_ACCEL_VISIBLE);
559 gtk_widget_add_accelerator(menu_item, "activate", accel_group, GDK_KEY_W, GDK_CONTROL_MASK, GTK_ACCEL_VISIBLE);
560
561 menu = add_menu(window->priv->menu_bar, MODE_MENU_LABEL);
562 window->priv->mode_basic_menu_item = add_menu_item(menu, radio_menu_item_new(&group, MODE_BASIC_LABEL), G_CALLBACK(mode_changed_cb), window);
563 g_object_set_data(G_OBJECT(window->priv->mode_basic_menu_item), "calcmode", GINT_TO_POINTER(BASIC));
564 window->priv->mode_advanced_menu_item = add_menu_item(menu, radio_menu_item_new(&group, MODE_ADVANCED_LABEL), G_CALLBACK(mode_changed_cb), window);
565 g_object_set_data(G_OBJECT(window->priv->mode_advanced_menu_item), "calcmode", GINT_TO_POINTER(ADVANCED));
566 window->priv->mode_financial_menu_item = add_menu_item(menu, radio_menu_item_new(&group, MODE_FINANCIAL_LABEL), G_CALLBACK(mode_changed_cb), window);
567 g_object_set_data(G_OBJECT(window->priv->mode_financial_menu_item), "calcmode", GINT_TO_POINTER(FINANCIAL));
568 window->priv->mode_programming_menu_item = add_menu_item(menu, radio_menu_item_new(&group, MODE_PROGRAMMING_LABEL), G_CALLBACK(mode_changed_cb), window);
569 g_object_set_data(G_OBJECT(window->priv->mode_programming_menu_item), "calcmode", GINT_TO_POINTER(PROGRAMMING));
570
571 menu = add_menu(window->priv->menu_bar, VIEW_MENU_LABEL);
572 window->priv->view_history_menu_item = add_menu_item(menu, gtk_check_menu_item_new_with_mnemonic(VIEW_HISTORY_LABEL), G_CALLBACK(history_check_toggled_cb), window);
573 gtk_widget_add_accelerator(window->priv->view_history_menu_item, "activate", accel_group, GDK_KEY_H, GDK_CONTROL_MASK | GDK_SHIFT_MASK, GTK_ACCEL_VISIBLE);
574
575 menu = add_menu(window->priv->menu_bar, HELP_MENU_LABEL);
576 menu_item = add_menu_item(menu, gtk_image_menu_item_new_from_icon("help-browser", HELP_CONTENTS_LABEL, accel_group), G_CALLBACK(help_cb), window);
577 gtk_widget_add_accelerator(menu_item, "activate", accel_group, GDK_KEY_F1, 0, GTK_ACCEL_VISIBLE);
578 add_menu_item(menu, gtk_image_menu_item_new_from_icon("help-about",_("_About"), accel_group), G_CALLBACK(about_cb), window);
579}
