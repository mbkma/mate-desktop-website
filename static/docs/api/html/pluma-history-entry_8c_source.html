71{
72 PlumaHistoryEntry *entry;

74 g_return_if_fail (PLUMA_IS_HISTORY_ENTRY (object));

186static GSList *
187get_history_list (PlumaHistoryEntry *entry)
188{
189 GtkListStore *store;
190 GtkTreeIter iter;
191 gboolean valid;
192 GSList *list = NULL;
193
194 store = get_history_store (entry);
195
196 valid = gtk_tree_model_get_iter_first (GTK_TREE_MODEL (store),
197 &iter);
198
199 while (valid)
200 {
201 gchar *str;
202
203 gtk_tree_model_get (GTK_TREE_MODEL (store),
204 &iter,
205 0, &str,
206 -1);
207
208 list = g_slist_prepend (list, str);
209
210 valid = gtk_tree_model_iter_next (GTK_TREE_MODEL (store),
211 &iter);
212 }
213
214 return g_slist_reverse (list);
215}

233static gboolean
234remove_item (GtkListStore *store,
235 const gchar *text)
236{
237 GtkTreeIter iter;
238
239 g_return_val_if_fail (text != NULL, FALSE);
240
241 if (!gtk_tree_model_get_iter_first (GTK_TREE_MODEL (store), &iter))
242 return FALSE;
243
244 do
245 {
246 gchar *item_text;
247
248 gtk_tree_model_get (GTK_TREE_MODEL (store),
249 &iter,
250 0,
251 &item_text,
252 -1);
253
254 if (item_text != NULL &&
255 strcmp (item_text, text) == 0)
256 {
257 gtk_list_store_remove (store, &iter);
258 g_free (item_text);
259 return TRUE;
260 }
261
262 g_free (item_text);
263
264 } while (gtk_tree_model_iter_next (GTK_TREE_MODEL (store), &iter));
265
266 return FALSE;
267}

269static void
270clamp_list_store (GtkListStore *store,
271 guint max)
272{
273 GtkTreePath *path;
274 GtkTreeIter iter;
275
276 /* -1 because TreePath counts from 0 */
277 path = gtk_tree_path_new_from_indices (max - 1, -1);
278
279 if (gtk_tree_model_get_iter (GTK_TREE_MODEL (store), &iter, path))
280 {
281 while (1)
282 {
283 if (!gtk_list_store_remove (store, &iter))
284 break;
285 }
286 }
287
288 gtk_tree_path_free (path);
289}

291static void
292insert_history_item (PlumaHistoryEntry *entry,
293 const gchar *text,
294 gboolean prepend)
295{
296 GtkListStore *store;
297 GtkTreeIter iter;
298
299 if (g_utf8_strlen (text, -1) <= MIN_ITEM_LEN)
300 return;
301
302 store = get_history_store (entry);
303
304 /* remove the text from the store if it was already
305 * present. If it wasn't, clamp to max history - 1
306 * before inserting the new row, otherwise appending
307 * would not work */
308
309 if (!remove_item (store, text))
310 clamp_list_store (store,
311 entry->priv->history_length - 1);
312
313 if (prepend)
314 gtk_list_store_insert (store, &iter, 0);
315 else
316 gtk_list_store_append (store, &iter);
317
318 gtk_list_store_set (store,
319 &iter,
320 0,
321 text,
322 -1);
323
324 pluma_history_entry_save_history (entry);
325}

347static void
348pluma_history_entry_load_history (PlumaHistoryEntry *entry)
349{
350 GSList *settings_items, *l;
351 GtkListStore *store;
352 GtkTreeIter iter;
353 guint i;
354
355 g_return_if_fail (PLUMA_IS_HISTORY_ENTRY (entry));
356
357 store = get_history_store (entry);
358
359 settings_items = pluma_settings_get_list (entry->priv->settings,
360 entry->priv->history_id);
361
362 gtk_list_store_clear (store);
363
364 for (l = settings_items, i = 0;
365 l != NULL && i < entry->priv->history_length;
366 l = l->next, i++)
367 {
368 gtk_list_store_append (store, &iter);
369 gtk_list_store_set (store,
370 &iter,
371 0,
372 l->data,
373 -1);
374 }
375
376 g_slist_free_full (settings_items, g_free);
377}

436void
437pluma_history_entry_set_enable_completion (PlumaHistoryEntry *entry,
438 gboolean enable)
439{
440 g_return_if_fail (PLUMA_IS_HISTORY_ENTRY (entry));
441
442 if (enable)
443 {
444 if (entry->priv->completion != NULL)
445 return;
446
447 entry->priv->completion = gtk_entry_completion_new ();
448 gtk_entry_completion_set_model (entry->priv->completion,
449 GTK_TREE_MODEL (get_history_store (entry)));
450
451 /* Use model column 0 as the text column */
452 gtk_entry_completion_set_text_column (entry->priv->completion, 0);
453
454 gtk_entry_completion_set_minimum_key_length (entry->priv->completion,
455 MIN_ITEM_LEN);
456
457 gtk_entry_completion_set_popup_completion (entry->priv->completion, FALSE);
458 gtk_entry_completion_set_inline_completion (entry->priv->completion, TRUE);
459
460 /* Assign the completion to the entry */
461 gtk_entry_set_completion (GTK_ENTRY (pluma_history_entry_get_entry(entry)),
462 entry->priv->completion);
463 }
464 else
465 {
466 if (entry->priv->completion == NULL)
467 return;
468
469 gtk_entry_set_completion (GTK_ENTRY (pluma_history_entry_get_entry (entry)),
470 NULL);
471
472 g_object_unref (entry->priv->completion);
473
474 entry->priv->completion = NULL;
475 }
476}

486GtkWidget *
487pluma_history_entry_new (const gchar *history_id,
488 gboolean enable_completion)
489{
490 GtkWidget *ret;
491 GtkListStore *store;
492
493 g_return_val_if_fail (history_id != NULL, NULL);
494
495 /* Note that we are setting the model, so
496 * user must be careful to always manipulate
497 * data in the history through pluma_history_entry_
498 * functions.
499 */
500
501 store = gtk_list_store_new (1, G_TYPE_STRING);
502
503 ret = g_object_new (PLUMA_TYPE_HISTORY_ENTRY,
504 "history-id", history_id,
505 "model", store,
506 "has-entry", TRUE,
507 "entry-text-column", 0,
508 NULL);
509
510 g_object_unref (store);
511
512 /* loading has to happen after the model
513 * has been set. However the model is not a
514 * G_PARAM_CONSTRUCT property of GtkComboBox
515 * so we cannot do this in the constructor.
516 * For now we simply do here since this widget is
517 * not bound to other programming languages.
518 * A maybe better alternative is to override the
519 * model property of combobox and mark CONTRUCT_ONLY.
520 * This would also ensure that the model cannot be
521 * set explicitely at a later time.
522 */
523 pluma_history_entry_load_history (PLUMA_HISTORY_ENTRY (ret));
524
525 pluma_history_entry_set_enable_completion (PLUMA_HISTORY_ENTRY (ret),
526 enable_completion);
527
528 return ret;
529}
