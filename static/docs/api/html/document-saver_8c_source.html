102static const gchar *
103read_file (const gchar *uri)
104{
105 GFile *file = g_file_new_for_commandline_arg (uri);
106 GError *error = NULL;
107 static gchar buffer[4096];
108 gsize read;
109
110 GInputStream *stream = G_INPUT_STREAM (g_file_read (file, NULL, &error));
111
112 g_assert_no_error (error);
113
114 g_input_stream_read_all (stream, buffer, sizeof (buffer) - 1, &read, NULL, &error);
115 g_assert_no_error (error);
116
117 buffer[read] = '\0';
118
119 g_input_stream_close (stream, NULL, NULL);
120
121 g_object_unref (stream);
122 g_object_unref (file);
123
124 return buffer;
125}

140static void
141mount_ready_callback (GObject *object,
142 GAsyncResult *result,
143 gpointer data)
144{
145 GError *error = NULL;
146 mount_success = g_file_mount_enclosing_volume_finish (G_FILE (object),
147 result,
148 &error);
149
150 if (error && error->code == G_IO_ERROR_ALREADY_MOUNTED)
151 {
152 mount_success = TRUE;
153 g_error_free (error);
154 }
155 else
156 {
157 g_assert_no_error (error);
158 }
159
160 mount_completed = TRUE;
161}

163static gboolean
164ensure_mounted (GFile *file)
165{
166 GMountOperation *mo;
167
168 mount_success = FALSE;
169 mount_completed = FALSE;
170
171 if (g_file_is_native (file))
172 {
173 return TRUE;
174 }
175
176 mo = gtk_mount_operation_new (NULL);
177
178 g_file_mount_enclosing_volume (file,
179 G_MOUNT_MOUNT_NONE,
180 mo,
181 NULL,
182 mount_ready_callback,
183 NULL);
184
185 while (!mount_completed)
186 {
187 g_main_context_iteration (NULL, TRUE);
188 }
189
190 g_object_unref (mo);
191
192 return mount_success;
193}

195static void
196test_saver (const gchar *filename_or_uri,
197 const gchar *contents,
198 PlumaDocumentNewlineType newline_type,
199 PlumaDocumentSaveFlags save_flags,
200 GCallback saved_callback,
201 SaverTestData *data)
202{
203 GFile *file;
204 gchar *uri;
205 PlumaDocument *document;
206 gboolean existed;
207
208 document = create_document (contents);
209 pluma_document_set_newline_type (document, newline_type);
210
211 g_signal_connect (document, "saved", G_CALLBACK (complete_test_error), data);
212
213 if (saved_callback)
214 {
215 g_signal_connect (document, "saved", saved_callback, data);
216 }
217
218 g_signal_connect_after (document, "saved", G_CALLBACK (complete_test), data);
219
220 test_completed = FALSE;
221
222 file = g_file_new_for_commandline_arg (filename_or_uri);
223 uri = g_file_get_uri (file);
224 existed = g_file_query_exists (file, NULL);
225
226 ensure_mounted (file);
227
228 pluma_document_save_as (document, uri, pluma_encoding_get_utf8 (), save_flags);
229
230 while (!test_completed)
231 {
232 g_main_context_iteration (NULL, TRUE);
233 }
234
235 if (!existed)
236 {
237 g_file_delete (file, NULL, NULL);
238 }
239
240 g_free (uri);
241 g_object_unref (file);
242
243 saver_test_data_free (data);
244}

401static void
402test_permissions (const gchar *uri,
403 guint permissions)
404{
405 GError *error = NULL;
406 GFile *file = g_file_new_for_commandline_arg (uri);
407 GFileOutputStream *stream;
408 GFileInfo *info;
409 guint mode;
410
411 g_file_delete (file, NULL, NULL);
412 stream = g_file_create (file, 0, NULL, &error);
413
414 g_assert_no_error (error);
415
416 g_output_stream_close (G_OUTPUT_STREAM (stream), NULL, NULL);
417 g_object_unref (stream);
418
419 info = g_file_query_info (file,
420 G_FILE_ATTRIBUTE_UNIX_MODE,
421 G_FILE_QUERY_INFO_NONE,
422 NULL,
423 &error);
424
425 g_assert_no_error (error);
426
427 mode = g_file_info_get_attribute_uint32 (info, G_FILE_ATTRIBUTE_UNIX_MODE);
428 g_object_unref (info);
429
430 g_file_set_attribute_uint32 (file,
431 G_FILE_ATTRIBUTE_UNIX_MODE,
432 (mode & ~ACCESSPERMS) | permissions,
433 G_FILE_QUERY_INFO_NONE,
434 NULL,
435 &error);
436 g_assert_no_error (error);
437
438 check_permissions (file, permissions);
439
440 test_saver (uri,
441 DEFAULT_CONTENT,
442 PLUMA_DOCUMENT_NEWLINE_TYPE_LF,
443 0,
444 G_CALLBACK (check_permissions_saved),
445 saver_test_data_new (uri,
446 DEFAULT_CONTENT_RESULT,
447 GINT_TO_POINTER ((gint)permissions)));
448
449 g_file_delete (file, NULL, NULL);
450 g_object_unref (file);
451}

552static gboolean
553check_unowned_directory ()
554{
555 GFile *unowned = g_file_new_for_path (UNOWNED_LOCAL_DIRECTORY);
556 GFile *unowned_file;
557 GFileInfo *info;
558 GError *error = NULL;
559
560 g_printf ("*** Checking for unowned directory test... ");
561
562 info = g_file_query_info (unowned,
563 G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE,
564 G_FILE_QUERY_INFO_NONE,
565 NULL,
566 &error);
567
568 if (error)
569 {
570 g_object_unref (unowned);
571 g_printf ("NO: directory does not exist\n");
572
573 g_error_free (error);
574 return FALSE;
575 }
576
577 if (g_file_info_get_attribute_boolean (info, G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE))
578 {
579 g_object_unref (unowned);
580
581 g_printf ("NO: directory is writable\n");
582 g_object_unref (info);
583 return FALSE;
584 }
585
586 g_object_unref (info);
587 g_object_unref (unowned);
588
589 unowned_file = g_file_new_for_commandline_arg (UNOWNED_LOCAL_URI);
590
591 info = g_file_query_info (unowned_file,
592 G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE,
593 G_FILE_QUERY_INFO_NONE,
594 NULL,
595 &error);
596
597 if (error)
598 {
599 g_object_unref (unowned_file);
600 g_error_free (error);
601
602 g_printf ("NO: file does not exist\n");
603 return FALSE;
604 }
605
606 if (!g_file_info_get_attribute_boolean (info, G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE))
607 {
608 g_object_unref (unowned_file);
609
610 g_printf ("NO: file is not writable\n");
611 g_object_unref (info);
612 return FALSE;
613 }
614
615 g_object_unref (info);
616 g_object_unref (unowned_file);
617
618 g_printf ("YES\n");
619 return TRUE;
620}

622static gboolean
623check_unowned_group ()
624{
625 GFile *unowned = g_file_new_for_path (UNOWNED_GROUP_LOCAL_URI);
626 GFileInfo *info;
627 GError *error = NULL;
628
629 g_printf ("*** Checking for unowned group test... ");
630
631 info = g_file_query_info (unowned,
632 G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE ","
633 G_FILE_ATTRIBUTE_OWNER_GROUP ","
634 G_FILE_ATTRIBUTE_UNIX_MODE,
635 G_FILE_QUERY_INFO_NONE,
636 NULL,
637 &error);
638
639 if (error)
640 {
641 g_object_unref (unowned);
642 g_printf ("NO: file does not exist\n");
643
644 g_error_free (error);
645 return FALSE;
646 }
647
648 if (!g_file_info_get_attribute_boolean (info, G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE))
649 {
650 g_object_unref (unowned);
651
652 g_printf ("NO: file is not writable\n");
653 g_object_unref (info);
654 return FALSE;
655 }
656
657 if (g_strcmp0 (g_file_info_get_attribute_string (info, G_FILE_ATTRIBUTE_OWNER_GROUP),
658 "root") != 0)
659 {
660 g_object_unref (unowned);
661
662 g_printf ("NO: group is not root (%s)\n", g_file_info_get_attribute_string (info, G_FILE_ATTRIBUTE_OWNER_GROUP));
663 g_object_unref (info);
664 return FALSE;
665 }
666
667 if ((g_file_info_get_attribute_uint32 (info, G_FILE_ATTRIBUTE_UNIX_MODE) & ACCESSPERMS) != 0660)
668 {
669 g_object_unref (unowned);
670
671 g_printf ("NO: file has wrong permissions\n");
672 g_object_unref (info);
673 return FALSE;
674 }
675
676 g_object_unref (info);
677 g_object_unref (unowned);
678
679 g_printf ("YES\n");
680 return TRUE;
681}
