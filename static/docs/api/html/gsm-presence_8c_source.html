109static void
110set_session_idle (GsmPresence *presence,
111 gboolean is_idle)
112{
113 GsmPresencePrivate *priv;
114
115 g_debug ("GsmPresence: setting idle: %d", is_idle);
116 priv = gsm_presence_get_instance_private (presence);
117
118 if (is_idle) {
119 if (priv->status == GSM_PRESENCE_STATUS_IDLE) {
120 g_debug ("GsmPresence: already idle, ignoring");
121 return;
122 }
123
124 /* save current status */
125 priv->saved_status = priv->status;
126 gsm_presence_set_status (presence, GSM_PRESENCE_STATUS_IDLE, NULL);
127 } else {
128 if (priv->status != GSM_PRESENCE_STATUS_IDLE) {
129 g_debug ("GsmPresence: already not idle, ignoring");
130 return;
131 }
132
133 /* restore saved status */
134 gsm_presence_set_status (presence, priv->saved_status, NULL);
135 priv->saved_status = GSM_PRESENCE_STATUS_AVAILABLE;
136 }
137}

153static void
154reset_idle_watch (GsmPresence *presence)
155{
156 GsmPresencePrivate *priv;
157
158 priv = gsm_presence_get_instance_private (presence);
159 if (priv->idle_monitor == NULL) {
160 return;
161 }
162
163 if (priv->idle_watch_id > 0) {
164 g_debug ("GsmPresence: removing idle watch");
165 gs_idle_monitor_remove_watch (priv->idle_monitor,
166 priv->idle_watch_id);
167 priv->idle_watch_id = 0;
168 }
169
170 if (! priv->screensaver_active
171 && priv->idle_enabled
172 && priv->idle_timeout > 0) {
173 g_debug ("GsmPresence: adding idle watch");
174
175 priv->idle_watch_id = gs_idle_monitor_add_watch (priv->idle_monitor,
176 priv->idle_timeout,
177 (GSIdleMonitorWatchFunc)on_idle_timeout,
178 presence);
179 }
180}

182static void
183on_screensaver_active_changed (DBusGProxy *proxy,
184 gboolean is_active,
185 GsmPresence *presence)
186{
187 GsmPresencePrivate *priv;
188
189 g_debug ("screensaver status changed: %d", is_active);
190 priv = gsm_presence_get_instance_private (presence);
191 if (priv->screensaver_active != is_active) {
192 priv->screensaver_active = is_active;
193 reset_idle_watch (presence);
194 set_session_idle (presence, is_active);
195 }

213static void
214on_bus_name_owner_changed (DBusGProxy *bus_proxy,
215 const char *service_name,
216 const char *old_service_name,
217 const char *new_service_name,
218 GsmPresence *presence)
219{
220 GError *error;
221 GsmPresencePrivate *priv;
222
223 priv = gsm_presence_get_instance_private (presence);
224
225 if (service_name == NULL
226 || strcmp (service_name, GS_NAME) != 0) {
227 /* ignore */
228 return;
229 }
230
231 if (strlen (new_service_name) == 0
232 && strlen (old_service_name) > 0) {
233 /* service removed */
234 /* let destroy signal handle this? */
235 } else if (strlen (old_service_name) == 0
236 && strlen (new_service_name) > 0) {
237 /* service added */
238 error = NULL;
239 priv->screensaver_proxy = dbus_g_proxy_new_for_name_owner (priv->bus_connection,
240 GS_NAME,
241 GS_PATH,
242 GS_INTERFACE,
243 &error);
244 if (priv->screensaver_proxy != NULL) {
245 g_signal_connect (priv->screensaver_proxy,
246 "destroy",
247 G_CALLBACK (on_screensaver_proxy_destroy),
248 presence);
249 dbus_g_proxy_add_signal (priv->screensaver_proxy,
250 "ActiveChanged",
251 G_TYPE_BOOLEAN,
252 G_TYPE_INVALID);
253 dbus_g_proxy_connect_signal (priv->screensaver_proxy,
254 "ActiveChanged",
255 G_CALLBACK (on_screensaver_active_changed),
256 presence,
257 NULL);
258 } else {
259 g_warning ("Unable to get screensaver proxy: %s", error->message);
260 g_error_free (error);
261 }
262 }
263}

265static gboolean
266register_presence (GsmPresence *presence)
267{
268 GError *error;
269 GsmPresencePrivate *priv;
270
271 error = NULL;
272
273 priv = gsm_presence_get_instance_private (presence);
274 priv->bus_connection = dbus_g_bus_get (DBUS_BUS_SESSION, &error);
275 if (priv->bus_connection == NULL) {
276 if (error != NULL) {
277 g_critical ("error getting session bus: %s", error->message);
278 g_error_free (error);
279 }
280 return FALSE;
281 }
282
283 dbus_g_connection_register_g_object (priv->bus_connection, GSM_PRESENCE_DBUS_PATH, G_OBJECT (presence));
284
285 return TRUE;
286}

355gboolean
356gsm_presence_set_status_text (GsmPresence *presence,
357 const char *status_text,
358 GError **error)
359{
360 GsmPresencePrivate *priv;
361 g_return_val_if_fail (GSM_IS_PRESENCE (presence), FALSE);
362
363 priv = gsm_presence_get_instance_private (presence);
364
365 g_free (priv->status_text);
366
367 /* check length */
368 if (status_text != NULL && strlen (status_text) > MAX_STATUS_TEXT) {
369 g_set_error (error,
370 GSM_PRESENCE_ERROR,
371 GSM_PRESENCE_ERROR_GENERAL,
372 "Status text too long");
373 return FALSE;
374 }
375
376 if (status_text != NULL) {
377 priv->status_text = g_strdup (status_text);
378 } else {
379 priv->status_text = g_strdup ("");
380 }
381 g_object_notify (G_OBJECT (presence), "status-text");
382 g_signal_emit (presence, signals[STATUS_TEXT_CHANGED], 0, priv->status_text);
383 return TRUE;
384}

481static void
482gsm_presence_finalize (GObject *object)
483{
484 GsmPresence *presence = (GsmPresence *) object;
485 GsmPresencePrivate *priv;
486
487 priv = gsm_presence_get_instance_private (presence);
488
489 if (priv->idle_watch_id > 0) {
490 gs_idle_monitor_remove_watch (priv->idle_monitor,
491 priv->idle_watch_id);
492 priv->idle_watch_id = 0;
493 }
494
495 if (priv->status_text != NULL) {
496 g_free (priv->status_text);
497 priv->status_text = NULL;
498 }
499
500 if (priv->idle_monitor != NULL) {
501 g_object_unref (priv->idle_monitor);
502 priv->idle_monitor = NULL;
503 }
504
505 G_OBJECT_CLASS (gsm_presence_parent_class)->finalize (object);
506}
