406static int
407application_compare_by_id (const GAppInfo *app_a,
408 const GAppInfo *app_b)
409{
410 const char *id_a, *id_b;
411
412 id_a = g_app_info_get_id ((GAppInfo *)app_a);
413 id_b = g_app_info_get_id ((GAppInfo *)app_b);
414
415 if (id_a == NULL && id_b == NULL)
416 {
417 if (g_app_info_equal ((GAppInfo *)app_a, (GAppInfo *)app_b))
418 {
419 return 0;
420 }
421 if ((gsize)app_a < (gsize) app_b)
422 {
423 return -1;
424 }
425 return 1;
426 }
427
428 if (id_a == NULL)
429 {
430 return -1;
431 }
432
433 if (id_b == NULL)
434 {
435 return 1;
436 }
437
438 return strcmp (id_a, id_b);
439}

443{
444 char *mime_type;
445 char *uri_scheme;
446 GList *result;
447
448 if (!caja_mime_actions_check_if_required_attributes_ready (file))

453 result = g_app_info_get_all_for_type (mime_type);

465 g_free (uri_scheme);

467

469 {
470 /* Filter out non-uri supporting apps */
471 result = filter_non_uri_apps (result);
472 }
473
474 result = g_list_sort (result, (GCompareFunc) application_compare_by_name);
475 g_free (mime_type);
476
477 return filter_caja_handler (result);

527GAppInfo *
528caja_mime_get_default_application_for_files (GList *files)
529{
530 GList *l, *sorted_files;
531 GAppInfo *app, *one_app;
532 CajaFile *file = NULL;
533
534 g_assert (files != NULL);
535
536 sorted_files = g_list_sort (g_list_copy (files), (GCompareFunc) file_compare_by_mime_type);
537
538 app = NULL;
539 for (l = sorted_files; l != NULL; l = l->next)
540 {
541 file = l->data;
542
543 if (l->prev &&
544 file_compare_by_mime_type (file, l->prev->data) == 0 &&
545 file_compare_by_parent_uri (file, l->prev->data) == 0)
546 {
547 continue;
548 }
549
550 one_app = caja_mime_get_default_application_for_file (file);
551 if (one_app == NULL || (app != NULL && !g_app_info_equal (app, one_app)))
552 {
553 if (app)
554 {
555 g_object_unref (app);
556 }
557 if (one_app)
558 {
559 g_object_unref (one_app);
560 }
561 app = NULL;
562 break;
563 }
564
565 if (app == NULL)
566 {
567 app = one_app;
568 }
569 else
570 {
571 g_object_unref (one_app);
572 }
573 }
574
575 g_list_free (sorted_files);
576
577 return app;
578}

580/* returns an intersection of two mime application lists,
581 * and returns a new list, freeing a, b and all applications
582 * that are not in the intersection set.
583 * The lists are assumed to be pre-sorted by their IDs */
584static GList *
585intersect_application_lists (GList *a,
586 GList *b)
587{
588 GList *l, *m;
589 GList *ret;
590 GAppInfo *a_app = NULL;
591 GAppInfo *b_app = NULL;
592
593 ret = NULL;
594
595 l = a;
596 m = b;
597
598 while (l != NULL && m != NULL)
599 {
600 int cmp;
601
602 a_app = (GAppInfo *) l->data;
603 b_app = (GAppInfo *) m->data;
604
605 cmp = application_compare_by_id (a_app, b_app);
606 if (cmp > 0)
607 {
608 g_object_unref (b_app);
609 m = m->next;
610 }
611 else if (cmp < 0)
612 {
613 g_object_unref (a_app);
614 l = l->next;
615 }
616 else
617 {
618 g_object_unref (b_app);
619 ret = g_list_prepend (ret, a_app);
620 l = l->next;
621 m = m->next;
622 }
623 }
624
625 g_list_foreach (l, (GFunc) g_object_unref, NULL);
626 g_list_foreach (m, (GFunc) g_object_unref, NULL);
627
628 g_list_free (a);
629 g_list_free (b);
630
631 return g_list_reverse (ret);
632}

634GList *
635caja_mime_get_applications_for_files (GList *files)
636{
637 GList *l, *sorted_files;
638 GList *one_ret, *ret;
639 CajaFile *file = NULL;
640
641 g_assert (files != NULL);
642
643 sorted_files = g_list_sort (g_list_copy (files), (GCompareFunc) file_compare_by_mime_type);
644
645 ret = NULL;
646 for (l = sorted_files; l != NULL; l = l->next)
647 {
648 file = l->data;
649
650 if (l->prev &&
651 file_compare_by_mime_type (file, l->prev->data) == 0 &&
652 file_compare_by_parent_uri (file, l->prev->data) == 0)
653 {
654 continue;
655 }
656
657 one_ret = caja_mime_get_applications_for_file (file);
658 one_ret = g_list_sort (one_ret, (GCompareFunc) application_compare_by_id);
659 if (ret != NULL)

661 ret = intersect_application_lists (ret, one_ret);

665 ret = one_ret;
666 }
667

669 {
670 break;
671 }
672 }
673
674 g_list_free (sorted_files);
675
676 ret = g_list_sort (ret, (GCompareFunc) application_compare_by_name);
677
678 return ret;

707 char *target_path;
708 char *display_name;

710 char *detail;
711 GtkDialog *dialog;

714
715 g_assert (caja_file_is_broken_symbolic_link (file));

729 if (target_path == NULL)
730 {
731 detail = g_strdup (_("This link cannot be used, because it has no target."));
732 }
733 else
734 {
735 detail = g_strdup_printf (_("This link cannot be used, because its target "
736 "\"%s\" doesn't exist."), target_path);
737 }
738
739 if (caja_file_is_in_trash (file))
740 {
741 eel_run_simple_dialog (GTK_WIDGET (parent_window), FALSE, GTK_MESSAGE_WARNING,
742 prompt, detail, "process-stop", NULL);
743 goto out;
744 }
745
746 dialog = eel_show_yes_no_dialog (prompt, detail, _("Mo_ve to Trash"), "process-stop",
747 parent_window);
748
749 gtk_dialog_set_default_response (dialog, GTK_RESPONSE_CANCEL);
750
751 /* Make this modal to avoid problems with reffing the view & file
752 * to keep them around in case the view changes, which would then
753 * cause the old view not to be destroyed, which would cause its
754 * merged MateComponent items not to be un-merged. Maybe we need to unmerge
755 * explicitly when disconnecting views instead of relying on the
756 * unmerge in Destroy. But since MateComponentUIHandler is probably going
757 * to change wildly, I don't want to mess with this now.
758 */
759
760 response = gtk_dialog_run (dialog);
761 gtk_widget_destroy (GTK_WIDGET (dialog));
762
763 if (response == GTK_RESPONSE_YES)
764 {
765 file_as_list.data = file;
766 file_as_list.next = NULL;
767 file_as_list.prev = NULL;
768 trash_or_delete_files (parent_window, &file_as_list, TRUE);
769 }
770
771out:
772 g_free (prompt);
773 g_free (target_path);
774 g_free (detail);
775}

975static GList *
976make_activation_parameters (GList *uris,
977 GList **unhandled_uris)
978{
979 GList *ret, *l, *app_uris;
980 GHashTable *app_table;
981 GAppInfo *old_app;
982 GAppInfo *app = NULL;
983 CajaFile *file = NULL;
984
985 ret = NULL;
986 *unhandled_uris = NULL;
987
988 app_table = g_hash_table_new_full
989 ((GHashFunc) mime_application_hash,
990 (GEqualFunc) g_app_info_equal,
991 (GDestroyNotify) g_object_unref,
992 (GDestroyNotify) g_list_free);
993
994 for (l = uris; l != NULL; l = l->next)
995 {
996 char *uri;
997
998 uri = l->data;
999 file = caja_file_get_by_uri (uri);
1000
1001 /* Double-check if a file's MIME type has changed before we commit to a
1002 choice of application for it. This can happen if, for instance, file
1003 was originally created with 0 bytes and then content was added to it
1004 later-- it will change from plaintext to something else. */
1005 caja_file_refresh_info (file);
1006
1007 app = caja_mime_get_default_application_for_file (file);
1008 if (app != NULL)
1009 {
1010 app_uris = NULL;
1011
1012 if (g_hash_table_lookup_extended (app_table, app,
1013 (gpointer *) &old_app,
1014 (gpointer *) &app_uris))
1015 {
1016 g_hash_table_steal (app_table, old_app);
1017
1018 app_uris = g_list_prepend (app_uris, uri);
1019
1020 g_object_unref (app);
1021 app = old_app;
1022 }
1023 else
1024 {
1025 app_uris = g_list_prepend (NULL, uri);
1026 }
1027
1028 g_hash_table_insert (app_table, app, app_uris);
1029 }
1030 else
1031 {
1032 *unhandled_uris = g_list_prepend (*unhandled_uris, uri);
1033 }
1034 caja_file_unref (file);
1035 }
1036
1037 g_hash_table_foreach (app_table,
1038 (GHFunc) list_to_parameters_foreach,
1039 &ret);
1040
1041 g_hash_table_destroy (app_table);
1042
1043 *unhandled_uris = g_list_reverse (*unhandled_uris);
1044
1045 return g_list_reverse (ret);
1046}
