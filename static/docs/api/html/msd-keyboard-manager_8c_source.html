99static void
100numlock_xkb_init (MsdKeyboardManager *manager)
101{
102 Display *dpy = GDK_DISPLAY_XDISPLAY (gdk_display_get_default ());
103 gboolean have_xkb;
104 int opcode, error_base, major, minor;
105
106 have_xkb = XkbQueryExtension (dpy,
107 &opcode,
108 &manager->priv->xkb_event_base,
109 &error_base,
110 &major,
111 &minor)
112 && XkbUseExtension (dpy, &major, &minor);
113
114 if (have_xkb) {
115 XkbSelectEventDetails (dpy,
116 XkbUseCoreKbd,
117 XkbStateNotifyMask,
118 XkbModifierLockMask,
119 XkbModifierLockMask);
120 } else {
121 g_warning ("XKB extension not available");
122 }
123
124 manager->priv->have_xkb = have_xkb;
125}

127static unsigned
128numlock_NumLock_modifier_mask (void)
129{
130 Display *dpy = GDK_DISPLAY_XDISPLAY (gdk_display_get_default ());
131 return XkbKeysymToModifiers (dpy, XK_Num_Lock);

134static void
135numlock_set_xkb_state (NumLockState new_state)
136{
137 unsigned int num_mask;
138 Display *dpy = GDK_DISPLAY_XDISPLAY (gdk_display_get_default ());
139 if (new_state != NUMLOCK_STATE_ON && new_state != NUMLOCK_STATE_OFF)
140 return;
141 num_mask = numlock_NumLock_modifier_mask ();
142 XkbLockModifiers (dpy, XkbUseCoreKbd, num_mask, new_state ? num_mask : 0);
143}

153static void numlock_set_settings_state(GSettings *settings, NumLockState new_state)
154{
155 if (new_state != NUMLOCK_STATE_ON && new_state != NUMLOCK_STATE_OFF) {

158 g_settings_set_enum (settings, KEY_NUMLOCK_STATE, new_state);

161static GdkFilterReturn
162numlock_xkb_callback (GdkXEvent *xev_,
163 GdkEvent *gdkev_,
164 gpointer xkb_event_code)
165{
166 XEvent *xev = (XEvent *) xev_;
167
168 if (xev->type == GPOINTER_TO_INT (xkb_event_code)) {
169 XkbEvent *xkbev = (XkbEvent *)xev;
170 if (xkbev->any.xkb_type == XkbStateNotify)
171 if (xkbev->state.changed & XkbModifierLockMask) {
172 unsigned num_mask = numlock_NumLock_modifier_mask ();
173 unsigned locked_mods = xkbev->state.locked_mods;
174 int numlock_state = !! (num_mask & locked_mods);
175 GSettings *settings = g_settings_new (MSD_KEYBOARD_SCHEMA);
176 numlock_set_settings_state (settings, numlock_state);
177 g_object_unref (settings);
178 }
179 }
180 return GDK_FILTER_CONTINUE;
181}

196static void
197apply_settings (GSettings *settings,
198 gchar *key,
199 MsdKeyboardManager *manager)
200{
201 XKeyboardControl kbdcontrol;
202 gboolean repeat;
203 gboolean click;
204 int click_volume;
205 int bell_volume;
206 int bell_pitch;
207 int bell_duration;
208 char *volume_string;
209 GdkDisplay *display;
210#ifdef HAVE_X11_EXTENSIONS_XKB_H
211 gboolean rnumlock;
212#endif /* HAVE_X11_EXTENSIONS_XKB_H */
213
214 repeat = g_settings_get_boolean (settings, KEY_REPEAT);
215 click = g_settings_get_boolean (settings, KEY_CLICK);
216 click_volume = g_settings_get_int (settings, KEY_CLICK_VOLUME);
217 bell_pitch = g_settings_get_int (settings, KEY_BELL_PITCH);
218 bell_duration = g_settings_get_int (settings, KEY_BELL_DURATION);
219
220 volume_string = g_settings_get_string (settings, KEY_BELL_MODE);
221 bell_volume = (volume_string && !strcmp (volume_string, "on")) ? 50 : 0;
222 g_free (volume_string);
223
224 display = gdk_display_get_default ();
225 gdk_x11_display_error_trap_push (display);
226 if (repeat) {
227 gboolean rate_set = FALSE;
228
229 XAutoRepeatOn (GDK_DISPLAY_XDISPLAY (display));
230 /* Use XKB in preference */
231#ifdef HAVE_X11_EXTENSIONS_XKB_H
232 rate_set = xkb_set_keyboard_autorepeat_rate (g_settings_get_int (settings, KEY_DELAY),
233 g_settings_get_int (settings, KEY_RATE));
234#endif /* HAVE_X11_EXTENSIONS_XKB_H */
235 if (!rate_set)
236 g_warning ("Neither XKeyboard not Xfree86's keyboard extensions are available,\n"
237 "no way to support keyboard autorepeat rate settings");
238 } else {
239 XAutoRepeatOff (GDK_DISPLAY_XDISPLAY (display));
240 }
241
242 /* as percentage from 0..100 inclusive */
243 if (click_volume < 0) {
244 click_volume = 0;
245 } else if (click_volume > 100) {
246 click_volume = 100;
247 }
248 kbdcontrol.key_click_percent = click ? click_volume : 0;
249 kbdcontrol.bell_percent = bell_volume;
250 kbdcontrol.bell_pitch = bell_pitch;
251 kbdcontrol.bell_duration = bell_duration;
252 XChangeKeyboardControl (GDK_DISPLAY_XDISPLAY(gdk_display_get_default()),
253 KBKeyClickPercent | KBBellPercent | KBBellPitch | KBBellDuration,
254 &kbdcontrol);
255
256#ifdef HAVE_X11_EXTENSIONS_XKB_H
257 rnumlock = g_settings_get_boolean (settings, KEY_NUMLOCK_REMEMBER);
258
259 if (rnumlock == 0 || key == NULL) {
260 if (manager->priv->have_xkb && rnumlock) {
261 numlock_set_xkb_state (numlock_get_settings_state (settings));
262 }
263 }
264#endif /* HAVE_X11_EXTENSIONS_XKB_H */
265
266 XSync (GDK_DISPLAY_XDISPLAY (display), FALSE);
267 gdk_x11_display_error_trap_pop_ignored (display);
268}

276static gboolean
277start_keyboard_idle_cb (MsdKeyboardManager *manager)
278{
279 mate_settings_profile_start (NULL);
280
281 g_debug ("Starting keyboard manager");
282
283 manager->priv->have_xkb = 0;
284 manager->priv->settings = g_settings_new (MSD_KEYBOARD_SCHEMA);
285
286 /* Essential - xkb initialization should happen before */
287 msd_keyboard_xkb_init (manager);
288
289#ifdef HAVE_X11_EXTENSIONS_XKB_H
290 numlock_xkb_init (manager);
291#endif /* HAVE_X11_EXTENSIONS_XKB_H */
292
293 /* apply current settings before we install the callback */
294 msd_keyboard_manager_apply_settings (manager);
295
296 g_signal_connect (manager->priv->settings, "changed", G_CALLBACK (apply_settings), manager);
297
298#ifdef HAVE_X11_EXTENSIONS_XKB_H
299 numlock_install_xkb_callback (manager);
300#endif /* HAVE_X11_EXTENSIONS_XKB_H */
301
302 mate_settings_profile_end (NULL);
303
304 return FALSE;
305}

320void
321msd_keyboard_manager_stop (MsdKeyboardManager *manager)
322{
323 MsdKeyboardManagerPrivate *p = manager->priv;
324
325 g_debug ("Stopping keyboard manager");
326
327 if (p->settings != NULL) {
328 g_object_unref (p->settings);
329 p->settings = NULL;
330 }
331
332#if HAVE_X11_EXTENSIONS_XKB_H
333 if (p->have_xkb) {
334 gdk_window_remove_filter (NULL,
335 numlock_xkb_callback,
336 GINT_TO_POINTER (p->xkb_event_base));
337 }
338#endif /* HAVE_X11_EXTENSIONS_XKB_H */
339
340 msd_keyboard_xkb_shutdown ();
341}
