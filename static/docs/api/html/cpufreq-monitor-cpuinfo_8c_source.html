62static gboolean
63cpufreq_monitor_cpuinfo_run (CPUFreqMonitor *monitor)
64{
65 gchar *file;
66 gchar **lines;
67 gchar *buffer = NULL;
68 gchar *p;
69 gint cpu, i;
70 gint cur_freq, max_freq;
71 gchar *governor;
72 GError *error = NULL;
73
74 file = g_strdup ("/proc/cpuinfo");
75 if (!cpufreq_file_get_contents (file, &buffer, NULL, &error)) {
76 g_warning ("%s", error->message);
77 g_error_free (error);
78
79 g_free (file);
80
81 return FALSE;
82 }
83 g_free (file);
84
85 /* TODO: SMP support */
86 lines = g_strsplit (buffer, "\n", -1);
87 for (i = 0; lines[i]; i++) {
88 if (g_ascii_strncasecmp ("cpu MHz", lines[i], strlen ("cpu MHz")) == 0) {
89 p = g_strrstr (lines[i], ":");
90
91 if (p == NULL) {
92 g_strfreev (lines);
93 g_free (buffer);
94
95 return FALSE;
96 }
97
98 if (strlen (lines[i]) < (size_t)(p - lines[i])) {
99 g_strfreev (lines);
100 g_free (buffer);
101
102 return FALSE;
103 }
104
105 if ((sscanf (p + 1, "%d.", &cpu)) != 1) {
106 g_strfreev (lines);
107 g_free (buffer);
108
109 return FALSE;
110 }
111
112 break;
113 }
114 }
115
116 g_strfreev (lines);
117 g_free (buffer);
118
119 governor = g_strdup (_("Frequency Scaling Unsupported"));
120 cur_freq = cpu * 1000;
121 max_freq = cur_freq;
122
123 g_object_set (G_OBJECT (monitor),
124 "governor", governor,
125 "frequency", cur_freq,
126 "max-frequency", max_freq,
127 NULL);
128
129 g_free (governor);
130
131 return TRUE;
132}
