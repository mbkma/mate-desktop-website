46static void
47append_directory_path (MateMenuTreeDirectory *directory,
48 GString *path)
49{
50 MateMenuTreeDirectory *parent;
51
52 parent = matemenu_tree_directory_get_parent(directory);
53
54 if (!parent)
55 {
56 g_string_append_c(path, '/');
57 return;
58 }
59
60 append_directory_path(parent, path);
61
62 g_string_append(path, matemenu_tree_directory_get_name (directory));
63 g_string_append_c(path, '/');
64
65 matemenu_tree_item_unref(parent);
66}

82static void
83print_entry (MateMenuTreeEntry *entry,
84 const char *path)
85{
86 char *utf8_path;
87 char *utf8_file_id;
88
89 utf8_path = g_filename_to_utf8(matemenu_tree_entry_get_desktop_file_path (entry),
90 -1, NULL, NULL, NULL);
91
92 utf8_file_id = g_filename_to_utf8(matemenu_tree_entry_get_desktop_file_id (entry),
93 -1, NULL, NULL, NULL);
94
95 g_print("%s\t%s\t%s%s\n",
96 path,
97 utf8_file_id ? utf8_file_id : _("Invalid desktop file ID"),
98 utf8_path ? utf8_path : _("[Invalid Filename]"),
99 matemenu_tree_entry_get_is_excluded(entry) ? _(" <excluded>") : "");
100
101 g_free(utf8_file_id);
102 g_free(utf8_path);
103}

105static void
106print_directory(MateMenuTreeDirectory *directory)
107{
108 MateMenuTreeIter *iter;
109 const char *path;
110 char *freeme;
111
112 freeme = make_path(directory);
113 if (!strcmp (freeme, "/"))
114 path = freeme;
115 else
116 path = freeme + 1;
117
118 iter = matemenu_tree_directory_iter(directory);
119
120 while(TRUE)
121 {
122 gpointer item;
123
124 switch (matemenu_tree_iter_next (iter))
125 {
126 case MATEMENU_TREE_ITEM_INVALID:
127 goto done;
128
129 case MATEMENU_TREE_ITEM_ENTRY:
130 item = matemenu_tree_iter_get_entry(iter);
131 print_entry((MateMenuTreeEntry*)item, path);
132 break;
133
134 case MATEMENU_TREE_ITEM_DIRECTORY:
135 item = matemenu_tree_iter_get_directory(iter);
136 print_directory((MateMenuTreeDirectory*)item);
137 break;
138
139 case MATEMENU_TREE_ITEM_HEADER:
140 case MATEMENU_TREE_ITEM_SEPARATOR:
141 item = NULL;
142 break;
143
144 case MATEMENU_TREE_ITEM_ALIAS:
145 {
146 item = matemenu_tree_iter_get_alias(iter);
147
148 if (matemenu_tree_alias_get_aliased_item_type (item) == MATEMENU_TREE_ITEM_ENTRY)
149 {
150 MateMenuTreeEntry *entry = matemenu_tree_alias_get_aliased_entry(item);
151 print_entry(entry, path);
152 matemenu_tree_item_unref(entry);
153 }
154 }
155 break;
156
157 default:
158 g_assert_not_reached();
159 break;
160 }
161
162 matemenu_tree_item_unref(item);
163 continue;
164done:
165 break;
166 }
167
168 matemenu_tree_iter_unref(iter);
169
170 g_free(freeme);
171}

173static void
174handle_tree_changed (MateMenuTree *tree)
175{
176 MateMenuTreeDirectory *root;
177 GError *error = NULL;
178
179 g_print(_("\n\n\n==== Menu changed, reloading ====\n\n\n"));
180
181 if(!matemenu_tree_load_sync (tree, &error))
182 {
183 g_printerr("Failed to load tree: %s\n", error->message);
184 g_clear_error(&error);
185 return;
186 }
187
188 root = matemenu_tree_get_root_directory(tree);
189 if (root == NULL)
190 {
191 g_warning(_("Menu tree is empty"));
192 return;
193 }
194
195 print_directory(root);
196 matemenu_tree_item_unref(root);
197}

199int
200main (int argc, char **argv)
201{
202 GOptionContext *options_context;
203 MateMenuTree *tree;
204 MateMenuTreeDirectory *root;
205 MateMenuTreeFlags flags;
206 GError *error = NULL;
207
208#ifdef ENABLE_NLS
209 setlocale(LC_ALL, "");
210 bindtextdomain (GETTEXT_PACKAGE, MATELOCALEDIR);
211 bind_textdomain_codeset (GETTEXT_PACKAGE, "UTF-8");
212 textdomain (GETTEXT_PACKAGE);
213#endif /* ENABLE_NLS */
214
215 options_context = g_option_context_new(_("- test MATE's implementation of the Desktop Menu Specification"));
216 g_option_context_add_main_entries(options_context, options, GETTEXT_PACKAGE);
217 g_option_context_parse(options_context, &argc, &argv, NULL);
218 g_option_context_free(options_context);
219
220 flags = MATEMENU_TREE_FLAGS_NONE;
221 if (include_excluded)
222 flags |= MATEMENU_TREE_FLAGS_INCLUDE_EXCLUDED;
223 if (include_nodisplay)
224 flags |= MATEMENU_TREE_FLAGS_INCLUDE_NODISPLAY;
225 if (include_unallocated)
226 flags |= MATEMENU_TREE_FLAGS_INCLUDE_UNALLOCATED;
227
228 tree = matemenu_tree_new(menu_file ? menu_file : "mate-applications.menu", flags);
229 g_assert(tree != NULL);
230
231 if (!matemenu_tree_load_sync (tree, &error))
232 {
233 g_printerr("Failed to load tree: %s\n", error->message);
234 return 1;
235 }
236
237 g_print("Loaded menu from %s\n", matemenu_tree_get_canonical_menu_path(tree));
238
239 root = matemenu_tree_get_root_directory(tree);
240 if (root != NULL)
241 {
242 print_directory(root);
243 matemenu_tree_item_unref(root);
244 }
245 else
246 {
247 g_warning(_("Menu tree is empty"));
248 }
249
250 if (monitor)
251 {
252 GMainLoop *main_loop;
253
254 g_signal_connect(tree, "changed", G_CALLBACK(handle_tree_changed), NULL);
255
256 main_loop = g_main_loop_new(NULL, FALSE);
257 g_main_loop_run(main_loop);
258 g_main_loop_unref(main_loop);
259 }
260
261 g_object_unref(tree);
262
263 return 0;
264}
