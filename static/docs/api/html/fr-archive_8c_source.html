385static void
386fr_archive_set_uri (FrArchive *archive,
387 const char *uri)
388{
389 if ((archive->local_copy != NULL) && archive->is_remote) {
390 GFile *temp_folder;
391 GError *err = NULL;
392
393 g_file_delete (archive->local_copy, NULL, &err);
394 if (err != NULL) {
395 g_warning ("Failed to delete the local copy: %s", err->message);
396 g_clear_error (&err);
397 }
398
399 temp_folder = g_file_get_parent (archive->local_copy);
400 g_file_delete (temp_folder, NULL, &err);
401 if (err != NULL) {
402 g_warning ("Failed to delete temp folder: %s", err->message);
403 g_clear_error (&err);
404 }
405
406 g_object_unref (temp_folder);
407 }
408
409 if (archive->file != NULL) {
410 g_object_unref (archive->file);
411 archive->file = NULL;
412 }
413 if (archive->local_copy != NULL) {
414 g_object_unref (archive->local_copy);
415 archive->local_copy = NULL;
416 }
417 archive->content_type = NULL;
418
419 if (uri == NULL)
420 return;
421
422 archive->file = g_file_new_for_uri (uri);
423 archive->is_remote = ! g_file_has_uri_scheme (archive->file, "file");
424 if (archive->is_remote)
425 archive->local_copy = get_local_copy_for_file (archive->file);
426 else
427 archive->local_copy = g_file_dup (archive->file);
428}

440static void
441fr_archive_finalize (GObject *object)
442{
443 FrArchive *archive;
444
445 g_return_if_fail (object != NULL);
446 g_return_if_fail (FR_IS_ARCHIVE (object));
447
448 archive = FR_ARCHIVE (object);
449
450 fr_archive_set_uri (archive, NULL);
451 fr_archive_remove_temp_work_dir (archive);
452 if (archive->command != NULL)
453 g_object_unref (archive->command);
454 g_object_unref (archive->process);
455 if (archive->priv->dropped_items_data != NULL) {
456 dropped_items_data_free (archive->priv->dropped_items_data);
457 archive->priv->dropped_items_data = NULL;
458 }
459 g_free (archive->priv->temp_extraction_dir);
460 g_free (archive->priv->extraction_destination);
461 g_free (archive->priv);
462
463 /* Chain up */
464
465 if (G_OBJECT_CLASS (parent_class)->finalize)
466 G_OBJECT_CLASS (parent_class)->finalize (object);
467}

491static const char *
492get_mime_type_from_magic_numbers (GFile *file)
493{
494#ifdef ENABLE_MAGIC
495 static magic_t magic = NULL;
496
497 if (! magic) {
498 magic = magic_open (MAGIC_MIME_TYPE);
499 if (magic)
500 magic_load (magic, NULL);
501 else
502 g_warning ("unable to open magic database");
503 }
504
505 if (magic) {
506 const char * mime_type = NULL;
507 char * filepath = g_file_get_path (file);
508
509 if (filepath) {
510 mime_type = magic_file (magic, filepath);
511 g_free (filepath);
512 }
513
514 if (mime_type)
515 return mime_type;
516
517 g_warning ("unable to detect filetype from magic: %s",
518 magic_error (magic));
519 }
520#else
521 static const struct magic {
522 const unsigned int off;
523 const unsigned int len;
524 const char * const id;
525 const char * const mime_type;
526 } magic_ids [] = {
527 /* magic ids taken from magic/Magdir/archive from the file-4.21 tarball */
528 { 0, 21, "!<arch>" "\x0a" "debian-binary", "application/vnd.debian.binary-package" },
529 { 0, 8, "!<arch>" "\x0a", "application/x-archive" },
530 { 0, 6, "7z\274\257\047\034", "application/x-7z-compressed" },
531 { 7, 7, "**ACE**", "application/x-ace" },
532 { 0, 2, "\x60\xea", "application/x-arj" },
533 { 0, 4, "\xce\xb2\xcf\x81", "application/x-brotli" },
534 { 0, 3, "BZh", "application/x-bzip2" },
535 { 0, 2, "\037\213", "application/x-gzip" },
536 { 2, 3, "-lh", "application/x-lzh-compressed"},
537 { 0, 4, "LZIP", "application/x-lzip" },
538 { 0, 9, "\x89\x4c\x5a\x4f\x00\x0d\x0a\x1a\x0a", "application/x-lzop", },
539 { 0, 4, "Rar!", "application/x-rar" },
540 { 0, 4, "\xed\xab\xee\xdb", "application/x-rpm" },
541 { 0, 4, "RZIP", "application/x-rzip" },
542 { 0, 6, "\3757zXZ\000", "application/x-xz" },
543 { 20, 4, "\xdc\xa7\xc4\xfd", "application/x-zoo", },
544 { 0, 4, "PK\003\004", "application/zip" },
545 { 0, 8, "PK00PK\003\004", "application/zip" },
546 { 0, 4, "LRZI", "application/x-lrzip" },
547 { 0, 2, "\x1f\x9d", "application/x-compress" }, /* using Lempel-Ziv-Welch algorithm */
548 { 0, 2, "\x1f\xa0", "application/x-compress" }, /* using Lempel-Ziv-Huffman algorithm */
549 };
550
551 char buffer[32];
552 size_t i;
553
554 if (! g_load_file_in_buffer (file, buffer, sizeof (buffer), NULL))
555 return NULL;
556
557 for (i = 0; i < G_N_ELEMENTS (magic_ids); i++) {
558 const struct magic * const magic = &magic_ids[i];
559
560 if (sizeof (buffer) < (magic->off + magic->len))
561 g_warning ("buffer underrun for mime type '%s' magic",
562 magic->mime_type);
563 else if (! memcmp (buffer + magic->off, magic->id, magic->len))
564 return magic->mime_type;
565 }
566#endif
567
568 return NULL;
569}

612static gboolean
613create_command_to_load_archive (FrArchive *archive,
614 const char *mime_type)
615{
616 FrCommandCaps requested_capabilities = FR_COMMAND_CAN_DO_NOTHING;
617 GType command_type;
618
619 if (mime_type == NULL)
620 return FALSE;
621
622 /* try with the WRITE capability even when loading, this way we give
623 * priority to the commands that can read and write over commands
624 * that can only read a specific file format. */
625
626 requested_capabilities |= FR_COMMAND_CAN_READ_WRITE;
627 command_type = get_command_type_from_mime_type (mime_type, requested_capabilities);
628
629 /* if no command was found, remove the write capability and try again */
630
631 if (command_type == 0) {
632 requested_capabilities ^= FR_COMMAND_CAN_WRITE;
633 command_type = get_command_type_from_mime_type (mime_type, requested_capabilities);
634 }
635
636 return create_command_from_type (archive,
637 mime_type,
638 command_type,
639 requested_capabilities);
640}

741/* -- copy_extracted_files_to_destination -- */
742
743static void
744move_here (FrArchive *archive)
745{
746 char *content_uri;
747 char *parent;
748 char *parent_parent;
749 char *new_content_uri;
750 GFile *source, *destination, *parent_file;
751 GError *error = NULL;
752
753 content_uri = get_dir_content_if_unique (archive->priv->extraction_destination);
754 if (content_uri == NULL)
755 return;
756
757 parent = remove_level_from_path (content_uri);
758
759 if (uricmp (parent, archive->priv->extraction_destination) == 0) {
760 char *new_uri;
761
762 new_uri = get_alternative_uri_for_uri (archive->priv->extraction_destination);
763
764 source = g_file_new_for_uri (archive->priv->extraction_destination);
765 destination = g_file_new_for_uri (new_uri);
766 if (! g_file_move (source, destination, 0, NULL, NULL, NULL, &error)) {
767 g_warning ("could not rename %s to %s: %s", archive->priv->extraction_destination, new_uri, error->message);

884static void
885action_performed (FrCommand *command,
886 FrAction action,
887 FrProcError *error,
888 FrArchive *archive)
889{
890#ifdef MATE_ENABLE_DEBUG
891 debug (DEBUG_INFO, "%s [DONE] (FR::Archive)\n", get_action_name (action));
892#endif
893
894 switch (action) {
895 case FR_ACTION_DELETING_FILES:
896 if (error->type == FR_PROC_ERROR_NONE) {
897 if (! g_file_has_uri_scheme (archive->file, "file")) {
898 copy_to_remote_location (archive, action);
899 return;
900 }
901 }
902 break;
903
904 case FR_ACTION_ADDING_FILES:
905 if (error->type == FR_PROC_ERROR_NONE) {
906 fr_archive_remove_temp_work_dir (archive);
907 if (archive->priv->continue_adding_dropped_items) {
908 add_dropped_items (archive->priv->dropped_items_data);
909 return;
910 }
911 if (archive->priv->dropped_items_data != NULL) {
912 dropped_items_data_free (archive->priv->dropped_items_data);
913 archive->priv->dropped_items_data = NULL;
914 }
915 /* the name of the volumes are different from the
916 * original name */
917 if (archive->command->multi_volume)
918 fr_archive_change_name (archive, archive->command->filename);
919 if (! g_file_has_uri_scheme (archive->file, "file")) {
920 copy_to_remote_location (archive, action);
921 return;
922 }
923 }
924 break;
925
926 case FR_ACTION_EXTRACTING_FILES:
927 if (error->type == FR_PROC_ERROR_NONE) {
928 if (archive->priv->remote_extraction) {
929 copy_extracted_files_to_destination (archive);
930 return;
931 }
932 else if (archive->priv->extract_here)
933 move_here (archive);
934 }
935 else {
936 /* if an error occurred during extraction remove the
937 * temp extraction dir, if used. */
938 g_print ("action_performed: ERROR!\n");
939
940 if ((archive->priv->remote_extraction) && (archive->priv->temp_extraction_dir != NULL)) {
941 remove_local_directory (archive->priv->temp_extraction_dir);
942 g_free (archive->priv->temp_extraction_dir);
943 archive->priv->temp_extraction_dir = NULL;
944 }
945
946 if (archive->priv->extract_here)
947 remove_directory (archive->priv->extraction_destination);
948 }
949 break;
950
951 case FR_ACTION_LISTING_CONTENT:
952 /* the name of the volumes are different from the
953 * original name */
954 if (archive->command->multi_volume)
955 fr_archive_change_name (archive, archive->command->filename);
956 fr_command_update_capabilities (archive->command);
957 if (! fr_command_is_capable_of (archive->command, FR_COMMAND_CAN_WRITE))
958 archive->read_only = TRUE;
959 break;
960
961 default:
962 /* nothing */
963 break;
964 }
965
966 g_signal_emit (G_OBJECT (archive),
967 fr_archive_signals[DONE],
968 0,
969 action,
970 error);
971}

1511
1512 tmp_base_dir = create_tmp_base_dir (base_dir, dest_dir);
1513 base_dir_created = TRUE;
1514
1515 if (dest_dir[0] == G_DIR_SEPARATOR)
1516 rel_dest_dir = dest_dir + 1;
1517
1518 new_file_list = NULL;
1519 for (scan = file_list; scan != NULL; scan = scan->next) {
1520 char *filename = scan->data;
1521 new_file_list = g_list_prepend (new_file_list, g_build_filename (rel_dest_dir, filename, NULL));
1522 }
1523 }
1524 else {
1525 tmp_base_dir = g_strdup (base_dir);
1526 new_file_list = path_list_dup(file_list);
1527 }
1528
1529 /* if the command cannot update, get the list of files that are
1530 * newer than the ones in the archive. */
1531
1532 if (update && ! archive->command->propAddCanUpdate) {
1533 GList *tmp_file_list;
1534
1535 tmp_file_list = new_file_list;
1536 new_file_list = newer_files_only (archive, tmp_file_list, tmp_base_dir);
1537 path_list_free (tmp_file_list);
1538 }
1539
1540 if (new_file_list == NULL) {
1541 debug (DEBUG_INFO, "nothing to update.\n");
1542
1543 if (base_dir_created)
1544 remove_local_directory (tmp_base_dir);
1545 g_free (tmp_base_dir);
1546
1547 archive->process->error.type = FR_PROC_ERROR_NONE;
1548 g_signal_emit_by_name (G_OBJECT (archive->process),
1549 "done",
1550 &archive->process->error);
1551 return;
1552 }
1553
1554 archive->command->creating_archive = ! g_file_query_exists (archive->local_copy, archive->priv->cancellable);
1555
1556 /* create the new archive in a temporary sub-directory, this allows
1557 * to cancel the operation without losing the original archive and
1558 * removing possible temporary files created by the command. */
1559
1560 {
1561 GFile *local_copy_parent;
1562 char *archive_dir;
1563 GFile *tmp_file;
1564
1565 /* create the new archive in a sub-folder of the original
1566 * archive this way the 'mv' command is fast. */
1567
1568 local_copy_parent = g_file_get_parent (archive->local_copy);
1569 archive_dir = g_file_get_path (local_copy_parent);
1570 tmp_archive_dir = get_temp_work_dir (archive_dir);
1571 archive_filename = g_file_get_path (archive->local_copy);
1572 tmp_archive_filename = g_build_filename (tmp_archive_dir, file_name_from_path (archive_filename), NULL);
1573 tmp_file = g_file_new_for_path (tmp_archive_filename);
1574 g_object_set (archive->command, "file", tmp_file, NULL);
1575
1576 if (! archive->command->creating_archive) {
1577 /* copy the original archive to the new position */
1578
1579 fr_process_begin_command (archive->process, "cp");
1580 fr_process_add_arg (archive->process, "-f");
1581 fr_process_add_arg (archive->process, archive_filename);
1582 fr_process_add_arg (archive->process, tmp_archive_filename);
1583 fr_process_end_command (archive->process);
1584 }
1585
1586 g_object_unref (tmp_file);
1587 g_free (archive_dir);
1588 g_object_unref (local_copy_parent);
1589 }
1590
1591 fr_command_uncompress (archive->command);
1592
1593 /* when files are already present in a tar archive and are added
1594 * again, they are not replaced, so we have to delete them first. */
1595
1596 /* if we are adding (== ! update) and 'add' cannot replace or
1597 * if we are updating and 'add' cannot update,
1598 * delete the files first. */
1599
1600 if ((! update && ! archive->command->propAddCanReplace)
1601 || (update && ! archive->command->propAddCanUpdate))
1602 {
1603 GList *del_list = NULL;
1604
1605 for (scan = new_file_list; scan != NULL; scan = scan->next) {
1606 char *filename = scan->data;
1607 if (find_file_in_archive (archive, filename))
1608 del_list = g_list_prepend (del_list, filename);
1609 }
1610
1611 /* delete */
1612
1613 if (del_list != NULL) {
1614 delete_from_archive (archive, del_list);
1615 fr_process_set_ignore_error (archive->process, TRUE);
1616 g_list_free (del_list);
1617 }
1618 }
1619
1620 /* add now. */
1621
1622 fr_command_set_n_files (archive->command, g_list_length (new_file_list));
1623
1624 if (archive->command->propListFromFile
1625 && (archive->command->n_files > LIST_LENGTH_TO_USE_FILE))
1626 {
1627 char *list_dir;
1628 char *list_filename;
1629 GError *error = NULL;
1630
1631 if (! save_list_to_temp_file (new_file_list, &list_dir, &list_filename, &error)) {
1632 archive->process->error.type = FR_PROC_ERROR_GENERIC;
1633 archive->process->error.status = 0;
1634 archive->process->error.gerror = g_error_copy (error);
1635 g_signal_emit_by_name (G_OBJECT (archive->process),
1636 "done",
1637 &archive->process->error);
1638 g_clear_error (&error);
1639 error_occurred = TRUE;
1640 }
1641 else {
1642 fr_command_add (archive->command,
1643 list_filename,
1644 new_file_list,
1645 tmp_base_dir,
1646 update,
1647 recursive);
1648
1649 /* remove the temp dir */
1650
1651 fr_process_begin_command (archive->process, "rm");
1652 fr_process_set_working_dir (archive->process, g_get_tmp_dir());
1653 fr_process_set_sticky (archive->process, TRUE);
1654 fr_process_add_arg (archive->process, "-rf");
1655 fr_process_add_arg (archive->process, list_dir);
1656 fr_process_end_command (archive->process);
1657 }
1658
1659 g_free (list_filename);
1660 g_free (list_dir);
1661 }
1662 else {
1663 GList *chunks = NULL;
1664
1665 /* specify the file list on the command line, splitting
1666 * in more commands to avoid to overflow the command line
1667 * length limit. */
1668
1669 chunks = split_in_chunks (new_file_list);
1670 for (scan = chunks; scan != NULL; scan = scan->next) {
1671 GList *chunk = scan->data;
1672
1673 fr_command_add (archive->command,
1674 NULL,
1675 chunk,
1676 tmp_base_dir,
1677 update,
1678 recursive);
1679 g_list_free (chunk);
1680 }
1681
1682 g_list_free (chunks);
1683 }
1684
1685 path_list_free (new_file_list);
1686
1687 if (! error_occurred) {
1688 fr_command_recompress (archive->command);
1689
1690 /* move the new archive to the original position */
1691
1692 fr_process_begin_command (archive->process, "mv");
1693 fr_process_add_arg (archive->process, "-f");
1694 fr_process_add_arg (archive->process, tmp_archive_filename);
1695 fr_process_add_arg (archive->process, archive_filename);
1696 fr_process_end_command (archive->process);
1697
1698 /* remove the temp sub-directory */
1699
1700 fr_process_begin_command (archive->process, "rm");
1701 fr_process_set_working_dir (archive->process, g_get_tmp_dir());
1702 fr_process_set_sticky (archive->process, TRUE);
1703 fr_process_add_arg (archive->process, "-rf");
1704 fr_process_add_arg (archive->process, tmp_archive_dir);
1705 fr_process_end_command (archive->process);
1706
1707 /* remove the base dir */
1708
1709 if (base_dir_created) {
1710 fr_process_begin_command (archive->process, "rm");
1711 fr_process_set_working_dir (archive->process, g_get_tmp_dir());
1712 fr_process_set_sticky (archive->process, TRUE);
1713 fr_process_add_arg (archive->process, "-rf");
1714 fr_process_add_arg (archive->process, tmp_base_dir);
1715 fr_process_end_command (archive->process);
1716 }
1717 }
1718
1719 g_free (tmp_archive_filename);
1720 g_free (archive_filename);
1721 g_free (tmp_archive_dir);
1722 g_free (tmp_base_dir);
1723}
