83static gint
84ev_sidebar_bookmarks_get_selected_page (EvSidebarBookmarks *sidebar_bookmarks,
85 GtkTreeSelection *selection)
86{
87 GtkTreeModel *model;
88 GtkTreeIter iter;
89
90 if (gtk_tree_selection_get_selected (selection, &model, &iter)) {
91 guint page;
92
93 gtk_tree_model_get (model, &iter,

162static gint
163compare_bookmarks (EvBookmark *a,
164 EvBookmark *b)
165{
166 if (a->page < b->page)
167 return -1;
168 if (a->page > b->page)
169 return 1;
170 return 0;
171}

173static void
174ev_sidebar_bookmarks_update (EvSidebarBookmarks *sidebar_bookmarks)
175{
176 EvSidebarBookmarksPrivate *priv = sidebar_bookmarks->priv;
177 GtkListStore *model;
178 GList *items, *l;
179 GtkTreeIter iter;
180
181 model = GTK_LIST_STORE (gtk_tree_view_get_model (GTK_TREE_VIEW (priv->tree_view)));
182 gtk_list_store_clear (model);
183
184 if (!priv->bookmarks) {
185 g_object_set (priv->tree_view, "has-tooltip", FALSE, NULL);
186 return;
187 }
188
189 items = ev_bookmarks_get_bookmarks (priv->bookmarks);
190 items = g_list_sort (items, (GCompareFunc)compare_bookmarks);
191 for (l = items; l; l = g_list_next (l)) {
192 EvBookmark *bm = (EvBookmark *)l->data;
193
194 gtk_list_store_append (model, &iter);
195 gtk_list_store_set (model, &iter,
196 COLUMN_MARKUP, bm->title,
197 COLUMN_PAGE, bm->page,
198 -1);
199 }
200 g_list_free (items);
201 g_object_set (priv->tree_view, "has-tooltip", TRUE, NULL);
202}

256static void
257ev_sidebar_bookmarks_bookmark_renamed (GtkCellRendererText *renderer,
258 const gchar *path_string,
259 const gchar *new_text,
260 EvSidebarBookmarks *sidebar_bookmarks)
261{
262 EvSidebarBookmarksPrivate *priv = sidebar_bookmarks->priv;
263 GtkTreePath *path = gtk_tree_path_new_from_string (path_string);
264 GtkTreeModel *model;
265 GtkTreeIter iter;
266 guint page;
267 EvBookmark bm;
268
269 if (!new_text || new_text[0] == '\0')
270 return;
271
272 model = gtk_tree_view_get_model (GTK_TREE_VIEW (priv->tree_view));
273 gtk_tree_model_get_iter (model, &iter, path);
274 gtk_tree_model_get (model, &iter,
275 COLUMN_PAGE, &page,
276 -1);
277 gtk_tree_path_free (path);
278
279 bm.page = page;
280 bm.title = g_strdup (new_text);
281 ev_bookmarks_update (priv->bookmarks, &bm);
282}

284static gboolean
285ev_sidebar_bookmarks_query_tooltip (GtkWidget *widget,
286 gint x,
287 gint y,
288 gboolean keyboard_tip,
289 GtkTooltip *tooltip,
290 EvSidebarBookmarks *sidebar_bookmarks)
291{
292 EvSidebarBookmarksPrivate *priv = sidebar_bookmarks->priv;
293 GtkTreeModel *model;
294 GtkTreeIter iter;
295 GtkTreePath *path = NULL;
296 EvDocument *document;
297 guint page;
298 gchar *page_label;
299 gchar *text;
300
301 model = gtk_tree_view_get_model (GTK_TREE_VIEW (priv->tree_view));
302 if (!gtk_tree_view_get_tooltip_context (GTK_TREE_VIEW (priv->tree_view),
303 &x, &y, keyboard_tip,
304 &model, &path, &iter))
305 return FALSE;
306
307 gtk_tree_model_get (model, &iter,
308 COLUMN_PAGE, &page,
309 -1);
310
311 document = ev_document_model_get_document (priv->model);
312 page_label = ev_document_get_page_label (document, page);
313 text = g_strdup_printf (_("Page %s"), page_label);
314 gtk_tooltip_set_text (tooltip, text);
315 g_free (text);
316 g_free (page_label);
317
318 gtk_tree_view_set_tooltip_row (GTK_TREE_VIEW (priv->tree_view),
319 tooltip, path);
320 gtk_tree_path_free (path);
321
322 return TRUE;
323}

325static gboolean
326ev_sidebar_bookmarks_popup_menu_show (EvSidebarBookmarks *sidebar_bookmarks,
327 gint x,
328 gint y,
329 gboolean keyboard_mode)
330{
331 EvSidebarBookmarksPrivate *priv = sidebar_bookmarks->priv;
332 GtkTreeView *tree_view = GTK_TREE_VIEW (sidebar_bookmarks->priv->tree_view);
333 GtkTreeSelection *selection = gtk_tree_view_get_selection (tree_view);
334
335 if (keyboard_mode) {
336 if (!gtk_tree_selection_get_selected (selection, NULL, NULL))
337 return FALSE;
338 } else {
339 GtkTreePath *path;
340
341 if (!gtk_tree_view_get_path_at_pos (tree_view, x, y, &path, NULL, NULL, NULL))
342 return FALSE;
343
344 g_signal_handlers_block_by_func (selection,
345 ev_sidebar_bookmarks_selection_changed,
346 sidebar_bookmarks);
347 gtk_tree_view_set_cursor (tree_view, path, NULL, FALSE);
348 g_signal_handlers_unblock_by_func (selection,
349 ev_sidebar_bookmarks_selection_changed,
350 sidebar_bookmarks);
351 gtk_tree_path_free (path);
352 }
353
354 if (!priv->popup)
355 priv->popup = gtk_ui_manager_get_widget (priv->ui_manager, "/BookmarksPopup");
356
357 gtk_menu_popup_at_pointer (GTK_MENU (priv->popup), NULL);
358 return TRUE;
359}

559 EvBookmarks *bookmarks)
560{
561 EvSidebarBookmarksPrivate *priv = sidebar_bookmarks->priv;
562
563 g_return_if_fail (EV_IS_BOOKMARKS (bookmarks));

566 return;
567
568 if (priv->bookmarks)
569 g_object_unref (priv->bookmarks);
570 priv->bookmarks = g_object_ref (bookmarks);
571 g_signal_connect (priv->bookmarks, "changed",
572 G_CALLBACK (ev_sidebar_bookmarks_changed),
573 sidebar_bookmarks);
574

576 ev_sidebar_bookmarks_update (sidebar_bookmarks);
