45void
46wayland_panel_toplevel_update_placement (PanelToplevel* toplevel)
47{
48 GtkWindow* window;
49 gboolean expand;
50 PanelOrientation orientation;
51 gboolean anchor[GTK_LAYER_SHELL_EDGE_ENTRY_NUMBER];
52
53 window = GTK_WINDOW (toplevel);
54 expand = panel_toplevel_get_expand (toplevel);
55 orientation = panel_toplevel_get_orientation (toplevel);
56 for (int i = 0; i < GTK_LAYER_SHELL_EDGE_ENTRY_NUMBER; i++)
57 anchor[i] = expand;
58
59 switch (orientation) {
60 case PANEL_ORIENTATION_LEFT:
61 anchor[GTK_LAYER_SHELL_EDGE_LEFT] = TRUE;
62 anchor[GTK_LAYER_SHELL_EDGE_RIGHT] = FALSE;
63 break;
64 case PANEL_ORIENTATION_RIGHT:
65 anchor[GTK_LAYER_SHELL_EDGE_RIGHT] = TRUE;
66 anchor[GTK_LAYER_SHELL_EDGE_LEFT] = FALSE;
67 break;
68 case PANEL_ORIENTATION_TOP:
69 anchor[GTK_LAYER_SHELL_EDGE_TOP] = TRUE;
70 anchor[GTK_LAYER_SHELL_EDGE_BOTTOM] = FALSE;
71 break;
72 case PANEL_ORIENTATION_BOTTOM:
73 anchor[GTK_LAYER_SHELL_EDGE_BOTTOM] = TRUE;
74 anchor[GTK_LAYER_SHELL_EDGE_TOP] = FALSE;
75 break;
76 default:
77 g_warning ("Invalid panel orientation %d", orientation);
78 }
79
80 for (int i = 0; i < GTK_LAYER_SHELL_EDGE_ENTRY_NUMBER; i++)
81 gtk_layer_set_anchor (window, i, anchor[i]);
82}
