315static void
316dump_entry( MateConfEntry *entry, void *user_data )
317{
318 static const gchar *thisfn = "na_mateconf_utils_dump_entry";
319 gchar *str = NULL;
320 gboolean str_free = FALSE;
321 GSList *value_list, *it;
322 MateConfValueType type_list;
323 GString *string;
324
325 gchar *key = g_path_get_basename( mateconf_entry_get_key( entry ));
326 MateConfValue *value = mateconf_entry_get_value( entry );
327
328 if( value ){
329 switch( value->type ){
330 case MATECONF_VALUE_STRING:
331 str = ( gchar * ) mateconf_value_get_string( value );
332 break;
333
334 case MATECONF_VALUE_INT:
335 str = g_strdup_printf( "%d", mateconf_value_get_int( value ));
336 str_free = TRUE;
337 break;
338
339 case MATECONF_VALUE_FLOAT:
340 str = g_strdup_printf( "%f", mateconf_value_get_float( value ));
341 str_free = TRUE;
342 break;
343
344 case MATECONF_VALUE_BOOL:
345 str = g_strdup_printf( "%s", mateconf_value_get_bool( value ) ? "True":"False" );
346 str_free = TRUE;
347 break;
348
349 case MATECONF_VALUE_LIST:
350 type_list = mateconf_value_get_list_type( value );
351 value_list = mateconf_value_get_list( value );
352 switch( type_list ){
353 case MATECONF_VALUE_STRING:
354 string = g_string_new( "[" );
355 for( it = value_list ; it ; it = it->next ){
356 if( g_utf8_strlen( string->str, -1 ) > 1 ){
357 string = g_string_append( string, "," );
358 }
359 string = g_string_append( string, ( const gchar * ) mateconf_value_get_string( it->data ));
360 }
361 string = g_string_append( string, "]" );
362 str = g_string_free( string, FALSE );
363 break;
364 default:
365 str = g_strdup( "(undetermined value)" );
366 }
367 str_free = TRUE;
368 break;
369
370 default:
371 str = g_strdup( "(undetermined value)" );
372 str_free = TRUE;
373 }
374 }
375
376 g_debug( "%s: key=%s, value=%s", thisfn, key, str );
377
378 if( str_free ){
379 g_free( str );
380 }
381
382 g_free( key );
383}

739GSList *
740na_mateconf_utils_slist_from_string( const gchar *value )
741{
742 GSList *slist;
743 gchar *tmp_string;
744
745 tmp_string = g_strdup( value );
746 g_strstrip( tmp_string );
747
748 if( !tmp_string || strlen( tmp_string ) < 3 ){
749 g_free( tmp_string );
750 return( NULL );
751 }
752
753 if( tmp_string[0] != '[' || tmp_string[strlen(tmp_string)-1] != ']' ){
754 g_free( tmp_string );
755 return( NULL );
756 }
757
758 tmp_string += 1;
759 tmp_string[strlen(tmp_string)-1] = '\0';
760 slist = na_core_utils_slist_from_split( tmp_string, "," );
761
762 return( slist );
763}

797static MateConfValue *
798read_value( MateConfClient *mateconf, const gchar *path, gboolean use_schema, MateConfValueType type )
799{
800 static const gchar *thisfn = "na_mateconf_utils_read_value";
801 GError *error = NULL;
802 MateConfValue *value = NULL;
803
804 if( use_schema ){
805 value = mateconf_client_get( mateconf, path, &error );
806 } else {
807 value = mateconf_client_get_without_default( mateconf, path, &error );
808 }
809
810 if( error ){
811 g_warning( "%s: path=%s, error=%s", thisfn, path, error->message );
812 g_error_free( error );
813 if( value ){
814 mateconf_value_free( value );
815 value = NULL;
816 }
817 }
818
819 if( value ){
820 if( value->type != type ){
821 g_warning( "%s: path=%s, found type '%u' while waiting for type '%u'", thisfn, path, value->type, type );
822 mateconf_value_free( value );
823 value = NULL;
824 }
825 }
826
827 return( value );
828}
