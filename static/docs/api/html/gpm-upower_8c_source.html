50static const gchar *
51gpm_upower_get_device_icon_index (UpDevice *device)
52{
53 gdouble percentage;
54 /* get device properties */
55 g_object_get (device, "percentage", &percentage, NULL);
56 if (percentage < 10)
57 return "000";
58 else if (percentage < 30)
59 return "020";
60 else if (percentage < 50)
61 return "040";
62 else if (percentage < 70)
63 return "060";
64 else if (percentage < 90)
65 return "080";
66 return "100";
67}

75gchar *
76gpm_upower_get_device_icon (UpDevice *device)
77{
78 gchar *filename = NULL;
79 const gchar *prefix = NULL;
80 const gchar *index_str;
81 UpDeviceKind kind;
82 UpDeviceState state;
83 gboolean is_present;
84 gdouble percentage;
85
86 g_return_val_if_fail (device != NULL, NULL);
87
88 /* get device properties */
89 g_object_get (device,
90 "kind", &kind,
91 "state", &state,
92 "percentage", &percentage,
93 "is-present", &is_present,
94 NULL);
95
96 /* get correct icon prefix */
97 prefix = up_device_kind_to_string (kind);
98
99 /* get the icon from some simple rules */
100 if (kind == UP_DEVICE_KIND_LINE_POWER) {
101 filename = g_strdup ("gpm-ac-adapter");
102 } else if (kind == UP_DEVICE_KIND_MONITOR) {
103 filename = g_strdup ("gpm-monitor");
104 } else if (kind == UP_DEVICE_KIND_UPS) {
105 if (!is_present) {
106 /* battery missing */
107 filename = g_strdup_printf ("gpm-%s-missing", prefix);
108
109 } else if (state == UP_DEVICE_STATE_FULLY_CHARGED) {
110 filename = g_strdup_printf ("gpm-%s-100", prefix);
111
112 } else if (state == UP_DEVICE_STATE_CHARGING) {
113 index_str = gpm_upower_get_device_icon_index (device);
114 filename = g_strdup_printf ("gpm-%s-%s-charging", prefix, index_str);
115
116 } else if (state == UP_DEVICE_STATE_DISCHARGING) {
117 index_str = gpm_upower_get_device_icon_index (device);
118 filename = g_strdup_printf ("gpm-%s-%s", prefix, index_str);
119 }
120 } else if (kind == UP_DEVICE_KIND_BATTERY) {
121 if (!is_present) {
122 /* battery missing */
123 filename = g_strdup_printf ("gpm-%s-missing", prefix);
124
125 } else if (state == UP_DEVICE_STATE_EMPTY) {
126 filename = g_strdup_printf ("gpm-%s-empty", prefix);
127
128 } else if (state == UP_DEVICE_STATE_FULLY_CHARGED) {
129 filename = g_strdup_printf ("gpm-%s-charged", prefix);
130
131 } else if (state == UP_DEVICE_STATE_CHARGING) {
132 index_str = gpm_upower_get_device_icon_index (device);
133 filename = g_strdup_printf ("gpm-%s-%s-charging", prefix, index_str);
134
135 } else if (state == UP_DEVICE_STATE_DISCHARGING) {
136 index_str = gpm_upower_get_device_icon_index (device);
137 filename = g_strdup_printf ("gpm-%s-%s", prefix, index_str);
138
139 } else if (state == UP_DEVICE_STATE_PENDING_CHARGE) {
140 index_str = gpm_upower_get_device_icon_index (device);
141 /* FIXME: do new grey icons */
142 filename = g_strdup_printf ("gpm-%s-%s-charging", prefix, index_str);
143
144 } else if (state == UP_DEVICE_STATE_PENDING_DISCHARGE) {
145 index_str = gpm_upower_get_device_icon_index (device);
146 filename = g_strdup_printf ("gpm-%s-%s", prefix, index_str);
147 } else {
148 filename = g_strdup ("gpm-battery-missing");
149 }
150
151 } else if (kind == UP_DEVICE_KIND_MOUSE ||
152 kind == UP_DEVICE_KIND_KEYBOARD ||
153 kind == UP_DEVICE_KIND_PHONE) {
154 if (!is_present) {
155 /* battery missing */
156 filename = g_strdup_printf ("gpm-%s-000", prefix);
157
158 } else if (state == UP_DEVICE_STATE_FULLY_CHARGED) {
159 filename = g_strdup_printf ("gpm-%s-100", prefix);
160
161 } else if (state == UP_DEVICE_STATE_DISCHARGING) {
162 index_str = gpm_upower_get_device_icon_index (device);
163 filename = g_strdup_printf ("gpm-%s-%s", prefix, index_str);
164 }
165 } else if (kind == UP_DEVICE_KIND_GAMING_INPUT) {
166 index_str = gpm_upower_get_device_icon_index (device);
167 filename = g_strdup_printf ("gpm-%s-%s", prefix, index_str);
168 }
169
170 /* nothing matched */
171 if (filename == NULL) {
172 g_warning ("nothing matched, falling back to default icon");
173 filename = g_strdup ("dialog-warning");
174 }
175
176 g_debug ("got filename: %s", filename);
177 return filename;
178}

183gchar *
184gpm_upower_get_device_summary (UpDevice *device)
185{
186 const gchar *kind_desc = NULL;
187 gchar *description = NULL;
188 guint time_to_full_round;
189 guint time_to_empty_round;
190 gchar *time_to_full_str;
191 gchar *time_to_empty_str;
192 UpDeviceKind kind;
193 UpDeviceState state;
194 gdouble percentage;
195 gboolean is_present;
196 gint64 time_to_full;
197 gint64 time_to_empty;
198
199 /* get device properties */
200 g_object_get (device,
201 "kind", &kind,
202 "state", &state,
203 "percentage", &percentage,
204 "is-present", &is_present,
205 "time-to-full", &time_to_full,
206 "time-to-empty", &time_to_empty,
207 NULL);
208
209 kind_desc = gpm_device_kind_to_localised_string (kind, 1);
210
211 /* not installed */
212 if (!is_present) {
213 /* TRANSLATORS: device not present */
214 return g_strdup_printf (_("%s not present"), kind_desc);
215 }
216
217 /* don't display all the extra stuff for keyboards and mice */
218 if (kind == UP_DEVICE_KIND_MOUSE ||
219 kind == UP_DEVICE_KIND_KEYBOARD ||
220 kind == UP_DEVICE_KIND_PDA)
221 return g_strdup_printf ("%s (%.1f%%)", kind_desc, percentage);
222
223 /* we care if we are on AC */
224 if (kind == UP_DEVICE_KIND_PHONE) {
225 if (state == UP_DEVICE_STATE_CHARGING || !(state == UP_DEVICE_STATE_DISCHARGING)) {
226 /* TRANSLATORS: a phone is charging */
227 return g_strdup_printf (_("%s charging (%.1f%%)"), kind_desc, percentage);
228 }
229 return g_strdup_printf ("%s (%.1f%%)", kind_desc, percentage);
230 }
231
232 /* precalculate so we don't get Unknown time remaining */
233 time_to_full_round = egg_precision_round_down (time_to_full, GPM_UP_TIME_PRECISION);
234 time_to_empty_round = egg_precision_round_down (time_to_empty, GPM_UP_TIME_PRECISION);
235
236 /* we always display "Laptop battery 16 minutes remaining" as we need to clarify what device we are refering to */
237 if (state == UP_DEVICE_STATE_FULLY_CHARGED) {
238
239 if (kind == UP_DEVICE_KIND_BATTERY && time_to_empty_round > GPM_UP_TEXT_MIN_TIME) {
240 time_to_empty_str = gpm_get_timestring (time_to_empty_round);
241 /* TRANSLATORS: The laptop battery is fully charged, and we know a time */
242 description = g_strdup_printf (_("Battery is fully charged.\nProvides %s laptop runtime"),
243 time_to_empty_str);
244 g_free (time_to_empty_str);
245 } else {
246 /* TRANSLATORS: the device is fully charged */
247 description = g_strdup_printf (_("%s is fully charged"), kind_desc);
248 }
249
250 } else if (state == UP_DEVICE_STATE_DISCHARGING) {
251
252 if (time_to_empty_round > GPM_UP_TEXT_MIN_TIME) {
253 time_to_empty_str = gpm_get_timestring (time_to_empty_round);
254 /* TRANSLATORS: the device is discharging, and we have a time remaining */
255 description = g_strdup_printf (_("%s %s remaining (%.1f%%)"),
256 kind_desc, time_to_empty_str, percentage);
257 g_free (time_to_empty_str);
258 } else {
259 /* TRANSLATORS: the device is discharging, but we only have a percentage */
260 description = g_strdup_printf (_("%s discharging (%.1f%%)"),
261 kind_desc, percentage);
262 }
263
264 } else if (state == UP_DEVICE_STATE_CHARGING) {
265
266 if (time_to_full_round > GPM_UP_TEXT_MIN_TIME &&
267 time_to_empty_round > GPM_UP_TEXT_MIN_TIME) {
268
269 /* display both discharge and charge time */
270 time_to_full_str = gpm_get_timestring (time_to_full_round);
271 time_to_empty_str = gpm_get_timestring (time_to_empty_round);
272
273 /* TRANSLATORS: the device is charging, and we have a time to full and empty */
274 description = g_strdup_printf (_("%s %s until charged (%.1f%%)\nProvides %s battery runtime"),
275 kind_desc, time_to_full_str, percentage, time_to_empty_str);
276 g_free (time_to_full_str);
277 g_free (time_to_empty_str);
278
279 } else if (time_to_full_round > GPM_UP_TEXT_MIN_TIME) {
280
281 /* display only charge time */
282 time_to_full_str = gpm_get_timestring (time_to_full_round);
283
284 /* TRANSLATORS: device is charging, and we have a time to full and a percentage */
285 description = g_strdup_printf (_("%s %s until charged (%.1f%%)"),
286 kind_desc, time_to_full_str, percentage);
287 g_free (time_to_full_str);
288 } else {
289
290 /* TRANSLATORS: device is charging, but we only have a percentage */
291 description = g_strdup_printf (_("%s charging (%.1f%%)"),
292 kind_desc, percentage);
293 }
294
295 } else if (state == UP_DEVICE_STATE_PENDING_DISCHARGE) {
296
297 /* TRANSLATORS: this is only shown for laptops with multiple batteries */
298 description = g_strdup_printf (_("%s waiting to discharge (%.1f%%)"),
299 kind_desc, percentage);
300
301 } else if (state == UP_DEVICE_STATE_PENDING_CHARGE) {
302
303 /* TRANSLATORS: this is only shown for laptops with multiple batteries */
304 description = g_strdup_printf (_("%s waiting to charge (%.1f%%)"), kind_desc, percentage);
305
306 } else if (state == UP_DEVICE_STATE_EMPTY) {
307
308 /* TRANSLATORS: when the device has no charge left */
309 description = g_strdup_printf (_("%s empty"), kind_desc);
310
311 } else {
312 g_warning ("in an undefined state we are not charging or "
313 "discharging and the batteries are also not charged");
314 description = g_strdup_printf ("%s (%.1f%%)", kind_desc, percentage);
315 }
316
317 return description;
318}

323gchar *
324gpm_upower_get_device_description (UpDevice *device)
325{
326 GString *details;
327 const gchar *text;
328 gchar *time_str;
329 UpDeviceKind kind;
330 UpDeviceState state;
331 UpDeviceTechnology technology;
332 gdouble percentage;
333 gdouble capacity;
334 gdouble energy;
335 gdouble energy_full;
336 gdouble energy_full_design;
337 gdouble energy_rate;
338 gboolean is_present;
339 gint64 time_to_full;
340 gint64 time_to_empty;
341 gchar *vendor = NULL;
342 gchar *serial = NULL;
343 gchar *model = NULL;
344
345 g_return_val_if_fail (device != NULL, NULL);
346
347 /* get device properties */
348 g_object_get (device,
349 "kind", &kind,
350 "state", &state,
351 "percentage", &percentage,
352 "is-present", &is_present,
353 "time-to-full", &time_to_full,
354 "time-to-empty", &time_to_empty,
355 "technology", &technology,
356 "capacity", &capacity,
357 "energy", &energy,
358 "energy-full", &energy_full,
359 "energy-full-design", &energy_full_design,
360 "energy-rate", &energy_rate,
361 "vendor", &vendor,
362 "serial", &serial,
363 "model", &model,
364 NULL);
365
366 details = g_string_new ("");
367 text = gpm_device_kind_to_localised_string (kind, 1);
368 /* TRANSLATORS: the type of data, e.g. Laptop battery */
369 g_string_append_printf (details, "<b>%s</b> %s\n", _("Product:"), text);
370
371 if (!is_present) {
372 /* TRANSLATORS: device is missing */
373 g_string_append_printf (details, "<b>%s</b> %s\n", _("Status:"), _("Missing"));
374 } else if (state == UP_DEVICE_STATE_FULLY_CHARGED) {
375 /* TRANSLATORS: device is charged */
376 g_string_append_printf (details, "<b>%s</b> %s\n", _("Status:"), _("Charged"));
377 } else if (state == UP_DEVICE_STATE_CHARGING) {
378 /* TRANSLATORS: device is charging */
379 g_string_append_printf (details, "<b>%s</b> %s\n", _("Status:"), _("Charging"));
380 } else if (state == UP_DEVICE_STATE_DISCHARGING) {
381 /* TRANSLATORS: device is discharging */
382 g_string_append_printf (details, "<b>%s</b> %s\n", _("Status:"), _("Discharging"));
383 }
384
385 if (percentage >= 0) {
386 /* TRANSLATORS: percentage */
387 g_string_append_printf (details, "<b>%s</b> %.1f%%\n", _("Percentage charge:"), percentage);
388 }
389 if (vendor) {
390 /* TRANSLATORS: manufacturer */
391 g_string_append_printf (details, "<b>%s</b> %s\n", _("Vendor:"), vendor);
392 }
393 if (technology != UP_DEVICE_TECHNOLOGY_UNKNOWN) {
394 text = gpm_device_technology_to_localised_string (technology);
395 /* TRANSLATORS: how the battery is made, e.g. Lithium Ion */
396 g_string_append_printf (details, "<b>%s</b> %s\n", _("Technology:"), text);
397 }
398 if (serial) {
399 /* TRANSLATORS: serial number of the battery */
400 g_string_append_printf (details, "<b>%s</b> %s\n", _("Serial number:"), serial);
401 }
402 if (model) {
403 /* TRANSLATORS: model number of the battery */
404 g_string_append_printf (details, "<b>%s</b> %s\n", _("Model:"), model);
405 }
406 if (time_to_full > 0) {
407 time_str = gpm_get_timestring (time_to_full);
408 /* TRANSLATORS: time to fully charged */
409 g_string_append_printf (details, "<b>%s</b> %s\n", _("Charge time:"), time_str);
410 g_free (time_str);
411 }
412 if (time_to_empty > 0) {
413 time_str = gpm_get_timestring (time_to_empty);
414 /* TRANSLATORS: time to empty */
415 g_string_append_printf (details, "<b>%s</b> %s\n", _("Discharge time:"), time_str);
416 g_free (time_str);
417 }
418 if (capacity > 0) {
419 const gchar *condition;
420 if (capacity > 99) {
421 /* TRANSLATORS: Excellent, Good, Fair and Poor are all related to battery Capacity */
422 condition = _("Excellent");
423 } else if (capacity > 90) {
424 condition = _("Good");
425 } else if (capacity > 70) {
426 condition = _("Fair");
427 } else {
428 condition = _("Poor");
429 }
430 /* TRANSLATORS: %.1f is a percentage and %s the condition (Excellent, Good, ...) */
431 g_string_append_printf (details, "<b>%s</b> %.1f%% (%s)\n",
432 _("Capacity:"), capacity, condition);
433 }
434 if (kind == UP_DEVICE_KIND_BATTERY) {
435 if (energy > 0) {
436 /* TRANSLATORS: current charge */
437 g_string_append_printf (details, "<b>%s</b> %.1f Wh\n",
438 _("Current charge:"), energy);
439 }
440 if (energy_full > 0 &&
441 energy_full_design != energy_full) {
442 /* TRANSLATORS: last full is the charge the battery was seen to charge to */
443 g_string_append_printf (details, "<b>%s</b> %.1f Wh\n",
444 _("Last full charge:"), energy_full);
445 }
446 if (energy_full_design > 0) {
447 /* Translators: */
448 /* TRANSLATORS: Design charge is the amount of charge the battery is designed to have when brand new */
449 g_string_append_printf (details, "<b>%s</b> %.1f Wh\n",
450 _("Design charge:"), energy_full_design);
451 }
452 if (energy_rate > 0) {
453 /* TRANSLATORS: the charge or discharge rate */
454 g_string_append_printf (details, "<b>%s</b> %.1f W\n",
455 _("Charge rate:"), energy_rate);
456 }
457 }
458 if (kind == UP_DEVICE_KIND_MOUSE ||
459 kind == UP_DEVICE_KIND_KEYBOARD) {
460 if (energy > 0) {
461 /* TRANSLATORS: the current charge for CSR devices */
462 g_string_append_printf (details, "<b>%s</b> %.0f/7\n",
463 _("Current charge:"), energy);
464 }
465 if (energy_full_design > 0) {
466 /* TRANSLATORS: the design charge for CSR devices */
467 g_string_append_printf (details, "<b>%s</b> %.0f/7\n",
468 _("Design charge:"), energy_full_design);
469 }
470 }
471 /* remove the last \n */
472 g_string_truncate (details, details->len-1);
473
474 g_free (vendor);
475 g_free (serial);
476 g_free (model);
477 return g_string_free (details, FALSE);
478}

483const gchar *
484gpm_device_kind_to_localised_string (UpDeviceKind kind, guint number)
485{
486 const gchar *text = NULL;
487 switch (kind) {
488 case UP_DEVICE_KIND_LINE_POWER:
489 /* TRANSLATORS: system power cord */
490 text = ngettext ("AC adapter", "AC adapters", number);
491 break;
492 case UP_DEVICE_KIND_BATTERY:
493 /* TRANSLATORS: laptop primary battery */
494 text = ngettext ("Laptop battery", "Laptop batteries", number);
495 break;
496 case UP_DEVICE_KIND_UPS:
497 /* TRANSLATORS: battery-backed AC power source */
498 text = ngettext ("UPS", "UPSs", number);
499 break;
500 case UP_DEVICE_KIND_MONITOR:
501 /* TRANSLATORS: a monitor is a device to measure voltage and current */
502 text = ngettext ("Monitor", "Monitors", number);
503 break;
504 case UP_DEVICE_KIND_MOUSE:
505 /* TRANSLATORS: wireless mice with internal batteries */
506 text = ngettext ("Mouse", "Mice", number);
507 break;
508 case UP_DEVICE_KIND_KEYBOARD:
509 /* TRANSLATORS: wireless keyboard with internal battery */
510 text = ngettext ("Keyboard", "Keyboards", number);
511 break;
512 case UP_DEVICE_KIND_PDA:
513 /* TRANSLATORS: portable device */
514 text = ngettext ("PDA", "PDAs", number);
515 break;
516 case UP_DEVICE_KIND_PHONE:
517 /* TRANSLATORS: cell phone (mobile...) */
518 text = ngettext ("Cell phone", "Cell phones", number);
519 break;
520 case UP_DEVICE_KIND_MEDIA_PLAYER:
521 /* TRANSLATORS: media player, mp3 etc */
522 text = ngettext ("Media player", "Media players", number);
523 break;
524 case UP_DEVICE_KIND_TABLET:
525 /* TRANSLATORS: tablet device */
526 text = ngettext ("Tablet", "Tablets", number);
527 break;
528 case UP_DEVICE_KIND_COMPUTER:
529 /* TRANSLATORS: tablet device */
530 text = ngettext ("Computer", "Computers", number);
531 break;
532 case UP_DEVICE_KIND_GAMING_INPUT:
533 /* TRANSLATORS: wireless gamepad, joystick etc */
534 text = ngettext ("Gaming input", "Gaming inputs", number);
535 break;
536 default:
537 g_warning ("enum unrecognised: %u", kind);
538 text = up_device_kind_to_string (kind);
539 }
540 return text;
541}

637const gchar *
638gpm_device_state_to_localised_string (UpDeviceState state)
639{
640 const gchar *state_string = NULL;
641
642 switch (state) {
643 case UP_DEVICE_STATE_CHARGING:
644 /* TRANSLATORS: battery state */
645 state_string = _("Charging");
646 break;
647 case UP_DEVICE_STATE_DISCHARGING:
648 /* TRANSLATORS: battery state */
649 state_string = _("Discharging");
650 break;
651 case UP_DEVICE_STATE_EMPTY:
652 /* TRANSLATORS: battery state */
653 state_string = _("Empty");
654 break;
655 case UP_DEVICE_STATE_FULLY_CHARGED:
656 /* TRANSLATORS: battery state */
657 state_string = _("Charged");
658 break;
659 case UP_DEVICE_STATE_PENDING_CHARGE:
660 /* TRANSLATORS: battery state */
661 state_string = _("Waiting to charge");
662 break;
663 case UP_DEVICE_STATE_PENDING_DISCHARGE:
664 /* TRANSLATORS: battery state */
665 state_string = _("Waiting to discharge");
666 break;
667 case UP_DEVICE_STATE_UNKNOWN:
668 /* TRANSLATORS: battery state */
669 state_string = _("Unknown state");
