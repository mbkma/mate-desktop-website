50void wncklet_display_help(GtkWidget* widget, const char* doc_id, const char* link_id, const char* icon_name)
51{
52 GError* error = NULL;
53 char* uri;
54
55 if (link_id)
56 uri = g_strdup_printf("help:%s/%s", doc_id, link_id);
57 else
58 uri = g_strdup_printf("help:%s", doc_id);
59
60 gtk_show_uri_on_window (NULL, uri, gtk_get_current_event_time (), &error);
61 g_free(uri);
62
63 if (error && g_error_matches(error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
64 {
65 g_error_free(error);
66 }
67 else if (error)
68 {
69 GtkWidget* parent;
70 GtkWidget* dialog;
71 char* primary;
72
73 if (GTK_IS_WINDOW(widget))
74 parent = widget;
75 else
76 parent = NULL;
77
78 primary = g_markup_printf_escaped(_("Could not display help document '%s'"), doc_id);
79 dialog = gtk_message_dialog_new(parent ? GTK_WINDOW(parent) : NULL, GTK_DIALOG_MODAL|GTK_DIALOG_DESTROY_WITH_PARENT, GTK_MESSAGE_ERROR, GTK_BUTTONS_CLOSE, "%s", primary);
80
81 gtk_message_dialog_format_secondary_text(GTK_MESSAGE_DIALOG(dialog), "%s", error->message);
82
83 g_error_free(error);
84 g_free(primary);
85
86 g_signal_connect(dialog, "response", G_CALLBACK(gtk_widget_destroy), NULL);
87
88 gtk_window_set_icon_name(GTK_WINDOW(dialog), icon_name);
89 gtk_window_set_screen(GTK_WINDOW(dialog), gtk_widget_get_screen(widget));
90
91 if (parent == NULL)
92 {
93 /* we have no parent window */
94 gtk_window_set_skip_taskbar_hint(GTK_WINDOW(dialog), FALSE);
95 gtk_window_set_title(GTK_WINDOW(dialog), _("Error displaying help document"));
96 }
97
98 gtk_widget_show(dialog);
99 }
100}

104{
105 g_return_val_if_fail (WNCK_IS_HANDLE (handle), NULL);

119void wncklet_connect_while_alive(gpointer object, const char* signal, GCallback func, gpointer func_data, gpointer alive_object)
120{
121 GClosure* closure;
122
123 closure = g_cclosure_new(func, func_data, NULL);
124 g_object_watch_closure(G_OBJECT(alive_object), closure);
125 g_signal_connect_closure_by_id(object, g_signal_lookup(signal, G_OBJECT_TYPE(object)), 0, closure, FALSE);
126}

128static gboolean wncklet_factory(MatePanelApplet* applet, const char* iid, gpointer data)
129{
130 gboolean retval = FALSE;
131
132 if (!strcmp(iid, "WindowMenuApplet"))

134#ifdef HAVE_X11
135 else if (!strcmp(iid, "WorkspaceSwitcherApplet") || !strcmp(iid, "PagerApplet"))
136 retval = workspace_switcher_applet_fill(applet);
137#endif
138 else if (!strcmp(iid, "WindowListApplet") || !strcmp(iid, "TasklistApplet"))
139 retval = window_list_applet_fill(applet);
140 else if (!strcmp(iid, "ShowDesktopApplet"))
141 retval = show_desktop_applet_fill(applet);
142
143 return retval;
144}
