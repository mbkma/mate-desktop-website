66gboolean
67gs_auth_verify_user (const char *username,
68 const char *display,
69 GSAuthMessageFunc func,
70 gpointer data,
71 GError **error)
72{
73 int res;
74 char *password;
75
76 /* ask for the password for user */
77 if (func != NULL) {
78 func (GS_AUTH_MESSAGE_PROMPT_ECHO_OFF,
79 "Password: ",
80 &password,
81 data);
82 }
83
84 if (password == NULL) {
85 return FALSE;
86 }
87
88 /* authenticate */
89 res = auth_userokay((char *)username, NULL, "auth-mate-screensaver", password);
90
91 return res;
92}
