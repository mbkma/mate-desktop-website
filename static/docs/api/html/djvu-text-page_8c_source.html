36static gboolean
37djvu_text_page_selection_process (DjvuTextPage *page,
38 miniexp_t p,
39 int delimit)
40{
41 if (page->text || p == page->start) {
42 char *token_text = (char *) miniexp_to_str (miniexp_nth (5, p));
43 if (page->text) {
44 char *new_text =
45 g_strjoin ((delimit & 2) ? "\n" :
46 (delimit & 1) ? " " : NULL,
47 page->text, token_text,
48 NULL);
49 g_free (page->text);
50 page->text = new_text;
51 } else
52 page->text = g_strdup (token_text);
53 if (p == page->end)
54 return FALSE;
55 }
56 return TRUE;
57}

71static gboolean
72djvu_text_page_selection (DjvuTextPage *page,
73 miniexp_t p,
74 int delimit)
75{
76 g_return_val_if_fail (miniexp_consp (p) && miniexp_symbolp
77 (miniexp_car (p)), FALSE);
78
79 if (miniexp_car (p) != page->char_symbol)
80 delimit |= miniexp_car (p) == page->word_symbol ? 1 : 2;
81
82 miniexp_t deeper = miniexp_cddr (miniexp_cdddr (p));
83 while (deeper != miniexp_nil) {
84 miniexp_t str = miniexp_car (deeper);
85 if (miniexp_stringp (str)) {
86 if (!djvu_text_page_selection_process
87 (page, p, delimit))
88 return FALSE;
89 } else {
90 if (!djvu_text_page_selection
91 (page, str, delimit))
92 return FALSE;
93 }
94 delimit = 0;
95 deeper = miniexp_cdr (deeper);
96 }
97 return TRUE;
98}

100static void
101djvu_text_page_limits_process (DjvuTextPage *page,
102 miniexp_t p,
103 EvRectangle *rect)
104{
105 EvRectangle current;
106
107 current.x1 = miniexp_to_int (miniexp_nth (1, p));
108 current.y1 = miniexp_to_int (miniexp_nth (2, p));
109 current.x2 = miniexp_to_int (miniexp_nth (3, p));
110 current.y2 = miniexp_to_int (miniexp_nth (4, p));
111 if (current.x2 >= rect->x1 && current.y1 <= rect->y2 &&
112 current.x1 <= rect->x2 && current.y2 >= rect->y1) {
113 if (page->start == miniexp_nil)
114 page->start = p;
115 page->end = p;
116 }
117}

119static void
120djvu_text_page_limits (DjvuTextPage *page,
121 miniexp_t p,
122 EvRectangle *rect)
123{
124 g_return_if_fail (miniexp_consp (p) &&
125 miniexp_symbolp (miniexp_car (p)));
126
127 miniexp_t deeper = miniexp_cddr (miniexp_cdddr (p));
128 while (deeper != miniexp_nil) {
129 miniexp_t str = miniexp_car (deeper);
130 if (miniexp_stringp (str))
131 djvu_text_page_limits_process (page, p, rect);
132 else
133 djvu_text_page_limits (page, str, rect);
134
135 deeper = miniexp_cdr (deeper);
136 }

140djvu_text_page_copy (DjvuTextPage *page,

142{
143 char* text;
144
145 page->start = miniexp_nil;

147 djvu_text_page_limits (page, page->text_structure, rectangle);
148 djvu_text_page_selection (page, page->text_structure, 0);

167static miniexp_t
168djvu_text_page_position (DjvuTextPage *page,
169 int position)
170{
171 GArray *links = page->links;
172 int low = 0;
173 int hi = links->len - 1;
174 int mid = 0;
175
176 g_return_val_if_fail (hi >= 0, miniexp_nil);
177
178 /* Shamelessly copied from GNU classpath */
179 while (low <= hi) {
180 mid = (low + hi) >> 1;
181 DjvuTextLink *link =
182 &g_array_index (links, DjvuTextLink, mid);
183 if (link->position == position)
184 break;
185 else if (link->position > position)
186 hi = --mid;
187 else
188 low = mid + 1;
189 }
190
191 return g_array_index (page->links, DjvuTextLink, mid).pair;
192}

316static void
317djvu_text_page_append_text (DjvuTextPage *page,
318 miniexp_t p,
319 gboolean case_sensitive,
320 gboolean delimit)
321{
322 char *token_text;
323
324 g_return_if_fail (miniexp_consp (p) &&
325 miniexp_symbolp (miniexp_car (p)));
326
327 delimit |= page->char_symbol != miniexp_car (p);
328
329 miniexp_t deeper = miniexp_cddr (miniexp_cdddr (p));
330 while (deeper != miniexp_nil) {
331 miniexp_t data = miniexp_car (deeper);
332 if (miniexp_stringp (data)) {
333 DjvuTextLink link;
334 link.position = page->text == NULL ? 0 :
335 strlen (page->text);
336 link.pair = p;
337 g_array_append_val (page->links, link);
338
339 token_text = (char *) miniexp_to_str (data);
340 if (!case_sensitive)
341 token_text = g_utf8_casefold (token_text, -1);
342 if (page->text == NULL)
343 page->text = g_strdup (token_text);
344 else {
345 char *new_text =
346 g_strjoin (delimit ? " " : NULL,
347 page->text, token_text,
348 NULL);
349 g_free (page->text);
350 page->text = new_text;
351 }
352 if (!case_sensitive)
353 g_free (token_text);
354 } else
355 djvu_text_page_append_text (page, data,
356 case_sensitive, delimit);
357 delimit = FALSE;
358 deeper = miniexp_cdr (deeper);
359 }
360}

371void
372djvu_text_page_search (DjvuTextPage *page,
373 const char *text,
374 gboolean case_sensitive)
375{
376 char *haystack = page->text;
377 char *search_text;
378 int search_len;
379 EvRectangle *result;
380 if (page->links->len == 0)
381 return;
382
383 search_len = strlen (text);
384
385 if (case_sensitive)
386 search_text = g_strdup (text);
387 else
388 search_text = g_utf8_casefold (text, search_len);
389
390 while ((haystack = strstr (haystack, search_text)) != NULL) {
391 int start_p = haystack - page->text;
392 miniexp_t start = djvu_text_page_position (page, start_p);
393 int end_p = start_p + search_len - 1;
394 miniexp_t end = djvu_text_page_position (page, end_p);
395 result = djvu_text_page_box (page, start, end);
396 g_assert (result);
397 page->results = g_list_prepend (page->results, result);
398 haystack = haystack + search_len;
399 }
400 page->results = g_list_reverse (page->results);
401
402 g_free (search_text);
403}
