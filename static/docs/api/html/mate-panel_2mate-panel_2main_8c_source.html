108 #if defined(HAVE_X11) && defined(HAVE_WAYLAND)

123

129 /* set the default layout */

131 {
132 GSettings *settings;

134 g_settings_set_string (settings, PANEL_DEFAULT_LAYOUT, layout);
135 g_object_unref (settings);

137 /* exit, except if reset argument is given */

139 return 0;

143 if (reset == TRUE)

147 }
148

160 gtk_main ();

185
186 panel_global_config_load ();

188 panel_profile_load ();
189

191 g_slist_foreach (panels,
192 (GFunc)panel_widget_add_forbidden,
193 NULL);
194
195#ifdef HAVE_X11
196 if (GDK_IS_X11_DISPLAY (display)) {
197 xstuff_init ();
198 }
199#endif
200
201 /* Flush to make sure our struts are seen by everyone starting
202 * immediate after (eg, the caja desktop). */
203 gdk_display_flush (display);
204
205 /* Do this at the end, to be sure that we're really ready when
206 * connecting to the session manager */
207 panel_session_init ();
208
209 /*Load a css file from a GResource so the drag handle image can be loaded*/
210 screen = gdk_screen_get_default ();
211 css = gtk_css_provider_new ();
212 provider = GTK_STYLE_PROVIDER (css);
213 resource = "/org/mate/panel/theme/mate-panel.css";
214 priority = GTK_STYLE_PROVIDER_PRIORITY_FALLBACK;
215
216 g_signal_connect (provider, "parsing-error", G_CALLBACK (parsing_error_cb), NULL);
217
218 gtk_css_provider_load_from_resource (css, resource);
219 gtk_style_context_add_provider_for_screen (screen, provider, priority);
220
221 g_object_unref (provider);
222
223 gtk_main ();
224
225 panel_lockdown_finalize ();
226
227 panel_cleanup_do ();
228
229 return 0;
230}
