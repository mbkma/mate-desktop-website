44static gint
45northwestcmp (gconstpointer a, gconstpointer b)
46{
47 MetaWindow *aw = (gpointer) a;
48 MetaWindow *bw = (gpointer) b;
49 double from_origin_a;
50 double from_origin_b;
51 int ax, ay, bx, by;
52
53 /* we're interested in the frame position for cascading,
54 * not meta_window_get_position()
55 */
56 if (aw->frame)
57 {
58 ax = aw->frame->rect.x;
59 ay = aw->frame->rect.y;
60 }
61 else
62 {
63 ax = aw->rect.x;
64 ay = aw->rect.y;
65 }
66
67 if (bw->frame)
68 {
69 bx = bw->frame->rect.x;
70 by = bw->frame->rect.y;
71 }
72 else
73 {
74 bx = bw->rect.x;
75 by = bw->rect.y;
76 }
77
78 /* probably there's a fast good-enough-guess we could use here. */
79 from_origin_a = sqrt ((double) (ax * ax + ay * ay));
80 from_origin_b = sqrt ((double) (bx * bx + by * by));
81
82 if (from_origin_a < from_origin_b)
83 return -1;
84 else if (from_origin_a > from_origin_b)
85 return 1;
86 else
87 return 0;
88}

90static gboolean
91place_by_pointer(MetaWindow *window,
92 MetaFrameBorders *borders,
93 MetaPlacementMode placement_mode,
94 int *new_x,
95 int *new_y)
96{
97 int window_width, window_height;
98 Window root_return, child_return;
99 int root_x_return, root_y_return;
100 int win_x_return, win_y_return;
101 unsigned int mask_return;
102
103 XQueryPointer (window->display->xdisplay,
104 window->screen->xroot,
105 &root_return,
106 &child_return,
107 &root_x_return,
108 &root_y_return,
109 &win_x_return,
110 &win_y_return,
111 &mask_return);
112
113 window_width = window->frame ? window->frame->rect.width : window->rect.width;
114 window_height = window->frame ? window->frame->rect.height : window->rect.height;
115
116 if (borders) {
117 *new_x = root_x_return + borders->visible.left - window_width / 2;
118 *new_y = root_y_return + borders->visible.top - window_height / 2;
119 }
120 else {
121 *new_x = root_x_return - window_width / 2;
122 *new_y = root_y_return - window_height / 2;
123 }
124
125 if (placement_mode == META_PLACEMENT_MODE_MANUAL)
126 window->move_after_placement = TRUE;
127
128 return TRUE;
129}

131static void
132find_next_cascade (MetaWindow *window,
133 MetaFrameBorders *borders,
134 /* visible windows on relevant workspaces */
135 GList *windows,
136 int x,
137 int y,
138 int *new_x,
139 int *new_y)
140{
141 GList *tmp;
142 GList *sorted;
143 int cascade_x, cascade_y;
144 int x_threshold, y_threshold;
145 int window_width, window_height;
146 int cascade_stage;
147 MetaRectangle work_area;
148 const MetaXineramaScreenInfo* current;
149
150 sorted = g_list_copy (windows);
151 sorted = g_list_sort (sorted, northwestcmp);
152
153 /* This is a "fuzzy" cascade algorithm.
154 * For each window in the list, we find where we'd cascade a
155 * new window after it. If a window is already nearly at that
156 * position, we move on.
157 */
158
159 /* arbitrary-ish threshold, honors user attempts to
160 * manually cascade.
161 */
162#define CASCADE_FUZZ 15
163 if (borders)
164 {
165 x_threshold = MAX (borders->visible.left, CASCADE_FUZZ);
166 y_threshold = MAX (borders->visible.top, CASCADE_FUZZ);
167 }
168 else
169 {
170 x_threshold = CASCADE_FUZZ;
171 y_threshold = CASCADE_FUZZ;
172 }
173
174 /* Find furthest-SE origin of all workspaces.
175 * cascade_x, cascade_y are the target position
176 * of NW corner of window frame.
177 */
178
179 current = meta_screen_get_current_xinerama (window->screen);
180 meta_window_get_work_area_for_xinerama (window, current->number, &work_area);
181
182 cascade_x = MAX (0, work_area.x);
183 cascade_y = MAX (0, work_area.y);
184
185 /* Find first cascade position that's not used. */
186
187 window_width = window->frame ? window->frame->rect.width : window->rect.width;
188 window_height = window->frame ? window->frame->rect.height : window->rect.height;
189
190 cascade_stage = 0;
191 tmp = sorted;
192 while (tmp != NULL)
193 {
194 MetaWindow *w;
195 int wx, wy;
196
197 w = tmp->data;
198
199 /* we want frame position, not window position */
200 if (w->frame)
201 {
202 wx = w->frame->rect.x;
203 wy = w->frame->rect.y;
204 }
205 else
206 {
207 wx = w->rect.x;
208 wy = w->rect.y;
209 }
210
211 if (ABS (wx - cascade_x) < x_threshold &&
212 ABS (wy - cascade_y) < y_threshold)
213 {
214 /* This window is "in the way", move to next cascade
215 * point. The new window frame should go at the origin
216 * of the client window we're stacking above.
217 */
218 meta_window_get_position (w, &wx, &wy);
219 cascade_x = wx;
220 cascade_y = wy;
221
222 /* If we go off the screen, start over with a new cascade */
223 if (((cascade_x + window_width) >
224 (work_area.x + work_area.width)) ||
225 ((cascade_y + window_height) >
226 (work_area.y + work_area.height)))
227 {
228 cascade_x = MAX (0, work_area.x);
229 cascade_y = MAX (0, work_area.y);
230
231#define CASCADE_INTERVAL 50 /* space between top-left corners of cascades */
232 cascade_stage += 1;
233 cascade_x += CASCADE_INTERVAL * cascade_stage;
234
235 /* start over with a new cascade translated to the right, unless
236 * we are out of space
237 */
238 if ((cascade_x + window_width) <
239 (work_area.x + work_area.width))
240 {
241 tmp = sorted;
242 continue;
243 }
244 else
245 {
246 /* All out of space, this cascade_x won't work */
247 cascade_x = MAX (0, work_area.x);
248 break;
249 }
250 }
251 }
252 else
253 {
254 /* Keep searching for a further-down-the-diagonal window. */
255 }
256
257 tmp = tmp->next;
258 }
259
260 /* cascade_x and cascade_y will match the last window in the list
261 * that was "in the way" (in the approximate cascade diagonal)
262 */
263
264 g_list_free (sorted);
265
266 /* Convert coords to position of window, not position of frame. */
267 if (borders == NULL)
268 {
269 *new_x = cascade_x;
270 *new_y = cascade_y;
271 }
272 else
273 {
274 *new_x = cascade_x + borders->visible.left;
275 *new_y = cascade_y + borders->visible.top;
276 }
277}

279static void
280find_most_freespace (MetaWindow *window,
281 MetaFrameBorders *borders,
282 /* visible windows on relevant workspaces */
283 MetaWindow *focus_window,
284 int x,
285 int y,
286 int *new_x,
287 int *new_y)
288{
289 MetaWindowDirection side;
290 int max_area;
291 int max_width, max_height, left, right, top, bottom;
292 int left_space, right_space, top_space, bottom_space;
293 int frame_size_left, frame_size_top;
294 MetaRectangle work_area;
295 MetaRectangle avoid;
296 MetaRectangle outer;
297
298 frame_size_left = borders ? borders->visible.left : 0;
299 frame_size_top = borders ? borders->visible.top : 0;
300
301 meta_window_get_work_area_current_xinerama (focus_window, &work_area);
302 meta_window_get_outer_rect (focus_window, &avoid);
303 meta_window_get_outer_rect (window, &outer);
304
305 /* Find the areas of choosing the various sides of the focus window */
306 max_width = MIN (avoid.width, outer.width);
307 max_height = MIN (avoid.height, outer.height);
308 left_space = avoid.x - work_area.x;
309 right_space = work_area.width - (avoid.x + avoid.width - work_area.x);
310 top_space = avoid.y - work_area.y;
311 bottom_space = work_area.height - (avoid.y + avoid.height - work_area.y);
312 left = MIN (left_space, outer.width);
313 right = MIN (right_space, outer.width);
314 top = MIN (top_space, outer.height);
315 bottom = MIN (bottom_space, outer.height);
316
317 /* Find out which side of the focus_window can show the most of the window */
318 side = META_LEFT;
319 max_area = left*max_height;
320 if (right*max_height > max_area)
321 {
322 side = META_RIGHT;
323 max_area = right*max_height;
324 }
325 if (top*max_width > max_area)
326 {
327 side = META_TOP;
328 max_area = top*max_width;
329 }
330 if (bottom*max_width > max_area)
331 {
332 side = META_BOTTOM;
333 max_area = bottom*max_width;
334 }
335
336 /* Give up if there's no where to put it (i.e. focus window is maximized) */
337 if (max_area == 0)
338 return;
339
340 /* Place the window on the relevant side; if the whole window fits,
341 * make it adjacent to the focus window; if not, make sure the
342 * window doesn't go off the edge of the screen.
343 */
344 switch (side)
345 {
346 case META_LEFT:
347 *new_y = avoid.y + frame_size_top;
348 if (left_space > outer.width)
349 *new_x = avoid.x - outer.width + frame_size_left;
350 else
351 *new_x = work_area.x + frame_size_left;
352 break;
353 case META_RIGHT:
354 *new_y = avoid.y + frame_size_top;
355 if (right_space > outer.width)
356 *new_x = avoid.x + avoid.width + frame_size_left;
357 else
358 *new_x = work_area.x + work_area.width - outer.width + frame_size_left;
359 break;
360 case META_TOP:
361 *new_x = avoid.x + frame_size_left;
362 if (top_space > outer.height)
363 *new_y = avoid.y - outer.height + frame_size_top;
364 else
365 *new_y = work_area.y + frame_size_top;
366 break;
367 case META_BOTTOM:
368 *new_x = avoid.x + frame_size_left;
369 if (bottom_space > outer.height)
370 *new_y = avoid.y + avoid.height + frame_size_top;
371 else
372 *new_y = work_area.y + work_area.height - outer.height + frame_size_top;
373 break;
374 }
375}

377static void
378avoid_being_obscured_as_second_modal_dialog (MetaWindow *window,
379 MetaFrameBorders *borders,
380 int *x,
381 int *y)
382{
383 /* We can't center this dialog if it was denied focus and it
384 * overlaps with the focus window and this dialog is modal and this
385 * dialog is in the same app as the focus window (*phew*...please
386 * don't make me say that ten times fast). See bug 307875 comment 11
387 * and 12 for details, but basically it means this is probably a
388 * second modal dialog for some app while the focus window is the
389 * first modal dialog. We should probably make them simultaneously
390 * visible in general, but it becomes mandatory to do so due to
391 * buggy apps (e.g. those using gtk+ *sigh*) because in those cases
392 * this second modal dialog also happens to be modal to the first
393 * dialog in addition to the main window, while it has only let us
394 * know about the modal-to-the-main-window part.
395 */
396
397 MetaWindow *focus_window;
398 MetaRectangle overlap;
399
400 focus_window = window->display->focus_window;
401
402 if (window->denied_focus_and_not_transient &&
403 window->wm_state_modal && /* FIXME: Maybe do this for all transients? */
404 meta_window_same_application (window, focus_window) &&
405 meta_rectangle_intersect (&window->rect,
406 &focus_window->rect,
407 &overlap))
408 {
409 find_most_freespace (window, borders, focus_window, *x, *y, x, y);
410 meta_topic (META_DEBUG_PLACEMENT,
411 "Dialog window %s was denied focus but may be modal "
412 "to the focus window; had to move it to avoid the "
413 "focus window\n",
414 window->desc);
415 }
416}

418static gboolean
419rectangle_overlaps_some_window (MetaRectangle *rect,
420 GList *windows)
421{
422 GList *tmp;
423 MetaRectangle dest;
424
425 tmp = windows;
426 while (tmp != NULL)
427 {
428 MetaWindow *other = tmp->data;
429 MetaRectangle other_rect;
430
431 switch (other->type)
432 {
433 case META_WINDOW_DOCK:
434 case META_WINDOW_SPLASHSCREEN:
435 case META_WINDOW_DESKTOP:
436 case META_WINDOW_DIALOG:
437 case META_WINDOW_MODAL_DIALOG:
438 break;
439
440 case META_WINDOW_NORMAL:
441 case META_WINDOW_UTILITY:
442 case META_WINDOW_TOOLBAR:
443 case META_WINDOW_MENU:
444 meta_window_get_outer_rect (other, &other_rect);
445
446 if (meta_rectangle_intersect (rect, &other_rect, &dest))
447 return TRUE;
448 break;
449 }
450
451 tmp = tmp->next;
452 }
453
454 return FALSE;
455}

457static gint
458leftmost_cmp (gconstpointer a, gconstpointer b)
459{
460 MetaWindow *aw = (gpointer) a;
461 MetaWindow *bw = (gpointer) b;
462 int ax, bx;
463
464 /* we're interested in the frame position for cascading,
465 * not meta_window_get_position()
466 */
467 if (aw->frame)
468 ax = aw->frame->rect.x;
469 else
470 ax = aw->rect.x;
471
472 if (bw->frame)
473 bx = bw->frame->rect.x;
474 else
475 bx = bw->rect.x;
476
477 if (ax < bx)
478 return -1;
479 else if (ax > bx)
480 return 1;
481 else
482 return 0;
483}

487{

513static void
514center_tile_rect_in_area (MetaRectangle *rect,
515 MetaRectangle *work_area)
516{
517 int fluff;
518
519 /* The point here is to tile a window such that "extra"
520 * space is equal on either side (i.e. so a full screen
521 * of windows tiled this way would center the windows
522 * as a group)
523 */
524
525 fluff = (work_area->width % (rect->width+1)) / 2;
526 rect->x = work_area->x + fluff;
527 fluff = (work_area->height % (rect->height+1)) / 3;
528 rect->y = work_area->y + fluff;
529}

531void
532center_rect_in_area (MetaRectangle *rect,
533 MetaRectangle *work_area)
534{
535 rect->x = work_area->x + ((work_area->width - rect->width) / 2);
536 rect->y = work_area->y + ((work_area->height - rect->height) / 2);
537}
538
539/* Find the leftmost, then topmost, empty area on the workspace
540 * that can contain the new window.
541 *
542 * Cool feature to have: if we can't fit the current window size,
543 * try shrinking the window (within geometry constraints). But
544 * beware windows such as Emacs with no sane minimum size, we
545 * don't want to create a 1x1 Emacs.
546 */
547static gboolean
548find_first_fit (MetaWindow *window,
549 MetaFrameBorders *borders,
550 /* visible windows on relevant workspaces */
551 GList *windows,
552 int xinerama,
553 int x,
554 int y,
555 int *new_x,
556 int *new_y)
557{
558 /* This algorithm is limited - it just brute-force tries
559 * to fit the window in a small number of locations that are aligned
560 * with existing windows. It tries to place the window on
561 * the bottom of each existing window, and then to the right
562 * of each existing window, aligned with the left/top of the
563 * existing window in each of those cases.
564 */
565 int retval;
566 GList *below_sorted;
567 GList *right_sorted;
568 GList *tmp;
569 MetaRectangle rect;
570 MetaRectangle work_area;
571
572 retval = FALSE;
573
574 /* Below each window */
575 below_sorted = g_list_copy (windows);
576 below_sorted = g_list_sort (below_sorted, leftmost_cmp);
577 below_sorted = g_list_sort (below_sorted, topmost_cmp);
578
579 /* To the right of each window */
580 right_sorted = g_list_copy (windows);
581 right_sorted = g_list_sort (right_sorted, topmost_cmp);
582 right_sorted = g_list_sort (right_sorted, leftmost_cmp);
583
584 rect.width = window->rect.width;
585 rect.height = window->rect.height;
586
587 if (borders)
588 {
589 rect.width += borders->visible.left + borders->visible.right;
590 rect.height += borders->visible.top + borders->visible.bottom;
591 }
592
593#ifdef WITH_VERBOSE_MODE
594 {
595 char xinerama_location_string[RECT_LENGTH];
596 meta_rectangle_to_string (&window->screen->xinerama_infos[xinerama].rect,
597 xinerama_location_string);
598 meta_topic (META_DEBUG_XINERAMA,
599 "Natural xinerama is %s\n",
600 xinerama_location_string);
601 }
602#endif
603
604 meta_window_get_work_area_for_xinerama (window, xinerama, &work_area);
605
606 if (meta_prefs_get_center_new_windows ())
607 center_rect_in_area (&rect, &work_area);
608 else
609 center_tile_rect_in_area (&rect, &work_area);
610
611 if (meta_rectangle_contains_rect (&work_area, &rect) &&
612 (meta_prefs_get_center_new_windows () ||
613 !rectangle_overlaps_some_window (&rect, windows)))
614 {
615 *new_x = rect.x;
616 *new_y = rect.y;
617 if (borders)
618 {
619 *new_x += borders->visible.left;
620 *new_y += borders->visible.top;
621 }
622
623 retval = TRUE;
624
625 goto out;
626 }
627
628 /* try below each window */
629 tmp = below_sorted;
630 while (tmp != NULL)
631 {
632 MetaWindow *w = tmp->data;
633 MetaRectangle outer_rect;
634
635 meta_window_get_outer_rect (w, &outer_rect);
636
637 rect.x = outer_rect.x;
638 rect.y = outer_rect.y + outer_rect.height;
639
640 if (meta_rectangle_contains_rect (&work_area, &rect) &&
641 !rectangle_overlaps_some_window (&rect, below_sorted))
642 {
643 *new_x = rect.x;
644 *new_y = rect.y;
645 if (borders)
646 {
647 *new_x += borders->visible.left;
648 *new_y += borders->visible.top;
649 }
650
651 retval = TRUE;
652
653 goto out;
654 }
655
656 tmp = tmp->next;
657 }
658
659 /* try to the right of each window */
660 tmp = right_sorted;
661 while (tmp != NULL)
662 {
663 MetaWindow *w = tmp->data;
664 MetaRectangle outer_rect;
665
666 meta_window_get_outer_rect (w, &outer_rect);
667
668 rect.x = outer_rect.x + outer_rect.width;
669 rect.y = outer_rect.y;
670
671 if (meta_rectangle_contains_rect (&work_area, &rect) &&
672 !rectangle_overlaps_some_window (&rect, right_sorted))
673 {
674 *new_x = rect.x;
675 *new_y = rect.y;
676 if (borders)
677 {
678 *new_x += borders->visible.left;
679 *new_y += borders->visible.top;
680 }
681
682 retval = TRUE;
683
684 goto out;
685 }
686
687 tmp = tmp->next;
688 }
689
690 out:
691
692 g_list_free (below_sorted);
693 g_list_free (right_sorted);
694 return retval;
695}

697void
698meta_window_place (MetaWindow *window,
699 MetaFrameBorders *borders,
700 int x,
701 int y,
702 int *new_x,
703 int *new_y)
704{
705 GList *windows;
706 const MetaXineramaScreenInfo *xi;
707 MetaPlacementMode placement_mode;
708
709 /* frame member variables should NEVER be used in here, only
710 * MetaFrameBorders. But remember borders == NULL
711 * for undecorated windows. Also, this function should
712 * NEVER have side effects other than computing the
713 * placement coordinates.
714 */
715
716 meta_topic (META_DEBUG_PLACEMENT, "Placing window %s\n", window->desc);
717
718 windows = NULL;
719
720 switch (window->type)
721 {
722 /* Run placement algorithm on these. */
723 case META_WINDOW_NORMAL:
724 case META_WINDOW_DIALOG:
725 case META_WINDOW_MODAL_DIALOG:
726 case META_WINDOW_SPLASHSCREEN:
727 break;
728
729 /* Assume the app knows best how to place these, no placement
730 * algorithm ever (other than "leave them as-is")
731 */
732 case META_WINDOW_DESKTOP:
733 case META_WINDOW_DOCK:
734 case META_WINDOW_TOOLBAR:
735 case META_WINDOW_MENU:
736 case META_WINDOW_UTILITY:
737 goto done_no_constraints;
738 }
739
740 if (meta_prefs_get_disable_workarounds ())
741 {
742 switch (window->type)
743 {
744 /* Only accept USPosition on normal windows because the app is full
745 * of shit claiming the user set -geometry for a dialog or dock
746 */
747 case META_WINDOW_NORMAL:
748 if (window->size_hints.flags & USPosition)
749 {
750 /* don't constrain with placement algorithm */
751 meta_topic (META_DEBUG_PLACEMENT,
752 "Honoring USPosition for %s instead of using placement algorithm\n", window->desc);
753
754 goto done;
755 }
756 break;
757
758 /* Ignore even USPosition on dialogs, splashscreen */
759 case META_WINDOW_DIALOG:
760 case META_WINDOW_MODAL_DIALOG:
761 case META_WINDOW_SPLASHSCREEN:
762 break;
763
764 /* Assume the app knows best how to place these. */
765 case META_WINDOW_DESKTOP:
766 case META_WINDOW_DOCK:
767 case META_WINDOW_TOOLBAR:
768 case META_WINDOW_MENU:
769 case META_WINDOW_UTILITY:
770 if (window->size_hints.flags & PPosition)
771 {
772 meta_topic (META_DEBUG_PLACEMENT,
773 "Not placing non-normal non-dialog window with PPosition set\n");
774 goto done_no_constraints;
775 }
776 break;
777 }
778 }
779 else
780 {
781 /* workarounds enabled */
782
783 if ((window->size_hints.flags & PPosition) ||
784 (window->size_hints.flags & USPosition))
785 {
786 meta_topic (META_DEBUG_PLACEMENT,
787 "Not placing window with PPosition or USPosition set\n");

886 }

914 * a hack here). Assume undecorated windows probably don't intend to
915 * be maximized.
916 */

918 !window->fullscreen)
919 {
920 MetaRectangle workarea;
921 MetaRectangle outer;
922

924 xi->number,
925 &workarea);
926 meta_window_get_outer_rect (window, &outer);

929 * auto-maximize the directions independently. See #419810.
930 */
931 if (outer.width >= workarea.width && outer.height >= workarea.height)
932 {
933 window->maximize_horizontally_after_placement = TRUE;
934 window->maximize_vertically_after_placement = TRUE;
935 }
936 }
937

939 * fully overlapping window (e.g. starting multiple terminals)
940 * */
941 if (!meta_prefs_get_center_new_windows() && (x == xi->rect.x && y == xi->rect.y))
942 find_next_cascade (window, borders, windows, x, y, &x, &y);
943
944 done_check_denied_focus:
945 /* If the window is being denied focus and isn't a transient of the
946 * focus window, we do NOT want it to overlap with the focus window
947 * if at all possible. This is guaranteed to only be called if the
948 * focus_window is non-NULL, and we try to avoid that window.
949 */
950 if (window->denied_focus_and_not_transient)
951 {
952 gboolean found_fit;
953 MetaWindow *focus_window;
954 MetaRectangle overlap;
955
956 focus_window = window->display->focus_window;
957 g_assert (focus_window != NULL);
958
959 /* No need to do anything if the window doesn't overlap at all */
960 found_fit = !meta_rectangle_intersect (&window->rect,
961 &focus_window->rect,
962 &overlap);
963
964 /* Try to do a first fit again, this time only taking into account the
965 * focus window.
966 */
967 if (!meta_prefs_get_center_new_windows() && !found_fit)
968 {
969 GList *focus_window_list;
970 focus_window_list = g_list_prepend (NULL, focus_window);
971
972 /* Reset x and y ("origin" placement algorithm) */
973 x = xi->rect.x;
974 y = xi->rect.y;
975
976 found_fit = find_first_fit (window, borders, focus_window_list,
977 xi->number,
978 x, y, &x, &y);
979 g_list_free (focus_window_list);
980 }
981
982 /* If that still didn't work, just place it where we can see as much
983 * as possible.
984 */
985 if (!found_fit)
986 find_most_freespace (window, borders, focus_window, x, y, &x, &y);
987 }
988
989 done:
990 g_list_free (windows);
991
992 done_no_constraints:
993
994 *new_x = x;
995 *new_y = y;
996}
