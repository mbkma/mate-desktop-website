116static void
117gdict_source_set_property (GObject *object,
118 guint prop_id,
119 const GValue *value,
120 GParamSpec *pspec)
121{
122 GdictSource *source = GDICT_SOURCE (object);
123
124 switch (prop_id)
125 {
126 case PROP_NAME:
127 gdict_source_set_name (source, g_value_get_string (value));
128 break;
129 case PROP_DESCRIPTION:
130 gdict_source_set_description (source, g_value_get_string (value));
131 break;
132 case PROP_TRANSPORT:
133 gdict_source_set_transport (source, g_value_get_enum (value), NULL);
134 break;
135 case PROP_DATABASE:
136 gdict_source_set_database (source, g_value_get_string (value));
137 break;
138 case PROP_STRATEGY:
139 gdict_source_set_strategy (source, g_value_get_string (value));
140 break;
141 default:
142 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
143 break;
144 }
145}

147static void
148gdict_source_get_property (GObject *object,
149 guint prop_id,
150 GValue *value,
151 GParamSpec *pspec)
152{
153 GdictSource *source = GDICT_SOURCE (object);
154 GdictSourcePrivate *priv = source->priv;
155
156 switch (prop_id)
157 {
158 case PROP_FILENAME:
159 g_value_set_string (value, priv->filename);
160 break;
161 case PROP_NAME:
162 g_value_set_string (value, priv->name);
163 break;
164 case PROP_DESCRIPTION:
165 g_value_set_string (value, priv->description);
166 break;
167 case PROP_DATABASE:
168 g_value_set_string (value, priv->database);
169 break;
170 case PROP_STRATEGY:
171 g_value_set_string (value, priv->strategy);
172 break;
173 case PROP_TRANSPORT:
174 g_value_set_enum (value, priv->transport);
175 break;
176 case PROP_CONTEXT:
177 g_value_set_object (value, gdict_source_peek_context (source));
178 break;
179 default:
180 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
181 break;
182 }
183}

185static void
186gdict_source_finalize (GObject *object)
187{
188 GdictSourcePrivate *priv = gdict_source_get_instance_private (GDICT_SOURCE (object));
189
190 g_free (priv->filename);
191
192 if (priv->keyfile)
193 g_key_file_free (priv->keyfile);
194
195 g_free (priv->name);
196 g_free (priv->description);
197

200
201 if (priv->context)
202 g_object_unref (priv->context);
203
204 G_OBJECT_CLASS (gdict_source_parent_class)->finalize (object);
205}

351static GdictSourceTransport
352gdict_source_resolve_transport (const gchar *transport)
353{
354 if (!transport)
355 return GDICT_SOURCE_TRANSPORT_INVALID;
356
357 if (strcmp (transport, "dictd") == 0)
358 return GDICT_SOURCE_TRANSPORT_DICTD;
359 else
360 return GDICT_SOURCE_TRANSPORT_INVALID;
361
362 g_assert_not_reached ();
363}

365static GdictContext *
366gdict_source_create_context (GdictSource *source,
367 GdictSourceTransport transport,
368 GError **error)
369{
370 GdictSourcePrivate *priv;
371 GdictContext *context;
372
373 g_assert (GDICT_IS_SOURCE (source));
374
375 priv = source->priv;
376
377 switch (transport)
378 {
379 case GDICT_SOURCE_TRANSPORT_DICTD:
380 {
381 gchar *hostname;
382 gint port;
383
384 hostname = g_key_file_get_string (priv->keyfile,
385 SOURCE_GROUP,
386 SOURCE_KEY_HOSTNAME,
387 NULL);
388
389 port = g_key_file_get_integer (priv->keyfile,
390 SOURCE_GROUP,
391 SOURCE_KEY_PORT,
392 NULL);
393 if (!port)
394 port = -1;
395
396 context = gdict_client_context_new (hostname, port);
397
398 g_free (hostname);
399 }
400 break;
401 default:
402 g_set_error (error, GDICT_SOURCE_ERROR,
403 GDICT_SOURCE_ERROR_PARSE,
404 _("Invalid transport type '%d'"),
405 transport);
406 return NULL;
407 }
408
409 g_assert (context != NULL);
410
411 if (priv->transport != transport)
412 priv->transport = transport;
413
414 return context;
415}

417static gboolean
418gdict_source_parse (GdictSource *source,
419 GError **error)
420{
421 GdictSourcePrivate *priv;
422 GError *parse_error;
423 gchar *transport;
424 GdictSourceTransport t;
425
426 priv = source->priv;
427
428 if (!g_key_file_has_group (priv->keyfile, SOURCE_GROUP))
429 {
430 g_set_error (error, GDICT_SOURCE_ERROR,
431 GDICT_SOURCE_ERROR_PARSE,
432 _("No '%s' group found inside the dictionary source definition"),
433 SOURCE_GROUP);
434
435 return FALSE;
436 }
437
438 /* fetch the name for the dictionary source definition */
439 parse_error = NULL;
440 priv->name = g_key_file_get_string (priv->keyfile,
441 SOURCE_GROUP,
442 SOURCE_KEY_NAME,
443 &parse_error);
444 if (parse_error)
445 {
446 g_set_error (error, GDICT_SOURCE_ERROR,
447 GDICT_SOURCE_ERROR_PARSE,
448 _("Unable to get the '%s' key inside the dictionary "
449 "source definition: %s"),
450 SOURCE_KEY_NAME,
451 parse_error->message);
452 g_error_free (parse_error);
453
454 g_key_file_free (priv->keyfile);
455 priv->keyfile = NULL;
456
457 return FALSE;
458 }
459
460 /* if present, fetch the localized description */
461 if (g_key_file_has_key (priv->keyfile, SOURCE_GROUP, SOURCE_KEY_DESCRIPTION, NULL))
462 {
463 priv->description = g_key_file_get_locale_string (priv->keyfile,
464 SOURCE_GROUP,
465 SOURCE_KEY_DESCRIPTION,
466 NULL,
467 &parse_error);
468 if (parse_error)
469 {
470 g_set_error (error, GDICT_SOURCE_ERROR,
471 GDICT_SOURCE_ERROR_PARSE,
472 _("Unable to get the '%s' key inside the dictionary "
473 "source definition: %s"),
474 SOURCE_KEY_DESCRIPTION,
475 parse_error->message);
476
477 g_error_free (parse_error);
478 g_key_file_free (priv->keyfile);
479 priv->keyfile = NULL;
480 g_free (priv->name);
481
482 return FALSE;
483 }
484 }
485
486 if (g_key_file_has_key (priv->keyfile, SOURCE_GROUP, SOURCE_KEY_DATABASE, NULL))
487 {
488 priv->database = g_key_file_get_string (priv->keyfile,
489 SOURCE_GROUP,
490 SOURCE_KEY_DATABASE,
491 &parse_error);
492 if (parse_error)
493 {
494 g_set_error (error, GDICT_SOURCE_ERROR,
495 GDICT_SOURCE_ERROR_PARSE,
496 _("Unable to get the '%s' key inside the dictionary "
497 "source definition: %s"),
498 SOURCE_KEY_DATABASE,
499 parse_error->message);
500
501 g_error_free (parse_error);
502 g_key_file_free (priv->keyfile);
503 priv->keyfile = NULL;
504 g_free (priv->name);
505 g_free (priv->description);
506
507 return FALSE;
508 }
509 }
510
511 if (g_key_file_has_key (priv->keyfile, SOURCE_GROUP, SOURCE_KEY_STRATEGY, NULL))
512 {
513 priv->strategy = g_key_file_get_string (priv->keyfile,
514 SOURCE_GROUP,
515 SOURCE_KEY_STRATEGY,
516 &parse_error);
517 if (parse_error)
518 {
519 g_set_error (error, GDICT_SOURCE_ERROR,
520 GDICT_SOURCE_ERROR_PARSE,
521 _("Unable to get the '%s' key inside the dictionary "
522 "source definition: %s"),
523 SOURCE_KEY_STRATEGY,
524 parse_error->message);
525
526 g_error_free (parse_error);
527 g_key_file_free (priv->keyfile);
528 priv->keyfile = NULL;
529
530 g_free (priv->name);
531 g_free (priv->description);
532 g_free (priv->database);
533
534 return FALSE;
535 }
536 }
537
538 transport = g_key_file_get_string (priv->keyfile,
539 SOURCE_GROUP,
540 SOURCE_KEY_TRANSPORT,
541 &parse_error);
542 if (parse_error)
543 {
544 g_set_error (error, GDICT_SOURCE_ERROR,
545 GDICT_SOURCE_ERROR_PARSE,
546 _("Unable to get the '%s' key inside the dictionary "
547 "source definition file: %s"),
548 SOURCE_KEY_TRANSPORT,
549 parse_error->message);
550
551 g_error_free (parse_error);
552 g_key_file_free (priv->keyfile);
553 priv->keyfile = NULL;
554 g_free (priv->name);
555 g_free (priv->description);
556 g_free (priv->database);
557 g_free (priv->strategy);
558
559 return FALSE;
560 }
561
562 t = gdict_source_resolve_transport (transport);
563 g_free (transport);
564
565 priv->context = gdict_source_create_context (source, t, &parse_error);
566 if (parse_error)
567 {
568 g_propagate_error (error, parse_error);
569
570 g_key_file_free (priv->keyfile);
571 priv->keyfile = NULL;
572
573 g_free (priv->name);
574 g_free (priv->description);
575 g_free (priv->database);
576 g_free (priv->strategy);
577
578 return FALSE;
579 }
580
581 return TRUE;
582}

597gboolean
598gdict_source_load_from_file (GdictSource *source,
599 const gchar *filename,
600 GError **error)
601{
602 GdictSourcePrivate *priv;
603 GError *read_error;
604 GError *parse_error;
605
606 g_return_val_if_fail (GDICT_IS_SOURCE (source), FALSE);
607 g_return_val_if_fail (filename != NULL, FALSE);
608
609 priv = source->priv;
610
611 if (!priv->keyfile)
612 priv->keyfile = g_key_file_new ();
613
614 read_error = NULL;
615 g_key_file_load_from_file (priv->keyfile,
616 filename,
617 G_KEY_FILE_KEEP_TRANSLATIONS,
618 &read_error);
619 if (read_error)
620 {
621 g_propagate_error (error, read_error);
622
623 return FALSE;
624 }
625
626 parse_error = NULL;
627 gdict_source_parse (source, &parse_error);
628 if (parse_error)
629 {
630 g_propagate_error (error, parse_error);
631
632 return FALSE;
633 }
634
635 g_assert (priv->context != NULL);
636
637 priv->filename = g_strdup (filename);
638
639 return TRUE;
640}

716gchar *
717gdict_source_to_data (GdictSource *source,
718 gsize *length,
719 GError **error)
720{
721 GdictSourcePrivate *priv;
722 gchar *retval = NULL;
723
724 g_return_val_if_fail (GDICT_IS_SOURCE (source), NULL);
725
726 priv = source->priv;
727
728 if (!priv->name)
729 {
730 g_set_error (error, GDICT_SOURCE_ERROR,
731 GDICT_SOURCE_ERROR_INVALID_NAME,
732 _("Dictionary source does not have name"));
733
734 return NULL;
735 }
736
737 if (!IS_VALID_TRANSPORT (priv->transport))
738 {
739 g_set_error (error, GDICT_SOURCE_ERROR,
740 GDICT_SOURCE_ERROR_INVALID_TRANSPORT,
741 _("Dictionary source '%s' has invalid transport '%s'"),
742 priv->name,
743 valid_transports[priv->transport]);
744
745 return NULL;
746 }
747
748 if (priv->keyfile)
749 {
750 GError *write_error = NULL;
751
752 retval = g_key_file_to_data (priv->keyfile,
753 length,
754 &write_error);
755 if (write_error)
756 g_propagate_error (error, write_error);
757 }
758
759 return retval;
760}

1006void
1007gdict_source_set_transportv (GdictSource *source,
1008 GdictSourceTransport transport,
1009 const gchar *first_transport_property,
1010 va_list var_args)
1011{
1012 GdictSourcePrivate *priv;
1013
1014 g_return_if_fail (GDICT_IS_SOURCE (source));
1015 g_return_if_fail (IS_VALID_TRANSPORT (transport));
1016
1017 priv = source->priv;
1018
1019 priv->transport = transport;
1020
1021 if (priv->context)
1022 g_object_unref (priv->context);
1023
1024 switch (priv->transport)
1025 {
1026 case GDICT_SOURCE_TRANSPORT_DICTD:
1027 priv->context = gdict_client_context_new (NULL, -1);
1028 g_assert (GDICT_IS_CLIENT_CONTEXT (priv->context));
1029
1030 g_object_set_valist (G_OBJECT (priv->context),
1031 first_transport_property,
1032 var_args);
1033
1034 break;
1035 case GDICT_SOURCE_TRANSPORT_INVALID:
1036 default:
1037 g_assert_not_reached ();
1038 break;
