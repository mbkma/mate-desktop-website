131static void
132set_combo_box_from_int(GtkWidget *combo, int value)
133{
134 GtkTreeModel *model;
135 GtkTreeIter iter;
136 gboolean valid;
137
138 model = gtk_combo_box_get_model(GTK_COMBO_BOX(combo));
139 valid = gtk_tree_model_get_iter_first(model, &iter);
140
141 while (valid) {
142 gint v;
143
144 gtk_tree_model_get(model, &iter, 1, &v, -1);
145 if (v == value)
146 break;
147 valid = gtk_tree_model_iter_next(model, &iter);
148 }
149 if (!valid)
150 valid = gtk_tree_model_get_iter_first(model, &iter);
151
152 gtk_combo_box_set_active_iter(GTK_COMBO_BOX(combo), &iter);
153}

248 gtk_cell_layout_pack_start(GTK_CELL_LAYOUT(widget), renderer, TRUE);
249 gtk_cell_layout_add_attribute(GTK_CELL_LAYOUT(widget), renderer, "text", 0);

320static void
321math_preferences_set_property(GObject *object,
322 guint prop_id,
323 const GValue *value,
324 GParamSpec *pspec)
325{
326 MathPreferencesDialog *self;
327
328 self = MATH_PREFERENCES(object);
329
330 switch (prop_id) {
331 case PROP_EQUATION:
332 self->priv->equation = g_value_get_object(value);
333 create_gui(self);
334 break;
335 default:
336 G_OBJECT_WARN_INVALID_PROPERTY_ID(object, prop_id, pspec);
337 break;
338 }
339}

341static void
342math_preferences_get_property(GObject *object,
343 guint prop_id,
344 GValue *value,
345 GParamSpec *pspec)
346{
347 MathPreferencesDialog *self;
348
349 self = MATH_PREFERENCES(object);
350

353 g_value_set_object(value, self->priv->equation);
354 break;
355 default:
356 G_OBJECT_WARN_INVALID_PROPERTY_ID(object, prop_id, pspec);
357 break;
358 }
359}
