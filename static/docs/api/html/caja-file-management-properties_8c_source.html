201static void
202preferences_show_help (GtkWindow *parent,
203 char const *helpfile,
204 char const *sect_id)
205{
206 GError *error = NULL;
207 char *help_string;
208
209 g_assert (helpfile != NULL);
210 g_assert (sect_id != NULL);
211
212 help_string = g_strdup_printf ("help:%s/%s", helpfile, sect_id);
213
214 gtk_show_uri_on_window (parent,
215 help_string, gtk_get_current_event_time (),
216 &error);
217 g_free (help_string);
218
219 if (error)
220 {
221 GtkWidget *dialog;
222
223 dialog = gtk_message_dialog_new (GTK_WINDOW (parent),
224 GTK_DIALOG_DESTROY_WITH_PARENT,
225 GTK_MESSAGE_ERROR,
226 GTK_BUTTONS_OK,
227 _("There was an error displaying help: \n%s"),
228 error->message);
229
230 g_signal_connect (dialog, "response",
231 G_CALLBACK (gtk_widget_destroy),
232 NULL);
233 gtk_window_set_resizable (GTK_WINDOW (dialog), FALSE);
234 gtk_widget_show (dialog);
235 g_error_free (error);
236 }
237}

309static void
310create_icon_caption_combo_box_items (GtkComboBoxText *combo_box,
311 GList *columns)
312{
313 GList *l;
314 GPtrArray *column_names;
315
316 column_names = g_ptr_array_new ();
317
318 /* Translators: this is referred to captions under icons. */
319 gtk_combo_box_text_append_text (combo_box, _("None"));
320 g_ptr_array_add (column_names, g_strdup ("none"));
321
322 for (l = columns; l != NULL; l = l->next)
323 {
324 CajaColumn *column;
325 char *name;
326 char *label;
327
328 column = CAJA_COLUMN (l->data);
329
330 g_object_get (G_OBJECT (column),
331 "name", &name, "label", &label,
332 NULL);
333
334 /* Don't show name here, it doesn't make sense */
335 if (!strcmp (name, "name"))
336 {
337 g_free (name);
338 g_free (label);
339 continue;
340 }
341
342 gtk_combo_box_text_append_text (combo_box, label);
343 g_ptr_array_add (column_names, name);
344
345 g_free (label);
346 }
347 g_object_set_data_full (G_OBJECT (combo_box), "column_names",
348 column_names,
349 (GDestroyNotify) free_column_names_array);
350}

352static void
353icon_captions_changed_callback (GtkComboBox *combo_box,
354 gpointer user_data)
355{
356 GPtrArray *captions;
357 GtkBuilder *builder;
358 int i;
359
360 builder = GTK_BUILDER (user_data);
361 captions = g_ptr_array_new ();
362
363 for (i = 0; icon_captions_components[i] != NULL; i++)
364 {
365 GObject *object;
366 GPtrArray *column_names;
367 int active;
368
369 object = gtk_builder_get_object (builder, icon_captions_components[i]);
370 column_names = g_object_get_data (object, "column_names");
371 active = gtk_combo_box_get_active (GTK_COMBO_BOX(object));
372 g_ptr_array_add (captions, g_ptr_array_index (column_names, active));
373 }
374 g_ptr_array_add (captions, NULL);
375
376 g_settings_set_strv (caja_icon_view_preferences,
377 CAJA_PREFERENCES_ICON_VIEW_CAPTIONS,
378 (const char **)captions->pdata);
379 g_ptr_array_free (captions, TRUE);
380}

382static void
383update_caption_combo_box (GtkBuilder *builder,
384 const char *combo_box_name,
385 const char *name)
386{
387 GtkWidget *combo_box;
388 guint i;
389 GPtrArray *column_names;
390
391 combo_box = GTK_WIDGET (gtk_builder_get_object (builder, combo_box_name));
392
393 g_signal_handlers_block_by_func
394 (combo_box,
395 G_CALLBACK (icon_captions_changed_callback),
396 builder);
397
398 column_names = g_object_get_data (G_OBJECT (combo_box),
399 "column_names");
400
401 for (i = 0; i < column_names->len; ++i)
402 {
403 if (!strcmp (name, g_ptr_array_index (column_names, i)))
404 {
405 gtk_combo_box_set_active (GTK_COMBO_BOX (combo_box), (int) i);
406 break;
407 }
408 }
409
410 g_signal_handlers_unblock_by_func
411 (combo_box,
412 G_CALLBACK (icon_captions_changed_callback),
413 builder);
414}

416static void
417update_icon_captions_from_settings (GtkBuilder *builder)
418{
419 char **captions;
420 int i, j;
421
422 captions = g_settings_get_strv (caja_icon_view_preferences, CAJA_PREFERENCES_ICON_VIEW_CAPTIONS);
423 if (captions == NULL)
424 return;
425
426 for (i = 0, j = 0;
427 icon_captions_components[i] != NULL;
428 i++)
429 {
430 char *data;
431
432 if (captions[j])
433 {
434 data = captions[j];
435 ++j;
436 }
437 else
438 {
439 data = "none";
440 }
441
442 update_caption_combo_box (builder,
443 icon_captions_components[i],
444 data);
445 }
446
447 g_strfreev (captions);
448}

481static void
482create_date_format_menu (GtkBuilder *builder)
483{
484 GtkComboBoxText *combo_box;
485 gchar *date_string;
486 GDateTime *now;
487
488 combo_box = GTK_COMBO_BOX_TEXT
489 (gtk_builder_get_object (builder,
490 CAJA_FILE_MANAGEMENT_PROPERTIES_DATE_FORMAT_WIDGET));
491
492 now = g_date_time_new_now_local ();
493
494 date_string = g_date_time_format (now, "%c");
495 gtk_combo_box_text_append_text (GTK_COMBO_BOX_TEXT (combo_box), date_string);
496 g_free (date_string);
497
498 date_string = g_date_time_format (now, "%Y-%m-%d %H:%M:%S");
499 gtk_combo_box_text_append_text (GTK_COMBO_BOX_TEXT (combo_box), date_string);
500 g_free (date_string);
501
502 date_string = g_date_time_format (now, _("today at %-I:%M:%S %p"));
503 gtk_combo_box_text_append_text (GTK_COMBO_BOX_TEXT (combo_box), date_string);
504 g_free (date_string);
505
506 g_date_time_unref (now);
507}
