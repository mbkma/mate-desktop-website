193static gboolean
194gpm_manager_play_loop_start (GpmManager *manager, GpmManagerSound action, gboolean force, guint timeout)
195{
196 const gchar *id = NULL;
197 const gchar *desc = NULL;
198 gboolean ret;
199 gint retval;
200 ca_context *context;
201
202 ret = g_settings_get_boolean (manager->priv->settings, GPM_SETTINGS_ENABLE_SOUND);
203 if (!ret && !force) {
204 g_debug ("ignoring sound due to policy");
205 return FALSE;
206 }
207
208 if (timeout == 0) {
209 g_warning ("received invalid timeout");
210 return FALSE;
211 }
212
213 /* if a sound loop is already running, stop the existing loop */
214 if (manager->priv->critical_alert_timeout_id != 0) {
215 g_warning ("was instructed to play a sound loop with one already playing");
216 gpm_manager_play_loop_stop (manager);
217 }
218
219 if (action == GPM_MANAGER_SOUND_BATTERY_LOW) {
220 id = "battery-low";
221 /* TRANSLATORS: this is the sound description */
222 desc = _("Battery is very low");
223 }
224
225 /* no match */
226 if (id == NULL) {
227 g_warning ("no sound match for %u", action);
228 return FALSE;
229 }
230
231 ca_proplist_create (&(manager->priv->critical_alert_loop_props));
232 ca_proplist_sets (manager->priv->critical_alert_loop_props,
233 CA_PROP_EVENT_ID, id);
234 ca_proplist_sets (manager->priv->critical_alert_loop_props,
235 CA_PROP_EVENT_DESCRIPTION, desc);
236
237 manager->priv->critical_alert_timeout_id = g_timeout_add_seconds (timeout,
238 (GSourceFunc) gpm_manager_play_loop_timeout_cb,
239 manager);
240
241 g_source_set_name_by_id (manager->priv->critical_alert_timeout_id, "[GpmManager] play-loop");
242
243 /* play the sound, using sounds from the naming spec */
244 context = ca_gtk_context_get_for_screen (gdk_screen_get_default ());
245 retval = ca_context_play (context, 0,
246 CA_PROP_EVENT_ID, id,
247 CA_PROP_EVENT_DESCRIPTION, desc, NULL);
248 if (retval < 0)
249 g_warning ("failed to play %s: %s", id, ca_strerror (retval));
250 return TRUE;
251}

288 id = "battery-caution";
289 /* TRANSLATORS: this is the sound description */
290 desc = _("Battery is low");
291 } else if (action == GPM_MANAGER_SOUND_BATTERY_LOW) {
292 id = "battery-low";
293 /* TRANSLATORS: this is the sound description */

308 id = "suspend-error";

369 gpm_idle_set_timeout_sleep (manager->priv->idle, sleep_computer);

471static gboolean
472gpm_manager_notify (GpmManager *manager, NotifyNotification **notification_class,
473 const gchar *title, const gchar *message,
474 guint timeout, const gchar *icon, NotifyUrgency urgency)
475{
476 gboolean ret;
477 GError *error = NULL;
478 NotifyNotification *notification;
479 GtkWidget *dialog;
480
481 /* close any existing notification of this class */
482 gpm_manager_notify_close (manager, *notification_class);
483
484 /* if the status icon is hidden, don't point at it */
485 if (manager->priv->status_icon != NULL &&
486 gtk_status_icon_is_embedded (manager->priv->status_icon))
487 notification = notify_notification_new (title, message, gtk_status_icon_get_icon_name(manager->priv->status_icon));
488 else
489 notification = notify_notification_new (title, message, icon);
490 notify_notification_set_timeout (notification, timeout);
491 notify_notification_set_urgency (notification, urgency);
492 g_signal_connect (notification, "closed", G_CALLBACK (gpm_manager_notification_closed_cb), notification_class);
493
494 g_debug ("notification %p: %s : %s", notification, title, message);
495
496 /* try to show */
497 ret = notify_notification_show (notification, &error);
498 if (!ret) {
499 g_warning ("failed to show notification: %s", error->message);
500 g_error_free (error);
501
502 /* show modal dialog as libmatenotify failed */
503 dialog = gtk_message_dialog_new_with_markup (NULL, GTK_DIALOG_DESTROY_WITH_PARENT,
504 GTK_MESSAGE_INFO, GTK_BUTTONS_CLOSE,
505 "<span size='larger'><b>%s</b></span>", title);
506 gtk_message_dialog_format_secondary_markup (GTK_MESSAGE_DIALOG (dialog), "%s", message);
507
508 /* wait async for close */
509 gtk_widget_show (dialog);
510 g_signal_connect_swapped (dialog, "response", G_CALLBACK (gtk_widget_destroy), dialog);
511
512 g_object_unref (notification);
513 goto out;
514 }
515
516 /* save this local instance as the class instance */
517 g_object_add_weak_pointer (G_OBJECT (notification), (gpointer) &notification);
518 *notification_class = notification;
519out:
520 return ret;
521}

553static void
554gpm_manager_sleep_failure (GpmManager *manager, gboolean is_suspend, const gchar *detail)
555{
556 gboolean show_sleep_failed;
557 GString *string = NULL;
558 const gchar *title;
559 gchar *uri = NULL;
560 const gchar *icon;
561 GtkWidget *dialog;
562
563 /* only show this if specified in settings */
564 show_sleep_failed = g_settings_get_boolean (manager->priv->settings, GPM_SETTINGS_NOTIFY_SLEEP_FAILED);
565
566 g_debug ("sleep failed");
567 gpm_manager_play (manager, GPM_MANAGER_SOUND_SUSPEND_ERROR, TRUE);
568
569 /* only emit if specified in settings */
570 if (!show_sleep_failed)
571 goto out;
572
573 /* TRANSLATORS: window title: there was a problem putting the machine to sleep */
574 string = g_string_new ("");
575 if (is_suspend) {
576 /* TRANSLATORS: message text */
577 g_string_append (string, _("Computer failed to suspend."));
578 /* TRANSLATORS: title text */
579 title = _("Failed to suspend");
580 icon = GPM_ICON_SUSPEND;
581 } else {
582 /* TRANSLATORS: message text */
583 g_string_append (string, _("Computer failed to hibernate."));
584 /* TRANSLATORS: title text */
585 title = _("Failed to hibernate");
586 icon = GPM_ICON_HIBERNATE;
587 }
588
589 /* TRANSLATORS: message text */
590 g_string_append_printf (string, "\n\n%s %s", _("Failure was reported as:"), detail);
591
592 /* show modal dialog */
593 dialog = gtk_message_dialog_new_with_markup (NULL, GTK_DIALOG_DESTROY_WITH_PARENT,
594 GTK_MESSAGE_INFO, GTK_BUTTONS_CLOSE,
595 "<span size='larger'><b>%s</b></span>", title);
596 gtk_message_dialog_format_secondary_markup (GTK_MESSAGE_DIALOG (dialog), "%s", string->str);
597 gtk_window_set_icon_name (GTK_WINDOW(dialog), icon);
598
599 /* show a button? */
600 uri = g_settings_get_string (manager->priv->settings, GPM_SETTINGS_NOTIFY_SLEEP_FAILED_URI);
601 if (uri != NULL && uri[0] != '\0') {
602 /* TRANSLATORS: button text, visit the suspend help website */
603 gtk_dialog_add_button (GTK_DIALOG (dialog), _("Visit help page"), GTK_RESPONSE_HELP);
604 }
605
606 /* wait async for close */
607 gtk_widget_show (dialog);
608 g_signal_connect (dialog, "response", G_CALLBACK (gpm_manager_sleep_failure_response_cb), manager);
609out:
610 g_free (uri);
611 g_string_free (string, TRUE);
612}

668static gboolean
669gpm_manager_perform_policy (GpmManager *manager, const gchar *policy_key, const gchar *reason)
670{
671 GpmActionPolicy policy;
672
673 /* are we inhibited? */
674 if (gpm_manager_is_inhibit_valid (manager, FALSE, "policy action") == FALSE)
675 return FALSE;
676
677 policy = g_settings_get_enum (manager->priv->settings, policy_key);
678 g_debug ("action: %s set to %u (%s)", policy_key, policy, reason);
679
680 if (policy == GPM_ACTION_POLICY_NOTHING) {
681 g_debug ("doing nothing, reason: %s", reason);
682 } else if (policy == GPM_ACTION_POLICY_SUSPEND) {
683 gpm_manager_action_suspend (manager, reason);
684
685 } else if (policy == GPM_ACTION_POLICY_HIBERNATE) {
686 gpm_manager_action_hibernate (manager, reason);
687
688 } else if (policy == GPM_ACTION_POLICY_BLANK) {
689 gpm_manager_blank_screen (manager, NULL);
690
691 } else if (policy == GPM_ACTION_POLICY_SHUTDOWN) {
692 g_debug ("shutting down, reason: %s", reason);
693 gpm_control_shutdown (manager->priv->control, NULL);
694
695 } else if (policy == GPM_ACTION_POLICY_INTERACTIVE) {
696 GpmSession *session;
697 g_debug ("logout, reason: %s", reason);
698 session = gpm_session_new ();
699 gpm_session_logout (session);
700 g_object_unref (session);
701 } else {
702 g_warning ("unknown action %u", policy);
703 }
704
705 return TRUE;
706}

715static void
716gpm_manager_idle_do_sleep (GpmManager *manager)
717{
718 gboolean ret;
719 GError *error = NULL;
720 GpmActionPolicy policy;
721
722 if (!manager->priv->on_battery)
723 policy = g_settings_get_enum (manager->priv->settings, GPM_SETTINGS_ACTION_SLEEP_TYPE_AC);
724 else
725 policy = g_settings_get_enum (manager->priv->settings, GPM_SETTINGS_ACTION_SLEEP_TYPE_BATT);
726
727 if (policy == GPM_ACTION_POLICY_NOTHING) {
728 g_debug ("doing nothing as system idle action");
729
730 } else if (policy == GPM_ACTION_POLICY_SUSPEND) {
731 g_debug ("suspending, reason: System idle");
732 ret = gpm_control_suspend (manager->priv->control, &error);
733 if (!ret) {
734 g_warning ("cannot suspend (error: %s), so trying hibernate", error->message);
735 g_error_free (error);
736 error = NULL;
737 ret = gpm_control_hibernate (manager->priv->control, &error);
738 if (!ret) {
739 g_warning ("cannot suspend or hibernate: %s", error->message);
740 g_error_free (error);

947static void
948gpm_manager_client_changed_cb (UpClient *client, GParamSpec *pspec, GpmManager *manager)
949{
950 gboolean event_when_closed;
951 gboolean on_battery;
952 gboolean lid_is_closed;
953
954 /* get the client state */
955 g_object_get (client,
956 "on-battery", &on_battery,
957 "lid-is-closed", &lid_is_closed,
958 NULL);
959 if (on_battery == manager->priv->on_battery) {
960 g_debug ("same state as before, ignoring");
961 return;
962 }
963
964 /* close any discharging notifications */
965 if (!on_battery) {
966 g_debug ("clearing notify due ac being present");
967 gpm_manager_notify_close (manager, manager->priv->notification_warning_low);
968 gpm_manager_notify_close (manager, manager->priv->notification_discharging);

974 gpm_manager_play_loop_stop (manager);

978 manager->priv->on_battery = on_battery;
979
980 /* ConsoleKit/systemd say we are not on active console */
981 if (!LOGIND_RUNNING() && !egg_console_kit_is_active (manager->priv->console)) {
982 g_debug ("ignoring as not on active console");

986 g_debug ("on_battery: %d", on_battery);
987
988 gpm_manager_sync_policy_sleep (manager);

992 /* simulate user input, but only when the lid is open */
993 if (!lid_is_closed)
994 gpm_screensaver_poke (manager->priv->screensaver);

1002 when the laptop is closed and on battery. Fixes #331655 */
1003 event_when_closed = g_settings_get_boolean (manager->priv->settings, GPM_SETTINGS_SLEEP_WHEN_CLOSED);
1004
1005 /* We keep track of the lid state so we can do the
1006 lid close on battery action if the ac adapter is removed when the laptop
1007 is closed. Fixes #331655 */

1010 "The lid has been closed, and the ac adapter "
1011 "removed (and GSettings is okay).");
1012 }

1729static gint32
1730gpm_manager_systemd_inhibit (GDBusProxy *proxy) {
1731 GError *error = NULL;
1732 gint32 r = -1;
1733 gint32 fd = -1;
1734 GVariant *res;
1735 GUnixFDList *fd_list = NULL;
1736
1737 /* Should we define these elsewhere? */
1738 const char* arg_what = "handle-power-key:handle-suspend-key:handle-lid-switch";
1739 const char* arg_who = g_get_user_name ();
1740 const char* arg_why = "Mate power manager handles these events";
1741 const char* arg_mode = "block";
1742
1743 g_debug ("Inhibiting systemd sleep");
1744 proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SYSTEM,
1745 G_DBUS_PROXY_FLAGS_DO_NOT_LOAD_PROPERTIES,
1746 NULL,
1747 "org.freedesktop.login1",
1748 "/org/freedesktop/login1",
1749 "org.freedesktop.login1.Manager",
1750 NULL,
1751 &error );
1752 //append all our arguments
1753 if (proxy == NULL) {
1754 g_error ("Error connecting to dbus - %s", error->message);
1755 g_error_free (error);
1756 return -1;
1757 }
1758 res = g_dbus_proxy_call_with_unix_fd_list_sync (proxy, "Inhibit",
1759 g_variant_new( "(ssss)",
1760 arg_what,
1761 arg_who,
1762 arg_why,
1763 arg_mode
1764 ),
1765 G_DBUS_CALL_FLAGS_NONE,
1766 -1,
1767 NULL,
1768 &fd_list,
1769 NULL,
1770 &error
1771 );
1772 if (error != NULL) {
1773 g_error ("Error in dbus - %s", error->message);
1774 g_error_free (error);
1775 return -EIO;
1776 }
1777 if (res == NULL)
1778 return -EIO;
1779
1780 g_variant_get(res, "(h)", &r);
1781 g_debug ("Inhibiting systemd sleep res = %i", r);
1782
1783 fd = g_unix_fd_list_get (fd_list, r, &error);
1784 if (fd == -1) {
1785 g_debug("Failed to get systemd inhibitor");
1786 return r;
1787 }
1788
1789 g_debug ("System inhibitor fd is %d", fd);
1790 g_object_unref (fd_list);
1791 g_variant_unref (res);
1792 g_debug ("Inhibiting systemd sleep - success");
1793 return r;
1794}
