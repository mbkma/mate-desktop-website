70static void
71show_bogus_bookmark_window (CajaWindow *window,
72 CajaBookmark *bookmark)
73{
74 GtkDialog *dialog;
75 GFile *location;
76 char *uri_for_display;
77 char *prompt;
78 char *detail;
79
80 location = caja_bookmark_get_location (bookmark);
81 uri_for_display = g_file_get_parse_name (location);
82
83 prompt = _("Do you want to remove any bookmarks with the "
84 "non-existing location from your list?");
85 detail = g_strdup_printf (_("The location \"%s\" does not exist."), uri_for_display);
86
87 dialog = eel_show_yes_no_dialog (prompt, detail,
88 _("Bookmark for Nonexistent Location"),
89 "process-stop",
90 GTK_WINDOW (window));
91
92 g_signal_connect (dialog, "response",
93 G_CALLBACK (remove_bookmarks_for_uri_if_yes), window);
94 g_object_set_data_full (G_OBJECT (dialog), "uri", g_file_get_uri (location), g_free);
95
96 gtk_dialog_set_default_response (dialog, GTK_RESPONSE_NO);
97
98 g_object_unref (location);
99 g_free (uri_for_display);
100 g_free (detail);
101}

210static void
211update_bookmarks (CajaWindow *window)
212{
213 guint bookmark_count;
214 guint index;
215 GtkUIManager *ui_manager;
216 CajaBookmarkList *bookmarks;
217 CajaBookmark *bookmark = NULL;
218
219 g_assert (CAJA_IS_WINDOW (window));
220 g_assert (window->details->bookmarks_merge_id == 0);
221 g_assert (window->details->bookmarks_action_group == NULL);
222
223 if (window->details->bookmark_list == NULL)
224 {
225 window->details->bookmark_list = caja_bookmark_list_new ();
226 }
227
228 bookmarks = window->details->bookmark_list;
229
230 ui_manager = caja_window_get_ui_manager (CAJA_WINDOW (window));
231
232 window->details->bookmarks_merge_id = gtk_ui_manager_new_merge_id (ui_manager);
233 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
234 window->details->bookmarks_action_group = gtk_action_group_new ("BookmarksGroup");
235 G_GNUC_END_IGNORE_DEPRECATIONS;
236 g_signal_connect (window->details->bookmarks_action_group, "connect-proxy",
237 G_CALLBACK (connect_proxy_cb), NULL);
238
239 gtk_ui_manager_insert_action_group (ui_manager,
240 window->details->bookmarks_action_group,
241 -1);
242 g_object_unref (window->details->bookmarks_action_group);
243
244 /* append new set of bookmarks */
245 bookmark_count = caja_bookmark_list_length (bookmarks);
246 for (index = 0; index < bookmark_count; ++index)
247 {
248 bookmark = caja_bookmark_list_item_at (bookmarks, index);
249
250 if (caja_bookmark_uri_known_not_to_exist (bookmark))
251 {
252 continue;
253 }
254
255 caja_menus_append_bookmark_to_menu
256 (CAJA_WINDOW (window),
257 bookmark,
258 CAJA_WINDOW_GET_CLASS (window)->bookmarks_placeholder,
259 "dynamic",
260 index,
261 window->details->bookmarks_action_group,
262 window->details->bookmarks_merge_id,
263 G_CALLBACK (refresh_bookmarks_menu),
264 show_bogus_bookmark_window);
265 }
266}
