165GList *
166caja_drag_build_selection_list (GtkSelectionData *data)
167{
168 GList *result;
169 const guchar *p, *oldp;
170 int size;
171
172 result = NULL;
173 oldp = gtk_selection_data_get_data (data);
174 size = gtk_selection_data_get_length (data);
175
176 while (size > 0)
177 {
178 CajaDragSelectionItem *item;
179 guint len;
180
181 /* The list is in the form:
182
183 name\rx:y:width:height\r\n
184
185 The geometry information after the first \r is optional. */
186
187 /* 1: Decode name. */
188
189 p = memchr (oldp, '\r', size);
190 if (p == NULL)
191 {
192 break;
193 }
194
195 item = caja_drag_selection_item_new ();
196
197 len = p - oldp;
198
199 item->uri = g_malloc (len + 1);
200 memcpy (item->uri, oldp, len);
201 item->uri[len] = 0;
202
203 p++;
204 if (*p == '\n' || *p == '\0')
205 {
206 result = g_list_prepend (result, item);
207 if (*p != '\n')
208 {
209 g_warning ("Invalid x-special/mate-icon-list data received: "
210 "missing newline character.");
211 break;
212 }
213 else
214 {
215 oldp = p + 1;
216 continue;
217 }
218 }
219
220 size -= p - oldp;
221 oldp = p;
222
223 /* 2: Decode geometry information. */
224
225 item->got_icon_position = sscanf (p, "%d:%d:%d:%d%*s",
226 &item->icon_x, &item->icon_y,
227 &item->icon_width, &item->icon_height) == 4;
228 if (!item->got_icon_position)
229 {
230 g_warning ("Invalid x-special/mate-icon-list data received: "
231 "invalid icon position specification.");
232 }
233
234 result = g_list_prepend (result, item);
235
236 p = memchr (p, '\r', size);
237 if (p == NULL || p[1] != '\n')
238 {
239 g_warning ("Invalid x-special/mate-icon-list data received: "
240 "missing newline character.");
241 if (p == NULL)
242 {
243 break;
244 }
245 }
246 else
247 {
248 p += 2;
249 }
250
251 size -= p - oldp;
252 oldp = p;
253 }
254
255 return g_list_reverse (result);
256}

258static gboolean
259caja_drag_file_local_internal (const char *target_uri_string,
260 const char *first_source_uri)
261{
262 /* check if the first item on the list has target_uri_string as a parent
263 * FIXME:
264 * we should really test each item but that would be slow for large selections
265 * and currently dropped items can only be from the same container
266 */
267 GFile *target, *item, *parent;
268 gboolean result;
269
270 result = FALSE;
271
272 target = g_file_new_for_uri (target_uri_string);
273
274 /* get the parent URI of the first item in the selection */
275 item = g_file_new_for_uri (first_source_uri);
276 parent = g_file_get_parent (item);
277 g_object_unref (item);
278
279 if (parent != NULL)
280 {
281 result = g_file_equal (parent, target);
282 g_object_unref (parent);
283 }
284
285 g_object_unref (target);
286
287 return result;
288}

348GdkDragAction
349caja_drag_default_drop_action_for_netscape_url (GdkDragContext *context)
350{
351 /* Mozilla defaults to copy, however by default caja creates a link
352 * if available.
353 *
354 * Enforced action X:
355 * ((actions & X) != 0) && (suggested_action == X)
356 * - GDK_ACTION_LINK enforced by Alt
357 * - GDK_ACTION_MOVE enforced by Shift
358 * - GDK_ACTION_COPY enforced by Ctrl
359 */
360 GdkDragAction suggested_action = gdk_drag_context_get_suggested_action (context);
361 GdkDragAction actions = gdk_drag_context_get_actions (context);
362 if ((actions & GDK_ACTION_LINK) == GDK_ACTION_LINK)
363 {
364 return GDK_ACTION_LINK;
365 }
366 else if (suggested_action == GDK_ACTION_MOVE)
367 {
368 /* Don't support move */
369 return GDK_ACTION_COPY;
370 }
371
372 return suggested_action;
373}

375static gboolean
376check_same_fs (const char *target_uri,
377 CajaFile *target_file,
378 const char *dropped_uri,
379 CajaFile *dropped_file)
380{
381 gboolean result;
382
383 result = FALSE;
384
385 char *target_fs = NULL, *dropped_fs = NULL;
386
387 if (target_file != NULL)
388 {
389 target_fs = caja_file_get_filesystem_id (target_file);
390 }
391 if (target_fs == NULL)
392 {
393 target_fs = caja_get_filesystem_id_by_uri (target_uri, TRUE);
394 }
395
396 if (dropped_file != NULL && !caja_file_is_symbolic_link (dropped_file))
397 {
398 dropped_fs = caja_file_get_filesystem_id (dropped_file);
399 }
400 if (dropped_fs == NULL)
401 {
402 dropped_fs = caja_get_filesystem_id_by_uri (dropped_uri, FALSE);
403 }
404
405 if (target_fs != NULL && dropped_fs != NULL)
406 {
407 result = (strcmp (target_fs, dropped_fs) == 0);
408 }
409
410 g_free (target_fs);
411 g_free (dropped_fs);
412
413 return result;
414}

435void
436caja_drag_default_drop_action_for_icons (GdkDragContext *context,
437 const char *target_uri_string, const GList *items,
438 int *action)
439{
440 gboolean same_fs;
441 gboolean target_is_source_parent;
442 gboolean source_deletable;
443 const char *dropped_uri;
444 GFile *target, *dropped, *dropped_directory;
445 GdkDragAction actions;
446 CajaFile *dropped_file, *target_file;
447
448 if (target_uri_string == NULL)
449 {
450 *action = 0;
451 return;
452 }
453
454 actions = gdk_drag_context_get_actions (context) & (GDK_ACTION_MOVE | GDK_ACTION_COPY);
455 if (actions == 0)
456 {
457 /* We can't use copy or move, just go with the suggested action. */
458 *action = gdk_drag_context_get_suggested_action (context);
459 return;
460 }
461
462 if (gdk_drag_context_get_suggested_action (context) == GDK_ACTION_ASK)
463 {
464 /* Don't override ask */
465 *action = gdk_drag_context_get_suggested_action (context);
466 return;
467 }
468
469 dropped_uri = ((CajaDragSelectionItem *)items->data)->uri;
470 dropped_file = caja_file_get_existing_by_uri (dropped_uri);
471 target_file = caja_file_get_existing_by_uri (target_uri_string);
472
473 /*
474 * Check for trash URI. We do a find_directory for any Trash directory.
475 * Passing 0 permissions as mate-vfs would override the permissions
476 * passed with 700 while creating .Trash directory
477 */
478 if (eel_uri_is_trash (target_uri_string))
479 {
480 /* Only move to Trash */
481 if (actions & GDK_ACTION_MOVE)
482 {
483 *action = GDK_ACTION_MOVE;
484 }
485
486 caja_file_unref (dropped_file);
487 caja_file_unref (target_file);
488 return;
489
490 }
491 else if (dropped_file != NULL && caja_file_is_launcher (dropped_file))
492 {
493 if (actions & GDK_ACTION_MOVE)
494 {
495 *action = GDK_ACTION_MOVE;
496 }
497 caja_file_unref (dropped_file);
498 caja_file_unref (target_file);
499 return;
500 }
501 else if (eel_uri_is_desktop (target_uri_string))
502 {
503 target = caja_get_desktop_location ();
504
505 caja_file_unref (target_file);
506 target_file = caja_file_get (target);
507
508 if (eel_uri_is_desktop (dropped_uri))
509 {
510 /* Only move to Desktop icons */
511 if (actions & GDK_ACTION_MOVE)
512 {
513 *action = GDK_ACTION_MOVE;
514 }
515
516 g_object_unref (target);
517 caja_file_unref (dropped_file);
518 caja_file_unref (target_file);
519 return;
520 }
521 }
522 else if (target_file != NULL && caja_file_is_archive (target_file))
523 {
524 *action = GDK_ACTION_COPY;
525
526 caja_file_unref (dropped_file);
527 caja_file_unref (target_file);
528 return;
529 }
530 else
531 {
532 target = g_file_new_for_uri (target_uri_string);
533 }
534
535 same_fs = check_same_fs (target_uri_string, target_file, dropped_uri, dropped_file);
536
537 caja_file_unref (dropped_file);
538 caja_file_unref (target_file);
539
540 /* Compare the first dropped uri with the target uri for same fs match. */
541 dropped = g_file_new_for_uri (dropped_uri);
542 dropped_directory = g_file_get_parent (dropped);
543 target_is_source_parent = FALSE;
544 if (dropped_directory != NULL)
545 {
546 /* If the dropped file is already in the same directory but
547 is in another filesystem we still want to move, not copy
548 as this is then just a move of a mountpoint to another
549 position in the dir */
550 target_is_source_parent = g_file_equal (dropped_directory, target);
551 g_object_unref (dropped_directory);
552 }
553 source_deletable = source_is_deletable (dropped);
554
555 if ((same_fs && source_deletable) || target_is_source_parent ||
556 g_file_has_uri_scheme (dropped, "trash"))
557 {
558 if (actions & GDK_ACTION_MOVE)
559 {
560 *action = GDK_ACTION_MOVE;
561 }
562 else
563 {
564 *action = gdk_drag_context_get_suggested_action (context);
565 }
566 }
567 else
568 {
569 if (actions & GDK_ACTION_COPY)
570 {
571 *action = GDK_ACTION_COPY;
572 }
573 else
574 {
575 *action = gdk_drag_context_get_suggested_action (context);
576 }
577 }
578
579 g_object_unref (target);
580 g_object_unref (dropped);
581
582}

617 */
618#ifdef THIS_WAS_REALLY_BROKEN
619static gboolean
620is_path_that_mate_uri_list_extract_filenames_can_parse (const char *path)
621{
622 if (path == NULL || path [0] == '\0')
623 {
624 return FALSE;
625 }
626
627 /* It strips leading and trailing spaces. So it can't handle
628 * file names with leading and trailing spaces.
629 */
630 if (g_ascii_isspace (path [0]))
631 {
632 return FALSE;
633 }
634 if (g_ascii_isspace (path [strlen (path) - 1]))
635 {
636 return FALSE;
637 }
638
639 /* # works as a comment delimiter, and \r and \n are used to
640 * separate the lines, so it can't handle file names with any
641 * of these.
642 */
643 if (strchr (path, '#') != NULL
644 || strchr (path, '\r') != NULL
645 || strchr (path, '\n') != NULL)
646 {
647 return FALSE;
648 }
649
650 return TRUE;
651}

653/* Encode a "text/plain" selection; this is a broken URL -- just
654 * "file:" with a path after it (no escaping or anything). We are
655 * trying to make the old mate_uri_list_extract_filenames function
656 * happy, so this is coded to its idiosyncrasises.
657 */
658static void
659add_one_compatible_uri (const char *uri, int x, int y, int w, int h, gpointer data)
660{
661 GString *result;
662
663 result = (GString *) data;
664
665 /* For URLs that do not have a file: scheme, there's no harm
666 * in passing the real URL. But for URLs that do have a file:
667 * scheme, we have to send a URL that will work with the old
668 * mate-libs function or nothing will be able to understand
669 * it.
670 */
671 if (!eel_istr_has_prefix (uri, "file:"))
672 {
673 g_string_append (result, uri);
674 g_string_append (result, "\r\n");
675 }
676 else
677 {
678 char *local_path;
679
680 local_path = g_filename_from_uri (uri, NULL, NULL);
681
682 /* Check for characters that confuse the old
683 * mate_uri_list_extract_filenames implementation, and just leave
684 * out any paths with those in them.
685 */
686 if (is_path_that_mate_uri_list_extract_filenames_can_parse (local_path))
687 {
688 g_string_append (result, "file:");
689 g_string_append (result, local_path);
690 g_string_append (result, "\r\n");
691 }
692
693 g_free (local_path);
694 }
695}

709/* Common function for drag_data_get_callback calls.
710 * Returns FALSE if it doesn't handle drag data */
711gboolean
712caja_drag_drag_data_get (GtkWidget *widget,
713 GdkDragContext *context,
714 GtkSelectionData *selection_data,
715 guint info,
716 guint32 time,
717 gpointer container_context,
718 CajaDragEachSelectedItemIterator each_selected_item_iterator)
719{
720 GString *result;
721
722 switch (info)
723 {
724 case CAJA_ICON_DND_MATE_ICON_LIST:
725 result = g_string_new (NULL);
726 (* each_selected_item_iterator) (add_one_mate_icon, container_context, result);
727 break;
728
729 case CAJA_ICON_DND_URI_LIST:
730 case CAJA_ICON_DND_TEXT:
731 result = g_string_new (NULL);
732 (* each_selected_item_iterator) (add_one_uri, container_context, result);
733 break;
734
735 default:
736 return FALSE;
737 }
738
739 gtk_selection_data_set (selection_data,
740 gtk_selection_data_get_target (selection_data),
741 8, result->str, result->len);
742 g_string_free (result, TRUE);
743
744 return TRUE;
745}

931void
932caja_drag_autoscroll_calculate_delta (GtkWidget *widget, float *x_scroll_delta, float *y_scroll_delta)
933{
934 GtkAllocation allocation;
935 GdkDisplay *display;
936 GdkSeat *seat;
937 GdkDevice *pointer;
938 int x, y;
939
940 g_assert (GTK_IS_WIDGET (widget));
941
942 display = gtk_widget_get_display (widget);
943 seat = gdk_display_get_default_seat (display);
944 pointer = gdk_seat_get_pointer (seat);
945 gdk_window_get_device_position (gtk_widget_get_window (widget), pointer,
946 &x, &y, NULL);
947
948 /* Find out if we are anywhere close to the tree view edges
949 * to see if we need to autoscroll.
950 */
951 *x_scroll_delta = 0;
952 *y_scroll_delta = 0;
953
954 if (x < AUTO_SCROLL_MARGIN)
955 {
956 *x_scroll_delta = (float)(x - AUTO_SCROLL_MARGIN);
957 }
958
959 gtk_widget_get_allocation (widget, &allocation);
960 if (x > allocation.width - AUTO_SCROLL_MARGIN)
961 {
962 if (*x_scroll_delta != 0)
963 {
964 /* Already trying to scroll because of being too close to
965 * the top edge -- must be the window is really short,
966 * don't autoscroll.
967 */
968 return;
969 }
970 *x_scroll_delta = (float)(x - (allocation.width - AUTO_SCROLL_MARGIN));
971 }
972
973 if (y < AUTO_SCROLL_MARGIN)
974 {
975 *y_scroll_delta = (float)(y - AUTO_SCROLL_MARGIN);
976 }
977
978 if (y > allocation.height - AUTO_SCROLL_MARGIN)
979 {
980 if (*y_scroll_delta != 0)
981 {
982 /* Already trying to scroll because of being too close to
983 * the top edge -- must be the window is really narrow,
984 * don't autoscroll.
985 */
986 return;
987 }
988 *y_scroll_delta = (float)(y - (allocation.height - AUTO_SCROLL_MARGIN));
989 }
990
991 if (*x_scroll_delta == 0 && *y_scroll_delta == 0)
992 {
993 /* no work */
994 return;
995 }
996
997 /* Adjust the scroll delta to the proper acceleration values depending on how far
998 * into the sroll margins we are.
999 * FIXME bugzilla.eazel.com 2486:
1000 * we could use an exponential acceleration factor here for better feel
1001 */
1002 if (*x_scroll_delta != 0)
1003 {
1004 *x_scroll_delta /= AUTO_SCROLL_MARGIN;
1005 *x_scroll_delta *= (MAX_AUTOSCROLL_DELTA - MIN_AUTOSCROLL_DELTA);
1006 *x_scroll_delta += MIN_AUTOSCROLL_DELTA;
1007 }
1008
1009 if (*y_scroll_delta != 0)
1010 {
1011 *y_scroll_delta /= AUTO_SCROLL_MARGIN;
1012 *y_scroll_delta *= (MAX_AUTOSCROLL_DELTA - MIN_AUTOSCROLL_DELTA);
1013 *y_scroll_delta += MIN_AUTOSCROLL_DELTA;
1014 }
1015
1016}
