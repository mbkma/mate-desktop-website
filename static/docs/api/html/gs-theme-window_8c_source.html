85static void
86gs_theme_window_real_realize (GtkWidget *widget)
87{
88 GdkWindow *window;
89 Window remote_xwindow;
90 GtkRequisition requisition;
91 GtkAllocation allocation;
92 const char *preview_xid;
93 int x;
94 int y;
95 int width;
96 int height;
97 int event_mask;
98
99 event_mask = 0;
100 window = NULL;
101 preview_xid = g_getenv ("XSCREENSAVER_WINDOW");
102
103 if (preview_xid != NULL)
104 {
105 char *end;
106
107 remote_xwindow = (Window) strtoul (preview_xid, &end, 0);
108
109 if ((remote_xwindow != 0) && (end != NULL) &&
110 ((*end == ' ') || (*end == '\0')) &&
111 ((remote_xwindow < G_MAXULONG) || (errno != ERANGE)))
112 {
113 window = gdk_x11_window_foreign_new_for_display (gdk_display_get_default (), remote_xwindow);
114 if (window != NULL)
115 {
116 /* This is a kludge; we need to set the same
117 * flags gs-window-x11.c does, to ensure they
118 * don't get unset by gtk_window_map() later.
119 */
120 gtk_window_set_decorated (GTK_WINDOW (widget), FALSE);
121
122 gtk_window_set_skip_taskbar_hint (GTK_WINDOW (widget), TRUE);
123 gtk_window_set_skip_pager_hint (GTK_WINDOW (widget), TRUE);
124
125 gtk_window_set_keep_above (GTK_WINDOW (widget), TRUE);
126
127 gtk_window_fullscreen (GTK_WINDOW (widget));
128
129 event_mask = GDK_EXPOSURE_MASK | GDK_STRUCTURE_MASK;
130 gtk_widget_set_events (widget, gtk_widget_get_events (widget) | event_mask);
131 }
132 }
133 }
134
135 if (window == NULL)
136 {
137 GtkWidgetClass *parent_class;
138
139 parent_class = GTK_WIDGET_CLASS (gs_theme_window_parent_class);
140
141 if (parent_class->realize != NULL)
142 parent_class->realize (widget);
143
144 return;
145 }
146
147 gtk_style_context_set_background (gtk_widget_get_style_context (widget),
148 window);
149 gdk_window_set_decorations (window, (GdkWMDecoration) 0);
150 gdk_window_set_events (window, gdk_window_get_events (window) | event_mask);
151
152 gtk_widget_set_window (widget, window);
153 gdk_window_set_user_data (window, widget);
154 gtk_widget_set_realized (widget, TRUE);
155
156 gdk_window_get_geometry (window, &x, &y, &width, &height);
157
158 if (width < MIN_SIZE || height < MIN_SIZE)
159 {
160 g_critical ("This window is way too small to use");
161 exit (1);
162 }
163
164 gtk_widget_get_preferred_size (widget, &requisition, NULL);
165 allocation.x = x;
166 allocation.y = y;
167 allocation.width = width;
168 allocation.height = height;
169 gtk_widget_size_allocate (widget, &allocation);
170 gtk_window_resize (GTK_WINDOW (widget), width, height);
171}
