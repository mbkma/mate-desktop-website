92static gboolean
93eq_events_pending (MetaEventQueue *eq)
94{
95 return eq->events->length > 0 || XPending (eq->display);
96}

98static void
99eq_queue_events (MetaEventQueue *eq)
100{
101 XEvent xevent;
102
103 while (XPending (eq->display))
104 {
105 XEvent *copy;
106
107 XNextEvent (eq->display, &xevent);
108
109 copy = g_new (XEvent, 1);
110 *copy = xevent;
111
112 g_queue_push_tail (eq->events, copy);
113 }
114}

128static gboolean
129eq_check (GSource *source)
130{
131 MetaEventQueue *eq;
132
133 eq = (MetaEventQueue*) source;
134
135 if (eq->poll_fd.revents & G_IO_IN)
136 return eq_events_pending (eq);
137 else
138 return FALSE;

141static gboolean
142eq_dispatch (GSource *source, GSourceFunc callback, gpointer user_data)
143{
144 MetaEventQueue *eq;
145
146 eq = (MetaEventQueue*) source;
147
148 eq_queue_events (eq);
149
150 if (eq->events->length > 0)
151 {
152 XEvent *event;
153 MetaEventQueueFunc func;
154
155 event = g_queue_pop_head (eq->events);
156 func = (MetaEventQueueFunc) G_CALLBACK (callback);
157
158 (* func) (event, user_data);
159
160 g_free (event);
161 }
162
163 return TRUE;
164}
