35/* the rules:
36 * 1) nothing here may be modified while trash_empty_update_pending.
37 * 2) an idle may only be scheduled if trash_empty_update_pending.
38 * 3) only the worker may set trash_empty_update_pending = TRUE.
39 * 4) only the UI updater may set trash_empty_update_pending = FALSE.
40 *
41 * i -think- this is threadsafe... ((famous last words...))
42 */
43static GFile * volatile trash_empty_current_file;
44static gsize volatile trash_empty_deleted_files;
45static gsize volatile trash_empty_total_files;
46static gboolean volatile trash_empty_update_pending;

48static gboolean
49trash_empty_update_dialog (gpointer user_data)
50{
51 gsize deleted, total;
52 GFile *file;
53
54 g_assert (trash_empty_update_pending);
55
56 deleted = trash_empty_deleted_files;
57 total = trash_empty_total_files;
58 file = trash_empty_current_file;
59
60 /* maybe the done() got processed first. */
61 if (trash_empty_dialog)
62 {
63 char *index_str, *total_str;
64 char *text_tmp, *text;
65 char *tmp;
66
67 /* The i18n tools can't handle a direct embedding of the
68 * size format using a macro. This is a work-around. */
69 index_str = g_strdup_printf ("%"G_GSIZE_FORMAT, deleted + 1);
70 total_str = g_strdup_printf ("%"G_GSIZE_FORMAT, total);
71 /* Translators: the %s in this string should be read as %d. */
72 text = g_strdup_printf (_("Removing item %s of %s"),
73 index_str, total_str);
74 gtk_progress_bar_set_text (trash_empty_progress_bar, text);
75 g_free (total_str);
76 g_free (index_str);
77 g_free (text);
78
79 if (deleted > total)
80 gtk_progress_bar_set_fraction (trash_empty_progress_bar, 1.0);
81 else
82 gtk_progress_bar_set_fraction (trash_empty_progress_bar,
83 (gdouble) deleted / (gdouble) total);
84
85 /* no g_file_get_basename? */
86 {
87 GFile *parent;
88
89 parent = g_file_get_parent (file);
90 text = g_file_get_uri (parent);
91 g_object_unref (parent);
92 }
93 gtk_label_set_text (trash_empty_location, text);
94 g_free (text);
95
96 tmp = g_file_get_basename (file);
97 /* Translators: %s is a file name */
98 text_tmp = g_strdup_printf (_("Removing: %s"), tmp);
99 text = g_markup_printf_escaped ("<i>%s</i>", text_tmp);
100 gtk_label_set_markup (trash_empty_file, text);
101 g_free (text);
102 g_free (text_tmp);
103 g_free (tmp);
104
105 /* unhide the labels */
106 gtk_widget_show_all (GTK_WIDGET (trash_empty_dialog));
107 }
108
109 trash_empty_current_file = NULL;
110 g_object_unref (file);
111
112 trash_empty_update_pending = FALSE;
113
114 return FALSE;
115}

128static void
129trash_empty_maybe_schedule_update (GFile *file,
130 gsize deleted)
131{
132 if (!trash_empty_update_pending)
133 {
134 g_assert (trash_empty_current_file == NULL);
135
136 trash_empty_current_file = g_object_ref (file);
137 trash_empty_deleted_files = deleted;
138
139 trash_empty_update_pending = TRUE;
140 g_main_context_invoke (NULL, trash_empty_update_dialog, NULL);
141 }
142}

144static void
145trash_empty_delete_contents (GCancellable *cancellable,
146 GFile *file,
147 gboolean actually_delete,
148 gsize *deleted)
149{
150 GFileEnumerator *enumerator;
151 GFileInfo *info;
152 GFile *child;
153
154 if (g_cancellable_is_cancelled (cancellable))
155 return;
156
157 enumerator = g_file_enumerate_children (file,
158 G_FILE_ATTRIBUTE_STANDARD_NAME ","
159 G_FILE_ATTRIBUTE_STANDARD_TYPE,
160 G_FILE_QUERY_INFO_NOFOLLOW_SYMLINKS,
161 cancellable, NULL);
162 if (enumerator)
163 {
164 while ((info = g_file_enumerator_next_file (enumerator,
165 cancellable, NULL)) != NULL)
166 {
167 child = g_file_get_child (file, g_file_info_get_name (info));
168
169 if (g_file_info_get_file_type (info) == G_FILE_TYPE_DIRECTORY)
170 trash_empty_delete_contents (cancellable, child,
171 actually_delete, deleted);
172
173 if (actually_delete)
174 {
175 trash_empty_maybe_schedule_update (child, *deleted);
176 g_file_delete (child, cancellable, NULL);
177 }
178
179 (*deleted)++;
180
181 g_object_unref (child);
182 g_object_unref (info);
183
184 if (g_cancellable_is_cancelled (cancellable))
185 break;
186 }
187
188 g_object_unref (enumerator);
189 }
190}

192static void
193trash_empty_job (GTask *task,
194 gpointer source_object,
195 gpointer user_data,
196 GCancellable *cancellable)
197{
198 gsize deleted;
199 GFile *trash;
200
201 trash = g_file_new_for_uri ("trash:///");
202
203 /* first do a dry run to count the number of files */
204 deleted = 0;
205 trash_empty_delete_contents (cancellable, trash, FALSE, &deleted);
206 trash_empty_total_files = deleted;
207
208 /* now do the real thing */
209 deleted = 0;
210 trash_empty_delete_contents (cancellable, trash, TRUE, &deleted);
211
212 /* done */
213 g_object_unref (trash);
214 g_task_return_boolean (task, TRUE);
215}

218static void
219trash_empty_start (GtkWidget *parent)
220{
221 struct { const char *name; gpointer *pointer; } widgets[] =
222 {
223 { "empty_trash", (gpointer *) &trash_empty_dialog },
224 { "progressbar", (gpointer *) &trash_empty_progress_bar },
225 { "location_label", (gpointer *) &trash_empty_location },
226 { "file_label", (gpointer *) &trash_empty_file }
227 };
228 GCancellable *cancellable;
229 GtkBuilder *builder;
230 GTask *task;
231 gint i;
232
233 builder = gtk_builder_new_from_resource (GRESOURCE "trashapplet-empty-progress.ui");
234
235 for (i = 0; i < G_N_ELEMENTS (widgets); i++)
236 {
237 GObject *object;
238
239 object = gtk_builder_get_object (builder, widgets[i].name);
240
241 if (object == NULL)
242 {
243 g_critical ("failed to parse trash-empty dialog markup");
244
245 if (trash_empty_dialog)
246 gtk_widget_destroy (GTK_WIDGET (trash_empty_dialog));
247
248 g_object_unref (builder);
249 return;
250 }
251
252 *widgets[i].pointer = object;
253 g_object_add_weak_pointer (object, widgets[i].pointer);
254 }
255 g_object_unref (builder);
256
257 cancellable = g_cancellable_new ();
258 g_signal_connect_object (trash_empty_dialog, "response",
259 G_CALLBACK (g_cancellable_cancel),
260 cancellable, G_CONNECT_SWAPPED);
261
262 task = g_task_new (parent, cancellable, trash_empty_done, NULL);
263 g_task_run_in_thread (task, trash_empty_job);
264 g_object_unref (task);
265 g_object_unref (cancellable);
266
267 gtk_window_set_screen (GTK_WINDOW (trash_empty_dialog),
268 gtk_widget_get_screen (parent));
269 gtk_widget_show (GTK_WIDGET (trash_empty_dialog));
270}
