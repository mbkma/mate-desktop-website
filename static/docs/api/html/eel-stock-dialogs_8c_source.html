109static void
110timed_wait_free (TimedWait *wait)
111{
112 guint delayed_close_handler_id;
113 guint64 time_up;
114
115 g_assert (g_hash_table_lookup (timed_wait_hash_table, wait) != NULL);
116
117 g_hash_table_remove (timed_wait_hash_table, wait);
118
119 g_free (wait->wait_message);
120 if (wait->parent_window != NULL)
121 {
122 g_object_unref (wait->parent_window);
123 }
124 if (wait->timeout_handler_id != 0)
125 {
126 g_source_remove (wait->timeout_handler_id);
127 }
128 if (wait->dialog != NULL)
129 {
130 /* Make sure to detach from the "destroy" signal, or we'll
131 * double-free.
132 */
133 g_signal_handlers_disconnect_by_func (G_OBJECT (wait->dialog),
134 G_CALLBACK (timed_wait_dialog_destroy_callback),
135 wait);
136
137 /* compute time up in milliseconds */
138 time_up = (g_get_monotonic_time () - wait->dialog_creation_time) / 1000;
139
140 if (time_up < TIMED_WAIT_MIN_TIME_UP)
141 {
142 delayed_close_handler_id = g_timeout_add (TIMED_WAIT_MIN_TIME_UP - time_up,
143 timed_wait_delayed_close_timeout_callback,
144 wait->dialog);
145 g_object_set_data (G_OBJECT (wait->dialog),
146 "eel-stock-dialogs/delayed_close_handler_timeout_id",
147 GUINT_TO_POINTER (delayed_close_handler_id));
148 g_signal_connect (wait->dialog, "destroy",
149 G_CALLBACK (timed_wait_delayed_close_destroy_dialog_callback),
150 GUINT_TO_POINTER (delayed_close_handler_id));
151 }
152 else
153 {
154 gtk_widget_destroy (GTK_WIDGET (wait->dialog));
155 }
156 }
157
158 /* And the wait object itself. */
159 g_free (wait);
160}

162static void
163timed_wait_dialog_destroy_callback (GtkWidget *object, gpointer callback_data)
164{
165 TimedWait *wait;
166
167 wait = callback_data;
168
169 g_assert (GTK_DIALOG (object) == wait->dialog);
170
171 wait->dialog = NULL;
172
173 /* When there's no cancel_callback, the originator will/must
174 * call eel_timed_wait_stop which will call timed_wait_free.
175 */
176
177 if (wait->cancel_callback != NULL)
178 {
179 (* wait->cancel_callback) (wait->callback_data);
180 timed_wait_free (wait);
181 }

269static gboolean
270timed_wait_callback (gpointer callback_data)
271{
272 TimedWait *wait;
273 GtkDialog *dialog;
274 const char *button;
275
276 wait = callback_data;
277
278 /* Put up the timed wait window. */
279 button = wait->cancel_callback != NULL ? "process-stop" : "gtk-ok";
280 dialog = GTK_DIALOG (gtk_message_dialog_new (wait->parent_window,
281 0,
282 GTK_MESSAGE_INFO,
283 GTK_BUTTONS_NONE,
284 NULL));
285
286 g_object_set (dialog,
287 "text", wait->wait_message,
288 "secondary-text", _("You can stop this operation by clicking cancel."),
289 NULL);
290
291 if (g_strcmp0 (button, "process-stop") == 0)
292 eel_dialog_add_button (GTK_DIALOG (dialog), _("_Cancel"), button, GTK_RESPONSE_OK);
293 else
294 eel_dialog_add_button (GTK_DIALOG (dialog), _("_OK"), button, GTK_RESPONSE_OK);
295
296 gtk_dialog_set_default_response (GTK_DIALOG (dialog), GTK_RESPONSE_OK);
297
298 /* The contents are often very small, causing tiny little
299 * dialogs with their titles clipped if you just let gtk
300 * sizing do its thing. This enforces a minimum width to
301 * make it more likely that the title won't be clipped.
302 */
303 gtk_window_set_default_size (GTK_WINDOW (dialog),
304 TIMED_WAIT_MINIMUM_DIALOG_WIDTH,
305 -1);
306 wait->dialog_creation_time = g_get_monotonic_time ();
307 gtk_widget_show (GTK_WIDGET (dialog));
308
309 /* FIXME bugzilla.eazel.com 2441:
310 * Could parent here, but it's complicated because we
311 * don't want this window to go away just because the parent
312 * would go away first.
313 */
314
315 /* Make the dialog cancel the timed wait when it goes away.
316 * Connect to "destroy" instead of "response" since we want
317 * to be called no matter how the dialog goes away.
318 */
319 g_signal_connect (dialog, "destroy",
320 G_CALLBACK (timed_wait_dialog_destroy_callback),
321 wait);
322 g_signal_connect (dialog, "response",
323 G_CALLBACK (trash_dialog_response_callback),
324 wait);
325
326 wait->timeout_handler_id = 0;
327 wait->dialog = dialog;
328
329 return FALSE;
330}

332void
333eel_timed_wait_start_with_duration (int duration,
334 EelCancelCallback cancel_callback,
335 gpointer callback_data,
336 const char *wait_message,
337 GtkWindow *parent_window)
338{
339 TimedWait *wait;
340
341 g_return_if_fail (callback_data != NULL);
342 g_return_if_fail (wait_message != NULL);
343 g_return_if_fail (parent_window == NULL || GTK_IS_WINDOW (parent_window));
344
345 /* Create the timed wait record. */
346 wait = g_new0 (TimedWait, 1);
347 wait->wait_message = g_strdup (wait_message);
348 wait->cancel_callback = cancel_callback;
349 wait->callback_data = callback_data;
350 wait->parent_window = parent_window;
351
352 if (parent_window != NULL)
353 {
354 g_object_ref (parent_window);
355 }
356
357 /* Start the timer. */
358 wait->timeout_handler_id = g_timeout_add (duration, timed_wait_callback, wait);
359
360 /* Put in the hash table so we can find it later. */
361 if (timed_wait_hash_table == NULL)
362 {
363 timed_wait_hash_table = eel_g_hash_table_new_free_at_exit
364 (timed_wait_hash, timed_wait_hash_equal, __FILE__ ": timed wait");
365 }
366 g_assert (g_hash_table_lookup (timed_wait_hash_table, wait) == NULL);
367 g_hash_table_insert (timed_wait_hash_table, wait, wait);
368 g_assert (g_hash_table_lookup (timed_wait_hash_table, wait) == wait);
369}

383void
384eel_timed_wait_stop (EelCancelCallback cancel_callback,
385 gpointer callback_data)
386{
387 TimedWait key;
388 TimedWait *wait;
389
390 g_return_if_fail (callback_data != NULL);
391
392 key.cancel_callback = cancel_callback;
393 key.callback_data = callback_data;
394 wait = g_hash_table_lookup (timed_wait_hash_table, &key);
395
396 g_return_if_fail (wait != NULL);
397
398 timed_wait_free (wait);
399}

401int
402eel_run_simple_dialog (GtkWidget *parent, gboolean ignore_close_box,
403 GtkMessageType message_type, const char *primary_text,
404 const char *secondary_text, ...)
405{
406 va_list button_title_args;
407 GtkWidget *dialog;
408 GtkWidget *top_widget, *chosen_parent;
409 int result;
410 int response_id;
411
412 /* Parent it if asked to. */
413 chosen_parent = NULL;
414 if (parent != NULL)
415 {
416 top_widget = gtk_widget_get_toplevel (parent);
417 if (GTK_IS_WINDOW (top_widget))
418 {
419 chosen_parent = top_widget;
420 }
421 }
422
423 /* Create the dialog. */
424 dialog = gtk_message_dialog_new (GTK_WINDOW (chosen_parent),
425 0,
426 message_type,
427 GTK_BUTTONS_NONE,
428 NULL);
429
430 g_object_set (dialog,
431 "text", primary_text,
432 "secondary-text", secondary_text,
433 NULL);
434
435 va_start (button_title_args, secondary_text);
436 response_id = 0;
437 while (1)
438 {
439 const char *button_title;
440
441 button_title = va_arg (button_title_args, const char *);
442 if (button_title == NULL)
443 {
444 break;
445 }
446
447 if (g_strcmp0 (button_title, "process-stop") == 0)
448 eel_dialog_add_button (GTK_DIALOG (dialog), _("_Cancel"), button_title, response_id);
449 else
450 eel_dialog_add_button (GTK_DIALOG (dialog), _("_OK"), button_title, response_id);
451
452 gtk_dialog_set_default_response (GTK_DIALOG (dialog), response_id);
453 response_id++;
454 }
455 va_end (button_title_args);
456
457 /* Run it. */
458 gtk_widget_show (dialog);
459 result = gtk_dialog_run (GTK_DIALOG (dialog));
460 while ((result == GTK_RESPONSE_NONE || result == GTK_RESPONSE_DELETE_EVENT) && ignore_close_box)
461 {
462 gtk_widget_show (GTK_WIDGET (dialog));
463 result = gtk_dialog_run (GTK_DIALOG (dialog));
464 }
465 gtk_widget_destroy (dialog);
466
467 return result;
468}
