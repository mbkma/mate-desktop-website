31static void
32test_consecutive_read (const gchar *inbuf,
33 const gchar *outbuf,
34 PlumaDocumentNewlineType type,
35 gsize read_chunk_len)
36{
37 GtkTextBuffer *buf;
38 GInputStream *in;
39 gsize outlen;
40 gssize n, r;
41 GError *err = NULL;
42 gchar *b;
43 gboolean close;
44
45 buf = gtk_text_buffer_new (NULL);
46 gtk_text_buffer_set_text (buf, inbuf, -1);
47
48 b = g_malloc (200);
49 in = pluma_document_input_stream_new (buf, type);
50
51 outlen = strlen (outbuf);
52 n = 0;
53
54 do
55 {
56 r = g_input_stream_read (in, b + n, read_chunk_len, NULL, &err);
57 g_assert_cmpint (r, >=, 0);
58 g_assert_no_error (err);
59
60 n += r;
61 } while (r != 0);
62
63 g_assert_cmpint (n, ==, outlen);
64
65 b[n] = '\0';
66
67 g_assert_cmpstr (b, ==, outbuf);
68
69 close = g_input_stream_close (in, NULL, &err);
70 g_assert (close);
71 g_assert_no_error (err);
72
73 g_object_unref (buf);
74 g_object_unref (in);
75 g_free (b);
76}
