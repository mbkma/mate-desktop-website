44char *
45mateweather_parser_get_value (MateWeatherParser *parser)
46{
47 char *value;
48
49 /* check for null node */
50 if (xmlTextReaderIsEmptyElement (parser->xml))
51 return NULL;
52
53 /* the next "node" is the text node containing the value we want to get */
54 if (xmlTextReaderRead (parser->xml) != 1)
55 return NULL;
56
57 value = (char *) xmlTextReaderValue (parser->xml);
58
59 /* move on to the end of this node */
60 while (xmlTextReaderNodeType (parser->xml) != XML_READER_TYPE_END_ELEMENT) {
61 if (xmlTextReaderRead (parser->xml) != 1) {
62 xmlFree (value);
63 return NULL;
64 }
65 }
66
67 /* consume the end element too */
68 if (xmlTextReaderRead (parser->xml) != 1) {
69 xmlFree (value);
70 return NULL;
71 }
72
73 return value;
74}

89char *
90mateweather_parser_get_localized_value (MateWeatherParser *parser)
91{
92 const char *this_language;
93 int best_match = INT_MAX;
94 const char *lang, *tagname, *next_tagname;
95 gboolean keep_going;
96 char *name = NULL;
97 int i;
98
99 tagname = (const char *) xmlTextReaderConstName (parser->xml);
100
101 do {
102 /* First let's get the language */
103 lang = (const char *) xmlTextReaderConstXmlLang (parser->xml);
104
105 if (lang == NULL)
106 this_language = "C";
107 else
108 this_language = lang;
109
110 /* the next "node" is text node containing the actual name */
111 if (xmlTextReaderRead (parser->xml) != 1) {
112 if (name)
113 xmlFree (name);
114 return NULL;
115 }
116
117 for (i = 0; parser->locales[i] && i < best_match; i++) {
118 if (!strcmp (parser->locales[i], this_language)) {
119 /* if we've already encounted a less accurate
120 translation, then free it */
121 g_free (name);
122
123 name = (char *) xmlTextReaderValue (parser->xml);
124 best_match = i;
125
126 break;
127 }
128 }
129
130 /* Skip to close tag */
131 while (xmlTextReaderNodeType (parser->xml) != XML_READER_TYPE_END_ELEMENT) {
132 if (xmlTextReaderRead (parser->xml) != 1) {
133 xmlFree (name);
134 return NULL;
135 }
136 }
137
138 /* Skip junk */
139 do {
140 if (xmlTextReaderRead (parser->xml) != 1) {
141 xmlFree (name);
142 return NULL;
143 }
144 } while (xmlTextReaderNodeType (parser->xml) != XML_READER_TYPE_ELEMENT &&
145 xmlTextReaderNodeType (parser->xml) != XML_READER_TYPE_END_ELEMENT);
146
147 /* if the next tag has the same name then keep going */
148 next_tagname = (const char *) xmlTextReaderConstName (parser->xml);
149 keep_going = !strcmp (next_tagname, tagname);
150
151 } while (keep_going);
152
153 return name;
154}

156MateWeatherParser *
157mateweather_parser_new (gboolean use_regions)
158{
159 MateWeatherParser *parser;
160 int zlib_support;
161 int i, keep_going;
162 char *filename;
163 char *tagname, *format;
164 time_t now;
165 struct tm tm;
166
167 parser = g_slice_new0 (MateWeatherParser);
168 parser->use_regions = use_regions;
169 parser->locales = g_get_language_names ();
170
171 zlib_support = xmlHasFeature (XML_WITH_ZLIB);
172
173 /* First try to load a locale-specific XML. It's much faster. */
174 filename = NULL;
175 for (i = 0; parser->locales[i] != NULL; i++) {
176 filename = g_strdup_printf ("%s/Locations.%s.xml",
177 MATEWEATHER_XML_LOCATION_DIR,
178 parser->locales[i]);
179
180 if (g_file_test (filename, G_FILE_TEST_IS_REGULAR))
181 break;
182
183 g_free (filename);
184 filename = NULL;
185
186 if (!zlib_support)
187 continue;
188
189 filename = g_strdup_printf ("%s/Locations.%s.xml.gz",
190 MATEWEATHER_XML_LOCATION_DIR,
191 parser->locales[i]);
192
193 if (g_file_test (filename, G_FILE_TEST_IS_REGULAR))
194 break;
195
196 g_free (filename);
197 filename = NULL;
198 }
199
200 /* Fall back on the file containing either all translations, or only
201 * the english names (depending on the configure flags).
202 */
203 if (!filename)
204 filename = g_build_filename (MATEWEATHER_XML_LOCATION_DIR, "Locations.xml", NULL);
205
206 if (!g_file_test (filename, G_FILE_TEST_IS_REGULAR) && zlib_support) {
207 g_free (filename);
208 filename = g_build_filename (MATEWEATHER_XML_LOCATION_DIR, "Locations.xml.gz", NULL);
209 }
210
211 /* Open the xml file containing the different locations */
212 parser->xml = xmlNewTextReaderFilename (filename);
213 g_free (filename);
214
215 if (parser->xml == NULL)
216 goto error_out;
217
218 /* fast forward to the first element */
219 do {
220 /* if we encounter a problem here, exit right away */
221 if (xmlTextReaderRead (parser->xml) != 1)
222 goto error_out;
223 } while (xmlTextReaderNodeType (parser->xml) != XML_READER_TYPE_ELEMENT);
224
225 /* check the name and format */
226 tagname = (char *) xmlTextReaderName (parser->xml);
227 keep_going = tagname && !strcmp (tagname, "mateweather");
228 xmlFree (tagname);
229
230 if (!keep_going)
231 goto error_out;
232
233 format = (char *) xmlTextReaderGetAttribute (parser->xml, (xmlChar *) "format");
234 keep_going = format && !strcmp (format, "1.0");
235 xmlFree (format);
236
237 if (!keep_going)
238 goto error_out;
239
240 /* Get timestamps for the start and end of this year */
241 now = time (NULL);
242 tm = *gmtime (&now);
243 tm.tm_mon = 0;
244 tm.tm_mday = 1;
245 tm.tm_hour = tm.tm_min = tm.tm_sec = 0;
246 parser->year_start = mktime (&tm);
247 tm.tm_year++;
248 parser->year_end = mktime (&tm);
249
250 return parser;
251
252error_out:
253 mateweather_parser_free (parser);
254 return NULL;
255}
