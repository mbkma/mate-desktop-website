130static void
131display_help (GtkAction *action,
132 DriveList *drive_list)
133{
134 GdkScreen *screen;
135 GError *error = NULL;
136
137 screen = gtk_widget_get_screen (GTK_WIDGET (drive_list));
138
139 gtk_show_uri_on_window (NULL,
140 "help:mate-drivemount",
141 gtk_get_current_event_time (),
142 &error);
143
144 if (error) {
145 GtkWidget *dialog;
146
147 dialog = gtk_message_dialog_new (NULL,
148 GTK_DIALOG_MODAL,
149 GTK_MESSAGE_ERROR,
150 GTK_BUTTONS_OK,
151 _("There was an error displaying help: %s"),
152 error->message);
153 g_signal_connect (dialog, "response",
154 G_CALLBACK (gtk_widget_destroy), NULL);
155 gtk_window_set_resizable (GTK_WINDOW (dialog), FALSE);
156 gtk_window_set_screen (GTK_WINDOW (dialog), screen);
157 gtk_widget_show (dialog);
158 g_error_free (error);
159 }
160}

171static gboolean
172applet_factory (MatePanelApplet *applet,
173 const char *iid,
174 gpointer user_data)
175{
176 gboolean ret = FALSE;
177 GtkWidget *drive_list;
178 AtkObject *ao;
179 GtkActionGroup *action_group;
180
181 if (!strcmp (iid, drivemount_iid)) {
182#ifndef ENABLE_IN_PROCESS
183 g_set_application_name (_("Disk Mounter"));
184#endif
185
186 gtk_window_set_default_icon_name ("media-floppy");
187
188 mate_panel_applet_set_flags (applet, MATE_PANEL_APPLET_EXPAND_MINOR);
189
190 drive_list = drive_list_new ();
191
192 gtk_container_add (GTK_CONTAINER (applet), drive_list);
193
194 g_signal_connect_object (applet, "change-orient",
195 G_CALLBACK (change_orient), drive_list, 0);
196 g_signal_connect_object (applet, "size-allocate",
197 G_CALLBACK (size_allocate), drive_list, 0);
198 g_signal_connect (applet, "change-background",
199 G_CALLBACK (change_background), drive_list);
200
201 /* set initial state */
202 change_orient (applet,
203 mate_panel_applet_get_orient (applet),
204 DRIVE_LIST (drive_list));
205
206 action_group = gtk_action_group_new ("DriveMount Applet Actions");
207 gtk_action_group_set_translation_domain (action_group, GETTEXT_PACKAGE);
208 gtk_action_group_add_actions (action_group,
209 applet_menu_actions,
210 G_N_ELEMENTS (applet_menu_actions),
211 drive_list);
212 mate_panel_applet_setup_menu_from_resource (applet,
213 DRIVEMOUNT_RESOURCE_PATH "drivemount-applet-menu.xml",
214 action_group);
215 g_object_unref (action_group);
216
217 ao = gtk_widget_get_accessible (GTK_WIDGET (applet));
218 atk_object_set_name (ao, _("Disk Mounter"));
219
220 gtk_widget_show_all (GTK_WIDGET (applet));
221
222 ret = TRUE;
223 }
224
225 return ret;
226}
