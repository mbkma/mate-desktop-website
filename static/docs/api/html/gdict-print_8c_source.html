55static void
56begin_print (GtkPrintOperation *operation,
57 GtkPrintContext *context,
58 gpointer user_data)
59{
60 GdictPrintData *data = user_data;
61 gchar *contents;
62 gdouble height;
63
64 height = gtk_print_context_get_height (context)
65 - HEADER_HEIGHT (10)
66 - HEADER_GAP (3);
67
68 contents = gdict_defbox_get_text (data->defbox, NULL);
69
70 data->lines = g_strsplit (contents, "\n", 0);
71 data->n_lines = g_strv_length (data->lines);
72 data->lines_per_page = floor (height / data->font_size);
73
74 data->n_pages = (data->n_lines - 1) / data->lines_per_page + 1;
75
76 gtk_print_operation_set_n_pages (operation, data->n_pages);
77
78 g_free (contents);
79}

81static void
82draw_page (GtkPrintOperation *operation,
83 GtkPrintContext *context,
84 gint page_number,
85 gpointer user_data)
86{
87 GdictPrintData *data = user_data;
88 cairo_t *cr;
89 PangoLayout *layout;
90 gint text_width, text_height;
91 gdouble width;
92 gint line, i;
93 PangoFontDescription *desc;
94 gchar *page_str;
95
96 cr = gtk_print_context_get_cairo_context (context);
97 width = gtk_print_context_get_width (context);
98
99 cairo_rectangle (cr, 0, 0, width, HEADER_HEIGHT (10));
100
101 cairo_set_source_rgb (cr, 0.8, 0.8, 0.8);
102 cairo_fill_preserve (cr);
103
104 cairo_set_source_rgb (cr, 0, 0, 0);
105 cairo_set_line_width (cr, 1);
106 cairo_stroke (cr);
107
108 /* header */
109 layout = gtk_print_context_create_pango_layout (context);
110
111 desc = pango_font_description_from_string ("sans 14");
112 pango_layout_set_font_description (layout, desc);
113 pango_font_description_free (desc);
114
115 pango_layout_set_text (layout, data->word, -1);
116 pango_layout_get_pixel_size (layout, &text_width, &text_height);
117
118 if (text_width > width)
119 {
120 pango_layout_set_width (layout, width);
121 pango_layout_set_ellipsize (layout, PANGO_ELLIPSIZE_START);
122 pango_layout_get_pixel_size (layout, &text_width, &text_height);
123 }
124
125 cairo_move_to (cr, (width - text_width) / 2,
126 (HEADER_HEIGHT (10) - text_height) / 2);
127 pango_cairo_show_layout (cr, layout);
128
129 page_str = g_strdup_printf ("%d/%d", page_number + 1, data->n_pages);
130 pango_layout_set_text (layout, page_str, -1);
131 g_free (page_str);
132
133 pango_layout_set_width (layout, -1);
134 pango_layout_get_pixel_size (layout, &text_width, &text_height);
135 cairo_move_to (cr, width - text_width - 4,
136 (HEADER_HEIGHT (10) - text_height) / 2);
137 pango_cairo_show_layout (cr, layout);
138
139 g_object_unref (layout);
140
141 /* text */
142 layout = gtk_print_context_create_pango_layout (context);
143 pango_font_description_set_size (data->font_desc,
144 data->font_size * PANGO_SCALE);
145 pango_layout_set_font_description (layout, data->font_desc);
146
147 cairo_move_to (cr, 0, HEADER_HEIGHT (10) + HEADER_GAP (3));
148 line = page_number * data->lines_per_page;
149 for (i = 0; i < data->lines_per_page && line < data->n_lines; i++)
150 {
151 pango_layout_set_text (layout, data->lines[line], -1);
152
153 pango_cairo_show_layout (cr, layout);
154 cairo_rel_move_to (cr, 0, data->font_size);
155 line++;
156 }
157
158 g_object_unref (layout);
159}

161static void
162end_print (GtkPrintOperation *operation,
163 GtkPrintContext *context,
164 gpointer user_data)
165{
166 GdictPrintData *data = user_data;
167
168 pango_font_description_free (data->font_desc);
169 g_free (data->word);
170 g_strfreev (data->lines);
171 g_free (data);
172}

185void
186gdict_show_print_preview (GtkWindow *parent,
187 GdictDefbox *defbox)
188{
189 GtkPrintOperation *operation;
190 GdictPrintData *data;
191 gchar *print_font;
192 gchar *word;
193 GError *error;
194
195 g_return_if_fail (parent == NULL || GTK_IS_WINDOW (parent));
196 g_return_if_fail (GDICT_IS_DEFBOX (defbox));
197
198 g_object_get (defbox, "word", &word, NULL);
199 if (!word)
200 {
201 g_warning ("Preview should be disabled.");
202 return;
203 }
204
205 data = g_new0 (GdictPrintData, 1);
206 data->defbox = defbox;
207 data->word = word;
208
209 operation = gtk_print_operation_new ();
210 print_font = get_print_font ();
211 data->font_desc = pango_font_description_from_string (print_font);
212 data->font_size = pango_font_description_get_size (data->font_desc)
213 / PANGO_SCALE;
214 g_free (print_font);
215
216 g_signal_connect (operation, "begin-print",
217 G_CALLBACK (begin_print), data);
218 g_signal_connect (operation, "draw-page",
219 G_CALLBACK (draw_page), data);
220 g_signal_connect (operation, "end-print",
221 G_CALLBACK (end_print), data);
222
223 error = NULL;
224 gtk_print_operation_run (operation,
225 GTK_PRINT_OPERATION_ACTION_PREVIEW,
226 parent,
227 &error);
228 g_object_unref (operation);
229
230 if (error)
231 {
232 GtkWidget *dialog;
233
234 dialog = gtk_message_dialog_new (parent,
235 GTK_DIALOG_DESTROY_WITH_PARENT,
236 GTK_MESSAGE_ERROR,
237 GTK_BUTTONS_CLOSE,
238 _("Unable to display the preview: %s"),
239 error->message);
240 g_error_free (error);
241
242 g_signal_connect (dialog, "response",
243 G_CALLBACK (gtk_widget_destroy), NULL);
244
245 gtk_widget_show (dialog);
246 }
247}
