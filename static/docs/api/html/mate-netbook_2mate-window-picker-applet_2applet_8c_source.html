152static gboolean
153cw_applet_fill (MatePanelApplet *applet,
154 const gchar *iid,
155 gpointer data)
156{
157 WinPickerApp *app;
158 GtkWidget *eb, *tasks, *title;
159 GtkActionGroup *action_group;
160 GObjectClass *object_class;
161
162 if (strcmp (iid, "MateWindowPicker") != 0)
163 return FALSE;
164
165 bindtextdomain (GETTEXT_PACKAGE, MATELOCALEDIR);
166 bind_textdomain_codeset (GETTEXT_PACKAGE, "UTF-8");
167 textdomain (GETTEXT_PACKAGE);
168
169 /* Have our background automatically painted. */
170 mate_panel_applet_set_background_widget (MATE_PANEL_APPLET (applet), GTK_WIDGET (applet));
171
172 wnck_set_client_type (WNCK_CLIENT_TYPE_PAGER);
173
174 app = g_slice_new0 (WinPickerApp);
175 mainapp = app;
176
177 object_class = G_OBJECT_GET_CLASS (G_OBJECT(applet));
178 object_class->finalize = cw_applet_finalize;
179 parent_class = g_type_class_peek_parent (object_class);
180
181 /* gsettings prefs */
182 app->settings = mate_panel_applet_settings_new (applet, APPLET_SCHEMA);
183 g_signal_connect (app->settings, "changed::" SHOW_WIN_KEY,
184 G_CALLBACK (on_show_all_windows_changed), app);
185 g_signal_connect (app->settings, "changed::" SHOW_HOME_TITLE_KEY,
186 G_CALLBACK (on_show_home_title_changed), app);
187 g_signal_connect (app->settings, "changed::" BOLD_WINDOW_TITLE_KEY,
188 G_CALLBACK (on_bold_window_title_changed), app);
189
190 app->applet = GTK_WIDGET (applet);
191 force_no_focus_padding (GTK_WIDGET (applet));
192 gtk_container_set_border_width (GTK_CONTAINER (applet), 0);
193
194 eb = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
195 gtk_container_add (GTK_CONTAINER (applet), eb);
196 gtk_container_set_border_width (GTK_CONTAINER (eb), 0);
197
198 tasks = app->tasks = task_list_get_default ();
199 gtk_box_pack_start (GTK_BOX (eb), tasks, FALSE, FALSE, 0);
200
201 title = app->title = task_title_new ();
202 gtk_box_pack_start (GTK_BOX (eb), title, TRUE, TRUE, 0);
203
204 gtk_widget_show_all (GTK_WIDGET (applet));
205
206 on_show_all_windows_changed (app->settings, SHOW_WIN_KEY, app);
207 on_show_home_title_changed (app->settings, SHOW_HOME_TITLE_KEY, app);
208 on_bold_window_title_changed (app->settings, BOLD_WINDOW_TITLE_KEY, app);
209
210 action_group = gtk_action_group_new ("MateWindowPicker Applet Actions");
211 gtk_action_group_set_translation_domain (action_group, GETTEXT_PACKAGE);
212 gtk_action_group_add_actions (action_group,
213 window_picker_menu_actions,
214 G_N_ELEMENTS (window_picker_menu_actions),
215 app);
216 mate_panel_applet_setup_menu_from_file (MATE_PANEL_APPLET (app->applet),
217 MATEWINDOWPICKER_MENU_UI_DIR "/mate-window-picker-applet-menu.xml",
218 action_group);
219 g_object_unref (action_group);
220
221 mate_panel_applet_set_flags (MATE_PANEL_APPLET (applet),
222 MATE_PANEL_APPLET_EXPAND_MAJOR
223 | MATE_PANEL_APPLET_EXPAND_MINOR
224 | MATE_PANEL_APPLET_HAS_HANDLE);
225
226 return TRUE;
227}
