43void
44baobab_get_filesystem (BaobabFS *fs)
45{
46 size_t i;
47 glibtop_mountlist mountlist;
48 glibtop_mountentry *mountentries;
49
50 memset (fs, 0, sizeof *fs);
51
52 mountentries = glibtop_get_mountlist (&mountlist, FALSE);
53
54 for (i = 0; i < mountlist.number; ++i) {
55 GFile *file;
56 glibtop_fsusage fsusage;
57
58 file = g_file_new_for_path (mountentries[i].mountdir);
59
60 if (!baobab_is_excluded_location (file)) {
61
62 glibtop_get_fsusage (&fsusage, mountentries[i].mountdir);
63
64 fs->total += fsusage.blocks * fsusage.block_size;
65 fs->avail += fsusage.bfree * fsusage.block_size;
66 fs->used += (fsusage.blocks - fsusage.bfree) * fsusage.block_size;
67 }
68
69 g_object_unref (file);
70 }
71
72 g_free (mountentries);
73}

149gboolean
150show_bars (GtkTreeModel *mdl,
151 GtkTreePath *path,
152 GtkTreeIter *iter,
153 gpointer data)
154{
155 GtkTreeIter parent;
156 gdouble perc;
157 gint readelements, size_col;
158 guint64 refsize, size;
159 char *sizecstr = NULL;
160
161 if (baobab.show_allocated)
162 size_col = (gint) COL_H_ALLOCSIZE;
163 else
164 size_col = (gint) COL_H_SIZE;
165
166 if (gtk_tree_model_iter_parent (mdl, &parent, iter)) {
167 gtk_tree_model_get (mdl, iter, COL_H_ELEMENTS,
168 &readelements, -1);
169
170 if (readelements == -1) {
171 gtk_tree_store_set (GTK_TREE_STORE (mdl), iter,
172 COL_DIR_SIZE, "--",
173 COL_ELEMENTS, "--", -1);
174 return FALSE;
175 }
176
177 gtk_tree_model_get (mdl, &parent, COL_H_ELEMENTS,
178 &readelements, -1);
179
180 gtk_tree_model_get (mdl, iter, size_col, &size, -1);
181
182 sizecstr = g_format_size (size);
183
184 if (readelements == -1) {
185 gtk_tree_store_set (GTK_TREE_STORE (mdl), iter,
186 COL_DIR_SIZE, sizecstr, -1);
187
188 g_free (sizecstr);
189 return FALSE;
190 }
191
192 gtk_tree_model_get (mdl, &parent, size_col, &refsize, -1);
193 perc = (refsize != 0) ? ((gdouble) size * 100) / (gdouble) refsize : 0.0;
194
195 gtk_tree_store_set (GTK_TREE_STORE (mdl), iter,
196 COL_DIR_SIZE, sizecstr,
197 COL_H_PERC, perc, -1);
198
199 g_free (sizecstr);
200 } else {
201 gtk_tree_model_get (mdl, iter, COL_H_ELEMENTS,
202 &readelements, -1);
203
204 if (readelements != -1) {
205 gtk_tree_model_get (mdl, iter, size_col, &size,
206 -1);
207
208 sizecstr = g_format_size (size);
209
210 gtk_tree_store_set (GTK_TREE_STORE (mdl), iter,
211 COL_H_PERC, 100.0,
212 COL_DIR_SIZE, sizecstr, -1);
213
214 g_free (sizecstr);
215 }
216 else {
217 gtk_tree_store_set (GTK_TREE_STORE (mdl), iter,
218 COL_DIR_SIZE, "--",
219 COL_ELEMENTS, "--", -1);
220 }
221 }
222
223 return FALSE;
224}

274gboolean
275baobab_check_dir (GFile *file)
276{
277 GFileInfo *info;
278 GError *error = NULL;
279 gboolean ret = TRUE;
280
281 if (baobab_is_excluded_location (file)) {
282 message("", _("Cannot check an excluded folder!"), GTK_MESSAGE_INFO, baobab.window);
283 return FALSE;
284 }
285
286 info = g_file_query_info (file,
287 G_FILE_ATTRIBUTE_STANDARD_TYPE,
288 G_FILE_QUERY_INFO_NONE,
289 NULL,
290 &error);
291
292 if (!info) {
293 message("", error->message, GTK_MESSAGE_INFO, baobab.window);
294 g_error_free (error);
295
296 return FALSE;
297 }
298
299 if ((g_file_info_get_file_type (info) != G_FILE_TYPE_DIRECTORY) ||
300 is_virtual_filesystem(file)) {
301
302 char *error_msg = NULL;
303 gchar *name = NULL;
304
305 name = g_file_get_parse_name (file);
306 error_msg = g_strdup_printf (_("\"%s\" is not a valid folder"),
307 name);
308
309 message (error_msg, _("Could not analyze disk usage."),
310 GTK_MESSAGE_ERROR, baobab.window);
311
312 g_free (error_msg);
313 g_free (name);
314 ret = FALSE;
315 }
316
317 g_object_unref(info);
318
319 return ret;
320}

361void
362open_file_with_application (GFile *file)
363{
364 GAppInfo *application;
365 gchar *primary;
366 GFileInfo *info;
367 const char *content;
368
369 info = g_file_query_info (file,
370 G_FILE_ATTRIBUTE_STANDARD_CONTENT_TYPE,
371 G_FILE_QUERY_INFO_NONE,
372 NULL,
373 NULL);
374 if (!info) return;
375
376 content = g_file_info_get_content_type (info);
377 application = g_app_info_get_default_for_type (content, TRUE);
378
379 if (!application) {
380 primary = g_strdup_printf (_("Could not open folder \"%s\""),
381 g_file_get_basename (file));
382 message (primary,
383 _("There is no installed viewer capable "
384 "of displaying the folder."),
385 GTK_MESSAGE_ERROR,
386 baobab.window);
387 g_free (primary);
388 }
389 else {
390 GList *uris = NULL;
391 gchar *uri;
392
393 uri = g_file_get_uri (file);
394 uris = g_list_append (uris, uri);
395 g_app_info_launch_uris (application, uris, NULL, NULL);
396
397 g_list_free (uris);
398 g_free (uri);
399 }
400
401 if (application)
402 g_object_unref (application);
403
404 g_object_unref (info);
405}

431gboolean
432trash_file (GFile *file)
433{
434 GError *error = NULL;
435
436 if (!g_file_trash (file, NULL, &error)) {
437 GFileInfo *info;
438 char *str = NULL;
439 char *mess;
440
441 info = g_file_query_info (file,
442 G_FILE_ATTRIBUTE_STANDARD_DISPLAY_NAME,
443 G_FILE_QUERY_INFO_NONE,
444 NULL,
445 NULL);
446
447 if (info) {
448 const char *displayname = g_file_info_get_display_name (info);
449 if (displayname)
450 str = g_strdup_printf (_("Could not move \"%s\" to the Trash"),
451 displayname);
452
453 g_object_unref (info);
454 }
455
456 /* fallback */
457 if (str == NULL)
458 str = g_strdup (_("Could not move file to the Trash"));
459
460 mess = g_strdup_printf (_("Details: %s"), error->message);
461 message (str, mess, GTK_MESSAGE_ERROR, baobab.window);
462 g_free (str);
463 g_free (mess);
464 g_error_free (error);
465
466 return FALSE;
467 }
468
469 return TRUE;
470}
