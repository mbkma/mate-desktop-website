31static void
32get_window_rect (const WnckWindowDisplayInfo *win,
33 int screen_width,
34 int screen_height,
35 const GdkRectangle *workspace_rect,
36 GdkRectangle *rect)
37{
38 double width_ratio, height_ratio;
39 int x, y, width, height;
40
41 width_ratio = (double) workspace_rect->width / (double) screen_width;
42 height_ratio = (double) workspace_rect->height / (double) screen_height;
43
44 x = win->x;
45 y = win->y;
46 width = win->width;
47 height = win->height;
48
49 x = (int) (width_ratio * (double) x);
50 y = (int) (height_ratio * (double) y);
51 width = (int) (width_ratio * (double) width);
52 height = (int) (height_ratio * (double) height);
53
54 x += workspace_rect->x;
55 y += workspace_rect->y;
56
57 if (width < 3)
58 width = 3;
59 if (height < 3)
60 height = 3;
61
62 rect->x = x;
63 rect->y = y;
64 rect->width = width;
65 rect->height = height;
66}

87static void
88draw_window (GtkWidget *widget,
89 cairo_t *cr,
90 const WnckWindowDisplayInfo *win,
91 const GdkRectangle *winrect,
92 GtkStateFlags state)
93{
94 cairo_surface_t *icon;
95 int icon_x, icon_y, icon_w, icon_h, scale;
96 gboolean is_active;
97 GdkRGBA color;
98 GtkStyleContext *style;
99
100 is_active = win->is_active;
101
102 cairo_save (cr);
103
104 cairo_rectangle (cr, winrect->x, winrect->y, winrect->width, winrect->height);
105 cairo_clip (cr);
106
107 style = gtk_widget_get_style_context (widget);
108 if (is_active)
109 meta_gtk_style_get_light_color (style, state, &color);
110 else
111 get_background_color (style, state, &color);
112
113 gdk_cairo_set_source_rgba (cr, &color);
114
115 cairo_rectangle (cr,
116 winrect->x + 1, winrect->y + 1,
117 MAX (0, winrect->width - 2), MAX (0, winrect->height - 2));
118 cairo_fill (cr);
119
120 scale = gtk_widget_get_scale_factor (widget);
121
122 icon = gdk_cairo_surface_create_from_pixbuf (win->icon, scale, NULL);
123
124 icon_w = icon_h = 0;
125
126 if (icon)
127 {
128 icon_w = cairo_image_surface_get_width (icon) / scale;
129 icon_h = cairo_image_surface_get_height (icon) / scale;
130
131 /* If the icon is too big, fall back to mini icon. */
132 if (icon_w > (winrect->width - 2) || icon_h > (winrect->height - 2))
133 {
134 cairo_surface_destroy (icon);
135 icon = gdk_cairo_surface_create_from_pixbuf (win->mini_icon, scale, NULL);
136 if (icon)
137 {
138 icon_w = cairo_image_surface_get_width (icon) / scale;
139 icon_h = cairo_image_surface_get_height (icon) / scale;
140
141 /* Give up. */
142 if (icon_w > (winrect->width - 2) || icon_h > (winrect->height - 2))
143 {
144 cairo_surface_destroy (icon);
145 icon = NULL;
146 }
147 }
148 }
149 }
150
151 if (icon)
152 {
153 icon_x = winrect->x + (winrect->width - icon_w) / 2;
154 icon_y = winrect->y + (winrect->height - icon_h) / 2;
155
156 cairo_save (cr);
157 cairo_set_source_surface (cr, icon, icon_x, icon_y);
158 cairo_rectangle (cr, icon_x, icon_y, icon_w, icon_h);
159 cairo_clip (cr);
160 cairo_paint (cr);
161 cairo_restore (cr);
162
163 cairo_surface_destroy (icon);
164 }
165
166 gtk_style_context_get_color (style, state, &color);
167 gdk_cairo_set_source_rgba (cr, &color);
168
169 cairo_set_line_width (cr, 1.0);
170 cairo_rectangle (cr,
171 winrect->x + 0.5, winrect->y + 0.5,
172 MAX (0, winrect->width - 1), MAX (0, winrect->height - 1));
173 cairo_stroke (cr);
174
175 cairo_restore(cr);
176}

178void
179wnck_draw_workspace (GtkWidget *widget,
180 cairo_t *cr,
181 int x,
182 int y,
183 int width,
184 int height,
185 int screen_width,
186 int screen_height,
187 GdkPixbuf *workspace_background,
188 gboolean is_active,
189 const WnckWindowDisplayInfo *windows,
190 int n_windows)
191{
192 int i;
193 GdkRectangle workspace_rect;
194 GtkStateFlags state;
195 GtkStyleContext *style;
196
197 workspace_rect.x = x;
198 workspace_rect.y = y;
199 workspace_rect.width = width;
200 workspace_rect.height = height;
201
202 if (is_active)
203 state = GTK_STATE_FLAG_SELECTED;
204 else if (workspace_background)
205 state = GTK_STATE_FLAG_PRELIGHT;
206 else
207 state = GTK_STATE_FLAG_NORMAL;
208
209 style = gtk_widget_get_style_context (widget);
210 cairo_save (cr);
211
212 if (workspace_background)
213 {
214 gdk_cairo_set_source_pixbuf (cr, workspace_background, x, y);
215 cairo_paint (cr);
216 }
217 else
218 {
219 GdkRGBA color;
220
221 meta_gtk_style_get_dark_color (style, state, &color);
222 color.alpha = 0.25;
223
224 gdk_cairo_set_source_rgba (cr, &color);
225 cairo_rectangle (cr, x, y, width, height);
226 cairo_fill (cr);
227 }
228
229 i = 0;
230 while (i < n_windows)
231 {
232 const WnckWindowDisplayInfo *win = &windows[i];
233 GdkRectangle winrect;
234
235 get_window_rect (win, screen_width,
236 screen_height, &workspace_rect, &winrect);
237
238 draw_window (widget,
239 cr,
240 win,
241 &winrect,
242 state);
243
244 ++i;
245 }
246
247 cairo_restore(cr);
248}
