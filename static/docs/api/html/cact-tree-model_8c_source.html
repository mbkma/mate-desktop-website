351/*
352 * Initializes the tree model.
353 *
354 * We use drag and drop:
355 * - inside of treeview, for duplicating items, or moving items between menus
356 * - from treeview to the outside world (e.g. Caja) to export actions
357 * - from outside world (e.g. Caja) to import actions
358 */
359static void
360instance_constructed( GObject *model )
361{
362 static const gchar *thisfn = "cact_tree_model_instance_constructed";
363 CactTreeModelPrivate *priv;
364
365 g_return_if_fail( CACT_IS_TREE_MODEL( model ));
366
367 priv = CACT_TREE_MODEL( model )->private;
368
369 if( !priv->dispose_has_run ){
370
371 /* chain up to the parent class */
372 if( G_OBJECT_CLASS( st_parent_class )->constructed ){
373 G_OBJECT_CLASS( st_parent_class )->constructed( model );
374 }
375
376 g_debug( "%s: model=%p (%s)", thisfn, ( void * ) model, G_OBJECT_TYPE_NAME( model ));
377
378 priv->clipboard = cact_clipboard_new( priv->window );
379
380 if( priv->mode == TREE_MODE_EDITION ){
381
382 egg_tree_multi_drag_add_drag_support(
383 EGG_TREE_MULTI_DRAG_SOURCE( model ),
384 priv->treeview );
385
386 gtk_tree_view_enable_model_drag_dest(
387 priv->treeview,
388 tree_model_dnd_dest_formats,
389 tree_model_dnd_dest_formats_count,
390 cact_tree_model_dnd_imulti_drag_source_get_drag_actions( EGG_TREE_MULTI_DRAG_SOURCE( model )));
391
392 base_window_signal_connect(
393 priv->window,
394 G_OBJECT( priv->treeview ),
395 "drag-begin",
396 G_CALLBACK( cact_tree_model_dnd_on_drag_begin ));
397
398 /* connect: implies that we have to do all hard stuff
399 * connect_after: no more triggers drag-drop signal
400 */
401 /*base_window_signal_connect_after( window,
402 G_OBJECT( model->private->treeview ), "drag-motion", G_CALLBACK( on_drag_motion ));*/
403
404 /*base_window_signal_connect( window,
405 G_OBJECT( model->private->treeview ), "drag-drop", G_CALLBACK( on_drag_drop ));*/
406
407 base_window_signal_connect(
408 priv->window,
409 G_OBJECT( priv->treeview ),
410 "drag-end",
411 G_CALLBACK( cact_tree_model_dnd_on_drag_end ));
412
413 na_settings_register_key_callback(
414 NA_IPREFS_ITEMS_LIST_ORDER_MODE,
415 G_CALLBACK( on_settings_order_mode_changed ),
416 model );
417
418 base_window_signal_connect_with_data(
419 priv->window,
420 G_OBJECT( priv->window ),
421 TAB_UPDATABLE_SIGNAL_ITEM_UPDATED,
422 G_CALLBACK( on_tab_updatable_item_updated ),
423 model );
424 }

494CactTreeModel *
495cact_tree_model_new( BaseWindow *window, GtkTreeView *treeview, CactTreeMode mode )
496{
497 static const gchar *thisfn = "cact_tree_model_new";
498 GtkTreeStore *ts_model;
499 CactTreeModel *model;
500 gint order_mode;
501
502 g_return_val_if_fail( BASE_IS_WINDOW( window ), NULL );
503 g_return_val_if_fail( GTK_IS_TREE_VIEW( treeview ), NULL );
504
505 g_debug( "%s: window=%p, treeview=%p, mode=%u", thisfn, ( void * ) window, ( void * ) treeview, mode );
506
507 ts_model = gtk_tree_store_new(
508 TREE_N_COLUMN, GDK_TYPE_PIXBUF, G_TYPE_STRING, NA_TYPE_OBJECT );
509
510 /* create the filter model
511 */
512 model = g_object_new( CACT_TYPE_TREE_MODEL,
513 "child-model", ts_model,
514 TREE_PROP_WINDOW, window,
515 TREE_PROP_TREEVIEW, treeview,
516 TREE_PROP_MODE, mode,
517 NULL );
518 g_object_unref( ts_model );
519
520 gtk_tree_model_filter_set_visible_func(
521 GTK_TREE_MODEL_FILTER( model ), ( GtkTreeModelFilterVisibleFunc ) filter_visible, model, NULL );
522
523 /* initialize the sortable interface
524 */
525 order_mode = na_iprefs_get_order_mode( NULL );
526 display_order_change( model, order_mode );
527
528 return( model );
529}

599GtkTreePath *
600cact_tree_model_delete( CactTreeModel *model, NAObject *object )
601{
602 GtkTreePath *path;
603 static const gchar *thisfn = "cact_tree_model_delete";
604 GtkTreeIter iter;
605 GtkTreeStore *store;
606 NAObjectItem *parent;
607
608 g_return_val_if_fail( CACT_IS_TREE_MODEL( model ), NULL );
609
610 path = NULL;
611
612 if( !model->private->dispose_has_run ){
613 g_debug( "%s: model=%p, object=%p (%s)",
614 thisfn, ( void * ) model, ( void * ) object, object ? G_OBJECT_TYPE_NAME( object ) : "null" );
615
616 path = cact_tree_model_object_to_path( model, object );
617
618 if( path != NULL ){
619
620 /* detaching to-be-deleted object from its current parent
621 */
622 parent = na_object_get_parent( object );
623 g_debug( "%s: object=%p, parent=%p", thisfn, ( void * ) object, ( void * ) parent );
624 if( parent ){
625 na_object_remove_item( parent, object );
626 }
627
628 /* then recursively remove the object and its children from the store
629 */
630 store = GTK_TREE_STORE( gtk_tree_model_filter_get_model( GTK_TREE_MODEL_FILTER( model )));
631 if( gtk_tree_model_get_iter( GTK_TREE_MODEL( store ), &iter, path )){
632 delete_items_rec( store, &iter );
633 }
634 }
635 }
636
637 return( path );
638}

653void
654cact_tree_model_fill( CactTreeModel *model, GList *items )
655{
656 static const gchar *thisfn = "cact_tree_model_fill";
657 GtkTreeStore *ts_model;
658 GList *it;
659 NAObject *duplicate;
660
661 g_return_if_fail( CACT_IS_TREE_MODEL( model ));
662
663 g_debug( "%s: model=%p, items=%p (count=%d)",
664 thisfn, ( void * ) model, ( void * ) items, g_list_length( items ));
665
666 if( !model->private->dispose_has_run ){
667
668 ts_model = GTK_TREE_STORE( gtk_tree_model_filter_get_model( GTK_TREE_MODEL_FILTER( model )));
669 gtk_tree_store_clear( ts_model );
670
671 for( it = items ; it ; it = it->next ){
672 duplicate = ( NAObject * ) na_object_duplicate( it->data, DUPLICATE_REC );
673 na_object_check_status( duplicate );
674 fill_tree_store( ts_model, model->private->treeview, duplicate, NULL );
675 na_object_unref( duplicate );
