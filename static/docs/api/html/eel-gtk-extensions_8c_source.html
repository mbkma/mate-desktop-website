47/* Used for window position & size sanity-checking. The sizes are big enough to prevent
48 * at least normal-sized mate panels from obscuring the window at the screen edges.

50#define MINIMUM_ON_SCREEN_WIDTH 100
51#define MINIMUM_ON_SCREEN_HEIGHT 100

62char*
63eel_gtk_window_get_geometry_string (GtkWindow *window)
64{
65 char *str;
66 int w, h, x, y;
67
68 g_return_val_if_fail (GTK_IS_WINDOW (window), NULL);
69 g_return_val_if_fail (gtk_window_get_gravity (window) ==

71
72 gtk_window_get_position (window, &x, &y);
73 gtk_window_get_size (window, &w, &h);
74
75 str = g_strdup_printf ("%dx%d+%d+%d", w, h, x, y);

80static void
81sanity_check_window_position (int *left, int *top)
82{
83 g_assert (left != NULL);
84 g_assert (top != NULL);
85 GdkDisplay *display;
86 GdkWindow *root_window;
87 GdkRectangle workarea = {0};
88
89 /* Make sure the top of the window is on screen, for
90 * draggability (might not be necessary with all window managers,
91 * but seems reasonable anyway). Make sure the top of the window
92 * isn't off the bottom of the screen, or so close to the bottom
93 * that it might be obscured by the panel.
94 *
95 */
96 display = gdk_display_get_default ();
97 /*This is x11 only, there is no root window in wayland*/
98 if (GDK_IS_X11_DISPLAY (display))
99 {
100 root_window = gdk_screen_get_root_window (gdk_screen_get_default());
101 gdk_monitor_get_workarea(gdk_display_get_monitor_at_window (display, root_window), &workarea);
102 *top = CLAMP (*top, 0, workarea.height - MINIMUM_ON_SCREEN_HEIGHT);
103
104 /* FIXME bugzilla.eazel.com 669:
105 * If window has negative left coordinate, set_uposition sends it
106 * somewhere else entirely. Not sure what level contains this bug (XWindows?).
107 * Hacked around by pinning the left edge to zero, which just means you
108 * can't set a window to be partly off the left of the screen using
109 * this routine.
110 */
111 /* Make sure the left edge of the window isn't off the right edge of
112 * the screen, or so close to the right edge that it might be
113 * obscured by the panel.
114 */
115 *left = CLAMP (*left, 0, workarea.width - MINIMUM_ON_SCREEN_WIDTH);
116 }
117}

119static void
120sanity_check_window_dimensions (guint *width, guint *height)
121{
122 g_assert (width != NULL);
123 g_assert (height != NULL);
124
125 /* Pin the size of the window to the screen, so we don't end up in
126 * a state where the window is so big essential parts of it can't
127 * be reached (might not be necessary with all window managers,
128 * but seems reasonable anyway).
129 */
130 *width = MIN ((int)*width, gdk_screen_width());
131 *height = MIN ((int)*height, gdk_screen_height());
132 }

150void
151eel_gtk_window_set_initial_geometry (GtkWindow *window,
152 EelGdkGeometryFlags geometry_flags,
153 int left,
154 int top,
155 guint width,
156 guint height)
157{
158 GdkScreen *screen;
159 int real_left, real_top;
160 int screen_width, screen_height;
161
162 g_return_if_fail (GTK_IS_WINDOW (window));
163
164 /* Setting the default size doesn't work when the window is already showing.
165 * Someday we could make this move an already-showing window, but we don't
166 * need that functionality yet.
167 */
168 g_return_if_fail (!gtk_widget_get_visible (GTK_WIDGET (window)));
169
170 if ((geometry_flags & EEL_GDK_X_VALUE) && (geometry_flags & EEL_GDK_Y_VALUE))
171 {
172 real_left = left;
173 real_top = top;
174
175 screen = gtk_window_get_screen (window);
176 screen_width = gdk_screen_get_width (screen);
177 screen_height = gdk_screen_get_height (screen);
178
179 /* This is sub-optimal. GDK doesn't allow us to set win_gravity
180 * to South/East types, which should be done if using negative
181 * positions (so that the right or bottom edge of the window
182 * appears at the specified position, not the left or top).
183 * However it does seem to be consistent with other MATE apps.
184 */
185 if (geometry_flags & EEL_GDK_X_NEGATIVE)
186 {
187 real_left = screen_width - real_left;
188 }
189 if (geometry_flags & EEL_GDK_Y_NEGATIVE)
190 {
191 real_top = screen_height - real_top;
192 }
193
194 sanity_check_window_position (&real_left, &real_top);
195 gtk_window_move (window, real_left, real_top);
196 }
197
198 if ((geometry_flags & EEL_GDK_WIDTH_VALUE) && (geometry_flags & EEL_GDK_HEIGHT_VALUE))
199 {
200 sanity_check_window_dimensions (&width, &height);
201 gtk_window_set_default_size (GTK_WINDOW (window), (int)width, (int)height);
202 }
203}

222void
223eel_gtk_window_set_initial_geometry_from_string (GtkWindow *window,
224 const char *geometry_string,
225 guint minimum_width,
226 guint minimum_height,
227 gboolean ignore_position)
228{
229 int left, top;
230 guint width, height;
231 EelGdkGeometryFlags geometry_flags;
232
233 g_return_if_fail (GTK_IS_WINDOW (window));
234 g_return_if_fail (geometry_string != NULL);
235
236 /* Setting the default size doesn't work when the window is already showing.
237 * Someday we could make this move an already-showing window, but we don't
238 * need that functionality yet.
239 */
240 g_return_if_fail (!gtk_widget_get_visible (GTK_WIDGET (window)));
241
242 geometry_flags = eel_gdk_parse_geometry (geometry_string, &left, &top, &width, &height);
243
244 /* Make sure the window isn't smaller than makes sense for this window.
245 * Other sanity checks are performed in set_initial_geometry.
246 */
247 if (geometry_flags & EEL_GDK_WIDTH_VALUE)
248 {
249 width = MAX (width, minimum_width);
250 }
251 if (geometry_flags & EEL_GDK_HEIGHT_VALUE)
252 {
253 height = MAX (height, minimum_height);
254 }
255
256 /* Ignore saved window position if requested. */
257 if (ignore_position)
258 {
259 geometry_flags &= ~(EEL_GDK_X_VALUE | EEL_GDK_Y_VALUE);
260 }
261
262 eel_gtk_window_set_initial_geometry (window, geometry_flags, left, top, width, height);
263}

500gboolean
501eel_notebook_scroll_event_cb (GtkWidget *widget,
502 GdkEventScroll *event)
503{
504 GtkNotebook *notebook = GTK_NOTEBOOK (widget);
505 GtkWidget *child, *event_widget, *action_widget;
506
507 child = gtk_notebook_get_nth_page (notebook, gtk_notebook_get_current_page (notebook));
508 if (child == NULL)
509 return FALSE;
510
511 event_widget = gtk_get_event_widget ((GdkEvent*) event);
512
513 /* Ignore scroll events from the content of the page */
514 if (event_widget == NULL || event_widget == child || gtk_widget_is_ancestor (event_widget, child))
515 return FALSE;
516
517 /* And also from the action widgets */
518 action_widget = gtk_notebook_get_action_widget (notebook, GTK_PACK_START);
519 if (event_widget == action_widget || (action_widget != NULL && gtk_widget_is_ancestor (event_widget, action_widget)))
520 return FALSE;
521
522 action_widget = gtk_notebook_get_action_widget (notebook, GTK_PACK_END);
523 if (event_widget == action_widget || (action_widget != NULL && gtk_widget_is_ancestor (event_widget, action_widget)))
524 return FALSE;
525
526 switch (event->direction) {
527 case GDK_SCROLL_RIGHT:
528 case GDK_SCROLL_DOWN:
529 gtk_notebook_next_page (notebook);
530 break;
531 case GDK_SCROLL_LEFT:
532 case GDK_SCROLL_UP:
533 gtk_notebook_prev_page (notebook);
534 break;
535 case GDK_SCROLL_SMOOTH:
536 switch (gtk_notebook_get_tab_pos (notebook)) {
537 case GTK_POS_LEFT:
538 case GTK_POS_RIGHT:
539 if (event->delta_y > 0)
540 gtk_notebook_next_page (notebook);
541 else if (event->delta_y < 0)
542 gtk_notebook_prev_page (notebook);
543 break;
544 case GTK_POS_TOP:
545 case GTK_POS_BOTTOM:
546 if (event->delta_x > 0)
547 gtk_notebook_next_page (notebook);
548 else if (event->delta_x < 0)
549 gtk_notebook_prev_page (notebook);
550 break;
551 }
552 break;
553 }
554
555 return TRUE;
556}
