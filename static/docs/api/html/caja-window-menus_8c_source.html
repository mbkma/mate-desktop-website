67/* Struct that stores all the info necessary to activate a bookmark. */
68typedef struct
69{
70 CajaBookmark *bookmark;
71 CajaWindow *window;
72 guint changed_handler_id;
73 CajaBookmarkFailedCallback failed_callback;
74} BookmarkHolder;
75
76static BookmarkHolder *
77bookmark_holder_new (CajaBookmark *bookmark,
78 CajaWindow *window,
79 GCallback refresh_callback,
80 CajaBookmarkFailedCallback failed_callback)
81{
82 BookmarkHolder *new_bookmark_holder;
83
84 new_bookmark_holder = g_new (BookmarkHolder, 1);
85 new_bookmark_holder->window = window;
86 new_bookmark_holder->bookmark = bookmark;
87 new_bookmark_holder->failed_callback = failed_callback;
88 /* Ref the bookmark because it might be unreffed away while
89 * we're holding onto it (not an issue for window).
90 */
91 g_object_ref (bookmark);
92 new_bookmark_holder->changed_handler_id =
93 g_signal_connect_object (bookmark, "appearance_changed",
94 refresh_callback,
95 window, G_CONNECT_SWAPPED);
96
97 return new_bookmark_holder;
98}
99
100static void
101bookmark_holder_free (BookmarkHolder *bookmark_holder)
102{
103 if (g_signal_handler_is_connected(bookmark_holder->bookmark,
104 bookmark_holder->changed_handler_id)){
105 g_signal_handler_disconnect (bookmark_holder->bookmark,
106 bookmark_holder->changed_handler_id);
107 }
108 g_object_unref (bookmark_holder->bookmark);
109 g_free (bookmark_holder);
110}

112static void
113bookmark_holder_free_cover (gpointer callback_data, GClosure *closure)
114{
115 bookmark_holder_free (callback_data);
116}

118static gboolean
119should_open_in_new_tab (void)
120{
121 /* FIXME this is duplicated */
122 GdkEvent *event;
123
124 event = gtk_get_current_event ();
125
126 if (event == NULL)
127 {
128 return FALSE;
129 }
130
131 if (event->type == GDK_BUTTON_PRESS || event->type == GDK_BUTTON_RELEASE)
132 {
133 return event->button.button == 2;
134 }
135
136 gdk_event_free (event);
137
138 return FALSE;
139}

141static void
142activate_bookmark_in_menu_item (GtkAction *action, gpointer user_data)
143{
144 BookmarkHolder *holder;
145
146 holder = (BookmarkHolder *)user_data;
147
148 if (caja_bookmark_uri_known_not_to_exist (holder->bookmark))
149 {
150 holder->failed_callback (holder->window, holder->bookmark);
151 }
152 else
153 {
154 CajaWindowSlot *slot;
155 GFile *location;
156
157 location = caja_bookmark_get_location (holder->bookmark);
158 slot = caja_window_get_active_slot (holder->window);
159 caja_window_slot_go_to (slot,
160 location,
161 should_open_in_new_tab ());
162 g_object_unref (location);
163 }
164}

166void
167caja_menus_append_bookmark_to_menu (CajaWindow *window,
168 CajaBookmark *bookmark,
169 const char *parent_path,
170 const char *parent_id,
171 guint index_in_parent,
172 GtkActionGroup *action_group,
173 guint merge_id,
174 GCallback refresh_callback,
175 CajaBookmarkFailedCallback failed_callback)
176{
177 BookmarkHolder *bookmark_holder;
178 char action_name[128];
179 char *name;
180 char *path;
181 cairo_surface_t *surface;
182 GtkAction *action;
183 GtkWidget *menuitem;
184
185 g_assert (CAJA_IS_WINDOW (window));
186 g_assert (CAJA_IS_BOOKMARK (bookmark));
187
188 bookmark_holder = bookmark_holder_new (bookmark, window, refresh_callback, failed_callback);
189 name = caja_bookmark_get_name (bookmark);
190
191 /* Create menu item with surface */
192 surface = caja_bookmark_get_surface (bookmark, GTK_ICON_SIZE_MENU);
193
194 g_snprintf (action_name, sizeof (action_name), "%s%d", parent_id, index_in_parent);
195
196 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
197 action = gtk_action_new (action_name,
198 name,
199 _("Go to the location specified by this bookmark"),
200 NULL);
201 G_GNUC_END_IGNORE_DEPRECATIONS;
202
203 g_object_set_data_full (G_OBJECT (action), "menu-icon",
204 cairo_surface_reference (surface),
205 (GDestroyNotify)cairo_surface_destroy);
206
207 g_signal_connect_data (action, "activate",
208 G_CALLBACK (activate_bookmark_in_menu_item),
209 bookmark_holder,
210 bookmark_holder_free_cover, 0);
211
212 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
213 gtk_action_group_add_action (action_group,
214 GTK_ACTION (action));
215 G_GNUC_END_IGNORE_DEPRECATIONS;
216
217 g_object_unref (action);
218
219 gtk_ui_manager_add_ui (window->details->ui_manager,
220 merge_id,
221 parent_path,
222 action_name,
223 action_name,
224 GTK_UI_MANAGER_MENUITEM,
225 FALSE);
226
227 path = g_strdup_printf ("%s/%s", parent_path, action_name);
228 menuitem = gtk_ui_manager_get_widget (window->details->ui_manager,
229 path);
230 gtk_image_menu_item_set_always_show_image (GTK_IMAGE_MENU_ITEM (menuitem),
231 TRUE);
232
233 cairo_surface_destroy (surface);
234 g_free (path);
235 g_free (name);
236}

548 GKeyFile *key_file;
549 GError *error = NULL;

670static GtkWidget *
671get_event_widget (GtkWidget *proxy)
672{
673 GtkWidget *widget;
674
680 if (GTK_IS_MENU_ITEM (proxy))
681 {
682 /* Menu items already forward middle clicks */
683 widget = NULL;
684 }
685 else if (GTK_IS_MENU_TOOL_BUTTON (proxy))
686 {
687 widget = eel_gtk_menu_tool_button_get_button (GTK_MENU_TOOL_BUTTON (proxy));
688 }
689 else if (GTK_IS_TOOL_BUTTON (proxy))
690 {
691 /* The tool button's button is the direct child */
692 widget = gtk_bin_get_child (GTK_BIN (proxy));
693 }
694 else if (GTK_IS_BUTTON (proxy))
695 {
696 widget = proxy;
697 }
698 else
699 {
700 /* Don't touch anything we don't know about */
701 widget = NULL;
702 }
703
704 return widget;
705}
