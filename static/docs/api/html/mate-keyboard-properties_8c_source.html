65static void
66setup_dialog (GtkBuilder * dialog)
67{
68 gchar *monitor;
69
70 g_settings_bind (keyboard_settings, "repeat",
71 gtk_builder_get_object (dialog, "repeat_toggle"), "active",
72 G_SETTINGS_BIND_DEFAULT);
73 g_settings_bind (keyboard_settings, "repeat",
74 gtk_builder_get_object (dialog, "repeat_table"), "sensitive",
75 G_SETTINGS_BIND_DEFAULT);
76 g_settings_bind (keyboard_settings, "delay",
77 gtk_range_get_adjustment (GET_RANGE ("repeat_delay_scale")), "value",
78 G_SETTINGS_BIND_DEFAULT);
79 g_settings_bind (keyboard_settings, "rate",
80 gtk_range_get_adjustment (GET_RANGE ("repeat_speed_scale")), "value",
81 G_SETTINGS_BIND_DEFAULT);
82 g_settings_bind (interface_settings, "cursor-blink",
83 gtk_builder_get_object (dialog, "cursor_toggle"), "active",
84 G_SETTINGS_BIND_DEFAULT);
85 g_settings_bind (interface_settings, "cursor-blink",
86 gtk_builder_get_object (dialog, "cursor_hbox"), "sensitive",
87 G_SETTINGS_BIND_DEFAULT);
88 g_settings_bind (interface_settings, "cursor-blink-time",
89 gtk_range_get_adjustment (GET_RANGE ("cursor_blink_time_scale")), "value",
90 G_SETTINGS_BIND_DEFAULT);
91
92 /* Ergonomics */
93 monitor = g_find_program_in_path ("mate-typing-monitor");
94 if (monitor != NULL) {
95 g_free (monitor);
96
97 g_settings_bind (typing_break_settings, "enabled",
98 gtk_builder_get_object (dialog, "break_enabled_toggle"), "active",
99 G_SETTINGS_BIND_DEFAULT);
100 g_settings_bind (typing_break_settings, "enabled",
101 gtk_builder_get_object (dialog, "break_details_table"), "sensitive",
102 G_SETTINGS_BIND_DEFAULT);
103 g_settings_bind (typing_break_settings, "type-time",
104 gtk_builder_get_object (dialog, "break_enabled_spin"), "value",
105 G_SETTINGS_BIND_DEFAULT);
106 g_settings_bind (typing_break_settings, "break-time",
107 gtk_builder_get_object (dialog, "break_interval_spin"), "value",
108 G_SETTINGS_BIND_DEFAULT);
109 g_settings_bind (typing_break_settings, "allow-postpone",
110 gtk_builder_get_object (dialog, "break_postponement_toggle"), "active",
111 G_SETTINGS_BIND_DEFAULT);
112
113 } else {
114 /* don't show the typing break tab if the daemon is not available */
115 GtkNotebook *nb = GTK_NOTEBOOK (gtk_builder_get_object (dialog, "keyboard_notebook"));
116 gint tb_page = gtk_notebook_page_num (nb, WID ("break_enabled_toggle"));
117 gtk_notebook_remove_page (nb, tb_page);
118 }
119
120 g_signal_connect (gtk_builder_get_object (dialog, "keyboard_dialog"), "response",
121 (GCallback) dialog_response,
122 NULL);
123
124 setup_xkb_tabs (dialog);
125 setup_a11y_tabs (dialog);
126}

128int
129main (int argc, char **argv)
130{
131 GtkBuilder *dialog;
132 GOptionContext *context;
133
134 static gboolean apply_only = FALSE;
135 static gboolean switch_to_typing_break_page = FALSE;
136 static gboolean switch_to_a11y_page = FALSE;
137
138 static GOptionEntry cap_options[] = {
139 {"apply", 0, 0, G_OPTION_ARG_NONE, &apply_only,
140 N_
141 ("Just apply settings and quit (compatibility only; now handled by daemon)"),
142 NULL},
143 {"init-session-settings", 0, 0, G_OPTION_ARG_NONE,
144 &apply_only,
145 N_
146 ("Just apply settings and quit (compatibility only; now handled by daemon)"),
147 NULL},
148 {"typing-break", 0, 0, G_OPTION_ARG_NONE,
149 &switch_to_typing_break_page,
150 N_
151 ("Start the page with the typing break settings showing"),
152 NULL},
153 {"a11y", 0, 0, G_OPTION_ARG_NONE,
154 &switch_to_a11y_page,
155 N_
156 ("Start the page with the accessibility settings showing"),
157 NULL},
158 { NULL, 0, 0, G_OPTION_ARG_NONE, NULL, NULL, NULL }
159 };
160
161 context = g_option_context_new (_("- MATE Keyboard Preferences"));
162 g_option_context_add_main_entries (context, cap_options,
163 GETTEXT_PACKAGE);
164
165 capplet_init (context, &argc, &argv);
166 g_option_context_free (context);
167
168 activate_settings_daemon ();
169
170 keyboard_settings = g_settings_new (KEYBOARD_SCHEMA);
171 interface_settings = g_settings_new (INTERFACE_SCHEMA);
172 typing_break_settings = g_settings_new (TYPING_BREAK_SCHEMA);
173
174 dialog = gtk_builder_new_from_resource ("/org/mate/mcc/keyboard/mate-keyboard-properties-dialog.ui");
175
176 setup_dialog (dialog);
177
178 GtkNotebook* nb = GTK_NOTEBOOK (gtk_builder_get_object (dialog, "keyboard_notebook"));
179 gtk_widget_add_events (GTK_WIDGET (nb), GDK_SCROLL_MASK);
180 g_signal_connect (nb, "scroll-event",
181 G_CALLBACK (capplet_notebook_scroll_event_cb),
182 NULL);
183
184 if (switch_to_typing_break_page) {
185 gtk_notebook_set_current_page (nb, 4);
186 }
187 else if (switch_to_a11y_page) {
188 gtk_notebook_set_current_page (nb, 2);
189
190 }
191
192 capplet_set_icon (WID ("keyboard_dialog"),
193 "input-keyboard");
194 gtk_widget_show (WID ("keyboard_dialog"));
195 gtk_main ();
196
197 finalize_a11y_tabs ();
198 g_object_unref (keyboard_settings);
199 g_object_unref (interface_settings);
200 g_object_unref (typing_break_settings);
201
202 return 0;
203}
