156gboolean
157cact_tree_model_dnd_idrag_dest_drag_data_received( GtkTreeDragDest *drag_dest, GtkTreePath *dest, GtkSelectionData *selection_data )
158{
159 static const gchar *thisfn = "cact_tree_model_dnd_idrag_dest_drag_data_received";
160 gboolean result = FALSE;
161 CactTreeModel *model;
162 gchar *atom_name;
163 guint info;
164 gchar *path_str;
165 GdkAtom selection_data_selection;
166 GdkAtom selection_data_target;
167 GdkAtom selection_data_type;
168 gint selection_data_format;
169 gint selection_data_length;
170
171 g_return_val_if_fail( CACT_IS_TREE_MODEL( drag_dest ), result );
172
173 model = CACT_TREE_MODEL( drag_dest );
174
175 if( !model->private->dispose_has_run ){
176
177 g_debug( "%s: drag_dest=%p (ref_count=%d), dest=%p, selection_data=%p",
178 thisfn,
179 ( void * ) drag_dest, G_OBJECT( drag_dest )->ref_count,
180 ( void * ) dest,
181 ( void * ) selection_data );
182
183 selection_data_selection = gtk_selection_data_get_selection( selection_data );
184 atom_name = gdk_atom_name( selection_data_selection );
185 g_debug( "%s: selection=%s", thisfn, atom_name );
186 g_free( atom_name );
187
188 selection_data_target = gtk_selection_data_get_target( selection_data );
189 atom_name = gdk_atom_name( selection_data_target );
190 g_debug( "%s: target=%s", thisfn, atom_name );
191 g_free( atom_name );
192
193 selection_data_type = gtk_selection_data_get_data_type( selection_data );
194 atom_name = gdk_atom_name( selection_data_type );
195 g_debug( "%s: type=%s", thisfn, atom_name );
196 g_free( atom_name );
197
198 selection_data_format = gtk_selection_data_get_format( selection_data );
199 selection_data_length = gtk_selection_data_get_length( selection_data );
200 g_debug( "%s: format=%d, length=%d", thisfn, selection_data_format, selection_data_length );
201
202 info = target_atom_to_id( selection_data_type );
203 g_debug( "%s: info=%u", thisfn, info );
204
205 path_str = gtk_tree_path_to_string( dest );
206 g_debug( "%s: dest_path=%s", thisfn, path_str );
207 g_free( path_str );
208
209 switch( info ){
210 case CACT_XCHANGE_FORMAT_CACT:
211 result = drop_inside( model, dest, selection_data );
212 break;
213
214 /* drop some actions from outside
215 * most probably from the file manager as a list of uris
216 */
217 case CACT_XCHANGE_FORMAT_URI_LIST:
218 result = drop_uri_list( model, dest, selection_data );
219 break;
220
221 default:
222 break;
223 }
224 }
225
226 return( result );
227}

283gboolean
284cact_tree_model_dnd_imulti_drag_source_drag_data_get( EggTreeMultiDragSource *drag_source,
285 GdkDragContext *context,
286 GtkSelectionData *selection_data,
287 GList *rows,
288 guint info )
289{
290 static const gchar *thisfn = "cact_tree_model_dnd_imulti_drag_source_drag_data_get";
291 gchar *atom_name;
292 CactTreeModel *model;
293 gchar *data;
294 gboolean ret = FALSE;
295 gchar *dest_folder, *folder;
296 gboolean is_writable;
297 gboolean copy_data;
298 GdkAtom selection_data_target;
299 GdkDragAction context_suggested_action;
300 GdkDragAction context_selected_action;
301
302 selection_data_target = gtk_selection_data_get_target( selection_data );
303
304 context_suggested_action = gdk_drag_context_get_suggested_action( context );
305 context_selected_action = gdk_drag_context_get_selected_action( context );
306
307 atom_name = gdk_atom_name( selection_data_target );
308 g_debug( "%s: drag_source=%p (ref_count=%d), context=%p, suggested action=%d, selection_data=%p, rows=%p (count=%d), atom=%s",
309 thisfn,
310 ( void * ) drag_source, G_OBJECT( drag_source )->ref_count,
311 ( void * ) context,
312 ( int ) context_suggested_action,
313 ( void * ) selection_data,
314 ( void * ) rows, g_list_length( rows ),
315 atom_name );
316 g_free( atom_name );
317
318 model = CACT_TREE_MODEL( drag_source );
319 g_return_val_if_fail( model->private->window, FALSE );
320
321 if( !model->private->dispose_has_run ){
322
323 if( !rows || !g_list_length( rows )){
324 return( FALSE );
325 }
326
327 switch( info ){
328 case CACT_XCHANGE_FORMAT_CACT:
329 copy_data = ( context_selected_action == GDK_ACTION_COPY );
330 gtk_selection_data_set( selection_data,
331 selection_data_target, 8, ( guchar * ) "", 0 );
332 cact_clipboard_dnd_set( model->private->clipboard, info, rows, NULL, copy_data );
333 ret = TRUE;
334 break;
335
336 case CACT_XCHANGE_FORMAT_XDS:
337 /* get the dest default filename as an uri
338 * e.g. file:///home/pierre/data/eclipse/caja-actions/trash/xds.txt
339 */
340 folder = get_xds_atom_value( context );
341 dest_folder = g_path_get_dirname( folder );
342
343 /* check that target folder is writable
344 */
345 is_writable = na_core_utils_dir_is_writable_uri( dest_folder );
346 g_debug( "%s: dest_folder=%s, is_writable=%s", thisfn, dest_folder, is_writable ? "True":"False" );
347 gtk_selection_data_set( selection_data,
348 selection_data_target, 8, ( guchar * )( is_writable ? "S" : "F" ), 1 );
349
350 if( is_writable ){
351 cact_clipboard_dnd_set( model->private->clipboard, info, rows, dest_folder, TRUE );
352 }
353
354 g_free( dest_folder );
355 g_free( folder );
356 ret = is_writable;
357 break;
358
359 case CACT_XCHANGE_FORMAT_APPLICATION_XML:
360 case CACT_XCHANGE_FORMAT_TEXT_PLAIN:
361 data = cact_clipboard_dnd_get_text( model->private->clipboard, rows );
362 gtk_selection_data_set( selection_data,
363 selection_data_target, 8, ( guchar * ) data, strlen( data ));
364 g_free( data );
365 ret = TRUE;
366 break;
367
368 default:
369 break;
370 }
371 }
372
373 return( ret );
374}

539/*
540 * called when a drop occurs in the treeview for a move/copy inside of
541 * the tree
542 *
543 * Returns: %TRUE if the specified rows were successfully inserted at
544 * the given dest, %FALSE else.
545 */
546static gboolean
547drop_inside( CactTreeModel *model, GtkTreePath *dest, GtkSelectionData *selection_data )
548{
549 static const gchar *thisfn = "cact_tree_model_dnd_inside_drag_and_drop";
550 CactApplication *application;
551 NAUpdater *updater;
552 CactMainWindow *main_window;
553 NAObjectItem *parent;
554 gboolean copy_data;
555 GList *rows;
556 GtkTreePath *new_dest;
557 GtkTreePath *path;
558 NAObject *current;
559 NAObject *inserted;
560 GList *object_list, *it;
561 GtkTreeIter iter;
562 GList *deletable;
563 gboolean relabel;
564 CactTreeView *items_view;
565
566 application = CACT_APPLICATION( base_window_get_application( model->private->window ));
567 updater = cact_application_get_updater( application );
568
569 g_return_val_if_fail( CACT_IS_MAIN_WINDOW( model->private->window ), FALSE );
570 main_window = CACT_MAIN_WINDOW( model->private->window );
571 items_view = cact_main_window_get_items_view( main_window );
572
573 /*
574 * CACT format (may embed profiles, or not)
575 * with profiles: only valid dest is inside an action
576 * without profile: only valid dest is outside (besides of) an action
577 */
578 parent = NULL;
579 rows = cact_clipboard_dnd_get_data( model->private->clipboard, &copy_data );
580
581 if( !is_drop_possible( model, dest, &parent )){
582 return( FALSE );
583 }
584
585 new_dest = gtk_tree_path_copy( dest );
586 if( !copy_data ){
587 drop_inside_move_dest( model, rows, &new_dest );
588 }
589
590 g_debug( "%s: rows has %d items, copy_data=%s", thisfn, g_list_length( rows ), copy_data ? "True":"False" );
591 object_list = NULL;
592 for( it = rows ; it ; it = it->next ){
593 path = gtk_tree_row_reference_get_path(( GtkTreeRowReference * ) it->data );
594 if( path ){
595 if( gtk_tree_model_get_iter( GTK_TREE_MODEL( model ), &iter, path )){
596 gtk_tree_model_get( GTK_TREE_MODEL( model ), &iter, TREE_COLUMN_NAOBJECT, &current, -1 );
597 g_object_unref( current );
598
599 if( copy_data ){
600 inserted = ( NAObject * ) na_object_duplicate( current, DUPLICATE_REC );
601 na_object_set_origin( inserted, NULL );
602 na_object_check_status( inserted );
603 relabel = na_updater_should_pasted_be_relabeled( updater, inserted );
604
605 } else {
606 inserted = na_object_ref( current );
607 deletable = g_list_prepend( NULL, inserted );
608 cact_tree_ieditable_delete( CACT_TREE_IEDITABLE( items_view ), deletable, TREE_OPE_MOVE );
609 g_list_free( deletable );
610 relabel = FALSE;
611 }
612
613 na_object_prepare_for_paste( inserted, relabel, copy_data, parent );
614 object_list = g_list_prepend( object_list, inserted );
615 g_debug( "%s: dropped=%s", thisfn, na_object_get_label( inserted ));
616 }
617 gtk_tree_path_free( path );
618 }
619 }
620 object_list = g_list_reverse( object_list );
621
622 cact_tree_ieditable_insert_at_path( CACT_TREE_IEDITABLE( items_view ), object_list, new_dest );
623
624 na_object_free_items( object_list );
625 gtk_tree_path_free( new_dest );
626
627 g_list_foreach( rows, ( GFunc ) gtk_tree_row_reference_free, NULL );
628 g_list_free( rows );
629
630 return( TRUE );
631}

633/*
634 * is a drop possible at given dest ?
635 *
636 * the only case where we would be led to have to modify the dest if
637 * we'd want be able to drop a profile into another profile, accepting
638 * it, actually dropping the profile just before the target
639 *
640 * -> it appears both clearer for the user interface and easyer from a
641 * code point of view to just refuse to drop a profile into a profile
642 *
643 * so this function is just to check if a drop is possible at the given
644 * dest
645 */
646static gboolean
647is_drop_possible( CactTreeModel *model, GtkTreePath *dest, NAObjectItem **parent )
648{
649 gboolean drop_ok;
650 CactApplication *application;
651 CactMainWindow *main_window;
652 GtkTreeIter iter;
653 NAObjectItem *parent_dest;
654
655 drop_ok = FALSE;
656 parent_dest = NULL;
657 application = CACT_APPLICATION( base_window_get_application( model->private->window ));
658
659 g_return_val_if_fail( CACT_IS_MAIN_WINDOW( model->private->window ), FALSE );
660 main_window = CACT_MAIN_WINDOW( model->private->window );
661
662 /* if we can have an iter on given dest, then the dest already exists
663 * so dropped items should be of the same type that already existing
664 */
665 if( gtk_tree_model_get_iter( GTK_TREE_MODEL( model ), &iter, dest )){
666 drop_ok = is_drop_possible_before_iter( model, &iter, main_window, &parent_dest );
667
668 /* inserting at the end of the list
669 * parent_dest is NULL
670 */
671 } else if( gtk_tree_path_get_depth( dest ) == 1 ){
672

679 }
680

826/*
827 * called when a drop from the outside occurs in the treeview
828 *
829 * Returns: %TRUE if the specified rows were successfully inserted at
830 * the given dest, %FALSE else.
831 *
832 * URI format only involves actions or menus
833 * so ony valid dest in outside (besides of) an action
834 */
835static gboolean
836drop_uri_list( CactTreeModel *model, GtkTreePath *dest, GtkSelectionData *selection_data )
837{
838 static const gchar *thisfn = "cact_tree_model_dnd_drop_uri_list";
839 gboolean drop_done;
840 CactApplication *application;
841 NAUpdater *updater;
842 CactMainWindow *main_window;
843 NAImporterParms parms;
844 GList *import_results, *it;
845 guint count;
846 GSList *im;
847 GList *imported, *overridden;
848 const gchar *selection_data_data;
849 CactTreeView *view;
850 GSList *messages;
851 gchar *dlg_message;
852 GtkWidget *dialog;
853
854 gchar *dest_str = gtk_tree_path_to_string( dest );
855 g_debug( "%s: model=%p, dest=%p (%s), selection_data=%p",
856 thisfn, ( void * ) model, ( void * ) dest, dest_str, ( void * ) selection_data );
857 g_free( dest_str );
858
859 drop_done = FALSE;
860 model->private->drag_has_profiles = FALSE;
861
862 if( !is_drop_possible( model, dest, NULL )){
863 return( FALSE );
864 }
865
866 application = CACT_APPLICATION( base_window_get_application( model->private->window ));
867 updater = cact_application_get_updater( application );
868
869 g_return_val_if_fail( CACT_IS_MAIN_WINDOW( model->private->window ), FALSE );
870 main_window = CACT_MAIN_WINDOW( model->private->window );
871
872 selection_data_data = ( const gchar * ) gtk_selection_data_get_data( selection_data );
873 g_debug( "%s", selection_data_data );
874
875 memset( &parms, '\0', sizeof( NAImporterParms ));
876 parms.uris = g_slist_reverse( na_core_utils_slist_from_split( selection_data_data, "\r\n" ));
877 parms.check_fn = ( NAImporterCheckFn ) is_dropped_already_exists;
878 parms.check_fn_data = main_window;
879 parms.preferred_mode = 0;
880 parms.parent_toplevel = base_window_get_gtk_toplevel( BASE_WINDOW( main_window ));
881
882 import_results = na_importer_import_from_uris( NA_PIVOT( updater ), &parms );
883
884 /* analysing output results, simultaneously building a concatenation
885 * of all lines of messages, and the list of imported items
886 */
887 imported = NULL;
888 overridden = NULL;
889 messages = NULL;
890
891 for( it = import_results ; it ; it = it->next ){
892 NAImporterResult *result = ( NAImporterResult * ) it->data;
893
894 for( im = result->messages ; im ; im = im->next ){
895 messages = g_slist_prepend( messages, im->data );
896 }
897 if( result->imported ){
898 if( !result->exist || result->mode == IMPORTER_MODE_RENUMBER ){
899 imported = g_list_prepend( imported, result->imported );
900 na_updater_check_item_writability_status( updater, result->imported );
901
902 } else if( result->mode == IMPORTER_MODE_OVERRIDE ){
903 overridden = g_list_prepend( overridden, result->imported );
904 }
905 }
906 }
907
908 /* if there is more than one message, display them in a dialog box
909 * else in the status bar
910 */
911 count = g_slist_length( messages );
912 g_debug( "%s: count=%d", thisfn, count );
913 if( count == 1 ){
914 cact_main_statusbar_display_with_timeout(
915 main_window, TREE_MODEL_STATUSBAR_CONTEXT, messages->data );
916 }
917 if( count > 1 ){
918 dlg_message = na_core_utils_slist_join_at_end( messages, "\n" );
919 g_debug( "%s: dlg_message='%s'", thisfn, dlg_message );
920 dialog = gtk_message_dialog_new(
921 parms.parent_toplevel,
922 GTK_DIALOG_MODAL, GTK_MESSAGE_WARNING, GTK_BUTTONS_CLOSE,
923 "%s", _( "Some messages have occurred during drop operation." ));
924 gtk_message_dialog_format_secondary_markup( GTK_MESSAGE_DIALOG( dialog ), "%s", dlg_message );
925 gtk_dialog_run( GTK_DIALOG( dialog ));
926 gtk_widget_destroy( dialog );
927 g_free( dlg_message );
928 }
929 g_slist_free( messages );
930
931 /* insert newly imported items in the list view
932 */
933 if( imported ){
934 na_object_dump_tree( imported );
935 view = cact_main_window_get_items_view( main_window );
936 cact_tree_ieditable_insert_at_path( CACT_TREE_IEDITABLE( view ), imported, dest );
937 }
938
939 /* override items if needed
940 * they may safely be released after having updated the store
941 */
942 if( overridden ){
943 na_object_dump_tree( overridden );
944 view = cact_main_window_get_items_view( main_window );
945 cact_tree_ieditable_set_items( CACT_TREE_IEDITABLE( view ), overridden );
946 na_object_free_items( overridden );
947 }
948
949 drop_done = TRUE;
950 na_object_free_items( imported );
951 na_object_free_items( overridden );
952 na_core_utils_slist_free( parms.uris );
953
954 for( it = import_results ; it ; it = it->next ){
955 na_importer_free_result( it->data );
956 }
957 g_list_free( import_results );
958
959 return( drop_done );
960}
