64static gchar *
65get_overwrite_mode_string (gboolean overwrite)
66{
67 return g_strconcat (" ", overwrite ? _("OVR") : _("INS"), NULL);
68}
69
70static gint
71get_overwrite_mode_length (void)
72{
73 return 2 + MAX (g_utf8_strlen (_("OVR"), -1), g_utf8_strlen (_("INS"), -1));
74}

76static void
77pluma_statusbar_dispose (GObject *object)
78{
79 PlumaStatusbar *statusbar = PLUMA_STATUSBAR (object);
80
81 if (statusbar->priv->flash_timeout > 0)
82 {
83 g_source_remove (statusbar->priv->flash_timeout);
84 statusbar->priv->flash_timeout = 0;
85 }
86
87 G_OBJECT_CLASS (pluma_statusbar_parent_class)->dispose (object);
88}

102{
103 GtkWidget *hbox;
104 GtkWidget *error_image;
105
106 statusbar->priv = pluma_statusbar_get_instance_private (statusbar);
107

110
111 statusbar->priv->overwrite_mode_label = gtk_label_new (NULL);
112 gtk_label_set_width_chars (GTK_LABEL (statusbar->priv->overwrite_mode_label),
113 get_overwrite_mode_length ());
114 gtk_widget_show (statusbar->priv->overwrite_mode_label);
115 gtk_box_pack_end (GTK_BOX (statusbar),
116 statusbar->priv->overwrite_mode_label,
117 FALSE, TRUE, 0);
118
119 statusbar->priv->cursor_position_label = gtk_label_new (NULL);
120 gtk_label_set_width_chars (GTK_LABEL (statusbar->priv->cursor_position_label),
121 CURSOR_POSITION_LABEL_WIDTH_CHARS);
122 gtk_widget_show (statusbar->priv->cursor_position_label);
123 gtk_box_pack_end (GTK_BOX (statusbar),
124 statusbar->priv->cursor_position_label,
125 FALSE, TRUE, 0);
126
127 statusbar->priv->state_frame = gtk_frame_new (NULL);
128 gtk_frame_set_shadow_type (GTK_FRAME (statusbar->priv->state_frame),
129 GTK_SHADOW_IN);
130
131 hbox = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);

141 statusbar->priv->load_image,
142 FALSE, TRUE, 4);
143 gtk_box_pack_start (GTK_BOX (hbox),
144 statusbar->priv->save_image,
145 FALSE, TRUE, 4);
146 gtk_box_pack_start (GTK_BOX (hbox),
147 statusbar->priv->print_image,
148 FALSE, TRUE, 4);
149
150 gtk_box_pack_start (GTK_BOX (statusbar),
151 statusbar->priv->state_frame,
152 FALSE, TRUE, 0);
153
154 statusbar->priv->error_frame = gtk_frame_new (NULL);

165 FALSE);
166 gtk_widget_show (statusbar->priv->error_event_box);
167
168 gtk_container_add (GTK_CONTAINER (statusbar->priv->error_frame),
169 statusbar->priv->error_event_box);
170 gtk_container_add (GTK_CONTAINER (statusbar->priv->error_event_box),
171 error_image);
172
173 gtk_box_pack_start (GTK_BOX (statusbar),
174 statusbar->priv->error_frame,
175 FALSE, TRUE, 0);
176
177 gtk_box_reorder_child (GTK_BOX (statusbar),
178 statusbar->priv->error_frame,
179 0);

233void
234pluma_statusbar_set_cursor_position (PlumaStatusbar *statusbar,
235 gint line,
236 gint col)
237{
238 gchar *msg = NULL;
239
240 g_return_if_fail (PLUMA_IS_STATUSBAR (statusbar));
241
242 if ((line >= 0) || (col >= 0))
243 {
244 /* Translators: "Ln" is an abbreviation for "Line", Col is an abbreviation for "Column". Please,
245 use abbreviations if possible to avoid space problems. */
246 msg = g_strdup_printf (_(" Ln %d, Col %d"), line, col);
247 }
248
249 gtk_label_set_text (GTK_LABEL (statusbar->priv->cursor_position_label), msg);
250
251 g_free (msg);
252}

254static gboolean
255remove_message_timeout (PlumaStatusbar *statusbar)
256{
257 gtk_statusbar_remove (GTK_STATUSBAR (statusbar),
258 statusbar->priv->flash_context_id,
259 statusbar->priv->flash_message_id);
260
261 /* remove the timeout */
262 statusbar->priv->flash_timeout = 0;
263 return FALSE;
264}

275void
276pluma_statusbar_flash_message (PlumaStatusbar *statusbar,
277 guint context_id,
278 const gchar *format, ...)
279{
280 const guint32 flash_length = 3000; /* three seconds */
281 va_list args;
282 gchar *msg;
283
284 g_return_if_fail (PLUMA_IS_STATUSBAR (statusbar));
285 g_return_if_fail (format != NULL);
286
287 va_start (args, format);
288 msg = g_strdup_vprintf (format, args);
289 va_end (args);
290
291 /* remove a currently ongoing flash message */
292 if (statusbar->priv->flash_timeout > 0)
293 {
294 g_source_remove (statusbar->priv->flash_timeout);
295 statusbar->priv->flash_timeout = 0;
296
297 gtk_statusbar_remove (GTK_STATUSBAR (statusbar),
298 statusbar->priv->flash_context_id,
299 statusbar->priv->flash_message_id);
300 }
301
302 statusbar->priv->flash_context_id = context_id;
303 statusbar->priv->flash_message_id = gtk_statusbar_push (GTK_STATUSBAR (statusbar),
304 context_id,
305 msg);
306
307 statusbar->priv->flash_timeout = g_timeout_add (flash_length,
308 (GSourceFunc) remove_message_timeout,
309 statusbar);
310
311 g_free (msg);
312}

314void
315pluma_statusbar_set_window_state (PlumaStatusbar *statusbar,
316 PlumaWindowState state,
317 gint num_of_errors)
318{
319 g_return_if_fail (PLUMA_IS_STATUSBAR (statusbar));
320
321 gtk_widget_hide (statusbar->priv->state_frame);
322 gtk_widget_hide (statusbar->priv->save_image);
323 gtk_widget_hide (statusbar->priv->load_image);
324 gtk_widget_hide (statusbar->priv->print_image);
325
326 if (state & PLUMA_WINDOW_STATE_SAVING)
327 {
328 gtk_widget_show (statusbar->priv->state_frame);
329 gtk_widget_show (statusbar->priv->save_image);
330 }
331 if (state & PLUMA_WINDOW_STATE_LOADING)
332 {
333 gtk_widget_show (statusbar->priv->state_frame);
334 gtk_widget_show (statusbar->priv->load_image);
335 }
336
337 if (state & PLUMA_WINDOW_STATE_PRINTING)
338 {
339 gtk_widget_show (statusbar->priv->state_frame);
340 gtk_widget_show (statusbar->priv->print_image);
341 }
342
343 if (state & PLUMA_WINDOW_STATE_ERROR)
344 {
345 gchar *tip;
346
347 tip = g_strdup_printf (ngettext("There is a tab with errors",
348 "There are %d tabs with errors",
349 num_of_errors),
350 num_of_errors);
351
352 gtk_widget_set_tooltip_text (statusbar->priv->error_event_box,
353 tip);
354 g_free (tip);
355
356 gtk_widget_show (statusbar->priv->error_frame);
357 }
358 else
359 {
360 gtk_widget_hide (statusbar->priv->error_frame);
361 }
362}
