38static void
39ev_image_finalize (GObject *object)
40{
41 EvImage *image = EV_IMAGE (object);
42
43 if (image->priv->pixbuf) {
44 g_object_unref (image->priv->pixbuf);
45 image->priv->pixbuf = NULL;
46 }
47
48 if (image->priv->tmp_uri) {
49 gchar *filename;
50
51 filename = g_filename_from_uri (image->priv->tmp_uri, NULL, NULL);
52 ev_tmp_filename_unlink (filename);
53 g_free (filename);
54 g_free (image->priv->tmp_uri);
55 image->priv->tmp_uri = NULL;
56 }
57
58 (* G_OBJECT_CLASS (ev_image_parent_class)->finalize) (object);
59}

128const gchar *
129ev_image_save_tmp (EvImage *image,
130 GdkPixbuf *pixbuf)
131{
132 GError *error = NULL;
133 gchar *filename = NULL;
134 int fd;
135
136 g_return_val_if_fail (EV_IS_IMAGE (image), NULL);
137 g_return_val_if_fail (GDK_IS_PIXBUF (pixbuf), NULL);
138
139 if (image->priv->tmp_uri)
140 return image->priv->tmp_uri;
141
142 if ((fd = ev_mkstemp ("image.XXXXXX.png", &filename, &error)) == -1)
143 goto had_error;
144
145 gdk_pixbuf_save (pixbuf, filename,
146 "png", &error,
147 "compression", "3", NULL);
148 close (fd);
149
150 if (!error) {
151 image->priv->tmp_uri = g_filename_to_uri (filename, NULL, &error);
152 if (image->priv->tmp_uri == NULL)
153 goto had_error;
154
155 g_free (filename);
156
157 return image->priv->tmp_uri;
158 }
159
160 had_error:
161
162 /* Erro saving image */
163 g_warning ("Error saving image: %s", error->message);
164 g_error_free (error);
165 g_free (filename);
166
167 return NULL;
168}
