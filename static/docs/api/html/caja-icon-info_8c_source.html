71static void
72pixbuf_toggle_notify (gpointer info,
73 GObject *object,
74 gboolean is_last_ref)
75{
76 CajaIconInfo *icon = info;
77
78 if (is_last_ref)
79 {
80 icon->sole_owner = TRUE;
81 g_object_remove_toggle_ref (object,
82 pixbuf_toggle_notify,
83 info);
84 icon->last_use_time = g_get_monotonic_time ();
85 schedule_reap_cache ();
86 }
87}

89static void
90caja_icon_info_finalize (GObject *object)
91{
92 CajaIconInfo *icon;
93
94 icon = CAJA_ICON_INFO (object);
95
96 if (!icon->sole_owner && icon->pixbuf)
97 {
98 g_object_remove_toggle_ref (G_OBJECT (icon->pixbuf),
99 pixbuf_toggle_notify,
100 icon);
101 }
102
103 if (icon->pixbuf)
104 {
105 g_object_unref (icon->pixbuf);
106 }
107 g_free (icon->attach_points);
108 g_free (icon->display_name);
109 g_free (icon->icon_name);
110
111 G_OBJECT_CLASS (caja_icon_info_parent_class)->finalize (object);
112}

143static CajaIconInfo *
144caja_icon_info_new_for_icon_info (GtkIconInfo *icon_info,
145 gint scale)
146{
147 CajaIconInfo *icon;
148 GdkPoint *points;
149 gint n_points;
150 const char *filename;
151 char *basename;
152
153 icon = g_object_new (CAJA_TYPE_ICON_INFO, NULL);
154
155 icon->pixbuf = gtk_icon_info_load_icon (icon_info, NULL);
156
157 icon->got_embedded_rect = gtk_icon_info_get_embedded_rect (icon_info,
158 &icon->embedded_rect);
159
160 if (gtk_icon_info_get_attach_points (icon_info, &points, &n_points))
161 {
162 icon->n_attach_points = n_points;
163 icon->attach_points = points;
164 }
165
166 icon->display_name = g_strdup (gtk_icon_info_get_display_name (icon_info));
167
168 filename = gtk_icon_info_get_filename (icon_info);
169 if (filename != NULL)
170 {
171 char *p;
172
173 basename = g_path_get_basename (filename);
174 p = strrchr (basename, '.');
175 if (p)
176 {
177 *p = 0;
178 }
179 icon->icon_name = basename;
180 }
181
182 icon->orig_scale = scale;
183
184 return icon;
185}

187typedef struct
188{
189 GIcon *icon;
190 int scale;
191 int size;

194static GHashTable *loadable_icon_cache = NULL;
195static GHashTable *themed_icon_cache = NULL;
196static guint reap_cache_timeout = 0;
197
198#define MICROSEC_PER_SEC ((guint64)1000000L)
199
200static guint time_now;
201
202static gboolean
203reap_old_icon (gpointer key,
204 gpointer value,
205 gpointer user_info)
206{
207 CajaIconInfo *icon = value;
208 gboolean *reapable_icons_left = user_info;
209
210 if (icon->sole_owner)
211 {
212 if (time_now - icon->last_use_time > 30 * MICROSEC_PER_SEC)
213 {
214 /* This went unused 30 secs ago. reap */
215 return TRUE;
216 }
217 else
218 {
219 /* We can reap this soon */
220 *reapable_icons_left = TRUE;
221 }
222 }
223
224 return FALSE;
225}
226
227static gboolean
228reap_cache (gpointer data)
229{
230 gboolean reapable_icons_left;
231
232 reapable_icons_left = TRUE;
233
234 time_now = g_get_monotonic_time ();
235
236 if (loadable_icon_cache)
237 {
238 g_hash_table_foreach_remove (loadable_icon_cache,
239 reap_old_icon,
240 &reapable_icons_left);
241 }
242
243 if (themed_icon_cache)
244 {
245 g_hash_table_foreach_remove (themed_icon_cache,
246 reap_old_icon,
247 &reapable_icons_left);
248 }
249
250 if (reapable_icons_left)
251 {
252 return TRUE;
253 }
254 else
255 {
256 reap_cache_timeout = 0;
257 return FALSE;
258 }
259}
260
261static void
262schedule_reap_cache (void)
263{
264 if (reap_cache_timeout == 0)
265 {
266 reap_cache_timeout = g_timeout_add_seconds_full (0, 5,
267 reap_cache,
268 NULL, NULL);
269 }
270}

323CajaIconInfo *
324caja_icon_info_lookup (GIcon *icon,
325 int size,
326 int scale)
327{
328 GtkIconTheme *icon_theme;
329 GtkIconInfo *gtkicon_info;
330
331 CajaIconInfo *icon_info;
332
333 icon_theme = gtk_icon_theme_get_default ();
334
335 if (G_IS_LOADABLE_ICON (icon)) {
336 GdkPixbuf *pixbuf;
337 IconKey lookup_key;
338 IconKey *key;
339 GInputStream *stream;
340
341 if (loadable_icon_cache == NULL) {
342 loadable_icon_cache = g_hash_table_new_full ((GHashFunc) icon_key_hash,
343 (GEqualFunc) icon_key_equal,
344 (GDestroyNotify) icon_key_free,
345 (GDestroyNotify) g_object_unref);
346 }
347
348 lookup_key.icon = icon;
349 lookup_key.scale = scale;
350 lookup_key.size = size * scale;
351
352 icon_info = g_hash_table_lookup (loadable_icon_cache, &lookup_key);
353 if (icon_info)
354 {
355 return g_object_ref (icon_info);
356 }
357
358 pixbuf = NULL;
359 stream = g_loadable_icon_load (G_LOADABLE_ICON (icon),
360 size * scale,
361 NULL, NULL, NULL);
362 if (stream)
363 {
364 pixbuf = gdk_pixbuf_new_from_stream_at_scale (stream,
365 size * scale, size * scale,
366 TRUE,
367 NULL, NULL);
368 g_input_stream_close (stream, NULL, NULL);
369 g_object_unref (stream);
370 }
371
372 if (!pixbuf) {
373 gtkicon_info = gtk_icon_theme_lookup_icon_for_scale (icon_theme,
374 "text-x-generic",
375 size,
376 scale,
377 GTK_ICON_LOOKUP_FORCE_SIZE);
378 pixbuf = gtk_icon_info_load_icon (gtkicon_info, NULL);
379 }
380
381 icon_info = caja_icon_info_new_for_pixbuf (pixbuf, scale);
382
383 key = icon_key_new (icon, scale, size);
384 g_hash_table_insert (loadable_icon_cache, key, icon_info);
385 g_clear_object (&pixbuf);
386
387 return g_object_ref (icon_info);
388 } else {
389 IconKey lookup_key;
390 IconKey *key;
391 if (themed_icon_cache == NULL) {
392 themed_icon_cache = g_hash_table_new_full ((GHashFunc) icon_key_hash,
393 (GEqualFunc) icon_key_equal,
394 (GDestroyNotify) icon_key_free,
395 (GDestroyNotify) g_object_unref);
396 }
397 lookup_key.icon = icon;
398 lookup_key.scale = scale;
399 lookup_key.size = size;
400
401 icon_info = g_hash_table_lookup (themed_icon_cache, &lookup_key);
402 if (icon_info) {
403 return g_object_ref (icon_info);
404 }
405
406 gtkicon_info = NULL;
407
408 gtkicon_info = gtk_icon_theme_lookup_by_gicon_for_scale (icon_theme,
409 icon,
410 size,
411 scale,
412 GTK_ICON_LOOKUP_FORCE_SIZE);
413
414 if (!gtkicon_info) {
415 gtkicon_info = gtk_icon_theme_lookup_icon_for_scale (icon_theme,
416 "text-x-generic",
417 size,
418 scale,
419 GTK_ICON_LOOKUP_FORCE_SIZE);
420 }
421
422 icon_info = caja_icon_info_new_for_icon_info (gtkicon_info, scale);
423 g_object_unref (gtkicon_info);
424
425 key = icon_key_new (icon,scale, size);
426 g_hash_table_insert (themed_icon_cache, key, icon_info);
427
428 return g_object_ref (icon_info);
429 }
430
431}

537GdkPixbuf *
538caja_icon_info_get_pixbuf_nodefault_at_size (CajaIconInfo *icon,
539 gsize forced_size)
540{
541 GdkPixbuf *pixbuf, *scaled_pixbuf;
542 int w, h, s;
543 double scale;
544
545 pixbuf = caja_icon_info_get_pixbuf_nodefault (icon);
546
547 if (pixbuf == NULL)
548 return NULL;
549
550 w = gdk_pixbuf_get_width (pixbuf) / icon->orig_scale;
551 h = gdk_pixbuf_get_height (pixbuf) / icon->orig_scale;
552 s = MAX (w, h);
553 if (s == forced_size)
554 {
555 return pixbuf;
556 }
557
558 scale = (double)forced_size / s;
559 scaled_pixbuf = gdk_pixbuf_scale_simple (pixbuf,
560 w * scale, h * scale,
561 GDK_INTERP_BILINEAR);
562 g_object_unref (pixbuf);
563 return scaled_pixbuf;
564}

654/* Return nominal icon size for given zoom level.
655 * @zoom_level: zoom level for which to find matching icon size.

801GIcon *
802caja_user_special_directory_get_gicon (GUserDirectory directory)
803{
804
805 #define ICON_CASE(x) \
806 case G_USER_DIRECTORY_ ## x:\
807 return g_themed_icon_new (CAJA_ICON_FOLDER_ ## x);
808
809 switch (directory) {
810
811 ICON_CASE (DESKTOP);
812 ICON_CASE (DOCUMENTS);
813 ICON_CASE (DOWNLOAD);
814 ICON_CASE (MUSIC);
815 ICON_CASE (PICTURES);
816 ICON_CASE (PUBLIC_SHARE);
817 ICON_CASE (TEMPLATES);
818 ICON_CASE (VIDEOS);
