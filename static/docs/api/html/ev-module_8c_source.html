67static gboolean
68ev_module_load (GTypeModule *gmodule)
69{
70 EvModule *module = EV_MODULE (gmodule);
71 EvModuleRegisterFunc register_func;
72
73 module->library = g_module_open (module->path, 0);
74
75 if (!module->library) {
76 g_warning ("%s", g_module_error ());
77
78 return FALSE;
79 }
80
81 /* extract symbols from the lib */
82 if (!g_module_symbol (module->library, "register_atril_backend",
83 (void *) &register_func)) {
84 g_warning ("%s", g_module_error ());
85 g_module_close (module->library);
86
87 return FALSE;
88 }
89
90 /* symbol can still be NULL even though g_module_symbol
91 * returned TRUE */
92 if (!register_func) {
93 g_warning ("Symbol 'register_atril_backend' should not be NULL");
94 g_module_close (module->library);
95
96 return FALSE;
97 }
98
99 module->type = register_func (gmodule);
100
101 if (module->type == 0) {
102 g_warning ("Invalid atril backend contained by module %s", module->path);
103
104 return FALSE;
105 }
106
107 if (module->resident)
108 g_module_make_resident (module->library);
109
110 return TRUE;
111}

166static void

168{

178EvModule *
179ev_module_new (const gchar *path,
180 gboolean resident)
181{
182 EvModule *result;
183
184 g_return_val_if_fail (path != NULL && path[0] != '\0', NULL);
185
186 result = g_object_new (EV_TYPE_MODULE, NULL);
187
188 g_type_module_set_name (G_TYPE_MODULE (result), path);
189 result->path = g_strdup (path);
190 result->resident = resident;
191
192 return result;
