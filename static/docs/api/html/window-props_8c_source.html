113void
114meta_window_reload_properties_from_xwindow (MetaWindow *window,
115 Window xwindow,
116 const Atom *properties,
117 int n_properties,
118 gboolean initial)
119{
120 int i;
121 MetaPropValue *values;
122
123 g_return_if_fail (properties != NULL);
124 g_return_if_fail (n_properties > 0);
125
126 values = g_new0 (MetaPropValue, n_properties);
127
128 for (i=0; i<n_properties; i++)
129 {
130 MetaWindowPropHooks *hooks = find_hooks (window->display,
131 properties[i]);
132
133 if (!hooks || hooks->type == META_PROP_VALUE_INVALID)
134 {
135 values[i].type = META_PROP_VALUE_INVALID;
136 values[i].atom = None;
137 }
138 else
139 {
140 values[i].type = hooks->type;
141 values[i].atom = properties[i];
142 }
143 }
144
145 meta_prop_get_values (window->display, xwindow,
146 values, n_properties);
147
148 for (i=0; i<n_properties; i++)
149 {
150 MetaWindowPropHooks *hooks = find_hooks (window->display,
151 properties[i]);
152
153 if (hooks && hooks->reload_func != NULL)
154 (* hooks->reload_func) (window, &values[i], initial);
155 }
156
157 meta_prop_free_values (values, n_properties);
158
159 g_free (values);
160}

298static pid_t
299get_local_pid (MetaWindow *window)
300{
301 pid_t pid;
302 XResClientIdSpec spec;
303 long num_ids;
304 XResClientIdValue *client_ids;
305 long i;
306
307 pid = -1;
308 if (window->display->have_xres == FALSE)
309 return pid;
310
311 spec.client = window->xwindow;
312 spec.mask = XRES_CLIENT_ID_PID_MASK;
313
314 if (XResQueryClientIds (window->display->xdisplay,
315 1,
316 &spec,
317 &num_ids,
318 &client_ids) != Success)
319 return pid;
320
321 for (i = 0; i < num_ids; i++)
322 {
323 pid = XResGetClientPid (&client_ids[i]);
324
325 if (pid != -1)
326 break;
327 }
328
329 XResClientIdsDestroy (num_ids, client_ids);
330
331 return pid;
332}

378static void
379reload_net_wm_user_time_window (MetaWindow *window,
380 MetaPropValue *value,
381 gboolean initial)
382{
383 if (value->type != META_PROP_VALUE_INVALID)
384 {
385 /* Unregister old NET_WM_USER_TIME_WINDOW */
386 if (window->user_time_window != None)
387 {
388 /* See the comment to the meta_display_register_x_window call below. */
389 meta_display_unregister_x_window (window->display,
390 window->user_time_window);
391 /* Don't get events on not-managed windows */
392 XSelectInput (window->display->xdisplay,
393 window->user_time_window,
394 NoEventMask);
395 }
396
397 /* Obtain the new NET_WM_USER_TIME_WINDOW and register it */
398 window->user_time_window = value->v.xwindow;
399 if (window->user_time_window != None)
400 {
401 /* Kind of a hack; display.c:event_callback() ignores events
402 * for unknown windows. We make window->user_time_window
403 * known by registering it with window (despite the fact
404 * that window->xwindow is already registered with window).
405 * This basically means that property notifies to either the
406 * window->user_time_window or window->xwindow will be
407 * treated identically and will result in functions for
408 * window being called to update it. Maybe we should ignore
409 * any property notifies to window->user_time_window other
410 * than atom__NET_WM_USER_TIME ones, but I just don't care
411 * and it's not specified in the spec anyway.
412 */
413 meta_display_register_x_window (window->display,
414 &window->user_time_window,
415 window);
416 /* Just listen for property notify events */
417 XSelectInput (window->display->xdisplay,
418 window->user_time_window,
419 PropertyChangeMask);
420
421 /* Manually load the _NET_WM_USER_TIME field from the given window
422 * at this time as well. If the user_time_window ever broadens in
423 * scope, we'll probably want to load all relevant properties here.
424 */
425 meta_window_reload_property_from_xwindow (
426 window,
427 window->user_time_window,
428 window->display->atom__NET_WM_USER_TIME,
429 initial);
430 }
431 }

442static gboolean
443owner_of_process (pid_t process, uid_t *result)
444{
445#ifdef HAVE_GTOP
446 glibtop_proc_uid process_details;
447
448 glibtop_get_proc_uid (&process_details, process);
449
450 if (process_details.flags & (1L << GLIBTOP_PROC_UID_UID))
451 {
452 *result = process_details.uid;
453 return TRUE;
454 }
455 else
456 {
457 return FALSE;
458 }
459#else
460 /* I don't know, maybe we could do something hairy like see whether
461 * /proc/$PID exists and who owns it, in case they have procfs.
462 */
463 return FALSE;
464#endif /* HAVE_GTOP */
465}

476static gboolean
477set_title_text (MetaWindow *window,
478 gboolean previous_was_modified,
479 const char *title,
480 Atom atom,
481 char **target)
482{
483 char hostname[HOST_NAME_MAX + 1];
484 gboolean modified = FALSE;
485
486 if (!target)
487 return FALSE;
488
489 g_free (*target);
490
491 if (!title || !*title)
492 *target = g_strdup ("");
493 else if (g_utf8_strlen (title, MAX_TITLE_LENGTH + 1) > MAX_TITLE_LENGTH)
494 {
495 *target = meta_g_utf8_strndup (title, MAX_TITLE_LENGTH);
496 modified = TRUE;
497 }
498 /* if WM_CLIENT_MACHINE indicates this machine is on a remote host
499 * let's place that hostname in the title */
500 else if (window->wm_client_machine &&
501 !gethostname (hostname, HOST_NAME_MAX + 1) &&
502 strcmp (hostname, window->wm_client_machine))
503 {
504 /* Translators: the title of a window from another machine */
505 *target = g_strdup_printf (_("%s (on %s)"),
506 title, window->wm_client_machine);
507 modified = TRUE;
508 }
509 else if (window->net_wm_pid != -1)
510 {
511 /* We know the process which owns this window; perhaps we can
512 * find out the name of its owner (if it's not us).
513 */
514
515 char *found_name = NULL;
516
517 uid_t window_owner = 0;
518 gboolean window_owner_known =
519 owner_of_process (window->net_wm_pid, &window_owner);
520
521 /* Assume a window with unknown ownership is ours (call it usufruct!) */
522 gboolean window_owner_is_us =
523 !window_owner_known || window_owner==getuid ();
524
525 if (window_owner_is_us)
526 {
527 /* we own it, so fall back to the simple case */
528 *target = g_strdup (title);
529 }
530 else
531 {
532 /* it belongs to window_owner. So what's their name? */
533
534 if (window_owner==0)
535 {
536 /* Simple case-- don't bother to look it up. It's root. */
537 *target = g_strdup_printf (_("%s (as superuser)"),
538 title);
539 }
540 else
541 {
542 /* Okay, let's look up the name. */
543 struct passwd *pwd;
544
545 errno = 0;
546 pwd = getpwuid (window_owner);
547 if (errno==0 && pwd!=NULL)
548 {
549 found_name = pwd->pw_name;
550 }
551
552 if (found_name)
553 /* Translators: the title of a window owned by another user
554 * on this machine */
555 *target = g_strdup_printf (_("%s (as %s)"),
556 title,
557 found_name);
558 else
559 /* Translators: the title of a window owned by another user
560 * on this machine, whose name we don't know */
561 *target = g_strdup_printf (_("%s (as another user)"),
562 title);
563 }
564 /* either way we changed it */
565 modified = TRUE;
566
567 }
568 }
569 else
570 *target = g_strdup (title);
571
572 if (modified && atom != None)
573 meta_prop_set_utf8_string_hint (window->display,
574 window->xwindow,
575 atom, *target);
576
577 /* Bug 330671 -- Don't forget to clear _NET_WM_VISIBLE_(ICON_)NAME */
578 if (!modified && previous_was_modified)
579 {
580 meta_error_trap_push (window->display);
581 XDeleteProperty (window->display->xdisplay,
582 window->xwindow,
583 atom);
584 meta_error_trap_pop (window->display, FALSE);
585 }
586
587 return modified;
588}

723static void
724reload_net_wm_state (MetaWindow *window,
725 MetaPropValue *value,
726 gboolean initial)
727{
728 int i;
729
730 /* We know this is only an initial window creation,
731 * clients don't change the property.
732 */
733
734 if (!initial) {
735 /* no, they DON'T change the property */
736 meta_verbose ("Ignoring _NET_WM_STATE: we should be the one who set "
737 "the property in the first place\n");
738 return;
739 }
740
741 window->shaded = FALSE;
742 window->maximized_horizontally = FALSE;
743 window->maximized_vertically = FALSE;
744 window->fullscreen = FALSE;
745 window->wm_state_modal = FALSE;
746 window->wm_state_skip_taskbar = FALSE;
747 window->wm_state_skip_pager = FALSE;
748 window->wm_state_above = FALSE;
749 window->wm_state_below = FALSE;
750 window->wm_state_demands_attention = FALSE;
751
752 if (value->type == META_PROP_VALUE_INVALID)
753 return;
754
755 i = 0;
756 while (i < value->v.atom_list.n_atoms)
757 {
758 if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_SHADED)
759 window->shaded = TRUE;
760 else if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_MAXIMIZED_HORZ)
761 window->maximize_horizontally_after_placement = TRUE;
762 else if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_MAXIMIZED_VERT)
763 window->maximize_vertically_after_placement = TRUE;
764 else if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_HIDDEN)
765 window->minimize_after_placement = TRUE;
766 else if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_MODAL)
767 window->wm_state_modal = TRUE;
768 else if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_SKIP_TASKBAR)
769 window->wm_state_skip_taskbar = TRUE;
770 else if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_SKIP_PAGER)
771 window->wm_state_skip_pager = TRUE;
772 else if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_FULLSCREEN)
773 window->fullscreen_after_placement = TRUE;
774 else if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_ABOVE)
775 window->wm_state_above = TRUE;
776 else if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_BELOW)
777 window->wm_state_below = TRUE;
778 else if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_DEMANDS_ATTENTION)
779 window->wm_state_demands_attention = TRUE;
780 else if (value->v.atom_list.atoms[i] == window->display->atom__NET_WM_STATE_STICKY)
781 window->on_all_workspaces = TRUE;
782
783 ++i;
784 }
785
786 meta_verbose ("Reloaded _NET_WM_STATE for %s\n",
787 window->desc);
788
789 meta_window_recalc_window_type (window);
790}

792static void
793reload_mwm_hints (MetaWindow *window,
794 MetaPropValue *value,
795 gboolean initial)
796{
797 MotifWmHints *hints;
798
799 window->mwm_decorated = TRUE;
800 window->mwm_border_only = FALSE;
801 window->mwm_has_close_func = TRUE;
802 window->mwm_has_minimize_func = TRUE;
803 window->mwm_has_maximize_func = TRUE;
804 window->mwm_has_move_func = TRUE;
805 window->mwm_has_resize_func = TRUE;
806
807 if (value->type == META_PROP_VALUE_INVALID)
808 {
809 meta_verbose ("Window %s has no MWM hints\n", window->desc);
810 meta_window_recalc_features (window);
811 return;
812 }
813
814 hints = value->v.motif_hints;
815
816 /* We support those MWM hints deemed non-stupid */
817
