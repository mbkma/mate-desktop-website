102void
103ev_sidebar_set_page (EvSidebar *ev_sidebar,
104 GtkWidget *main_widget)
105{
106 GtkTreeIter iter;
107 gboolean valid;
108
109 valid = gtk_tree_model_get_iter_first (ev_sidebar->priv->page_model, &iter);
110
111 while (valid) {
112 GtkWidget *widget;
113
114 gtk_tree_model_get (ev_sidebar->priv->page_model, &iter,
115 PAGE_COLUMN_MAIN_WIDGET, &widget,
116 -1);
117
118 if (widget == main_widget) {
119 ev_sidebar_select_page (ev_sidebar, &iter);
120 valid = FALSE;
121 } else {
122 valid = gtk_tree_model_iter_next (ev_sidebar->priv->page_model, &iter);
123 }
124 g_object_unref (widget);
125 }
126
127 g_object_notify (G_OBJECT (ev_sidebar), "current-page");
128}

195static gboolean
196ev_sidebar_select_button_press_cb (GtkWidget *widget,
197 GdkEventButton *event,
198 gpointer user_data)
199{
200 EvSidebar *ev_sidebar = EV_SIDEBAR (user_data);
201
202 if (event->button == 1) {
203 GtkRequisition requisition;
204 GtkAllocation allocation;
205 gint width;
206
207 gtk_widget_get_allocation (widget, &allocation);
208 width = allocation.width;
209 gtk_widget_set_size_request (ev_sidebar->priv->menu, -1, -1);
210 gtk_widget_get_preferred_size (ev_sidebar->priv->menu, &requisition, NULL);
211 gtk_widget_set_size_request (ev_sidebar->priv->menu,
212 MAX (width, requisition.width), -1);
213
214 gtk_widget_grab_focus (widget);
215
216 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (widget), TRUE);
217 gtk_menu_popup_at_widget (GTK_MENU (ev_sidebar->priv->menu),
218 widget,
219 GDK_GRAVITY_SOUTH_WEST,
220 GDK_GRAVITY_NORTH_WEST,
221 (const GdkEvent*) event);
222
223 return TRUE;
224 }
225
226 return FALSE;
227}

411void
412ev_sidebar_add_page (EvSidebar *ev_sidebar,
413 GtkWidget *main_widget)
414{
415 GtkTreeIter iter;
416 GtkWidget *menu_item;
417 gchar *label_title;
418 const gchar *title;
419 int index;
420
421 g_return_if_fail (EV_IS_SIDEBAR (ev_sidebar));
422 g_return_if_fail (EV_IS_SIDEBAR_PAGE (main_widget));
423 g_return_if_fail (GTK_IS_WIDGET (main_widget));
424
425 ev_sidebar_page_set_model (EV_SIDEBAR_PAGE (main_widget),
426 ev_sidebar->priv->model);
427 title = ev_sidebar_page_get_label (EV_SIDEBAR_PAGE (main_widget));
428
429 index = gtk_notebook_append_page (GTK_NOTEBOOK (ev_sidebar->priv->notebook),
430 main_widget, NULL);
431
432 menu_item = mate_image_menu_item_new_with_label (title);
433 g_signal_connect (menu_item, "activate",
434 G_CALLBACK (ev_sidebar_menu_item_activate_cb),
435 ev_sidebar);
436 gtk_widget_show (menu_item);
437 gtk_menu_shell_append (GTK_MENU_SHELL (ev_sidebar->priv->menu),
438 menu_item);
439
440 /* Insert and move to end */
441 gtk_list_store_insert_with_values (GTK_LIST_STORE (ev_sidebar->priv->page_model),
442 &iter, 0,
443 PAGE_COLUMN_TITLE, title,
444 PAGE_COLUMN_MENU_ITEM, menu_item,
445 PAGE_COLUMN_MAIN_WIDGET, main_widget,
446 PAGE_COLUMN_NOTEBOOK_INDEX, index,
447 -1);
448 gtk_list_store_move_before(GTK_LIST_STORE(ev_sidebar->priv->page_model),
449 &iter, NULL);
450
451 /* Set the first item added as active */
452 gtk_tree_model_get_iter_first (ev_sidebar->priv->page_model, &iter);
453 gtk_tree_model_get (ev_sidebar->priv->page_model,
454 &iter,
455 PAGE_COLUMN_TITLE, &label_title,
456 PAGE_COLUMN_NOTEBOOK_INDEX, &index,
457 -1);
458
459 gtk_menu_set_active (GTK_MENU (ev_sidebar->priv->menu), index);
460 gtk_label_set_text (GTK_LABEL (ev_sidebar->priv->label), label_title);
461 gtk_notebook_set_current_page (GTK_NOTEBOOK (ev_sidebar->priv->notebook),
462 index);
463 g_free (label_title);
464}

466static void
467ev_sidebar_document_changed_cb (EvDocumentModel *model,
468 GParamSpec *pspec,
469 EvSidebar *sidebar)
470{
471 EvSidebarPrivate *priv = sidebar->priv;
472 EvDocument *document = ev_document_model_get_document (model);
473 GtkTreeIter iter;
474 gboolean valid;
475 gboolean has_pages = FALSE;
476
477 for (valid = gtk_tree_model_get_iter_first (priv->page_model, &iter);
478 valid;
479 valid = gtk_tree_model_iter_next (priv->page_model, &iter)) {
480 GtkWidget *widget;
481 GtkWidget *menu_widget;
482
483 gtk_tree_model_get (priv->page_model, &iter,
484 PAGE_COLUMN_MAIN_WIDGET, &widget,
485 PAGE_COLUMN_MENU_ITEM, &menu_widget,
486 -1);
487
488 if (ev_sidebar_page_support_document (EV_SIDEBAR_PAGE (widget), document)) {
489 has_pages = TRUE;
490 } else {
491 gtk_widget_set_sensitive (menu_widget, FALSE);
492 }
493 g_object_unref (widget);
494 g_object_unref (menu_widget);
495 }
496
497 if (!has_pages) {
498 gtk_widget_hide (GTK_WIDGET (sidebar));
499 } else {
500 gtk_widget_set_sensitive (GTK_WIDGET (sidebar->priv->notebook), TRUE);
501 gtk_widget_set_sensitive (GTK_WIDGET (sidebar->priv->select_button), TRUE);
502 }
503}
