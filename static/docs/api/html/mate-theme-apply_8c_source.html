49void
50mate_meta_theme_set (MateThemeMetaInfo *meta_theme_info)
51{
52 GSettings *interface_settings;
53 GSettings *interface_gnome_settings = NULL;
54 GSettings *marco_settings;
55 GSettings *mouse_settings;
56 GSettings *notification_settings = NULL;
57 gchar *old_key;
58 gint old_key_int;
59
60 /*Load the gnome interface schema if we are running under wayland and it is present*/
61 if (!(GDK_IS_X11_DISPLAY (gdk_display_get_default())))
62 {
63 GSettingsSchemaSource *source = g_settings_schema_source_get_default ();
64
65 if (source)
66 {
67 GSettingsSchema *schema = g_settings_schema_source_lookup (source, INTERFACE_GNOME_SCHEMA, TRUE);
68
69 if (schema)
70 interface_gnome_settings = g_settings_new_full (schema, NULL, NULL);
71 }
72 }
73
74 interface_settings = g_settings_new (INTERFACE_SCHEMA);
75 marco_settings = g_settings_new (MARCO_SCHEMA);
76 mouse_settings = g_settings_new (MOUSE_SCHEMA);
77
78 if (mate_gsettings_schema_exists (NOTIFICATION_SCHEMA))
79 {
80 notification_settings = g_settings_new (NOTIFICATION_SCHEMA);
81 }
82
83 /* Set the gtk+ key */
84 old_key = g_settings_get_string (interface_settings, GTK_THEME_KEY);
85 if (compare (old_key, meta_theme_info->gtk_theme_name))
86 {
87 g_settings_set_string (interface_settings, GTK_THEME_KEY, meta_theme_info->gtk_theme_name);
88
89 if (interface_gnome_settings)
90 g_settings_set_string (interface_gnome_settings,
91 GTK_THEME_KEY, meta_theme_info->gtk_theme_name);
92
93 }
94 g_free (old_key);
95
96 /* Set the color scheme key */
97 old_key = g_settings_get_string (interface_settings, COLOR_SCHEME_KEY);
98 if (compare (old_key, meta_theme_info->gtk_color_scheme))
99 {
100 /* only save the color scheme if it differs from the default
101 scheme for the selected gtk theme */
102 gchar *newval, *gtkcols;
103
104 newval = meta_theme_info->gtk_color_scheme;
105 gtkcols = gtkrc_get_color_scheme_for_theme (meta_theme_info->gtk_theme_name);
106
107 if (newval == NULL || !strcmp (newval, "") ||
108 mate_theme_color_scheme_equal (newval, gtkcols))
109 {
110 g_settings_reset (interface_settings, COLOR_SCHEME_KEY);
111 }
112 else
113 {
114 g_settings_set_string (interface_settings, COLOR_SCHEME_KEY, newval);
115 }
116 g_free (gtkcols);
117 }
118 g_free (old_key);
119
120 /* Set the wm key */
121 g_settings_set_string (marco_settings, MARCO_THEME_KEY, meta_theme_info->marco_theme_name);
122 /*To also control decoration theme in wayland we need a decorator that uses marco themes
123 *and queries gsettings to determine which theme to use to this is a TODO */
124
125 /* set the icon theme */
126 old_key = g_settings_get_string (interface_settings, ICON_THEME_KEY);
127 if (compare (old_key, meta_theme_info->icon_theme_name))
128 {
129 g_settings_set_string (interface_settings, ICON_THEME_KEY, meta_theme_info->icon_theme_name);
130 if (interface_gnome_settings)
131 g_settings_set_string (interface_gnome_settings,
132 ICON_THEME_KEY, meta_theme_info->icon_theme_name);
133 }
134 g_free (old_key);
135
136 /* set the notification theme */
137 if (notification_settings != NULL)
138 {
139 if (meta_theme_info->notification_theme_name != NULL)
140 {
141 old_key = g_settings_get_string (notification_settings, NOTIFICATION_THEME_KEY);
142 if (compare (old_key, meta_theme_info->notification_theme_name))
143 {
144 g_settings_set_string (notification_settings, NOTIFICATION_THEME_KEY, meta_theme_info->notification_theme_name);
145 }
146 g_free (old_key);
147 }
148 }
149
150 /* Set the cursor theme key */
151 old_key = g_settings_get_string (mouse_settings, CURSOR_THEME_KEY);
152 if (compare (old_key, meta_theme_info->cursor_theme_name))
153 {
154 g_settings_set_string (mouse_settings, CURSOR_THEME_KEY, meta_theme_info->cursor_theme_name);
155 if (interface_gnome_settings)
156 {
157 /*Note that this key is in a different place in GNOME than it is in MATE*/
158 g_settings_set_string (interface_gnome_settings,
159 CURSOR_THEME_KEY, meta_theme_info->cursor_theme_name);
160 }
161 }
162
163 old_key_int = g_settings_get_int (mouse_settings, CURSOR_SIZE_KEY);
164 if (old_key_int != meta_theme_info->cursor_size)
165 {
166 g_settings_set_int (mouse_settings, CURSOR_SIZE_KEY, meta_theme_info->cursor_size);
167 }
168
169 g_free (old_key);
170 g_object_unref (interface_settings);
171 g_object_unref (marco_settings);
172 g_object_unref (mouse_settings);
173 if (notification_settings != NULL)
174 g_object_unref (notification_settings);
175}
