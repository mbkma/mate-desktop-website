41static gboolean
42session_manager_connect (void)
43{
44 GError *error = NULL;
45 if (sm_proxy == NULL) {
46 sm_proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SESSION,
47 G_DBUS_PROXY_FLAGS_NONE,
48 NULL,
49 SM_DBUS_NAME,
50 SM_DBUS_PATH,
51 SM_DBUS_INTERFACE,
52 NULL,
53 &error);
54 }
55 return (sm_proxy != NULL);
56}

64static gboolean
65do_inhibit_for_window (GdkWindow *window)
66{
67 GError *error;
68 GVariant *ret;
69 const char *startup_id G_GNUC_UNUSED;
70 const char *app_id;
71 const char *reason;
72 guint toplevel_xid;
73 guint flags;
74
75 startup_id = g_getenv ("DESKTOP_AUTOSTART_ID");
76 app_id = "caja-cd-burner";
77 reason = "A CD burn is in progress.";
78 toplevel_xid = gdk_x11_window_get_xid (window);
79
80 flags = GSM_INHIBITOR_FLAG_LOGOUT
81 | GSM_INHIBITOR_FLAG_SWITCH_USER
82 | GSM_INHIBITOR_FLAG_SUSPEND;
83
84 error = NULL;
85 ret = g_dbus_proxy_call_sync (sm_proxy,
86 "Inhibit",
87 g_variant_new ("(susu)", app_id, toplevel_xid, reason, flags),
88 G_DBUS_CALL_FLAGS_NONE,
89 -1,
90 NULL,
91 &error);
92 if (ret == NULL) {
93 g_warning ("Failed to inhibit: %s", error->message);
94 g_error_free (error);
95 return FALSE;
96 }
97
98 g_variant_get (ret, "(u)", &cookie);
99 g_variant_unref (ret);
100 g_debug ("Inhibiting session manager: %u", cookie);
101
102 return TRUE;
103}
