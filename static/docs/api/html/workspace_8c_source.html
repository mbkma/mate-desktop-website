53MetaWorkspace*
54meta_workspace_new (MetaScreen *screen)
55{
56 MetaWorkspace *workspace;
57
58 workspace = g_new (MetaWorkspace, 1);
59
60 workspace->screen = screen;
61 workspace->screen->workspaces =
62 g_list_append (workspace->screen->workspaces, workspace);
63 workspace->windows = NULL;
64 workspace->mru_list = NULL;
65 meta_screen_foreach_window (screen, maybe_add_to_list, &workspace->mru_list);
66
67 workspace->work_areas_invalid = TRUE;
68 workspace->work_area_xinerama = NULL;
69 workspace->work_area_screen.x = 0;
70 workspace->work_area_screen.y = 0;
71 workspace->work_area_screen.width = 0;
72 workspace->work_area_screen.height = 0;
73
74 workspace->screen_region = NULL;
75 workspace->xinerama_region = NULL;
76 workspace->screen_edges = NULL;
77 workspace->xinerama_edges = NULL;
78 workspace->list_containing_self = g_list_prepend (NULL, workspace);
79
80 workspace->all_struts = NULL;
81
82 workspace->showing_desktop = FALSE;
83
84 return workspace;
85}

110void
111meta_workspace_free (MetaWorkspace *workspace)
112{
113 GList *tmp;
114 MetaScreen *screen;
115 int i;
116
117 g_return_if_fail (workspace != workspace->screen->active_workspace);
118
119 /* Here we assume all the windows are already on another workspace
120 * as well, so they won't be "orphaned"
121 */
122
123 tmp = workspace->windows;
124 while (tmp != NULL)
125 {
126 GList *next;
127 MetaWindow *window = tmp->data;
128 next = tmp->next;
129
130 /* pop front of list we're iterating over */
131 meta_workspace_remove_window (workspace, window);
132 g_assert (window->workspace != NULL);
133
134 tmp = next;
135 }
136
137 g_assert (workspace->windows == NULL);
138
139 screen = workspace->screen;
140
141 workspace->screen->workspaces =
142 g_list_remove (workspace->screen->workspaces, workspace);
143
144 g_free (workspace->work_area_xinerama);
145
146 g_list_free (workspace->mru_list);
147 g_list_free (workspace->list_containing_self);
148
149 /* screen.c:update_num_workspaces(), which calls us, removes windows from
150 * workspaces first, which can cause the workareas on the workspace to be
151 * invalidated (and hence for struts/regions/edges to be freed).
152 * So, no point trying to double free it; that causes a crash
153 * anyway. #361804.
154 */
155
156 if (!workspace->work_areas_invalid)
157 {
158 workspace_free_struts (workspace);
159 for (i = 0; i < screen->n_xinerama_infos; i++)
160 g_list_free_full (workspace->xinerama_region[i], g_free);
161 g_free (workspace->xinerama_region);
162 g_list_free_full (workspace->screen_region, g_free);
163 g_list_free_full (workspace->screen_edges, g_free);
164 g_list_free_full (workspace->xinerama_edges, g_free);
165 }
166
167 g_free (workspace);
168
169 /* don't bother to reset names, pagers can just ignore
170 * extra ones
171 */
172}

174void
175meta_workspace_add_window (MetaWorkspace *workspace,
176 MetaWindow *window)
177{
178 g_return_if_fail (window->workspace == NULL);
179
180 /* If the window is on all workspaces, we want to add it to all mru
181 * lists, otherwise just add it to this workspaces mru list
182 */
183 if (window->on_all_workspaces)
184 {
185 if (window->workspace == NULL)
186 {
187 GList* tmp = window->screen->workspaces;
188 while (tmp)
189 {
190 MetaWorkspace* work = (MetaWorkspace*) tmp->data;
191 if (!g_list_find (work->mru_list, window))
192 work->mru_list = g_list_prepend (work->mru_list, window);
193
194 tmp = tmp->next;
195 }
196 }
197 }
198 else
199 {
200 g_assert (g_list_find (workspace->mru_list, window) == NULL);
201 workspace->mru_list = g_list_prepend (workspace->mru_list, window);
202 }
203
204 workspace->windows = g_list_prepend (workspace->windows, window);
205 window->workspace = workspace;
206
207 meta_window_set_current_workspace_hint (window);
208
209 if (window->struts)
210 {
211 meta_topic (META_DEBUG_WORKAREA,
212 "Invalidating work area of workspace %d since we're adding window %s to it\n",
213 meta_workspace_index (workspace), window->desc);
214 meta_workspace_invalidate_work_area (workspace);
215 }
216
217 /* queue a move_resize since changing workspaces may change
218 * the relevant struts
219 */
220 meta_window_queue (window, META_QUEUE_CALC_SHOWING|META_QUEUE_MOVE_RESIZE);
221
222 /* Update dynamic workspaces (this will add a new workspace if necessary) */
223 meta_screen_update_dynamic_workspaces (workspace->screen);
224}

226void
227meta_workspace_remove_window (MetaWorkspace *workspace,
228 MetaWindow *window)
229{
230 g_return_if_fail (window->workspace == workspace);
231
232 workspace->windows = g_list_remove (workspace->windows, window);
233 window->workspace = NULL;
234
235 /* If the window is on all workspaces, we don't want to remove it
236 * from the MRU list unless this causes it to be removed from all
237 * workspaces
238 */
239 if (window->on_all_workspaces)
240 {
241 GList* tmp = window->screen->workspaces;
242 while (tmp)
243 {
244 MetaWorkspace* work = (MetaWorkspace*) tmp->data;
245 work->mru_list = g_list_remove (work->mru_list, window);
246
247 tmp = tmp->next;
248 }
249 }
250 else
251 {
252 workspace->mru_list = g_list_remove (workspace->mru_list, window);
253 g_assert (g_list_find (workspace->mru_list, window) == NULL);
254 }
255
256 meta_window_set_current_workspace_hint (window);
257
258 if (window->struts)
259 {
260 meta_topic (META_DEBUG_WORKAREA,
261 "Invalidating work area of workspace %d since we're removing window %s from it\n",
262 meta_workspace_index (workspace), window->desc);
263 meta_workspace_invalidate_work_area (workspace);
264 }
265
266 /* queue a move_resize since changing workspaces may change
267 * the relevant struts
268 */
269 meta_window_queue (window, META_QUEUE_CALC_SHOWING|META_QUEUE_MOVE_RESIZE);
270
271 /* Update dynamic workspaces (this will remove an empty workspace if necessary) */
272 meta_screen_update_dynamic_workspaces (workspace->screen);
273}

275void
276meta_workspace_relocate_windows (MetaWorkspace *workspace,
277 MetaWorkspace *new_home)
278{
279 GList *tmp;
280 GList *copy;
281
282 g_return_if_fail (workspace != new_home);
283
284 /* can't modify list we're iterating over */
285 copy = g_list_copy (workspace->windows);
286
287 tmp = copy;
288 while (tmp != NULL)
289 {
290 MetaWindow *window = tmp->data;
291
292 meta_workspace_remove_window (workspace, window);
293 meta_workspace_add_window (new_home, window);
294
295 tmp = tmp->next;
296 }
297
298 g_list_free (copy);
299
300 g_assert (workspace->windows == NULL);
301}

317static void workspace_switch_sound(MetaWorkspace *from,
318 MetaWorkspace *to) {
319
320 MetaWorkspaceLayout layout;
321 int i, nw, x, y, fi, ti;
322 const char *e;
323
324 nw = meta_screen_get_n_workspaces(from->screen);
325 fi = meta_workspace_index(from);
326 ti = meta_workspace_index(to);
327
328 meta_screen_calc_workspace_layout(from->screen,
329 nw,
330 fi,
331 &layout);
332
333 for (i = 0; i < nw; i++)
334 if (layout.grid[i] == ti)
335 break;
336
337 if (i >= nw) {
338 meta_bug("Failed to find destination workspace in layout\n");
339 goto finish;
340 }
341
342 y = i / layout.cols;
343 x = i % layout.cols;
344
345 /* We priorize horizontal over vertical movements here. The
346 rationale for this is that horizontal movements are probably more
347 interesting for sound effects because speakers are usually
348 positioned on a horizontal and not a vertical axis. i.e. your
349 spatial "Woosh!" effects will easily be able to encode horizontal
350 movement but not such much vertical movement. */
351
352 if (x < layout.current_col)
353 e = "desktop-switch-left";
354 else if (x > layout.current_col)
355 e = "desktop-switch-right";
356 else if (y < layout.current_row)
357 e = "desktop-switch-up";
358 else if (y > layout.current_row)
359 e = "desktop-switch-down";
360 else {
361 meta_bug("Uh, origin and destination workspace at same logic position!\n");
362 goto finish;
363 }
364
365 ca_context_play(ca_gtk_context_get(), 1,
366 CA_PROP_EVENT_ID, e,
367 CA_PROP_EVENT_DESCRIPTION, "Desktop switched",
368 CA_PROP_CANBERRA_CACHE_CONTROL, "permanent",
369 NULL);
370
371 finish:
372 meta_screen_free_workspace_layout (&layout);
373}

375void
376meta_workspace_activate_with_focus (MetaWorkspace *workspace,
377 MetaWindow *focus_this,
378 guint32 timestamp)
379{
380 MetaWorkspace *old;
381 MetaWindow *move_window;
382
383 meta_verbose ("Activating workspace %d\n",
384 meta_workspace_index (workspace));
385
386 if (workspace->screen->active_workspace == workspace)
387 return;
388
389 if (workspace->screen->active_workspace)
390 workspace_switch_sound(workspace->screen->active_workspace, workspace);
391
392 /* Free any cached pointers to the workspaces's edges from
393 * a current resize or move operation */
394 meta_display_cleanup_edges (workspace->screen->display);
395
396 /* Note that old can be NULL; e.g. when starting up */
397 old = workspace->screen->active_workspace;
398
399 /* Save old workspace, to be able to switch back. */
400 workspace->screen->prev_workspace = old;
401
402 workspace->screen->active_workspace = workspace;
403
404 set_active_space_hint (workspace->screen);
405
406 /* If the "show desktop" mode is active for either the old workspace
407 * or the new one *but not both*, then update the
408 * _net_showing_desktop hint
409 */
410 if (old && (old->showing_desktop ^ workspace->showing_desktop))
411 meta_screen_update_showing_desktop_hint (workspace->screen);
412
413 if (old == NULL)
414 return;
415
416 move_window = NULL;
417 if (workspace->screen->display->grab_op == META_GRAB_OP_MOVING ||
418 workspace->screen->display->grab_op == META_GRAB_OP_KEYBOARD_MOVING)
419 move_window = workspace->screen->display->grab_window;
420
421 if (move_window != NULL)
422 {
423 if (move_window->on_all_workspaces)
424 move_window = NULL; /* don't move it after all */
425
426 /* We put the window on the new workspace, flip spaces,
427 * then remove from old workspace, so the window
428 * never gets unmapped and we maintain the button grab
429 * on it.
430 *
431 * \bug This comment appears to be the reverse of what happens
432 */
433 if (move_window && (move_window->workspace != workspace))
434 {
435 meta_workspace_remove_window (old, move_window);
436 meta_workspace_add_window (workspace, move_window);
437 }
438 }
439
440 meta_workspace_queue_calc_showing (old);
441 meta_workspace_queue_calc_showing (workspace);
442
443 /* FIXME: Why do we need this?!? Isn't it handled in the lines above? */
444 if (move_window)
445 /* Removes window from other spaces */
446 meta_window_change_workspace (move_window, workspace);
447
448 if (focus_this)
449 {
450 meta_window_focus (focus_this, timestamp);
451 meta_window_raise (focus_this);
452 }
453 else if (move_window)
454 {
455 meta_window_raise (move_window);
456 }
457 else
458 {
459 meta_topic (META_DEBUG_FOCUS, "Focusing default window on new workspace\n");
460 meta_workspace_focus_default_window (workspace, NULL, timestamp);
461 }
462}

484/* get windows contained on workspace, including workspace->windows
485 * and also sticky windows.
486 */
487GList*
488meta_workspace_list_windows (MetaWorkspace *workspace)
489{
490 GSList *display_windows;
491 GSList *tmp;
492 GList *workspace_windows;
493
494 display_windows = meta_display_list_windows (workspace->screen->display);
495
496 workspace_windows = NULL;
497 tmp = display_windows;
498 while (tmp != NULL)
499 {
500 MetaWindow *window = tmp->data;
501
502 if (meta_window_located_on_workspace (window, workspace))
503 workspace_windows = g_list_prepend (workspace_windows,
504 window);
505
506 tmp = tmp->next;
507 }
508
509 g_slist_free (display_windows);
510
511 return workspace_windows;
512}

514static void
515set_active_space_hint (MetaScreen *screen)
516{
517 unsigned long data[1];
518
519 /* this is because we destroy the spaces in order,
520 * so we always end up setting a current desktop of
521 * 0 when closing a screen, so lose the current desktop
522 * on restart. By doing this we keep the current
523 * desktop on restart.
524 */
525 if (screen->closing > 0)
526 return;
527
528 data[0] = meta_workspace_index (screen->active_workspace);
529
530 meta_verbose ("Setting _NET_CURRENT_DESKTOP to %lu\n", data[0]);
531
532 meta_error_trap_push (screen->display);
533 XChangeProperty (screen->display->xdisplay, screen->xroot,
534 screen->display->atom__NET_CURRENT_DESKTOP,
535 XA_CARDINAL,
536 32, PropModeReplace, (guchar*) data, 1);
537 meta_error_trap_pop (screen->display, FALSE);
538}

540void
541meta_workspace_invalidate_work_area (MetaWorkspace *workspace)
542{
543 GList *tmp;
544 GList *windows;
545 int i;
546
547 if (workspace->work_areas_invalid)
548 {
549 meta_topic (META_DEBUG_WORKAREA,
550 "Work area for workspace %d is already invalid\n",
551 meta_workspace_index (workspace));
552 return;
553 }
554
555 meta_topic (META_DEBUG_WORKAREA,
556 "Invalidating work area for workspace %d\n",
557 meta_workspace_index (workspace));
558
559 /* If we are in the middle of a resize or move operation, we
560 * might have cached pointers to the workspace's edges */
561 if (workspace == workspace->screen->active_workspace)
562 meta_display_cleanup_edges (workspace->screen->display);
563
564 g_free (workspace->work_area_xinerama);
565 workspace->work_area_xinerama = NULL;
566
567 workspace_free_struts (workspace);
568
569 for (i = 0; i < workspace->screen->n_xinerama_infos; i++)
570 g_list_free_full (workspace->xinerama_region[i], g_free);
571 g_free (workspace->xinerama_region);
572 g_list_free_full (workspace->screen_region, g_free);
573 g_list_free_full (workspace->screen_edges, g_free);
574 g_list_free_full (workspace->xinerama_edges, g_free);
575 workspace->xinerama_region = NULL;
576 workspace->screen_region = NULL;
577 workspace->screen_edges = NULL;
578 workspace->xinerama_edges = NULL;
579
580 workspace->work_areas_invalid = TRUE;
581
582 /* redo the size/position constraints on all windows */
583 windows = meta_workspace_list_windows (workspace);
584 tmp = windows;
585 while (tmp != NULL)
586 {
587 MetaWindow *w = tmp->data;
588
589 meta_window_queue (w, META_QUEUE_MOVE_RESIZE);
590
591 tmp = tmp->next;
592 }
593
594 g_list_free (windows);
595
596 meta_screen_queue_workarea_recalc (workspace->screen);
597}

599static void
600ensure_work_areas_validated (MetaWorkspace *workspace)
601{
602 GList *windows;
603 GList *tmp;
604 MetaRectangle work_area;
605 int i; /* C89 absolutely sucks... */
606
607 if (!workspace->work_areas_invalid)
608 return;
609
610 g_assert (workspace->all_struts == NULL);
611 g_assert (workspace->xinerama_region == NULL);
612 g_assert (workspace->screen_region == NULL);
613 g_assert (workspace->screen_edges == NULL);
614 g_assert (workspace->xinerama_edges == NULL);
615
616 /* STEP 1: Get the list of struts */
617 windows = meta_workspace_list_windows (workspace);
618 for (tmp = windows; tmp != NULL; tmp = tmp->next)
619 {
620 MetaWindow *win = tmp->data;
621 GSList *s_iter;
622
623 for (s_iter = win->struts; s_iter != NULL; s_iter = s_iter->next) {
624 MetaStrut *cpy = g_new (MetaStrut, 1);
625 *cpy = *((MetaStrut *)s_iter->data);
626 workspace->all_struts = g_slist_prepend (workspace->all_struts,
627 cpy);
628 }
629 }
630 g_list_free (windows);
631
632 /* STEP 2: Get the maximal/spanning rects for the onscreen and
633 * on-single-xinerama regions
634 */
635 g_assert (workspace->xinerama_region == NULL);
636 g_assert (workspace->screen_region == NULL);
637
638 workspace->xinerama_region = g_new (GList*,
639 workspace->screen->n_xinerama_infos);
640 for (i = 0; i < workspace->screen->n_xinerama_infos; i++)
641 {
642 workspace->xinerama_region[i] =
643 meta_rectangle_get_minimal_spanning_set_for_region (
644 &workspace->screen->xinerama_infos[i].rect,
645 workspace->all_struts,
646 FALSE);
647 }
648 workspace->screen_region =
649 meta_rectangle_get_minimal_spanning_set_for_region (
650 &workspace->screen->rect,
651 workspace->all_struts,
652 TRUE);
653
654 /* STEP 3: Get the work areas (region-to-maximize-to) for the screen and
655 * xineramas.
656 */
657 work_area = workspace->screen->rect; /* start with the screen */
658 if (workspace->screen_region == NULL)
659 work_area = meta_rect (0, 0, -1, -1);
660 else
661 meta_rectangle_clip_to_region (workspace->screen_region,
662 FIXED_DIRECTION_NONE,
663 &work_area);
664
665 /* Lots of paranoia checks, forcing work_area_screen to be sane */
666#define MIN_SANE_AREA 100
667 if (work_area.width < MIN_SANE_AREA)
668 {
669 meta_warning ("struts occupy an unusually large percentage of the screen; "
670 "available remaining width = %d < %d",
671 work_area.width, MIN_SANE_AREA);
672 if (work_area.width < 1)
673 {
674 work_area.x = (workspace->screen->rect.width - MIN_SANE_AREA)/2;
675 work_area.width = MIN_SANE_AREA;
676 }
677 else
678 {
679 int amount = (MIN_SANE_AREA - work_area.width)/2;
680 work_area.x -= amount;
681 work_area.width += 2*amount;
682 }
683 }
684 if (work_area.height < MIN_SANE_AREA)
685 {
686 meta_warning ("struts occupy an unusually large percentage of the screen; "
687 "available remaining height = %d < %d",
688 work_area.height, MIN_SANE_AREA);
689 if (work_area.height < 1)
690 {
691 work_area.y = (workspace->screen->rect.height - MIN_SANE_AREA)/2;
692 work_area.height = MIN_SANE_AREA;
693 }
694 else
695 {
696 int amount = (MIN_SANE_AREA - work_area.height)/2;
697 work_area.y -= amount;
698 work_area.height += 2*amount;
699 }
700 }
701 workspace->work_area_screen = work_area;
702 meta_topic (META_DEBUG_WORKAREA,
703 "Computed work area for workspace %d: %d,%d %d x %d\n",
704 meta_workspace_index (workspace),
705 workspace->work_area_screen.x,
706 workspace->work_area_screen.y,
707 workspace->work_area_screen.width,
708 workspace->work_area_screen.height);
709
710 /* Now find the work areas for each xinerama */
711 g_free (workspace->work_area_xinerama);
712 workspace->work_area_xinerama = g_new (MetaRectangle,
713 workspace->screen->n_xinerama_infos);
714
715 for (i = 0; i < workspace->screen->n_xinerama_infos; i++)
716 {
717 work_area = workspace->screen->xinerama_infos[i].rect;
718
719 if (workspace->xinerama_region[i] == NULL)
720 /* FIXME: constraints.c untested with this, but it might be nice for
721 * a screen reader or magnifier.
722 */
723 work_area = meta_rect (work_area.x, work_area.y, -1, -1);
724 else
725 meta_rectangle_clip_to_region (workspace->xinerama_region[i],
726 FIXED_DIRECTION_NONE,
727 &work_area);
728
729 workspace->work_area_xinerama[i] = work_area;
730 meta_topic (META_DEBUG_WORKAREA,
731 "Computed work area for workspace %d "
732 "xinerama %d: %d,%d %d x %d\n",
733 meta_workspace_index (workspace),
734 i,
735 workspace->work_area_xinerama[i].x,
736 workspace->work_area_xinerama[i].y,
737 workspace->work_area_xinerama[i].width,
738 workspace->work_area_xinerama[i].height);
739 }
740
741 /* STEP 4: Make sure the screen_region is nonempty (separate from step 2
742 * since it relies on step 3).
743 */
744 if (workspace->screen_region == NULL)
745 {
746 MetaRectangle *nonempty_region;
747 nonempty_region = g_new (MetaRectangle, 1);
748 *nonempty_region = workspace->work_area_screen;
749 workspace->screen_region = g_list_prepend (NULL, nonempty_region);
750 }
751
752 /* STEP 5: Cache screen and xinerama edges for edge resistance and snapping */

830MetaWorkspace*
831meta_workspace_get_neighbor (MetaWorkspace *workspace,
832 MetaMotionDirection direction)
833{
834 MetaWorkspaceLayout layout;
835 int i, current_space, num_workspaces;
836 gboolean ltr;
837 MetaWrapStyle wrap;
838
839 current_space = meta_workspace_index (workspace);
840 num_workspaces = meta_screen_get_n_workspaces (workspace->screen);
841 meta_screen_calc_workspace_layout (workspace->screen, num_workspaces,
842 current_space, &layout);
843 wrap = meta_prefs_get_wrap_style();
844
845 meta_verbose ("Getting neighbor of %d in direction %s\n",
846 current_space, meta_motion_direction_to_string (direction));
847
848 ltr = meta_ui_get_direction() == META_UI_DIRECTION_LTR;
849
850 switch (direction)
851 {
852 case META_MOTION_LEFT:
853 layout.current_col -= ltr ? 1 : -1;
854 break;
855 case META_MOTION_RIGHT:
856 layout.current_col += ltr ? 1 : -1;
857 break;
858 case META_MOTION_UP:
859 layout.current_row -= 1;
860 break;
861 case META_MOTION_DOWN:
862 layout.current_row += 1;
863 break;
864 case META_MOTION_PREV:
865 break;
866 }
867
868 /* LEFT */
869 if (layout.current_col < 0)
870 switch (wrap)
871 {
872 case META_WRAP_NONE:
873 layout.current_col = 0;
874 break;
875 case META_WRAP_CLASSIC:
876 layout.current_row = layout.current_row > 0 ? layout.current_row - 1 : layout.rows - 1;
877 /* fall through */
878 case META_WRAP_TOROIDAL:
879 layout.current_col = layout.cols - 1;
880 }
881 /* RIGHT */
882 if (layout.current_col >= layout.cols)
883 switch (wrap)
884 {
885 case META_WRAP_NONE:
886 layout.current_col = layout.cols - 1;
887 break;
888 case META_WRAP_CLASSIC:
889 layout.current_row = layout.current_row < layout.rows - 1 ? layout.current_row + 1 : 0;
890 /* fall through */
891 case META_WRAP_TOROIDAL:
892 layout.current_col = 0;
893 }
894 /* UP */
895 if (layout.current_row < 0)
896 switch (wrap)
897 {
898 case META_WRAP_NONE:
899 layout.current_row = 0;
900 break;
901 case META_WRAP_CLASSIC:
902 layout.current_col = layout.current_col > 0 ? layout.current_col - 1 : layout.cols - 1;
903 /* fall through */
904 case META_WRAP_TOROIDAL:
905 layout.current_row = layout.rows - 1;
906 }
907 /* DOWN */
908 if (layout.current_row >= layout.rows)
909 switch (wrap)
910 {
911 case META_WRAP_NONE:
912 layout.current_row = layout.rows - 1;
913 break;
914 case META_WRAP_CLASSIC:
915 layout.current_col = layout.current_col < layout.cols - 1 ? layout.current_col + 1 : 0;
916 /* fall through */
917 case META_WRAP_TOROIDAL:
918 layout.current_row = 0;
919 }
920
921 /* If we have an uneven arrangement of workspaces, (layout.cols - n, layout.rows - 1) may be an invalid workspace
922 e.g. we have 7 workspaces on a 3x3 pane */
923 if (wrap != META_WRAP_NONE && (layout.current_row * layout.cols + layout.current_col >= num_workspaces))
924 switch (direction)
925 {
926 case META_MOTION_LEFT:
927 layout.current_col = num_workspaces - (layout.current_row * layout.cols + 1);
928 break;
929 case META_MOTION_RIGHT:
930 layout.current_col = 0;
931 if (wrap == META_WRAP_CLASSIC)
932 layout.current_row = 0;
933 break;
934 case META_MOTION_UP:
935 layout.current_row -= 1;
936 break;
937 case META_MOTION_DOWN:
938 layout.current_row = 0;
939 if (wrap == META_WRAP_CLASSIC)
940 layout.current_col = layout.current_col < layout.cols - 1 ? layout.current_col + 1 : 0;
941 break;
942 case META_MOTION_PREV:
943 break;
944 }
945
946 i = layout.grid[layout.current_row * layout.cols + layout.current_col];
947
948 if (i < 0)
949 i = current_space;
950
951 if (i >= num_workspaces)
952 meta_bug ("calc_workspace_layout left an invalid (too-high) workspace number %d in the grid\n",
953 i);
954
955 meta_verbose ("Neighbor workspace is %d at row %d col %d\n",
956 i, layout.current_row, layout.current_col);
957
958 meta_screen_free_workspace_layout (&layout);
959
960 return meta_screen_get_workspace_by_index (workspace->screen, i);
961}

970meta_workspace_focus_default_window (MetaWorkspace *workspace,
971 MetaWindow *not_this_one,
972 guint32 timestamp)
973{
974 if (timestamp == CurrentTime)
975 {
976 meta_warning ("CurrentTime used to choose focus window; "
977 "focus window may not be correct.\n");
978 }
979
980 if (meta_prefs_get_focus_mode () == META_FOCUS_MODE_CLICK ||
981 !workspace->screen->display->mouse_mode)
982 focus_ancestor_or_top_window (workspace, not_this_one, timestamp);
983 else
984 {
985 MetaWindow * window;
986 window = meta_screen_get_mouse_window (workspace->screen, not_this_one);
987 if (window &&
988 window->type != META_WINDOW_DOCK &&
989 window->type != META_WINDOW_DESKTOP)
990 {
991 if (timestamp == CurrentTime)
992 {
993
994 /* We would like for this to never happen. However, if
995 * it does happen then we kludge since using CurrentTime
996 * can mean ugly race conditions--and we can avoid these
997 * by allowing EnterNotify events (which come with
998 * timestamps) to handle focus.
999 */
1000
1001 meta_topic (META_DEBUG_FOCUS,
1002 "Not focusing mouse window %s because EnterNotify events should handle that\n", window->desc);
1003 }
1004 else
1005 {
1006 meta_topic (META_DEBUG_FOCUS,
1007 "Focusing mouse window %s\n", window->desc);
1008 meta_window_focus (window, timestamp);
1009 }
1010
1011 if (workspace->screen->display->autoraise_window != window &&
1012 meta_prefs_get_auto_raise ())
1013 {
1014 meta_display_queue_autoraise_callback (workspace->screen->display,
1015 window);
1016 }
1017 }
1018 else if (meta_prefs_get_focus_mode () == META_FOCUS_MODE_SLOPPY)
1019 focus_ancestor_or_top_window (workspace, not_this_one, timestamp);
1020 else if (meta_prefs_get_focus_mode () == META_FOCUS_MODE_MOUSE)
1021 {
1022 meta_topic (META_DEBUG_FOCUS,
1023 "Setting focus to no_focus_window, since no valid "
1024 "window to focus found.\n");
1025 meta_display_focus_the_no_focus_window (workspace->screen->display,
1026 workspace->screen,
1027 timestamp);
1028 }
1029 }
