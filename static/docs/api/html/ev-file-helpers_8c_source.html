36/*
37 * ev_dir_ensure_exists:
38 * @dir: the directory name
39 * @mode: permissions to use when creating the directory
40 * @error: a location to store a #GError
41 *
42 * Create @dir recursively with permissions @mode.
43 *
44 * Returns: %TRUE on success, or %FALSE on error with @error filled in
45 */
46static gboolean
47_ev_dir_ensure_exists (const gchar *dir,
48 int mode,
49 GError **error)
50{
51 int errsv;
52 char *display_name;
53
54 g_return_val_if_fail (dir != NULL, FALSE);
55
56 errno = 0;
57 if (g_mkdir_with_parents (dir, mode) == 0)
58 return TRUE;
59
60 errsv = errno;
61 if (errsv == EEXIST && g_file_test (dir, G_FILE_TEST_IS_DIR))
62 return TRUE;
63
64 display_name = g_filename_display_name (dir);
65 g_set_error (error, G_IO_ERROR, g_io_error_from_errno (errsv),
66 "Failed to create directory '%s': %s",
67 display_name, g_strerror (errsv));
68 g_free (display_name);
69
70 return FALSE;
71}

73/*
74 * _ev_tmp_dir:
75 * @error: a location to store a #GError
76 *
77 * Returns the tmp directory.
78 *
79 * Returns: the tmp directory, or %NULL with @error filled in if the
80 * directory could not be created
81 */
82static const char *
83_ev_tmp_dir (GError **error)
84{
85
86 if (tmp_dir == NULL) {
87 const gchar *prgname;
88 gchar *dirname;
89
90 prgname = g_get_prgname ();
91 dirname = g_strdup_printf ("%s-%u", prgname ? prgname : "unknown", getpid ());
92 tmp_dir = g_build_filename (g_get_tmp_dir (), dirname, NULL);
93 g_free (dirname);
94 }
95
96 if (!_ev_dir_ensure_exists (tmp_dir, 0700, error))
97 return NULL;
98
99 return tmp_dir;
100}

128int
129ev_mkstemp (const char *template,
130 char **file_name,
131 GError **error)
132{
133 const char *tmp;
134 char *name;
135 int fd;
136
137 if ((tmp = _ev_tmp_dir (error)) == NULL)
138 return -1;
139
140 name = g_build_filename (tmp, template, NULL);
141 fd = g_mkstemp (name);
142
143 if (fd == -1) {
144 int errsv = errno;
145
146 g_set_error (error, G_IO_ERROR, g_io_error_from_errno (errsv),
147 _("Failed to create a temporary file: %s"),
148 g_strerror (errsv));
149
150 g_free (name);
151 return -1;
152 }
153
154 if (file_name)
155 *file_name = name;
156
157 return fd;
158}

346get_mime_type_from_uri (const gchar *uri, GError **error)
347{
348 GFile *file;
349 GFileInfo *file_info;
350 const gchar *content_type;

352
353 file = g_file_new_for_uri (uri);
354 file_info = g_file_query_info (file,
355 G_FILE_ATTRIBUTE_STANDARD_CONTENT_TYPE,
356 0, NULL, error);
357 g_object_unref (file);
358
359 if (file_info == NULL)
360 return NULL;
361
362 content_type = g_file_info_get_content_type (file_info);
363 if (content_type) {

365 }
366
367 g_object_unref (file_info);
368 return mime_type;

371static gchar *
372get_mime_type_from_data (const gchar *uri, GError **error)
373{
374 GFile *file;
375 GFileInputStream *input_stream;
376 gssize size_read;
377 guchar buffer[1024];
378 gboolean retval;
379 gchar *content_type, *mime_type;
380
381 file = g_file_new_for_uri (uri);
382
383 input_stream = g_file_read (file, NULL, error);
384 if (!input_stream) {
385 g_object_unref (file);
386 return NULL;
387 }
388
389 size_read = g_input_stream_read (G_INPUT_STREAM (input_stream),
390 buffer, sizeof (buffer), NULL, error);
391 if (size_read == -1) {
392 g_object_unref (input_stream);
393 g_object_unref (file);
394 return NULL;
395 }
396
397 retval = g_input_stream_close (G_INPUT_STREAM (input_stream), NULL, error);
398
399 g_object_unref (input_stream);
400 g_object_unref (file);
401 if (!retval)
402 return NULL;
403
404 content_type = g_content_type_guess (NULL, /* no filename */
405 buffer, size_read,
406 NULL);
407 if (!content_type)
408 return NULL;
409
410 mime_type = g_content_type_get_mime_type (content_type);
411 g_free (content_type);
412 return mime_type;
413}

438 NULL,
439 "bzip2",
440 "gzip",
441 "xz"

445#define BUFFER_SIZE 1024

447static gchar *
448compression_run (const gchar *uri,
449 EvCompressionType type,
450 gboolean compress,
451 GError **error)
452{
453 gchar *argv[N_ARGS];
454 gchar *uri_dst = NULL;
455 gchar *filename, *filename_dst = NULL;
456 gchar *cmd;
457 gint fd, pout;
458 GError *err = NULL;
459
460 if (type == EV_COMPRESSION_NONE)
461 return NULL;
462
463 cmd = g_find_program_in_path (compressor_cmds[type]);
464 if (!cmd) {
465 /* FIXME: better error codes! */
466 /* FIXME: i18n later */
467 g_set_error (error, G_FILE_ERROR, G_FILE_ERROR_FAILED,
468 "Failed to find the \"%s\" command in the search path.",
469 compressor_cmds[type]);
470 return NULL;
471 }
472
473 filename = g_filename_from_uri (uri, NULL, error);
474 if (!filename) {
475 g_free (cmd);
476 return NULL;
477 }
478
479 fd = ev_mkstemp ("comp.XXXXXX", &filename_dst, error);
480 if (fd == -1) {
481 g_free (cmd);
482 g_free (filename);
483
484 return NULL;
485 }
486
487 argv[0] = cmd;
488 argv[1] = compress ? "-c" : "-cd";
489 argv[2] = filename;
490 argv[3] = NULL;
491
492 if (g_spawn_async_with_pipes (NULL, argv, NULL,
493 G_SPAWN_STDERR_TO_DEV_NULL,
494 NULL, NULL, NULL,
495 NULL, &pout, NULL, &err)) {
496 GIOChannel *in, *out;
497 gchar buf[BUFFER_SIZE];
498 GIOStatus read_st, write_st;
499 gsize bytes_read, bytes_written;
500
501 in = g_io_channel_unix_new (pout);
502 g_io_channel_set_encoding (in, NULL, NULL);
503 out = g_io_channel_unix_new (fd);
504 g_io_channel_set_encoding (out, NULL, NULL);
505
506 do {
507 read_st = g_io_channel_read_chars (in, buf,
508 BUFFER_SIZE,
509 &bytes_read,
510 error);
511 if (read_st == G_IO_STATUS_NORMAL) {
512 write_st = g_io_channel_write_chars (out, buf,
513 bytes_read,
514 &bytes_written,
515 error);
516 if (write_st == G_IO_STATUS_ERROR)
517 break;
518 } else if (read_st == G_IO_STATUS_ERROR) {
519 break;
520 }
521 } while (bytes_read > 0);
522
523 g_io_channel_unref (in);
524 g_io_channel_unref (out);
525 }
526
527 close (fd);
528
529 if (err) {
530 g_propagate_error (error, err);
531 } else {
532 uri_dst = g_filename_to_uri (filename_dst, NULL, error);
533 }
534
535 g_free (cmd);
536 g_free (filename);
537 g_free (filename_dst);
538
539 return uri_dst;
540}
