172static void
173update_info_from_link (CajaDesktopIconFile *icon_file)
174{
175 CajaFile *file;
176 CajaDesktopLink *link;
177 char *display_name;
178 GMount *mount;
179
180 file = CAJA_FILE (icon_file);
181
182 link = icon_file->details->link;
183
184 if (link == NULL)
185 {
186 return;
187 }
188
189 g_clear_pointer (&file->details->mime_type, g_ref_string_release);
190 file->details->mime_type = g_ref_string_new_intern ("application/x-caja-link");
191 file->details->type = G_FILE_TYPE_SHORTCUT;
192 file->details->size = 0;
193 file->details->has_permissions = FALSE;
194 file->details->can_read = TRUE;
195 file->details->can_write = TRUE;
196
197 file->details->can_mount = FALSE;
198 file->details->can_unmount = FALSE;
199 file->details->can_eject = FALSE;
200 if (file->details->mount)
201 {
202 g_object_unref (file->details->mount);
203 }
204 mount = caja_desktop_link_get_mount (link);
205 file->details->mount = mount;
206 if (mount)
207 {
208 file->details->can_unmount = (g_mount_can_unmount (mount) != FALSE);
209 file->details->can_eject = (g_mount_can_eject (mount) != FALSE);
210 }
211
212 file->details->file_info_is_up_to_date = TRUE;
213
214 display_name = caja_desktop_link_get_display_name (link);
215 caja_file_set_display_name (file,
216 display_name, NULL, TRUE);
217 g_free (display_name);
218
219 if (file->details->icon != NULL)
220 {
221 g_object_unref (file->details->icon);
222 }
223 file->details->icon = caja_desktop_link_get_icon (link);
224 g_free (file->details->activation_uri);
225 file->details->activation_uri = caja_desktop_link_get_activation_uri (link);
226 file->details->got_link_info = TRUE;
227 file->details->link_info_is_up_to_date = TRUE;
228
229 file->details->directory_count = 0;
230 file->details->got_directory_count = TRUE;
231 file->details->directory_count_is_up_to_date = TRUE;
232}

244void
245caja_desktop_icon_file_remove (CajaDesktopIconFile *icon_file)
246{
247 CajaFile *file;
248 GList list;
249
250 icon_file->details->link = NULL;
251
252 file = CAJA_FILE (icon_file);
253
254 /* ref here because we might be removing the last ref when we
255 * mark the file gone below, but we need to keep a ref at
256 * least long enough to send the change notification.
257 */
258 caja_file_ref (file);
259
260 file->details->is_gone = TRUE;
261
262 list.data = file;
263 list.next = NULL;
264 list.prev = NULL;
265
266 caja_directory_remove_file (file->details->directory, file);
267 caja_directory_emit_change_signals (file->details->directory, &list);
268
269 caja_file_unref (file);
270}

272CajaDesktopIconFile *
273caja_desktop_icon_file_new (CajaDesktopLink *link)
274{
275 CajaFile *file;
276 CajaDirectory *directory;
277 CajaDesktopIconFile *icon_file;
278 GList list;
279 char *name;
280
281 directory = caja_directory_get_by_uri (EEL_DESKTOP_URI);
282
283 file = CAJA_FILE (g_object_new (CAJA_TYPE_DESKTOP_ICON_FILE, NULL));
284
285#ifdef CAJA_FILE_DEBUG_REF
286 printf("%10p ref'd\n", file);
287 eazel_dump_stack_trace ("\t", 10);
288#endif
289
290 file->details->directory = directory;
291
292 icon_file = CAJA_DESKTOP_ICON_FILE (file);
293 icon_file->details->link = link;
294
295 name = caja_desktop_link_get_file_name (link);
296 file->details->name = g_ref_string_new (name);
297 g_free (name);
298
299 update_info_from_link (icon_file);
300
301 caja_desktop_update_metadata_from_keyfile (file, file->details->name);
302
303 caja_directory_add_file (directory, file);
304
305 list.data = file;
306 list.next = NULL;
307 list.prev = NULL;
308 caja_directory_emit_files_added (directory, &list);
309
310 return icon_file;
311}
