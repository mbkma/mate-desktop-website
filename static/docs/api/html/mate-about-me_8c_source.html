89static void
90about_me_load_photo (MateAboutMe *me)
91{
92 gchar *file = NULL;
93 GError *error = NULL;
94#if HAVE_ACCOUNTSSERVICE
95 const gchar *act_file;
96
97 if (act_user_is_loaded (me->user)) {
98 act_file = act_user_get_icon_file (me->user);
99 if ( act_file != NULL && strlen (act_file) > 1) {
100 file = g_strdup (act_file);
101 }
102 }
103#endif
104 if (file == NULL) {
105 file = g_build_filename (g_get_home_dir (), ".face", NULL);
106 }
107
108 me->image = gdk_pixbuf_new_from_file(file, &error);
109
110 if (me->image != NULL) {
111 e_image_chooser_set_from_file (E_IMAGE_CHOOSER (me->image_chooser), file);
112 me->have_image = TRUE;
113 } else {
114 me->have_image = FALSE;
115 g_warning ("Could not load %s: %s", file, error->message);
116 e_image_chooser_set_from_file (E_IMAGE_CHOOSER (me->image_chooser), me->person);
117 g_error_free (error);
118 }
119 g_free (file);

122static void
123about_me_update_photo (MateAboutMe *me)
124{
125 gchar *file;
126 GError *error = NULL;
127
128 guchar *data;
129 gsize length;
130
131 if (me->image_changed && me->have_image) {
132 GdkPixbufLoader *loader = gdk_pixbuf_loader_new ();
133 GdkPixbuf *pixbuf = NULL, *scaled = NULL;
134 int height, width;
135 gboolean do_scale = FALSE;
136 float scale = 1.0;
137 float scalex = 1.0, scaley = 1.0;
138
139 e_image_chooser_get_image_data (E_IMAGE_CHOOSER (me->image_chooser), (char **) &data, &length);
140
141 /* Before updating the image in EDS scale it to a reasonable size
142 so that the user doesn't get an application that does not respond
143 or that takes 100% CPU */
144 gdk_pixbuf_loader_write (loader, data, length, NULL);
145 gdk_pixbuf_loader_close (loader, NULL);
146
147 pixbuf = gdk_pixbuf_loader_get_pixbuf (loader);
148
149 if (pixbuf)
150 g_object_ref (pixbuf);
151
152 g_object_unref (loader);
153
154 height = gdk_pixbuf_get_height (pixbuf);
155 width = gdk_pixbuf_get_width (pixbuf);
156
157 if (width > MAX_WIDTH) {
158 scalex = (float)MAX_WIDTH/width;
159 if (scalex < scale) {
160 scale = scalex;
161 }
162 do_scale = TRUE;
163 }
164
165 if (height > MAX_HEIGHT) {
166 scaley = (float)MAX_HEIGHT/height;
167 if (scaley < scale) {
168 scale = scaley;
169 }
170 do_scale = TRUE;
171 }
172
173 if (do_scale) {
174 char *scaled_data = NULL;
175 gsize scaled_length;
176
177 scaled = gdk_pixbuf_scale_simple (pixbuf, width*scale, height*scale, GDK_INTERP_BILINEAR);
178 gdk_pixbuf_save_to_buffer (scaled, &scaled_data, &scaled_length, "png", NULL,
179 "compression", "9", NULL);
180
181 g_free (data);
182 data = (guchar *) scaled_data;
183 length = scaled_length;
184 }
185
186 /* Save the image for MDM */
187 /* FIXME: I would have to read the default used by the mdmgreeter program */
188 error = NULL;
189 file = g_build_filename (g_get_home_dir (), ".face", NULL);
190 if (g_file_set_contents (file, (gchar *)data, length, &error) == TRUE) {
191 g_chmod (file, 0644);
192#if HAVE_ACCOUNTSSERVICE
193 act_user_set_icon_file (me->user, file);
194#endif
195 } else {
196 g_warning ("Could not create %s: %s", file, error->message);
197 g_error_free (error);
198 }
199
200 g_free (file);
201 g_object_unref (pixbuf);
202 g_free (data);
203 } else if (me->image_changed && !me->have_image) {
204 /* Update the image in the card */
205 file = g_build_filename (g_get_home_dir (), ".face", NULL);
206
207 g_unlink (file);
208
209 g_free (file);
210#if HAVE_ACCOUNTSSERVICE
211 act_user_set_icon_file (me->user, "");
212#endif
213 }
214}

223static void
224about_me_update_preview (GtkFileChooser *chooser,
225 MateAboutMe *me)
226{
227 gchar *uri;
228
229 uri = gtk_file_chooser_get_preview_uri (chooser);
230
231 if (uri) {
232 GtkWidget *image;
233 GdkPixbuf *pixbuf = NULL;
234 GFile *file;
235 GFileInfo *file_info;
236
237 if (!me->thumbs)
238 me->thumbs = mate_desktop_thumbnail_factory_new (MATE_DESKTOP_THUMBNAIL_SIZE_NORMAL);
239
240 file = g_file_new_for_uri (uri);
241 file_info = g_file_query_info (file,
242 G_FILE_ATTRIBUTE_STANDARD_CONTENT_TYPE,
243 G_FILE_QUERY_INFO_NONE,
244 NULL, NULL);
245 g_object_unref (file);
246
247 if (file_info != NULL) {
248 const gchar *content_type;
249
250 content_type = g_file_info_get_content_type (file_info);
251 if (content_type) {
252 gchar *mime_type;
253
254 mime_type = g_content_type_get_mime_type (content_type);
255
256 pixbuf = mate_desktop_thumbnail_factory_generate_thumbnail (me->thumbs,
257 uri,
258 mime_type);
259 g_free (mime_type);
260 }
261 g_object_unref (file_info);
262 }
263
264 image = gtk_file_chooser_get_preview_widget (chooser);
265
266 if (pixbuf != NULL) {
267 gtk_image_set_from_pixbuf (GTK_IMAGE (image), pixbuf);
268 g_object_unref (pixbuf);
269 } else {
270 gtk_image_set_from_icon_name (GTK_IMAGE (image),
271 "dialog-question",
272 GTK_ICON_SIZE_DIALOG);
273 }
274 }
275 gtk_file_chooser_set_preview_widget_active (chooser, TRUE);
276}

278static void
279about_me_image_clicked_cb (GtkWidget *button, MateAboutMe *me)
280{
281 GtkFileChooser *chooser_dialog;
282 gint response;
283 GtkBuilder *dialog;
284 GtkWidget *image;
285 const gchar *chooser_dir = DATADIR"/pixmaps/faces";
286 const gchar *pics_dir;
287 GtkFileFilter *filter;
288
289 dialog = me->dialog;

291 chooser_dialog =
292 GTK_FILE_CHOOSER (
293 gtk_file_chooser_dialog_new (_("Select Image"),
294 GTK_WINDOW (gtk_builder_get_object (dialog, "about-me-dialog")),
295 GTK_FILE_CHOOSER_ACTION_OPEN,
296 _("No Image"), GTK_RESPONSE_NO,
297 "gtk-cancel", GTK_RESPONSE_CANCEL,
298 "gtk-open", GTK_RESPONSE_ACCEPT,
299 NULL));
300 gtk_window_set_modal (GTK_WINDOW (chooser_dialog), TRUE);
301 gtk_dialog_set_default_response (GTK_DIALOG (chooser_dialog), GTK_RESPONSE_ACCEPT);
302
303 gtk_file_chooser_add_shortcut_folder (chooser_dialog, chooser_dir, NULL);
304 pics_dir = g_get_user_special_dir (G_USER_DIRECTORY_PICTURES);
305 if (pics_dir != NULL)
306 gtk_file_chooser_add_shortcut_folder (chooser_dialog, pics_dir, NULL);
307
308 if (!g_file_test (chooser_dir, G_FILE_TEST_IS_DIR))
309 chooser_dir = g_get_home_dir ();
310
311 gtk_file_chooser_set_current_folder (chooser_dialog, chooser_dir);
312 gtk_file_chooser_set_use_preview_label (chooser_dialog, FALSE);
313
314 image = gtk_image_new ();
315 gtk_file_chooser_set_preview_widget (chooser_dialog, image);
316 gtk_widget_set_size_request (image, 128, -1);
317
318 gtk_widget_show (image);
319
320 g_signal_connect (chooser_dialog, "update-preview",
321 G_CALLBACK (about_me_update_preview), me);
322
323 filter = gtk_file_filter_new ();
324 gtk_file_filter_set_name (filter, _("Images"));
325 gtk_file_filter_add_pixbuf_formats (filter);
326 gtk_file_chooser_add_filter (chooser_dialog, filter);
327 filter = gtk_file_filter_new ();
328 gtk_file_filter_set_name (filter, _("All Files"));
329 gtk_file_filter_add_pattern(filter, "*");
330 gtk_file_chooser_add_filter (chooser_dialog, filter);
331
332 response = gtk_dialog_run (GTK_DIALOG (chooser_dialog));
333
334 if (response == GTK_RESPONSE_ACCEPT) {
335 gchar* filename;
336
337 filename = gtk_file_chooser_get_filename (chooser_dialog);
338 me->have_image = TRUE;
339 me->image_changed = TRUE;
340
341 e_image_chooser_set_from_file (E_IMAGE_CHOOSER (me->image_chooser), filename);
342 g_free (filename);
343 about_me_update_photo (me);
344 } else if (response == GTK_RESPONSE_NO) {
345 me->have_image = FALSE;
346 me->image_changed = TRUE;
347 e_image_chooser_set_from_file (E_IMAGE_CHOOSER (me->image_chooser), me->person);
348 about_me_update_photo (me);
349 }
350
351 gtk_widget_destroy (GTK_WIDGET (chooser_dialog));
352}

425static gint
426about_me_setup_dialog (void)
427{
428 GtkWidget *widget;
429 GtkWidget *main_dialog;
430 GtkIconInfo *icon;
431 GtkBuilder *dialog;
432 GError *error = NULL;
433 gchar *str;
434#if HAVE_ACCOUNTSSERVICE
435 ActUserManager* manager;
436#endif
437
438 me = g_new0 (MateAboutMe, 1);
439 me->image = NULL;
440
441 dialog = gtk_builder_new ();
442 if (gtk_builder_add_from_resource (dialog, "/org/mate/mcc/am/mate-about-me-dialog.ui", &error) == 0)
443 {
444 g_warning ("Could not parse UI definition: %s", error->message);
445 g_error_free (error);
446 }
447
448 me->image_chooser = e_image_chooser_new_with_size (MAX_WIDTH, MAX_HEIGHT);
449 gtk_container_add (GTK_CONTAINER (gtk_builder_get_object (dialog, "button-image")), me->image_chooser);
450
451 if (dialog == NULL) {
452 about_me_destroy ();
453 return -1;
454 }
455
456 me->dialog = dialog;
457
458 /* Connect the close button signal */
459 main_dialog = WID ("about-me-dialog");
460 g_signal_connect (main_dialog, "response",
461 G_CALLBACK (about_me_button_clicked_cb), me);
462
463 gtk_window_set_resizable (GTK_WINDOW (main_dialog), FALSE);
464 capplet_set_icon (main_dialog, "user-info");
465
466 /* Setup theme details */
467 me->screen = gtk_window_get_screen (GTK_WINDOW (main_dialog));
468 me->theme = gtk_icon_theme_get_for_screen (me->screen);
469
470 icon = gtk_icon_theme_lookup_icon (me->theme, "avatar-default", 80, 0);
471 if (icon != NULL) {
472 me->person = g_strdup (gtk_icon_info_get_filename (icon));
473 g_object_unref (icon);
474 }
475
476 g_signal_connect_object (me->theme, "changed",
477 G_CALLBACK (about_me_icon_theme_changed),
478 main_dialog,
479 G_CONNECT_SWAPPED);
480
481 me->login = g_strdup (g_get_user_name ());
482 me->username = g_strdup (g_get_real_name ());
483
484#if HAVE_ACCOUNTSSERVICE
485 manager = act_user_manager_get_default ();
486 me->user = act_user_manager_get_user (manager, me->login);
487 g_signal_connect (me->user, "notify::is-loaded", G_CALLBACK (on_user_is_loaded_changed), me);
488#endif
489 /* Contact Tab */
490 about_me_load_photo (me);
491
492 widget = WID ("fullname");
493 str = g_strdup_printf ("<b><span size=\"xx-large\">%s</span></b>", me->username);
494
495 gtk_label_set_markup (GTK_LABEL (widget), str);
496 g_free (str);
497
498 widget = WID ("login");
499 gtk_label_set_text (GTK_LABEL (widget), me->login);
500
501 str = g_strdup_printf (_("About %s"), me->username);
502 gtk_window_set_title (GTK_WINDOW (main_dialog), str);
503 g_free (str);
504
505 widget = WID ("password");
506 g_signal_connect (widget, "clicked",
507 G_CALLBACK (about_me_passwd_clicked_cb), me);
508
509 widget = WID ("button-image");
510 g_signal_connect (widget, "clicked",
511 G_CALLBACK (about_me_image_clicked_cb), me);
512
513 me->enable_fingerprint_button = WID ("enable_fingerprint_button");
514 me->disable_fingerprint_button = WID ("disable_fingerprint_button");
515
516 g_signal_connect (me->enable_fingerprint_button, "clicked",
517 G_CALLBACK (about_me_fingerprint_button_clicked_cb), me);
518 g_signal_connect (me->disable_fingerprint_button, "clicked",
519 G_CALLBACK (about_me_fingerprint_button_clicked_cb), me);
520
521 g_signal_connect (me->image_chooser, "changed",
522 G_CALLBACK (about_me_image_changed_cb), me);
523
524 about_me_load_info (me);
525
526 gtk_widget_show_all (main_dialog);
527
528 return 0;
529}
