77static SidebarPage *
78sidebar_page_new (const gchar *id,

107static void
108gdict_sidebar_finalize (GObject *object)
109{
110 GdictSidebar *sidebar = GDICT_SIDEBAR (object);
111 GdictSidebarPrivate *priv = sidebar->priv;
112
113 if (priv->pages_by_id)
114 g_hash_table_destroy (priv->pages_by_id);
115
116 if (priv->pages)
117 {
118 g_slist_free_full (priv->pages, (GDestroyNotify) sidebar_page_free);
119 }
120
121 G_OBJECT_CLASS (gdict_sidebar_parent_class)->finalize (object);
122}

138static gboolean
139gdict_sidebar_select_button_press_cb (GtkWidget *widget,
140 GdkEventButton *event,
141 gpointer user_data)
142{
143 GdictSidebar *sidebar = GDICT_SIDEBAR (user_data);
144 GtkAllocation allocation;
145
146 if (event->button == 1)
147 {
148 GtkRequisition req;
149 gint width;
150
151 gtk_widget_get_allocation (widget, &allocation);
152 width = allocation.width;
153 gtk_widget_set_size_request (sidebar->priv->menu, -1, -1);
154 gtk_widget_get_preferred_size (sidebar->priv->menu, &req, NULL);
155 gtk_widget_set_size_request (sidebar->priv->menu,
156 MAX (width, req.width), -1);
157 gtk_widget_grab_focus (widget);
158
159 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (widget), TRUE);
160 gtk_menu_popup_at_widget (GTK_MENU (sidebar->priv->menu),
161 widget,
162 GDK_GRAVITY_SOUTH_WEST,
163 GDK_GRAVITY_NORTH_WEST,
164 (const GdkEvent*) event);
165
166 return TRUE;
167 }
168
169 return FALSE;
170}

226static void
227gdict_sidebar_menu_item_activate (GtkWidget *widget,
228 gpointer user_data)
229{
230 GdictSidebar *sidebar = GDICT_SIDEBAR (user_data);
231 GdictSidebarPrivate *priv = sidebar->priv;
232 GtkWidget *menu_item;
233 const gchar *id;
234 SidebarPage *page;
235 gint current_index;
236
237 menu_item = gtk_menu_get_active (GTK_MENU (priv->menu));
238 id = g_object_get_qdata (G_OBJECT (menu_item), sidebar_page_id_quark);
239 g_assert (id != NULL);
240
241 page = g_hash_table_lookup (priv->pages_by_id, id);
242 g_assert (page != NULL);
243
244 current_index = gtk_notebook_get_current_page (GTK_NOTEBOOK (priv->notebook));
245 if (current_index == page->index)
246 return;
247
248 gtk_notebook_set_current_page (GTK_NOTEBOOK (priv->notebook),
249 page->index);
250 gtk_label_set_text (GTK_LABEL (priv->label), page->name);
251
252 g_signal_emit (sidebar, sidebar_signals[PAGE_CHANGED], 0);
253}

377void
378gdict_sidebar_add_page (GdictSidebar *sidebar,
379 const gchar *page_id,
380 const gchar *page_name,
381 GtkWidget *page_widget)
382{
383 GdictSidebarPrivate *priv;
384 SidebarPage *page;
385 GtkWidget *menu_item;
386
387 g_return_if_fail (GDICT_IS_SIDEBAR (sidebar));
388 g_return_if_fail (page_id != NULL);
389 g_return_if_fail (page_name != NULL);
390 g_return_if_fail (GTK_IS_WIDGET (page_widget));
391
392 priv = sidebar->priv;
393
394 if (g_hash_table_lookup (priv->pages_by_id, page_id))
395 {
396 g_warning ("Attempting to add a page to the sidebar with "
397 "id `%s', but there already is a page with the "
398 "same id. Aborting...",
399 page_id);
400 return;
401 }
402
403 /* add the page inside the page list */
404 page = sidebar_page_new (page_id, page_name, page_widget);
405
406 priv->pages = g_slist_append (priv->pages, page);
407 g_hash_table_insert (priv->pages_by_id, page->id, page);
408
409 page->index = gtk_notebook_append_page (GTK_NOTEBOOK (priv->notebook),
410 page_widget,
411 NULL);
412
413 /* add the menu item for the page */
414 menu_item = mate_image_menu_item_new_with_label (page_name);
415 g_object_set_qdata_full (G_OBJECT (menu_item),
416 sidebar_page_id_quark,
417 g_strdup (page_id),
418 g_free);
419 g_signal_connect (menu_item, "activate",
420 G_CALLBACK (gdict_sidebar_menu_item_activate),
421 sidebar);
422 gtk_menu_shell_append (GTK_MENU_SHELL (priv->menu), menu_item);
423 gtk_widget_show (menu_item);
424 page->menu_item = menu_item;
425
426 if (gtk_widget_get_realized (priv->menu))
427 gtk_menu_shell_select_item (GTK_MENU_SHELL (priv->menu), menu_item);
428 gtk_label_set_text (GTK_LABEL (priv->label), page_name);
429 gtk_notebook_set_current_page (GTK_NOTEBOOK (priv->notebook), page->index);
430}

432void
433gdict_sidebar_remove_page (GdictSidebar *sidebar,
434 const gchar *page_id)
435{
436 GdictSidebarPrivate *priv;
437 SidebarPage *page;
438 GList *children, *l;
439
440 g_return_if_fail (GDICT_IS_SIDEBAR (sidebar));
441 g_return_if_fail (page_id != NULL);
442
443 priv = sidebar->priv;
444
445 if ((page = g_hash_table_lookup (priv->pages_by_id, page_id)) == NULL)
446 {
447 g_warning ("Attempting to remove a page from the sidebar with "
448 "id `%s', but there is no page with this id. Aborting...",
449 page_id);
450 return;
451 }
452
453 children = gtk_container_get_children (GTK_CONTAINER (priv->menu));
454 for (l = children; l != NULL; l = l->next)
455 {
456 GtkWidget *menu_item = l->data;
457
458 if (menu_item == page->menu_item)
459 {
460 gtk_container_remove (GTK_CONTAINER (priv->menu), menu_item);
461 break;
462 }
463 }
464 g_list_free (children);
465
466 gtk_notebook_remove_page (GTK_NOTEBOOK (priv->notebook), page->index);
467
468 g_hash_table_remove (priv->pages_by_id, page->id);
469 priv->pages = g_slist_remove (priv->pages, page);
470
471 sidebar_page_free (page);
472
473 /* select the first page, if present */
474 page = priv->pages->data;
475 if (page)
476 {
477 if (gtk_widget_get_realized (priv->menu))
478 gtk_menu_shell_select_item (GTK_MENU_SHELL (priv->menu), page->menu_item);
479 gtk_label_set_text (GTK_LABEL (priv->label), page->name);
480 gtk_notebook_set_current_page (GTK_NOTEBOOK (priv->notebook), page->index);
481 }
482 else
483 gtk_widget_hide (GTK_WIDGET (sidebar));
484}
