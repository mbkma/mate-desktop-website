67static void
68save_window_session (GKeyFile *state_file,
69 const gchar *group_name,
70 PlumaWindow *window)
71{
72 const gchar *role;
73 int width, height;
74 PlumaPanel *panel;
75 GList *docs, *l;
76 GPtrArray *doc_array;
77 PlumaDocument *active_document;
78 gchar *uri;
79
80 pluma_debug (DEBUG_SESSION);
81
82 role = gtk_window_get_role (GTK_WINDOW (window));
83 g_key_file_set_string (state_file, group_name, "role", role);
84 gtk_window_get_size (GTK_WINDOW (window), &width, &height);
85 g_key_file_set_integer (state_file, group_name, "width", width);
86 g_key_file_set_integer (state_file, group_name, "height", height);
87
88 panel = pluma_window_get_side_panel (window);
89 g_key_file_set_boolean (state_file, group_name, "side-panel-visible",
90 gtk_widget_get_visible (GTK_WIDGET (panel)));
91
92 panel = pluma_window_get_bottom_panel (window);
93 g_key_file_set_boolean (state_file, group_name, "bottom-panel-visible",
94 gtk_widget_get_visible (GTK_WIDGET (panel)));
95
96 active_document = pluma_window_get_active_document (window);
97 if (active_document)
98 {
99 uri = pluma_document_get_uri (active_document);
100 g_key_file_set_string (state_file, group_name,
101 "active-document", uri);
102 }
103
104 docs = pluma_window_get_documents (window);
105
106 doc_array = g_ptr_array_new ();
107 for (l = docs; l != NULL; l = g_list_next (l))
108 {
109 uri = pluma_document_get_uri (PLUMA_DOCUMENT (l->data));
110
111 if (uri != NULL)
112 g_ptr_array_add (doc_array, uri);
113
114 }
115 g_list_free (docs);
116
117 if (doc_array->len)
118 {
119 guint i;
120
121 g_key_file_set_string_list (state_file, group_name,
122 "documents",
123 (const char **)doc_array->pdata,
124 doc_array->len);
125 for (i = 0; i < doc_array->len; i++)
126 g_free (doc_array->pdata[i]);
127 }
128 g_ptr_array_free (doc_array, TRUE);
129}

131static void
132client_save_state_cb (EggSMClient *client,
133 GKeyFile *state_file,
134 gpointer user_data)
135{
136 const GList *windows;
137 gchar *group_name;
138 int n;
139
140 windows = pluma_app_get_windows (pluma_app_get_default ());
141 n = 1;
142
143 while (windows != NULL)
144 {
145 group_name = g_strdup_printf ("pluma window %d", n);
146 save_window_session (state_file,
147 group_name,
148 PLUMA_WINDOW (windows->data));
149 g_free (group_name);
150
151 windows = g_list_next (windows);
152 n++;
153 }

168static void
169window_state_change (PlumaWindow *window,
170 GParamSpec *pspec,
171 gpointer data)
172{
173 PlumaWindowState state;
174 GList *unsaved_docs;
175 GList *docs_to_save;
176 GList *l;
177 gboolean done = TRUE;
178
179 state = pluma_window_get_state (window);
180
181 /* we are still saving */
182 if (state & PLUMA_WINDOW_STATE_SAVING)
183 return;
184
185 unsaved_docs = pluma_window_get_unsaved_documents (window);
186
187 docs_to_save = g_object_get_data (G_OBJECT (window),
188 PLUMA_SESSION_LIST_OF_DOCS_TO_SAVE);
189
190
191 for (l = docs_to_save; l != NULL; l = l->next)
192 {
193 if (g_list_find (unsaved_docs, l->data))
194 {
195 done = FALSE;
196 break;
197 }
198 }
199
200 if (done)
201 {
202 g_signal_handlers_disconnect_by_func (window, window_state_change, data);
203 g_list_free (docs_to_save);
204 g_object_set_data (G_OBJECT (window),
205 PLUMA_SESSION_LIST_OF_DOCS_TO_SAVE,
206 NULL);
207
208 window_handled (window);
209 }
210
211 g_list_free (unsaved_docs);
212}

214static void
215close_confirmation_dialog_response_handler (PlumaCloseConfirmationDialog *dlg,
216 gint response_id,
217 PlumaWindow *window)
218{
219 GList *selected_documents;
220 GSList *l;
221
222 pluma_debug (DEBUG_COMMANDS);
223
224 switch (response_id)
225 {
226 case GTK_RESPONSE_YES:
227 /* save selected docs */
228
229 g_signal_connect (window,
230 "notify::state",
231 G_CALLBACK (window_state_change),
232 NULL);
233
234 selected_documents = pluma_close_confirmation_dialog_get_selected_documents (dlg);
235
236 g_return_if_fail (g_object_get_data (G_OBJECT (window),
237 PLUMA_SESSION_LIST_OF_DOCS_TO_SAVE) == NULL);
238
239 g_object_set_data (G_OBJECT (window),
240 PLUMA_SESSION_LIST_OF_DOCS_TO_SAVE,
241 selected_documents);
242
243 _pluma_cmd_file_save_documents_list (window, selected_documents);
244
245 /* FIXME: also need to lock the window to prevent further changes... */
246
247 break;
248
249 case GTK_RESPONSE_NO:
250 /* dont save */
251 window_handled (window);
252 break;
253
254 default:
255 /* disconnect window_state_changed where needed */
256 for (l = window_dirty_list; l != NULL; l = l->next)
257 g_signal_handlers_disconnect_by_func (window,
258 window_state_change, NULL);
259 g_slist_free (window_dirty_list);
260 window_dirty_list = NULL;
261
262 /* cancel shutdown */
263 egg_sm_client_will_quit (master_client, FALSE);
264
265 break;
266 }
267
268 gtk_widget_destroy (GTK_WIDGET (dlg));
269}

271static void
272show_confirmation_dialog (PlumaWindow *window)
273{
274 GList *unsaved_docs;
275 GtkWidget *dlg;
276
277 pluma_debug (DEBUG_SESSION);
278
279 unsaved_docs = pluma_window_get_unsaved_documents (window);
280
281 g_return_if_fail (unsaved_docs != NULL);
282
283 if (unsaved_docs->next == NULL)
284 {
285 /* There is only one unsaved document */
286 PlumaTab *tab;
287 PlumaDocument *doc;
288
289 doc = PLUMA_DOCUMENT (unsaved_docs->data);
290
291 tab = pluma_tab_get_from_document (doc);
292 g_return_if_fail (tab != NULL);
293
294 pluma_window_set_active_tab (window, tab);
295
296 dlg = pluma_close_confirmation_dialog_new_single (
297 GTK_WINDOW (window),
298 doc,
299 TRUE);
300 }
301 else
302 {
303 dlg = pluma_close_confirmation_dialog_new (GTK_WINDOW (window),
304 unsaved_docs,
305 TRUE);
306 }
307
308 g_list_free (unsaved_docs);
309
310 g_signal_connect (dlg,
311 "response",
312 G_CALLBACK (close_confirmation_dialog_response_handler),
313 window);
314
315 gtk_widget_show (dlg);
316}

330/* quit_requested handler for the master client */
331static void
332client_quit_requested_cb (EggSMClient *client, gpointer data)
333{
334 PlumaApp *app;
335 const GList *l;
336
337 pluma_debug (DEBUG_SESSION);
338
339 app = pluma_app_get_default ();
340
341 if (window_dirty_list != NULL)
342 {
343 g_critical ("global variable window_dirty_list not NULL");
344 window_dirty_list = NULL;
345 }
346
347 for (l = pluma_app_get_windows (app); l != NULL; l = l->next)
348 {
349 if (pluma_window_get_unsaved_documents (PLUMA_WINDOW (l->data)) != NULL)
350 {
351 window_dirty_list = g_slist_prepend (window_dirty_list, l->data);
352 }
353 }
354
355 /* no modified docs */
356 if (window_dirty_list == NULL)
357 {
358 egg_sm_client_will_quit (client, TRUE);
359
360 return;
361 }
362
363 ask_next_confirmation ();
364
365 pluma_debug_message (DEBUG_SESSION, "END");
366}

455static void
456parse_window (GKeyFile *state_file, const char *group_name)
457{
458 PlumaWindow *window;
459 gchar *role, *active_document, **documents;
460 int width, height;
461 gboolean visible;
462 PlumaPanel *panel;
463 GError *error = NULL;
464
465 role = g_key_file_get_string (state_file, group_name, "role", NULL);
466
467 pluma_debug_message (DEBUG_SESSION, "Window role: %s", role);
468
469 window = _pluma_app_restore_window (pluma_app_get_default (), (gchar *) role);
470 g_free (role);
471
472 if (window == NULL)
473 {
474 g_warning ("Couldn't restore window");
475 return;
476 }
477
478 width = g_key_file_get_integer (state_file, group_name,
479 "width", &error);
480 if (error)
481 {
482 g_clear_error (&error);
483 width = -1;
484 }
485 height = g_key_file_get_integer (state_file, group_name,
486 "height", &error);
487 if (error)
488 {
489 g_clear_error (&error);
490 height = -1;
491 }
492 gtk_window_set_default_size (GTK_WINDOW (window), width, height);
493
494
495 visible = g_key_file_get_boolean (state_file, group_name,
496 "side-panel-visible", &error);
497 if (error)
498 {
499 g_clear_error (&error);
500 visible = FALSE;
501 }
502
503 panel = pluma_window_get_side_panel (window);
504
505 if (visible)
506 {
507 pluma_debug_message (DEBUG_SESSION, "Side panel visible");
508 gtk_widget_show (GTK_WIDGET (panel));
509 }
510 else
511 {
512 pluma_debug_message (DEBUG_SESSION, "Side panel _NOT_ visible");
513 gtk_widget_hide (GTK_WIDGET (panel));
514 }
515
516 visible = g_key_file_get_boolean (state_file, group_name,
517 "bottom-panel-visible", &error);
518 if (error)
519 {
520 g_clear_error (&error);
521 visible = FALSE;
522 }
523
524 panel = pluma_window_get_bottom_panel (window);
525 if (visible)
526 {
527 pluma_debug_message (DEBUG_SESSION, "Bottom panel visible");
528 gtk_widget_show (GTK_WIDGET (panel));
529 }
530 else
531 {
532 pluma_debug_message (DEBUG_SESSION, "Bottom panel _NOT_ visible");
533 gtk_widget_hide (GTK_WIDGET (panel));
534 }
535
536 active_document = g_key_file_get_string (state_file, group_name,
537 "active-document", NULL);
538 documents = g_key_file_get_string_list (state_file, group_name,
539 "documents", NULL, NULL);
540 if (documents)
541 {
542 int i;
543 gboolean jump_to = FALSE;
544
545 for (i = 0; documents[i]; i++)
546 {
547 if (active_document != NULL)
548 jump_to = strcmp (active_document,
549 documents[i]) == 0;
550
551 pluma_debug_message (DEBUG_SESSION,
552 "URI: %s (%s)",
553 documents[i],
554 jump_to ? "active" : "not active");
555 pluma_window_create_tab_from_uri (window,
556 documents[i],
557 NULL,
558 0,
559 FALSE,
560 jump_to);
561 }
562 g_strfreev (documents);
563 }
564
565 g_free (active_document);
566
567 gtk_widget_show (GTK_WIDGET (window));
568}

578gboolean
579pluma_session_load (void)
580{
581 GKeyFile *state_file;
582 gchar **groups;
583 int i;
584
585 pluma_debug (DEBUG_SESSION);
586
587 state_file = egg_sm_client_get_state_file (master_client);
588 if (state_file == NULL)
589 return FALSE;
590
591 groups = g_key_file_get_groups (state_file, NULL);
592
593 for (i = 0; groups[i] != NULL; i++)
594 {
595 if (g_str_has_prefix (groups[i], "pluma window "))
596 parse_window (state_file, groups[i]);
597 }
598
599 g_strfreev (groups);
600 g_key_file_free (state_file);
601
602 return TRUE;
