76static PlumaTab *
77get_tab_from_file (GList *docs, GFile *file)
78{
79 PlumaTab *tab = NULL;
80
81 while (docs != NULL)
82 {
83 PlumaDocument *d;
84 GFile *l;
85
86 d = PLUMA_DOCUMENT (docs->data);
87
88 l = pluma_document_get_location (d);
89 if (l != NULL)
90 {
91 if (g_file_equal (l, file))
92 {
93 tab = pluma_tab_get_from_document (d);
94 g_object_unref (l);
95 break;
96 }
97
98 g_object_unref (l);
99 }
100
101 docs = g_list_next (docs);
102 }
103
104 return tab;
105}

107static gboolean
108is_duplicated_file (GSList *files, GFile *file)
109{
110 while (files != NULL)
111 {
112 if (g_file_equal (files->data, file))
113 return TRUE;
114
115 files = g_slist_next (files);
116 }
117
118 return FALSE;
119}

121/* File loading */
122static gint
123load_file_list (PlumaWindow *window,
124 GSList *files,
125 const PlumaEncoding *encoding,
126 gint line_pos,
127 gboolean create)
128{
129 PlumaTab *tab;
130 gint loaded_files = 0; /* Number of files to load */
131 gboolean jump_to = TRUE; /* Whether to jump to the new tab */
132 GList *win_docs;
133 GSList *files_to_load = NULL;
134 GSList *l;
135
136 pluma_debug (DEBUG_COMMANDS);
137
138 win_docs = pluma_window_get_documents (window);
139
140 /* Remove the uris corresponding to documents already open
141 * in "window" and remove duplicates from "uris" list */
142 for (l = files; l != NULL; l = l->next)
143 {
144 if (!is_duplicated_file (files_to_load, l->data))
145 {
146 tab = get_tab_from_file (win_docs, l->data);
147 if (tab != NULL)
148 {
149 if (l == files)
150 {
151 pluma_window_set_active_tab (window, tab);
152 jump_to = FALSE;
153
154 if (line_pos > 0)
155 {
156 PlumaDocument *doc;
157 PlumaView *view;
158
159 doc = pluma_tab_get_document (tab);
160 view = pluma_tab_get_view (tab);
161
162 /* document counts lines starting from 0 */
163 pluma_document_goto_line (doc, line_pos - 1);
164 pluma_view_scroll_to_cursor (view);
165 }
166 }
167
168 ++loaded_files;
169 }
170 else
171 {
172 files_to_load = g_slist_prepend (files_to_load,
173 l->data);
174 }
175 }
176 }
177
178 g_list_free (win_docs);
179
180 if (files_to_load == NULL)
181 return loaded_files;
182
183 files_to_load = g_slist_reverse (files_to_load);
184 l = files_to_load;
185
186 tab = pluma_window_get_active_tab (window);
187 if (tab != NULL)
188 {
189 PlumaDocument *doc;
190
191 doc = pluma_tab_get_document (tab);
192
193 if (pluma_document_is_untouched (doc) &&
194 (pluma_tab_get_state (tab) == PLUMA_TAB_STATE_NORMAL))
195 {
196 gchar *uri;
197
198 // FIXME: pass the GFile to tab when api is there
199 uri = g_file_get_uri (l->data);
200 _pluma_tab_load (tab,
201 uri,
202 encoding,
203 line_pos,
204 create);
205 g_free (uri);
206
207 l = g_slist_next (l);
208 jump_to = FALSE;
209
210 ++loaded_files;
211 }
212 }
213
214 while (l != NULL)
215 {
216 gchar *uri;
217
218 g_return_val_if_fail (l->data != NULL, 0);
219
220 // FIXME: pass the GFile to tab when api is there
221 uri = g_file_get_uri (l->data);
222 tab = pluma_window_create_tab_from_uri (window,
223 uri,
224 encoding,
225 line_pos,
226 create,
227 jump_to);
228 g_free (uri);
229
230 if (tab != NULL)
231 {
232 jump_to = FALSE;
233 ++loaded_files;
234 }
235
236 l = g_slist_next (l);
237 }
238
239 if (loaded_files == 1)
240 {
241 PlumaDocument *doc;
242 gchar *uri_for_display;
243
244 g_return_val_if_fail (tab != NULL, loaded_files);
245
246 doc = pluma_tab_get_document (tab);
247 uri_for_display = pluma_document_get_uri_for_display (doc);
248
249 pluma_statusbar_flash_message (PLUMA_STATUSBAR (window->priv->statusbar),
250 window->priv->generic_message_cid,
251 _("Loading file '%s'\342\200\246"),
252 uri_for_display);
253
254 g_free (uri_for_display);
255 }
256 else
257 {
258 pluma_statusbar_flash_message (PLUMA_STATUSBAR (window->priv->statusbar),
259 window->priv->generic_message_cid,
260 ngettext("Loading %d file\342\200\246",
261 "Loading %d files\342\200\246",
262 loaded_files),
263 loaded_files);
264 }
265
266 /* Free uris_to_load. Note that l points to the first element of uris_to_load */
267 g_slist_free (files_to_load);
268
269 return loaded_files;
270}

276static gint
277load_uri_list (PlumaWindow *window,
278 const GSList *uris,
279 const PlumaEncoding *encoding,
280 gint line_pos,
281 gboolean create)
282{
283 GSList *files = NULL;
284 const GSList *u;
285 gint ret;
286
287 for (u = uris; u != NULL; u = u->next)
288 {
289 gchar *uri = u->data;
290
291 if (pluma_utils_is_valid_uri (uri))
292 files = g_slist_prepend (files, g_file_new_for_uri (uri));
293 else
294 g_warning ("invalid uri: %s", uri);
295 }
296 files = g_slist_reverse (files);
297
298 ret = load_file_list (window, files, encoding, line_pos, create);
299
300 g_slist_free_full (files, g_object_unref);
301
302 return ret;
303}

548/* FIXME: modify this dialog to be similar to the one provided by gtk+ for
549 * already existing files - Paolo (Oct. 11, 2005) */
550static gboolean
551replace_read_only_file (GtkWindow *parent, GFile *file)
552{
553 GtkWidget *dialog;
554 gint ret;
555 gchar *parse_name;
556 gchar *name_for_display;
557
558 pluma_debug (DEBUG_COMMANDS);
559
560 parse_name = g_file_get_parse_name (file);
561
562 /* Truncate the name so it doesn't get insanely wide. Note that even
563 * though the dialog uses wrapped text, if the name doesn't contain
564 * white space then the text-wrapping code is too stupid to wrap it.
565 */
566 name_for_display = pluma_utils_str_middle_truncate (parse_name, 50);
567 g_free (parse_name);
568
569 dialog = gtk_message_dialog_new (parent,
570 GTK_DIALOG_DESTROY_WITH_PARENT,
571 GTK_MESSAGE_QUESTION,
572 GTK_BUTTONS_NONE,
573 _("The file \"%s\" is read-only."),
574 name_for_display);
575 g_free (name_for_display);
576
577 gtk_message_dialog_format_secondary_text (GTK_MESSAGE_DIALOG (dialog),
578 _("Do you want to try to replace it "
579 "with the one you are saving?"));
580
581 pluma_dialog_add_button (GTK_DIALOG (dialog),
582 _("_Cancel"),
583 "process-stop",
584 GTK_RESPONSE_CANCEL);
585
586 pluma_dialog_add_button (GTK_DIALOG (dialog),
587 _("_Replace"),
588 "document-save-as",
589 GTK_RESPONSE_YES);
590
591 gtk_dialog_set_default_response (GTK_DIALOG (dialog),
592 GTK_RESPONSE_CANCEL);
593
594 gtk_window_set_resizable (GTK_WINDOW (dialog), FALSE);
595
596 ret = gtk_dialog_run (GTK_DIALOG (dialog));
597
598 gtk_widget_destroy (dialog);
599
600 return (ret == GTK_RESPONSE_YES);
601}

603static void
604save_dialog_response_cb (PlumaFileChooserDialog *dialog,
605 gint response_id,
606 PlumaWindow *window)
607{
608 GFile *file;
609 const PlumaEncoding *encoding;
610 PlumaTab *tab;
611 gpointer data;
612 GSList *tabs_to_save_as;
613 PlumaDocumentNewlineType newline_type;
614
615 pluma_debug (DEBUG_COMMANDS);
616
617 tab = PLUMA_TAB (g_object_get_data (G_OBJECT (dialog),
618 PLUMA_TAB_TO_SAVE_AS));
619
620 if (response_id != GTK_RESPONSE_OK)
621 {
622 gtk_widget_destroy (GTK_WIDGET (dialog));
623
624 goto save_next_tab;
625 }
626
627 file = gtk_file_chooser_get_file (GTK_FILE_CHOOSER (dialog));
628 g_return_if_fail (file != NULL);
629
630 encoding = pluma_file_chooser_dialog_get_encoding (dialog);
631 newline_type = pluma_file_chooser_dialog_get_newline_type (dialog);
632
633 gtk_widget_destroy (GTK_WIDGET (dialog));
634
635 if (tab != NULL)
636 {
637 PlumaDocument *doc;
638 gchar *parse_name;
639 gchar *uri;
640
641 doc = pluma_tab_get_document (tab);
642 g_return_if_fail (PLUMA_IS_DOCUMENT (doc));
643
644 parse_name = g_file_get_parse_name (file);
645
646 pluma_statusbar_flash_message (PLUMA_STATUSBAR (window->priv->statusbar),
647 window->priv->generic_message_cid,
648 _("Saving file '%s'\342\200\246"),
649 parse_name);
650
651 g_free (parse_name);
652
653 /* let's remember the dir we navigated too,
654 * even if the saving fails... */
655 _pluma_window_set_default_location (window, file);
656
657 // FIXME: pass the GFile to tab when api is there
658 uri = g_file_get_uri (file);
659 _pluma_tab_save_as (tab, uri, encoding, newline_type);
660 g_free (uri);
661 }
662
663 g_object_unref (file);
664
665save_next_tab:
666
667 data = g_object_get_data (G_OBJECT (window),
668 PLUMA_LIST_OF_TABS_TO_SAVE_AS);
669 if (data == NULL)
670 return;
671
672 /* Save As the next tab of the list (we are Saving All files) */
673 tabs_to_save_as = (GSList *)data;
674 g_return_if_fail (tab == PLUMA_TAB (tabs_to_save_as->data));
675
676 /* Remove the first item of the list */
677 tabs_to_save_as = g_slist_delete_link (tabs_to_save_as,
678 tabs_to_save_as);
679
680 g_object_set_data (G_OBJECT (window),
681 PLUMA_LIST_OF_TABS_TO_SAVE_AS,
682 tabs_to_save_as);
683
684 if (tabs_to_save_as != NULL)
685 {
686 tab = PLUMA_TAB (tabs_to_save_as->data);
687
688 if (GPOINTER_TO_BOOLEAN (g_object_get_data (G_OBJECT (tab),
689 PLUMA_IS_CLOSING_TAB)) == TRUE)
690 {
691 g_object_set_data (G_OBJECT (tab),
692 PLUMA_IS_CLOSING_TAB,
693 NULL);
694
695 /* Trace tab state changes */
696 g_signal_connect (tab,
697 "notify::state",
698 G_CALLBACK (tab_state_changed_while_saving),
699 window);
700 }
701
702 pluma_window_set_active_tab (window, tab);
703 file_save_as (tab, window);
704 }

707static GtkFileChooserConfirmation
708confirm_overwrite_callback (GtkFileChooser *dialog,
709 gpointer data)
710{
711 gchar *uri;
712 GFile *file;
713 GtkFileChooserConfirmation res;
714
715 pluma_debug (DEBUG_COMMANDS);
716
717 uri = gtk_file_chooser_get_uri (dialog);
718 file = g_file_new_for_uri (uri);
719 g_free (uri);
720
721 if (is_read_only (file))
722 {
723 if (replace_read_only_file (GTK_WINDOW (dialog), file))
724 res = GTK_FILE_CHOOSER_CONFIRMATION_ACCEPT_FILENAME;
725 else
726 res = GTK_FILE_CHOOSER_CONFIRMATION_SELECT_AGAIN;
727 }
728 else
729 {
730 /* fall back to the default confirmation dialog */
731 res = GTK_FILE_CHOOSER_CONFIRMATION_CONFIRM;
732 }
733
734 g_object_unref (file);
735
736 return res;
737}

739static void
740file_save_as (PlumaTab *tab,
741 PlumaWindow *window)
742{
743 GtkWidget *save_dialog;
744 GtkWindowGroup *wg;
745 PlumaDocument *doc;
746 GFile *file;
747 gboolean uri_set = FALSE;
748 const PlumaEncoding *encoding;
749 PlumaDocumentNewlineType newline_type;
750
751 g_return_if_fail (PLUMA_IS_TAB (tab));
752 g_return_if_fail (PLUMA_IS_WINDOW (window));
753
754 pluma_debug (DEBUG_COMMANDS);
755
756 save_dialog = pluma_file_chooser_dialog_new (_("Save As\342\200\246"),
757 GTK_WINDOW (window),
758 GTK_FILE_CHOOSER_ACTION_SAVE,
759 NULL,
760 "process-stop", GTK_RESPONSE_CANCEL,
761 "document-save", GTK_RESPONSE_OK,
762 NULL);
763
764 gtk_file_chooser_set_do_overwrite_confirmation (GTK_FILE_CHOOSER (save_dialog),
765 TRUE);
766 g_signal_connect (save_dialog,
767 "confirm-overwrite",
768 G_CALLBACK (confirm_overwrite_callback),
769 NULL);
770
771 wg = pluma_window_get_group (window);
772
773 gtk_window_group_add_window (wg,
774 GTK_WINDOW (save_dialog));
775
776 /* Save As dialog is modal to its main window */
777 gtk_window_set_modal (GTK_WINDOW (save_dialog), TRUE);
778
779 /* Set the suggested file name */
780 doc = pluma_tab_get_document (tab);
781 file = pluma_document_get_location (doc);
782
783 if (file != NULL)
784 {
785 uri_set = gtk_file_chooser_set_file (GTK_FILE_CHOOSER (save_dialog),
786 file,
787 NULL);
788
789 g_object_unref (file);
790 }
791
792
793 if (!uri_set)
794 {
795 GFile *default_path;
796 gchar *docname;
797
798 default_path = _pluma_window_get_default_location (window);
799 docname = pluma_document_get_short_name_for_display (doc);
800
801 if (default_path != NULL)
802 {
803 gchar *uri;
804
805 uri = g_file_get_uri (default_path);
806 gtk_file_chooser_set_current_folder_uri (GTK_FILE_CHOOSER (save_dialog),
807 uri);
808
809 g_free (uri);
810 g_object_unref (default_path);

924 * The docs in the list must belong to the same PlumaWindow.
925 */
926void
927_pluma_cmd_file_save_documents_list (PlumaWindow *window,
928 GList *docs)
929{
930 GList *l;
931 GSList *tabs_to_save_as = NULL;
932
933 pluma_debug (DEBUG_COMMANDS);
934
935 g_return_if_fail (!(pluma_window_get_state (window) &
936 (PLUMA_WINDOW_STATE_PRINTING |
937 PLUMA_WINDOW_STATE_SAVING_SESSION)));
938
939 l = docs;
940 while (l != NULL)
941 {
942 PlumaDocument *doc;
943 PlumaTab *t;
944 PlumaTabState state;
945
946 g_return_if_fail (PLUMA_IS_DOCUMENT (l->data));
947
948 doc = PLUMA_DOCUMENT (l->data);
949 t = pluma_tab_get_from_document (doc);
950 state = pluma_tab_get_state (t);
951
952 g_return_if_fail (state != PLUMA_TAB_STATE_PRINTING);
953 g_return_if_fail (state != PLUMA_TAB_STATE_PRINT_PREVIEWING);
954 g_return_if_fail (state != PLUMA_TAB_STATE_CLOSING);
955
956 if ((state == PLUMA_TAB_STATE_NORMAL) ||
957 (state == PLUMA_TAB_STATE_SHOWING_PRINT_PREVIEW) ||
958 (state == PLUMA_TAB_STATE_GENERIC_NOT_EDITABLE))
959 {
960 /* FIXME: manage the case of local readonly files owned by the
961 user is running pluma - Paolo (Dec. 8, 2005) */
962 if (pluma_document_is_untitled (doc) ||
963 pluma_document_get_readonly (doc))
964 {
965 if (document_needs_saving (doc))
966 {
967 tabs_to_save_as = g_slist_prepend (tabs_to_save_as,
968 t);
969 }
970 }
971 else
972 {
973 file_save (t, window);
974 }
975 }
976 else
977 {
978 /* If the state is:
979 - PLUMA_TAB_STATE_LOADING: we do not save since we are sure the file is unmodified
980 - PLUMA_TAB_STATE_REVERTING: we do not save since the user wants
981 to return back to the version of the file she previously saved
982 - PLUMA_TAB_STATE_SAVING: well, we are already saving (no need to save again)
983 - PLUMA_TAB_STATE_PRINTING, PLUMA_TAB_STATE_PRINT_PREVIEWING: there is not a
984 real reason for not saving in this case, we do not save to avoid to run
985 two operations using the message area at the same time (may be we can remove
986 this limitation in the future). Note that SaveAll, ClosAll
987 and Quit are unsensitive if the window state is PRINTING.
988 - PLUMA_TAB_STATE_GENERIC_ERROR: we do not save since the document contains
989 errors (I don't think this is a very frequent case, we should probably remove
990 this state)
991 - PLUMA_TAB_STATE_LOADING_ERROR: there is nothing to save
992 - PLUMA_TAB_STATE_REVERTING_ERROR: there is nothing to save and saving the current
993 document will overwrite the copy of the file the user wants to go back to
994 - PLUMA_TAB_STATE_SAVING_ERROR: we do not save since we just failed to save, so there is
995 no reason to automatically retry... we wait for user intervention
996 - PLUMA_TAB_STATE_CLOSING: this state is invalid in this case
997 */
998
999 gchar *uri_for_display;
1000
1001 uri_for_display = pluma_document_get_uri_for_display (doc);
1002 pluma_debug_message (DEBUG_COMMANDS,
1003 "File '%s' not saved. State: %d",
1004 uri_for_display,
1005 state);
1006 g_free (uri_for_display);
1007 }
1008
1009 l = g_list_next (l);
1010 }
1011
1012 if (tabs_to_save_as != NULL)
1013 {
1014 PlumaTab *tab;
1015
1016 tabs_to_save_as = g_slist_reverse (tabs_to_save_as );
1017
1018 g_return_if_fail (g_object_get_data (G_OBJECT (window),
1019 PLUMA_LIST_OF_TABS_TO_SAVE_AS) == NULL);
1020
1021 g_object_set_data (G_OBJECT (window),
1022 PLUMA_LIST_OF_TABS_TO_SAVE_AS,
1023 tabs_to_save_as);
1024
1025 tab = PLUMA_TAB (tabs_to_save_as->data);
1026
1027 pluma_window_set_active_tab (window, tab);
1028 file_save_as (tab, window);
1029 }

1389static void
1390save_and_close_all_documents (const GList *docs,
1391 PlumaWindow *window)
1392{
1393 GList *tabs;
1394 GList *l;
1395 GSList *sl;
1396 GSList *tabs_to_save_as;
1397 GSList *tabs_to_save_and_close;
1398 GList *tabs_to_close;
1399
1400 pluma_debug (DEBUG_COMMANDS);
1401
1402 g_return_if_fail (!(pluma_window_get_state (window) & PLUMA_WINDOW_STATE_PRINTING));
1403
1404 tabs = gtk_container_get_children (
1405 GTK_CONTAINER (_pluma_window_get_notebook (window)));
1406
1407 tabs_to_save_as = NULL;
1408 tabs_to_save_and_close = NULL;
1409 tabs_to_close = NULL;
1410
1411 l = tabs;
1412 while (l != NULL)
1413 {
1414 PlumaTab *t;
1415 PlumaTabState state;
1416 PlumaDocument *doc;
1417
1418 t = PLUMA_TAB (l->data);
1419
1420 state = pluma_tab_get_state (t);
1421 doc = pluma_tab_get_document (t);
1422
1423 /* If the state is: ([*] invalid states)
1424 - PLUMA_TAB_STATE_NORMAL: close (and if needed save)
1425 - PLUMA_TAB_STATE_LOADING: close, we are sure the file is unmodified
1426 - PLUMA_TAB_STATE_REVERTING: since the user wants
1427 to return back to the version of the file she previously saved, we can close
1428 without saving (CHECK: are we sure this is the right behavior, suppose the case
1429 the original file has been deleted)
1430 - [*] PLUMA_TAB_STATE_SAVING: invalid, ClosAll
1431 and Quit are unsensitive if the window state is SAVING.
1432 - [*] PLUMA_TAB_STATE_PRINTING, PLUMA_TAB_STATE_PRINT_PREVIEWING: there is not a
1433 real reason for not closing in this case, we do not save to avoid to run
1434 two operations using the message area at the same time (may be we can remove
1435 this limitation in the future). Note that ClosAll
1436 and Quit are unsensitive if the window state is PRINTING.
1437 - PLUMA_TAB_STATE_SHOWING_PRINT_PREVIEW: close (and if needed save)
1438 - PLUMA_TAB_STATE_LOADING_ERROR: close without saving (if the state is LOADING_ERROR then the
1439 document is not modified)
1440 - PLUMA_TAB_STATE_REVERTING_ERROR: we do not close since the document contains errors
1441 - PLUMA_TAB_STATE_SAVING_ERROR: we do not close since the document contains errors
1442 - PLUMA_TAB_STATE_GENERIC_ERROR: we do not close since the document contains
1443 errors (CHECK: we should problably remove this state)
1444 - [*] PLUMA_TAB_STATE_CLOSING: this state is invalid in this case
1445 */
1446
1447 g_return_if_fail (state != PLUMA_TAB_STATE_PRINTING);
1448 g_return_if_fail (state != PLUMA_TAB_STATE_PRINT_PREVIEWING);
1449 g_return_if_fail (state != PLUMA_TAB_STATE_CLOSING);
1450 g_return_if_fail (state != PLUMA_TAB_STATE_SAVING);
1451
1452 if ((state != PLUMA_TAB_STATE_SAVING_ERROR) &&
1453 (state != PLUMA_TAB_STATE_GENERIC_ERROR) &&
1454 (state != PLUMA_TAB_STATE_REVERTING_ERROR))
1455 {
1456 if ((g_list_index ((GList *)docs, doc) >= 0) &&
1457 (state != PLUMA_TAB_STATE_LOADING) &&
1458 (state != PLUMA_TAB_STATE_LOADING_ERROR) &&
1459 (state != PLUMA_TAB_STATE_REVERTING)) /* CHECK: is this the right behavior with REVERTING ?*/
1460 {
1461 /* The document must be saved before closing */
1462 g_return_if_fail (document_needs_saving (doc));
1463
1464 /* FIXME: manage the case of local readonly files owned by the
1465 user is running pluma - Paolo (Dec. 8, 2005) */
1466 if (pluma_document_is_untitled (doc) ||
1467 pluma_document_get_readonly (doc))
1468 {
1469 g_object_set_data (G_OBJECT (t),
1470 PLUMA_IS_CLOSING_TAB,
1471 GBOOLEAN_TO_POINTER (TRUE));
1472
1473 tabs_to_save_as = g_slist_prepend (tabs_to_save_as,
1474 t);
1475 }
1476 else
1477 {
1478 tabs_to_save_and_close = g_slist_prepend (tabs_to_save_and_close,
1479 t);
1480 }
1481 }
1482 else
1483 {
1484 /* The document must be closed without saving */
1485 tabs_to_close = g_list_prepend (tabs_to_close,
1486 t);
1487 }
1488 }
1489
1490 l = g_list_next (l);
1491 }
1492
1493 g_list_free (tabs);
1494
1495 /* Close all tabs to close (in a sync way) */
1496 pluma_window_close_tabs (window, tabs_to_close);
1497 g_list_free (tabs_to_close);
1498
1499 /* Save and close all the files in tabs_to_save_and_close */
1500 sl = tabs_to_save_and_close;
1501 while (sl != NULL)
1502 {
1503 save_and_close (PLUMA_TAB (sl->data),
1504 window);
1505 sl = g_slist_next (sl);
1506 }
1507 g_slist_free (tabs_to_save_and_close);
1508
1509 /* Save As and close all the files in tabs_to_save_as */
1510 if (tabs_to_save_as != NULL)
1511 {
