210gboolean
211egg_accelerator_parse_virtual (const gchar *accelerator,
212 guint *accelerator_key,
213 guint *keycode,
214 EggVirtualModifierType *accelerator_mods)
215{
216 guint keyval;
217 GdkModifierType mods;
218 gint len;
219 gboolean bad_keyval;
220
221 if (accelerator_key)
222 *accelerator_key = 0;
223 if (accelerator_mods)
224 *accelerator_mods = 0;
225 if (keycode)
226 *keycode = 0;
227
228 g_return_val_if_fail (accelerator != NULL, FALSE);
229
230 bad_keyval = FALSE;
231
232 keyval = 0;
233 mods = 0;
234 len = strlen (accelerator);
235 while (len)
236 {
237 if (*accelerator == '<')
238 {
239 if (len >= 9 && is_release (accelerator))
240 {
241 accelerator += 9;
242 len -= 9;
243 mods |= EGG_VIRTUAL_RELEASE_MASK;
244 }
245 else if (len >= 9 && is_control (accelerator))
246 {
247 accelerator += 9;
248 len -= 9;
249 mods |= EGG_VIRTUAL_CONTROL_MASK;
250 }
251 else if (len >= 9 && is_primary (accelerator))
252 {
253 accelerator += 9;
254 len -= 9;
255 mods |= EGG_VIRTUAL_CONTROL_MASK;
256 }
257 else if (len >= 7 && is_shift (accelerator))
258 {
259 accelerator += 7;
260 len -= 7;
261 mods |= EGG_VIRTUAL_SHIFT_MASK;
262 }
263 else if (len >= 6 && is_shft (accelerator))
264 {
265 accelerator += 6;
266 len -= 6;
267 mods |= EGG_VIRTUAL_SHIFT_MASK;
268 }
269 else if (len >= 6 && is_ctrl (accelerator))
270 {
271 accelerator += 6;
272 len -= 6;
273 mods |= EGG_VIRTUAL_CONTROL_MASK;
274 }
275 else if (len >= 6 && is_modx (accelerator))
276 {
277 static const guint mod_vals[] = {
278 EGG_VIRTUAL_ALT_MASK, EGG_VIRTUAL_MOD2_MASK, EGG_VIRTUAL_MOD3_MASK,
279 EGG_VIRTUAL_MOD4_MASK, EGG_VIRTUAL_MOD5_MASK
280 };
281
282 len -= 6;
283 accelerator += 4;
284 mods |= mod_vals[*accelerator - '1'];
285 accelerator += 2;
286 }
287 else if (len >= 5 && is_ctl (accelerator))
288 {
289 accelerator += 5;
290 len -= 5;
291 mods |= EGG_VIRTUAL_CONTROL_MASK;
292 }
293 else if (len >= 5 && is_alt (accelerator))
294 {
295 accelerator += 5;
296 len -= 5;
297 mods |= EGG_VIRTUAL_ALT_MASK;
298 }
299 else if (len >= 6 && is_meta (accelerator))
300 {
301 accelerator += 6;
302 len -= 6;
303 mods |= EGG_VIRTUAL_META_MASK;
304 }
305 else if (len >= 7 && is_hyper (accelerator))
306 {
307 accelerator += 7;
308 len -= 7;
309 mods |= EGG_VIRTUAL_HYPER_MASK;
310 }
311 else if (len >= 7 && is_super (accelerator))
312 {
313 accelerator += 7;
314 len -= 7;
315 mods |= EGG_VIRTUAL_SUPER_MASK;
316 }
317 else
318 {
319 gchar last_ch;
320
321 last_ch = *accelerator;
322 while (last_ch && last_ch != '>')
323 {
324 last_ch = *accelerator;
325 accelerator += 1;
326 len -= 1;
327 }
328 }
329 }
330 else
331 {
332 keyval = gdk_keyval_from_name (accelerator);
333
334 if (keyval == 0)
335 {
336 /* If keyval is 0, then maybe it's a keycode. Check for 0x## */
337 if (len >= 4 && is_keycode (accelerator))
338 {
339 char keystring[5];
340 gchar *endptr;
341 gint tmp_keycode;
342
343 memcpy (keystring, accelerator, 4);
344 keystring [4] = '\000';
345
346 tmp_keycode = strtol (keystring, &endptr, 16);
347
348 if (endptr == NULL || *endptr != '\000')
349 {
350 bad_keyval = TRUE;
351 }
352 else if (keycode != NULL)
353 {
354 *keycode = tmp_keycode;
355 /* 0x00 is an invalid keycode too. */
356 if (*keycode == 0)
357 bad_keyval = TRUE;
358 }
359 }
360 }
361 else if (keycode != NULL)
362 {
363 *keycode = XKeysymToKeycode (GDK_DISPLAY_XDISPLAY(gdk_display_get_default()), keyval);
364 if (*keycode == 0)
365 bad_keyval = TRUE;
366 }
367
368 accelerator += len;
369 len -= len;
370 }
371 }
372
373 if (accelerator_key)
374 *accelerator_key = gdk_keyval_to_lower (keyval);
375 if (accelerator_mods)
376 *accelerator_mods = mods;
377
378 return !bad_keyval;
379}

454void egg_keymap_resolve_virtual_modifiers (GdkKeymap* keymap, EggVirtualModifierType virtual_mods, GdkModifierType* concrete_mods)
455{
456 GdkModifierType concrete;
457 int i;
458 const EggModmap* modmap;
459
460 g_return_if_fail (concrete_mods != NULL);
461 g_return_if_fail (keymap == NULL || GDK_IS_KEYMAP (keymap));
462
463 modmap = egg_keymap_get_modmap(keymap);
464
465 /* Not so sure about this algorithm. */
466 concrete = 0;
467
468 for (i = 0; i < EGG_MODMAP_ENTRY_LAST; ++i)
469 {
470 if (modmap->mapping[i] & virtual_mods)
471 {
472 concrete |= MODMAP_ENTRY_TO_MODIFIER (i);
473 }
474 }
475
476 *concrete_mods = concrete;
477}

479void
480egg_keymap_virtualize_modifiers (GdkKeymap *keymap,
481 GdkModifierType concrete_mods,
482 EggVirtualModifierType *virtual_mods)
483{
484 GdkModifierType virtual;
485 int i;
486 const EggModmap *modmap;
487
488 g_return_if_fail (virtual_mods != NULL);
489 g_return_if_fail (keymap == NULL || GDK_IS_KEYMAP (keymap));
490
491 modmap = egg_keymap_get_modmap (keymap);
492
493 /* Not so sure about this algorithm. */
494
495 virtual = 0;
496 for (i = 0; i < EGG_MODMAP_ENTRY_LAST; ++i)
497 {
498 if (MODMAP_ENTRY_TO_MODIFIER (i) & concrete_mods)
499 {
500 EggVirtualModifierType cleaned;
501
502 cleaned = modmap->mapping[i] & ~(EGG_VIRTUAL_MOD2_MASK |
503 EGG_VIRTUAL_MOD3_MASK |
504 EGG_VIRTUAL_MOD4_MASK |
505 EGG_VIRTUAL_MOD5_MASK);
506
507 if (cleaned != 0)
508 {
509 virtual |= cleaned;
510 }
511 else
512 {
513 /* Rather than dropping mod2->mod5 if not bound,
514 * go ahead and use the concrete names
515 */
516 virtual |= modmap->mapping[i];
517 }
518 }
519 }
520
521 *virtual_mods = virtual;
522}

524static void
525reload_modmap (GdkKeymap *keymap,
526 EggModmap *modmap)
527{
528 XModifierKeymap *xmodmap;
529 int map_size;
530 int i;
531
532 /* FIXME multihead */
533 xmodmap = XGetModifierMapping (gdk_x11_get_default_xdisplay ());
534
535 memset (modmap->mapping, 0, sizeof (modmap->mapping));
536
537 /* there are 8 modifiers in the order shift, shift lock,
538 * control, mod1-5 with up to max_keypermod bindings each
539 */
540 map_size = 8 * xmodmap->max_keypermod;
541 for (i = 3 * xmodmap->max_keypermod; i < map_size; ++i)
542 {
543 /* get the key code at this point in the map,
544 * see if its keysym is one we're interested in
545 */
546 int keycode = xmodmap->modifiermap[i];
547 GdkKeymapKey *keys;
548 guint *keyvals;
549 int n_entries;
550 int j;
551 EggVirtualModifierType mask;
552
553 keys = NULL;
554 keyvals = NULL;
555 n_entries = 0;
556
557 gdk_keymap_get_entries_for_keycode (keymap,
558 keycode,
559 &keys, &keyvals, &n_entries);
560
561 mask = 0;
562 for (j = 0; j < n_entries; ++j)
563 {
564 if (keyvals[j] == GDK_KEY_Num_Lock)
565 mask |= EGG_VIRTUAL_NUM_LOCK_MASK;
566 else if (keyvals[j] == GDK_KEY_Scroll_Lock)
567 mask |= EGG_VIRTUAL_SCROLL_LOCK_MASK;
568 else if (keyvals[j] == GDK_KEY_Meta_L ||
569 keyvals[j] == GDK_KEY_Meta_R)
570 mask |= EGG_VIRTUAL_META_MASK;
571 else if (keyvals[j] == GDK_KEY_Hyper_L ||
572 keyvals[j] == GDK_KEY_Hyper_R)
573 mask |= EGG_VIRTUAL_HYPER_MASK;
574 else if (keyvals[j] == GDK_KEY_Super_L ||
575 keyvals[j] == GDK_KEY_Super_R)
576 mask |= EGG_VIRTUAL_SUPER_MASK;
577 else if (keyvals[j] == GDK_KEY_Mode_switch)
578 mask |= EGG_VIRTUAL_MODE_SWITCH_MASK;
579 }
580
581 /* Mod1Mask is 1 << 3 for example, i.e. the
582 * fourth modifier, i / keyspermod is the modifier
583 * index
584 */
585 modmap->mapping[i/xmodmap->max_keypermod] |= mask;
586
587 g_free (keyvals);
588 g_free (keys);
589 }
590
591 /* Add in the not-really-virtual fixed entries */
592 modmap->mapping[EGG_MODMAP_ENTRY_SHIFT] |= EGG_VIRTUAL_SHIFT_MASK;
593 modmap->mapping[EGG_MODMAP_ENTRY_CONTROL] |= EGG_VIRTUAL_CONTROL_MASK;
594 modmap->mapping[EGG_MODMAP_ENTRY_LOCK] |= EGG_VIRTUAL_LOCK_MASK;
595 modmap->mapping[EGG_MODMAP_ENTRY_MOD1] |= EGG_VIRTUAL_ALT_MASK;
596 modmap->mapping[EGG_MODMAP_ENTRY_MOD2] |= EGG_VIRTUAL_MOD2_MASK;
597 modmap->mapping[EGG_MODMAP_ENTRY_MOD3] |= EGG_VIRTUAL_MOD3_MASK;
598 modmap->mapping[EGG_MODMAP_ENTRY_MOD4] |= EGG_VIRTUAL_MOD4_MASK;
599 modmap->mapping[EGG_MODMAP_ENTRY_MOD5] |= EGG_VIRTUAL_MOD5_MASK;
600
601 XFreeModifiermap (xmodmap);
602}

604const EggModmap*
605egg_keymap_get_modmap (GdkKeymap *keymap)
606{
607 EggModmap *modmap;
608
609 if (keymap == NULL)
610 keymap = gdk_keymap_get_for_display (gdk_display_get_default ());
611
612 /* This is all a hack, much simpler when we can just
613 * modify GDK directly.
614 */
615
616 modmap = g_object_get_data (G_OBJECT (keymap), "egg-modmap");
617
618 if (modmap == NULL)
619 {
620 modmap = g_new0 (EggModmap, 1);
621
622 /* FIXME modify keymap change events with an event filter
623 * and force a reload if we get one
624 */
625
626 reload_modmap (keymap, modmap);
627
628 g_object_set_data_full (G_OBJECT (keymap),
629 "egg-modmap",
630 modmap,
631 g_free);
632 }
633
634 g_assert (modmap != NULL);
635
636 return modmap;
