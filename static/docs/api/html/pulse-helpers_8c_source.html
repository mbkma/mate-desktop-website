73MateMixerStreamControlMediaRole
74pulse_convert_media_role_name (const gchar *name)
75{
76 g_return_val_if_fail (name != NULL, MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_UNKNOWN);
77
78 if (!strcmp (name, "video")) {
79 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_VIDEO;
80 }
81 else if (!strcmp (name, "music")) {
82 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_MUSIC;
83 }
84 else if (!strcmp (name, "game")) {
85 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_GAME;
86 }
87 else if (!strcmp (name, "event")) {
88 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_EVENT;
89 }
90 else if (!strcmp (name, "phone")) {
91 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_PHONE;
92 }
93 else if (!strcmp (name, "animation")) {
94 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_ANIMATION;
95 }
96 else if (!strcmp (name, "production")) {
97 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_PRODUCTION;
98 }
99 else if (!strcmp (name, "a11y")) {
100 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_A11Y;
101 }
102 else if (!strcmp (name, "test")) {
103 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_TEST;
104 }
105 else if (!strcmp (name, "abstract")) {
106 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_ABSTRACT;
107 }
108 else if (!strcmp (name, "filter")) {
109 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_FILTER;
110 }
111
112 return MATE_MIXER_STREAM_CONTROL_MEDIA_ROLE_UNKNOWN;
113}
