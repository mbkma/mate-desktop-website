38static gboolean
39remove_job_from_queue (GQueue *queue, EomJob *job)
40{
41 GList *list;
42
43 list = g_queue_find (queue, job);
44
45 if (list) {
46 g_object_unref (G_OBJECT (job));
47 g_queue_delete_link (queue, list);
48
49 return TRUE;
50 }
51
52 return FALSE;
53}

128static gpointer
129eom_render_thread (gpointer data)
130{
131 while (TRUE) {
132 EomJob *job;
133
134 g_mutex_lock (&eom_queue_mutex);
135
136 if (no_jobs_available_unlocked ()) {
137 g_cond_wait (&render_cond, &eom_queue_mutex);
138 }
139
140 job = search_for_jobs_unlocked ();
141
142 g_mutex_unlock (&eom_queue_mutex);
143
144 /* Now that we have our job, we handle it */
145 if (job) {
146 handle_job (job);
147 g_object_unref (G_OBJECT (job));
148 }
149 }
150 return NULL;
151
152}

170static GQueue *
171find_queue (EomJob *job)
172{
173 if (EOM_IS_JOB_THUMBNAIL (job)) {
174 return thumbnail_queue;

184 return copy_queue;

187 g_assert_not_reached ();
188
189 return NULL;

192void
193eom_job_queue_add_job (EomJob *job)
194{
195 GQueue *queue;
196
197 g_return_if_fail (EOM_IS_JOB (job));
198
199 queue = find_queue (job);
200
201 g_mutex_lock (&eom_queue_mutex);
202
203 add_job_to_queue_locked (queue, job);
204
205 g_mutex_unlock (&eom_queue_mutex);
206}

208gboolean
209eom_job_queue_remove_job (EomJob *job)
210{
211 gboolean retval = FALSE;
212
213 g_return_val_if_fail (EOM_IS_JOB (job), FALSE);
214
215 g_mutex_lock (&eom_queue_mutex);
216
217 if (EOM_IS_JOB_THUMBNAIL (job)) {
218 retval = remove_job_from_queue (thumbnail_queue, job);
219 } else if (EOM_IS_JOB_LOAD (job)) {
220 retval = remove_job_from_queue (load_queue, job);
221 } else if (EOM_IS_JOB_MODEL (job)) {
222 retval = remove_job_from_queue (model_queue, job);
223 } else if (EOM_IS_JOB_TRANSFORM (job)) {
224 retval = remove_job_from_queue (transform_queue, job);
225 } else if (EOM_IS_JOB_SAVE (job)) {
226 retval = remove_job_from_queue (save_queue, job);
227 } else if (EOM_IS_JOB_COPY (job)) {
228 retval = remove_job_from_queue (copy_queue, job);
229 } else {
230 g_assert_not_reached ();
231 }
232
233 g_mutex_unlock (&eom_queue_mutex);
234
235 return retval;
