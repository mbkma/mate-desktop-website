59{
60 GError *error = NULL;
61
62 manager->proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SESSION,
63 G_DBUS_PROXY_FLAGS_NONE,
64 NULL,
65 "org.gnome.SessionManager",
66 "/org/gnome/SessionManager",
67 "org.gnome.SessionManager",
68 NULL,
69 &error);
70 if (manager->proxy == NULL) {
71 g_warning ("Unable to contact session manager daemon: %s\n", error->message);

76void
77panel_session_manager_request_logout (PanelSessionManager *manager,
78 PanelSessionManagerLogoutType mode)
79{
80 GError *error = NULL;
81 GVariant *ret;
82
83 g_return_if_fail (PANEL_IS_SESSION_MANAGER (manager));
84
85 if (manager->proxy == NULL)
86 return;
87
88 ret = g_dbus_proxy_call_sync (manager->proxy, "Logout",
89 g_variant_new ("(u)", mode),
90 G_DBUS_CALL_FLAGS_NONE,
91 -1,
92 NULL,
93 &error);
94 if (ret == NULL) {
95 g_warning ("Could not ask session manager to log out: %s",
96 error->message);
97 g_error_free (error);
98 } else {
99 g_variant_unref (ret);
100 }
101}

129gboolean
130panel_session_manager_is_shutdown_available (PanelSessionManager *manager)
131{
132 GError *error = NULL;
133 gboolean is_shutdown_available;
134 GVariant *ret;
135
136 g_return_val_if_fail (PANEL_IS_SESSION_MANAGER (manager), FALSE);
137
138 if (manager->proxy == NULL)
139 return FALSE;
140
141 ret = g_dbus_proxy_call_sync (manager->proxy, "CanShutdown",
142 g_variant_new ("()"),
143 G_DBUS_CALL_FLAGS_NONE,
144 -1,
145 NULL,
146 &error);
147 if (ret == NULL) {
148 g_warning ("Could not ask session manager if shut down is available: %s",
149 error->message);
150 g_error_free (error);
151 return FALSE;
152 } else {
153 g_variant_get (ret, "(b)", &is_shutdown_available);
154 g_variant_unref (ret);
155 }
156
157 return is_shutdown_available;
158}
