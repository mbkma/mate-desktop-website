49static time_t
50mktime_from_string (const char *time_s)
51{
52 struct tm tm = {0, };
53 tm.tm_isdst = -1;
54 strptime (time_s, LSNOMARCH_DATE_FORMAT, &tm);
55 return mktime (&tm);
56}

58static void
59list__process_line (char *line,
60 gpointer data)
61{
62 FileData *fdata;
63 FrCommand *comm = FR_COMMAND (data);
64 char **fields;
65 int date_idx;
66 char *field_date, *field_name;
67
68 g_return_if_fail (line != NULL);
69
70 date_idx = file_list__get_index_from_pattern (line, "%n%n%n%n-%n%n-%n%n %n%n:%n%n");
71 if (date_idx < 0)
72 return;
73
74 fdata = file_data_new ();
75 fields = split_line (line, 8);
76
77 field_name = fields[0];
78 fdata->full_path = g_strconcat ("/", field_name, NULL);
79 fdata->original_path = fdata->full_path + 1;
80 fdata->name = g_strdup (file_name_from_path (fdata->full_path));
81 fdata->path = remove_level_from_path (fdata->full_path);
82
83 fdata->size = g_ascii_strtoull (fields[4], NULL, 10);
84
85 field_date = g_strndup (line + date_idx, 16);
86 fdata->modified = mktime_from_string (field_date);
87 g_free (field_date);
88
89 g_strfreev (fields);
90
91 if (*fdata->name == 0)
92 file_data_free (fdata);
93 else
94 fr_command_add_file (comm, fdata);
95}

109static void
110fr_command_nomarch_add (FrCommand *comm,
111 const char *from_file,
112 GList *file_list,
113 const char *base_dir,
114 gboolean update,
115 gboolean recursive)
116{
117 GList *scan;
118
119 fr_process_begin_command (comm->process, "arc");
120
121 if (base_dir != NULL)
122 fr_process_set_working_dir (comm->process, base_dir);
123
124 if (update)
125 fr_process_add_arg (comm->process, "u");
126 else
127 fr_process_add_arg (comm->process, "a");
128
129 fr_process_add_arg (comm->process, comm->filename);
130
131 for (scan = file_list; scan; scan = scan->next)
132 fr_process_add_arg (comm->process, (gchar*) scan->data);
133
134 fr_process_end_command (comm->process);
135}
