24gboolean
25panel_a11y_get_is_a11y_enabled (GtkWidget *widget)
26{
27 static gboolean initialised = FALSE;
28 static gboolean a11y_enabled = FALSE;
29
30 if (!initialised) {
31 a11y_enabled = GTK_IS_ACCESSIBLE (gtk_widget_get_accessible (widget));
32 initialised = TRUE;
33 }
34
35 return a11y_enabled;
36}

67void
68panel_a11y_set_atk_relation (GtkWidget *widget,
69 GtkLabel *label)
70{
71 AtkObject *aobject;
72 AtkRelationSet *relation_set;
73 AtkRelation *relation;
74 AtkObject *targets [1];
75
76 g_return_if_fail (GTK_IS_WIDGET(widget));
77 g_return_if_fail (GTK_IS_LABEL(label));
78
79 if (!panel_a11y_get_is_a11y_enabled (widget))
80 return;
81
82 aobject = gtk_widget_get_accessible (widget);
83
84 gtk_label_set_mnemonic_widget (label, widget);
85
86 targets [0] = gtk_widget_get_accessible (GTK_WIDGET (label));
87
88 relation_set = atk_object_ref_relation_set (aobject);
89
90 relation = atk_relation_new (targets, 1, ATK_RELATION_LABELLED_BY);
91 atk_relation_set_add (relation_set, relation);
92 g_object_unref (relation);
93}

110GType
111panel_a11y_query_accessible_parent_type (GType type,
112 GTypeInfo *type_info)
113{
114 AtkObjectFactory *factory;
115 GType parent_type;
116 GType accessible_parent_type;
117
118 g_return_val_if_fail (G_TYPE_IS_OBJECT (type), G_TYPE_INVALID);
119
120 parent_type = g_type_parent (type);
121
122 factory = atk_registry_get_factory (atk_get_default_registry (), parent_type);
123
124 accessible_parent_type = atk_object_factory_get_accessible_type (factory);
125
126 if (type_info) {
127 GTypeQuery query;
128
129 g_type_query (accessible_parent_type, &query);
130
131 type_info->class_size = query.class_size;
132 type_info->instance_size = query.instance_size;
133 }
134
135 return atk_object_factory_get_accessible_type (factory);
