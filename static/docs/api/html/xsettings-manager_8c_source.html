60static Bool
61timestamp_predicate (Display *display,
62 XEvent *xevent,
63 XPointer arg)
64{
65 TimeStampInfo *info = (TimeStampInfo *)arg;
66
67 if (xevent->type == PropertyNotify &&
68 xevent->xproperty.window == info->window &&
69 xevent->xproperty.atom == info->timestamp_prop_atom)
70 return True;
71
72 return False;
73}

86static Time
87get_server_time (Display *display,
88 Window window)
89{
90 unsigned char c = 'a';
91 XEvent xevent;
92 TimeStampInfo info;
93
94 info.timestamp_prop_atom = XInternAtom (display, "_TIMESTAMP_PROP", False);
95 info.window = window;
96
97 XChangeProperty (display, window,
98 info.timestamp_prop_atom, info.timestamp_prop_atom,
99 8, PropModeReplace, &c, 1);
100
101 XIfEvent (display, &xevent,
102 timestamp_predicate, (XPointer)&info);
103
104 return xevent.xproperty.time;
105}

123XSettingsManager *
124xsettings_manager_new (Display *display,
125 int screen,
126 XSettingsTerminateFunc terminate,
127 void *cb_data)
128{
129 XSettingsManager *manager;
130 Time timestamp;
131 XClientMessageEvent xev;
132
133 char buffer[256];
134
135 manager = malloc (sizeof *manager);
136 if (!manager)
137 return NULL;
138
139 manager->display = display;
140 manager->screen = screen;
141
142 sprintf(buffer, "_XSETTINGS_S%d", screen);
143 manager->selection_atom = XInternAtom (display, buffer, False);
144 manager->xsettings_atom = XInternAtom (display, "_XSETTINGS_SETTINGS", False);
145 manager->manager_atom = XInternAtom (display, "MANAGER", False);
146
147 manager->terminate = terminate;
148 manager->cb_data = cb_data;
149
150 manager->settings = NULL;
151 manager->serial = 0;
152 manager->overrides = NULL;
153
154 manager->window = XCreateSimpleWindow (display,
155 RootWindow (display, screen),
156 0, 0, 10, 10, 0,
157 WhitePixel (display, screen),
158 WhitePixel (display, screen));
159
160 XSelectInput (display, manager->window, PropertyChangeMask);
161 timestamp = get_server_time (display, manager->window);
162
163 XSetSelectionOwner (display, manager->selection_atom,
164 manager->window, timestamp);
165
166 /* Check to see if we managed to claim the selection. If not,
167 * we treat it as if we got it then immediately lost it
168 */
169
170 if (XGetSelectionOwner (display, manager->selection_atom) ==
171 manager->window)
172 {
173 xev.type = ClientMessage;
174 xev.window = RootWindow (display, screen);
175 xev.message_type = manager->manager_atom;
176 xev.format = 32;
177 xev.data.l[0] = timestamp;
178 xev.data.l[1] = manager->selection_atom;
179 xev.data.l[2] = manager->window;
180 xev.data.l[3] = 0; /* manager specific data */
181 xev.data.l[4] = 0; /* manager specific data */
182
183 XSendEvent (display, RootWindow (display, screen),
184 False, StructureNotifyMask, (XEvent *)&xev);
185 }
186 else
187 {
188 manager->terminate (manager->cb_data);
189 }
190
191 return manager;
192}

235XSettingsResult
236xsettings_manager_set_setting (XSettingsManager *manager,
237 XSettingsSetting *setting)
238{
239 XSettingsSetting *old_setting = xsettings_list_lookup (settings, setting->name);
240 XSettingsSetting *new_setting;
241 XSettingsResult result;
242
243 if (old_setting)
244 {
245 if (xsettings_setting_equal (old_setting, setting))
246 return XSETTINGS_SUCCESS;
247
248 xsettings_list_delete (&settings, setting->name);
249 }
250
251 new_setting = xsettings_setting_copy (setting);
252 if (!new_setting)
253 return XSETTINGS_NO_MEM;
254
255 new_setting->last_change_serial = manager->serial;
256
257 result = xsettings_list_insert (&settings, new_setting);
258
259 if (result != XSETTINGS_SUCCESS)
260 xsettings_setting_free (new_setting);
261
262 return result;
263}

307static size_t
308setting_length (XSettingsSetting *setting)
309{
310 size_t length = 8; /* type + pad + name-len + last-change-serial */
311 length += XSETTINGS_PAD (strlen (setting->name), 4);
312
313 switch (setting->type)
314 {
315 case XSETTINGS_TYPE_INT:
316 length += 4;
317 break;
318 case XSETTINGS_TYPE_STRING:
319 length += 4 + XSETTINGS_PAD (strlen (setting->data.v_string), 4);
320 break;
321 case XSETTINGS_TYPE_COLOR:
322 length += 8;
323 break;
324 }
325
326 return length;
327}

329static void
330setting_store (XSettingsSetting *setting,
331 XSettingsBuffer *buffer)
332{
333 size_t string_len;
334 size_t length;
335
336 *(buffer->pos++) = setting->type;
337 *(buffer->pos++) = 0;
338
339 string_len = strlen (setting->name);
340 *(CARD16 *)(buffer->pos) = string_len;
341 buffer->pos += 2;
342
343 length = XSETTINGS_PAD (string_len, 4);
344 memcpy (buffer->pos, setting->name, string_len);
345 length -= string_len;
346 buffer->pos += string_len;
347
348 while (length > 0)
349 {
350 *(buffer->pos++) = 0;
351 length--;
352 }
353
354 *(CARD32 *)(buffer->pos) = setting->last_change_serial;
355 buffer->pos += 4;
356
357 switch (setting->type)
358 {
359 case XSETTINGS_TYPE_INT:
360 *(CARD32 *)(buffer->pos) = setting->data.v_int;
361 buffer->pos += 4;
362 break;
363 case XSETTINGS_TYPE_STRING:
364 string_len = strlen (setting->data.v_string);
365 *(CARD32 *)(buffer->pos) = string_len;
366 buffer->pos += 4;
367
368 length = XSETTINGS_PAD (string_len, 4);
369 memcpy (buffer->pos, setting->data.v_string, string_len);
370 length -= string_len;
371 buffer->pos += string_len;
372
373 while (length > 0)
374 {
375 *(buffer->pos++) = 0;
376 length--;
377 }
378 break;
379 case XSETTINGS_TYPE_COLOR:
380 *(CARD16 *)(buffer->pos) = setting->data.v_color.red;
381 *(CARD16 *)(buffer->pos + 2) = setting->data.v_color.green;
382 *(CARD16 *)(buffer->pos + 4) = setting->data.v_color.blue;
383 *(CARD16 *)(buffer->pos + 6) = setting->data.v_color.alpha;
384 buffer->pos += 8;
385 break;
386 }
387}

389XSettingsResult
390xsettings_manager_notify (XSettingsManager *manager)
391{
392 XSettingsBuffer buffer;
393 XSettingsList *iter;
394 int n_settings = 0;
395
396 buffer.len = 12; /* byte-order + pad + SERIAL + N_SETTINGS */
397
398 iter = settings;
399 while (iter)
400 {
401 buffer.len += setting_length (iter->setting);
402 n_settings++;
403 iter = iter->next;
404 }
405
406 buffer.data = buffer.pos = malloc (buffer.len);
407 if (!buffer.data)
408 return XSETTINGS_NO_MEM;
409
410 *buffer.pos = xsettings_byte_order ();
411
412 buffer.pos += 4;
413 *(CARD32 *)buffer.pos = manager->serial++;
414 buffer.pos += 4;
415 *(CARD32 *)buffer.pos = n_settings;
416 buffer.pos += 4;
417
418 iter = settings;
419 while (iter)
420 {
421 setting_store (iter->setting, &buffer);
422 iter = iter->next;
423 }
424
425 XChangeProperty (manager->display, manager->window,
426 manager->xsettings_atom, manager->xsettings_atom,
427 8, PropModeReplace, buffer.data, buffer.len);
428
429 free (buffer.data);
430
431 return XSETTINGS_SUCCESS;
432}

436void
437xsettings_manager_set_overrides (XSettingsManager *manager,
438 GVariant *overrides)
439{
440 GVariantIter iter;
441 const gchar *key;
442 GVariant *value;
443
444 g_return_if_fail (overrides != NULL && g_variant_is_of_type (overrides, G_VARIANT_TYPE_VARDICT));
445
446 if (manager->overrides)
447 {
448 /* unset the existing overrides */
449
450 g_variant_iter_init (&iter, manager->overrides);
451 while (g_variant_iter_next (&iter, "{&sv}", &key, NULL))
452 /* only unset it at this point if it's not in the new list */
453 if (!g_variant_lookup (overrides, key, "*", NULL))
454 xsettings_manager_delete_setting (manager, key);
455 g_variant_unref (manager->overrides);
456 }
457
458 /* save this so we can do the unsets next time */
459 manager->overrides = g_variant_ref_sink (overrides);
460
461 /* set the new values */
462 g_variant_iter_init (&iter, overrides);
463 while (g_variant_iter_loop (&iter, "{&sv}", &key, &value))
464 {
465 XSettingsSetting *setting;
466
467 /* only accept recognised types... */
468 if (!g_variant_is_of_type (value, G_VARIANT_TYPE_STRING) &&
469 !g_variant_is_of_type (value, G_VARIANT_TYPE_INT32) &&
470 !g_variant_is_of_type (value, XSETTINGS_VARIANT_TYPE_COLOR))
471 continue;
472
473 setting = g_new0 (XSettingsSetting, 1);
474 setting->name = g_strdup (key);
475
476 if (g_variant_is_of_type (value, G_VARIANT_TYPE_STRING))
477 {
478 setting->type = XSETTINGS_TYPE_STRING;
479 setting->data.v_string = g_strdup (g_variant_get_string (value, NULL));
480 }
481 else if (g_variant_is_of_type (value, G_VARIANT_TYPE_INT32))
482 {
483 setting->type = XSETTINGS_TYPE_INT;
484 setting->data.v_int = g_variant_get_int32 (value);
485 }
486 else if (g_variant_is_of_type (value, XSETTINGS_VARIANT_TYPE_COLOR))
487 {
488 guint16 red, green, blue, alpha;
489
490 g_variant_get (value, "(qqqq)", &red, &green, &blue, &alpha);
491 setting->type = XSETTINGS_TYPE_COLOR;
492 setting->data.v_color.red = red;
493 setting->data.v_color.green = green;
494 setting->data.v_color.blue = blue;
495 setting->data.v_color.alpha = alpha;
496 }
497
498 xsettings_manager_set_setting (manager, setting);
499 }
