229static GSList *
230panel_addto_prepend_internal_applets (GSList *list)
231{
232 static gboolean translated = FALSE;
233 gsize i;
234 PanelActionButtonType p;
235
236 for (i = 0; i < G_N_ELEMENTS (internal_addto_items); i++) {
237 if (!translated) {
238 internal_addto_items [i].name = _(internal_addto_items [i].name);
239 internal_addto_items [i].description = _(internal_addto_items [i].description);

241

243 }
244
245 translated = TRUE;
246
247 for (p = PANEL_ACTION_LOCK; p < PANEL_ACTION_LAST; p++) {
248 PanelAddtoItemInfo *info;
249
250 if (panel_action_get_is_disabled (p))

255 info->action_type = p;

258 info->icon = g_strdup (panel_action_get_icon_name (p));
259 info->iid = g_strdup (panel_action_get_drag_id (p));
260 info->enabled = TRUE;
261 info->static_data = FALSE;
262
263 list = g_slist_prepend (list, info);
264 }
265
266 return list;

302static void
303panel_addto_drag_begin_cb (GtkWidget *widget,
304 GdkDragContext *context,
305 gpointer data)
306{
307 GtkTreeModel *filter_model;
308 GtkTreeModel *child_model;
309 GtkTreePath *path;
310 GtkTreeIter iter;
311 GtkTreeIter filter_iter;
312 gchar *icon_name;
313
314 filter_model = gtk_tree_view_get_model (GTK_TREE_VIEW (widget));
315

317 gtk_tree_model_get_iter (filter_model, &filter_iter, path);
318 gtk_tree_path_free (path);

320 &iter, &filter_iter);
321

323 gtk_tree_model_get (child_model, &iter,
324 COLUMN_ICON_NAME, &icon_name,
325 -1);
326
327 gtk_drag_set_icon_name (context, icon_name, 0, 0);
328 g_free (icon_name);

392static GSList *
393panel_addto_query_applets (GSList *list)
394{
395 GList *applet_list, *l;
396
397 applet_list = mate_panel_applets_manager_get_applets ();
398
399 for (l = applet_list; l; l = g_list_next (l)) {
400 MatePanelAppletInfo *info;
401 const char *iid, *name, *description, *icon;
402 gboolean enabled;
403 PanelAddtoItemInfo *applet;
404
405 info = (MatePanelAppletInfo *)l->data;
406
407 iid = mate_panel_applet_info_get_iid (info);
408 name = mate_panel_applet_info_get_name (info);
409 description = mate_panel_applet_info_get_description (info);
410 icon = mate_panel_applet_info_get_icon (info);
411
412 if (!name || panel_lockdown_is_applet_disabled (iid)) {
413 continue;
414 }
415
416 enabled = TRUE;
417#ifdef HAVE_X11
418 if (GDK_IS_X11_DISPLAY (gdk_display_get_default ()) &&
419 !mate_panel_applet_info_get_x11_supported (info)) {
420 enabled = FALSE;
421 description = _("Not compatible with X11");
422 }
423#endif
424#ifdef HAVE_WAYLAND
425 if (GDK_IS_WAYLAND_DISPLAY (gdk_display_get_default ()) &&
426 !mate_panel_applet_info_get_wayland_supported (info)) {
427 enabled = FALSE;
428 description = _("Not compatible with Wayland");
429 }
430#endif
431
432 applet = g_new0 (PanelAddtoItemInfo, 1);
433 applet->type = PANEL_ADDTO_APPLET;
434 applet->name = g_strdup (name);
435 applet->description = g_strdup (description);
436 applet->icon = g_strdup (icon);
437 applet->iid = g_strdup (iid);
438 applet->enabled = enabled;
439 applet->static_data = FALSE;
440
441 list = g_slist_prepend (list, applet);
442 }
443
444 g_list_free (applet_list);
445
446 return list;
447}

506static void
507panel_addto_make_applet_model (PanelAddtoDialog *dialog)
508{
509 GtkListStore *model;
510 GSList *l;
511
512 if (dialog->filter_applet_model != NULL)
513 return;
514
515 if (panel_profile_id_lists_are_writable ()) {
516 dialog->applet_list = panel_addto_query_applets (dialog->applet_list);
517 dialog->applet_list = panel_addto_prepend_internal_applets (dialog->applet_list);
518 }
519
520 dialog->applet_list = g_slist_sort (dialog->applet_list,
521 (GCompareFunc) panel_addto_applet_info_sort_func);
522
523 model = gtk_list_store_new (NUMBER_COLUMNS,
524 G_TYPE_STRING,
525 G_TYPE_STRING,
526 G_TYPE_POINTER,
527 G_TYPE_STRING,
528 G_TYPE_BOOLEAN);
529
530 if (panel_profile_id_lists_are_writable ()) {
531 panel_addto_append_special_applets (dialog, model);
532 if (dialog->applet_list)
533 panel_addto_append_item (dialog, model, NULL);
534 }
535
536 for (l = dialog->applet_list; l; l = l->next)
537 panel_addto_append_item (dialog, model, l->data);
538
539 dialog->applet_model = GTK_TREE_MODEL (model);
540 dialog->filter_applet_model = gtk_tree_model_filter_new (GTK_TREE_MODEL (dialog->applet_model),
541 NULL);
542 gtk_tree_model_filter_set_visible_func (GTK_TREE_MODEL_FILTER (dialog->filter_applet_model),
543 panel_addto_filter_func,
544 dialog, NULL);
545}

638static void
639panel_addto_make_application_list (GSList **parent_list,
640 MateMenuTreeDirectory *directory,
641 const char *filename)
642{
643 MateMenuTreeIter *iter;
644 iter = matemenu_tree_directory_iter (directory);
645 MateMenuTreeItemType type;
646 while ((type = matemenu_tree_iter_next (iter)) != MATEMENU_TREE_ITEM_INVALID) {
647 gpointer item;
648 switch (type) {
649 case MATEMENU_TREE_ITEM_DIRECTORY:
650 item = matemenu_tree_iter_get_directory(iter);
651 panel_addto_prepend_directory (parent_list, item, filename);
652 matemenu_tree_item_unref (item);
653 break;
654
655 case MATEMENU_TREE_ITEM_ENTRY:
656 item = matemenu_tree_iter_get_entry (iter);
657 panel_addto_prepend_entry (parent_list, item, filename);
658 matemenu_tree_item_unref (item);
659 break;
660
661 case MATEMENU_TREE_ITEM_ALIAS:
662 item = matemenu_tree_iter_get_alias(iter);
663 panel_addto_prepend_alias (parent_list, item, filename);
664 matemenu_tree_item_unref (item);
665 break;
666 default:
667 break;
668 }
669 }
670 matemenu_tree_iter_unref (iter);
671
672 *parent_list = g_slist_reverse (*parent_list);
673}

706static void panel_addto_make_application_model(PanelAddtoDialog* dialog)
707{
708 GtkTreeStore* store;
709 MateMenuTree* tree;
710 MateMenuTreeDirectory* root;
711 GError *error = NULL;
712
713 if (dialog->filter_application_model != NULL)
714 return;
715
716 store = gtk_tree_store_new (NUMBER_COLUMNS,
717 G_TYPE_STRING,
718 G_TYPE_STRING,
719 G_TYPE_POINTER,
720 G_TYPE_STRING,
721 G_TYPE_BOOLEAN);
722
723 tree = matemenu_tree_new ("mate-applications.menu", MATEMENU_TREE_FLAGS_SORT_DISPLAY_NAME);
724 if (! matemenu_tree_load_sync (tree, &error)) {
725 g_warning("Applications menu tree loading got error:%s\n", error->message);
726 g_error_free(error);
727 g_clear_object(&tree);
728 }
729
730 if ((root = matemenu_tree_get_root_directory (tree)) != NULL )
