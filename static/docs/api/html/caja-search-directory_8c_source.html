110static void
111reset_file_list (CajaSearchDirectory *search)
112{
113 GList *list, *monitor_list;
114 SearchMonitor *monitor;
115 CajaFile *file = NULL;
116
117 /* Remove file connections */
118 for (list = search->details->files; list != NULL; list = list->next)
119 {
120 file = list->data;
121
122 /* Disconnect change handler */
123 g_signal_handlers_disconnect_by_func (file, file_changed, search);
124
125 /* Remove monitors */
126 for (monitor_list = search->details->monitor_list; monitor_list;
127 monitor_list = monitor_list->next)
128 {
129 monitor = monitor_list->data;
130 caja_file_monitor_remove (file, monitor);
131 }
132 }
133
134 caja_file_list_free (search->details->files);
135 search->details->files = NULL;
136}

138static void
139start_or_stop_search_engine (CajaSearchDirectory *search, gboolean adding)
140{
141 if (adding && (search->details->monitor_list ||
142 search->details->pending_callback_list) &&
143 search->details->query &&
144 !search->details->search_running)
145 {
146 /* We need to start the search engine */
147 search->details->search_running = TRUE;
148 search->details->search_finished = FALSE;
149 ensure_search_engine (search);
150 caja_search_engine_set_query (search->details->engine, search->details->query);
151
152 reset_file_list (search);
153
154 caja_search_engine_start (search->details->engine);
155 }
156 else if (!adding && !search->details->monitor_list &&
157 !search->details->pending_callback_list &&
158 search->details->engine &&
159 search->details->search_running)
160 {
161 search->details->search_running = FALSE;
162 caja_search_engine_stop (search->details->engine);
163
164 reset_file_list (search);
165 }

180static void
181search_monitor_add (CajaDirectory *directory,
182 gconstpointer client,
183 gboolean monitor_hidden_files,
184 CajaFileAttributes file_attributes,
185 CajaDirectoryCallback callback,
186 gpointer callback_data)
187{
188 GList *list;
189 SearchMonitor *monitor;
190 CajaSearchDirectory *search;
191 CajaFile *file = NULL;
192
193 search = CAJA_SEARCH_DIRECTORY (directory);
194
195 monitor = g_new0 (SearchMonitor, 1);
196 monitor->monitor_hidden_files = monitor_hidden_files;
197 monitor->monitor_attributes = file_attributes;
198 monitor->client = client;
199
200 search->details->monitor_list = g_list_prepend (search->details->monitor_list, monitor);
201
202 if (callback != NULL)
203 {
204 (* callback) (directory, search->details->files, callback_data);
205 }
206
207 for (list = search->details->files; list != NULL; list = list->next)
208 {
209 file = list->data;
210
211 /* Add monitors */
212 caja_file_monitor_add (file, monitor, file_attributes);
213 }
214
215 start_or_stop_search_engine (search, TRUE);
216}

240static void
241search_monitor_remove (CajaDirectory *directory,
242 gconstpointer client)
243{
244 CajaSearchDirectory *search;
245 GList *list;
246 SearchMonitor *monitor = NULL;
247
248 search = CAJA_SEARCH_DIRECTORY (directory);
249
250 for (list = search->details->monitor_list; list != NULL; list = list->next)
251 {
252 monitor = list->data;
253
254 if (monitor->client == client)
255 {
256 search->details->monitor_list = g_list_delete_link (search->details->monitor_list, list);
257
258 search_monitor_destroy (monitor, search);
259
260 break;
261 }
262 }
263

380static GHashTable *
381file_list_to_hash_table (GList *file_list)
382{
383 GList *list;
384 GHashTable *table;
385
386 if (!file_list)
387 return NULL;
388
389 table = g_hash_table_new (NULL, NULL);
390
391 for (list = file_list; list != NULL; list = list->next)
392 {
393 g_hash_table_insert (table, list->data, list->data);
394 }
395
396 return table;
397}

399static void
400search_call_when_ready (CajaDirectory *directory,
401 CajaFileAttributes file_attributes,
402 gboolean wait_for_file_list,
403 CajaDirectoryCallback callback,
404 gpointer callback_data)
405{
406 CajaSearchDirectory *search;
407 SearchCallback *search_callback;
408
409 search = CAJA_SEARCH_DIRECTORY (directory);
410
411 search_callback = search_callback_find (search, callback, callback_data);
412 if (search_callback == NULL)
413 {
414 search_callback = search_callback_find_pending (search, callback, callback_data);
415 }
416
417 if (search_callback)
418 {
419 g_warning ("tried to add a new callback while an old one was pending");
420 return;
421 }
422
423 search_callback = g_new0 (SearchCallback, 1);
424 search_callback->search_directory = search;
425 search_callback->callback = callback;
426 search_callback->callback_data = callback_data;
427 search_callback->wait_for_attributes = file_attributes;
428 search_callback->wait_for_file_list = wait_for_file_list;
429
430 if (wait_for_file_list && !search->details->search_finished)
431 {
432 /* Add it to the pending callback list, which will be
433 * processed when the directory has finished loading
434 */
435 search->details->pending_callback_list =
436 g_list_prepend (search->details->pending_callback_list, search_callback);
437
438 /* We might need to start the search engine */
439 start_or_stop_search_engine (search, TRUE);
440 }
441 else
442 {
443 search_callback->file_list = caja_file_list_copy (search->details->files);
444 search_callback->non_ready_hash = file_list_to_hash_table (search->details->files);
445
446 if (!search_callback->non_ready_hash)
447 {
448 /* If there are no ready files, we invoke the callback
449 with an empty list.
450 */
451 search_callback_invoke_and_destroy (search_callback);
452 }
453 else
454 {
455 search->details->callback_list = g_list_prepend (search->details->callback_list, search_callback);
456 search_callback_add_file_callbacks (search_callback);
457 }
458 }
459}

461static void
462search_cancel_callback (CajaDirectory *directory,
463 CajaDirectoryCallback callback,
464 gpointer callback_data)
465{
466 CajaSearchDirectory *search;
467 SearchCallback *search_callback;
468
469 search = CAJA_SEARCH_DIRECTORY (directory);
470 search_callback = search_callback_find (search, callback, callback_data);
471
472 if (search_callback)
473 {
474 search->details->callback_list = g_list_remove (search->details->callback_list, search_callback);
475
476 search_callback_destroy (search_callback);
477
478 return;
479 }
480
481 /* Check for a pending callback */
482 search_callback = search_callback_find_pending (search, callback, callback_data);
483
484 if (search_callback)
485 {
486 search->details->pending_callback_list = g_list_remove (search->details->pending_callback_list, search_callback);
487
488 search_callback_destroy (search_callback);
489
490 /* We might need to stop the search engine now */
491 start_or_stop_search_engine (search, FALSE);
492 }

495static void
496search_engine_hits_added (CajaSearchEngine *engine, GList *hits,
497 CajaSearchDirectory *search)
498{
499 GList *hit_list;
500 GList *file_list;
501 CajaFile *file;
502 SearchMonitor *monitor;
503 GList *monitor_list;
504
505 file_list = NULL;
506
507 for (hit_list = hits; hit_list != NULL; hit_list = hit_list->next)
508 {
509 char *uri;
510
511 uri = hit_list->data;
512
513 if (g_str_has_suffix (uri, CAJA_SAVED_SEARCH_EXTENSION))
514 {
515 /* Never return saved searches themselves as hits */
516 continue;
517 }
518
519 file = caja_file_get_by_uri (uri);
520
521 for (monitor_list = search->details->monitor_list; monitor_list; monitor_list = monitor_list->next)
522 {
523 monitor = monitor_list->data;
524
525 /* Add monitors */
526 caja_file_monitor_add (file, monitor, monitor->monitor_attributes);
527 }
528
529 g_signal_connect (file, "changed", G_CALLBACK (file_changed), search);
530
531 file_list = g_list_prepend (file_list, file);
532 }
533
534 search->details->files = g_list_concat (search->details->files, file_list);
535
536 caja_directory_emit_files_added (CAJA_DIRECTORY (search), file_list);
537
538 file = caja_directory_get_corresponding_file (CAJA_DIRECTORY (search));
539 caja_file_emit_changed (file);
540 caja_file_unref (file);
541}

543static void
544search_engine_hits_subtracted (CajaSearchEngine *engine, GList *hits,
545 CajaSearchDirectory *search)
546{
547 GList *hit_list;
548 GList *monitor_list;
549 SearchMonitor *monitor;
550 GList *file_list;
551 CajaFile *file;
552
553 file_list = NULL;
554
555 for (hit_list = hits; hit_list != NULL; hit_list = hit_list->next)
556 {
557 char *uri;
558
559 uri = hit_list->data;
560 file = caja_file_get_by_uri (uri);
561
562 for (monitor_list = search->details->monitor_list; monitor_list;
563 monitor_list = monitor_list->next)
564 {
565 monitor = monitor_list->data;
566 /* Remove monitors */
567 caja_file_monitor_remove (file, monitor);
568 }
569
570 g_signal_handlers_disconnect_by_func (file, file_changed, search);
571
572 search->details->files = g_list_remove (search->details->files, file);
573
574 file_list = g_list_prepend (file_list, file);
575 }
576
577 caja_directory_emit_files_changed (CAJA_DIRECTORY (search), file_list);
578
579 caja_file_list_free (file_list);
580
581 file = caja_directory_get_corresponding_file (CAJA_DIRECTORY (search));
582 caja_file_emit_changed (file);
583 caja_file_unref (file);
584}

624static void
625search_force_reload (CajaDirectory *directory)
626{
627 CajaSearchDirectory *search;
628
629 search = CAJA_SEARCH_DIRECTORY (directory);
630
631 if (!search->details->query)
632 {
633 return;
634 }
635
636 search->details->search_finished = FALSE;
637
638 if (!search->details->engine)
639 {
640 return;
641 }
642
643 /* Remove file monitors */
644 reset_file_list (search);
645
646 if (search->details->search_running)
647 {
648 caja_search_engine_stop (search->details->engine);
649 caja_search_engine_set_query (search->details->engine, search->details->query);
650 caja_search_engine_start (search->details->engine);
651 }

693static void
694search_dispose (GObject *object)
695{
696 CajaSearchDirectory *search;
697 GList *list;
698
699 search = CAJA_SEARCH_DIRECTORY (object);
700
701 /* Remove search monitors */
702 if (search->details->monitor_list)
703 {
704 for (list = search->details->monitor_list; list != NULL; list = list->next)
705 {
706 search_monitor_destroy ((SearchMonitor *)list->data, search);
707 }
708
709 g_list_free (search->details->monitor_list);
710 search->details->monitor_list = NULL;
711 }
712
713 reset_file_list (search);
714
715 if (search->details->callback_list)
716 {
717 /* Remove callbacks */
718 g_list_foreach (search->details->callback_list,
719 (GFunc)search_callback_destroy, NULL);
720 g_list_free (search->details->callback_list);
721 search->details->callback_list = NULL;
722 }
723
724 if (search->details->pending_callback_list)
725 {
726 g_list_foreach (search->details->pending_callback_list,
727 (GFunc)search_callback_destroy, NULL);
728 g_list_free (search->details->pending_callback_list);
729 search->details->pending_callback_list = NULL;
730 }
731
732 if (search->details->query)
733 {
734 g_object_unref (search->details->query);
735 search->details->query = NULL;
736 }
737
738 if (search->details->engine)
739 {
740 if (search->details->search_running)
741 {
742 caja_search_engine_stop (search->details->engine);
743 }
744
745 g_object_unref (search->details->engine);
746 search->details->engine = NULL;
747 }
748
749 G_OBJECT_CLASS (caja_search_directory_parent_class)->dispose (object);
750}

806void
807caja_search_directory_set_query (CajaSearchDirectory *search,
808 CajaQuery *query)
809{
810 CajaDirectory *dir;
811 CajaFile *as_file;
812
813 if (search->details->query != query)
814 {
815 search->details->modified = TRUE;
816 }
817
818 if (query)
819 {
820 g_object_ref (query);
821 }
822
823 if (search->details->query)
824 {
825 g_object_unref (search->details->query);
826 }
827
