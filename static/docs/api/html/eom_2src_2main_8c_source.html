109#ifdef ENABLE_NLS

115 gdk_set_allowed_backends ("wayland,x11");

117 ctx = g_option_context_new (_("[FILEâ€¦]"));
118 g_option_context_add_main_entries (ctx, goption_options, PACKAGE);
119 /* Option groups are free'd together with the context
120 * Using gtk_get_option_group here initializes gtk during parsing */
121 g_option_context_add_group (ctx, gtk_get_option_group (TRUE));
122#ifdef HAVE_INTROSPECTION
123#ifdef HAVE_GIREPOSITORY_2
124 g_option_context_add_group (ctx, gi_repository_get_option_group ());
125#else
126 g_option_context_add_group (ctx, g_irepository_get_option_group ());
127#endif
128#endif
129
130 if (!g_option_context_parse (ctx, &argc, &argv, &error)) {
131 gchar *help_msg;
132
133 /* I18N: The '%s' is replaced with eom's command name. */
134 help_msg = g_strdup_printf (_("Run '%s --help' to see a full "
135 "list of available command line "
136 "options."), argv[0]);
137 g_printerr ("%s\n%s\n", error->message, help_msg);
138 g_error_free (error);
139 g_free (help_msg);
140 g_option_context_free (ctx);
141
142 return 1;
143 }

155 /* Load special style properties for EomThumbView's scrollbar */

172
173 /* Add application specific icons to search path */
174 gtk_icon_theme_append_search_path (gtk_icon_theme_get_default (),
175 EOM_DATA_DIR G_DIR_SEPARATOR_S "icons");

177 gtk_window_set_default_icon_name ("eom");
178 g_set_application_name (_("Eye of MATE Image Viewer"));
179
180 EOM_APP->priv->flags = flags;
181 if (force_new_instance) {
182 GApplicationFlags app_flags = g_application_get_flags (G_APPLICATION (EOM_APP));
183 app_flags |= G_APPLICATION_NON_UNIQUE;
184 g_application_set_flags (G_APPLICATION (EOM_APP), app_flags);
185 }
186
187 g_application_run (G_APPLICATION (EOM_APP), argc, argv);
188 g_object_unref (EOM_APP);
189
190 if (startup_files)
191 g_strfreev (startup_files);
192
193#ifdef HAVE_EXEMPI
194 xmp_terminate();
195#endif
