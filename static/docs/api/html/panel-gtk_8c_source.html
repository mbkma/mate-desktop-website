43static void
44panel_gtk_file_chooser_preview_update (GtkFileChooser *chooser,
45 gpointer data)
46{
47 GtkWidget *preview;
48 char *filename;
49 GdkPixbuf *pixbuf;
50 gboolean have_preview;
51
52 preview = GTK_WIDGET (data);
53 filename = gtk_file_chooser_get_preview_filename (chooser);
54
55 if (filename == NULL)
56 return;
57
58 pixbuf = gdk_pixbuf_new_from_file_at_size (filename, 128, 128, NULL);
59 have_preview = (pixbuf != NULL);
60 g_free (filename);
61
62 gtk_image_set_from_pixbuf (GTK_IMAGE (preview), pixbuf);
63 if (pixbuf)
64 g_object_unref (pixbuf);
65
66 gtk_file_chooser_set_preview_widget_active (chooser,
67 have_preview);
68}

70void
71panel_gtk_file_chooser_add_image_preview (GtkFileChooser *chooser)
72{
73 GtkFileFilter *filter;
74 GtkWidget *chooser_preview;
75
76 g_return_if_fail (GTK_IS_FILE_CHOOSER (chooser));
77
78 filter = gtk_file_filter_new ();
79 gtk_file_filter_add_pixbuf_formats (filter);
80 gtk_file_chooser_set_filter (chooser, filter);
81
82 chooser_preview = gtk_image_new ();
83 gtk_file_chooser_set_preview_widget (chooser, chooser_preview);
84 g_signal_connect (chooser, "update-preview",
85 G_CALLBACK (panel_gtk_file_chooser_preview_update),
86 chooser_preview);
87}

93GtkWidget*
94panel_dialog_add_button (GtkDialog *dialog,
95 const gchar *button_text,
96 const gchar *icon_name,
97 gint response_id)
98{
99 GtkWidget *button;
100
101 button = gtk_button_new_with_mnemonic (button_text);
102 gtk_button_set_image (GTK_BUTTON (button), gtk_image_new_from_icon_name (icon_name, GTK_ICON_SIZE_BUTTON));
103
104 gtk_button_set_use_underline (GTK_BUTTON (button), TRUE);
105 gtk_style_context_add_class (gtk_widget_get_style_context (button), "text-button");
106 gtk_widget_set_can_default (button, TRUE);
107 gtk_widget_show (button);
108 gtk_dialog_add_action_widget (GTK_DIALOG (dialog), button, response_id);
109
110 return button;
111}

113static GtkWidget *
114panel_file_chooser_dialog_new_valist (const gchar *title,
115 GtkWindow *parent,
116 GtkFileChooserAction action,
117 const gchar *first_button_text,
118 va_list varargs)
119{
120 GtkWidget *result;
121 const char *button_text = first_button_text;
122
123 result = g_object_new (GTK_TYPE_FILE_CHOOSER_DIALOG,
124 "title", title,
125 "action", action,
126 NULL);
127
128 if (parent)
129 gtk_window_set_transient_for (GTK_WINDOW (result), parent);
130
131 while (button_text)
132 {
133 gint response_id = va_arg (varargs, gint);
134
135 if (g_strcmp0 (button_text, "process-stop") == 0)
136 panel_dialog_add_button (GTK_DIALOG (result), _("_Cancel"), button_text, response_id);
137 else if (g_strcmp0 (button_text, "document-open") == 0)
138 panel_dialog_add_button (GTK_DIALOG (result), _("_Open"), button_text, response_id);
139 else if (g_strcmp0 (button_text, "gtk-ok") == 0)
140 panel_dialog_add_button (GTK_DIALOG (result), _("_OK"), button_text, response_id);
141 else
142 gtk_dialog_add_button (GTK_DIALOG (result), button_text, response_id);
143
144 button_text = va_arg (varargs, const gchar *);
145 }
146
147 return result;
148}

181GtkWidget *
182panel_image_menu_item_new_from_icon (const gchar *icon_name,
183 const gchar *label_name)
184{
185 gchar *concat;
186 GtkWidget *icon;
187 GtkStyleContext *context;
188 GtkWidget *box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
189 GtkWidget *icon_box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);
190
191 if (icon_name)
192 icon = gtk_image_new_from_icon_name (icon_name, GTK_ICON_SIZE_MENU);
193 else
194 icon = gtk_image_new ();
195
196 concat = g_strconcat (label_name, " ", NULL);
197 GtkWidget *label_menu = gtk_label_new_with_mnemonic (concat);
198 GtkWidget *menuitem = gtk_menu_item_new ();
199
200 context = gtk_widget_get_style_context (GTK_WIDGET(icon_box));
201 gtk_style_context_add_class(context,"mate-panel-menu-icon-box");
202
203 gtk_container_add (GTK_CONTAINER (icon_box), icon);
204 gtk_container_add (GTK_CONTAINER (box), icon_box);
205 gtk_container_add (GTK_CONTAINER (box), label_menu);
206
207 gtk_container_add (GTK_CONTAINER (menuitem), box);
208 gtk_widget_show_all (menuitem);
209
210 ensure_icon_settings();
211 g_settings_bind (icon_settings, "menus-have-icons", icon, "visible",
212 G_SETTINGS_BIND_GET);
213
214 g_free (concat);
215
216 return menuitem;
217}
