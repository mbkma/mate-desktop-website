29static void
30bom_finish (SoupSession *session, SoupMessage *msg, gpointer data)
31{
32 char *p, *rp;
33 WeatherInfo *info = (WeatherInfo *)data;
34
35 g_return_if_fail (info != NULL);
36
37 if (!SOUP_STATUS_IS_SUCCESSFUL (msg->status_code)) {
38 g_warning ("Failed to get BOM forecast data: %d %s.\n",
39 msg->status_code, msg->reason_phrase);
40 request_done (info, FALSE);
41 return;
42 }
43
44 p = strstr (msg->response_body->data, "Forecast for the rest");
45 if (p != NULL) {
46 rp = strstr (p, "The next routine forecast will be issued");
47 if (rp == NULL)
48 info->forecast = g_strdup (p);
49 else
50 info->forecast = g_strndup (p, rp - p);
51 }
52
53 if (info->forecast == NULL)
54 info->forecast = g_strdup (msg->response_body->data);
55
56 g_print ("%s\n", info->forecast);
57 request_done (info, TRUE);
58}

61bom_start_open (WeatherInfo *info)

63 gchar *url;
