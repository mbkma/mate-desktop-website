59static void
60mate_color_selection_dialog_get_property (GObject *object,
61 guint prop_id,
62 GValue *value,
63 GParamSpec *pspec)
64{
65 MateColorSelectionDialog *colorsel;
66
67 colorsel = MATE_COLOR_SELECTION_DIALOG (object);
68
69 switch (prop_id)
70 {
71 case PROP_COLOR_SELECTION:
72 g_value_set_object (value, colorsel->colorsel);
73 break;
74 case PROP_OK_BUTTON:
75 g_value_set_object (value, colorsel->ok_button);
76 break;
77 case PROP_CANCEL_BUTTON:
78 g_value_set_object (value, colorsel->cancel_button);
79 break;
80 case PROP_HELP_BUTTON:
81 g_value_set_object (value, colorsel->help_button);
82 break;
83 default:
84 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
85 break;
86 }
87}

130{

142static void
143mate_color_selection_dialog_init (MateColorSelectionDialog *colorseldiag)
144{
145 GtkDialog *dialog = GTK_DIALOG (colorseldiag);
146
147 _mate_desktop_init_i18n ();
148
149 gtk_container_set_border_width (GTK_CONTAINER (dialog), 5);
150 gtk_box_set_spacing (GTK_BOX (gtk_dialog_get_content_area (dialog)), 6);
151
152 colorseldiag->colorsel = mate_color_selection_new ();
153 gtk_container_set_border_width (GTK_CONTAINER (colorseldiag->colorsel), 5);
154 mate_color_selection_set_has_palette (MATE_COLOR_SELECTION(colorseldiag->colorsel), FALSE);
155 mate_color_selection_set_has_opacity_control (MATE_COLOR_SELECTION(colorseldiag->colorsel), FALSE);
156 gtk_container_add (GTK_CONTAINER (gtk_dialog_get_content_area (GTK_DIALOG (colorseldiag))), colorseldiag->colorsel);
157 gtk_widget_show (colorseldiag->colorsel);
158
159 colorseldiag->cancel_button = mate_add_dialog_button (GTK_DIALOG (colorseldiag),
160 _("_Cancel"),
161 "gtk-cancel",
162 GTK_RESPONSE_CANCEL);
163
164 colorseldiag->ok_button = mate_add_dialog_button (GTK_DIALOG (colorseldiag),
165 _("_OK"),
166 "gtk-ok",
167 GTK_RESPONSE_OK);
168
169 gtk_widget_set_can_default (colorseldiag->ok_button, TRUE);
170 gtk_widget_grab_default (colorseldiag->ok_button);
171
172 colorseldiag->help_button = mate_add_dialog_button (GTK_DIALOG (colorseldiag),
173 _("Help"),
174 "help-browser",
175 GTK_RESPONSE_HELP);
176
177 gtk_widget_hide (colorseldiag->help_button);
178 gtk_widget_show (colorseldiag->cancel_button);
179 gtk_widget_show (colorseldiag->ok_button);
180
181 gtk_window_set_title (GTK_WINDOW (colorseldiag),
182 _("Color Selection"));
183
184 //_gtk_dialog_set_ignore_separator (dialog, TRUE);
185}

234 else if (strcmp(childname, "cancel_button") == 0)

237 return G_OBJECT (MATE_COLOR_SELECTION_DIALOG(buildable)->help_button);

239 return G_OBJECT (MATE_COLOR_SELECTION_DIALOG(buildable)->colorsel);
240
241 return parent_buildable_iface->get_internal_child (buildable, builder, childname);
