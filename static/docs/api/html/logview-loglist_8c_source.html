71static void
72update_days_and_lines_for_log (LogviewLoglist *loglist,
73 GtkTreeIter *log, GSList *days)
74{
75 gboolean res;
76 GtkTreeIter iter, dummy;
77 GSList *l;
78 int i;
79 char date[200];
80 Day *day;
81
82 /* if we have some days, we can't remove all the items immediately, otherwise,
83 * if the row is expanded, it will be collapsed because there are no items,
84 * so we create a dummy entry, remove all the others and then remove the
85 * dummy one.
86 */
87 res = gtk_tree_model_iter_children (GTK_TREE_MODEL (loglist->priv->model),
88 &iter, log);
89 if (res) {
90 gtk_tree_store_insert_before (loglist->priv->model, &dummy, log, &iter);
91 gtk_tree_store_set (loglist->priv->model, &dummy,
92 LOG_NAME, "", -1);
93 do {
94 gtk_tree_store_remove (loglist->priv->model, &iter);
95 } while (gtk_tree_store_iter_is_valid (loglist->priv->model, &iter));
96 }
97
98 for (i = 1, l = days; l; l = l->next) {
99 /* now insert all the days */
100 day = l->data;
101
102 /* TRANSLATORS: "strftime format options, see the man page for strftime(3) for further information." */
103 g_date_strftime (date, 200, _("%A, %e %b"), day->date);
104
105 gtk_tree_store_insert (GTK_TREE_STORE (loglist->priv->model),
106 &iter, log, i);
107 gtk_tree_store_set (GTK_TREE_STORE (loglist->priv->model),
108 &iter, LOG_NAME, date, LOG_DAY, day, -1);
109 i++;
110 }
111
112 if (res) {
113 gtk_tree_store_remove (loglist->priv->model, &dummy);
114 }
115}

117static GtkTreeIter *
118logview_loglist_find_log (LogviewLoglist *list, LogviewLog *log)
119{
120 GtkTreeIter iter;
121 GtkTreeModel *model;
122 GtkTreeIter *retval = NULL;
123 LogviewLog *current;
124
125 model = GTK_TREE_MODEL (list->priv->model);
126
127 if (!gtk_tree_model_get_iter_first (model, &iter)) {
128 return NULL;
129 }
130
131 do {
132 gtk_tree_model_get (model, &iter, LOG_OBJECT, &current, -1);
133 if (current == log) {
134 retval = gtk_tree_iter_copy (&iter);
135 }
136 if (current)
137 g_object_unref (current);
138 } while (gtk_tree_model_iter_next (model, &iter) != FALSE && retval == NULL);
139
140 return retval;
141}

143static void
144log_changed_cb (LogviewLog *log,
145 gpointer user_data)
146{
147 LogviewLoglist *list = user_data;
148 LogviewLog *active;
149 GtkTreeIter *iter;
150

152
153 if (log == active) {
154 g_object_unref (active);
155 return;
156 }
157
158 iter = logview_loglist_find_log (list, log);
159
160 if (!iter) {
161 return;
162 }
163
164 /* make the log bold in the list */
165 gtk_tree_store_set (list->priv->model, iter,
166 LOG_WEIGHT, PANGO_WEIGHT_BOLD,
167 LOG_WEIGHT_SET, TRUE, -1);
168
169 gtk_tree_iter_free (iter);

172static void
173tree_selection_changed_cb (GtkTreeSelection *selection,
174 gpointer user_data)
175{
176 LogviewLoglist *list = user_data;
177 GtkTreeModel *model;
178 GtkTreeIter iter, parent;
179 LogviewLog *log;
180 gboolean is_bold, is_active;
181 Day *day;
182
183 if (!gtk_tree_selection_get_selected (selection, &model, &iter)) {
184 return;
185 }
186
187 gtk_tree_model_get (model, &iter, LOG_OBJECT, &log,
188 LOG_WEIGHT_SET, &is_bold,
189 LOG_DAY, &day, -1);
190 if (log) {
191 is_active = logview_manager_log_is_active (list->priv->manager, log);
192
193 if (is_active && list->priv->has_day_selection) {
194 list->priv->has_day_selection = FALSE;
195 g_signal_emit (list, signals[DAY_CLEARED], 0, NULL);
196 } else if (!is_active) {
197 logview_manager_set_active_log (list->priv->manager, log);
198 }
199 } else if (day) {
200 list->priv->has_day_selection = TRUE;
201 gtk_tree_model_iter_parent (model, &parent, &iter);
202 gtk_tree_model_get (model, &parent, LOG_OBJECT, &log, -1);
203
204 if (!logview_manager_log_is_active (list->priv->manager, log)) {
205 save_day_selection (list, &iter);
206 logview_manager_set_active_log (list->priv->manager, log);
207 } else {
208 g_signal_emit (list, signals[DAY_SELECTED], 0, day, NULL);
209 }
210 }
211
212 if (is_bold) {
213 gtk_tree_store_set (GTK_TREE_STORE (model), &iter,
214 LOG_WEIGHT_SET, FALSE, -1);
215 }
216
217 if (log) {
218 g_object_unref (log);
219 }
220}

222static void
223manager_active_changed_cb (LogviewManager *manager,
224 LogviewLog *log,
225 LogviewLog *old_log,
226 gpointer user_data)
227{
228 LogviewLoglist *list = user_data;
229 GtkTreeIter * iter, sel_iter;
230 GtkTreeSelection * selection;
231
232 if (list->priv->selection &&
233 gtk_tree_model_get_iter (GTK_TREE_MODEL (list->priv->model),
234 &sel_iter, list->priv->selection))
235 {
236 Day *day;
237
238 iter = gtk_tree_iter_copy (&sel_iter);
239
240 gtk_tree_model_get (GTK_TREE_MODEL (list->priv->model), iter,
241 LOG_DAY, &day, -1);
242
243 if (day) {
244 g_signal_emit (list, signals[DAY_SELECTED], 0, day, NULL);
245 }
246
247 gtk_tree_path_free (list->priv->selection);
248 list->priv->selection = NULL;
249 } else {
250 iter = logview_loglist_find_log (list, log);
251 }
252
253 if (!iter) {
254 return;
255 }
256
257 selection = gtk_tree_view_get_selection (GTK_TREE_VIEW (list));
258 g_signal_handlers_block_by_func (selection, tree_selection_changed_cb, list);
259
260 gtk_tree_selection_select_iter (selection, iter);
261
262 g_signal_handlers_unblock_by_func (selection, tree_selection_changed_cb, list);
263 gtk_tree_iter_free (iter);
264}

267manager_log_closed_cb (LogviewManager *manager,
268 LogviewLog *log,
269 gpointer user_data)
270{
271 LogviewLoglist *list = user_data;
272 GtkTreeIter *iter;
273 gboolean res;
274
275 iter = logview_loglist_find_log (list, log);
276
277 if (!iter) {
278 return;
279 }
280
281 g_signal_handlers_disconnect_by_func (log, log_changed_cb, list);
282
283 res = gtk_tree_store_remove (list->priv->model, iter);
284 if (res) {
285 GtkTreeSelection *selection;
286
287 /* iter now points to the next valid row */
288 selection = gtk_tree_view_get_selection (GTK_TREE_VIEW (list));
289 gtk_tree_selection_select_iter (selection, iter);
290 } else {
291 /* FIXME: what shall we do here? */
292 }
293
294 gtk_tree_iter_free (iter);

338static int
339loglist_sort_func (GtkTreeModel *model,
340 GtkTreeIter *a,
341 GtkTreeIter *b,
342 gpointer user_data)
343{
344 char *name_a, *name_b;
345 Day *day_a, *day_b;
346 int retval = 0;
347
348 switch (gtk_tree_store_iter_depth (GTK_TREE_STORE (model), a)) {
349 case 0:
350 gtk_tree_model_get (model, a, LOG_NAME, &name_a, -1);
351 gtk_tree_model_get (model, b, LOG_NAME, &name_b, -1);
352 retval = g_utf8_collate (name_a, name_b);
353 g_free (name_a);
354 g_free (name_b);
355
356 break;
357 case 1:
358 gtk_tree_model_get (model, a, LOG_DAY, &day_a, -1);
359 gtk_tree_model_get (model, b, LOG_DAY, &day_b, -1);
360 if (day_a && day_b) {
361 retval = days_compare (day_a, day_b);
362 } else {
363 retval = 0;
364 }
365
366 break;
367 default:
368 g_assert_not_reached ();
369
370 break;
371 }
372
373 return retval;
374}

392static void
393logview_loglist_init (LogviewLoglist *list)
394{
395 GtkTreeStore *model;
396 GtkTreeViewColumn *column;
397 GtkTreeSelection *selection;
398 GtkCellRenderer *cell;
399
400 list->priv = logview_loglist_get_instance_private (list);
401 list->priv->has_day_selection = FALSE;
402 list->priv->selection = NULL;
403
404 model = gtk_tree_store_new (5, LOGVIEW_TYPE_LOG, G_TYPE_STRING, G_TYPE_INT,
405 G_TYPE_BOOLEAN, G_TYPE_POINTER);
406 gtk_tree_view_set_model (GTK_TREE_VIEW (list), GTK_TREE_MODEL (model));
407 list->priv->model = model;
408 gtk_tree_view_set_headers_visible (GTK_TREE_VIEW (list), FALSE);
409
410 selection = gtk_tree_view_get_selection (GTK_TREE_VIEW (list));
411 gtk_tree_selection_set_mode (selection, GTK_SELECTION_BROWSE);
412 g_signal_connect (selection, "changed",
413 G_CALLBACK (tree_selection_changed_cb), list);
414
415 cell = gtk_cell_renderer_text_new ();
416 column = gtk_tree_view_column_new ();
417 gtk_tree_view_column_pack_start (column, cell, TRUE);
418 gtk_tree_view_column_set_attributes (column, cell,
419 "text", LOG_NAME,
420 "weight-set", LOG_WEIGHT_SET,
421 "weight", LOG_WEIGHT,
422 NULL);
423
424 gtk_tree_sortable_set_sort_column_id (GTK_TREE_SORTABLE (list->priv->model), LOG_NAME, GTK_SORT_ASCENDING);
425 gtk_tree_sortable_set_sort_func (GTK_TREE_SORTABLE (list->priv->model),
426 LOG_NAME,
427 (GtkTreeIterCompareFunc) loglist_sort_func,
428 list, NULL);
429 gtk_tree_view_append_column (GTK_TREE_VIEW (list), column);
430 gtk_tree_view_set_search_column (GTK_TREE_VIEW (list), -1);
431
432 list->priv->manager = logview_manager_get ();
433
434 g_signal_connect (list->priv->manager, "log-added",
435 G_CALLBACK (manager_log_added_cb), list);
436 g_signal_connect (list->priv->manager, "log-closed",
437 G_CALLBACK (manager_log_closed_cb), list);
438 g_signal_connect_after (list->priv->manager, "active-changed",
439 G_CALLBACK (manager_active_changed_cb), list);
440 g_signal_connect (list, "row-expanded",
441 G_CALLBACK (row_expanded_cb), list);
442}
