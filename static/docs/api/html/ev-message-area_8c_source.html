86static void
87ev_message_area_init (EvMessageArea *area)
88{
89 GtkWidget *hbox, *vbox;
90 GtkWidget *content_area;
91
92 area->priv = ev_message_area_get_instance_private (area);
93
94 area->priv->main_box = gtk_box_new (GTK_ORIENTATION_VERTICAL, 12);
95
96 hbox = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 12);
97 vbox = gtk_box_new (GTK_ORIENTATION_VERTICAL, 12);
98
99 area->priv->label = gtk_label_new (NULL);
100 gtk_label_set_use_markup (GTK_LABEL (area->priv->label), TRUE);
101 gtk_label_set_line_wrap (GTK_LABEL (area->priv->label), TRUE);
102 gtk_label_set_selectable (GTK_LABEL (area->priv->label), TRUE);
103 gtk_label_set_xalign (GTK_LABEL (area->priv->label), 0.0);
104 gtk_widget_set_can_focus (area->priv->label, TRUE);
105 gtk_box_pack_start (GTK_BOX (vbox), area->priv->label, TRUE, TRUE, 0);
106 gtk_widget_show (area->priv->label);
107
108 area->priv->secondary_label = gtk_label_new (NULL);
109 gtk_label_set_use_markup (GTK_LABEL (area->priv->secondary_label), TRUE);
110 gtk_label_set_line_wrap (GTK_LABEL (area->priv->secondary_label), TRUE);
111 gtk_label_set_selectable (GTK_LABEL (area->priv->secondary_label), TRUE);
112 gtk_label_set_xalign (GTK_LABEL (area->priv->secondary_label), 0.0);
113 gtk_widget_set_can_focus (area->priv->secondary_label, TRUE);
114 gtk_box_pack_start (GTK_BOX (vbox), area->priv->secondary_label, TRUE, TRUE, 0);
115
116 area->priv->image = gtk_image_new_from_icon_name (NULL, GTK_ICON_SIZE_DIALOG);
117 gtk_widget_set_halign (area->priv->image, GTK_ALIGN_CENTER);
118 gtk_widget_set_valign (area->priv->image, GTK_ALIGN_START);
119 gtk_box_pack_start (GTK_BOX (hbox), area->priv->image, FALSE, FALSE, 0);
120 gtk_widget_show (area->priv->image);
121
122 gtk_box_pack_start (GTK_BOX (hbox), vbox, TRUE, TRUE, 0);
123 gtk_widget_show (vbox);
124
125 gtk_box_pack_start (GTK_BOX (area->priv->main_box), hbox, TRUE, TRUE, 0);
126 gtk_widget_show (hbox);
127
128 content_area = gtk_info_bar_get_content_area (GTK_INFO_BAR (area));
129 gtk_container_add (GTK_CONTAINER (content_area), area->priv->main_box);
130 gtk_widget_show (area->priv->main_box);
131}

133static void
134ev_message_area_set_image_for_type (EvMessageArea *area,
135 GtkMessageType type)
136{
137 const gchar *icon_name = NULL;
138 AtkObject *atk_obj;
139
140 switch (type) {
141 case GTK_MESSAGE_INFO:
142 icon_name = "dialog-information";
143 break;
144 case GTK_MESSAGE_QUESTION:
145 icon_name = "dialog-question";
146 break;
147 case GTK_MESSAGE_WARNING:
148 icon_name = "dialog-warning";
149 break;
150 case GTK_MESSAGE_ERROR:
151 icon_name = "dialog-error";
152 break;
153 case GTK_MESSAGE_OTHER:
154 break;
155 default:
156 g_warning ("Unknown GtkMessageType %u", type);
157 break;
158 }
159
160 if (icon_name)
161 gtk_image_set_from_icon_name (GTK_IMAGE (area->priv->image), icon_name,
162 GTK_ICON_SIZE_DIALOG);
163
164 atk_obj = gtk_widget_get_accessible (GTK_WIDGET (area));
165 if (GTK_IS_ACCESSIBLE (atk_obj)) {
166 atk_object_set_role (atk_obj, ATK_ROLE_ALERT);
167 if (icon_name) {
168 atk_object_set_name (atk_obj, icon_name);
169 }
170 }
171}

173static void
174ev_message_area_set_property (GObject *object,
175 guint prop_id,
176 const GValue *value,
177 GParamSpec *pspec)
178{
179 EvMessageArea *area = EV_MESSAGE_AREA (object);
180
181 switch (prop_id) {
182 case PROP_TEXT:
183 ev_message_area_set_text (area, g_value_get_string (value));
184 break;
185 case PROP_SECONDARY_TEXT:
186 ev_message_area_set_secondary_text (area, g_value_get_string (value));
187 break;
188 case PROP_IMAGE:
189 ev_message_area_set_image (area, (GtkWidget *)g_value_get_object (value));
190 break;
191 default:
192 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
193 break;
194 }
195}

197static void
198ev_message_area_get_property (GObject *object,
199 guint prop_id,
200 GValue *value,
201 GParamSpec *pspec)
202{
203 EvMessageArea *area = EV_MESSAGE_AREA (object);
204
205 switch (prop_id) {
206 case PROP_TEXT:
207 g_value_set_string (value, gtk_label_get_label (GTK_LABEL (area->priv->label)));
208 break;
209 case PROP_SECONDARY_TEXT:
210 g_value_set_string (value, gtk_label_get_label (GTK_LABEL (area->priv->secondary_label)));
211 break;
212 case PROP_IMAGE:
213 g_value_set_object (value, area->priv->image);
214 break;
215 default:
216 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
217 break;
218 }

221void
222_ev_message_area_add_buttons_valist (EvMessageArea *area,
223 const gchar *first_button_text,
224 va_list args)
225{
226 const gchar* text;
227 gint response_id;
228
229 if (first_button_text == NULL)
230 return;
231
232 text = first_button_text;
233 response_id = va_arg (args, gint);
234
235 while (text != NULL) {
236 gtk_info_bar_add_button (GTK_INFO_BAR (area), text, response_id);
237
238 text = va_arg (args, gchar*);
239 if (text == NULL)
240 break;
241
242 response_id = va_arg (args, int);
243 }
244}

252GtkWidget *
253ev_message_area_new (GtkMessageType type,
254 const gchar *text,
255 const gchar *first_button_text,
256 ...)
257{
258 GtkWidget *widget;
259
260 widget = g_object_new (EV_TYPE_MESSAGE_AREA,
261 "message-type", type,
262 "text", text,
263 NULL);
264 ev_message_area_set_image_for_type (EV_MESSAGE_AREA (widget), type);
265 if (first_button_text) {
266 va_list args;
267
268 va_start (args, first_button_text);
269 _ev_message_area_add_buttons_valist (EV_MESSAGE_AREA (widget),
270 first_button_text, args);
271 va_end (args);
272 }
273
274 return widget;
275}

277void
278ev_message_area_set_image (EvMessageArea *area,
279 GtkWidget *image)
280{
281 GtkWidget *parent;
282
283 g_return_if_fail (EV_IS_MESSAGE_AREA (area));
284
285 area->priv->message_type = GTK_MESSAGE_OTHER;
286
287 parent = gtk_widget_get_parent (area->priv->image);
288 gtk_container_add (GTK_CONTAINER (parent), image);
289 gtk_container_remove (GTK_CONTAINER (parent), area->priv->image);
290 gtk_box_reorder_child (GTK_BOX (parent), image, 0);
291
292 area->priv->image = image;
293
294 g_object_notify (G_OBJECT (area), "image");
295}

314
315 if (str) {
316 gchar *msg;
317
318 msg = g_strdup_printf ("<b>%s</b>", str);
319 gtk_label_set_markup (GTK_LABEL (area->priv->label), msg);
320 g_free (msg);
321 } else {
322 gtk_label_set_markup (GTK_LABEL (area->priv->label), NULL);
