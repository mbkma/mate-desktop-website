54void
55caja_file_queue_enqueue (CajaFileQueue *queue,
56 CajaFile *file)
57{
58 if (g_hash_table_lookup (queue->item_to_link_map, file) != NULL)
59 {
60 /* It's already on the queue. */
61 return;
62 }
63
64 if (queue->tail == NULL)
65 {
66 queue->head = g_list_append (NULL, file);
67 queue->tail = queue->head;
68 }
69 else
70 {
71 queue->tail = g_list_append (queue->tail, file);
72 queue->tail = queue->tail->next;
73 }
74
75 caja_file_ref (file);
76 g_hash_table_insert (queue->item_to_link_map, file, queue->tail);
77}

79CajaFile *
80caja_file_queue_dequeue (CajaFileQueue *queue)
81{
82 CajaFile *file;
83
84 file = caja_file_queue_head (queue);
85 caja_file_queue_remove (queue, file);
86
87 return file;

90void
91caja_file_queue_remove (CajaFileQueue *queue,
92 CajaFile *file)
93{
94 GList *link;
95
96 link = g_hash_table_lookup (queue->item_to_link_map, file);
97
98 if (link == NULL)
99 {
100 /* It's not on the queue */
101 return;
102 }
103
104 if (link == queue->tail)
105 {
106 /* Need to special-case removing the tail. */
107 queue->tail = queue->tail->prev;
108 }
109
110 queue->head = g_list_remove_link (queue->head, link);
111 g_list_free (link);
112 g_hash_table_remove (queue->item_to_link_map, file);
113
114 caja_file_unref (file);
115}

123 }
