152static void
153caja_directory_finalize (GObject *object)
154{
155 CajaDirectory *directory;
156
157 directory = CAJA_DIRECTORY (object);
158
159 g_hash_table_remove (directories, directory->details->location);
160
161 caja_directory_cancel (directory);
162 g_assert (directory->details->count_in_progress == NULL);
163 g_assert (directory->details->top_left_read_state == NULL);
164
165 if (directory->details->monitor_list != NULL)
166 {
167 g_warning ("destroying a CajaDirectory while it's being monitored");
168 g_list_free_full (directory->details->monitor_list, g_free);
169 }
170
171 if (directory->details->monitor != NULL)
172 {
173 caja_monitor_cancel (directory->details->monitor);
174 }
175
176 if (directory->details->dequeue_pending_idle_id != 0)
177 {
178 g_source_remove (directory->details->dequeue_pending_idle_id);
179 }
180
181 if (directory->details->call_ready_idle_id != 0)
182 {
183 g_source_remove (directory->details->call_ready_idle_id);
184 }
185
186 if (directory->details->location)
187 {
188 g_object_unref (directory->details->location);
189 }
190
191 g_assert (directory->details->file_list == NULL);
192 g_hash_table_destroy (directory->details->file_hash);
193
194 caja_file_queue_destroy (directory->details->high_priority_queue);
195 caja_file_queue_destroy (directory->details->low_priority_queue);
196 caja_file_queue_destroy (directory->details->extension_queue);
197 g_assert (directory->details->directory_load_in_progress == NULL);
198 g_assert (directory->details->count_in_progress == NULL);
199 g_assert (directory->details->dequeue_pending_idle_id == 0);
200 g_list_free_full (directory->details->pending_file_info, g_object_unref);
201
202 G_OBJECT_CLASS (caja_directory_parent_class)->finalize (object);
203}

264CajaDirectory *
265caja_directory_get_internal (GFile *location, gboolean create)
266{
267 CajaDirectory *directory;
268
269 /* Create the hash table first time through. */
270 if (directories == NULL) {
271 directories = g_hash_table_new (g_file_hash, (GCompareFunc) g_file_equal);
272 caja_global_preferences_init ();
273 }
274
275 /* If the object is already in the hash table, look it up. */
276
277 directory = g_hash_table_lookup (directories,
278 location);
279 if (directory != NULL)
280 {
281 caja_directory_ref (directory);
282 }
283 else if (create)
284 {
285 /* Create a new directory object instead. */
286 directory = caja_directory_new (location);
287 if (directory == NULL)
288 {
289 return NULL;
290 }
291
292 /* Put it in the hash table. */
293 g_hash_table_insert (directories,
294 directory->details->location,
295 directory);
296 }
297
298 return directory;
299}

414static char *
415real_get_name_for_self_as_new_file (CajaDirectory *directory)
416{
417 char *directory_uri;
418 char *name, *colon;
419
420 directory_uri = caja_directory_get_uri (directory);
421
422 colon = strchr (directory_uri, ':');
423 if (colon == NULL || colon == directory_uri)
424 {
425 name = g_strdup (directory_uri);
426 }
427 else
428 {
429 name = g_strndup (directory_uri, colon - directory_uri);
430 }
431 g_free (directory_uri);
432
433 return name;
434}

452static CajaDirectory *
453caja_directory_new (GFile *location)
454{
455 CajaDirectory *directory;
456 char *uri;
457
458 uri = g_file_get_uri (location);
459
460 if (eel_uri_is_desktop (uri))
461 {
462 directory = CAJA_DIRECTORY (g_object_new (CAJA_TYPE_DESKTOP_DIRECTORY, NULL));
463 }
464 else if (eel_uri_is_search (uri))
465 {
466 directory = CAJA_DIRECTORY (g_object_new (CAJA_TYPE_SEARCH_DIRECTORY, NULL));
467 }
468 else if (g_str_has_suffix (uri, CAJA_SAVED_SEARCH_EXTENSION))
469 {
470 directory = CAJA_DIRECTORY (caja_search_directory_new_from_saved_search (uri));
471 }
472 else
473 {
474 directory = CAJA_DIRECTORY (g_object_new (CAJA_TYPE_VFS_DIRECTORY, NULL));
475 }
476
477 set_directory_location (directory, location);
478
479 g_free (uri);
480
481 return directory;
482}

537static GList *
538extract_from_hash_table (CajaDirectory *directory, CajaFile *file)
539{
540 const char *name;

542
543 name = file->details->name;
544 if (name == NULL)
545 {
546 return NULL;
547 }
548
549 /* Find the list node in the hash table. */
550 node = g_hash_table_lookup (directory->details->file_hash, name);
551 g_hash_table_remove (directory->details->file_hash, name);
552
553 return node;

556void
557caja_directory_add_file (CajaDirectory *directory, CajaFile *file)
558{
559 GList *node;
560 gboolean add_to_work_queue;
561
562 g_assert (CAJA_IS_DIRECTORY (directory));
563 g_assert (CAJA_IS_FILE (file));
564 g_assert (file->details->name != NULL);
565
566 /* Add to list. */
567 node = g_list_prepend (directory->details->file_list, file);
568 directory->details->file_list = node;
569
570 /* Add to hash table. */
571 add_to_hash_table (directory, file, node);
572
573 directory->details->confirmed_file_count++;
574
575 add_to_work_queue = FALSE;
576 if (caja_directory_is_file_list_monitored (directory))
577 {
578 /* Ref if we are monitoring, since monitoring owns the file list. */
579 caja_file_ref (file);
580 add_to_work_queue = TRUE;
581 }
582 else if (caja_directory_has_active_request_for_file (directory, file))
583 {
584 /* We're waiting for the file in a call_when_ready. Make sure
585 we add the file to the work queue so that said waiter won't
586 wait forever for e.g. all files in the directory to be done */
587 add_to_work_queue = TRUE;
588 }
589
590 if (add_to_work_queue)
591 {
592 caja_directory_add_file_to_work_queue (directory, file);
593 }
594}

596void
597caja_directory_remove_file (CajaDirectory *directory, CajaFile *file)
598{
599 GList *node;
600
601 g_assert (CAJA_IS_DIRECTORY (directory));
602 g_assert (CAJA_IS_FILE (file));
603 g_assert (file->details->name != NULL);
604
605 /* Find the list node in the hash table. */
606 node = extract_from_hash_table (directory, file);
607 g_assert (node != NULL);
608 g_assert (node->data == file);
609
610 /* Remove the item from the list. */
611 directory->details->file_list = g_list_remove_link
612 (directory->details->file_list, node);
613 g_list_free_1 (node);
614
615 caja_directory_remove_file_from_work_queue (directory, file);
616
617 if (!file->details->unconfirmed)
618 {
619 directory->details->confirmed_file_count--;
620 }
621
622 /* Unref if we are monitoring. */
623 if (caja_directory_is_file_list_monitored (directory))
624 {
625 caja_file_unref (file);
626 }
627}

857void
858caja_directory_notify_files_added (GList *files)
859{
860 GHashTable *added_lists;
861 GList *p;
862 GHashTable *parent_directories;
863 CajaFile *file;
864 GFile *parent;
865 CajaDirectory *directory = NULL;
866 GFile *location = NULL;
867
868 /* Make a list of added files in each directory. */
869 added_lists = g_hash_table_new (NULL, NULL);
870
871 /* Make a list of parent directories that will need their counts updated. */
872 parent_directories = g_hash_table_new (NULL, NULL);
873
874 for (p = files; p != NULL; p = p->next)
875 {
876 location = p->data;
877
878 /* See if the directory is already known. */
879 directory = get_parent_directory_if_exists (location);
880 if (directory == NULL)
881 {
882 /* In case the directory is not being
883 * monitored, but the corresponding file is,
884 * we must invalidate it's item count.
885 */
886
887 file = NULL;
888 parent = g_file_get_parent (location);
889 if (parent)
890 {
891 file = caja_file_get_existing (parent);
892 g_object_unref (parent);
893 }
894
895 if (file != NULL)
896 {
897 caja_file_invalidate_count_and_mime_list (file);
898 caja_file_unref (file);
899 }
900
901 continue;
902 }
903
904 collect_parent_directories (parent_directories, directory);
905
906 /* If no one is monitoring files in the directory, nothing to do. */
907 if (!caja_directory_is_file_list_monitored (directory))
908 {
909 caja_directory_unref (directory);
910 continue;
911 }
912
913 file = caja_file_get_existing (location);
914 /* We check is_added here, because the file could have been added
915 * to the directory by a caja_file_get() but not gotten
916 * files_added emitted
917 */
918 if (file && file->details->is_added)
919 {
920 /* A file already exists, it was probably renamed.
921 * If it was renamed this could be ignored, but
922 * queue a change just in case */
923 caja_file_changed (file);
924 }
925 else
926 {
927 hash_table_list_prepend (added_lists,
928 directory,
929 g_object_ref (location));
930 }
931 caja_file_unref (file);
932 caja_directory_unref (directory);
933 }
934
935 /* Now get file info for the new files. This creates CajaFile
936 * objects for the new files, and sends out a files_added signal.
937 */
938 g_hash_table_foreach (added_lists, call_get_file_info_free_list, NULL);
939 g_hash_table_destroy (added_lists);
940
941 /* Invalidate count for each parent directory. */
942 g_hash_table_foreach (parent_directories, invalidate_count_and_unref, NULL);
943 g_hash_table_destroy (parent_directories);
944}

1100 CajaDirectory *directory = NULL;

1102

1104 collection.directories = NULL;
1105

1118 {
1119 new_directory_location = g_object_ref (new_location);

1122 {

1124 directory->details->location);
1125 if (relative_path != NULL)
1126 {
1127 new_directory_location = g_file_resolve_relative_path (new_location, relative_path);
1128 g_free (relative_path);
1129
1130 }

1143 caja_file_ref (directory->details->as_file));
1144 }

1146 (affected_files,
1147 caja_file_list_copy (directory->details->file_list));

1190void
1191caja_directory_notify_files_moved (GList *file_pairs)
1192{
1193 GList *p, *affected_files, *node;
1194 GFilePair *pair;
1195 CajaFile *file;
1196 CajaDirectory *old_directory, *new_directory;
1197 GHashTable *parent_directories;
1198 GList *new_files_list, *unref_list;
1199 GHashTable *added_lists, *changed_lists;
1200 char *name;
1201 CajaFileAttributes cancel_attributes;
1202 GFile *to_location = NULL;
1203 GFile *from_location = NULL;
1204
1205 /* Make a list of added and changed files in each directory. */
1206 new_files_list = NULL;
1207 added_lists = g_hash_table_new (NULL, NULL);
1208 changed_lists = g_hash_table_new (NULL, NULL);
1209 unref_list = NULL;
1210
1211 /* Make a list of parent directories that will need their counts updated. */
1212 parent_directories = g_hash_table_new (NULL, NULL);
1213
1214 cancel_attributes = caja_file_get_all_attributes ();
1215
1216 for (p = file_pairs; p != NULL; p = p->next)
1217 {
1218 pair = p->data;
1219 from_location = pair->from;
1220 to_location = pair->to;
1221
1222 /* Handle overwriting a file. */
1223 file = caja_file_get_existing (to_location);
1224 if (file != NULL)
1225 {
1226 /* Mark it gone and prepare to send the changed signal. */
1227 caja_file_mark_gone (file);
1228 new_directory = file->details->directory;
1229 hash_table_list_prepend (changed_lists,
1230 new_directory,
1231 file);
1232 collect_parent_directories (parent_directories,
1233 new_directory);
1234 }
1235
1236 /* Update any directory objects that are affected. */
1237 affected_files = caja_directory_moved_internal (from_location,
1238 to_location);
1239 for (node = affected_files; node != NULL; node = node->next)
1240 {
1241 file = CAJA_FILE (node->data);
1242 hash_table_list_prepend (changed_lists,
1243 file->details->directory,
1244 file);
1245 }
1246 unref_list = g_list_concat (unref_list, affected_files);
1247
1248 /* Move an existing file. */
1249 file = caja_file_get_existing (from_location);
1250 if (file == NULL)
1251 {
1252 /* Handle this as if it was a new file. */
1253 new_files_list = g_list_prepend (new_files_list,
1254 to_location);
1255 }
1256 else
1257 {
1258 /* Handle notification in the old directory. */
1259 old_directory = file->details->directory;
1260 collect_parent_directories (parent_directories, old_directory);
1261
1262 /* Cancel loading of attributes in the old directory */
1263 caja_directory_cancel_loading_file_attributes
1264 (old_directory, file, cancel_attributes);
1265
1266 /* Locate the new directory. */
1267 new_directory = get_parent_directory (to_location);
1268 collect_parent_directories (parent_directories, new_directory);
1269 /* We can unref now -- new_directory is in the
1270 * parent directories list so it will be
1271 * around until the end of this function
1272 * anyway.
1273 */
1274 caja_directory_unref (new_directory);
1275
1276 /* Update the file's name and directory. */
1277 name = g_file_get_basename (to_location);
1278 caja_file_update_name_and_directory
1279 (file, name, new_directory);
1280 g_free (name);
1281
1282 /* Update file attributes */
1283 caja_file_invalidate_attributes (file, CAJA_FILE_ATTRIBUTE_INFO);
1284
1285 hash_table_list_prepend (changed_lists,
1286 old_directory,
1287 file);
1288 if (old_directory != new_directory)
1289 {
1290 hash_table_list_prepend (added_lists,
1291 new_directory,
1292 file);
1293 }
1294
1295 /* Unref each file once to balance out caja_file_get_by_uri. */
1296 unref_list = g_list_prepend (unref_list, file);
1297 }
1298 }
1299
1300 /* Now send out the changed and added signals for existing file objects. */
1301 g_hash_table_foreach (changed_lists, call_files_changed_free_list, NULL);
1302 g_hash_table_destroy (changed_lists);
1303 g_hash_table_foreach (added_lists, call_files_added_free_list, NULL);
1304 g_hash_table_destroy (added_lists);
1305
1306 /* Let the file objects go. */
1307 caja_file_list_free (unref_list);
1308
1309 /* Invalidate count for each parent directory. */
1310 g_hash_table_foreach (parent_directories, invalidate_count_and_unref, NULL);
1311 g_hash_table_destroy (parent_directories);
1312
1313 /* Separate handling for brand new file objects. */
1314 caja_directory_notify_files_added (new_files_list);
1315 g_list_free (new_files_list);
1316}
