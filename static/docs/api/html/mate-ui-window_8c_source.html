36static void
37mate_ui_window_rebuild_layout(MateUiWindow *self)
38{
39 MateUiWindowPrivate *priv = mate_ui_window_get_instance_private(self);
40
41 /* Remove all children from main_box */
42 GList *children = gtk_container_get_children(GTK_CONTAINER(priv->main_box));
43 for (GList *l = children; l != NULL; l = l->next)
44 {
45 gtk_container_remove(GTK_CONTAINER(priv->main_box), GTK_WIDGET(l->data));
46 }
47 g_list_free(children);
48
49 /* Re-add in order: menubar, toolbar, content, statusbar */
50 if (priv->menubar != NULL)
51 {
52 gtk_box_pack_start(GTK_BOX(priv->main_box), priv->menubar, FALSE, FALSE, 0);
53 gtk_widget_show(priv->menubar);
54 }
55
56 if (priv->toolbar != NULL)
57 {
58 gtk_box_pack_start(GTK_BOX(priv->main_box), priv->toolbar, FALSE, FALSE, 0);
59 gtk_widget_show(priv->toolbar);
60 }
61
62 if (priv->content != NULL)
63 {
64 gtk_box_pack_start(GTK_BOX(priv->main_box), priv->content, TRUE, TRUE, 0);
65 gtk_widget_show(priv->content);
66 }
67
68 if (priv->statusbar != NULL)
69 {
70 gtk_box_pack_end(GTK_BOX(priv->main_box), priv->statusbar, FALSE, FALSE, 0);
71 gtk_widget_show(priv->statusbar);
72 }
73}

75static gboolean
76mate_ui_window_configure_event(GtkWidget *widget,
77 GdkEventConfigure *event,
78 gpointer user_data G_GNUC_UNUSED)
79{
80 MateUiWindow *self = MATE_UI_WINDOW(widget);
81 MateUiWindowPrivate *priv = mate_ui_window_get_instance_private(self);
82
83 if (priv->settings == NULL)
84 return FALSE;
85
86 /* Don't save size when maximized */
87 GdkWindow *gdk_window = gtk_widget_get_window(widget);
88 if (gdk_window != NULL)
89 {
90 GdkWindowState state = gdk_window_get_state(gdk_window);
91 if (state & GDK_WINDOW_STATE_MAXIMIZED)
92 return FALSE;
93 }
94
95 if (priv->width_key != NULL)
96 g_settings_set_int(priv->settings, priv->width_key, event->width);
97
98 if (priv->height_key != NULL)
99 g_settings_set_int(priv->settings, priv->height_key, event->height);
100
101 return FALSE;
102}

104static gboolean
105mate_ui_window_state_event(GtkWidget *widget,
106 GdkEventWindowState *event,
107 gpointer user_data G_GNUC_UNUSED)
108{
109 MateUiWindow *self = MATE_UI_WINDOW(widget);
110 MateUiWindowPrivate *priv = mate_ui_window_get_instance_private(self);
111
112 if (priv->settings == NULL || priv->maximized_key == NULL)
113 return FALSE;
114
115 if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED)
116 {
117 gboolean maximized = (event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED) != 0;
118 g_settings_set_boolean(priv->settings, priv->maximized_key, maximized);
119 }
120
121 return FALSE;
122}

184GtkWidget *
185mate_ui_window_new(GtkApplication *app,
186 const gchar *title,
187 MateUiWindowFlags flags)
188{
189 g_return_val_if_fail(GTK_IS_APPLICATION(app), NULL);
190
191 MateUiWindow *window = g_object_new(MATE_UI_TYPE_WINDOW,
192 "application", app,
193 NULL);
194
195 MateUiWindowPrivate *priv = mate_ui_window_get_instance_private(window);
196 priv->flags = flags;
197
198 if (title != NULL)
199 gtk_window_set_title(GTK_WINDOW(window), title);
200
201 /* Set up headerbar if requested */
202 if (flags & MATE_UI_WINDOW_USE_HEADERBAR)
203 {
204 GtkWidget *headerbar = gtk_header_bar_new();
205 gtk_header_bar_set_show_close_button(GTK_HEADER_BAR(headerbar), TRUE);
206 if (title != NULL)
207 gtk_header_bar_set_title(GTK_HEADER_BAR(headerbar), title);
208 gtk_window_set_titlebar(GTK_WINDOW(window), headerbar);
209 gtk_widget_show(headerbar);
210 }
211
212 return GTK_WIDGET(window);
213}

236
237 priv->menubar = menubar;

241
242 mate_ui_window_rebuild_layout(window);

412void
413mate_ui_window_bind_settings(MateUiWindow *window,
414 GSettings *settings,
415 const gchar *width_key,
416 const gchar *height_key,
417 const gchar *maximized_key)
418{
419 g_return_if_fail(MATE_UI_IS_WINDOW(window));
420 g_return_if_fail(G_IS_SETTINGS(settings));
421
422 MateUiWindowPrivate *priv = mate_ui_window_get_instance_private(window);
423
424 /* Clear previous binding */
425 g_clear_object(&priv->settings);
426 g_clear_pointer(&priv->width_key, g_free);
427 g_clear_pointer(&priv->height_key, g_free);
428 g_clear_pointer(&priv->maximized_key, g_free);
429
430 if (priv->configure_handler > 0)
431 {
432 g_signal_handler_disconnect(window, priv->configure_handler);
433 priv->configure_handler = 0;
434 }
435 if (priv->state_handler > 0)
436 {
437 g_signal_handler_disconnect(window, priv->state_handler);
438 priv->state_handler = 0;
439 }
440
441 /* Set new binding */
442 priv->settings = g_object_ref(settings);
443 priv->width_key = g_strdup(width_key);
444 priv->height_key = g_strdup(height_key);
445 priv->maximized_key = g_strdup(maximized_key);
446
447 /* Connect signals */
448 priv->configure_handler = g_signal_connect(window, "configure-event",
449 G_CALLBACK(mate_ui_window_configure_event), NULL);
450 priv->state_handler = g_signal_connect(window, "window-state-event",
451 G_CALLBACK(mate_ui_window_state_event), NULL);
452}

465void
466mate_ui_window_set_default_size_from_settings(MateUiWindow *window,
467 GSettings *settings,
468 const gchar *width_key,
469 const gchar *height_key,
470 gint default_width,
471 gint default_height)
472{
473 g_return_if_fail(MATE_UI_IS_WINDOW(window));
474 g_return_if_fail(G_IS_SETTINGS(settings));
475 g_return_if_fail(width_key != NULL);
476 g_return_if_fail(height_key != NULL);
477
478 gint width = g_settings_get_int(settings, width_key);
479 gint height = g_settings_get_int(settings, height_key);
480
481 if (width <= 0)
482 width = default_width;
483 if (height <= 0)
484 height = default_height;
485
486 gtk_window_set_default_size(GTK_WINDOW(window), width, height);
487}
