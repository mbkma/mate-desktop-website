72static void
73fm_widget_view_begin_loading (FMDirectoryView *view)
74{
75 GtkWindow *window;
76 CajaFile *file;
77 gchar *uri;
78 GList *providers, *l;
79 char *mimetype;
80 GtkWidget *widget;
81 FMWidgetView *widget_view;
82
83 widget_view = FM_WIDGET_VIEW (view);
84
85 uri = fm_directory_view_get_uri (view);
86 file = caja_file_get_by_uri (uri);
87 mimetype = caja_file_get_mime_type (file);
88
89 providers = caja_extensions_get_for_type (CAJA_TYPE_WIDGET_VIEW_PROVIDER);
90 for (l = providers; l != NULL; l = l->next)
91 {
92 CajaWidgetViewProvider *provider;
93
94 provider = CAJA_WIDGET_VIEW_PROVIDER (l->data);
95 if (caja_widget_view_provider_supports_uri (provider, uri,
96 caja_file_get_file_type (file),
97 mimetype)) {
98 widget_view->provider = provider;
99 break;
100 }
101 }
102 caja_file_unref (file);
103 g_free (mimetype);
104 caja_module_extension_list_free (providers);
105
106 if (widget_view->provider == NULL) {
107 g_free (uri);
108 return;
109 }
110
111 caja_widget_view_provider_set_location (widget_view->provider, uri);
112 g_free (uri);
113
114 widget = caja_widget_view_provider_get_widget (widget_view->provider);
115 gtk_container_add (GTK_CONTAINER(widget_view), widget);
116
117 window = fm_directory_view_get_containing_window (view);
118 caja_widget_view_provider_set_window (widget_view->provider, window);
119}

415static gboolean
416fm_widget_view_supports_uri (const char *uri,
417 GFileType file_type,
418 const char *mime_type)
419{
420 GList *providers, *l;
421 gboolean result = FALSE;
422
423 providers = caja_extensions_get_for_type (CAJA_TYPE_WIDGET_VIEW_PROVIDER);
424
425 for (l = providers; l != NULL; l = l->next)
426 {
427 CajaWidgetViewProvider *provider;
428
429 provider = CAJA_WIDGET_VIEW_PROVIDER (l->data);
430 if (caja_widget_view_provider_supports_uri (provider, uri, file_type, mime_type)) {
431 result = TRUE;
432 }
433 }
434 caja_module_extension_list_free (providers);
435
436 return result;
437}
