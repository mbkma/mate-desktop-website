147static void
148create_log_cb (LogviewLog *log,
149 GError *error,
150 gpointer user_data)
151{
152 CreateCBData *data = user_data;
153
154 if (log) {
155 char *log_uri;
156 LogviewPrefs *prefs;
157 GFile *file;
158
159 log_uri = logview_log_get_uri (log);
160
161 /* creation went well, store the log and notify */
162 g_hash_table_insert (data->manager->priv->logs,
163 log_uri, log);
164
165 prefs = logview_prefs_get ();
166 file = logview_log_get_gfile (log);
167 logview_prefs_store_log (prefs, file);
168
169 g_object_unref (file);
170
171 g_signal_emit (data->manager, signals[LOG_ADDED], 0, log, NULL);
172
173 if (data->set_active) {
174 logview_manager_set_active_log (data->manager, log);
175 }
176 } else {
177 char *path;
178
179 /* notify the error */
180 path = g_file_get_path (data->file);
181
182 if (!data->is_multiple) {
183 logview_app_add_error (logview_app_get (),
184 path, error->message);
185 } else {
186 char **error_arr = g_new0 (char *, 3);
187
188 error_arr[0] = g_strdup (path);
189 error_arr[1] = g_strdup (error->message);
190 error_arr[2] = NULL;
191
192 g_ptr_array_add (op->errors, error_arr);
193 }
194
195 g_free (path);
196 }
197
198 if (data->is_multiple) {
199 op->current++;
200
201 if (op->total == op->current) {
202 logview_app_add_errors (logview_app_get (), op->errors);
203 multiple_creation_op_free (op);
204 op = NULL;
205 }
206 }
207
208 g_object_unref (data->file);
209 g_slice_free (CreateCBData, data);
210}

212static void
213add_log_from_gfile_internal (LogviewManager *manager,
214 GFile *file,
215 gboolean set_active,
216 gboolean is_multiple)
217{
218 char *file_uri;
219 LogviewLog *log;
220 CreateCBData *data;
221
222 file_uri = g_file_get_uri (file);
223
224 if (set_active == FALSE) {
225 /* if it's the first log being added, set it as active anyway */
226 set_active = (manager->priv->logs == NULL);
227 }
228
229 if ((log = g_hash_table_lookup (manager->priv->logs, file_uri)) != NULL) {
230 /* log already exists, don't load it */
231 if (set_active) {
232 logview_manager_set_active_log (manager, log);
233 }
234 } else {
235 data = g_slice_new0 (CreateCBData);
236 data->manager = manager;
237 data->set_active = set_active;
238 data->is_multiple = is_multiple;
239 data->file = g_object_ref (file);
240
241 logview_log_create_from_gfile (file, create_log_cb, data);
242 }
243
244 g_free (file_uri);
245}

273void
274logview_manager_set_active_log (LogviewManager *manager,
275 LogviewLog *log)
276{
277 LogviewLog *old_log = NULL;
278 GFile *file;
279 char *path;
280
281 g_assert (LOGVIEW_IS_MANAGER (manager));
282
283 if (manager->priv->active_log) {
284 old_log = manager->priv->active_log;
285 }
286
287 manager->priv->active_log = g_object_ref (log);
288
289 file = logview_log_get_gfile (log);
290 path = g_file_get_path (file);
291 logview_prefs_store_active_logfile (logview_prefs_get (), path);
292 g_free (path);
293 g_object_unref (file);
294
295 g_signal_emit (manager, signals[ACTIVE_CHANGED], 0, log, old_log, NULL);
296
297 if (old_log) {
298 g_object_unref (old_log);
299 }
300}

322void
323logview_manager_add_logs_from_name_list (LogviewManager *manager,
324 GSList *names,
325 const char *active)
326{
327 GSList *l;
328
329 g_assert (LOGVIEW_IS_MANAGER (manager));
330 g_assert (op == NULL);
331
332 op = multiple_creation_op_new (g_slist_length (names));
333
334 for (l = names; l; l = l->next) {
335 logview_manager_add_log_from_name (manager, l->data,
336 (g_ascii_strcasecmp (active, l->data) == 0),

385void
386logview_manager_close_active_log (LogviewManager *manager)
387{
388 LogviewLog *active_log;
389 char *log_uri;
390 GFile *file;
391
392 g_assert (LOGVIEW_IS_MANAGER (manager));
393
394 active_log = manager->priv->active_log;
395 if (active_log == NULL) {
396 return;
397 }
398
399 log_uri = logview_log_get_uri (active_log);
400 file = logview_log_get_gfile (active_log);
401
402 g_signal_emit (manager, signals[LOG_CLOSED], 0, active_log, NULL);
403
404 logview_prefs_remove_stored_log (logview_prefs_get (), file);
405
406 g_object_unref (file);
407
408 /* drop the hash table ref */
409 g_hash_table_remove (manager->priv->logs, log_uri);
410
411 g_free (log_uri);
412
413 /* someone else will take care of setting the next active log to us */
414}
