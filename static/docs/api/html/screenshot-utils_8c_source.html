38static GtkWidget *selection_window;
39
40#define SELECTION_NAME "_MATE_PANEL_SCREENSHOT"
41
42/* To make sure there is only one screenshot taken at a time,
43 * (Imagine key repeat for the print screen key) we hold a selection
44 * until we are done taking the screenshot
45 */
46gboolean
47screenshot_grab_lock (void)
48{
49 GdkAtom selection_atom;
50 gboolean result = FALSE;
51 GdkDisplay *display;
52
53 selection_atom = gdk_atom_intern (SELECTION_NAME, FALSE);
54 gdk_x11_grab_server ();
55
56 if (gdk_selection_owner_get (selection_atom) != NULL)
57 goto out;
58
59 selection_window = gtk_invisible_new ();
60 gtk_widget_show (selection_window);
61
62 if (!gtk_selection_owner_set (selection_window,
63 gdk_atom_intern (SELECTION_NAME, FALSE),
64 GDK_CURRENT_TIME))
65 {
66 gtk_widget_destroy (selection_window);
67 selection_window = NULL;
68 goto out;
69 }
70
71 result = TRUE;
72
73 out:
74 gdk_x11_ungrab_server ();
75
76 display = gdk_display_get_default ();
77 gdk_display_flush (display);
78
79 return result;
80}
81
82void
83screenshot_release_lock (void)
84{
85 GdkDisplay *display;
86
87 if (selection_window)
88 {
89 gtk_widget_destroy (selection_window);
90 selection_window = NULL;
91 }
92
93 display = gdk_display_get_default ();
94 gdk_display_flush (display);
95}

97static GdkWindow *
98screen_get_active_window (GdkScreen *screen)
99{
100 GdkWindow *ret = NULL;
101 Atom type_return;
102 gint format_return;
103 gulong nitems_return;
104 gulong bytes_after_return;
105 guchar *data = NULL;
106
107 if (!gdk_x11_screen_supports_net_wm_hint (screen,
108 gdk_atom_intern_static_string ("_NET_ACTIVE_WINDOW")))
109 return NULL;
110
111 if (XGetWindowProperty (GDK_DISPLAY_XDISPLAY (gdk_screen_get_display (screen)),
112 RootWindow (GDK_DISPLAY_XDISPLAY (gdk_screen_get_display (screen)),
113 GDK_SCREEN_XNUMBER (screen)),
114 gdk_x11_get_xatom_by_name_for_display (gdk_screen_get_display (screen),
115 "_NET_ACTIVE_WINDOW"),
116 0, 1, False, XA_WINDOW, &type_return,
117 &format_return, &nitems_return,
118 &bytes_after_return, &data)
119 == Success)
120 {
121 if ((type_return == XA_WINDOW) && (format_return == 32) && (data))
122 {
123 Window window = *(Window *) data;
124
125 if (window != None)
126 {
127 ret = gdk_x11_window_foreign_new_for_display (gdk_screen_get_display (screen),
128 window);
129 }
130 }
131 }
132
133 if (data)
134 XFree (data);
135
136 return ret;
137}

151static gboolean
152screenshot_window_is_desktop (GdkWindow *window)
153{
154 GdkWindow *root_window = gdk_get_default_root_window ();
155 GdkWindowTypeHint window_type_hint;
156
157 if (window == root_window)
158 return TRUE;
159
160 window_type_hint = gdk_window_get_type_hint (window);
161 if (window_type_hint == GDK_WINDOW_TYPE_HINT_DESKTOP)
162 return TRUE;
163
164 return FALSE;
165
166}

168GdkWindow *
169screenshot_find_current_window ()
170{
171 GdkWindow *current_window;
172 GdkDisplay *display;
173 GdkSeat *seat;
174 GdkDevice *device;
175
176 current_window = screenshot_find_active_window ();
177 display = gdk_window_get_display (current_window);
178 seat = gdk_display_get_default_seat (display);
179 device = gdk_seat_get_pointer (seat);
180
181 /* If there's no active window, we fall back to returning the
182 * window that the cursor is in.
183 */
184 if (!current_window)
185 current_window = gdk_device_get_window_at_position (device, NULL, NULL);
186
187 if (current_window)
188 {
189 if (screenshot_window_is_desktop (current_window))
190 /* if the current window is the desktop (e.g. caja), we
191 * return NULL, as getting the whole screen makes more sense.
192 */
193 return NULL;
194
195 /* Once we have a window, we take the toplevel ancestor. */
196 current_window = gdk_window_get_toplevel (current_window);
197 }
198
199 return current_window;
200}

208static gboolean
209select_area_button_press (GtkWidget *window,
210 GdkEventButton *event,
211 select_area_filter_data *data)
212{
213 if (data->button_pressed)
214 return TRUE;
215
216 data->button_pressed = TRUE;
217 data->rect.x = event->x_root;
218 data->rect.y = event->y_root;
219
220 return TRUE;
221}

223static gboolean
224select_area_motion_notify (GtkWidget *window,
225 GdkEventMotion *event,
226 select_area_filter_data *data)
227{
228 GdkRectangle draw_rect;
229
230 if (!data->button_pressed)
231 return TRUE;
232
233 draw_rect.width = ABS (data->rect.x - event->x_root);
234 draw_rect.height = ABS (data->rect.y - event->y_root);
235 draw_rect.x = MIN (data->rect.x, event->x_root);
236 draw_rect.y = MIN (data->rect.y, event->y_root);
237
238 if (draw_rect.width <= 0 || draw_rect.height <= 0)
239 {
240 gtk_window_move (GTK_WINDOW (window), -100, -100);
241 gtk_window_resize (GTK_WINDOW (window), 10, 10);
242 return TRUE;
243 }
244
245 gtk_window_move (GTK_WINDOW (window), draw_rect.x, draw_rect.y);
246 gtk_window_resize (GTK_WINDOW (window), draw_rect.width, draw_rect.height);
247
248 /* We (ab)use app-paintable to indicate if we have an RGBA window */
249 if (!gtk_widget_get_app_paintable (window))
250 {
251 GdkWindow *gdkwindow = gtk_widget_get_window (window);
252
253 /* Shape the window to make only the outline visible */
254 if (draw_rect.width > 2 && draw_rect.height > 2)
255 {
256 cairo_region_t *region, *region2;
257 cairo_rectangle_int_t region_rect = {
258 0, 0,
259 draw_rect.width - 2, draw_rect.height - 2
260 };
261
262 region = cairo_region_create_rectangle (&region_rect);
263 region_rect.x++;
264 region_rect.y++;
265 region_rect.width -= 2;
266 region_rect.height -= 2;
267 region2 = cairo_region_create_rectangle (&region_rect);
268 cairo_region_subtract (region, region2);
269
270 gdk_window_shape_combine_region (gdkwindow, region, 0, 0);
271
272 cairo_region_destroy (region);
273 cairo_region_destroy (region2);
274 }
275 else
276 gdk_window_shape_combine_region (gdkwindow, NULL, 0, 0);
277 }
278
279 return TRUE;
280}

282static gboolean
283select_area_button_release (GtkWidget *window,
284 GdkEventButton *event,
285 select_area_filter_data *data)
286{
287 if (!data->button_pressed)
288 return TRUE;
289
290 data->rect.width = ABS (data->rect.x - event->x_root);
291 data->rect.height = ABS (data->rect.y - event->y_root);
292 data->rect.x = MIN (data->rect.x, event->x_root);
293 data->rect.y = MIN (data->rect.y, event->y_root);
294
295 gtk_main_quit ();
296
297 return TRUE;
298}

301select_area_key_press (GtkWidget *window,

317static gboolean
318draw (GtkWidget *window, cairo_t *cr, gpointer unused)
319{
320 GtkStyleContext *style;
321
322 style = gtk_widget_get_style_context (window);
323
324 if (gtk_widget_get_app_paintable (window))
325 {
326 cairo_set_operator (cr, CAIRO_OPERATOR_SOURCE);
327 cairo_set_source_rgba (cr, 0, 0, 0, 0);
328 cairo_paint (cr);
329
330 gtk_style_context_save (style);
331 gtk_style_context_add_class (style, GTK_STYLE_CLASS_RUBBERBAND);
332
333 gtk_render_background (style, cr,
334 0, 0,
335 gtk_widget_get_allocated_width (window),
336 gtk_widget_get_allocated_height (window));
337 gtk_render_frame (style, cr,
338 0, 0,
339 gtk_widget_get_allocated_width (window),
340 gtk_widget_get_allocated_height (window));
341
342 gtk_style_context_restore (style);
343 }
344
345 return TRUE;
346}

348static GtkWidget *
349create_select_window (void)
350{
351 GdkScreen *screen = gdk_screen_get_default ();
352 GtkWidget *window = gtk_window_new (GTK_WINDOW_POPUP);
353
354 GdkVisual *visual = gdk_screen_get_rgba_visual (screen);
355 if (gdk_screen_is_composited (screen) && visual)
356 {
357 gtk_widget_set_visual (window, visual);
358 gtk_widget_set_app_paintable (window, TRUE);
359 }
360
361 g_signal_connect (window, "draw", G_CALLBACK (draw), NULL);
362
363 gtk_window_move (GTK_WINDOW (window), -100, -100);
364 gtk_window_resize (GTK_WINDOW (window), 10, 10);
365 gtk_widget_show (window);
366 return window;
367}

374static gboolean
375emit_select_callback_in_idle (gpointer user_data)
376{
377 CallbackData *data = user_data;
378
379 data->callback (&data->rectangle);
380
381 g_slice_free (CallbackData, data);
382
383 return FALSE;

386void
387screenshot_select_area_async (SelectAreaCallback callback)
388{
389 GdkDisplay *display;
390 GdkCursor *cursor;
391 GdkSeat *seat;
392 GdkGrabStatus res;
393 select_area_filter_data data;
394 CallbackData *cb_data;
395
396 data.rect.x = 0;
397 data.rect.y = 0;
398 data.rect.width = 0;
399 data.rect.height = 0;
400 data.button_pressed = FALSE;
401 data.window = create_select_window();
402
403 cb_data = g_slice_new0 (CallbackData);
404 cb_data->callback = callback;
405
406 g_signal_connect (data.window, "key-press-event", G_CALLBACK (select_area_key_press), &data);
407 g_signal_connect (data.window, "button-press-event", G_CALLBACK (select_area_button_press), &data);
408 g_signal_connect (data.window, "button-release-event", G_CALLBACK (select_area_button_release), &data);
409 g_signal_connect (data.window, "motion-notify-event", G_CALLBACK (select_area_motion_notify), &data);
410
411 display = gdk_display_get_default ();
412 cursor = gdk_cursor_new_for_display (display, GDK_CROSSHAIR);
413
414 seat = gdk_display_get_default_seat (display);
415
416 res = gdk_seat_grab (seat,
417 gtk_widget_get_window (data.window),
418 GDK_SEAT_CAPABILITY_ALL,
419 FALSE,
420 cursor,
421 NULL,
422 NULL,
423 NULL);
424
425 if (res != GDK_GRAB_SUCCESS)
426 {
427 g_object_unref (cursor);
428 goto out;
429 }
430
431 gtk_main ();
432
433 gdk_seat_ungrab (seat);
434
435 gtk_widget_destroy (data.window);
436 g_object_unref (cursor);
437 gdk_display_flush (display);
438
439out:
440 cb_data->rectangle = data.rect;
441
442 /* FIXME: we should actually be emitting the callback When
443 * the compositor has finished re-drawing, but there seems to be no easy
444 * way to know that.
445 */
446 g_timeout_add (200, emit_select_callback_in_idle, cb_data);
447}

449static Window
450find_wm_window (Window xid)
451{
452 Window root, parent, *children;
453 unsigned int nchildren;
454
455 do
456 {
457 if (XQueryTree (GDK_DISPLAY_XDISPLAY (gdk_display_get_default ()), xid, &root,
458 &parent, &children, &nchildren) == 0)
459 {
460 g_warning ("Couldn't find window manager window");
461 return None;
462 }
463
464 if (root == parent)
465 return xid;
466
467 xid = parent;
468 }
469 while (TRUE);
470}

472static cairo_region_t *
473make_region_with_monitors (GdkScreen *screen)
474{
475 GdkDisplay *display;
476 cairo_region_t *region;
477 int num_monitors;
478 int i;
479
480 display = gdk_screen_get_display (screen);
481 num_monitors = gdk_display_get_n_monitors (display);
482
483 region = cairo_region_create ();
484
485 for (i = 0; i < num_monitors; i++)
486 {
487 GdkRectangle rect;
488
489 gdk_monitor_get_geometry (gdk_display_get_monitor (display, i), &rect);
490 cairo_region_union_rectangle (region, &rect);
491 }
492
493 return region;
494}

496static void
497blank_rectangle_in_pixbuf (GdkPixbuf *pixbuf, GdkRectangle *rect)
498{
499 int x, y;
500 int x2, y2;
501 guchar *pixels;
502 int rowstride;
503 int n_channels;
504 guchar *row;
505 gboolean has_alpha;
506
507 g_assert (gdk_pixbuf_get_colorspace (pixbuf) == GDK_COLORSPACE_RGB);
508
509 x2 = rect->x + rect->width;
510 y2 = rect->y + rect->height;
511
512 pixels = gdk_pixbuf_get_pixels (pixbuf);
513 rowstride = gdk_pixbuf_get_rowstride (pixbuf);
514 has_alpha = gdk_pixbuf_get_has_alpha (pixbuf);
515 n_channels = gdk_pixbuf_get_n_channels (pixbuf);
516
517 for (y = rect->y; y < y2; y++)
518 {
519 guchar *p;
520
521 row = pixels + y * rowstride;
522 p = row + rect->x * n_channels;
523
524 for (x = rect->x; x < x2; x++)
525 {
526 *p++ = 0;
527 *p++ = 0;
528 *p++ = 0;
529
530 if (has_alpha)
531 *p++ = 255; /* opaque black */
532 }
533 }
534}

536static void
537blank_region_in_pixbuf (GdkPixbuf *pixbuf, cairo_region_t *region)
538{
539 int n_rects;
540 int i;
541 int width, height;
542 cairo_rectangle_int_t pixbuf_rect;
543
544 n_rects = cairo_region_num_rectangles (region);
545
546 width = gdk_pixbuf_get_width (pixbuf);
547 height = gdk_pixbuf_get_height (pixbuf);
548
549 pixbuf_rect.x = 0;
550 pixbuf_rect.y = 0;
551 pixbuf_rect.width = width;
552 pixbuf_rect.height = height;
553
554 for (i = 0; i < n_rects; i++)
555 {
556 cairo_rectangle_int_t rect, dest;
557
558 cairo_region_get_rectangle (region, i, &rect);
559 if (gdk_rectangle_intersect (&rect, &pixbuf_rect, &dest))
560 blank_rectangle_in_pixbuf (pixbuf, &dest);
561
562 }
563}

565/* When there are multiple monitors with different resolutions, the visible area
566 * within the root window may not be rectangular (it may have an L-shape, for
567 * example). In that case, mask out the areas of the root window which would
568 * not be visible in the monitors, so that screenshot do not end up with content
569 * that the user won't ever see.
570 */
571static void
572mask_monitors (GdkPixbuf *pixbuf, GdkWindow *root_window)
573{
574 GdkScreen *screen;
575 cairo_region_t *region_with_monitors;
576 cairo_region_t *invisible_region;
577 cairo_rectangle_int_t rect;
578 gint scale;
579
580 screen = gdk_window_get_screen (root_window);
581 scale = gdk_window_get_scale_factor (root_window);
582
583 region_with_monitors = make_region_with_monitors (screen);
584
585 rect.x = 0;
586 rect.y = 0;
587 rect.width = WidthOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale;
588 rect.height = HeightOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale;
589
590 invisible_region = cairo_region_create_rectangle (&rect);
591 cairo_region_subtract (invisible_region, region_with_monitors);
592
593 blank_region_in_pixbuf (pixbuf, invisible_region);
594
595 cairo_region_destroy (region_with_monitors);
596 cairo_region_destroy (invisible_region);
597}

599GdkPixbuf *
600screenshot_get_pixbuf (GdkWindow *window,
601 GdkRectangle *rectangle,
602 gboolean include_pointer,
603 gboolean include_border,
604 gboolean include_mask)
605{
606 GdkWindow *root;
607 GdkPixbuf *screenshot;
608 gint x_real_orig, y_real_orig, x_orig, y_orig;
609 gint width, real_width, height, real_height;
610 gint screen_width, screen_height, scale;
611
612 /* If the screenshot should include the border, we look for the WM window. */
613
614 if (include_border)
615 {
616 Window xid, wm;
617
618 xid = GDK_WINDOW_XID (window);
619 wm = find_wm_window (xid);
620
621 if (wm != None)
622 window = gdk_x11_window_foreign_new_for_display (gdk_display_get_default (), wm);
623
624 /* fallback to no border if we can't find the WM window. */
625 }
626
627 root = gdk_get_default_root_window ();
628 scale = gdk_window_get_scale_factor (root);
629
630 real_width = gdk_window_get_width (window);
631 real_height = gdk_window_get_height (window);
632
633 screen_width = WidthOfScreen (gdk_x11_screen_get_xscreen (gdk_screen_get_default ())) / scale;
634 screen_height = HeightOfScreen (gdk_x11_screen_get_xscreen (gdk_screen_get_default ())) / scale;
635
636 gdk_window_get_origin (window, &x_real_orig, &y_real_orig);
637
638 x_orig = x_real_orig;
639 y_orig = y_real_orig;
640 width = real_width;
641 height = real_height;
642
643 if (x_orig < 0)
644 {
645 width = width + x_orig;
646 x_orig = 0;
647 }
648
649 if (y_orig < 0)
650 {
651 height = height + y_orig;
652 y_orig = 0;
653 }
654
655 if (x_orig + width > screen_width)
656 width = screen_width - x_orig;
657
658 if (y_orig + height > screen_height)
659 height = screen_height - y_orig;
660
661 if (rectangle)
662 {
663 x_orig = rectangle->x - x_orig;
664 y_orig = rectangle->y - y_orig;
665 width = rectangle->width;
666 height = rectangle->height;
667 }
668
669 screenshot = gdk_pixbuf_get_from_window (root,
670 x_orig, y_orig,
671 width, height);
672
673 /*
674 * Masking currently only works properly with full-screen shots
675 */
676 if (include_mask)
677 mask_monitors (screenshot, root);
678
679#ifdef HAVE_X11_EXTENSIONS_SHAPE_H
680 if (include_border)
681 {
682 XRectangle *rectangles;
683 GdkPixbuf *tmp;
684 int rectangle_count, rectangle_order, i;
685
686 /* we must use XShape to avoid showing what's under the rounder corners
687 * of the WM decoration.
688 */
689
690 rectangles = XShapeGetRectangles (GDK_DISPLAY_XDISPLAY (gdk_display_get_default ()),
691 GDK_WINDOW_XID (window),
692 ShapeBounding,
693 &rectangle_count,
694 &rectangle_order);
695 if (rectangles && rectangle_count > 0 && window != root)
696 {
697 gboolean has_alpha = gdk_pixbuf_get_has_alpha (screenshot);
698
699 if (scale)
700 {
701 width *= scale;
702 height *= scale;
703 }
704
705 tmp = gdk_pixbuf_new (GDK_COLORSPACE_RGB, TRUE, 8, width, height);
706 gdk_pixbuf_fill (tmp, 0);
707
708 for (i = 0; i < rectangle_count; i++)
709 {
710 gint rec_x, rec_y;
711 gint rec_width, rec_height;
712 gint y;
713
714 rec_x = rectangles[i].x;
715 rec_y = rectangles[i].y;
716 rec_width = rectangles[i].width / scale;
717 rec_height = rectangles[i].height / scale;
718
719 if (x_real_orig < 0)
720 {
721 rec_x += x_real_orig;
722 rec_x = MAX(rec_x, 0);
723 rec_width += x_real_orig;
724 }
725
726 if (y_real_orig < 0)
727 {
728 rec_y += y_real_orig;
729 rec_y = MAX(rec_y, 0);
730 rec_height += y_real_orig;
731 }
732
733 if (x_orig + rec_x + rec_width > screen_width)
734 rec_width = screen_width - x_orig - rec_x;
735
736 if (y_orig + rec_y + rec_height > screen_height)
737 rec_height = screen_height - y_orig - rec_y;
738
739 if (scale)
740 {
741 rec_width *= scale;
742 rec_height *= scale;
743 }
744
745 for (y = rec_y; y < rec_y + rec_height; y++)
746 {
747 guchar *src_pixels, *dest_pixels;
748 gint x;
749
750 src_pixels = gdk_pixbuf_get_pixels (screenshot)
751 + y * gdk_pixbuf_get_rowstride (screenshot)
752 + rec_x * (has_alpha ? 4 : 3);
753 dest_pixels = gdk_pixbuf_get_pixels (tmp)
754 + y * gdk_pixbuf_get_rowstride (tmp)
755 + rec_x * 4;
756
757 for (x = 0; x < rec_width; x++)
758 {
759 *dest_pixels++ = *src_pixels++;
760 *dest_pixels++ = *src_pixels++;
761 *dest_pixels++ = *src_pixels++;
762
763 if (has_alpha)
764 *dest_pixels++ = *src_pixels++;
765 else
766 *dest_pixels++ = 255;
767 }
768 }
769 }
770
771 g_object_unref (screenshot);
772 screenshot = tmp;
773 }
774 }

825 cx - xhot, cy - yhot,
826 r2.width, r2.height,
827 cx - xhot, cy - yhot,
828 1.0, 1.0,
829 GDK_INTERP_BILINEAR,
830 255);
831 }
832
833 g_object_unref (cursor_pixbuf);
834 g_object_unref (cursor);
835 }
836 }

838 return screenshot;
