126static void
127place_widgets (MateWeatherApplet *gw_applet)
128{
129 GtkRequisition req;
130 int total_size = 0;
131 gboolean horizontal = FALSE;
132 int panel_size = (int) gw_applet->size;
133 const gchar *temp = NULL;
134 const gchar *icon_name = NULL;
135
136 switch (gw_applet->orient) {
137 case MATE_PANEL_APPLET_ORIENT_LEFT:
138 case MATE_PANEL_APPLET_ORIENT_RIGHT:
139 horizontal = FALSE;
140 break;
141 case MATE_PANEL_APPLET_ORIENT_UP:
142 case MATE_PANEL_APPLET_ORIENT_DOWN:
143 horizontal = TRUE;
144 break;
145 }
146
147 /* Create the weather icon */
148 if ((gw_applet->mateweather_info) && ((icon_name = weather_info_get_icon_name (gw_applet->mateweather_info)) != NULL)) {
149 gw_applet->image = gtk_image_new_from_icon_name (icon_name, GTK_ICON_SIZE_BUTTON);
150 } else {
151 gw_applet->image = gtk_image_new_from_icon_name ("weather-storm", GTK_ICON_SIZE_BUTTON);;
152 }
153 gtk_widget_show (gw_applet->image);
154 gtk_widget_get_preferred_size (gw_applet->image, &req, NULL);
155 if (horizontal)
156 total_size += req.height;
157 else
158 total_size += req.width;
159
160 /* Update temperature text */
161 if ((gw_applet->mateweather_info != NULL) && ((temp = weather_info_get_temp_summary (gw_applet->mateweather_info)) != NULL))
162 gw_applet->label = gtk_label_new (temp);
163 else
164 gw_applet->label = gtk_label_new(_("?"));
165
166 /* Check the label size to determine box layout */
167 gtk_widget_show (gw_applet->label);
168 gtk_widget_get_preferred_size (gw_applet->label, &req, NULL);
169 if (horizontal)
170 total_size += req.height;
171 else
172 total_size += req.width;
173
174 /* Pack the box */
175 if (gw_applet->box)
176 gtk_widget_destroy (gw_applet->box);
177
178 if (horizontal && (total_size <= panel_size))
179 gw_applet->box = gtk_box_new (GTK_ORIENTATION_VERTICAL, 0);
180 else if (horizontal && (total_size > panel_size))
181 gw_applet->box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 2);
182 else if (!horizontal && (total_size <= panel_size))
183 gw_applet->box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 2);
184 else
185 gw_applet->box = gtk_box_new (GTK_ORIENTATION_VERTICAL, 0);
186
187 /* better for vertical panels */
188 if (horizontal)
189 gtk_widget_set_valign (gw_applet->box, GTK_ALIGN_CENTER);
190 else
191 gtk_widget_set_halign (gw_applet->box, GTK_ALIGN_CENTER);
192
193 /* Rebuild the applet it's visual area */
194 gtk_container_add (GTK_CONTAINER (gw_applet->applet), gw_applet->box);
195 gtk_box_pack_start (GTK_BOX (gw_applet->box), gw_applet->image, TRUE, TRUE, 0);
196 gtk_box_pack_start (GTK_BOX (gw_applet->box), gw_applet->label, TRUE, TRUE, 0);
197
198 gtk_widget_show_all (GTK_WIDGET (gw_applet->applet));
199}

251static gboolean
252key_press_cb (GtkWidget *widget, GdkEventKey *event, MateWeatherApplet *gw_applet)
253{
254 switch (event->keyval) {
255 case GDK_KEY_u:
256 if (event->state == GDK_CONTROL_MASK) {
257 mateweather_update (gw_applet);
258 return TRUE;
259 }
260 break;
261 case GDK_KEY_d:
262 if (event->state == GDK_CONTROL_MASK) {
263 details_cb (NULL, gw_applet);
264 return TRUE;
265 }
266 break;
267 case GDK_KEY_KP_Enter:
268 case GDK_KEY_ISO_Enter:
269 case GDK_KEY_3270_Enter:
270 case GDK_KEY_Return:
271 case GDK_KEY_space:
272 case GDK_KEY_KP_Space:
273 details_cb (NULL, gw_applet);
274 return TRUE;
275 default:
276 break;
277 }
278
279 return FALSE;

414static void
415update_finish (WeatherInfo *info, gpointer data)
416{
417 static int gw_fault_counter = 0;
418#ifdef HAVE_LIBNOTIFY
419 char *message, *detail;
420#endif
421 MateWeatherApplet *gw_applet = (MateWeatherApplet *)data;
422
423 /* Update timer */
424 if (gw_applet->timeout_tag > 0)
425 g_source_remove(gw_applet->timeout_tag);
426 if (gw_applet->mateweather_pref.update_enabled)
427 {
428 gint nxtSunEvent;
429
430 gw_applet->timeout_tag
431 = g_timeout_add_seconds ((guint) gw_applet->mateweather_pref.update_interval,
432 timeout_cb, gw_applet);
433
434 if ((info != NULL) && ((nxtSunEvent = weather_info_next_sun_event (info)) >= 0))
435 {
436 gw_applet->suncalc_timeout_tag
437 = g_timeout_add_seconds ((guint) nxtSunEvent,
438 suncalc_timeout_cb, gw_applet);
439 }
440 }
441
442 if ((info != NULL) && weather_info_is_valid (info))
443 {
444 char *s;
445
446 gw_fault_counter = 0;
447
448 /* update tooltip */
449 s = weather_info_get_weather_summary (info);
450 gtk_widget_set_tooltip_text (GTK_WIDGET (gw_applet->applet), s);
451 g_free (s);
452
453 /* Update dialog -- if one is present */
454 if (gw_applet->details_dialog)
455 mateweather_dialog_update (MATEWEATHER_DIALOG (gw_applet->details_dialog));
456
457 /* update applet */
458 place_widgets (gw_applet);
459
460#ifdef HAVE_LIBNOTIFY
461 if (gw_applet->mateweather_pref.show_notifications)
462 {
463 NotifyNotification *n;
464
465 /* Show notifications if possible */
466 if (!notify_is_initted ())
467 notify_init (_("Weather Forecast"));
468
469 if (notify_is_initted ())
470 {
471 GError *error = NULL;
472 const char *icon;
473
474 /* Show notification */
475 message = g_strdup_printf ("%s: %s",
476 weather_info_get_location_name (info),
477 weather_info_get_sky (info));
478 detail = g_strdup_printf (_("City: %s\nSky: %s\nTemperature: %s"),
479 weather_info_get_location_name (info),
480 weather_info_get_sky (info),
481 weather_info_get_temp_summary (info));
482
483 icon = weather_info_get_icon_name (gw_applet->mateweather_info);
484 if (icon == NULL)
485 icon = "stock-unknown";
486
487 n = notify_notification_new (message, detail, icon);
488
489 notify_notification_show (n, &error);
490 if (error)
491 {
492 g_warning ("%s", error->message);
493 g_error_free (error);
494 }
495
496 g_free (message);
497 g_free (detail);
498 }
499 }
500#endif
501 } else {
502 if (gw_fault_counter >= MAX_CONSECUTIVE_FAULTS) {
503 gw_fault_counter = 0;
504
505 /* Update dialog -- if one is present */
506 if (gw_applet->details_dialog)
507 mateweather_dialog_update (MATEWEATHER_DIALOG (gw_applet->details_dialog));
508
509 /* update applet */
510 place_widgets (gw_applet);
511 } else {
512 /* there has been an error during retrival
513 * just update the fault counter
514 */
515 gw_fault_counter++;
516 }
517 }
518}

527void mateweather_update (MateWeatherApplet *gw_applet)
528{
529 WeatherPrefs prefs;
530
531 gtk_widget_set_tooltip_text (GTK_WIDGET(gw_applet->applet), _("Updating..."));
532
533 /* Set preferred forecast type */
534 prefs.type = gw_applet->mateweather_pref.detailed ? FORECAST_ZONE : FORECAST_STATE;
535
536 /* Set radar map retrieval option */
537 prefs.radar = gw_applet->mateweather_pref.radar_enabled;
538 prefs.radar_custom_url = (gw_applet->mateweather_pref.use_custom_radar_url &&
539 gw_applet->mateweather_pref.radar) ?
540 gw_applet->mateweather_pref.radar : NULL;
541
542 /* Set the units */
543 prefs.temperature_unit = gw_applet->mateweather_pref.temperature_unit;
544 prefs.speed_unit = gw_applet->mateweather_pref.speed_unit;
545 prefs.pressure_unit = gw_applet->mateweather_pref.pressure_unit;
546 prefs.distance_unit = gw_applet->mateweather_pref.distance_unit;
547
548 /* Update current conditions */
549 if (gw_applet->mateweather_info &&
550 weather_location_equal(weather_info_get_location(gw_applet->mateweather_info),
551 gw_applet->mateweather_pref.location)) {
552 weather_info_update(gw_applet->mateweather_info, &prefs,
553 update_finish, gw_applet);
554 } else {
555 weather_info_free(gw_applet->mateweather_info);
556 gw_applet->mateweather_info = weather_info_new(gw_applet->mateweather_pref.location,
557 &prefs,
558 update_finish, gw_applet);
559 }
