151static void
152pluma_get_command_line_data (void)
153{
154 if (remaining_args)
155 {
156 gint i;
157
158 for (i = 0; remaining_args[i]; i++)
159 {
160 if (*remaining_args[i] == '+')
161 {
162 if (*(remaining_args[i] + 1) == '\0')
163 /* goto the last line of the document */
164 line_position = G_MAXINT;
165 else
166 line_position = atoi (remaining_args[i] + 1);
167 }
168 else
169 {
170 GFile *file;
171
172 file = g_file_new_for_commandline_arg (remaining_args[i]);
173 file_list = g_slist_prepend (file_list, file);
174 }
175 }
176
177 file_list = g_slist_reverse (file_list);
178 }
179
180 if (encoding_charset &&
181 (pluma_encoding_get_from_charset (encoding_charset) == NULL))
182 {
183 g_print (_("%s: invalid encoding.\n"),
184 encoding_charset);
185 }
186}

188static guint32
189get_startup_timestamp (void)
190{
191 const gchar *startup_id_env;
192 gchar *startup_id = NULL;
193 gchar *time_str;
194 gchar *end;
195 gulong retval = 0;
196
197 /* we don't unset the env, since startup-notification
198 * may still need it */
199 startup_id_env = g_getenv ("DESKTOP_STARTUP_ID");
200 if (startup_id_env == NULL)
201 goto out;
202
203 startup_id = g_strdup (startup_id_env);
204
205 time_str = g_strrstr (startup_id, "_TIME");
206 if (time_str == NULL)
207 goto out;
208
209 errno = 0;
210
211 /* Skip past the "_TIME" part */
212 time_str += 5;
213
214 retval = strtoul (time_str, &end, 0);
215 if (end == time_str || errno != 0)
216 retval = 0;
217
218 out:
219 g_free (startup_id);
220
221 return (retval > 0) ? retval : 0;
222}

224static GdkDisplay *
225display_open_if_needed (const gchar *name)
226{
227 GSList *displays;
228 GSList *l;
229 GdkDisplay *display = NULL;
230
231 displays = gdk_display_manager_list_displays (gdk_display_manager_get ());
232
233 for (l = displays; l != NULL; l = l->next)
234 {
235 if (strcmp (gdk_display_get_name ((GdkDisplay *) l->data), name) == 0)
236 {
237 display = l->data;
238 break;
239 }
240 }
241
242 g_slist_free (displays);
243
244 return display != NULL ? display : gdk_display_open (name);
245}

247/* serverside */
248static void
249on_message_received (const char *message,
250 gpointer data)
251{
252 const PlumaEncoding *encoding = NULL;
253 gchar **commands;
254 gchar **params;
255 gint workspace;
256 gint viewport_x;
257 gint viewport_y;
258 gchar *display_name;
259 gint i;
260 PlumaApp *app;
261 PlumaWindow *window;
262 GdkDisplay *display;
263 GdkScreen *screen;
264
265 g_return_if_fail (message != NULL);
266
267 pluma_debug_message (DEBUG_APP, "Received message:\n%s\n", message);
268
269 commands = g_strsplit (message, "\v", -1);
270
271 /* header */
272 params = g_strsplit (commands[0], "\t", 6);
273 startup_timestamp = atoi (params[0]);
274 display_name = params[1];
275 workspace = atoi (params[3]);
276 viewport_x = atoi (params[4]);
277 viewport_y = atoi (params[5]);
278
279 display = display_open_if_needed (display_name);
280 if (display == NULL)
281 {
282 g_warning ("Could not open display %s\n", display_name);
283 g_strfreev (params);
284 goto out;
285 }
286
287 screen = gdk_display_get_default_screen (display);
288
289 g_strfreev (params);
290
291 /* body */
292 for (i = 1; commands[i] != NULL; i++)
293 {
294 params = g_strsplit (commands[i], "\t", -1);
295
296 if (strcmp (params[0], "NEW-WINDOW") == 0)
297 {
298 new_window_option = TRUE;
299 }
300 else if (strcmp (params[0], "NEW-DOCUMENT") == 0)
301 {
302 new_document_option = TRUE;
303 }
304 else if (strcmp (params[0], "OPEN-URIS") == 0)
305 {
306 gint n_uris, j;
307 gchar **uris;
308
309 line_position = atoi (params[1]);
310
311 if (*params[2] != '\0')
312 encoding = pluma_encoding_get_from_charset (params[2]);
313
314 n_uris = atoi (params[3]);
315 uris = g_strsplit (params[4], " ", n_uris);
316
317 for (j = 0; j < n_uris; j++)
318 {
319 GFile *file;
320
321 file = g_file_new_for_uri (uris[j]);
322 file_list = g_slist_prepend (file_list, file);
323 }
324
325 file_list = g_slist_reverse (file_list);
326
327 /* the list takes ownerhip of the strings,
328 * only free the array */
329 g_free (uris);
330 }
331 else
332 {
333 g_warning ("Unexpected bacon command");
334 }
335
336 g_strfreev (params);
337 }
338
339 /* execute the commands */
340
341 app = pluma_app_get_default ();
342
343 if (new_window_option)
344 {
345 window = pluma_app_create_window (app, screen);
346 }
347 else
348 {
349 /* get a window in the current workspace (if exists) and raise it */
350 window = _pluma_app_get_window_in_viewport (app,
351 screen,
352 workspace,
353 viewport_x,
354 viewport_y);
355 }
356
357 if (file_list != NULL)
358 {
359 _pluma_cmd_load_files_from_prompt (window,
360 file_list,
361 encoding,
362 line_position);
363
364 if (new_document_option)
365 pluma_window_create_tab (window, TRUE);
366 }
367 else
368 {
369 PlumaDocument *doc;
370 doc = pluma_window_get_active_document (window);
371
372 if (doc == NULL ||
373 !pluma_document_is_untouched (doc) ||
374 new_document_option)
375 pluma_window_create_tab (window, TRUE);
376 }
377
378 /* set the proper interaction time on the window.
379 * Fall back to roundtripping to the X server when we
380 * don't have the timestamp, e.g. when launched from
381 * terminal. We also need to make sure that the window
382 * has been realized otherwise it will not work. lame.
383 */
384 if (!gtk_widget_get_realized (GTK_WIDGET (window)))
385 gtk_widget_realize (GTK_WIDGET (window));
386
387 if (GDK_IS_X11_DISPLAY (gdk_display_get_default ()))
388 {
389 if (startup_timestamp <= 0)
390 startup_timestamp = gdk_x11_get_server_time (gtk_widget_get_window (GTK_WIDGET (window)));
391
392 gdk_x11_window_set_user_time (gtk_widget_get_window (GTK_WIDGET (window)), startup_timestamp);
393 gtk_window_present (GTK_WINDOW (window));
394 }
395 else
396 gtk_window_present_with_time (GTK_WINDOW (window), GPOINTER_TO_UINT (data));
397
398 out:
399 g_strfreev (commands);
400
401 free_command_line_data ();
402}

404/* clientside */
405static void
406send_bacon_message (void)
407{
408 GdkScreen *screen;
409 GdkDisplay *display;
410 const gchar *display_name;
411 gint screen_number;
412 gint ws;
413 gint viewport_x;
414 gint viewport_y;
415 GString *command;
416
417 /* the messages have the following format:
418 * <--- header ---> <---- body ----->
419 * timestamp \t display_name \t screen_number \t workspace \t viewport_x \t viewport_y \v OP1 \t arg \t arg \v OP2 \t arg \t arg|...
420 *
421 * when the arg is a list of uri, they are separated by a space.
422 * So the delimiters are \v for the commands, \t for the tokens in
423 * a command and ' ' for the uris: note that such delimiters cannot
424 * be part of an uri, this way parsing is easier.
425 */
426
427 pluma_debug (DEBUG_APP);
428
429 screen = gdk_screen_get_default ();
430 display = gdk_screen_get_display (screen);
431
432 display_name = gdk_display_get_name (display);
433
434 if (GDK_IS_X11_DISPLAY (gdk_display_get_default ()))
435 screen_number = gdk_x11_screen_get_screen_number (screen);
436 else
437 screen_number = 0;
438
439 pluma_debug_message (DEBUG_APP, "Display: %s", display_name);
440 pluma_debug_message (DEBUG_APP, "Screen: %d", screen_number);
441
442 ws = pluma_utils_get_current_workspace (screen);
443 pluma_utils_get_current_viewport (screen, &viewport_x, &viewport_y);
444
445 command = g_string_new (NULL);
446
447 /* header */
448 g_string_append_printf (command,
449 "%" G_GUINT32_FORMAT "\t%s\t%d\t%d\t%d\t%d",
450 startup_timestamp,
451 display_name,
452 screen_number,
453 ws,
454 viewport_x,
455 viewport_y);
456
457 /* NEW-WINDOW command */
458 if (new_window_option)
459 {
460 command = g_string_append_c (command, '\v');
461 command = g_string_append (command, "NEW-WINDOW");
462 }
463
464 /* NEW-DOCUMENT command */
465 if (new_document_option)
466 {
467 command = g_string_append_c (command, '\v');
468 command = g_string_append (command, "NEW-DOCUMENT");
469 }
470
471 /* OPEN_URIS command, optionally specify line_num and encoding */
472 if (file_list)
473 {
474 GSList *l;
475
476 command = g_string_append_c (command, '\v');
477 command = g_string_append (command, "OPEN-URIS");
478
479 g_string_append_printf (command,
480 "\t%d\t%s\t%u\t",
481 line_position,
482 encoding_charset ? encoding_charset : "",
483 g_slist_length (file_list));
484
485 for (l = file_list; l != NULL; l = l->next)
486 {
487 gchar *uri;
488
489 uri = g_file_get_uri (G_FILE (l->data));
490 command = g_string_append (command, uri);
491 if (l->next != NULL)
492 command = g_string_append_c (command, ' ');
493
494 g_free (uri);
495 }
496 }
497
498 pluma_debug_message (DEBUG_APP, "Bacon Message: %s", command->str);
499
500 bacon_message_connection_send (connection,
501 command->str);
502
503 g_string_free (command, TRUE);
504}

506int
507main (int argc, char *argv[])
508{
509 GOptionContext *context;
510 PlumaApplication *application;
511 GError *error = NULL;
512 int status;
513
514 /* Setup debugging */
515 pluma_debug_init ();
516 pluma_debug_message (DEBUG_APP, "Startup");
517
518 setlocale (LC_ALL, "");
519
520 bindtextdomain (GETTEXT_PACKAGE, PLUMA_LOCALEDIR);
521 bind_textdomain_codeset (GETTEXT_PACKAGE, "UTF-8");
522 textdomain (GETTEXT_PACKAGE);
523
524 startup_timestamp = get_startup_timestamp();
525
526 /* Setup command line options */
527 context = g_option_context_new (_("- Edit text files"));
528 g_option_context_add_main_entries (context, options, GETTEXT_PACKAGE);
529 g_option_context_add_group (context, gtk_get_option_group (TRUE));
530 g_option_context_add_group (context, egg_sm_client_get_option_group ());
531
532#ifdef HAVE_INTROSPECTION
533#ifdef HAVE_GIREPOSITORY_2
534 g_option_context_add_group (context, gi_repository_get_option_group ());
535#else
536 g_option_context_add_group (context, g_irepository_get_option_group ());
537#endif
538#endif
539
540 if (!g_option_context_parse (context, &argc, &argv, &error))
541 {
542 g_print(_("%s\nRun '%s --help' to see a full list of available command line options.\n"),
543 error->message, argv[0]);
544 g_error_free (error);
545 g_option_context_free (context);
546 return 1;
547 }
548
549 g_option_context_free (context);
550
551 pluma_get_command_line_data ();
552
553 pluma_debug_message (DEBUG_APP, "Create bacon connection");
554
555 connection = bacon_message_connection_new ("pluma");
556
557 if (connection != NULL)
558 {
559 if (!bacon_message_connection_get_is_server (connection))
560 {
561 pluma_debug_message (DEBUG_APP, "I'm a client");
562
563 send_bacon_message ();
564
565 free_command_line_data ();
566
567 /* we never popup a window... tell startup-notification
568 * that we are done.
569 */
570 gdk_notify_startup_complete ();
571
572 bacon_message_connection_free (connection);
573
574 exit (EXIT_SUCCESS);
575 }
576 else
577 {
578 pluma_debug_message (DEBUG_APP, "I'm a server");
579
580 bacon_message_connection_set_callback (connection,
581 on_message_received,
582 NULL);
583 }
584 }
585 else
586 {
587 g_warning ("Cannot create the 'pluma' connection.");
588 }
589
590 /* Only the server reaches this point - now use GtkApplication */
591 application = pluma_application_new ();
592
593 /* Send parsed command line data to application */
594 pluma_application_set_command_line_options (application,
595 line_position,
596 encoding_charset,
597 new_window_option,
598 new_document_option,
599 file_list);
600
601 status = g_application_run (G_APPLICATION (application), argc, argv);
602 g_object_unref (application);
603
604 bacon_message_connection_free (connection);
605
606 return status;
607}
