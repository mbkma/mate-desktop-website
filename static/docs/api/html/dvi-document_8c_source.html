83static gboolean
84dvi_document_load (EvDocument *document,
85 const char *uri,
86 GError **error)
87{
88 gchar *filename;
89 DviDocument *dvi_document = DVI_DOCUMENT(document);
90
91 filename = g_filename_from_uri (uri, NULL, error);
92 if (!filename)
93 return FALSE;
94
95 g_mutex_lock (&dvi_context_mutex);
96 if (dvi_document->context)
97 mdvi_destroy_context (dvi_document->context);
98
99 dvi_document->context = mdvi_init_context(dvi_document->params, dvi_document->spec, filename);
100 g_mutex_unlock (&dvi_context_mutex);
101 g_free (filename);
102
103 if (!dvi_document->context) {
104 g_set_error_literal (error,
105 EV_DOCUMENT_ERROR,
106 EV_DOCUMENT_ERROR_INVALID,
107 _("DVI document has incorrect format"));
108 return FALSE;
109 }
110
111 mdvi_cairo_device_init (&dvi_document->context->device);
112
113 dvi_document->base_width = dvi_document->context->dvi_page_w * dvi_document->context->params.conv
114 + 2 * unit2pix(dvi_document->params->dpi, MDVI_HMARGIN) / dvi_document->params->hshrink;
115
116 dvi_document->base_height = dvi_document->context->dvi_page_h * dvi_document->context->params.vconv
117 + 2 * unit2pix(dvi_document->params->vdpi, MDVI_VMARGIN) / dvi_document->params->vshrink;
118
119 g_free (dvi_document->uri);
120 dvi_document->uri = g_strdup (uri);
121
122 return TRUE;
123}

155static cairo_surface_t *
156dvi_document_render (EvDocument *document,
157 EvRenderContext *rc)
158{
159 cairo_surface_t *surface;
160 cairo_surface_t *rotated_surface;
161 DviDocument *dvi_document = DVI_DOCUMENT(document);
162 gint required_width, required_height;
163 gint proposed_width, proposed_height;
164 gint xmargin = 0, ymargin = 0;
165
166 /* We should protect our context since it's not
167 * thread safe. The work to the future -
168 * let context render page independently
169 */
170 g_mutex_lock (&dvi_context_mutex);
171
172 mdvi_setpage (dvi_document->context, rc->page->index);
173
174 mdvi_set_shrink (dvi_document->context,
175 (int)((dvi_document->params->hshrink - 1) / rc->scale) + 1,
176 (int)((dvi_document->params->vshrink - 1) / rc->scale) + 1);
177
178 required_width = dvi_document->base_width * rc->scale + 0.5;
179 required_height = dvi_document->base_height * rc->scale + 0.5;
180 proposed_width = dvi_document->context->dvi_page_w * dvi_document->context->params.conv;
181 proposed_height = dvi_document->context->dvi_page_h * dvi_document->context->params.vconv;
182
183 if (required_width >= proposed_width)
184 xmargin = (required_width - proposed_width) / 2;
185 if (required_height >= proposed_height)
186 ymargin = (required_height - proposed_height) / 2;
187
188 mdvi_cairo_device_set_margins (&dvi_document->context->device, xmargin, ymargin);
189 mdvi_cairo_device_set_scale (&dvi_document->context->device, rc->scale);
190 mdvi_cairo_device_render (dvi_document->context);
191 surface = mdvi_cairo_device_get_surface (&dvi_document->context->device);
192
193 g_mutex_unlock (&dvi_context_mutex);
194
195 rotated_surface = ev_document_misc_surface_rotate_and_scale (surface,
196 required_width,
197 required_height,
198 rc->rotation);
199 cairo_surface_destroy (surface);
200
201 return rotated_surface;
202}

276static GdkPixbuf *
277dvi_document_thumbnails_get_thumbnail (EvDocumentThumbnails *document,
278 EvRenderContext *rc,
279 gboolean border)
280{
281 DviDocument *dvi_document = DVI_DOCUMENT (document);
282 GdkPixbuf *pixbuf;
283 GdkPixbuf *rotated_pixbuf;
284 cairo_surface_t *surface;
285 gint thumb_width, thumb_height;
286 gint proposed_width, proposed_height;
287
288 thumb_width = (gint) (dvi_document->base_width * rc->scale);
289 thumb_height = (gint) (dvi_document->base_height * rc->scale);
290
291 g_mutex_lock (&dvi_context_mutex);
292
293 mdvi_setpage (dvi_document->context, rc->page->index);
294
295 mdvi_set_shrink (dvi_document->context,
296 (int)dvi_document->base_width * dvi_document->params->hshrink / thumb_width,
297 (int)dvi_document->base_height * dvi_document->params->vshrink / thumb_height);
298
299 proposed_width = dvi_document->context->dvi_page_w * dvi_document->context->params.conv;
300 proposed_height = dvi_document->context->dvi_page_h * dvi_document->context->params.vconv;
301
302 if (border) {
303 mdvi_cairo_device_set_margins (&dvi_document->context->device,
304 MAX (thumb_width - proposed_width, 0) / 2,
305 MAX (thumb_height - proposed_height, 0) / 2);
306 } else {
307 mdvi_cairo_device_set_margins (&dvi_document->context->device,
308 MAX (thumb_width - proposed_width - 2, 0) / 2,
309 MAX (thumb_height - proposed_height - 2, 0) / 2);
310 }
311
312 mdvi_cairo_device_set_scale (&dvi_document->context->device, rc->scale);
313 mdvi_cairo_device_render (dvi_document->context);
314 surface = mdvi_cairo_device_get_surface (&dvi_document->context->device);
315 g_mutex_unlock (&dvi_context_mutex);
316
317 pixbuf = ev_document_misc_pixbuf_from_surface (surface);
318 cairo_surface_destroy (surface);
319
320 rotated_pixbuf = gdk_pixbuf_rotate_simple (pixbuf, 360 - rc->rotation);
321 g_object_unref (pixbuf);
322
323 if (border) {
324 GdkPixbuf *tmp_pixbuf = rotated_pixbuf;
325
326 rotated_pixbuf = ev_document_misc_get_thumbnail_frame (-1, -1, tmp_pixbuf);
327 g_object_unref (tmp_pixbuf);
328 }
329
330 return rotated_pixbuf;
331}

366static void
367dvi_document_file_exporter_end (EvFileExporter *exporter)
368{
369 gchar *command_line;
370 gint exit_stat;
371 GError *err = NULL;
372 gboolean success;
373
374 DviDocument *dvi_document = DVI_DOCUMENT(exporter);
375 gchar* quoted_filename = g_shell_quote (dvi_document->context->filename);
376
377 command_line = g_strdup_printf ("dvipdfm %s -o %s %s", /* dvipdfm -s 1,2,.., -o exporter_filename dvi_filename */
378 dvi_document->exporter_opts->str,
379 dvi_document->exporter_filename,
380 quoted_filename);
381 g_free (quoted_filename);
382
383 success = g_spawn_command_line_sync (command_line,
384 NULL,
385 NULL,
386 &exit_stat,
387 &err);
388
389 g_free (command_line);
390
391 if (success == FALSE) {
392 g_warning ("Error: %s", err->message);
393 } else if (!WIFEXITED(exit_stat) || WEXITSTATUS(exit_stat) != EXIT_SUCCESS){
394 g_warning ("Error: dvipdfm does not end normally or exit with a failure status.");
395 }
396
397 if (err)
398 g_error_free (err);
399}

420#define RGB2ULONG(r,g,b) ((0xFF<<24)|(r<<16)|(g<<8)|(b))

422static gboolean
423hsb2rgb (float h, float s, float v, guchar *red, guchar *green, guchar *blue)
424{
425 float f, p, q, t, r, g, b;
426 int i;
427
428 s /= 100;
429 v /= 100;
430 h /= 60;
431 i = floor (h);
432 if (i == 6)
433 i = 0;
434 else if ((i > 6) || (i < 0))
435 return FALSE;
436 f = h - i;
437 p = v * (1 - s);
438 q = v * (1 - (s * f));
439 t = v * (1 - (s * (1 - f)));
440
441 if (i == 0) {
442 r = v;
443 g = t;
444 b = p;
445 } else if (i == 1) {
446 r = q;
447 g = v;
448 b = p;
449 } else if (i == 2) {
450 r = p;
451 g = v;
452 b = t;
453 } else if (i == 3) {
454 r = p;
455 g = q;
456 b = v;
457 } else if (i == 4) {
458 r = t;
459 g = p;
460 b = v;
461 } else if (i == 5) {
462 r = v;
463 g = p;
464 b = q;
465 }
466
467 *red = (guchar)floor(r * 255.0);
468 *green = (guchar)floor(g * 255.0);
469 *blue = (guchar)floor(b * 255.0);
470
471 return TRUE;
472}

474static void
475parse_color (const gchar *ptr,
476 gdouble *color,
477 gint n_color)
478{
479 gchar *p = (gchar *)ptr;
480 gint i;
481
482 for (i = 0; i < n_color; i++) {
483 while (isspace (*p)) p++;
484 color[i] = g_ascii_strtod (p, NULL);
485 while (!isspace (*p) && *p != '\0') p++;
486 if (*p == '\0')
487 break;
488 }
489}

491static void
492dvi_document_do_color_special (DviContext *dvi, const char *prefix, const char *arg)
493{
494 if (strncmp (arg, "pop", 3) == 0) {
495 mdvi_pop_color (dvi);
496 } else if (strncmp (arg, "push", 4) == 0) {
497 /* Find color source: Named, CMYK or RGB */
498 const char *tmp = arg + 4;
499
500 while (isspace (*tmp)) tmp++;
501
502 if (!strncmp ("rgb", tmp, 3)) {
503 gdouble rgb[3];
504 guchar red, green, blue;
505
506 parse_color (tmp + 4, rgb, 3);
507
508 red = 255 * rgb[0];
509 green = 255 * rgb[1];
510 blue = 255 * rgb[2];
511
512 mdvi_push_color (dvi, RGB2ULONG (red, green, blue), 0xFFFFFFFF);
513 } else if (!strncmp ("hsb", tmp, 4)) {
514 gdouble hsb[3];
515 guchar red, green, blue;
516
517 parse_color (tmp + 4, hsb, 3);
518
519 if (hsb2rgb (hsb[0], hsb[1], hsb[2], &red, &green, &blue))
520 mdvi_push_color (dvi, RGB2ULONG (red, green, blue), 0xFFFFFFFF);
521 } else if (!strncmp ("cmyk", tmp, 4)) {
522 gdouble cmyk[4];
523 double r, g, b;
524 guchar red, green, blue;
525
526 parse_color (tmp + 5, cmyk, 4);
527
528 r = 1.0 - cmyk[0] - cmyk[3];
529 if (r < 0.0)
530 r = 0.0;
531 g = 1.0 - cmyk[1] - cmyk[3];
532 if (g < 0.0)
533 g = 0.0;
534 b = 1.0 - cmyk[2] - cmyk[3];
535 if (b < 0.0)
536 b = 0.0;
537
538 red = r * 255 + 0.5;
539 green = g * 255 + 0.5;
540 blue = b * 255 + 0.5;
541
542 mdvi_push_color (dvi, RGB2ULONG (red, green, blue), 0xFFFFFFFF);
543 } else if (!strncmp ("gray ", tmp, 5)) {
544 gdouble gray;
545 guchar rgb;
546
547 parse_color (tmp + 5, &gray, 1);
548
549 rgb = gray * 255 + 0.5;
550
551 mdvi_push_color (dvi, RGB2ULONG (rgb, rgb, rgb), 0xFFFFFFFF);
552 } else {
553 GdkColor color;
554
555 if (gdk_color_parse (tmp, &color)) {
556 guchar red, green, blue;
557
558 red = color.red * 255 / 65535.;
559 green = color.green * 255 / 65535.;
560 blue = color.blue * 255 / 65535.;
561
562 mdvi_push_color (dvi, RGB2ULONG (red, green, blue), 0xFFFFFFFF);
563 }
564 }
565 }
