88/*
89 * CappDesktopProvider is able to write if user data dir exists (or
90 * can be created) and is writable
91 *
92 * This is implementation of NAIIOProvider::is_able_to_write method
93 */
94gboolean
95cadp_iio_provider_is_able_to_write( const NAIIOProvider *provider )
96{
97 static const gchar *thisfn = "cadp_writer_iio_provider_is_able_to_write";
98 gboolean able_to;
99 gchar *userdir;
100
101 g_return_val_if_fail( CADP_IS_DESKTOP_PROVIDER( provider ), FALSE );
102
103 able_to = FALSE;
104
105 userdir = cadp_xdg_dirs_get_user_data_dir();
106
107 if( g_file_test( userdir, G_FILE_TEST_IS_DIR )){
108 able_to = na_core_utils_dir_is_writable_path( userdir );
109
110 } else if( g_mkdir_with_parents( userdir, 0750 )){
111 g_warning( "%s: %s: %s", thisfn, userdir, g_strerror( errno ));
112
113 } else {
114 na_core_utils_dir_list_perms( userdir, thisfn );
115 able_to = na_core_utils_dir_is_writable_path( userdir );
116 }
117
118 g_free( userdir );
119
120 return( able_to );
121}

123/*
124 * This is implementation of NAIIOProvider::write_item method
125 */
126guint
127cadp_iio_provider_write_item( const NAIIOProvider *provider, const NAObjectItem *item, GSList **messages )
128{
129 static const gchar *thisfn = "cadp_iio_provider_write_item";
130 guint ret;
131 CappDesktopFile *ndf;
132 gchar *path;
133 gchar *userdir;
134 gchar *id;
135 gchar *bname;
136 GSList *subdirs;
137 gchar *fulldir;
138 gboolean dir_ok;
139
140 ret = NA_IIO_PROVIDER_CODE_PROGRAM_ERROR;
141
142 g_return_val_if_fail( CADP_IS_DESKTOP_PROVIDER( provider ), ret );
143 g_return_val_if_fail( NA_IS_OBJECT_ITEM( item ), ret );
144
145 if( na_object_is_readonly( item )){
146 g_warning( "%s: item=%p is read-only", thisfn, ( void * ) item );
147 return( ret );
148 }
149
150 ndf = ( CappDesktopFile * ) na_object_get_provider_data( item );
151
152 /* write into the current key file and write it to current path */
153 if( ndf ){
154 g_return_val_if_fail( CADP_IS_DESKTOP_FILE( ndf ), ret );
155
156 } else {
157 userdir = cadp_xdg_dirs_get_user_data_dir();
158 subdirs = na_core_utils_slist_from_split( CADP_DESKTOP_PROVIDER_SUBDIRS, G_SEARCHPATH_SEPARATOR_S );
159 fulldir = g_build_filename( userdir, ( gchar * ) subdirs->data, NULL );
160 dir_ok = TRUE;
161
162 if( !g_file_test( fulldir, G_FILE_TEST_IS_DIR )){
163 if( g_mkdir_with_parents( fulldir, 0750 )){
164 g_warning( "%s: %s: %s", thisfn, userdir, g_strerror( errno ));
165 dir_ok = FALSE;
166 } else {
167 na_core_utils_dir_list_perms( userdir, thisfn );
168 }
169 }
170 g_free( userdir );
171 na_core_utils_slist_free( subdirs );
172
173 if( dir_ok ){
174 id = na_object_get_id( item );
175 bname = g_strdup_printf( "%s%s", id, CADP_DESKTOP_FILE_SUFFIX );
176 g_free( id );
177 path = g_build_filename( fulldir, bname, NULL );
178 g_free( bname );
179 }
180 g_free( fulldir );
181
182 if( dir_ok ){
183 ndf = cadp_desktop_file_new_for_write( path );
184 na_object_set_provider_data( item, ndf );
185 g_object_weak_ref( G_OBJECT( item ), ( GWeakNotify ) desktop_weak_notify, ndf );
186 g_free( path );
187 }
188 }
189
190 if( ndf ){
191 ret = write_item( provider, item, ndf, messages );
192 }
193
194 return( ret );
195}

197/*
198 * actually writes the item to the existing CappDesktopFile
199 * as we have chosen to take advantage of data factory management system
200 * we do not need to enumerate each and every elementary data
201 *
202 * As we want keep comments between through multiple updates, we cannot
203 * just delete the .desktop file and recreate it as we are doing for MateConf.
204 * Instead of that, we delete at end groups that have not been walked through
205 * -> as a side effect, we lose comments inside of these groups :(
206 */
207static guint
208write_item( const NAIIOProvider *provider, const NAObjectItem *item, CappDesktopFile *ndf, GSList **messages )
209{
210 static const gchar *thisfn = "cadp_iio_provider_write_item";
211 guint ret;
212 CappDesktopProvider *self;
213
214 g_debug( "%s: provider=%p (%s), item=%p (%s), ndf=%p, messages=%p",
215 thisfn,
216 ( void * ) provider, G_OBJECT_TYPE_NAME( provider ),
217 ( void * ) item, G_OBJECT_TYPE_NAME( item ),
218 ( void * ) ndf,
219 ( void * ) messages );
220
221 ret = NA_IIO_PROVIDER_CODE_PROGRAM_ERROR;
222
223 g_return_val_if_fail( NA_IS_IIO_PROVIDER( provider ), ret );
224 g_return_val_if_fail( CADP_IS_DESKTOP_PROVIDER( provider ), ret );
225 g_return_val_if_fail( NA_IS_IFACTORY_PROVIDER( provider ), ret );
226
227 g_return_val_if_fail( NA_IS_OBJECT_ITEM( item ), ret );
228 g_return_val_if_fail( NA_IS_IFACTORY_OBJECT( item ), ret );
229
230 g_return_val_if_fail( CADP_IS_DESKTOP_FILE( ndf ), ret );
231
232 self = CADP_DESKTOP_PROVIDER( provider );
233
234 if( self->private->dispose_has_run ){
235 return( NA_IIO_PROVIDER_CODE_NOT_WILLING_TO_RUN );
236 }
237
238 ret = NA_IIO_PROVIDER_CODE_OK;
239
240 na_ifactory_provider_write_item( NA_IFACTORY_PROVIDER( provider ), ndf, NA_IFACTORY_OBJECT( item ), messages );
241
242 if( !cadp_desktop_file_write( ndf )){
243 ret = NA_IIO_PROVIDER_CODE_WRITE_ERROR;
244 }
245
246 return( ret );
247}

354guint
355cadp_writer_iexporter_export_to_buffer( const NAIExporter *instance, NAIExporterBufferParmsv2 *parms )
356{
357 static const gchar *thisfn = "cadp_writer_iexporter_export_to_buffer";
358 guint code, write_code;
359 ExportFormatFn *fmt;
360 GKeyFile *key_file;
361 CappDesktopFile *ndf;
362
363 g_debug( "%s: instance=%p, parms=%p", thisfn, ( void * ) instance, ( void * ) parms );
364
365 parms->buffer = NULL;
366 code = NA_IEXPORTER_CODE_OK;
367
368 if( !parms->exported || !NA_IS_OBJECT_ITEM( parms->exported )){
369 code = NA_IEXPORTER_CODE_INVALID_ITEM;
370 }
371
372 if( code == NA_IEXPORTER_CODE_OK ){
373
374#ifdef NA_ENABLE_DEPRECATED
375 if( parms->version == 1 ){
376 fmt = find_export_format_fn_from_quark((( NAIExporterBufferParms * ) parms )->format );
377 } else {
378 fmt = find_export_format_fn( parms->format );
379 }
380#else
381 fmt = find_export_format_fn( parms->format );
382#endif
383
384 if( !fmt ){
385 code = NA_IEXPORTER_CODE_INVALID_FORMAT;
386
387 } else {
388 ndf = cadp_desktop_file_new();
389 write_code = na_ifactory_provider_write_item( NA_IFACTORY_PROVIDER( instance ), ndf, NA_IFACTORY_OBJECT( parms->exported ), &parms->messages );
390
391 if( write_code != NA_IIO_PROVIDER_CODE_OK ){
392 code = NA_IEXPORTER_CODE_ERROR;
393
394 } else {
395 key_file = cadp_desktop_file_get_key_file( ndf );
396 parms->buffer = g_key_file_to_data( key_file, NULL, NULL );
397 }
398
399 g_object_unref( ndf );
400 }
401 }
402
403 g_debug( "%s: returning code=%u", thisfn, code );
404 return( code );
405}

414guint
415cadp_writer_iexporter_export_to_file( const NAIExporter *instance, NAIExporterFileParmsv2 *parms )
416{
417 static const gchar *thisfn = "cadp_writer_iexporter_export_to_file";
418 guint code, write_code;
419 gchar *id, *folder_path, *dest_path;
420 ExportFormatFn *fmt;
421 CappDesktopFile *ndf;
422
423 g_debug( "%s: instance=%p, parms=%p", thisfn, ( void * ) instance, ( void * ) parms );
424
425 parms->basename = NULL;
426 code = NA_IEXPORTER_CODE_OK;
427
428 if( !parms->exported || !NA_IS_OBJECT_ITEM( parms->exported )){
429 code = NA_IEXPORTER_CODE_INVALID_ITEM;
430 }
431
432 if( code == NA_IEXPORTER_CODE_OK ){
433
434#ifdef NA_ENABLE_DEPRECATED
435 if( parms->version == 1 ){
436 fmt = find_export_format_fn_from_quark((( NAIExporterFileParms * ) parms )->format );
437 } else {
438 fmt = find_export_format_fn( parms->format );
439 }
440#else
441 fmt = find_export_format_fn( parms->format );
442#endif
443
444 if( !fmt ){
445 code = NA_IEXPORTER_CODE_INVALID_FORMAT;
446
447 } else {
448 id = na_object_get_id( parms->exported );
449 parms->basename = g_strdup_printf( "%s%s", id, CADP_DESKTOP_FILE_SUFFIX );
450 g_free( id );
451
452 folder_path = g_filename_from_uri( parms->folder, NULL, NULL );
453 dest_path = g_strdup_printf( "%s/%s", folder_path, parms->basename );
454 g_free( folder_path );
455
456 ndf = cadp_desktop_file_new_for_write( dest_path );
457 write_code = na_ifactory_provider_write_item( NA_IFACTORY_PROVIDER( instance ), ndf, NA_IFACTORY_OBJECT( parms->exported ), &parms->messages );
458
459 if( write_code != NA_IIO_PROVIDER_CODE_OK ){
460 code = NA_IEXPORTER_CODE_ERROR;
461
462 } else if( !cadp_desktop_file_write( ndf )){
463 code = NA_IEXPORTER_CODE_UNABLE_TO_WRITE;
464 }
465
466 g_free( dest_path );
467 g_object_unref( ndf );
468 }
469 }
470
471 g_debug( "%s: returning code=%u", thisfn, code );
472 return( code );
473}

496/*
497 * when writing to .desktop file a profile which has both a path and parameters,
498 * then concatenate these two fields to the 'Exec' key
499 */
500guint
501cadp_writer_ifactory_provider_write_data(
502 const NAIFactoryProvider *provider, void *writer_data, const NAIFactoryObject *object,
503 const NADataBoxed *boxed, GSList **messages )
504{
505 static const gchar *thisfn = "cadp_writer_ifactory_provider_write_data";
506 CappDesktopFile *ndf;
507 guint code;
508 const NADataDef *def;
509 gchar *profile_id;
510 gchar *group_name;
511 gchar *str_value;
512 gboolean bool_value;
513 GSList *slist_value;
514 guint uint_value;
515 gchar *parms, *tmp;
516
517 g_return_val_if_fail( CADP_IS_DESKTOP_FILE( writer_data ), NA_IIO_PROVIDER_CODE_PROGRAM_ERROR );
518 /*g_debug( "%s: object=%p (%s)", thisfn, ( void * ) object, G_OBJECT_TYPE_NAME( object ));*/
519
520 code = NA_IIO_PROVIDER_CODE_OK;
521 ndf = CADP_DESKTOP_FILE( writer_data );
522 def = na_data_boxed_get_data_def( boxed );
523
524 if( def->desktop_entry && strlen( def->desktop_entry )){
525
526 if( NA_IS_OBJECT_PROFILE( object )){
527 profile_id = na_object_get_id( object );
528 group_name = g_strdup_printf( "%s %s", CADP_GROUP_PROFILE, profile_id );
529 g_free( profile_id );
530
531 } else {
532 group_name = g_strdup( CADP_GROUP_DESKTOP );
533 }
534
535 if( !na_data_boxed_is_default( boxed ) || def->write_if_default ){
536
537 switch( def->type ){
538
539 case NA_DATA_TYPE_STRING:
540 str_value = na_boxed_get_string( NA_BOXED( boxed ));
541
542 if( !strcmp( def->name, NAFO_DATA_PATH )){
543 parms = na_object_get_parameters( object );
544 tmp = g_strdup_printf( "%s %s", str_value, parms );
545 g_free( str_value );
546 g_free( parms );
547 str_value = tmp;
548 }
549
550 cadp_desktop_file_set_string( ndf, group_name, def->desktop_entry, str_value );
551 g_free( str_value );
552 break;
553
554 case NA_DATA_TYPE_LOCALE_STRING:
555 str_value = na_boxed_get_string( NA_BOXED( boxed ));
556 cadp_desktop_file_set_locale_string( ndf, group_name, def->desktop_entry, str_value );
557 g_free( str_value );
558 break;
559
560 case NA_DATA_TYPE_BOOLEAN:
561 bool_value = GPOINTER_TO_UINT( na_boxed_get_as_void( NA_BOXED( boxed )));
562 cadp_desktop_file_set_boolean( ndf, group_name, def->desktop_entry, bool_value );
563 break;
564
565 case NA_DATA_TYPE_STRING_LIST:
566 slist_value = ( GSList * ) na_boxed_get_as_void( NA_BOXED( boxed ));
567 cadp_desktop_file_set_string_list( ndf, group_name, def->desktop_entry, slist_value );
568 na_core_utils_slist_free( slist_value );
569 break;
570
571 case NA_DATA_TYPE_UINT:
572 uint_value = GPOINTER_TO_UINT( na_boxed_get_as_void( NA_BOXED( boxed )));
573 cadp_desktop_file_set_uint( ndf, group_name, def->desktop_entry, uint_value );
574 break;
575
576 default:
577 g_warning( "%s: unknown type=%u for %s", thisfn, def->type, def->name );
578 code = NA_IIO_PROVIDER_CODE_PROGRAM_ERROR;
579 }
580
581 } else {
582 cadp_desktop_file_remove_key( ndf, group_name, def->desktop_entry );
583 }
584
585 g_free( group_name );
586 }
587
588 return( code );
589}

602static void
603write_done_write_subitems_list( CappDesktopFile *ndp, NAObjectItem *item )
604{
605 static const gchar *thisfn = "cadp_writer_write_done_write_subitems_list";
606 GSList *subitems;
607 GSList *profile_groups, *ip;
608 gchar *tmp;
609
610 subitems = na_object_get_items_slist( item );
611 tmp = g_strdup_printf( "%s (written subitems)", thisfn );
612 na_core_utils_slist_dump( tmp, subitems );
613 g_free( tmp );
614
615 cadp_desktop_file_set_string_list(
616 ndp,
617 CADP_GROUP_DESKTOP,
618 NA_IS_OBJECT_ACTION( item ) ? CADP_KEY_PROFILES : CADP_KEY_ITEMS_LIST,
619 subitems );
620
621 profile_groups = cadp_desktop_file_get_profiles( ndp );
622 tmp = g_strdup_printf( "%s (existing profiles)", thisfn );
623 na_core_utils_slist_dump( tmp, profile_groups );
624 g_free( tmp );
625
626 for( ip = profile_groups ; ip ; ip = ip->next ){
627 if( na_core_utils_slist_count( subitems, ( const gchar * ) ip->data ) == 0 ){
628 g_debug( "%s: deleting (removed) profile %s", thisfn, ( const gchar * ) ip->data );
629 cadp_desktop_file_remove_profile( ndp, ( const gchar * ) ip->data );
630 }
631 }
632
633 na_core_utils_slist_free( profile_groups );
634 na_core_utils_slist_free( subitems );
635}

637static ExportFormatFn *
638find_export_format_fn( const gchar *format )
639{
640 ExportFormatFn *found;
641 ExportFormatFn *i;
642
643 found = NULL;
644 i = st_export_format_fn;
645
646 while( i->format && !found ){
647 if( !strcmp( i->format, format )){
648 found = i;
649 }
650 i++;
651 }
652
653 return( found );
654}
