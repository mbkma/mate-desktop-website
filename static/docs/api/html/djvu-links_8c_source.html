63static guint64
64get_djvu_link_page (const DjvuDocument *djvu_document, const gchar *link_name, int base_page)
65{
66 guint64 page_num = 0;
67
68 /* #pagenum, #+pageoffset, #-pageoffset */
69 if (g_str_has_prefix (link_name, "#")) {
70 if (base_page > 0 && g_str_has_prefix (link_name+1, "+")) {
71 if (number_from_string_10 (link_name + 2, &page_num)) {
72 return base_page + page_num;
73 }
74 } else if (base_page > 0 && g_str_has_prefix (link_name+1, "-")) {
75 if (number_from_string_10 (link_name + 2, &page_num)) {
76 return base_page - page_num;
77 }
78 } else {
79 if (number_from_string_10 (link_name + 1, &page_num)) {
80 return page_num - 1;
81 }
82 }
83 } else {
84 /* FIXME: component file identifiers */
85 }
86
87 return page_num;
88}

116static gchar *
117str_to_utf8 (const gchar *text)
118{
119 static const gchar *encodings_to_try[2];
120 static gint n_encodings_to_try = 0;
121 gchar *utf8_text = NULL;
122 gint i;
123
124 if (n_encodings_to_try == 0) {
125 const gchar *charset;
126 gboolean charset_is_utf8;
127
128 charset_is_utf8 = g_get_charset (&charset);
129 if (!charset_is_utf8) {
130 encodings_to_try[n_encodings_to_try++] = charset;
131 }
132
133 if (g_ascii_strcasecmp (charset, "ISO-8859-1") != 0) {
134 encodings_to_try[n_encodings_to_try++] = "ISO-8859-1";
135 }
136 }
137
138 for (i = 0; i < n_encodings_to_try; i++) {
139 utf8_text = g_convert (text, -1, "UTF-8",
140 encodings_to_try[i],
141 NULL, NULL, NULL);
142 if (utf8_text)
143 break;
144 }
145
146 return utf8_text;
147}

161static void
162build_tree (const DjvuDocument *djvu_document,
163 GtkTreeModel *model,
164 GtkTreeIter *parent,
165 miniexp_t iter)
166{
167 const char *title, *link_dest;
168 char *title_markup;
169
170 EvLinkAction *ev_action = NULL;
171 EvLink *ev_link = NULL;
172 GtkTreeIter tree_iter;
173
174 if (miniexp_car (iter) == miniexp_symbol ("bookmarks")) {
175 /* The (bookmarks) cons */
176 iter = miniexp_cdr (iter);
177 } else if ( miniexp_length (iter) >= 2 ) {
178 gchar *utf8_title = NULL;
179
180 /* An entry */
181 if (!string_from_miniexp (miniexp_car (iter), &title)) goto unknown_entry;
182 if (!string_from_miniexp (miniexp_cadr (iter), &link_dest)) goto unknown_entry;
183
184 if (!g_utf8_validate (title, -1, NULL)) {
185 utf8_title = str_to_utf8 (title);
186 title_markup = g_markup_escape_text (utf8_title, -1);
187 } else {
188 title_markup = g_markup_escape_text (title, -1);
189 }
190
191 ev_action = get_djvu_link_action (djvu_document, link_dest, -1);
192
193 if (g_str_has_suffix (link_dest, ".djvu")) {
194 /* FIXME: component file identifiers */
195 } else if (ev_action) {
196 ev_link = ev_link_new (utf8_title ? utf8_title : title, ev_action);
197 gtk_tree_store_append (GTK_TREE_STORE (model), &tree_iter, parent);
198 gtk_tree_store_set (GTK_TREE_STORE (model), &tree_iter,
199 EV_DOCUMENT_LINKS_COLUMN_MARKUP, title_markup,
200 EV_DOCUMENT_LINKS_COLUMN_LINK, ev_link,
201 EV_DOCUMENT_LINKS_COLUMN_EXPAND, FALSE,
202 -1);
203 g_object_unref (ev_link);
204 } else {
205 gtk_tree_store_append (GTK_TREE_STORE (model), &tree_iter, parent);
206 gtk_tree_store_set (GTK_TREE_STORE (model), &tree_iter,
207 EV_DOCUMENT_LINKS_COLUMN_MARKUP, title_markup,
208 EV_DOCUMENT_LINKS_COLUMN_EXPAND, FALSE,
209 -1);
210 }
211
212 g_free (title_markup);
213 g_free (utf8_title);
214 iter = miniexp_cddr (iter);
215 parent = &tree_iter;
216 } else {
217 goto unknown_entry;
218 }
219
220 for (; iter != miniexp_nil; iter = miniexp_cdr (iter)) {
221 build_tree (djvu_document, model, parent, miniexp_car (iter));
222 }
223 return;
224
225 unknown_entry:
226 g_warning ("DjvuLibre error: Unknown entry in bookmarks");
227 return;
228}

230static gboolean
231get_djvu_hyperlink_area (ddjvu_pageinfo_t *page_info,
232 miniexp_t sexp,
233 EvMapping *ev_link_mapping)
234{
235 miniexp_t iter;
236
237 iter = sexp;
238
239 if ((miniexp_car (iter) == miniexp_symbol ("rect") || miniexp_car (iter) == miniexp_symbol ("oval"))
240 && miniexp_length (iter) == 5) {
241 /* FIXME: get bounding box for (oval) since Atril doesn't support shaped links */
242 int minx, miny, width, height;
243
244 iter = miniexp_cdr (iter);
245 if (!number_from_miniexp (miniexp_car (iter), &minx)) goto unknown_link;
246 iter = miniexp_cdr (iter);
247 if (!number_from_miniexp (miniexp_car (iter), &miny)) goto unknown_link;
248 iter = miniexp_cdr (iter);
249 if (!number_from_miniexp (miniexp_car (iter), &width)) goto unknown_link;
250 iter = miniexp_cdr (iter);
251 if (!number_from_miniexp (miniexp_car (iter), &height)) goto unknown_link;
252
253 ev_link_mapping->area.x1 = minx;
254 ev_link_mapping->area.x2 = (minx + width);
255 ev_link_mapping->area.y1 = (page_info->height - (miny + height));
256 ev_link_mapping->area.y2 = (page_info->height - miny);
257 } else if (miniexp_car (iter) == miniexp_symbol ("poly")
258 && miniexp_length (iter) >= 5 && miniexp_length (iter) % 2 == 1) {
259
260 /* FIXME: get bounding box since Atril doesn't support shaped links */
261 int minx = G_MAXINT, miny = G_MAXINT;
262 int maxx = G_MININT, maxy = G_MININT;
263
264 iter = miniexp_cdr(iter);
265 while (iter != miniexp_nil) {
266 int x, y;
267
268 if (!number_from_miniexp (miniexp_car(iter), &x)) goto unknown_link;
269 iter = miniexp_cdr (iter);
270 if (!number_from_miniexp (miniexp_car(iter), &y)) goto unknown_link;
271 iter = miniexp_cdr (iter);
272
273 minx = MIN (minx, x);
274 miny = MIN (miny, y);
275 maxx = MAX (maxx, x);
276 maxy = MAX (maxy, y);
277 }
278
279 ev_link_mapping->area.x1 = minx;
280 ev_link_mapping->area.x2 = maxx;
281 ev_link_mapping->area.y1 = (page_info->height - maxy);
282 ev_link_mapping->area.y2 = (page_info->height - miny);
283 } else {
284 /* unknown */
285 goto unknown_link;
286 }
287
288 return TRUE;
289
290 unknown_link:
291 g_warning("DjvuLibre error: Unknown hyperlink area %s", miniexp_to_name(miniexp_car(sexp)));
292 return FALSE;
293}

295static EvMapping *
296get_djvu_hyperlink_mapping (DjvuDocument *djvu_document,
297 int page,
298 ddjvu_pageinfo_t *page_info,
299 miniexp_t sexp)
300{
301 EvMapping *ev_link_mapping = NULL;
302 EvLinkAction *ev_action = NULL;
303 miniexp_t iter;
304 const char *url, *url_target, *comment;
305
306 ev_link_mapping = g_new (EvMapping, 1);
307
308 iter = sexp;
309
310 if (miniexp_car (iter) != miniexp_symbol ("maparea")) goto unknown_mapping;
311
312 iter = miniexp_cdr(iter);
313
314 if (miniexp_caar(iter) == miniexp_symbol("url")) {
315 if (!string_from_miniexp (miniexp_cadr (miniexp_car (iter)), &url)) goto unknown_mapping;
316 if (!string_from_miniexp (miniexp_caddr (miniexp_car (iter)), &url_target)) goto unknown_mapping;
317 } else {
318 if (!string_from_miniexp (miniexp_car(iter), &url)) goto unknown_mapping;
319 url_target = NULL;
320 }
321
322 iter = miniexp_cdr (iter);
323 if (!string_from_miniexp (miniexp_car(iter), &comment)) goto unknown_mapping;
324
325 iter = miniexp_cdr (iter);
326 if (!get_djvu_hyperlink_area (page_info, miniexp_car(iter), ev_link_mapping)) goto unknown_mapping;
327
328 iter = miniexp_cdr (iter);
329 /* FIXME: DjVu hyperlink attributes are ignored */
330
331 ev_action = get_djvu_link_action (djvu_document, url, page);
332 if (!ev_action) goto unknown_mapping;
333
334 ev_link_mapping->data = ev_link_new (comment, ev_action);
335
336 return ev_link_mapping;
337
338 unknown_mapping:
339 if (ev_link_mapping) g_free(ev_link_mapping);
340 g_warning("DjvuLibre error: Unknown hyperlink %s", miniexp_to_name(miniexp_car(sexp)));
341 return NULL;
342}

361EvMappingList *
362djvu_links_get_links (EvDocumentLinks *document_links,
363 gint page,
364 double scale_factor)
365{
366 DjvuDocument *djvu_document = DJVU_DOCUMENT (document_links);
367 GList *retval = NULL;
368 miniexp_t page_annotations = miniexp_nil;
369 miniexp_t *hyperlinks = NULL, *iter = NULL;
370 EvMapping *ev_link_mapping;
371 ddjvu_pageinfo_t page_info;
372
373 while ((page_annotations = ddjvu_document_get_pageanno (djvu_document->d_document, page)) == miniexp_dummy)
374 djvu_handle_events (djvu_document, TRUE, NULL);
375
376 while (ddjvu_document_get_pageinfo (djvu_document->d_document, page, &page_info) < DDJVU_JOB_OK)
377 djvu_handle_events(djvu_document, TRUE, NULL);
378
379 if (page_annotations) {
380 hyperlinks = ddjvu_anno_get_hyperlinks (page_annotations);
381 if (hyperlinks) {
382 for (iter = hyperlinks; *iter; ++iter) {
383 ev_link_mapping = get_djvu_hyperlink_mapping (djvu_document, page, &page_info, *iter);
384 if (ev_link_mapping) {
385 ev_link_mapping->area.x1 *= scale_factor;
386 ev_link_mapping->area.x2 *= scale_factor;
387 ev_link_mapping->area.y1 *= scale_factor;
388 ev_link_mapping->area.y2 *= scale_factor;
389 retval = g_list_prepend (retval, ev_link_mapping);
390 }
391 }
392 free (hyperlinks);
393 }
394 ddjvu_miniexp_release (djvu_document->d_document, page_annotations);
395 }
396
397 return ev_mapping_list_new (page, retval, (GDestroyNotify)g_object_unref);
398}
