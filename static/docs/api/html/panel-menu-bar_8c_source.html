98static void panel_menu_bar_setup_tooltip(PanelMenuBar* menubar)
99{
100 panel_util_set_tooltip_text(menubar->priv->applications_item, _("Browse and run installed applications"));
101 panel_util_set_tooltip_text(menubar->priv->places_item, _("Access documents, folders and network places"));
102 panel_util_set_tooltip_text(menubar->priv->desktop_item, _("Change desktop appearance and behavior, get help, or log out"));
103
104 /* FIXME: this doesn't handle the right-click case. Sigh. */
105 /* Hide tooltip if a menu is activated */
106 g_signal_connect(menubar->priv->applications_item, "activate", G_CALLBACK (panel_menu_bar_hide_tooltip_and_focus), menubar);
107 g_signal_connect(menubar->priv->places_item, "activate", G_CALLBACK (panel_menu_bar_hide_tooltip_and_focus), menubar);
108 g_signal_connect(menubar->priv->desktop_item, "activate", G_CALLBACK (panel_menu_bar_hide_tooltip_and_focus), menubar);
109
110 /* Reset tooltip when the menu bar is not used */
111 g_signal_connect(menubar, "deactivate", G_CALLBACK (panel_menu_bar_reinit_tooltip), menubar);

114static void
115panel_menu_bar_update_visibility (GSettings *settings,
116 gchar *key,
117 PanelMenuBar *menubar)
118{
119 if (!GTK_IS_WIDGET (menubar))
120 return;
121
122 gtk_widget_set_visible (GTK_WIDGET (menubar->priv->applications_item),
123 g_settings_get_boolean (settings, PANEL_MENU_BAR_SHOW_APPLICATIONS_KEY));
124 gtk_widget_set_visible (GTK_WIDGET (menubar->priv->places_item),
125 g_settings_get_boolean (settings, PANEL_MENU_BAR_SHOW_PLACES_KEY));
126 gtk_widget_set_visible (GTK_WIDGET (menubar->priv->desktop_item),
127 g_settings_get_boolean (settings, PANEL_MENU_BAR_SHOW_DESKTOP_KEY));
128
129 if (g_settings_get_boolean (settings, PANEL_MENU_BAR_SHOW_ICON_KEY)) {
130 GtkWidget *image;
131 gint icon_height;
132 gchar *str = g_settings_get_string (settings, PANEL_MENU_BAR_ICON_NAME_KEY);
133 GtkIconSize icon_size = panel_menu_bar_icon_get_size ();
134
135 gtk_icon_size_lookup (icon_size, NULL, &icon_height);
136 if (str != NULL && str[0] != 0)
137 image = gtk_image_new_from_icon_name(str, icon_size);
138 else
139 image = gtk_image_new_from_icon_name(PANEL_ICON_MAIN_MENU, icon_size);
140 mate_image_menu_item_set_image (MATE_IMAGE_MENU_ITEM (menubar->priv->applications_item), image);
141 gtk_image_set_pixel_size (GTK_IMAGE (image), icon_height);
142 g_free (str);
143 }
144 else
145 mate_image_menu_item_set_image (MATE_IMAGE_MENU_ITEM (menubar->priv->applications_item), NULL);
146}

148static void panel_menu_bar_init(PanelMenuBar* menubar)
149{
150 GtkCssProvider *provider;
151
152 menubar->priv = panel_menu_bar_get_instance_private(menubar);
153
154 provider = gtk_css_provider_new ();
155 gtk_css_provider_load_from_data (provider,
156 "PanelMenuBar {\n"
157 " border-width: 0px;\n"
158 "}",
159 -1, NULL);
160 gtk_style_context_add_provider (gtk_widget_get_style_context (GTK_WIDGET (menubar)),
161 GTK_STYLE_PROVIDER (provider),
162 GTK_STYLE_PROVIDER_PRIORITY_APPLICATION);
163 g_object_unref (provider);
164
165 menubar->priv->info = NULL;
166
167 menubar->priv->settings = g_settings_new (PANEL_MENU_BAR_SCHEMA);
168
169 menubar->priv->applications_menu = create_applications_menu("mate-applications.menu", NULL, TRUE);
170
171 menubar->priv->applications_item = panel_image_menu_item_new();
172 gtk_menu_item_set_label(GTK_MENU_ITEM(menubar->priv->applications_item), _("Applications"));
173
174 gtk_menu_item_set_submenu(GTK_MENU_ITEM(menubar->priv->applications_item), menubar->priv->applications_menu);
175 gtk_menu_shell_append(GTK_MENU_SHELL(menubar), menubar->priv->applications_item);
176
177 menubar->priv->places_item = panel_place_menu_item_new(FALSE);
178 gtk_menu_shell_append(GTK_MENU_SHELL(menubar), menubar->priv->places_item);
179
180 menubar->priv->desktop_item = panel_desktop_menu_item_new(FALSE, TRUE);
181 gtk_menu_shell_append(GTK_MENU_SHELL(menubar), menubar->priv->desktop_item);
182
183 panel_menu_bar_setup_tooltip(menubar);
184
185 panel_menu_bar_update_visibility(menubar->priv->settings, NULL, menubar);
186 g_signal_connect(menubar->priv->settings, "changed", G_CALLBACK (panel_menu_bar_update_visibility), menubar);
187
188 panel_menu_bar_update_text_gravity(menubar);
189 g_signal_connect(menubar, "screen-changed", G_CALLBACK(panel_menu_bar_update_text_gravity), NULL);
190}

257static void panel_menu_bar_size_allocate(GtkWidget* widget, GtkAllocation* allocation)
258{
259 GtkAllocation old_allocation;
260 GtkAllocation widget_allocation;
261 PanelBackground* background;
262
263 gtk_widget_get_allocation(widget, &widget_allocation);
264
265 old_allocation.x = widget_allocation.x;
266 old_allocation.y = widget_allocation.y;
267 old_allocation.width = widget_allocation.width;
268 old_allocation.height = widget_allocation.height;
269
270 GTK_WIDGET_CLASS(panel_menu_bar_parent_class)->size_allocate (widget, allocation);
271
272 if (old_allocation.x == allocation->x && old_allocation.y == allocation->y && old_allocation.width == allocation->width && old_allocation.height == allocation->height)
273 {
274 return;
275 }
276
277 background = &PANEL_MENU_BAR(widget)->priv->panel->toplevel->background;
278 if (background->type == PANEL_BACK_NONE || (background->type == PANEL_BACK_COLOR && !background->has_alpha))
279 {
280 return;
281 }
282
283 panel_menu_bar_change_background(PANEL_MENU_BAR(widget));
284}

314static gboolean panel_menu_bar_on_draw (GtkWidget* widget, cairo_t* cr, gpointer data)
315{
316 PanelMenuBar* menubar = data;
317
318 if (gtk_widget_has_focus(GTK_WIDGET(menubar))) {
319 GtkStyleContext *context;
320
321 context = gtk_widget_get_style_context (widget);
322 gtk_style_context_save (context);
323 gtk_style_context_set_state (context, gtk_widget_get_state_flags (widget));

325 cairo_save (cr);
326 gtk_render_focus (context, cr,

337static void panel_menu_bar_load(PanelWidget* panel, gboolean locked, int position, gboolean exactpos, const char* id)
338{
339 PanelMenuBar* menubar;
340 GtkSettings* settings;
341
342 g_return_if_fail (panel != NULL);
343
344 menubar = g_object_new(PANEL_TYPE_MENU_BAR, NULL);
345
346 menubar->priv->info = mate_panel_applet_register(GTK_WIDGET(menubar), NULL, NULL, panel, locked, position, exactpos, PANEL_OBJECT_MENU_BAR, id);
347
348 if (!menubar->priv->info)
349 {
350 gtk_widget_destroy(GTK_WIDGET(menubar));
351 return;
352 }
353
354 settings = gtk_settings_get_for_screen (gtk_widget_get_screen (GTK_WIDGET (panel)));
355 g_object_set (settings, "gtk-shell-shows-app-menu", FALSE, "gtk-shell-shows-menubar", FALSE, NULL);
356
357 mate_panel_applet_add_callback(menubar->priv->info, "help", "help-browser", _("_Help"), NULL);
358
359 /* Menu editors */
360 if (!panel_lockdown_get_locked_down () && (panel_is_program_in_path("mozo") || panel_is_program_in_path("menulibre")))
361 {
362 mate_panel_applet_add_callback (menubar->priv->info, "edit", "document-properties", _("_Edit Menus"), NULL);
363 }
364
365 g_signal_connect_after(menubar, "focus-in-event", G_CALLBACK(gtk_widget_queue_draw), menubar);
366 g_signal_connect_after(menubar, "focus-out-event", G_CALLBACK(gtk_widget_queue_draw), menubar);
367 g_signal_connect_after(menubar, "draw", G_CALLBACK(panel_menu_bar_on_draw), menubar);
368
369 gtk_widget_set_can_focus(GTK_WIDGET(menubar), TRUE);
370
371 panel_widget_set_applet_expandable(panel, GTK_WIDGET(menubar), FALSE, TRUE);
372 panel_menu_bar_update_visibility(menubar->priv->settings, NULL, menubar);
373}

469static void panel_menu_bar_update_orientation(PanelMenuBar* menubar)
470{
471 GtkPackDirection pack_direction;
472 double text_angle;
473 float text_xalign;
474 float text_yalign;
475
476 pack_direction = GTK_PACK_DIRECTION_LTR;
477 text_angle = 0.0;
478 text_xalign = 0.0;
479 text_yalign = 0.5;
480
481 switch (menubar->priv->orientation)
482 {
483 case PANEL_ORIENTATION_TOP:
484 case PANEL_ORIENTATION_BOTTOM:
485 break;
486 case PANEL_ORIENTATION_LEFT:
487 pack_direction = GTK_PACK_DIRECTION_BTT;
488 text_angle = 90.0;
489 text_xalign = 0.5;
490 text_yalign = 0.0;
491 break;
492 case PANEL_ORIENTATION_RIGHT:
493 pack_direction = GTK_PACK_DIRECTION_TTB;
494 text_angle = 270.0;
495 text_xalign = 0.5;
496 text_yalign = 0.0;
497 break;
498 default:
499 g_assert_not_reached();
500 break;
501 }
502
503 gtk_menu_bar_set_pack_direction(GTK_MENU_BAR(menubar), pack_direction);
504 gtk_menu_bar_set_child_pack_direction(GTK_MENU_BAR(menubar), pack_direction);
505
506 set_item_text_angle_and_alignment(menubar->priv->applications_item, text_angle, text_xalign, text_yalign);
507 set_item_text_angle_and_alignment(menubar->priv->places_item, text_angle, text_xalign, text_yalign);
508 set_item_text_angle_and_alignment(menubar->priv->desktop_item, text_angle, text_xalign, text_yalign);
509}
