55PulseSourceControl *
56pulse_source_control_new (PulseConnection *connection,
57 const pa_source_info *info,
58 PulseSource *parent)
59{
60 PulseSourceControl *control;
61 MateMixerStreamControlFlags flags = MATE_MIXER_STREAM_CONTROL_MUTE_READABLE |
62 MATE_MIXER_STREAM_CONTROL_MUTE_WRITABLE |
63 MATE_MIXER_STREAM_CONTROL_VOLUME_READABLE |
64 MATE_MIXER_STREAM_CONTROL_VOLUME_WRITABLE |
65 MATE_MIXER_STREAM_CONTROL_HAS_MONITOR;
66 MateMixerStreamControlRole role;
67
68 g_return_val_if_fail (PULSE_IS_CONNECTION (connection), NULL);
69 g_return_val_if_fail (info != NULL, NULL);
70 g_return_val_if_fail (PULSE_IS_SOURCE (parent), NULL);
71
72 if (info->active_port != NULL)
73 role = MATE_MIXER_STREAM_CONTROL_ROLE_PORT;
74 else
75 role = MATE_MIXER_STREAM_CONTROL_ROLE_MASTER;
76
77 /* Build the flag list */
78 if (info->flags & PA_SOURCE_DECIBEL_VOLUME)
79 flags |= MATE_MIXER_STREAM_CONTROL_HAS_DECIBEL;
80
81 control = g_object_new (PULSE_TYPE_SOURCE_CONTROL,
82 "name", info->name,
83 "label", info->description,
84 "flags", flags,
85 "role", role,
86 "stream", parent,
87 "connection", connection,
88 NULL);
89
90 pulse_source_control_update (control, info);
91 return control;
92}

94void
95pulse_source_control_update (PulseSourceControl *control, const pa_source_info *info)
96{
97 g_return_if_fail (PULSE_IS_SOURCE_CONTROL (control));
98 g_return_if_fail (info != NULL);
99
100 /* Let all the information update before emitting notify signals */
101 g_object_freeze_notify (G_OBJECT (control));
102
103 _mate_mixer_stream_control_set_mute (MATE_MIXER_STREAM_CONTROL (control),
104 info->mute ? TRUE : FALSE);
105
106 pulse_stream_control_set_channel_map (PULSE_STREAM_CONTROL (control),
107 &info->channel_map);
108
109 pulse_stream_control_set_cvolume (PULSE_STREAM_CONTROL (control),
110 &info->volume,
111 info->base_volume);
112
113 g_object_thaw_notify (G_OBJECT (control));
114}

137static PulseMonitor *
138pulse_source_control_create_monitor (PulseStreamControl *psc)
139{
140 guint32 index;
141
142 g_return_val_if_fail (PULSE_IS_SOURCE_CONTROL (psc), NULL);
143
144 index = pulse_stream_control_get_stream_index (psc);
145 if (G_UNLIKELY (index == PA_INVALID_INDEX)) {
146 g_debug ("Monitor of stream control %s is not available",
147 mate_mixer_stream_control_get_name (MATE_MIXER_STREAM_CONTROL (psc)));
148 return NULL;
149 }
150
151 return pulse_connection_create_monitor (pulse_stream_control_get_connection (psc),
152 index,
153 PA_INVALID_INDEX);
