47static gboolean
48 matekbd_desktop_config_get_lv_descriptions
49 (MatekbdDesktopConfig * config,
50 XklConfigRegistry * registry,
51 const gchar ** layout_ids,
52 const gchar ** variant_ids,
53 gchar *** short_layout_descriptions,
54 gchar *** long_layout_descriptions,
55 gchar *** short_variant_descriptions,
56 gchar *** long_variant_descriptions) {
57 const gchar **pl, **pv;
58 guint total_layouts;
59 gchar **sld, **lld, **svd, **lvd;
60 XklConfigItem *item = xkl_config_item_new ();
61
62 if (!
63 (xkl_engine_get_features (config->engine) &
64 XKLF_MULTIPLE_LAYOUTS_SUPPORTED))
65 return FALSE;
66
67 pl = layout_ids;
68 pv = variant_ids;
69 total_layouts = g_strv_length ((char **) layout_ids);
70 sld = *short_layout_descriptions =
71 g_new0 (gchar *, total_layouts + 1);
72 lld = *long_layout_descriptions =
73 g_new0 (gchar *, total_layouts + 1);
74 svd = *short_variant_descriptions =
75 g_new0 (gchar *, total_layouts + 1);
76 lvd = *long_variant_descriptions =
77 g_new0 (gchar *, total_layouts + 1);
78
79 while (pl != NULL && *pl != NULL) {
80
81 xkl_debug (100, "ids: [%s][%s]\n", *pl,
82 pv == NULL ? NULL : *pv);
83
84 g_snprintf (item->name, sizeof item->name, "%s", *pl);
85 if (xkl_config_registry_find_layout (registry, item)) {
86 *sld = g_strdup (item->short_description);
87 *lld = g_strdup (item->description);
88 } else {
89 *sld = g_strdup ("");
90 *lld = g_strdup ("");
91 }
92
93 if (pv != NULL && *pv != NULL) {
94 g_snprintf (item->name, sizeof item->name, "%s",
95 *pv);
96 if (xkl_config_registry_find_variant
97 (registry, *pl, item)) {
98 *svd = g_strdup (item->short_description);
99 *lvd = g_strdup (item->description);
100 } else {
101 *svd = g_strdup ("");
102 *lvd = g_strdup ("");
103 }
104 } else {
105 *svd = g_strdup ("");
106 *lvd = g_strdup ("");
107 }
108
109 xkl_debug (100, "description: [%s][%s][%s][%s]\n",
110 *sld, *lld, *svd, *lvd);
111 sld++;
112 lld++;
113 svd++;
114 lvd++;
115
116 pl++;
117
118 if (pv != NULL && *pv != NULL)
119 pv++;
120 }
121
122 g_object_unref (item);
123 return TRUE;
124}

145void
146matekbd_desktop_config_load_from_gsettings (MatekbdDesktopConfig * config)
147{
148 config->group_per_app =
149 g_settings_get_boolean (config->settings,
150 MATEKBD_DESKTOP_CONFIG_KEY_GROUP_PER_WINDOW);
151 xkl_debug (150, "group_per_app: %d\n", config->group_per_app);
152
153 config->handle_indicators =
154 g_settings_get_boolean (config->settings,
155 MATEKBD_DESKTOP_CONFIG_KEY_HANDLE_INDICATORS);
156 xkl_debug (150, "handle_indicators: %d\n",
157 config->handle_indicators);
158
159 config->layout_names_as_group_names =
160 g_settings_get_boolean (config->settings,
161 MATEKBD_DESKTOP_CONFIG_KEY_LAYOUT_NAMES_AS_GROUP_NAMES);
162 xkl_debug (150, "layout_names_as_group_names: %d\n",
163 config->layout_names_as_group_names);
164
165 config->load_extra_items =
166 g_settings_get_boolean (config->settings,
167 MATEKBD_DESKTOP_CONFIG_KEY_LOAD_EXTRA_ITEMS);
168 xkl_debug (150, "load_extra_items: %d\n",
169 config->load_extra_items);
170
171 config->default_group =
172 g_settings_get_int (config->settings,
173 MATEKBD_DESKTOP_CONFIG_KEY_DEFAULT_GROUP);
174
175 if (config->default_group < -1
176 || config->default_group >=
177 xkl_engine_get_max_num_groups (config->engine))
178 config->default_group = -1;
179 xkl_debug (150, "default_group: %d\n", config->default_group);
180}

182void
183matekbd_desktop_config_save_to_gsettings (MatekbdDesktopConfig * config)
184{
185 g_settings_delay (config->settings);
186
187 g_settings_set_boolean (config->settings,
188 MATEKBD_DESKTOP_CONFIG_KEY_GROUP_PER_WINDOW,
189 config->group_per_app);
190 g_settings_set_boolean (config->settings,
191 MATEKBD_DESKTOP_CONFIG_KEY_HANDLE_INDICATORS,
192 config->handle_indicators);
193 g_settings_set_boolean (config->settings,
194 MATEKBD_DESKTOP_CONFIG_KEY_LAYOUT_NAMES_AS_GROUP_NAMES,
195 config->layout_names_as_group_names);
196 g_settings_set_boolean (config->settings,
197 MATEKBD_DESKTOP_CONFIG_KEY_LOAD_EXTRA_ITEMS,
198 config->load_extra_items);
199 g_settings_set_int (config->settings,
200 MATEKBD_DESKTOP_CONFIG_KEY_DEFAULT_GROUP,
201 config->default_group);
202
203 g_settings_apply (config->settings);
204}

271gboolean
272matekbd_desktop_config_load_group_descriptions (MatekbdDesktopConfig
273 * config,
274 XklConfigRegistry *
275 registry,
276 const gchar **
277 layout_ids,
278 const gchar **
279 variant_ids,
280 gchar ***
281 short_group_names,
282 gchar *** full_group_names)
283{
284 gchar **sld, **lld, **svd, **lvd;
285 gchar **psld, **plld, **plvd;
286 gchar **psgn, **pfgn, **psvd;
287 gint total_descriptions;
288
289 if (!matekbd_desktop_config_get_lv_descriptions
290 (config, registry, layout_ids, variant_ids, &sld, &lld, &svd,
291 &lvd)) {
292 return False;
293 }
294
295 total_descriptions = g_strv_length (sld);
296
297 *short_group_names = psgn =
298 g_new0 (gchar *, total_descriptions + 1);
299 *full_group_names = pfgn =
300 g_new0 (gchar *, total_descriptions + 1);
301
302 plld = lld;
303 psld = sld;
304 plvd = lvd;
305 psvd = svd;
306 while (plld != NULL && *plld != NULL) {
307 gchar *sd = (*psvd[0] == '\0') ? *psld : *psvd;
308 psld++, psvd++;
309 *psgn++ = g_strdup (sd);
310 *pfgn++ = g_strdup (matekbd_keyboard_config_format_full_layout
311 (*plld++, *plvd++));
312 }
313 g_strfreev (sld);
314 g_strfreev (lld);
315 g_strfreev (svd);
316 g_strfreev (lvd);
317
318 return True;
319}
