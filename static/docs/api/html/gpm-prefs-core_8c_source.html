213static void
214gpm_prefs_setup_action_combo (GpmPrefs *prefs, const gchar *widget_name,
215 const gchar *gpm_pref_key, const GpmActionPolicy *actions)
216{
217 guint i;
218 gboolean is_writable;
219 GtkWidget *widget;
220 GpmActionPolicy policy;
221 GpmActionPolicy value;
222 GPtrArray *array;
223 GpmActionPolicy *actions_added;
224
225 widget = GET_WIDGET (widget_name);
226
227 value = g_settings_get_enum (prefs->priv->settings, gpm_pref_key);
228 is_writable = g_settings_is_writable (prefs->priv->settings, gpm_pref_key);
229
230 gtk_widget_set_sensitive (widget, is_writable);
231
232 array = g_ptr_array_new ();
233 g_object_set_data (G_OBJECT (widget), "settings_key", (gpointer) gpm_pref_key);
234 g_signal_connect (G_OBJECT (widget), "changed",
235 G_CALLBACK (gpm_prefs_action_combo_changed_cb), prefs);
236
237 for (i=0; (gint)actions[i] != -1; i++) {
238 policy = actions[i];
239 if (policy == GPM_ACTION_POLICY_SHUTDOWN && !prefs->priv->can_shutdown) {
240 g_debug ("Cannot add option, as cannot shutdown.");
241 } else if (policy == GPM_ACTION_POLICY_SHUTDOWN && prefs->priv->can_shutdown) {
242 gtk_combo_box_text_append_text(GTK_COMBO_BOX_TEXT (widget), _("Shutdown"));
243 g_ptr_array_add(array, GINT_TO_POINTER (policy));
244 } else if (policy == GPM_ACTION_POLICY_SUSPEND && !prefs->priv->can_suspend) {
245 g_debug ("Cannot add option, as cannot suspend.");
246 } else if (policy == GPM_ACTION_POLICY_HIBERNATE && !prefs->priv->can_hibernate) {
247 g_debug ("Cannot add option, as cannot hibernate.");
248 } else if (policy == GPM_ACTION_POLICY_SUSPEND && prefs->priv->can_suspend) {
249 gtk_combo_box_text_append_text(GTK_COMBO_BOX_TEXT (widget), _("Suspend"));
250 g_ptr_array_add (array, GINT_TO_POINTER (policy));
251 } else if (policy == GPM_ACTION_POLICY_HIBERNATE && prefs->priv->can_hibernate) {
252 gtk_combo_box_text_append_text(GTK_COMBO_BOX_TEXT (widget), _("Hibernate"));
253 g_ptr_array_add(array, GINT_TO_POINTER (policy));
254 } else if (policy == GPM_ACTION_POLICY_BLANK) {
255 gtk_combo_box_text_append_text(GTK_COMBO_BOX_TEXT (widget), _("Blank screen"));
256 g_ptr_array_add (array, GINT_TO_POINTER (policy));
257 } else if (policy == GPM_ACTION_POLICY_INTERACTIVE) {
258 gtk_combo_box_text_append_text(GTK_COMBO_BOX_TEXT (widget), _("Ask me"));
259 g_ptr_array_add(array, GINT_TO_POINTER (policy));
260 } else if (policy == GPM_ACTION_POLICY_NOTHING) {
261 gtk_combo_box_text_append_text(GTK_COMBO_BOX_TEXT (widget), _("Do nothing"));
262 g_ptr_array_add(array, GINT_TO_POINTER (policy));
263 } else {
264 g_warning ("Unknown action read from settings: %u", policy);
265 }
266 }
267
268 /* save as array _only_ the actions we could add */
269 actions_added = g_new0 (GpmActionPolicy, array->len+1);
270 for (i=0; i<array->len; i++)
271 actions_added[i] = GPOINTER_TO_INT (g_ptr_array_index (array, i));
272 actions_added[i] = -1;
273
274 g_object_set_data_full (G_OBJECT (widget), "actions", (gpointer) actions_added, (GDestroyNotify) gpm_prefs_actions_destroy_cb);
275
276 /* set what we have in the settings */
277 for (i=0; (gint)actions_added[i] != -1; i++) {
278 policy = actions_added[i];
279 if (value == policy)
280 gtk_combo_box_set_active (GTK_COMBO_BOX (widget), i);
281 }
282
283 g_ptr_array_unref (array);
284}

293static void
294gpm_prefs_setup_time_combo (GpmPrefs *prefs, const gchar *widget_name,
295 const gchar *gpm_pref_key, const gint *values)
296{
297 guint value;
298 gchar *text;
299 guint i;
300 gboolean is_writable;
301 GtkWidget *widget;
302
303 widget = GET_WIDGET (widget_name);
304
305 value = g_settings_get_int (prefs->priv->settings, gpm_pref_key);
306 is_writable = g_settings_is_writable (prefs->priv->settings, gpm_pref_key);
307 gtk_widget_set_sensitive (widget, is_writable);
308
309 g_object_set_data (G_OBJECT (widget), "settings_key", (gpointer) gpm_pref_key);
310 g_object_set_data (G_OBJECT (widget), "values", (gpointer) values);
311
312 /* add each time */
313 for (i=0; values[i] != -1; i++) {
314
315 /* get translation for number of seconds */
316 if (values[i] != 0) {
317 text = gpm_get_timestring (values[i]);
318 gtk_combo_box_text_append_text(GTK_COMBO_BOX_TEXT (widget), text);
319 g_free (text);
320 } else {
321 gtk_combo_box_text_append_text(GTK_COMBO_BOX_TEXT (widget), _("Never"));
322 }
323
324 /* matches, so set default */
325 if ((gint)value == values[i])
326 gtk_combo_box_set_active (GTK_COMBO_BOX (widget), i);
327 }
328
329 /* connect after set */
330 g_signal_connect (G_OBJECT (widget), "changed",
331 G_CALLBACK (gpm_prefs_action_time_changed_cb), prefs);
332}

582static void
583gpm_prefs_init (GpmPrefs *prefs)
584{
585 GError *error = NULL;
586 GPtrArray *devices = NULL;
587 UpDevice *device;
588 UpDeviceKind kind;
589 GpmBrightness *brightness;
590 gint icon_policy;
592 GtkWidget *radiobutton_icon_always;
593 GtkWidget *radiobutton_icon_present;
594 GtkWidget *radiobutton_icon_charge;
595 GtkWidget *radiobutton_icon_low;
596 GtkWidget *radiobutton_icon_never;
597
598 guint i;
599
600 GDBusProxy *proxy;
601 GVariant *res, *inner;
602 gchar * r;
603
604 prefs->priv = gpm_prefs_get_instance_private (prefs);
605
606 prefs->priv->client = up_client_new ();
607 prefs->priv->console = egg_console_kit_new ();
608 prefs->priv->settings = g_settings_new (GPM_SETTINGS_SCHEMA);
609
610 prefs->priv->can_shutdown = FALSE;
611 prefs->priv->can_suspend = FALSE;
612 prefs->priv->can_hibernate = FALSE;
613
614 if (LOGIND_RUNNING()) {
615 /* get values from logind */
616
617 proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SYSTEM,
618 G_DBUS_PROXY_FLAGS_DO_NOT_LOAD_PROPERTIES,
619 NULL,
620 "org.freedesktop.login1",
621 "/org/freedesktop/login1",
622 "org.freedesktop.login1.Manager",
623 NULL,
624 &error );
625 if (proxy == NULL) {
626 g_error ("Error connecting to dbus - %s", error->message);
627 g_error_free (error);
628 return;
629 }
630
631 res = g_dbus_proxy_call_sync (proxy, "CanPowerOff",
632 NULL,
633 G_DBUS_CALL_FLAGS_NONE,
634 -1,
635 NULL,
636 &error
637 );
638 if (error == NULL && res != NULL) {
639 g_variant_get(res,"(&s)", &r);
640 prefs->priv->can_shutdown = g_strcmp0(r,"yes")==0?TRUE:FALSE;
641 g_variant_unref (res);
642 } else if (error != NULL ) {
643 g_error ("Error in dbus - %s", error->message);
644 g_error_free (error);
645 }
646
647 res = g_dbus_proxy_call_sync (proxy, "CanSuspend",
648 NULL,
649 G_DBUS_CALL_FLAGS_NONE,
650 -1,
651 NULL,
652 &error
653 );
654 if (error == NULL && res != NULL) {
655 g_variant_get(res,"(&s)", &r);
656 prefs->priv->can_suspend = g_strcmp0(r,"yes")==0?TRUE:FALSE;
657 g_variant_unref (res);
658 } else if (error != NULL ) {
659 g_error ("Error in dbus - %s", error->message);
660 g_error_free (error);
661 }
662
663 res = g_dbus_proxy_call_sync (proxy, "CanHibernate",
664 NULL,
665 G_DBUS_CALL_FLAGS_NONE,
666 -1,
667 NULL,
668 &error
669 );
670 if (error == NULL && res != NULL) {
671 g_variant_get(res,"(&s)", &r);
672 prefs->priv->can_hibernate = g_strcmp0(r,"yes")==0?TRUE:FALSE;
673 g_variant_unref (res);
674 } else if (error != NULL ) {
675 g_error ("Error in dbus - %s", error->message);
676 g_error_free (error);
677 }
678 g_object_unref(proxy);
679 }
680 else {
681 /* Get values from ConsoleKit */
682 egg_console_kit_can_stop (prefs->priv->console, &prefs->priv->can_shutdown, NULL);
683 egg_console_kit_can_suspend (prefs->priv->console, &prefs->priv->can_suspend, NULL);
