43/* Sticky Window Callback : Resize the window. */
44gboolean stickynote_resize_cb (GtkWidget *widget,
45 GdkEventButton *event,
46 StickyNote *note)
47{
48 if (event->type == GDK_BUTTON_PRESS && event->button == 1) {
49 if (widget == note->w_resize_se)
50 gtk_window_begin_resize_drag (GTK_WINDOW (note->w_window),
51 GDK_WINDOW_EDGE_SOUTH_EAST,
52 event->button,
53 (gint) event->x_root,
54 (gint) event->y_root,
55 event->time);
56 else /* if (widget == note->w_resize_sw) */
57 gtk_window_begin_resize_drag (GTK_WINDOW (note->w_window),
58 GDK_WINDOW_EDGE_SOUTH_WEST,
59 event->button,
60 (gint) event->x_root,
61 (gint) event->y_root,
62 event->time);
63 }
64 else
65 return FALSE;
66
67 return TRUE;
68}

72stickynote_move_cb (GtkWidget *widget,

75{

77 gtk_window_begin_move_drag (GTK_WINDOW (note->w_window),
78 event->button, event->x_root,
79 event->y_root, event->time);
80 else if (event->type == GDK_2BUTTON_PRESS && event->button == 1)
81 stickynote_change_properties (note);
82 else
83 return FALSE;

129/* Sticky Window Callback : Exit entry field on key press or popup
130 * the right click menu. */
131gboolean
132stickynote_keypress_cb (GtkWidget *widget,
133 GdkEventKey *event,
134 GtkMenu *popup_menu)
135{
136 GdkModifierType state = event->state & gtk_accelerator_get_default_mod_mask ();
137
138 switch (event->keyval) {
139 case GDK_KEY_F6:
140 if (state == 0)
141 gtk_widget_child_focus(widget, GTK_DIR_TAB_FORWARD);
142 else if (state == GDK_SHIFT_MASK)
143 gtk_widget_child_focus(widget, GTK_DIR_TAB_BACKWARD);
144 else
145 break;
146 return TRUE;
147 default:
148 break;
149 }
150
151 if ((event->keyval == GDK_KEY_F10 && (state == 0 || state == GDK_SHIFT_MASK)) ||
152 (event->keyval == GDK_KEY_Menu && state == 0)) {
153 GtkWidget *focus_widget = NULL;
154
155 if (GTK_IS_WINDOW (widget))
156 focus_widget = gtk_window_get_focus (GTK_WINDOW (widget));
157 if (! focus_widget)
158 focus_widget = widget;
159
160 gtk_menu_popup_at_widget (popup_menu, focus_widget,
161 GDK_GRAVITY_SOUTH_WEST,
162 GDK_GRAVITY_NORTH_WEST,
163 (const GdkEvent*) event);
164 return TRUE;
165 }
166
167 return FALSE;
168}

210/* Properties Dialog Callback : Apply color */
211void
212properties_apply_color_cb (StickyNote *note)
213{
214 char *color_str = NULL;
215 char *font_color_str = NULL;
216
217 if (!gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (note->w_def_color))) {
218 GdkRGBA color, font_color;
219
220 gtk_color_chooser_get_rgba (GTK_COLOR_CHOOSER (note->w_color),
221 &color);
222 gtk_color_chooser_get_rgba (GTK_COLOR_CHOOSER (note->w_font_color),
223 &font_color);
224
225 color_str = gdk_rgba_to_string (&color);
226 font_color_str = gdk_rgba_to_string (&font_color);
227 }
228
229 stickynote_set_color (note, color_str,
230 font_color_str, TRUE);
231
232 g_free (color_str);
233 g_free (font_color_str);
234}
