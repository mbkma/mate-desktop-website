47static char *
48get_target_uri (GFile *file)
49{
50 GFileInfo *info;
51 char *target;
52
53 info = g_file_query_info (file, G_FILE_ATTRIBUTE_STANDARD_TARGET_URI, G_FILE_QUERY_INFO_NONE, NULL, NULL);
54 if (info == NULL)
55 return NULL;
56 target = g_strdup (g_file_info_get_attribute_string (info, G_FILE_ATTRIBUTE_STANDARD_TARGET_URI));
57 g_object_unref (info);
58
59 return target;
60}

62static char *
63get_target_path (GFile *input)
64{
65 if (g_file_has_uri_scheme (input, "trash") != FALSE ||
66 g_file_has_uri_scheme (input, "recent") != FALSE) {
67 GFile *file;
68 char *input_uri;
69 char *input_path;
70
71 input_uri = get_target_uri (input);
72 file = g_file_new_for_uri (input_uri);
73 g_free (input_uri);
74 input_path = g_file_get_path (file);
75 g_object_unref (file);
76 return input_path;
77 }
78 return g_file_get_path (input);
79}

88int main (int argc, char **argv)
89{
90 MagickBooleanType status;
91 MagickWand *magick_wand;
92 char *input_filename;
93 GError *error = NULL;
94 GOptionContext *context;
95 GFile *input;
96 const char *output;

99 /* Options parsing */
100 context = g_option_context_new ("- thumbnail images");
101 g_option_context_add_main_entries (context, entries, NULL);
102
103 (void) g_option_context_parse (context, &argc, &argv, &error);
104 g_option_context_free (context);
105 if (error) {
106 g_warning ("Couldn't parse command-line options: %s", error->message);
107 g_error_free (error);
108 goto arguments_error;
109 }
110
111 /* Set fatal warnings if required */
112 if (g_fatal_warnings) {
113 GLogLevelFlags fatal_mask;
114
115 fatal_mask = g_log_set_always_fatal (G_LOG_FATAL_MASK);
116 fatal_mask |= G_LOG_LEVEL_WARNING | G_LOG_LEVEL_CRITICAL;
117 g_log_set_always_fatal (fatal_mask);
118 }
119
120 if (filenames == NULL || g_strv_length (filenames) != 2) {
121 g_print ("Expects an input and an output file\n");
122 goto arguments_error;
123 }
124
125 if (output_size < 1) {
126 g_warning ("Size cannot be smaller than 1 pixel");
127 goto arguments_error;
128 }
129
130 input = g_file_new_for_commandline_arg (filenames[0]);
131 input_filename = get_target_path (input);
132 g_object_unref (input);
133 if (input_filename == NULL) {
134 g_warning ("Could not get file path for %s", filenames[0]);
135 goto arguments_error;
136 }
137
138 output = filenames[1];
139
140 /* Read an image */
141 MagickWandGenesis ();
142 magick_wand = NewMagickWand ();
143 status = MagickReadImage (magick_wand, input_filename);
144 g_free (input_filename);
145 if (status == MagickFalse) {
146 g_warning ("Could not load input file %s", filenames[0]);
147 goto imagemagick_error;
148 }
149
150 /* Get the image's width and height */
151 width = MagickGetImageWidth (magick_wand);
152 height = MagickGetImageHeight (magick_wand);
153
154 /* Thumbnail */
155 if ((height > output_size) || (width > output_size)) {
156 double scale;
157 scale = (double) output_size / MAX (width, height);
158 MagickThumbnailImage (magick_wand,
159 (size_t) floor (width * scale + 0.5),
160 (size_t) floor (height * scale + 0.5));
161 }
162
163 /* Write the image then destroy it */
164 status = MagickWriteImages (magick_wand, output, MagickTrue);
165 if (status == MagickFalse) {
166 g_warning ("Could not save output file %s", output);
167 goto imagemagick_error;
168 }
169 g_strfreev (filenames);
170 if (magick_wand)
171 DestroyMagickWand (magick_wand);
172 MagickWandTerminus ();
173
174 return 0;
175
176arguments_error:
177 if (filenames)
178 g_strfreev (filenames);
179 return 1;
180
181imagemagick_error:
182 g_strfreev (filenames);
183 if (magick_wand)
184 DestroyMagickWand (magick_wand);
185 return 1;
186}
