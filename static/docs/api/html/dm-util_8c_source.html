25static GDBusProxy *get_sys_proxy (void)
26{
27 GError *error = NULL;
28 GDBusProxy *proxy = NULL;
29
30 proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SYSTEM,
31 G_DBUS_PROXY_FLAGS_NONE,
32 NULL,
33 "org.freedesktop.DBus",
34 "/org/freedesktop/DBus",
35 "org.freedesktop.DBus",
36 NULL,
37 &error);
38 if (proxy == NULL) {
39 g_warning ("Couldn't connect to system bus: %s", error->message);
40 g_error_free (error);
41 }
42 return proxy;
43}

45static gboolean dm_is_running (void)
46{
47 GDBusProxy *proxy;
48 GError *error = NULL;
49 GVariant *ret;
50 gboolean running = FALSE;
51
52 proxy = get_sys_proxy ();
53 if (proxy == NULL)
54 return FALSE;
55
56 ret = g_dbus_proxy_call_sync (proxy,
57 "NameHasOwner",
58 g_variant_new ("(s)", "org.freedesktop.DisplayManager"),
59 G_DBUS_CALL_FLAGS_NONE,
60 -1,
61 NULL,
62 &error);
63 if (ret == NULL) {
64 g_warning ("Couldn't call dbus method: %s", error->message);
65 g_error_free (error);
66 } else {
67 g_variant_get (ret, "(b)", &running);
68 g_variant_unref (ret);
69 }
70
71 if (proxy)
72 g_object_unref (proxy);
73
74 return running;
75}

94 NULL,
95 &error);
96 if (ret == NULL) {
97 g_propagate_error (err, error);

99 g_variant_get (ret, "(u)", &pid);

102
103 if (proxy)
104 g_object_unref (proxy);

109static gchar* get_cmdline_from_pid (gint pid)
110{
111 gchar path[255];
112 gchar *text = NULL;
113 gchar *cmdline = NULL;
114 GError *error = NULL;
115
116 g_snprintf(path, sizeof(path), "/proc/%d/cmdline", pid);
117 if (!g_file_get_contents (path, &text, NULL, &error)) {
118 g_warning ("get cmdline error: %s\n", error->message);
119 g_error_free (error);
120 return NULL;
121 } else {
122 cmdline = g_path_get_basename (text);
123 g_free (text);
124 }
125 return cmdline;
126}

128DMType dm_get_type(void)
129{
130 gint pid;
131 gchar *cmdline;
132
133 DMType dmtype = DM_TYPE_UNKNOWN;
134
135 if (!dm_is_running()) {
136 goto ret;
137 }
138
139 pid = dm_get_pid (NULL);
140 if (pid <= 1) {
141 goto ret;
142 }
143
144 cmdline = get_cmdline_from_pid (pid);
145 if (cmdline == NULL) {
146 goto ret;
147 }
148
149 if (g_strcmp0(cmdline, "lightdm") == 0) {
150 dmtype = DM_TYPE_LIGHTDM;
151 } else if (g_strcmp0(cmdline, "mdm") == 0) {
152 dmtype = DM_TYPE_MDM;
153 } else if (g_strcmp0(cmdline, "gdm") == 0) {
154 dmtype = DM_TYPE_GDM;
155 }
156 g_free (cmdline);
157ret:
158 return dmtype;
159}
