799/* which = 0 means draw old sample, which = 1 means draw new */
800static void
801color_sample_draw_sample (MateColorSelection *colorsel, cairo_t *cr, int which)
802{
803 GtkWidget *da;
804 gint x, y, wid, heig, goff;
805 MateColorSelectionPrivate *priv;
806 GtkAllocation allocation;
807
808 g_return_if_fail (colorsel != NULL);
809 priv = colorsel->private_data;
810
811 g_return_if_fail (priv->sample_area != NULL);
812 if (!gtk_widget_is_drawable (priv->sample_area))
813 return;
814
815 if (which == 0)
816 {
817 da = priv->old_sample;
818 goff = 0;
819 }
820 else
821 {
822 da = priv->cur_sample;
823 gtk_widget_get_allocation (priv->old_sample, &allocation);
824 goff = allocation.width % 32;
825 }
826
827 gtk_widget_get_allocation (da, &allocation);
828 wid = allocation.width;
829 heig = allocation.height;
830
831 /* Below needs tweaking for non-power-of-two */
832
833 if (priv->has_opacity)
834 {
835 /* Draw checks in background */
836
837 cairo_set_source_rgb (cr, 0.5, 0.5, 0.5);
838 cairo_rectangle (cr, 0, 0, wid, heig);
839 cairo_fill (cr);
840
841 cairo_set_source_rgb (cr, 0.75, 0.75, 0.75);
842 for (x = goff & -CHECK_SIZE; x < goff + wid; x += CHECK_SIZE)
843 for (y = 0; y < heig; y += CHECK_SIZE)
844 if ((x / CHECK_SIZE + y / CHECK_SIZE) % 2 == 0)
845 cairo_rectangle (cr, x - goff, y, CHECK_SIZE, CHECK_SIZE);
846 cairo_fill (cr);
847 }
848
849 if (which == 0)
850 {
851 cairo_set_source_rgba (cr,
852 priv->old_color[COLORSEL_RED],
853 priv->old_color[COLORSEL_GREEN],
854 priv->old_color[COLORSEL_BLUE],
855 priv->has_opacity ?
856 priv->old_color[COLORSEL_OPACITY] : 1.0);
857 }
858 else
859 {
860 cairo_set_source_rgba (cr,
861 priv->color[COLORSEL_RED],
862 priv->color[COLORSEL_GREEN],
863 priv->color[COLORSEL_BLUE],
864 priv->has_opacity ?
865 priv->color[COLORSEL_OPACITY] : 1.0);
866 }
867
868 cairo_rectangle (cr, 0, 0, wid, heig);
869 cairo_fill (cr);
870}

1029{
1030 gint focus_width;

1056static void
1057set_focus_line_attributes (GtkWidget *drawing_area,
1058 cairo_t *cr,
1059 gint *focus_width)
1060{
1061 gdouble color[4];
1062 gint8 *dash_list;
1063
1064 gtk_widget_style_get (drawing_area,
1065 "focus-line-width", focus_width,
1066 "focus-line-pattern", (gchar *)&dash_list,
1067 NULL);
1068
1069 palette_get_color (drawing_area, color);
1070
1071 if (INTENSITY (color[0], color[1], color[2]) > 0.5)
1072 cairo_set_source_rgb (cr, 0., 0., 0.);
1073 else
1074 cairo_set_source_rgb (cr, 1., 1., 1.);
1075
1076 cairo_set_line_width (cr, *focus_width);
1077
1078 if (dash_list[0])
1079 {
1080 gint n_dashes = strlen ((gchar *)dash_list);
1081 gdouble *dashes = g_new (gdouble, n_dashes);
1082 gdouble total_length = 0;
1083 gdouble dash_offset;
1084 gint i;
1085
1086 for (i = 0; i < n_dashes; i++)
1087 {
1088 dashes[i] = dash_list[i];
1089 total_length += dash_list[i];
1090 }
1091
1092 /* The dash offset here aligns the pattern to integer pixels
1093 * by starting the dash at the right side of the left border
1094 * Negative dash offsets in cairo don't work
1095 * (https://bugs.freedesktop.org/show_bug.cgi?id=2729)
1096 */
1097 dash_offset = - *focus_width / 2.;
1098 while (dash_offset < 0)
1099 dash_offset += total_length;
1100
1101 cairo_set_dash (cr, dashes, n_dashes, dash_offset);
1102 g_free (dashes);
1103 }
1104
1105 g_free (dash_list);
1106}

1187{

1189 MateColorSelectionPrivate *priv;
1190 GdkRGBA gdk_color;
