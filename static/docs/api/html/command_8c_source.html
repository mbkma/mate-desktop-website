93/* GSettings signal callbacks */
94static void
95settings_command_changed (GSettings *settings, gchar *key, CommandApplet *command_applet)
96{
97 GError *error = NULL;
98 gchar *cmdline;
99 gchar **argv;
100
101 cmdline = g_settings_get_string (command_applet->settings, COMMAND_KEY);
102 if (strlen (cmdline) == 0 || g_strcmp0(command_applet->cmdline, cmdline) == 0)
103 {
104 g_free (cmdline);
105 return;
106 }
107
108 if (!g_shell_parse_argv (cmdline, NULL, &argv, &error))
109 {
110 gtk_label_set_text (command_applet->label, ERROR_OUTPUT);
111 g_clear_error (&error);
112 g_free (cmdline);
113 return;
114 }
115 g_strfreev(argv);
116
117 if (command_applet->cmdline)
118 g_free (command_applet->cmdline);
119 command_applet->cmdline = cmdline;
120
121 command_execute (command_applet);
122}

172static gboolean
173command_text_changed (GtkWidget *widget, GdkEvent *event, gpointer user_data)
174{
175 const gchar *text;
176 CommandApplet *command_applet;
177
178 command_applet = (CommandApplet*) user_data;
179 text = gtk_entry_get_text (GTK_ENTRY(widget));
180 if (g_strcmp0(command_applet->cmdline, text) == 0) {
181 return TRUE;
182 }
183
184 if (strlen (text) == 0) {
185 gtk_label_set_text (command_applet->label, ERROR_OUTPUT);
186 return TRUE;
187 }
188
189 g_settings_set_string (command_applet->settings, COMMAND_KEY, text);
190 return TRUE;
191}

285static void
286process_command_output (CommandApplet *command_applet, gchar *output)
287{
288 gtk_widget_set_tooltip_text (GTK_WIDGET (command_applet->label), command_applet->cmdline);
289
290 if ((output == NULL) || (output[0] == '\0'))
291 {
292 gtk_label_set_text (command_applet->label, ERROR_OUTPUT);
293 return;
294 }
295
296 /* check if output is a custom GKeyFile */
297 if (g_str_has_prefix (output, "[Command]"))
298 {
299 GKeyFile *file = g_key_file_new ();
300 if (g_key_file_load_from_data (file, output, -1, G_KEY_FILE_NONE, NULL))
301 {
302 gchar *goutput = g_key_file_get_string (file, GK_COMMAND_GROUP, GK_COMMAND_OUTPUT, NULL);
303 gchar *icon = g_key_file_get_string (file, GK_COMMAND_GROUP, GK_COMMAND_ICON, NULL);
304
305 if (goutput)
306 {
307 gtk_label_set_use_markup (command_applet->label, TRUE);
308 gtk_label_set_markup (command_applet->label, goutput);
309 }
310
311 if (icon)
312 gtk_image_set_from_icon_name (command_applet->image, icon, GTK_ICON_SIZE_LARGE_TOOLBAR);
313
314 g_free (goutput);
315 g_free (icon);
316 }
317 else
318 gtk_label_set_text (command_applet->label, ERROR_OUTPUT);
319
320 g_key_file_free (file);
321 }
322 else
323 {
324 /* Remove leading and trailing whitespace */
325 g_strstrip (output);
326
327 /* check output length */
328 if (g_utf8_strlen (output, strlen(output)) > command_applet->width)
329 {
330 *g_utf8_offset_to_pointer(output, command_applet->width) = '\0';
331 }
332
333 gtk_label_set_text (command_applet->label, output);
334 }
335}

337static void command_async_ready_callback (GObject *source_object, GAsyncResult *res, gpointer user_data)
338{
339 gchar *output;
340 GError *error = NULL;
341 CommandApplet *command_applet;
342
343 command_applet = (CommandApplet*) user_data;
344
345 output = ma_command_run_finish (command_applet->command, res, &error);
346 if (error == NULL) {
347 process_command_output (command_applet, output);
348 } else {
349 if (!g_error_matches (error, G_IO_ERROR, G_IO_ERROR_FAILED)) {
350 gtk_label_set_text (command_applet->label, ERROR_OUTPUT);
351 }
352 g_error_free (error);
353 }
354 g_free (output);
355 command_applet->running = FALSE;
356}

358static gboolean timeout_callback (CommandApplet *command_applet)
359{
360 /* command is empty, wait for next timer execution */
361 if (strlen (command_applet->cmdline) == 0) {
362 return G_SOURCE_CONTINUE;
363 }
364
365 /* command running, wait for next timer execution */
366 if (command_applet->running) {
367 return G_SOURCE_CONTINUE;
368 } else {
369 gchar **argv;
370 GError *error = NULL;
371 if (!g_shell_parse_argv (command_applet->cmdline, NULL, &argv, &error)) {
372 gtk_label_set_text (command_applet->label, ERROR_OUTPUT);
373 g_clear_error (&error);
374 return G_SOURCE_CONTINUE;
375 }
376 g_strfreev(argv);
377 command_execute (command_applet);
378 return G_SOURCE_REMOVE;
379 }
380}

382static gboolean
383command_execute (CommandApplet *command_applet)
384{
385 /* stop current timer */
386 if (command_applet->timeout_id != 0)
387 {
388 g_source_remove (command_applet->timeout_id);
389 command_applet->timeout_id = 0;
390 }
391
392 if (command_applet->running) {
393 g_cancellable_cancel (command_applet->cancellable);
394 }
395
396 g_object_set (G_OBJECT(command_applet->command), "command", command_applet->cmdline, NULL);
397 ma_command_run_async (command_applet->command,
398 command_applet->cancellable,
399 command_async_ready_callback,
400 command_applet);
401 if (!command_applet->running) {
402 command_applet->running = TRUE;
403 }
404
405 if (g_cancellable_is_cancelled (command_applet->cancellable)) {
406 g_cancellable_reset (command_applet->cancellable);
407 }
408
409 command_applet->timeout_id = g_timeout_add_seconds (command_applet->interval,
410 (GSourceFunc) timeout_callback,
411 command_applet);
412 return G_SOURCE_CONTINUE;
413}
