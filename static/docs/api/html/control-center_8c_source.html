31#define CONTROL_CENTER_SCHEMA "org.mate.control-center"
32#define CONTROL_CENTER_ACTIONS_LIST_KEY "cc-actions-list"
33#define CONTROL_CENTER_ACTIONS_SEPARATOR ";"
34#define EXIT_SHELL_ON_STATIC_ACTION "cc-exit-shell-on-static-action"

36static GSList* get_actions_list(void)
37{
38 GSettings *settings;
39 GSList* l;
40 GSList* key_list = NULL;
41 GSList* actions_list = NULL;
42 AppAction* action;
43 gchar **array;
44 gint i;
45
46 settings = g_settings_new (CONTROL_CENTER_SCHEMA);
47 array = g_settings_get_strv (settings, CONTROL_CENTER_ACTIONS_LIST_KEY);
48 if (array != NULL) {
49 for (i = 0; array[i]; i++) {
50 key_list = g_slist_append (key_list, g_strdup (array[i]));
51 }
52 }
53 g_strfreev (array);
54 g_object_unref (settings);
55
56 if (!key_list)
57 {
58 g_warning(_("%s key is empty\n"), CONTROL_CENTER_ACTIONS_LIST_KEY);
59 return NULL;
60 }
61
62 for (l = key_list; l != NULL; l = l->next)
63 {
64 gchar* entry = (gchar*) l->data;
65 gchar** temp;
66
67 action = g_new(AppAction, 1);
68 temp = g_strsplit(entry, CONTROL_CENTER_ACTIONS_SEPARATOR, 2);
69 action->name = g_strdup(temp[0]);
70
71 if ((action->item = load_desktop_item_from_unknown(temp[1])) == NULL)
72 {
73 g_free (action->name);
74 g_free (action);
75 g_warning("get_actions_list() - PROBLEM - Can't load %s\n", temp[1]);
76 }
77 else
78 {
79 actions_list = g_slist_prepend(actions_list, action);
80 }
81
82 g_strfreev(temp);
83 g_free(entry);
84 }
85
86 g_slist_free(key_list);
87
88 return g_slist_reverse(actions_list);
89}

91void handle_static_action_clicked(Tile* tile, TileEvent* event, gpointer data)
92{
93 AppShellData* app_data = (AppShellData*) data;
94 MateDesktopItem* item = (MateDesktopItem*) g_object_get_data(G_OBJECT(tile), APP_ACTION_KEY);
95 GSettings *settings;
96 GApplication *app;
97
98 if (event->type == TILE_EVENT_ACTIVATED_DOUBLE_CLICK)
99 {
100 return;
101 }
102
103 open_desktop_item_exec(item);
104
105 settings = g_settings_new (CONTROL_CENTER_SCHEMA);
106
107 if (g_settings_get_boolean(settings, EXIT_SHELL_ON_STATIC_ACTION))
108 {
109 if (app_data->exit_on_close)
110 {
111 app=g_application_get_default();
112 g_application_quit(app);
113 }
114 else
115 {
116 hide_shell(app_data);
117 }
118 }
119
120 g_object_unref(settings);
121}

123static void
124activate (GtkApplication *app)
125{
126 GList *list;
127 GSList* actions;
128 gboolean hidden = FALSE;
129
130 list = gtk_application_get_windows (app);
131
132 AppShellData* app_data = appshelldata_new("matecc.menu", GTK_ICON_SIZE_DND, FALSE, TRUE, 0);
133
134 generate_categories(app_data);
135
136 actions = get_actions_list();
137 layout_shell(app_data, _("Filter"), _("Groups"), _("Common Tasks"), actions, handle_static_action_clicked);
138
139 if (list)
140 {
141 gtk_window_present (GTK_WINDOW (list->data));
142 }
143 else
144 {
145 create_main_window(app_data, "MyControlCenter", _("Control Center"), "preferences-desktop", 975, 600, hidden);
146 gtk_application_add_window (app, GTK_WINDOW(app_data->main_app));
147 }
148}

172#endif /* ENABLE_NLS */

174 error = NULL;
175
176 if (!gtk_init_with_args(&argc, &argv, NULL, options, GETTEXT_PACKAGE, &error))
177 {
178 g_printerr("%s\n", error->message);
179 g_error_free(error);
180 return 1;
181 }
182
183 g_signal_connect (app, "activate", G_CALLBACK (activate), NULL);
184 g_signal_connect (app, "window-removed", G_CALLBACK (quit), NULL);
185 retval = g_application_run (G_APPLICATION (app), argc, argv);
186 return retval;
