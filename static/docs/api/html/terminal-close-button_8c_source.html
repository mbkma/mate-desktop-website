48static void
49terminal_close_button_init (TerminalCloseButton *button)
50{
51 GtkWidget *image;
52 GtkStyleContext *context;
53
54 gtk_widget_set_name (GTK_WIDGET (button), "mate-terminal-tab-close-button");
55
56 image = gtk_image_new_from_icon_name ("window-close", GTK_ICON_SIZE_MENU);
57 gtk_widget_show (image);
58
59 gtk_container_add (GTK_CONTAINER (button), image);
60
61 context = gtk_widget_get_style_context (GTK_WIDGET (button));
62 gtk_style_context_add_provider (context,
63 GTK_STYLE_PROVIDER (TERMINAL_CLOSE_BUTTON_GET_CLASS (button)->priv->css),
64 GTK_STYLE_PROVIDER_PRIORITY_APPLICATION);
65}
