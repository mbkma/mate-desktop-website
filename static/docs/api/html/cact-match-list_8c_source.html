212static void
213create_tree_model( MatchListData *data )
214{
215 GtkListStore *model;
216 GtkCellRenderer *text_cell, *radio_cell;
217 GtkTreeViewColumn *column;
218 GtkTreeSelection *selection;
219
220 model = gtk_list_store_new( N_COLUMN, G_TYPE_STRING, G_TYPE_BOOLEAN, G_TYPE_BOOLEAN );
221 gtk_tree_view_set_model( data->listview, GTK_TREE_MODEL( model ));
222 g_object_unref( model );
223
224 text_cell = gtk_cell_renderer_text_new();
225 column = gtk_tree_view_column_new_with_attributes(
226 data->item_header,
227 text_cell,
228 "text", ITEM_COLUMN,
229 NULL );
230 gtk_tree_view_append_column( data->listview, column );
231
232 radio_cell = gtk_cell_renderer_toggle_new();
233 gtk_cell_renderer_toggle_set_radio( GTK_CELL_RENDERER_TOGGLE( radio_cell ), TRUE );
234 column = gtk_tree_view_column_new_with_attributes(
235 get_must_match_header( data->match_header ),
236 radio_cell,
237 "active", MUST_MATCH_COLUMN,
238 NULL );
239 gtk_tree_view_append_column( data->listview, column );
240
241 radio_cell = gtk_cell_renderer_toggle_new();
242 gtk_cell_renderer_toggle_set_radio( GTK_CELL_RENDERER_TOGGLE( radio_cell ), TRUE );
243 column = gtk_tree_view_column_new_with_attributes(
244 /* Translators: label of the header of a column which let the user select a negative filter */
245 _( "Must not match any of" ),
246 radio_cell,
247 "active", MUST_NOT_MATCH_COLUMN,
248 NULL );
249 gtk_tree_view_append_column( data->listview, column );
250
251 /* an empty column to fill out the view
252 */
253 column = gtk_tree_view_column_new();
254 gtk_tree_view_append_column( data->listview, column );
255
256 gtk_tree_view_set_headers_visible( data->listview, TRUE );
257 gtk_tree_view_set_headers_clickable( data->listview, TRUE );
258
259 selection = gtk_tree_view_get_selection( data->listview );
260 gtk_tree_selection_set_mode( selection, GTK_SELECTION_BROWSE );
261}

384/*
385 * Called each time the selection changes in the Actions tree view.
386 *
387 * Basically we are using here a rather common scheme:
388 * - object has a GSList of strings, each of one being a filter description,
389 * which may be negated
390 * - the list is displayed in a listview with radio toggle buttons
391 * so that a user cannot have both positive and negative assertions
392 * for the same basename filter
393 * - update the object with a summary of the listbox contents
394 */
395static void
396on_main_selection_changed( BaseWindow *window, GList *selected_items, MatchListData *data )
397{
398 static const gchar *thisfn = "cact_match_list_on_main_selection_changed";
399 NAIContext *context;
400 gboolean enable_tab;
401 GSList *filters;
402 GtkTreeModel *model;
403 GtkTreeSelection *selection;
404 GtkTreeViewColumn *column;
405 GtkTreePath *path;
406
407 g_return_if_fail( BASE_IS_WINDOW( window ));
408 g_return_if_fail( data != NULL );
409
410 g_object_get( G_OBJECT( window ),
411 MAIN_PROP_CONTEXT, &context, MAIN_PROP_EDITABLE, &data->editable_item,
412 NULL );
413
414 enable_tab = ( context != NULL );
415 cact_main_tab_enable_page( CACT_MAIN_WINDOW( data->window ), data->tab_id, enable_tab );
416
417 data->on_selection_change = TRUE;
418
419 filters = context ? ( *data->pget )( context ) : NULL;
420 g_debug( "%s: filters=%p (count=%d)", thisfn, ( void * ) filters, filters ? g_slist_length( filters ) : -1 );
421
422 model = gtk_tree_view_get_model( data->listview );
423 selection = gtk_tree_view_get_selection( data->listview );
424 gtk_tree_selection_unselect_all( selection );
425 gtk_list_store_clear( GTK_LIST_STORE( model ));
426
427 if( filters ){
428 na_core_utils_slist_dump( thisfn, filters );
429 g_slist_foreach( filters, ( GFunc ) iter_for_setup, model );
430 }
431
432 column = gtk_tree_view_get_column( data->listview, ITEM_COLUMN );
433 base_gtk_utils_set_editable( G_OBJECT( column ), data->editable_item && data->editable_filter );
434
435 base_gtk_utils_set_editable( G_OBJECT( data->addbutton ), data->editable_item );
436 base_gtk_utils_set_editable( G_OBJECT( data->removebutton ), data->editable_item );
437 gtk_widget_set_sensitive( data->removebutton, FALSE );
438
439 data->on_selection_change = FALSE;
440
441 path = gtk_tree_path_new_first();
442 if( path ){
443 selection = gtk_tree_view_get_selection( data->listview );
444 gtk_tree_selection_select_path( selection, path );
445 gtk_tree_path_free( path );
446 }
447}

511static void
512on_filter_edited( GtkCellRendererText *renderer, const gchar *path_str, const gchar *text, MatchListData *data )
513{
514 static const gchar *thisfn = "cact_match_list_on_filter_edited";
515 GtkTreeModel *model;
516 GtkTreeIter iter;
517 GtkTreePath *path;
518 gchar *old_text;
519 NAIContext *context;
520 gboolean must_match, must_not_match;
521 gchar *to_add, *to_remove;
522 GSList *filters;
523 GtkWidget *dialog;
524
525 g_return_if_fail( data->editable_filter );
526
527 g_object_get( G_OBJECT( data->window ), MAIN_PROP_CONTEXT, &context, NULL );
528 g_return_if_fail( NA_IS_ICONTEXT( context ));
529
530 model = gtk_tree_view_get_model( data->listview );
531 path = gtk_tree_path_new_from_string( path_str );
532 gtk_tree_model_get_iter( model, &iter, path );
533 gtk_tree_path_free( path );
534
535 gtk_tree_model_get( model, &iter, ITEM_COLUMN, &old_text, -1 );
536
537 if( strcmp( text, old_text ) == 0 ){
538 return;
539 }
540
541 dump_current_rows( data );
542 g_debug( "%s: new filter=%s, count=%d", thisfn, text, count_filters( text, data ));
543
544 if( count_filters( text, data ) >= 1 ){
545 dialog = gtk_message_dialog_new(
546 base_window_get_gtk_toplevel( BASE_WINDOW( data->window )),
547 GTK_DIALOG_MODAL, GTK_MESSAGE_WARNING, GTK_BUTTONS_OK,
548 _( "'%s' filter already exists in the list.\nPlease provide another one." ), text );
549 gtk_dialog_run( GTK_DIALOG( dialog ));
550 gtk_widget_destroy( dialog );
551
552 return;
553 }
554
555 gtk_tree_model_get( model, &iter,
556 MUST_MATCH_COLUMN, &must_match,
557 MUST_NOT_MATCH_COLUMN, &must_not_match,
558 -1 );
559
560 gtk_list_store_set( GTK_LIST_STORE( model ), &iter, ITEM_COLUMN, text, -1 );
561
562 filters = ( *data->pget )( context );
563
564 if( filters ){
565 to_remove = g_strdup( old_text );
566 filters = na_core_utils_slist_remove_ascii( filters, to_remove );
567 g_free( to_remove );
568 to_remove = g_strdup_printf( "!%s", old_text );
569 filters = na_core_utils_slist_remove_ascii( filters, to_remove );
570 g_free( to_remove );
571 }
572
573 if( must_match ){
574 filters = g_slist_prepend( filters, g_strdup( text ));
575
576 } else if( must_not_match ){
577 to_add = g_strdup_printf( "!%s", text );
578 filters = g_slist_prepend( filters, to_add );
579 }
580
581 ( *data->pset )( context, filters );
582 na_core_utils_slist_free( filters );
583 g_free( old_text );
584
585 g_signal_emit_by_name( G_OBJECT( data->window ), TAB_UPDATABLE_SIGNAL_ITEM_UPDATED, context, 0 );
586}

588static gboolean
589on_key_pressed_event( GtkWidget *widget, GdkEventKey *event, MatchListData *data )
590{
591 gboolean stop;
592
593 stop = FALSE;
594
595 if( event->keyval == CACT_KEY_F2 ){
596 if( data->editable_filter ){
597 edit_inline( data );
598 stop = TRUE;
599 }
600 }
601
602 if( event->keyval == CACT_KEY_Insert || event->keyval == CACT_KEY_KP_Insert ){
603 if( data->editable_item ){
604 insert_new_row( data );
605 stop = TRUE;
606 }
607 }
608
609 if( event->keyval == CACT_KEY_Delete || event->keyval == CACT_KEY_KP_Delete ){
610 if( data->editable_item ){
611 delete_current_row( data );
612 stop = TRUE;
613 }
614 }
615
616 return( stop );
617}

625/*
626 * clicking on an already active toggle button has no effect
627 * clicking on an icactive toggle button has a double effect:
628 * - the other toggle button becomes icactive
629 * - this toggle button becomes active
630 * the corresponding strings must be respectively removed/added to the
631 * filters list
632 */
633static void
634on_must_match_toggled( GtkCellRendererToggle *cell_renderer, gchar *path_str, MatchListData *data )
635{
636 /*static const gchar *thisfn = "cact_match_list_on_must_match_toggled";*/
637 gchar *filter;
638 NAIContext *context;
639 GSList *filters;
640 gchar *to_remove;
641 gboolean active;
642
643 /*gboolean is_active = gtk_cell_renderer_toggle_get_active( cell_renderer );
644 g_debug( "%s: is_active=%s", thisfn, is_active ? "True":"False" );*/
645
646 active = gtk_cell_renderer_toggle_get_active( cell_renderer );
647
648 if( data->editable_item ){
649 if( !active ){
650 g_object_get( G_OBJECT( data->window ), MAIN_PROP_CONTEXT, &context, NULL );
651 g_return_if_fail( NA_IS_ICONTEXT( context ));
652
653 set_match_status( path_str, TRUE, FALSE, data );
654
655 filter = get_filter_from_path( path_str, data );
656 filters = ( *data->pget )( context );
657
658 if( filters ){
659 to_remove = g_strdup_printf( "!%s", filter );
660 filters = na_core_utils_slist_remove_ascii( filters, to_remove );
661 g_free( to_remove );
662 }
663
664 filters = g_slist_prepend( filters, g_strdup( filter ));
665 ( *data->pset )( context, filters );
666
667 na_core_utils_slist_free( filters );
668 g_free( filter );
669
670 g_signal_emit_by_name( G_OBJECT( data->window ), TAB_UPDATABLE_SIGNAL_ITEM_UPDATED, context, 0 );
671 }
672 } else {
673 g_signal_handlers_block_by_func(( gpointer ) cell_renderer, on_must_match_toggled, data );
674 gtk_cell_renderer_toggle_set_active( cell_renderer, !active );
675 g_signal_handlers_unblock_by_func(( gpointer ) cell_renderer, on_must_match_toggled, data );
676 }
677}
