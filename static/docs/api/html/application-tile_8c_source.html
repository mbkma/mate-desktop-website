122GtkWidget *
123application_tile_new_full (const gchar *desktop_item_id,
124 GtkIconSize image_size, gboolean show_generic_name)
125{
126 ApplicationTile *this;
127 ApplicationTilePrivate *priv;
128
129 const gchar *uri = NULL;
130
131 MateDesktopItem *desktop_item;
132
133 desktop_item = load_desktop_item_from_unknown (desktop_item_id);
134
135 if (
136 desktop_item &&
137 mate_desktop_item_get_entry_type (desktop_item) == MATE_DESKTOP_ITEM_TYPE_APPLICATION
138 )
139 uri = mate_desktop_item_get_location (desktop_item);
140
141 if (! uri) {
142 if (desktop_item)
143 mate_desktop_item_unref (desktop_item);
144
145 return NULL;
146 }
147
148 this = g_object_new (APPLICATION_TILE_TYPE, "tile-uri", uri, NULL);
149 priv = application_tile_get_instance_private (this);
150
151 priv->image_size = image_size;
152 priv->desktop_item = desktop_item;
153 priv->show_generic_name = show_generic_name;
154
155 application_tile_setup (this);
156
157 return GTK_WIDGET (this);
158}

177static void
178application_tile_finalize (GObject *g_object)
179{
180 ApplicationTile *tile = APPLICATION_TILE (g_object);
181 ApplicationTilePrivate *priv = application_tile_get_instance_private (tile);
182
183 if (tile->name) {
184 g_free (tile->name);
185 tile->name = NULL;
186 }
187 if (tile->description) {
188 g_free (tile->description);
189 tile->description = NULL;
190 }
191
192 if (priv->desktop_item) {
193 mate_desktop_item_unref (priv->desktop_item);
194 priv->desktop_item = NULL;
195 }
196 if (priv->image_id) {
197 g_free (priv->image_id);
198 priv->image_id = NULL;
199 }
200
201 if (priv->notify_signal_id)
202 g_signal_handler_disconnect (priv->agent, priv->notify_signal_id);
203
204 g_object_unref (G_OBJECT (priv->agent));
205
206 G_OBJECT_CLASS (application_tile_parent_class)->finalize (g_object);
207}

251static void
252application_tile_setup (ApplicationTile *this)
253{
254 ApplicationTilePrivate *priv = application_tile_get_instance_private (this);
255
256 GtkWidget *image;
257 GtkWidget *header;
258 GtkWidget *subheader;
259 GtkMenu *context_menu;
260 AtkObject *accessible;
261
262 TileAction **actions;
263 TileAction *action;
264 GtkWidget *menu_item;
265 GtkContainer *menu_ctnr;
266
267 gchar *name;
268 gchar *desc;
269
270 gchar *comment;
271
272 gchar *markup;
273 gchar *str;
274
275 if (! priv->desktop_item) {
276 priv->desktop_item = load_desktop_item_from_unknown (TILE (this)->uri);
277
278 if (! priv->desktop_item)
279 return;
280 }
281
282 priv->image_id = g_strdup (mate_desktop_item_get_localestring (priv->desktop_item, "Icon"));
283 image = themed_icon_new (priv->image_id, priv->image_size);
284
285 gchar *filename = g_filename_from_uri (mate_desktop_item_get_location (priv->desktop_item), NULL, NULL);
286 GKeyFile *keyfile = g_key_file_new ();
287 g_key_file_load_from_file (keyfile, filename, G_KEY_FILE_KEEP_COMMENTS | G_KEY_FILE_KEEP_TRANSLATIONS, NULL);
288
289 name = g_key_file_get_locale_string (keyfile, G_KEY_FILE_DESKTOP_GROUP, "Name", NULL, NULL);
290 desc = g_key_file_get_locale_string (keyfile, G_KEY_FILE_DESKTOP_GROUP, "GenericName", NULL, NULL);
291 comment = g_key_file_get_locale_string (keyfile, G_KEY_FILE_DESKTOP_GROUP, "Comment", NULL, NULL);
292
293 accessible = gtk_widget_get_accessible (GTK_WIDGET (this));
294 if (name)
295 atk_object_set_name (accessible, name);
296 if (desc)
297 atk_object_set_description (accessible, desc);
298
299 header = create_header (name);
300
301 /*if no GenericName then just show and center the Name */
302 if (desc && priv->show_generic_name
303 && (!name || strcmp(name, desc) != 0))
304 subheader = create_subheader (desc);
305 else
306 subheader = NULL;
307
308 context_menu = GTK_MENU (gtk_menu_new ());
309
310 g_object_set (
311 G_OBJECT (this),
312 "nameplate-image", image,
313 "nameplate-header", header,
314 "nameplate-subheader", subheader,
315 "context-menu", context_menu,
316 "application-name", name,
317 "application-description", desc,
318 NULL);
319 gtk_widget_set_tooltip_text (GTK_WIDGET (this), comment);
320
321 priv->agent = bookmark_agent_get_instance (BOOKMARK_STORE_USER_APPS);
322 g_object_get (G_OBJECT (priv->agent), BOOKMARK_AGENT_STORE_STATUS_PROP, & priv->agent_status, NULL);
323
324 priv->notify_signal_id = g_signal_connect (
325 G_OBJECT (priv->agent), "notify", G_CALLBACK (agent_notify_cb), this);
326
327 priv->startup_status = get_desktop_item_startup_status (priv->desktop_item);
328
329 actions = g_new0 (TileAction *, 6);
330
331 TILE (this)->actions = actions;
332 TILE (this)->n_actions = 6;
333
334 menu_ctnr = GTK_CONTAINER (TILE (this)->context_menu);
335
336/* make start action */
337
338 str = g_strdup_printf (_("Start %s"), this->name);
339 markup = g_markup_printf_escaped ("<b>%s</b>", str);
340 action = tile_action_new (TILE (this), start_trigger, markup, TILE_ACTION_OPENS_NEW_WINDOW);
341 actions [APPLICATION_TILE_ACTION_START] = action;
342 g_free (markup);
343 g_free (str);
344
345 menu_item = GTK_WIDGET (tile_action_get_menu_item (action));
346
347 gtk_container_add (menu_ctnr, menu_item);
348
349 TILE (this)->default_action = action;
350
351/* insert separator */
352
353 gtk_container_add (menu_ctnr, gtk_separator_menu_item_new ());
354
355/* make help action */
356
357 if (mate_desktop_item_get_string (priv->desktop_item, "DocPath")) {
358 action = tile_action_new (
359 TILE (this), help_trigger, _("Help"),
360 TILE_ACTION_OPENS_NEW_WINDOW | TILE_ACTION_OPENS_HELP);
361
362 menu_item = GTK_WIDGET (tile_action_get_menu_item (action));
363 gtk_container_add (menu_ctnr, menu_item);
364 }
365 else {
366 action = NULL;
367 }
368
369 actions [APPLICATION_TILE_ACTION_HELP] = action;
370
371/* insert separator */
372
373 if (action != NULL)
374 gtk_container_add (menu_ctnr, gtk_separator_menu_item_new ());
375
376/* make "add/remove to favorites" action */
377
378 update_user_list_menu_item (this);
379
380/* make "add/remove to startup" action */
381
382 if (priv->startup_status != APP_NOT_ELIGIBLE) {
383 action = tile_action_new (TILE (this), startup_trigger, NULL, 0);
384 actions [APPLICATION_TILE_ACTION_UPDATE_STARTUP] = action;
385
386 update_startup_menu_item (this);
387
388 menu_item = GTK_WIDGET (tile_action_get_menu_item (action));
389
390 gtk_container_add (menu_ctnr, menu_item);
391 }
392
393 gtk_widget_show_all (GTK_WIDGET (TILE (this)->context_menu));
394
395 g_free (name);
396 g_free (desc);
397 g_free (comment);
398 g_free (filename);
399 g_key_file_unref (keyfile);
400}

540static void
541add_to_startup_list (ApplicationTile *this)
542{
543 ApplicationTilePrivate *priv = application_tile_get_instance_private (this);
544
545 gchar *desktop_item_filename;
546 gchar *desktop_item_basename;
547
548 gchar *startup_dir;
549 gchar *dst_filename;
550
551 const gchar *src_uri;
552 gchar *dst_uri;
553
554 desktop_item_filename =
555 g_filename_from_uri (mate_desktop_item_get_location (priv->desktop_item), NULL,
556 NULL);
557
558 g_return_if_fail (desktop_item_filename != NULL);
559
560 desktop_item_basename = g_path_get_basename (desktop_item_filename);
561
562 startup_dir = g_build_filename (g_get_user_config_dir (), "autostart", NULL);
563
564 if (! g_file_test (startup_dir, G_FILE_TEST_EXISTS))
565 g_mkdir_with_parents (startup_dir, 0700);
566
567 dst_filename = g_build_filename (startup_dir, desktop_item_basename, NULL);
568
569 src_uri = mate_desktop_item_get_location (priv->desktop_item);
570 dst_uri = g_filename_to_uri (dst_filename, NULL, NULL);
571
572 copy_file (src_uri, dst_uri);
573 priv->startup_status = APP_IN_USER_STARTUP_DIR;
574
575 g_free (desktop_item_filename);
576 g_free (desktop_item_basename);
577 g_free (startup_dir);
578 g_free (dst_filename);
579 g_free (dst_uri);
580}

623static void
624update_user_list_menu_item (ApplicationTile *this)
625{
626 ApplicationTilePrivate *priv = application_tile_get_instance_private (this);
627
628 TileAction *action;
629 GtkWidget *item;
630
631 if (priv->agent_status == BOOKMARK_STORE_ABSENT) {
632 if (TILE (this)->actions [APPLICATION_TILE_ACTION_UPDATE_MAIN_MENU])
633 g_object_unref (TILE (this)->actions [APPLICATION_TILE_ACTION_UPDATE_MAIN_MENU]);
634
635 TILE (this)->actions [APPLICATION_TILE_ACTION_UPDATE_MAIN_MENU] = NULL;
636 }
637 else if (! TILE (this)->actions [APPLICATION_TILE_ACTION_UPDATE_MAIN_MENU]) {
638 TILE (this)->actions [APPLICATION_TILE_ACTION_UPDATE_MAIN_MENU] =
639 tile_action_new (TILE (this), user_apps_trigger, NULL, 0);
640
641 tile_action_set_menu_item_label (
642 TILE (this)->actions [APPLICATION_TILE_ACTION_UPDATE_MAIN_MENU], "blah");
643
644 item = GTK_WIDGET (tile_action_get_menu_item (
645 TILE (this)->actions [APPLICATION_TILE_ACTION_UPDATE_MAIN_MENU]));
646 gtk_menu_shell_insert (GTK_MENU_SHELL (TILE (this)->context_menu), item, 4);
647
648 gtk_widget_show_all (item);
649 }
650 else {
651 /* do nothing */ ;
652 }
653
654 action = TILE (this)->actions [APPLICATION_TILE_ACTION_UPDATE_MAIN_MENU];
655
656 if (! action)
657 return;
658
659 priv->is_bookmarked = bookmark_agent_has_item (priv->agent, TILE (this)->uri);
660
661 if (priv->is_bookmarked)
662 tile_action_set_menu_item_label (action, _("Remove from Favorites"));
663 else
664 tile_action_set_menu_item_label (action, _("Add to Favorites"));
665
666 item = GTK_WIDGET (tile_action_get_menu_item (action));
667
668 if (! GTK_IS_MENU_ITEM (item))
669 return;
670
671 g_object_get (G_OBJECT (priv->agent), BOOKMARK_AGENT_STORE_STATUS_PROP, & priv->agent_status, NULL);
672
673 gtk_widget_set_sensitive (item, (priv->agent_status != BOOKMARK_STORE_DEFAULT_ONLY));
674}

676static StartupStatus
677get_desktop_item_startup_status (MateDesktopItem *desktop_item)
678{
679 gchar *filename;
680 gchar *basename;
681
682 const gchar * const * global_dirs;
683 gchar *global_target;
684 gchar *user_target;
685
686 StartupStatus retval;
687 gint x;
688
689 filename = g_filename_from_uri (mate_desktop_item_get_location (desktop_item), NULL, NULL);
690 if (!filename)
691 return APP_NOT_ELIGIBLE;
692 basename = g_path_get_basename (filename);
693
694 retval = APP_NOT_IN_STARTUP_DIR;
695 global_dirs = g_get_system_config_dirs();
696 for(x=0; global_dirs[x]; x++)
697 {
698 global_target = g_build_filename (global_dirs[x], "autostart", basename, NULL);
699 if (g_file_test (global_target, G_FILE_TEST_EXISTS))
700 {
701 retval = APP_NOT_ELIGIBLE;
702 g_free (global_target);
703 break;
704 }
705 g_free (global_target);
706 }
707
708 /* mate-session currently checks these dirs also. see startup-programs.c */
709 if (retval != APP_NOT_ELIGIBLE)
710 {
711 global_dirs = g_get_system_data_dirs();
712 for(x=0; global_dirs[x]; x++)
713 {
714 global_target = g_build_filename (global_dirs[x], "mate", "autostart", basename, NULL);
715 if (g_file_test (global_target, G_FILE_TEST_EXISTS))
716 {
717 retval = APP_NOT_ELIGIBLE;
718 g_free (global_target);
719 break;
720 }
721 g_free (global_target);
722 }
723 }
724
725 if (retval != APP_NOT_ELIGIBLE)
726 {
