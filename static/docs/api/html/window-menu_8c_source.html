135static gboolean window_menu_on_draw (GtkWidget* widget,
136 cairo_t* cr,
137 gpointer data)
138{
139 GtkStyleContext *context;
140 GtkStateFlags state;
141 WindowMenu *window_menu = data;
142
143 if (!gtk_widget_has_focus (window_menu->applet))
144 return FALSE;
145
146 state = gtk_widget_get_state_flags (widget);
147 context = gtk_widget_get_style_context (widget);
148 gtk_style_context_save (context);
149 gtk_style_context_set_state (context, state);
150
151 cairo_save (cr);
152 gtk_render_focus (context, cr,
153 0., 0.,
154 gtk_widget_get_allocated_width (widget),
155 gtk_widget_get_allocated_height (widget));
156 cairo_restore (cr);
157
158 gtk_style_context_restore (context);
159
160 return FALSE;
161}

163static void window_menu_size_allocate(MatePanelApplet* applet, GtkAllocation* allocation, WindowMenu* window_menu)
164{
165 MatePanelAppletOrient orient;
166 GList* children;
167 GtkWidget* child;
168
169 orient = mate_panel_applet_get_orient(applet);
170
171 if (!GTK_IS_CONTAINER (window_menu->selector))
172 return;
173
174 children = gtk_container_get_children(GTK_CONTAINER(window_menu->selector));
175 child = GTK_WIDGET(children->data);
176 g_list_free(children);
177
178 if (orient == MATE_PANEL_APPLET_ORIENT_LEFT || orient == MATE_PANEL_APPLET_ORIENT_RIGHT)
179 {
180 if (window_menu->size == allocation->width && orient == window_menu->orient)
181 return;
182
183 window_menu->size = allocation->width;
184 gtk_widget_set_size_request(child, window_menu->size, -1);
185 }
186 else
187 {
188 if (window_menu->size == allocation->height && orient == window_menu->orient)
189 return;
190
191 window_menu->size = allocation->height;
192 gtk_widget_set_size_request(child, -1, window_menu->size);
193 }
194
195 window_menu->orient = orient;
196}

198static gboolean window_menu_key_press_event(GtkWidget* widget, GdkEventKey* event, WindowMenu* window_menu)
199{
200 GtkMenuShell* menu_shell;
201
202 switch (event->keyval)
203 {
204 case GDK_KEY_KP_Enter:
205 case GDK_KEY_ISO_Enter:
206 case GDK_KEY_3270_Enter:
207 case GDK_KEY_Return:
208 case GDK_KEY_space:
209 case GDK_KEY_KP_Space:
210 /*
211 * We need to call _gtk_menu_shell_activate() here as is done in
212 * window_key_press_handler in gtkmenubar.c which pops up menu
213 * when F10 is pressed.
214 *
215 * As that function is private its code is replicated here.
216 */
217 menu_shell = GTK_MENU_SHELL(window_menu->selector);
218
219 gtk_menu_shell_select_first(menu_shell, FALSE);
220 return TRUE;
221 default:
222 break;
223 }
224
225 return FALSE;
226}

236gboolean window_menu_applet_fill(MatePanelApplet* applet)
237{
238 WindowMenu* window_menu;
239 GtkActionGroup* action_group;
240
241 window_menu = g_new0(WindowMenu, 1);
242
243 window_menu->applet = GTK_WIDGET(applet);
244 gtk_widget_set_name (window_menu->applet, "window-menu-applet-button");
245 gtk_widget_set_tooltip_text(window_menu->applet, _("Window Selector"));
246
247 mate_panel_applet_set_flags(applet, MATE_PANEL_APPLET_EXPAND_MINOR);
248 window_menu->size = mate_panel_applet_get_size(applet);
249 window_menu->orient = mate_panel_applet_get_orient(applet);
250
251 g_signal_connect(window_menu->applet, "destroy", G_CALLBACK(window_menu_destroy), window_menu);
252 g_signal_connect(window_menu->applet, "key-press-event", G_CALLBACK(window_menu_key_press_event), window_menu);
253
254 action_group = gtk_action_group_new("WindowMenu Applet Actions");
255 gtk_action_group_set_translation_domain(action_group, GETTEXT_PACKAGE);
256 gtk_action_group_add_actions(action_group, window_menu_actions, G_N_ELEMENTS(window_menu_actions), window_menu);
257 mate_panel_applet_setup_menu_from_resource (MATE_PANEL_APPLET (window_menu->applet),
258 WNCKLET_RESOURCE_PATH "window-menu-menu.xml",
259 action_group);
260 g_object_unref(action_group);
261
262#ifdef HAVE_X11
263 if (GDK_IS_X11_DISPLAY (gdk_display_get_default ()))
264 {
265 window_menu->wnck_handle = wnck_handle_new(WNCK_CLIENT_TYPE_PAGER);
266 window_menu->selector = wnck_selector_new_with_handle(window_menu->wnck_handle);
267 }
268 else
269#endif /* HAVE_X11 */
270
271#ifdef HAVE_WAYLAND
272 if (GDK_IS_WAYLAND_DISPLAY (gdk_display_get_default ()))
273 {
274 window_menu->selector = gtk_label_new ("[Window menu not supported on Wayland]");
275 }
276 else
277#endif /* HAVE_WAYLAND */
278
279 {
280 window_menu->selector = gtk_label_new ("[Window menu not supported on this platform]");
281 }
282
283 gtk_container_add(GTK_CONTAINER(window_menu->applet), window_menu->selector);
284
285 g_signal_connect (window_menu->applet, "size-allocate",
286 G_CALLBACK(window_menu_size_allocate),
287 window_menu);
288
289 g_signal_connect_after (window_menu->applet, "focus-in-event",
290 G_CALLBACK (gtk_widget_queue_draw),
291 window_menu);
292 g_signal_connect_after (window_menu->applet, "focus-out-event",
293 G_CALLBACK (gtk_widget_queue_draw),
294 window_menu);
295 g_signal_connect_after (window_menu->selector, "draw",
296 G_CALLBACK (window_menu_on_draw),
297 window_menu);
298
299 g_signal_connect (window_menu->selector, "button_press_event",
300 G_CALLBACK (filter_button_press),
301 window_menu);
302
303 gtk_widget_show_all(GTK_WIDGET(window_menu->applet));
304
305 return TRUE;
306}
