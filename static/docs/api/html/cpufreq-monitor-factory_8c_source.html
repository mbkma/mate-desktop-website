39CPUFreqMonitor *
40cpufreq_monitor_factory_create_monitor (guint cpu)
41{
42#ifdef HAVE_LIBCPUFREQ
43 return cpufreq_monitor_libcpufreq_new (cpu);
44#else
45 CPUFreqMonitor *monitor = NULL;
46
47 if (g_file_test ("/sys/devices/system/cpu/cpu0/cpufreq", G_FILE_TEST_EXISTS)) {
48 monitor = cpufreq_monitor_sysfs_new (cpu);
49 } else if (g_file_test ("/proc/cpuinfo", G_FILE_TEST_EXISTS)) {
50 /* If there is no cpufreq support it shows only the cpu frequency,
51 * I think is better than do nothing. I have to notify it to the user, because
52 * he could think that cpufreq is supported but it doesn't work succesfully
53 */
54
55 cpufreq_utils_display_error (_("CPU frequency scaling unsupported"),
56 _("You will not be able to modify the frequency of your machine. "
57 "Your machine may be misconfigured or not have hardware support "
58 "for CPU frequency scaling."));
59
60 monitor = cpufreq_monitor_cpuinfo_new (cpu);
61 }
62
63 return monitor;
64#endif
65}
