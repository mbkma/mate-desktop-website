52/* called when the "ok" button is pressed. */
53static void
54ok_clicked_cb (GtkWidget *widget,
55 DialogData *data)
56{
57 gboolean selected_files;
58 gboolean pattern_files;
59 FrWindow *window = data->window;
60 GList *file_list = NULL;
61 gboolean do_not_remove_if_null = FALSE;
62
63 selected_files = gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (data->d_selected_files_radio));
64 pattern_files = gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (data->d_files_radio));
65
66 /* create the file list. */
67
68 if (selected_files) {
69 file_list = data->selected_files;
70 data->selected_files = NULL; /* do not the list when destroying the dialog. */
71 }
72 else if (pattern_files) {
73 const char *pattern;
74
75 pattern = gtk_entry_get_text (GTK_ENTRY (data->d_files_entry));
76 file_list = fr_window_get_file_list_pattern (window, pattern);
77 if (file_list == NULL)
78 do_not_remove_if_null = TRUE;
79 }
80
81 /* close the dialog. */
82
83 gtk_widget_destroy (data->dialog);
84
85 /* remove ! */
86
87 if (! do_not_remove_if_null || (file_list != NULL))
88 fr_window_archive_remove (window, file_list);
89
90 path_list_free (file_list);
91}

101static void
102dlg_delete__common (FrWindow *window,
103 GList *selected_files)
104{
105 GtkBuilder *builder;
106 DialogData *data;
107
108 data = g_new (DialogData, 1);
109 data->window = window;
110 data->selected_files = selected_files;
111 builder = gtk_builder_new_from_resource (ENGRAMPA_RESOURCE_UI_PATH G_DIR_SEPARATOR_S "delete.ui");
112
113 /* Get the widgets. */
114
115 data->dialog = GET_WIDGET ("delete_dialog");
116 data->d_all_files_radio = GET_WIDGET ("d_all_files_radio");
117 data->d_selected_files_radio = GET_WIDGET ("d_selected_files_radio");
118 data->d_files_radio = GET_WIDGET ("d_files_radio");
119 data->d_files_entry = GET_WIDGET ("d_files_entry");
120
121 /* Set widgets data. */
122
123 if (data->selected_files != NULL)
124 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (data->d_selected_files_radio), TRUE);
125 else {
126 gtk_widget_set_sensitive (data->d_selected_files_radio, FALSE);
127 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (data->d_all_files_radio), TRUE);
128 }
129
130 /* Set the signals handlers. */
131 gtk_builder_add_callback_symbols (builder,
132 "on_delete_dialog_destroy", G_CALLBACK (destroy_cb),
133 "on_d_ok_button_clicked", G_CALLBACK (ok_clicked_cb),
134 "on_d_files_entry_changed", G_CALLBACK (entry_changed_cb),
135 NULL);
136
137 gtk_builder_connect_signals (builder, data);
138
139 g_signal_connect_swapped (gtk_builder_get_object (builder, "d_cancel_button"),
140 "clicked",
141 G_CALLBACK (gtk_widget_destroy),
142 G_OBJECT (data->dialog));
143
144 g_object_unref (builder);
145
146 /* Run dialog. */
147
148 gtk_window_set_transient_for (GTK_WINDOW (data->dialog),
149 GTK_WINDOW (window));
150 gtk_window_set_modal (GTK_WINDOW (data->dialog), TRUE);
151
152 gtk_widget_show (data->dialog);
153}
