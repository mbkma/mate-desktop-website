102gboolean
103open_desktop_item_help (MateDesktopItem * desktop_item)
104{
105 const gchar *doc_path;
106 gchar *help_uri;
107
108 GError *error;
109
110 if (!desktop_item)
111 return FALSE;
112
113 doc_path = mate_desktop_item_get_string (desktop_item, "DocPath");
114
115 if (doc_path)
116 {
117 help_uri = g_strdup_printf ("help:%s", doc_path);
118
119 error = NULL;
120 if (!gtk_show_uri_on_window (NULL, help_uri, gtk_get_current_event_time (), &error))
121 {
122 g_warning ("error opening %s [%s]\n", help_uri, error->message);
123
124 g_free (help_uri);
125 g_error_free (error);
126 return FALSE;
127 }
128
129 g_free (help_uri);
130 }
131 else
132 return FALSE;
133
134 return TRUE;
135}
