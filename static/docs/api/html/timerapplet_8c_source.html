115/* timer management */
116static gboolean
117timer_callback (TimerApplet *applet)
118{
119 AtkObject *atk_obj;
120 gboolean retval = TRUE;
121 gchar *label;
122 gchar *name;
123 gchar *tooltip;
124 gint hours, minutes, seconds, duration, remaining;
125
126 label = NULL;
127 tooltip = NULL;
128
129 if (!GTK_IS_WIDGET (applet->label))
130 return FALSE;
131
132 name = g_settings_get_string (applet->settings, NAME_KEY);
133 atk_obj = gtk_widget_get_accessible (GTK_WIDGET (applet->applet));
134
135 if (!applet->active)
136 {
137 applet->pause = FALSE;
138 applet->elapsed = 0;
139
140 gtk_label_set_text (applet->label, name);
141 gtk_widget_set_tooltip_text (GTK_WIDGET (applet->label), "");
142 gtk_widget_hide (GTK_WIDGET (applet->pause_image));
143 atk_object_set_name (atk_obj, name);
144 }
145 else
146 {
147 if (!applet->pause)
148 applet->elapsed += STEP;
149
150 duration = g_settings_get_int (applet->settings, DURATION_KEY);
151
152 remaining = duration - (applet->elapsed / 1000);
153
154 if (remaining <= 0)
155 {
156 applet->active = FALSE;
157 applet->timeout_id = 0;
158
159 /* Translators: %s is a placeholder for the timer name, 'Timer' by default */
160 label = g_strdup_printf (_("Finished %s"), name);
161 gtk_label_set_text (applet->label, label);
162 gtk_widget_set_tooltip_text (GTK_WIDGET (applet->label), name);
163 gtk_widget_hide (GTK_WIDGET (applet->pause_image));
164 atk_object_set_name (atk_obj, label);
165 atk_object_set_description (atk_obj, "");
166
167 if (g_settings_get_boolean (applet->settings, SHOW_NOTIFICATION_KEY))
168 {
169 NotifyNotification *n;
170 n = notify_notification_new (name, _("Timer finished!"), APPLET_ICON);
171 notify_notification_set_timeout (n, 30000);
172 notify_notification_show (n, NULL);
173 g_object_unref (G_OBJECT (n));
174 }
175
176 if (g_settings_get_boolean (applet->settings, SHOW_DIALOG_KEY))
177 {
178 GtkWidget *dialog = gtk_message_dialog_new_with_markup (NULL,
179 GTK_DIALOG_MODAL,
180 GTK_MESSAGE_INFO,
181 GTK_BUTTONS_OK,
182 "<b>%s</b>\n\n%s", name, _("Timer finished!"));
183 gtk_dialog_run (GTK_DIALOG (dialog));
184 gtk_widget_destroy (dialog);
185 }
186
187 /* stop further calls */
188 retval = FALSE;
189 }
190 else
191 {
192 hours = remaining / 60 / 60;
193 minutes = remaining / 60 % 60;
194 seconds = remaining % 60;
195
196 if (hours > 0)
197 label = g_strdup_printf ("%02d:%02d:%02d", hours, minutes, seconds);
198 else
199 label = g_strdup_printf ("%02d:%02d", minutes, seconds);
200
201 hours = duration / 60 / 60;
202 minutes = duration / 60 % 60;
203 seconds = duration % 60;
204
205 if (hours > 0)
206 tooltip = g_strdup_printf ("%s (%02d:%02d:%02d)", name, hours, minutes, seconds);
207 else
208 tooltip = g_strdup_printf ("%s (%02d:%02d)", name, minutes, seconds);
209
210 gtk_label_set_text (applet->label, label);
211 gtk_widget_set_tooltip_text (GTK_WIDGET (applet->label), tooltip);
212 gtk_widget_set_visible (GTK_WIDGET (applet->pause_image), applet->pause);
213 atk_object_set_name (atk_obj, label);
214 }
215
216 g_free (label);
217 g_free (tooltip);
218 }
219
220 /* update actions sensitiveness */
221 gtk_action_set_sensitive (gtk_action_group_get_action (applet->action_group, "Start"), !applet->active || applet->pause);
222 gtk_action_set_sensitive (gtk_action_group_get_action (applet->action_group, "Pause"), applet->active && !applet->pause);
223 gtk_action_set_sensitive (gtk_action_group_get_action (applet->action_group, "Stop"), applet->active);
224 gtk_action_set_sensitive (gtk_action_group_get_action (applet->action_group, "Reset"), !applet->active && !applet->pause && applet->elapsed);
225 gtk_action_set_sensitive (gtk_action_group_get_action (applet->action_group, "Preferences"), !applet->active && !applet->pause);
226
227 g_free (name);
228
229 return retval;
230}

232/* start action */
233static void
234timer_start_callback (GtkAction *action, TimerApplet *applet)
235{
236 applet->active = TRUE;
237 if (applet->pause)
238 applet->pause = FALSE;
239 else
240 applet->elapsed = 0;
241 applet->timeout_id = g_timeout_add (STEP, (GSourceFunc) timer_callback, applet);
242}

252 applet->timeout_id = 0;

298/* calculate duration and save in GSettings */
299static void
300timer_spin_button_value_changed (GtkSpinButton *spinbutton, TimerApplet *applet)
301{
302 gint duration;
303
304 duration = gtk_spin_button_get_value_as_int (applet->seconds);
305 duration += gtk_spin_button_get_value_as_int (applet->minutes) * 60;
306 duration += gtk_spin_button_get_value_as_int (applet->hours) * 3600;
307
308 g_settings_set_int (applet->settings, DURATION_KEY, duration);
309}

312static void
313timer_preferences_callback (GtkAction *action, TimerApplet *applet)
314{
315 GtkBuilder *builder;
316 GtkDialog *dialog;
317 gint duration, hours, minutes, seconds;
318
319 builder = gtk_builder_new_from_resource ("/org/mate/mate-applets/timerapplet/timerapplet-preferences.ui");
320
321 dialog = GET_DIALOG ("preferences_dialog");
322 applet->hours = GET_SPIN_BUTTON ("hours_spinbutton");
323 applet->minutes = GET_SPIN_BUTTON ("minutes_spinbutton");
324 applet->seconds = GET_SPIN_BUTTON ("seconds_spinbutton");
325
326 duration = g_settings_get_int (applet->settings, DURATION_KEY);
327 hours = duration / 60 / 60;
328 minutes = duration / 60 % 60;
329 seconds = duration % 60;
330
331 gtk_spin_button_set_value (applet->hours, hours);
332 gtk_spin_button_set_value (applet->minutes, minutes);
333 gtk_spin_button_set_value (applet->seconds, seconds);
334
335 g_settings_bind (applet->settings, NAME_KEY, GET_WIDGET ("name_entry"), "text", G_SETTINGS_BIND_DEFAULT);
336 g_settings_bind (applet->settings, SHOW_NOTIFICATION_KEY, GET_WIDGET ("show_notification_popup_check"), "active", G_SETTINGS_BIND_DEFAULT);
337 g_settings_bind (applet->settings, SHOW_DIALOG_KEY, GET_WIDGET ("show_dialog_check"), "active", G_SETTINGS_BIND_DEFAULT);
338
339 /* signals */
340 gtk_builder_add_callback_symbols (builder,
341 "on_hours_spinbutton_value_changed", G_CALLBACK (timer_spin_button_value_changed),
342 "on_minutes_spinbutton_value_changed", G_CALLBACK (timer_spin_button_value_changed),
343 "on_seconds_spinbutton_value_changed", G_CALLBACK (timer_spin_button_value_changed),
344 NULL);
345 gtk_builder_connect_signals (builder, applet);
346
347 g_signal_connect (dialog, "response",
348 G_CALLBACK (gtk_widget_destroy),
349 dialog);
350
351 g_object_unref (builder);
352
353 gtk_widget_show_all (GTK_WIDGET (dialog));
354}

356static gboolean
357timer_applet_click (TimerApplet *applet, GdkEventButton *event)
358{
359 if ( event->button == 1)
360 {
361 if (!applet->active && !applet->pause && applet->elapsed)
362 timer_reset_callback (NULL, applet);
363 else if (applet->active && !applet->pause)
364 timer_pause_callback (NULL, applet);
365 else if (!applet->active || applet->pause)
366 timer_start_callback (NULL, applet);
367 return TRUE;
368 }
369 else
370 return FALSE;
371}

379static gboolean
380timer_applet_fill (MatePanelApplet* applet_widget)
381{
382 TimerApplet *applet;
383 AtkObject *atk_obj;
384
385#ifndef ENABLE_IN_PROCESS
386 g_set_application_name (_("Timer Applet"));
387#endif
388 gtk_window_set_default_icon_name (APPLET_ICON);
389
390 if (!notify_is_initted ())
391 notify_init ("timer-applet");
392
393 mate_panel_applet_set_flags (applet_widget, MATE_PANEL_APPLET_EXPAND_MINOR);
394
395 applet = g_malloc0(sizeof(TimerApplet));
396 applet->applet = applet_widget;
397 applet->settings = mate_panel_applet_settings_new (applet_widget,TIMER_SCHEMA);
398 applet->timeout_id = 0;
399 applet->active = FALSE;
400 applet->pause = FALSE;
401
402 applet->box = GTK_BOX (gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0));
403 applet->image = GTK_IMAGE (gtk_image_new_from_icon_name (APPLET_ICON, GTK_ICON_SIZE_BUTTON));
404 applet->pause_image = GTK_IMAGE (gtk_image_new_from_icon_name ("media-playback-pause", GTK_ICON_SIZE_BUTTON));
405 applet->label = GTK_LABEL (gtk_label_new (""));
406
407 atk_obj = gtk_widget_get_accessible (GTK_WIDGET (applet->applet));
408 if (GTK_IS_ACCESSIBLE (atk_obj)) {
409 atk_object_set_name (atk_obj, _("Timer Applet"));
410 atk_object_set_description (atk_obj, _("Start a timer and receive a notification when it is finished"));
411 }
412 /* we add the Gtk label into the applet */
413 gtk_box_pack_start (applet->box,
414 GTK_WIDGET (applet->image),
415 TRUE, TRUE, 0);
416 gtk_box_pack_start (applet->box,
417 GTK_WIDGET (applet->pause_image),
418 TRUE, TRUE, 0);
419 gtk_box_pack_start (applet->box,
420 GTK_WIDGET (applet->label),
421 TRUE, TRUE, 3);
422
423 gtk_container_add (GTK_CONTAINER (applet_widget),
424 GTK_WIDGET (applet->box));
425
426 gtk_widget_show_all (GTK_WIDGET (applet->applet));
427 gtk_widget_hide (GTK_WIDGET (applet->pause_image));
428
429 g_signal_connect (applet->applet, "destroy",
430 G_CALLBACK (timer_applet_destroy),
431 applet);
432
433 g_signal_connect_swapped (applet->applet, "button-press-event",
434 G_CALLBACK (timer_applet_click),
435 applet);
436
437 /* set up context menu */
438 applet->action_group = gtk_action_group_new ("Timer Applet Actions");
439 gtk_action_group_set_translation_domain (applet->action_group, GETTEXT_PACKAGE);
440 gtk_action_group_add_actions (applet->action_group, applet_menu_actions,
441 G_N_ELEMENTS (applet_menu_actions), applet);
442 mate_panel_applet_setup_menu (applet->applet, ui, applet->action_group);
443
444 /* execute callback to set actions sensitiveness */
445 timer_callback (applet);
446
447 /* GSettings callback */
448 g_signal_connect (applet->settings, "changed",
449 G_CALLBACK (timer_settings_changed),
450 applet);
451
452 return TRUE;
453}
