66static gchar*
67msd_ldsm_dialog_get_primary_text (MsdLdsmDialog *dialog)
68{
69 gchar *primary_text, *free_space;
70
71 g_return_val_if_fail (MSD_IS_LDSM_DIALOG (dialog), NULL);
72
73 free_space = g_format_size (dialog->priv->space_remaining);
74
75 if (dialog->priv->other_partitions) {
76 primary_text = g_strdup_printf (_("The volume \"%s\" has only %s disk space remaining."),
77 dialog->priv->partition_name, free_space);
78 } else {
79 primary_text = g_strdup_printf (_("This computer has only %s disk space remaining."),
80 free_space);
81 }
82
83 g_free (free_space);
84
85 return primary_text;

119static gboolean
120update_ignore_paths (GSList **ignore_paths,
121 const gchar *mount_path,
122 gboolean ignore)
123{
124 GSList *found;
125 gchar *path_to_remove;
126
127 found = g_slist_find_custom (*ignore_paths, mount_path, (GCompareFunc) ignore_path_compare);
128
129 if (ignore && (found == NULL)) {
130 *ignore_paths = g_slist_prepend (*ignore_paths, g_strdup (mount_path));
131 return TRUE;
132 }
133
134 if (!ignore && (found != NULL)) {
135 path_to_remove = found->data;
136 *ignore_paths = g_slist_remove (*ignore_paths, path_to_remove);
137 g_free (path_to_remove);
138 return TRUE;
139 }
140
141 return FALSE;
142}

144static void
145ignore_check_button_toggled_cb (GtkToggleButton *button,
146 gpointer user_data)
147{
148 MsdLdsmDialog *dialog = (MsdLdsmDialog *)user_data;
149 GSettings *settings;
150 gchar **settings_list;
151 GSList *ignore_paths = NULL;
152 gboolean ignore, updated;
153 gint i;
154
155 settings = g_settings_new (SETTINGS_SCHEMA);
156
157 settings_list = g_settings_get_strv (settings, SETTINGS_IGNORE_PATHS);
158
159 for (i = 0; i < g_strv_length (settings_list); i++) {
160 ignore_paths = g_slist_prepend (ignore_paths, g_strdup (settings_list[i]));
161 }
162 g_strfreev (settings_list);
163
164 if (i > 0)
165 ignore_paths = g_slist_reverse (ignore_paths);
166
167 ignore = gtk_toggle_button_get_active (button);
168 updated = update_ignore_paths (&ignore_paths, dialog->priv->mount_path, ignore);
169
170 if (updated) {
171 GSList *l;
172 GPtrArray *array = g_ptr_array_new ();
173
174 for (l = ignore_paths; l != NULL; l = l->next)
175 g_ptr_array_add (array, l->data);
176
177 g_ptr_array_add (array, NULL);
178
179 if (!g_settings_set_strv (settings, SETTINGS_IGNORE_PATHS, (const gchar **) array->pdata)) {
180 g_warning ("Cannot change ignore preference - failed to commit changes");
181 }
182
183 g_ptr_array_free (array, FALSE);
184 }
185
186 g_slist_free_full (ignore_paths, g_free);
187 g_object_unref (settings);
188}

192{

410msd_ldsm_dialog_new (gboolean other_usable_partitions,
411 gboolean other_partitions,
412 gboolean display_baobab,
413 gboolean display_empty_trash,

417{
418 MsdLdsmDialog *dialog;
419 GtkWidget *button_empty_trash, *button_ignore, *button_analyze;

421 gchar *primary_text, *primary_text_markup;
422 const gchar *secondary_text, *checkbutton_text;
423
424 dialog = MSD_LDSM_DIALOG (g_object_new (MSD_TYPE_LDSM_DIALOG,
425 "other-usable-partitions", other_usable_partitions,
426 "other-partitions", other_partitions,
427 "has-trash", display_empty_trash,
428 "space-remaining", space_remaining,
429 "partition-name", partition_name,

447 gtk_button_set_image (GTK_BUTTON (button_analyze), analyze_image);

451 _("Ignore"),
452 GTK_RESPONSE_CANCEL);
453 ignore_image = gtk_image_new_from_icon_name ("process-stop", GTK_ICON_SIZE_BUTTON);
454 gtk_button_set_image (GTK_BUTTON (button_ignore), ignore_image);

467 gtk_button_set_label (GTK_BUTTON (dialog->priv->ignore_check_button), checkbutton_text);
468
469 g_free (primary_text);
470 g_free (primary_text_markup);
471
472 return dialog;
