156EggDesktopFile *
157egg_desktop_file_new_from_key_file (GKeyFile *key_file,
158 const char *source,
159 GError **error)
160{
161 EggDesktopFile *desktop_file;
162 char *version, *type;
163
164 if (!g_key_file_has_group (key_file, EGG_DESKTOP_FILE_GROUP))
165 {
166 g_set_error (error, EGG_DESKTOP_FILE_ERROR,
167 EGG_DESKTOP_FILE_ERROR_INVALID,
168 _("File is not a valid .desktop file"));
169 g_key_file_free (key_file);
170 return NULL;
171 }
172
173 version = g_key_file_get_value (key_file, EGG_DESKTOP_FILE_GROUP,
174 EGG_DESKTOP_FILE_KEY_VERSION,
175 NULL);
176 if (version)
177 {
178 double version_num;
179 char *end;
180
181 version_num = g_ascii_strtod (version, &end);
182 if (*end)
183 {
184 g_warning ("Invalid Version string '%s' in %s",
185 version, source ? source : "(unknown)");
186 }
187 else if (version_num > 1.0)
188 {
189 /* Translators: Version here is capitalized because it refers to a
190 * field in the desktop file */
191 g_set_error (error, EGG_DESKTOP_FILE_ERROR,
192 EGG_DESKTOP_FILE_ERROR_INVALID,
193 _("Unrecognized desktop file Version '%s'"), version);
194 g_free (version);
195 g_key_file_free (key_file);
196 return NULL;
197 }
198 g_free (version);
199 }
200
201 desktop_file = g_new0 (EggDesktopFile, 1);
202 desktop_file->key_file = key_file;
203
204 if (g_path_is_absolute (source))
205 desktop_file->source = g_filename_to_uri (source, NULL, NULL);
206 else
207 desktop_file->source = g_strdup (source);
208
209 desktop_file->name = g_key_file_get_string (key_file, EGG_DESKTOP_FILE_GROUP,
210 EGG_DESKTOP_FILE_KEY_NAME, error);
211 if (!desktop_file->name)
212 {
213 egg_desktop_file_free (desktop_file);
214 return NULL;
215 }
216
217 type = g_key_file_get_string (key_file, EGG_DESKTOP_FILE_GROUP,
218 EGG_DESKTOP_FILE_KEY_TYPE, error);
219 if (!type)
220 {
221 egg_desktop_file_free (desktop_file);
222 return NULL;
223 }
224
225 if (!strcmp (type, "Application"))
226 {
227 char *exec, *p;
228
229 desktop_file->type = EGG_DESKTOP_FILE_TYPE_APPLICATION;
230
231 exec = g_key_file_get_string (key_file,
232 EGG_DESKTOP_FILE_GROUP,
233 EGG_DESKTOP_FILE_KEY_EXEC,
234 error);
235 if (!exec)
236 {
237 egg_desktop_file_free (desktop_file);
238 g_free (type);
239 return NULL;
240 }
241
242 /* See if it takes paths or URIs or neither */
243 for (p = exec; *p; p++)
244 {
245 if (*p == '%')
246 {
247 if (p[1] == '\0' || strchr ("FfUu", p[1]))
248 {
249 desktop_file->document_code = p[1];
250 break;
251 }
252 p++;
253 }
254 }
255
256 g_free (exec);
257 }
258 else if (!strcmp (type, "Link"))
259 {
260 char *url;
261
262 desktop_file->type = EGG_DESKTOP_FILE_TYPE_LINK;
263
264 url = g_key_file_get_string (key_file,
265 EGG_DESKTOP_FILE_GROUP,
266 EGG_DESKTOP_FILE_KEY_URL,
267 error);
268 if (!url)
269 {
270 egg_desktop_file_free (desktop_file);
271 g_free (type);
272 return NULL;
273 }
274 g_free (url);
275 }
276 else if (!strcmp (type, "Directory"))
277 desktop_file->type = EGG_DESKTOP_FILE_TYPE_DIRECTORY;
278 else
279 desktop_file->type = EGG_DESKTOP_FILE_TYPE_UNRECOGNIZED;
280
281 g_free (type);
282
283 /* Check the Icon key */
284 desktop_file->icon = g_key_file_get_string (key_file,
285 EGG_DESKTOP_FILE_GROUP,
286 EGG_DESKTOP_FILE_KEY_ICON,
287 NULL);
288 if (desktop_file->icon && !g_path_is_absolute (desktop_file->icon))
289 {
290 char *ext;
291
292 /* Lots of .desktop files still get this wrong */
293 ext = strrchr (desktop_file->icon, '.');
294 if (ext && (!strcmp (ext, ".png") ||
295 !strcmp (ext, ".xpm") ||
296 !strcmp (ext, ".svg")))
297 {
298 g_warning ("Desktop file '%s' has malformed Icon key '%s'"
299 "(should not include extension)",
300 source ? source : "(unknown)",
301 desktop_file->icon);
302 *ext = '\0';
303 }
304 }
305
306 return desktop_file;
307}

493gboolean
494egg_desktop_file_can_launch (EggDesktopFile *desktop_file,
495 const char *desktop_environment)
496{
497 char *try_exec, *found_program;
498 char **only_show_in, **not_show_in;
499 gboolean found;
500 int i;
501
502 if (desktop_file->type != EGG_DESKTOP_FILE_TYPE_APPLICATION &&
503 desktop_file->type != EGG_DESKTOP_FILE_TYPE_LINK)
504 return FALSE;
505
506 if (desktop_environment)
507 {
508 only_show_in = g_key_file_get_string_list (desktop_file->key_file,
509 EGG_DESKTOP_FILE_GROUP,
510 EGG_DESKTOP_FILE_KEY_ONLY_SHOW_IN,
511 NULL, NULL);
512 if (only_show_in)
513 {
514 for (i = 0, found = FALSE; only_show_in[i] && !found; i++)
515 {
516 if (!strcmp (only_show_in[i], desktop_environment))
517 found = TRUE;
518 }
519
520 g_strfreev (only_show_in);
521
522 if (!found)
523 return FALSE;
524 }
525
526 not_show_in = g_key_file_get_string_list (desktop_file->key_file,
527 EGG_DESKTOP_FILE_GROUP,
528 EGG_DESKTOP_FILE_KEY_NOT_SHOW_IN,
529 NULL, NULL);
530 if (not_show_in)
531 {
532 for (i = 0, found = FALSE; not_show_in[i] && !found; i++)
533 {
534 if (!strcmp (not_show_in[i], desktop_environment))
535 found = TRUE;
536 }
537
538 g_strfreev (not_show_in);
539
540 if (found)
541 return FALSE;
542 }
543 }
544
545 if (desktop_file->type == EGG_DESKTOP_FILE_TYPE_APPLICATION)
546 {
547 try_exec = g_key_file_get_string (desktop_file->key_file,
548 EGG_DESKTOP_FILE_GROUP,
549 EGG_DESKTOP_FILE_KEY_TRY_EXEC,
550 NULL);
551 if (try_exec)
552 {
553 found_program = g_find_program_in_path (try_exec);
554 g_free (try_exec);
555
556 if (!found_program)
557 return FALSE;
558 g_free (found_program);
559 }
560 }
561
562 return TRUE;
563}

616static void
617append_quoted_word (GString *str,
618 const char *s,
619 gboolean in_single_quotes,
620 gboolean in_double_quotes)
621{
622 const char *p;
623
624 if (!in_single_quotes && !in_double_quotes)
625 g_string_append_c (str, '\'');
626 else if (!in_single_quotes && in_double_quotes)
627 g_string_append (str, "\"'");
628
629 if (!strchr (s, '\''))
630 g_string_append (str, s);
631 else
632 {
633 for (p = s; *p != '\0'; p++)
634 {
635 if (*p == '\'')
636 g_string_append (str, "'\\''");
637 else
638 g_string_append_c (str, *p);
639 }
640 }
641
642 if (!in_single_quotes && !in_double_quotes)
643 g_string_append_c (str, '\'');
644 else if (!in_single_quotes && in_double_quotes)
645 g_string_append (str, "'\"");
646}

648static void
649do_percent_subst (EggDesktopFile *desktop_file,
650 char code,
651 GString *str,
652 GSList **documents,
653 gboolean in_single_quotes,
654 gboolean in_double_quotes)
655{
656 GSList *d;
657 char *doc;
658
659 switch (code)
660 {
661 case '%':
662 g_string_append_c (str, '%');
663 break;
664
665 case 'F':
666 case 'U':
667 for (d = *documents; d; d = d->next)
668 {
669 doc = d->data;
670 g_string_append (str, " ");
671 append_quoted_word (str, doc, in_single_quotes, in_double_quotes);
672 }
673 *documents = NULL;
674 break;
675
676 case 'f':
677 case 'u':
678 if (*documents)
679 {
680 doc = (*documents)->data;
681 g_string_append (str, " ");
682 append_quoted_word (str, doc, in_single_quotes, in_double_quotes);
683 *documents = (*documents)->next;
684 }
685 break;
686
687 case 'i':
688 if (desktop_file->icon)
689 {
690 g_string_append (str, "--icon ");
691 append_quoted_word (str, desktop_file->icon,
692 in_single_quotes, in_double_quotes);
693 }
694 break;
695
696 case 'c':
697 if (desktop_file->name)
698 {
699 append_quoted_word (str, desktop_file->name,
700 in_single_quotes, in_double_quotes);
701 }
702 break;
703
704 case 'k':
705 if (desktop_file->source)
706 {
707 append_quoted_word (str, desktop_file->source,
708 in_single_quotes, in_double_quotes);
709 }
710 break;
711
712 case 'D':
713 case 'N':
714 case 'd':
715 case 'n':
716 case 'v':
717 case 'm':
718 /* Deprecated; skip */
719 break;
720
721 default:
722 g_warning ("Unrecognized %%-code '%%%c' in Exec", code);
723 break;
724 }
725}

727static char *
728parse_exec (EggDesktopFile *desktop_file,
729 GSList **documents,
730 GError **error)
731{
732 char *exec, *p, *command;
733 gboolean escape, single_quot, double_quot;
734 GString *gs;
735
736 exec = g_key_file_get_string (desktop_file->key_file,
737 EGG_DESKTOP_FILE_GROUP,
738 EGG_DESKTOP_FILE_KEY_EXEC,
739 error);
740 if (!exec)
741 return NULL;
742
743 /* Build the command */
744 gs = g_string_new (NULL);
745 escape = single_quot = double_quot = FALSE;
746
747 for (p = exec; *p != '\0'; p++)
748 {
749 if (escape)
750 {
751 escape = FALSE;
752 g_string_append_c (gs, *p);
753 }
754 else if (*p == '\\')
755 {
756 if (!single_quot)
757 escape = TRUE;
758 g_string_append_c (gs, *p);
759 }
760 else if (*p == '\'')
761 {
762 g_string_append_c (gs, *p);
763 if (!single_quot && !double_quot)
764 single_quot = TRUE;
765 else if (single_quot)
766 single_quot = FALSE;
767 }
768 else if (*p == '"')
769 {
770 g_string_append_c (gs, *p);
771 if (!single_quot && !double_quot)
772 double_quot = TRUE;
773 else if (double_quot)
774 double_quot = FALSE;
775 }
776 else if (*p == '%' && p[1])
777 {
778 do_percent_subst (desktop_file, p[1], gs, documents,
779 single_quot, double_quot);
780 p++;
781 }
782 else
783 g_string_append_c (gs, *p);
784 }
785
786 g_free (exec);
787 command = g_string_free (gs, FALSE);
788
789 /* Prepend "xdg-terminal " if needed (FIXME: use gvfs) */
790 if (g_key_file_has_key (desktop_file->key_file,
791 EGG_DESKTOP_FILE_GROUP,
792 EGG_DESKTOP_FILE_KEY_TERMINAL,
793 NULL))
794 {
795 GError *terminal_error = NULL;
796 gboolean use_terminal =
797 g_key_file_get_boolean (desktop_file->key_file,
798 EGG_DESKTOP_FILE_GROUP,
799 EGG_DESKTOP_FILE_KEY_TERMINAL,
800 &terminal_error);
801 if (terminal_error)
802 {
803 g_free (command);
804 g_propagate_error (error, terminal_error);
805 return NULL;
806 }
807
808 if (use_terminal)
809 {
810 gs = g_string_new ("xdg-terminal ");
811 append_quoted_word (gs, command, FALSE, FALSE);
812 g_free (command);
813 command = g_string_free (gs, FALSE);
814 }
815 }
816
817 return command;
818}

820static GSList *
821translate_document_list (EggDesktopFile *desktop_file, GSList *documents)
822{
823 gboolean accepts_uris = egg_desktop_file_accepts_uris (desktop_file);
824 GSList *ret, *d;
825
826 for (d = documents, ret = NULL; d; d = d->next)
827 {
828 const char *document = d->data;
829 gboolean is_uri = !g_path_is_absolute (document);
830 char *translated;
831
832 if (accepts_uris)
833 {
834 if (is_uri)
835 translated = g_strdup (document);
836 else
837 translated = g_filename_to_uri (document, NULL, NULL);
838 }
839 else
840 {
841 if (is_uri)
842 translated = g_filename_from_uri (document, NULL, NULL);
843 else
844 translated = g_strdup (document);
845 }
846
847 if (translated)
848 ret = g_slist_prepend (ret, translated);
849 }
850
851 return g_slist_reverse (ret);
852}

854static void
855free_document_list (GSList *documents)
856{
857 GSList *d;
858
859 for (d = documents; d; d = d->next)
860 g_free (d->data);
861 g_slist_free (documents);
862}

880char *
881egg_desktop_file_parse_exec (EggDesktopFile *desktop_file,
882 GSList *documents,
883 GError **error)
884{
885 GSList *translated, *docs;
886 char *command;
887
888 docs = translated = translate_document_list (desktop_file, documents);
889 command = parse_exec (desktop_file, &docs, error);
890 free_document_list (translated);
891
892 return command;

895static gboolean
896parse_link (EggDesktopFile *desktop_file,
897 EggDesktopFile **app_desktop_file,
898 GSList **documents,
899 GError **error)
900{
901 char *url;
902 GKeyFile *key_file;
903
904 url = g_key_file_get_string (desktop_file->key_file,
905 EGG_DESKTOP_FILE_GROUP,
906 EGG_DESKTOP_FILE_KEY_URL,
907 error);
908 if (!url)
909 return FALSE;
910 *documents = g_slist_prepend (NULL, url);
911
912 /* FIXME: use gvfs */
913 key_file = g_key_file_new ();
914 g_key_file_set_string (key_file, EGG_DESKTOP_FILE_GROUP,
915 EGG_DESKTOP_FILE_KEY_NAME,
916 "xdg-open");
917 g_key_file_set_string (key_file, EGG_DESKTOP_FILE_GROUP,
918 EGG_DESKTOP_FILE_KEY_TYPE,
919 "Application");
920 g_key_file_set_string (key_file, EGG_DESKTOP_FILE_GROUP,
921 EGG_DESKTOP_FILE_KEY_EXEC,
922 "xdg-open %u");
923 *app_desktop_file = egg_desktop_file_new_from_key_file (key_file, NULL, NULL);
924 return TRUE;
925}

929start_startup_notification (GdkDisplay *display,

1042static GPtrArray *
1043array_putenv (GPtrArray *env, char *variable)
1044{
1045 guint i, keylen;
1046
1047 if (!env)
1048 {
1049 char **envp;
1050
1051 env = g_ptr_array_new ();
1052
1053 envp = g_listenv ();
1054 for (i = 0; envp[i]; i++)
1055 {
1056 const char *value;
1057
1058 value = g_getenv (envp[i]);
1059 g_ptr_array_add (env, g_strdup_printf ("%s=%s", envp[i],
1060 value ? value : ""));
1061 }
1062 g_strfreev (envp);
1063 }
1064
1065 keylen = strcspn (variable, "=");
1066
1067 /* Remove old value of key */
1068 for (i = 0; i < env->len; i++)
1069 {
1070 char *envvar = env->pdata[i];
1071
1072 if (!strncmp (envvar, variable, keylen) && envvar[keylen] == '=')
1073 {
1074 g_free (envvar);
1075 g_ptr_array_remove_index_fast (env, i);
1076 break;
1077 }
1078 }
1079
1080 /* Add new value */
1081 g_ptr_array_add (env, g_strdup (variable));
1082
1083 return env;
1084}

1086static gboolean
1087egg_desktop_file_launchv (EggDesktopFile *desktop_file,
1088 GSList *documents, va_list args,
1089 GError **error)
1090{
1091 EggDesktopFileLaunchOption option;
1092 GSList *translated_documents = NULL, *docs = NULL;
1093 char *command, **argv;
1094 int argc, i;
1095 gboolean success, current_success;
1096 GdkDisplay *display;
1097 char *startup_id;
1098
1099 GPtrArray *env = NULL;
1100 char **variables = NULL;
1101 GdkScreen *screen = NULL;
1102 int workspace = -1;
1103 const char *directory = NULL;
1104 guint32 launch_time = (guint32)-1;
1105 GSpawnFlags flags = G_SPAWN_SEARCH_PATH;
1106 GSpawnChildSetupFunc setup_func = NULL;
1107 gpointer setup_data = NULL;
1108
1109 GPid *ret_pid = NULL;
1110 int *ret_stdin = NULL, *ret_stdout = NULL, *ret_stderr = NULL;
1111 char **ret_startup_id = NULL;
1112
1113 if (documents && desktop_file->document_code == 0)
1114 {
1115 g_set_error (error, EGG_DESKTOP_FILE_ERROR,
1116 EGG_DESKTOP_FILE_ERROR_NOT_LAUNCHABLE,
1117 _("Application does not accept documents on command line"));
1118 return FALSE;
1119 }
1120
1121 /* Read the options: technically it's incorrect for the caller to
1122 * NULL-terminate the list of options (rather than 0-terminating
1123 * it), but NULL-terminating lets us use G_GNUC_NULL_TERMINATED,
1124 * it's more consistent with other glib/gtk methods, and it will
1125 * work as long as sizeof (int) <= sizeof (NULL), and NULL is
1126 * represented as 0. (Which is true everywhere we care about.)
1127 */
1128 while ((option = va_arg (args, EggDesktopFileLaunchOption)))
1129 {
1130 switch (option)
1131 {
1132 case EGG_DESKTOP_FILE_LAUNCH_CLEARENV:
1133 if (env)
1134 g_ptr_array_free (env, TRUE);
1135 env = g_ptr_array_new ();
1136 break;
1137 case EGG_DESKTOP_FILE_LAUNCH_PUTENV:
1138 variables = va_arg (args, char **);
1139 for (i = 0; variables[i]; i++)
1140 env = array_putenv (env, variables[i]);
1141 break;
1142
1143 case EGG_DESKTOP_FILE_LAUNCH_SCREEN:
1144 screen = va_arg (args, GdkScreen *);
1145 break;
1146 case EGG_DESKTOP_FILE_LAUNCH_WORKSPACE:
1147 workspace = va_arg (args, int);
1148 break;
1149
1150 case EGG_DESKTOP_FILE_LAUNCH_DIRECTORY:
1151 directory = va_arg (args, const char *);
1152 break;
1153 case EGG_DESKTOP_FILE_LAUNCH_TIME:
1154 launch_time = va_arg (args, guint32);
1155 break;
1156 case EGG_DESKTOP_FILE_LAUNCH_FLAGS:
1157 flags |= va_arg (args, GSpawnFlags);
1158 /* Make sure they didn't set any flags that don't make sense. */
1159 flags &= ~G_SPAWN_FILE_AND_ARGV_ZERO;
1160 break;
1161 case EGG_DESKTOP_FILE_LAUNCH_SETUP_FUNC:
1162 setup_func = va_arg (args, GSpawnChildSetupFunc);
1163 setup_data = va_arg (args, gpointer);
1164 break;
1165
1166 case EGG_DESKTOP_FILE_LAUNCH_RETURN_PID:
1167 ret_pid = va_arg (args, GPid *);
1168 break;
1169 case EGG_DESKTOP_FILE_LAUNCH_RETURN_STDIN_PIPE:
1170 ret_stdin = va_arg (args, int *);
1171 break;
1172 case EGG_DESKTOP_FILE_LAUNCH_RETURN_STDOUT_PIPE:
1173 ret_stdout = va_arg (args, int *);
1174 break;
1175 case EGG_DESKTOP_FILE_LAUNCH_RETURN_STDERR_PIPE:
1176 ret_stderr = va_arg (args, int *);
1177 break;
1178 case EGG_DESKTOP_FILE_LAUNCH_RETURN_STARTUP_ID:
1179 ret_startup_id = va_arg (args, char **);
1180 break;
1181
1182 default:
1183 g_set_error (error, EGG_DESKTOP_FILE_ERROR,
1184 EGG_DESKTOP_FILE_ERROR_UNRECOGNIZED_OPTION,
1185 _("Unrecognized launch option: %d"),
1186 GPOINTER_TO_INT (option));
1187 success = FALSE;
1188 goto out;
1189 }
1190 }
1191
1192 if (screen)
1193 {
1194 display = gdk_screen_get_display (screen);
1195 char *display_name = g_strdup (gdk_display_get_name (display));
1196 char *display_env = g_strdup_printf ("DISPLAY=%s", display_name);
1197 env = array_putenv (env, display_env);
1198 g_free (display_name);
1199 g_free (display_env);
1200 }
1201 else
1202 {
1203 display = gdk_display_get_default ();
1204 screen = gdk_display_get_default_screen (display);
1205 }
1206
1207 translated_documents = translate_document_list (desktop_file, documents);
1208 docs = translated_documents;
1209
1210 success = FALSE;
1211
1212 do
1213 {
1214 command = parse_exec (desktop_file, &docs, error);
1215 if (!command)
1216 goto out;
1217
1218 if (!g_shell_parse_argv (command, &argc, &argv, error))
1219 {
1220 g_free (command);
1221 goto out;
1222 }
1223 g_free (command);
1224
1225 startup_id = NULL;
1226
1227#if defined(GDK_WINDOWING_X11) || defined(HAVE_X11)
1228 if (GDK_IS_X11_DISPLAY (display))
1229 {
1230 int screen_num;
1231
1232 screen_num = gdk_x11_screen_get_screen_number (screen);
1233 startup_id = start_startup_notification (display, desktop_file,
1234 argv[0], screen_num,
1235 workspace, launch_time);
1236 if (startup_id)
1237 {
1238 char *startup_id_env = g_strdup_printf ("DESKTOP_STARTUP_ID=%s",
1239 startup_id);
1240 env = array_putenv (env, startup_id_env);
1241 g_free (startup_id_env);
1242 }
1243 }
1244#else
1245 // Suppress unused variable warnings when not compiling with X
1246 (void)workspace;
1247 (void)launch_time;
1248#endif
1249
1250 if (env != NULL)
1251 g_ptr_array_add (env, NULL);
1252
1253 current_success =
1254 g_spawn_async_with_pipes (directory,
1255 argv,
1256 env ? (char **)(env->pdata) : NULL,
1257 flags,
1258 setup_func, setup_data,
1259 ret_pid,
1260 ret_stdin, ret_stdout, ret_stderr,
1261 error);
1262 g_strfreev (argv);
1263
1264 if (startup_id)
1265 {
1266 if (current_success)
1267 {
1268#if defined(GDK_WINDOWING_X11) || defined(HAVE_X11)
1269 if (GDK_IS_X11_DISPLAY (display))
1270 {
1271 set_startup_notification_timeout (display, startup_id);
1272 }
1273#endif
1274
1275 if (ret_startup_id)
1276 *ret_startup_id = startup_id;
1277 else
1278 g_free (startup_id);
1279 }
1280 else
1281 g_free (startup_id);
1282 }
1283 else if (ret_startup_id)
1284 *ret_startup_id = NULL;
1285
1286 if (current_success)
1287 {
1288 /* If we successfully launch any instances of the app, make
1289 * sure we return TRUE and don't set @error.
1290 */
1291 success = TRUE;
1292 error = NULL;
1293
1294 /* Also, only set the output params on the first one */
1295 ret_pid = NULL;
1296 ret_stdin = ret_stdout = ret_stderr = NULL;
1297 ret_startup_id = NULL;
1298 }
1299 }
1300 while (docs && current_success);
1301
1302 out:
1303 if (env)
1304 {
1305 g_ptr_array_set_free_func (env, g_free);
1306 g_ptr_array_free (env, TRUE);
1307 }
1308 free_document_list (translated_documents);
1309
1310 return success;
1311}
