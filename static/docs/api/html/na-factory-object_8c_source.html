159/*
160 * na_factory_object_get_data_def:
161 * @object: this #NAIFactoryObject object.
162 * @name: the searched name.
163 *
164 * Returns: the #NADataDef structure which describes this @name, or %NULL.
165 */
166NADataDef *
167na_factory_object_get_data_def( const NAIFactoryObject *object, const gchar *name )
168{
169 NADataDef *def = NULL;
170
171 g_return_val_if_fail( NA_IS_IFACTORY_OBJECT( object ), NULL );
172
173 NADataGroup *groups = v_get_groups( object );
174 while( groups->group ){
175 def = groups->def;
176 if( def ){
177 while( def->name ){
178
179 if( !strcmp( def->name, name )){
180 return( def );
181 }
182 def++;
183 }
184 }
185 groups++;
186 }
187
188 return( def );
189}

296/*
297 * because this function is called very early in the NAIFactoryObject life,
298 * we assume here that if a NADataBoxed has been allocated, then this is
299 * most probably because it is set. Thus a 'null' value is not considered
300 * as an 'unset' value.
301 */
302static gboolean
303set_defaults_iter( NADataDef *def, NafoDefaultIter *data )
304{
305 NADataBoxed *boxed = na_ifactory_object_get_data_boxed( data->object, def->name );
306
307 if( !boxed ){
308 boxed = na_data_boxed_new( def );
309 attach_boxed_to_object( data->object, boxed );
310 na_boxed_set_from_string( NA_BOXED( boxed ), def->default_value );
311 }
312
313 /* do not stop */
314 return( FALSE );
315}

318 * na_factory_object_move_boxed:
319 * @target: the target #NAIFactoryObject instance.
320 * @source: the source #NAIFactoryObject instance.
321 * @boxed: a #NADataBoxed.
322 *
323 * Move the @boxed from @source to @target, detaching from @source list
324 * to be attached to @target one.

326void
327na_factory_object_move_boxed( NAIFactoryObject *target, const NAIFactoryObject *source, NADataBoxed *boxed )
328{

330 g_return_if_fail( NA_IS_IFACTORY_OBJECT( source ));

335 src_list = g_list_remove( src_list, boxed );
336 g_object_set_data( G_OBJECT( source ), NA_IFACTORY_OBJECT_PROP_DATA, src_list );

339

341 NADataDef *tgt_def = na_factory_object_get_data_def( target, src_def->name );
342 na_data_boxed_set_data_def( boxed, tgt_def );
343 }

346/*
347 * na_factory_object_copy:
348 * @target: the target #NAIFactoryObject instance.
349 * @source: the source #NAIFactoryObject instance.
350 *
351 * Copies one instance to another.
352 * Takes care of not overriding provider data.
353 */
354void
355na_factory_object_copy( NAIFactoryObject *target, const NAIFactoryObject *source )
356{
357 static const gchar *thisfn = "na_factory_object_copy";
358 GList *dest_list, *idest, *inext;
359 GList *src_list, *isrc;
360 NADataBoxed *boxed;
361 const NADataDef *def;
362 void *provider, *provider_data;
363
364 g_return_if_fail( NA_IS_IFACTORY_OBJECT( target ));
365 g_return_if_fail( NA_IS_IFACTORY_OBJECT( source ));
366
367 g_debug( "%s: target=%p (%s), source=%p (%s)",
368 thisfn,
369 ( void * ) target, G_OBJECT_TYPE_NAME( target ),
370 ( void * ) source, G_OBJECT_TYPE_NAME( source ));
371
372 /* first remove copyable data from target
373 */
374 provider = na_object_get_provider( target );
375 provider_data = na_object_get_provider_data( target );
376
377 idest = dest_list = g_object_get_data( G_OBJECT( target ), NA_IFACTORY_OBJECT_PROP_DATA );
378 while( idest ){
379 boxed = NA_DATA_BOXED( idest->data );
380 inext = idest->next;
381 def = na_data_boxed_get_data_def( boxed );
382 if( def->copyable ){
383 dest_list = g_list_remove_link( dest_list, idest );
384 g_object_unref( idest->data );
385 }
386 idest = inext;
387 }
388 g_object_set_data( G_OBJECT( target ), NA_IFACTORY_OBJECT_PROP_DATA, dest_list );
389
390 /* only then copy copyable data from source
391 */
392 src_list = g_object_get_data( G_OBJECT( source ), NA_IFACTORY_OBJECT_PROP_DATA );
393 for( isrc = src_list ; isrc ; isrc = isrc->next ){
394 boxed = NA_DATA_BOXED( isrc->data );
395 def = na_data_boxed_get_data_def( boxed );
396 if( def->copyable ){
397 NADataBoxed *tgt_boxed = na_ifactory_object_get_data_boxed( target, def->name );
398 if( !tgt_boxed ){
399 tgt_boxed = na_data_boxed_new( def );
400 attach_boxed_to_object( target, tgt_boxed );
401 }
402 na_boxed_set_from_boxed( NA_BOXED( tgt_boxed ), NA_BOXED( boxed ));
403 }
404 }
405
406 if( provider ){
407 na_object_set_provider( target, provider );
408 if( provider_data ){
409 na_object_set_provider_data( target, provider_data );
410 }
411 }
412
413 v_copy( target, source );
414}

416/*
417 * na_factory_object_are_equal:
418 * @a: the first (original) #NAIFactoryObject instance.
419 * @b: the second (current) #NAIFactoryObject isntance.
420 *
421 * Returns: %TRUE if @a is equal to @b, %FALSE else.
422 */
423gboolean
424na_factory_object_are_equal( const NAIFactoryObject *a, const NAIFactoryObject *b )
425{
426 static const gchar *thisfn = "na_factory_object_are_equal";
427 gboolean are_equal;
428 GList *a_list, *b_list, *ia, *ib;
429
430 are_equal = FALSE;
431
432 a_list = g_object_get_data( G_OBJECT( a ), NA_IFACTORY_OBJECT_PROP_DATA );
433 b_list = g_object_get_data( G_OBJECT( b ), NA_IFACTORY_OBJECT_PROP_DATA );
434
435 g_debug( "%s: a=%p, b=%p", thisfn, ( void * ) a, ( void * ) b );
436
437 are_equal = TRUE;
438 for( ia = a_list ; ia && are_equal ; ia = ia->next ){
439
440 NADataBoxed *a_boxed = NA_DATA_BOXED( ia->data );
441 const NADataDef *a_def = na_data_boxed_get_data_def( a_boxed );
442 if( a_def->comparable ){
443
444 NADataBoxed *b_boxed = na_ifactory_object_get_data_boxed( b, a_def->name );
445 if( b_boxed ){
446 are_equal = na_boxed_are_equal( NA_BOXED( a_boxed ), NA_BOXED( b_boxed ));
447 if( !are_equal ){
448 g_debug( "%s: %s not equal as %s different", thisfn, G_OBJECT_TYPE_NAME( a ), a_def->name );
449 }
450
451 } else {
452 are_equal = FALSE;
453 g_debug( "%s: %s not equal as %s has disappeared", thisfn, G_OBJECT_TYPE_NAME( a ), a_def->name );
454 }
455 }
456 }
457
458 for( ib = b_list ; ib && are_equal ; ib = ib->next ){
459
460 NADataBoxed *b_boxed = NA_DATA_BOXED( ib->data );
461 const NADataDef *b_def = na_data_boxed_get_data_def( b_boxed );
462 if( b_def->comparable ){
463
464 NADataBoxed *a_boxed = na_ifactory_object_get_data_boxed( a, b_def->name );
465 if( !a_boxed ){
466 are_equal = FALSE;
467 g_debug( "%s: %s not equal as %s was not set", thisfn, G_OBJECT_TYPE_NAME( a ), b_def->name );
468 }
469 }
470 }
471
472 are_equal &= v_are_equal( a, b );
473
474 return( are_equal );
475}

477/*
478 * na_factory_object_is_valid:
479 * @object: the #NAIFactoryObject instance whose validity is to be checked.
480 *
481 * Returns: %TRUE if @object is valid, %FALSE else.
482 */
483gboolean
484na_factory_object_is_valid( const NAIFactoryObject *object )
485{
486 static const gchar *thisfn = "na_factory_object_is_valid";
487 gboolean is_valid;
488 NADataGroup *groups;
489 GList *list, *iv;
490
491 g_return_val_if_fail( NA_IS_IFACTORY_OBJECT( object ), FALSE );
492
493 g_debug( "%s: object=%p (%s)", thisfn, ( void * ) object, G_OBJECT_TYPE_NAME( object ));
494
495 list = g_object_get_data( G_OBJECT( object ), NA_IFACTORY_OBJECT_PROP_DATA );
496 is_valid = TRUE;
497
498 /* mandatory data must be set
499 */
500 NafoValidIter iter_data;
501 iter_data.object = ( NAIFactoryObject * ) object;
502 iter_data.is_valid = TRUE;
503
504 groups = v_get_groups( object );
505 if( groups ){
506 iter_on_data_defs( groups, DATA_DEF_ITER_IS_VALID, ( NADataDefIterFunc ) is_valid_mandatory_iter, &iter_data );
507 }
508 is_valid = iter_data.is_valid;
509
510 for( iv = list ; iv && is_valid ; iv = iv->next ){
511 is_valid = na_data_boxed_is_valid( NA_DATA_BOXED( iv->data ));
512 }
513
514 is_valid &= v_is_valid( object );
515
516 return( is_valid );
517}
518
519static gboolean
520is_valid_mandatory_iter( const NADataDef *def, NafoValidIter *data )
521{
522 NADataBoxed *boxed;
523
524 if( def->mandatory ){
525 boxed = na_ifactory_object_get_data_boxed( data->object, def->name );
526 if( !boxed ){
527 g_debug( "na_factory_object_is_valid_mandatory_iter: invalid %s: mandatory but not set", def->name );
528 data->is_valid = FALSE;
529 }
530 }
531
532 /* do not stop iteration while valid */
533 return( !data->is_valid );
534}

538 * @object: this #NAIFactoryObject instance.
539 *
540 * Dumps the content of @object.
541 */
542void
543na_factory_object_dump( const NAIFactoryObject *object )
544{
545 static const gchar *thisfn = "na_factory_object_dump";
546 static const gchar *prefix = "na-factory-data-";
547 GList *list, *it;
548 guint length;
549 guint l_prefix;
550
551 length = 0;
552 l_prefix = strlen( prefix );
553 list = g_object_get_data( G_OBJECT( object ), NA_IFACTORY_OBJECT_PROP_DATA );
554
555 for( it = list ; it ; it = it->next ){
556 NADataBoxed *boxed = NA_DATA_BOXED( it->data );
557 const NADataDef *def = na_data_boxed_get_data_def( boxed );
558 length = MAX( length, strlen( def->name ));
559 }
560
561 length -= l_prefix;
562 length += 1;
563
564 for( it = list ; it ; it = it->next ){
565 NADataBoxed *boxed = NA_DATA_BOXED( it->data );
566 const NADataDef *def = na_data_boxed_get_data_def( boxed );
567 gchar *value = na_boxed_get_string( NA_BOXED( boxed ));
568 g_debug( "| %s: %*s=%s", thisfn, length, def->name+l_prefix, value );
569 g_free( value );
570 }
571}

585/*
586 * na_factory_object_read_item:
587 * @object: this #NAIFactoryObject instance.
588 * @reader: the #NAIFactoryProvider which is at the origin of this read.
589 * @reader_data: reader data.
590 * @messages: a pointer to a #GSList list of strings; the implementation
591 * may append messages to this list, but shouldn't reinitialize it.
592 *
593 * Unserializes the object.
594 */
595void
596na_factory_object_read_item( NAIFactoryObject *object, const NAIFactoryProvider *reader, void *reader_data, GSList **messages )
597{
598 static const gchar *thisfn = "na_factory_object_read_item";
599
600 g_return_if_fail( NA_IS_IFACTORY_OBJECT( object ));
601 g_return_if_fail( NA_IS_IFACTORY_PROVIDER( reader ));
602
603 NADataGroup *groups = v_get_groups( object );
604
605 if( groups ){
606 v_read_start( object, reader, reader_data, messages );
607
608 NafoReadIter *iter = g_new0( NafoReadIter, 1 );
609 iter->object = object;
610 iter->reader = ( NAIFactoryProvider * ) reader;
611 iter->reader_data = reader_data;
612 iter->messages = messages;
613
614 iter_on_data_defs( groups, DATA_DEF_ITER_READ_ITEM, ( NADataDefIterFunc ) read_data_iter, iter );
615
616 g_free( iter );
617
618 v_read_done( object, reader, reader_data, messages );
619
620 } else {
621 g_warning( "%s: class %s doesn't return any NADataGroup structure",
622 thisfn, G_OBJECT_TYPE_NAME( object ));
623 }
624}

626static gboolean
627read_data_iter( NADataDef *def, NafoReadIter *iter )
628{
629 gboolean stop;
630
631 stop = FALSE;
632
633 NADataBoxed *boxed = na_factory_provider_read_data( iter->reader, iter->reader_data, iter->object, def, iter->messages );
634
635 if( boxed ){
636 NADataBoxed *exist = na_ifactory_object_get_data_boxed( iter->object, def->name );
637
638 if( exist ){
639 na_boxed_set_from_boxed( NA_BOXED( exist ), NA_BOXED( boxed ));
640 g_object_unref( boxed );
641
642 } else {
643 attach_boxed_to_object( iter->object, boxed );
644 }
645 }
646
647 return( stop );
648}

650/*
651 * na_factory_object_write_item:
652 * @object: this #NAIFactoryObject instance.
653 * @writer: the #NAIFactoryProvider which is at the origin of this write.
654 * @writer_data: writer data.
655 * @messages: a pointer to a #GSList list of strings; the implementation

658 * Serializes the object down to the @writer.

664{
665 static const gchar *thisfn = "na_factory_object_write_item";
666 guint code;
667 NADataGroup *groups;
668 gchar *msg;
669

682 thisfn, G_OBJECT_TYPE_NAME( object ));

685 }
686

690 iter->writer = ( NAIFactoryProvider * ) writer;
691 iter->writer_data = writer_data;
692 iter->messages = messages;
693 iter->code = code;
694

696
697 code = iter->code;
698 g_free( iter );
699 }
700
701 if( code == NA_IIO_PROVIDER_CODE_OK ){
702 code = v_write_done( object, writer, writer_data, messages );
703 }
