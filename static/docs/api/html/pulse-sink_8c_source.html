118PulseSink *
119pulse_sink_new (PulseConnection *connection,
120 const pa_sink_info *info,
121 PulseDevice *device)
122{
123 PulseSink *sink;
124
125 g_return_val_if_fail (PULSE_IS_CONNECTION (connection), NULL);
126 g_return_val_if_fail (info != NULL, NULL);
127 g_return_val_if_fail (device == NULL || PULSE_IS_DEVICE (device), NULL);
128
129 sink = g_object_new (PULSE_TYPE_SINK,
130 "name", info->name,
131 "label", info->description,
132 "device", device,
133 "direction", MATE_MIXER_DIRECTION_OUTPUT,
134 "connection", connection,
135 "index", info->index,
136 NULL);
137
138 sink->priv->control = pulse_sink_control_new (connection, info, sink);
139
140 if (info->n_ports > 0) {
141 pa_sink_port_info **ports = info->ports;
142
143 /* Create the port switch */
144 sink->priv->pswitch = pulse_sink_switch_new ("port", _("Connector"), sink);
145
146 while (*ports != NULL) {
147 pa_sink_port_info *p = *ports++;
148 PulsePort *port;
149 const gchar *icon = NULL;
150
151 /* A port may include an icon but in PulseAudio sink and source ports
152 * the property is not included, for this reason ports are also read from
153 * devices where the icons may be present */
154 if (device != NULL) {
155 port = pulse_device_get_port (PULSE_DEVICE (device), p->name);
156 if (port != NULL)
157 icon = mate_mixer_switch_option_get_icon (MATE_MIXER_SWITCH_OPTION (port));
158 }
159
160 port = pulse_port_new (p->name,
161 p->description,
162 icon,
163 p->priority);
164
165 pulse_port_switch_add_port (sink->priv->pswitch, port);
166
167 if (p == info->active_port)
168 pulse_port_switch_set_active_port (sink->priv->pswitch, port);
169 }
170 sink->priv->pswitch_list = g_list_prepend (NULL, sink->priv->pswitch);
171
172 g_debug ("Created port list for sink %s", info->name);
173 }
174
175 pulse_sink_update (sink, info);
176
177 _mate_mixer_stream_set_default_control (MATE_MIXER_STREAM (sink),
178 MATE_MIXER_STREAM_CONTROL (sink->priv->control));
179 return sink;
180}

182gboolean
183pulse_sink_add_input (PulseSink *sink, const pa_sink_input_info *info)
184{
185 PulseSinkInput *input;
186
187 g_return_val_if_fail (PULSE_IS_SINK (sink), FALSE);
188 g_return_val_if_fail (info != NULL, FALSE);
189
190 /* This function is used for both creating and refreshing sink inputs */
191 input = g_hash_table_lookup (sink->priv->inputs, GUINT_TO_POINTER (info->index));
192 if (input == NULL) {
193 const gchar *name;
194 PulseConnection *connection;
195
196 connection = pulse_stream_get_connection (PULSE_STREAM (sink));
197 input = pulse_sink_input_new (connection,
198 info,
199 sink);
200
201 g_hash_table_insert (sink->priv->inputs,
202 GUINT_TO_POINTER (info->index),
203 input);
204
205 free_list_controls (sink);
206
207 name = mate_mixer_stream_control_get_name (MATE_MIXER_STREAM_CONTROL (input));
208 g_signal_emit_by_name (G_OBJECT (sink),
209 "control-added",
210 name);
211 return TRUE;
212 }
213
214 pulse_sink_input_update (input, info);
215 return FALSE;
216}

218void
219pulse_sink_remove_input (PulseSink *sink, guint32 index)
220{
221 PulseSinkInput *input;
222 gchar *name;
223
224 g_return_if_fail (PULSE_IS_SINK (sink));
225
226 input = g_hash_table_lookup (sink->priv->inputs, GUINT_TO_POINTER (index));
227 if (G_UNLIKELY (input == NULL))
228 return;
229
230 name = g_strdup (mate_mixer_stream_control_get_name (MATE_MIXER_STREAM_CONTROL (input)));
231
232 g_hash_table_remove (sink->priv->inputs, GUINT_TO_POINTER (index));
233
234 free_list_controls (sink);
235 g_signal_emit_by_name (G_OBJECT (sink),
236 "control-removed",
237 name);
238 g_free (name);
239}

241void
242pulse_sink_update (PulseSink *sink, const pa_sink_info *info)
243{
244 g_return_if_fail (PULSE_IS_SINK (sink));
245 g_return_if_fail (info != NULL);
246
247 /* The switch doesn't allow being unset, PulseAudio should always include
248 * the active port name if the are any ports available */
249 if (info->active_port != NULL)
250 pulse_port_switch_set_active_port_by_name (sink->priv->pswitch,
251 info->active_port->name);
252
253 sink->priv->monitor = info->monitor_source;
254
255 pulse_sink_control_update (sink->priv->control, info);
256}

266static const GList *
267pulse_sink_list_controls (MateMixerStream *mms)
268{
269 PulseSink *sink;
270
271 g_return_val_if_fail (PULSE_IS_SINK (mms), NULL);
272
273 sink = PULSE_SINK (mms);
274
275 if (sink->priv->inputs_list == NULL) {
276 sink->priv->inputs_list = g_hash_table_get_values (sink->priv->inputs);
277 if (sink->priv->inputs_list != NULL)
278 g_list_foreach (sink->priv->inputs_list, (GFunc) g_object_ref, NULL);
279
280 sink->priv->inputs_list = g_list_prepend (sink->priv->inputs_list,
281 g_object_ref (sink->priv->control));
282 }
283 return sink->priv->inputs_list;
284}
