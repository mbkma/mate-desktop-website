50static void
51query_editor_changed_callback (CajaSearchBar *bar,
52 CajaQuery *query,
53 gboolean reload,
54 CajaWindowSlot *slot)
55{
56 CajaDirectory *directory;
57
58 directory = caja_directory_get_for_file (slot->viewed_file);
59 g_assert (CAJA_IS_SEARCH_DIRECTORY (directory));
60
61 caja_search_directory_set_query (CAJA_SEARCH_DIRECTORY (directory),
62 query);
63 if (reload)
64 {
65 caja_window_slot_reload (slot);
66 }
67
68 caja_directory_unref (directory);

71static void
72real_update_query_editor (CajaWindowSlot *slot)
73{
74 CajaDirectory *directory;
75
76 directory = caja_directory_get (slot->location);
77
78 if (CAJA_IS_SEARCH_DIRECTORY (directory))
79 {
80 GtkWidget *query_editor;
81 CajaQuery *query;
82 CajaSearchDirectory *search_directory;
83
84 search_directory = CAJA_SEARCH_DIRECTORY (directory);
85
86 query_editor = caja_query_editor_new (caja_search_directory_is_saved_search (search_directory),
87 caja_search_directory_is_indexed (search_directory));
88
89 slot->query_editor = CAJA_QUERY_EDITOR (query_editor);
90
91 caja_window_slot_add_extra_location_widget (slot, query_editor);
92 gtk_widget_show (query_editor);
93 g_signal_connect_object (query_editor, "changed",
94 G_CALLBACK (query_editor_changed_callback), slot, 0);
95
96 query = caja_search_directory_get_query (search_directory);
97 if (query != NULL)
98 {
99 caja_query_editor_set_query (CAJA_QUERY_EDITOR (query_editor),
100 query);
101 g_object_unref (query);
102 }
103 else
104 {
105 caja_query_editor_set_default_query (CAJA_QUERY_EDITOR (query_editor));
106 }
107 }
108
109 caja_directory_unref (directory);
110}

180{
181 GtkWidget *content_box, *eventbox, *extras_vbox, *frame;
182
183 content_box = gtk_box_new (GTK_ORIENTATION_VERTICAL, 0);
184 slot->content_box = content_box;
185 gtk_widget_show (content_box);
186
187 frame = gtk_frame_new (NULL);

193 gtk_widget_set_name (eventbox, "caja-extra-view-widget");
194 gtk_container_add (GTK_CONTAINER (frame), eventbox);
195 gtk_widget_show (eventbox);
196
197 extras_vbox = gtk_box_new (GTK_ORIENTATION_VERTICAL, 6);
198 gtk_container_set_border_width (GTK_CONTAINER (extras_vbox), 6);
199 slot->extra_location_widgets = extras_vbox;
200 gtk_container_add (GTK_CONTAINER (eventbox), extras_vbox);
201 gtk_widget_show (extras_vbox);
202
203 slot->view_box = gtk_box_new (GTK_ORIENTATION_VERTICAL, 0);
204 gtk_box_pack_start (GTK_BOX (content_box), slot->view_box, TRUE, TRUE, 0);
205 gtk_widget_show (slot->view_box);
206
207 slot->title = g_strdup (_("Loading..."));

298/* caja_window_slot_set_title:
299 *
300 * Sets slot->title, and if it changed
301 * synchronizes the actual GtkWindow title which
302 * might look a bit different (e.g. with "file browser:" added)
303 */
304static void
305caja_window_slot_set_title (CajaWindowSlot *slot,
306 const char *title)
307{
308 CajaWindow *window;
309 gboolean changed;
310
311 g_assert (CAJA_IS_WINDOW_SLOT (slot));
312
313 window = CAJA_WINDOW (slot->pane->window);
314
315 changed = FALSE;
316
317 if (eel_strcmp (title, slot->title) != 0)
318 {
319 changed = TRUE;
320
321 g_free (slot->title);
322 slot->title = g_strdup (title);
323 }
324
325 if (eel_strlen (slot->title) > 0 && slot->current_location_bookmark &&
326 caja_bookmark_set_name (slot->current_location_bookmark,
327 slot->title))
328 {
329 changed = TRUE;
330
331 /* Name of item in history list changed, tell listeners. */
332 caja_send_history_list_changed ();
333 }
334
335 if (changed)
336 {
337 caja_window_sync_title (window, slot);
338 }
339}

358/* caja_window_slot_update_icon:
359 *
360 * Re-calculate the slot icon
361 * Called when the location or view or icon set has changed.
362 * @slot: The CajaWindowSlot in question.
363 */
364void
365caja_window_slot_update_icon (CajaWindowSlot *slot)
366{
367 CajaWindow *window;
368 CajaIconInfo *info;
369 const char *icon_name;
370
371 window = slot->pane->window;
372
373 g_return_if_fail (CAJA_IS_WINDOW (window));
374
375 info = EEL_CALL_METHOD_WITH_RETURN_VALUE (CAJA_WINDOW_CLASS, window,
376 get_icon, (window, slot));
377
378 if (slot != slot->pane->active_slot)
379 return;
380
381 icon_name = NULL;
382 if (info)
383 {
384 GtkWindow *gtk_window = GTK_WINDOW (window);
385
386 icon_name = caja_icon_info_get_used_name (info);
387 if (icon_name != NULL)
388 {
389 /* frees the icon if it was previously set by using a pixbuf */
390 if (gtk_window_get_icon (gtk_window))
391 gtk_window_set_icon_list (gtk_window, NULL);
392
393 /* Gtk+ doesn't short circuit this (yet), so avoid lots of work
394 * if we're setting to the same icon. This happens a lot e.g. when
395 * the trash directory changes due to the file count changing.
396 */
397 const gchar *current_icon_name;
398
399 current_icon_name = gtk_window_get_icon_name (gtk_window);
400 if ((current_icon_name == NULL) || (strcmp (icon_name, current_icon_name) != 0))
401 {
402 if (strcmp (icon_name, "text-x-generic") == 0)
403 gtk_window_set_icon_name (gtk_window, "folder-saved-search");
404 else
405 gtk_window_set_icon_name (gtk_window, icon_name);
406 }
407 }
408 else
409 {
410 GdkPixbuf *pixbuf = caja_icon_info_get_pixbuf_nodefault (info);
411
412 if (pixbuf)
413 {
414 gtk_window_set_icon (gtk_window, pixbuf);
415 g_object_unref (pixbuf);
416 }
417 }
418 g_object_unref (info);
419 }

473{

494 gtk_box_pack_start (GTK_BOX (slot->view_box), widget,
495 TRUE, TRUE, 0);
496
497 gtk_widget_show (widget);
498
499 slot->content_view = new_view;
500 g_object_ref (slot->content_view);
501
502 /* connect new view */
503 caja_window_slot_connect_content_view (slot, new_view);
504 }

641static void
642caja_window_slot_dispose (GObject *object)
643{
644 CajaWindowSlot *slot;
645 GtkWidget *widget;
646
647 slot = CAJA_WINDOW_SLOT (object);
648
649 if (slot->content_view)
650 {
651 widget = caja_view_get_widget (slot->content_view);
652 gtk_widget_destroy (widget);
653 g_object_unref (slot->content_view);
654 slot->content_view = NULL;
655 }
656
657 if (slot->new_content_view)
658 {
659 widget = caja_view_get_widget (slot->new_content_view);
660 gtk_widget_destroy (widget);
661 g_object_unref (slot->new_content_view);
662 slot->new_content_view = NULL;
663 }
664
665 caja_window_slot_set_viewed_file (slot, NULL);
666
667 g_clear_object (&slot->location);
668
669 g_list_free_full (slot->pending_selection, g_free);
670 slot->pending_selection = NULL;
671
672 if (slot->current_location_bookmark != NULL)
673 {
674 g_object_unref (slot->current_location_bookmark);
675 slot->current_location_bookmark = NULL;
676 }
677 if (slot->last_location_bookmark != NULL)
678 {
679 g_object_unref (slot->last_location_bookmark);
680 slot->last_location_bookmark = NULL;
681 }
682
683 if (slot->find_mount_cancellable != NULL)
684 {
685 g_cancellable_cancel (slot->find_mount_cancellable);
686 slot->find_mount_cancellable = NULL;
687 }
688
689 slot->pane = NULL;
690
691 g_free (slot->title);
692 slot->title = NULL;
693
694 g_free (slot->status_text);
695 slot->status_text = NULL;
696
697 G_OBJECT_CLASS (parent_class)->dispose (object);
698}
