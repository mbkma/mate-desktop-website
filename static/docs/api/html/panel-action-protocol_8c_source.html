56static void
57panel_action_protocol_main_menu (GdkScreen *screen,
58 guint32 activate_time, GdkEvent *event)
59{
60 PanelWidget *panel_widget;
61 GtkWidget *menu;
62 AppletInfo *info;
63 GdkVisual *visual;
64 GtkWidget *toplevel;
65 GtkStyleContext *context;
66 GdkSeat *seat;
67 GdkDevice *device;
68
69 info = mate_panel_applet_get_by_type (PANEL_OBJECT_MENU_BAR, screen);
70 if (info) {
71 panel_menu_bar_popup_menu (PANEL_MENU_BAR (info->widget),
72 activate_time);
73 return;
74 }
75
76 info = mate_panel_applet_get_by_type (PANEL_OBJECT_MENU, screen);
77 if (info && !panel_menu_button_get_use_menu_path (PANEL_MENU_BUTTON (info->widget))) {
78 panel_menu_button_popup_menu (PANEL_MENU_BUTTON (info->widget),
79 1, activate_time);
80 return;
81 }
82
83 panel_widget = panels->data;
84 menu = create_main_menu (panel_widget);
85
86 panel_toplevel_push_autohide_disabler (panel_widget->toplevel);
87
88 gtk_menu_set_screen (GTK_MENU (menu), screen);
89/* Set up theme and transparency support */
90 toplevel = gtk_widget_get_toplevel (menu);
91/* Fix any failures of compiz/other wm's to communicate with gtk for transparency */
92 visual = gdk_screen_get_rgba_visual(screen);
93 gtk_widget_set_visual(GTK_WIDGET(toplevel), visual);
94/* Set menu and it's toplevel window to follow panel theme */
95 context = gtk_widget_get_style_context (GTK_WIDGET(toplevel));
96 gtk_style_context_add_class(context,"gnome-panel-menu-bar");
97 gtk_style_context_add_class(context,"mate-panel-menu-bar");
98
99 seat = gdk_display_get_default_seat (gdk_display_get_default());
100 device = gdk_seat_get_pointer (seat);
101 gdk_event_set_device (event, device);
102
103 gtk_menu_popup_at_pointer (GTK_MENU (menu),event);
104}

120static GdkFilterReturn
121panel_action_protocol_filter (GdkXEvent *gdk_xevent,
122 GdkEvent *event,
123 gpointer data)
124{
125 GdkWindow *window;
126 GdkScreen *screen;
127 GdkDisplay *display;
128 XEvent *xevent = (XEvent *) gdk_xevent;
129
130 if (xevent->type != ClientMessage)
131 return GDK_FILTER_CONTINUE;
132
133 if ((xevent->xclient.message_type != atom_mate_panel_action) &&
134 (xevent->xclient.message_type != atom_gnome_panel_action))
135 return GDK_FILTER_CONTINUE;
136
137 screen = gdk_event_get_screen (event);
138 display = gdk_screen_get_display (screen);
139 if (!GDK_IS_X11_DISPLAY (display))
140 return GDK_FILTER_CONTINUE;
141 window = gdk_x11_window_lookup_for_display (display, xevent->xclient.window);
142 if (!window)
143 return GDK_FILTER_CONTINUE;
144
145 if (window != gdk_screen_get_root_window (screen))
146 return GDK_FILTER_CONTINUE;
147
148 if (xevent->xclient.data.l [0] < 0)
149 return GDK_FILTER_CONTINUE;
150
151 if (xevent->xclient.data.l [0] == (long) atom_mate_panel_action_main_menu)
152 panel_action_protocol_main_menu (screen, xevent->xclient.data.l [1], event);
153 else if (xevent->xclient.data.l [0] == (long) atom_mate_panel_action_run_dialog)
154 panel_action_protocol_run_dialog (screen, xevent->xclient.data.l [1]);
155 else if (xevent->xclient.data.l [0] == (long) atom_gnome_panel_action_main_menu)
156 panel_action_protocol_main_menu (screen, xevent->xclient.data.l [1], event);
157 else if (xevent->xclient.data.l [0] == (long) atom_gnome_panel_action_run_dialog)
158 panel_action_protocol_run_dialog (screen, xevent->xclient.data.l [1]);
159 else if (xevent->xclient.data.l [0] == (long) atom_mate_panel_action_kill_dialog)
160 panel_action_protocol_kill_dialog (screen, xevent->xclient.data.l [1]);
161 else
162 return GDK_FILTER_CONTINUE;
163
164 return GDK_FILTER_REMOVE;
165}
