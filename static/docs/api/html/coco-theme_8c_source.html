143/* Draw fuctions */
144/* Standard rounded rectangle */
145static void
146nodoka_rounded_rectangle (cairo_t * cr,
147 double x, double y, double w, double h,
148 int radius)
149{
150 cairo_move_to (cr, x + radius, y);
151 cairo_arc (cr, x + w - radius, y + radius, radius, G_PI * 1.5, G_PI * 2);
152 cairo_arc (cr, x + w - radius, y + h - radius, radius, 0, G_PI * 0.5);
153 cairo_arc (cr, x + radius, y + h - radius, radius, G_PI * 0.5, G_PI);
154 cairo_arc (cr, x + radius, y + radius, radius, G_PI, G_PI * 1.5);
155}

157/* Fill background */
158static void
159fill_background(GtkWidget *widget, WindowData *windata, cairo_t *cr)
160{
161 double alpha;
162 if (windata->composited)
163 alpha = BACKGROUND_OPACITY;
164 else
165 alpha = 1.0;
166
167 cairo_pattern_t *pattern;
168 pattern = cairo_pattern_create_linear (0, 0, 0, windata->height);
169 cairo_pattern_add_color_stop_rgba (pattern, 0,
170 19/255.0, 19/255.0, 19/255.0, alpha);
171 cairo_pattern_add_color_stop_rgba (pattern, GRADIENT_CENTER,
172 19/255.0, 19/255.0, 19/255.0, alpha);
173 cairo_pattern_add_color_stop_rgba (pattern, 1,
174 19/255.0, 19/255.0, 19/255.0, alpha);
175 cairo_set_source (cr, pattern);
176 cairo_pattern_destroy (pattern);
177
178 nodoka_rounded_rectangle (cr, 0, 8, windata->width-8,
179 windata->height-8, 6);
180
181 cairo_fill (cr);
182}

184static void
185draw_pie(GtkWidget *pie, WindowData *windata, cairo_t *cr)
186{
187 if (windata->timeout == 0)
188 return;
189
190 gdouble arc_angle = 1.0 - (gdouble)windata->remaining / (gdouble)windata->timeout;
191 GtkStyleContext *context;
192 GdkRGBA orig, bg;
193
194 // :selected { background-color:#aabbcc; } ignored -> 1.0, 1.0, 1.0, 0.3
195 context = gtk_widget_get_style_context (windata->win);
196 gtk_style_context_save (context);
197 gtk_style_context_set_state (context, GTK_STATE_FLAG_SELECTED);
198 get_background_color (context, GTK_STATE_FLAG_SELECTED, &orig);
199 gtk_style_context_restore (context);
200
201 // .notification-box .countdown:selected { background-color:#aabbcc; }
202 context = gtk_widget_get_style_context (pie);
203 gtk_style_context_save (context);
204 gtk_style_context_set_state (context, GTK_STATE_FLAG_SELECTED);
205 get_background_color (context, GTK_STATE_FLAG_SELECTED, &bg);
206 gtk_style_context_restore (context);
207
208 if (gdk_rgba_equal (&orig, &bg))
209 cairo_set_source_rgba (cr, 1.0, 1.0, 1.0, 0.3);
210 else
211 cairo_set_source_rgba (cr, bg.red, bg.green, bg.blue, bg.alpha);
212
213 cairo_move_to(cr, PIE_RADIUS, PIE_RADIUS);
214 cairo_arc_negative(cr, PIE_RADIUS, PIE_RADIUS, PIE_RADIUS,
215 -G_PI/2, (-0.25 + arc_angle)*2*G_PI);
216 cairo_line_to(cr, PIE_RADIUS, PIE_RADIUS);
217
218 cairo_fill (cr);
219}

221static void
222update_shape_region (cairo_surface_t *surface,
223 WindowData *windata)
224{
225 if (windata->width == windata->last_width && windata->height == windata->last_height)
226 {
227 return;
228 }
229
230 if (windata->width == 0 || windata->height == 0)
231 {
232 GtkAllocation allocation;
233 gtk_widget_get_allocation (windata->win, &allocation);
234
235 windata->width = MAX (allocation.width, 1);
236 windata->height = MAX (allocation.height, 1);
237 }
238
239 if (!windata->composited) {
240 cairo_region_t *region;
241
242 region = gdk_cairo_region_create_from_surface (surface);
243 gtk_widget_shape_combine_region (windata->win, region);
244 cairo_region_destroy (region);
245 } else {
246 gtk_widget_shape_combine_region (windata->win, NULL);
247 return;
248 }
249
250 windata->last_width = windata->width;
251 windata->last_height = windata->height;
252}

254static void
255paint_window (GtkWidget *widget,
256 cairo_t *cr,
257 WindowData *windata)
258{
259 cairo_surface_t *surface;
260 cairo_t *cr2;
261
262 if (windata->width == 0 || windata->height == 0) {
263 GtkAllocation allocation;
264
265 gtk_widget_get_allocation(windata->win, &allocation);
266 windata->width = allocation.width;
267 windata->height = allocation.height;
268 }
269
270 surface = cairo_surface_create_similar(cairo_get_target(cr),
271 CAIRO_CONTENT_COLOR_ALPHA,
272 windata->width,
273 windata->height);
274
275 cr2 = cairo_create (surface);
276
277 /* transparent background */
278 cairo_rectangle (cr2, 0, 0, windata->width, windata->height);
279 cairo_set_source_rgba (cr2, 0.0, 0.0, 0.0, 0.0);
280 cairo_fill (cr2);
281
282 nodoka_rounded_rectangle (cr2, 0, 0, windata->width , windata->height, 6);
283 fill_background(widget, windata, cr2);
284 cairo_fill (cr2);
285
286 cairo_destroy (cr2);
287
288 cairo_save (cr);
289 cairo_set_operator (cr, CAIRO_OPERATOR_SOURCE);
290 cairo_set_source_surface (cr, surface, 0, 0);
291 cairo_paint (cr);
292 cairo_restore (cr);
293
294 update_shape_region (surface, windata);
295 cairo_surface_destroy (surface);
296}

320static gboolean
321countdown_expose_cb(GtkWidget *pie, cairo_t *cr, WindowData *windata)
322{
323 cairo_t *cr2;
324 cairo_surface_t *surface;
325 GtkAllocation alloc;
326
327 cairo_set_operator (cr, CAIRO_OPERATOR_OVER);
328 gtk_widget_get_allocation (pie, &alloc);
329
330 surface = cairo_surface_create_similar (cairo_get_target (cr), CAIRO_CONTENT_COLOR_ALPHA, alloc.width, alloc.height);
331 cr2 = cairo_create (surface);
332 cairo_set_source_rgba (cr2, 0.0, 0.0, 0.0, 0.0); // transparent background color
333 cairo_paint (cr2);
334 draw_pie (pie, windata, cr2); // countdown
335 cairo_fill (cr2);
336 cairo_destroy (cr2);
337
338 cairo_save (cr);
339 cairo_set_source_surface (cr, surface, 0, 0);
340 cairo_paint (cr);
341 cairo_restore (cr);
342
343 cairo_surface_destroy (surface);
344 return TRUE;
345}

526}

528/* Set the notification text */
529void
530set_notification_text(GtkWindow *nw, const char *summary, const char *body)
531{
532 char *str;
533 size_t str_len;
534 char *quoted;
535 WindowData *windata = g_object_get_data(G_OBJECT(nw), "windata");
536 g_assert(windata != NULL);
537
538 /* title */
539 quoted = g_markup_escape_text(summary, -1);
540 str = g_strdup_printf(
541 "<span color=\"#FFFFFF\"><big><b>%s</b></big></span>", quoted);
542 g_free(quoted);
543 gtk_label_set_markup(GTK_LABEL(windata->summary_label), str);
544 g_free(str);
545
546 /* body */
547 xmlDocPtr doc;
548 xmlInitParser();
549 str = g_strconcat ("<markup>", "<span color=\"#EAEAEA\">", body, "</span>", "</markup>", NULL);
550 /* parse notification body */
551 str_len = strlen (str);
552 doc = xmlReadMemory(str, (int) str_len, "noname.xml", NULL, 0);
553 g_free (str);
554 if (doc != NULL) {
555 xmlXPathContextPtr xpathCtx;
556 xmlXPathObjectPtr xpathObj;
557 xmlNodeSetPtr nodes;
558 const char *body_label_text;
559 int i, size;
560
561 /* filterout img nodes */
562 xpathCtx = xmlXPathNewContext(doc);
563 xpathObj = xmlXPathEvalExpression((unsigned char *)"//img", xpathCtx);
564 nodes = xpathObj->nodesetval;
565 size = (nodes) ? nodes->nodeNr : 0;
566 for(i = size - 1; i >= 0; i--) {
567 xmlUnlinkNode (nodes->nodeTab[i]);
568 xmlFreeNode (nodes->nodeTab[i]);
569 }
570
571 /* write doc to string */
572 xmlBufferPtr buf = xmlBufferCreate();
573 (void) xmlNodeDump(buf, doc, xmlDocGetRootElement (doc), 0, 0);
574 str = (char *)buf->content;
575 gtk_label_set_markup (GTK_LABEL (windata->body_label), str);
576
577 /* cleanup */
578 xmlBufferFree (buf);
579 xmlXPathFreeObject (xpathObj);
580 xmlXPathFreeContext (xpathCtx);
581 xmlFreeDoc (doc);
582
583 /* Does it render properly? */
584 body_label_text = gtk_label_get_text (GTK_LABEL (windata->body_label));
585 if ((body_label_text == NULL) || (strlen (body_label_text) == 0)) {
586 goto render_fail;
587 }
588 goto render_ok;
589 }
590
591render_fail:
592 /* could not parse notification body */
593 quoted = g_markup_escape_text(body, -1);
594 str = g_strconcat ("<span color=\"#EAEAEA\">", quoted, "</span>", NULL);
595 gtk_label_set_markup (GTK_LABEL (windata->body_label), str);
596 g_free (quoted);
597 g_free (str);
598
599render_ok:
600 xmlCleanupParser ();
601
602 if (body == NULL || *body == '\0')
603 gtk_widget_hide(windata->body_label);
604 else
605 gtk_widget_show(windata->body_label);
606
607 gtk_widget_set_size_request(
608 ((body != NULL && *body != '\0')
609 ? windata->body_label : windata->summary_label),
610 WIDTH - (IMAGE_SIZE + IMAGE_PADDING) - 10,
611 -1);
612}

614/* Set notification icon */
615void
616set_notification_icon(GtkWindow *nw, GdkPixbuf *pixbuf)
617{
618 WindowData *windata = g_object_get_data(G_OBJECT(nw), "windata");
619 g_assert(windata != NULL);
620
621 gtk_image_set_from_pixbuf(GTK_IMAGE(windata->icon), pixbuf);
622
623 if (pixbuf != NULL)
624 {
625 int pixbuf_width = gdk_pixbuf_get_width(pixbuf);
