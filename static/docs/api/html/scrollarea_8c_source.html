319typedef void (* PathForeachFunc) (double *x,
320 double *y,
321 gpointer data);
322
323static void
324path_foreach_point (cairo_path_t *path,
325 PathForeachFunc func,
326 gpointer user_data)
327{
328 int i;
329
330 for (i = 0; i < path->num_data; i += path->data[i].header.length)
331 {
332 cairo_path_data_t *data = &(path->data[i]);
333
334 switch (data->header.type)
335 {
336 case CAIRO_PATH_MOVE_TO:
337 case CAIRO_PATH_LINE_TO:
338 func (&(data[1].point.x), &(data[1].point.y), user_data);
339 break;
340
341 case CAIRO_PATH_CURVE_TO:
342 func (&(data[1].point.x), &(data[1].point.y), user_data);
343 func (&(data[2].point.x), &(data[2].point.y), user_data);
344 func (&(data[3].point.x), &(data[3].point.y), user_data);
345 break;
346
347 case CAIRO_PATH_CLOSE_PATH:
348 break;
349 }
350 }
351}

402static void
403clear_exposed_input_region (FooScrollArea *area,
404 cairo_region_t *exposed) /* in canvas coordinates */
405{
406 guint i;
407 cairo_region_t *viewport;
408 GdkRectangle allocation;
409
410 gtk_widget_get_allocation (GTK_WIDGET (area), &allocation);
411 allocation.x = 0;
412 allocation.y = 0;
413 allocation_to_canvas (area, &allocation.x, &allocation.y);
414 viewport = cairo_region_create_rectangle (&allocation);
415 cairo_region_subtract (viewport, exposed);
416
417 for (i = 0; i < area->priv->input_regions->len; ++i)
418 {
419 InputRegion *region = area->priv->input_regions->pdata[i];
420
421 cairo_region_intersect (region->region, viewport);
422
423 if (cairo_region_is_empty (region->region))
424 {
425 input_region_free (region);
426 g_ptr_array_remove_index_fast (area->priv->input_regions, i--);
427 }
428 }
429
430 cairo_region_destroy (viewport);
431}

433static void
434setup_background_cr (GtkWidget *widget,
435 cairo_t *cr,
436 int x_offset,
437 int y_offset)
438{
439 GdkWindow *window = gtk_widget_get_window (widget);
440 GtkWidget *parent = gtk_widget_get_parent (widget);
441 GtkStyleContext *context;
442 GdkRGBA *pattern_rgba = NULL;
443 GtkStateFlags state;
444
445 context = gtk_widget_get_style_context (widget);
446 state = gtk_style_context_get_state (context);
447
448 gtk_style_context_get (context, state, GTK_STYLE_PROPERTY_BACKGROUND_COLOR, &pattern_rgba, NULL);
449
450 if (parent)
451 {
452 gint window_x, window_y;
453
454 gdk_window_get_position (window, &window_x, &window_y);
455 setup_background_cr (parent, cr, x_offset + window_x, y_offset + window_y);
456 }
457 else if (pattern_rgba)
458 {
459 cairo_translate (cr, - x_offset, - y_offset);
460 gdk_cairo_set_source_rgba (cr, pattern_rgba);
461 cairo_translate (cr, x_offset, y_offset);
462 gdk_rgba_free (pattern_rgba);
463 }
464}

475static gboolean
476foo_scroll_area_draw (GtkWidget *widget,
477 cairo_t *widget_cr)
478{
479 FooScrollArea *scroll_area = FOO_SCROLL_AREA (widget);
480 cairo_t *cr;
481 cairo_region_t *region;
482 GtkAllocation widget_allocation;
483
484 /* Setup input areas */
485 clear_exposed_input_region (scroll_area, scroll_area->priv->update_region);
486
487 scroll_area->priv->current_input = g_new0 (InputRegion, 1);
488 scroll_area->priv->current_input->region = cairo_region_copy (scroll_area->priv->update_region);
489 scroll_area->priv->current_input->paths = NULL;
490 g_ptr_array_add (scroll_area->priv->input_regions,
491 scroll_area->priv->current_input);
492
493 region = scroll_area->priv->update_region;
494 scroll_area->priv->update_region = cairo_region_create ();
495
496 /* Create cairo context */
497 cr = cairo_create (scroll_area->priv->surface);
498 initialize_background (widget, cr);
499
500 g_signal_emit (widget, signals[PAINT], 0, cr);
501
502 /* Destroy stuff */
503 cairo_destroy (cr);
504
505 scroll_area->priv->current_input = NULL;
506
507 /* Finally draw the backing pixmap */
508 gtk_widget_get_allocation (widget, &widget_allocation);
509 cairo_set_source_surface (widget_cr, scroll_area->priv->surface, widget_allocation.x, widget_allocation.y);
510 cairo_paint (widget_cr);
511
512 cairo_region_destroy (region);
513
514 return TRUE;
515}

535static void
536emit_viewport_changed (FooScrollArea *scroll_area,
537 GdkRectangle *new_viewport,
538 GdkRectangle *old_viewport)
539{
540 GdkDisplay *display;
541 GdkSeat *seat;
542 GdkDevice *pointer;
543

545 g_signal_emit (scroll_area, signals[VIEWPORT_CHANGED], 0,

555 pointer,
556 &px,
557 &py,
558 NULL);
559
560 process_event (scroll_area, FOO_MOTION, px, py);
561}

563static void
564clamp_adjustment (GtkAdjustment *adj)
565{
566 if (gtk_adjustment_get_upper (adj) >= gtk_adjustment_get_page_size (adj))
567 gtk_adjustment_set_value (adj, CLAMP (gtk_adjustment_get_value (adj), 0.0,
568 gtk_adjustment_get_upper (adj)
569 - gtk_adjustment_get_page_size (adj)));
570 else
571 gtk_adjustment_set_value (adj, 0.0);
572}

574static gboolean
575set_adjustment_values (FooScrollArea *scroll_area)
576{
577 GtkAllocation allocation;
578
579 GtkAdjustment *hadj = scroll_area->priv->hadj;
580 GtkAdjustment *vadj = scroll_area->priv->vadj;
581
582 /* Horizontal */
583 gtk_widget_get_allocation (GTK_WIDGET (scroll_area), &allocation);
584 g_object_freeze_notify (G_OBJECT (hadj));
585 gtk_adjustment_set_page_size (hadj, allocation.width);
586 gtk_adjustment_set_step_increment (hadj, 0.1 * allocation.width);
587 gtk_adjustment_set_page_increment (hadj, 0.9 * allocation.width);
588 gtk_adjustment_set_lower (hadj, 0.0);
589 gtk_adjustment_set_upper (hadj, scroll_area->priv->width);
590 g_object_thaw_notify (G_OBJECT (hadj));
591
592 /* Vertical */
593 g_object_freeze_notify (G_OBJECT (vadj));
594 gtk_adjustment_set_page_size (vadj, allocation.height);
595 gtk_adjustment_set_step_increment (vadj, 0.1 * allocation.height);
596 gtk_adjustment_set_page_increment (vadj, 0.9 * allocation.height);
597 gtk_adjustment_set_lower (vadj, 0.0);
598 gtk_adjustment_set_upper (vadj, scroll_area->priv->height);
599 g_object_thaw_notify (G_OBJECT (vadj));
600
601 clamp_adjustment (hadj);
602 clamp_adjustment (vadj);
603
604 return TRUE;
605}

607static void
608foo_scroll_area_realize (GtkWidget *widget)
609{
610 FooScrollArea *area = FOO_SCROLL_AREA (widget);
611 GdkWindowAttr attributes;
612 GtkAllocation widget_allocation;
613 GdkWindow *window;
614 gint attributes_mask;
615 GdkDrawingContext *gdc;
616 cairo_region_t *cairo_region;
617 cairo_t *cr;
618
619 gtk_widget_get_allocation (widget, &widget_allocation);
620 gtk_widget_set_realized (widget, TRUE);
621
622 attributes.window_type = GDK_WINDOW_CHILD;
623 attributes.x = widget_allocation.x;
624 attributes.y = widget_allocation.y;
625 attributes.width = widget_allocation.width;
626 attributes.height = widget_allocation.height;
627 attributes.wclass = GDK_INPUT_ONLY;
628 attributes.event_mask = gtk_widget_get_events (widget);
629 attributes.event_mask |= (GDK_BUTTON_PRESS_MASK |
630 GDK_BUTTON_RELEASE_MASK |
631 GDK_BUTTON1_MOTION_MASK |
632 GDK_BUTTON2_MOTION_MASK |
633 GDK_BUTTON3_MOTION_MASK |
634 GDK_POINTER_MOTION_MASK |
635 GDK_ENTER_NOTIFY_MASK |
636 GDK_LEAVE_NOTIFY_MASK);
637
638 attributes_mask = GDK_WA_X | GDK_WA_Y;
639
640 window = gtk_widget_get_parent_window (widget);
641 gtk_widget_set_window (widget, window);
642 g_object_ref (window);
643
644 area->priv->input_window = gdk_window_new (window,
645 &attributes, attributes_mask);
646
647 cairo_region = cairo_region_create ();
648 gdc = gdk_window_begin_draw_frame (gtk_widget_get_window (widget), cairo_region);
649 cr = gdk_drawing_context_get_cairo_context (gdc);
650
651 area->priv->surface = cairo_surface_create_similar (cairo_get_target (cr), CAIRO_CONTENT_COLOR,
652 widget_allocation.width, widget_allocation.height);
653
654 gdk_window_end_draw_frame (gtk_widget_get_window (widget), gdc);
655 cairo_region_destroy (cairo_region);
656
657 gdk_window_set_user_data (area->priv->input_window, area);
658}

675static cairo_surface_t *
676create_new_surface (GtkWidget *widget,
677 cairo_surface_t *old)
678{
679 GtkAllocation widget_allocation;
680 GdkDrawingContext *gdc;
681 cairo_region_t *cairo_region;
682 cairo_t *cr;
683 cairo_surface_t *new;
684
685 gtk_widget_get_allocation (widget, &widget_allocation);
686
687 cairo_region = cairo_region_create ();
688 gdc = gdk_window_begin_draw_frame (gtk_widget_get_window (widget), cairo_region);
689 cr = gdk_drawing_context_get_cairo_context (gdc);
690 new = cairo_surface_create_similar (cairo_get_target (cr),
691 CAIRO_CONTENT_COLOR,
692 widget_allocation.width,
693 widget_allocation.height);
694 gdk_window_end_draw_frame (gtk_widget_get_window (widget), gdc);
695 cairo_region_destroy (cairo_region);
696
697 /* Unfortunately we don't know in which direction we were resized,
698 * so we just assume we were dragged from the south-east corner.
699 *
700 * Although, maybe we could get the root coordinates of the input-window?
701 * That might just work, actually. We need to make sure marco uses
702 * static gravity for the window before this will be useful.
703 */
704
705 cr = cairo_create (new);
706 cairo_set_operator (cr, CAIRO_OPERATOR_SOURCE);
707 cairo_set_source_surface (cr, old, 0, 0);
708
709 cairo_paint (cr);
710 cairo_destroy (cr);
711
712 return new;
713}

722static void
723_cairo_region_xor (cairo_region_t *dst, const cairo_region_t *src)
724{
725 cairo_region_t *trb;
726
727 trb = cairo_region_copy (src);
728
729 cairo_region_subtract (trb, dst);
730 cairo_region_subtract (dst, src);

732 cairo_region_destroy (trb);
733}

735static void
736foo_scroll_area_size_allocate (GtkWidget *widget,
737 GtkAllocation *allocation)
738{
739 FooScrollArea *scroll_area = FOO_SCROLL_AREA (widget);
740 GdkRectangle new_viewport;
741 GdkRectangle old_viewport;
742 cairo_region_t *old_allocation;
743 cairo_region_t *invalid;
744 GtkAllocation widget_allocation;
745
746 get_viewport (scroll_area, &old_viewport);
747
748 gtk_widget_get_allocation (widget, &widget_allocation);
749 old_allocation = cairo_region_create_rectangle (&widget_allocation);
750 cairo_region_translate (old_allocation,
751 -widget_allocation.x, -widget_allocation.y);
752 invalid = cairo_region_create_rectangle (allocation);
753 cairo_region_translate (invalid, -allocation->x, -allocation->y);
754 _cairo_region_xor (invalid, old_allocation);
755 allocation_to_canvas_region (scroll_area, invalid);
756 foo_scroll_area_invalidate_region (scroll_area, invalid);
757 cairo_region_destroy (old_allocation);
758 cairo_region_destroy (invalid);
759
760 gtk_widget_set_allocation (widget, allocation);
761
762 if (scroll_area->priv->input_window)
763 {
764 cairo_surface_t *new_surface;
765
766 gdk_window_move_resize (scroll_area->priv->input_window,
767 allocation->x, allocation->y,
768 allocation->width, allocation->height);
769
770 new_surface = create_new_surface (widget, scroll_area->priv->surface);
771 cairo_surface_destroy (scroll_area->priv->surface);
772 scroll_area->priv->surface = new_surface;
773 }
774
775 get_viewport (scroll_area, &new_viewport);
776
777 emit_viewport_changed (scroll_area, &new_viewport, &old_viewport);
778}

782 FooScrollAreaEventType type,
783 int x,
784 int y,
785 FooScrollAreaEventFunc func,
786 gpointer data)
787{

791 return;

794 emit_input (scroll_area, FOO_MOTION, x, y, func, data);

803static void
804process_event (FooScrollArea *scroll_area,
805 FooScrollAreaEventType input_type,
806 int x,
807 int y)
808{
809 GtkWidget *widget = GTK_WIDGET (scroll_area);
810 guint i;
811
812 allocation_to_canvas (scroll_area, &x, &y);
813
814 if (scroll_area->priv->grabbed)
815 {
816 emit_input (scroll_area, input_type, x, y,
817 scroll_area->priv->grab_func,
818 scroll_area->priv->grab_data);
819 return;
820 }
821
822 for (i = 0; i < scroll_area->priv->input_regions->len; ++i)
823 {
824 InputRegion *region = scroll_area->priv->input_regions->pdata[i];
825
826 if (cairo_region_contains_point (region->region, x, y))
827 {
828 InputPath *path;
829
830 path = region->paths;
831 while (path)
832 {
833 cairo_t *cr;
834 gboolean inside;
835 GdkDrawingContext *gdc;
836 cairo_region_t *cairo_region;
837
838 cairo_region = cairo_region_create ();
839 gdc = gdk_window_begin_draw_frame (gtk_widget_get_window (widget), cairo_region);
840 cr = gdk_drawing_context_get_cairo_context (gdc);
841 cairo_set_fill_rule (cr, path->fill_rule);
842 cairo_set_line_width (cr, path->line_width);
843 cairo_append_path (cr, path->path);
844
845 if (path->is_stroke)
846 inside = cairo_in_stroke (cr, x, y);
847 else
848 inside = cairo_in_fill (cr, x, y);
849 gdk_window_end_draw_frame (gtk_widget_get_window (widget), gdc);
850 cairo_region_destroy (cairo_region);
851
852
853 if (inside)
854 {
855 emit_input (scroll_area, input_type,
856 x, y,
857 path->func,
858 path->data);
859 return;
860 }
861
862 path = path->next;
863 }
864
865 /* Since the regions are all disjoint, no other region
866 * can match. Of course we could be clever and try and
867 * sort the regions, but so far I have been unable to
868 * make this loop show up on a profile.
869 */
870 return;
871 }
872 }
873}

878 int y,
879 GdkEvent *event)

1002static void
1003foo_scroll_area_scroll (FooScrollArea *area,
1004 gint dx,
1005 gint dy)
1006{
1007 GdkRectangle allocation;
1008 GdkRectangle src_area;
1009 GdkRectangle move_area;
1010 cairo_region_t *invalid_region;
1011
1012 gtk_widget_get_allocation (GTK_WIDGET (area), &allocation);
1013 allocation.x = 0;
1014 allocation.y = 0;
1015
1016 src_area = allocation;
1017 src_area.x -= dx;
1018 src_area.y -= dy;
1019
1020 invalid_region = cairo_region_create_rectangle (&allocation);
1021
1022 if (gdk_rectangle_intersect (&allocation, &src_area, &move_area))
1023 {
1024 cairo_region_t *move_region;
1025 cairo_t *cr;
1026
1027 cr = cairo_create (area->priv->surface);
1028
1029 /* Cairo doesn't allow self-copies, so we do this little trick instead:
1030 * 1) Clip so the group size is small.
1031 * 2) Call push_group() which creates a temporary pixmap as a workaround
1032 */
1033 gdk_cairo_rectangle (cr, &move_area);
1034 cairo_clip (cr);
1035 cairo_push_group (cr);
1036
1037 cairo_set_source_surface (cr, area->priv->surface, dx, dy);
1038 gdk_cairo_rectangle (cr, &move_area);
1039 cairo_fill (cr);
1040
1041 cairo_pop_group_to_source (cr);
1042 cairo_paint (cr);
1043
1044 cairo_destroy (cr);
1045
1046 gtk_widget_queue_draw (GTK_WIDGET (area));
1047
1048 move_region = cairo_region_create_rectangle (&move_area);
1049 cairo_region_translate (move_region, dx, dy);
1050 cairo_region_subtract (invalid_region, move_region);
1051 cairo_region_destroy (move_region);
1052 }
1053
1054 allocation_to_canvas_region (area, invalid_region);
1055
1056 foo_scroll_area_invalidate_region (area, invalid_region);
1057
1058 cairo_region_destroy (invalid_region);
1059}

1061static void
1062foo_scrollbar_adjustment_changed (GtkAdjustment *adj,
1063 FooScrollArea *scroll_area)
1064{
1065 GtkWidget *widget = GTK_WIDGET (scroll_area);
1066 gint dx = 0;
1067 gint dy = 0;
1068 GdkRectangle old_viewport, new_viewport;
1069 gdouble aux;
1070 int offset;
1071
1072 get_viewport (scroll_area, &old_viewport);
1073
1074 aux = gtk_adjustment_get_value (adj);
1075 offset = (int) aux;
1076 if (adj == scroll_area->priv->hadj)
1077 {
1078 dx = offset - scroll_area->priv->x_offset;
1079 scroll_area->priv->x_offset = offset;
1080 }
1081 else if (adj == scroll_area->priv->vadj)
1082 {
1083 dy = offset - scroll_area->priv->y_offset;
1084 scroll_area->priv->y_offset = offset;
1085 }
1086 else
1087 {
1088 g_assert_not_reached ();
1089 }
1090
1091 if (gtk_widget_get_realized (widget))
1092 {
1093 foo_scroll_area_scroll (scroll_area, -dx, -dy);
1094 }
1095
1096 get_viewport (scroll_area, &new_viewport);
1097
1098 emit_viewport_changed (scroll_area, &new_viewport, &old_viewport);
1099}

1101static void
1102set_one_adjustment (FooScrollArea *scroll_area,
1103 GtkAdjustment *adjustment,
1104 GtkAdjustment **location)
1105{
1106 g_return_if_fail (location != NULL);
1107
1108 if (adjustment == *location)
1109 return;
1110
1111 if (!adjustment)
1112 adjustment = new_adjustment ();
1113
1114 g_return_if_fail (GTK_IS_ADJUSTMENT (adjustment));
1115
1116 if (*location)
1117 {
1118 g_signal_handlers_disconnect_by_func (
1119 *location, foo_scrollbar_adjustment_changed, scroll_area);
1120
1121 g_object_unref (*location);
1122 }
1123
1124 *location = adjustment;
1125
1126 g_object_ref_sink (*location);
1127
1128 g_signal_connect (*location, "value_changed",
1129 G_CALLBACK (foo_scrollbar_adjustment_changed),
1130 scroll_area);
1131}

1387static gboolean
1388scroll_idle (gpointer data)
1389{
1390 GdkRectangle viewport, new_viewport;
1391 FooScrollArea *area = data;
1392 AutoScrollInfo *info = area->priv->auto_scroll_info;
1393 int new_x, new_y;
1394 double elapsed;
1395
1396 get_viewport (area, &viewport);
1397
1398 elapsed = g_timer_elapsed (info->timer, NULL);
1399
1400 info->res_x = elapsed * info->dx / 0.2;
1401 info->res_y = elapsed * info->dy / 0.2;
1402
1403 new_x = viewport.x + (int) info->res_x;
1404 new_y = viewport.y + (int) info->res_y;
1405
1406 foo_scroll_area_set_viewport_pos (area, new_x, new_y);
1407
1408 get_viewport (area, &new_viewport);
1409
1410 if (viewport.x == new_viewport.x &&
1411 viewport.y == new_viewport.y &&
1412 (info->res_x > 1.0 ||
1413 info->res_y > 1.0 ||
1414 info->res_x < -1.0 ||
1415 info->res_y < -1.0))
1416 {
1417 stop_scrolling (area);
1418
1419 /* stop scrolling if it didn't have an effect */
1420 return FALSE;
1421 }
1422
1423 return TRUE;
1424}

1426static void
1427ensure_scrolling (FooScrollArea *area,
1428 int dx,
1429 int dy)
1430{
1431 if (!area->priv->auto_scroll_info)
1432 {
