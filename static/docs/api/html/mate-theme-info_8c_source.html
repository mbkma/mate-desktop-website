146static void
147add_theme_to_hash_by_name (GHashTable *hash_table,
148 gpointer data)
149{
150 MateThemeCommonInfo *info = data;
151 GList *list;
152
153 list = g_hash_table_lookup (hash_table, info->name);
154 if (list == NULL) {
155 list = g_list_append (list, info);
156 } else {
157 GList *list_ptr = list;
158 gboolean added = FALSE;
159
160 while (list_ptr) {
161 gint theme_priority;
162
163 theme_priority = ((MateThemeCommonInfo *) list_ptr->data)->priority;
164
165 if (theme_priority == info->priority) {
166 /* Swap it in */
167 list_ptr->data = info;
168 added = TRUE;
169 break;
170 } else if (theme_priority > info->priority) {
171 list = g_list_insert_before (list, list_ptr, info);
172 added = TRUE;
173 break;
174 }
175 list_ptr = list_ptr->next;
176 }
177 if (!added)
178 list = g_list_append (list, info);
179 }
180 g_hash_table_insert (hash_table,
181 g_strdup (info->name),
182 list);
183}

185static void
186remove_theme_from_hash_by_name (GHashTable *hash_table,
187 gpointer data)
188{
189 MateThemeCommonInfo *info = data;
190 GList *list;
191
192 list = g_hash_table_lookup (hash_table, info->name);
193
194 list = g_list_remove (list, info);
195 if (list == NULL)
196 g_hash_table_remove (hash_table, info->name);
197 else
198 g_hash_table_insert (hash_table, g_strdup (info->name), list);
199}

201static MateThemeCommonInfo *
202get_theme_from_hash_by_name (GHashTable *hash_table,
203 const gchar *name,
204 gint priority)
205{
206 GList *list;
207
208 list = g_hash_table_lookup (hash_table, name);
209
210 /* -1 implies return the first one */
211 if (priority == -1) {
212 return list ? list->data : NULL;
213 }
214
215 while (list) {
216 MateThemeCommonInfo *info = (MateThemeCommonInfo *) list->data;
217
218 if (info->priority == priority)
219 return info;
220
221 list = list->next;
222 }
223 return NULL;
224}

226static gint
227theme_compare (MateThemeCommonInfo *a,
228 MateThemeCommonInfo *b)
229{
230 gint cmp;
231
232 g_return_val_if_fail (a->type == b->type, a->type - b->type);
233
234 switch (a->type) {
235 case MATE_THEME_TYPE_METATHEME:
236 cmp = mate_theme_meta_info_compare (
237 (MateThemeMetaInfo *) a, (MateThemeMetaInfo *) b);
238 break;
239 case MATE_THEME_TYPE_ICON:
240 cmp = mate_theme_icon_info_compare (
241 (MateThemeIconInfo *) a, (MateThemeIconInfo *) b);
242 break;
243 case MATE_THEME_TYPE_CURSOR:
244 cmp = mate_theme_cursor_info_compare (
245 (MateThemeCursorInfo *) a, (MateThemeCursorInfo *) b);
246 break;
247 default:
248 /* not supported at this time */
249 g_assert_not_reached ();
250 }
251
252 return cmp;
253}

295

298
299 common_theme_dir_uri = g_file_get_parent (meta_theme_uri);

301 meta_theme_info = mate_theme_meta_info_new ();
302

305

307 meta_theme_info->name = g_file_get_basename (common_theme_dir_uri);
308
309 g_object_unref (common_theme_dir_uri);
310
311 /* meta_theme_info->readable_name */
312 str = mate_desktop_item_get_localestring (meta_theme_ditem,
313 THEME_NAME);
314 if (str == NULL) {
315 str = mate_desktop_item_get_localestring (meta_theme_ditem,
316 MATE_DESKTOP_ITEM_NAME);
317 if (str == NULL) /* shouldn't reach */
318 goto mate_theme_read_meta_theme_out;
319 }
320 meta_theme_info->readable_name = g_strdup (str);
321
322 /* meta_theme_info->comment */
323 str = mate_desktop_item_get_localestring (meta_theme_ditem,
324 THEME_COMMENT);

326 str = mate_desktop_item_get_localestring (meta_theme_ditem,

341 goto mate_theme_read_meta_theme_out;
342 meta_theme_info->gtk_theme_name = g_strdup (str);
343
344 /* meta_theme_info->gtk_color_scheme */

346 GTK_COLOR_SCHEME_KEY);
347 if (str == NULL || str[0] == '\0')
348 scheme = gtkrc_get_color_scheme_for_theme (meta_theme_info->gtk_theme_name);
349 else
350 scheme = g_strdup(str);
351
352 if (scheme != NULL) {

356 if (*scheme == ',')

359 }
360

362 str = mate_desktop_item_get_string (meta_theme_ditem,
363 MARCO_THEME_KEY);

513static GdkPixbuf *
514gdk_pixbuf_from_xcursor_image (XcursorImage *cursor)
515{
516 GdkPixbuf *pixbuf;
517#define BUF_SIZE sizeof(guint32) * cursor->width * cursor->height
518 guchar *buf = g_malloc0 (BUF_SIZE);
519 guchar *it;
520
521 for (it = buf; it < (buf + BUF_SIZE); it += 4) {
522#if G_BYTE_ORDER == G_LITTLE_ENDIAN
523 /* on little endianess it's BGRA to RGBA */
524 it[0] = ((guchar *) (cursor->pixels))[it - buf + 2];
525 it[1] = ((guchar *) (cursor->pixels))[it - buf + 1];
526 it[2] = ((guchar *) (cursor->pixels))[it - buf + 0];
527 it[3] = ((guchar *) (cursor->pixels))[it - buf + 3];
528#else
529 /* on big endianess it's ARGB to RGBA */
530 it[0] = ((guchar *) cursor->pixels)[it - buf + 1];
531 it[1] = ((guchar *) cursor->pixels)[it - buf + 2];
532 it[2] = ((guchar *) cursor->pixels)[it - buf + 3];
533 it[3] = ((guchar *) cursor->pixels)[it - buf + 0];
534#endif
535 }
536
537 pixbuf = gdk_pixbuf_new_from_data ((const guchar *) buf,
538 GDK_COLORSPACE_RGB, TRUE, 8,
539 cursor->width, cursor->height,
540 cursor->width * 4,
541 (GdkPixbufDestroyNotify) g_free,
542 NULL);
543
544 if (!pixbuf)
545 g_free (buf);
546
547 return pixbuf;
548}

550static MateThemeCursorInfo *
551read_cursor_theme (GFile *cursor_theme_uri)
552{
553 MateThemeCursorInfo *cursor_theme_info = NULL;
554 GFile *parent_uri, *cursors_uri;
555
556 const gint filter_sizes[] = { 12, 16, 18, 24, 32, 36, 40, 48, 64, 96, 128 };
557 const gint num_sizes = G_N_ELEMENTS (filter_sizes);
558
559 parent_uri = g_file_get_parent (cursor_theme_uri);
560 cursors_uri = g_file_get_child (parent_uri, "cursors");
561
562 if (get_file_type (cursors_uri) == G_FILE_TYPE_DIRECTORY) {
563 GArray *sizes;
564 XcursorImage *cursor;
565 GdkPixbuf *thumbnail = NULL;
566 gchar *name;
567 gint i;
568
569 name = g_file_get_basename (parent_uri);
570
571 sizes = g_array_sized_new (FALSE, FALSE, sizeof (gint), num_sizes);
572
573 for (i = 0; i < num_sizes; ++i) {
574 cursor = XcursorLibraryLoadImage ("left_ptr", name, filter_sizes[i]);
575
576 if (cursor) {
577 if (cursor->size == filter_sizes[i]) {
578 g_array_append_val (sizes, filter_sizes[i]);
579
580 if (thumbnail == NULL && i >= 1)
581 thumbnail = gdk_pixbuf_from_xcursor_image (cursor);
582 }
583
584 XcursorImageDestroy (cursor);
585 }
586 }
587
588 if (sizes->len == 0) {
589 g_array_free (sizes, TRUE);
590 g_free (name);
591 } else {
592 MateDesktopItem *cursor_theme_ditem;
593 gchar *cursor_theme_file;
594
595 if (!thumbnail) {
596 cursor = XcursorLibraryLoadImage ("left_ptr", name,
597 g_array_index (sizes, gint, 0));
598 if (cursor) {
599 thumbnail = gdk_pixbuf_from_xcursor_image (cursor);
600 XcursorImageDestroy (cursor);
601 }
602 }
603
604 cursor_theme_info = mate_theme_cursor_info_new ();
605 cursor_theme_info->path = g_file_get_path (parent_uri);
606 cursor_theme_info->name = name;
607 cursor_theme_info->sizes = sizes;
608 cursor_theme_info->thumbnail = thumbnail;
609
610 cursor_theme_file = g_file_get_path (cursor_theme_uri);
611 cursor_theme_ditem = mate_desktop_item_new_from_file (cursor_theme_file, 0, NULL);
612 g_free (cursor_theme_file);
613
614 if (cursor_theme_ditem != NULL) {
615 const gchar *readable_name;
616
617 readable_name = mate_desktop_item_get_string (cursor_theme_ditem,
618 "Icon Theme/Name");
619 if (readable_name)
620 cursor_theme_info->readable_name = g_strdup (readable_name);
621 else
622 cursor_theme_info->readable_name = g_strdup (name);
623
624 cursor_theme_info->hidden = mate_desktop_item_get_boolean (cursor_theme_ditem,
625 "Icon Theme/Hidden");
626
627 mate_desktop_item_unref (cursor_theme_ditem);
628 } else {
629 cursor_theme_info->readable_name = g_strdup (name);
630 }
631 }
632 }
633
634 g_object_unref (cursors_uri);
635 g_object_unref (parent_uri);
636
637 return cursor_theme_info;
638}

640static void
641handle_change_signal (gpointer data,
642 MateThemeChangeType change_type,
643 MateThemeElement element_type)
644{
645#ifdef DEBUG
646 gchar *type_str = NULL;
647 gchar *change_str = NULL;
648 gchar *element_str = NULL;
649#endif
650 MateThemeCommonInfo *theme = data;
651 GList *list;
652
653 if (initting)
654 return;
655
656 for (list = callbacks; list; list = list->next) {
657 ThemeCallbackData *callback_data = list->data;
658 (* callback_data->func) (theme, change_type, element_type, callback_data->data);
659 }
660
661#ifdef DEBUG
662 if (theme->type == MATE_THEME_TYPE_METATHEME)
663 type_str = "meta";
664 else if (theme->type == MATE_THEME_TYPE_ICON)
665 type_str = "icon";
666 else if (theme->type == MATE_THEME_TYPE_CURSOR)
667 type_str = "cursor";
668 else if (theme->type == MATE_THEME_TYPE_REGULAR) {
669 if (element_type & MATE_THEME_GTK_2)
670 element_str = "gtk-2";
671 else if (element_type & MATE_THEME_GTK_2_KEYBINDING)
672 element_str = "keybinding";
673 else if (element_type & MATE_THEME_MARCO)
674 element_str = "marco";

700/* index_uri should point to the gtkrc file that was modified */
701static void
702update_theme_index (GFile *index_uri,
703 MateThemeElement key_element,
704 gint priority)
705{
706 gboolean theme_exists;
707 MateThemeInfo *theme_info;
708 GFile *parent;
709 GFile *common_theme_dir_uri;
710 gchar *common_theme_dir;
711
712 /* First, we determine the new state of the file. We do no more
713 * sophisticated a test than "files exists and is a file" */
714 theme_exists = (get_file_type (index_uri) == G_FILE_TYPE_REGULAR);
715
716 /* Next, we see what currently exists */
717 parent = g_file_get_parent (index_uri);
718 common_theme_dir_uri = g_file_get_parent (parent);
719 common_theme_dir = g_file_get_path (common_theme_dir_uri);
720
721 theme_info = g_hash_table_lookup (theme_hash_by_uri, common_theme_dir);
722 if (theme_info == NULL) {
723 if (theme_exists) {
724 theme_info = mate_theme_info_new ();
725 theme_info->path = g_strdup (common_theme_dir);
726 theme_info->name = g_file_get_basename (common_theme_dir_uri);
727 theme_info->readable_name = g_strdup (theme_info->name);
728 theme_info->priority = priority;
729 if (key_element & MATE_THEME_GTK_2)
730 theme_info->has_gtk = TRUE;
731 else if (key_element & MATE_THEME_GTK_2_KEYBINDING)
732 theme_info->has_keybinding = TRUE;
733 else if (key_element & MATE_THEME_MARCO)
734 theme_info->has_marco = TRUE;
735
736 g_hash_table_insert (theme_hash_by_uri, g_strdup (common_theme_dir), theme_info);
737 add_theme_to_hash_by_name (theme_hash_by_name, theme_info);
738 handle_change_signal (theme_info, MATE_THEME_CHANGE_CREATED, key_element);
739 }
740 } else {
741 gboolean theme_used_to_exist = FALSE;
742
743 if (key_element & MATE_THEME_GTK_2) {
744 theme_used_to_exist = theme_info->has_gtk;
745 theme_info->has_gtk = (theme_exists != FALSE);
746 } else if (key_element & MATE_THEME_GTK_2_KEYBINDING) {
747 theme_used_to_exist = theme_info->has_keybinding;
748 theme_info->has_keybinding = (theme_exists != FALSE);
749 } else if (key_element & MATE_THEME_MARCO) {
750 theme_used_to_exist = theme_info->has_marco;
751 theme_info->has_marco = (theme_exists != FALSE);
752 }
753
754 if (!theme_info->has_marco && !theme_info->has_keybinding && !theme_info->has_gtk) {
755 g_hash_table_remove (theme_hash_by_uri, common_theme_dir);
756 remove_theme_from_hash_by_name (theme_hash_by_name, theme_info);
757 }
758
759 if (theme_exists && theme_used_to_exist) {
760 handle_change_signal (theme_info, MATE_THEME_CHANGE_CHANGED, key_element);
761 } else if (theme_exists && !theme_used_to_exist) {
762 handle_change_signal (theme_info, MATE_THEME_CHANGE_CREATED, key_element);
763 } else if (!theme_exists && theme_used_to_exist) {
764 handle_change_signal (theme_info, MATE_THEME_CHANGE_DELETED, key_element);
765 }
766
767 if (!theme_info->has_marco && !theme_info->has_keybinding && !theme_info->has_gtk) {
768 mate_theme_info_free (theme_info);
769 }
770 }
771
772 g_free (common_theme_dir);
773 g_object_unref (parent);
774 g_object_unref (common_theme_dir_uri);
775}

804static void
805update_common_theme_dir_index (GFile *theme_index_uri,
806 MateThemeType type,
807 gint priority)
808{
809 gboolean theme_exists;
810 MateThemeCommonInfo *theme_info = NULL;
811 MateThemeCommonInfo *old_theme_info;
812 GFile *common_theme_dir_uri;
813 gchar *common_theme_dir;
814 GHashTable *hash_by_uri;
815 GHashTable *hash_by_name;
816
817 if (type == MATE_THEME_TYPE_ICON) {
818 hash_by_uri = icon_theme_hash_by_uri;
819 hash_by_name = icon_theme_hash_by_name;
820 } else if (type == MATE_THEME_TYPE_CURSOR) {
821 hash_by_uri = cursor_theme_hash_by_uri;
822 hash_by_name = cursor_theme_hash_by_name;
823 } else {
824 hash_by_uri = meta_theme_hash_by_uri;
825 hash_by_name = meta_theme_hash_by_name;
826 }
827
828 if (type != MATE_THEME_TYPE_CURSOR) {
829 /* First, we determine the new state of the file. */
830 if (get_file_type (theme_index_uri) == G_FILE_TYPE_REGULAR) {
831 /* It's an interesting file. Let's try to load it. */
832 if (type == MATE_THEME_TYPE_ICON)
833 theme_info = (MateThemeCommonInfo *) read_icon_theme (theme_index_uri);
834 else
835 theme_info = (MateThemeCommonInfo *) mate_theme_read_meta_theme (theme_index_uri);
836 } else {
837 theme_info = NULL;
838 }
839
840 }
841 /* cursor themes don't necessarily have an index file, so try those in any case */
842 else {
843 theme_info = (MateThemeCommonInfo *) read_cursor_theme (theme_index_uri);
844 }
845
846 if (theme_info) {
847 theme_info->priority = priority;
848 theme_exists = TRUE;
849 } else {
850 theme_exists = FALSE;
851 }
852
853 /* Next, we see what currently exists */
854 common_theme_dir_uri = g_file_get_parent (theme_index_uri);
855 common_theme_dir = g_file_get_path (common_theme_dir_uri);
856 g_object_unref (common_theme_dir_uri);
857
858 old_theme_info = (MateThemeCommonInfo *) g_hash_table_lookup (hash_by_uri, common_theme_dir);
859
860 if (old_theme_info == NULL) {
861 if (theme_exists) {
862 g_hash_table_insert (hash_by_uri, g_strdup (common_theme_dir), theme_info);
863 add_theme_to_hash_by_name (hash_by_name, theme_info);
864 handle_change_signal (theme_info, MATE_THEME_CHANGE_CREATED, 0);
865 }
866 } else {
867 if (theme_exists) {
868 if (theme_compare (theme_info, old_theme_info) != 0) {
869 /* Remove old theme */
870 g_hash_table_remove (hash_by_uri, common_theme_dir);
871 remove_theme_from_hash_by_name (hash_by_name, old_theme_info);
872 g_hash_table_insert (hash_by_uri, g_strdup (common_theme_dir), theme_info);
873 add_theme_to_hash_by_name (hash_by_name, theme_info);
874 handle_change_signal (theme_info, MATE_THEME_CHANGE_CHANGED, 0);
875 theme_free (old_theme_info);
876 } else {
877 theme_free (theme_info);
878 }
879 } else {
880 g_hash_table_remove (hash_by_uri, common_theme_dir);
881 remove_theme_from_hash_by_name (hash_by_name, old_theme_info);
882
883 handle_change_signal (old_theme_info, MATE_THEME_CHANGE_DELETED, 0);
884 theme_free (old_theme_info);
885 }
886 }
887
888 g_free (common_theme_dir);
889}

1223/* Add a monitor to a top dir. These monitors persist for the duration of the
1224 * lib.
1225 */
1226static gboolean
1227real_add_top_theme_dir_monitor (GFile *uri,
1228 gint priority,
1229 gboolean icon_theme,
1230 GError **error)
1231{
1232 GFileInfo *file_info;
1233 GFileMonitor *monitor;
1234 GFileEnumerator *enumerator;
1235 CallbackTuple *tuple;
1236
1237 /* Check the URI */
1238 if (get_file_type (uri) != G_FILE_TYPE_DIRECTORY)
1239 return FALSE;
1240
1241 /* handle_hash is a hash of common_theme_dir names to their monitor_data. We
1242 * use it to remove the monitor handles when a dir is removed.
1243 */
1244 tuple = g_new (CallbackTuple, 1);
1245 tuple->handle_hash = g_hash_table_new_full (g_str_hash, g_str_equal, (GDestroyNotify) g_free, (GDestroyNotify) g_free);
1246 tuple->priority = priority;
1247
1248 /* Monitor the top directory */
1249 if ((monitor = g_file_monitor_directory (uri, G_FILE_MONITOR_NONE, NULL, NULL)) != NULL) {
1250 g_signal_connect (monitor, "changed",
1251 (GCallback) (icon_theme ? top_icon_theme_dir_changed : top_theme_dir_changed),
1252 tuple);
1253 } else {
1254 g_hash_table_destroy (tuple->handle_hash);
1255 g_free (tuple);
1256 return FALSE;
1257 }
1258
1259 /* Go through the directory to add monitoring */
1260 enumerator = g_file_enumerate_children (uri,
1261 G_FILE_ATTRIBUTE_STANDARD_TYPE ","
1262 G_FILE_ATTRIBUTE_STANDARD_NAME,
1263 G_FILE_QUERY_INFO_NONE,
1264 NULL, NULL);
1265 if (enumerator == NULL)
1266 return FALSE;
1267
1268 while ((file_info = g_file_enumerator_next_file (enumerator, NULL, NULL))) {
1269 GFileType type = g_file_info_get_file_type (file_info);
1270
1271 if (type == G_FILE_TYPE_DIRECTORY || type == G_FILE_TYPE_SYMBOLIC_LINK) {
1272 GFile *child;
1273 const gchar *name;
1274 gpointer data;
1275
1276 /* Add the directory */
1277 name = g_file_info_get_name (file_info);
1278 child = g_file_get_child (uri, name);
1279
1280 if (icon_theme) {
1281 CommonIconThemeDirMonitorData *monitor_data;
1282 monitor_data = g_new0 (CommonIconThemeDirMonitorData, 1);
1283 monitor_data->priority = priority;
1284 add_common_icon_theme_dir_monitor (child, monitor_data, error);
1285 data = monitor_data;
1286 } else {
1287 CommonThemeDirMonitorData *monitor_data;
1288 monitor_data = g_new0 (CommonThemeDirMonitorData, 1);
1289 monitor_data->priority = priority;
1290 add_common_theme_dir_monitor (child, monitor_data, error);
1291 data = monitor_data;
1292 }
1293 g_object_unref (child);
1294
1295 g_hash_table_insert (tuple->handle_hash, g_strdup (name), data);
1296 }
1297 g_object_unref (file_info);
1298 }
1299 g_file_enumerator_close (enumerator, NULL, NULL);
1300
1301 return TRUE;
1302}

1721gboolean
1722mate_theme_color_scheme_parse (const gchar *scheme,
1723 GdkRGBA *colors)
1724{
1725 gchar **color_scheme_strings, **color_scheme_pair, *current_string;
1726 gint i;
1727
1728 if (!scheme || !strcmp (scheme, ""))
1729 return FALSE;
1730
1731 /* initialise the array */
1732 for (i = 0; i < NUM_SYMBOLIC_COLORS; i++)
1733 colors[i].red = colors[i].green = colors[i].blue = 0;
1734
1735 /* The color scheme string consists of name:color pairs, separated by
1736 * newlines, so first we split the string up by new line */
1737
1738 color_scheme_strings = g_strsplit (scheme, "\n", 0);
1739
1740 /* loop through the name:color pairs, and save the color if we recognise the name */
1741 i = 0;
1742 while ((current_string = color_scheme_strings[i++])) {
1743 color_scheme_pair = g_strsplit (current_string, ":", 0);
1744
1745 if (color_scheme_pair[0] != NULL && color_scheme_pair[1] != NULL) {
1746 g_strstrip (color_scheme_pair[0]);
1747 g_strstrip (color_scheme_pair[1]);
1748
1749 if (!strcmp ("fg_color", color_scheme_pair[0]))
1750 gdk_rgba_parse (&colors[COLOR_FG], color_scheme_pair[1]);
1751 else if (!strcmp ("bg_color", color_scheme_pair[0]))
1752 gdk_rgba_parse (&colors[COLOR_BG], color_scheme_pair[1]);
1753 else if (!strcmp ("text_color", color_scheme_pair[0]))
1754 gdk_rgba_parse (&colors[COLOR_TEXT], color_scheme_pair[1]);
1755 else if (!strcmp ("base_color", color_scheme_pair[0]))
1756 gdk_rgba_parse (&colors[COLOR_BASE], color_scheme_pair[1]);
1757 else if (!strcmp ("selected_fg_color", color_scheme_pair[0]))
1758 gdk_rgba_parse (&colors[COLOR_SELECTED_FG], color_scheme_pair[1]);
1759 else if (!strcmp ("selected_bg_color", color_scheme_pair[0]))
1760 gdk_rgba_parse ( &colors[COLOR_SELECTED_BG], color_scheme_pair[1]);
1761 else if (!strcmp ("tooltip_fg_color", color_scheme_pair[0]))
1762 gdk_rgba_parse (&colors[COLOR_TOOLTIP_FG], color_scheme_pair[1]);

1765 }

1767 g_strfreev (color_scheme_pair);
