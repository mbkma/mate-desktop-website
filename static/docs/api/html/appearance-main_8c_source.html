36static AppearanceData *
37init_appearance_data (int *argc, char ***argv, GOptionContext *context)
38{
39 AppearanceData *data = NULL;
40 GtkBuilder *ui;
41
42 theme_thumbnail_factory_init (*argc, *argv);
43 capplet_init (context, argc, argv);
44 activate_settings_daemon ();
45
46 /* set up the data */
47 ui = gtk_builder_new_from_resource ("/org/mate/mcc/appearance/data/appearance.ui");
48
49 data = g_new (AppearanceData, 1);
50 data->settings = g_settings_new (APPEARANCE_SCHEMA);
51 data->wp_settings = g_settings_new (WP_SCHEMA);
52
53 if (mate_gsettings_schema_exists (CAJA_SCHEMA))
54 data->caja_settings = g_settings_new (CAJA_SCHEMA);
55 else
56 data->caja_settings = NULL;
57
58 data->filechooser_settings = g_settings_new (FILECHOOSER_SCHEMA);
59 data->interface_settings = g_settings_new (INTERFACE_SCHEMA);
60
61 data->interface_gnome_settings = NULL;
62 /*Load the gnome interface schema if we are running under wayland and it is present*/
63 if (!(GDK_IS_X11_DISPLAY (gdk_display_get_default())))
64 {
65 GSettingsSchemaSource *source = g_settings_schema_source_get_default ();
66
67 if (source)
68 {
69 GSettingsSchema *schema = g_settings_schema_source_lookup (source, INTERFACE_GNOME_SCHEMA, TRUE);
70
71 if (schema)
72 data->interface_gnome_settings = g_settings_new_full (schema, NULL, NULL);
73 }
74 }
75 data->marco_settings = g_settings_new (MARCO_SCHEMA);
76 data->mouse_settings = g_settings_new (MOUSE_SCHEMA);
77 data->font_settings = g_settings_new (FONT_RENDER_SCHEMA);
78 data->ui = ui;
79 data->thumb_factory = mate_desktop_thumbnail_factory_new (MATE_DESKTOP_THUMBNAIL_SIZE_NORMAL);
80
81 return data;
82}

84static void
85main_window_response (GtkWidget *widget,
86 gint response_id,
87 AppearanceData *data)
88{
89 if (response_id == GTK_RESPONSE_CLOSE ||
90 response_id == GTK_RESPONSE_DELETE_EVENT)
91 {
92 gtk_main_quit ();
93
94 themes_shutdown (data);
95 style_shutdown (data);
96 desktop_shutdown (data);
97 font_shutdown (data);
98 support_shutdown (data);
99
100 g_object_unref (data->thumb_factory);
101 g_object_unref (data->settings);
102 g_object_unref (data->wp_settings);
103
104 if (data->caja_settings)
105 g_object_unref (data->caja_settings);
106
107 g_object_unref (data->filechooser_settings);
108 g_object_unref (data->interface_settings);
109 g_object_unref (data->marco_settings);
110 g_object_unref (data->mouse_settings);
111 g_object_unref (data->font_settings);
112 g_object_unref (data->ui);
113 }
114 else if (response_id == GTK_RESPONSE_HELP)
115 {
116 GtkNotebook *nb;
117 gint pindex;
118
119 nb = GTK_NOTEBOOK (appearance_capplet_get_widget (data, "main_notebook"));
120 pindex = gtk_notebook_get_current_page (nb);
121
122 switch (pindex)
123 {
124 case 0: /* theme */
125 capplet_help (GTK_WINDOW (widget), "goscustdesk-12");
126 break;
127 case 1: /* background */
128 capplet_help (GTK_WINDOW (widget), "goscustdesk-7");
129 break;
130 case 2: /* fonts */
131 capplet_help (GTK_WINDOW (widget), "goscustdesk-38");
132 break;
133 case 3: /* interface */
134 capplet_help (GTK_WINDOW (widget), "goscustuserinter-2");
135 break;
136 default:
137 capplet_help (GTK_WINDOW (widget), "prefs-look-and-feel");
138 break;
139 }
140 }
141}

143int
144main (int argc, char **argv)
145{
146 AppearanceData *data;
147 GtkWidget *w;
148 GtkWidget *nb;
149 GtkWidget *nb_custom_theme;
150 GtkStyleContext *context;
151
152 gchar *install_filename = NULL;
153 gchar *start_page = NULL;
154 gchar **wallpaper_files = NULL;
155 GOptionContext *option_context;
156 GOptionEntry option_entries[] = {
157 { "install-theme",
158 'i',
159 G_OPTION_FLAG_IN_MAIN,
160 G_OPTION_ARG_FILENAME,
161 &install_filename,
162 N_("Specify the filename of a theme to install"),
163 N_("filename") },
164 { "show-page",
165 'p',
166 G_OPTION_FLAG_IN_MAIN,
167 G_OPTION_ARG_STRING,
168 &start_page,
169 /* TRANSLATORS: don't translate the terms in brackets */
170 N_("Specify the name of the page to show (theme|background|fonts|interface)"),
171 N_("page") },
172 { G_OPTION_REMAINING,
173 0,
174 G_OPTION_FLAG_IN_MAIN,
175 G_OPTION_ARG_FILENAME_ARRAY,
176 &wallpaper_files,
177 NULL,
178 N_("[WALLPAPER...]") },
179 { NULL, 0, 0, G_OPTION_ARG_NONE, NULL, NULL, NULL }
180 };
181
182 option_context = g_option_context_new (NULL);
183 g_option_context_add_main_entries (option_context, option_entries, GETTEXT_PACKAGE);
184
185 /* init */
186 data = init_appearance_data (&argc, &argv, option_context);
187 if (!data)
188 return 1;
189
190 /* init tabs */
191 themes_init (data);
192 style_init (data);
193 desktop_init (data, (const gchar **) wallpaper_files);
194 g_strfreev (wallpaper_files);
195 font_init (data);
196 ui_init (data);
197
198 /* init support for other window managers */
199 support_init (data);
200
201 /* prepare the main window */
202 w = appearance_capplet_get_widget (data, "appearance_window");
203 context = gtk_widget_get_style_context (w);
204 gtk_style_context_add_class (context, "appearance-window");
205
206 capplet_set_icon (w, "preferences-desktop-theme");
207 gtk_widget_show_all (w);
208
209#if !GTK_CHECK_VERSION(3, 24, 12)
210 gtk_widget_hide (GTK_WIDGET (gtk_builder_get_object (data->ui, "filechooser_type_column_toggle")));
211#endif /* !GTK_CHECK_VERSION(3, 24, 12) */
212
213 g_signal_connect_after (w, "response",
214 (GCallback) main_window_response, data);
215
216 /* default to background page if files were given on the command line */
217 if (wallpaper_files && !install_filename && !start_page)
218 start_page = g_strdup ("background");
219
220 nb = appearance_capplet_get_widget (data, "main_notebook");
221 gtk_widget_add_events (nb, GDK_SCROLL_MASK);
222 g_signal_connect (nb,
223 "scroll-event",
224 G_CALLBACK (capplet_notebook_scroll_event_cb),
225 NULL);
226
227 if (start_page != NULL) {
228 gchar *page_name;
229
230 page_name = g_strconcat (start_page, "_vbox", NULL);
231 g_free (start_page);
232
233 w = appearance_capplet_get_widget (data, page_name);
234 if (w != NULL) {
235 gint pindex;
236 GtkNotebook *notebook;
237
238 notebook = GTK_NOTEBOOK (nb);
239 pindex = gtk_notebook_page_num (notebook, w);
240 if (pindex != -1)
241 gtk_notebook_set_current_page (notebook, pindex);
242 }
243 g_free (page_name);
244 }
245
246 if (install_filename != NULL) {
247 GFile *inst = g_file_new_for_commandline_arg (install_filename);
248 g_free (install_filename);
249 mate_theme_install (inst, GTK_WINDOW (w));
250 g_object_unref (inst);
251 }
252
253 g_option_context_free (option_context);
254
255 nb_custom_theme = appearance_capplet_get_widget (data, "notebook2");
256 gtk_widget_add_events (nb_custom_theme, GDK_SCROLL_MASK);
257 g_signal_connect (nb_custom_theme,
258 "scroll-event",
259 G_CALLBACK (capplet_notebook_scroll_event_cb),
260 NULL);
261
262 /* start the mainloop */
263 gtk_main ();
264
265 /* free stuff */
266 g_free (data);
267
268 return 0;
269}
