57dropbox_client_util_command_parse_arg(const gchar *line, GHashTable *return_table) {
58 gchar **argval;
59 guint len;
60 gboolean retval;
61
62 argval = g_strsplit(line, "\t", 0);
63 len = g_strv_length(argval);
64
65 /* debug("parsed: (%d) %s", len, line); */
66
67 if (len > 1) {
68 int i;
69 gchar **vals;
70
71 vals = g_new(gchar *, len);
72 vals[len - 1] = NULL;
73
74 for (i = 1; argval[i] != NULL; i++) {
75 vals[i-1] = dropbox_client_util_desanitize(argval[i]);
76
77 }
78
79 g_hash_table_insert(return_table,
80 dropbox_client_util_desanitize(argval[0]),
81 vals);
82 retval = TRUE;
83 }
84 else {
85 retval = FALSE;
86 }
87
88 g_strfreev(argval);
89 return retval;
90}
