152void
153alsa_stream_control_set_data (AlsaStreamControl *control, AlsaControlData *data)
154{
155 MateMixerStreamControlFlags flags = MATE_MIXER_STREAM_CONTROL_NO_FLAGS;
156 MateMixerStreamControl *mmsc;
157 gboolean mute = FALSE;
158
159 g_return_if_fail (ALSA_IS_STREAM_CONTROL (control));
160 g_return_if_fail (data != NULL);
161
162 mmsc = MATE_MIXER_STREAM_CONTROL (control);
163
164 control->priv->data = *data;
165
166 g_object_freeze_notify (G_OBJECT (control));
167
168 if (data->channels > 0) {
169 if (data->switch_usable == TRUE) {
170 /* If the mute switch is joined, all the channels get the same value,
171 * otherwise the element has per-channel mute, which we don't support.
172 * In that case, treat the control as unmuted if any channel is
173 * unmuted. */
174 if (data->channels == 1 || data->switch_joined == TRUE) {
175 mute = data->m[0];
176 } else {
177 gint i;
178 mute = TRUE;
179 for (i = 0; i < data->channels; i++)
180 if (data->m[i] == FALSE) {
181 mute = FALSE;
182 break;
183 }
184 }
185
186 flags |= MATE_MIXER_STREAM_CONTROL_MUTE_READABLE;
187 if (data->active == TRUE)
188 flags |= MATE_MIXER_STREAM_CONTROL_MUTE_WRITABLE;
189 }
190
191 flags |= MATE_MIXER_STREAM_CONTROL_VOLUME_READABLE;
192 if (data->active == TRUE)
193 flags |= MATE_MIXER_STREAM_CONTROL_VOLUME_WRITABLE;
194
195 if (data->max_decibel > -MATE_MIXER_INFINITY)
196 flags |= MATE_MIXER_STREAM_CONTROL_HAS_DECIBEL;
197
198 control->priv->channel_mask = _mate_mixer_create_channel_mask (data->c, data->channels);
199
200 if (data->volume_joined == FALSE) {
201 if (MATE_MIXER_CHANNEL_MASK_HAS_LEFT (control->priv->channel_mask) &&
202 MATE_MIXER_CHANNEL_MASK_HAS_RIGHT (control->priv->channel_mask))
203 flags |= MATE_MIXER_STREAM_CONTROL_CAN_BALANCE;
204
205 if (MATE_MIXER_CHANNEL_MASK_HAS_FRONT (control->priv->channel_mask) &&
206 MATE_MIXER_CHANNEL_MASK_HAS_BACK (control->priv->channel_mask))
207 flags |= MATE_MIXER_STREAM_CONTROL_CAN_FADE;
208 }
209
210 g_object_notify (G_OBJECT (control), "volume");
211 } else {
212 control->priv->channel_mask = 0;
213 }
214
215 _mate_mixer_stream_control_set_mute (mmsc, mute);
216 _mate_mixer_stream_control_set_flags (mmsc, flags);
217
218 if (flags & MATE_MIXER_STREAM_CONTROL_CAN_BALANCE)
219 _mate_mixer_stream_control_set_balance (mmsc, control_data_get_balance (data));
220 if (flags & MATE_MIXER_STREAM_CONTROL_CAN_FADE)
221 _mate_mixer_stream_control_set_fade (mmsc, control_data_get_fade (data));
222
223 g_object_thaw_notify (G_OBJECT (control));
224}

254static gboolean
255alsa_stream_control_set_mute (MateMixerStreamControl *mmsc, gboolean mute)
256{
257 AlsaStreamControl *control;
258 gboolean change = FALSE;
259 gint i;
260
261 g_return_val_if_fail (ALSA_IS_STREAM_CONTROL (mmsc), FALSE);
262
263 control = ALSA_STREAM_CONTROL (mmsc);
264
265 /* If the switch is joined, only verify the first channel */
266 if (control->priv->data.switch_joined == TRUE) {
267 if (control->priv->data.m[0] != mute)
268 change = TRUE;
269 } else {
270 /* Avoid trying to set the mute if all channels are already at the
271 * selected mute value */
272 for (i = 0; i < control->priv->data.channels; i++)
273 if (control->priv->data.m[i] != mute) {
274 change = TRUE;
275 break;
276 }
277 }
278
279 if (change == TRUE) {
280 AlsaStreamControlClass *klass;
281
282 klass = ALSA_STREAM_CONTROL_GET_CLASS (control);
283 if (klass->set_mute (control, mute) == FALSE)
284 return FALSE;
285
286 for (i = 0; i < control->priv->data.channels; i++)
287 control->priv->data.m[i] = mute;
288 }
289 return TRUE;
290}

308static gboolean
309alsa_stream_control_set_volume (MateMixerStreamControl *mmsc, guint volume)
310{
311 AlsaStreamControl *control;
312 gboolean change = FALSE;
313 gint i;
314
315 g_return_val_if_fail (ALSA_IS_STREAM_CONTROL (mmsc), FALSE);
316
317 control = ALSA_STREAM_CONTROL (mmsc);
318 volume = CLAMP (volume, control->priv->data.min, control->priv->data.max);
319
320 /* If the volume is joined, only verify the first channel */
321 if (control->priv->data.volume_joined == TRUE) {
322 if (control->priv->data.v[0] != volume)
323 change = TRUE;
324 } else {
325 /* Avoid trying to set the volume if all channels are already at the
326 * selected volume */
327 for (i = 0; i < control->priv->data.channels; i++)
328 if (control->priv->data.v[i] != volume) {
329 change = TRUE;
330 break;
331 }
332 }
333
334 if (change == TRUE) {
335 AlsaStreamControlClass *klass;
336
337 klass = ALSA_STREAM_CONTROL_GET_CLASS (control);
338 if (klass->set_volume (control, volume) == FALSE)
339 return FALSE;
340
341 for (i = 0; i < control->priv->data.channels; i++)
342 control->priv->data.v[i] = volume;
343
344 control->priv->data.volume = volume;
345
346 g_object_notify (G_OBJECT (control), "volume");
347 }
348 return TRUE;
349}

435static gboolean
436alsa_stream_control_set_channel_volume (MateMixerStreamControl *mmsc, guint channel, guint volume)
437{
438 AlsaStreamControl *control;
439
440 g_return_val_if_fail (ALSA_IS_STREAM_CONTROL (mmsc), FALSE);
441
442 control = ALSA_STREAM_CONTROL (mmsc);
443
444 if (channel >= control->priv->data.channels)
445 return FALSE;
446
447 /* Set volume for all channels at once when channels are joined */
448 if (control->priv->data.volume_joined == TRUE)
449 return alsa_stream_control_set_volume (mmsc, volume);
450
451 volume = CLAMP (volume, control->priv->data.min, control->priv->data.max);
452
453 if (volume != control->priv->data.v[channel]) {
454 AlsaStreamControlClass *klass;
455
456 /* Convert channel index to ALSA channel position and make sure it is valid */
457 snd_mixer_selem_channel_id_t c = alsa_channel_map_to[control->priv->data.c[channel]];
458 if (G_UNLIKELY (c == SND_MIXER_SCHN_UNKNOWN)) {
459 g_warn_if_reached ();
460 return FALSE;
461 }
462
463 klass = ALSA_STREAM_CONTROL_GET_CLASS (control);
464 if (klass->set_channel_volume (control, c, volume) == FALSE)
465 return FALSE;
466
467 control->priv->data.v[channel] = volume;
468
469 /* The global volume is always set to the highest channel volume */
470 control->priv->data.volume = MAX (control->priv->data.volume, volume);
471
472 g_object_notify (G_OBJECT (control), "volume");
473 }
474 return TRUE;
475}

521static gboolean
522alsa_stream_control_set_balance (MateMixerStreamControl *mmsc, gfloat balance)
523{
524 AlsaStreamControlClass *klass;
525 AlsaStreamControl *control;
526 AlsaControlData *data;
527 guint left,
528 right;
529 guint nleft,
530 nright;
531 guint max;
532 guint channel;
533
534 g_return_val_if_fail (ALSA_IS_STREAM_CONTROL (mmsc), FALSE);
535
536 control = ALSA_STREAM_CONTROL (mmsc);
537 klass = ALSA_STREAM_CONTROL_GET_CLASS (control);
538
539 data = &control->priv->data;
540 control_data_get_average_left_right (data, &left, &right);
541
542 max = MAX (left, right);
543 if (balance <= 0) {
544 nright = (balance + 1.0f) * max;
545 nleft = max;
546 } else {
547 nleft = (1.0f - balance) * max;
548 nright = max;
549 }
550
551 for (channel = 0; channel < data->channels; channel++) {
552 gboolean lc = MATE_MIXER_IS_LEFT_CHANNEL (data->c[channel]);
553 gboolean rc = MATE_MIXER_IS_RIGHT_CHANNEL (data->c[channel]);
554
555 if (lc == TRUE || rc == TRUE) {
556 guint volume;
557 if (lc == TRUE) {
558 if (left == 0)
559 volume = nleft;
560 else
561 volume = CLAMP (((guint64) data->v[channel] * (guint64) nleft) / (guint64) left,
562 data->min,
563 data->max);
564 } else {
565 if (right == 0)
566 volume = nright;
567 else
568 volume = CLAMP (((guint64) data->v[channel] * (guint64) nright) / (guint64) right,
569 data->min,
570 data->max);
571 }
572
573 if (klass->set_channel_volume (control,
574 alsa_channel_map_to[data->c[channel]],
575 volume) == TRUE)
576 data->v[channel] = volume;
577 }
578 }
579 return TRUE;
580}

675static void
