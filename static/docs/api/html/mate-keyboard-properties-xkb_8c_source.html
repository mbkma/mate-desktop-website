58static void
59set_model_text (GtkWidget * picker, gchar * value)
60{
61 XklConfigItem *ci = xkl_config_item_new ();
62 char *model = NULL;
63
64 if (value != NULL && value[0] != '\0') {
65 model = g_strdup(value);
66 }
67
68 if (model == NULL) {
69 model = g_strdup(initial_config.model);
70 if (model == NULL)
71 model = g_strdup("");
72 }
73
74 g_snprintf (ci->name, sizeof (ci->name), "%s", model);
75
76 if (xkl_config_registry_find_model (config_registry, ci)) {
77 char *d;
78
79 d = xci_desc_to_utf8 (ci);
80 gtk_button_set_label (GTK_BUTTON (picker), d);
81 g_free (d);
82 } else {
83 gtk_button_set_label (GTK_BUTTON (picker), _("Unknown"));
84 }
85 g_object_unref (G_OBJECT (ci));
86 g_free (model);
87}

175void

179 WID ("chk_new_windows_inherit_layout");

184 engine = xkl_engine_get_instance (GDK_DISPLAY_XDISPLAY(gdk_display_get_default()));

186
187 matekbd_desktop_config_init (&desktop_config, engine);
188 matekbd_desktop_config_load_from_gsettings (&desktop_config);
189
190 xkl_config_registry_load (config_registry, desktop_config.load_extra_items);
191
192 matekbd_keyboard_config_init (&initial_config, engine);
193 matekbd_keyboard_config_load_from_x_initial (&initial_config, NULL);
194
195 setup_model_entry (dialog);
196
197 g_settings_bind (xkb_general_settings,
198 "group-per-window",
199 WID ("chk_separate_group_per_window"),
200 "active",
201 G_SETTINGS_BIND_DEFAULT);

279static void
280xkb_save_gslist_as_strv (gchar *schema, gchar *key, GSList *list)
281{
282 GSettings *settings;
283 GArray *array;
284 GSList *l;
285 array = g_array_new (TRUE, TRUE, sizeof (gchar *));
286 for (l = list; l; l = l->next) {
287 array = g_array_append_val (array, l->data);
288 }
289 settings = g_settings_new (schema);
290 g_settings_set_strv (settings, key, (const gchar **) array->data);
291 g_array_free (array, TRUE);
292 g_object_unref (settings);
293}
