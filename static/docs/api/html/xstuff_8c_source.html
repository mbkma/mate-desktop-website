80static gboolean
81zoom_draw (GtkWidget *widget,
82 cairo_t *cr,
83 gpointer user_data)
84{
85 CompositedZoomData *zoom;
86
87 zoom = user_data;
88
89 if (zoom->size >= zoom->size_end) {
90 if (zoom->timeout_id)
91 g_source_remove (zoom->timeout_id);
92 zoom->timeout_id = 0;
93
94 gtk_widget_hide (widget);
95 g_idle_add (idle_destroy, widget);
96
97 g_object_unref (zoom->pixbuf);
98 zoom->pixbuf = NULL;
99
100 g_slice_free (CompositedZoomData, zoom);
101 } else {
102 GdkPixbuf *scaled;
103 int width, height;
104 int x = 0, y = 0;
105
106 gtk_window_get_size (GTK_WINDOW (widget), &width, &height);
107
108 zoom->size += MAX ((zoom->size_end - zoom->size_start) / ZOOM_STEPS, 1);
109 zoom->opacity -= 1.0 / ((double) ZOOM_STEPS + 1);
110
111 scaled = gdk_pixbuf_scale_simple (zoom->pixbuf,
112 zoom->size, zoom->size,
113 GDK_INTERP_BILINEAR);
114
115 switch (zoom->orientation) {
116 case PANEL_ORIENTATION_TOP:
117 x = (width - gdk_pixbuf_get_width (scaled)) / 2;
118 y = 0;
119 break;
120
121 case PANEL_ORIENTATION_RIGHT:
122 x = width - gdk_pixbuf_get_width (scaled);
123 y = (height - gdk_pixbuf_get_height (scaled)) / 2;
124 break;
125
126 case PANEL_ORIENTATION_BOTTOM:
127 x = (width - gdk_pixbuf_get_width (scaled)) / 2;
128 y = height - gdk_pixbuf_get_height (scaled);
129 break;
130
131 case PANEL_ORIENTATION_LEFT:
132 x = 0;
133 y = (height - gdk_pixbuf_get_height (scaled)) / 2;
134 break;
135 }
136
137 cairo_set_operator (cr, CAIRO_OPERATOR_SOURCE);
138 cairo_set_source_rgba (cr, 0, 0, 0, 0.0);
139 cairo_rectangle (cr, 0, 0, width, height);
140 cairo_fill (cr);
141
142 gdk_cairo_set_source_pixbuf (cr, scaled, x, y);
143 cairo_set_operator (cr, CAIRO_OPERATOR_OVER);
144 cairo_paint_with_alpha (cr, MAX (zoom->opacity, 0));
145
146 g_object_unref (scaled);
147 }
148
149 return FALSE;
150}

152static void
153draw_zoom_animation_composited (GdkScreen *gscreen,
154 int x, int y, int w, int h,
155 GdkPixbuf *pixbuf,
156 PanelOrientation orientation)
157{
158 GtkWidget *win;
159 CompositedZoomData *zoom;
160 int wx = 0, wy = 0;
161
162 w += 2;
163 h += 2;
164
165 zoom = g_slice_new (CompositedZoomData);
166 zoom->size = w;
167 zoom->size_start = w;
168 zoom->size_end = w * ZOOM_FACTOR;
169 zoom->orientation = orientation;
170 zoom->opacity = 1.0;
171 zoom->pixbuf = g_object_ref (pixbuf);
172 zoom->timeout_id = 0;
173
174 win = gtk_window_new (GTK_WINDOW_POPUP);
175
176 gtk_window_set_screen (GTK_WINDOW (win), gscreen);
177 gtk_window_set_keep_above (GTK_WINDOW (win), TRUE);
178 gtk_window_set_decorated (GTK_WINDOW (win), FALSE);
179 gtk_widget_set_app_paintable(win, TRUE);
180 gtk_widget_set_visual (win, gdk_screen_get_rgba_visual (gscreen));
181
182 gtk_window_set_gravity (GTK_WINDOW (win), GDK_GRAVITY_STATIC);
183 gtk_window_set_default_size (GTK_WINDOW (win),
184 w * ZOOM_FACTOR, h * ZOOM_FACTOR);
185
186 switch (zoom->orientation) {
187 case PANEL_ORIENTATION_TOP:
188 wx = x - w * (ZOOM_FACTOR / 2);
189 wy = y;
190 break;
191
192 case PANEL_ORIENTATION_RIGHT:
193 wx = x - w * (ZOOM_FACTOR - 1);
194 wy = y - h * (ZOOM_FACTOR / 2);
195 break;
196
197 case PANEL_ORIENTATION_BOTTOM:
198 wx = x - w * (ZOOM_FACTOR / 2);
199 wy = y - h * (ZOOM_FACTOR - 1);
200 break;
201
202 case PANEL_ORIENTATION_LEFT:
203 wx = x;
204 wy = y - h * (ZOOM_FACTOR / 2);
205 break;
206 }
207
208 gtk_window_move (GTK_WINDOW (win), wx, wy);
209
210 g_signal_connect (win, "draw",
211 G_CALLBACK (zoom_draw),
212 zoom);
213
214 /* see doc for gtk_widget_set_app_paintable() */
215 gtk_widget_realize (win);
216 gdk_window_set_background_pattern (gtk_widget_get_window (win), NULL);
217 gtk_widget_show (win);
218
219 zoom->timeout_id = g_timeout_add (ZOOM_DELAY,
220 (GSourceFunc) zoom_timeout,
221 win);
222}

224static void
225draw_zoom_animation (GdkScreen *gscreen,
226 int x, int y, int w, int h,
227 int fx, int fy, int fw, int fh,
228 int steps)
229{
230#define FRAMES (MINIATURIZE_ANIMATION_FRAMES_Z)
231 float cx[FRAMES], cy[FRAMES], cw[FRAMES], ch[FRAMES];
232 int cxi[FRAMES], cyi[FRAMES], cwi[FRAMES], chi[FRAMES];
233 float xstep, ystep, wstep, hstep;
234 int i, j;
235 GC frame_gc;
236 XGCValues gcv;
237 GdkColor color = { .pixel = 0, .red = 65535, .green = 65535, .blue = 65535 };
238 Display *dpy;
239 Window root_win;
240 int screen;
241 int depth;
242
243 dpy = gdk_x11_display_get_xdisplay (gdk_screen_get_display (gscreen));
244 root_win = GDK_WINDOW_XID (gdk_screen_get_root_window (gscreen));
245 screen = gdk_x11_screen_get_screen_number (gscreen);
246 depth = DefaultDepth(dpy,screen);
247
248 /* frame GC */
249 gcv.function = GXxor;
250 /* this will raise the probability of the XORed color being different
251 * of the original color in PseudoColor when not all color cells are
252 * initialized */
253 if (DefaultVisual(dpy, screen)->class==PseudoColor)
254 gcv.plane_mask = (1<<(depth-1))|1;
255 else
256 gcv.plane_mask = AllPlanes;
257 gcv.foreground = color.pixel;
258 if (gcv.foreground == 0)
259 gcv.foreground = 1;
260 gcv.line_width = 1;
261 gcv.subwindow_mode = IncludeInferiors;
262 gcv.graphics_exposures = False;
263
264 frame_gc = XCreateGC(dpy, root_win, GCForeground|GCGraphicsExposures
265 |GCFunction|GCSubwindowMode|GCLineWidth
266 |GCPlaneMask, &gcv);
267
268 xstep = (float)(fx-x)/steps;
269 ystep = (float)(fy-y)/steps;
270 wstep = (float)(fw-w)/steps;
271 hstep = (float)(fh-h)/steps;
272
273 for (j=0; j<FRAMES; j++) {
274 cx[j] = (float)x;
275 cy[j] = (float)y;
276 cw[j] = (float)w;
277 ch[j] = (float)h;
278 cxi[j] = (int)cx[j];
279 cyi[j] = (int)cy[j];
280 cwi[j] = (int)cw[j];
281 chi[j] = (int)ch[j];
282 }
283 XGrabServer(dpy);
284 for (i=0; i<steps; i++) {
285 for (j=0; j<FRAMES; j++) {
286 XDrawRectangle(dpy, root_win, frame_gc, cxi[j], cyi[j], cwi[j], chi[j]);
287 }
288 XFlush(dpy);
289#if (MINIATURIZE_ANIMATION_DELAY_Z > 0)
290 usleep(MINIATURIZE_ANIMATION_DELAY_Z);
291#else
292 usleep(10);
293#endif
294 for (j=0; j<FRAMES; j++) {
295 XDrawRectangle(dpy, root_win, frame_gc,
296 cxi[j], cyi[j], cwi[j], chi[j]);
297 if (j<FRAMES-1) {
298 cx[j]=cx[j+1];
299 cy[j]=cy[j+1];
300 cw[j]=cw[j+1];
301 ch[j]=ch[j+1];
302
303 cxi[j]=cxi[j+1];
304 cyi[j]=cyi[j+1];
305 cwi[j]=cwi[j+1];
306 chi[j]=chi[j+1];
307
308 } else {
309 cx[j]+=xstep;
310 cy[j]+=ystep;
311 cw[j]+=wstep;
312 ch[j]+=hstep;
313
314 cxi[j] = (int)cx[j];
315 cyi[j] = (int)cy[j];
316 cwi[j] = (int)cw[j];
317 chi[j] = (int)ch[j];
318 }
319 }
320 }
321
322 for (j=0; j<FRAMES; j++) {
323 XDrawRectangle(dpy, root_win, frame_gc,
324 cxi[j], cyi[j], cwi[j], chi[j]);
325 }
326 XFlush(dpy);
327#if (MINIATURIZE_ANIMATION_DELAY_Z > 0)
328 usleep(MINIATURIZE_ANIMATION_DELAY_Z);
329#else
330 usleep(10);
331#endif
332 for (j=0; j<FRAMES; j++) {
333 XDrawRectangle(dpy, root_win, frame_gc,
334 cxi[j], cyi[j], cwi[j], chi[j]);
335 }
336
337 XUngrabServer(dpy);
338 XFreeGC (dpy, frame_gc);
339}
340#undef FRAMES

342void
343xstuff_zoom_animate (GtkWidget *widget,
344 cairo_surface_t *surface,
345 PanelOrientation orientation,
346 GdkRectangle *opt_rect)
347{
348 GdkScreen *gscreen;
349 GdkRectangle rect, dest;
350 GtkAllocation allocation;
351
352 if (opt_rect)
353 rect = *opt_rect;
354 else {
355 gdk_window_get_origin (gtk_widget_get_window (widget), &rect.x, &rect.y);
356 gtk_widget_get_allocation (widget, &allocation);
357 if (!gtk_widget_get_has_window (widget)) {
358 rect.x += allocation.x;
359 rect.y += allocation.y;
360 }
361 rect.height = allocation.height;
362 rect.width = allocation.width;
363 }
364
365 gscreen = gtk_widget_get_screen (widget);
366
367 if (gdk_screen_is_composited (gscreen) && surface) {
368 GdkPixbuf *pixbuf = gdk_pixbuf_get_from_surface (surface,
369 0, 0,
370 cairo_image_surface_get_width (surface),
371 cairo_image_surface_get_height (surface));
372 draw_zoom_animation_composited (gscreen,
373 rect.x, rect.y,
374 rect.width, rect.height,
375 pixbuf, orientation);
376 g_object_unref (pixbuf);
377 } else {
378 GdkDisplay *display = gdk_screen_get_display (gscreen);
379 GdkMonitor *monitor = gdk_display_get_monitor_at_window (display,
380 gtk_widget_get_window (widget));
381 gdk_monitor_get_geometry (monitor, &dest);
382
383 draw_zoom_animation (gscreen,
384 rect.x, rect.y, rect.width, rect.height,
385 dest.x, dest.y, dest.width, dest.height,
386 MINIATURIZE_ANIMATION_STEPS_Z);
387 }
388}

407static int
408xstuff_x_error_handler (Display *display, XErrorEvent *error)
409{
410 if (!error->error_code)
411 return 0;
412
413 /* If we got a BadDrawable or a BadWindow, we ignore it for now.
414 * FIXME: We need to somehow distinguish real errors from
415 * X-server-induced errors. Keeping a list of windows for which we will
416 * ignore BadDrawables would be a good idea. */
417 if (error->error_code == BadDrawable ||
418 error->error_code == BadWindow)
419 return 0;
420
421 return xstuff_old_x_error_handler (display, error);
422}
