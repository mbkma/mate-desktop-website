51static void
52matekbd_indicator_config_load_font (MatekbdIndicatorConfig * ind_config)
53{
54 ind_config->font_family =
55 g_settings_get_string (ind_config->settings,
56 MATEKBD_INDICATOR_CONFIG_KEY_FONT_FAMILY);
57
58 if (ind_config->font_family == NULL ||
59 ind_config->font_family[0] == '\0') {
60 PangoFontDescription *fd = NULL;
61 GtkWidgetPath *widget_path = gtk_widget_path_new ();
62 GtkStyleContext *context = gtk_style_context_new ();
63
64 gtk_widget_path_append_type (widget_path, GTK_TYPE_WINDOW);
65 gtk_widget_path_iter_set_name (widget_path, -1 , "PanelWidget");
66
67 gtk_style_context_set_path (context, widget_path);
68 gtk_style_context_set_screen (context, gdk_screen_get_default ());
69 gtk_style_context_set_state (context, GTK_STATE_FLAG_NORMAL);
70 gtk_style_context_add_class (context, GTK_STYLE_CLASS_DEFAULT);
71 gtk_style_context_add_class (context, "gnome-panel-menu-bar");
72 gtk_style_context_add_class (context, "mate-panel-menu-bar");
73
74 gtk_style_context_get (context, GTK_STATE_FLAG_NORMAL,
75 GTK_STYLE_PROPERTY_FONT, &fd, NULL);
76
77 if (fd != NULL) {
78 ind_config->font_family =
79 g_strdup (pango_font_description_to_string(fd));
80 }
81
82 g_object_unref (G_OBJECT (context));
83 gtk_widget_path_unref (widget_path);
84 }
85 xkl_debug (150, "font: [%s]\n", ind_config->font_family);
86
87}

89static void
90matekbd_indicator_config_load_colors (MatekbdIndicatorConfig * ind_config)
91{
92 ind_config->foreground_color =
93 g_settings_get_string (ind_config->settings,
94 MATEKBD_INDICATOR_CONFIG_KEY_FOREGROUND_COLOR);
95
96 if (ind_config->foreground_color == NULL ||
97 ind_config->foreground_color[0] == '\0') {
98 GtkWidgetPath *widget_path = gtk_widget_path_new ();
99 GtkStyleContext *context = gtk_style_context_new ();
100 GdkRGBA fg_color;
101
102 gtk_widget_path_append_type (widget_path, GTK_TYPE_WINDOW);
103 gtk_widget_path_iter_set_name (widget_path, -1 , "PanelWidget");
104
105 gtk_style_context_set_path (context, widget_path);
106 gtk_style_context_set_screen (context, gdk_screen_get_default ());
107 gtk_style_context_set_state (context, GTK_STATE_FLAG_NORMAL);
108 gtk_style_context_add_class (context, GTK_STYLE_CLASS_DEFAULT);
109 gtk_style_context_add_class (context, "gnome-panel-menu-bar");
110 gtk_style_context_add_class (context, "mate-panel-menu-bar");
111
112 gtk_style_context_get_color (context,
113 GTK_STATE_FLAG_NORMAL, &fg_color);
114 ind_config->foreground_color =
115 g_strdup_printf ("%g %g %g",
116 fg_color.red,
117 fg_color.green,
118 fg_color.blue);
119
120 g_object_unref (G_OBJECT (context));
121 gtk_widget_path_unref (widget_path);
122 }
123
124 ind_config->background_color =
125 g_settings_get_string (ind_config->settings,
126 MATEKBD_INDICATOR_CONFIG_KEY_BACKGROUND_COLOR);
127}

139gchar *
140matekbd_indicator_config_get_images_file (MatekbdIndicatorConfig *
141 ind_config,
142 MatekbdKeyboardConfig *
143 kbd_config, int group)
144{
145 char *image_file = NULL;
146 GtkIconInfo *icon_info = NULL;
147
148 if (!ind_config->show_flags)
149 return NULL;
150
151 if ((kbd_config->layouts_variants != NULL) &&
152 (g_strv_length (kbd_config->layouts_variants) > group)) {
153 char *full_layout_name =
154 kbd_config->layouts_variants[group];
155
156 if (full_layout_name != NULL) {
157 char *l, *v;
158 matekbd_keyboard_config_split_items (full_layout_name,
159 &l, &v);
160 if (l != NULL) {
161 /* probably there is something in theme? */
162 icon_info = gtk_icon_theme_lookup_icon
163 (ind_config->icon_theme, l, 48, 0);
164
165 /* Unbelievable but happens */
166 if (icon_info != NULL &&
167 gtk_icon_info_get_filename (icon_info) == NULL) {
168 g_object_unref (icon_info);
169 icon_info = NULL;
170 }
171 }
172 }
173 }
174 /* fallback to the default value */
175 if (icon_info == NULL) {
176 icon_info = gtk_icon_theme_lookup_icon
177 (ind_config->icon_theme, "stock_dialog-error", 48, 0);
178 }
179 if (icon_info != NULL) {
180 image_file =
181 g_strdup (gtk_icon_info_get_filename (icon_info));
182 g_object_unref (icon_info);
183 }
184
185 return image_file;
186}

188void
189matekbd_indicator_config_load_image_filenames (MatekbdIndicatorConfig *
190 ind_config,
191 MatekbdKeyboardConfig *
192 kbd_config)
193{
194 int i;
195 ind_config->image_filenames = NULL;
196
197 if (!ind_config->show_flags)
198 return;
199
200 for (i = xkl_engine_get_max_num_groups (ind_config->engine);
201 --i >= 0;) {
202 gchar *image_file =
203 matekbd_indicator_config_get_images_file (ind_config,
204 kbd_config,
205 i);
206 ind_config->image_filenames =
207 g_slist_prepend (ind_config->image_filenames,
208 image_file);
209 }
210}

244 gtk_icon_theme_append_search_path (ind_config->icon_theme,
245 sp =
246 g_build_filename (DATADIR,

252 sp =
