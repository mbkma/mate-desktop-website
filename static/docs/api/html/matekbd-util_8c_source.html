36static void
37matekbd_log_appender (const char file[], const char function[],

64GdkRectangle *
65matekbd_preview_load_position (void)
66{
67 GdkRectangle *rv = NULL;
68 gint x, y, w, h;
69 GSettings* settings = g_settings_new (MATEKBD_PREVIEW_CONFIG_SCHEMA);
70
71 x = g_settings_get_int (settings, MATEKBD_PREVIEW_CONFIG_KEY_X);
72 y = g_settings_get_int (settings, MATEKBD_PREVIEW_CONFIG_KEY_Y);
73 w = g_settings_get_int (settings, MATEKBD_PREVIEW_CONFIG_KEY_WIDTH);
74 h = g_settings_get_int (settings, MATEKBD_PREVIEW_CONFIG_KEY_HEIGHT);
75
76 g_object_unref (settings);
77
78 rv = g_new (GdkRectangle, 1);
79 if (x == -1 || y == -1 || w == -1 || h == -1) {
80 /* default values should be treated as
81 * "0.75 of the screen size" */
82 GdkScreen *scr = gdk_screen_get_default ();
83 w = WidthOfScreen (gdk_x11_screen_get_xscreen (scr));
84 h = HeightOfScreen (gdk_x11_screen_get_xscreen (scr));
85 rv->x = w >> 3;
86 rv->y = h >> 3;
87 rv->width = w - (w >> 2);
88 rv->height = h - (h >> 2);
89 } else {
90 rv->x = x;
91 rv->y = y;
92 rv->width = w;
93 rv->height = h;
94 }
95 return rv;
96}

98void
99matekbd_preview_save_position (GdkRectangle * rect)
100{
101 GSettings* settings = g_settings_new (MATEKBD_PREVIEW_CONFIG_SCHEMA);
102
103 g_settings_delay (settings);
104
105 g_settings_set_int (settings, MATEKBD_PREVIEW_CONFIG_KEY_X, rect->x);
106 g_settings_set_int (settings, MATEKBD_PREVIEW_CONFIG_KEY_Y, rect->y);
107 g_settings_set_int (settings, MATEKBD_PREVIEW_CONFIG_KEY_WIDTH, rect->width);
108 g_settings_set_int (settings, MATEKBD_PREVIEW_CONFIG_KEY_HEIGHT, rect->height);
109
110 g_settings_apply (settings);
111
112 g_object_unref (settings);
113}

120gchar **
121matekbd_strv_append (gchar ** arr, gchar * element)
122{
123 gint old_length = (arr == NULL) ? 0 : g_strv_length (arr);
124 gchar **new_arr = g_new0 (gchar *, old_length + 2);
125 if (arr != NULL) {
126 memcpy (new_arr, arr, old_length * sizeof (gchar *));
127 g_free (arr);
128 }
129 new_arr[old_length] = element;
130 return new_arr;
131}
