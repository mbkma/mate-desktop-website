66static int
67get_current_desktop (GdkScreen *screen)
68{
69 Display *display;
70 Window win;
71 Atom current_desktop, type;
72 int format;
73 unsigned long n_items, bytes_after;
74 unsigned char *data_return = NULL;
75 int workspace = 0;
76
77 display = GDK_DISPLAY_XDISPLAY (gdk_screen_get_display (screen));
78 win = XRootWindow (display, GDK_SCREEN_XNUMBER (screen));
79
80 current_desktop = XInternAtom (display, "_NET_CURRENT_DESKTOP", True);
81
82 XGetWindowProperty (display,
83 win,
84 current_desktop,
85 0, G_MAXLONG,
86 False, XA_CARDINAL,
87 &type, &format, &n_items, &bytes_after,
88 &data_return);
89
90 if (type == XA_CARDINAL && format == 32 && n_items > 0)
91 workspace = (int) data_return[0];
92 if (data_return)
93 XFree (data_return);
94
95 return workspace;
96}

98static gboolean
99get_work_area (MateRRLabeler *labeler,
100 GdkRectangle *rect)
101{
102 Atom workarea;
103 Atom type;
104 Window win;
105 int format;
106 gulong num;
107 gulong leftovers;
108 gulong max_len = 4 * 32;
109 guchar *ret_workarea;
110 long *workareas;
111 int result;
112 int disp_screen;
113 int desktop;
114 Display *display;
115
116 display = GDK_DISPLAY_XDISPLAY (gdk_screen_get_display (labeler->priv->screen));
117 workarea = XInternAtom (display, "_NET_WORKAREA", True);
118
119 disp_screen = GDK_SCREEN_XNUMBER (labeler->priv->screen);
120
121 /* Defaults in case of error */
122 rect->x = 0;
123 rect->y = 0;
124 rect->width = WidthOfScreen (gdk_x11_screen_get_xscreen (labeler->priv->screen));
125 rect->height = HeightOfScreen (gdk_x11_screen_get_xscreen (labeler->priv->screen));
126
127 if (workarea == None)
128 return FALSE;
129
130 win = XRootWindow (display, disp_screen);
131 result = XGetWindowProperty (display,
132 win,
133 workarea,
134 0,
135 max_len,
136 False,
137 AnyPropertyType,
138 &type,
139 &format,
140 &num,
141 &leftovers,
142 &ret_workarea);
143
144 if (result != Success
145 || type == None
146 || format == 0
147 || leftovers
148 || num % 4) {
149 return FALSE;
150 }
151
152 desktop = get_current_desktop (labeler->priv->screen);
153
154 workareas = (long *) ret_workarea;
155 rect->x = workareas[desktop * 4];
156 rect->y = workareas[desktop * 4 + 1];
157 rect->width = workareas[desktop * 4 + 2];
158 rect->height = workareas[desktop * 4 + 3];
159
160 XFree (ret_workarea);
161
162 return TRUE;
163}

282static void
283make_palette (MateRRLabeler *labeler)
284{
285 /* The idea is that we go around an hue color wheel. We want to start
286 * at red, go around to green/etc. and stop at blue --- because magenta
287 * is evil. Eeeeek, no magenta, please!
288 *
289 * Purple would be nice, though. Remember that we are watered down
290 * (i.e. low saturation), so that would be like Like berries with cream.
291 * Mmmmm, berries.
292 */
293 double start_hue;
294 double end_hue;
295 int i;
296
297 g_assert (labeler->priv->num_outputs > 0);
298
299 labeler->priv->palette = g_new (GdkRGBA, labeler->priv->num_outputs);
300
301 start_hue = 0.0; /* red */
302 end_hue = 2.0/3; /* blue */
303
304 for (i = 0; i < labeler->priv->num_outputs; i++) {
305 double h, s, v;
306 double r, g, b;
307
308 h = start_hue + (end_hue - start_hue) / labeler->priv->num_outputs * i;
309 s = 1.0 / 3;
310 v = 1.0;
311
312 gtk_hsv_to_rgb (h, s, v, &r, &g, &b);
313
314 labeler->priv->palette[i].red = r;
315 labeler->priv->palette[i].green = g;
316 labeler->priv->palette[i].blue = b;
317 labeler->priv->palette[i].alpha = 1.0;
318 }
319}

324static gboolean
325label_window_draw_event_cb (GtkWidget *widget, cairo_t *cr, gpointer data)
326{
327 GdkRGBA *color;
328 GtkAllocation allocation;
329
330 color = g_object_get_data (G_OBJECT (widget), "color");
331 gtk_widget_get_allocation (widget, &allocation);
332
333 /* edge outline */
334
335 cairo_set_source_rgb (cr, 0, 0, 0);
336 cairo_rectangle (cr,
337 LABEL_WINDOW_EDGE_THICKNESS / 2.0,
338 LABEL_WINDOW_EDGE_THICKNESS / 2.0,
339 allocation.width - LABEL_WINDOW_EDGE_THICKNESS,
340 allocation.height - LABEL_WINDOW_EDGE_THICKNESS);
341 cairo_set_line_width (cr, LABEL_WINDOW_EDGE_THICKNESS);
342 cairo_stroke (cr);
343
344 /* fill */
345 gdk_cairo_set_source_rgba (cr, color);
346 cairo_rectangle (cr,
347 LABEL_WINDOW_EDGE_THICKNESS,
348 LABEL_WINDOW_EDGE_THICKNESS,
349 allocation.width - LABEL_WINDOW_EDGE_THICKNESS * 2,
350 allocation.height - LABEL_WINDOW_EDGE_THICKNESS * 2);
351 cairo_fill (cr);
352
353 return FALSE;
354}

394static GtkWidget *
395create_label_window (MateRRLabeler *labeler, MateRROutputInfo *output, GdkRGBA *color)
396{
397 GtkWidget *window;
398 GtkWidget *widget;
399 char *str;
400 char *display_name;
401 GdkRGBA black = { 0, 0, 0, 1.0 };
402 int x,y;
403
404 window = gtk_window_new (GTK_WINDOW_POPUP);
405 gtk_widget_set_app_paintable (window, TRUE);
406
407 gtk_container_set_border_width (GTK_CONTAINER (window), LABEL_WINDOW_PADDING + LABEL_WINDOW_EDGE_THICKNESS);
408
409 /* This is semi-dangerous. The color is part of the labeler->palette
410 * array. Note that in mate_rr_labeler_finalize(), we are careful to
411 * free the palette only after we free the windows.
412 */
413 g_object_set_data (G_OBJECT (window), "color", color);
414
415 g_signal_connect (window, "draw",
416 G_CALLBACK (label_window_draw_event_cb), labeler);
417
418 if (mate_rr_config_get_clone (labeler->priv->config)) {
419 /* Keep this string in sync with mate-control-center/capplets/display/xrandr-capplet.c:get_display_name() */
420
421 /* Translators: this is the feature where what you see on your laptop's
422 * screen is the same as your external monitor. Here, "Mirror" is being
423 * used as an adjective, not as a verb. For example, the Spanish
424 * translation could be "Pantallas en Espejo", *not* "Espejar Pantallas".
425 */
426 display_name = g_strdup_printf (_("Mirror Screens"));
427 str = g_strdup_printf ("<b>%s</b>", display_name);
428 } else {
429 display_name = g_strdup_printf ("<b>%s</b>\n<small>%s</small>", mate_rr_output_info_get_display_name (output), mate_rr_output_info_get_name (output));
430 str = g_strdup_printf ("%s", display_name);
431 }
432 g_free (display_name);
433
434 widget = gtk_label_new (NULL);
435 gtk_label_set_markup (GTK_LABEL (widget), str);
436 g_free (str);
437
438 /* Make the label explicitly black. We don't want it to follow the
439 * theme's colors, since the label is always shown against a light
440 * pastel background. See bgo#556050
441 */
442 set_override_color (widget, &black);
443
444 gtk_container_add (GTK_CONTAINER (window), widget);
445
446 /* Should we center this at the top edge of the monitor, instead of using the upper-left corner? */
447 mate_rr_output_info_get_geometry (output, &x, &y, NULL, NULL);
448 position_window (labeler, window, x, y);
449
450 gtk_widget_show_all (window);
451
452 return window;
453}

455static void
456create_label_windows (MateRRLabeler *labeler)
457{
458 int i;
459 gboolean created_window_for_clone;
460 MateRROutputInfo **outputs;
461
462 labeler->priv->windows = g_new (GtkWidget *, labeler->priv->num_outputs);
463
464 created_window_for_clone = FALSE;
465
466 outputs = mate_rr_config_get_outputs (labeler->priv->config);
467
468 for (i = 0; i < labeler->priv->num_outputs; i++) {
469 if (!created_window_for_clone && mate_rr_output_info_is_active (outputs[i])) {
470 labeler->priv->windows[i] = create_label_window (labeler, outputs[i], labeler->priv->palette + i);
471
472 if (mate_rr_config_get_clone (labeler->priv->config))
473 created_window_for_clone = TRUE;
474 } else
475 labeler->priv->windows[i] = NULL;
476 }
477}
