52static void
53ev_properties_fonts_dispose (GObject *object)
54{

78static void
79font_cell_data_func (GtkTreeViewColumn *col, GtkCellRenderer *renderer,
80 GtkTreeModel *model, GtkTreeIter *iter, gpointer user_data)
81{
82 char *name;
83 char *details;
84 char *markup;
85
86 gtk_tree_model_get (model, iter,
87 EV_DOCUMENT_FONTS_COLUMN_NAME, &name,
88 EV_DOCUMENT_FONTS_COLUMN_DETAILS, &details,
89 -1);
90
91 if (details) {
92 markup = g_strdup_printf ("<b><big>%s</big></b>\n<small>%s</small>",
93 name, details);
94 } else {
95 markup = g_strdup_printf ("<b><big>%s</big></b>", name);
96 }
97
98 g_object_set (renderer, "markup", markup, NULL);
99
100 g_free (markup);
101 g_free (details);
102 g_free (name);
103}

194void
195ev_properties_fonts_set_document (EvPropertiesFonts *properties,
196 EvDocument *document)
197{
198 GtkTreeView *tree_view = GTK_TREE_VIEW (properties->fonts_treeview);
199 GtkListStore *list_store;
200
201 properties->document = document;
202
203 list_store = gtk_list_store_new (EV_DOCUMENT_FONTS_COLUMN_NUM_COLUMNS,
204 G_TYPE_STRING, G_TYPE_STRING);
205 gtk_tree_view_set_model (tree_view, GTK_TREE_MODEL (list_store));
206
207 properties->fonts_job = ev_job_fonts_new (properties->document);
208 g_signal_connect (properties->fonts_job, "updated",
209 G_CALLBACK (job_fonts_updated_cb),
210 properties);
211 g_signal_connect (properties->fonts_job, "finished",
212 G_CALLBACK (job_fonts_finished_cb),
213 properties);
214 ev_job_scheduler_push_job (properties->fonts_job, EV_JOB_PRIORITY_NONE);
215}
