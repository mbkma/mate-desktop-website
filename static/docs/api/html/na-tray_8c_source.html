170static void
171icon_tip_free (gpointer data)
172{
173 IconTip *icontip;
174
175 if (data == NULL)
176 return;
177
178 icontip = data;
179
180 if (icontip->fixedtip != NULL)
181 gtk_widget_destroy (GTK_WIDGET (icontip->fixedtip));
182 icontip->fixedtip = NULL;
183
184 if (icontip->source_id != 0)
185 g_source_remove (icontip->source_id);
186 icontip->source_id = 0;
187
188 if (icontip->buffer != NULL)
189 {
190 g_slist_foreach (icontip->buffer, icon_tip_buffer_free, NULL);
191 g_slist_free (icontip->buffer);
192 }
193 icontip->buffer = NULL;
194
195 g_free (icontip);
196}

198static int
199icon_tip_buffer_compare (gconstpointer a,
200 gconstpointer b)
201{
202 const IconTipBuffer *buffer_a = a;
203 const IconTipBuffer *buffer_b = b;
204
205 if (buffer_a == NULL || buffer_b == NULL)
206 return !(buffer_a == buffer_b);
207
208 return buffer_a->id - buffer_b->id;
209}

228static void
229icon_tip_show_next (IconTip *icontip)
230{
231 IconTipBuffer *buffer;
232
233 if (icontip->buffer == NULL)
234 {
235 /* this will also destroy the tip window */
236 g_hash_table_remove (icontip->tray->priv->trays_screen->tip_table,
237 icontip->icon);
238 return;
239 }
240
241 if (icontip->source_id != 0)
242 g_source_remove (icontip->source_id);
243 icontip->source_id = 0;
244
245 buffer = icontip->buffer->data;
246 icontip->buffer = g_slist_remove (icontip->buffer, buffer);
247
248 if (icontip->fixedtip == NULL)
249 {
250 icontip->fixedtip = na_fixed_tip_new (icontip->icon,
251 gtk_orientable_get_orientation (GTK_ORIENTABLE (icontip->tray)));
252
253 g_signal_connect (icontip->fixedtip, "clicked",
254 G_CALLBACK (icon_tip_show_next_clicked), icontip);
255 }
256
257 na_fixed_tip_set_markup (icontip->fixedtip, buffer->text);
258
259 if (!gtk_widget_get_mapped (icontip->fixedtip))
260 gtk_widget_show (icontip->fixedtip);
261
262 icontip->id = buffer->id;
263
264 if (buffer->timeout > 0)
265 icontip->source_id = g_timeout_add_seconds (buffer->timeout,
266 icon_tip_show_next_timeout,
267 icontip);
268
269 icon_tip_buffer_free (buffer, NULL);
270}

272static void
273message_sent (NaTrayManager *manager,
274 GtkWidget *icon,
275 const char *text,
276 glong id,
277 glong timeout,
278 TraysScreen *trays_screen)
279{
280 IconTip *icontip;
281 IconTipBuffer find_buffer;
282 IconTipBuffer *buffer;
283 gboolean show_now;
284
285 icontip = g_hash_table_lookup (trays_screen->tip_table, icon);
286
287 find_buffer.id = id;
288 if (icontip &&
289 (icontip->id == id ||
290 g_slist_find_custom (icontip->buffer, &find_buffer,
291 icon_tip_buffer_compare) != NULL))
292 /* we already have this message, so ignore it */
293 /* FIXME: in an ideal world, we'd remember all the past ids and ignore them
294 * too */
295 return;
296
297 show_now = FALSE;
298
299 if (icontip == NULL)
300 {
301 NaTray *tray;
302
303 tray = g_hash_table_lookup (trays_screen->icon_table, icon);
304 if (tray == NULL)
305 {
306 /* We don't know about the icon sending the message, so ignore it.
307 * But this should never happen since NaTrayManager shouldn't send
308 * us the message if there's no socket for it. */
309 g_critical ("Ignoring a message sent by a tray icon "
310 "we don't know: \"%s\".\n", text);
311 return;
312 }
313
314 icontip = g_new0 (IconTip, 1);
315 icontip->tray = tray;
316 icontip->icon = icon;
317
318 g_hash_table_insert (trays_screen->tip_table, icon, icontip);
319
320 show_now = TRUE;
321 }
322
323 buffer = g_new0 (IconTipBuffer, 1);
324
325 buffer->text = g_strdup (text);
326 buffer->id = id;
327 buffer->timeout = timeout;
328
329 icontip->buffer = g_slist_append (icontip->buffer, buffer);
330
331 if (show_now)
332 icon_tip_show_next (icontip);
333}

335static void
336message_cancelled (NaTrayManager *manager,
337 GtkWidget *icon,
338 glong id,
339 TraysScreen *trays_screen)
340{
341 IconTip *icontip;
342 IconTipBuffer find_buffer;
343 GSList *cancel_buffer_l;
344 IconTipBuffer *cancel_buffer;
345
346 icontip = g_hash_table_lookup (trays_screen->tip_table, icon);
347 if (icontip == NULL)
348 return;
349
350 if (icontip->id == id)
351 {
352 icon_tip_show_next (icontip);
353 return;
354 }
355
356 find_buffer.id = id;
357 cancel_buffer_l = g_slist_find_custom (icontip->buffer, &find_buffer,
358 icon_tip_buffer_compare);
359 if (cancel_buffer_l == NULL)
360 return;
361
362 cancel_buffer = cancel_buffer_l->data;
363 icon_tip_buffer_free (cancel_buffer, NULL);
364
365 icontip->buffer = g_slist_remove_link (icontip->buffer, cancel_buffer_l);
366 g_slist_free_1 (cancel_buffer_l);
367}

389static void
390update_size_and_orientation (NaTray *tray)
391{
392 NaTrayPrivate *priv = tray->priv;
393
394 /* This only happens when setting the property during object construction */
395 if (!priv->trays_screen)
396 return;
397
398 g_hash_table_foreach (priv->trays_screen->tip_table,
399 update_orientation_for_messages, tray);
400
401 if (get_tray (priv->trays_screen) == tray)
402 na_tray_manager_set_orientation (priv->trays_screen->tray_manager,
403 priv->orientation);
404}

419static GObject *
420na_tray_constructor (GType type,
421 guint n_construct_properties,
422 GObjectConstructParam *construct_params)
423{
424 GObject *object;
425 NaTray *tray;
426 NaTrayPrivate *priv;
427 int screen_number;
428
429 object = G_OBJECT_CLASS (na_tray_parent_class)->constructor (type,
430 n_construct_properties,
431 construct_params);
432 tray = NA_TRAY (object);
433 priv = tray->priv;
434
435 g_assert (priv->screen != NULL);
436
437 if (!initialized)
438 {
439 trays_screens = g_new0 (TraysScreen, 1);
440 initialized = TRUE;
441 }
442
443 screen_number = gdk_x11_screen_get_screen_number (priv->screen);
444
445 if (trays_screens [screen_number].tray_manager == NULL)
446 {
447 NaTrayManager *tray_manager;
448
449 tray_manager = na_tray_manager_new ();
450
451 if (na_tray_manager_manage_screen (tray_manager, priv->screen))
452 {
453 trays_screens [screen_number].tray_manager = tray_manager;
454
455 g_signal_connect (tray_manager, "tray-icon-added",
456 G_CALLBACK (tray_added),
457 &trays_screens [screen_number]);
458 g_signal_connect (tray_manager, "tray-icon-removed",
459 G_CALLBACK (tray_removed),
460 &trays_screens [screen_number]);
461 g_signal_connect (tray_manager, "message-sent",
462 G_CALLBACK (message_sent),
463 &trays_screens [screen_number]);
464 g_signal_connect (tray_manager, "message-cancelled",
465 G_CALLBACK (message_cancelled),
466 &trays_screens [screen_number]);
467
468 trays_screens [screen_number].icon_table = g_hash_table_new (NULL,
469 NULL);
470 trays_screens [screen_number].tip_table = g_hash_table_new_full (
471 NULL,
472 NULL,
473 NULL,
474 icon_tip_free);
475 }
476 else
477 {
478 g_printerr ("System tray didn't get the system tray manager selection for screen %d\n",
479 screen_number);
480 g_object_unref (tray_manager);
481 }
482 }
483
484 priv->trays_screen = &trays_screens [screen_number];
485 trays_screens [screen_number].all_trays = g_slist_append (trays_screens [screen_number].all_trays,
486 tray);
487
488 update_size_and_orientation (tray);
489
490 return object;
491}

493static void
494na_tray_dispose (GObject *object)
495{
496 NaTray *tray = NA_TRAY (object);
497 NaTrayPrivate *priv = tray->priv;
498 TraysScreen *trays_screen = priv->trays_screen;
499
500 if (trays_screen != NULL)
501 {
502 trays_screen->all_trays = g_slist_remove (trays_screen->all_trays, tray);
503
504 if (trays_screen->all_trays == NULL)
505 {
506 /* Make sure we drop the manager selection */
507 g_object_unref (trays_screen->tray_manager);
508 trays_screen->tray_manager = NULL;
509
510 g_hash_table_destroy (trays_screen->icon_table);
511 trays_screen->icon_table = NULL;
512
513 g_hash_table_destroy (trays_screen->tip_table);
514 trays_screen->tip_table = NULL;
515 }
516 else
517 {
518 NaTray *new_tray;
519
520 new_tray = get_tray (trays_screen);
521 if (new_tray != NULL)
522 na_tray_manager_set_orientation (trays_screen->tray_manager,
523 gtk_orientable_get_orientation (GTK_ORIENTABLE (new_tray)));
524 }
525 }
526
527 priv->trays_screen = NULL;
528
529 if (priv->idle_redraw_id != 0)
530 {
531 g_source_remove (priv->idle_redraw_id);
532 priv->idle_redraw_id = 0;
533 }
534
535 G_OBJECT_CLASS (na_tray_parent_class)->dispose (object);
536}

552static void
553na_tray_set_property (GObject *object,
554 guint prop_id,
555 const GValue *value,
556 GParamSpec *pspec)
557{
558 NaTray *tray = NA_TRAY (object);
559 NaTrayPrivate *priv = tray->priv;
560
561 switch (prop_id)
562 {
563 case PROP_ORIENTATION:
564 na_tray_set_orientation (tray, g_value_get_enum (value));
565 break;
566 case PROP_ICON_PADDING:

568 break;
569 case PROP_ICON_SIZE:
570 na_tray_set_icon_size (tray, g_value_get_int (value));
571 break;
572 case PROP_SCREEN:
573 priv->screen = g_value_get_object (value);
574 break;
575 default:
576 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, prop_id, pspec);
577 break;
578 }
579}

682static void
683na_tray_style_updated (NaHost *host,
684 GtkStyleContext *context)
685{
686 GdkRGBA fg;
687 GdkRGBA error;
688 GdkRGBA warning;
689 GdkRGBA success;
690
691 gtk_style_context_save (context);
692 gtk_style_context_set_state (context, GTK_STATE_FLAG_NORMAL);
693
694 gtk_style_context_get_color (context, GTK_STATE_FLAG_NORMAL, &fg);
695
696 if (!gtk_style_context_lookup_color (context, "error_color", &error))
697 error = fg;
698 if (!gtk_style_context_lookup_color (context, "warning_color", &warning))
699 warning = fg;
700 if (!gtk_style_context_lookup_color (context, "success_color", &success))
701 success = fg;
702
703 gtk_style_context_restore (context);
704
705 na_tray_set_colors (NA_TRAY (host), &fg, &error, &warning, &success);
706}
