120static void
121icon_container_set_workarea (CajaIconContainer *icon_container,
122 GdkScreen *screen,
123 long *workareas,
124 int n_items)
125{
126 int left, right, top, bottom;
127 int screen_width, screen_height;
128 int scale;
129 int i;
130 GdkDisplay *display;
131
132 left = right = top = bottom = 0;
133
134 display = gdk_screen_get_display (screen);
135 if (GDK_IS_X11_DISPLAY (display))
136 {
137 scale = gdk_window_get_scale_factor (gdk_screen_get_root_window (screen));
138 screen_width = WidthOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale;
139 screen_height = HeightOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale;
140 }
141 else
142 {
143 scale = 1; /*wayland handles this for us*/
144 GdkRectangle geometry = {0};
145 GdkMonitor *monitor;
146 monitor = gdk_display_get_monitor (display, 0);
147 gdk_monitor_get_geometry (monitor, &geometry);
148 screen_width = geometry.width;
149 screen_height = geometry.height;
150 }
151
152 for (i = 0; i < n_items; i += 4)
153 {
154 int x = workareas [i] / scale;
155 int y = workareas [i + 1] / scale;
156 int width = workareas [i + 2] / scale;
157 int height = workareas [i + 3] / scale;
158
159 if ((x + width) > screen_width || (y + height) > screen_height)
160 continue;
161
162 left = MAX (left, x);
163 right = MAX (right, screen_width - width - x);
164 top = MAX (top, y);
165 bottom = MAX (bottom, screen_height - height - y);
166 }
167
168 caja_icon_container_set_margins (icon_container,
169 left, right, top, bottom);
170}

173static void
174net_workarea_changed (FMDesktopIconView *icon_view,
175 GdkWindow *window)
176{
177 GdkDisplay *display;
178 long *nworkareas = NULL;
179 long *workareas = NULL;
180 GdkAtom type_returned;
181 int format_returned;
182 int length_returned;
183 CajaIconContainer *icon_container;
184
185 g_return_if_fail (FM_IS_DESKTOP_ICON_VIEW (icon_view));
186
187 icon_container = get_icon_container (icon_view);
188
189 /* Find the number of desktops so we know how long the
190 * workareas array is going to be (each desktop will have four
191 * elements in the workareas array describing
192 * x,y,width,height) */
193 display = gtk_widget_get_display (GTK_WIDGET (icon_view));
194 gdk_x11_display_error_trap_push (display);
195 if (!gdk_property_get (window,
196 gdk_atom_intern ("_NET_NUMBER_OF_DESKTOPS", FALSE),
197 gdk_x11_xatom_to_atom (XA_CARDINAL),
198 0, 4, FALSE,
199 &type_returned,
200 &format_returned,
201 &length_returned,
202 (guchar **) &nworkareas))
203 {
204 g_warning("Can not calculate _NET_NUMBER_OF_DESKTOPS");
205 }
206 if (gdk_x11_display_error_trap_pop (display)
207 || nworkareas == NULL
208 || type_returned != gdk_x11_xatom_to_atom (XA_CARDINAL)
209 || format_returned != 32)
210 g_warning("Can not calculate _NET_NUMBER_OF_DESKTOPS");
211
212 /* Note : gdk_property_get() is broken (API documents admit
213 * this). As a length argument, it expects the number of
214 * _bytes_ of data you require. Internally, gdk_property_get
215 * converts that value to a count of 32 bit (4 byte) elements.
216 * However, the length returned is in bytes, but is calculated
217 * via the count of returned elements * sizeof(long). This
218 * means on a 64 bit system, the number of bytes you have to
219 * request does not correspond to the number of bytes you get
220 * back, and is the reason for the workaround below.
221 */
222 gdk_x11_display_error_trap_push (display);
223 if (nworkareas == NULL || (*nworkareas < 1)
224 || !gdk_property_get (window,
225 gdk_atom_intern ("_NET_WORKAREA", FALSE),
226 gdk_x11_xatom_to_atom (XA_CARDINAL),
227 0, ((*nworkareas) * 4 * 4), FALSE,
228 &type_returned,
229 &format_returned,
230 &length_returned,
231 (guchar **) &workareas))
232 {
233 g_warning("Can not get _NET_WORKAREA");
234 workareas = NULL;
235 }
236
237 if (gdk_x11_display_error_trap_pop (display)
238 || workareas == NULL
239 || type_returned != gdk_x11_xatom_to_atom (XA_CARDINAL)
240 || ((*nworkareas) * 4 * sizeof(long)) != length_returned
241 || format_returned != 32)
242 {
243 g_warning("Can not determine workarea, guessing at layout");
244 caja_icon_container_set_margins (icon_container,
245 0, 0, 0, 0);
246 }
247 else
248 {
249 GdkScreen *screen;
250
251 screen = gdk_window_get_screen (window);
252
253 icon_container_set_workarea (
254 icon_container, screen, workareas, length_returned / sizeof (long));
255 }
256
257 if (nworkareas != NULL)
258 g_free (nworkareas);
259
260 if (workareas != NULL)
261 g_free (workareas);
262}

265static GdkFilterReturn
266desktop_icon_view_property_filter (GdkXEvent *gdk_xevent,
267 GdkEvent *event,
268 gpointer data)
269{
270 XEvent *xevent = gdk_xevent;
271 FMDesktopIconView *icon_view;
272
273 icon_view = FM_DESKTOP_ICON_VIEW (data);
274
275 switch (xevent->type)
276 {
277 case PropertyNotify:
278 if (xevent->xproperty.atom == gdk_x11_get_xatom_by_name ("_NET_WORKAREA"))
279 net_workarea_changed (icon_view, event->any.window);
280 break;
281 default:
282 break;
283 }
284 return GDK_FILTER_CONTINUE;
285}

287static void
288fm_desktop_icon_view_dispose (GObject *object)
289{
290 FMDesktopIconView *icon_view;
291 GtkUIManager *ui_manager;
292
293 icon_view = FM_DESKTOP_ICON_VIEW (object);
294
295 /* Remove desktop rescan timeout. */
296 if (icon_view->priv->reload_desktop_timeout != 0)
297 {
298 g_source_remove (icon_view->priv->reload_desktop_timeout);
299 icon_view->priv->reload_desktop_timeout = 0;
300 }
301
302 ui_manager = fm_directory_view_get_ui_manager (FM_DIRECTORY_VIEW (icon_view));
303 if (ui_manager != NULL)
304 {
305 caja_ui_unmerge_ui (ui_manager,
306 &icon_view->priv->desktop_merge_id,
307 &icon_view->priv->desktop_action_group);
308 }
309
310 g_signal_handlers_disconnect_by_func (caja_icon_view_preferences,
311 default_zoom_level_changed,
312 icon_view);
313 g_signal_handlers_disconnect_by_func (caja_desktop_preferences,
314 font_changed_callback,
315 icon_view);
316
317 g_signal_handlers_disconnect_by_func (mate_lockdown_preferences,
318 fm_directory_view_update_menus,
319 icon_view);
320 g_signal_handlers_disconnect_by_func (caja_preferences,
321 desktop_directory_changed_callback,
322 NULL);
323
324 G_OBJECT_CLASS (fm_desktop_icon_view_parent_class)->dispose (object);
325}

344static void
345fm_desktop_icon_view_handle_middle_click (CajaIconContainer *icon_container,
346 GdkEventButton *event,
347 FMDesktopIconView *desktop_icon_view)
348{
349 XButtonEvent x_event;
350 GdkDevice *keyboard = NULL, *pointer = NULL;
351 GdkSeat *seat;
352 GdkDisplay *display;
353
354 seat = gdk_display_get_default_seat (gtk_widget_get_display (GTK_WIDGET (icon_container)));
355 pointer = gdk_seat_get_pointer (seat);
356 keyboard = gdk_seat_get_keyboard (seat);
357
358 /* During a mouse click we have the pointer and keyboard grab.
359 * We will send a fake event to the root window which will cause it
360 * to try to get the grab so we need to let go ourselves.
361 */
362
363 if (pointer != NULL) {
364 gdk_seat_ungrab (seat);
365 }
366
367 if (keyboard != NULL) {
368 gdk_seat_ungrab (seat);
369 }
370
371 /* Stop the event because we don't want anyone else dealing with it. */
372 display = gtk_widget_get_display (GTK_WIDGET (icon_container));
373 gdk_display_flush (display);
374 g_signal_stop_emission_by_name (icon_container, "middle_click");
375
376 /* build an X event to represent the middle click. */
377 x_event.type = ButtonPress;
378 x_event.send_event = True;
379 x_event.display = GDK_DISPLAY_XDISPLAY (gdk_display_get_default ());
380 x_event.window = GDK_ROOT_WINDOW ();
381 x_event.root = GDK_ROOT_WINDOW ();
382 x_event.subwindow = 0;
383 x_event.time = event->time;
384 x_event.x = event->x;
385 x_event.y = event->y;
386 x_event.x_root = event->x_root;
387 x_event.y_root = event->y_root;
388 x_event.state = event->state;
389 x_event.button = event->button;
390 x_event.same_screen = True;
391
392 /* Send it to the root window, the window manager will handle it. */
393 XSendEvent (GDK_DISPLAY_XDISPLAY (gdk_display_get_default ()), GDK_ROOT_WINDOW (), True,
394 ButtonPressMask, (XEvent *) &x_event);
395}

397static void
398unrealized_callback (GtkWidget *widget, FMDesktopIconView *desktop_icon_view)
399{
400 if (!GDK_IS_X11_DISPLAY (gdk_display_get_default()))
401 return;
402
403 g_return_if_fail (desktop_icon_view->priv->root_window != NULL);
404
405 /* Remove the property filter */
406 gdk_window_remove_filter (desktop_icon_view->priv->root_window,
407 desktop_icon_view_property_filter,
408 desktop_icon_view);
409 desktop_icon_view->priv->root_window = NULL;

412static void
413realized_callback (GtkWidget *widget, FMDesktopIconView *desktop_icon_view)
414{
415 GdkWindow *root_window;
416 GdkScreen *screen;
417 GdkDisplay *display;
418 GtkAllocation allocation;
419
420 g_return_if_fail (desktop_icon_view->priv->root_window == NULL);
421
422 screen = gtk_widget_get_screen (widget);
423 display = gdk_screen_get_display (screen);
424
425 /* Ugly HACK for the problem that the views realize at the
426 * wrong size and then get resized. (This is a problem with
427 * MateComponentPlug.) This was leading to problems where initial
428 * layout was done at 60x60 stacking all desktop icons in
429 * the top left corner.
430 */
431 allocation.x = 0;
432 allocation.y = 0;
433 if (GDK_IS_X11_DISPLAY (display))
434 {
435 gint scale;
436 scale = gtk_widget_get_scale_factor (widget);
437 allocation.width = WidthOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale;
438 allocation.height = HeightOfScreen (gdk_x11_screen_get_xscreen (screen)) / scale;
439 }
440 else
441 {
442 /*No real root window or primary monitor in wayland unless compositors add it back*/
443 GdkRectangle geometry = {0};
444 GdkMonitor *monitor;
445 monitor = gdk_display_get_monitor (display, 0);
446 gdk_monitor_get_geometry (monitor, &geometry);
447 allocation.width = geometry.width;
448 allocation.height = geometry.height;
449 }
450
451 gtk_widget_size_allocate (GTK_WIDGET(get_icon_container(desktop_icon_view)),
452 &allocation);
453
454 if (GDK_IS_X11_DISPLAY (display))
455 {
456 root_window = gdk_screen_get_root_window (screen);
457 desktop_icon_view->priv->root_window = root_window;
458
459 /* Read out the workarea geometry and update the icon container accordingly */
460 net_workarea_changed (desktop_icon_view, root_window);
461
462 /* Setup the property filter */
463
464 gdk_window_set_events (root_window, GDK_PROPERTY_CHANGE_MASK);
465 gdk_window_add_filter (root_window,
466 desktop_icon_view_property_filter,
467 desktop_icon_view);
468 }
469 else
470 {
471 desktop_icon_view->priv->root_window = NULL;
472 }
473}

505static gboolean
506do_desktop_rescan (gpointer data)
507{
508 FMDesktopIconView *desktop_icon_view;
509 struct stat buf;
510
511 desktop_icon_view = FM_DESKTOP_ICON_VIEW (data);
512 if (desktop_icon_view->priv->pending_rescan)
513 {
514 return TRUE;
515 }
516
517 if (stat (desktop_directory, &buf) == -1)
518 {
519 return TRUE;
520 }
521
522 if (buf.st_ctime == desktop_dir_modify_time)
523 {
524 return TRUE;
525 }
526
527 desktop_icon_view->priv->pending_rescan = TRUE;
528
529 caja_directory_force_reload (
530 fm_directory_view_get_model (
531 FM_DIRECTORY_VIEW (desktop_icon_view)));
532 return TRUE;
533}

535static void
536done_loading (CajaDirectory *model,
537 FMDesktopIconView *desktop_icon_view)
538{
539 struct stat buf;
540
541 desktop_icon_view->priv->pending_rescan = FALSE;
542 if (stat (desktop_directory, &buf) == -1)
543 {
544 return;
545 }
546
547 desktop_dir_modify_time = buf.st_ctime;
548}

550/* This function is used because the CajaDirectory model does not
551 * exist always in the desktop_icon_view, so we wait until it has been
552 * instantiated.
553 */
554static void
555delayed_init (FMDesktopIconView *desktop_icon_view)
556{
557 /* Keep track of the load time. */
558 g_signal_connect_object (fm_directory_view_get_model (FM_DIRECTORY_VIEW (desktop_icon_view)),
559 "done_loading",
560 G_CALLBACK (done_loading), desktop_icon_view, 0);
561
562 /* Monitor desktop directory. */
563 desktop_icon_view->priv->reload_desktop_timeout =
564 g_timeout_add_seconds (RESCAN_TIMEOUT, do_desktop_rescan, desktop_icon_view);
565
566 g_signal_handler_disconnect (desktop_icon_view,
567 desktop_icon_view->priv->delayed_init_signal);
568
569 desktop_icon_view->priv->delayed_init_signal = 0;
570}

596static void
597fm_desktop_icon_view_init (FMDesktopIconView *desktop_icon_view)
598{
599 CajaIconContainer *icon_container;
600 GtkAllocation allocation;
601 GtkAdjustment *hadj, *vadj;
602
603 desktop_icon_view->priv = fm_desktop_icon_view_get_instance_private (desktop_icon_view);
604
605 if (desktop_directory == NULL)
606 {
607 g_signal_connect_swapped (caja_preferences, "changed::" CAJA_PREFERENCES_DESKTOP_IS_HOME_DIR,
608 G_CALLBACK(desktop_directory_changed_callback),
609 NULL);
610 desktop_directory_changed_callback (NULL);
611 }
612
613 fm_icon_view_filter_by_screen (FM_ICON_VIEW (desktop_icon_view), TRUE);
614 icon_container = get_icon_container (desktop_icon_view);
615 caja_icon_container_set_use_drop_shadows (icon_container, TRUE);
616 fm_icon_container_set_sort_desktop (FM_ICON_CONTAINER (icon_container), TRUE);
617
618 /* Do a reload on the desktop if we don't have FAM, a smarter
619 * way to keep track of the items on the desktop.
620 */
621 if (!caja_monitor_active ())
622 {
623 desktop_icon_view->priv->delayed_init_signal = g_signal_connect_object
624 (desktop_icon_view, "begin_loading",
625 G_CALLBACK (delayed_init), desktop_icon_view, 0);
626 }
627
628 caja_icon_container_set_is_fixed_size (icon_container, TRUE);
629 caja_icon_container_set_is_desktop (icon_container, TRUE);
630 caja_icon_container_set_store_layout_timestamps (icon_container, TRUE);
631
632 /* Set allocation to be at 0, 0 */
633 gtk_widget_get_allocation (GTK_WIDGET (icon_container), &allocation);
634 allocation.x = 0;
635 allocation.y = 0;
636 gtk_widget_set_allocation (GTK_WIDGET (icon_container), &allocation);
637
638 gtk_widget_queue_resize (GTK_WIDGET (icon_container));
639
640 hadj = gtk_scrollable_get_hadjustment (GTK_SCROLLABLE (icon_container));
641 vadj = gtk_scrollable_get_vadjustment (GTK_SCROLLABLE (icon_container));
642
643 gtk_adjustment_set_value (hadj, 0);
644 gtk_adjustment_set_value (vadj, 0);
645
646 gtk_scrolled_window_set_shadow_type (GTK_SCROLLED_WINDOW (desktop_icon_view),
647 GTK_SHADOW_NONE);
648 gtk_scrolled_window_set_overlay_scrolling (GTK_SCROLLED_WINDOW (desktop_icon_view), FALSE);
649
650 fm_directory_view_ignore_hidden_file_preferences
651 (FM_DIRECTORY_VIEW (desktop_icon_view));
652
653 fm_directory_view_set_show_foreign (FM_DIRECTORY_VIEW (desktop_icon_view),
654 FALSE);
