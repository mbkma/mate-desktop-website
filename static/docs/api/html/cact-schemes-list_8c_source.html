132void
133cact_schemes_list_create_model( GtkTreeView *treeview, guint mode )
134{
135 static const char *thisfn = "cact_schemes_list_create_model";
136 GtkListStore *model;
137 GtkTreeViewColumn *column;
138 GtkCellRenderer *text_cell;
139 GtkTreeSelection *selection;
140 SchemesListData *data;
141
142 g_return_if_fail( GTK_IS_TREE_VIEW( treeview ));
143 g_debug( "%s: treeview=%p, mode=%d", thisfn, ( void * ) treeview, mode );
144
145 data = get_schemes_list_data( treeview );
146 data->mode = mode;
147
148 model = gtk_list_store_new( SCHEMES_N_COLUMN, G_TYPE_STRING, G_TYPE_STRING, G_TYPE_BOOLEAN );
149 gtk_tree_view_set_model( treeview, GTK_TREE_MODEL( model ));
150 g_object_unref( model );
151
152 /* scheme */
153 text_cell = gtk_cell_renderer_text_new();
154 column = gtk_tree_view_column_new_with_attributes(
155 _( "Keyword" ),
156 text_cell,
157 "text", SCHEMES_KEYWORD_COLUMN,
158 NULL );
159 gtk_tree_view_append_column( treeview, column );
160 gtk_tree_sortable_set_sort_column_id( GTK_TREE_SORTABLE( model ), SCHEMES_KEYWORD_COLUMN, GTK_SORT_ASCENDING );
161 gtk_tree_view_column_set_cell_data_func(
162 column, text_cell, ( GtkTreeCellDataFunc ) display_keyword, data, NULL );
163
164 /* description */
165 text_cell = gtk_cell_renderer_text_new();
166 column = gtk_tree_view_column_new_with_attributes(
167 _( "Description" ),
168 text_cell,
169 "text", SCHEMES_DESC_COLUMN,
170 NULL );
171 gtk_tree_view_append_column( treeview, column );
172 gtk_tree_view_column_set_cell_data_func(
173 column, text_cell, ( GtkTreeCellDataFunc ) display_description, data, NULL );
174
175 gtk_tree_view_set_headers_visible( treeview, TRUE );
176
177 selection = gtk_tree_view_get_selection( treeview );
178 gtk_tree_selection_set_mode( selection, GTK_SELECTION_BROWSE );
179}

224
225 init_view_setup_defaults( data );

229static void
230init_view_setup_defaults( SchemesListData *data )
231{
232 GtkListStore *model;
233 GSList *schemes, *iter;
234 GtkTreeIter row;
235 gchar **tokens;
236
237 model = GTK_LIST_STORE( gtk_tree_view_get_model( data->treeview ));
238
239 schemes = init_view_get_default_list( data );
240
241 for( iter = schemes ; iter ; iter = iter->next ){
242
243 tokens = g_strsplit(( gchar * ) iter->data, "|", 2 );
244 gtk_list_store_append( model, &row );
245 gtk_list_store_set( model, &row,
246 SCHEMES_KEYWORD_COLUMN, tokens[0],
247 SCHEMES_DESC_COLUMN, tokens[1],
248 SCHEMES_ALREADY_USED_COLUMN, FALSE,
249 -1 );
250 g_strfreev( tokens );
251 }
252
253 na_core_utils_slist_free( schemes );
254}

271

273

296static void
297init_view_connect_signals( SchemesListData *data )
298{
299 GtkTreeViewColumn *column;
300 GList *renderers;
301 GtkButton *add_button, *remove_button;
302
303 base_window_signal_connect(
304 data->window,
305 G_OBJECT( gtk_tree_view_get_selection( data->treeview )),
306 "changed",
307 G_CALLBACK( on_selection_changed ));
308
309 column = gtk_tree_view_get_column( data->treeview, SCHEMES_KEYWORD_COLUMN );
310 base_gtk_utils_set_editable( G_OBJECT( column ), data->editable );
311 if( data->editable ){
312 renderers = gtk_cell_layout_get_cells( GTK_CELL_LAYOUT( column ));
313 base_window_signal_connect(
314 data->window,
315 G_OBJECT( renderers->data ),
316 "edited",
317 G_CALLBACK( on_keyword_edited ));
318 }
319
320 column = gtk_tree_view_get_column( data->treeview, SCHEMES_DESC_COLUMN );
321 base_gtk_utils_set_editable( G_OBJECT( column ), data->editable );
322 if( data->editable ){
323 renderers = gtk_cell_layout_get_cells( GTK_CELL_LAYOUT( column ));
324 base_window_signal_connect(
325 data->window,
326 G_OBJECT( renderers->data ),
327 "edited",
328 G_CALLBACK( on_desc_edited ));
329 }
330
331 add_button = get_add_button( data->window );
332 gtk_widget_set_sensitive( GTK_WIDGET( add_button ), data->editable );
333 if( data->editable ){
334 base_window_signal_connect(
335 data->window,
336 G_OBJECT( add_button ),
337 "clicked",
338 G_CALLBACK( on_add_clicked ));
339 }
340
341 remove_button = get_remove_button( data->window );
342 gtk_widget_set_sensitive( GTK_WIDGET( remove_button ), data->editable );
343 if( data->editable ){
344 base_window_signal_connect(
345 data->window,
346 G_OBJECT( remove_button ),
347 "clicked",
348 G_CALLBACK( on_remove_clicked ));
349 }

390static gboolean
391setup_values_iter( GtkTreeModel *model, GtkTreePath *path, GtkTreeIter* iter, GSList *schemes )
392{
393 gchar *keyword;
394 gchar *description, *new_description;
395
396 gtk_tree_model_get( model, iter, SCHEMES_KEYWORD_COLUMN, &keyword, SCHEMES_DESC_COLUMN, &description, -1 );
397
398 if( na_core_utils_slist_find_negated( schemes, keyword )){
399 /* Translators: add a comment when a scheme is already used by current item */
400 new_description = g_strdup_printf( _( "%s (already used)"), description );
401 gtk_list_store_set( GTK_LIST_STORE( model ), iter, SCHEMES_DESC_COLUMN, new_description, SCHEMES_ALREADY_USED_COLUMN, TRUE, -1 );
402 g_free( new_description );
403 }
404
405 g_free( description );
406 g_free( keyword );
407
408 return( FALSE ); /* don't stop looping */
409}

505static gboolean
506get_list_schemes_iter( GtkTreeModel *model, GtkTreePath *path, GtkTreeIter* iter, GSList **list )
507{
508 gchar *keyword;
509 gchar *description;
510 gchar *scheme;
511
512 gtk_tree_model_get( model, iter, SCHEMES_KEYWORD_COLUMN, &keyword, SCHEMES_DESC_COLUMN, &description, -1 );
513 scheme = g_strdup_printf( "%s|%s", keyword, description );
514 g_free( description );
515 g_free( keyword );
516
517 ( *list ) = g_slist_append(( *list ), scheme );
518
519 return( FALSE ); /* don't stop looping */
520}

544static gboolean
545on_key_pressed_event( GtkWidget *widget, GdkEventKey *event, BaseWindow *window )
546{
547 gboolean stop;
548
549 /*g_debug( "cact_schemes_list_on_key_pressed_event" );*/
550
551 stop = FALSE;
552
553 if( event->keyval == CACT_KEY_F2 ){
554 edit_inline( window );
555 stop = TRUE;
556 }
557
558 if( event->keyval == CACT_KEY_Insert || event->keyval == CACT_KEY_KP_Insert ){
559 insert_new_row( window );
560 stop = TRUE;
561 }
562
563 if( event->keyval == CACT_KEY_Delete || event->keyval == CACT_KEY_KP_Delete ){
564 delete_row( window );
565 stop = TRUE;
566 }
567
568 return( stop );
569}

571static void
572on_selection_changed( GtkTreeSelection *selection, BaseWindow *window )
573{
574 /*static const gchar *thisfn = "cact_schemes_list_on_selection_changed";*/
575 GtkButton *button;
576 GtkTreeView *listview;
577 SchemesListData *data;
578 GtkTreeModel *model;
579 GList *rows;
580 GtkTreePath *path;
581 GtkTreeIter iter;
582 gchar *keyword;
583 gboolean used;
584
585 /*g_debug( "%s: selection=%p, window=%p", thisfn, ( void * ) selection, ( void * ) window );*/
586
587 listview = GTK_TREE_VIEW( g_object_get_data( G_OBJECT( window ), SCHEMES_LIST_TREEVIEW ));
588 data = ( SchemesListData * ) g_object_get_data( G_OBJECT( listview ), SCHEMES_LIST_DATA );
589
590 button = get_remove_button( window );
591 gtk_widget_set_sensitive( GTK_WIDGET( button ),
592 data->editable && gtk_tree_selection_count_selected_rows( selection ) > 0);
593
594 if( data->pf_on_sel_changed ){
595 rows = gtk_tree_selection_get_selected_rows( selection, &model );
596 keyword = NULL;
597 used = FALSE;
598
599 if( g_list_length( rows ) == 1 ){
600 path = ( GtkTreePath * ) rows->data;
601 gtk_tree_model_get_iter( model, &iter, path );
602 gtk_tree_model_get( model, &iter, SCHEMES_KEYWORD_COLUMN, &keyword, SCHEMES_ALREADY_USED_COLUMN, &used, -1 );
603 }
604
605 data->pf_on_sel_changed( keyword, used, data->user_data );
606
607 g_free( keyword );
608 }
609}
