180static int RecordPid(void)
181{
182 int pid = 0;
183 int fd;
184 int Length = 0;
185 char WriteBuf[30] = { 0 };
186
187 fd = open(LOCKFILE,O_WRONLY|O_CREAT|O_TRUNC,0777);
188 if(fd < 0)
189 {
190 ErrorMessage (_("open file"), _("Create pid file failed"));
191 return -1;
192 }
193 chmod(LOCKFILE,0777);
194 pid = getpid();
195 sprintf(WriteBuf,"%d",pid);
196 Length = write(fd,WriteBuf,strlen(WriteBuf));
197 if(Length <= 0 )
198 {
199 ErrorMessage (_("write file"), _("write pid file failed"));
200 return -1;
201 }
202 close(fd);
203
204 return 0;
205}

207/******************************************************************************
208* Function: ProcessRuning
209*
210* Explain: Check whether the process has been started,If the process is not started,
211* record the current process ID =====>"/tmp/user-admin.pid"
212*
213* Input:
214*
215*
216* Output: start :TRUE
217* not start :FALSE
218*
219* Author: zhuyaliang 31/07/2018
220******************************************************************************/
221static gboolean ProcessRuning(void)
222{
223 int fd;
224 int pid = 0;
225 gboolean Run = FALSE;
226 char ReadBuf[30] = { 0 };
227
228 if(access(LOCKFILE,F_OK) == 0)
229 {
230 fd = open(LOCKFILE,O_RDONLY);
231 if(fd < 0)
232 {
233 ErrorMessage (_("open file"), _("open pid file failed"));
234 return TRUE;
235 }
236 if(read(fd,ReadBuf,sizeof(ReadBuf)) <= 0)
237 {
238 ErrorMessage (_("read file"), _("read pid file failed"));
239 goto ERROREXIT;
240 }
241 pid = atoi(ReadBuf);
242 if(kill(pid,0) == 0)
243 {
244 goto ERROREXIT;
245 }
246 }
247
248 if(RecordPid() < 0)
249 Run = TRUE;
250
251 return Run;
252ERROREXIT:
253 close(fd);
254 return TRUE;
255}

257static gboolean InitDbusProxy(TimeAdmin *ta)
258{
259 GError *error = NULL;
260
261 ta->Connection = g_bus_get_sync (G_BUS_TYPE_SYSTEM, NULL, &error);
262 if(ta->Connection == NULL)
263 {
264 ErrorMessage ("g_bus_get_sync", error->message);
265 goto EXIT;
266 }
267 ta->proxy = g_dbus_proxy_new_sync (ta->Connection,
268 G_DBUS_PROXY_FLAGS_NONE,
269 NULL,
270 "org.freedesktop.timedate1",
271 "/org/freedesktop/timedate1",
272 "org.freedesktop.timedate1",
273 NULL,
274 &error);
275 if(ta->proxy == NULL)
276 {
277 ErrorMessage ("g_bus_proxy_new", error->message);
278 goto EXIT;
279 }
280
281 return TRUE;
282EXIT:
283 g_error_free(error);
284 return FALSE;
285}
