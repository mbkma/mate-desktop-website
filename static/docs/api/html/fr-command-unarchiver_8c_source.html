68static void
69list_command_completed (gpointer data)
70{
71 FrCommandUnarchiver *unar_comm = FR_COMMAND_UNARCHIVER (data);
72 JsonParser *parser;
73 GError *error = NULL;
74
75 parser = json_parser_new ();
76 if (json_parser_load_from_stream (parser, unar_comm->stream, NULL, &error)) {
77 JsonObject *root;
78
79 root = json_node_get_object (json_parser_get_root (parser));
80
81 if (json_object_get_int_member (root, "lsarFormatVersion") == LSAR_SUPPORTED_FORMAT) {
82 JsonArray *content;
83 guint i;
84
85 content = json_object_get_array_member (root, "lsarContents");
86 for (i = 0; i < json_array_get_length (content); i++) {
87 JsonObject *entry;
88 FileData *fdata;
89 const char *filename;
90
91 entry = json_array_get_object_element (content, i);
92 fdata = file_data_new ();
93 fdata->size = json_object_get_int_member (entry, "XADFileSize");
94 fdata->modified = mktime_from_string (json_object_get_string_member (entry, "XADLastModificationDate"));
95 if (json_object_has_member (entry, "XADIsEncrypted"))
96 fdata->encrypted = json_object_get_int_member (entry, "XADIsEncrypted") == 1;
97
98 filename = json_object_get_string_member (entry, "XADFileName");
99 if (*filename == '/') {
100 fdata->full_path = g_strdup (filename);
101 fdata->original_path = fdata->full_path;
102 }
103 else {
104 fdata->full_path = g_strconcat ("/", filename, NULL);
105 fdata->original_path = fdata->full_path + 1;
106 }
107
108 fdata->link = NULL;
109 if (json_object_has_member (entry, "XADIsDirectory"))
110 fdata->dir = json_object_get_int_member (entry, "XADIsDirectory") == 1;
111 if (fdata->dir)
112 fdata->name = dir_name_from_path (fdata->full_path);
113 else
114 fdata->name = g_strdup (file_name_from_path (fdata->full_path));
115 fdata->path = remove_level_from_path (fdata->full_path);
116
117 fr_command_add_file (FR_COMMAND (unar_comm), fdata);
118 }
119 }
120 }
121
122 g_object_unref (parser);
123}

125static void
126fr_command_unarchiver_list (FrCommand *comm)
127{
128 FrCommandUnarchiver *unar_comm = FR_COMMAND_UNARCHIVER (comm);
129
130 _g_object_unref (unar_comm->stream);
131 unar_comm->stream = g_memory_input_stream_new ();
132
133 fr_process_set_out_line_func (comm->process, process_line, comm);
134
135 fr_process_begin_command (comm->process, "lsar");
136 fr_process_set_end_func (comm->process, list_command_completed, comm);
137 fr_process_add_arg (comm->process, "-j");
138 if ((comm->password != NULL) && (comm->password[0] != '\0'))
139 fr_process_add_arg_concat (comm->process, "-password=", comm->password, NULL);
140 fr_process_add_arg (comm->process, comm->filename);
141 fr_process_end_command (comm->process);
142
143 fr_process_start (comm->process);
144}

146static void
147process_line__extract (char *line,
148 gpointer data)
149{
150 FrCommand *comm = FR_COMMAND (data);
151 FrCommandUnarchiver *unar_comm = FR_COMMAND_UNARCHIVER (comm);
152
153 if (line == NULL)
154 return;
155
156 unar_comm->n_line++;
157
158 /* the first line is the name of the archive */
159 if (unar_comm->n_line == 1)
160 return;
161
162 if (comm->n_files > 1) {
163 double fraction = (double) ++comm->n_file / (comm->n_files + 1);
164 fr_command_progress (comm, CLAMP (fraction, 0.0, 1.0));
165 }
166 else
167 fr_command_message (comm, line);
168}

170static void
171fr_command_unarchiver_extract (FrCommand *comm,
172 const char *from_file,
173 GList *file_list,
174 const char *dest_dir,
175 gboolean overwrite,
176 gboolean skip_older,
177 gboolean junk_paths)
178{
179 FrCommandUnarchiver *unar_comm = FR_COMMAND_UNARCHIVER (comm);
180 GList *scan;
181
182 unar_comm->n_line = 0;
183
184 fr_process_use_standard_locale (comm->process, TRUE);
185 fr_process_set_out_line_func (comm->process,
186 process_line__extract,
187 comm);
188
189 fr_process_begin_command (comm->process, "unar");
190
191 if (overwrite)
192 fr_process_add_arg (comm->process, "-f");
193 else
194 fr_process_add_arg (comm->process, "-s");
195
196 fr_process_add_arg (comm->process, "-D");
197
198 if ((comm->password != NULL) && (comm->password[0] != '\0'))
199 fr_process_add_arg_concat (comm->process, "-password=", comm->password, NULL);
200
201 if (dest_dir != NULL)
202 fr_process_add_arg_concat (comm->process, "-output-directory=", dest_dir, NULL);
203
204 fr_process_add_arg (comm->process, comm->filename);
205
206 for (scan = file_list; scan; scan = scan->next) {
207 char *escaped;
208
209 escaped = escape_str (scan->data, "[?");
210 fr_process_add_arg (comm->process, escaped);
211 g_free (escaped);
212 }
213
214 fr_process_end_command (comm->process);
215}
