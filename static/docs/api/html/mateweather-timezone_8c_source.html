60static gboolean
61parse_tzdata (const char *tzname, time_t start, time_t end,
62 int *offset, gboolean *has_dst, int *dst_offset)
63{
64 char *filename, *contents;
65 const char *tzdir;
66 gsize length;
67 int timecnt, transitions_size, ttinfo_map_size;
68 int initial_transition = -1, second_transition = -1;
69 gint32 *transitions;
70 char *ttinfo_map, *ttinfos;
71 gint32 initial_offset, second_offset;
72 char initial_isdst, second_isdst;
73 int i;
74
75 tzdir = g_getenv ("TZDIR");
76 if (tzdir == NULL)
77 tzdir = ZONEINFO_DIR;
78 filename = g_build_filename (tzdir, tzname, NULL);
79 if (!g_file_get_contents (filename, &contents, &length, NULL)) {
80 g_free (filename);
81 return FALSE;
82 }
83 g_free (filename);
84
85 if (length < TZ_HEADER_SIZE ||
86 strncmp (contents, TZ_MAGIC, strlen (TZ_MAGIC)) != 0) {
87 g_free (contents);
88 return FALSE;
89 }
90
91 timecnt = GUINT32_FROM_BE (*(guint32 *)(contents + TZ_TIMECNT_OFFSET));
92 transitions = (void *)(contents + TZ_TRANSITIONS_OFFSET);
93 transitions_size = timecnt * sizeof (*transitions);
94 ttinfo_map = (void *)(contents + TZ_TRANSITIONS_OFFSET + transitions_size);
95 ttinfo_map_size = timecnt;
96 ttinfos = (void *)(ttinfo_map + ttinfo_map_size);
97
98 /* @transitions is an array of @timecnt time_t values. We need to
99 * find the transition into the current offset, which is the last
100 * transition before @start. If the following transition is before
101 * @end, then note that one too, since it presumably means we're
102 * doing DST.
103 */
104 for (i = 1; i < timecnt && initial_transition == -1; i++) {
105 if (GINT32_FROM_BE (transitions[i]) > start) {
106 initial_transition = ttinfo_map[i - 1];
107 if (GINT32_FROM_BE (transitions[i]) < end)
108 second_transition = ttinfo_map[i];
109 }
110 }
111 if (initial_transition == -1) {
112 if (timecnt)
113 initial_transition = ttinfo_map[timecnt - 1];
114 else
115 initial_transition = 0;
116 }
117
118 /* Copy the data out of the corresponding ttinfo structs */
119 initial_offset = *(gint32 *)(ttinfos +
120 initial_transition * TZ_TTINFO_SIZE +
121 TZ_TTINFO_GMTOFF_OFFSET);
122 initial_offset = GINT32_FROM_BE (initial_offset);
123 initial_isdst = *(ttinfos +
124 initial_transition * TZ_TTINFO_SIZE +
125 TZ_TTINFO_ISDST_OFFSET);
126
127 if (second_transition != -1) {
128 second_offset = *(gint32 *)(ttinfos +
129 second_transition * TZ_TTINFO_SIZE +
130 TZ_TTINFO_GMTOFF_OFFSET);
131 second_offset = GINT32_FROM_BE (second_offset);
132 second_isdst = *(ttinfos +
133 second_transition * TZ_TTINFO_SIZE +
134 TZ_TTINFO_ISDST_OFFSET);
135
136 *has_dst = (initial_isdst != second_isdst);
137 } else
138 *has_dst = FALSE;
139
140 if (!*has_dst)
141 *offset = initial_offset / 60;
142 else {
143 if (initial_isdst) {
144 *offset = second_offset / 60;
145 *dst_offset = initial_offset / 60;
146 } else {
147 *offset = initial_offset / 60;
148 *dst_offset = second_offset / 60;
149 }
150 }
151
152 g_free (contents);
153 return TRUE;
154}

156static MateWeatherTimezone *
157parse_timezone (MateWeatherParser *parser)
158{
159 MateWeatherTimezone *zone = NULL;
160 char *id = NULL, *name = NULL;
161 int offset = 0, dst_offset = 0;
162 gboolean has_dst = FALSE;
163
164 id = (char *) xmlTextReaderGetAttribute (parser->xml, (xmlChar *) "id");
165 if (!id) {
166 xmlTextReaderNext (parser->xml);
167 return NULL;
168 }
169
170 if (!xmlTextReaderIsEmptyElement (parser->xml)) {
171 if (xmlTextReaderRead (parser->xml) != 1) {
172 xmlFree (id);
173 return NULL;
174 }
175
176 while (xmlTextReaderNodeType (parser->xml) != XML_READER_TYPE_END_ELEMENT) {
177 if (xmlTextReaderNodeType (parser->xml) != XML_READER_TYPE_ELEMENT) {
178 if (xmlTextReaderRead (parser->xml) != 1)
179 break;
180 continue;
181 }
182
183 if (!strcmp ((const char *) xmlTextReaderConstName (parser->xml), "name"))
184 name = mateweather_parser_get_localized_value (parser);
185 else {
186 if (xmlTextReaderNext (parser->xml) != 1)
187 break;
188 }
189 }
190 }
191
192 if (parse_tzdata (id, parser->year_start, parser->year_end,
193 &offset, &has_dst, &dst_offset)) {
194 zone = g_slice_new0 (MateWeatherTimezone);
195 zone->ref_count = 1;
196 zone->id = g_strdup (id);
197 zone->name = g_strdup (name);
198 zone->offset = offset;
199 zone->has_dst = has_dst;
200 zone->dst_offset = dst_offset;
201 }
202
203 xmlFree (id);
204 if (name)
205 xmlFree (name);
206
207 return zone;
208}

210MateWeatherTimezone **
211mateweather_timezones_parse_xml (MateWeatherParser *parser)
212{
213 GPtrArray *zones;
214 MateWeatherTimezone *zone;
215 const char *tagname;
216 int tagtype, i;
217
218 zones = g_ptr_array_new ();
219
220 if (xmlTextReaderRead (parser->xml) != 1)
221 goto error_out;
222 while ((tagtype = xmlTextReaderNodeType (parser->xml)) !=
223 XML_READER_TYPE_END_ELEMENT) {
224 if (tagtype != XML_READER_TYPE_ELEMENT) {
225 if (xmlTextReaderRead (parser->xml) != 1)
226 goto error_out;
227 continue;
228 }
229
230 tagname = (const char *) xmlTextReaderConstName (parser->xml);
231
232 if (!strcmp (tagname, "timezone")) {
233 zone = parse_timezone (parser);
234 if (zone)
235 g_ptr_array_add (zones, zone);
236 }
237
238 if (xmlTextReaderNext (parser->xml) != 1)
239 goto error_out;
240 }
241 if (xmlTextReaderRead (parser->xml) != 1)
242 goto error_out;
243
244 g_ptr_array_add (zones, NULL);
245 return (MateWeatherTimezone **)g_ptr_array_free (zones, FALSE);
246
247error_out:
248 for (i = 0; i < zones->len; i++)
249 mateweather_timezone_unref (zones->pdata[i]);
250 g_ptr_array_free (zones, TRUE);
251 return NULL;
252}
