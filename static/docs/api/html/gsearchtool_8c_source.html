184static gboolean
185handle_locate_command_stdout_io (GIOChannel * ioc,
186 GIOCondition condition,
187 gpointer data)
188{
189 GSearchWindow * gsearch = data;
190 gboolean broken_pipe = FALSE;
191
192 if (condition & G_IO_IN) {
193
194 GError * error = NULL;
195 GString * string;
196
197 string = g_string_new (NULL);
198
199 while (ioc->is_readable != TRUE);
200
201 do {
202 gint status;
203
204 do {
205 status = g_io_channel_read_line_string (ioc, string, NULL, &error);
206
207 if (status == G_IO_STATUS_EOF) {
208 broken_pipe = TRUE;
209 }
210 else if (status == G_IO_STATUS_AGAIN) {
211 if (gtk_events_pending ()) {
212 while (gtk_events_pending ()) {
213 gtk_main_iteration ();
214 }
215 }
216 }
217 else if ((string->len != 0) && (strncmp (string->str, "/", 1) == 0)) {
218 gsearch->is_locate_database_available = TRUE;
219 broken_pipe = TRUE;
220 }
221
222 } while (status == G_IO_STATUS_AGAIN && broken_pipe == FALSE);
223
224 if (broken_pipe == TRUE) {
225 break;
226 }
227
228 if (status != G_IO_STATUS_NORMAL) {
229 if (error != NULL) {
230 g_warning ("handle_locate_command_stdout_io(): %s", error->message);
231 g_error_free (error);
232 }
233 }
234
235 } while (g_io_channel_get_buffer_condition (ioc) & G_IO_IN);
236
237 waitpid (locate_database_check_command_pid, NULL, 0);
238 g_string_free (string, TRUE);
239 }
240
241 if (!(condition & G_IO_IN) || broken_pipe == TRUE) {
242 gsearch->is_locate_database_check_finished = TRUE;
243 g_io_channel_shutdown (ioc, TRUE, NULL);
244 return FALSE;
245 }
246 return TRUE;
247}

249static void
250setup_case_insensitive_arguments (GSearchWindow * gsearch)
251{
252 static gboolean case_insensitive_arguments_initialized = FALSE;
253 gchar * cmd_stderr;
254 gchar * grep_cmd;
255 gchar * locate;
256
257 if (case_insensitive_arguments_initialized == TRUE) {
258 return;
259 }
260 case_insensitive_arguments_initialized = TRUE;
261
262 /* check find command for -iname argument compatibility */
263 g_spawn_command_line_sync ("find /dev/null -iname 'string'", NULL, &cmd_stderr, NULL, NULL);
264
265 if ((cmd_stderr != NULL) && (strlen (cmd_stderr) == 0)) {
266 find_command_default_name_argument = g_strdup ("-iname");
267 GSearchOptionTemplates[SEARCH_CONSTRAINT_FILE_IS_NOT_NAMED].option = g_strdup ("'!' -iname '*%s*'");
268 }
269 else {
270 find_command_default_name_argument = g_strdup ("-name");
271 }
272 g_free (cmd_stderr);
273
274 /* check grep command for -i argument compatibility */
275 grep_cmd = g_strdup_printf ("%s -i 'string' /dev/null", GREP_COMMAND);
276 g_spawn_command_line_sync (grep_cmd, NULL, &cmd_stderr, NULL, NULL);
277
278 if ((cmd_stderr != NULL) && (strlen (cmd_stderr) == 0)) {
279 g_free (cmd_stderr);
280 g_free (grep_cmd);
281
282 /* check grep command for -I argument compatibility, bug 568840 */
283 grep_cmd = g_strdup_printf ("%s -i -I 'string' /dev/null", GREP_COMMAND);
284 g_spawn_command_line_sync (grep_cmd, NULL, &cmd_stderr, NULL, NULL);
285
286 if ((cmd_stderr != NULL) && (strlen (cmd_stderr) == 0)) {
287 GSearchOptionTemplates[SEARCH_CONSTRAINT_CONTAINS_THE_TEXT].option =
288 g_strdup_printf ("'!' -type p -exec %s -i -I -c '%%s' {} \\;", GREP_COMMAND);
289 }
290 else {
291 GSearchOptionTemplates[SEARCH_CONSTRAINT_CONTAINS_THE_TEXT].option =
292 g_strdup_printf ("'!' -type p -exec %s -i -c '%%s' {} \\;", GREP_COMMAND);
293 }
294 }
295 else {
296 GSearchOptionTemplates[SEARCH_CONSTRAINT_CONTAINS_THE_TEXT].option =
297 g_strdup_printf ("'!' -type p -exec %s -c '%%s' {} \\;", GREP_COMMAND);
298 }
299 g_free (cmd_stderr);
300
301 locate = g_find_program_in_path ("locate");
302
303 if (locate != NULL) {
304 GIOChannel * ioc_stdout;
305 gchar ** argv = NULL;
306 gchar *command = NULL;
307 gint child_stdout;
308
309 /* check locate command for -i argument compatibility */
310 command = g_strconcat (locate, " -i /", NULL);
311 g_shell_parse_argv (command, NULL, &argv, NULL);
312 g_free (command);
313
314 gsearch->is_locate_database_check_finished = FALSE;
315 gsearch->is_locate_database_available = FALSE;
316
317 /* run locate command asynchronously because on some systems it can be slow */
318 if (g_spawn_async_with_pipes (g_get_home_dir (), argv, NULL,
319 G_SPAWN_SEARCH_PATH, NULL, NULL,
320 &locate_database_check_command_pid, NULL, &child_stdout,
321 NULL, NULL)) {
322
323 ioc_stdout = g_io_channel_unix_new (child_stdout);
324 g_io_channel_set_encoding (ioc_stdout, NULL, NULL);
325 g_io_channel_set_flags (ioc_stdout, G_IO_FLAG_NONBLOCK, NULL);
326 g_io_add_watch (ioc_stdout, G_IO_IN | G_IO_HUP,
327 handle_locate_command_stdout_io, gsearch);
328 g_io_channel_unref (ioc_stdout);
329 }
330 else {
331 gsearch->is_locate_database_check_finished = TRUE;
332 }
333
334 g_strfreev (argv);
335
336 while (gsearch->is_locate_database_check_finished == FALSE) {
337 if (gtk_events_pending ()) {
338 while (gtk_events_pending ()) {
339 gtk_main_iteration ();
340 }
341 }
342 }
343
344 if (gsearch->is_locate_database_available == TRUE) {
345 locate_command_default_options = g_strdup ("-i");
346 }
347 else {
348 /* run locate again to check if it can find anything */
349 command = g_strconcat (locate, " /", NULL);
350 g_shell_parse_argv (command, NULL, &argv, NULL);
351 g_free (command);
352
353 gsearch->is_locate_database_check_finished = FALSE;
354 locate_command_default_options = g_strdup ("");
355
356 /* run locate command asynchronously because on some systems it can be slow */
357 if (g_spawn_async_with_pipes (g_get_home_dir (), argv, NULL,
358 G_SPAWN_SEARCH_PATH, NULL, NULL,
359 &locate_database_check_command_pid, NULL, &child_stdout,
360 NULL, NULL)) {
361
362 ioc_stdout = g_io_channel_unix_new (child_stdout);
363 g_io_channel_set_encoding (ioc_stdout, NULL, NULL);
364 g_io_channel_set_flags (ioc_stdout, G_IO_FLAG_NONBLOCK, NULL);
365 g_io_add_watch (ioc_stdout, G_IO_IN | G_IO_HUP,
366 handle_locate_command_stdout_io, gsearch);
367 g_io_channel_unref (ioc_stdout);
368 }
369 else {
370 gsearch->is_locate_database_check_finished = TRUE;
371 }
372
373 g_strfreev (argv);
374
375 while (gsearch->is_locate_database_check_finished == FALSE) {
376 if (gtk_events_pending ()) {
377 while (gtk_events_pending ()) {
378 gtk_main_iteration ();
379 }
380 }
381 }
382
383 if (gsearch->is_locate_database_available == FALSE) {
384 g_warning (_("A locate database has probably not been created."));
385 }
386 }
387 }
388 else {
389 /* locate is not installed */
390 locate_command_default_options = g_strdup ("");
391 gsearch->is_locate_database_available = FALSE;
392 }
393 g_free (grep_cmd);
394 g_free (locate);
395}

397static gchar *
398setup_find_name_options (gchar * file)
399{
400 /* This function builds the name options for the find command. This in
401 done to insure that the find command returns hidden files and folders. */
402
403 GString * command;
404 command = g_string_new ("");
405
406 if (strstr (file, "*") == NULL) {
407
408 if ((strlen (file) == 0) || (file[0] != '.')) {
409 g_string_append_printf (command, "\\( %s \"*%s*\" -o %s \".*%s*\" \\) ",
410 find_command_default_name_argument, file,
411 find_command_default_name_argument, file);
412 }
413 else {
414 g_string_append_printf (command, "\\( %s \"*%s*\" -o %s \".*%s*\" -o %s \"%s*\" \\) ",
415 find_command_default_name_argument, file,
416 find_command_default_name_argument, file,
417 find_command_default_name_argument, file);
418 }
419 }
420 else {
421 if (file[0] == '.') {
422 g_string_append_printf (command, "\\( %s \"%s\" -o %s \".*%s\" \\) ",
423 find_command_default_name_argument, file,
424 find_command_default_name_argument, file);
425 }
426 else if (file[0] != '*') {
427 g_string_append_printf (command, "%s \"%s\" ",
428 find_command_default_name_argument, file);
429 }
430 else {
431 if ((strlen (file) >= 1) && (file[1] == '.')) {
432 g_string_append_printf (command, "\\( %s \"%s\" -o %s \"%s\" \\) ",
433 find_command_default_name_argument, file,
434 find_command_default_name_argument, &file[1]);
435 }
436 else {
437 g_string_append_printf (command, "\\( %s \"%s\" -o %s \".%s\" \\) ",
438 find_command_default_name_argument, file,
439 find_command_default_name_argument, file);
440 }
441 }
442 }
443 return g_string_free (command, FALSE);
444}

446static gboolean
447has_additional_constraints (GSearchWindow * gsearch)
448{
449 GList * list;
450
451 if (gsearch->available_options_selected_list != NULL) {
452
453 for (list = gsearch->available_options_selected_list; list != NULL; list = g_list_next (list)) {
454
455 GSearchConstraint * constraint = list->data;
456
457 switch (GSearchOptionTemplates[constraint->constraint_id].type) {
458 case SEARCH_CONSTRAINT_TYPE_BOOLEAN:
459 case SEARCH_CONSTRAINT_TYPE_NUMERIC:
460 case SEARCH_CONSTRAINT_TYPE_DATE_BEFORE:
461 case SEARCH_CONSTRAINT_TYPE_DATE_AFTER:
462 return TRUE;
463 case SEARCH_CONSTRAINT_TYPE_TEXT:
464 if (strlen (constraint->data.text) > 0) {
465 return TRUE;
466 }
467 default:
468 break;
469 }
470 }
471 }
472 return FALSE;
473}

927 COLUMN_READABLE_DATE, readable_date,
928#if GLIB_CHECK_VERSION(2,61,2)

939 reference = gtk_tree_row_reference_new (GTK_TREE_MODEL (store), path);
940 gtk_tree_path_free (path);
941

946 monitor->reference = reference;

949 COLUMN_MONITOR, monitor, -1);

955 gtk_tree_row_reference_free (reference);
956 g_slice_free (GSearchMonitor, monitor);
957 g_clear_error (&error);
958 }
959 }
960
961 g_object_unref (g_file);
962 g_object_unref (file_info);
963 g_date_time_unref (file_dt);
964 g_free (base_name);
965 g_free (dir_name);
966 g_free (relative_dir_name);
967 g_free (utf8_base_name);
968 g_free (utf8_relative_dir_name);
969 g_free (look_in_folder);
970 g_free (description);
971 g_free (readable_size);
972 g_free (readable_date);

1468{

1474 GError * error = NULL;
1475 GTimer * timer;

1477 GdkRectangle prior_rect;
1478 GdkRectangle after_rect;
1479 gulong duration;

1485 timer = g_timer_new ();

1489

1491 gchar * utf8 = NULL;
1492 gchar * filename = NULL;
1493 gint status;
1494

1496 broken_pipe = TRUE;
1497 break;
1498 }
1499 else if (gsearch->command_details->command_status != RUNNING) {
1500 break;
1501 }

1503 do {
1504 status = g_io_channel_read_line_string (ioc, string, NULL, &error);
1505

1507 broken_pipe = TRUE;
1508 }

1511 intermediate_file_count_update (gsearch);
1512 while (gtk_events_pending ()) {
1513 if (gsearch->command_details->command_status == MAKE_IT_QUIT) {
1514 return FALSE;
1515 }
1516 gtk_main_iteration ();
1517 }
1518
1519 }

1523

1532 }

1534 }
1535
1536 string = g_string_truncate (string, string->len - 1);
1537 if (string->len <= 1) {
1538 continue;
1539 }
1540
1541 utf8 = g_filename_display_name (string->str);
1542 if (utf8 == NULL) {
1543 continue;
1544 }
1545
1546 if (strncmp (string->str, gsearch->command_details->look_in_folder_string, look_in_folder_string_length) == 0) {
1547
1548 if (strlen (string->str) != look_in_folder_string_length) {
1549
1550 filename = g_path_get_basename (utf8);
1551
1552 if (fnmatch (gsearch->command_details->name_contains_pattern_string, filename, FNM_NOESCAPE | FNM_CASEFOLD ) != FNM_NOMATCH) {
1553 if (gsearch->command_details->is_command_show_hidden_files_enabled) {
1554 if (gsearch->command_details->is_command_regex_matching_enabled == FALSE) {
1555 add_file_to_search_results (string->str, gsearch->search_results_list_store, &gsearch->search_results_iter, gsearch);
1556 }
1557 else if (compare_regex (gsearch->command_details->name_contains_regex_string, filename)) {
1558 add_file_to_search_results (string->str, gsearch->search_results_list_store, &gsearch->search_results_iter, gsearch);
1559 }
1560 }
1561 else if ((is_path_hidden (string->str) == FALSE ||
1562 is_path_hidden (gsearch->command_details->look_in_folder_string) == TRUE) &&
1563 (!g_str_has_suffix (string->str, "~"))) {
1564 if (gsearch->command_details->is_command_regex_matching_enabled == FALSE) {
1565 add_file_to_search_results (string->str, gsearch->search_results_list_store, &gsearch->search_results_iter, gsearch);

1568 add_file_to_search_results (string->str, gsearch->search_results_list_store, &gsearch->search_results_iter, gsearch);
1569 }
1570 }
1571 }
1572 }
1573 }
1574 g_free (utf8);
1575 g_free (filename);
1576
1577 gtk_tree_view_get_visible_rect (GTK_TREE_VIEW (gsearch->search_results_tree_view), &prior_rect);
1578
1579 if (prior_rect.y == 0) {
1580 gtk_tree_view_get_visible_rect (GTK_TREE_VIEW (gsearch->search_results_tree_view), &after_rect);

1582 gtk_tree_view_scroll_to_point (GTK_TREE_VIEW (gsearch->search_results_tree_view), -1, 0);
1583 }
1584 }
1585
1586 g_timer_elapsed (timer, &duration);
1587
1588 if (duration > MATE_SEARCH_TOOL_REFRESH_DURATION) {
1589 if (gtk_events_pending ()) {
1590 intermediate_file_count_update (gsearch);
1591 while (gtk_events_pending ()) {
1592 if (gsearch->command_details->command_status == MAKE_IT_QUIT) {
1593 return FALSE;
1594 }
1595 gtk_main_iteration ();
1596 }
1597 }
1598 g_timer_reset (timer);
1599 }
1600
1601 } while (g_io_channel_get_buffer_condition (ioc) & G_IO_IN);
1602
1603 g_string_free (string, TRUE);
1604 g_timer_destroy (timer);
1605 }
1606
1607 if (!(condition & G_IO_IN) || broken_pipe == TRUE) {
1608
1609 g_io_channel_shutdown (ioc, TRUE, NULL);
1610
1611 if ((gsearch->command_details->command_status != MAKE_IT_STOP)
1612 && (gsearch->command_details->is_command_using_quick_mode == TRUE)
1613 && (gsearch->command_details->is_command_first_pass == TRUE)
1614 && (gsearch->command_details->is_command_second_pass_enabled == TRUE)

1618

1620 g_free (gsearch->command_details->name_contains_pattern_string);
1621 g_free (gsearch->command_details->name_contains_regex_string);

1624 if (command != NULL) {
1625 spawn_search_command (gsearch, command);

1638
1639 /* Free the gchar fields of search_command structure. */
1640 g_free (gsearch->command_details->name_contains_pattern_string);
1641 g_free (gsearch->command_details->name_contains_regex_string);
1642
1643 }
1644 return FALSE;
1645 }
1646 return TRUE;
