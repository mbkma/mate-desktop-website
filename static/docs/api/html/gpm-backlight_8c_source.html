100gboolean
101gpm_backlight_get_brightness (GpmBacklight *backlight, guint *brightness, GError **error)
102{
103 guint level;
104 gboolean ret;
105 g_return_val_if_fail (backlight != NULL, FALSE);
106 g_return_val_if_fail (GPM_IS_BACKLIGHT (backlight), FALSE);
107 g_return_val_if_fail (brightness != NULL, FALSE);
108
109 /* check if we have the hw */
110 if (backlight->priv->can_dim == FALSE) {
111 g_set_error_literal (error, gpm_backlight_error_quark (),
112 GPM_BACKLIGHT_ERROR_HARDWARE_NOT_PRESENT,
113 "Dim capable hardware not present");
114 return FALSE;
115 }
116
117 /* gets the current brightness */
118 ret = gpm_brightness_get (backlight->priv->brightness, &level);
119 if (ret) {
120 *brightness = level;
121 } else {
122 g_set_error_literal (error, gpm_backlight_error_quark (),
123 GPM_BACKLIGHT_ERROR_DATA_NOT_AVAILABLE,
124 "Data not available");
125 }
126 return ret;
127}

132gboolean
133gpm_backlight_set_brightness (GpmBacklight *backlight, guint percentage, GError **error)
134{
135 gboolean ret;
136 gboolean hw_changed;
137
138 g_return_val_if_fail (backlight != NULL, FALSE);
139 g_return_val_if_fail (GPM_IS_BACKLIGHT (backlight), FALSE);
140
141 /* check if we have the hw */
142 if (backlight->priv->can_dim == FALSE) {
143 g_set_error_literal (error, gpm_backlight_error_quark (),
144 GPM_BACKLIGHT_ERROR_HARDWARE_NOT_PRESENT,
145 "Dim capable hardware not present");
146 return FALSE;
147 }
148
149 /* just set the master percentage for now, don't try to be clever */
150 backlight->priv->master_percentage = percentage;
151
152 /* sets the current policy brightness */
153 ret = gpm_brightness_set (backlight->priv->brightness, percentage, &hw_changed);
154 if (!ret) {
155 g_set_error_literal (error, gpm_backlight_error_quark (),
156 GPM_BACKLIGHT_ERROR_GENERAL,
157 "Cannot set policy brightness");
158 }
159 /* we emit a signal for the brightness applet */
160 if (ret && hw_changed) {
161 g_debug ("emitting brightness-changed : %u", percentage);
162 g_signal_emit (backlight, signals [BRIGHTNESS_CHANGED], 0, percentage);
163 }
164 return ret;
165}

195static void
196gpm_backlight_dialog_show (GpmBacklight *backlight)
197{
198 int orig_w;
199 int orig_h;
200 int screen_w;
201 int screen_h;
202 int x;
203 int y;
204 int pointer_x;
205 int pointer_y;
206 GtkRequisition win_req;
207 GdkScreen *pointer_screen;
208 GdkRectangle geometry;
209 GdkMonitor *monitor;
210 GdkDisplay *display;
211 GdkSeat *seat;
212 GdkDevice *device;
213
214 /*
215 * get the window size
216 * if the window hasn't been mapped, it doesn't necessarily
217 * know its true size, yet, so we need to jump through hoops
218 */
219 gtk_window_get_default_size (GTK_WINDOW (backlight->priv->popup), &orig_w, &orig_h);
220 gtk_widget_get_preferred_size (backlight->priv->popup, NULL, &win_req);
221
222 if (win_req.width > orig_w) {
223 orig_w = win_req.width;
224 }
225 if (win_req.height > orig_h) {
226 orig_h = win_req.height;
227 }
228
229 pointer_screen = NULL;
230 display = gtk_widget_get_display (backlight->priv->popup);
231 seat = gdk_display_get_default_seat (display);
232 device = gdk_seat_get_pointer (seat);
233 gdk_device_get_position (device,
234 &pointer_screen,
235 &pointer_x,
236 &pointer_y);
237
238 monitor = gdk_display_get_monitor_at_point (gdk_screen_get_display (pointer_screen),
239 pointer_x,
240 pointer_y);
241
242 gdk_monitor_get_geometry (monitor, &geometry);
243
244 screen_w = geometry.width;
245 screen_h = geometry.height;
246
247 x = ((screen_w - orig_w) / 2) + geometry.x;
248 y = geometry.y + (screen_h / 2) + (screen_h / 2 - orig_h) / 2;
249
250 gtk_window_move (GTK_WINDOW (backlight->priv->popup), x, y);
251
252 gtk_widget_show (backlight->priv->popup);
253
254 gdk_display_sync (gtk_widget_get_display (backlight->priv->popup));
255}

260static gboolean
261gpm_backlight_brightness_evaluate_and_set (GpmBacklight *backlight, gboolean interactive, gboolean use_initial)
262{
263 gfloat brightness;
264 gfloat scale;
265 gboolean ret;
266 gboolean on_battery;
267 gboolean do_laptop_lcd;
268 gboolean enable_action;
269 gboolean battery_reduce;
270 gboolean hw_changed;
271 guint value;
272 guint old_value;
273
274 if (backlight->priv->can_dim == FALSE) {
275 g_debug ("no dimming hardware");
276 return FALSE;
277 }
278
279 do_laptop_lcd = g_settings_get_boolean (backlight->priv->settings, GPM_SETTINGS_BACKLIGHT_ENABLE);
280 if (do_laptop_lcd == FALSE) {
281 g_warning ("policy is no dimming");
282 return FALSE;
283 }
284
285 /* get the last set brightness */
286 brightness = backlight->priv->master_percentage / 100.0f;
287 g_debug ("1. main brightness %f", brightness);
288
289 /* get battery status */
290 g_object_get (backlight->priv->client,
291 "on-battery", &on_battery,
292 NULL);
293
294 /* reduce if on battery power if we should */
295 if (use_initial) {
296 g_debug ("Setting initial brightness level");
297 battery_reduce = g_settings_get_boolean (backlight->priv->settings, GPM_SETTINGS_BACKLIGHT_BATTERY_REDUCE);
298 if (on_battery && battery_reduce) {
299 value = g_settings_get_int (backlight->priv->settings, GPM_SETTINGS_BRIGHTNESS_DIM_BATT);
300 if (value > 100) {
301 g_warning ("cannot use battery brightness value %u, correcting to 50", value);
302 value = 50;
303 }
304 scale = (100 - value) / 100.0f;
305 brightness *= scale;
306 } else {
307 scale = 1.0f;
308 }
309 g_debug ("2. battery scale %f, brightness %f", scale, brightness);
310 }
311
312 /* reduce if system is momentarily idle */
313 if (!on_battery)
314 enable_action = g_settings_get_boolean (backlight->priv->settings, GPM_SETTINGS_IDLE_DIM_AC);
315 else
316 enable_action = g_settings_get_boolean (backlight->priv->settings, GPM_SETTINGS_IDLE_DIM_BATT);
317 if (enable_action && backlight->priv->system_is_idle) {
318 value = g_settings_get_int (backlight->priv->settings, GPM_SETTINGS_IDLE_BRIGHTNESS);
319 if (value > 100) {
320 g_warning ("cannot use idle brightness value %u, correcting to 50", value);
321 value = 50;
322 }
323 scale = value / 100.0f;
324 brightness *= scale;
325 } else {
326 scale = 1.0f;
327 }
328 g_debug ("3. idle scale %f, brightness %f", scale, brightness);
329
330 /* convert to percentage */
331 value = (guint) ((brightness * 100.0f) + 0.5);
332
333 /* only do stuff if the brightness is different */
334 gpm_brightness_get (backlight->priv->brightness, &old_value);
335 if (old_value == value) {
336 g_debug ("values are the same, no action");
337 return FALSE;
338 }
339
340 /* only show dialog if interactive */
341 if (interactive) {
342 gpm_backlight_dialog_init (backlight);
343 msd_media_keys_window_set_volume_level (MSD_MEDIA_KEYS_WINDOW (backlight->priv->popup),
344 round (brightness));
345 gpm_backlight_dialog_show (backlight);
346 }
347
348 ret = gpm_brightness_set (backlight->priv->brightness, value, &hw_changed);
349 /* we emit a signal for the brightness applet */
350 if (ret && hw_changed) {
351 g_debug ("emitting brightness-changed : %u", value);
352 g_signal_emit (backlight, signals [BRIGHTNESS_CHANGED], 0, value);
353 }
354 return TRUE;
355}

362static void
363gpm_settings_key_changed_cb (GSettings *settings, const gchar *key, GpmBacklight *backlight)
364{
365 gboolean on_battery;
366
367 /* get battery status */
368 g_object_get (backlight->priv->client,
369 "on-battery", &on_battery,
370 NULL);
371
372 if (g_strcmp0 (key, GPM_SETTINGS_BRIGHTNESS_AC) == 0) {
373 backlight->priv->master_percentage = g_settings_get_double (settings, key);
374 gpm_backlight_brightness_evaluate_and_set (backlight, FALSE, TRUE);
375
376 } else if (on_battery && g_strcmp0 (key, GPM_SETTINGS_BRIGHTNESS_DIM_BATT) == 0) {
377 gpm_backlight_brightness_evaluate_and_set (backlight, FALSE, TRUE);
378
379 } else if (g_strcmp0 (key, GPM_SETTINGS_IDLE_DIM_AC) == 0 ||
380 g_strcmp0 (key, GPM_SETTINGS_BACKLIGHT_ENABLE) == 0 ||
381 g_strcmp0 (key, GPM_SETTINGS_SLEEP_DISPLAY_BATT) == 0 ||
382 g_strcmp0 (key, GPM_SETTINGS_BACKLIGHT_BATTERY_REDUCE) == 0 ||
383 g_strcmp0 (key, GPM_SETTINGS_IDLE_BRIGHTNESS) == 0) {
384 gpm_backlight_brightness_evaluate_and_set (backlight, FALSE, TRUE);
385
386 } else if (g_strcmp0 (key, GPM_SETTINGS_IDLE_DIM_TIME) == 0) {
387 backlight->priv->idle_dim_timeout = g_settings_get_int (settings, key);
388 gpm_idle_set_timeout_dim (backlight->priv->idle, backlight->priv->idle_dim_timeout);
389 } else {
390 g_debug ("unknown key %s", key);
391 }
392}

412static void
413gpm_backlight_save_settings (GpmBacklight *backlight, guint percentage)
414{
415 gint battery_reduce;
416 gfloat brightness_ac;
417 gboolean on_battery;
418
419 backlight->priv->master_percentage = percentage;
420 g_object_get (backlight->priv->client, "on-battery", &on_battery, NULL);
421 if (on_battery) {
422 /* If using battery, saving settings needs a bit of trickery.
423 * IMHO this shows that using two factors for setting
424 * brightness level is a crooked idea to begin with, but
425 * this is still less random than only saving changes when
426 * running on AC.
427 */
428 brightness_ac = g_settings_get_double (backlight->priv->settings, GPM_SETTINGS_BRIGHTNESS_AC);
429 if (brightness_ac) {
430 battery_reduce = 100 - (gint) (percentage * 100.0f / brightness_ac);
431 } else {
432 /* Any negative number indicates we surpassed brightness_ac. 0 indicates nothing changed. */
433 battery_reduce = - (gint) percentage;
434 }
435 if (battery_reduce < 0) {
436 /* Brightness set higher than brightness_ac - we have to adjust that value. */
437 g_settings_set_double (backlight->priv->settings, GPM_SETTINGS_BRIGHTNESS_AC,
438 percentage * 1.0);
439 battery_reduce = 0;
440 g_debug ("saving brightness for ac supply: %u", percentage);
441 }
442 g_debug ("saving brightness for battery - master percent: %u; ac supply: %f; dim battery: %u",
443 percentage, brightness_ac, battery_reduce);
444 g_settings_set_uint (backlight->priv->settings, GPM_SETTINGS_BRIGHTNESS_DIM_BATT, (guint) battery_reduce);
445 } else {
446 /* if using AC power supply, save the new brightness settings */
447 g_debug ("saving brightness for ac supply: %u", percentage);
448 g_settings_set_double (backlight->priv->settings, GPM_SETTINGS_BRIGHTNESS_AC,
449 percentage * 1.0);
450 }
451}

460static void
461gpm_backlight_button_pressed_cb (GpmButton *button, const gchar *type, GpmBacklight *backlight)
462{
463 gboolean ret;
464 GError *error = NULL;
465 guint percentage;
466 gboolean hw_changed;
467 g_debug ("Button press event type=%s", type);
468
469 if (g_strcmp0 (type, GPM_BUTTON_BRIGHT_UP) == 0) {
470 /* go up one step */
471 ret = gpm_brightness_up (backlight->priv->brightness, &hw_changed);
472
473 /* show the new value */
474 if (ret) {
475 gpm_brightness_get (backlight->priv->brightness, &percentage);
476 gpm_backlight_dialog_init (backlight);
477 msd_media_keys_window_set_volume_level (MSD_MEDIA_KEYS_WINDOW (backlight->priv->popup),
478 percentage);
479 gpm_backlight_dialog_show (backlight);
480 /* save the new percentage */
481 gpm_backlight_save_settings (backlight, percentage);
482 }
483 /* we emit a signal for the brightness applet */
484 if (ret && hw_changed) {
485 g_debug ("emitting brightness-changed : %u", percentage);
486 g_signal_emit (backlight, signals [BRIGHTNESS_CHANGED], 0, percentage);
487 }
488 } else if (g_strcmp0 (type, GPM_BUTTON_BRIGHT_DOWN) == 0) {
489 /* go up down step */
490 ret = gpm_brightness_down (backlight->priv->brightness, &hw_changed);
491
492 /* show the new value */
493 if (ret) {
494 gpm_brightness_get (backlight->priv->brightness, &percentage);
495 gpm_backlight_dialog_init (backlight);
496 msd_media_keys_window_set_volume_level (MSD_MEDIA_KEYS_WINDOW (backlight->priv->popup),
497 percentage);
498 gpm_backlight_dialog_show (backlight);
499 /* save the new percentage */
500 gpm_backlight_save_settings (backlight, percentage);
501 }
502 /* we emit a signal for the brightness applet */
503 if (ret && hw_changed) {
504 g_debug ("emitting brightness-changed : %u", percentage);
505 g_signal_emit (backlight, signals [BRIGHTNESS_CHANGED], 0, percentage);
506 }
507 } else if (g_strcmp0 (type, GPM_BUTTON_LID_OPEN) == 0) {
508 /* make sure we undim when we lift the lid */
509 gpm_backlight_brightness_evaluate_and_set (backlight, FALSE, TRUE);
510
511 /* ensure backlight is on */
512 ret = gpm_dpms_set_mode (backlight->priv->dpms, GPM_DPMS_MODE_ON, &error);
513 if (!ret) {
514 g_warning ("failed to turn on DPMS: %s", error->message);
515 g_error_free (error);
516 }
517 }

523static gboolean
524gpm_backlight_notify_system_idle_changed (GpmBacklight *backlight, gboolean is_idle)
525{
526 gdouble elapsed;
527
528 /* no point continuing */
529 if (backlight->priv->system_is_idle == is_idle) {
530 g_debug ("state not changed");
531 return FALSE;
532 }
533
534 /* get elapsed time and reset timer */
535 elapsed = g_timer_elapsed (backlight->priv->idle_timer, NULL);
536 g_timer_reset (backlight->priv->idle_timer);
537
538 if (is_idle == FALSE) {
539 g_debug ("we have just been idle for %lfs", elapsed);
540
541 /* The user immediatly undimmed the screen!
542 * We should double the timeout to avoid this happening again */
543 if (elapsed < 10) {
544 /* double the event time */
545 backlight->priv->idle_dim_timeout *= 2.0;
546 g_debug ("increasing idle dim time to %us", backlight->priv->idle_dim_timeout);
547 gpm_idle_set_timeout_dim (backlight->priv->idle, backlight->priv->idle_dim_timeout);
548 }
549
550 /* We reset the dimming after 2 minutes of idle,
551 * as the user will have changed tasks */
552 if (elapsed > 2*60) {
553 /* reset back to our default dimming */
554 backlight->priv->idle_dim_timeout =
555 g_settings_get_int (backlight->priv->settings,
556 GPM_SETTINGS_IDLE_DIM_TIME);
557 g_debug ("resetting idle dim time to %us", backlight->priv->idle_dim_timeout);
558 gpm_idle_set_timeout_dim (backlight->priv->idle, backlight->priv->idle_dim_timeout);
559 }
560 } else {
561 g_debug ("we were active for %lfs", elapsed);
562 }
563
564 g_debug ("changing powersave idle status to %i", is_idle);
565 backlight->priv->system_is_idle = is_idle;
566 return TRUE;
567}

580static void
581idle_changed_cb (GpmIdle *idle, GpmIdleMode mode, GpmBacklight *backlight)
582{
583 gboolean ret;
584 GError *error = NULL;
585 gboolean on_battery;
586 GpmDpmsMode dpms_mode;
587
588 /* don't dim or undim the screen when the lid is closed */
589 if (gpm_button_is_lid_closed (backlight->priv->button))
590 return;
591
592 /* don't dim or undim the screen unless ConsoleKit/systemd say we are on the active console */
593 if (!LOGIND_RUNNING() && !egg_console_kit_is_active (backlight->priv->console)) {
594 g_debug ("ignoring as not on active console");
595 return;
596 }
597
598 if (mode == GPM_IDLE_MODE_NORMAL) {
599 /* sync lcd brightness */
600 gpm_backlight_notify_system_idle_changed (backlight, FALSE);
601 gpm_backlight_brightness_evaluate_and_set (backlight, FALSE, TRUE);
602
603 /* ensure backlight is on */
604 ret = gpm_dpms_set_mode (backlight->priv->dpms, GPM_DPMS_MODE_ON, &error);
605 if (!ret) {
606 g_warning ("failed to turn on DPMS: %s", error->message);
607 g_error_free (error);
608 }
609
610 } else if (mode == GPM_IDLE_MODE_DIM) {
611
612 /* sync lcd brightness */
613 gpm_backlight_notify_system_idle_changed (backlight, TRUE);
614 gpm_backlight_brightness_evaluate_and_set (backlight, FALSE, TRUE);
615
616 /* ensure backlight is on */
617 ret = gpm_dpms_set_mode (backlight->priv->dpms, GPM_DPMS_MODE_ON, &error);
618 if (!ret) {
619 g_warning ("failed to turn on DPMS: %s", error->message);
620 g_error_free (error);
621 }
622
623 } else if (mode == GPM_IDLE_MODE_BLANK) {
624
625 /* sync lcd brightness */
626 gpm_backlight_notify_system_idle_changed (backlight, TRUE);
627 gpm_backlight_brightness_evaluate_and_set (backlight, FALSE, TRUE);
628
629 /* get the DPMS state we're supposed to use on the power state */
630 g_object_get (backlight->priv->client,
631 "on-battery", &on_battery,
632 NULL);
633 if (!on_battery)
634 dpms_mode = g_settings_get_enum (backlight->priv->settings, GPM_SETTINGS_DPMS_METHOD_AC);
635 else
636 dpms_mode = g_settings_get_enum (backlight->priv->settings, GPM_SETTINGS_DPMS_METHOD_BATT);
637
638 /* check if method is valid */
639 if (dpms_mode == GPM_DPMS_MODE_UNKNOWN || dpms_mode == GPM_DPMS_MODE_ON) {
640 g_warning ("BACKLIGHT method %u unknown. Using OFF.", dpms_mode);
641 dpms_mode = GPM_DPMS_MODE_OFF;
642 }
643
644 /* turn backlight off */
645 ret = gpm_dpms_set_mode (backlight->priv->dpms, dpms_mode, &error);
646 if (!ret) {
647 g_warning ("failed to change DPMS: %s", error->message);
648 g_error_free (error);
649 }
650
651 }
652}

697static void
698gpm_backlight_finalize (GObject *object)
699{
700 GpmBacklight *backlight;
701 g_return_if_fail (object != NULL);
702 g_return_if_fail (GPM_IS_BACKLIGHT (object));
703 backlight = GPM_BACKLIGHT (object);
704
705 g_timer_destroy (backlight->priv->idle_timer);
706 gtk_widget_destroy (backlight->priv->popup);
707
708 g_object_unref (backlight->priv->dpms);
709 g_object_unref (backlight->priv->control);
710 g_object_unref (backlight->priv->settings);
711 g_object_unref (backlight->priv->client);
712 g_object_unref (backlight->priv->button);
713 g_object_unref (backlight->priv->idle);
714 g_object_unref (backlight->priv->brightness);

750{

754 backlight->priv->idle_timer = g_timer_new ();
755
756 /* watch for manual brightness changes (for the popup widget) */

760
761 /* we use up_client for the ac-adapter-changed signal */
762 backlight->priv->client = up_client_new ();
763 g_signal_connect (backlight->priv->client, "notify",
764 G_CALLBACK (gpm_backlight_client_changed_cb), backlight);
765
766 /* gets caps */
767 backlight->priv->can_dim = gpm_brightness_has_hw (backlight->priv->brightness);
768
769 /* watch for dim value changes */

771 g_signal_connect (backlight->priv->settings, "changed", G_CALLBACK (gpm_settings_key_changed_cb), backlight);
772
773 /* set the main brightness, this is designed to be updated if the user changes the
774 * brightness so we can undim to the 'correct' value */
775 backlight->priv->master_percentage = g_settings_get_double (backlight->priv->settings, GPM_SETTINGS_BRIGHTNESS_AC);
776
777 /* watch for brightness up and down buttons and also check lid state */
778 backlight->priv->button = gpm_button_new ();
779 g_signal_connect (backlight->priv->button, "button-pressed",

781
782 /* watch for idle mode changes */

786
787 /* assumption */
788 backlight->priv->system_is_idle = FALSE;
789 backlight->priv->idle_dim_timeout = g_settings_get_int (backlight->priv->settings, GPM_SETTINGS_IDLE_DIM_TIME);

791

793 backlight->priv->popup = msd_media_keys_window_new ();

795 "gpm-brightness-lcd",

806
807 /* Don't do dimming on inactive console */
