66static void
67gnome_keyring_daemon_startup (void)
68{
69 GError *error = NULL;
70 GPid pid;
71 gchar *argv[3];
72
73 error = NULL;
74 argv[0] = GNOME_KEYRING_DAEMON;
75 argv[1] = "--start";
76 argv[2] = NULL;
77 g_spawn_async (NULL, argv, NULL,
78 G_SPAWN_SEARCH_PATH | G_SPAWN_DO_NOT_REAP_CHILD,
79 NULL, NULL, &pid,
80 &error);
81
82 if (error != NULL)
83 {
84 g_printerr ("Failed to spawn gnome-keyring-daemon: %s\n",
85 error->message);
86 g_error_free (error);
87 return;
88 }
89
90 g_child_watch_add (pid, gnome_keyring_daemon_finished, NULL);
91}

93static void
94msm_compat_gnome_smproxy_startup (void)
95{
96 Atom gnome_sm_proxy;
97 Display *dpy;
98 Window root;
99 GdkDisplay *gdkdisplay;
100
101 gdkdisplay = gdk_display_get_default ();
102 gdk_x11_display_error_trap_push (gdkdisplay);
103
104 /* Set GNOME_SM_PROXY property, since some apps (like OOo) seem to require
105 * it to behave properly. Thanks to Jasper/Francois for reporting this.
106 * This has another advantage, since it prevents people from running
107 * gnome-smproxy in xfce4, which would cause trouble otherwise.
108 */
109 dpy = GDK_DISPLAY_XDISPLAY(gdkdisplay);
110 root = RootWindow (dpy, 0);
111
112 if (gnome_smproxy_window != None)
113 XDestroyWindow (dpy, gnome_smproxy_window);
114
115 gnome_sm_proxy = XInternAtom (dpy, "GNOME_SM_PROXY", False);
116 gnome_smproxy_window = XCreateSimpleWindow (dpy, root, 1, 1, 1, 1, 0, 0, 0);
117
118 XChangeProperty (dpy, gnome_smproxy_window, gnome_sm_proxy,
119 XA_CARDINAL, 32, PropModeReplace,
120 (unsigned char *) (void *) &gnome_smproxy_window, 1);
121 XChangeProperty (dpy, root, gnome_sm_proxy,
122 XA_CARDINAL, 32, PropModeReplace,
123 (unsigned char *) (void *) &gnome_smproxy_window, 1);
124
125 XSync (dpy, False);
126 gdk_x11_display_error_trap_pop_ignored (gdkdisplay);
127}

129static void
130msm_compat_gnome_smproxy_shutdown (void)
131{
132 GdkDisplay *gdkdisplay;
133
134 gdkdisplay = gdk_display_get_default ();
135 gdk_x11_display_error_trap_push (gdkdisplay);
136
137 if (gnome_smproxy_window != None)
138 {
139 XDestroyWindow (GDK_DISPLAY_XDISPLAY(gdkdisplay), gnome_smproxy_window);
140 XSync (GDK_DISPLAY_XDISPLAY(gdkdisplay), False);
141 gnome_smproxy_window = None;
142 }
143 gdk_x11_display_error_trap_pop_ignored (gdkdisplay);
144}

146void
147msm_gnome_start (void)
148{
149 GSettings* settings;
150 gchar **array;
151
152 if (gnome_compat_started == TRUE)
153 return;
154
155 settings = g_settings_new (GSM_SCHEMA);
156 array = g_settings_get_strv (settings, GSM_GNOME_COMPAT_STARTUP_KEY);
157 if (array) {
158 guint i;
159
160 for (i = 0; array[i]; i++) {
161 if (strcmp (array[i], "smproxy") == 0) {
162 g_debug ("MsmGnome: starting smproxy");
163 msm_compat_gnome_smproxy_startup ();
164 gnome_compat_started = TRUE;
165 } else if (strcmp (array[i], "keyring") == 0) {
166 g_debug ("MsmGnome: starting keyring");
167 gnome_keyring_daemon_startup ();
168 gnome_compat_started = TRUE;
169 } else {
170 g_debug ("MsmGnome: ignoring unknown component \"%s\"", array[i]);
171 }
172 }
173 g_strfreev (array);
174 } else {
175 g_debug ("MsmGnome: No components found to start");
176 }
177 g_object_unref (settings);
178}

180void
181msm_gnome_stop (void)
182{
183 if (gnome_compat_started == FALSE)
184 return;
185
186 g_debug ("MsmGnome: stopping");
187
188 msm_compat_gnome_smproxy_shutdown ();
189
190 gnome_compat_started = FALSE;
191}
