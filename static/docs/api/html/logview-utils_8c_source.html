82static GDate *
83string_get_date (const char *line, char **time_string, int *timestamp_len)
84{
85 GDate *date = NULL;
86 struct tm tp;
87 char *cp = NULL, *timestamp = NULL;
88
89 /* it's safe to assume that if strptime returns NULL, it's
90 * because of an error (format unmatched). being a log file, it's very
91 * unlikely that there aren't any more characters after the date.
92 */
93
94 if (line == NULL || line[0] == '\0') {
95 return NULL;
96 }
97
98 /* this parses the "MonthName DayNo" format */
99 cp = strptime (line, "%b %d", &tp);
100 if (cp) {
101 goto out;
102 }
103
104 /* this parses the YYYY-MM-DD format */
105 cp = strptime (line, "%F", &tp);
106 if (cp) {
107 goto out;
108 }
109
110out:
111 if (cp) {
112 /* the year doesn't matter to us now */
113 date = g_date_new_dmy (tp.tm_mday, tp.tm_mon + 1, 1);
114 *time_string = g_strndup (line, cp - line);
115
116 timestamp = strptime (cp, "%X", &tp);
117 if (timestamp) {
118 *timestamp_len = timestamp - line;
119 }
120 }
121
122 return date;
123}

139GSList *
140log_read_dates (const char **buffer_lines, time_t current)
141{
142 int current_year, offsetyear, i, n, rangemin, rangemax, timestamp_len = 0;
143 GSList *days = NULL;
144 GDate *date = NULL;
145 struct tm *tmptm;
146 char *date_string = NULL;
147 Day *day;
148 gboolean done = FALSE;
149
150 g_return_val_if_fail (buffer_lines != NULL, NULL);
151
152 n = g_strv_length ((char **) buffer_lines);
153
154 tmptm = localtime (&current);
155 current_year = tmptm->tm_year + 1900;
156 offsetyear = 0;
157
158 /* find the first line with a date we're able to parse */
159 for (i = 0; buffer_lines[i]; i++) {
160 if ((date = string_get_date (buffer_lines[i], &date_string, &timestamp_len)) != NULL)
161 break;
162 }
163
164 if (!date) {
165 /* no valid dates in the array, return NULL */
166 return NULL;
167 }
168
169 if (!g_date_valid (date)) {
170 g_date_free (date);
171 g_free (date_string);
172 return NULL;
173 }
174
175 g_date_set_year (date, current_year);
176
177 day = g_slice_new0 (Day);
178 days = g_slist_append (days, day);
179
180 /* $i now contains the line number for the first good date */
181 day->date = date;
182 day->first_line = i;
183 day->last_line = -1;
184 day->timestamp_len = timestamp_len;
185
186 /* now scan the logfile to get the last line of the day */
187 rangemin = i;
188 rangemax = n - 1;
189
190 while (!done) {
191 /* find out the last line of the day we're currently building */
192
193 i = n - 1;
194
195 while (day->last_line < 0) {
196 if (strstr (buffer_lines[i], date_string)) {
197 /* if we find the same string on the last line of the log, we're done */
198 if (i == n - 1) {
199 done = TRUE;
200 day->last_line = i;
201 break;
202 }
203
204 /* we're still in a section of lines with the same date;
205 * - if the next one changes, then we're on the last.
206 * - else we keep searching in the following.
207 */
208
209 if (!strstr (buffer_lines[i + 1], date_string)) {
210 day->last_line = i;
211 break;
212 } else {
213 rangemin = i;
214 i = floor (((float) i + (float) rangemax) / 2.);
215 }
216 } else {
217 /* we can't find the same date here; go back to a safer range. */
218 rangemax = i;
219 i = floor (((float) rangemin + (float) i) / 2.);
220 }
221 }
222
223 g_free (date_string);
224 date_string = NULL;
225
226 if (!done) {
227 /* this means we finished the current day but we're not at the end
228 * of the buffer: reset the parameters for the next day.
229 */
230 GDate *newdate = NULL;
231
232 for (i = day->last_line + 1; buffer_lines[i]; i++) {
233 if ((newdate = string_get_date (buffer_lines[i], &date_string, &timestamp_len)) != NULL)
234 break;
235 }
236
237 /* this will set the last line of the "old" log to either:
238 * - "n - 1" if we can't find another date
239 * - the line before the new date else.
240 */
241 day->last_line = i - 1;
242
243 if (newdate) {
244 /* append a new day to the list */
245
246 g_date_set_year (newdate, current_year + offsetyear);
247
248 if (g_date_compare (newdate, date) < 1) {
249 /* this isn't possible, as we're reading the log forward.
250 * so it means that newdate is the next year.
251 */
252 g_date_add_years (newdate, 1);
253 offsetyear++;
254 }
255
256 date = newdate;
257 day = g_slice_new0 (Day);
258 days = g_slist_prepend (days, day);
259
260 day->date = date;
261 day->first_line = i;
262 day->last_line = -1;
263 day->timestamp_len = timestamp_len;
264 rangemin = i;
265 rangemax = n - 1;
266 } else {
267 /* no other date found */
268 done = TRUE;
269 }
270 }
271 }
272
273 g_free (date_string);
274
275 /* sort the days in chronological order */
276 days = g_slist_sort (days, days_compare);
277
278 return days;
279}
