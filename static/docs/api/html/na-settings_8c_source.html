92/* Each consumer may register a callback function which will be triggered
93 * when a key is modified.
94 *
95 * The monitored key usually is the real key read in the file;
96 * as a special case, composite keys are defined:
97 * - NA_IPREFS_IO_PROVIDERS_READ_STATUS monitors the 'readable' key for all i/o providers
98 *
99 * Note that we actually monitor the _user_view_ of the configuration:
100 * e.g. if a key has a mandatory value in global conf, then the same
101 * key in user conf will just be ignored.
102 */
103typedef struct {
104 gchar *monitored_key;
105 GCallback callback;
106 gpointer user_data;
107}
108 Consumer;

280static void
281class_init( NASettingsClass *klass )
282{
283 static const gchar *thisfn = "na_settings_class_init";
284 GObjectClass *object_class;
285
286 g_debug( "%s: klass=%p", thisfn, ( void * ) klass );
287
288 st_parent_class = g_type_class_peek_parent( klass );
289
290 object_class = G_OBJECT_CLASS( klass );
291 object_class->dispose = instance_dispose;
292 object_class->finalize = instance_finalize;
293
294 klass->private = g_new0( NASettingsClassPrivate, 1 );
295
296 /*
297 * NASettings::settings-key-changed:
298 *
299 * This signal is sent by NASettings when the value of a key is modified.
300 *
301 * Arguments are the group, the key, the new value as a NABoxed,
302 * and whether it is mandatory.
303 *
304 * Handler is of type:
305 * void ( *handler )( NASettings *settings,
306 * const gchar *group,
307 * const gchar *key,
308 * NABoxed *value,
309 * gboolean mandatory,
310 * gpointer user_data );
311 *
312 * The default class handler frees these datas.
313 */
314 st_signals[ KEY_CHANGED ] = g_signal_new_class_handler(
315 SETTINGS_SIGNAL_KEY_CHANGED,
316 NA_SETTINGS_TYPE,
317 G_SIGNAL_RUN_CLEANUP | G_SIGNAL_ACTION,
318 G_CALLBACK( on_key_changed_final_handler ),
319 NULL, /* accumulator */
320 NULL, /* accumulator data */
321 na_cclosure_marshal_VOID__STRING_STRING_POINTER_BOOLEAN,
322 G_TYPE_NONE,
323 4,
324 G_TYPE_STRING, G_TYPE_STRING, G_TYPE_POINTER, G_TYPE_BOOLEAN );
325}

354static void
355instance_dispose( GObject *object )
356{
357 static const gchar *thisfn = "na_settings_instance_dispose";
358 NASettings *self;
359
360 g_return_if_fail( NA_IS_SETTINGS( object ));
361
362 self = NA_SETTINGS( object );
363
364 if( !self->private->dispose_has_run ){
365
366 g_debug( "%s: object=%p (%s)", thisfn, ( void * ) object, G_OBJECT_TYPE_NAME( object ));
367
368 self->private->dispose_has_run = TRUE;
369
370 release_key_file( self->private->mandatory );
371 release_key_file( self->private->user );
372
373 if( G_OBJECT_CLASS( st_parent_class )->dispose ){
374 G_OBJECT_CLASS( st_parent_class )->dispose( object );
375 }
376 }
377}

379static void
380instance_finalize( GObject *object )
381{
382 static const gchar *thisfn = "na_settings_instance_finalize";
383 NASettings *self;
384
385 g_return_if_fail( NA_IS_SETTINGS( object ));
386
387 g_debug( "%s: object=%p (%s)", thisfn, ( void * ) object, G_OBJECT_TYPE_NAME( object ));
388
389 self = NA_SETTINGS( object );
390
391 g_list_foreach( self->private->content, ( GFunc ) release_key_value, NULL );
392 g_list_free( self->private->content );
393
394 g_list_foreach( self->private->consumers, ( GFunc ) release_consumer, NULL );
395 g_list_free( self->private->consumers );
396
397 g_free( self->private );
398
399 /* chain call to parent class */
400 if( G_OBJECT_CLASS( st_parent_class )->finalize ){
401 G_OBJECT_CLASS( st_parent_class )->finalize( object );
402 }
403}

411static void
412settings_new( void )
413{
414 static const gchar *thisfn = "na_settings_new";
415 gchar *dir;
416 GList *content;
417
418 if( !st_settings ){
419 st_settings = g_object_new( NA_SETTINGS_TYPE, NULL );
420
421 g_debug( "%s: reading mandatory configuration", thisfn );
422 dir = g_build_filename( SYSCONFDIR, "xdg", PACKAGE, NULL );
423 st_settings->private->mandatory = key_file_new( dir );
424 g_free( dir );
425 st_settings->private->mandatory->mandatory = TRUE;
426 content = content_load_keys( NULL, st_settings->private->mandatory );
427
428 g_debug( "%s: reading user configuration", thisfn );
429 dir = g_build_filename( g_get_home_dir(), ".config", PACKAGE, NULL );
430 g_mkdir_with_parents( dir, 0750 );
431 st_settings->private->user = key_file_new( dir );
432 g_free( dir );
433 st_settings->private->mandatory->mandatory = FALSE;
434 content = content_load_keys( content, st_settings->private->user );
435
436 st_settings->private->content = g_list_copy( content );
437 g_list_free( content );
438 }
439}

522{
523 gboolean value;
524 KeyValue *key_value;
525 KeyDef *key_def;
526
527 value = FALSE;
528 key_value = read_key_value( group, key, found, mandatory );
529
530 if( key_value ){
531 value = na_boxed_get_boolean( key_value->boxed );
532 release_key_value( key_value );
533
534 } else {
535 key_def = get_key_def( key );
536 if( key_def ){
537 value = ( key_def->default_value ? ( strcasecmp( key_def->default_value, "true" ) == 0 || atoi( key_def->default_value ) != 0 ) : FALSE );
538 }
539 }
540
541 return( value );
542}

955/*
956 * returns a list of modified KeyValue
957 * - order in the lists is not signifiant
958 * - the mandatory flag is not signifiant
959 * - a key is modified:
960 * > if it appears in new
961 * > if it disappears: the value is so reset to its default
962 * > if the value has been modified
963 *
964 * we return here a new list, with newly allocated KeyValue structs
965 * which hold the new value of each modified key
966 */
967static GList *
968content_diff( GList *old, GList *new )
969{
970 GList *diffs, *io, *in;
971 KeyValue *kold, *knew, *kdiff;
972 gboolean found;
973
974 diffs = NULL;
975
976 for( io = old ; io ; io = io->next ){
977 kold = ( KeyValue * ) io->data;
978 found = FALSE;
979 for( in = new ; in && !found ; in = in->next ){
980 knew = ( KeyValue * ) in->data;
981 if( !strcmp( kold->group, knew->group ) && ( gpointer ) kold->def == ( gpointer ) knew->def ){
982 found = TRUE;
983 if( !na_boxed_are_equal( kold->boxed, knew->boxed )){
984 /* a key has been modified */
985 kdiff = g_new0( KeyValue, 1 );
986 kdiff->group = g_strdup( knew->group );
987 kdiff->def = knew->def;
988 kdiff->mandatory = knew->mandatory;
989 kdiff->boxed = na_boxed_copy( knew->boxed );
990 diffs = g_list_prepend( diffs, kdiff );
991 }
992 }
993 }
994 if( !found ){
995 /* a key has disappeared */
996 kdiff = g_new0( KeyValue, 1 );
997 kdiff->group = g_strdup( kold->group );
998 kdiff->def = kold->def;
999 kdiff->mandatory = FALSE;
1000 kdiff->boxed = na_boxed_new_from_string( kold->def->type, kold->def->default_value );
1001 diffs = g_list_prepend( diffs, kdiff );
1002 }
1003 }
1004
1005 for( in = new ; in ; in = in->next ){
1006 knew = ( KeyValue * ) in->data;
1007 found = FALSE;
1008 for( io = old ; io && !found ; io = io->next ){
1009 kold = ( KeyValue * ) io->data;
1010 if( !strcmp( kold->group, knew->group ) && ( gpointer ) kold->def == ( gpointer ) knew->def ){
1011 found = TRUE;
1012 }
1013 }
1014 if( !found ){
1015 /* a key is new */
1016 kdiff = g_new0( KeyValue, 1 );
1017 kdiff->group = g_strdup( knew->group );
1018 kdiff->def = knew->def;
1019 kdiff->mandatory = knew->mandatory;
1020 kdiff->boxed = na_boxed_copy( knew->boxed );
1021 diffs = g_list_prepend( diffs, kdiff );
1022 }
1023 }
1024
1025 return( diffs );
1026}
