109static void
110caja_navigation_window_init (CajaNavigationWindow *window)
111{
112 GtkUIManager *ui_manager;
113 GtkWidget *toolbar;
114 CajaWindow *win;
115 CajaNavigationWindowPane *pane;
116 GtkWidget *hpaned;
117 GtkWidget *vbox;
118
119 win = CAJA_WINDOW (window);
120
121 window->details = caja_navigation_window_get_instance_private (window);
122
123 GtkStyleContext *context;
124
125 context = gtk_widget_get_style_context (GTK_WIDGET (window));
126 gtk_style_context_add_class (context, "caja-navigation-window");
127
128 pane = caja_navigation_window_pane_new (win);
129 win->details->panes = g_list_prepend (win->details->panes, pane);
130
131 window->details->header_size_group = gtk_size_group_new (GTK_SIZE_GROUP_VERTICAL);
132 gtk_size_group_set_ignore_hidden (window->details->header_size_group, FALSE);

134 window->details->content_paned = gtk_paned_new (GTK_ORIENTATION_HORIZONTAL);
135 gtk_widget_set_hexpand (window->details->content_paned, TRUE);
136 gtk_widget_set_vexpand (window->details->content_paned, TRUE);
137 gtk_grid_attach (GTK_GRID (CAJA_WINDOW (window)->details->grid),
138 window->details->content_paned,
139 0, 3, 1, 1);
140 gtk_widget_show (window->details->content_paned);
141
142 vbox = gtk_box_new (GTK_ORIENTATION_VERTICAL, 0);
143 gtk_paned_pack2 (GTK_PANED (window->details->content_paned), vbox,

148 gtk_box_pack_start (GTK_BOX (vbox), hpaned, TRUE, TRUE, 0);

155 caja_navigation_window_pane_setup (pane);
156
157 gtk_paned_pack1 (GTK_PANED(hpaned), pane->widget, TRUE, FALSE);
158 gtk_widget_show (pane->widget);
159
160 /* this has to be done after the location bar has been set up,
161 * but before menu stuff is being called */

163
164 caja_navigation_window_initialize_actions (window);
165
166 caja_navigation_window_initialize_menus (window);
167
168 ui_manager = caja_window_get_ui_manager (CAJA_WINDOW (window));
169 toolbar = gtk_ui_manager_get_widget (ui_manager, "/Toolbar");
170 gtk_style_context_add_class (gtk_widget_get_style_context (toolbar), GTK_STYLE_CLASS_PRIMARY_TOOLBAR);

172 gtk_widget_set_hexpand (toolbar, TRUE);
173 gtk_grid_attach (GTK_GRID (CAJA_WINDOW (window)->details->grid),
174 toolbar,
175 0, 1, 1, 1);
176 gtk_widget_show (toolbar);
177
178 caja_navigation_window_initialize_toolbars (window);
179
180 /* Set initial sensitivity of some buttons & menu items
181 * now that they're all created.
182 */
183 caja_navigation_window_allow_back (window, FALSE);
184 caja_navigation_window_allow_forward (window, FALSE);
185
186 g_signal_connect_swapped (caja_preferences,
187 "changed::" CAJA_PREFERENCES_ALWAYS_USE_LOCATION_ENTRY,

192 "changed::" CAJA_PREFERENCES_ALWAYS_USE_BROWSER,
193 G_CALLBACK(always_use_browser_changed),

517static gboolean
518caja_navigation_window_key_press_event (GtkWidget *widget,
519 GdkEventKey *event)
520{
521 CajaNavigationWindow *window;
522 gsize i;
523
524 window = CAJA_NAVIGATION_WINDOW (widget);
525
526 if (event->state & GDK_CONTROL_MASK)
527 {
528 GSettings *settings = g_settings_new ("org.mate.caja.preferences");
529 gboolean handled = FALSE;
530
531 if (g_settings_get_boolean (settings, "ctrl-tab-switch-tabs"))
532 {
533 CajaWindow *cajawin;
534 CajaWindowSlot *slot;
535 CajaNavigationWindowPane *pane;
536 CajaNotebook *cajanotebook;
537 GtkNotebook *notebook;
538 int pages;
539 int page_num;
540
541 cajawin = CAJA_WINDOW (window);
542 slot = caja_window_get_active_slot (cajawin);
543 pane = CAJA_NAVIGATION_WINDOW_PANE (slot->pane);
544 cajanotebook = CAJA_NOTEBOOK (pane->notebook);
545 notebook = GTK_NOTEBOOK (cajanotebook);
546 pages = gtk_notebook_get_n_pages (notebook);
547 page_num = gtk_notebook_get_current_page (notebook);
548
549 if (event->keyval == GDK_KEY_ISO_Left_Tab)
550 {
551 if (page_num != 0)
552 gtk_notebook_prev_page (notebook);
553 else
554 gtk_notebook_set_current_page (notebook, (pages - 1));
555 handled = TRUE;
556 }
557 if (event->keyval == GDK_KEY_Tab)
558 {
559 if (page_num != (pages -1))
560 gtk_notebook_next_page (notebook);
561 else
562 gtk_notebook_set_current_page (notebook, 0);
563 handled = TRUE;
564 }
565 }
566 g_object_unref (settings);
567
568 if (handled)
569 return TRUE;
570 }
571
572 for (i = 0; i < G_N_ELEMENTS (extra_navigation_window_keybindings); i++)
573 {
574 if (extra_navigation_window_keybindings[i].keyval == event->keyval)
575 {
576 GtkAction *action;
577
578 G_GNUC_BEGIN_IGNORE_DEPRECATIONS;
579 action = gtk_action_group_get_action (window->details->navigation_action_group,
580 extra_navigation_window_keybindings[i].action);
581
582 g_assert (action != NULL);
583 if (gtk_action_is_sensitive (action))
584 {
585 gtk_action_activate (action);
586 return TRUE;
587 }
588 G_GNUC_END_IGNORE_DEPRECATIONS;
589
590 break;
591 }
592 }
593
594 return GTK_WIDGET_CLASS (caja_navigation_window_parent_class)->key_press_event (widget, event);
595}

597static gboolean
598caja_navigation_window_button_press_event (GtkWidget *widget,
599 GdkEventButton *event)
600{
601 CajaNavigationWindow *window;
602 gboolean handled;
603
604 handled = FALSE;
605 window = CAJA_NAVIGATION_WINDOW (widget);
606
607 if (mouse_extra_buttons && (((int) event->button) == mouse_back_button))
608 {
609 caja_navigation_window_go_back (window);
610 handled = TRUE;
611 }
612 else if (mouse_extra_buttons && (((int) event->button) == mouse_forward_button))
613 {
614 caja_navigation_window_go_forward (window);
615 handled = TRUE;
616 }
617 else if (GTK_WIDGET_CLASS (caja_navigation_window_parent_class)->button_press_event)
618 {
619 handled = GTK_WIDGET_CLASS (caja_navigation_window_parent_class)->button_press_event (widget, event);
620 }
621 else
622 {
623 handled = FALSE;
624 }
625 return handled;
626}
