56static void
57update_sensitivity_for_ext (DlgNewData *data,
58 const char *ext)
59{
60 const char *mime_type;
61 int i;
62
63 data->can_encrypt = FALSE;
64 data->can_encrypt_header = FALSE;
65 data->can_create_volumes = FALSE;
66
67 mime_type = get_mime_type_from_extension (ext);
68
69 if (mime_type == NULL) {
70 gtk_widget_set_sensitive (data->n_password_entry, FALSE);
71 gtk_widget_set_sensitive (data->n_password_label, FALSE);
72 gtk_widget_set_sensitive (data->n_encrypt_header_checkbutton, FALSE);
73 gtk_widget_set_sensitive (data->n_volume_box, FALSE);
74 return;
75 }
76
77 for (i = 0; mime_type_desc[i].mime_type != NULL; i++) {
78 if (strcmp (mime_type_desc[i].mime_type, mime_type) == 0) {
79 data->can_encrypt = mime_type_desc[i].capabilities & FR_COMMAND_CAN_ENCRYPT;
80 gtk_widget_set_sensitive (data->n_password_entry, data->can_encrypt);
81 gtk_widget_set_sensitive (data->n_password_label, data->can_encrypt);
82
83 data->can_encrypt_header = mime_type_desc[i].capabilities & FR_COMMAND_CAN_ENCRYPT_HEADER;
84 gtk_widget_set_sensitive (data->n_encrypt_header_checkbutton, data->can_encrypt_header);
85
86 data->can_create_volumes = mime_type_desc[i].capabilities & FR_COMMAND_CAN_CREATE_VOLUMES;
87 gtk_widget_set_sensitive (data->n_volume_box, data->can_create_volumes);
88
89 break;
90 }
91 }
92
93 update_sensitivity (data);
94}

223static char *
224get_icon_name_for_type (const char *mime_type)
225{
226 char *name = NULL;
227
228 if (mime_type != NULL) {
229 char *s;
230
231 name = g_strconcat ("mate-mime-", mime_type, NULL);
232 for (s = name; *s; ++s)
233 if (! g_ascii_isalpha (*s))
234 *s = '-';
235 }
236
237 if ((name == NULL) || ! gtk_icon_theme_has_icon (gtk_icon_theme_get_default (), name)) {
238 g_free (name);
239 name = g_strdup ("package-x-generic");
240 }
241
242 return name;

252static DlgNewData *
253dlg_new_archive (FrWindow *window,
254 int *supported_types,
255 const char *default_name)
256{
257 GtkBuilder *builder;
258 DlgNewData *data;
259 GSettings *settings;
260 int i;
261 int size;
262
263 data = g_new0 (DlgNewData, 1);
264 builder = gtk_builder_new_from_resource (ENGRAMPA_RESOURCE_UI_PATH G_DIR_SEPARATOR_S "new.ui");
265 data->window = window;
266 data->supported_types = supported_types;
267 sort_mime_types_by_description (data->supported_types);
268
269 /* Get the widgets. */
270
271 data->dialog = GET_WIDGET ("dialog");
272
273 data->n_password_entry = GET_WIDGET ("n_password_entry");
274 data->n_password_label = GET_WIDGET ("n_password_label");
275 data->n_other_options_expander = GET_WIDGET ("n_other_options_expander");
276 data->n_encrypt_header_checkbutton = GET_WIDGET ("n_encrypt_header_checkbutton");
277
278 data->n_volume_checkbutton = GET_WIDGET ("n_volume_checkbutton");
279 data->n_volume_spinbutton = GET_WIDGET ("n_volume_spinbutton");
280 data->n_volume_box = GET_WIDGET ("n_volume_box");
281
282 /* Set widgets data. */
283
284 gtk_dialog_set_default_response (GTK_DIALOG (data->dialog), GTK_RESPONSE_OK);
285 gtk_file_chooser_set_current_folder_uri (GTK_FILE_CHOOSER (data->dialog), fr_window_get_open_default_dir (window));
286
287 if (default_name != NULL)
288 gtk_file_chooser_set_current_name (GTK_FILE_CHOOSER (data->dialog), default_name);
289
290 /**/
291
292 gtk_expander_set_expanded (GTK_EXPANDER (data->n_other_options_expander), FALSE);
293 settings = g_settings_new (ENGRAMPA_SCHEMA_GENERAL);
294 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (data->n_encrypt_header_checkbutton), g_settings_get_boolean (settings, PREF_GENERAL_ENCRYPT_HEADER));
295 g_object_unref (settings);
296
297 settings = g_settings_new (ENGRAMPA_SCHEMA_BATCH_ADD);
298 size = g_settings_get_int (settings, PREF_BATCH_ADD_VOLUME_SIZE);
299 gtk_spin_button_set_value (GTK_SPIN_BUTTON (data->n_volume_spinbutton),
300 ((gdouble) size) / MEGABYTE);
301 g_object_unref (settings);
302
303 /* format chooser */
304
305 data->format_chooser = (EggFileFormatChooser *) egg_file_format_chooser_new ();
306 for (i = 0; data->supported_types[i] != -1; i++) {
307 int idx = data->supported_types[i];
308 const char *exts[4];
309 int e;
310 int n_exts;
311 char *icon_name;
312
313 n_exts = 0;
314 for (e = 0; (n_exts < 4) && file_ext_type[e].ext != NULL; e++) {
315 if (strcmp (file_ext_type[e].ext, mime_type_desc[idx].default_ext) == 0)
316 continue;
317 if (strcmp (file_ext_type[e].mime_type, mime_type_desc[idx].mime_type) == 0)
318 exts[n_exts++] = file_ext_type[e].ext;
319 }
320 while (n_exts < 4)
321 exts[n_exts++] = NULL;
322
323 /* g_print ("%s => %s, %s, %s, %s\n", mime_type_desc[idx].mime_type, exts[0], exts[1], exts[2], exts[3]); */
324
325 icon_name = get_icon_name_for_type (mime_type_desc[idx].mime_type);
326 egg_file_format_chooser_add_format (data->format_chooser,
327 0,
328 _(mime_type_desc[idx].name),
329 icon_name,
330 mime_type_desc[idx].default_ext,
331 exts[0],
332 exts[1],
333 exts[2],
334 exts[3],
335 NULL);
336
337 g_free (icon_name);
338 }
339 egg_file_format_chooser_set_format (data->format_chooser, 0);
340 gtk_widget_show (GTK_WIDGET (data->format_chooser));
341 gtk_box_pack_start (GTK_BOX (GET_WIDGET ("format_chooser_box")), GTK_WIDGET (data->format_chooser), TRUE, TRUE, 0);
342 gtk_widget_set_vexpand (GET_WIDGET ("extra_widget"), FALSE);
343
344 /* Set the signals handlers. */
345
346 gtk_builder_add_callback_symbols (builder,
347 "on_dialog_destroy", G_CALLBACK (destroy_cb),
348 "on_n_password_entry_changed", G_CALLBACK (password_entry_changed_cb),
349 "on_n_volume_checkbutton_toggled", G_CALLBACK (volume_toggled_cb),
350 NULL);
351 gtk_builder_connect_signals (builder, data);
352
353 g_signal_connect (G_OBJECT (data->format_chooser),
354 "selection-changed",
355 G_CALLBACK (format_chooser_selection_changed_cb),
356 data);
357
358 g_signal_connect_after (GET_WIDGET ("other_oprtions_alignment"),
359 "unmap",
360 G_CALLBACK (options_expander_unmap_cb),
361 data->format_chooser);
362
363 g_object_unref (builder);
364
365 /* Run dialog. */
366
367 update_sensitivity (data);
368
369 gtk_window_set_modal (GTK_WINDOW (data->dialog), TRUE);
370 gtk_window_set_transient_for (GTK_WINDOW (data->dialog), GTK_WINDOW (data->window));
371 /*gtk_window_present (GTK_WINDOW (data->dialog));*/
372
373 return data;
374}
