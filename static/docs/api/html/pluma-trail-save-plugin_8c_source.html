51static void
52strip_trailing_spaces (GtkTextBuffer *text_buffer)
53{
54 gint line_count, line_num;
55 GtkTextIter line_start, line_end;
56 gchar *slice;
57 gchar byte;
58 gint byte_index;
59 gint strip_start_index, strip_end_index;
60 gboolean should_strip;
61 GtkTextIter strip_start, strip_end;
62
63 g_assert (text_buffer != NULL);
64
65 line_count = gtk_text_buffer_get_line_count (text_buffer);
66
67 for (line_num = 0; line_num < line_count; ++line_num)
68 {
69 /* Get line text */
70 gtk_text_buffer_get_iter_at_line (text_buffer, &line_start, line_num);
71
72 if (line_num == line_count - 1)
73 {
74 gtk_text_buffer_get_end_iter (text_buffer, &line_end);
75 }
76 else
77 {
78 gtk_text_buffer_get_iter_at_line (text_buffer, &line_end, line_num + 1);
79 }
80
81 slice = gtk_text_buffer_get_slice (text_buffer, &line_start, &line_end, TRUE);
82
83 if (slice == NULL)
84 {
85 continue;
86 }
87
88 /* Find indices of bytes that should be stripped */
89 should_strip = FALSE;
90
91 for (byte_index = 0; slice [byte_index] != 0; ++byte_index)
92 {
93 byte = slice [byte_index];
94
95 if ((byte == ' ') || (byte == '\t'))
96 {
97 if (!should_strip)
98 {
99 strip_start_index = byte_index;
100 should_strip = TRUE;
101 }
102
103 strip_end_index = byte_index + 1;
104 }
105 else if ((byte == '\r') || (byte == '\n'))
106 {
107 break;
108 }
109 else
110 {
111 should_strip = FALSE;
112 }
113 }
114
115 g_free (slice);
116
117 /* Strip trailing spaces */
118 if (should_strip)
119 {
120 gtk_text_buffer_get_iter_at_line_index (text_buffer, &strip_start, line_num, strip_start_index);
121 gtk_text_buffer_get_iter_at_line_index (text_buffer, &strip_end, line_num, strip_end_index);
122 gtk_text_buffer_delete (text_buffer, &strip_start, &strip_end);
123 }
124 }
125}

161static void
162pluma_trail_save_plugin_activate (PlumaWindowActivatable *activatable)
163{
164 PlumaTrailSavePlugin *plugin;
165 PlumaWindow *window;
166 GList *documents;
167 GList *documents_iter;
168 PlumaDocument *document;
169
170 pluma_debug (DEBUG_PLUGINS);
171
172 plugin = PLUMA_TRAIL_SAVE_PLUGIN (activatable);
173 window = PLUMA_WINDOW (plugin->priv->window);
174
175 g_signal_connect (window, "tab_added", G_CALLBACK (on_tab_added), plugin);
176 g_signal_connect (window, "tab_removed", G_CALLBACK (on_tab_removed), plugin);
177
178 documents = pluma_window_get_documents (window);
179
180 for (documents_iter = documents;
181 documents_iter && documents_iter->data;
182 documents_iter = documents_iter->next)
183 {
184 document = (PlumaDocument *) documents_iter->data;
185 g_signal_connect (document, "save", G_CALLBACK (on_save), plugin);
186 }
187
188 g_list_free (documents);
189}
