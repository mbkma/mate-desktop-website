247EelIRect
248eel_irect_align (EelIRect container,
249 int aligned_width,
250 int aligned_height,
251 float x_alignment,
252 float y_alignment)
253{
254 EelIRect aligned;
255 int available_width;
256 int available_height;
257
258 if (eel_irect_is_empty (&container))
259 {
260 return eel_irect_empty;
261 }
262
263 if (aligned_width == 0 || aligned_height == 0)
264 {
265 return eel_irect_empty;
266 }
267
268 /* Make sure the aligment parameters are within range */
269 x_alignment = MAX (0, x_alignment);
270 x_alignment = MIN (1.0, x_alignment);
271 y_alignment = MAX (0, y_alignment);
272 y_alignment = MIN (1.0, y_alignment);
273
274 available_width = eel_irect_get_width (container) - aligned_width;
275 available_height = eel_irect_get_height (container) - aligned_height;
276
277 aligned.x0 = floor (container.x0 + (available_width * x_alignment) + 0.5);
278 aligned.y0 = floor (container.y0 + (available_height * y_alignment) + 0.5);
279 aligned.x1 = aligned.x0 + aligned_width;
280 aligned.y1 = aligned.y0 + aligned_height;
281
282 return aligned;
283}
