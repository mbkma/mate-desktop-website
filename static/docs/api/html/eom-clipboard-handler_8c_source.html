68static void
69eom_clipboard_handler_set_pixbuf (EomClipboardHandler *handler, GdkPixbuf *pixbuf)
70{
71 g_return_if_fail (EOM_IS_CLIPBOARD_HANDLER (handler));
72 g_return_if_fail (pixbuf == NULL || GDK_IS_PIXBUF (pixbuf));
73
74 if (handler->priv->pixbuf == pixbuf)
75 return;
76
77 if (handler->priv->pixbuf)
78 g_object_unref (handler->priv->pixbuf);
79
80 handler->priv->pixbuf = g_object_ref (pixbuf);
81
82 g_object_notify (G_OBJECT (handler), "pixbuf");
83}

85static void
86eom_clipboard_handler_set_uri (EomClipboardHandler *handler, const gchar *uri)
87{
88 g_return_if_fail (EOM_IS_CLIPBOARD_HANDLER (handler));
89
90 if (handler->priv->uri != NULL)
91 g_free (handler->priv->uri);
92
93 handler->priv->uri = g_strdup (uri);
94 g_object_notify (G_OBJECT (handler), "uri");

97static void
98eom_clipboard_handler_get_property (GObject *object, guint property_id,
99 GValue *value, GParamSpec *pspec)
100{
101 EomClipboardHandler *handler;
102
103 g_return_if_fail (EOM_IS_CLIPBOARD_HANDLER (object));
104
105 handler = EOM_CLIPBOARD_HANDLER (object);
106
107 switch (property_id) {
108 case PROP_PIXBUF:
109 g_value_set_object (value,
110 eom_clipboard_handler_get_pixbuf (handler));
111 break;
112 case PROP_URI:
113 g_value_set_string (value,
114 eom_clipboard_handler_get_uri (handler));
115 break;
116 default:
117 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, property_id, pspec);
118 }
119}

154static void
155eom_clipboard_handler_dispose (GObject *obj)
156{
157 EomClipboardHandlerPrivate *priv;
158
159 g_return_if_fail (EOM_IS_CLIPBOARD_HANDLER (obj));
160
161 priv = EOM_CLIPBOARD_HANDLER (obj)->priv;
162
163 if (priv->pixbuf != NULL) {
164 g_object_unref (priv->pixbuf);
165 priv->pixbuf = NULL;
166 }
167 if (priv->uri) {
168 g_free (priv->uri);
169 priv->uri = NULL;
170 }
171
172 G_OBJECT_CLASS (eom_clipboard_handler_parent_class)->dispose (obj);
173}

204eom_clipboard_handler_new (EomImage *img)
205{
206 GObject *obj;
207 GFile *file;
208 GdkPixbuf *pbuf;
209 gchar *uri;
210
211 g_object_ref (img);
212 pbuf = eom_image_get_pixbuf (img);
213 file = eom_image_get_file (img);
214 uri = g_file_get_uri (file);
215 obj = g_object_new (EOM_TYPE_CLIPBOARD_HANDLER,
216 "pixbuf", pbuf,
217 "uri", uri,
218 NULL);
219 g_free (uri);
220 g_object_unref (file);
221 g_object_unref (pbuf);
222 g_object_unref (img);
223
224 return EOM_CLIPBOARD_HANDLER (obj);

228static void
229eom_clipboard_handler_get_func (GtkClipboard *clipboard,
230 GtkSelectionData *selection,
231 guint info, gpointer owner)
232{
233 EomClipboardHandler *handler;
234
235 g_return_if_fail (EOM_IS_CLIPBOARD_HANDLER (owner));
236
237 handler = EOM_CLIPBOARD_HANDLER (owner);
238
239 switch (info) {
240 case TARGET_PIXBUF:
241 {
242 GdkPixbuf *pixbuf = eom_clipboard_handler_get_pixbuf (handler);
243 g_object_ref (pixbuf);
244 gtk_selection_data_set_pixbuf (selection, pixbuf);
245 g_object_unref (pixbuf);
246 break;
247 }
248 case TARGET_TEXT:
249 {
250 gtk_selection_data_set_text (selection,
251 eom_clipboard_handler_get_uri (handler),
252 -1);
253 break;
254 }
255 case TARGET_URI:
256 {
257 gchar *uris[2];
258 uris[0] = g_strdup (eom_clipboard_handler_get_uri (handler));
259 uris[1] = NULL;
260
261 gtk_selection_data_set_uris (selection, uris);
262 g_free (uris[0]);
263 break;
264 }
265 default:
266 g_return_if_reached ();
267 }

279void
280eom_clipboard_handler_copy_to_clipboard (EomClipboardHandler *handler,
281 GtkClipboard *clipboard)
282{
283 GtkTargetList *tlist;
284 GtkTargetEntry *targets;
285 gint n_targets = 0;
286 gboolean set = FALSE;
287
288 tlist = gtk_target_list_new (NULL, 0);
289
290 if (handler->priv->pixbuf != NULL)
291 gtk_target_list_add_image_targets (tlist, TARGET_PIXBUF, TRUE);
292
293 if (handler->priv->uri != NULL) {
294 gtk_target_list_add_text_targets (tlist, TARGET_TEXT);
295 gtk_target_list_add_uri_targets (tlist, TARGET_URI);
296 }
297
298 targets = gtk_target_table_new_from_list (tlist, &n_targets);
299
300 // We need to take ownership here if nobody else did
301 g_object_ref_sink (handler);
302
303 if (n_targets > 0) {
304 set = gtk_clipboard_set_with_owner (clipboard,
305 targets, n_targets,
306 eom_clipboard_handler_get_func,
307 eom_clipboard_handler_clear_func,
308 G_OBJECT (handler));
309
310 }
311
312 if (!set) {
313 gtk_clipboard_clear (clipboard);
314 g_object_unref (handler);
315 }
316
317 gtk_target_table_free (targets, n_targets);
318 gtk_target_list_unref (tlist);
319}
