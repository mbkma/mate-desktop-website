40void
41caja_recent_add_file (CajaFile *file,
42 GAppInfo *application)
43{
44 GtkRecentData recent_data;
45 char *uri;
46
47 uri = caja_file_get_uri (file);
48
49 /* do not add trash:// etc */
50 if (eel_uri_is_trash (uri) ||
51 eel_uri_is_search (uri) ||
52 eel_uri_is_desktop (uri))
53 {
54 g_free (uri);
55 return;
56 }
57
58 recent_data.display_name = NULL;
59 recent_data.description = NULL;
60
61 recent_data.mime_type = caja_file_get_mime_type (file);
62 recent_data.app_name = g_strdup (g_get_application_name ());
63
64 if (application != NULL)
65 recent_data.app_exec = g_strdup (g_app_info_get_commandline (application));
66 else
67 recent_data.app_exec = g_strdup (DEFAULT_APP_EXEC);
68
69 recent_data.groups = NULL;
70 recent_data.is_private = FALSE;
71
72 gtk_recent_manager_add_full (caja_recent_get_manager (),
73 uri, &recent_data);
74
75 g_free (recent_data.mime_type);
76 g_free (recent_data.app_name);
77 g_free (recent_data.app_exec);
78
79 g_free (uri);
80}
