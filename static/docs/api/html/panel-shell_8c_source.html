71gboolean
72panel_shell_register (gboolean replace)
73{
74 GBusNameOwnerFlags flags;
75 guint32 request_name_reply;
76 GVariant *result;
77 gboolean retval = FALSE;
78 GError *error = NULL;
79
80 if (dbus_connection != NULL)
81 return TRUE;
82
83 panel_cleanup_register (PANEL_CLEAN_FUNC (panel_shell_cleanup), NULL);
84
85 /* There isn't a sync version of g_bus_own_name, so we have to requeste
86 * the name manually here. There's no ui yet so it's safe to use this
87 * sync api */
88 dbus_connection = g_bus_get_sync (G_BUS_TYPE_SESSION, NULL, &error);
89 if (dbus_connection == NULL) {
90 g_warning ("Cannot register the panel shell: %s", error->message);
91 goto register_out;
92 }
93
94 flags = G_BUS_NAME_OWNER_FLAGS_ALLOW_REPLACEMENT;
95 if (replace)
96 flags |= G_BUS_NAME_OWNER_FLAGS_REPLACE;
97
98 result = g_dbus_connection_call_sync (dbus_connection,
99 "org.freedesktop.DBus",
100 "/org/freedesktop/DBus",
101 "org.freedesktop.DBus",
102 "RequestName",
103 g_variant_new ("(su)",
104 PANEL_DBUS_SERVICE,
105 flags),
106 G_VARIANT_TYPE ("(u)"),
107 G_DBUS_CALL_FLAGS_NONE,
108 -1, NULL, &error);
109 if (!result) {
110 g_warning ("Cannot register the panel shell: %s",
111 error->message);
112 g_error_free (error);
113
114 goto register_out;
115 }
116
117 g_variant_get (result, "(u)", &request_name_reply);
118 g_variant_unref (result);
119
120 switch (request_name_reply) {
121 case 1: /* DBUS_REQUEST_NAME_REPLY_PRIMARY_OWNER */
122 case 4: /* DBUS_REQUEST_NAME_REPLY_ALREADY_OWNER */
123 retval = TRUE;
124 g_dbus_connection_signal_subscribe (dbus_connection,
125 "org.freedesktop.DBus",
126 "org.freedesktop.DBus",
127 "NameLost",
128 "/org/freedesktop/DBus",
129 PANEL_DBUS_SERVICE,
130 G_DBUS_SIGNAL_FLAGS_NONE,
131 (GDBusSignalCallback)panel_shell_on_name_lost,
132 NULL, NULL);
133 break;
134 case 2: /* DBUS_REQUEST_NAME_REPLY_IN_QUEUE */
135 case 3: /* DBUS_REQUEST_NAME_REPLY_EXISTS */
136 g_printerr ("Cannot register the panel shell: there is "
137 "already one running.\n");
138 break;
139 default:
140 g_warning ("Cannot register the panel shell: unhandled "
141 "reply %u from RequestName", request_name_reply);
142 }
143
144register_out:
145
146 if (!retval) {
147 panel_session_do_not_restart ();
148 panel_shell_cleanup (NULL);
149 }
150
151 return retval;
152}
