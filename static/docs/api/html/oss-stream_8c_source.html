126void
127oss_stream_load (OssStream *stream)
128{
129 GList *list;
130
131 g_return_if_fail (OSS_IS_STREAM (stream));
132
133 list = stream->priv->controls;
134 while (list != NULL) {
135 OssStreamControl *control = OSS_STREAM_CONTROL (list->data);
136
137 oss_stream_control_load (control);
138 list = list->next;
139 }
140
141 if (stream->priv->swtch != NULL)
142 oss_switch_load (stream->priv->swtch);
143}

194void
195oss_stream_set_switch_data (OssStream *stream, gint fd, GList *options)
196{
197 g_return_if_fail (OSS_IS_STREAM (stream));
198 g_return_if_fail (fd != -1);
199 g_return_if_fail (options != NULL);
200
201 /* Function may only be called once */
202 if (G_UNLIKELY (stream->priv->swtch != NULL)) {
203 g_warn_if_reached ();
204 return;
205 }
206
207 /* Takes ownership of options */
208 stream->priv->swtch = oss_switch_new (stream,
209 OSS_STREAM_SWITCH_NAME,
210 _("Connector"),
211 fd,
212 options);
213 if (G_UNLIKELY (stream->priv->swtch == NULL))
214 return;
215
216 /* Read the active selection */
217 oss_switch_load (stream->priv->swtch);
218
219 stream->priv->switches = g_list_prepend (NULL, g_object_ref (stream->priv->swtch));
220 g_signal_emit_by_name (G_OBJECT (stream),
221 "switch-added",
222 OSS_STREAM_SWITCH_NAME);
223}

225void
226oss_stream_remove_all (OssStream *stream)
227{
228 GList *list;
229
230 g_return_if_fail (OSS_IS_STREAM (stream));
231
232 list = stream->priv->controls;
233 while (list != NULL) {
234 MateMixerStreamControl *control = MATE_MIXER_STREAM_CONTROL (list->data);
235 GList *next = list->next;
236
237 oss_stream_control_close (OSS_STREAM_CONTROL (control));
238
239 stream->priv->controls = g_list_delete_link (stream->priv->controls, list);
240 g_signal_emit_by_name (G_OBJECT (stream),
241 "control-removed",
242 mate_mixer_stream_control_get_name (control));
243
244 g_object_unref (control);
245 list = next;
246 }
247
248 /* Unset the default stream control */
249 oss_stream_set_default_control (stream, NULL);
250
251 if (stream->priv->swtch != NULL) {
252 oss_switch_close (stream->priv->swtch);
253
254 g_list_free_full (stream->priv->switches, g_object_unref);
255 stream->priv->switches = NULL;
256
257 g_signal_emit_by_name (G_OBJECT (stream),
258 "switch-removed",
259 OSS_STREAM_SWITCH_NAME);
260
261 g_clear_object (&stream->priv->swtch);
262 }
263}
