58void
59meta_group_reload_properties (MetaGroup *group,
60 const Atom *properties,
61 int n_properties)
62{
63 int i;
64 MetaPropValue *values;
65
66 g_return_if_fail (properties != NULL);
67 g_return_if_fail (n_properties > 0);
68
69 values = g_new0 (MetaPropValue, n_properties);
70
71 i = 0;
72 while (i < n_properties)
73 {
74 init_prop_value (group->display, properties[i], &values[i]);
75 ++i;
76 }
77
78 meta_prop_get_values (group->display, group->group_leader,
79 values, n_properties);
80
81 i = 0;
82 while (i < n_properties)
83 {
84 reload_prop_value (group, &values[i]);
85
86 ++i;
87 }
88
89 meta_prop_free_values (values, n_properties);
90
91 g_free (values);
92}

95static void
96init_prop_value (MetaDisplay *display,
97 Atom property,
98 MetaPropValue *value)
99{
100 MetaGroupPropHooks *hooks;
101
102 value->type = META_PROP_VALUE_INVALID;
103 value->atom = None;
104
105 hooks = find_hooks (display, property);
106 if (hooks && hooks->init_func != NULL)
107 (* hooks->init_func) (display, property, value);
108}
109
110static void
111reload_prop_value (MetaGroup *group,
112 MetaPropValue *value)
113{
114 MetaGroupPropHooks *hooks;
115
116 hooks = find_hooks (group->display, value->atom);
117 if (hooks && hooks->reload_func != NULL)
118 (* hooks->reload_func) (group, value);
119}

133{

138 group->wm_client_machine = g_strdup (value->v.str);

140 meta_verbose ("Group has client machine \"%s\"\n",

167#define N_HOOKS 3
168
169void
170meta_display_init_group_prop_hooks (MetaDisplay *display)
171{
172 int i;
173 MetaGroupPropHooks *hooks;
174
175 g_assert (display->group_prop_hooks == NULL);
176
177 display->group_prop_hooks = g_new0 (MetaGroupPropHooks, N_HOOKS);
178 hooks = display->group_prop_hooks;
179
180 i = 0;
181
182 hooks[i].property = display->atom_WM_CLIENT_MACHINE;
183 hooks[i].init_func = init_wm_client_machine;
184 hooks[i].reload_func = reload_wm_client_machine;
185 ++i;
186
187 hooks[i].property = display->atom__NET_WM_PID;
188 hooks[i].init_func = NULL;
189 hooks[i].reload_func = NULL;
190 ++i;
191
192 hooks[i].property = display->atom__NET_STARTUP_ID;
193 hooks[i].init_func = init_net_startup_id;
194 hooks[i].reload_func = reload_net_startup_id;
195 ++i;
196
197 if (i != N_HOOKS)
198 {
199 g_error ("Initialized %d group hooks should have been %d\n", i, N_HOOKS);
200 }
201}

212static MetaGroupPropHooks*

214 Atom property)
215{
216 int i;
217
218 /* FIXME we could sort the array and do binary search or
219 * something

223 while (i < N_HOOKS)
224 {
225 if (display->group_prop_hooks[i].property == property)
226 return &display->group_prop_hooks[i];
227
228 ++i;
229 }
230
231 return NULL;
