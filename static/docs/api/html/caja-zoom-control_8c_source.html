156/* handle button presses */
157static gboolean
158caja_zoom_control_button_press_event (GtkWidget *widget,
159 GdkEventButton *event,
160 CajaZoomControl *zoom_control)
161{
162 if (event->type != GDK_BUTTON_PRESS)
163 {
164 return FALSE;
165 }
166
167 /* check for the context menu button and show the menu */
168 if (event->button == CONTEXTUAL_MENU_BUTTON)
169 {
170 zoom_popup_menu_show (widget, event, zoom_control);
171 return TRUE;
172 }
173 /* We don't change our state (to reflect the new zoom) here.
174 The zoomable will call back with the new level.
175 Actually, the callback goes to the viewframe containing the
176 zoomable which, in turn, emits zoom_level_changed,
177 which someone (e.g. caja_window) picks up and handles by
178 calling into is - caja_zoom_control_set_zoom_level.
179 */
180
181 return FALSE;
182}

229static void
230caja_zoom_control_init (CajaZoomControl *zoom_control)
231{
232 GtkWidget *image;
233 int i;
234
235 zoom_control->details = caja_zoom_control_get_instance_private (zoom_control);
236
237 zoom_control->details->zoom_level = CAJA_ZOOM_LEVEL_STANDARD;
238 zoom_control->details->min_zoom_level = CAJA_ZOOM_LEVEL_SMALLEST;
239 zoom_control->details->max_zoom_level = CAJA_ZOOM_LEVEL_LARGEST;
240 zoom_control->details->has_min_zoom_level = TRUE;
241 zoom_control->details->has_max_zoom_level = TRUE;
242
243 for (i = CAJA_ZOOM_LEVEL_LARGEST; i >= CAJA_ZOOM_LEVEL_SMALLEST; i--)
244 {
245 zoom_control->details->preferred_zoom_levels = g_list_prepend (
246 zoom_control->details->preferred_zoom_levels,
247 GINT_TO_POINTER (i));
248 }
249
250 image = gtk_image_new_from_icon_name ("zoom-out", GTK_ICON_SIZE_MENU);
251 zoom_control->details->zoom_out = gtk_button_new ();
252 gtk_widget_set_focus_on_click (zoom_control->details->zoom_out, FALSE);
253 gtk_button_set_relief (GTK_BUTTON (zoom_control->details->zoom_out),
254 GTK_RELIEF_NONE);
255 gtk_widget_set_tooltip_text (zoom_control->details->zoom_out,
256 _("Decrease the view size"));
257 g_signal_connect (G_OBJECT (zoom_control->details->zoom_out),
258 "clicked", G_CALLBACK (zoom_out_clicked),
259 zoom_control);
260
261 gtk_orientable_set_orientation (GTK_ORIENTABLE (zoom_control), GTK_ORIENTATION_HORIZONTAL);
262
263 gtk_container_add (GTK_CONTAINER (zoom_control->details->zoom_out), image);
264 gtk_box_pack_start (GTK_BOX (zoom_control),
265 zoom_control->details->zoom_out, FALSE, FALSE, 0);
266
267 zoom_control->details->zoom_button = gtk_button_new ();
268 gtk_widget_set_focus_on_click (zoom_control->details->zoom_button, FALSE);
269 gtk_button_set_relief (GTK_BUTTON (zoom_control->details->zoom_button),
270 GTK_RELIEF_NONE);
271 gtk_widget_set_tooltip_text (zoom_control->details->zoom_button,
272 _("Use the normal view size"));
273
274 gtk_widget_add_events (GTK_WIDGET (zoom_control->details->zoom_button),
275 GDK_BUTTON_PRESS_MASK
276 | GDK_BUTTON_RELEASE_MASK
277 | GDK_POINTER_MOTION_MASK
278 | GDK_SCROLL_MASK);
279
280 g_signal_connect (G_OBJECT (zoom_control->details->zoom_button),
281 "button-press-event",
282 G_CALLBACK (caja_zoom_control_button_press_event),
283 zoom_control);
284
285 g_signal_connect (G_OBJECT (zoom_control->details->zoom_button),
286 "clicked", G_CALLBACK (zoom_button_clicked),
287 zoom_control);
288
289 g_signal_connect (G_OBJECT (zoom_control->details->zoom_button),
290 "popup-menu", G_CALLBACK (zoom_popup_menu),
291 zoom_control);
292
293 zoom_control->details->zoom_label = gtk_label_new ("100%");
294 g_signal_connect (zoom_control->details->zoom_label,
295 "style_set",
296 G_CALLBACK (label_style_set_callback),
297 zoom_control);
298 set_label_size (zoom_control);
299
300 gtk_container_add (GTK_CONTAINER (zoom_control->details->zoom_button), zoom_control->details->zoom_label);
301
302 gtk_box_pack_start (GTK_BOX (zoom_control),
303 zoom_control->details->zoom_button, TRUE, TRUE, 0);
304
305 image = gtk_image_new_from_icon_name ("zoom-in", GTK_ICON_SIZE_MENU);
306 zoom_control->details->zoom_in = gtk_button_new ();
307 gtk_widget_set_focus_on_click (zoom_control->details->zoom_in, FALSE);
308 gtk_button_set_relief (GTK_BUTTON (zoom_control->details->zoom_in),
309 GTK_RELIEF_NONE);
310 gtk_widget_set_tooltip_text (zoom_control->details->zoom_in,
311 _("Increase the view size"));
312 g_signal_connect (G_OBJECT (zoom_control->details->zoom_in),
313 "clicked", G_CALLBACK (zoom_in_clicked),
314 zoom_control);
315
316 gtk_container_add (GTK_CONTAINER (zoom_control->details->zoom_in), image);
317 gtk_box_pack_start (GTK_BOX (zoom_control),
318 zoom_control->details->zoom_in, FALSE, FALSE, 0);
319
320 gtk_widget_show_all (zoom_control->details->zoom_out);
321 gtk_widget_show_all (zoom_control->details->zoom_button);
322 gtk_widget_show_all (zoom_control->details->zoom_in);
323}

332static void
333caja_zoom_control_redraw (CajaZoomControl *zoom_control)
334{
335 int percent;
336 char *num_str;
337
338 gtk_widget_set_sensitive (zoom_control->details->zoom_in,
339 caja_zoom_control_can_zoom_in (zoom_control));
340 gtk_widget_set_sensitive (zoom_control->details->zoom_out,
341 caja_zoom_control_can_zoom_out (zoom_control));
342
343 percent = floor ((100.0 * caja_get_relative_icon_size_for_zoom_level (zoom_control->details->zoom_level)) + .2);
344 num_str = g_strdup_printf ("%d%%", percent);
345 gtk_label_set_text (GTK_LABEL (zoom_control->details->zoom_label), num_str);
346 g_free (num_str);
347}

351static void
352zoom_menu_callback (GtkMenuItem *item, gpointer callback_data)
353{
354 CajaZoomLevel zoom_level;
355 CajaZoomControl *zoom_control;
356 gboolean can_zoom;
357
358 zoom_control = CAJA_ZOOM_CONTROL (callback_data);
359
360 /* Don't do anything if we're just setting the toggle state of menu items. */
361 if (zoom_control->details->marking_menu_items)
362 {
363 return;
364 }
365
366 /* Don't send the signal if the menuitem was toggled off */
367 if (!gtk_check_menu_item_get_active (GTK_CHECK_MENU_ITEM (item)))
368 {
369 return;
370 }
371
372 zoom_level = (CajaZoomLevel) GPOINTER_TO_INT (g_object_get_data (G_OBJECT (item), "zoom_level"));
373
374 /* Assume we can zoom and then check whether we're right. */
375 can_zoom = TRUE;
376 if (zoom_control->details->has_min_zoom_level &&
377 zoom_level < zoom_control->details->min_zoom_level)
378 can_zoom = FALSE; /* no, we're below the minimum zoom level. */
379 if (zoom_control->details->has_max_zoom_level &&
380 zoom_level > zoom_control->details->max_zoom_level)
381 can_zoom = FALSE; /* no, we're beyond the upper zoom level. */
382
383 /* if we can zoom */
384 if (can_zoom)
385 {
386 g_signal_emit (zoom_control, signals[ZOOM_TO_LEVEL], 0, zoom_level);
387 }
388}

390static GtkRadioMenuItem *
391create_zoom_menu_item (CajaZoomControl *zoom_control, GtkMenu *menu,
392 CajaZoomLevel zoom_level,
393 GtkRadioMenuItem *previous_radio_item)
394{
395 GtkWidget *menu_item;
396 char *item_text;
397 GSList *radio_item_group;
398 int percent;
399
400 /* Set flag so that callback isn't activated when set_active called
401 * to set toggle state of other radio items.
402 */
403 zoom_control->details->marking_menu_items = TRUE;
404
405 percent = floor ((100.0 * caja_get_relative_icon_size_for_zoom_level (zoom_level)) + .5);
406 item_text = g_strdup_printf ("%d%%", percent);
407
408 radio_item_group = previous_radio_item == NULL
409 ? NULL
410 : gtk_radio_menu_item_get_group (previous_radio_item);
411 menu_item = gtk_radio_menu_item_new_with_label (radio_item_group, item_text);
412 g_free (item_text);
413
414 gtk_check_menu_item_set_active (GTK_CHECK_MENU_ITEM (menu_item),
415 zoom_level == zoom_control->details->zoom_level);
416
417 g_object_set_data (G_OBJECT (menu_item), "zoom_level", GINT_TO_POINTER (zoom_level));
418 g_signal_connect_object (menu_item, "activate",
419 G_CALLBACK (zoom_menu_callback), zoom_control, 0);
420
421 gtk_widget_show (menu_item);
422 gtk_menu_shell_append (GTK_MENU_SHELL (menu), menu_item);
423
424 zoom_control->details->marking_menu_items = FALSE;
425
426 return GTK_RADIO_MENU_ITEM (menu_item);
427}

429static GtkMenu *
430create_zoom_menu (CajaZoomControl *zoom_control)
431{
432 GtkMenu *menu;
433 GtkRadioMenuItem *previous_item;
434 GList *node;
435
436 menu = GTK_MENU (gtk_menu_new ());
437
438 previous_item = NULL;
439 for (node = zoom_control->details->preferred_zoom_levels; node != NULL; node = node->next)
440 {
441 previous_item = create_zoom_menu_item
442 (zoom_control, menu, GPOINTER_TO_INT (node->data), previous_item);
443 }
444
445 return menu;
446}

448static void
449caja_zoom_control_change_value (CajaZoomControl *zoom_control,
450 GtkScrollType scroll)
451{
452 switch (scroll)
453 {
454 case GTK_SCROLL_STEP_DOWN :
455 if (caja_zoom_control_can_zoom_out (zoom_control))
456 {
457 g_signal_emit (zoom_control, signals[ZOOM_OUT], 0);
458 }
459 break;
460 case GTK_SCROLL_STEP_UP :
461 if (caja_zoom_control_can_zoom_in (zoom_control))
462 {
463 g_signal_emit (zoom_control, signals[ZOOM_IN], 0);
464 }
465 break;
466 default :
467 g_warning ("Invalid scroll type %d for CajaZoomControl:change_value", scroll);
468 }
469}

546static gboolean
547caja_zoom_control_scroll_event (GtkWidget *widget, GdkEventScroll *event)
548{
549 CajaZoomControl *zoom_control;
550
551 zoom_control = CAJA_ZOOM_CONTROL (widget);
552
553 if (event->type != GDK_SCROLL)
554 {
555 return FALSE;
556 }
557
558 if (event->direction == GDK_SCROLL_DOWN &&
559 caja_zoom_control_can_zoom_out (zoom_control))
560 {
561 g_signal_emit (widget, signals[ZOOM_OUT], 0);
562 }
563 else if (event->direction == GDK_SCROLL_UP &&
564 caja_zoom_control_can_zoom_in (zoom_control))
565 {
566 g_signal_emit (widget, signals[ZOOM_IN], 0);
567 }
568
569 /* We don't change our state (to reflect the new zoom) here. The zoomable will
570 * call back with the new level. Actually, the callback goes to the view-frame
571 * containing the zoomable which, in turn, emits zoom_level_changed, which
572 * someone (e.g. caja_window) picks up and handles by calling into us -
573 * caja_zoom_control_set_zoom_level.
574 */
575 return TRUE;
576}

580{

583

585
586 widget_class = GTK_WIDGET_CLASS (class);

592
593 class->change_value = caja_zoom_control_change_value;
594
595 signals[ZOOM_IN] =
596 g_signal_new ("zoom_in",
597 G_TYPE_FROM_CLASS (class),
598 G_SIGNAL_RUN_LAST,
599 G_STRUCT_OFFSET (CajaZoomControlClass,
600 zoom_in),
601 NULL, NULL,
602 g_cclosure_marshal_VOID__VOID,
603 G_TYPE_NONE, 0);
604
605 signals[ZOOM_OUT] =
606 g_signal_new ("zoom_out",
607 G_TYPE_FROM_CLASS (class),
608 G_SIGNAL_RUN_LAST,
609 G_STRUCT_OFFSET (CajaZoomControlClass,
610 zoom_out),
611 NULL, NULL,
612 g_cclosure_marshal_VOID__VOID,
613 G_TYPE_NONE, 0);
614
615 signals[ZOOM_TO_LEVEL] =
616 g_signal_new ("zoom_to_level",
617 G_TYPE_FROM_CLASS (class),
618 G_SIGNAL_RUN_LAST,
619 G_STRUCT_OFFSET (CajaZoomControlClass,
620 zoom_to_level),
621 NULL, NULL,
622 g_cclosure_marshal_VOID__INT,
623 G_TYPE_NONE,
624 1,
625 G_TYPE_INT);
626
627 signals[ZOOM_TO_DEFAULT] =
628 g_signal_new ("zoom_to_default",
629 G_TYPE_FROM_CLASS (class),
630 G_SIGNAL_RUN_LAST | G_SIGNAL_ACTION,
631 G_STRUCT_OFFSET (CajaZoomControlClass,
632 zoom_to_default),
633 NULL, NULL,
634 g_cclosure_marshal_VOID__VOID,
635 G_TYPE_NONE, 0);
636
637 signals[CHANGE_VALUE] =
638 g_signal_new ("change_value",
639 G_TYPE_FROM_CLASS (class),
640 G_SIGNAL_RUN_LAST | G_SIGNAL_ACTION,
641 G_STRUCT_OFFSET (CajaZoomControlClass,
642 change_value),
643 NULL, NULL,
644 g_cclosure_marshal_VOID__ENUM,
645 G_TYPE_NONE, 1, GTK_TYPE_SCROLL_TYPE);
646
647 binding_set = gtk_binding_set_by_class (class);
648
649 gtk_binding_entry_add_signal (binding_set,
650 GDK_KEY_KP_Subtract, 0,
651 "change_value",
652 1, GTK_TYPE_SCROLL_TYPE,
653 GTK_SCROLL_STEP_DOWN);
654 gtk_binding_entry_add_signal (binding_set,
655 GDK_KEY_minus, 0,
656 "change_value",
657 1, GTK_TYPE_SCROLL_TYPE,
658 GTK_SCROLL_STEP_DOWN);
659
660 gtk_binding_entry_add_signal (binding_set,
661 GDK_KEY_KP_Equal, 0,
662 "zoom_to_default",
663 0);
664 gtk_binding_entry_add_signal (binding_set,
665 GDK_KEY_KP_Equal, 0,
666 "zoom_to_default",
667 0);
668
669 gtk_binding_entry_add_signal (binding_set,
670 GDK_KEY_KP_Add, 0,
671 "change_value",
672 1, GTK_TYPE_SCROLL_TYPE,
673 GTK_SCROLL_STEP_UP);
674 gtk_binding_entry_add_signal (binding_set,
675 GDK_KEY_plus, 0,

784static CajaZoomLevel
785nearest_preferred (CajaZoomControl *zoom_control, CajaZoomLevel value)
786{
787 CajaZoomLevel last_value;
788 CajaZoomLevel current_value;
789 GList *l;
790
791 if (!zoom_control->details->preferred_zoom_levels)
792 {
793 return value;
794 }
795
796 last_value = GPOINTER_TO_INT (zoom_control->details->preferred_zoom_levels->data);
797 current_value = last_value;
798
799 for (l = zoom_control->details->preferred_zoom_levels; l != NULL; l = l->next)
800 {
801 current_value = GPOINTER_TO_INT (l->data);
802
803 if (current_value > value)
804 {
805 float center = (last_value + current_value) / 2;
806
807 return (value < center) ? last_value : current_value;
808
809 }
810
811 last_value = current_value;
812 }
813
814 return current_value;
815}
