86void
87ev_history_add_link (EvHistory *history, EvLink *link)
88{
89 GList *l;
90
91 g_return_if_fail (EV_IS_HISTORY (history));
92 g_return_if_fail (EV_IS_LINK (link));
93
94 for (l = history->priv->links; l; l = l->next) {
95 if (!strcmp (ev_link_get_title (EV_LINK (l->data)), ev_link_get_title (link))) {
96 g_object_unref (G_OBJECT (l->data));
97 history->priv->links = g_list_delete_link (history->priv->links, l);
98 break;
99 }
100 }
101
102 g_object_ref (link);
103 history->priv->links = g_list_append (history->priv->links,
104 link);
105
106 if (g_list_length (history->priv->links) > HISTORY_LENGTH) {
107 g_object_unref (G_OBJECT (history->priv->links->data));
108 history->priv->links = g_list_delete_link (history->priv->links,
109 history->priv->links);
110 }
111
112 g_signal_emit (history, signals[HISTORY_CHANGED], 0);
113}
