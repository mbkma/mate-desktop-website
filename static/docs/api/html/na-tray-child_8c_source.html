62static void
63na_tray_child_realize (GtkWidget *widget)
64{
65 NaTrayChild *child = NA_TRAY_CHILD (widget);
66 GdkVisual *visual = gtk_widget_get_visual (widget);
67 GdkWindow *window;
68
69 GTK_WIDGET_CLASS (na_tray_child_parent_class)->realize (widget);
70
71 window = gtk_widget_get_window (widget);
72
73 if (child->has_alpha)
74 {
75 /* We have real transparency with an ARGB visual and the Composite
76 * extension. */
77
78 /* Set a transparent background */
79 cairo_pattern_t *transparent = cairo_pattern_create_rgba (0, 0, 0, 0);
80 gdk_window_set_background_pattern (window, transparent);
81 gdk_window_set_composited (window, TRUE);
82 cairo_pattern_destroy (transparent);
83
84 child->parent_relative_bg = FALSE;
85 }
86 else if (visual == gdk_window_get_visual(gdk_window_get_parent(window)))
87 {
88 /* Otherwise, if the visual matches the visual of the parent window, we
89 * can use a parent-relative background and fake transparency. */
90 gdk_window_set_background_pattern (window, NULL);
91
92 child->parent_relative_bg = TRUE;
93 }
94 else
95 {
96 /* Nothing to do; the icon will sit on top of an ugly gray box */
97 child->parent_relative_bg = FALSE;
98 }
99
100 gdk_window_set_composited (window, child->composited);
101
102 gtk_widget_set_app_paintable (GTK_WIDGET (child),
103 child->parent_relative_bg || child->has_alpha);
104}

160/* The plug window should completely occupy the area of the child, so we won't
161 * get an expose event. But in case we do (the plug unmaps itself, say), this
162 * expose handler draws with real or fake transparency.
163 */
164static gboolean
165na_tray_child_draw (GtkWidget *widget,
166 cairo_t *cr)
167{
168 NaTrayChild *child = NA_TRAY_CHILD (widget);
169
170 if (na_tray_child_has_alpha (child))
171 {
172 /* Clear to transparent */
173 cairo_set_source_rgba (cr, 0, 0, 0, 0);
174 cairo_set_operator (cr, CAIRO_OPERATOR_SOURCE);
175 cairo_paint (cr);
176 }
177 else if (child->parent_relative_bg)
178 {
179 /* Clear to parent-relative pixmap */
180 GdkWindow *window;
181 cairo_surface_t *target;
182 GdkRectangle clip_rect;
183
184 window = gtk_widget_get_window (widget);
185 target = cairo_get_group_target (cr);
186
187 gdk_cairo_get_clip_rectangle (cr, &clip_rect);
188
189 /* Clear to parent-relative pixmap
190 * We need to use direct X access here because GDK doesn't know about
191 * the parent relative pixmap. */
192 cairo_surface_flush (target);
193
194 XClearArea (GDK_WINDOW_XDISPLAY (window),
195 GDK_WINDOW_XID (window),
196 clip_rect.x, clip_rect.y,
197 clip_rect.width, clip_rect.height,
198 False);
199 cairo_surface_mark_dirty_rectangle (target,
200 clip_rect.x, clip_rect.y,
201 clip_rect.width, clip_rect.height);
202 }
203
204 return FALSE;
205}

207/* Children with alpha channels have been set to be composited by calling
208 * gdk_window_set_composited(). We need to paint these children ourselves.
209 *
210 * FIXME: is that still needed on GTK3? Seems like it could be done in draw().
211 */
212static gboolean
213na_tray_child_draw_on_parent (NaItem *item,
214 GtkWidget *parent,
215 cairo_t *parent_cr)
216{
217 if (na_tray_child_has_alpha (NA_TRAY_CHILD (item)))
218 {
219 GtkWidget *widget = GTK_WIDGET (item);
220 GtkAllocation parent_allocation = { 0 };
221 GtkAllocation allocation;
222
223 /* if the parent doesn't have a window, our allocation is not relative to
224 * the context coordinates but to the parent's allocation */
225 if (! gtk_widget_get_has_window (parent))
226 gtk_widget_get_allocation (parent, &parent_allocation);
227
228 gtk_widget_get_allocation (widget, &allocation);
229 allocation.x -= parent_allocation.x;
230 allocation.y -= parent_allocation.y;
231
232 cairo_save (parent_cr);
233 gdk_cairo_set_source_window (parent_cr,
234 gtk_widget_get_window (widget),
235 allocation.x,
236 allocation.y);
237 cairo_rectangle (parent_cr, allocation.x, allocation.y, allocation.width, allocation.height);
238 cairo_clip (parent_cr);
239 cairo_paint (parent_cr);
240 cairo_restore (parent_cr);
241 }
242
243 return TRUE;
244}

283/* Hack to keep order of some known system-tray elements. For a wm_class
284 * match, give it category @category and ID @id.
285 *
286 * TODO: improve this to play well if one of those elements were to start
287 * using SNI instead */
288static const struct
289{
290 const gchar *const wm_class;
291 const gchar *const id;
292 NaItemCategory category;
293} wmclass_categories[] = {
294 /* order is LTR, so higher category and higher ASCII ordering on the right */
295 { "keyboard", "~01-keyboard", NA_ITEM_CATEGORY_HARDWARE },
296 { "Mate-volume-control-applet", "~02-volume", NA_ITEM_CATEGORY_HARDWARE },
297 { "Bluetooth-applet", "~03-bluetooth", NA_ITEM_CATEGORY_HARDWARE },
298 { "Nm-applet", "~04-network", NA_ITEM_CATEGORY_HARDWARE },
299 { "Mate-power-manager", "~05-battery", NA_ITEM_CATEGORY_HARDWARE },
300};

302static const gchar *
303na_tray_child_get_id (NaItem *item)
304{
305 NaTrayChild *child = NA_TRAY_CHILD (item);
306
307 if (! child->id)
308 {
309 char *res_name = NULL;
310 char *res_class = NULL;
311 guint i;
312
313 na_tray_child_get_wm_class (child, &res_name, &res_class);
314
315 for (i = 0; i < G_N_ELEMENTS (wmclass_categories) && ! child->id; i++)
316 {
317 if (g_strcmp0 (res_class, wmclass_categories[i].wm_class) == 0)
318 child->id = g_strdup (wmclass_categories[i].id);
319 }
320
321 if (! child->id)
322 child->id = res_name;
323 else
324 g_free (res_name);
325
326 g_free (res_class);
327 }
328
329 return child->id;
330}

332static NaItemCategory
333na_tray_child_get_category (NaItem *item)
334{
335 guint i;
336 NaItemCategory category = NA_ITEM_CATEGORY_APPLICATION_STATUS;
337 char *res_class = NULL;
338
339 na_tray_child_get_wm_class (NA_TRAY_CHILD (item), NULL, &res_class);
340
341 for (i = 0; i < G_N_ELEMENTS (wmclass_categories); i++)
342 {
343 if (g_strcmp0 (res_class, wmclass_categories[i].wm_class) == 0)
344 {
345 category = wmclass_categories[i].category;
346 break;
347 }
348 }
349
350 g_free (res_class);
351
352 return category;
353}

395GtkWidget *
396na_tray_child_new (GdkScreen *screen,
397 Window icon_window)
398{
399 XWindowAttributes window_attributes;
400 Display *xdisplay;
401 GdkDisplay *display;
402 NaTrayChild *child;
403 GdkVisual *visual;
404 gboolean visual_has_alpha;
405 int red_prec, green_prec, blue_prec, depth;
406 int result;
407
408 g_return_val_if_fail (GDK_IS_SCREEN (screen), NULL);
409 g_return_val_if_fail (icon_window != None, NULL);
410
411 xdisplay = GDK_SCREEN_XDISPLAY (screen);
412
413 /* We need to determine the visual of the window we are embedding and create
414 * the socket in the same visual.
415 */
416
417 display = gdk_screen_get_display (screen);
418 if (!GDK_IS_X11_DISPLAY (display)) {
419 g_warning ("na_tray only works on X11");
420 return NULL;
421 }
422 gdk_x11_display_error_trap_push (display);
423 result = XGetWindowAttributes (xdisplay, icon_window,
424 &window_attributes);
425 gdk_x11_display_error_trap_pop_ignored (display);
426
427 if (!result) /* Window already gone */
428 return NULL;
429
430 visual = gdk_x11_screen_lookup_visual (screen,
431 window_attributes.visual->visualid);
432 if (!visual) /* Icon window is on another screen? */
433 return NULL;
434
435 child = g_object_new (NA_TYPE_TRAY_CHILD, NULL);
436 child->icon_window = icon_window;
437
438 gtk_widget_set_visual (GTK_WIDGET (child), visual);
439
440 /* We have alpha if the visual has something other than red, green,
441 * and blue */
442 gdk_visual_get_red_pixel_details (visual, NULL, NULL, &red_prec);
443 gdk_visual_get_green_pixel_details (visual, NULL, NULL, &green_prec);
444 gdk_visual_get_blue_pixel_details (visual, NULL, NULL, &blue_prec);
445 depth = gdk_visual_get_depth (visual);
446
447 visual_has_alpha = red_prec + blue_prec + green_prec < depth;
448 child->has_alpha = (visual_has_alpha &&
449 gdk_display_supports_composite (gdk_screen_get_display (screen)));
450
451 child->composited = child->has_alpha;
452
453 return GTK_WIDGET (child);
454}

456char *
457na_tray_child_get_title (NaTrayChild *child)
458{
459 char *retval = NULL;
460 GdkDisplay *display;
461 Atom utf8_string, atom, type;
462 int result;
463 int format;
464 gulong nitems;
465 gulong bytes_after;
466 gchar *val;
467
468 g_return_val_if_fail (NA_IS_TRAY_CHILD (child), NULL);
469
470 display = gtk_widget_get_display (GTK_WIDGET (child));
471
472 utf8_string = gdk_x11_get_xatom_by_name_for_display (display, "UTF8_STRING");
473 atom = gdk_x11_get_xatom_by_name_for_display (display, "_NET_WM_NAME");
474
475 gdk_x11_display_error_trap_push (display);
476
477 result = XGetWindowProperty (GDK_DISPLAY_XDISPLAY (display),
478 child->icon_window,
479 atom,
480 0, G_MAXLONG,
481 False, utf8_string,
482 &type, &format, &nitems,
483 &bytes_after, (guchar **)&val);
484
485 if (gdk_x11_display_error_trap_pop (display) || result != Success)
486 return NULL;
487
488 if (type != utf8_string ||
489 format != 8 ||
490 nitems == 0)
491 {
492 if (val)
493 XFree (val);
494 return NULL;
495 }
496
497 if (!g_utf8_validate (val, nitems, NULL))
498 {
499 XFree (val);
500 return NULL;
501 }
502
503 retval = g_strndup (val, nitems);
504
505 XFree (val);
506
507 return retval;
508}

580static char *
581latin1_to_utf8 (const char *latin1)
582{
583 GString *str;
584 const char *p;
585
586 str = g_string_new (NULL);
587
588 p = latin1;

628 }
629
630 if (ch.res_class)
631 {
632 if (res_class)
633 *res_class = latin1_to_utf8 (ch.res_class);
634
635 XFree (ch.res_class);
