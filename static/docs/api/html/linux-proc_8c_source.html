48void
49GetLoad (guint64 Maximum,
50 guint64 data [cpuload_n],
51 LoadGraph *g)
52{
53 MultiloadApplet *multiload;
54 glibtop_cpu cpu;
55 guint64 cpu_aux [cpuload_n], used = 0, total = 0;
56 guint64 current_scaled, used_scaled = 0;
57 unsigned i;
58
59 glibtop_get_cpu (&cpu);
60
61 g_return_if_fail ((cpu.flags & needed_cpu_flags) == needed_cpu_flags);
62
63 multiload = g->multiload;
64
65 multiload->cpu_time [cpuload_usr] = cpu.user;
66 multiload->cpu_time [cpuload_nice] = cpu.nice;
67 multiload->cpu_time [cpuload_sys] = cpu.sys;
68 multiload->cpu_time [cpuload_iowait] = cpu.iowait + cpu.irq + cpu.softirq;
69 multiload->cpu_time [cpuload_free] = cpu.idle;
70
71 if (!multiload->cpu_initialized) {
72 memcpy (multiload->cpu_last, multiload->cpu_time, sizeof (multiload->cpu_last));
73 multiload->cpu_initialized = TRUE;
74 }
75
76 for (i = 0; i < cpuload_n; i++) {
77 cpu_aux [i] = multiload->cpu_time [i] - multiload->cpu_last [i];
78 total += cpu_aux [i];
79 }
80
81 for (i = 0; i < cpuload_free; i++) {
82 used += cpu_aux [i];
83 current_scaled = (guint64) ((float)(cpu_aux [i] * Maximum) / (float)total);
84 used_scaled += current_scaled;
85 data [i] = current_scaled;
86 }
87 data [cpuload_free] = Maximum - used_scaled;
88
89 multiload->cpu_used_ratio = (float)(used) / (float)total;
90
91 memcpy (multiload->cpu_last, multiload->cpu_time, sizeof multiload->cpu_last);
92}

94void
95GetDiskLoad (guint64 Maximum,
96 guint64 data [diskload_n],
97 LoadGraph *g)
98{
99 static gboolean first_call = TRUE;
100 static guint64 lastread = 0, lastwrite = 0;
101 static AutoScaler scaler;
102
103 guint64 max;
104 guint64 read, write;
105 guint64 readdiff, writediff;
106 guint i;
107
108 MultiloadApplet *multiload;
109
110 multiload = g->multiload;
111
112 if(first_call)
113 {
114 autoscaler_init (&scaler, g->speed, 500);
115 }
116
117 read = write = 0;
118
119 if (multiload->nvme_diskstats)
120 {
121 FILE *fdr;
122 char line[255];
123 guint64 s_read, s_write;
124
125 fdr = fopen("/proc/diskstats", "r");
126 if (!fdr)
127 {
128 multiload->nvme_diskstats = FALSE;
129 g_settings_set_boolean (multiload->settings, "diskload-nvme-diskstats", FALSE);
130 return;
131 }
132
133 while (fgets(line, 255, fdr))
134 {
135 /* Match main device, rather than individual partitions (e.g. nvme0n1) */
136 if (!g_regex_match_simple("\\snvme\\d+\\w+\\d+\\s", line, 0, 0))
137 {
138 continue;
139 }
140
141 /*
142 6 - sectors read
143 10 - sectors written
144 */
145 if (sscanf(line, "%*d %*d %*s %*d %*d %ld %*d %*d %*d %ld", &s_read, &s_write) == 2)
146 {
147 read += 512 * s_read;
148 write += 512 * s_write;
149 }
150 }
151 fclose(fdr);
152 }
153 else
154 {
155 glibtop_mountlist mountlist;
156 glibtop_mountentry *mountentries;
157
158 mountentries = glibtop_get_mountlist (&mountlist, FALSE);
159
160 for (i = 0; i < mountlist.number; i++)
161 {
162 struct statvfs statresult;
163 glibtop_fsusage fsusage;
164
165 if (strstr (mountentries[i].devname, "/dev/") == NULL)
166 continue;
167
168 if (strstr (mountentries[i].mountdir, "/media/") != NULL)
169 continue;
170
171 if (statvfs (mountentries[i].mountdir, &statresult) < 0)
172 {
173 g_debug ("Failed to get statistics for mount entry: %s. Reason: %s. Skipping entry.",
174 mountentries[i].mountdir, strerror(errno));
175 continue;
176 }
177
178 glibtop_get_fsusage(&fsusage, mountentries[i].mountdir);
179 read += fsusage.read;
180 write += fsusage.write;
181 }
182
183 g_free(mountentries);
184 }
185
186 readdiff = read - lastread;
187 writediff = write - lastwrite;
188
189 lastread = read;
190 lastwrite = write;
191
192 if(first_call)
193 {
194 first_call = FALSE;
195 memset(data, 0, 3 * sizeof data[0]);
196 return;
197 }
198
199 max = autoscaler_get_max(&scaler, readdiff + writediff);
200
201 multiload->diskload_used_ratio = (float)(readdiff + writediff) / (float)max;
202
203 data [diskload_read] = (guint64) ((float)Maximum * (float)readdiff / (float)max);
204 data [diskload_write] = (guint64) ((float)Maximum * (float)writediff / (float)max);
205 data [diskload_free] = Maximum - (data [0] + data[1]);
206}

208/* GNU/Linux:
209 * aux [memload_user] = (mem.total - mem.free) - (mem.cached + mem.buffer)
210 * aux [memload_shared] = mem.shared;
211 * aux [memload_cached] = mem.cached - mem.shared;
212 * aux [memload_buffer] = mem.buffer;
213 *
214 * Other operating systems:
215 * aux [memload_user] = mem.user;
216 * aux [memload_shared] = mem.shared;
217 * aux [memload_cached] = mem.cached;
218 * aux [memload_buffer] = mem.buffer;

220void
221GetMemory (guint64 Maximum,
222 guint64 data [memload_n],
223 LoadGraph *g)
224{
225 MultiloadApplet *multiload;
226 glibtop_mem mem;
227 guint64 aux [memload_n], cache = 0;
228 guint64 current_scaled, used_scaled = 0;
229 int i;
230
231 glibtop_get_mem (&mem);
232
233 g_return_if_fail ((mem.flags & needed_mem_flags) == needed_mem_flags);
234
235#ifndef __linux__
236 aux [memload_user] = mem.user;
237 aux [memload_cached] = mem.cached;
238#else
239 aux [memload_user] = mem.total - mem.free - mem.buffer - mem.cached;;
240 aux [memload_cached] = mem.cached - mem.shared;
241#endif /* __linux__ */
242 aux [memload_shared] = mem.shared;
243 aux [memload_buffer] = mem.buffer;
244
245 for (i = 0; i < memload_free; i++) {
246 current_scaled = (guint64) ((float)(aux [i] * Maximum) / (float)mem.total);
247 if (i != memload_user) {
248 cache += aux [i];
249 }
250 used_scaled += current_scaled;
251 data [i] = current_scaled;
252 }
253 data [memload_free] = MAX (Maximum - used_scaled, 0);
254
255 multiload = g->multiload;
256 multiload->memload_user = aux [memload_user];
257 multiload->memload_cache = cache;
258 multiload->memload_total = mem.total;
259}

310/*
311 * Return true if a network device (identified by its name) is virtual
312 * (ie: not corresponding to a physical device). In case it is a physical
313 * device or unknown, returns false.
314 */
315static gboolean
316is_net_device_virtual(char *device)
317{
318 /*
319 * There is not definitive way to find out. On some systems (Linux
320 * kernels 2.19 without option SYSFS_DEPRECATED), there exist a
321 * directory /sys/devices/virtual/net which only contains virtual
322 * devices. It's also possible to detect by the fact that virtual
323 * devices do not have a symlink "device" in
324 * /sys/class/net/name-of-dev/ . This second method is more complex
325 * but more reliable.
326 */
327 gboolean ret = FALSE;
328 char *path = malloc (strlen (device) + strlen ("/sys/class/net//device") + 1);
329
330 if (path == NULL)
331 return FALSE;
332
333 /* Check if /sys/class/net/name-of-dev/ exists (may be old linux kernel
334 * or not linux at all). */
335 do {
336 if (sprintf(path, "/sys/class/net/%s", device) < 0)
337 break;
338 if (access(path, F_OK) != 0)
339 break; /* unknown */
340
341 if (sprintf(path, "/sys/class/net/%s/device", device) < 0)
342 break;
343 if (access(path, F_OK) != 0)
344 ret = TRUE;
345 } while (0);
346
347 free (path);
348 return ret;
349}

351void
352GetNet (guint64 Maximum,
353 guint64 data [4],
354 LoadGraph *g)
355{
356 enum Types {
357 IN_COUNT = 0,
358 OUT_COUNT = 1,
359 LOCAL_COUNT = 2,
360 COUNT_TYPES = 3
361 };
362
363 static int ticks = 0;
364 static guint64 past[COUNT_TYPES] = {0};
365
366 guint64 present[COUNT_TYPES] = {0};
367
368 guint i;
369 gchar **devices;
370 glibtop_netlist netlist;
371
372 MultiloadApplet *multiload;
373
374 multiload = g->multiload;
375 devices = glibtop_get_netlist(&netlist);
376
377 for(i = 0; i < netlist.number; ++i)
378 {
379 glibtop_netload netload;
380
381 glibtop_get_netload(&netload, devices[i]);
382
383 g_return_if_fail((netload.flags & needed_netload_flags) == needed_netload_flags);
384
385 if (!(netload.if_flags & (1L << GLIBTOP_IF_FLAGS_UP)))
386 continue;
387
388 if (netload.if_flags & (1L << GLIBTOP_IF_FLAGS_LOOPBACK)) {
389 /* for loopback in and out are identical, so only count in */
390 present[LOCAL_COUNT] += netload.bytes_in;
391 continue;
392 }
393
394 /*
395 * Do not include virtual devices (VPN, PPPOE...) to avoid
396 * counting the same throughput several times.
397 */
398 if (is_net_device_virtual(devices[i]))
399 continue;
400
401 present[IN_COUNT] += netload.bytes_in;
402 present[OUT_COUNT] += netload.bytes_out;
403 }
404
405 g_strfreev(devices);
406 netspeed_add (multiload->netspeed_in, present[IN_COUNT]);
407 netspeed_add (multiload->netspeed_out, present[OUT_COUNT]);
408
409 if(ticks < 2) /* avoid initial spike */
410 {
411 ticks++;
412 memset(data, 0, (COUNT_TYPES + 1) * sizeof data[0]);
413 }
414 else
415 {
416 data[COUNT_TYPES] = 0;
417 float seconds = (float) g->speed / 1000.0f;
418 for (i = 0; i < COUNT_TYPES; i++)
419 {
420 /* protect against weirdness */
421 if (present[i] >= past[i])
422 data[i] = (guint) ((float) (present[i] - past[i]) / seconds);
423 else
424 data[i] = 0;
425 data[COUNT_TYPES] += data[i];
426 }
427 }
428
429 memcpy(past, present, sizeof past);
430}
