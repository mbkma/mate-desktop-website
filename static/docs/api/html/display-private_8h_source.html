327/* Xserver time can wraparound, thus comparing two timestamps needs to take
328 * this into account. Here's a little macro to help out. If no wraparound
329 * has occurred, this is equivalent to
330 * time1 < time2
331 * Of course, the rest of the ugliness of this macro comes from accounting
332 * for the fact that wraparound can occur and the fact that a timestamp of
333 * 0 must be special-cased since it means older than anything else.
334 *
335 * Note that this is NOT an equivalent for time1 <= time2; if that's what
336 * you need then you'll need to swap the order of the arguments and negate
337 * the result.
338 */
339#define XSERVER_TIME_IS_BEFORE_ASSUMING_REAL_TIMESTAMPS(time1, time2) \
340 ( (( (time1) < (time2) ) && ( (time2) - (time1) < ((guint32)-1)/2 )) || \
341 (( (time1) > (time2) ) && ( (time1) - (time2) > ((guint32)-1)/2 )) \
342 )
343#define XSERVER_TIME_IS_BEFORE(time1, time2) \
344 ( (time1) == 0 || \
345 (XSERVER_TIME_IS_BEFORE_ASSUMING_REAL_TIMESTAMPS(time1, time2) && \
346 (time2) != 0) \
347 )
