68static void
69on_stock_data_received (GObject *source_object,
70 GAsyncResult *result,
71 gpointer user_data)
72{
73 gpointer *user_data_with_index = (gpointer *)user_data;
74 InvestApplet *applet = INVEST_APPLET (user_data_with_index[0]);
75 gint symbol_index = GPOINTER_TO_INT (user_data_with_index[1]);
76 SoupMessage *msg = SOUP_MESSAGE (user_data_with_index[2]);
77
78 JsonParser *parser = NULL;
79 JsonNode *root;
80 JsonObject *root_obj;
81 GError *error = NULL;
82 GBytes *response_body;
83
84 response_body = soup_session_send_and_read_finish (SOUP_SESSION (source_object), result, &error);
85 if (error) {
86 g_warning ("Failed to fetch stock data for symbol %d: %s", symbol_index, error->message);
87 g_error_free (error);
88 goto cleanup;
89 }
90
91 if (soup_message_get_status (msg) != SOUP_STATUS_OK) {
92 g_warning ("Failed to fetch stock data for symbol %d: %s", symbol_index, soup_message_get_reason_phrase (msg));
93 g_bytes_unref (response_body);
94 goto cleanup;
95 }
96
97 parser = json_parser_new ();
98 gsize data_length;
99 const gchar *data = g_bytes_get_data (response_body, &data_length);
100 if (!json_parser_load_from_data (parser, data, data_length, &error)) {
101 g_warning ("Failed to parse JSON for symbol %d: %s", symbol_index, error->message);
102 g_error_free (error);
103 g_bytes_unref (response_body);
104 goto cleanup;
105 }
106 g_bytes_unref (response_body);
107
108 root = json_parser_get_root (parser);
109 root_obj = json_node_get_object (root);
110
111 /* Parse data for single stock. The expected JSON from the Yahoo Finance API should look like this:
112 * {
113 * "chart": {
114 * "result": [{
115 * "meta": {
116 * "currency": "USD",
117 * "symbol": "IBM",
118 * "exchangeName": "NYQ",
119 * "instrumentType": "EQUITY",
120 * "regularMarketPrice": 288.998,
121 * "longName": "International Business Machines Corporation",
122 * "shortName": "International Business Machines",
123 * "previousClose": 291.2,
124 * ...
125 * },
126 * "timestamp": [...],
127 * "indicators": {...}
128 * }],
129 * "error": null
130 * }
131 * }
132 */
133 if (json_object_has_member (root_obj, "chart")) {
134 JsonObject *chart = json_object_get_object_member (root_obj, "chart");
135 if (json_object_has_member (chart, "result") && !json_object_get_null_member (chart, "result")) {
136 JsonArray *results = json_object_get_array_member (chart, "result");
137 if (results && json_array_get_length (results) > 0) {
138 JsonObject *result = json_array_get_object_element (results, 0);
139 if (result) {
140 JsonObject *meta = json_object_get_object_member (result, "meta");
141
142 if (meta) {
143 gdouble current_price = json_object_get_double_member (meta, "regularMarketPrice");
144 gdouble prev_close = json_object_get_double_member (meta, "previousClose");
145
146 if (current_price > 0 && prev_close > 0) {
147 gdouble change_percent = ((current_price - prev_close) / prev_close) * 100.0;
148
149 /* store individual stock data */
150 applet->stock_prices[symbol_index] = current_price;
151 applet->stock_changes[symbol_index] = change_percent;
152 applet->stock_valid[symbol_index] = TRUE;
153 }
154 }
155 }
156 }
157 }
158 }
159
160cleanup:
161 if (parser) {
162 g_object_unref (parser);
163 }
164 if (msg) {
165 g_object_unref (msg);
166 }
167 applet->pending_requests--;
168
169 if (applet->pending_requests == 0) {
170 if (applet->total_symbols > 0) {
171 clear_timeout (&applet->cycle_timeout_id);
172
173 applet->cycle_position = 0;
174
175 gint valid_indices[applet->total_symbols];
176 gint valid_count = get_valid_stock_indices (applet, valid_indices);
177
178 if (valid_count > 0) {
179 display_stock_at_index (applet, valid_indices[0]);
180
181 /* start cycle timer if we have multiple stocks */
182 if (!applet->cycle_timeout_id && valid_count > 1) {
183 applet->cycle_timeout_id = g_timeout_add_seconds (applet->cycle_interval, (GSourceFunc) invest_applet_cycle_stocks, applet);
184 }
185
186 gchar *tooltip_text = create_stock_tooltip (applet);
187 gtk_widget_set_tooltip_text (GTK_WIDGET (applet), tooltip_text);
188 g_free (tooltip_text);
189 } else {
190 update_applet_text (applet, _("No valid stock data"), 0.0);
191 }
192 } else {
193 update_applet_text (applet, _("No valid stock data"), 0.0);
194 }
195 }
196
197 g_free (user_data_with_index);
198}

200/* fetch stock data from Yahoo! Finance for all configured symbols */
201static gboolean
202invest_applet_update_stocks (gpointer user_data)
203{
204 InvestApplet *applet = INVEST_APPLET (user_data);
205 gchar **symbols;
206 gint symbol_count;
207
208 if (!G_IS_SETTINGS (applet->settings)) {
209 g_warning ("Settings not available yet");
210 return G_SOURCE_CONTINUE;
211 }
212
213 symbols = g_settings_get_strv (applet->settings, "stock-symbols");
214
215 if (!symbols || !symbols[0]) {
216 update_applet_text (applet, _("No stocks configured"), 0.0);
217 g_strfreev (symbols);
218 return G_SOURCE_CONTINUE;
219 }
220
221 symbol_count = g_strv_length (symbols);
222
223 free_stock_data (applet);
224
225 applet->pending_requests = symbol_count;
226 applet->total_symbols = symbol_count;
227 applet->stock_symbols = g_strdupv (symbols);
228 applet->stock_prices = g_malloc0 (symbol_count * sizeof (gdouble));
229 applet->stock_changes = g_malloc0 (symbol_count * sizeof (gdouble));
230 applet->stock_valid = g_malloc0 (symbol_count * sizeof (gboolean));
231
232 applet->cycle_position = 0;
233 clear_timeout (&applet->cycle_timeout_id);
234
235 /* initialize networking */
236 if (!applet->soup_session) {
237 applet->soup_session = soup_session_new ();
238 }
239
240 /* make separate request for each stock symbol */
241 for (gint i = 0; i < symbol_count; i++) {
242 gchar *url = g_strdup_printf ("https://query2.finance.yahoo.com/v8/finance/chart/%s", symbols[i]);
243 SoupMessage *msg = soup_message_new ("GET", url);
244
245 /* HACK: avoid rate limiting */
246 SoupMessageHeaders *headers = soup_message_get_request_headers (msg);
247 soup_message_headers_replace (headers, "User-Agent",
248 "Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36");
249
250 gpointer *user_data_with_index = g_malloc (3 * sizeof (gpointer));
251 user_data_with_index[0] = applet;
252 user_data_with_index[1] = GINT_TO_POINTER (i);
253 user_data_with_index[2] = msg;
254
255 soup_session_send_and_read_async (applet->soup_session, msg, G_PRIORITY_DEFAULT, NULL, on_stock_data_received, user_data_with_index);
256 g_free (url);
257 }
258
259 g_strfreev (symbols);
260 return G_SOURCE_CONTINUE;
261}

263static void
264invest_applet_preferences_cb (GtkAction *action,
265 InvestApplet *applet)
266{
267 GtkWidget *dialog, *content_area, *entry, *label, *spin_button, *refresh_label, *cycle_label, *cycle_spin;
268 GtkWidget *hbox1, *hbox2, *hbox3;
269 gchar **symbols;
270 gchar *symbols_text;
271 gint response, refresh_interval, cycle_interval;
272
273 dialog = gtk_dialog_new_with_buttons (_("Investment Applet Preferences"),
274 NULL,
275 GTK_DIALOG_MODAL,
276 _("_Cancel"), GTK_RESPONSE_CANCEL,
277 _("_OK"), GTK_RESPONSE_OK,
278 NULL);
279
280 gtk_dialog_set_default_response (GTK_DIALOG (dialog), GTK_RESPONSE_OK);
281 gtk_window_set_default_size (GTK_WINDOW (dialog), 350, 150);
282
283 content_area = gtk_dialog_get_content_area (GTK_DIALOG (dialog));
284 gtk_container_set_border_width (GTK_CONTAINER (content_area), 12);
285
286 /* Stock symbols */
287 hbox1 = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
288 label = gtk_label_new (_("Stock symbols:"));
289 gtk_label_set_xalign (GTK_LABEL (label), 0.0);
290 gtk_widget_set_size_request (label, 120, -1);
291 entry = gtk_entry_new ();
292 gtk_box_pack_start (GTK_BOX (hbox1), label, FALSE, FALSE, 0);
293 gtk_box_pack_start (GTK_BOX (hbox1), entry, TRUE, TRUE, 0);
294
295 /* Refresh interval */
296 hbox2 = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
297 refresh_label = gtk_label_new (_("Refresh interval (minutes):"));
298 gtk_label_set_xalign (GTK_LABEL (refresh_label), 0.0);
299 gtk_widget_set_size_request (refresh_label, 120, -1);
300 spin_button = gtk_spin_button_new_with_range (1, 60, 1);
301 gtk_spin_button_set_digits (GTK_SPIN_BUTTON (spin_button), 0);
302 gtk_box_pack_start (GTK_BOX (hbox2), refresh_label, FALSE, FALSE, 0);
303 gtk_box_pack_start (GTK_BOX (hbox2), spin_button, FALSE, FALSE, 0);
304
305 /* Cycle interval */
306 hbox3 = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 6);
307 cycle_label = gtk_label_new (_("Cycle interval (seconds):"));
308 gtk_label_set_xalign (GTK_LABEL (cycle_label), 0.0);
309 gtk_widget_set_size_request (cycle_label, 120, -1);
310 cycle_spin = gtk_spin_button_new_with_range (1, 60, 1);
311 gtk_spin_button_set_digits (GTK_SPIN_BUTTON (cycle_spin), 0);
312 gtk_box_pack_start (GTK_BOX (hbox3), cycle_label, FALSE, FALSE, 0);
313 gtk_box_pack_start (GTK_BOX (hbox3), cycle_spin, FALSE, FALSE, 0);
314
315 if (!G_IS_SETTINGS (applet->settings)) {
316 g_warning ("Settings not available in preferences");
317 gtk_widget_destroy (dialog);
318 return;
319 }
320
321 symbols = g_settings_get_strv (applet->settings, "stock-symbols");
322 symbols_text = g_strjoinv (",", symbols);
323 gtk_entry_set_text (GTK_ENTRY (entry), symbols_text);
324 g_free (symbols_text);
325 g_strfreev (symbols);
326
327 refresh_interval = g_settings_get_int (applet->settings, "refresh-interval");
328 gtk_spin_button_set_value (GTK_SPIN_BUTTON (spin_button), refresh_interval);
329
330 cycle_interval = g_settings_get_int (applet->settings, "cycle-interval");
331 gtk_spin_button_set_value (GTK_SPIN_BUTTON (cycle_spin), cycle_interval);
332
333 /* Pack all rows into content area */
334 gtk_box_pack_start (GTK_BOX (content_area), hbox1, FALSE, FALSE, 6);
335 gtk_box_pack_start (GTK_BOX (content_area), hbox2, FALSE, FALSE, 6);
336 gtk_box_pack_start (GTK_BOX (content_area), hbox3, FALSE, FALSE, 6);
337
338 gtk_widget_show_all (dialog);
339
340 response = gtk_dialog_run (GTK_DIALOG (dialog));
341
342 if (response == GTK_RESPONSE_OK) {
343 const gchar *text = gtk_entry_get_text (GTK_ENTRY (entry));
344 gchar **new_symbols = g_strsplit (text, ",", -1);
345 gint new_refresh_interval = gtk_spin_button_get_value_as_int (GTK_SPIN_BUTTON (spin_button));
346 gint new_cycle_interval = gtk_spin_button_get_value_as_int (GTK_SPIN_BUTTON (cycle_spin));
347
348 /* trim whitespaces */
349 for (gint i = 0; new_symbols[i]; i++) {
350 g_strstrip (new_symbols[i]);
351 }
352
353 g_settings_set_strv (applet->settings, "stock-symbols", (const gchar * const *)new_symbols);
354 g_settings_set_int (applet->settings, "refresh-interval", new_refresh_interval);
355 g_settings_set_int (applet->settings, "cycle-interval", new_cycle_interval);
356 g_strfreev (new_symbols);
357
358 /* Update refresh interval if changed */
359 if (new_refresh_interval != applet->refresh_interval) {
360 clear_timeout (&applet->update_timeout_id);
361 applet->refresh_interval = new_refresh_interval;
362 applet->update_timeout_id = g_timeout_add_seconds (new_refresh_interval * 60,
363 invest_applet_update_stocks,
364 applet);
365 }
366
367 /* Update cycle interval if changed */
368 if (new_cycle_interval != applet->cycle_interval) {
369 clear_timeout (&applet->cycle_timeout_id);
370 applet->cycle_interval = new_cycle_interval;
371 /* Restart cycle timer with new interval */
372 if (applet->total_symbols > 0) {
373 applet->cycle_timeout_id = g_timeout_add_seconds (new_cycle_interval, (GSourceFunc) invest_applet_cycle_stocks, applet);
374 }
375 }
376
377 /* Trigger update */
378 invest_applet_update_stocks (applet);
379
380 /* Refresh chart data if chart is visible */
381 invest_chart_refresh_data (applet->chart);
382 }
383
384 gtk_widget_destroy (dialog);
385}

448static gboolean
449invest_applet_cycle_stocks (InvestApplet *applet)
450{
451 if (applet->total_symbols == 0) {
452 return G_SOURCE_CONTINUE;
453 }
454
455 /* find valid stocks */
456 gint valid_indices[applet->total_symbols];
457 gint valid_count = get_valid_stock_indices (applet, valid_indices);
458
459 if (valid_count == 0) {
460 update_applet_text (applet, _("No valid stock data"), 0.0);
461 return G_SOURCE_CONTINUE;
462 }
463
464 if (valid_count == 1) {
465 /* single stock, nothing to cycle, stop timer */
466 return G_SOURCE_REMOVE;
467 }
468
469 /* multiple stocks, cycle to next stock */
470 applet->cycle_position = (applet->cycle_position + 1) % valid_count;
471
472 /* display current stock */
473 display_stock_at_index (applet, valid_indices[applet->cycle_position]);
474
475 return G_SOURCE_CONTINUE;
476}

596static gboolean
597invest_applet_factory (MatePanelApplet *applet,
598 const gchar *iid,
599 gpointer data)
600{
601 if (!g_strcmp0 (iid, "InvestApplet")) {
602 InvestApplet *invest_applet = INVEST_APPLET (applet);
603
604#ifndef ENABLE_IN_PROCESS
605 g_set_application_name (_("Investment Applet"));
606#endif
607 gtk_window_set_default_icon_name ("mate-invest-applet");
608
609 /* Set applet flags first */
610 mate_panel_applet_set_flags (applet, MATE_PANEL_APPLET_EXPAND_MINOR);
611
612 /* Initialize settings after applet is set up */
613 invest_applet->settings = mate_panel_applet_settings_new (applet, "org.mate.panel.applet.invest");
614
615 /* Load refresh interval from settings */
616 if (G_IS_SETTINGS (invest_applet->settings)) {
