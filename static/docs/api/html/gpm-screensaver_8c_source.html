53gboolean
54gpm_screensaver_lock (GpmScreensaver *screensaver)
55{
56 guint sleepcount = 0;
57
58 g_return_val_if_fail (GPM_IS_SCREENSAVER (screensaver), FALSE);
59
60 if (screensaver->priv->proxy == NULL) {
61 g_warning ("not connected");
62 return FALSE;
63 }
64
65 g_debug ("doing mate-screensaver lock");
66 dbus_g_proxy_call_no_reply (screensaver->priv->proxy,
67 "Lock", G_TYPE_INVALID);
68
69 /* When we send the Lock signal to g-ss it takes maybe a second
70 or so to fade the screen and lock. If we suspend mid fade then on
71 resume the X display is still present for a split second
72 (since fade is gamma) and as such it can leak information.
73 Instead we wait until g-ss reports running and thus blanked
74 solidly before we continue from the screensaver_lock action.
75 The interior of g-ss is async, so we cannot get the dbus method
76 to block until lock is complete. */
77 while (! gpm_screensaver_check_running (screensaver)) {
78 /* Sleep for 1/10s */
79 g_usleep (1000 * 100);
80 if (sleepcount++ > 50) {
81 g_debug ("timeout waiting for mate-screensaver");
82 break;
83 }
84 }
85
86 return TRUE;
87}

95guint
96gpm_screensaver_add_throttle (GpmScreensaver *screensaver,
97 const char *reason)
98{
99 GError *error = NULL;
100 gboolean ret;
101 guint32 cookie;
102
103 g_return_val_if_fail (GPM_IS_SCREENSAVER (screensaver), 0);
104 g_return_val_if_fail (reason != NULL, 0);
105
106 if (screensaver->priv->proxy == NULL) {
107 g_warning ("not connected");
108 return 0;
109 }
110
111 ret = dbus_g_proxy_call (screensaver->priv->proxy,
112 "Throttle", &error,
113 G_TYPE_STRING, "Power screensaver",
114 G_TYPE_STRING, reason,
115 G_TYPE_INVALID,
116 G_TYPE_UINT, &cookie,
117 G_TYPE_INVALID);
118 if (error) {
119 g_debug ("ERROR: %s", error->message);
120 g_error_free (error);
121 }
122 if (!ret) {
123 /* abort as the DBUS method failed */
124 g_warning ("Throttle failed!");
125 return 0;
126 }
127
128 g_debug ("adding throttle reason: '%s': id %u", reason, cookie);
129 return cookie;
130}

137{
138 gboolean ret;
139 GError *error = NULL;
140
141 g_return_val_if_fail (GPM_IS_SCREENSAVER (screensaver), FALSE);
142
143 if (screensaver->priv->proxy == NULL) {
144 g_warning ("not connected");
145 return FALSE;
146 }
147
148 g_debug ("removing throttle: id %u", cookie);
149 ret = dbus_g_proxy_call (screensaver->priv->proxy,
150 "UnThrottle", &error,
151 G_TYPE_UINT, cookie,
152 G_TYPE_INVALID,
153 G_TYPE_INVALID);
154 if (error) {
155 g_debug ("ERROR: %s", error->message);
156 g_error_free (error);
157 }
158 if (!ret) {
159 /* abort as the DBUS method failed */
160 g_warning ("UnThrottle failed!");
161 return FALSE;
162 }
163
164 return TRUE;
165}

239static void
240gpm_screensaver_init (GpmScreensaver *screensaver)
241{
242 DBusGConnection *connection;
243
244 screensaver->priv = gpm_screensaver_get_instance_private (screensaver);
245
246 connection = dbus_g_bus_get (DBUS_BUS_SESSION, NULL);
247 screensaver->priv->proxy = dbus_g_proxy_new_for_name (connection,
248 GS_LISTENER_SERVICE,
249 GS_LISTENER_PATH,
250 GS_LISTENER_INTERFACE);
251}

257static void
258gpm_screensaver_finalize (GObject *object)
259{
260 GpmScreensaver *screensaver;
261 g_return_if_fail (object != NULL);
262 g_return_if_fail (GPM_IS_SCREENSAVER (object));
263
264 screensaver = GPM_SCREENSAVER (object);
265 screensaver->priv = gpm_screensaver_get_instance_private (screensaver);
266
267 g_object_unref (screensaver->priv->proxy);
268
269 G_OBJECT_CLASS (gpm_screensaver_parent_class)->finalize (object);
270}
