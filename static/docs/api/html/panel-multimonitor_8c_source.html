64static gboolean
65_panel_multimonitor_output_should_be_first (Display *xdisplay,
66 RROutput output,
67 XRROutputInfo *info,
68 RROutput primary)
69{
70 if (primary)
71 return output == primary;
72
73 Atom connector_type_atom;
74 Atom actual_type;
75 int actual_format;
76 unsigned long nitems;
77 unsigned long bytes_after;
78 unsigned char *prop;
79
80 connector_type_atom = XInternAtom (xdisplay, "ConnectorType", False);
81
82 if (XRRGetOutputProperty (xdisplay, output, connector_type_atom,
83 0, 100, False, False, None,
84 &actual_type, &actual_format,
85 &nitems, &bytes_after, &prop) == Success) {
86 if (actual_type == XA_ATOM && nitems == 1 && actual_format == 32) {
87 char *connector_type = XGetAtomName (xdisplay, prop[0]);
88 gboolean retval = g_strcmp0 (connector_type, "Panel") == 0;
89 XFree (connector_type);
90 return retval;
91 }
92 }
93
94 /* Fallback (see https://bugs.freedesktop.org/show_bug.cgi?id=26736)
95 * "LVDS" is the oh-so-intuitive name that X gives to laptop LCDs.
96 * It can actually be LVDS0, LVDS-0, Lvds, etc.
97 */
98 return (g_ascii_strncasecmp (info->name, "LVDS", strlen ("LVDS")) == 0);
99}

101static gboolean
102panel_multimonitor_get_randr_monitors (int *monitors_ret,
103 GdkRectangle **geometries_ret)
104{
105 GdkDisplay *display;
106 GdkScreen *screen;
107 GdkMonitor *monitor;
108 Display *xdisplay;
109 Window xroot;
110 XRRScreenResources *resources;
111 RROutput primary;
112 GArray *geometries_array;
113 int scale;
114 int i;
115
116 display = gdk_display_get_default ();
117
118 g_return_val_if_fail (have_randr, FALSE);
119 g_return_val_if_fail (GDK_IS_X11_DISPLAY (display), FALSE);
120
121 /* GTK+ 2.14.x uses the Xinerama API, instead of RANDR, to get the
122 * monitor geometries. It does this to avoid calling
123 * XRRGetScreenResources(), which is slow as it re-detects all the
124 * monitors --- note that XRRGetScreenResourcesCurrent() had not been
125 * introduced yet. Using Xinerama in GTK+ has the bad side effect that
126 * gdk_screen_get_monitor_plug_name() will return NULL, as Xinerama
127 * does not provide that information, unlike RANDR.
128 *
129 * Here we need to identify the output names, so that we can put the
130 * built-in LCD in a laptop *before* all other outputs. This is so
131 * that mate-panel will normally prefer to appear on the "native"
132 * display rather than on an external monitor.
133 *
134 * To get the output names and geometries, we will not use
135 * gdk_screen_get_n_monitors() and friends, but rather we will call
136 * XRR*() directly.
137 *
138 * See https://bugzilla.novell.com/show_bug.cgi?id=479684 for this
139 * particular bug, and and
140 * http://bugzilla.gnome.org/show_bug.cgi?id=562944 for a more
141 * long-term solution.
142 */
143 screen = gdk_display_get_default_screen (display);
144 xdisplay = GDK_SCREEN_XDISPLAY (screen);
145 xroot = GDK_WINDOW_XID (gdk_screen_get_root_window (screen));
146
147 resources = XRRGetScreenResourcesCurrent (xdisplay, xroot);
148 if (resources->noutput == 0) {
149 /* This might happen if nothing tried to get randr
150 * resources from the server before, so we need an
151 * active probe. See comment #27 in
152 * https://bugzilla.gnome.org/show_bug.cgi?id=597101 */
153 XRRFreeScreenResources (resources);
154 resources = XRRGetScreenResources (xdisplay, xroot);
155 }
156
157 if (!resources)
158 return FALSE;
159
160 primary = XRRGetOutputPrimary (xdisplay, xroot);
161 monitor = gdk_display_get_primary_monitor (display);
162
163 /* Use scale factor to bring geometries down to device pixels to support HiDPI displays */
164 scale = gdk_monitor_get_scale_factor (monitor);
165
166 geometries_array = g_array_sized_new (FALSE, FALSE,
167 sizeof (GdkRectangle),
168 resources->noutput);
169
170 for (i = 0; i < resources->noutput; i++) {
171 XRROutputInfo *output;
172
173 output = XRRGetOutputInfo (xdisplay, resources,
174 resources->outputs[i]);
175
176 if (output->connection != RR_Disconnected &&
177 output->crtc != 0) {
178 XRRCrtcInfo *crtc;
179 GdkRectangle rect;
180
181 crtc = XRRGetCrtcInfo (xdisplay, resources,
182 output->crtc);
183
184 rect.x = crtc->x / scale;
185 rect.y = crtc->y / scale;
186 rect.width = crtc->width / scale;
187 rect.height = crtc->height / scale;
188
189 XRRFreeCrtcInfo (crtc);
190
191 if (_panel_multimonitor_output_should_be_first (xdisplay,
192 resources->outputs[i],
193 output, primary))
194 g_array_prepend_vals (geometries_array, &rect, 1);
195 else
196 g_array_append_vals (geometries_array, &rect, 1);
197 }
198
199 XRRFreeOutputInfo (output);
200 }
201
202 XRRFreeScreenResources (resources);
203
204 if (geometries_array->len == 0) {
205 /* This can happen in at least one case:
206 * https://bugzilla.novell.com/show_bug.cgi?id=543876 where all
207 * monitors appear disconnected (possibly because the screen
208 * is behing a KVM switch) -- see comment #8.
209 * There might be other cases too, so we stay on the safe side.
210 */
211 g_array_free (geometries_array, TRUE);
212 return FALSE;
213 }
214
215 *monitors_ret = geometries_array->len;
216 *geometries_ret = (GdkRectangle *) g_array_free (geometries_array, FALSE);
217
218 return TRUE;
219}

221#endif /* HAVE_X11 */

280static void
281panel_multimonitor_compress_overlapping_monitors (int *num_monitors_inout,
282 GdkRectangle **geometries_inout)
283{
284 int num_monitors;
285 GdkRectangle *geometries_array;
286 int i;
287
288 num_monitors = *num_monitors_inout;
289 geometries_array = *geometries_inout;
290
291 /* http://bugzilla.gnome.org/show_bug.cgi?id=530969
292 * https://bugzilla.novell.com/show_bug.cgi?id=310208
293 * and many other such bugs...
294 *
295 * RANDR sometimes gives us monitors that overlap (i.e. outputs whose
296 * bounding rectangles overlap). This is sometimes right and sometimes
297 * wrong:
298 *
299 * * Right - two 1024x768 outputs at the same offset (0, 0) that show
300 * the same thing. Think "laptop plus projector with the same
301 * resolution".
302 *
303 * * Wrong - one 1280x1024 output ("laptop internal LCD") and another
304 * 1024x768 output ("external monitor"), both at offset (0, 0).
305 * There is no way for the monitor with the small resolution to
306 * show the complete image from the laptop's LCD, unless one uses
307 * panning (but nobody wants panning, right!?).
308 *
309 * With overlapping monitors, we may end up placing the panel with
310 * respect to the "wrong" one. This is always wrong, as the panel
311 * appears "in the middle of the screen" of the monitor with the
312 * smaller resolution, instead of at the edge.
313 *
314 * Our strategy is to find the subsets of overlapping monitors, and
315 * "compress" each such set to being like if there were a single
316 * monitor with the biggest resolution of each of that set's monitors.
317 * Say we have four monitors
318 *
319 * A, B, C, D
320 *
321 * where B and D overlap. In that case, we'll generate a new list that
322 * looks like
323 *
324 * A, MAX(B, D), C
325 *
326 * with three monitors.
327 *
328 * NOTE FOR THE FUTURE: We could avoid most of this mess if we had a
329 * concept of a "primary monitor". Also, we could look at each
330 * output's name or properties to see if it is the built-in LCD in a
331 * laptop. However, with GTK+ 2.14.x we don't get output names, since
332 * it gets the list outputs from Xinerama, not RANDR (and Xinerama
333 * doesn't provide output names).
334 */
335
336 for (i = 0; i < num_monitors; i++) {
337 long max_pixels;
338 int j;
339
340 max_pixels = pixels_in_rectangle (&geometries_array[i]);
341
342 j = i + 1;
343
344 while (j < num_monitors) {
345 if (rectangle_overlaps (&geometries_array[i],
346 &geometries_array[j])) {
347 long pixels;
348
349 pixels = pixels_in_rectangle (&geometries_array[j]);
350 if (pixels > max_pixels) {
351 max_pixels = pixels;
352 /* keep the maximum */
353 geometries_array[i] = geometries_array[j];
354 }
355
356 /* Shift the remaining monitors to the left */
357 if (num_monitors - j - 1 > 0)
358 memmove (&geometries_array[j],
359 &geometries_array[j + 1],
360 sizeof (geometries_array[0]) * (num_monitors - j - 1));
361
362 num_monitors--;
363 g_assert (num_monitors > 0);
364 } else
365 j++;
366 }
367 }
368
369 *num_monitors_inout = num_monitors;
370 *geometries_inout = geometries_array;
371}

415static void
416panel_multimonitor_init_randr (GdkDisplay *display)
417{
418 Display *xdisplay;
419 int event_base, error_base;
420
421 g_return_if_fail (GDK_IS_X11_DISPLAY (display));
422
423 have_randr = FALSE;
424
425 if (!GDK_IS_X11_DISPLAY (display))
426 return;
427
428 xdisplay = GDK_DISPLAY_XDISPLAY (display);
429
430 /* We don't remember the event/error bases, as we expect to get monitor
431 * added/removed events from the display instead.
432 */
433
434 if (XRRQueryExtension (xdisplay, &event_base, &error_base)) {
435 int major, minor;
436
437 XRRQueryVersion (xdisplay, &major, &minor);
438 if ((major == 1 && minor >= 3) || major > 1)
439 have_randr = TRUE;
440 }
441}

445void
446panel_multimonitor_init (void)
447{
448 GdkDisplay *display;
449 GdkScreen *screen;
450 int i;
451
452 if (initialized)
453 return;
454
455 display = gdk_display_get_default ();
456 screen = gdk_display_get_default_screen (display);
457
458 have_randr = FALSE;
459
460#ifdef HAVE_X11
461#ifdef HAVE_RANDR
462 if (GDK_IS_X11_DISPLAY (display))
463 panel_multimonitor_init_randr (display);
464#endif /* HAVE_RANDR */
465#endif /* HAVE_X11 */
466
467 /*
468 * The screen signals probably shouldn't be needed, but sometimes on X11 they are
469 * the only ones that get fired
470 */
471
472 g_signal_handlers_disconnect_by_func (screen, panel_multimonitor_handle_screen_changed, NULL);
473 g_signal_connect (screen, "size-changed",
474 G_CALLBACK (panel_multimonitor_handle_screen_changed), NULL);
475 g_signal_connect (screen, "monitors-changed",
476 G_CALLBACK (panel_multimonitor_handle_screen_changed), NULL);
477
478 g_signal_handlers_disconnect_by_func (display, panel_multimonitor_handle_monitor_changed, NULL);
479 g_signal_connect (display, "monitor-added",
480 G_CALLBACK (panel_multimonitor_handle_monitor_changed), NULL);
481 g_signal_connect (display, "monitor-removed",
482 G_CALLBACK (panel_multimonitor_handle_monitor_changed), NULL);
483
484 for (i = 0; i < gdk_display_get_n_monitors (display); i++) {
485 GdkMonitor *monitor;
486
487 monitor = gdk_display_get_monitor (display, i);
488 g_signal_handlers_disconnect_by_func (display, panel_multimonitor_handle_monitor_invalidate, NULL);
489 g_signal_connect (monitor, "invalidate",
490 G_CALLBACK (panel_multimonitor_handle_monitor_invalidate), NULL);
491 }
492
493 panel_multimonitor_get_raw_monitors (&monitor_count, &geometries);
494 panel_multimonitor_compress_overlapping_monitors (&monitor_count, &geometries);
495
496 initialized = TRUE;
497}

567static int
568axis_distance (int p, int axis_start, int axis_size)
569{
570 if (p >= axis_start && p < axis_start + axis_size)
571 return 0;
572 else if (p < axis_start)
573 return (axis_start - p);
574 else
575 return (p - (axis_start + axis_size - 1));
576}

578/* The panel can't use gdk_screen_get_monitor_at_point() since it has its own
579 * view of which monitors are present. Look at get_monitors_for_screen() above
580 * to see why. */
581int
582panel_multimonitor_get_monitor_at_point (int x, int y)
583{
584 int i;
585 int min_dist_squared;
586 int closest_monitor;
587
588 min_dist_squared = G_MAXINT32;
589 closest_monitor = 0;
590
591 for (i = 0; i < monitor_count; i++) {
592 int dist_x, dist_y;
593 int dist_squared;
594
595 dist_x = axis_distance (x, geometries[i].x, geometries[i].width);
596 dist_y = axis_distance (y, geometries[i].y, geometries[i].height);
597
598 if (dist_x == 0 && dist_y == 0)
599 return i;
600
601 dist_squared = dist_x * dist_x + dist_y * dist_y;
602
603 if (dist_squared < min_dist_squared) {
604 min_dist_squared = dist_squared;
605 closest_monitor = i;
606 }
607 }
608
609 return closest_monitor;
610}

632/* determines whether a given monitor is along the visible
633 * edge of the logical screen.
634 */
635void
636panel_multimonitor_is_at_visible_extreme (int n_monitor,
637 gboolean *leftmost,
638 gboolean *rightmost,
639 gboolean *topmost,
640 gboolean *bottommost)
641{
642 MonitorBounds monitor;
643 int i;
644
645 *leftmost = TRUE;
646 *rightmost = TRUE;
647 *topmost = TRUE;
648 *bottommost = TRUE;
649
650 g_return_if_fail (n_monitor >= 0 && n_monitor < monitor_count);
651
652 get_monitor_bounds (n_monitor, &monitor);
653
654 /* go through each monitor and try to find one either right,
655 * below, above, or left of the specified monitor
656 */
657
658 for (i = 0; i < monitor_count; i++) {
659 MonitorBounds iter;
660
661 if (i == n_monitor) continue;
662
663 get_monitor_bounds (i, &iter);
664
665 if ((iter.y0 >= monitor.y0 && iter.y0 < monitor.y1) ||
666 (iter.y1 > monitor.y0 && iter.y1 <= monitor.y1)) {
667 if (iter.x0 < monitor.x0)
668 *leftmost = FALSE;
669 if (iter.x1 > monitor.x1)
670 *rightmost = FALSE;
671 }
672
673 if ((iter.x0 >= monitor.x0 && iter.x0 < monitor.x1) ||
674 (iter.x1 > monitor.x0 && iter.x1 <= monitor.x1)) {
675 if (iter.y0 < monitor.y0)
676 *topmost = FALSE;
677 if (iter.y1 > monitor.y1)
678 *bottommost = FALSE;
679 }
680 }
681}
