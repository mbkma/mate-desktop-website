57static GOptionContext *
58create_option_context (void)
59{
60 GOptionContext *context;
61
62 const GOptionEntry entries[] = {
63 { "version", 'V', G_OPTION_FLAG_NO_ARG, G_OPTION_ARG_CALLBACK,
64 logview_show_version_and_quit, N_("Show the application's version"), NULL },
65 { G_OPTION_REMAINING, '\0', 0, G_OPTION_ARG_FILENAME_ARRAY, &log_files,
66 NULL, N_("[LOGFILE...]") },
67 { NULL, 0, 0, G_OPTION_ARG_NONE, NULL, NULL, NULL },
68 };
69
70 context = g_option_context_new (_(" - Browse and monitor logs"));
71#ifdef ENABLE_NLS
72 g_option_context_set_translation_domain (context, GETTEXT_PACKAGE);
73#endif /* ENABLE_NLS */
74 g_option_context_add_main_entries (context, entries, GETTEXT_PACKAGE);
75 g_option_context_set_ignore_unknown_options (context, TRUE);
76 g_option_context_add_group (context, gtk_get_option_group (TRUE));
77
78 return context;

81int
82main (int argc, char *argv[])
83{
84 GError *error = NULL;
85 GOptionContext *context;
86 LogviewApp *app;
87
88#ifdef ENABLE_NLS
89 bindtextdomain (GETTEXT_PACKAGE, MATELOCALEDIR);
90 bind_textdomain_codeset (GETTEXT_PACKAGE, "UTF-8");
91 textdomain (GETTEXT_PACKAGE);
92#endif /* ENABLE_NLS */
93
94 context = create_option_context ();
95
96 g_option_context_parse (context, &argc, &argv, &error);
97
98 if (error) {
99 g_critical ("Unable to parse arguments: %s", error->message);
100 g_error_free (error);
101 g_option_context_free (context);
102
103 exit (1);
104 }
105
106 g_option_context_free (context);
107 g_set_application_name (_("Log Viewer"));
108
109 app = logview_app_get ();
110
111 if (!app) {
112 g_critical ("Unable to create the user interface.");
113
114 exit (1);
115 } else {
116 g_signal_connect (app, "app-quit",
117 G_CALLBACK (app_quit_cb), NULL);
118 }
119
120 logview_app_initialize (app, log_files);
121
122 gtk_main ();
123
124 g_object_unref (app);
125
126 return EXIT_SUCCESS;
