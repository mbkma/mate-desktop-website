63static void
64pluma_application_activate (GApplication *application)
65{
66 PlumaApp *app;
67 PlumaWindow *window;
68 gboolean restored = FALSE;
69
70 pluma_debug_message (DEBUG_APP, "PlumaApplication activate");
71
72 /* Chain up to parent */
73 G_APPLICATION_CLASS (pluma_application_parent_class)->activate (application);
74
75 if (pluma_session_is_restored ())
76 restored = pluma_session_load ();
77
78 if (!restored)
79 {
80 PlumaApplication *pluma_app = PLUMA_APPLICATION (application);
81 PlumaApplicationPrivate *priv = pluma_app->priv;
82
83 pluma_debug_message (DEBUG_APP, "Get default app");
84 app = pluma_app_get_default ();
85
86 pluma_debug_message (DEBUG_APP, "Create main window");
87 window = pluma_app_create_window (app, NULL);
88 gtk_application_add_window (GTK_APPLICATION (application), GTK_WINDOW (window));
89 gtk_widget_set_size_request (GTK_WIDGET (window), 250, 250);
90
91 if (priv->file_list != NULL)
92 {
93 const PlumaEncoding *encoding = NULL;
94
95 if (priv->encoding_charset)
96 encoding = pluma_encoding_get_from_charset (priv->encoding_charset);
97
98 pluma_debug_message (DEBUG_APP, "Load files");
99 _pluma_cmd_load_files_from_prompt (window,
100 priv->file_list,
101 encoding,
102 priv->line_position);
103 }
104 else
105 {
106 pluma_debug_message (DEBUG_APP, "Create tab");
107 pluma_window_create_tab (window, TRUE);
108 }
109
110 if (priv->new_document_option)
111 pluma_window_create_tab (window, TRUE);
112
113 pluma_debug_message (DEBUG_APP, "Show window");
114 gtk_widget_show (GTK_WIDGET (window));
115
116 pluma_application_free_command_line_data (pluma_app);
117 }
118}

120static void
121pluma_application_startup (GApplication *application)
122{
123 pluma_debug_message (DEBUG_APP, "PlumaApplication startup");
124
125 /* Chain up to parent first */
126 G_APPLICATION_CLASS (pluma_application_parent_class)->startup (application);
127
128 /* Most initialization is done in main() before GtkApplication starts */
129 /* Only do the minimal setup here that needs to happen in the GtkApplication context */
130
131 pluma_debug_message (DEBUG_APP, "Set icon");
132 gtk_icon_theme_append_search_path (gtk_icon_theme_get_default (),
133 PLUMA_DATADIR "/icons");
134
135 /* Set the associated .desktop file */
136 egg_set_desktop_file (DATADIR "/applications/pluma.desktop");
137
138 /* Init plugins engine */
139 pluma_debug_message (DEBUG_APP, "Init plugins");
140 pluma_plugins_engine_get_default ();
141
142 /* Initialize session management */
143 pluma_debug_message (DEBUG_APP, "Init session manager");
144 pluma_session_init ();
145}

162static void
163pluma_application_free_command_line_data (PlumaApplication *app)
164{
165 PlumaApplicationPrivate *priv = app->priv;
166
167 if (priv->file_list)
168 {
169 g_slist_free_full (priv->file_list, g_object_unref);
170 priv->file_list = NULL;
171 }
172
173 g_free (priv->encoding_charset);
174 priv->encoding_charset = NULL;
175
176 priv->new_window_option = FALSE;
177 priv->new_document_option = FALSE;
178 priv->line_position = 0;
179}

230void
231pluma_application_set_command_line_options (PlumaApplication *app,
232 gint line_pos,
233 const gchar *encoding,
234 gboolean new_window,
235 gboolean new_document,
236 GSList *file_list)
237{
238 PlumaApplicationPrivate *priv = app->priv;
239
240 pluma_application_free_command_line_data (app);
241
242 priv->line_position = line_pos;
243 priv->new_window_option = new_window;
244 priv->new_document_option = new_document;
245 priv->encoding_charset = g_strdup (encoding);
246
247 if (file_list)
248 {
249 GSList *l;
250 for (l = file_list; l != NULL; l = l->next)
251 {
252 priv->file_list = g_slist_prepend (priv->file_list, g_object_ref (l->data));
253 }
254 priv->file_list = g_slist_reverse (priv->file_list);
255 }
256
257 if (priv->encoding_charset &&
258 (pluma_encoding_get_from_charset (priv->encoding_charset) == NULL))
259 {
260 g_free (priv->encoding_charset);
261 priv->encoding_charset = NULL;
262 }
263}
