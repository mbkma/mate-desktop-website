228gboolean
229eel_g_lists_sort_and_check_for_intersection (GList **list_1,
230 GList **list_2)
231
232{
233 GList *node_1, *node_2;
234
235 *list_1 = g_list_sort (*list_1, compare_pointers);
236 *list_2 = g_list_sort (*list_2, compare_pointers);
237
238 node_1 = *list_1;
239 node_2 = *list_2;
240
241 while (node_1 != NULL && node_2 != NULL)
242 {
243 int compare_result;
244
245 compare_result = compare_pointers (node_1->data, node_2->data);
246 if (compare_result == 0)
247 {
248 return TRUE;
249 }
250 if (compare_result <= 0)
251 {
252 node_1 = node_1->next;
253 }
254 if (compare_result >= 0)
255 {
256 node_2 = node_2->next;
257 }
258 }
259
260 return FALSE;
261}

280GList *
281eel_g_list_partition (GList *list,
282 EelPredicateFunction predicate,
283 gpointer user_data,
284 GList **failed)
285{
286 GList *predicate_true;
287 GList *predicate_false;
288 GList *reverse;
289 GList *p;
290 GList *next;
291
292 predicate_true = NULL;
293 predicate_false = NULL;
294
295 reverse = g_list_reverse (list);
296
297 for (p = reverse; p != NULL; p = next)
298 {
299 next = p->next;
300
301 if (next != NULL)
302 {
303 next->prev = NULL;
304 }
305
306 if (predicate (p->data, user_data))
307 {
308 p->next = predicate_true;
309 if (predicate_true != NULL)
310 {
311 predicate_true->prev = p;
312 }
313 predicate_true = p;
314 }
315 else
316 {
317 p->next = predicate_false;
318 if (predicate_false != NULL)
319 {
320 predicate_false->prev = p;
321 }
322 predicate_false = p;
323 }
324 }
325
326 *failed = predicate_false;
327 return predicate_true;
328}

338static void
339free_hash_tables_at_exit (void)
340{
341 GList *p;
342 guint size;
343 HashTableToFree *hash_table_to_free = NULL;
344
345 for (p = hash_tables_to_free_at_exit; p != NULL; p = p->next)
346 {
347 hash_table_to_free = p->data;
348
349 size = g_hash_table_size (hash_table_to_free->hash_table);
350 if (size != 0)
351 {
352 if (hash_table_to_free->keys_known_to_be_strings)
353 {
354 g_print ("\n--- Hash table keys for warning below:\n");
355 g_hash_table_foreach (hash_table_to_free->hash_table,
356 print_key_string,
357 NULL);
358 }
359 g_warning ("\"%s\" hash table still has %u element%s at quit time%s",
360 hash_table_to_free->display_name, size,
361 size == 1 ? "" : "s",
362 hash_table_to_free->keys_known_to_be_strings
363 ? " (keys above)" : "");
364 }
365
366 g_hash_table_destroy (hash_table_to_free->hash_table);
367 g_free (hash_table_to_free->display_name);
368 g_free (hash_table_to_free);
369 }
370 g_list_free (hash_tables_to_free_at_exit);
371 hash_tables_to_free_at_exit = NULL;
372}

374GHashTable *
375eel_g_hash_table_new_free_at_exit (GHashFunc hash_func,
376 GCompareFunc key_compare_func,
377 const char *display_name)
378{
379 GHashTable *hash_table;
380 HashTableToFree *hash_table_to_free;
381
382 /* FIXME: We can take out the CAJA_DEBUG check once we
383 * have fixed more of the leaks. For now, it's a bit too noisy
384 * for the general public.
385 */
386 if (hash_tables_to_free_at_exit == NULL)
387 {
388 eel_debug_call_at_shutdown (free_hash_tables_at_exit);
389 }
390
391 hash_table = g_hash_table_new (hash_func, key_compare_func);
392
393 hash_table_to_free = g_new (HashTableToFree, 1);
394 hash_table_to_free->hash_table = hash_table;
395 hash_table_to_free->display_name = g_strdup (display_name);
396 hash_table_to_free->keys_known_to_be_strings =
397 hash_func == g_str_hash;
398
399 hash_tables_to_free_at_exit = g_list_prepend
400 (hash_tables_to_free_at_exit, hash_table_to_free);
401
402 return hash_table;
403}

411static void
412flatten_hash_table_element (gpointer key, gpointer value, gpointer callback_data)
413{
414 FlattenedHashTable *flattened_table;

416 flattened_table = callback_data;

423void
424eel_g_hash_table_safe_for_each (GHashTable *hash_table,
425 GHFunc callback,
426 gpointer callback_data)
427{
428 FlattenedHashTable flattened;
429 GList *p, *q;
430
431 flattened.keys = NULL;
432 flattened.values = NULL;
433
434 g_hash_table_foreach (hash_table,
435 flatten_hash_table_element,
436 &flattened);
437
438 for (p = flattened.keys, q = flattened.values;
439 p != NULL;
440 p = p->next, q = q->next)
441 {
442 (* callback) (p->data, q->data, callback_data);
443 }
444
445 g_list_free (flattened.keys);
446 g_list_free (flattened.values);
447}

449int
450eel_round (double d)
451{
452 double val;
453
454 val = floor (d + .5);
455
456 /* The tests are needed because the result of floating-point to integral
457 * conversion is undefined if the floating point value is not representable
458 * in the new type. E.g. the magnititude is too large or a negative
459 * floating-point value being converted to an unsigned.
460 */
461 g_return_val_if_fail (val <= INT_MAX, INT_MAX);
462 g_return_val_if_fail (val >= INT_MIN, INT_MIN);
463
464 return val;
465}
