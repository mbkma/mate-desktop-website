64static gboolean
65_check_file (GFile *mount_root, const char *file_path, gboolean must_be_executable)
66{
67 GFile *file;
68 GFileInfo *file_info;
69 gboolean ret;
70
71 ret = FALSE;
72
73 file = g_file_get_child (mount_root, file_path);
74 file_info = g_file_query_info (file,
75 G_FILE_ATTRIBUTE_ACCESS_CAN_EXECUTE,
76 G_FILE_QUERY_INFO_NONE,
77 NULL,
78 NULL);
79 if (file_info != NULL)
80 {
81 if (must_be_executable)
82 {
83 if (g_file_info_get_attribute_boolean (file_info, G_FILE_ATTRIBUTE_ACCESS_CAN_EXECUTE))
84 {
85 ret = TRUE;
86 }
87 }
88 else
89 {
90 ret = TRUE;
91 }
92 g_object_unref (file_info);
93 }
94 g_object_unref (file);
95
96 return ret;
97}

99static void
100autorun (GMount *mount)
101{
102 char *error_string;
103 GFile *root;
104 GFile *program_to_spawn;
105 char *path_to_spawn;
106 char *cwd_for_program;
107
108 root = g_mount_get_root (mount);
109
110 /* Careful here, according to
111 *
112 * https://standards.freedesktop.org/autostart-spec/autostart-spec-latest.html
113 *
114 * the ordering does matter.
115 */
116
117 program_to_spawn = NULL;
118 path_to_spawn = NULL;
119
120 if (_check_file (root, ".autorun", TRUE))
121 {
122 program_to_spawn = g_file_get_child (root, ".autorun");
123 }
124 else if (_check_file (root, "autorun", TRUE))
125 {
126 program_to_spawn = g_file_get_child (root, "autorun");
127 }
128 else if (_check_file (root, "autorun.sh", TRUE))
129 {
130 program_to_spawn = g_file_get_child (root, "autorun.sh");
131 }
132 else if (_check_file (root, "autorun.exe", TRUE))
133 {
134 /* TODO */
135 }
136 else if (_check_file (root, "AUTORUN.EXE", TRUE))
137 {
138 /* TODO */
139 }
140 else if (_check_file (root, "autorun.inf", FALSE))
141 {
142 /* TODO */
143 }
144 else if (_check_file (root, "AUTORUN.INF", FALSE))
145 {
146 /* TODO */
147 }
148
149 if (program_to_spawn != NULL)
150 {
151 path_to_spawn = g_file_get_path (program_to_spawn);
152 }
153
154 cwd_for_program = g_file_get_path (root);
155
156 error_string = NULL;
157 if (path_to_spawn != NULL && cwd_for_program != NULL)
158 {
159 if (chdir (cwd_for_program) == 0)
160 {
161 execl (path_to_spawn, path_to_spawn, NULL);
162 error_string = g_strdup_printf (_("Error starting autorun program: %s"), strerror (errno));
163 goto out;
164 }
165 error_string = g_strdup_printf (_("Error starting autorun program: %s"), strerror (errno));
166 goto out;
167 }
168 error_string = g_strdup_printf (_("Cannot find the autorun program"));
169
170out:
171 if (program_to_spawn != NULL)
172 {
173 g_object_unref (program_to_spawn);
174 }
175 if (root != NULL)
176 {
177 g_object_unref (root);
178 }
179 g_free (path_to_spawn);
180 g_free (cwd_for_program);
181
182 if (error_string != NULL)
183 {
184 GtkWidget *dialog;
185 dialog = gtk_message_dialog_new_with_markup (NULL, /* TODO: parent window? */
186 0,
187 GTK_MESSAGE_ERROR,
188 GTK_BUTTONS_OK,
189 _("<big><b>Error autorunning software</b></big>"));
190 gtk_message_dialog_format_secondary_text (GTK_MESSAGE_DIALOG (dialog), "%s", error_string);
191 gtk_dialog_run (GTK_DIALOG (dialog));
192 gtk_widget_destroy (dialog);
193 g_free (error_string);
194 }
195}

197static void
198present_autorun_for_software_dialog (GMount *mount)
199{
200 GIcon *icon;
201 int icon_size;
202 CajaIconInfo *icon_info;
203 GdkPixbuf *pixbuf;
204 GtkWidget *image;
205 char *mount_name;
206 GtkWidget *dialog;
207 AutorunSoftwareDialogData *data;
208
209 mount_name = g_mount_get_name (mount);
210
211 dialog = gtk_message_dialog_new_with_markup (NULL, /* TODO: parent window? */
212 0,
213 GTK_MESSAGE_OTHER,
214 GTK_BUTTONS_CANCEL,
215 _("<big><b>This medium contains software intended to be automatically started. Would you like to run it?</b></big>"));
216 gtk_message_dialog_format_secondary_text (GTK_MESSAGE_DIALOG (dialog),
217 _("The software will run directly from the medium \"%s\". "
218 "You should never run software that you don't trust.\n"
219 "\n"
220 "If in doubt, press Cancel."),
221 mount_name);
222
223 /* TODO: in a star trek future add support for verifying
224 * software on media (e.g. if it has a certificate, check it
225 * etc.)
226 */
227
228 icon = g_mount_get_icon (mount);
229 icon_size = caja_get_icon_size_for_stock_size (GTK_ICON_SIZE_DIALOG);
230 icon_info = caja_icon_info_lookup (icon, icon_size,
231 gtk_widget_get_scale_factor (GTK_WIDGET (dialog)));
232 pixbuf = caja_icon_info_get_pixbuf_at_size (icon_info, icon_size);
233 image = gtk_image_new_from_pixbuf (pixbuf);
234 gtk_widget_set_halign (image, GTK_ALIGN_CENTER);
235 gtk_widget_set_valign (image, GTK_ALIGN_START);
236 gtk_message_dialog_set_image (GTK_MESSAGE_DIALOG (dialog), image);
237
238 gtk_window_set_title (GTK_WINDOW (dialog), mount_name);
239 gtk_window_set_icon (GTK_WINDOW (dialog), pixbuf);
240
241 data = g_new0 (AutorunSoftwareDialogData, 1);
242 data->dialog = dialog;
243 data->mount = g_object_ref (mount);
244
245 g_signal_connect (G_OBJECT (mount),
246 "unmounted",
247 G_CALLBACK (autorun_software_dialog_mount_unmounted),
248 data);
249
250 gtk_dialog_add_button (GTK_DIALOG (dialog),
251 _("_Run"),
252 GTK_RESPONSE_OK);
253
254 gtk_widget_show_all (dialog);
255
256 if (gtk_dialog_run (GTK_DIALOG (dialog)) == GTK_RESPONSE_OK)
257 {
258 gtk_widget_destroy (dialog);
259 autorun (mount);
260 }
261
262 g_object_unref (icon_info);
263 g_object_unref (pixbuf);
264 g_free (mount_name);
265}
