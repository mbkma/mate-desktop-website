45void
46eom_debug_init (void)
47{
48 if (g_getenv ("EOM_DEBUG") != NULL)
49 {
50 /* Enable all debugging */
51 debug = ~EOM_DEBUG_NO_DEBUG;
52 goto out;
53 }
54
55 if (g_getenv ("EOM_DEBUG_WINDOW") != NULL)
56 debug = debug | EOM_DEBUG_WINDOW;
57
58 if (g_getenv ("EOM_DEBUG_VIEW") != NULL)
59 debug = debug | EOM_DEBUG_VIEW;
60
61 if (g_getenv ("EOM_DEBUG_JOBS") != NULL)
62 debug = debug | EOM_DEBUG_JOBS;
63
64 if (g_getenv ("EOM_DEBUG_THUMBNAIL") != NULL)
65 debug = debug | EOM_DEBUG_THUMBNAIL;
66
67 if (g_getenv ("EOM_DEBUG_IMAGE_DATA") != NULL)
68 debug = debug | EOM_DEBUG_IMAGE_DATA;
69
70 if (g_getenv ("EOM_DEBUG_IMAGE_LOAD") != NULL)
71 debug = debug | EOM_DEBUG_IMAGE_LOAD;
72
73 if (g_getenv ("EOM_DEBUG_IMAGE_SAVE") != NULL)
74 debug = debug | EOM_DEBUG_IMAGE_SAVE;
75
76 if (g_getenv ("EOM_DEBUG_LIST_STORE") != NULL)
77 debug = debug | EOM_DEBUG_LIST_STORE;
78
79 if (g_getenv ("EOM_DEBUG_PREFERENCES") != NULL)
80 debug = debug | EOM_DEBUG_PREFERENCES;
81
82 if (g_getenv ("EOM_DEBUG_PRINTING") != NULL)
83 debug = debug | EOM_DEBUG_PRINTING;
84
85 if (g_getenv ("EOM_DEBUG_LCMS") != NULL)
86 debug = debug | EOM_DEBUG_LCMS;

88 if (g_getenv ("EOM_DEBUG_PLUGINS") != NULL)

91out:

100void
101eom_debug_message (EomDebug section,
102 const gchar *file,
103 gint line,
104 const gchar *function,
105 const gchar *format, ...)
106{
107 if (G_UNLIKELY (debug & section))
108 {
109#ifdef ENABLE_PROFILING
110 gdouble seconds;
111 g_return_if_fail (timer != NULL);
112#endif
113
114 va_list args;
115 gchar *msg;
116
117 g_return_if_fail (format != NULL);
118
119 va_start (args, format);
120 msg = g_strdup_vprintf (format, args);
121 va_end (args);
122
123#ifdef ENABLE_PROFILING
124 seconds = g_timer_elapsed (timer, NULL);
125 g_print ("[%f (%f)] %s:%d (%s) %s\n",
126 seconds, seconds - last, file, line, function, msg);
127 last = seconds;
128#else
129 g_print ("%s:%d (%s) %s\n", file, line, function, msg);
130#endif
131
132 fflush (stdout);
133
134 g_free (msg);
135 }
136}
