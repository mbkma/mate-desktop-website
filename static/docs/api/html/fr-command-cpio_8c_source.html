46static time_t
47mktime_from_string (char *month,
48 char *mday,
49 char *year)
50{
51 struct tm tm = {0, };
52
53 tm.tm_isdst = -1;
54
55 if (month != NULL) {
56 static const char *months[] = { "Jan", "Feb", "Mar", "Apr", "May", "Jun",
57 "Jul", "Aug", "Sep", "Oct", "Nov", "Dec" };
58 int i;
59
60 for (i = 0; i < 12; i++) {
61 if (strcmp (months[i], month) == 0) {
62 tm.tm_mon = i;
63 break;
64 }
65 }
66 }
67 tm.tm_mday = atoi (mday);
68 if (strchr (year, ':') != NULL) {
69 char **fields = g_strsplit (year, ":", 2);
70 if ((fields != NULL) && (g_strv_length (fields) == 2)) {
71 time_t now;
72 struct tm *now_tm;
73
74 tm.tm_hour = atoi (fields[0]);
75 tm.tm_min = atoi (fields[1]);
76
77 now = time(NULL);
78 now_tm = localtime (&now);
79 tm.tm_year = now_tm->tm_year;
80 }
81 } else
82 tm.tm_year = atoi (year) - 1900;
83
84 return mktime (&tm);
85}

87static void
88list__process_line (char *line,
89 gpointer data)
90{
91 FileData *fdata;
92 FrCommand *comm = FR_COMMAND (data);
93 char **fields;
94 const char *name_field;
95 char *name;
96 int ofs = 0;
97
98 g_return_if_fail (line != NULL);
99
100 fdata = file_data_new ();
101
102#ifdef __sun
103 fields = split_line (line, 9);
104 fdata->size = g_ascii_strtoull (fields[4], NULL, 10);
105 fdata->modified = mktime_from_string (fields[5], fields[6], fields[8]);
106 g_strfreev (fields);
107
108 name_field = get_last_field (line, 10);
109#else /* !__sun */
110 /* Handle char and block device files */
111 if ((line[0] == 'c') || (line[0] == 'b')) {
112 fields = split_line (line, 9);
113 ofs = 1;
114 fdata->size = 0;
115 /* FIXME: We should also specify the content type */
116 }
117 else {
118 fields = split_line (line, 8);
119 fdata->size = g_ascii_strtoull (fields[4], NULL, 10);
120 }
121 fdata->modified = mktime_from_string (fields[5+ofs], fields[6+ofs], fields[7+ofs]);
122 g_strfreev (fields);
123
124 name_field = get_last_field (line, 9+ofs);
125#endif /* !__sun */
126
127 fields = g_strsplit (name_field, " -> ", 2);
128
129 if (fields[1] == NULL) {
130 g_strfreev (fields);
131 fields = g_strsplit (name_field, " link to ", 2);
132 }
133
134 fdata->dir = line[0] == 'd';
135
136 name = g_strcompress (fields[0]);
137 if (*(fields[0]) == '/') {
138 fdata->full_path = g_strdup (name);
139 fdata->original_path = fdata->full_path;
140 }
141 else {
142 fdata->full_path = g_strconcat ("/", name, NULL);
143 fdata->original_path = fdata->full_path + 1;
144 }
145
146 if (fdata->dir && (name[strlen (name) - 1] != '/')) {
147 char *old_full_path = fdata->full_path;
148 fdata->full_path = g_strconcat (old_full_path, "/", NULL);
149 g_free (old_full_path);
150 fdata->original_path = g_strdup (name);
151 fdata->free_original_path = TRUE;
152 }
153 g_free (name);
154
155 if (fields[1] != NULL)
156 fdata->link = g_strcompress (fields[1]);
157 g_strfreev (fields);
158
159 if (fdata->dir)
160 fdata->name = dir_name_from_path (fdata->full_path);
161 else
162 fdata->name = g_strdup (file_name_from_path (fdata->full_path));
163 fdata->path = remove_level_from_path (fdata->full_path);
164
165 if (*fdata->name == 0)
166 file_data_free (fdata);
167 else
168 fr_command_add_file (comm, fdata);
169}

183static void
184fr_command_cpio_extract (FrCommand *comm,
185 const char *from_file,
186 GList *file_list,
187 const char *dest_dir,
188 gboolean overwrite,
189 gboolean skip_older,
190 gboolean junk_paths)
191{
192 GList *scan;
193 GString *cmd;
194
195 fr_process_begin_command (comm->process, "sh");
196 if (dest_dir != NULL)
197 fr_process_set_working_dir (comm->process, dest_dir);
198 fr_process_add_arg (comm->process, "-c");
199
200 cmd = g_string_new ("cpio -idu --no-absolute-filenames ");
201 for (scan = file_list; scan; scan = scan->next) {
202 char *filepath = scan->data;
203 char *filename;
204
205 if (filepath[0] == '/')
206 filename = g_shell_quote (filepath + 1);
207 else
208 filename = g_shell_quote (filepath);
209 g_string_append (cmd, filename);
210 g_string_append (cmd, " ");
211
212 g_free (filename);
213 }
214 g_string_append (cmd, " < ");
215 g_string_append (cmd, comm->e_filename);
216 fr_process_add_arg (comm->process, cmd->str);
217 g_string_free (cmd, TRUE);
218
219 fr_process_end_command (comm->process);
220}
