80static void
81caja_file_changes_queue_add_common (CajaFileChangesQueue *queue,
82 CajaFileChange *new_item)
83{
84 /* enqueue the new queue item while locking down the list */
85 g_mutex_lock (&queue->mutex);
86
87 queue->head = g_list_prepend (queue->head, new_item);
88 if (queue->tail == NULL)
89 queue->tail = queue->head;
90
91 g_mutex_unlock (&queue->mutex);
92}

184static CajaFileChange *
185caja_file_changes_queue_get_change (CajaFileChangesQueue *queue)
186{
187 GList *new_tail;
188 CajaFileChange *result;
189
190 g_assert (queue != NULL);
191
192 /* dequeue the tail item while locking down the list */
193 g_mutex_lock (&queue->mutex);
194
195 if (queue->tail == NULL)
196 {
197 result = NULL;
198 }
199 else
200 {
201 new_tail = queue->tail->prev;
202 result = queue->tail->data;
203 queue->head = g_list_remove_link (queue->head,
204 queue->tail);
205 g_list_free_1 (queue->tail);
206 queue->tail = new_tail;
207 }
208
209 g_mutex_unlock (&queue->mutex);
210
211 return result;
212}

219static void
220pairs_list_free (GList *pairs)
221{
222 GList *p;
223 GFilePair *pair = NULL;
224
225 /* deep delete the list of pairs */
226
227 for (p = pairs; p != NULL; p = p->next)
228 {
229 /* delete the strings in each pair */
230 pair = p->data;
231 g_object_unref (pair->from);
232 g_object_unref (pair->to);
233 }
234
235 /* delete the list and the now empty pair structs */
236 g_list_free_full (pairs, g_free);
237}

254/* go through changes in the change queue, send ones with the same kind
255 * in a list to the different caja_directory_notify calls
256 */
257void
258caja_file_changes_consume_changes (gboolean consume_all)
259{
260 CajaFileChange *change;
261 GList *additions, *changes, *deletions, *moves;
262 GList *position_set_requests;
263 GFilePair *pair;
264 CajaFileChangesQueuePosition *position_set;
265 guint chunk_count;
266 CajaFileChangesQueue *queue;
267 gboolean flush_needed;
268
269 additions = NULL;
270 changes = NULL;
271 deletions = NULL;
272 moves = NULL;
273 position_set_requests = NULL;
274
275 queue = caja_file_changes_queue_get();
276
277 /* Consume changes from the queue, stuffing them into one of three lists,
278 * keep doing it while the changes are of the same kind, then send them off.
279 * This is to ensure that the changes get sent off in the same order that they
280 * arrived.
281 */
282 for (chunk_count = 0; ; chunk_count++)
283 {
284 change = caja_file_changes_queue_get_change (queue);
285
286 /* figure out if we need to flush the pending changes that we collected sofar */
287
288 if (change == NULL)
289 {
290 flush_needed = TRUE;
291 /* no changes left, flush everything */
292 }
293 else
294 {
295 flush_needed = additions != NULL
296 && change->kind != CHANGE_FILE_ADDED
297 && change->kind != CHANGE_POSITION_SET
298 && change->kind != CHANGE_POSITION_REMOVE;
299
300 flush_needed |= changes != NULL
301 && change->kind != CHANGE_FILE_CHANGED;
302
303 flush_needed |= moves != NULL
304 && change->kind != CHANGE_FILE_MOVED
305 && change->kind != CHANGE_POSITION_SET
306 && change->kind != CHANGE_POSITION_REMOVE;
307
308 flush_needed |= deletions != NULL
309 && change->kind != CHANGE_FILE_REMOVED;
310
311 flush_needed |= position_set_requests != NULL
312 && change->kind != CHANGE_POSITION_SET
313 && change->kind != CHANGE_POSITION_REMOVE
314 && change->kind != CHANGE_FILE_ADDED
315 && change->kind != CHANGE_FILE_MOVED;
316
317 flush_needed |= !consume_all && chunk_count >= CONSUME_CHANGES_MAX_CHUNK;
318 /* we have reached the chunk maximum */
319 }
320
321 if (flush_needed)
322 {
323 /* Send changes we collected off.
324 * At one time we may only have one of the lists
325 * contain changes.
326 */
327
328 if (deletions != NULL)
329 {
330 deletions = g_list_reverse (deletions);
331 caja_directory_notify_files_removed (deletions);
332 g_list_free_full (deletions, g_object_unref);
333 deletions = NULL;
334 }
335 if (moves != NULL)
336 {
337 moves = g_list_reverse (moves);
338 caja_directory_notify_files_moved (moves);
339 pairs_list_free (moves);
340 moves = NULL;
341 }
342 if (additions != NULL)
343 {
344 additions = g_list_reverse (additions);
345 caja_directory_notify_files_added (additions);
346 g_list_free_full (additions, g_object_unref);
347 additions = NULL;
348 }
349 if (changes != NULL)
350 {
351 changes = g_list_reverse (changes);
352 caja_directory_notify_files_changed (changes);
353 g_list_free_full (changes, g_object_unref);
354 changes = NULL;
355 }
356 if (position_set_requests != NULL)
357 {
358 position_set_requests = g_list_reverse (position_set_requests);
359 caja_directory_schedule_position_set (position_set_requests);
360 position_set_list_free (position_set_requests);
361 position_set_requests = NULL;
362 }
363 }
364
365 if (change == NULL)
366 {
367 /* we are done */
368 return;
369 }
370
371 /* add the new change to the list */
372 switch (change->kind)
373 {
374 case CHANGE_FILE_ADDED:
375 additions = g_list_prepend (additions, change->from);
376 break;
377
378 case CHANGE_FILE_CHANGED:
379 changes = g_list_prepend (changes, change->from);
380 break;
381
382 case CHANGE_FILE_REMOVED:
383 deletions = g_list_prepend (deletions, change->from);
384 break;
385
386 case CHANGE_FILE_MOVED:
387 pair = g_new (GFilePair, 1);
388 pair->from = change->from;
389 pair->to = change->to;
390 moves = g_list_prepend (moves, pair);
391 break;
392
393 case CHANGE_POSITION_SET:
394 position_set = g_new (CajaFileChangesQueuePosition, 1);
395 position_set->location = change->from;
396 position_set->set = TRUE;
397 position_set->point = change->point;
398 position_set->screen = change->screen;
399 position_set_requests = g_list_prepend (position_set_requests,
400 position_set);
401 break;
402
403 case CHANGE_POSITION_REMOVE:
404 position_set = g_new (CajaFileChangesQueuePosition, 1);
405 position_set->location = change->from;
406 position_set->set = FALSE;
407 position_set_requests = g_list_prepend (position_set_requests,
408 position_set);
409 break;
410
411 default:
412 g_assert_not_reached ();
413 break;
414 }
415
416 g_free (change);
417 }
