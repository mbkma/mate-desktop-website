74static GtkWidget *
75get_license_uri_widget (const gchar *uri)
76{
77 GtkWidget *label;
78 gchar *checked_uri;
79 gchar *markup;
80
81 label = gtk_label_new (NULL);
82 gtk_widget_set_margin_start (label, 12);
83 g_object_set (G_OBJECT (label),
84 "xalign", 0.0,
85 "width_chars", 25,
86 "selectable", TRUE,
87 "ellipsize", PANGO_ELLIPSIZE_END,
88 NULL);
89
90 checked_uri = g_uri_parse_scheme (uri);
91 if (checked_uri) {
92 markup = g_markup_printf_escaped ("<a href=\"%s\">%s</a>", uri, uri);
93 gtk_label_set_markup (GTK_LABEL (label), markup);
94 g_free (markup);
95 g_free (checked_uri);
96 } else {
97 gtk_label_set_text (GTK_LABEL (label), uri);
98 }
99
100 return label;
101}

124void
125ev_properties_license_set_license (EvPropertiesLicense *properties,
126 EvDocumentLicense *license)
127{
128 const gchar *text = ev_document_license_get_text (license);
129 const gchar *uri = ev_document_license_get_uri (license);

131
132 if (text) {
133 ev_properties_license_add_section (properties,

138 if (uri) {

140 _("Text License"),
141 get_license_uri_widget (uri));
142 }
143
144 if (web_statement) {
145 ev_properties_license_add_section (properties,
146 _("Further Information"),
