109static void
110pluma_notebook_dispose (GObject *object)
111{
112 PlumaNotebook *notebook = PLUMA_NOTEBOOK (object);
113
114 if (!notebook->priv->destroy_has_run)
115 {
116 GList *children, *l;
117
118 children = gtk_container_get_children (GTK_CONTAINER (notebook));
119
120 for (l = children; l != NULL; l = g_list_next (l))
121 {
122 pluma_notebook_remove_tab (notebook,
123 PLUMA_TAB (l->data));
124 }
125
126 g_list_free (children);
127 notebook->priv->destroy_has_run = TRUE;
128 }
129
130 g_clear_object (&notebook->priv->editor_settings);
131
132 G_OBJECT_CLASS (pluma_notebook_parent_class)->dispose (object);
133}

192static PlumaNotebook *
193find_notebook_at_pointer (gint abs_x, gint abs_y)
194{
195 GdkWindow *win_at_pointer;
196 GdkWindow *toplevel_win;
197 gpointer toplevel = NULL;
198 GdkSeat *seat;
199 GdkDevice *device;
200 gint x, y;
201
202 /* FIXME multi-head */
203 seat = gdk_display_get_default_seat (gdk_display_get_default ());
204 device = gdk_seat_get_pointer (seat);
205 win_at_pointer = gdk_device_get_window_at_position (device, &x, &y);
206
207 if (win_at_pointer == NULL)
208 {
209 /* We are outside all windows of the same application */
210 return NULL;
211 }
212
213 toplevel_win = gdk_window_get_toplevel (win_at_pointer);
214
215 /* get the GtkWidget which owns the toplevel GdkWindow */
216 gdk_window_get_user_data (toplevel_win, &toplevel);
217
218 /* toplevel should be an PlumaWindow */
219 if ((toplevel != NULL) &&
220 PLUMA_IS_WINDOW (toplevel))
221 {
222 return PLUMA_NOTEBOOK (_pluma_window_get_notebook
223 (PLUMA_WINDOW (toplevel)));
224 }
225
226 /* We are outside all windows containing a notebook */
227 return NULL;
228}

230static gboolean
231is_in_notebook_window (PlumaNotebook *notebook,
232 gint abs_x,
233 gint abs_y)
234{
235 PlumaNotebook *nb_at_pointer;
236
237 g_return_val_if_fail (notebook != NULL, FALSE);
238
239 nb_at_pointer = find_notebook_at_pointer (abs_x, abs_y);
240
241 return (nb_at_pointer == notebook);
242}

244static gint
245find_tab_num_at_pos (PlumaNotebook *notebook,
246 gint abs_x,
247 gint abs_y)
248{
249 GtkPositionType tab_pos;
250 int page_num = 0;
251 GtkNotebook *nb = GTK_NOTEBOOK (notebook);
252 GtkWidget *page;
253
254 tab_pos = gtk_notebook_get_tab_pos (GTK_NOTEBOOK (notebook));
255
256 /* For some reason unfullscreen + quick click can
257 cause a wrong click event to be reported to the tab */
258 if (!is_in_notebook_window (notebook, abs_x, abs_y))
259 {
260 return NOT_IN_APP_WINDOWS;
261 }
262
263 while ((page = gtk_notebook_get_nth_page (nb, page_num)) != NULL)
264 {
265 GtkAllocation allocation;
266 GtkWidget *tab;
267 gint max_x, max_y;
268 gint x_root, y_root;
269
270 tab = gtk_notebook_get_tab_label (nb, page);
271 g_return_val_if_fail (tab != NULL, AFTER_ALL_TABS);
272
273 if (!gtk_widget_get_mapped (tab))
274 {
275 ++page_num;
276 continue;
277 }
278
279 gdk_window_get_origin (GDK_WINDOW (gtk_widget_get_window (tab)),
280 &x_root, &y_root);
281
282 gtk_widget_get_allocation(tab, &allocation);
283
284 max_x = x_root + allocation.x + allocation.width;
285 max_y = y_root + allocation.y + allocation.height;
286
287 if (((tab_pos == GTK_POS_TOP) ||
288 (tab_pos == GTK_POS_BOTTOM)) &&
289 (abs_x <= max_x))
290 {
291 return page_num;
292 }
293 else if (((tab_pos == GTK_POS_LEFT) ||
294 (tab_pos == GTK_POS_RIGHT)) &&
295 (abs_y <= max_y))
296 {
297 return page_num;
298 }
299
300 ++page_num;
301 }
302
303 return AFTER_ALL_TABS;
304}

306static gint
307find_notebook_and_tab_at_pos (gint abs_x,
308 gint abs_y,
309 PlumaNotebook **notebook,
310 gint *page_num)
311{
312 *notebook = find_notebook_at_pointer (abs_x, abs_y);
313 if (*notebook == NULL)
314 {
315 return NOT_IN_APP_WINDOWS;
316 }
317
318 *page_num = find_tab_num_at_pos (*notebook, abs_x, abs_y);
319
320 if (*page_num < 0)
321 {
322 return *page_num;
323 }
324 else
325 {
326 return 0;
327 }

342void
343pluma_notebook_move_tab (PlumaNotebook *src,
344 PlumaNotebook *dest,
345 PlumaTab *tab,
346 gint dest_position)
347{
348 g_return_if_fail (PLUMA_IS_NOTEBOOK (src));
349 g_return_if_fail (PLUMA_IS_NOTEBOOK (dest));
350 g_return_if_fail (src != dest);
351 g_return_if_fail (PLUMA_IS_TAB (tab));
352
353 /* make sure the tab isn't destroyed while we move it */
354 g_object_ref (tab);
355 pluma_notebook_remove_tab (src, tab);
356 pluma_notebook_add_tab (dest, tab, dest_position, TRUE);
357 g_object_unref (tab);
358}

371void
372pluma_notebook_reorder_tab (PlumaNotebook *src,
373 PlumaTab *tab,
374 gint dest_position)
375{
376 gint old_position;
377
378 g_return_if_fail (PLUMA_IS_NOTEBOOK (src));
379 g_return_if_fail (PLUMA_IS_TAB (tab));
380
381 old_position = gtk_notebook_page_num (GTK_NOTEBOOK (src),
382 GTK_WIDGET (tab));
383
384 if (old_position == dest_position)
385 return;
386
387 gtk_notebook_reorder_child (GTK_NOTEBOOK (src),
388 GTK_WIDGET (tab),
389 dest_position);
390
391 if (!src->priv->drag_in_progress)
392 {
393 g_signal_emit (G_OBJECT (src),
394 signals[TABS_REORDERED],
395 0);
396 }
397}

399static void
400drag_start (PlumaNotebook *notebook,
401 GdkEvent *event)
402{
403 GdkSeat *seat;
404 GdkDevice *device;
405 GdkDisplay *display;
406
407 display = gtk_widget_get_display (GTK_WIDGET (notebook));
408 seat = gdk_display_get_default_seat (display);
409 device = gdk_seat_get_pointer (seat);
410
411 if (!leftdown) return;
412
413 notebook->priv->drag_in_progress = TRUE;
414
415 /* get a new cursor, if necessary */
416 /* FIXME multi-head */
417 if (cursor == NULL)
418 cursor = gdk_cursor_new_for_display (display, GDK_FLEUR);
419
420 /* grab the pointer */
421 gtk_grab_add (GTK_WIDGET (notebook));
422
423 /* FIXME multi-head */
424 if (!gdk_display_device_is_grabbed (display, device))
425 {
426 gdk_seat_grab (seat,
427 gtk_widget_get_window (GTK_WIDGET (notebook)),
428 GDK_SEAT_CAPABILITY_POINTER,
429 FALSE,
430 cursor,
431 event,
432 NULL,
433 NULL);
434 }
435}

437static void
438drag_stop (PlumaNotebook *notebook)
439{
440 if (notebook->priv->drag_in_progress)
441 {
442 g_signal_emit (G_OBJECT (notebook),
443 signals[TABS_REORDERED],
444 0);
445 }
446
447 notebook->priv->drag_in_progress = FALSE;
448#if GLIB_CHECK_VERSION(2,62,0)
449 g_clear_signal_handler (&notebook->priv->motion_notify_handler_id,
450 notebook);
451#else
452 if (notebook->priv->motion_notify_handler_id != 0)
453 {
454 g_signal_handler_disconnect (G_OBJECT (notebook),
455 notebook->priv->motion_notify_handler_id);
456 notebook->priv->motion_notify_handler_id = 0;
457 }
458#endif
459}

464static void
465move_current_tab (PlumaNotebook *notebook,
466 gint dest_position)
467{
468 gint cur_page_num;
469
470 cur_page_num = gtk_notebook_get_current_page (GTK_NOTEBOOK (notebook));
471
472 if (dest_position != cur_page_num)
473 {
474 GtkWidget *cur_tab;
475
476 cur_tab = gtk_notebook_get_nth_page (GTK_NOTEBOOK (notebook),
477 cur_page_num);
478
479 pluma_notebook_reorder_tab (PLUMA_NOTEBOOK (notebook),
480 PLUMA_TAB (cur_tab),
481 dest_position);
482 }

485static gboolean
486motion_notify_cb (PlumaNotebook *notebook,
487 GdkEventMotion *event,
488 gpointer data)
489{
490 PlumaNotebook *dest;
491 gint page_num;
492 gint result;
493
494 if (notebook->priv->drag_in_progress == FALSE)
495 {
496 if (notebook->priv->tab_drag_and_drop_enabled == FALSE)
497 return FALSE;
498
499 if (gtk_drag_check_threshold (GTK_WIDGET (notebook),
500 notebook->priv->x_start,
501 notebook->priv->y_start,
502 event->x_root,
503 event->y_root))
504 {
505 if (drag_ready)
506 drag_start (notebook, (GdkEvent *) event);
507 else
508 drag_stop (notebook);
509
510 return TRUE;
511 }
512
513 return FALSE;
514 }
515
516 result = find_notebook_and_tab_at_pos ((gint)event->x_root,
517 (gint)event->y_root,
518 &dest,
519 &page_num);
520
521 if (result != NOT_IN_APP_WINDOWS)
522 {
523 if (dest != notebook)
524 {
525 move_current_tab_to_another_notebook (notebook,
526 dest,
527 event,
528 page_num);
529 }
530 else
531 {
532 g_return_val_if_fail (page_num >= -1, FALSE);
533 move_current_tab (notebook, page_num);
534 }
535 }
536
537 return FALSE;
538}

540static void
541move_current_tab_to_another_notebook (PlumaNotebook *src,
542 PlumaNotebook *dest,
543 GdkEventMotion *event,
544 gint dest_position)
545{
546 PlumaTab *tab;
547 gint cur_page;
548 GdkSeat *seat;
549 GdkDevice *device;
550 GdkDisplay *display;
551
552 display = gtk_widget_get_display (GTK_WIDGET (GTK_NOTEBOOK (src)));
553 seat = gdk_display_get_default_seat (display);
554 device = gdk_seat_get_pointer (seat);
555
556 /* This is getting tricky, the tab was dragged in a notebook
557 * in another window of the same app, we move the tab
558 * to that new notebook, and let this notebook handle the
559 * drag
560 */
561 g_return_if_fail (PLUMA_IS_NOTEBOOK (dest));
562 g_return_if_fail (dest != src);
563
564 cur_page = gtk_notebook_get_current_page (GTK_NOTEBOOK (src));
565 tab = PLUMA_TAB (gtk_notebook_get_nth_page (GTK_NOTEBOOK (src),
566 cur_page));
567
568 /* stop drag in origin window */
569 /* ungrab the pointer if it's grabbed */
570 drag_stop (src);
571 if (gdk_display_device_is_grabbed (display, device))
572 {
573 gdk_seat_ungrab (seat);
574 }
575 gtk_grab_remove (GTK_WIDGET (src));
576
577 pluma_notebook_move_tab (src, dest, tab, dest_position);
578
579 /* start drag handling in dest notebook */
580 dest->priv->motion_notify_handler_id =
581 g_signal_connect (dest, "motion-notify-event",
582 G_CALLBACK (motion_notify_cb),
583 NULL);
584
585 drag_start (dest, (GdkEvent *) event);
586}

588static gboolean
589button_release_cb (PlumaNotebook *notebook,
590 GdkEventButton *event,
591 gpointer data)
592{
593 GdkSeat *seat;
594 GdkDevice *device;
595 GdkDisplay *display;
596
597 display = gtk_widget_get_display (GTK_WIDGET (GTK_NOTEBOOK (notebook)));
598 seat = gdk_display_get_default_seat (display);
599 device = gdk_seat_get_pointer (seat);
600
601 if (event->button == 1) leftdown = FALSE;
602
603 if (notebook->priv->drag_in_progress)
604 {
605 gint cur_page_num;
606 GtkWidget *cur_page;
607
608 cur_page_num = gtk_notebook_get_current_page (GTK_NOTEBOOK (notebook));
609 cur_page = gtk_notebook_get_nth_page (GTK_NOTEBOOK (notebook),
610 cur_page_num);
611
612 /* CHECK: I don't follow the code here -- Paolo */
613 if (!is_in_notebook_window (notebook, event->x_root, event->y_root) &&
614 (gtk_notebook_get_n_pages (GTK_NOTEBOOK (notebook)) > 1))
615 {
616 /* Tab was detached */
617 g_signal_emit (G_OBJECT (notebook),
618 signals[TAB_DETACHED],
619 0,
620 cur_page);
621 }
622
623 /* ungrab the pointer if it's grabbed */
624 if (gdk_display_device_is_grabbed (display, device))
625 {
626 gdk_seat_ungrab (seat);
627 }
628 gtk_grab_remove (GTK_WIDGET (notebook));
629 }
630
631 /* This must be called even if a drag isn't happening */
632 drag_stop (notebook);
633 drag_ready = FALSE;
634
635 return FALSE;
636}

638static gboolean
639button_press_cb (PlumaNotebook *notebook,
640 GdkEventButton *event,
641 gpointer data)
642{
643 static gboolean newfile = FALSE;
644 static gint tab1click = -1;
645
646 gint tab_clicked;
647
648 if (notebook->priv->drag_in_progress)
649 return TRUE;
650
651 tab_clicked = find_tab_num_at_pos (notebook,
652 event->x_root,
653 event->y_root);
654
655 if ((event->button == 1) &&
656 (event->type == GDK_BUTTON_PRESS) &&
657 (tab_clicked >= 0))
658 {
659 notebook->priv->x_start = event->x_root;
660 notebook->priv->y_start = event->y_root;
661
662 notebook->priv->motion_notify_handler_id =
663 g_signal_connect (notebook, "motion-notify-event",
664 G_CALLBACK (motion_notify_cb),
665 NULL);
666 }
667 else if ((event->type == GDK_BUTTON_PRESS) &&
668 (event->button == 3 || event->button == 2))
669 {
670 if (tab_clicked == -1)
671 {
672 // CHECK: do we really need it?
673
674 /* consume event, so that we don't pop up the context menu when
675 * the mouse if not over a tab label
676 */
677 return TRUE;
678 }
679 else
680 {
681 if (leftdown) {
682 leftdown = FALSE;
683 return TRUE;
684 }
685
686 /* Switch to the page the mouse is over, but don't consume the event */
687 gtk_notebook_set_current_page (GTK_NOTEBOOK (notebook),
688 tab_clicked);
689 }
690 }
691
692 if (event->button == 1)
693 {
694 if (event->type == GDK_BUTTON_PRESS)
695 {
696 tab1click = gtk_notebook_get_current_page (GTK_NOTEBOOK (notebook));
697
698 if (newfile_ready)
699 newfile = (tab_clicked == -1);
700 else
701 newfile = FALSE;
702 }
703 else if (event->type == GDK_2BUTTON_PRESS)
704 {
705 if ((tab1click != gtk_notebook_get_current_page (GTK_NOTEBOOK (notebook))) ||
706 (tab_clicked >= 0) || ((tab_clicked == -1) && (!newfile)) || (!leftdown))
707 return TRUE;
708
709 newfile = FALSE;
710 }
711
712 leftdown = TRUE;
713 }
714
715 return FALSE;
716}

758static void
759pluma_notebook_switch_page_cb (GtkNotebook *notebook,
760 GtkWidget *page,
761 guint page_num,
762 gpointer data)
763{
764 PlumaNotebook *nb = PLUMA_NOTEBOOK (notebook);
765 GtkWidget *child;
766 PlumaView *view;
767
768 child = gtk_notebook_get_nth_page (notebook, page_num);
769
770 /* Remove the old page, we dont want to grow unnecessarily
771 * the list */
772 if (nb->priv->focused_pages)
773 {
774 nb->priv->focused_pages =
775 g_list_remove (nb->priv->focused_pages, child);
776 }
777
778 nb->priv->focused_pages = g_list_append (nb->priv->focused_pages,
779 child);
780
781 /* give focus to the view */
782 view = pluma_tab_get_view (PLUMA_TAB (child));
783 gtk_widget_grab_focus (GTK_WIDGET (view));
784}

786/*
787 * update_tabs_visibility: Hide tabs if there is only one tab
788 * and the pref is not set.
789 */
790static void
791update_tabs_visibility (PlumaNotebook *nb)
792{
793 gboolean show_tabs;
794 guint num;
795
796 num = gtk_notebook_get_n_pages (GTK_NOTEBOOK (nb));
797
798 show_tabs = (g_settings_get_boolean (nb->priv->editor_settings, PLUMA_SETTINGS_SHOW_SINGLE_TAB) || (num > 1));
799
800 if (g_settings_get_boolean (nb->priv->editor_settings, PLUMA_SETTINGS_SHOW_TABS_WITH_SIDE_PANE))
801 gtk_notebook_set_show_tabs (GTK_NOTEBOOK (nb), show_tabs);
802 else
803 {
804 gboolean visible = g_settings_get_boolean (nb->priv->editor_settings, PLUMA_SETTINGS_SIDE_PANE_VISIBLE);
805
806 if (visible)
807 gtk_notebook_set_show_tabs (GTK_NOTEBOOK (nb), FALSE);
808 else
809 gtk_notebook_set_show_tabs (GTK_NOTEBOOK (nb), show_tabs);
810 }
811}

855/*
856 * We need to override this because when we don't show the tabs, like in
857 * fullscreen we need to have wrap around too
858 */
859static gboolean
860pluma_notebook_change_current_page (GtkNotebook *notebook,
861 gint offset)
862{
863 gboolean wrap_around;
864 gint current;
865
866 current = gtk_notebook_get_current_page (notebook);
867
868 if (current != -1)
869 {
870 current = current + offset;
871
872 g_object_get (gtk_widget_get_settings (GTK_WIDGET (notebook)),
873 "gtk-keynav-wrap-around", &wrap_around,
874 NULL);
875
876 if (wrap_around)
877 {
878 if (current < 0)
879 {
880 current = gtk_notebook_get_n_pages (notebook) - 1;
881 }
882 else if (current >= gtk_notebook_get_n_pages (notebook))
883 {
884 current = 0;
885 }
886 }
887
888 gtk_notebook_set_current_page (notebook, current);
889 }
890 else
891 {
892 gtk_widget_error_bell (GTK_WIDGET (notebook));
893 }
894
895 return TRUE;
896}

999smart_tab_switching_on_closure (PlumaNotebook *nb,
1000 PlumaTab *tab)
1001{
1002 gboolean jump_to;
1003
1004 jump_to = GPOINTER_TO_INT (g_object_get_data

1012 {

1023 page_num);
