105static void
106pluma_taglist_plugin_activate (PlumaWindowActivatable *activatable)
107{
108 PlumaTaglistPluginPrivate *priv;
109 PlumaWindow *window;
110 PlumaPanel *side_panel;
111 gchar *data_dir;
112
113 pluma_debug (DEBUG_PLUGINS);
114
115 priv = PLUMA_TAGLIST_PLUGIN (activatable)->priv;
116 window = PLUMA_WINDOW (priv->window);
117 side_panel = pluma_window_get_side_panel (window);
118
119 data_dir = peas_extension_base_get_data_dir (PEAS_EXTENSION_BASE (activatable));
120 priv->taglist_panel = pluma_taglist_plugin_panel_new (window, data_dir);
121 g_free (data_dir);
122
123 pluma_panel_add_item_with_icon (side_panel,
124 priv->taglist_panel,
125 _("Tags"),
126 "list-add");
127}

146static void
147pluma_taglist_plugin_update_state (PlumaWindowActivatable *activatable)
148{
149 PlumaTaglistPluginPrivate *priv;
150 PlumaWindow *window;
151 PlumaView *view;
152
153 pluma_debug (DEBUG_PLUGINS);
154
155 priv = PLUMA_TAGLIST_PLUGIN (activatable)->priv;
156 window = PLUMA_WINDOW (priv->window);
157 view = pluma_window_get_active_view (window);
158
159 gtk_widget_set_sensitive (priv->taglist_panel,
160 (view != NULL) &&
161 gtk_text_view_get_editable (GTK_TEXT_VIEW (view)));

164static void
165pluma_taglist_plugin_set_property (GObject *object,
166 guint prop_id,
167 const GValue *value,
168 GParamSpec *pspec)
169{
170 PlumaTaglistPlugin *plugin = PLUMA_TAGLIST_PLUGIN (object);
171
172 switch (prop_id)

174 case PROP_WINDOW:
175 plugin->priv->window = PLUMA_WINDOW (g_value_dup_object (value));
