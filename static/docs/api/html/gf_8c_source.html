78static int gf_read_bitmap(FILE *p, DviFontChar *ch)
79{
80 int op;
81 int min_n, max_n;
82 int min_m, max_m;
83 int paint_switch;
84 int x, y;
85 int bpl;
86 Int32 par;
87 BmUnit *line;
88 BITMAP *map;
89
90 fseek(p, (long)ch->offset, SEEK_SET);
91 op = fuget1(p);
92 if(op == GF_BOC) {
93 /* skip character code */
94 fuget4(p);
95 /* skip pointer */
96 fuget4(p);
97 min_m = fsget4(p);
98 max_m = fsget4(p);
99 min_n = fsget4(p);
100 max_n = fsget4(p);
101 } else if(op == GF_BOC1) {
102 /* skip character code */
103 fuget1(p);
104 min_m = fuget1(p); /* this is max_m - min_m */
105 max_m = fuget1(p);
106 min_n = fuget1(p); /* this is max_n - min_n */
107 max_n = fuget1(p);
108 min_m = max_m - min_m;
109 min_n = max_n - min_n;
110 } else {
111 mdvi_error(_("GF: invalid opcode %d in character %d\n"),
112 op, ch->code);
113 return -1;
114 }
115
116 ch->x = -min_m;
117 ch->y = max_n;
118 ch->width = max_m - min_m + 1;
119 ch->height = max_n - min_n + 1;
120 map = bitmap_alloc(ch->width, ch->height);
121
122 ch->glyph.data = map;
123 ch->glyph.x = ch->x;
124 ch->glyph.y = ch->y;
125 ch->glyph.w = ch->width;
126 ch->glyph.h = ch->height;
127
128#define COLOR(x) ((x) ? "BLACK" : "WHITE")
129
130 paint_switch = WHITE;
131 x = y = 0;
132 line = map->data;
133 bpl = map->stride;
134 DEBUG((DBG_BITMAPS, "(gf) reading character %d\n", ch->code));
135 while((op = fuget1(p)) != GF_EOC) {
136 Int32 n;
137
138 if(feof(p))
139 break;
140 if(op == GF_PAINT0) {
141 DEBUG((DBG_BITMAPS, "(gf) Paint0 %s -> %s\n",
142 COLOR(paint_switch), COLOR(!paint_switch)));
143 paint_switch = !paint_switch;
144 } else if(op <= GF_PAINT3) {
145 if(op < GF_PAINT1)
146 par = op;
147 else
148 par = fugetn(p, op - GF_PAINT1 + 1);
149 if(y >= ch->height || x + par >= ch->width)
150 goto toobig;
151 /* paint everything between columns x and x + par - 1 */
152 DEBUG((DBG_BITMAPS, "(gf) Paint %d %s from (%d,%d)\n",
153 par, COLOR(paint_switch), x, y));
154 if(paint_switch == BLACK)
155 bitmap_paint_bits(line + (x / BITMAP_BITS),
156 x % BITMAP_BITS, par);
157 paint_switch = !paint_switch;
158 x += par;
159 } else if(op >= GF_NEW_ROW_0 && op <= GF_NEW_ROW_MAX) {
160 y++;
161 line = bm_offset(line, bpl);
162 x = op - GF_NEW_ROW_0;
163 paint_switch = BLACK;
164 DEBUG((DBG_BITMAPS, "(gf) new_row_%d\n", x));
165 } else switch(op) {
166 case GF_SKIP0:
167 y++;
168 line = bm_offset(line, bpl);
169 x = 0;
170 paint_switch = WHITE;
171 DEBUG((DBG_BITMAPS, "(gf) skip_0\n"));
172 break;
173 case GF_SKIP1:
174 case GF_SKIP2:
175 case GF_SKIP3:
176 par = fugetn(p, op - GF_SKIP1 + 1);
177 y += par + 1;
178 line = bm_offset(line, (par + 1) * bpl);
179 x = 0;
180 paint_switch = WHITE;
181 DEBUG((DBG_BITMAPS, "(gf) skip_%d\n", op - GF_SKIP1));
182 break;
183 case GF_XXX1:
184 case GF_XXX2:
185 case GF_XXX3:
186 case GF_XXX4: {
187#ifndef NODEBUG
188 char *s;
189
190 s = read_string(p, op - GF_XXX1 + 1, NULL, 0);
191 DEBUG((DBG_SPECIAL, "(gf) Character %d: Special \"%s\"\n",
192 ch->code, s));
193 mdvi_free(s);
194#else
195 n = fugetn(p, op - GF_XXX1 + 1);
196 fseek(p, (long)n, SEEK_CUR);
197#endif
198 break;
199 }
200 case GF_YYY:
201 n = fuget4(p);
202 DEBUG((DBG_SPECIAL, "(gf) Character %d: MF special %u\n",
203 ch->code, n));
204 break;
205 case GF_NOOP:
206 DEBUG((DBG_BITMAPS, "(gf) no_op\n"));
207 break;
208 default:
209 mdvi_error(_("(gf) Character %d: invalid opcode %d\n"),
210 ch->code, op);
211 goto error;
212 }
213 /* chech that we're still inside the bitmap */
214 if(x > ch->width || y > ch->height)
215 goto toobig;
216 DEBUG((DBG_BITMAPS, "(gf) curr_loc @ (%d,%d)\n", x, y));
217 }
218
219 if(op != GF_EOC)
220 goto error;
221 DEBUG((DBG_BITMAPS, "(gf) end of character %d\n", ch->code));
222 return 0;
223
224toobig:
225 mdvi_error(_("(gf) character %d has an incorrect bounding box\n"),
226 ch->code);
227error:
228 bitmap_destroy(map);
229 ch->glyph.data = NULL;
230 return -1;
231}

233static int gf_load_font(DviParams *unused, DviFont *font)
234{
235 int i;
236 int n;
237 int loc;
238 int hic;
239 FILE *p;
240 Int32 word;
241 int op;
242 long alpha, beta, z;
243#ifndef NODEBUG
244 char s[256];
245#endif
246
247 p = font->in;
248
249 /* check preamble */
250 loc = fuget1(p); hic = fuget1(p);
251 if(loc != GF_PRE || hic != GF_ID)
252 goto badgf;
253 loc = fuget1(p);
254#ifndef NODEBUG
255 for(i = 0; i < loc; i++)
256 s[i] = fuget1(p);
257 s[i] = 0;
258 DEBUG((DBG_FONTS, "(gf) %s: %s\n", font->fontname, s));
259#else
260 fseek(p, (long)loc, SEEK_CUR);
261#endif
262 /* now read character locators in postamble */
263 if(fseek(p, (long)-1, SEEK_END) == -1)
264 return -1;
265
266 n = 0;
267 while((op = fuget1(p)) == GF_TRAILER) {
268 if(fseek(p, (long)-2, SEEK_CUR) < 0)
269 break;
270 n++;
271 }
272 if(op != GF_ID || n < 4)
273 goto badgf;
274 /* get the pointer to the postamble */
275 fseek(p, (long)-5, SEEK_CUR);
276 op = fuget4(p);
277 /* jump to it */
278 fseek(p, (long)op, SEEK_SET);
279 if(fuget1(p) != GF_POST)
280 goto badgf;
281 /* skip pointer to last EOC */
282 fuget4(p);
283 /* get the design size */
284 font->design = fuget4(p);
285 /* the checksum */
286 word = fuget4(p);
287 if(word && font->checksum && font->checksum != word) {
288 mdvi_warning(_("%s: bad checksum (expected %u, found %u)\n"),
289 font->fontname, font->checksum, word);
290 } else if(!font->checksum)
291 font->checksum = word;
292 /* skip pixels per point ratio */
293 fuget4(p);
294 fuget4(p);
295 font->chars = xnalloc(DviFontChar, 256);
296 for(loc = 0; loc < 256; loc++)
297 font->chars[loc].offset = 0;
298 /* skip glyph "bounding box" */
299 fseek(p, (long)16, SEEK_CUR);
300 loc = 256;
301 hic = -1;
302 TFMPREPARE(font->scale, z, alpha, beta);
303 while((op = fuget1(p)) != GF_POST_POST) {
304 DviFontChar *ch;
305 int cc;
306
307 /* get the character code */
308 cc = fuget1(p);
309 if(cc < loc)
310 loc = cc;
311 if(cc > hic)
312 hic = cc;
313 ch = &font->chars[cc];
314 switch(op) {
315 case GF_LOC:
316 fsget4(p); /* skip dx */
317 fsget4(p); /* skip dy */
318 break;
319 case GF_LOC0:
320 fuget1(p); /* skip dx */
321 /* dy assumed 0 */
322 break;
323 default:
324 mdvi_error(_("%s: junk in postamble\n"), font->fontname);
325 goto error;
326 }
327 ch->code = cc;
328 ch->tfmwidth = fuget4(p);
329 ch->tfmwidth = TFMSCALE(ch->tfmwidth, z, alpha, beta);
330 ch->offset = fuget4(p);
331 if(ch->offset == -1)
332 ch->offset = 0;
333 /* initialize the rest of the glyph information */
334 ch->x = 0;
335 ch->y = 0;
336 ch->width = 0;
337 ch->height = 0;
338 ch->glyph.data = NULL;
339 ch->shrunk.data = NULL;
340 ch->grey.data = NULL;
341 ch->flags = 0;
342 ch->loaded = 0;
343 }
344
345 if(op != GF_POST_POST)
346 goto badgf;
347
348 if(loc > 0 || hic < 255) {
349 /* shrink to optimal size */
350 memmove(font->chars, font->chars + loc,
351 (hic - loc + 1) * sizeof(DviFontChar));
352 font->chars = xresize(font->chars,
353 DviFontChar, hic - loc + 1);
354 }
355 font->loc = loc;
356 font->hic = hic;
357
358 return 0;
359
360badgf:
361 mdvi_error(_("%s: File corrupted, or not a GF file\n"), font->fontname);
362error:
363 if(font->chars) {
364 mdvi_free(font->chars);
365 font->chars = NULL;
366 }
367 font->loc = font->hic = 0;
368 return -1;
369}

371static int gf_font_get_glyph(DviParams *params, DviFont *font, int code)
372{
373 DviFontChar *ch;
374
375 if(code < font->loc || code > font->hic || !font->chars)
376 return -1;
377 ch = &font->chars[code - font->loc];
378
379 if(!ch->loaded) {
380 if(ch->offset == 0)
381 return -1;
382 DEBUG((DBG_GLYPHS, "(gf) %s: loading GF glyph for character %d\n",
383 font->fontname, code));
384 if(font->in == NULL && font_reopen(font) < 0)
385 return -1;
386 if(fseek(font->in, ch->offset, SEEK_SET) == -1)
387 return -1;
388 if(gf_read_bitmap(font->in, ch) < 0)
389 return -1;
390 ch->loaded = 1;
391 }
392 return 0;
393}
