52/* caja_keep_last_vertical_box_new:
53 *
54 * Create a new vertical box that clips off items from the end that don't
55 * fit, except the last item, which is always kept. When packing this widget
56 * into another vbox, use TRUE for expand and TRUE for fill or this class's
57 * special clipping magic won't work because this widget's allocation might
58 * be larger than the available space.
59 *
60 * @spacing: Vertical space between items.
61 *
62 * Return value: A new CajaKeepLastVerticalBox
63 */
64GtkWidget *
65caja_keep_last_vertical_box_new (gint spacing)
66{
67 CajaKeepLastVerticalBox *box;
68
69 box = CAJA_KEEP_LAST_VERTICAL_BOX (gtk_widget_new (caja_keep_last_vertical_box_get_type (), NULL));
70
71 gtk_box_set_spacing (GTK_BOX (box), spacing);
72
73 /* If homogeneous is TRUE and there are too many items to fit
74 * naturally, they will be squashed together to fit in the space.
75 * We want the ones that don't fit to be not shown at all, so
76 * we set homogeneous to FALSE.
77 */
78 gtk_box_set_homogeneous (GTK_BOX (box), FALSE);
79
80 return GTK_WIDGET (box);
81}

83static void
84caja_keep_last_vertical_box_size_allocate (GtkWidget *widget,
85 GtkAllocation *allocation)
86{
87 GList *children, *l;
88 GtkAllocation last_child_allocation, child_allocation, tiny_allocation;
89
90 g_return_if_fail (CAJA_IS_KEEP_LAST_VERTICAL_BOX (widget));
91 g_return_if_fail (allocation != NULL);
92
93 GTK_WIDGET_CLASS (caja_keep_last_vertical_box_parent_class)->size_allocate (widget, allocation);
94
95 children = gtk_container_get_children (GTK_CONTAINER (widget));
96 l = g_list_last (children);
97
98 if (l != NULL)
99 {
100 GtkWidget *last_child;
101
102 last_child = l->data;
103 l = l->prev;
104
105 gtk_widget_get_allocation (last_child, &last_child_allocation);
106
107 /* If last child doesn't fit vertically, prune items from the end of the
108 * list one at a time until it does.
109 */
110 if (last_child_allocation.y + last_child_allocation.height >
111 allocation->y + allocation->height)
112 {
113 GtkWidget *child = NULL;
114
115 while (l != NULL)
116 {
117 child = l->data;
118 l = l->prev;
119
120 gtk_widget_get_allocation (child, &child_allocation);
121
122 /* Reallocate this child's position so that it does not appear.
123 * Setting the width & height to 0 is not enough, as
124 * one pixel is still drawn. Must also move it outside
125 * visible range. For the cases I've seen, -1, -1 works fine.
126 * This might not work in all future cases. Alternatively, the
127 * items that don't fit could be hidden, but that would interfere
128 * with having other hidden children.
129 *
130 * Note that these children are having their size allocated twice,
131 * once by gtk_vbox_size_allocate and then again here. I don't
132 * know of any problems with this, but holler if you do.
133 */
134 tiny_allocation.x = tiny_allocation.y = -1;
135 tiny_allocation.height = tiny_allocation.width = 0;
136 gtk_widget_size_allocate (child, &tiny_allocation);
137
138 /* We're done if the special last item fits now. */
139 if (child_allocation.y + last_child_allocation.height <=
140 allocation->y + allocation->height)
141 {
142 last_child_allocation.y = child_allocation.y;
143 gtk_widget_size_allocate (last_child, &last_child_allocation);
144 break;
145 }
146
147 /* If the special last item still doesn't fit, but we've
148 * run out of earlier items, then the special last item is
149 * just too darn tall. Let's squash it down to fit in the box's
150 * allocation.
151 */
152 if (l == NULL)
153 {
154 last_child_allocation.y = allocation->y;
155 last_child_allocation.height = allocation->height;
156 gtk_widget_size_allocate (last_child, &last_child_allocation);
157 }
158 }
159 }
160 }
161 g_list_free (children);
162}
