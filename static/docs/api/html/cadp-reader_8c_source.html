82/*
83 * Returns an unordered list of NAIFactoryObject-derived objects
84 *
85 * This is implementation of NAIIOProvider::read_items method
86 */
87GList *
88cadp_iio_provider_read_items( const NAIIOProvider *provider, GSList **messages )
89{
90 static const gchar *thisfn = "cadp_iio_provider_read_items";
91 GList *items;
92 GList *desktop_paths, *ip;
93 NAIFactoryObject *item;
94
95 g_debug( "%s: provider=%p (%s), messages=%p",
96 thisfn, ( void * ) provider, G_OBJECT_TYPE_NAME( provider ), ( void * ) messages );
97
98 g_return_val_if_fail( NA_IS_IIO_PROVIDER( provider ), NULL );
99
100 items = NULL;
101 cadp_desktop_provider_release_monitors( CADP_DESKTOP_PROVIDER( provider ));
102
103 desktop_paths = get_list_of_desktop_paths( CADP_DESKTOP_PROVIDER( provider ), messages );
104 for( ip = desktop_paths ; ip ; ip = ip->next ){
105
106 item = item_from_desktop_path( CADP_DESKTOP_PROVIDER( provider ), ( DesktopPath * ) ip->data, messages );
107
108 if( item ){
109 items = g_list_prepend( items, item );
110 na_object_dump( item );
111 }
112 }
113
114 free_desktop_paths( desktop_paths );
115
116 g_debug( "%s: count=%d", thisfn, g_list_length( items ));
117 return( items );
118}

120/*
121 * returns a list of DesktopPath items
122 *
123 * we get the ordered list of XDG_DATA_DIRS, and the ordered list of
124 * subdirs to add; then for each item of each list, we search for
125 * .desktop files in the resulted built path
126 *
127 * the returned list is so a list of DesktopPath struct, in
128 * the ordered of preference (most preferred first)
129 */
130static GList *
131get_list_of_desktop_paths( CappDesktopProvider *provider, GSList **messages )
132{
133 GList *files;
134 GSList *xdg_dirs, *idir;
135 GSList *subdirs, *isub;
136 gchar *dir;
137
138 files = NULL;
139 xdg_dirs = cadp_xdg_dirs_get_data_dirs();
140 subdirs = na_core_utils_slist_from_split( CADP_DESKTOP_PROVIDER_SUBDIRS, G_SEARCHPATH_SEPARATOR_S );
141
142 /* explore each directory from XDG_DATA_DIRS
143 */
144 for( idir = xdg_dirs ; idir ; idir = idir->next ){
145
146 /* explore each N-A candidate subdirectory for each XDG dir
147 */
148 for( isub = subdirs ; isub ; isub = isub->next ){
149
150 dir = g_build_filename(( gchar * ) idir->data, ( gchar * ) isub->data, NULL );
151 cadp_desktop_provider_add_monitor( provider, dir );
152 get_list_of_desktop_files( provider, &files, dir, messages );
153 g_free( dir );
154 }
155 }
156
157 na_core_utils_slist_free( subdirs );
158 na_core_utils_slist_free( xdg_dirs );
159
160 return( files );
161}

163/*
164 * scans the directory for .desktop files
165 * only adds to the list those which have not been yet loaded
166 */
167static void
168get_list_of_desktop_files( const CappDesktopProvider *provider, GList **files, const gchar *dir, GSList **messages )
169{
170 static const gchar *thisfn = "cadp_reader_get_list_of_desktop_files";
171 GDir *dir_handle;
172 GError *error;
173 const gchar *name;
174 gchar *desktop_id;
175
176 g_debug( "%s: provider=%p, files=%p (count=%d), dir=%s, messages=%p",
177 thisfn, ( void * ) provider, ( void * ) files, g_list_length( *files ), dir, ( void * ) messages );
178
179 error = NULL;
180 dir_handle = NULL;
181
182 /* do not warn when the directory just doesn't exist
183 */
184 if( g_file_test( dir, G_FILE_TEST_IS_DIR )){
185 dir_handle = g_dir_open( dir, 0, &error );
186 if( error ){
187 g_warning( "%s: %s: %s", thisfn, dir, error->message );
188 g_error_free( error );
189 goto close_dir_handle;
190 }
191 } else {
192 g_debug( "%s: %s: directory doesn't exist", thisfn, dir );
193 }
194
195 if( dir_handle ){
196 while(( name = g_dir_read_name( dir_handle ))){
197 if( g_str_has_suffix( name, CADP_DESKTOP_FILE_SUFFIX )){
198 desktop_id = na_core_utils_str_remove_suffix( name, CADP_DESKTOP_FILE_SUFFIX );
199 if( !is_already_loaded( provider, *files, desktop_id )){
200 *files = desktop_path_from_id( provider, *files, dir, desktop_id );
201 }
202 g_free( desktop_id );
203 }
204 }
205 }
206
207close_dir_handle:
208 if( dir_handle ){
209 g_dir_close( dir_handle );
210 }
211}

213static gboolean
214is_already_loaded( const CappDesktopProvider *provider, GList *files, const gchar *desktop_id )
215{
216 gboolean found;
217 GList *ip;
218 DesktopPath *dps;
219
220 found = FALSE;
221 for( ip = files ; ip && !found ; ip = ip->next ){
222 dps = ( DesktopPath * ) ip->data;
223 if( !g_ascii_strcasecmp( dps->id, desktop_id )){
224 found = TRUE;
225 }
226 }
227
228 return( found );
229}

231static GList *
232desktop_path_from_id( const CappDesktopProvider *provider, GList *files, const gchar *dir, const gchar *id )
233{
234 DesktopPath *dps;
235 gchar *bname;
236 GList *list;
237
238 dps = g_new0( DesktopPath, 1 );
239
240 bname = g_strdup_printf( "%s%s", id, CADP_DESKTOP_FILE_SUFFIX );
241 dps->path = g_build_filename( dir, bname, NULL );
242 g_free( bname );
243
244 dps->id = g_strdup( id );
245
246 list = g_list_prepend( files, dps );
247
248 return( list );
249}

268/*
269 * Returns a newly allocated NAIFactoryObject-derived object, initialized
270 * from the .desktop file
271 */
272static NAIFactoryObject *
273item_from_desktop_file( const CappDesktopProvider *provider, CappDesktopFile *ndf, GSList **messages )
274{
275 /*static const gchar *thisfn = "cadp_reader_item_from_desktop_file";*/
276 NAIFactoryObject *item;
277 gchar *type;
278 CappReaderData *reader_data;
279 gchar *id;
280
281 item = NULL;
282 type = cadp_desktop_file_get_file_type( ndf );
283
284 if( !strcmp( type, CADP_VALUE_TYPE_ACTION )){
285 item = NA_IFACTORY_OBJECT( na_object_action_new());
286
287 } else if( !strcmp( type, CADP_VALUE_TYPE_MENU )){
288 item = NA_IFACTORY_OBJECT( na_object_menu_new());
289
290 } else {
291 /* Translators: 'type' is the nature of the item: Action or Menu */
292 na_core_utils_slist_add_message( messages, _( "unknown type: %s" ), type );
293 }
294
295 if( item ){
296 id = cadp_desktop_file_get_id( ndf );
297 na_object_set_id( item, id );
298 g_free( id );
299
300 reader_data = g_new0( CappReaderData, 1 );
301 reader_data->ndf = ndf;
302
303 na_ifactory_provider_read_item( NA_IFACTORY_PROVIDER( provider ), reader_data, item, messages );
304
305 na_object_set_provider_data( item, ndf );
306 g_object_weak_ref( G_OBJECT( item ), ( GWeakNotify ) desktop_weak_notify, ndf );
307
308 g_free( reader_data );
309 }
310
311 g_free( type );
312
313 return( item );
314}

363guint
364cadp_reader_iimporter_import_from_uri( const NAIImporter *instance, void *parms_ptr )
365{
366 static const gchar *thisfn = "cadp_reader_iimporter_import_from_uri";
367 guint code;
368 NAIImporterImportFromUriParmsv2 *parms;
369 CappDesktopFile *ndf;
370
371 g_debug( "%s: instance=%p, parms=%p", thisfn, ( void * ) instance, parms_ptr );
372
373 g_return_val_if_fail( NA_IS_IIMPORTER( instance ), IMPORTER_CODE_PROGRAM_ERROR );
374 g_return_val_if_fail( CADP_IS_DESKTOP_PROVIDER( instance ), IMPORTER_CODE_PROGRAM_ERROR );
375
376 parms = ( NAIImporterImportFromUriParmsv2 * ) parms_ptr;
377
378 if( !na_core_utils_file_is_loadable( parms->uri )){
379 code = IMPORTER_CODE_NOT_LOADABLE;
380 return( code );
381 }
382
383 code = IMPORTER_CODE_NOT_WILLING_TO;
384
385 ndf = cadp_desktop_file_new_from_uri( parms->uri );
386 if( ndf ){
387 parms->imported = ( NAObjectItem * ) item_from_desktop_file(
388 ( const CappDesktopProvider * ) CADP_DESKTOP_PROVIDER( instance ),
389 ndf, &parms->messages );
390
391 if( parms->imported ){
392 g_return_val_if_fail( NA_IS_OBJECT_ITEM( parms->imported ), IMPORTER_CODE_NOT_WILLING_TO );
393
394 /* remove the weak reference on desktop file set by 'item_from_desktop_file'
395 * as we must consider this #NAObjectItem as a new one
396 */
397 na_object_set_provider_data( parms->imported, NULL );
398 g_object_weak_unref( G_OBJECT( parms->imported ), ( GWeakNotify ) desktop_weak_notify, ndf );
399 g_object_unref( ndf );
400
401 /* also remove the 'writable' status'
402 */
403 na_object_set_readonly( parms->imported, FALSE );
404
405 code = IMPORTER_CODE_OK;
406 }
407 }
408
409 if( code == IMPORTER_CODE_NOT_WILLING_TO ){
410 na_core_utils_slist_add_message( &parms->messages, ERR_NOT_DESKTOP );
411 }
412
413 return( code );
414}

475/*
476 * reading any data from a desktop file requires:
477 * - a CappDesktopFile object which has been initialized with the .desktop file
478 * -> has been attached to the NAObjectItem in get_item() above
479 * - the data type (+ reading default value)
480 * - group and key names
481 *
482 * Returns: NULL if the key has not been found
483 * letting the caller deal with default values
484 */
485NADataBoxed *
486cadp_reader_ifactory_provider_read_data( const NAIFactoryProvider *reader, void *reader_data, const NAIFactoryObject *object, const NADataDef *def, GSList **messages )
487{
488 static const gchar *thisfn = "cadp_reader_ifactory_provider_read_data";
489 NADataBoxed *boxed;
490 gboolean found;
491 CappReaderData *nrd;
492 gchar *group, *id;
493 gchar *msg;
494 gchar *str_value;
495 gboolean bool_value;
496 GSList *slist_value;
497 guint uint_value;
498
499 g_return_val_if_fail( NA_IS_IFACTORY_PROVIDER( reader ), NULL );
500 g_return_val_if_fail( CADP_IS_DESKTOP_PROVIDER( reader ), NULL );
501 g_return_val_if_fail( NA_IS_IFACTORY_OBJECT( object ), NULL );
502
503 boxed = NULL;
504
505 if( !CADP_DESKTOP_PROVIDER( reader )->private->dispose_has_run ){
506
507 nrd = ( CappReaderData * ) reader_data;
508 g_return_val_if_fail( CADP_IS_DESKTOP_FILE( nrd->ndf ), NULL );
509
510 if( def->desktop_entry ){
511
512 if( NA_IS_OBJECT_ITEM( object )){
513 group = g_strdup( CADP_GROUP_DESKTOP );
514
515 } else {
516 g_return_val_if_fail( NA_IS_OBJECT_PROFILE( object ), NULL );
517 id = na_object_get_id( object );
518 group = g_strdup_printf( "%s %s", CADP_GROUP_PROFILE, id );
519 g_free( id );
520 }
521
522 switch( def->type ){
523
524 case NA_DATA_TYPE_LOCALE_STRING:
525 str_value = cadp_desktop_file_get_locale_string( nrd->ndf, group, def->desktop_entry, &found, def->default_value );
526 if( found ){
527 boxed = na_data_boxed_new( def );
528 na_boxed_set_from_void( NA_BOXED( boxed ), str_value );
529 }
530 g_free( str_value );
531 break;
532
533 case NA_DATA_TYPE_STRING:
534 str_value = cadp_desktop_file_get_string( nrd->ndf, group, def->desktop_entry, &found, def->default_value );
535 if( found ){
536 boxed = na_data_boxed_new( def );
537 na_boxed_set_from_void( NA_BOXED( boxed ), str_value );
538 }
539 g_free( str_value );
540 break;
541
542 case NA_DATA_TYPE_BOOLEAN:
543 bool_value = cadp_desktop_file_get_boolean( nrd->ndf, group, def->desktop_entry, &found, na_core_utils_boolean_from_string( def->default_value ));
544 if( found ){
545 boxed = na_data_boxed_new( def );
546 na_boxed_set_from_void( NA_BOXED( boxed ), GUINT_TO_POINTER( bool_value ));
547 }
548 break;
549
550 case NA_DATA_TYPE_STRING_LIST:
551 slist_value = cadp_desktop_file_get_string_list( nrd->ndf, group, def->desktop_entry, &found, def->default_value );
552 if( found ){
553 boxed = na_data_boxed_new( def );
554 na_boxed_set_from_void( NA_BOXED( boxed ), slist_value );
555 }
556 na_core_utils_slist_free( slist_value );
557 break;
558
559 case NA_DATA_TYPE_UINT:
560 uint_value = cadp_desktop_file_get_uint( nrd->ndf, group, def->desktop_entry, &found, atoi( def->default_value ));
561 if( found ){
562 boxed = na_data_boxed_new( def );
563 na_boxed_set_from_void( NA_BOXED( boxed ), GUINT_TO_POINTER( uint_value ));
564 }
565 break;
566
567 default:
568 msg = g_strdup_printf( "%s: %d: invalid data type.", thisfn, def->type );
569 g_warning( "%s", msg );
570 *messages = g_slist_append( *messages, msg );
571 }
572
573 g_free( group );
574 }
575 }
576
577 return( boxed );
578}

610

630/*
631 * Read and attach profiles in the specified order
632 * - profiles which may exist in .desktop files, but are not referenced
633 * in the 'Profiles' string list are just ignored
634 * - profiles which may be referenced in the action string list, but are not
635 * found in the .desktop file are recreated with default values (plus a warning)
636 * - ensure that there is at least one profile attached to the action
637 */
638static void
639read_done_action_read_profiles( const NAIFactoryProvider *provider, NAObjectAction *action, CappReaderData *reader_data, GSList **messages )
640{
641 static const gchar *thisfn = "cadp_reader_read_done_action_read_profiles";
642 GSList *order;
643 GSList *ip;
644 gchar *profile_id;
645 NAObjectId *found;
646 NAObjectProfile *profile;
647
648 reader_data->action = action;
649 order = na_object_get_items_slist( action );
650
651 for( ip = order ; ip ; ip = ip->next ){
652 profile_id = ( gchar * ) ip->data;
