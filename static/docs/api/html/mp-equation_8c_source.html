40static int
41get_variable(ParserState *state, const char *name, MPNumber *z)
42{
43 int result = 1;
44
45 if (strcmp(name, "e") == 0)
46 mp_get_eulers(z);
47 else if (strcmp(name, "i") == 0)
48 mp_get_i(z);
49 else if (strcmp(name, "π") == 0 || strcmp(name, "pi") == 0)
50 mp_get_pi(z);
51 else if (strcmp(name, "c₀") == 0)
52 mp_set_from_string("299792458", 10, z); /* velocity of light */
53 else if (strcmp(name, "μ₀") == 0)
54 mp_set_from_string("0.0000012566370614", 10, z); /* magnetic constant */
55 else if (strcmp(name, "ε₀") == 0)
56 mp_set_from_string("0.00000000000885418782", 10, z); /* electric constant */
57 else if (strcmp(name, "G") == 0)
58 mp_set_from_string("0.0000000000667408", 10, z); /* Newtonian constant of gravitation */
59 else if (strcmp(name, "h") == 0)
60 mp_set_from_string("0.000000000000000000000000000000000662607004", 10, z); /* Planck constant */
61 else if (strcmp(name, "ｅ") == 0)
62 mp_set_from_string("0.00000000000000000016021766208", 10, z); /* elementary charge */
63 else if (strcmp(name, "mₑ") == 0)
64 mp_set_from_string("0.000000000000000000000000000000910938356", 10, z); /* electron mass */
65 else if (strcmp(name, "mₚ") == 0)
66 mp_set_from_string("0.000000000000000000000000001672621898", 10, z); /* proton mass */
67 else if (strcmp(name, "Nₐ") == 0)
68 mp_set_from_string("602214086000000000000000", 10, z); /* Avogadro constant */
69 else if (state->options->get_variable)
70 result = state->options->get_variable(name, z, state->options->callback_data);
71 else
72 result = 0;
73
74 return result;

105int
106sub_atoi(const char *data)
107{
108 int i, value = 0;
109 const char *digits[] = {"₀", "₁", "₂", "₃", "₄", "₅", "₆", "₇", "₈", "₉", NULL};
110
111 do {
112 for(i = 0; digits[i] != NULL && strncmp(data, digits[i], strlen(digits[i])) != 0; i++);
113 if(digits[i] == NULL)
114 return -1;
115 data += strlen(digits[i]);
116 value = value * 10 + i;
117 } while(*data != '\0');
118
119 return value;
120}

122int
123super_atoi(const char *data)
124{
125 int i, sign = 1, value = 0;
126 const char *digits[11] = {"⁰", "¹", "²", "³", "⁴", "⁵", "⁶", "⁷", "⁸", "⁹", NULL};
127
128 if(strncmp(data, "⁻", strlen("⁻")) == 0) {
129 sign = -1;
130 data += strlen("⁻");
131 }
132
133 do {
134 for(i = 0; digits[i] != NULL && strncmp(data, digits[i], strlen(digits[i])) != 0; i++);
135 if(digits[i] == NULL)
136 return 0;
137 value = value * 10 + i;
138 data += strlen(digits[i]);
139 } while(*data != '\0');
140
141 return sign * value;
142}

288MPErrorCode
289mp_equation_parse(const char *expression, MPEquationOptions *options, MPNumber *result, char **error_token)
290{
291 int ret;
292 int err;
293 ParserState* state;
294 state = p_create_parser (expression, options);
295
296 if (!(expression && result) || strlen(expression) == 0)
297 return PARSER_ERR_INVALID;
298
299 state->variable_is_defined = variable_is_defined;
300 state->get_variable = get_variable;
301 state->set_variable = set_variable;
302 state->function_is_defined = function_is_defined;
303 state->get_function = get_function;
304 state->convert = convert;
305 state->error = 0;
306 mp_clear_error();
307 ret = p_parse (state);
308 if (state->error_token != NULL && error_token != NULL) {
309 *error_token = state->error_token;
310 }
311 /* Error during parsing */
312 if (state->error) {
313 err = state->error;
314 p_destroy_parser (state);
315 return err;
316 }
317
318 if (mp_get_error()) {
319 p_destroy_parser (state);
320 return PARSER_ERR_MP;
321 }
322
323 /* Failed to parse */
324 if (ret) {
325 p_destroy_parser (state);
326 return PARSER_ERR_INVALID;
327 }
328 mp_set_from_mp(&state->ret, result);
329 p_destroy_parser (state);
330 return PARSER_ERR_NONE;
331}
