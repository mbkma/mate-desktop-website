39static void
40ev_form_field_accessible_get_extents (AtkComponent *atk_component,
41 gint *x,
42 gint *y,
43 gint *width,
44 gint *height,
45 AtkCoordType coord_type)
46{
47 EvFormFieldAccessible *self;

58 *width = atk_rect.x2 - atk_rect.x1;

90static AtkRole
91ev_form_field_accessible_get_role (AtkObject *atk_object)
92{
93 EvFormField *ev_form_field;
94
95 ev_form_field = EV_FORM_FIELD_ACCESSIBLE (atk_object)->priv->form_field;
96 if (EV_IS_FORM_FIELD_BUTTON (ev_form_field)) {
97 EvFormFieldButton *field_button = EV_FORM_FIELD_BUTTON (ev_form_field);
98
99 switch (field_button->type) {
100 case EV_FORM_FIELD_BUTTON_CHECK:
101 return ATK_ROLE_CHECK_BOX;
102 case EV_FORM_FIELD_BUTTON_RADIO:
103 return ATK_ROLE_RADIO_BUTTON;
104 case EV_FORM_FIELD_BUTTON_PUSH:
105 return ATK_ROLE_PUSH_BUTTON;
106 default:
107 return ATK_ROLE_UNKNOWN;
108 }
109 }
110
111 if (EV_IS_FORM_FIELD_CHOICE (ev_form_field)) {
112 EvFormFieldChoice *field_choice = EV_FORM_FIELD_CHOICE (ev_form_field);
113
114 switch (field_choice->type) {
115 case EV_FORM_FIELD_CHOICE_COMBO:
116 return ATK_ROLE_COMBO_BOX;
117 case EV_FORM_FIELD_CHOICE_LIST:
118 return ATK_ROLE_LIST_BOX;
119 default:
120 return ATK_ROLE_UNKNOWN;
121 }
122 }
123
124 if (EV_IS_FORM_FIELD_TEXT (ev_form_field)) {
125 EvFormFieldText *field_text = EV_FORM_FIELD_TEXT (ev_form_field);
126
127 if (field_text->is_password)
128 return ATK_ROLE_PASSWORD_TEXT;
129 else if (field_text->type == EV_FORM_FIELD_TEXT_MULTILINE)
130 return ATK_ROLE_TEXT;
131 else
132 return ATK_ROLE_ENTRY;
133 }
134
135 return ATK_ROLE_UNKNOWN;
136}

138static AtkStateSet *
139ev_form_field_accessible_ref_state_set (AtkObject *atk_object)
140{
141 AtkStateSet *state_set;
142 AtkStateSet *copy_set;
143 AtkStateSet *page_accessible_state_set;
144 EvFormFieldAccessible *self;
145 EvFormField *ev_form_field;
146 EvViewAccessible *view_accessible;
147 gint page;
148
149 self = EV_FORM_FIELD_ACCESSIBLE (atk_object);
150 state_set = ATK_OBJECT_CLASS (ev_form_field_accessible_parent_class)->ref_state_set (atk_object);
151 atk_state_set_clear_states (state_set);
152
153 page_accessible_state_set = atk_object_ref_state_set (ATK_OBJECT (self->priv->page));
154 copy_set = atk_state_set_or_sets (state_set, page_accessible_state_set);
155
156 view_accessible = ev_page_accessible_get_view_accessible (self->priv->page);
157 page = ev_page_accessible_get_page (self->priv->page);
158 if (!ev_view_accessible_is_doc_rect_showing (view_accessible, page, &self->priv->area))
159 atk_state_set_remove_state (copy_set, ATK_STATE_SHOWING);
160
161 ev_form_field = EV_FORM_FIELD_ACCESSIBLE (atk_object)->priv->form_field;
162 if (EV_IS_FORM_FIELD_BUTTON (ev_form_field)) {
163 EvFormFieldButton *field_button = EV_FORM_FIELD_BUTTON (ev_form_field);
164
165 if (field_button->state) {
166 if (field_button->type == EV_FORM_FIELD_BUTTON_PUSH)
167 atk_state_set_add_state (copy_set, ATK_STATE_PRESSED);
168 else
169 atk_state_set_add_state (copy_set, ATK_STATE_CHECKED);
170 }
171 }
172
173 else if (EV_IS_FORM_FIELD_CHOICE (ev_form_field)) {
174 EvFormFieldChoice *field_choice = EV_FORM_FIELD_CHOICE (ev_form_field);
175
176 if (field_choice->is_editable && !ev_form_field->is_read_only)
177 atk_state_set_add_state (copy_set, ATK_STATE_EDITABLE);
178 if (field_choice->multi_select)
179 atk_state_set_add_state (copy_set, ATK_STATE_MULTISELECTABLE);
180 }
181
182 else if (EV_IS_FORM_FIELD_TEXT (ev_form_field)) {
183 EvFormFieldText *field_text = EV_FORM_FIELD_TEXT (ev_form_field);
184
185 if (!ev_form_field->is_read_only)
186 atk_state_set_add_state (copy_set, ATK_STATE_EDITABLE);
187 if (field_text->type == EV_FORM_FIELD_TEXT_NORMAL)
188 atk_state_set_add_state (copy_set, ATK_STATE_SINGLE_LINE);
189 else if (field_text->type == EV_FORM_FIELD_TEXT_MULTILINE)
190 atk_state_set_add_state (copy_set, ATK_STATE_MULTI_LINE);
191 }
192
193 g_object_unref (state_set);
194 g_object_unref (page_accessible_state_set);
195
196 return copy_set;
197}
