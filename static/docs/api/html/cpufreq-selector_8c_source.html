121static gboolean
122cpufreq_selector_create_proxy (CPUFreqSelector *selector,
123 GError **error)
124{
125 if (selector->proxy)
126 return TRUE;
127
128 selector->proxy = g_dbus_proxy_new_sync (selector->system_bus,
129 G_DBUS_PROXY_FLAGS_NONE,
130 NULL,
131 "org.mate.CPUFreqSelector",
132 "/org/mate/cpufreq_selector/selector",
133 "org.mate.CPUFreqSelector",
134 NULL,
135 error);
136
137 return (selector->proxy != NULL);

157static void
158selector_set_frequency_async (SelectorAsyncData *data)
159{
160 GError *error = NULL;
161
162 if (!cpufreq_selector_connect_to_system_bus (data->selector, &error)) {
163 g_warning ("%s", error->message);
164 g_error_free (error);
165
166 selector_async_data_free (data);
167
168 return;
169 }
170
171 if (!cpufreq_selector_create_proxy (data->selector, &error)) {
172 g_warning ("%s", error->message);
173 g_error_free (error);
174
175 selector_async_data_free (data);
176 return;
177 }
178
179 g_dbus_proxy_call (data->selector->proxy,
180 "SetFrequency",
181 g_variant_new ("(uu)",
182 data->cpu,
183 data->frequency),
184 G_DBUS_CALL_FLAGS_NONE,
185 -1,
186 NULL,
187 selector_setter_cb,
188 data);
189}

191void
192cpufreq_selector_set_frequency_async (CPUFreqSelector *selector,
193 guint cpu,
194 guint frequency)
195{
196 guint cores;
197 cores = get_nprocs() ;
198 for (cpu = 0; cpu < cores; cpu = cpu+1) {
199 SelectorAsyncData *data;
200
201 data = g_new0 (SelectorAsyncData, 1);
202
203 data->selector = selector;
204 data->call = FREQUENCY;
205 data->cpu = cpu;
206 data->frequency = frequency;
207
208 selector_set_frequency_async (data);
209 }

269cpufreq_selector_run_command (CPUFreqSelector *selector,

271{

276 path = g_find_program_in_path ("cpufreq-selector");
277
278 if (!path)
279 return;
280
281 command = g_strdup_printf ("%s %s", path, args);
282 g_free (path);
283
284 g_spawn_command_line_async (command, &error);
285 g_free (command);
286
287 if (error) {
288 g_warning ("%s", error->message);
