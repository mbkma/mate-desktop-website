127mate_mixer_stream_toggle_get_property (GObject *object,

131{
132 MateMixerStreamToggle *toggle;
133
134 toggle = MATE_MIXER_STREAM_TOGGLE (object);

136 switch (param_id) {
137 case PROP_STATE:

143 case PROP_OFF_STATE_OPTION:

148 G_OBJECT_WARN_INVALID_PROPERTY_ID (object, param_id, pspec);
149 break;
150 }

215gboolean
216mate_mixer_stream_toggle_get_state (MateMixerStreamToggle *toggle)
217{
218 MateMixerSwitchOption *active;
219
220 g_return_val_if_fail (MATE_MIXER_IS_STREAM_TOGGLE (toggle), FALSE);
221
222 active = mate_mixer_switch_get_active_option (MATE_MIXER_SWITCH (toggle));
223 if (G_UNLIKELY (active == NULL))
224 return FALSE;
225
226 if (active == toggle->priv->on)
227 return TRUE;
228 else
229 return FALSE;
230}

264gboolean

267 MateMixerSwitchOption *active;
268
269 g_return_val_if_fail (MATE_MIXER_IS_STREAM_TOGGLE (toggle), FALSE);
270
271 if (state == TRUE)
272 active = toggle->priv->on;
273 else
274 active = toggle->priv->off;
275
276 if (G_UNLIKELY (active == NULL))
277 return FALSE;
278
279 return mate_mixer_switch_set_active_option (MATE_MIXER_SWITCH (toggle), active);

282static MateMixerSwitchOption *
283mate_mixer_stream_toggle_get_option (MateMixerSwitch *swtch, const gchar *name)
284{
285 MateMixerStreamToggle *toggle;
286
287 g_return_val_if_fail (MATE_MIXER_IS_STREAM_TOGGLE (swtch), NULL);
288 g_return_val_if_fail (name != NULL, NULL);
289
290 toggle = MATE_MIXER_STREAM_TOGGLE (swtch);
291
292 if (g_strcmp0 (name, mate_mixer_switch_option_get_name (toggle->priv->on)) == 0)
293 return toggle->priv->on;
294 if (g_strcmp0 (name, mate_mixer_switch_option_get_name (toggle->priv->off)) == 0)
295 return toggle->priv->off;
296
297 return NULL;
298}

300static const GList *

302{

305 g_return_val_if_fail (MATE_MIXER_IS_STREAM_TOGGLE (swtch), NULL);
306
307 toggle = MATE_MIXER_STREAM_TOGGLE (swtch);
308
309 if (toggle->priv->options == NULL) {
310 if (G_LIKELY (toggle->priv->off != NULL))
311 toggle->priv->options = g_list_prepend (toggle->priv->options,
312 toggle->priv->off);
313 if (G_LIKELY (toggle->priv->on != NULL))
314 toggle->priv->options = g_list_prepend (toggle->priv->options,
315 toggle->priv->on);
316 }
317 return toggle->priv->options;
