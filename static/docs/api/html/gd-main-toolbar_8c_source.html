99static GtkWidget *
100get_symbolic_button (const gchar *icon_name,
101 ChildType type)
102{
103 GtkWidget *button, *w;
104
105 switch (type)
106 {
107 case CHILD_MENU:
108 button = gtk_menu_button_new ();
109 gtk_widget_destroy (gtk_bin_get_child (GTK_BIN (button)));
110 break;
111 case CHILD_TOGGLE:
112 button = gtk_toggle_button_new ();
113 break;
114 case CHILD_NORMAL:
115 default:
116 button = gtk_button_new ();
117 break;
118 }
119
120 gtk_style_context_add_class (gtk_widget_get_style_context (button), "raised");
121 gtk_style_context_add_class (gtk_widget_get_style_context (button), "image-button");
122
123 w = gtk_image_new_from_icon_name (icon_name, GTK_ICON_SIZE_MENU);
124 gtk_widget_show (w);
125 gtk_container_add (GTK_CONTAINER (button), w);
126
127 return button;

161static GtkSizeGroup *
162get_vertical_size_group (GdMainToolbar *self)
163{
164 GtkSizeGroup *retval;
165 GtkWidget *dummy;
166 GtkToolItem *container;
167
168 dummy = get_text_button ("Dummy", CHILD_NORMAL);
169 container = gtk_tool_item_new ();
170 gtk_widget_set_no_show_all (GTK_WIDGET (container), TRUE);
171 gtk_container_add (GTK_CONTAINER (container), dummy);
172 gtk_toolbar_insert (GTK_TOOLBAR (self), container, -1);
173
174 retval = gtk_size_group_new (GTK_SIZE_GROUP_VERTICAL);
175 gtk_size_group_add_widget (retval, dummy);
176
177 return retval;

186void
187gd_main_toolbar_set_show_modes (GdMainToolbar *self,
188 gboolean show_modes)
189{
190 if (self->priv->show_modes == show_modes)
191 return;
192
193 self->priv->show_modes = show_modes;
194 if (self->priv->show_modes)
195 {
196 gtk_widget_set_no_show_all (self->priv->labels_grid, TRUE);
197 gtk_widget_hide (self->priv->labels_grid);
198
199 gtk_widget_set_valign (self->priv->center_grid, GTK_ALIGN_FILL);
200 gtk_widget_set_no_show_all (self->priv->modes_box, FALSE);
201 gtk_widget_show_all (self->priv->modes_box);
202 }
203 else
204 {
205 gtk_widget_set_no_show_all (self->priv->modes_box, TRUE);
206 gtk_widget_hide (self->priv->modes_box);
207
208 gtk_widget_set_valign (self->priv->center_grid, GTK_ALIGN_CENTER);
209 gtk_widget_set_no_show_all (self->priv->labels_grid, FALSE);
210 gtk_widget_show_all (self->priv->labels_grid);
211 }
212
213 g_object_notify (G_OBJECT (self), "show-modes");
214}

255static void
256gd_main_toolbar_constructed (GObject *obj)
257{
258 GdMainToolbar *self = GD_MAIN_TOOLBAR (obj);
259 GtkToolbar *tb = GTK_TOOLBAR (obj);
260 GtkWidget *grid;
261
262 G_OBJECT_CLASS (gd_main_toolbar_parent_class)->constructed (obj);
263
264 self->priv->vertical_size_group = get_vertical_size_group (self);
265
266 /* left section */
267 self->priv->left_group = gtk_tool_item_new ();
268 gtk_widget_set_margin_end (GTK_WIDGET (self->priv->left_group), 12);
269 gtk_toolbar_insert (tb, self->priv->left_group, -1);
270 gtk_size_group_add_widget (self->priv->vertical_size_group,
271 GTK_WIDGET (self->priv->left_group));
272
273 /* left button group */
274 self->priv->left_grid = gtk_grid_new ();
275 gtk_grid_set_column_spacing (GTK_GRID (self->priv->left_grid), 12);
276 gtk_container_add (GTK_CONTAINER (self->priv->left_group), self->priv->left_grid);
277 gtk_widget_set_halign (self->priv->left_grid, GTK_ALIGN_START);
278
279 /* center section */
280 self->priv->center_group = gtk_tool_item_new ();
281 gtk_tool_item_set_expand (self->priv->center_group, TRUE);
282 gtk_toolbar_insert (tb, self->priv->center_group, -1);
283 self->priv->center_grid = gtk_grid_new ();
284 gtk_widget_set_halign (self->priv->center_grid, GTK_ALIGN_CENTER);
285 gtk_widget_set_valign (self->priv->center_grid, GTK_ALIGN_CENTER);
286 gtk_container_add (GTK_CONTAINER (self->priv->center_group), self->priv->center_grid);
287 gtk_size_group_add_widget (self->priv->vertical_size_group,
288 GTK_WIDGET (self->priv->center_group));
289
290 /* centered label group */
291 self->priv->labels_grid = grid = gtk_grid_new ();
292 gtk_grid_set_column_spacing (GTK_GRID (grid), 12);
293 gtk_container_add (GTK_CONTAINER (self->priv->center_grid), grid);
294
295 self->priv->title_label = gtk_label_new (NULL);
296 gtk_label_set_ellipsize (GTK_LABEL (self->priv->title_label), PANGO_ELLIPSIZE_END);
297 gtk_container_add (GTK_CONTAINER (grid), self->priv->title_label);
298
299 self->priv->detail_label = gtk_label_new (NULL);
300 gtk_label_set_ellipsize (GTK_LABEL (self->priv->detail_label), PANGO_ELLIPSIZE_END);
301 gtk_widget_set_no_show_all (self->priv->detail_label, TRUE);
302 gtk_style_context_add_class (gtk_widget_get_style_context (self->priv->detail_label), "dim-label");
303 gtk_container_add (GTK_CONTAINER (grid), self->priv->detail_label);
304
305 /* centered mode group */
306 self->priv->modes_box = gtk_box_new (GTK_ORIENTATION_HORIZONTAL, 0);
307 gtk_box_set_homogeneous (GTK_BOX (self->priv->modes_box), TRUE);
308 gtk_widget_set_no_show_all (self->priv->modes_box, TRUE);
309 gtk_style_context_add_class (gtk_widget_get_style_context (self->priv->modes_box), "linked");
310 gtk_container_add (GTK_CONTAINER (self->priv->center_grid), self->priv->modes_box);
311
312 /* right section */
313 self->priv->right_group = gtk_tool_item_new ();
314 gtk_widget_set_margin_start (GTK_WIDGET (self->priv->right_group), 12);
315 gtk_toolbar_insert (tb, self->priv->right_group, -1);
316 gtk_size_group_add_widget (self->priv->vertical_size_group,
317 GTK_WIDGET (self->priv->right_group));
318
319 self->priv->right_grid = gtk_grid_new ();
320 gtk_grid_set_column_spacing (GTK_GRID (self->priv->right_grid), 12);
321 gtk_container_add (GTK_CONTAINER (self->priv->right_group), self->priv->right_grid);
322 gtk_widget_set_halign (self->priv->right_grid, GTK_ALIGN_END);
323
324 self->priv->size_group = gtk_size_group_new (GTK_SIZE_GROUP_HORIZONTAL);
325 gtk_size_group_add_widget (self->priv->size_group, GTK_WIDGET (self->priv->left_group));
326 gtk_size_group_add_widget (self->priv->size_group, GTK_WIDGET (self->priv->right_group));
327}

379gd_main_toolbar_set_labels (GdMainToolbar *self,

382{

419static GtkWidget *
420add_button_internal (GdMainToolbar *self,
421 const gchar *icon_name,
422 const gchar *label,
423 gboolean pack_start,
424 ChildType type)
425{
426 GtkWidget *button;
427
428 if (icon_name != NULL)
429 {
430 button = get_symbolic_button (icon_name, type);
431 if (label != NULL)
432 gtk_widget_set_tooltip_text (button, label);
433 }
434 else if (label != NULL)
435 {
436 button = get_text_button (label, type);
437 }
438 else
439 {
440 button = get_empty_button (type);
441 }
442
443 gd_main_toolbar_add_widget (self, button, pack_start);
444
445 gtk_widget_show_all (button);
446
447 return button;
448}

456void
457gd_main_toolbar_set_labels_menu (GdMainToolbar *self,
458 GMenuModel *menu)
459{
460 GtkWidget *button, *grid, *w;
461
462 if (menu == NULL &&
463 ((gtk_widget_get_parent (self->priv->labels_grid) == self->priv->center_grid) ||
464 self->priv->center_menu_child == NULL))
465 return;
466
467 if (menu != NULL)
468 {
469 g_object_ref (self->priv->labels_grid);
470 gtk_container_remove (GTK_CONTAINER (self->priv->center_grid),
471 self->priv->labels_grid);
472
473 self->priv->center_menu_child = grid = gtk_grid_new ();
474 gtk_grid_set_column_spacing (GTK_GRID (grid), 6);
475 gtk_container_add (GTK_CONTAINER (grid), self->priv->labels_grid);
476 g_object_unref (self->priv->labels_grid);
477
478 w = gtk_image_new_from_icon_name ("pan-down-symbolic", GTK_ICON_SIZE_BUTTON);
479 gtk_container_add (GTK_CONTAINER (grid), w);
480
481 self->priv->center_menu = button = gtk_menu_button_new ();
482 gtk_style_context_add_class (gtk_widget_get_style_context (self->priv->center_menu),
483 "selection-menu");
484 gtk_widget_destroy (gtk_bin_get_child (GTK_BIN (button)));
485 gtk_widget_set_halign (button, GTK_ALIGN_CENTER);
486 gtk_menu_button_set_menu_model (GTK_MENU_BUTTON (button), menu);
487 gtk_container_add (GTK_CONTAINER (self->priv->center_menu), grid);
488
489 gtk_container_add (GTK_CONTAINER (self->priv->center_grid), button);
490 }
491 else
492 {
493 g_object_ref (self->priv->labels_grid);
494 gtk_container_remove (GTK_CONTAINER (self->priv->center_menu_child),
495 self->priv->labels_grid);
496 gtk_widget_destroy (self->priv->center_menu);
497
498 self->priv->center_menu = NULL;
499 self->priv->center_menu_child = NULL;
500
501 gtk_container_add (GTK_CONTAINER (self->priv->center_grid),
502 self->priv->labels_grid);
503 g_object_unref (self->priv->labels_grid);
504 }
505
506 gtk_widget_show_all (self->priv->center_grid);
507}

516GtkWidget *
517gd_main_toolbar_add_mode (GdMainToolbar *self,
518 const gchar *label)
519{
520 GtkWidget *button;
521 GList *group;
522
523 button = gtk_radio_button_new_with_label (NULL, label);
524 gtk_toggle_button_set_mode (GTK_TOGGLE_BUTTON (button), FALSE);
525 gtk_widget_set_size_request (button, 100, -1);
526 gtk_style_context_add_class (gtk_widget_get_style_context (button), "raised");
527 gtk_style_context_add_class (gtk_widget_get_style_context (button), "text-button");
528
529 group = gtk_container_get_children (GTK_CONTAINER (self->priv->modes_box));
530 if (group != NULL)
531 {
532 gtk_radio_button_join_group (GTK_RADIO_BUTTON (button), GTK_RADIO_BUTTON (group->data));
533 g_list_free (group);
534 }
535
536 gtk_container_add (GTK_CONTAINER (self->priv->modes_box), button);
537 gtk_widget_show (button);
538
539 return button;
