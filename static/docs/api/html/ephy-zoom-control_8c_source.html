64static void
65combo_changed_cb (GtkComboBox *combo, EphyZoomControl *control)
66{
67 gint index;
68 float zoom;
69
70 index = gtk_combo_box_get_active (combo);
71 zoom = zoom_levels[index].level;
72
73 if (zoom != control->priv->zoom)
74 {
75 g_signal_emit (control, signals[ZOOM_TO_LEVEL_SIGNAL], 0, zoom);
76 }
77}

79static void
80sync_zoom_cb (EphyZoomControl *control, GParamSpec *pspec, gpointer data)
81{

92static void
93sync_zoom_max_min_cb (EphyZoomControl *control, GParamSpec *pspec, gpointer data)
94{
95 EphyZoomControlPrivate *p = control->priv;
96 GtkListStore *model = (GtkListStore *)gtk_combo_box_get_model (p->combo);
97 GtkTreeIter iter;
98 gint i;
99
100 g_signal_handler_block (p->combo, p->handler_id);
101 gtk_list_store_clear (model);
102
103 for (i = 0; i < n_zoom_levels; i++)
104 {
105 if (zoom_levels[i].level > 0) {
106 if (zoom_levels[i].level < p->min_zoom)
107 continue;
108
109 if (zoom_levels[i].level > p->max_zoom)
110 break;
111 }
112
113 gtk_list_store_append (model, &iter);
114
115 if (zoom_levels[i].name != NULL) {
116 gtk_list_store_set (model, &iter,
117 COL_TEXT, _(zoom_levels[i].name),
118 -1);
119 } else {
120 gtk_list_store_set (model, &iter,
121 COL_IS_SEP, zoom_levels[i].name == NULL,
122 -1);
123 }
124 }
125
126 gtk_combo_box_set_active (p->combo, ephy_zoom_get_zoom_level_index (p->zoom));
127 g_signal_handler_unblock (p->combo, p->handler_id);
128}

150static void
151ephy_zoom_control_init (EphyZoomControl *control)
152{
153 EphyZoomControlPrivate *p;
154 GtkWidget *vbox;
155 GtkCellRenderer *renderer;
156 GtkListStore *store;
157 GtkTreeIter iter;
158 guint i;
159
160 p = ephy_zoom_control_get_instance_private (control);
161 control->priv = p;
162
163 p->zoom = 1.0;
164
165 store = gtk_list_store_new (2, G_TYPE_STRING, G_TYPE_BOOLEAN);
166
167 for (i = 0; i < n_zoom_levels; i++)
168 {
169 gtk_list_store_append (store, &iter);
170
171 if (zoom_levels[i].name != NULL) {
172 gtk_list_store_set (store, &iter,
173 COL_TEXT, _(zoom_levels[i].name),
174 -1);
175 } else {
176 gtk_list_store_set (store, &iter,
177 COL_IS_SEP, zoom_levels[i].name == NULL,
178 -1);
179 }
180 }
181
182 p->combo = GTK_COMBO_BOX (gtk_combo_box_new_with_model (GTK_TREE_MODEL (store)));
183 g_object_unref (store);
184
185 renderer = gtk_cell_renderer_text_new ();
186 gtk_cell_layout_pack_start (GTK_CELL_LAYOUT (p->combo), renderer, TRUE);
187 gtk_cell_layout_set_attributes (GTK_CELL_LAYOUT (p->combo), renderer,
188 "text", COL_TEXT, NULL);
189 gtk_combo_box_set_row_separator_func (p->combo,
190 (GtkTreeViewRowSeparatorFunc) row_is_separator,
191 NULL, NULL);
192
193 gtk_widget_set_focus_on_click (GTK_WIDGET (p->combo), FALSE);
194 g_object_ref_sink (G_OBJECT (p->combo));
195 gtk_widget_show (GTK_WIDGET (p->combo));
196
197 i = ephy_zoom_get_zoom_level_index (p->zoom);
198 gtk_combo_box_set_active (p->combo, i);
199
200 vbox = gtk_box_new (GTK_ORIENTATION_VERTICAL, 0);
201 gtk_box_set_homogeneous (GTK_BOX (vbox), TRUE);
202 gtk_box_pack_start (GTK_BOX (vbox), GTK_WIDGET (p->combo), TRUE, FALSE, 0);
203 gtk_widget_show (vbox);
204
205 gtk_container_add (GTK_CONTAINER (control), vbox);
206
207 p->handler_id = g_signal_connect (p->combo, "changed",
208 G_CALLBACK (combo_changed_cb), control);
209
210 g_signal_connect_object (control, "notify::zoom",
211 G_CALLBACK (sync_zoom_cb), NULL, 0);
212 g_signal_connect_object (control, "notify::min-zoom",
213 G_CALLBACK (sync_zoom_max_min_cb), NULL, 0);
214 g_signal_connect_object (control, "notify::max-zoom",
215 G_CALLBACK (sync_zoom_max_min_cb), NULL, 0);
216}

322void
323ephy_zoom_control_set_zoom_level (EphyZoomControl *control, float zoom)
324{
325 g_return_if_fail (EPHY_IS_ZOOM_CONTROL (control));
326
327 if (zoom < ZOOM_MINIMAL || zoom > ZOOM_MAXIMAL) return;
328
329 control->priv->zoom = zoom;
330 g_object_notify (G_OBJECT (control), "zoom");
331}
