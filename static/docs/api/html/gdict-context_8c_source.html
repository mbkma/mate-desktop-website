54GType
55gdict_context_get_type (void)
56{
57 static GType context_type = 0;
58
59 if (G_UNLIKELY (context_type == 0))
60 {
61 static GTypeInfo context_info =
62 {
63 sizeof (GdictContextIface),
64 NULL, /* base_init */
65 NULL, /* base_finalize */
66 (GClassInitFunc) gdict_context_class_init,
67 NULL, /* class_finalize */
68 NULL, /* class_data */
69 0, /* instance_size */
70 0, /* n_preallocs */
71 NULL, /* instance_init */
72 NULL /* value_table */
73 };
74
75 context_type = g_type_register_static (G_TYPE_INTERFACE,
76 "GdictContext",
77 &context_info, 0);
78 g_type_interface_add_prerequisite (context_type, G_TYPE_OBJECT);
79 }
80
81 return context_type;
82}
