62static gboolean
63do_run_overwrite_confirm_dialog (gpointer _data)
64{
65 ErrorDialogData *data = _data;
66 GtkWidget *dialog;
67 gint response;
68
69 /* we need to ask the user if they want to overwrite this file */
70 dialog = gtk_message_dialog_new (GTK_WINDOW (data->parent),
71 GTK_DIALOG_DESTROY_WITH_PARENT,
72 GTK_MESSAGE_QUESTION,
73 GTK_BUTTONS_NONE,
74 _("File already exists"));
75
76 gtk_message_dialog_format_secondary_text (GTK_MESSAGE_DIALOG (dialog),
77 _("The file \"%s\" already exists. "
78 "Would you like to replace it?"),
79 data->basename);
80 gtk_dialog_add_button (GTK_DIALOG (dialog),
81 "gtk-cancel",
82 GTK_RESPONSE_CANCEL);
83 gtk_dialog_add_button (GTK_DIALOG (dialog),
84 _("_Replace"),
85 GTK_RESPONSE_OK);
86
87 response = gtk_dialog_run (GTK_DIALOG (dialog));
88 gtk_widget_destroy (dialog);
89
90 data->resp = response;
91
92 return FALSE;
93}

109static gboolean
110transfer_progress_dialog_new (TransferJob *job)
111{

122 _("Saving file..."));
123 widget = gtk_progress_bar_new ();
124 gtk_box_pack_start (GTK_BOX (gtk_message_dialog_get_message_area GTK_MESSAGE_DIALOG (gdialog)),

127 dialog->progress_bar = widget;

130 g_signal_connect (gdialog,
131 "response",

151static int
152run_overwrite_confirm_dialog (TransferJob *job)
153{
154 ErrorDialogData *data;
155 gboolean need_timeout;
156 int response;
157 char *basename;
158
159 basename = g_file_get_basename (job->dest);
160
161 data = g_slice_new0 (ErrorDialogData);
162 data->parent = job->parent;
163 data->basename = basename;
164
165 need_timeout = (job->dialog_timeout_id > 0);
166
167 if (need_timeout)
168 {
169 g_source_remove (job->dialog_timeout_id);
170 job->dialog_timeout_id = 0;
171 }
172
173 g_io_scheduler_job_send_to_mainloop (job->io_job,
174 do_run_overwrite_confirm_dialog,
175 data,
176 NULL);
177 response = data->resp;
178
179 if (need_timeout)
180 transfer_progress_dialog_start (job);
181
182 g_free (basename);
183 g_slice_free (ErrorDialogData, data);
184
185 return response;
186}

202static gboolean
203transfer_job_done (gpointer user_data)
204{
205 TransferJob *job = user_data;
206 TransferDialog *dialog;
207
208 dialog = job->dialog;
209
210 if (job->dialog_timeout_id)
211 {
212 g_source_remove (job->dialog_timeout_id);
213 job->dialog_timeout_id = 0;
214 }
215 if (dialog)
216 gtk_widget_destroy (dialog->dialog);
217
218 if (job->callback)
219 {
220 (job->callback) (job->result,
221 job->error,
222 job->callback_data);
223 }
224
225 g_object_unref (job->source);
226 g_object_unref (job->dest);
227 g_object_unref (job->cancellable);
228
229 g_free (dialog);
230 g_free (job->error);
231 g_slice_free (TransferJob, job);
232
233 return FALSE;
234}

252static goffset
253get_file_size (GFile *file)
254{
255 GFileInfo *file_info;
256 goffset size;
257
258 file_info = g_file_query_info (file,
259 G_FILE_ATTRIBUTE_STANDARD_SIZE,
260 0, NULL, NULL);
261 if (file_info != NULL)
262 {
263 size = g_file_info_get_size (file_info);
264 g_object_unref (file_info);
265 }
266 else
267 {
268 /* this should never fail as the source file is always local and in /tmp,
269 * but you never know.
270 */
271 size = -1;
272 }
273
274 return size;
275}

277static gboolean
278transfer_file (GIOSchedulerJob *io_job,
279 GCancellable *cancellable,
280 gpointer user_data)
281{
282 TransferJob *job = user_data;
283 GError *error;
284
285 job->io_job = io_job;
286 job->total_bytes = get_file_size (job->source);
287 if (job->total_bytes == -1)
288 {
289 /* we can't access the source file, abort early */
290 error = NULL;
291 job->result = TRANSFER_ERROR;
292 job->error = g_strdup (_("Can't access source file"));
293
294 goto out;
295 }
296
297 transfer_progress_dialog_start (job);
298
299retry:
300 error = NULL;
301 if (!g_file_copy (job->source,
302 job->dest,
303 job->flags,
304 job->cancellable,
305 (GFileProgressCallback) transfer_progress_cb,
306 job,
307 &error))
308 {
309 if (error->code == G_IO_ERROR_EXISTS)
310 {
311 int response;
312 /* ask the user if he wants to overwrite, otherwise
313 * return and report what happened.
314 */
315 response = run_overwrite_confirm_dialog (job);
316 if (response == GTK_RESPONSE_OK)
317 {
318 job->flags |= G_FILE_COPY_OVERWRITE;
319 g_error_free (error);
320
321 goto retry;
322 }
323 else
324 {
325 g_cancellable_cancel (job->cancellable);
326 job->result = TRANSFER_OVERWRITE;
327 goto out;
328 }
329 }
330 else if (error->code == G_IO_ERROR_CANCELLED)
331 {
332 job->result = TRANSFER_CANCELLED;
333
334 goto out;
335 }
336 else
337 {
338 /* other vfs error, abort the transfer and report
339 * the error.
340 */
341 g_cancellable_cancel (job->cancellable);
342 job->result = TRANSFER_ERROR;
343 job->error = g_strdup (error->message);
344
345 goto out;
346 }
347 }
348 else
349 {
350 /* success */
351 job->result = TRANSFER_OK;
352
353 goto out;
354 }
355
356out:
357 if (error)
358 g_error_free (error);
359
360 g_io_scheduler_job_send_to_mainloop_async (io_job,
361 transfer_job_done,
362 job,
363 NULL);
364 return FALSE;
365}
