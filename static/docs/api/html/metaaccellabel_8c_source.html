241static gboolean
242meta_accel_label_draw (GtkWidget *widget,
243 cairo_t *cr)
244{
245 MetaAccelLabel *accel_label = META_ACCEL_LABEL (widget);
246 GtkTextDirection direction;
247 int ac_width;
248 GtkAllocation allocation;
249 GtkRequisition requisition;
250
251 direction = gtk_widget_get_direction (widget);
252 ac_width = meta_accel_label_get_accel_width (accel_label);
253 gtk_widget_get_allocation (widget, &allocation);
254 gtk_widget_get_preferred_size (widget,
255 &requisition, NULL);
256
257 if (allocation.width >= requisition.width + ac_width)
258 {
259 GtkStyleContext *style;
260 PangoLayout *label_layout;
261 PangoLayout *accel_layout;
262 GtkLabel *label = GTK_LABEL (widget);
263 gint x, y, xpad, ypad;
264 gint margin_start, margin_end, margin_top, margin_bottom;
265 gfloat yalign;
266
267 label_layout = gtk_label_get_layout (GTK_LABEL (accel_label));
268 yalign = gtk_label_get_yalign (GTK_LABEL (accel_label));
269
270 cairo_save (cr);
271
272 /* XXX: Mad hack: We modify the label's width so it renders
273 * properly in its draw function that we chain to. */
274 if (direction == GTK_TEXT_DIR_RTL)
275 cairo_translate (cr, ac_width, 0);
276 if (gtk_label_get_ellipsize (label))
277 pango_layout_set_width (label_layout,
278 pango_layout_get_width (label_layout)
279 - ac_width * PANGO_SCALE);
280
281 allocation.width -= ac_width;
282 gtk_widget_set_allocation (widget, &allocation);
283 if (GTK_WIDGET_CLASS (meta_accel_label_parent_class)->draw)
284 GTK_WIDGET_CLASS (meta_accel_label_parent_class)->draw (widget,
285 cr);
286 allocation.width += ac_width;
287 gtk_widget_set_allocation (widget, &allocation);
288 if (gtk_label_get_ellipsize (label))
289 pango_layout_set_width (label_layout,
290 pango_layout_get_width (label_layout)
291 + ac_width * PANGO_SCALE);
292
293 cairo_restore (cr);
294
295 margin_start = gtk_widget_get_margin_start (widget);
296 margin_end = gtk_widget_get_margin_end (widget);
297 margin_top = gtk_widget_get_margin_top (widget);
298 margin_bottom = gtk_widget_get_margin_bottom (widget);
299
300 xpad = margin_start + margin_end;
301 ypad = margin_top + margin_bottom;
302
303 if (direction == GTK_TEXT_DIR_RTL)
304 x = xpad;
305 else
306 x = gtk_widget_get_allocated_width (widget) - xpad - ac_width;
307
308 gtk_label_get_layout_offsets (GTK_LABEL (accel_label), NULL, &y);
309
310 accel_layout = gtk_widget_create_pango_layout (widget, accel_label->accel_string);
311
312 y = (allocation.height - (requisition.height - ypad * 2)) * yalign + 1.5;
313
314 style = gtk_widget_get_style_context (widget);
315 gtk_style_context_save (style);
316 gtk_style_context_set_state (style,
317 gtk_widget_get_state_flags (widget));
318 gtk_render_layout (gtk_widget_get_style_context (widget),
319 cr,
320 x, y,
321 accel_layout);
322 gtk_style_context_restore (style);
323
324 g_object_unref (accel_layout);
325 }
326 else
327 {
328 if (GTK_WIDGET_CLASS (meta_accel_label_parent_class)->draw)
329 GTK_WIDGET_CLASS (meta_accel_label_parent_class)->draw (widget, cr);
330 }
331
332 return FALSE;
333}

335static void
336meta_accel_label_update (MetaAccelLabel *accel_label)
337{
338 MetaAccelLabelClass *class;
339 GString *gstring;
340 gboolean seen_mod = FALSE;
341 gunichar ch;
342
343 g_return_if_fail (META_IS_ACCEL_LABEL (accel_label));
344
345 class = META_ACCEL_LABEL_GET_CLASS (accel_label);
346
347 g_free (accel_label->accel_string);
348 accel_label->accel_string = NULL;
349
350 gstring = g_string_new (accel_label->accel_string);
351 g_string_append (gstring, gstring->len ? class->accel_seperator : " ");
352
353 if (accel_label->accel_mods & META_VIRTUAL_SHIFT_MASK)
354 {
355 g_string_append (gstring, class->mod_name_shift);
356 seen_mod = TRUE;
357 }
358 if (accel_label->accel_mods & META_VIRTUAL_CONTROL_MASK)
359 {
360 if (seen_mod)
361 g_string_append (gstring, class->mod_separator);
362 g_string_append (gstring, class->mod_name_control);
363 seen_mod = TRUE;
364 }
365 if (accel_label->accel_mods & META_VIRTUAL_ALT_MASK)
366 {
367 if (seen_mod)
368 g_string_append (gstring, class->mod_separator);
369 g_string_append (gstring, class->mod_name_alt);
370 seen_mod = TRUE;
371 }
372 if (accel_label->accel_mods & META_VIRTUAL_META_MASK)
373 {
374 if (seen_mod)
375 g_string_append (gstring, class->mod_separator);
376 g_string_append (gstring, class->mod_name_meta);
377 seen_mod = TRUE;
378 }
379 if (accel_label->accel_mods & META_VIRTUAL_SUPER_MASK)
380 {
381 if (seen_mod)
382 g_string_append (gstring, class->mod_separator);
383 g_string_append (gstring, class->mod_name_super);
384 seen_mod = TRUE;
385 }
386 if (accel_label->accel_mods & META_VIRTUAL_HYPER_MASK)
387 {
388 if (seen_mod)
389 g_string_append (gstring, class->mod_separator);
390 g_string_append (gstring, class->mod_name_hyper);
391 seen_mod = TRUE;
392 }
393 if (accel_label->accel_mods & META_VIRTUAL_MOD2_MASK)
394 {
395 if (seen_mod)
396 g_string_append (gstring, class->mod_separator);
397 g_string_append (gstring, class->mod_name_mod2);
398 seen_mod = TRUE;
399 }
400 if (accel_label->accel_mods & META_VIRTUAL_MOD3_MASK)
401 {
402 if (seen_mod)
403 g_string_append (gstring, class->mod_separator);
404 g_string_append (gstring, class->mod_name_mod3);
405 seen_mod = TRUE;
406 }
407 if (accel_label->accel_mods & META_VIRTUAL_MOD4_MASK)
408 {
409 if (seen_mod)
410 g_string_append (gstring, class->mod_separator);
411 g_string_append (gstring, class->mod_name_mod4);
412 seen_mod = TRUE;
413 }
414 if (accel_label->accel_mods & META_VIRTUAL_MOD5_MASK)
415 {
416 if (seen_mod)
417 g_string_append (gstring, class->mod_separator);
418 g_string_append (gstring, class->mod_name_mod5);
419 seen_mod = TRUE;
420 }
421
422 if (seen_mod)
423 g_string_append (gstring, class->mod_separator);
424
425 ch = gdk_keyval_to_unicode (accel_label->accel_key);
426 if (ch && (g_unichar_isgraph (ch) || ch == ' ') &&
427 (ch < 0x80 || class->latin1_to_char))
428 {
429 switch (ch)
430 {
431 case ' ':
432 g_string_append (gstring, "Space");
433 break;
434 case '\\':
435 g_string_append (gstring, "Backslash");
436 break;
437 default:
438 g_string_append_unichar (gstring, g_unichar_toupper (ch));
439 break;
440 }
441 }
442 else
443 {
444 gchar *tmp;
445
446 tmp = gtk_accelerator_name (accel_label->accel_key, 0);
447 if (tmp[0] != 0 && tmp[1] == 0)
448 tmp[0] = g_ascii_toupper (tmp[0]);
449 g_string_append (gstring, tmp);
450 g_free (tmp);
451 }
452
453 g_free (accel_label->accel_string);
454 accel_label->accel_string = gstring->str;
455 g_string_free (gstring, FALSE);
456
457 g_assert (accel_label->accel_string);
458 /* accel_label->accel_string = g_strdup ("-/-"); */
459
460 gtk_widget_queue_resize (GTK_WIDGET (accel_label));
461}
