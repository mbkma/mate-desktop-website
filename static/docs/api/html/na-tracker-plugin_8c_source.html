46/*
47 * A caja extension must implement three functions :
48 *
49 * - caja_module_initialize
50 * - caja_module_list_types
51 * - caja_module_shutdown
52 *
53 * The first two functions are called at caja startup.
54 *
55 * The prototypes for these functions are defined in caja-extension-types.h
56 */
57
58void
59caja_module_initialize( GTypeModule *module )
60{
61 static const gchar *thisfn = "caja_module_initialize";
62
63 syslog( LOG_USER | LOG_INFO, "[N-A] %s Tracker %s initializing...", PACKAGE_NAME, PACKAGE_VERSION );
64
65 set_log_handler();
66
67 g_debug( "%s: module=%p", thisfn, ( void * ) module );
68
69 g_type_module_set_name( module, PACKAGE_STRING );
70
71 na_tracker_register_type( module );
72}

87void
88caja_module_shutdown( void )
89{
90 static const gchar *thisfn = "caja_module_shutdown";
91
92 g_debug( "%s", thisfn );
93
94 /* remove the log handler
95 * almost useless as the process is nonetheless terminating at this time
96 * but this is the art of coding...
97 */
98 if( st_default_log_func ){
99 g_log_set_default_handler( st_default_log_func, NULL );
100 st_default_log_func = NULL;
101 }
102}

104/*
105 * a log handler that we install when in development mode in order to be
106 * able to log plugin runtime
107 * TODO: the debug flag should be dynamic, so that an advanced user could
108 * setup a given key and obtain a full log to send to Bugzilla..
109 * For now, is always install when compiled in maintainer mode, never else
110 */
111static void
112set_log_handler( void )
113{
114 st_default_log_func = g_log_set_default_handler(( GLogFunc ) log_handler, NULL );
115}

117/*
118 * we used to install a log handler for each and every log domain used
119 * in Caja-Actions ; this led to a fastidious enumeration
120 * instead we install a default log handler which will receive all
121 * debug messages, i.e. not only from N-A, but also from other code
122 * in the Caja process
123 */
124static void
125log_handler( const gchar *log_domain, GLogLevelFlags log_level, const gchar *message, gpointer user_data )
126{
127 gchar *tmp;
128
129 tmp = g_strdup( "" );
130 if( log_domain && strlen( log_domain )){
131 g_free( tmp );
132 tmp = g_strdup_printf( "[%s] ", log_domain );
133 }
134
135#ifdef NA_MAINTAINER_MODE
136 /*( *st_default_log_func )( log_domain, log_level, message, user_data );*/
137 syslog( LOG_USER | LOG_DEBUG, "%s%s", tmp, message );
138#else
139 if( g_getenv( CAJA_ACTIONS_DEBUG )){
140 syslog( LOG_USER | LOG_DEBUG, "%s%s", tmp, message );
141 }
142#endif
143
144 g_free( tmp );
145}
