33/*
34 * SECTION: ioptions_list
35 * @title: NAIOptionsList
36 * @short_description: The OptionsList Interface v 1
37 * @include: core/na-ioptions-list.h
38 *
39 * The #NAIOptionsList interface is to be used when we have one option to choose
40 * among several. The interface takes care of:
41 * <itemizedlist>
42 * <listitem>
43 * <para>
44 * displaying the items, either as a radio button group or as a tree view,
45 * inside of a parent container;
46 * </para>
47 * <para>
48 * maybe displaying a tooltip and/or a pixbuf for each option;
49 * </para>
50 * <para>
51 * maybe adding an 'Ask me' option;
52 * </para>
53 * <para>
54 * setting a default option, identified by its identifier, which can
55 * be anything resolvable to a #gpointer value;
56 * </para>
57 * <para>
58 * returning the selected option as a #gpointer value;
59 * </para>
60 * <para>
61 * freeing the allocated resources at end.
62 * </para>
63 * </listitem>
64 * </itemizedlist>
65 *
66 * In order this interface to work, each option has to be seen as a #GObject
67 * -derived object, which itself should implement the #NAIOption companion
68 * interface.
69 *
70 * Note that the instance which implements this interface is able to provide
71 * several list of values. This is the reason why each list of values is
72 * identified by its container inside of the window's instance.
73 *
74 * Instance initialization
75 *
76 * Rather that having to call a na_ioptions_list_instance_init() on each
77 * instance implementation, we prefer to check when entering in each public
78 * method for an 'initialized' flag both at the instance level and at the
79 * container level.
80 *
81 * Alternative might be be to have the na_ioptions_list_instance_init()
82 * initialization interface method, which itself connect to #BaseWindow
83 * signals. But this would not prevent of initializing for both managed
84 * containers...
85 *
86 * Initialization mainly consists of defining weak references both on
87 * instance and on containers levels.
88 *
89 * <refsect2>
90 * <title>Versions historic</title>
91 * <table>
92 * <title>Historic of the versions of the #NAIOptionsList interface</title>
93 * <tgroup rowsep="1" colsep="1" align="center" cols="3">
94 * <colspec colname="na-version" />
95 * <colspec colname="api-version" />
96 * <colspec colname="current" />
97 * <thead>
98 * <row>
99 * <entry>&prodname; version</entry>
100 * <entry>#NAIOptionsList interface version</entry>
101 * <entry></entry>
102 * </row>
103 * </thead>
104 * <tbody>
105 * <row>
106 * <entry>since 3.2</entry>
107 * <entry>1</entry>
108 * <entry>current version</entry>
109 * </row>
110 * </tbody>
111 * </tgroup>
112 * </table>
113 * </refsect2>
114 */

132/*
133 * NAIOptionsListInterface:
134 * @get_version: returns the version of this interface that the
135 * instance implements.
136 * @get_options: returns the list of #NAIOption.
137 * @free_options: releases the list of #NAIOption.
138 * @get_ask_option: returns the 'Ask me' #NAIOption.
139 * @free_ask_option: releases the 'Ask me' #NAIOption.
140 *
141 * This defines the interface that a #NAIOptionsList implementation should provide.
142 */
143typedef struct {
144 /*< private >*/
145 GTypeInterface parent;
146 NAIOptionsListInterfacePrivate *private;
147
148 /*< public >*/
149 /*
150 * get_version:
151 * @instance: the #NAIOptionsList instance of the implementation.
152 *
153 * This method is supposed to let know to any caller which version of this
154 * interface the implementation provides. This may be useful when this
155 * interface will itself be upgraded.
156 *
157 * If this method is not provided by the implementation, one should suppose
158 * that the implemented version is at last the version 1.
159 *
160 * Returns: the version of this interface provided by the implementation.
161 *
162 * Since: 3.2
163 */
164 guint ( *get_version ) ( const NAIOptionsList *instance );
165
166 /*
167 * get_options:
168 * @instance: the #NAIOptionsList instance of the implementation.
169 * @container: the #GtkWidget which embeds the list of values.
170 *
171 * This method may be called at more or less early stage of the build
172 * of the display, either rather early when displaying a radio button
173 * group, or later in the case of a tree view.
174 *
175 * Returns: a #GList list of #NAIOptions object instances.
176 *
177 * Since: 3.2
178 */
179 GList * ( *get_options ) ( const NAIOptionsList *instance, GtkWidget *container );
180
181 /*
182 * free_options:
183 * @instance: the #NAIOptionsList instance of the implementation.
184 * @container: the #GtkWidget which embeds the list of values.
185 * @options: a #GList of #NAIoption objects as returned by get_options() method.
186 *
187 * Release the resources allocated to the @options list.
188 *
189 * Note that the interface defaults to just g_object_unref() each
190 * instance of the @options list, the g_list_free() the list itself.
191 * So if your implementation may satisfy itself with this default, you
192 * just do not need to implement the method.
193 *
194 * Since: 3.2
195 */
196 void ( *free_options )( const NAIOptionsList *instance, GtkWidget *container, GList *options );
197
198 /*
199 * get_ask_option:
200 * @instance: the #NAIOptionsList instance of the implementation.
201 * @container: the #GtkWidget which embeds the list of values.
202 *
203 * Ask the implementation to provide a #NAIOption which defines the
204 * 'Ask me' option.
205 *
206 * Returns: a #NAIOption which defines the 'Ask me' option.
207 *
208 * Since: 3.2
209 */
210 NAIOption * ( *get_ask_option ) ( const NAIOptionsList *instance, GtkWidget *container );
211
212 /*
213 * free_ask_option:
214 * @instance: the #NAIOptionsList instance of the implementation.
215 * @container: the #GtkWidget which embeds the list of values.
216 * @ask_option: the #NAIoption to be released.
217 *
218 * Release the resources allocated to the @ask_option instance.
219 *
220 * Note that the interface defaults to just g_object_unref() the object.
221 * So if your implementation may satisfy itself with this default, you just
222 * do not need to implement the method.
223 *
224 * Since: 3.2
225 */
226 void ( *free_ask_option )( const NAIOptionsList *instance, GtkWidget *container, NAIOption *ask_option );
227}
