49static void
50namespace_watcher_stop (gpointer data)
51{
52 NamespaceWatcher *watcher = data;
53
54 g_cancellable_cancel (watcher->cancellable);
55 g_object_unref (watcher->cancellable);
56
57 if (watcher->subscription_id)
58 g_dbus_connection_signal_unsubscribe (watcher->connection, watcher->subscription_id);
59
60 if (watcher->vanished_handler)
61 {
62 GHashTableIter it;
63 const gchar *name;
64
65 g_hash_table_iter_init (&it, watcher->names);
66 while (g_hash_table_iter_next (&it, (gpointer *) &name, NULL))
67 watcher->vanished_handler (watcher->connection, name, watcher->user_data);
68 }
69
70 if (watcher->user_data_destroy)
71 watcher->user_data_destroy (watcher->user_data);
72
73 if (watcher->connection)
74 {
75 g_signal_handlers_disconnect_by_func (watcher->connection, namespace_watcher_stop, watcher);
76 g_object_unref (watcher->connection);
77 }
78
79 g_hash_table_unref (watcher->names);
80
81 g_hash_table_remove (namespace_watcher_watchers, GUINT_TO_POINTER (watcher->id));
82 if (g_hash_table_size (namespace_watcher_watchers) == 0)
83 g_clear_pointer (&namespace_watcher_watchers, g_hash_table_destroy);
84
85 g_free (watcher);
86}

88static void
89namespace_watcher_name_appeared (NamespaceWatcher *watcher,
90 const gchar *name,
91 const gchar *owner)
92{
93 /* There's a race between NameOwnerChanged signals arriving and the
94 * ListNames/GetNameOwner sequence returning, so this function might
95 * be called more than once for the same name. To ensure that
96 * appeared_handler is only called once for each name, it is only
97 * called when inserting the name into watcher->names (each name is
98 * only inserted once there).
99 */
100 if (g_hash_table_contains (watcher->names, name))
101 return;
102
103 g_hash_table_add (watcher->names, g_strdup (name));
104
105 if (watcher->appeared_handler)
106 watcher->appeared_handler (watcher->connection, name, owner, watcher->user_data);
107}

109static void
110namespace_watcher_name_vanished (NamespaceWatcher *watcher,
111 const gchar *name)
112{
113 if (g_hash_table_remove (watcher->names, name) && watcher->vanished_handler)
114 watcher->vanished_handler (watcher->connection, name, watcher->user_data);
115}

117static gboolean
118dbus_name_has_namespace (const gchar *name,
119 const gchar *name_space)
120{
121 gint len_name;
122 gint len_namespace;
123
124 len_name = strlen (name);
125 len_namespace = strlen (name_space);
126
127 if (len_name < len_namespace)
128 return FALSE;
129
130 if (memcmp (name_space, name, len_namespace) != 0)
131 return FALSE;
132
133 return len_namespace == len_name || name[len_namespace] == '.';
134}

136static void
137name_owner_changed (GDBusConnection *connection,
138 const gchar *sender_name,
139 const gchar *object_path,
140 const gchar *interface_name,
141 const gchar *signal_name,
142 GVariant *parameters,
143 gpointer user_data)
144{
145 NamespaceWatcher *watcher = user_data;
146 const gchar *name;
147 const gchar *old_owner;
148 const gchar *new_owner;
149
150 g_variant_get (parameters, "(&s&s&s)", &name, &old_owner, &new_owner);
151
152 if (old_owner[0] != '\0')
153 namespace_watcher_name_vanished (watcher, name);
154
155 if (new_owner[0] != '\0')
156 namespace_watcher_name_appeared (watcher, name, new_owner);
157}

159static void
160got_name_owner (GObject *object,
161 GAsyncResult *result,
162 gpointer user_data)
163{
164 GetNameOwnerData *data = user_data;
165 GError *error = NULL;
166 GVariant *reply;
167 const gchar *owner;
168
169 reply = g_dbus_connection_call_finish (G_DBUS_CONNECTION (object), result, &error);
170
171 if (g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
172 {
173 g_error_free (error);
174 goto out;
175 }
176
177 if (reply == NULL)
178 {
179 if (!g_error_matches (error, G_DBUS_ERROR, G_DBUS_ERROR_NAME_HAS_NO_OWNER))
180 g_warning ("bus_watch_namespace: error calling org.freedesktop.DBus.GetNameOwner: %s", error->message);
181 g_error_free (error);
182 goto out;
183 }
184
185 g_variant_get (reply, "(&s)", &owner);
186 namespace_watcher_name_appeared (data->watcher, data->name, owner);
187
188 g_variant_unref (reply);
189
190out:
191 g_free (data->name);
192 g_slice_free (GetNameOwnerData, data);
193}

195static void
196names_listed (GObject *object,
197 GAsyncResult *result,
198 gpointer user_data)
199{
200 NamespaceWatcher *watcher;
201 GError *error = NULL;
202 GVariant *reply;
203 GVariantIter *iter;
204 const gchar *name;
205
206 reply = g_dbus_connection_call_finish (G_DBUS_CONNECTION (object), result, &error);
207
208 if (g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
209 {
210 g_error_free (error);
211 return;
212 }
213
214 watcher = user_data;
215
216 if (reply == NULL)
217 {
218 g_warning ("bus_watch_namespace: error calling org.freedesktop.DBus.ListNames: %s", error->message);
219 g_error_free (error);
220 return;
221 }
222
223 g_variant_get (reply, "(as)", &iter);
224 while (g_variant_iter_next (iter, "&s", &name))
225 {
226 if (dbus_name_has_namespace (name, watcher->name_space))
227 {
228 GetNameOwnerData *data = g_slice_new (GetNameOwnerData);
229 data->watcher = watcher;
230 data->name = g_strdup (name);
231 g_dbus_connection_call (watcher->connection, "org.freedesktop.DBus", "/",
232 "org.freedesktop.DBus", "GetNameOwner",
233 g_variant_new ("(s)", name), G_VARIANT_TYPE ("(s)"),
234 G_DBUS_CALL_FLAGS_NONE, -1, watcher->cancellable,
235 got_name_owner, data);
236 }
237 }
238
239 g_variant_iter_free (iter);
240 g_variant_unref (reply);
241}

254static void
255got_bus (GObject *object,
256 GAsyncResult *result,
257 gpointer user_data)
258{
259 GDBusConnection *connection;
260 NamespaceWatcher *watcher;
261 GError *error = NULL;
262
263 connection = g_bus_get_finish (result, &error);
264
265 if (g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
266 {
267 g_error_free (error);
268 return;
269 }
270
271 watcher = user_data;
272
273 if (connection == NULL)
274 {
275 namespace_watcher_stop (watcher);
276 return;
277 }
278
279 watcher->connection = connection;
280 g_signal_connect (watcher->connection, "closed", G_CALLBACK (connection_closed), watcher);
281
282 watcher->subscription_id =
283 g_dbus_connection_signal_subscribe (watcher->connection, "org.freedesktop.DBus",
284 "org.freedesktop.DBus", "NameOwnerChanged", "/org/freedesktop/DBus",
285 watcher->name_space, G_DBUS_SIGNAL_FLAGS_MATCH_ARG0_NAMESPACE,
286 name_owner_changed, watcher, NULL);
287
288 g_dbus_connection_call (watcher->connection, "org.freedesktop.DBus", "/",
289 "org.freedesktop.DBus", "ListNames", NULL, G_VARIANT_TYPE ("(as)"),
290 G_DBUS_CALL_FLAGS_NONE, -1, watcher->cancellable,
291 names_listed, watcher);
292}

294guint
295bus_watch_namespace (GBusType bus_type,
296 const gchar *name_space,
297 GBusNameAppearedCallback appeared_handler,
298 GBusNameVanishedCallback vanished_handler,
299 gpointer user_data,
300 GDestroyNotify user_data_destroy)
301{
302 NamespaceWatcher *watcher;
303
304 /* same rules for interfaces and well-known names */
305 g_return_val_if_fail (name_space != NULL && g_dbus_is_interface_name (name_space), 0);
306 g_return_val_if_fail (appeared_handler || vanished_handler, 0);
307
308 watcher = g_new0 (NamespaceWatcher, 1);
309 watcher->id = namespace_watcher_next_id++;
310 watcher->name_space = g_strdup (name_space);
311 watcher->appeared_handler = appeared_handler;
312 watcher->vanished_handler = vanished_handler;
313 watcher->user_data = user_data;
314 watcher->user_data_destroy = user_data_destroy;
315 watcher->cancellable = g_cancellable_new ();;
316 watcher->names = g_hash_table_new_full (g_str_hash, g_str_equal, g_free, NULL);
317
318 if (namespace_watcher_watchers == NULL)
319 namespace_watcher_watchers = g_hash_table_new (g_direct_hash, g_direct_equal);
320 g_hash_table_insert (namespace_watcher_watchers, GUINT_TO_POINTER (watcher->id), watcher);
321
322 g_bus_get (bus_type, watcher->cancellable, got_bus, watcher);
323
324 return watcher->id;
325}

327void
328bus_unwatch_namespace (guint id)
329{
330 /* namespace_watcher_stop() might have already removed the watcher
331 * with @id in the case of a connection error. Thus, this function
332 * doesn't warn when @id is absent from the hash table.
333 */
334
335 if (namespace_watcher_watchers)
336 {
337 NamespaceWatcher *watcher;
338
339 watcher = g_hash_table_lookup (namespace_watcher_watchers, GUINT_TO_POINTER (id));
340 if (watcher)
341 {
342 /* make sure vanished() is not called as a result of this function */
343 g_hash_table_remove_all (watcher->names);
344
345 namespace_watcher_stop (watcher);
346 }
347 }
