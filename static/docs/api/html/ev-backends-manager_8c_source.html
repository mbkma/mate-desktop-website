72static EvBackendInfo *
73ev_backends_manager_load_backend (const gchar *file)
74{
75 EvBackendInfo *info;
76 GKeyFile *backend_file = NULL;
77 GError *error = NULL;
78
79 backend_file = g_key_file_new ();
80 if (!g_key_file_load_from_file (backend_file, file, G_KEY_FILE_NONE, &error)) {
81 g_warning ("Error opening backend file %s: %s",
82 file, error->message);
83 g_error_free (error);
84 g_key_file_free (backend_file);
85
86 return NULL;
87 }
88
89 info = g_new0 (EvBackendInfo, 1);
90 info->module_name = g_key_file_get_string (backend_file, EV_BACKENDS_GROUP,
91 "Module", NULL);
92 if (!info->module_name) {
93 g_warning ("Bad atril backend file %s: Could not find 'Module'",
94 file);
95 ev_backend_info_free (info);
96 g_key_file_free (backend_file);
97
98 return NULL;
99 }
100
101 info->resident = g_key_file_get_boolean (backend_file, EV_BACKENDS_GROUP,
102 "Resident", NULL);
103
104 info->type_desc = g_key_file_get_locale_string (backend_file, EV_BACKENDS_GROUP,
105 "TypeDescription", NULL, NULL);
106 if (!info->type_desc) {
107 g_warning ("Bad atril backend file %s: Could not find 'TypeDescription'",
108 file);
109 ev_backend_info_free (info);
110 g_key_file_free (backend_file);
111
112 return NULL;
113 }
114
115 info->mime_types = g_key_file_get_string_list (backend_file, EV_BACKENDS_GROUP,
116 "MimeType", NULL, NULL);
117 if (!info->mime_types) {
118 g_warning ("Bad atril backend file %s: Could not find 'MimeType'",
119 file);
120 ev_backend_info_free (info);
121 g_key_file_free (backend_file);
122
123 return NULL;
124 }
125
126 g_key_file_free (backend_file);
127
128 return info;
129}

131static gboolean
132ev_backends_manager_load (void)
133{
134 GDir *dir;
135 const gchar *dirent;
136 GError *error = NULL;
137
138 dir = g_dir_open (backends_dir(), 0, &error);
139 if (!dir) {
140 g_warning ("%s", error->message);
141 g_error_free (error);
142
143 return FALSE;
144 }
145
146 while ((dirent = g_dir_read_name (dir))) {
147 EvBackendInfo *info;
148 gchar *file;
149
150 if (!g_str_has_suffix (dirent, EV_BACKENDS_EXTENSION))
151 continue;
152
153 file = g_build_filename (backends_dir(), dirent, NULL);
154 info = ev_backends_manager_load_backend (file);
155 g_free (file);
156
157 if (!info)
158 continue;
159
160 ev_backends_list = g_list_prepend (ev_backends_list, info);
161 }
162
163 g_dir_close (dir);
164
165 return ev_backends_list != NULL;
166}

199static EvBackendInfo *
200ev_backends_manager_get_backend_info (const gchar *mime_type)
201{
202 GList *l;
203
204 for (l = ev_backends_list; l; l = g_list_next (l)) {
205 EvBackendInfo *info;
206 gint i = 0;
207 const char *mime;
208
209 info = (EvBackendInfo *)l->data;
210
211 while ((mime = info->mime_types[i++])) {
212 if (g_ascii_strcasecmp (mime, mime_type) == 0)
213 return info;
214 }
215 }
216
217 return NULL;
218}

220EvDocument *
221ev_backends_manager_get_document (const gchar *mime_type)
222{
223 EvDocument *document;
224 EvBackendInfo *info;
225
226 info = ev_backends_manager_get_backend_info (mime_type);
227 if (!info)
228 return NULL;
229
230 if (!info->module) {
231 gchar *path;
232
233 path = g_module_build_path (backends_dir(), info->module_name);
234 info->module = G_TYPE_MODULE (ev_module_new (path, info->resident));
235 g_free (path);
236 }
237
238 if (!g_type_module_use (info->module)) {
239 g_warning ("Cannot load backend '%s' since file '%s' cannot be read.",
240 info->module_name,
241 ev_module_get_path (EV_MODULE (info->module)));
242 g_object_unref (G_OBJECT (info->module));
243 info->module = NULL;
244
245 return NULL;
246 }
247
248 document = EV_DOCUMENT (ev_module_new_object (EV_MODULE (info->module)));
249 g_type_module_unuse (info->module);
250
251 return document;
252}

254static EvBackendInfo *
255get_document_backend_info (EvDocument *document)
256{
257 GList *l;
258
259 for (l = ev_backends_list; l; l = g_list_next (l)) {
260 EvBackendInfo *info;
261 GType type_id;
262
263 info = (EvBackendInfo *)l->data;
264
265 if (!info->module)
266 continue;
267
268 type_id = ev_module_get_object_type (EV_MODULE (info->module));
269
270 if (G_TYPE_CHECK_INSTANCE_TYPE (document, type_id)) {
271 return info;
272 }
273 }
274
275 return NULL;
276}
