58void
59panel_init_stock_icons_and_items (void)
60{
61 GSettings *settings;
62 gint icon_size;
63
64 settings = g_settings_new (PANEL_MENU_BAR_SCHEMA);
65
66 icon_size = g_settings_get_enum (settings, "item-icon-size");
67 if (icon_size <= 0) {
68 panel_menu_icon_size = gtk_icon_size_register ("panel-menu",
69 PANEL_DEFAULT_MENU_ICON_SIZE,
70 PANEL_DEFAULT_MENU_ICON_SIZE);
71 } else {
72 /* underscores to prevent themes from altering these settings */
73 panel_menu_icon_size = gtk_icon_size_register ("__panel-menu",
74 icon_size,
75 icon_size);
76 }
77
78 icon_size = g_settings_get_enum (settings, "icon-size");
79 if (icon_size <= 0) {
80 panel_menu_bar_icon_size = gtk_icon_size_register ("panel-foobar",
81 PANEL_DEFAULT_MENU_BAR_ICON_SIZE,
82 PANEL_DEFAULT_MENU_BAR_ICON_SIZE);
83 } else {
84 panel_menu_bar_icon_size = gtk_icon_size_register ("__panel-foobar",
85 icon_size,
86 icon_size);
87 }
88
89 panel_add_to_icon_size = gtk_icon_size_register ("panel-add-to",
90 PANEL_ADD_TO_DEFAULT_ICON_SIZE,
91 PANEL_ADD_TO_DEFAULT_ICON_SIZE);
92
93
94 g_object_unref (settings);
95}
