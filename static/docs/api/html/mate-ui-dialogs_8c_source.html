142GtkWidget *
143mate_ui_dialog_about_new(GtkWindow *parent,
144 const MateUiAboutInfo *info)
145{
146 g_return_val_if_fail(info != NULL, NULL);
147 g_return_val_if_fail(info->program_name != NULL, NULL);
148
149 GtkWidget *dialog = gtk_about_dialog_new();
150 GtkAboutDialog *about = GTK_ABOUT_DIALOG(dialog);
151
152 gtk_about_dialog_set_program_name(about, info->program_name);
153
154 if (info->version != NULL)
155 gtk_about_dialog_set_version(about, info->version);
156
157 if (info->copyright != NULL)
158 gtk_about_dialog_set_copyright(about, info->copyright);
159
160 if (info->comments != NULL)
161 gtk_about_dialog_set_comments(about, info->comments);
162
163 if (info->license != NULL)
164 {
165 gtk_about_dialog_set_license(about, info->license);
166 gtk_about_dialog_set_wrap_license(about, TRUE);
167 }
168 else
169 {
170 gtk_about_dialog_set_license(about, gpl_2_0_text);
171 gtk_about_dialog_set_wrap_license(about, TRUE);
172 }
173
174 if (info->website != NULL)
175 gtk_about_dialog_set_website(about, info->website);
176
177 if (info->website_label != NULL)
178 gtk_about_dialog_set_website_label(about, info->website_label);
179
180 if (info->authors != NULL)
181 gtk_about_dialog_set_authors(about, info->authors);
182
183 if (info->documenters != NULL)
184 gtk_about_dialog_set_documenters(about, info->documenters);
185
186 if (info->translator_credits != NULL)
187 gtk_about_dialog_set_translator_credits(about, info->translator_credits);
188
189 if (info->artists != NULL)
190 gtk_about_dialog_set_artists(about, info->artists);
191
192 if (info->logo_icon_name != NULL)
193 gtk_about_dialog_set_logo_icon_name(about, info->logo_icon_name);
194
195 if (parent != NULL)
196 {
197 gtk_window_set_transient_for(GTK_WINDOW(dialog), parent);
198 gtk_window_set_modal(GTK_WINDOW(dialog), TRUE);
199 }
200
201 gtk_window_set_destroy_with_parent(GTK_WINDOW(dialog), TRUE);
202
203 /* Auto-close on response */
204 g_signal_connect(dialog, "response", G_CALLBACK(gtk_widget_destroy), NULL);
205
206 return dialog;
207}

260gint
261mate_ui_dialog_message(GtkWindow *parent,
262 GtkMessageType type,
263 GtkButtonsType buttons,
264 const gchar *primary,
265 const gchar *secondary)
266{
267 g_return_val_if_fail(primary != NULL, GTK_RESPONSE_NONE);
268

270 GTK_DIALOG_MODAL | GTK_DIALOG_DESTROY_WITH_PARENT,
271 type,

273 "%s", primary);
274
275 if (secondary != NULL)
276 {
277 gtk_message_dialog_format_secondary_text(GTK_MESSAGE_DIALOG(dialog),
278 "%s", secondary);
279 }
280
281 gint response = gtk_dialog_run(GTK_DIALOG(dialog));
282 gtk_widget_destroy(dialog);
283

370gboolean
371mate_ui_dialog_confirm(GtkWindow *parent,
372 const gchar *primary,
373 const gchar *secondary,
374 const gchar *confirm_label,
375 gboolean is_destructive)
376{
377 g_return_val_if_fail(primary != NULL, FALSE);
378 g_return_val_if_fail(confirm_label != NULL, FALSE);
379
380 GtkWidget *dialog = gtk_message_dialog_new(parent,
381 GTK_DIALOG_MODAL | GTK_DIALOG_DESTROY_WITH_PARENT,
382 is_destructive ? GTK_MESSAGE_WARNING : GTK_MESSAGE_QUESTION,
383 GTK_BUTTONS_NONE,
384 "%s", primary);
385
386 if (secondary != NULL)
387 {
388 gtk_message_dialog_format_secondary_text(GTK_MESSAGE_DIALOG(dialog),
389 "%s", secondary);
390 }
391
392 gtk_dialog_add_button(GTK_DIALOG(dialog), "_Cancel", GTK_RESPONSE_CANCEL);
393 GtkWidget *confirm_btn = gtk_dialog_add_button(GTK_DIALOG(dialog),
394 confirm_label, GTK_RESPONSE_ACCEPT);
395
396 if (is_destructive)
397 {
398 GtkStyleContext *ctx = gtk_widget_get_style_context(confirm_btn);
399 gtk_style_context_add_class(ctx, "destructive-action");
400 }
401
402 gtk_dialog_set_default_response(GTK_DIALOG(dialog), GTK_RESPONSE_CANCEL);
403
404 gint response = gtk_dialog_run(GTK_DIALOG(dialog));
405 gtk_widget_destroy(dialog);
406
407 return (response == GTK_RESPONSE_ACCEPT);
408}

421gchar *
422mate_ui_dialog_file_chooser_open(GtkWindow *parent,
423 const gchar *title,
424 const gchar *filter_name,
425 const gchar *filter_pattern)
426{
427 GtkWidget *dialog = gtk_file_chooser_dialog_new(title,
428 parent,
429 GTK_FILE_CHOOSER_ACTION_OPEN,
430 "_Cancel", GTK_RESPONSE_CANCEL,
431 "_Open", GTK_RESPONSE_ACCEPT,
432 NULL);
433
434 if (filter_name != NULL && filter_pattern != NULL)
435 {
436 GtkFileFilter *filter = gtk_file_filter_new();
437 gtk_file_filter_set_name(filter, filter_name);
438 gtk_file_filter_add_pattern(filter, filter_pattern);
439 gtk_file_chooser_add_filter(GTK_FILE_CHOOSER(dialog), filter);
440
441 /* Also add "All files" filter */
442 GtkFileFilter *all_filter = gtk_file_filter_new();
443 gtk_file_filter_set_name(all_filter, "All files");
444 gtk_file_filter_add_pattern(all_filter, "*");
445 gtk_file_chooser_add_filter(GTK_FILE_CHOOSER(dialog), all_filter);
446 }
447
448 gchar *filename = NULL;
449 if (gtk_dialog_run(GTK_DIALOG(dialog)) == GTK_RESPONSE_ACCEPT)
450 {
451 filename = gtk_file_chooser_get_filename(GTK_FILE_CHOOSER(dialog));
452 }
453
454 gtk_widget_destroy(dialog);
455 return filename;
456}
