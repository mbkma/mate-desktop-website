52gboolean
53procman_get_tree_state (GSettings *settings, GtkWidget *tree, const gchar *child_schema)
54{
55 GtkTreeModel *model;
56 GList *columns, *it;
57 gint sort_col;
58 GtkSortType order;
59
60 g_assert(tree);
61 g_assert(child_schema);
62
63 GSettings *pt_settings = g_settings_get_child (settings, child_schema);
64
65 model = gtk_tree_view_get_model (GTK_TREE_VIEW (tree));
66
67 sort_col = g_settings_get_int (pt_settings, "sort-col");
68
69 order = static_cast<GtkSortType>(g_settings_get_int (pt_settings, "sort-order"));
70
71 if (sort_col != -1)
72 gtk_tree_sortable_set_sort_column_id (GTK_TREE_SORTABLE (model),
73 sort_col,
74 order);
75
76 columns = gtk_tree_view_get_columns (GTK_TREE_VIEW (tree));
77
78 if (g_strcmp0(child_schema, "proctree") == 0 ||
79 g_strcmp0(child_schema, "disktreenew") == 0)
80 {
81 for (it = columns; it; it = it->next)
82 {
83 GtkTreeViewColumn *column;
84 gint width;
85 gboolean visible;
86 int id;
87 gchar *key;
88
89 column = static_cast<GtkTreeViewColumn*>(it->data);
90 id = gtk_tree_view_column_get_sort_column_id (column);
91
92 key = g_strdup_printf ("col-%d-width", id);
93 g_settings_get (pt_settings, key, "i", &width);
94 g_free (key);
95
96 key = g_strdup_printf ("col-%d-visible", id);
97 visible = g_settings_get_boolean (pt_settings, key);
98 g_free (key);
99
100 gtk_tree_view_column_set_visible (column, visible);
101 if (visible) {
102 /* ensure column is really visible */
103 width = MAX(width, 20);
104 gtk_tree_view_column_set_fixed_width(column, width);
105 }
106 }
107
108 GVariant *value;
109 GVariantIter iter;
110 int sortIndex;
111
112 GSList *order = NULL;
113
114 value = g_settings_get_value(pt_settings, "columns-order");
115 g_variant_iter_init(&iter, value);
116
117 while (g_variant_iter_loop (&iter, "i", &sortIndex))
118 order = g_slist_append(order, GINT_TO_POINTER(sortIndex));
119
120 proctable_set_columns_order(GTK_TREE_VIEW(tree), order);
121
122 g_variant_unref(value);
123 g_slist_free(order);
124 }
125
126 g_object_unref(pt_settings);
127 pt_settings = NULL;
128
129 g_list_free(columns);
130
131 return TRUE;
132}

134void
135procman_save_tree_state (GSettings *settings, GtkWidget *tree, const gchar *child_schema)
136{
137 GtkTreeModel *model;
138 GList *columns;
139 gint sort_col;
140 GtkSortType order;
141
142 if (ProcData::get_instance()->terminating)
143 return;
144
145 g_assert(tree);
146 g_assert(child_schema);
147
148 GSettings *pt_settings = g_settings_get_child (settings, child_schema);
149
150 model = gtk_tree_view_get_model (GTK_TREE_VIEW (tree));
151 if (gtk_tree_sortable_get_sort_column_id (GTK_TREE_SORTABLE (model), &sort_col,
152 &order)) {
153 g_settings_set_int (pt_settings, "sort-col", sort_col);
154 g_settings_set_int (pt_settings, "sort-order", order);
155 }
156
157 columns = gtk_tree_view_get_columns (GTK_TREE_VIEW (tree));
158
159 if (g_strcmp0(child_schema, "proctree") == 0 ||
160 g_strcmp0(child_schema, "disktreenew") == 0)
161 {
162 GSList *order;
163 GSList *order_node;
164 GVariantBuilder *builder;
165 GVariant *order_variant;
166
167 order = proctable_get_columns_order(GTK_TREE_VIEW(tree));
168
169 builder = g_variant_builder_new (G_VARIANT_TYPE ("ai"));
170
171 for (order_node = order; order_node; order_node = order_node->next)
172 g_variant_builder_add(builder, "i", GPOINTER_TO_INT(order_node->data));
173
174 order_variant = g_variant_new("ai", builder);
175 g_settings_set_value(pt_settings, "columns-order", order_variant);
176
177 g_slist_free(order);
178 }
179
180 g_list_free(columns);
181}

183void
184procman_save_config (ProcData *data)
185{
186 GSettings *settings = data->settings;
187
188 g_assert(data);
189
190 data->config.maximized = gdk_window_get_state(gtk_widget_get_window (data->app)) & GDK_WINDOW_STATE_MAXIMIZED;
191 if (!data->config.maximized) {
192 // we only want to store/overwrite size and position info with non-maximized state info
193 data->config.width = gdk_window_get_width (gtk_widget_get_window (data->app));
194 data->config.height = gdk_window_get_height(gtk_widget_get_window (data->app));
195
196 gtk_window_get_position(GTK_WINDOW(data->app), &data->config.xpos, &data->config.ypos);
197
198 g_settings_set (settings, "window-state", "(iiii)",
199 data->config.width, data->config.height,
200 data->config.xpos, data->config.ypos);
201 }
202 g_settings_set_boolean (settings, "maximized", data->config.maximized);
203
204 g_settings_set_int (settings, "current-tab", data->config.current_tab);
205}
