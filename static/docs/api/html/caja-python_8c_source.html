48caja_operationhandle_get_handle(PyGBoxed *self, void *closure)

50 return PyLong_FromSsize_t((Py_ssize_t) (size_t) self->boxed);

53static int
54caja_operationhandle_set_handle(PyGBoxed *self, PyObject *value, void *closure)
55{
56 Py_ssize_t val = PyLong_AsSsize_t(value);
57
58 if (!PyErr_Occurred()) {
59 if (val) {
60 self->boxed = (void *) val;
61 return 0;
62 }
63 PyErr_SetString(PyExc_ValueError, "invalid operation handle value");
64 }
65 return -1;
66}

89static void
90caja_python_load_file(GTypeModule *type_module,
91 const gchar *filename)
92{
93 PyObject *main_module, *main_locals, *locals, *key, *value;
94 PyObject *module;
95 GType gtype;
96 Py_ssize_t pos = 0;
97
98 debug_enter_args("filename=%s", filename);
99
100 main_module = PyImport_AddModule("__main__");
101 if (main_module == NULL)
102 {
103 g_warning("Could not get __main__.");
104 return;
105 }
106
107 main_locals = PyModule_GetDict(main_module);
108 module = PyImport_ImportModuleEx(filename, main_locals, main_locals, NULL);
109 if (!module)
110 {
111 PyErr_Print();
112 return;
113 }
114
115 locals = PyModule_GetDict(module);
116
117 while (PyDict_Next(locals, &pos, &key, &value))
118 {
119 if (!PyType_Check(value))
120 continue;
121
122 if (PyObject_IsSubclass(value, (PyObject*)&PyCajaColumnProvider_Type) ||
123 PyObject_IsSubclass(value, (PyObject*)&PyCajaInfoProvider_Type) ||
124 PyObject_IsSubclass(value, (PyObject*)&PyCajaLocationWidgetProvider_Type) ||
125 PyObject_IsSubclass(value, (PyObject*)&PyCajaMenuProvider_Type) ||
126 PyObject_IsSubclass(value, (PyObject*)&PyCajaPropertyPageProvider_Type))
127 {
128 gtype = caja_python_object_get_type(type_module, value);
129 g_array_append_val(all_types, gtype);
130
131 all_pyfiles = g_list_append(all_pyfiles, g_strdup(filename));
132 }
133 }
134
135 debug("Loaded python modules");
136}

138static void
139caja_python_load_dir (GTypeModule *module,
140 const char *dirname)
141{
142 GDir *dir;
143 const char *name;
144 gboolean initialized = FALSE;
145
146 debug_enter_args("dirname=%s", dirname);
147
148 dir = g_dir_open(dirname, 0, NULL);
149 if (!dir)
150 return;
151
152 while ((name = g_dir_read_name(dir)))
153 {
154 if (g_str_has_suffix(name, ".py"))
155 {
156 char *modulename;
157 size_t len;
158
159 len = strlen(name) - 3;
160 modulename = g_new0(char, len + 1 );
161 strncpy(modulename, name, len);
162
163 if (!initialized)
164 {
165 PyObject *sys_path, *py_path;
166
167 /* n-p python part is initialized on demand (or not
168 * at all if no extensions are found) */
169 if (!caja_python_init_python())
170 {
171 g_warning("caja_python_init_python failed");
172 g_dir_close(dir);
173 g_free(modulename);
174 break;
175 }
176
177 /* sys.path.insert(0, dirname) */
178 sys_path = PySys_GetObject("path");
179 py_path = PyUnicode_FromString(dirname);
180 PyList_Insert(sys_path, 0, py_path);
181 Py_DECREF(py_path);
182 }
183 caja_python_load_file(module, modulename);
184 g_free(modulename);
185 }
186 }

189static gboolean
190caja_python_init_python (void)
191{
192 PyObject *gi, *require_version, *args, *caja, *descr;
193 GModule *libpython;
194
195 if (Py_IsInitialized())
196 return TRUE;
197
198 debug("g_module_open " PY_LIB_LOC "/libpython" PYTHON_VERSION PYTHON_ABIFLAGS "." G_MODULE_SUFFIX ".1.0");
199 libpython = g_module_open(PY_LIB_LOC "/libpython" PYTHON_VERSION PYTHON_ABIFLAGS "." G_MODULE_SUFFIX ".1.0", 0);
200 if (!libpython)
201 g_warning("g_module_open libpython failed: %s", g_module_error());
202
203 debug("Py_Initialize");
204 Py_Initialize();
205 if (PyErr_Occurred())
206 {
207 PyErr_Print();
208 return FALSE;
209 }
210
211 debug("Sanitize the python search path and set sys.argv");
212 PyRun_SimpleString("import sys; "
213 "sys.path = list(filter(None, sys.path)); "
214 "sys.argv = ['caja']");
215 if (PyErr_Occurred())
216 {
217 PyErr_Print();
218 return FALSE;
219 }
220
221 /* import gobject */
222 debug("init_pygobject");
223 if (!np_init_pygobject())
224 {
225 g_warning("pygobject initialization failed");
226 return FALSE;
227 }
228
229 /* import caja */
230 g_setenv("INSIDE_CAJA_PYTHON", "", FALSE);
231 debug("import caja");
232 gi = PyImport_ImportModule ("gi");
233 if (!gi) {
234 PyErr_Print();
235 return FALSE;
236 }
237
238 require_version = PyObject_GetAttrString (gi, (char *) "require_version");
239 args = PyTuple_Pack (2, PyUnicode_FromString ("Caja"),
240 PyUnicode_FromString ("2.0"));
241 PyObject_CallObject (require_version, args);
242 Py_DECREF (require_version);
243 Py_DECREF (args);
244 Py_DECREF (gi);
245 caja = PyImport_ImportModule("gi.repository.Caja");
246 if (!caja)
247 {
248 PyErr_Print();
249 return FALSE;
250 }
251
252 _PyGtkWidget_Type = pygobject_lookup_class(GTK_TYPE_WIDGET);
253 g_assert(_PyGtkWidget_Type != NULL);
254
255#define IMPORT(x, y) \
256 _PyCaja##x##_Type = (PyTypeObject *)PyObject_GetAttrString(caja, y); \
257 if (_PyCaja##x##_Type == NULL) { \
258 PyErr_Print(); \
259 return FALSE; \
260 }
261
262 IMPORT(Column, "Column");
263 IMPORT(ColumnProvider, "ColumnProvider");
264 IMPORT(InfoProvider, "InfoProvider");
265 IMPORT(LocationWidgetProvider, "LocationWidgetProvider");
266 IMPORT(Menu, "Menu");
267 IMPORT(MenuItem, "MenuItem");
268 IMPORT(MenuProvider, "MenuProvider");
269 IMPORT(PropertyPage, "PropertyPage");
270 IMPORT(PropertyPageProvider, "PropertyPageProvider");
271 IMPORT(OperationHandle, "OperationHandle");
272
273#undef IMPORT
274
275 /* Add the "handle" member to the OperationHandle type. */
276 descr = PyDescr_NewGetSet(_PyCajaOperationHandle_Type,
277 &caja_operationhandle_handle);
278 if (!descr) {
279 PyErr_Print();
280 return FALSE;
281 }
282 if (PyDict_SetItemString(_PyCajaOperationHandle_Type->tp_dict,
283 caja_operationhandle_handle.name, descr)) {
284 Py_DECREF(descr);
285 PyErr_Print();
286 return FALSE;
287 }
288 Py_DECREF(descr);
289
290 return TRUE;
291}

293void
294caja_module_initialize(GTypeModule *module)
295{
296 gchar *user_extensions_dir;
297 const gchar *env_string;
298
299 env_string = g_getenv("CAJA_PYTHON_DEBUG");
300 if (env_string != NULL)
301 {
302 caja_python_debug = g_parse_debug_string(env_string,
303 caja_python_debug_keys,
304 caja_python_ndebug_keys);
305 env_string = NULL;
306 }
307
308 debug_enter();
309
310 all_types = g_array_new(FALSE, FALSE, sizeof(GType));
311
312 // Look in the new global path, $DATADIR/caja-python/extensions
313 caja_python_load_dir(module, DATADIR "/caja-python/extensions");
314
315 // Look in XDG_DATA_DIR, ~/.local/share/caja-python/extensions
316 user_extensions_dir = g_build_filename(g_get_user_data_dir(),
317 "caja-python", "extensions", NULL);
318 caja_python_load_dir(module, user_extensions_dir);
319}
