32/* Returns a new GdkPixbuf that is suitable for placing in the thumbnail view.
33 * It is four pixels wider and taller than the source. If source_pixbuf is not
34 * NULL, then it will fill the return pixbuf with the contents of
35 * source_pixbuf.
36 */
37static GdkPixbuf *
38create_thumbnail_frame (int width,
39 int height,
40 GdkPixbuf *source_pixbuf,
41 gboolean fill_bg)
42{
43 GdkPixbuf *retval;
44 guchar *data;
45 gint rowstride;
46 int i;
47 int width_r, height_r;
48
49 if (source_pixbuf)
50 g_return_val_if_fail (GDK_IS_PIXBUF (source_pixbuf), NULL);
51
52 if (source_pixbuf) {
53 width_r = gdk_pixbuf_get_width (source_pixbuf);
54 height_r = gdk_pixbuf_get_height (source_pixbuf);
55 } else {
56 width_r = width;
57 height_r = height;
58 }
59
60 /* make sure no one is passing us garbage */
61 g_return_val_if_fail (width_r >= 0 && height_r >= 0, NULL);
62
63 retval = gdk_pixbuf_new (GDK_COLORSPACE_RGB,
64 TRUE, 8,
65 width_r + 4,
66 height_r + 4);
67
68 /* make it black and fill in the middle */
69 data = gdk_pixbuf_get_pixels (retval);
70 rowstride = gdk_pixbuf_get_rowstride (retval);
71
72 gdk_pixbuf_fill (retval, 0x000000ff);
73 if (fill_bg) {
74 for (i = 1; i < height_r + 1; i++)
75 memset (data + (rowstride * i) + 4, 0xffffffff, width_r * 4);
76 }
77
78 /* copy the source pixbuf */
79 if (source_pixbuf)
80 gdk_pixbuf_copy_area (source_pixbuf, 0, 0,
81 width_r,
82 height_r,
83 retval,
84 1, 1);
85 /* Add the corner */
86 data [(width_r + 2) * 4 + 3] = 0;
87 data [(width_r + 3) * 4 + 3] = 0;
88 data [(width_r + 2) * 4 + (rowstride * 1) + 3] = 0;
89 data [(width_r + 3) * 4 + (rowstride * 1) + 3] = 0;
90
91 data [(height_r + 2) * rowstride + 3] = 0;
92 data [(height_r + 3) * rowstride + 3] = 0;
93 data [(height_r + 2) * rowstride + 4 + 3] = 0;
94 data [(height_r + 3) * rowstride + 4 + 3] = 0;
95
96 return retval;
97}

152void
153ev_document_misc_paint_one_page (cairo_t *cr,
154 GtkWidget *widget,
155 GdkRectangle *area,
156 GtkBorder *border,
157 gboolean highlight,
158 gboolean inverted_colors)
159{
160 GtkStyleContext *context = gtk_widget_get_style_context (widget);
161 GtkStateFlags state = gtk_widget_get_state_flags (widget);
162 GdkRGBA fg, bg, shade_bg;
163
164 gtk_style_context_save (context);
165 gtk_style_context_get_background_color (context, state, &bg);
166 gtk_style_context_get_color (context, state, &fg);
167 gtk_style_context_get_color (context, state, &shade_bg);
168 gtk_style_context_restore (context);
169 shade_bg.alpha *= 0.5;
170
171 gdk_cairo_set_source_rgba (cr, highlight ? &fg : &shade_bg);
172 cairo_rectangle (cr,
173 area->x,
174 area->y,
175 area->width - border->right + border->left,
176 area->height - border->bottom + border->top);
177 cairo_rectangle (cr,
178 area->x + area->width - border->right,
179 area->y + border->right - border->left,
180 border->right,
181 area->height - border->right + border->left);
182 cairo_rectangle (cr,
183 area->x + border->bottom - border->top,
184 area->y + area->height - border->bottom,
185 area->width - border->bottom + border->top,
186 border->bottom);
187 cairo_fill (cr);
188
189 if (inverted_colors)
190 cairo_set_source_rgb (cr, 0, 0, 0);
191 else
192 cairo_set_source_rgb (cr, 1, 1, 1);
193 cairo_rectangle (cr,
194 area->x + border->left,
195 area->y + border->top,
196 area->width - (border->left + border->right),
197 area->height - (border->top + border->bottom));
198 cairo_fill (cr);
199}

203{

206
207 g_return_val_if_fail (GDK_IS_PIXBUF (pixbuf), NULL);
208
209 surface = cairo_image_surface_create (gdk_pixbuf_get_has_alpha (pixbuf) ?
210 CAIRO_FORMAT_ARGB32 : CAIRO_FORMAT_RGB24,

238cairo_surface_t *
239ev_document_misc_surface_rotate_and_scale (cairo_surface_t *surface,
240 gint dest_width,
241 gint dest_height,
242 gint dest_rotation)
243{
244 cairo_surface_t *new_surface;
245 cairo_t *cr;
246 gint width, height;
247 gint new_width = dest_width;
248 gint new_height = dest_height;
249
250 width = cairo_image_surface_get_width (surface);
251 height = cairo_image_surface_get_height (surface);
252
253 if (dest_width == width &&
254 dest_height == height &&
255 dest_rotation == 0) {
256 return cairo_surface_reference (surface);
257 }
258
259 if (dest_rotation == 90 || dest_rotation == 270) {
260 new_width = dest_height;
261 new_height = dest_width;
262 }
263
264 new_surface = cairo_surface_create_similar (surface,
265 cairo_surface_get_content (surface),
266 new_width, new_height);
267
268 cr = cairo_create (new_surface);
269 switch (dest_rotation) {
270 case 90:
271 cairo_translate (cr, new_width, 0);
272 break;
273 case 180:
274 cairo_translate (cr, new_width, new_height);
275 break;
276 case 270:
277 cairo_translate (cr, 0, new_height);
278 break;
279 default:
280 cairo_translate (cr, 0, 0);
281 }
282 cairo_rotate (cr, dest_rotation * G_PI / 180.0);
283
284 if (dest_width != width || dest_height != height) {
285 cairo_pattern_set_filter (cairo_get_source (cr), CAIRO_FILTER_BILINEAR);
286 cairo_scale (cr,
287 (gdouble)dest_width / width,
288 (gdouble)dest_height / height);
289 }
290
291 cairo_set_source_surface (cr, surface, 0, 0);
292 cairo_paint (cr);
293 cairo_destroy (cr);
294
295 return new_surface;
296}

311void
312ev_document_misc_invert_pixbuf (GdkPixbuf *pixbuf)
313{
314 guchar *data, *p;
315 guint width, height, x, y, rowstride, n_channels;
316
317 n_channels = gdk_pixbuf_get_n_channels (pixbuf);
318 g_assert (gdk_pixbuf_get_colorspace (pixbuf) == GDK_COLORSPACE_RGB);
319 g_assert (gdk_pixbuf_get_bits_per_sample (pixbuf) == 8);
320
321 /* First grab a pointer to the raw pixel data. */
322 data = gdk_pixbuf_get_pixels (pixbuf);
323
324 /* Find the number of bytes per row (could be padded). */
325 rowstride = gdk_pixbuf_get_rowstride (pixbuf);
326
327 width = gdk_pixbuf_get_width (pixbuf);
328 height = gdk_pixbuf_get_height (pixbuf);
329 for (x = 0; x < width; x++) {
330 for (y = 0; y < height; y++) {
331 /* Calculate pixel's offset into the data array. */
332 p = data + x * n_channels + y * rowstride;
333 /* Change the RGB values*/
334 p[0] = 255 - p[0];
335 p[1] = 255 - p[1];
336 p[2] = 255 - p[2];
337 }
338 }
339}

341gdouble
342ev_document_misc_get_monitor_dpi (GdkMonitor *monitor)
343{
344 GdkRectangle geometry;
345 int s;
346 gdouble dp, di;
347
348 gdk_monitor_get_geometry (monitor, &geometry);
349 s = gdk_monitor_get_scale_factor (monitor);
350
351 /*diagonal in pixels*/
352 dp = hypot (geometry.width * s, geometry.height * s);
353
354 /*diagonal in inches*/
355 di = hypot (gdk_monitor_get_width_mm(monitor), gdk_monitor_get_height_mm (monitor)) / 25.4;
356
357 return (dp / di);
358}

375void
376ev_document_misc_get_pointer_position (GtkWidget *widget,
377 gint *x,
378 gint *y)
379{
380 GdkSeat *seat;
381 GdkDevice *device_pointer;
382 GdkRectangle allocation;
383
384 if (x)
385 *x = -1;
386 if (y)
387 *y = -1;
388
389 if (!gtk_widget_get_realized (widget))
390 return;
391
392 seat = gdk_display_get_default_seat (gtk_widget_get_display (widget));
393 device_pointer = gdk_seat_get_pointer (seat);
394 gdk_window_get_device_position (gtk_widget_get_window (widget),
395 device_pointer,
396 x, y, NULL);
397
398 if (gtk_widget_get_has_window (widget))
399 return;
400
401 gtk_widget_get_allocation (widget, &allocation);
402 if (x)
403 *x -= allocation.x;
404 if (y)
405 *y -= allocation.y;
406}
