134static void
135ev_link_action_get_property (GObject *object,
136 guint prop_id,
137 GValue *value,
138 GParamSpec *param_spec)
139{
140 EvLinkAction *self;
141
142 self = EV_LINK_ACTION (object);
143
144 switch (prop_id) {
145 case PROP_TYPE:
146 g_value_set_enum (value, self->priv->type);
147 break;
148 case PROP_DEST:
149 g_value_set_pointer (value, self->priv->dest);
150 break;
151 case PROP_URI:
152 g_value_set_string (value, self->priv->uri);
153 break;
154 case PROP_FILENAME:
155 g_value_set_string (value, self->priv->filename);
156 break;
157 case PROP_PARAMS:
158 g_value_set_string (value, self->priv->params);
159 break;
160 case PROP_NAME:
161 g_value_set_string (value, self->priv->name);
162 break;
163 case PROP_SHOW_LIST:
164 g_value_set_pointer (value, self->priv->show_list);
165 break;
166 case PROP_HIDE_LIST:
167 g_value_set_pointer (value, self->priv->hide_list);
168 break;
169 case PROP_TOGGLE_LIST:
170 g_value_set_pointer (value, self->priv->toggle_list);
171 break;
172 default:
173 G_OBJECT_WARN_INVALID_PROPERTY_ID (object,
174 prop_id,
175 param_spec);
176 break;
177 }
178}

180static void
181ev_link_action_set_property (GObject *object,
182 guint prop_id,
183 const GValue *value,
184 GParamSpec *param_spec)
185{
186 EvLinkAction *self = EV_LINK_ACTION (object);
187
188 switch (prop_id) {
189 case PROP_TYPE:
190 self->priv->type = g_value_get_enum (value);
191 break;
192 case PROP_DEST:
193 self->priv->dest = g_value_get_pointer (value);
194 break;
195 case PROP_URI:
196 g_free (self->priv->uri);
197 self->priv->uri = g_value_dup_string (value);
198 break;
199 case PROP_FILENAME:
200 g_free (self->priv->filename);
201 self->priv->filename = g_value_dup_string (value);
202 break;
203 case PROP_PARAMS:
204 g_free (self->priv->params);
205 self->priv->params = g_value_dup_string (value);
206 break;
207 case PROP_NAME:
208 g_free (self->priv->name);
209 self->priv->name = g_value_dup_string (value);
210 break;
211 case PROP_SHOW_LIST:
212 self->priv->show_list = g_value_get_pointer (value);
213 break;
214 case PROP_HIDE_LIST:
215 self->priv->hide_list = g_value_get_pointer (value);
216 break;
217 case PROP_TOGGLE_LIST:
218 self->priv->toggle_list = g_value_get_pointer (value);
219 break;
220 default:
221 G_OBJECT_WARN_INVALID_PROPERTY_ID (object,
222 prop_id,
223 param_spec);
224 break;
225 }
226}

228static void
229ev_link_action_finalize (GObject *object)
230{
231 EvLinkActionPrivate *priv;
232
233 priv = EV_LINK_ACTION (object)->priv;
234
235 if (priv->dest) {
236 g_object_unref (priv->dest);
237 priv->dest = NULL;
238 }
239
240 if (priv->uri) {
241 g_free (priv->uri);
242 priv->uri = NULL;
243 }
244
245 if (priv->filename) {
246 g_free (priv->filename);
247 priv->filename = NULL;
248 }
249
250 if (priv->params) {
251 g_free (priv->params);
252 priv->params = NULL;
253 }
254
255 if (priv->name) {
256 g_free (priv->name);
257 priv->name = NULL;
258 }
259
260 if (priv->show_list) {
261 g_list_foreach (priv->show_list, (GFunc)g_object_unref, NULL);
262 g_list_free (priv->show_list);
263 priv->show_list = NULL;
264 }
265
266 if (priv->hide_list) {
267 g_list_foreach (priv->hide_list, (GFunc)g_object_unref, NULL);
268 g_list_free (priv->hide_list);
269 priv->hide_list = NULL;
270 }
271
272 if (priv->toggle_list) {
273 g_list_foreach (priv->toggle_list, (GFunc)g_object_unref, NULL);
274 g_list_free (priv->toggle_list);
275 priv->toggle_list = NULL;
276 }
277
278 G_OBJECT_CLASS (ev_link_action_parent_class)->finalize (object);
279}
