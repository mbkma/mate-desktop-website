38/* these are the mods whose combinations are ignored by the keygrabbing code */
39static GdkModifierType msd_ignored_mods = 0;
40
41/* these are the ones we actually use for global keys, we always only check
42 * for these set */
43static GdkModifierType msd_used_mods = 0;
44
45static void
46setup_modifiers (void)
47{
48 if (msd_used_mods == 0 || msd_ignored_mods == 0) {
49 GdkModifierType dynmods;
50
51 /* default modifiers */
52 msd_ignored_mods = \
53 0x2000 /*Xkb modifier*/ | GDK_LOCK_MASK | GDK_HYPER_MASK;
54 msd_used_mods = \
55 GDK_SHIFT_MASK | GDK_CONTROL_MASK |\
56 GDK_MOD1_MASK | GDK_MOD2_MASK | GDK_MOD3_MASK | GDK_MOD4_MASK |\
57 GDK_MOD5_MASK | GDK_SUPER_MASK | GDK_META_MASK;
58
59 /* NumLock can be assigned to varying keys so we need to
60 * resolve and ignore it specially */
61 dynmods = 0;
62 egg_keymap_resolve_virtual_modifiers (gdk_keymap_get_for_display (gdk_display_get_default ()),
63 EGG_VIRTUAL_NUM_LOCK_MASK,
64 &dynmods);
65
66 msd_ignored_mods |= dynmods;
67 msd_used_mods &= ~dynmods;
68 }
69}

93/* Grab the key. In order to ignore MSD_IGNORED_MODS we need to grab
94 * all combinations of the ignored modifiers and those actually used
95 * for the binding (if any).
96 *
97 * inspired by all_combinations from mate-panel/mate-panel/global-keys.c
98 *
99 * This may generate X errors. The correct way to use this is like:
100 *
101 * gdk_error_trap_push ();
102 *
103 * grab_key_unsafe (key, grab, screens);
104 *
105 * gdk_flush ();
106 * if (gdk_error_trap_pop ())
107 * g_warning ("Grab failed, another application may already have access to key '%u'",
108 * key->keycode);
109 *
110 * This is not done in the function itself, to allow doing multiple grab_key
111 * operations with one flush only.
112 */
113#define N_BITS 32
114void
115grab_key_unsafe (Key *key,
116 gboolean grab,
117 GSList *screens)
118{
119 int indexes[N_BITS]; /* indexes of bits we need to flip */
120 int i;
121 int bit;
122 int bits_set_cnt;
123 int uppervalue;
124 guint mask;
125
126 setup_modifiers ();
127
128 mask = msd_ignored_mods & ~key->state & GDK_MODIFIER_MASK;
129
130 bit = 0;
131 /* store the indexes of all set bits in mask in the array */
132 for (i = 0; mask; ++i, mask >>= 1) {
133 if (mask & 0x1) {
134 indexes[bit++] = i;
135 }
136 }
137
138 bits_set_cnt = bit;
139
140 uppervalue = 1 << bits_set_cnt;
141 /* grab all possible modifier combinations for our mask */
142 for (i = 0; i < uppervalue; ++i) {
143 GSList *l;
144 int j;
145 int result = 0;
146
147 /* map bits in the counter to those in the mask */
148 for (j = 0; j < bits_set_cnt; ++j) {
149 if (i & (1 << j)) {
150 result |= (1 << indexes[j]);
151 }
152 }
153
154 for (l = screens; l; l = l->next) {
155 GdkScreen *screen = l->data;
156 guint *code;
157
158 for (code = key->keycodes; *code; ++code) {
159 grab_key_real (*code,
160 gdk_screen_get_root_window (screen),
161 grab,
162 result | key->state);
163 }
164 }
165 }
166}

168static gboolean
169have_xkb (Display *dpy)
170{
171 static int have_xkb = -1;
172
173 if (have_xkb == -1) {
174#ifdef HAVE_X11_EXTENSIONS_XKB_H
175 int opcode, error_base, major, minor, xkb_event_base;
176
177 have_xkb = XkbQueryExtension (dpy,
178 &opcode,
179 &xkb_event_base,
180 &error_base,
181 &major,
182 &minor)
183 && XkbUseExtension (dpy, &major, &minor);
184#else
185 have_xkb = 0;
186#endif
187 }
188
189 return have_xkb;
190}

206gboolean
207match_key (Key *key, XEvent *event)
208{
209 guint keyval;
210 GdkModifierType consumed;
211 gint group;
212
213 if (key == NULL)
214 return FALSE;
215
216 setup_modifiers ();
217
218#ifdef HAVE_X11_EXTENSIONS_XKB_H
219 if (have_xkb (event->xkey.display))
220 group = XkbGroupForCoreState (event->xkey.state);
221 else
222#endif
223 group = (event->xkey.state & GDK_KEY_Mode_switch) ? 1 : 0;
224
225 /* Check if we find a keysym that matches our current state */
226 if (gdk_keymap_translate_keyboard_state (gdk_keymap_get_for_display (gdk_display_get_default ()), event->xkey.keycode,
227 event->xkey.state, group,
228 &keyval, NULL, NULL, &consumed)) {
229 guint lower, upper;
230
231 gdk_keyval_convert_case (keyval, &lower, &upper);
232
233 /* If we are checking against the lower version of the
234 * keysym, we might need the Shift state for matching,
235 * so remove it from the consumed modifiers */
236 if (lower == key->keysym)
237 consumed &= ~GDK_SHIFT_MASK;
238
239 return ((lower == key->keysym || upper == key->keysym)
240 && (event->xkey.state & ~consumed & msd_used_mods) == key->state);
241 }
242
243 /* The key we passed doesn't have a keysym, so try with just the keycode */
244 return (key->state == (event->xkey.state & msd_used_mods)
245 && key_uses_keycode (key, event->xkey.keycode));
246}
