93static gboolean
94gpm_control_systemd_shutdown (void) {
95 GError *error = NULL;
96 GDBusProxy *proxy;
97 GVariant *res = NULL;
98
99 g_debug ("Requesting systemd to shutdown");
100 proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SYSTEM,
101 G_DBUS_PROXY_FLAGS_DO_NOT_LOAD_PROPERTIES,
102 NULL,
103 "org.freedesktop.login1",
104 "/org/freedesktop/login1",
105 "org.freedesktop.login1.Manager",
106 NULL,
107 &error );
108 //append all our arguments
109 if (proxy == NULL) {
110 g_warning ("Error connecting to dbus - %s", error->message);
111 g_error_free (error);
112 return FALSE;
113 }
114
115 res = g_dbus_proxy_call_sync (proxy, "PowerOff",
116 g_variant_new( "(b)", FALSE),
117 G_DBUS_CALL_FLAGS_NONE,
118 -1,
119 NULL,
120 &error
121 );
122 if (error != NULL) {
123 g_warning ("Error in dbus - %s", error->message);
124 g_error_free (error);
125 return FALSE;
126 }
127
128 g_variant_unref(res);
129 return TRUE;
130}

165gboolean
166gpm_control_get_lock_policy (GpmControl *control, const gchar *policy)
167{
168 gboolean do_lock;
169 gboolean use_ss_setting;
170 gchar **schemas = NULL;
171 gboolean schema_exists;
172 gint i;
173
174 /* Check if the mate-screensaver schema exists before trying to read
175 the lock setting to prevent crashing. See GNOME bug #651225. */
176 g_settings_schema_source_list_schemas (g_settings_schema_source_get_default (), TRUE, &schemas, NULL);
177 schema_exists = FALSE;
178 for (i = 0; schemas[i] != NULL; i++) {
179 if (g_strcmp0 (schemas[i], GS_SETTINGS_SCHEMA) == 0) {
180 schema_exists = TRUE;
181 break;
182 }
183 }
184
185 g_strfreev (schemas);
186
187 /* This allows us to over-ride the custom lock settings set
188 with a system default set in mate-screensaver.
189 See bug #331164 for all the juicy details. :-) */
190 use_ss_setting = g_settings_get_boolean (control->priv->settings, GPM_SETTINGS_LOCK_USE_SCREENSAVER);
191 if (use_ss_setting && schema_exists) {
192 GSettings *settings_ss;
193 settings_ss = g_settings_new (GS_SETTINGS_SCHEMA);
194 do_lock = g_settings_get_boolean (settings_ss, GS_SETTINGS_PREF_LOCK_ENABLED);
195 g_debug ("Using ScreenSaver settings (%i)", do_lock);
196 g_object_unref (settings_ss);
197 } else {
198 do_lock = g_settings_get_boolean (control->priv->settings, policy);
199 g_debug ("Using custom locking settings (%i)", do_lock);
200 }
201 return do_lock;
202}

207gboolean
208gpm_control_suspend (GpmControl *control, GError **error)
209{
210 gboolean allowed = FALSE;
211 gboolean ret = FALSE;
212 gboolean do_lock;
213 gboolean nm_sleep;
214 EggConsoleKit *console;
215 GpmScreensaver *screensaver;
216 guint32 throttle_cookie = 0;
217#ifdef WITH_LIBSECRET
218 gboolean lock_libsecret;
219 GCancellable *libsecret_cancellable = NULL;
220 SecretService *secretservice_proxy = NULL;
221 gint num_secrets_locked;
222 GList *libsecret_collections = NULL;
223#endif /* WITH_LIBSECRET */
224#ifdef WITH_KEYRING
225 gboolean lock_gnome_keyring;
226 GnomeKeyringResult keyres;
227#endif /* WITH_KEYRING */
228
229 GError *dbus_error = NULL;
230 GDBusProxy *proxy;
231 GVariant *res = NULL;
232
233 screensaver = gpm_screensaver_new ();
234
235 if (!LOGIND_RUNNING()) {
236 console = egg_console_kit_new ();
237 egg_console_kit_can_suspend (console, &allowed, NULL);
238 g_object_unref (console);
239
240 if (!allowed) {
241 g_debug ("cannot suspend as not allowed from policy");
242 g_set_error_literal (error, GPM_CONTROL_ERROR, GPM_CONTROL_ERROR_GENERAL, "Cannot suspend");
243 goto out;
244 }
245 }
246
247#ifdef WITH_LIBSECRET
248 /* we should perhaps lock keyrings when sleeping #375681 */
249 lock_libsecret = g_settings_get_boolean (control->priv->settings,
250 GPM_SETTINGS_LOCK_KEYRING_SUSPEND);
251 if (lock_libsecret) {
252 libsecret_cancellable = g_cancellable_new ();
253 secretservice_proxy = secret_service_get_sync (SECRET_SERVICE_LOAD_COLLECTIONS,
254 libsecret_cancellable,
255 error);
256 if (secretservice_proxy == NULL) {
257 g_warning ("failed to connect to secret service");
258 } else {
259 libsecret_collections = secret_service_get_collections (secretservice_proxy);
260 if ( libsecret_collections == NULL) {
261 g_warning ("failed to get secret collections");
262 } else {
263 num_secrets_locked = secret_service_lock_sync (secretservice_proxy,
264 libsecret_collections,
265 libsecret_cancellable,
266 NULL,
267 error);
268 if (num_secrets_locked <= 0)
269 g_warning ("could not lock keyring");
270 g_list_free (libsecret_collections);
271 }
272 g_object_unref (secretservice_proxy);
273 }
274 g_object_unref (libsecret_cancellable);
275 }
276#endif /* WITH_LIBSECRET */
277#ifdef WITH_KEYRING
278 /* we should perhaps lock keyrings when sleeping #375681 */
279 lock_gnome_keyring = g_settings_get_boolean (control->priv->settings, GPM_SETTINGS_LOCK_KEYRING_SUSPEND);
280 if (lock_gnome_keyring) {
281 keyres = gnome_keyring_lock_all_sync ();
282 if (keyres != GNOME_KEYRING_RESULT_OK)
283 g_warning ("could not lock keyring");
284 }
285#endif /* WITH_KEYRING */
286
287 do_lock = gpm_control_get_lock_policy (control, GPM_SETTINGS_LOCK_ON_SUSPEND);
288 if (do_lock) {
289 throttle_cookie = gpm_screensaver_add_throttle (screensaver, "suspend");
290 gpm_screensaver_lock (screensaver);
291 }
292
293 nm_sleep = g_settings_get_boolean (control->priv->settings, GPM_SETTINGS_NETWORKMANAGER_SLEEP);
294 if (nm_sleep)
295 gpm_networkmanager_sleep ();
296
297 /* Do the suspend */
298 g_debug ("emitting sleep");
299 g_signal_emit (control, signals [SLEEP], 0, GPM_CONTROL_ACTION_SUSPEND);
300
301 if (LOGIND_RUNNING()) {
302 /* sleep via logind */
303 proxy = g_dbus_proxy_new_for_bus_sync (G_BUS_TYPE_SYSTEM,
304 G_DBUS_PROXY_FLAGS_DO_NOT_LOAD_PROPERTIES,
305 NULL,
306 "org.freedesktop.login1",
307 "/org/freedesktop/login1",
308 "org.freedesktop.login1.Manager",
309 NULL,
310 &dbus_error );
311 if (proxy == NULL) {
312 g_warning ("Error connecting to dbus - %s", dbus_error->message);
313 g_error_free (dbus_error);
314 ret = FALSE;
315 goto out;
316 }
317 res = g_dbus_proxy_call_sync (proxy, "Suspend",
318 g_variant_new( "(b)",FALSE),
319 G_DBUS_CALL_FLAGS_NONE,
320 -1,
321 NULL,
322 &dbus_error
323 );
324 if (dbus_error != NULL ) {
325 g_warning ("Error in dbus - %s", dbus_error->message);
326 g_error_free (dbus_error);
327 ret = TRUE;
328 }
329 else {
330 g_variant_unref(res);
331 ret = TRUE;
332 }
333 g_object_unref(proxy);
334 }
335 else {
336 console = egg_console_kit_new ();
337 ret = egg_console_kit_suspend (console, error);
338 g_object_unref (console);
339 }
340
341 g_debug ("emitting resume");
342 g_signal_emit (control, signals [RESUME], 0, GPM_CONTROL_ACTION_SUSPEND);
343
344 if (do_lock) {
345 gpm_screensaver_poke (screensaver);
346 if (throttle_cookie)
347 gpm_screensaver_remove_throttle (screensaver, throttle_cookie);
348 }
349
350 nm_sleep = g_settings_get_boolean (control->priv->settings, GPM_SETTINGS_NETWORKMANAGER_SLEEP);
351 if (nm_sleep)
352 gpm_networkmanager_wake ();
353
354out:
355 g_object_unref (screensaver);
356 return ret;
357}
