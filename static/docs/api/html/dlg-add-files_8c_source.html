51static int
52file_sel_response_cb (GtkWidget *widget,
53 int response,
54 DialogData *data)
55{
56 GtkFileChooser *file_sel = GTK_FILE_CHOOSER (widget);
57 FrWindow *window = data->window;
58 char *current_folder;
59 char *uri;
60 gboolean update;
61 GSList *selections, *iter;
62 GList *item_list = NULL;
63
64 current_folder = gtk_file_chooser_get_current_folder_uri (file_sel);
65 uri = gtk_file_chooser_get_uri (file_sel);
66
67 if (current_folder != NULL) {
68 g_settings_set_string (data->settings, PREF_ADD_CURRENT_FOLDER, current_folder);
69 fr_window_set_add_default_dir (window, current_folder);
70 }
71
72 if (uri != NULL)
73 {
74 g_settings_set_string (data->settings, PREF_ADD_FILENAME, uri);
75 g_free (uri);
76 }
77
78 if ((response == GTK_RESPONSE_CANCEL) || (response == GTK_RESPONSE_DELETE_EVENT)) {
79 gtk_widget_destroy (data->dialog);
80 g_free (current_folder);
81 return TRUE;
82 }
83
84 if (response == GTK_RESPONSE_HELP) {
85 show_help_dialog (GTK_WINDOW (data->dialog), "engrampa-add-options");
86 g_free (current_folder);
87 return TRUE;
88 }
89
90 /* check folder permissions. */
91
92 if (uri_is_dir (current_folder) && ! check_permissions (current_folder, R_OK)) {
93 GtkWidget *d;
94 char *utf8_path;
95
96 utf8_path = g_filename_display_name (current_folder);
97
98 d = _gtk_error_dialog_new (GTK_WINDOW (window),
99 GTK_DIALOG_MODAL,
100 NULL,
101 _("Could not add the files to the archive"),
102 _("You don't have the right permissions to read files from folder \"%s\""),
103 utf8_path);
104 gtk_dialog_run (GTK_DIALOG (d));
105 gtk_widget_destroy (GTK_WIDGET (d));
106
107 g_free (utf8_path);
108 g_free (current_folder);
109 return FALSE;
110 }
111
112 update = gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (data->add_if_newer_checkbutton));
113
114 /**/
115
116 selections = gtk_file_chooser_get_uris (file_sel);
117 for (iter = selections; iter != NULL; iter = iter->next)
118 item_list = g_list_prepend (item_list, g_file_new_for_uri (iter->data));
119
120 if (item_list != NULL)
121 fr_window_archive_add_files (window, item_list, update);
122
123 gio_file_list_free (item_list);
124 g_slist_free_full (selections, g_free);
125 g_free (current_folder);
126
127 gtk_widget_destroy (data->dialog);
128
129 return TRUE;
130}

132/* create the "add" dialog. */
133void
134add_files_cb (GtkWidget *widget,
135 void *callback_data)
136{
137 GtkBuilder *builder;
138 DialogData *data;
139 char *folder;
140
141 builder = gtk_builder_new_from_resource (ENGRAMPA_RESOURCE_UI_PATH G_DIR_SEPARATOR_S "dlg-add-files.ui");
142
143 data = g_new0 (DialogData, 1);
144 data->window = callback_data;
145 data->settings = g_settings_new (ENGRAMPA_SCHEMA_ADD);
146 data->dialog = GET_WIDGET ("dialog_add_files");
147 data->add_if_newer_checkbutton = GET_WIDGET ("add_if_newer_checkbutton");
148
149 /* set data */
150 folder = g_settings_get_string (data->settings, PREF_ADD_CURRENT_FOLDER);
151 if ((folder == NULL) || (strcmp (folder, "") == 0))
152 folder = g_strdup (fr_window_get_add_default_dir (data->window));
153 gtk_file_chooser_set_current_folder_uri (GTK_FILE_CHOOSER (GTK_WIDGET (data->dialog)), folder);
154 g_free (folder);
155
156 /* signals */
157 gtk_builder_add_callback_symbols (builder,
158 "on_dlg_add_files_destroy", G_CALLBACK (open_file_destroy_cb),
159 "on_dlg_add_files_response", G_CALLBACK (file_sel_response_cb),
160 NULL);
161 gtk_builder_connect_signals (builder, data);
162
163 g_object_unref (builder);
164
165 gtk_window_set_modal (GTK_WINDOW (data->dialog), TRUE);
166 gtk_widget_show (data->dialog);
167}
