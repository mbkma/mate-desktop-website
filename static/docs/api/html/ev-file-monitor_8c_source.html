102static void
103ev_file_monitor_timeout_start (EvFileMonitor *ev_monitor)
104{
105 ev_file_monitor_timeout_stop (ev_monitor);
106
107 ev_monitor->priv->timeout_id =
108 g_timeout_add_seconds (5, (GSourceFunc)timeout_cb, ev_monitor);

111static void
112ev_file_monitor_timeout_stop (EvFileMonitor *ev_monitor)
113{
114 if (ev_monitor->priv->timeout_id > 0) {
115 g_source_remove (ev_monitor->priv->timeout_id);
116 ev_monitor->priv->timeout_id = 0;
117 }

120static void
121ev_file_monitor_changed_cb (GFileMonitor *monitor,
122 GFile *file,
123 GFile *other_file,
124 GFileMonitorEvent event_type,
125 EvFileMonitor *ev_monitor)
126{
127 switch (event_type) {
128 case G_FILE_MONITOR_EVENT_CHANGES_DONE_HINT:
129 ev_file_monitor_timeout_stop (ev_monitor);
130 g_signal_emit (ev_monitor, signals[CHANGED], 0);
131
132 break;
133 case G_FILE_MONITOR_EVENT_CHANGED:
134 ev_file_monitor_timeout_start (ev_monitor);
135 break;
136 default:
137 break;
138 }
139}

141EvFileMonitor *
142ev_file_monitor_new (const gchar *uri)
143{
144 EvFileMonitor *ev_monitor;
145 GFile *file;
146 GError *error = NULL;
147
148 ev_monitor = EV_FILE_MONITOR (g_object_new (EV_TYPE_FILE_MONITOR, NULL));
149
150 file = g_file_new_for_uri (uri);
151 ev_monitor->priv->monitor = g_file_monitor_file (file, G_FILE_MONITOR_NONE, NULL, &error);
152 if (ev_monitor->priv->monitor) {
153 g_signal_connect (ev_monitor->priv->monitor, "changed",
154 G_CALLBACK (ev_file_monitor_changed_cb), ev_monitor);
155 } else if (error) {
156 g_warning ("%s", error->message);
157 g_error_free (error);
158 }
159
160 g_object_unref (file);
161
162 return ev_monitor;
163}
