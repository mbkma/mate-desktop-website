173 g_option_context_add_group (context, gtk_get_option_group(TRUE)); \
174 \
175 error = NULL; \
176 if (!g_option_context_parse (context, &argc, &argv, &error)) \
177 { \
178 if (error) \
179 { \
180 g_printerr ("Cannot parse arguments: %s.\n", error->message); \
181 g_error_free (error); \
182 } \
183 else \
184 { \
185 g_printerr ("Cannot parse arguments.\n"); \
186 } \
187 g_option_context_free (context); \
188 return 1; \
189 } \
190 \
191 gtk_init (&argc, &argv); \
192 \
193 retval = mate_panel_applet_factory_main (factory_id,TRUE, type, callback, data); \
194 g_option_context_free (context); \
195 \
196 return retval; \
197}
