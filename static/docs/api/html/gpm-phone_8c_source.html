67

139static void
140gpm_phone_battery_state_changed (DBusGProxy *proxy, guint idx, guint percentage, gboolean on_ac, GpmPhone *phone)
141{
142 g_return_if_fail (GPM_IS_PHONE (phone));
143
144 g_debug ("got BatteryStateChanged %u = %u (%i)", idx, percentage, on_ac);
145 phone->priv->percentage = percentage;
146 phone->priv->onac = on_ac;
147 phone->priv->present = TRUE;
148 g_debug ("emitting device-refresh : (%u)", idx);
149 g_signal_emit (phone, signals [DEVICE_REFRESH], 0, idx);

154static void
155gpm_phone_num_batteries_changed (DBusGProxy *proxy, guint number, GpmPhone *phone)
156{
157 g_return_if_fail (GPM_IS_PHONE (phone));
158
159 g_debug ("got NumberBatteriesChanged %u", number);
160 if (number > 1) {
161 g_warning ("number not 0 or 1, not valid!");
162 return;
163 }
164
165 /* are we removed? */
166 if (number == 0) {
167 phone->priv->present = FALSE;
168 phone->priv->percentage = 0;
169 phone->priv->onac = FALSE;
170 g_debug ("emitting device-removed : (%i)", 0);
171 g_signal_emit (phone, signals [DEVICE_REMOVED], 0, 0);
172 return;
173 }
174
175 if (phone->priv->present) {
176 g_warning ("duplicate NumberBatteriesChanged with no change");
177 return;
178 }
179
180 /* reset to defaults until we get BatteryStateChanged */
181 phone->priv->present = TRUE;
182 phone->priv->percentage = 0;
183 phone->priv->onac = FALSE;
184 g_debug ("emitting device-added : (%i)", 0);
185 g_signal_emit (phone, signals [DEVICE_ADDED], 0, 0);
186}

226static void
227gpm_phone_service_appeared_cb (GDBusConnection *connection,
228 const gchar *name, const gchar *name_owner,
229 GpmPhone *phone)
230{
231 GError *error = NULL;
232
233 g_return_if_fail (GPM_IS_PHONE (phone));
234
235 if (phone->priv->connection == NULL) {
236 g_debug ("get connection");
237 g_clear_error (&error);
238 phone->priv->connection = dbus_g_bus_get (DBUS_BUS_SESSION, &error);
239 if (error != NULL) {
240 g_warning ("Could not connect to DBUS daemon: %s", error->message);
241 g_error_free (error);
242 phone->priv->connection = NULL;
243 return;
244 }
245 }
246 if (phone->priv->proxy == NULL) {
247 g_debug ("get proxy");
248 g_clear_error (&error);
249 phone->priv->proxy = dbus_g_proxy_new_for_name_owner (phone->priv->connection,
250 MATE_PHONE_MANAGER_DBUS_SERVICE,
251 MATE_PHONE_MANAGER_DBUS_PATH,
252 MATE_PHONE_MANAGER_DBUS_INTERFACE,
253 &error);
254 if (error != NULL) {
255 g_warning ("Cannot connect, maybe the daemon is not running: %s", error->message);
256 g_error_free (error);
257 phone->priv->proxy = NULL;
258 return;
259 }
260
261 /* complicated type. ick */
262 dbus_g_object_register_marshaller(gpm_marshal_VOID__UINT_UINT_BOOLEAN,
263 G_TYPE_NONE, G_TYPE_UINT, G_TYPE_UINT,
264 G_TYPE_BOOLEAN, G_TYPE_INVALID);
265
266 /* get BatteryStateChanged */
267 dbus_g_proxy_add_signal (phone->priv->proxy, "BatteryStateChanged",
268 G_TYPE_UINT, G_TYPE_UINT, G_TYPE_BOOLEAN, G_TYPE_INVALID);
269 dbus_g_proxy_connect_signal (phone->priv->proxy, "BatteryStateChanged",
270 G_CALLBACK (gpm_phone_battery_state_changed),
271 phone, NULL);
272
273 /* get NumberBatteriesChanged */
274 dbus_g_proxy_add_signal (phone->priv->proxy, "NumberBatteriesChanged",
275 G_TYPE_UINT, G_TYPE_INVALID);
276 dbus_g_proxy_connect_signal (phone->priv->proxy, "NumberBatteriesChanged",
277 G_CALLBACK (gpm_phone_num_batteries_changed),
278 phone, NULL);
279
280 }
281}

311static void
312gpm_phone_init (GpmPhone *phone)
313{
314 phone->priv = gpm_phone_get_instance_private (phone);
315
316 phone->priv->connection = NULL;
317 phone->priv->proxy = NULL;
318 phone->priv->present = FALSE;
319 phone->priv->percentage = 0;
320 phone->priv->onac = FALSE;
321
322 phone->priv->watch_id = g_bus_watch_name (G_BUS_TYPE_SESSION,
323 MATE_PHONE_MANAGER_DBUS_SERVICE,
324 G_BUS_NAME_WATCHER_FLAGS_NONE,
325 (GBusNameAppearedCallback) gpm_phone_service_appeared_cb,
326 (GBusNameVanishedCallback) gpm_phone_service_vanished_cb,
327 phone, NULL);
328}

334static void

336{
337 GpmPhone *phone;
338 g_return_if_fail (GPM_IS_PHONE (object));
339
340 phone = GPM_PHONE (object);
341 phone->priv = gpm_phone_get_instance_private (phone);
342

344 g_object_unref (phone->priv->proxy);
