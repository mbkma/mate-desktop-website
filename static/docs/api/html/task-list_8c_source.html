62static void
63on_window_opened (WnckScreen *screen,
64 WnckWindow *window,
65 TaskList *list)
66{
67 WnckWindowType type;
68
69 g_return_if_fail (TASK_IS_LIST (list));
70
71 type = wnck_window_get_window_type (window);
72
73 if (type == WNCK_WINDOW_DESKTOP
74 || type == WNCK_WINDOW_DOCK
75 || type == WNCK_WINDOW_SPLASHSCREEN
76 || type == WNCK_WINDOW_MENU)
77 return;
78
79 GtkWidget *item = task_item_new (window);
80
81 if (item)
82 {
83 gtk_box_pack_start (GTK_BOX (list), item, FALSE, FALSE, 0);
84 g_signal_connect (TASK_ITEM (item), "task-item-closed",
85 G_CALLBACK (on_task_item_closed), list);
86 }
87}

196gboolean
197task_list_get_desktop_visible (TaskList *list)
198{
199 GList *windows, *w;
200 gboolean all_minimised = TRUE;
201
202 g_return_val_if_fail (TASK_IS_LIST (list), TRUE);
203
204 windows = wnck_screen_get_windows (list->priv->screen);
205 for (w = windows; w; w = w->next)
206 {
207 WnckWindow *window;
208
209 window = w->data;
210
211 if (WNCK_IS_WINDOW (window) && !wnck_window_is_minimized (window))
212 all_minimised = FALSE;
213 }
214
215 return all_minimised;
216}
