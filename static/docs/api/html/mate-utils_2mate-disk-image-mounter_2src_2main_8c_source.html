151int
152main (int argc, char *argv[])
153{
154 gint ret = 1;
155 GError *error = NULL;
156 gchar *s = NULL;
157 GOptionContext *o = NULL;
158 gint n;
159 GSList *uris = NULL;
160 GSList *l;
161
162#ifdef ENABLE_NLS
163 bindtextdomain (GETTEXT_PACKAGE, MATELOCALEDIR);
164 bind_textdomain_codeset (GETTEXT_PACKAGE, "UTF-8");
165 textdomain (GETTEXT_PACKAGE);
166#endif /* ENABLE_NLS */
167
168 have_gtk = gtk_init_check (&argc, &argv);
169
170 if (have_gtk)
171 gtk_window_set_default_icon_name ("drive-removable-media");
172
173 main_loop = g_main_loop_new (NULL, FALSE);
174
175 udisks_client = udisks_client_new_sync (NULL, &error);
176 if (udisks_client == NULL)
177 {
178 g_printerr (_("Error connecting to udisks daemon: %s (%s, %d)"),
179 error->message, g_quark_to_string (error->domain), error->code);
180 g_error_free (error);
181 goto out;
182 }
183
184 o = g_option_context_new (NULL);
185 g_option_context_set_help_enabled (o, FALSE);
186 g_option_context_set_summary (o, _("Attach and mount one or more disk image files."));
187 g_option_context_add_main_entries (o, opt_entries, GETTEXT_PACKAGE);
188
189 if (!g_option_context_parse (o, &argc, &argv, NULL))
190 {
191 s = g_option_context_get_help (o, FALSE, NULL);
192 g_printerr ("%s", s);
193 g_free (s);
194 g_option_context_free (o);
195 goto out;
196 }
197 g_option_context_free (o);
198
199 if (argc > 1)
200 {
201 for (n = 1; n < argc; n++)
202 uris = g_slist_prepend (uris, g_strdup (argv[n]));
203 uris = g_slist_reverse (uris);
204 }
205 else
206 {
207 if (!have_gtk)
208 {
209 show_error ("No files given and GTK+ not available");
210 goto out;
211 }
212 else
213 {
214 uris = do_filechooser ();
215 }
216 }
217
218 /* Files to attach are positional arguments */
219 for (l = uris; l != NULL; l = l->next)
220 {
221 const gchar *uri;
222 gchar *filename;
223 GUnixFDList *fd_list = NULL;
224 GVariantBuilder options_builder;
225 gint fd;
226 gchar *loop_object_path = NULL;
227 GFile *file;
228
229 uri = l->data;
230 file = g_file_new_for_commandline_arg (uri);
231 filename = g_file_get_path (file);
232 g_object_unref (file);
233
234 if (filename == NULL)
235 {
236 show_error (_("Cannot open `%s' - maybe the volume isn't mounted?"), uri);
237 goto done_with_image;
238 }
239
240 fd = open (filename, opt_writable ? O_RDWR : O_RDONLY);
241 if (fd == -1)
242 {
243 show_error (_("Error opening `%s': %m"), filename);
244 goto done_with_image;
245 }
246
247 g_variant_builder_init (&options_builder, G_VARIANT_TYPE ("a{sv}"));
248 if (!opt_writable)
249 g_variant_builder_add (&options_builder, "{sv}", "read-only", g_variant_new_boolean (TRUE));
250
251 fd_list = g_unix_fd_list_new_from_array (&fd, 1); /* adopts the fd */
252
253 /* Set up the disk image... */
254 error = NULL;
255 if (!udisks_manager_call_loop_setup_sync (udisks_client_get_manager (udisks_client),
256 g_variant_new_handle (0),
257 g_variant_builder_end (&options_builder),
258 fd_list,
259 &loop_object_path,
260 NULL, /* out_fd_list */
261 NULL, /* GCancellable */
262 &error))
263 {
264 show_error (_("Error attaching disk image: %s (%s, %d)"),
265 error->message, g_quark_to_string (error->domain), error->code);
266 g_clear_error (&error);
267 goto done_with_image;
268 }
269
270 /* Note that the desktop automounter is responsible for mounting,
271 * unlocking etc. partitions etc. inside the image...
272 */
273
274 done_with_image:
275
276 g_clear_object (&fd_list);
277 g_free (filename);
278 g_free (loop_object_path);
279
280 } /* for each image */
281
282 ret = 0;
283
284 out:
285 if (main_loop != NULL)
286 g_main_loop_unref (main_loop);
287 g_slist_free_full (uris, g_free);
288 g_clear_object (&udisks_client);
289 return ret;
290}
