35/* shared utility to create a new pixbuf from the passed-in one */
36
37static GdkPixbuf *
38create_new_pixbuf (GdkPixbuf *src)
39{
40 g_assert (gdk_pixbuf_get_colorspace (src) == GDK_COLORSPACE_RGB);
41 g_assert ((!gdk_pixbuf_get_has_alpha (src)
42 && gdk_pixbuf_get_n_channels (src) == 3)
43 || (gdk_pixbuf_get_has_alpha (src)
44 && gdk_pixbuf_get_n_channels (src) == 4));
45
46 return gdk_pixbuf_new (gdk_pixbuf_get_colorspace (src),
47 gdk_pixbuf_get_has_alpha (src),
48 gdk_pixbuf_get_bits_per_sample (src),
49 gdk_pixbuf_get_width (src),
50 gdk_pixbuf_get_height (src));
51}

53/* utility routine to bump the level of a color component with pinning */
54
55static guchar
56lighten_component (guchar cur_value)
57{
58 int new_value = cur_value;
59 new_value += 24 + (new_value >> 3);
60 if (new_value > 255)
61 {
62 new_value = 255;
63 }
64 return (guchar) new_value;
65}

67GdkPixbuf *
68eel_create_spotlight_pixbuf (GdkPixbuf* src)
69{
70 GdkPixbuf *dest;
71 int i, j;
72 int width, height, has_alpha, src_row_stride, dst_row_stride;
73 guchar *target_pixels, *original_pixels;
74
75 g_return_val_if_fail (gdk_pixbuf_get_colorspace (src) == GDK_COLORSPACE_RGB, NULL);
76 g_return_val_if_fail ((!gdk_pixbuf_get_has_alpha (src)
77 && gdk_pixbuf_get_n_channels (src) == 3)
78 || (gdk_pixbuf_get_has_alpha (src)
79 && gdk_pixbuf_get_n_channels (src) == 4), NULL);
80 g_return_val_if_fail (gdk_pixbuf_get_bits_per_sample (src) == 8, NULL);
81
82 dest = create_new_pixbuf (src);
83
84 has_alpha = gdk_pixbuf_get_has_alpha (src);
85 width = gdk_pixbuf_get_width (src);
86 height = gdk_pixbuf_get_height (src);
87 dst_row_stride = gdk_pixbuf_get_rowstride (dest);
88 src_row_stride = gdk_pixbuf_get_rowstride (src);
89 target_pixels = gdk_pixbuf_get_pixels (dest);
90 original_pixels = gdk_pixbuf_get_pixels (src);
91
92 for (i = 0; i < height; i++)
93 {
94 guchar *pixsrc, *pixdest;
95
96 pixdest = target_pixels + i * dst_row_stride;
97 pixsrc = original_pixels + i * src_row_stride;
98 for (j = 0; j < width; j++)
99 {
100 *pixdest++ = lighten_component (*pixsrc++);
101 *pixdest++ = lighten_component (*pixsrc++);
102 *pixdest++ = lighten_component (*pixsrc++);
103 if (has_alpha)
104 {
105 *pixdest++ = *pixsrc++;
106 }
107 }
108 }
109 return dest;
110}

127/* the following routine was stolen from the panel to darken a pixbuf, by manipulating the saturation */
128
129/* saturation is 0-255, darken is 0-255 */
130
131GdkPixbuf *
132eel_create_darkened_pixbuf (GdkPixbuf *src, int saturation, int darken)
133{
134 gint i, j;
135 gint width, height, src_row_stride, dest_row_stride;
136 gboolean has_alpha;
137 guchar *target_pixels, *original_pixels;
138 guchar intensity;
139 guchar alpha;
140 guchar negalpha;
141 guchar r, g, b;
142 GdkPixbuf *dest;
143
144 g_return_val_if_fail (gdk_pixbuf_get_colorspace (src) == GDK_COLORSPACE_RGB, NULL);
145 g_return_val_if_fail ((!gdk_pixbuf_get_has_alpha (src)
146 && gdk_pixbuf_get_n_channels (src) == 3)
147 || (gdk_pixbuf_get_has_alpha (src)
148 && gdk_pixbuf_get_n_channels (src) == 4), NULL);
149 g_return_val_if_fail (gdk_pixbuf_get_bits_per_sample (src) == 8, NULL);
150
151 dest = create_new_pixbuf (src);
152
153 has_alpha = gdk_pixbuf_get_has_alpha (src);
154 width = gdk_pixbuf_get_width (src);
155 height = gdk_pixbuf_get_height (src);
156 dest_row_stride = gdk_pixbuf_get_rowstride (dest);
157 src_row_stride = gdk_pixbuf_get_rowstride (src);
158 target_pixels = gdk_pixbuf_get_pixels (dest);
159 original_pixels = gdk_pixbuf_get_pixels (src);
160
161 for (i = 0; i < height; i++)
162 {
163 guchar *pixsrc, *pixdest;
164
165 pixdest = target_pixels + i * dest_row_stride;
166 pixsrc = original_pixels + i * src_row_stride;
167 for (j = 0; j < width; j++)
168 {
169 r = *pixsrc++;
170 g = *pixsrc++;
171 b = *pixsrc++;
172 intensity = (r * 77 + g * 150 + b * 28) >> 8;
173 negalpha = ((255 - saturation) * darken) >> 8;
174 alpha = (saturation * darken) >> 8;
175 *pixdest++ = (negalpha * intensity + alpha * r) >> 8;
176 *pixdest++ = (negalpha * intensity + alpha * g) >> 8;
177 *pixdest++ = (negalpha * intensity + alpha * b) >> 8;
178 if (has_alpha)
179 {
180 *pixdest++ = *pixsrc++;
181 }
182 }
183 }
184 return dest;
185}

187/* this routine colorizes the passed-in pixbuf by multiplying each pixel with the passed in color */

189GdkPixbuf *
190eel_create_colorized_pixbuf (GdkPixbuf *src,
191 GdkRGBA *color)
192{
193 int i, j;
194 int width, height, has_alpha, src_row_stride, dst_row_stride;
195 guchar *target_pixels;
196 guchar *original_pixels;
197 GdkPixbuf *dest;
198
199 gint red_value, green_value, blue_value;
200
201 g_return_val_if_fail (gdk_pixbuf_get_colorspace (src) == GDK_COLORSPACE_RGB, NULL);
202 g_return_val_if_fail ((!gdk_pixbuf_get_has_alpha (src)
203 && gdk_pixbuf_get_n_channels (src) == 3)
204 || (gdk_pixbuf_get_has_alpha (src)
205 && gdk_pixbuf_get_n_channels (src) == 4), NULL);
206 g_return_val_if_fail (gdk_pixbuf_get_bits_per_sample (src) == 8, NULL);
207
208 red_value = eel_round (color->red * 255);
209 green_value = eel_round (color->green * 255);
210 blue_value = eel_round (color->blue * 255);
211
212 dest = create_new_pixbuf (src);
213
214 has_alpha = gdk_pixbuf_get_has_alpha (src);
215 width = gdk_pixbuf_get_width (src);
216 height = gdk_pixbuf_get_height (src);
217 src_row_stride = gdk_pixbuf_get_rowstride (src);
218 dst_row_stride = gdk_pixbuf_get_rowstride (dest);
219 target_pixels = gdk_pixbuf_get_pixels (dest);
220 original_pixels = gdk_pixbuf_get_pixels (src);
221
222 for (i = 0; i < height; i++)
223 {
224 guchar *pixdest;
225 guchar *pixsrc;
226
227 pixdest = target_pixels + i*dst_row_stride;
228 pixsrc = original_pixels + i*src_row_stride;
229 for (j = 0; j < width; j++)
230 {
231 *pixdest++ = (*pixsrc++ * red_value) >> 8;
232 *pixdest++ = (*pixsrc++ * green_value) >> 8;
233 *pixdest++ = (*pixsrc++ * blue_value) >> 8;
234 if (has_alpha)
235 {
236 *pixdest++ = *pixsrc++;
237 }
238 }
239 }
240 return dest;
241}

243/* utility to stretch a frame to the desired size */
244
245static void
246draw_frame_row (GdkPixbuf *frame_image, int target_width, int source_width, int source_v_position, int dest_v_position, GdkPixbuf *result_pixbuf, int left_offset, int height)
247{
248 int remaining_width, h_offset;
249
250 remaining_width = target_width;
251 h_offset = 0;
252 while (remaining_width > 0)
253 {
254 int slab_width;
255
256 slab_width = remaining_width > source_width ? source_width : remaining_width;
257 gdk_pixbuf_copy_area (frame_image, left_offset, source_v_position, slab_width, height, result_pixbuf, left_offset + h_offset, dest_v_position);
258 remaining_width -= slab_width;
259 h_offset += slab_width;
260 }
261}

283GdkPixbuf *
284eel_stretch_frame_image (GdkPixbuf *frame_image, int left_offset, int top_offset, int right_offset, int bottom_offset,
285 int dest_width, int dest_height, gboolean fill_flag)
286{
287 GdkPixbuf *result_pixbuf;
288 guchar *pixels_ptr;
289 int frame_width, frame_height;
290 int row_stride;
291 int target_width, target_frame_width;
292 int target_height, target_frame_height;
293
294 frame_width = gdk_pixbuf_get_width (frame_image);
295 frame_height = gdk_pixbuf_get_height (frame_image );
296
297 if (fill_flag)
298 {
299 result_pixbuf = gdk_pixbuf_scale_simple (frame_image, dest_width, dest_height, GDK_INTERP_NEAREST);
300 }
301 else
302 {
303 result_pixbuf = gdk_pixbuf_new (GDK_COLORSPACE_RGB, TRUE, 8, dest_width, dest_height);
304 }
305 row_stride = gdk_pixbuf_get_rowstride (result_pixbuf);
306 pixels_ptr = gdk_pixbuf_get_pixels (result_pixbuf);
307
308 /* clear the new pixbuf */
309 if (!fill_flag)
310 {
311 int y;
312
313 for (y = 0; y < dest_height; y++)
314 {
315 memset (pixels_ptr, 255, row_stride);
316 pixels_ptr += row_stride;
317 }
318 }
319
320 target_width = dest_width - left_offset - right_offset;
321 target_frame_width = frame_width - left_offset - right_offset;
322
323 target_height = dest_height - top_offset - bottom_offset;
324 target_frame_height = frame_height - top_offset - bottom_offset;
325
326 /* draw the left top corner and top row */
327 gdk_pixbuf_copy_area (frame_image, 0, 0, left_offset, top_offset, result_pixbuf, 0, 0);
328 draw_frame_row (frame_image, target_width, target_frame_width, 0, 0, result_pixbuf, left_offset, top_offset);
329
330 /* draw the right top corner and left column */
331 gdk_pixbuf_copy_area (frame_image, frame_width - right_offset, 0, right_offset, top_offset, result_pixbuf, dest_width - right_offset, 0);
332 draw_frame_column (frame_image, target_height, target_frame_height, 0, 0, result_pixbuf, top_offset, left_offset);
333
334 /* draw the bottom right corner and bottom row */
335 gdk_pixbuf_copy_area (frame_image, frame_width - right_offset, frame_height - bottom_offset, right_offset, bottom_offset, result_pixbuf, dest_width - right_offset, dest_height - bottom_offset);
336 draw_frame_row (frame_image, target_width, target_frame_width, frame_height - bottom_offset, dest_height - bottom_offset, result_pixbuf, left_offset, bottom_offset);
337
338 /* draw the bottom left corner and the right column */
339 gdk_pixbuf_copy_area (frame_image, 0, frame_height - bottom_offset, left_offset, bottom_offset, result_pixbuf, 0, dest_height - bottom_offset);
340 draw_frame_column (frame_image, target_height, target_frame_height, frame_width - right_offset, dest_width - right_offset, result_pixbuf, top_offset, right_offset);
341
342 return result_pixbuf;
343}

345/* draw an arbitrary frame around an image, with the result passed back in a newly allocated pixbuf */
346GdkPixbuf *
347eel_embed_image_in_frame (GdkPixbuf *source_image, GdkPixbuf *frame_image, int left_offset, int top_offset, int right_offset, int bottom_offset)
348{
349 GdkPixbuf *result_pixbuf;
350 int source_width, source_height;
351 int dest_width, dest_height;
352
353 source_width = gdk_pixbuf_get_width (source_image);
354 source_height = gdk_pixbuf_get_height (source_image);
355
356 dest_width = source_width + left_offset + right_offset;
357 dest_height = source_height + top_offset + bottom_offset;
358
359 result_pixbuf = eel_stretch_frame_image (frame_image, left_offset, top_offset, right_offset, bottom_offset,
360 dest_width, dest_height, FALSE);
361
362 /* Finally, copy the source image into the framed area */
363 gdk_pixbuf_copy_area (source_image, 0, 0, source_width, source_height, result_pixbuf, left_offset, top_offset);
364
365 return result_pixbuf;
366}
