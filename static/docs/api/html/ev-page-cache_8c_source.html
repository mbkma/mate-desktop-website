143static void
144ev_page_cache_finalize (GObject *object)
145{
146 EvPageCache *cache = EV_PAGE_CACHE (object);
147 gint i;
148
149 if (cache->page_list) {
150 for (i = 0; i < cache->n_pages; i++) {
151 EvPageCacheData *data;
152
153 data = &cache->page_list[i];
154
155 if (data->job) {
156 g_signal_handlers_disconnect_by_func (data->job,
157 G_CALLBACK (job_page_data_finished_cb),
158 cache);
159 g_signal_handlers_disconnect_by_func (data->job,
160 G_CALLBACK (job_page_data_cancelled_cb),
161 data);
162 }
163 ev_page_cache_data_free (data);
164 }
165
166 g_free (cache->page_list);
167 cache->page_list = NULL;
168 cache->n_pages = 0;
169 }
170
171 if (cache->document) {
172 g_object_unref (cache->document);
173 cache->document = NULL;
174 }
175
176 G_OBJECT_CLASS (ev_page_cache_parent_class)->finalize (object);
177}

192static EvJobPageDataFlags
193ev_page_cache_get_flags_for_data (EvPageCache *cache,
194 EvPageCacheData *data)
195{
196 EvJobPageDataFlags flags = EV_PAGE_DATA_INCLUDE_NONE;
197
198 if (data->flags == cache->flags && !data->dirty)
199 return cache->flags;
200
201 /* Flags changed or data is dirty */
202 if (cache->flags & EV_PAGE_DATA_INCLUDE_LINKS) {
203 flags = (data->link_mapping) ?
204 flags & ~EV_PAGE_DATA_INCLUDE_LINKS :
205 flags | EV_PAGE_DATA_INCLUDE_LINKS;
206 }
207
208 if (cache->flags & EV_PAGE_DATA_INCLUDE_IMAGES) {
209 flags = (data->image_mapping) ?
210 flags & ~EV_PAGE_DATA_INCLUDE_IMAGES :
211 flags | EV_PAGE_DATA_INCLUDE_IMAGES;
212 }
213
214 if (cache->flags & EV_PAGE_DATA_INCLUDE_FORMS) {
215 flags = (data->form_field_mapping) ?
216 flags & ~EV_PAGE_DATA_INCLUDE_FORMS :
217 flags | EV_PAGE_DATA_INCLUDE_FORMS;
218 }
219
220 if (cache->flags & EV_PAGE_DATA_INCLUDE_ANNOTS) {
221 flags = (data->annot_mapping) ?
222 flags & ~EV_PAGE_DATA_INCLUDE_ANNOTS :
223 flags | EV_PAGE_DATA_INCLUDE_ANNOTS;
224 }
225
226 if (cache->flags & EV_PAGE_DATA_INCLUDE_TEXT_MAPPING) {
227 flags = (data->text_mapping) ?
228 flags & ~EV_PAGE_DATA_INCLUDE_TEXT_MAPPING :
229 flags | EV_PAGE_DATA_INCLUDE_TEXT_MAPPING;
230 }
231
232 if (cache->flags & EV_PAGE_DATA_INCLUDE_TEXT) {
233 flags = (data->text) ?
234 flags & ~EV_PAGE_DATA_INCLUDE_TEXT :
235 flags | EV_PAGE_DATA_INCLUDE_TEXT;
236 }
237
238 if (cache->flags & EV_PAGE_DATA_INCLUDE_TEXT_LAYOUT) {
239 flags = (data->text_layout_length > 0) ?
240 flags & ~EV_PAGE_DATA_INCLUDE_TEXT_LAYOUT :
241 flags | EV_PAGE_DATA_INCLUDE_TEXT_LAYOUT;
242 }
243
244 if (cache->flags & EV_PAGE_DATA_INCLUDE_TEXT_ATTRS) {
245 flags = (data->text_attrs) ?
246 flags & ~EV_PAGE_DATA_INCLUDE_TEXT_ATTRS :
247 flags | EV_PAGE_DATA_INCLUDE_TEXT_ATTRS;
248 }
249
250 if (cache->flags & EV_PAGE_DATA_INCLUDE_TEXT_LOG_ATTRS) {
251 flags = (data->text_log_attrs) ?
252 flags & ~EV_PAGE_DATA_INCLUDE_TEXT_LOG_ATTRS :
253 flags | EV_PAGE_DATA_INCLUDE_TEXT_LOG_ATTRS;
254 }
255
256 return flags;
257}

275static void
276job_page_data_finished_cb (EvJob *job,
277 EvPageCache *cache)
278{
279 EvJobPageData *job_data = EV_JOB_PAGE_DATA (job);
280 EvPageCacheData *data;
281
282 data = &cache->page_list[job_data->page];
283
284 if (job_data->flags & EV_PAGE_DATA_INCLUDE_LINKS)
285 data->link_mapping = job_data->link_mapping;
286 if (job_data->flags & EV_PAGE_DATA_INCLUDE_IMAGES)
287 data->image_mapping = job_data->image_mapping;
288 if (job_data->flags & EV_PAGE_DATA_INCLUDE_FORMS)
289 data->form_field_mapping = job_data->form_field_mapping;
290 if (job_data->flags & EV_PAGE_DATA_INCLUDE_ANNOTS)
291 data->annot_mapping = job_data->annot_mapping;
292 if (job_data->flags & EV_PAGE_DATA_INCLUDE_TEXT_MAPPING)
293 data->text_mapping = job_data->text_mapping;
294 if (job_data->flags & EV_PAGE_DATA_INCLUDE_TEXT_LAYOUT) {
295 data->text_layout = job_data->text_layout;
296 data->text_layout_length = job_data->text_layout_length;
297 }
298 if (job_data->flags & EV_PAGE_DATA_INCLUDE_TEXT)
299 data->text = job_data->text;
300 if (job_data->flags & EV_PAGE_DATA_INCLUDE_TEXT_ATTRS)
301 data->text_attrs = job_data->text_attrs;
302 if (job_data->flags & EV_PAGE_DATA_INCLUDE_TEXT_LOG_ATTRS) {
303 data->text_log_attrs = job_data->text_log_attrs;
304 data->text_log_attrs_length = job_data->text_log_attrs_length;
305 }
306
307 data->done = TRUE;
308 data->dirty = FALSE;
309
310 g_object_unref (data->job);
311 data->job = NULL;

322static void
323ev_page_cache_schedule_job_if_needed (EvPageCache *cache,
324 gint page)
325{
326 EvPageCacheData *data = &cache->page_list[page];
327 EvJobPageDataFlags flags;
328
329 if (data->flags == cache->flags && !data->dirty && (data->done || data->job))
330 return;
331
332 if (data->job)
333 ev_job_cancel (data->job);
334
335 flags = ev_page_cache_get_flags_for_data (cache, data);
336
337 data->flags = cache->flags;
338 data->job = ev_job_page_data_new (cache->document, page, flags);
339 g_signal_connect (data->job, "finished",
340 G_CALLBACK (job_page_data_finished_cb),
341 cache);
342 g_signal_connect (data->job, "cancelled",
343 G_CALLBACK (job_page_data_cancelled_cb),
344 data);
345 ev_job_scheduler_push_job (data->job, EV_JOB_PRIORITY_NONE);
346
347}

349void
350ev_page_cache_set_page_range (EvPageCache *cache,
351 gint start,
352 gint end)
353{
354 gint i;
355 gint pages_to_pre_cache;
356
357 if (cache->flags == EV_PAGE_DATA_INCLUDE_NONE)
358 return;
359
360 for (i = start; i <= end; i++)
361 ev_page_cache_schedule_job_if_needed (cache, i);
362
363 cache->start_page = start;
364 cache->end_page = end;
365
366 i = 1;
367 pages_to_pre_cache = PRE_CACHE_SIZE * 2;
368 while ((start - i > 0) || (end + i < cache->n_pages)) {
369 if (end + i < cache->n_pages) {
370 ev_page_cache_schedule_job_if_needed (cache, end + i);
371 if (--pages_to_pre_cache == 0)
372 break;
373 }
374
375 if (start - i > 0) {
376 ev_page_cache_schedule_job_if_needed (cache, start - i);
377 if (--pages_to_pre_cache == 0)
378 break;
379 }
380 i++;
381 }
382}

403void
404ev_page_cache_mark_dirty (EvPageCache *cache,
405 gint page,
406 EvJobPageDataFlags flags)
407{
408 EvPageCacheData *data;
409
410 g_return_if_fail (EV_IS_PAGE_CACHE (cache));
411
412 data = &cache->page_list[page];
413 data->dirty = TRUE;
414
415 if (flags & EV_PAGE_DATA_INCLUDE_LINKS)
416 g_clear_pointer (&data->link_mapping, ev_mapping_list_unref);
417
418 if (flags & EV_PAGE_DATA_INCLUDE_IMAGES)
419 g_clear_pointer (&data->image_mapping, ev_mapping_list_unref);
420
421 if (flags & EV_PAGE_DATA_INCLUDE_FORMS)
422 g_clear_pointer (&data->form_field_mapping, ev_mapping_list_unref);
423
424 if (flags & EV_PAGE_DATA_INCLUDE_ANNOTS)
425 g_clear_pointer (&data->annot_mapping, ev_mapping_list_unref);
426
427 if (flags & EV_PAGE_DATA_INCLUDE_TEXT_MAPPING)
428 g_clear_pointer (&data->text_mapping, cairo_region_destroy);
429
430 if (flags & EV_PAGE_DATA_INCLUDE_TEXT)
431 g_clear_pointer (&data->text, g_free);
432
433 if (flags & EV_PAGE_DATA_INCLUDE_TEXT_LAYOUT)
434 g_clear_pointer (&data->text_layout, g_free);
435
436 /* Update the current range */
437 ev_page_cache_set_page_range (cache, cache->start_page, cache->end_page);
438}
