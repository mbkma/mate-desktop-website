113static void
114pluma_print_job_get_property (GObject *object,
115 guint prop_id,
116 GValue *value,
117 GParamSpec *pspec)
118{
119 PlumaPrintJob *job = PLUMA_PRINT_JOB (object);
120

151static void
152pluma_print_job_finalize (GObject *object)
153{
154 PlumaPrintJob *job = PLUMA_PRINT_JOB (object);
155
156 g_free (job->priv->status_string);
157
158 if (job->priv->compositor != NULL)
159 g_object_unref (job->priv->compositor);
160
161 if (job->priv->operation != NULL)
162 g_object_unref (job->priv->operation);
163
164 G_OBJECT_CLASS (pluma_print_job_parent_class)->finalize (object);
165}

177static void
178pluma_print_job_class_init (PlumaPrintJobClass *klass)
179{
180 GObjectClass *object_class;
181
182 object_class = G_OBJECT_CLASS (klass);
183
184 object_class->get_property = pluma_print_job_get_property;
185 object_class->set_property = pluma_print_job_set_property;

200 g_signal_new ("printing",

289static GObject *
290create_custom_widget_cb (GtkPrintOperation *operation,
291 PlumaPrintJob *job)
292{
293 gboolean ret;
294 GtkWidget *widget;
295 GtkWidget *error_widget;
296 guint line_numbers;
297 GtkWrapMode wrap_mode;
298 gboolean syntax_hl;
299 gboolean print_header;
300 gchar *font_body, *font_header, *font_numbers;
301 gchar *root_objects[] = {
302 "adjustment1",
303 "contents",
304 NULL
305 };
306
307 ret = pluma_utils_get_ui_objects (PLUMA_DATADIR "/ui/pluma-print-preferences.ui",
308 root_objects,
309 &error_widget,
310 "contents", &widget,
311 "syntax_checkbutton", &job->priv->syntax_checkbutton,
312 "line_numbers_checkbutton", &job->priv->line_numbers_checkbutton,
313 "line_numbers_hbox", &job->priv->line_numbers_hbox,
314 "line_numbers_spinbutton", &job->priv->line_numbers_spinbutton,
315 "page_header_checkbutton", &job->priv->page_header_checkbutton,
316 "text_wrapping_checkbutton", &job->priv->text_wrapping_checkbutton,
317 "do_not_split_checkbutton", &job->priv->do_not_split_checkbutton,
318 "fonts_table", &job->priv->fonts_table,
319 "body_font_label", &job->priv->body_font_label,
320 "body_fontbutton", &job->priv->body_fontbutton,
321 "headers_font_label", &job->priv->headers_font_label,
322 "headers_fontbutton", &job->priv->headers_fontbutton,
323 "numbers_font_label", &job->priv->numbers_font_label,
324 "numbers_fontbutton", &job->priv->numbers_fontbutton,
325 "restore_button", &job->priv->restore_button,
326 NULL);
327
328 if (!ret)
329 {
330 return G_OBJECT (error_widget);
331 }
332
333 /* Get all settings values */
334 syntax_hl = g_settings_get_boolean (job->priv->print_settings,
335 PLUMA_SETTINGS_PRINT_SYNTAX_HIGHLIGHTING);
336 print_header = g_settings_get_boolean (job->priv->print_settings,
337 PLUMA_SETTINGS_PRINT_HEADER);
338 line_numbers = g_settings_get_uint (job->priv->print_settings, PLUMA_SETTINGS_PRINT_LINE_NUMBERS);
339
340 font_body = g_settings_get_string (job->priv->print_settings,
341 PLUMA_SETTINGS_PRINT_FONT_BODY_PANGO);
342 font_header = g_settings_get_string (job->priv->print_settings,
343 PLUMA_SETTINGS_PRINT_FONT_HEADER_PANGO);
344 font_numbers = g_settings_get_string (job->priv->print_settings,
345 PLUMA_SETTINGS_PRINT_FONT_NUMBERS_PANGO);
346
347 /* Print syntax */
348 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (job->priv->syntax_checkbutton),
349 syntax_hl);
350
351 /* Print page headers */
352 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (job->priv->page_header_checkbutton),
353 print_header);
354
355
356 /* Line numbers */
357 gtk_toggle_button_set_active (GTK_TOGGLE_BUTTON (job->priv->line_numbers_checkbutton),
358 line_numbers > 0);
359
360 if (line_numbers > 0)
361 {
362 gtk_spin_button_set_value (GTK_SPIN_BUTTON (job->priv->line_numbers_spinbutton),
363 (guint) line_numbers);
364 gtk_widget_set_sensitive (job->priv->line_numbers_hbox, TRUE);
365 }
366 else
367 {
368 gtk_spin_button_set_value (GTK_SPIN_BUTTON (job->priv->line_numbers_spinbutton),
369 1);
370 gtk_widget_set_sensitive (job->priv->line_numbers_hbox, FALSE);
371 }
372
373 /* Text wrapping */
374 wrap_mode = pluma_settings_get_wrap_mode (job->priv->print_settings,
375 PLUMA_SETTINGS_PRINT_WRAP_MODE);
376
377 switch (wrap_mode)
378 {
379 case GTK_WRAP_WORD:
380 gtk_toggle_button_set_active (
381 GTK_TOGGLE_BUTTON (job->priv->text_wrapping_checkbutton), TRUE);
382 gtk_toggle_button_set_active (
383 GTK_TOGGLE_BUTTON (job->priv->do_not_split_checkbutton), TRUE);
384 break;
385 case GTK_WRAP_CHAR:
386 gtk_toggle_button_set_active (
387 GTK_TOGGLE_BUTTON (job->priv->text_wrapping_checkbutton), TRUE);
388 gtk_toggle_button_set_active (
389 GTK_TOGGLE_BUTTON (job->priv->do_not_split_checkbutton), FALSE);
390 break;
391 default:
392 gtk_toggle_button_set_active (
393 GTK_TOGGLE_BUTTON (job->priv->text_wrapping_checkbutton), FALSE);
394 gtk_toggle_button_set_inconsistent (
395 GTK_TOGGLE_BUTTON (job->priv->do_not_split_checkbutton), TRUE);
396 }
397
398 gtk_widget_set_sensitive (job->priv->do_not_split_checkbutton,
399 (wrap_mode != GTK_WRAP_NONE));
400
401 /* Set initial values */
402 gtk_font_chooser_set_font (GTK_FONT_CHOOSER (job->priv->body_fontbutton),
403 font_body);
404 g_free (font_body);
405
406 gtk_font_chooser_set_font (GTK_FONT_CHOOSER (job->priv->headers_fontbutton),
407 font_header);
408 g_free (font_header);
409
410 gtk_font_chooser_set_font (GTK_FONT_CHOOSER (job->priv->numbers_fontbutton),
411 font_numbers);
412 g_free (font_numbers);
413
414 g_signal_connect (job->priv->line_numbers_checkbutton,
415 "toggled",
416 G_CALLBACK (line_numbers_checkbutton_toggled),
417 job);
418 g_signal_connect (job->priv->text_wrapping_checkbutton,
419 "toggled",
420 G_CALLBACK (wrap_mode_checkbutton_toggled),
421 job);
422 g_signal_connect (job->priv->do_not_split_checkbutton,
423 "toggled",
424 G_CALLBACK (wrap_mode_checkbutton_toggled),
425 job);
426 g_signal_connect (job->priv->restore_button,
427 "clicked",
428 G_CALLBACK (restore_button_clicked),
429 job);
430
431 return G_OBJECT (widget);
432}

434static void
435custom_widget_apply_cb (GtkPrintOperation *operation,
436 GtkWidget *widget,
437 PlumaPrintJob *job)
438{
439 gboolean syntax, page_header;
440 const gchar *body, *header, *numbers;
441
442 syntax = gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (job->priv->syntax_checkbutton));
443 page_header = gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (job->priv->page_header_checkbutton));
444 body = gtk_font_chooser_get_font (GTK_FONT_CHOOSER (job->priv->body_fontbutton));
445 header = gtk_font_chooser_get_font (GTK_FONT_CHOOSER (job->priv->headers_fontbutton));
446 numbers = gtk_font_chooser_get_font (GTK_FONT_CHOOSER (job->priv->numbers_fontbutton));
447
448 g_settings_set_boolean (job->priv->print_settings,
449 PLUMA_SETTINGS_PRINT_SYNTAX_HIGHLIGHTING, syntax);
450 g_settings_set_boolean (job->priv->print_settings, PLUMA_SETTINGS_PRINT_HEADER,
451 page_header);
452 g_settings_set_string (job->priv->print_settings, PLUMA_SETTINGS_PRINT_FONT_BODY_PANGO,
453 body);
454 g_settings_set_string (job->priv->print_settings, PLUMA_SETTINGS_PRINT_FONT_HEADER_PANGO,
455 header);
456 g_settings_set_string (job->priv->print_settings, PLUMA_SETTINGS_PRINT_FONT_NUMBERS_PANGO,
457 numbers);
458
459
460 if (gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (job->priv->line_numbers_checkbutton)))
461 {
462 g_settings_set_uint (job->priv->print_settings,
463 PLUMA_SETTINGS_PRINT_LINE_NUMBERS,
464 MAX (1, gtk_spin_button_get_value_as_int (GTK_SPIN_BUTTON (job->priv->line_numbers_spinbutton))));
465
466 }
467 else
468 {
469 g_settings_set_uint (job->priv->print_settings,
470 PLUMA_SETTINGS_PRINT_LINE_NUMBERS, 0);
471 }
472
473 if (!gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (job->priv->text_wrapping_checkbutton)))
474 {
475 pluma_settings_set_wrap_mode (job->priv->print_settings,
476 PLUMA_SETTINGS_PRINT_WRAP_MODE,
477 GTK_WRAP_NONE);
478 }
479 else
480 {
481 if (gtk_toggle_button_get_active (GTK_TOGGLE_BUTTON (job->priv->do_not_split_checkbutton)))
482 {
483 pluma_settings_set_wrap_mode (job->priv->print_settings,
484 PLUMA_SETTINGS_PRINT_WRAP_MODE,
485 GTK_WRAP_WORD);
486
487 }
488 else
489 {
490 pluma_settings_set_wrap_mode (job->priv->print_settings,
491 PLUMA_SETTINGS_PRINT_WRAP_MODE,
492 GTK_WRAP_CHAR);
493
494 }
495 }

498static void
499create_compositor (PlumaPrintJob *job)
500{
501 gchar *print_font_body;
502 gchar *print_font_header;
503 gchar *print_font_numbers;
504 gboolean syntax_hl;
505 GtkWrapMode wrap_mode;
506 guint print_line_numbers;
507 gboolean print_header;
508
509
510 /* Create and initialize print compositor */
511 print_font_body = g_settings_get_string (job->priv->print_settings,
512 PLUMA_SETTINGS_PRINT_FONT_BODY_PANGO);
513 print_font_header = g_settings_get_string (job->priv->print_settings,
514 PLUMA_SETTINGS_PRINT_FONT_HEADER_PANGO);
515 print_font_numbers = g_settings_get_string (job->priv->print_settings,
516 PLUMA_SETTINGS_PRINT_FONT_NUMBERS_PANGO);
517 syntax_hl = g_settings_get_boolean (job->priv->print_settings,
518 PLUMA_SETTINGS_PRINT_SYNTAX_HIGHLIGHTING);
519 print_line_numbers = g_settings_get_uint (job->priv->print_settings, PLUMA_SETTINGS_PRINT_LINE_NUMBERS);
520 print_header = g_settings_get_boolean (job->priv->print_settings,
521 PLUMA_SETTINGS_PRINT_HEADER);
522 wrap_mode = pluma_settings_get_wrap_mode (job->priv->print_settings,
523 PLUMA_SETTINGS_PRINT_WRAP_MODE);
524
525
526 job->priv->compositor = GTK_SOURCE_PRINT_COMPOSITOR (
527 g_object_new (GTK_SOURCE_TYPE_PRINT_COMPOSITOR,
528 "buffer", GTK_SOURCE_BUFFER (job->priv->doc),
529 "tab-width", gtk_source_view_get_tab_width (GTK_SOURCE_VIEW (job->priv->view)),
530 "highlight-syntax", gtk_source_buffer_get_highlight_syntax (GTK_SOURCE_BUFFER (job->priv->doc)) &&
531 syntax_hl,
532 "wrap-mode", wrap_mode,
533 "print-line-numbers", print_line_numbers,
534 "print-header", print_header,
535 "print-footer", FALSE,
536 "body-font-name", print_font_body,
537 "line-numbers-font-name", print_font_numbers,
538 "print-footer", FALSE,
539 "body-font-name", print_font_body,
540 "line-numbers-font-name", print_font_numbers,
541 "header-font-name", print_font_header,
542 NULL));
543
544 g_free (print_font_body);
545 g_free (print_font_header);
546 g_free (print_font_numbers);
547
548 if (print_header)
549 {
550 gchar *doc_name;
551 gchar *name_to_display;
552 gchar *left;
553
554 doc_name = pluma_document_get_uri_for_display (job->priv->doc);
555 name_to_display = pluma_utils_str_middle_truncate (doc_name, 60);
556
557 left = g_strdup_printf (_("File: %s"), name_to_display);
558
559 /* Translators: %N is the current page number, %Q is the total
560 * number of pages (ex. Page 2 of 10)
561 */
562 gtk_source_print_compositor_set_header_format (job->priv->compositor,
563 TRUE,
564 left,
565 NULL,
566 _("Page %N of %Q"));
567
568 g_free (doc_name);
569 g_free (name_to_display);
570 g_free (left);
571 }
572}

628static gboolean
629paginate_cb (GtkPrintOperation *operation,
630 GtkPrintContext *context,
631 PlumaPrintJob *job)
632{
633 gboolean res;
634
635 job->priv->status = PLUMA_PRINT_JOB_STATUS_PAGINATING;
636
637 res = gtk_source_print_compositor_paginate (job->priv->compositor, context);
638
639 if (res)
640 {
641 gint n_pages;
642
643 n_pages = gtk_source_print_compositor_get_n_pages (job->priv->compositor);
644 gtk_print_operation_set_n_pages (job->priv->operation, n_pages);
645 }
646
647 job->priv->progress = gtk_source_print_compositor_get_pagination_progress (job->priv->compositor);
648
649 /* When previewing, the progress is just for pagination, when printing
650 * it's split between pagination and rendering */
651 if (!job->priv->is_preview)
652 job->priv->progress /= 2.0;
653
654 g_signal_emit (job, print_job_signals[PRINTING], 0, job->priv->status);
655
656 return res;
657}

659static void
660draw_page_cb (GtkPrintOperation *operation,
661 GtkPrintContext *context,
662 gint page_nr,
663 PlumaPrintJob *job)
664{
665 gint n_pages;
666
667 /* In preview, pages are drawn on the fly, so rendering is
668 * not part of the progress */
669 if (!job->priv->is_preview)
670 {
671 g_free (job->priv->status_string);
672
673 n_pages = gtk_source_print_compositor_get_n_pages (job->priv->compositor);
674
675 job->priv->status = PLUMA_PRINT_JOB_STATUS_DRAWING;
676
677 job->priv->status_string = g_strdup_printf ("Rendering page %d of %d...",
678 page_nr + 1,
679 n_pages);
680
681 job->priv->progress = page_nr / (2.0 * n_pages) + 0.5;
682

698static void
699done_cb (GtkPrintOperation *operation,
700 GtkPrintOperationResult result,
701 PlumaPrintJob *job)
702{
703 GError *error = NULL;
704 PlumaPrintJobResult print_result;
705
706 switch (result)
707 {
708 case GTK_PRINT_OPERATION_RESULT_CANCEL:
709 print_result = PLUMA_PRINT_JOB_RESULT_CANCEL;
710 break;
711
712 case GTK_PRINT_OPERATION_RESULT_APPLY:
713 print_result = PLUMA_PRINT_JOB_RESULT_OK;
714 break;
715
716 case GTK_PRINT_OPERATION_RESULT_ERROR:
717 print_result = PLUMA_PRINT_JOB_RESULT_ERROR;
718 gtk_print_operation_get_error (operation, &error);
719 break;
720
721 default:
722 g_return_if_reached ();
723 }
724
725 /* Avoid job is destroyed in the handler of the "done" message */
726 g_object_ref (job);
727
728 g_signal_emit (job, print_job_signals[DONE], 0, print_result, error);
729
730 g_object_unref (operation);
731 job->priv->operation = NULL;
732
733 g_object_unref (job);
734}
