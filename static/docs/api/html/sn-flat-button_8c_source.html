33static gboolean
34sn_flat_button_draw (GtkWidget *widget,
35 cairo_t *cr)
36{
37 GtkWidget *child = gtk_bin_get_child (GTK_BIN (widget));
38
39 if (child)
40 gtk_container_propagate_draw (GTK_CONTAINER (widget), child, cr);
41
42 if (gtk_widget_is_drawable (widget) && gtk_widget_has_focus (widget))
43 {
44 GtkStyleContext *context = gtk_widget_get_style_context (widget);
45 gdouble x1, y1, x2, y2;
46
47 cairo_clip_extents (cr, &x1, &y1, &x2, &y2);
48 gtk_render_focus (context, cr, x1, y1, x2 - x1, y2 - y1);
49 }
50
51 return child != NULL;
52}
