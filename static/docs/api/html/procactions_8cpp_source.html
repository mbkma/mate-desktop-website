36static void
37renice_single_process (GtkTreeModel *model, GtkTreePath *path, GtkTreeIter *iter, gpointer data)
38{
39 const struct ReniceArgs * const args = static_cast<ReniceArgs*>(data);
40
41 ProcInfo *info = NULL;
42 gint error;
43 int saved_errno;
44 gchar *error_msg;
45 GtkWidget *dialog;
46
47 gtk_tree_model_get (model, iter, COL_POINTER, &info, -1);
48
49 if (!info)
50 return;
51 if (info->nice == args->nice_value)
52 return;
53 error = setpriority (PRIO_PROCESS, info->pid, args->nice_value);
54
55 /* success */
56 if(error != -1) return;
57
58 saved_errno = errno;
59
60 /* need to be root */
61 if(errno == EPERM || errno == EACCES) {
62 gboolean success;
63
64 success = procdialog_create_root_password_dialog (
65 PROCMAN_ACTION_RENICE, args->procdata, info->pid,
66 args->nice_value);
67
68 if(success) return;
69
70 if(errno) {
71 saved_errno = errno;
72 }
73 }
74
75 /* failed */
76 error_msg = g_strdup_printf (
77 _("Cannot change the priority of process with PID %d to %d.\n"
78 "%s"),
79 info->pid, args->nice_value, g_strerror(saved_errno));
80
81 dialog = gtk_message_dialog_new (
82 NULL,
83 GTK_DIALOG_DESTROY_WITH_PARENT,
84 GTK_MESSAGE_ERROR,
85 GTK_BUTTONS_OK,
86 "%s", error_msg);
87
88 gtk_dialog_run (GTK_DIALOG (dialog));
89 gtk_widget_destroy (dialog);
90 g_free (error_msg);
91}

96{
97 struct ReniceArgs args = { procdata, nice };

99 /* EEEK - ugly hack - make sure the table is not updated as a crash
100 ** occurs if you first kill a process and the tree node is removed while
101 ** still in the foreach function

104
