33static UpClient *upc;
34static void (*status_updated_callback) (void);
35
36/* status_updated_callback () can not be called directly because at the time of
37 * the device-remove signal, the device is not actually removed from the list
38 * of devices known to the up_client object (see libupower-glib/up-client.c in
39 * upower). Waiting for the next idle timer works around this issue and has has
40 * the additionaly benefit of possibly running status_updated_callback only
41 * once when several events happen very soon after each other.
42 */
43static gboolean status_update_scheduled;
44
45static gboolean
46update_status_idle (gpointer junk)
47{
48 if (status_updated_callback)
49 status_updated_callback ();
50
51 return status_update_scheduled = FALSE;
52}
53
54static void
55schedule_status_callback (void)
56{
57 if (status_update_scheduled)
58 return;
59
60 status_update_scheduled = TRUE;
61 g_idle_add (update_status_idle, NULL);
62}

76char *
77battstat_upower_initialise (void (*callback) (void))
78{
79 status_updated_callback = callback;
80
81 if (upc != NULL)
82 return g_strdup ("Already initialised!");
83
84 if ((upc = up_client_new ()) == NULL)
85 goto error_out;
86
87 GPtrArray *devices;
88 devices = up_client_get_devices2 (upc);
89 if (!devices) {
90 goto error_shutdownclient;
91 }
92 g_ptr_array_unref (devices);
93
94 g_signal_connect_after (upc, "device-added", G_CALLBACK (device_cb), NULL);
95 g_signal_connect_after (upc, "device-removed", G_CALLBACK (device_removed_cb), NULL);
96
97 return NULL;
98
99error_shutdownclient:
100 g_object_unref (upc);
101 upc = NULL;
102
103error_out:
104 return g_strdup ("Can not initialize upower");
105}

119/* This function currently exists to allow the multiple batteries supported
120 * by the upower backend to appear as a single composite battery device (since
121 * at the current time this is all that battstat supports).
122 *
123 * This entire function is filled with logic to make multiple batteries
124 * appear as one "composite" battery. Comments included as appropriate.
125 *
126 * For more information about some of the assumptions made in the following
127 * code please see the following mailing list post and the resulting thread:
128 *
129 * http://lists.freedesktop.org/archives/hal/2005-July/002841.html
130 */
131void
132battstat_upower_get_battery_info (BatteryStatus *status)
133{
134
135 GPtrArray *devices = up_client_get_devices2 (upc);
136
137 /* The calculation to get overall percentage power remaining is as follows:
138 *
139 * Sum (Current charges) / Sum (Full Capacities)
140 *
141 * We can't just take an average of all of the percentages since this
142 * doesn't deal with the case that one battery might have a larger
143 * capacity than the other.
144 *
145 * In order to do this calculation, we need to keep a running total of
146 * current charge and full capacities.
147 */
148 double current_charge_total = 0, full_capacity_total = 0;
149
150 /* Record the time remaining as reported by upower. This is used in the event
151 * that the system has exactly one battery (since, then, upower is capable
152 * of providing an accurate time remaining report and we should trust it.)
153 */
154 gint64 remaining_time = 0;
155
156 /* The total (dis)charge rate of the system is the sum of the rates of
157 * the individual batteries.
158 */
159 double rate_total = 0;
160
161 /* We need to know if we should report the composite battery as present
162 * at all. The logic is that if at least one actual battery is installed
163 * then the composite battery will be reported to exist.
164 */
165 int present = 0;
166
167 /* We need to know if we are on AC power or not. Eventually, we can look
168 * at the AC adaptor upower devices to determine that. For now, we assume that
169 * if any battery is discharging then we must not be on AC power. Else, by
170 * default, we must be on AC.
171 */
172 int on_ac_power = 1;
173
174 /* Finally, we consider the composite battery to be "charging" if at least
175 * one of the actual batteries in the system is charging.
176 */
177 int charging = 0;
178
179 /* For each physical battery bay... */
180 int i;
181 for (i = 0; i < devices->len; i++)
182 {
183 UpDevice *upd = g_ptr_array_index (devices, i);
184
185 int type, state;
186 double current_charge, full_capacity, rate;
187 gint64 time_to_full, time_to_empty;
188
189 g_object_get (upd,
190 "kind", &type,
191 "state", &state,
192 "energy", &current_charge,
193 "energy-full", &full_capacity,
194 "energy-rate", &rate,
195 "time-to-full", &time_to_full,
196 "time-to-empty", &time_to_empty,
197 NULL);
198
199 /* Only count batteries here */
200
201 if (type != UP_DEVICE_KIND_BATTERY)
202 continue;
203
204 /* At least one battery present -> composite battery is present. */
205 present++;
206
207 /* At least one battery charging -> composite battery is charging. */
208 if (state == UP_DEVICE_STATE_CHARGING)
209 charging = 1;
210
211 /* At least one battery is discharging -> we're not on AC. */
212 if (state == UP_DEVICE_STATE_DISCHARGING)
213 on_ac_power = 0;
214
215 /* Sum the totals for current charge, design capacity, (dis)charge rate. */
216 current_charge_total += current_charge;
217 full_capacity_total += full_capacity;
218 rate_total += rate;
219
220 /* Record remaining time too, incase this is the only battery. */
221 remaining_time = (state == UP_DEVICE_STATE_DISCHARGING ? time_to_empty : time_to_full);
222 }
223
224 if (!present || full_capacity_total <= 0 || (charging && !on_ac_power))
225 {
226 /* Either no battery is present or something has gone horribly wrong.
227 * In either case we must return that the composite battery is not
228 * present.
229 */
230 status->present = FALSE;
231 status->percent = 0;
232 status->minutes = -1;
233 status->on_ac_power = TRUE;
234 status->charging = FALSE;
235
236 g_ptr_array_unref (devices);
237 return;
238 }
239
240 /* Else, our composite battery is present. */
241 status->present = TRUE;
242
243 /* As per above, overall charge is:
244 *
245 * Sum (Current charges) / Sum (Full Capacities)
246 */
247 status->percent = (current_charge_total / full_capacity_total) * 100.0 + 0.5;
248
249 if (present == 1)
250 {
251 /* In the case of exactly one battery, report the time remaining figure
252 * from upower directly since it might have come from an authorative source
253 * (ie: the PMU or APM subsystem).
254 *
255 * upower gives remaining time in seconds with a 0 to mean that the
256 * remaining time is unknown. Battstat uses minutes and -1 for
257 * unknown time remaining.
258 */
259
260 if (remaining_time == 0)
261 status->minutes = -1;
262 else
263 status->minutes = (remaining_time + 30) / 60;
264 }
265 /* Rest of cases to deal with multiple battery systems... */
266 else if (!on_ac_power && rate_total != 0)
267 {
268 /* Then we're discharging. Calculate time remaining until at zero. */
269
270 double remaining;
271
272 remaining = current_charge_total;
273 remaining /= rate_total;
274 status->minutes = (int) floor (remaining * 60.0 + 0.5);
275 }
276 else if (charging && rate_total != 0)
277 {
278 /* Calculate time remaining until charged. For systems with more than
279 * one battery, this code is very approximate. The assumption is that if
280 * one battery reaches full charge before the other that the other will
281 * start charging faster due to the increase in available power (similar
282 * to how a laptop will charge faster if you're not using it).
283 */
284
285 double remaining;
286
287 remaining = full_capacity_total - current_charge_total;
288 if (remaining < 0)
289 remaining = 0;
290 remaining /= rate_total;
291
292 status->minutes = (int) floor (remaining * 60.0 + 0.5);
293 }
294 else
295 {
296 /* On AC power and not charging -or- rate is unknown. */
297 status->minutes = -1;
298 }
299
300 /* These are simple and well-explained above. */
301 status->charging = charging;
302 status->on_ac_power = on_ac_power;
303
304 g_ptr_array_unref (devices);
305}
