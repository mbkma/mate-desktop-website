101static gboolean
102engine_get_airplane_mode_helper (GHashTable *killswitches)
103{
104 GHashTableIter iter;
105 gpointer key, value;
106
107 if (g_hash_table_size (killswitches) == 0)
108 return FALSE;
109
110 g_hash_table_iter_init (&iter, killswitches);
111 while (g_hash_table_iter_next (&iter, &key, &value)) {
112 int state;
113
114 state = GPOINTER_TO_INT (value);
115
116 /* A single rfkill switch that's unblocked? Airplane mode is off */
117 if (state == RFKILL_STATE_UNBLOCKED)
118 return FALSE;
119 }
120
121 return TRUE;
122}

131engine_get_bluetooth_hardware_airplane_mode (MsdRfkillManager *manager)

133 GHashTableIter iter;
134 gpointer key, value;

136 /* If we have no killswitches, hw airplane mode is off. */
137 if (g_hash_table_size (manager->priv->bt_killswitches) == 0)
138 return FALSE;
139
140 g_hash_table_iter_init (&iter, manager->priv->bt_killswitches);
141 while (g_hash_table_iter_next (&iter, &key, &value)) {
142 int state;
143
144 state = GPOINTER_TO_INT (value);

146 /* A single rfkill switch that's not hw blocked? Hw airplane mode is off */
147 if (state != RFKILL_STATE_HARD_BLOCKED) {
148 return FALSE;
149 }
150 }
151
152 return TRUE;

161static gboolean
162engine_get_airplane_mode (MsdRfkillManager *manager)
163{
164 if (!manager->priv->wwan_interesting)
165 return engine_get_airplane_mode_helper (manager->priv->killswitches);
166 /* wwan enabled? then airplane mode is off (because an USB modem
167 could be on in this state) */
168 return engine_get_airplane_mode_helper (manager->priv->killswitches) && !manager->priv->wwan_enabled;
169}

212static void
213engine_properties_changed (MsdRfkillManager *manager)
214{
215 GVariantBuilder props_builder;
216 GVariant *props_changed = NULL;
217
218 /* not yet connected to the session bus */
219 if (manager->priv->connection == NULL)
220 return;
221
222 g_variant_builder_init (&props_builder, G_VARIANT_TYPE ("a{sv}"));
223
224 g_variant_builder_add (&props_builder, "{sv}", "AirplaneMode",
225 g_variant_new_boolean (engine_get_airplane_mode (manager)));
226 g_variant_builder_add (&props_builder, "{sv}", "HardwareAirplaneMode",
227 g_variant_new_boolean (engine_get_hardware_airplane_mode (manager)));
228 g_variant_builder_add (&props_builder, "{sv}", "HasAirplaneMode",
229 g_variant_new_boolean (engine_get_has_airplane_mode (manager)));
230 g_variant_builder_add (&props_builder, "{sv}", "ShouldShowAirplaneMode",
231 g_variant_new_boolean (engine_get_should_show_airplane_mode (manager)));
232 g_variant_builder_add (&props_builder, "{sv}", "BluetoothAirplaneMode",
233 g_variant_new_boolean (engine_get_bluetooth_airplane_mode (manager)));
234 g_variant_builder_add (&props_builder, "{sv}", "BluetoothHardwareAirplaneMode",
235 g_variant_new_boolean (engine_get_bluetooth_hardware_airplane_mode (manager)));
236 g_variant_builder_add (&props_builder, "{sv}", "BluetoothHasAirplaneMode",
237 g_variant_new_boolean (engine_get_has_bluetooth_airplane_mode (manager)));
238
239 props_changed = g_variant_new ("(s@a{sv}@as)", MSD_RFKILL_DBUS_NAME,
240 g_variant_builder_end (&props_builder),
241 g_variant_new_strv (NULL, 0));
242
243 g_dbus_connection_emit_signal (manager->priv->connection,
244 NULL,
245 MSD_RFKILL_DBUS_PATH,
246 "org.freedesktop.DBus.Properties",
247 "PropertiesChanged",
248 props_changed, NULL);
249}

251static void
252rfkill_changed (CcRfkillGlib *rfkill G_GNUC_UNUSED,
253 GList *events,
254 MsdRfkillManager *manager)
255{
256 GList *l;
257 int value;
258
259 for (l = events; l != NULL; l = l->next) {
260 struct rfkill_event *event = l->data;
261
262 switch (event->op) {
263 case RFKILL_OP_ADD:
264 case RFKILL_OP_CHANGE:
265 if (event->hard)
266 value = RFKILL_STATE_HARD_BLOCKED;
267 else if (event->soft)
268 value = RFKILL_STATE_SOFT_BLOCKED;
269 else
270 value = RFKILL_STATE_UNBLOCKED;
271
272 g_hash_table_insert (manager->priv->killswitches,
273 GINT_TO_POINTER (event->idx),
274 GINT_TO_POINTER (value));
275 if (event->type == RFKILL_TYPE_BLUETOOTH)
276 g_hash_table_insert (manager->priv->bt_killswitches,
277 GINT_TO_POINTER (event->idx),
278 GINT_TO_POINTER (value));
279 g_debug ("%s %srfkill with ID %d",
280 event->op == RFKILL_OP_ADD ? "Added" : "Changed",
281 event->type == RFKILL_TYPE_BLUETOOTH ? "Bluetooth " : "",
282 event->idx);
283 break;
284 case RFKILL_OP_DEL:
285 g_hash_table_remove (manager->priv->killswitches,
286 GINT_TO_POINTER (event->idx));
287 if (event->type == RFKILL_TYPE_BLUETOOTH)
288 g_hash_table_remove (manager->priv->bt_killswitches,
289 GINT_TO_POINTER (event->idx));
290 g_debug ("Removed %srfkill with ID %d", event->type == RFKILL_TYPE_BLUETOOTH ? "Bluetooth " : "",
291 event->idx);
292 break;
293 }
294 }
295
296 engine_properties_changed (manager);
297}

299static void
300rfkill_set_cb (GObject *source_object,
301 GAsyncResult *res,
302 gpointer user_data)
303{
304 gboolean ret;
305 GError *error = NULL;
306
307 ret = cc_rfkill_glib_send_change_all_event_finish (CC_RFKILL_GLIB (source_object), res, &error);
308 if (!ret) {
309 if (g_error_matches (error, G_IO_ERROR, G_IO_ERROR_TIMED_OUT))
310 g_debug ("Timed out waiting for blocked rfkills");
311 else if (!g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
312 g_warning ("Failed to set RFKill: %s", error->message);
313 g_error_free (error);
314 }
315}

348static gboolean
349engine_set_airplane_mode (MsdRfkillManager *manager,
350 gboolean enable)
351{
352 cc_rfkill_glib_send_change_all_event (manager->priv->rfkill, RFKILL_TYPE_ALL,
353 enable, manager->priv->cancellable, rfkill_set_cb, manager);
354
355 /* Note: we set the the NM property even if there are no modems, so we don't
356 need to resync when one is plugged in */
357 if (manager->priv->nm_client) {
358 g_dbus_proxy_call (manager->priv->nm_client,
359 "org.freedesktop.DBus.Properties.Set",
360 g_variant_new ("(ssv)",
361 "org.freedesktop.NetworkManager",
362 "WwanEnabled",
363 g_variant_new_boolean (!enable)),
364 G_DBUS_CALL_FLAGS_NONE,
365 -1, /* timeout */
366 manager->priv->cancellable,
367 set_wwan_complete, NULL);
368 }
369
370 return TRUE;
371}

373static gboolean
374handle_set_property (GDBusConnection *connection,
375 const gchar *sender,
376 const gchar *object_path,
377 const gchar *interface_name,
378 const gchar *property_name,
379 GVariant *value,
380 GError **error,
381 gpointer user_data)
382{
383 MsdRfkillManager *manager = MSD_RFKILL_MANAGER (user_data);
384
385 if (g_strcmp0 (property_name, "AirplaneMode") == 0) {
386 gboolean airplane_mode;
387 g_variant_get (value, "b", &airplane_mode);
388 return engine_set_airplane_mode (manager, airplane_mode);
389 } else if (g_strcmp0 (property_name, "BluetoothAirplaneMode") == 0) {
390 gboolean airplane_mode;
391 g_variant_get (value, "b", &airplane_mode);
392 return engine_set_bluetooth_airplane_mode (manager, airplane_mode);
393 }
394
395 return FALSE;
396}

398static GVariant *
399handle_get_property (GDBusConnection *connection,
400 const gchar *sender,
401 const gchar *object_path,
402 const gchar *interface_name,
403 const gchar *property_name,
404 GError **error,
405 gpointer user_data)
406{
407 MsdRfkillManager *manager = MSD_RFKILL_MANAGER (user_data);
408
409 /* Check session pointer as a proxy for whether the manager is in the
410 start or stop state */
411 if (manager->priv->connection == NULL) {
412 return NULL;
413 }
414
415 if (g_strcmp0 (property_name, "AirplaneMode") == 0) {
416 gboolean airplane_mode;
417 airplane_mode = engine_get_airplane_mode (manager);
418 return g_variant_new_boolean (airplane_mode);
419 }
420
421 if (g_strcmp0 (property_name, "HardwareAirplaneMode") == 0) {
422 gboolean hw_airplane_mode;
423 hw_airplane_mode = engine_get_hardware_airplane_mode (manager);
424 return g_variant_new_boolean (hw_airplane_mode);
425 }
426
427 if (g_strcmp0 (property_name, "ShouldShowAirplaneMode") == 0) {
428 gboolean should_show_airplane_mode;
429 should_show_airplane_mode = engine_get_should_show_airplane_mode (manager);
430 return g_variant_new_boolean (should_show_airplane_mode);
431 }
432
433 if (g_strcmp0 (property_name, "HasAirplaneMode") == 0) {
434 gboolean has_airplane_mode;
435 has_airplane_mode = engine_get_has_airplane_mode (manager);
436 return g_variant_new_boolean (has_airplane_mode);
437 }
438
439 if (g_strcmp0 (property_name, "BluetoothAirplaneMode") == 0) {
440 gboolean airplane_mode;
441 airplane_mode = engine_get_bluetooth_airplane_mode (manager);
442 return g_variant_new_boolean (airplane_mode);
443 }
444
445 if (g_strcmp0 (property_name, "BluetoothHardwareAirplaneMode") == 0) {
446 gboolean hw_airplane_mode;
447 hw_airplane_mode = engine_get_bluetooth_hardware_airplane_mode (manager);
448 return g_variant_new_boolean (hw_airplane_mode);
449 }
450
451 if (g_strcmp0 (property_name, "BluetoothHasAirplaneMode") == 0) {
452 gboolean has_airplane_mode;
453 has_airplane_mode = engine_get_has_bluetooth_airplane_mode (manager);
454 return g_variant_new_boolean (has_airplane_mode);
455 }
456
457 return NULL;
458}

468static void
469on_bus_gotten (GObject *source_object,
470 GAsyncResult *res,
471 MsdRfkillManager *manager)
472{
473 GDBusConnection *connection;
474 GError *error = NULL;
475
476 connection = g_bus_get_finish (res, &error);
477 if (connection == NULL) {
478 if (!g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED))
479 g_warning ("Could not get session bus: %s", error->message);
480 g_error_free (error);
481 return;
482 }
483 manager->priv->connection = connection;
484
485 g_dbus_connection_register_object (connection,
486 MSD_RFKILL_DBUS_PATH,
487 manager->priv->introspection_data->interfaces[0],
488 &interface_vtable,
489 manager,
490 NULL,
491 NULL);
492
493 manager->priv->name_id = g_bus_own_name_on_connection (connection,
494 MSD_DBUS_NAME,
495 G_BUS_NAME_OWNER_FLAGS_ALLOW_REPLACEMENT,
496 NULL,
497 NULL,
498 NULL,
499 NULL);
500}

502static void
503sync_wwan_enabled (MsdRfkillManager *manager)
504{
505 GVariant *property;
506
507 property = g_dbus_proxy_get_cached_property (manager->priv->nm_client,
508 "WwanEnabled");
509
510 if (property == NULL) {
511 /* GDBus telling us NM went down */
512 return;
513 }
514
515 manager->priv->wwan_enabled = g_variant_get_boolean (property);
516 engine_properties_changed (manager);
517
518 g_variant_unref (property);

521static void
522nm_signal (GDBusProxy *proxy,
523 char *sender_name G_GNUC_UNUSED,
524 char *signal_name,
525 GVariant *parameters,
526 gpointer user_data)
527{
528 MsdRfkillManager *manager = user_data;
529 GVariant *changed;
530 GVariant *property;
531
532 if (g_strcmp0 (signal_name, "PropertiesChanged") == 0) {
533 changed = g_variant_get_child_value (parameters, 0);
534 property = g_variant_lookup_value (changed, "WwanEnabled", G_VARIANT_TYPE ("b"));
535 g_dbus_proxy_set_cached_property (proxy, "WwanEnabled", property);
536
537 if (property != NULL) {
538 sync_wwan_enabled (manager);
539 g_variant_unref (property);
540 }
541
542 g_variant_unref (changed);
543 }

546static void
547on_nm_proxy_gotten (GObject *source,
548 GAsyncResult *result,
549 gpointer user_data)
550{
551 MsdRfkillManager *manager = user_data;
552 GDBusProxy *proxy;
553 GError *error;
554
555 error = NULL;
556 proxy = g_dbus_proxy_new_for_bus_finish (result, &error);
557
558 if (proxy == NULL) {
559 if (!g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED) &&
560 !g_error_matches (error, G_DBUS_ERROR, G_DBUS_ERROR_SERVICE_UNKNOWN))
561 g_warning ("Failed to acquire NetworkManager proxy: %s", error->message);
562
563 g_error_free (error);
564 goto out;
565 }
566
567 manager->priv->nm_client = proxy;
568
569 g_signal_connect (manager->priv->nm_client, "g-signal",
570 G_CALLBACK (nm_signal), manager);
571 sync_wwan_enabled (manager);
572
573 out:
574 g_object_unref (manager);
575}

593static void
594on_mm_proxy_gotten (GObject *source,
595 GAsyncResult *result,
596 gpointer user_data)
597{
598 MsdRfkillManager *manager = user_data;
599 GDBusObjectManager *proxy;
600 GError *error;
601
602 error = NULL;
603 proxy = g_dbus_object_manager_client_new_for_bus_finish (result, &error);
604
605 if (proxy == NULL) {
606 if (!g_error_matches (error, G_IO_ERROR, G_IO_ERROR_CANCELLED) &&
607 !g_error_matches (error, G_DBUS_ERROR, G_DBUS_ERROR_SERVICE_UNKNOWN))
608 g_warning ("Failed to acquire ModemManager proxy: %s", error->message);
609
610 g_error_free (error);
611 goto out;
612 }
613
614 manager->priv->mm_client = proxy;
615
616 g_signal_connect (manager->priv->mm_client, "interface-added",
617 G_CALLBACK (sync_wwan_interesting), manager);
618 g_signal_connect (manager->priv->mm_client, "interface-removed",
619 G_CALLBACK (sync_wwan_interesting), manager);
620 sync_wwan_interesting (manager->priv->mm_client, NULL, NULL, manager);
621
622 out:
623 g_object_unref (manager);
624}

626gboolean
627msd_rfkill_manager_start (MsdRfkillManager *manager,
628 GError **error)
629{
630 mate_settings_profile_start (NULL);
631
632 manager->priv->introspection_data = g_dbus_node_info_new_for_xml (introspection_xml, NULL);
633 g_assert (manager->priv->introspection_data != NULL);
634
635 manager->priv->killswitches = g_hash_table_new (g_direct_hash, g_direct_equal);
636 manager->priv->bt_killswitches = g_hash_table_new (g_direct_hash, g_direct_equal);
637 manager->priv->rfkill = cc_rfkill_glib_new ();
638 g_signal_connect (G_OBJECT (manager->priv->rfkill), "changed",
639 G_CALLBACK (rfkill_changed), manager);
640 cc_rfkill_glib_open (manager->priv->rfkill);
641
642 manager->priv->cancellable = g_cancellable_new ();
643
644 manager->priv->chassis_type = mate_settings_get_chassis_type ();
645
646 g_dbus_proxy_new_for_bus (G_BUS_TYPE_SYSTEM,
647 G_DBUS_PROXY_FLAGS_NONE,
648 NULL, /* g-interface-info */
649 "org.freedesktop.NetworkManager",
650 "/org/freedesktop/NetworkManager",
651 "org.freedesktop.NetworkManager",
652 manager->priv->cancellable,
653 on_nm_proxy_gotten, g_object_ref (manager));
654
655 g_dbus_object_manager_client_new_for_bus (G_BUS_TYPE_SYSTEM,
656 G_DBUS_OBJECT_MANAGER_CLIENT_FLAGS_DO_NOT_AUTO_START,
657 "org.freedesktop.ModemManager1",
658 "/org/freedesktop/ModemManager1",
659 NULL, NULL, NULL, /* get_proxy_type and closure */
660 manager->priv->cancellable,
661 on_mm_proxy_gotten, g_object_ref (manager));
662
663 /* Start process of owning a D-Bus name */
664 g_bus_get (G_BUS_TYPE_SESSION,
665 manager->priv->cancellable,
666 (GAsyncReadyCallback) on_bus_gotten,
667 manager);
668
669 mate_settings_profile_end (NULL);
670
671 return TRUE;
672}

674void
675msd_rfkill_manager_stop (MsdRfkillManager *manager)
676{
677 MsdRfkillManagerPrivate *p = manager->priv;
678
679 g_debug ("Stopping rfkill manager");
