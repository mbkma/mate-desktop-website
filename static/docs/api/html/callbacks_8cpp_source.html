87void
88cb_renice (GtkAction *action, GtkRadioAction *current, gpointer data)
89{
90 ProcData * const procdata = static_cast<ProcData*>(data);
91
92 gint selected = gtk_radio_action_get_current_value(current);
93
94 if (selected == CUSTOM_PRIORITY)
95 {
96 procdialog_create_renice_dialog (procdata);
97 } else {
98 gint new_nice_value = 0;
99 switch (selected) {
100 case VERY_HIGH_PRIORITY: new_nice_value = -20; break;
101 case HIGH_PRIORITY: new_nice_value = -5; break;
102 case NORMAL_PRIORITY: new_nice_value = 0; break;
103 case LOW_PRIORITY: new_nice_value = 5; break;
104 case VERY_LOW_PRIORITY: new_nice_value = 19; break;
105 }
106 renice(procdata, new_nice_value);
107 }
108}

351void
352cb_row_selected (GtkTreeSelection *selection, gpointer data)
353{
354 ProcData * const procdata = static_cast<ProcData*>(data);
355
356 procdata->selection = selection;
357
358 procdata->selected_process = NULL;
359
360 /* get the most recent selected process and determine if there are
361 ** no selected processes
362 */
363 gtk_tree_selection_selected_foreach (procdata->selection, get_last_selected,
364 &procdata->selected_process);
365 if (procdata->selected_process) {
366 gint value;
367 gint nice = procdata->selected_process->nice;
368 if (nice < -7)
369 value = VERY_HIGH_PRIORITY;
370 else if (nice < -2)
371 value = HIGH_PRIORITY;
372 else if (nice < 3)
373 value = NORMAL_PRIORITY;
374 else if (nice < 7)
375 value = LOW_PRIORITY;
376 else
377 value = VERY_LOW_PRIORITY;
378
379 GtkRadioAction* normal = GTK_RADIO_ACTION(gtk_action_group_get_action(procdata->action_group, "Normal"));
380 block_priority_changed_handlers(procdata, TRUE);
381 gtk_radio_action_set_current_value(normal, value);
382 block_priority_changed_handlers(procdata, FALSE);
383
384 }
385 update_sensitivity(procdata);
386}

421void
422cb_change_current_page (GtkNotebook *nb, gint num, gpointer data)
423{
424 ProcData * const procdata = static_cast<ProcData*>(data);
425
426 procdata->config.current_tab = num;
427
428
429 if (num == PROCMAN_TAB_PROCESSES) {
430
431 cb_timeout (procdata);
432
433 if (!procdata->timeout)
434 procdata->timeout = g_timeout_add (
435 procdata->config.update_interval,
436 cb_timeout, procdata);
437
438 update_sensitivity(procdata);
439 }
440 else {
441 if (procdata->timeout) {
442 g_source_remove (procdata->timeout);
443 procdata->timeout = 0;
444 }
445
446 update_sensitivity(procdata);
447 }
448
449
450 if (num == PROCMAN_TAB_RESOURCES) {
451 load_graph_start (procdata->cpu_graph);
452 load_graph_start (procdata->mem_graph);
453 load_graph_start (procdata->net_graph);
454 }
455 else {
456 load_graph_stop (procdata->cpu_graph);
457 load_graph_stop (procdata->mem_graph);
458 load_graph_stop (procdata->net_graph);
459 }
460
461
462 if (num == PROCMAN_TAB_DISKS) {
463
464 cb_update_disks (procdata);
465
466 if(!procdata->disk_timeout) {
467 procdata->disk_timeout =
468 g_timeout_add (procdata->config.disks_update_interval,
469 cb_update_disks,
470 procdata);
471 }
472 }
473 else {
474 if(procdata->disk_timeout) {
475 g_source_remove (procdata->disk_timeout);
476 procdata->disk_timeout = 0;
477 }
478 }
479
480 if (num == PROCMAN_TAB_SYSINFO) {
481 procman::build_sysinfo_ui();
482 }
483}

497cb_timeout (gpointer data)
498{
499 ProcData * const procdata = static_cast<ProcData*>(data);
500 guint new_interval;
501
502 if (!procdata->terminating) {
503 proctable_update (procdata);
504
505 if (procdata->smooth_refresh->get(new_interval))

508 cb_timeout,

511 }
512 }
513 return TRUE;
