156int
157na_core_utils_str_collate( const gchar *str1, const gchar *str2 )
158{
159 int res;
160
161 if( str1 && str2 ){
162 res = g_utf8_collate( str1, str2 );
163
164 } else if( !str1 && !str2 ){
165 res = 0;
166
167 } else if( !str1 ){
168 res = -1;
169
170 } else {
171 g_return_val_if_fail( str2 == NULL, 0 );
172 res = 1;
173 }
174 return( res );
175}

189gchar *
190na_core_utils_str_remove_char( const gchar *string, const gchar *to_remove )
191{
192 static const gchar *thisfn = "na_core_utils_str_remove_char";
193 gchar *removed;
194 GRegex *regex;
195 GError *error;
196
197 removed = g_strdup( string );
198
199 if( g_utf8_validate( string, -1, NULL )){
200
201 error = NULL;
202 regex = g_regex_new( to_remove, 0, 0, &error );
203 if( error ){
204 g_warning( "%s [g_regex_new] %s", thisfn, error->message );
205 g_error_free( error );
206
207 } else {
208 g_free( removed );
209 removed = g_regex_replace_literal( regex, string, -1, 0, "", 0, &error );
210 if( error ){
211 g_warning( "%s [g_regex_replace_literal] %s", thisfn, error->message );
212 g_error_free( error );
213 }
214 }
215 }
216
217 return( removed );

782/***
783 * na_core_utils_selcount_get_ope:
784 * @selcount: the selection count condition string.
785 * @ope: a pointer to a newly allocated string which will contains the
786 * operation code.
787 * @uint: a pointer to a newly allocated string which will contains the
788 * relevant integer.
789 *
790 * Parses a selection count string, and extract the operation code and the
791 * relevant integer.
792 *
793 * The two returned strings must be g_free() by the caller.
794 *
795 * Since: 2.30
796 */
797void
798na_core_utils_selcount_get_ope_int( const gchar *selcount, gchar **ope, gchar **uint )
799{
800 gchar *dup, *dup2;
801 guint uint_int;
802
803 g_return_if_fail( ope && uint );
804
805 *ope = NULL;
806 *uint = NULL;
807
808 dup = g_strstrip( g_strdup( selcount ));
809 *ope = g_strdup( " " );
810 *ope[0] = dup[0];
811
812 dup2 = g_strstrip( g_strdup( dup+1 ));
813 uint_int = abs( atoi( dup2 ));
814 *uint = g_strdup_printf( "%d", uint_int );
815
816 g_free( dup2 );
817 g_free( dup );
818}

886static gboolean
887info_dir_is_writable( GFile *file, const gchar *path_or_uri )
888{
889 static const gchar *thisfn = "na_core_utils_info_dir_is_writable";
890 GError *error = NULL;
891 GFileInfo *info;
892 GFileType type;
893 gboolean writable;
894
895 info = g_file_query_info( file,
896 G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE "," G_FILE_ATTRIBUTE_STANDARD_TYPE,
897 G_FILE_QUERY_INFO_NONE, NULL, &error );
898
899 if( error ){
900 if( error->code != G_IO_ERROR_NOT_FOUND ){
901 g_warning( "%s: g_file_query_info error: %s", thisfn, error->message );
902 }
903 g_error_free( error );
904 return( FALSE );
905 }
906
907 type = g_file_info_get_file_type( info );
908 if( type != G_FILE_TYPE_DIRECTORY ){
909 g_debug( "%s: %s is not a directory", thisfn, path_or_uri );
910 g_object_unref( info );
911 return( FALSE );
912 }
913
914 writable = g_file_info_get_attribute_boolean( info, G_FILE_ATTRIBUTE_ACCESS_CAN_WRITE );
915 if( !writable ){
916 g_debug( "%s: %s is not writable", thisfn, path_or_uri );
917 }
918
919 g_object_unref( info );
920
921 return( writable );
922}

1067static gboolean
1068file_is_loadable( GFile *file )
1069{
1070 static const gchar *thisfn = "na_core_utils_file_is_loadable";
1071 GError *error;
1072 GFileInfo *info;
1073 guint64 size;
1074 GFileType type;
1075 gboolean isok;
1076 GFile *target_file;
1077
1078 error = NULL;
1079 isok = FALSE;
1080 info = g_file_query_info( file,
1081 G_FILE_ATTRIBUTE_STANDARD_SIZE "," G_FILE_ATTRIBUTE_STANDARD_TYPE,
1082 G_FILE_QUERY_INFO_NONE, NULL, &error );
1083
1084 if( !info ){
1085 if( error ){
1086 g_debug( "%s: %s", thisfn, error->message );
1087 g_error_free( error );
1088 }
1089
1090 } else {
1091 size = g_file_info_get_attribute_uint64( info, G_FILE_ATTRIBUTE_STANDARD_SIZE );
1092 g_debug( "%s: size=%lu", thisfn, ( unsigned long ) size );
1093 isok = ( size >= CA_SIZE_MIN && size <= CA_SIZE_MAX );
1094 }
1095
1096 if( isok ){
1097 type = g_file_info_get_file_type( info );
1098 g_debug( "%s: type=%u", thisfn, ( unsigned ) type );
1099
1100 if( type != G_FILE_TYPE_REGULAR ){
1101 isok = FALSE;
1102
1103 if( type == G_FILE_TYPE_SYMBOLIC_LINK ){
1104 const char *target = g_file_info_get_symlink_target( info );
1105 if( target && strlen( target )){
1106 target_file = g_file_resolve_relative_path( file, target );
1107 if( target_file ){
1108 isok = file_is_loadable( target_file );
1109 g_object_unref( target_file );
1110 }
1111 }
1112 }
1113 }
1114 }
1115
