281/*
282 * @a is the original object
283 * @b is the current one
284 *
285 * Even if they have both the same children list, the current action is
286 * considered modified as soon as one of its profile is itself modified.
287 */
288static gboolean
289object_are_equal( const NAObject *a, const NAObject *b )
290{
291 static const gchar *thisfn = "na_object_action_object_are_equal";
292 GList *it;
293 gboolean are_equal;
294
295 g_debug( "%s: a=%p, b=%p", thisfn, ( void * ) a, ( void * ) b );
296
297 for( it = na_object_get_items( b ) ; it ; it = it->next ){
298 if( na_object_is_modified( it->data )){
299 return( FALSE );
300 }
301 }
302
303 are_equal = TRUE;
304
305 /* chain call to parent class */
306 if( NA_OBJECT_CLASS( st_parent_class )->are_equal ){
307 are_equal &= NA_OBJECT_CLASS( st_parent_class )->are_equal( a, b );
308 }
309
310 return( are_equal );
311}

313static gboolean
314object_is_valid( const NAObject *object )
315{
316 static const gchar *thisfn = "na_object_action_object_is_valid";
317 gboolean is_valid;
318 NAObjectAction *action;
319
320 g_return_val_if_fail( NA_IS_OBJECT_ACTION( object ), FALSE );
321
322 is_valid = FALSE;
323 action = NA_OBJECT_ACTION( object );
324
325 if( !action->private->dispose_has_run ){
326 g_debug( "%s: action=%p (%s)", thisfn, ( void * ) action, G_OBJECT_TYPE_NAME( action ));
327
328 is_valid = TRUE;
329
330 if( na_object_is_target_toolbar( action )){
331 is_valid &= is_valid_toolbar_label( action );
332 }
333
334 if( na_object_is_target_selection( action ) || na_object_is_target_location( action )){
335 is_valid &= is_valid_label( action );
336 }
337
338 if( !is_valid ){
339 na_object_debug_invalid( action, "no valid profile" );
340 }
341 }
342
343 /* chain up to the parent class */
344 if( NA_OBJECT_CLASS( st_parent_class )->is_valid ){
345 is_valid &= NA_OBJECT_CLASS( st_parent_class )->is_valid( object );
346 }
347
348 return( is_valid );
349}

377/*
378 * at this time, we don't yet have read the profiles as this will be
379 * triggered by ifactory_provider_read_done - we so just are able to deal with
380 * action-specific properties (not to check for profiles consistency)
381 */
382static void
383ifactory_object_read_done( NAIFactoryObject *instance, const NAIFactoryProvider *reader, void *reader_data, GSList **messages )
384{
385 guint iversion;
386 NAObjectProfile *profile;
387
388 g_debug( "na_object_action_ifactory_object_read_done: instance=%p", ( void * ) instance );
389
390 na_object_item_deals_with_version( NA_OBJECT_ITEM( instance ));
391
392 /* should attach a new profile if we detect a pre-v2 action
393 * the v1_to_v2 conversion must be followed by a v2_to_v3 one
394 */
395 iversion = na_object_get_iversion( instance );
396 if( iversion < 2 ){
397 profile = read_done_convert_v1_to_v2( instance );
398 na_object_profile_convert_v2_to_last( profile );
399 }
400
401 /* deals with obsoleted data, i.e. data which may have been written in the past
402 * but are no long written by now - not relevant for a menu
403 */
404 read_done_deals_with_toolbar_label( instance );
405
406 /* prepare the context after the reading
407 */
408 na_icontext_read_done( NA_ICONTEXT( instance ));
409
410 /* last, set action defaults
411 */
412 na_factory_object_set_defaults( instance );
413}

451/*
452 * if we have a pre-v2 action
453 * any data found in data_def_action_v1 (defined in na-object-action-factory.c)
454 * is obsoleted and should be moved to a new profile
455 *
456 * actions read from .desktop already have iversion=3 (cf. desktop_read_start)
457 * so v1 actions may only come from xml or mateconf
458 *
459 * returns the newly defined profile
460 */
461static NAObjectProfile *
462read_done_convert_v1_to_v2( NAIFactoryObject *instance )
463{
464 static const gchar *thisfn = "na_object_action_read_done_read_done_convert_v1_to_last";
465 GList *to_move;
466 NADataDef *def;
467 NADataBoxed *boxed;
468 GList *ibox;
469 NAObjectProfile *profile;
470
471 /* search for old data in the body: this is only possible before profiles
472 * because starting with contexts, iversion was written
473 */
474 to_move = NULL;
475 def = data_def_action_v1;
476
477 while( def->name ){
478 boxed = na_ifactory_object_get_data_boxed( instance, def->name );
479 if( boxed ){
480 g_debug( "%s: boxed=%p (%s) marked to be moved from action body to profile",
481 thisfn, ( void * ) boxed, def->name );
482 to_move = g_list_prepend( to_move, boxed );
483 }
484 def++;
485 }
486
487 /* now create a new profile
488 */
489 profile = na_object_profile_new();
490 na_object_set_id( profile, "profile-pre-v2" );
491 na_object_set_label( profile, _( "Profile automatically created from pre-v2 action" ));
492 na_object_attach_profile( instance, profile );
493
494 for( ibox = to_move ; ibox ; ibox = ibox->next ){
495 na_factory_object_move_boxed(
496 NA_IFACTORY_OBJECT( profile ), instance, NA_DATA_BOXED( ibox->data ));
497 }
498
499 return( profile );
500}

503 * if toolbar-same-label is true, then ensure that this is actually true
504 */
505static void
506read_done_deals_with_toolbar_label( NAIFactoryObject *instance )
507{
508 gchar *toolbar_label;
509 gchar *action_label;
510 gboolean same_label;
511
512 toolbar_label = na_object_get_toolbar_label( instance );
513 action_label = na_object_get_label( instance );
514
515 if( !toolbar_label || !g_utf8_strlen( toolbar_label, -1 )){
516 na_object_set_toolbar_label( instance, action_label );
517 na_object_set_toolbar_same_label( instance, TRUE );
518
519 } else {
520 same_label = ( na_core_utils_str_collate( action_label, toolbar_label ) == 0 );
521 na_object_set_toolbar_same_label( instance, same_label );
522 }
523
524 g_free( action_label );
525 g_free( toolbar_label );
526}

528/*
529 * write the profiles of the action
530 * note that subitems string list has been rebuilt on write_start
531 */
532static guint
533write_done_write_profiles( NAIFactoryObject *instance, const NAIFactoryProvider *writer, void *writer_data, GSList **messages )
534{
535 static const gchar *thisfn = "na_object_action_write_done_write_profiles";
536 guint code;
537 GSList *children_slist, *ic;
538 NAObjectProfile *profile;
539
540 code = NA_IIO_PROVIDER_CODE_OK;
541 children_slist = na_object_get_items_slist( instance );
542
543 for( ic = children_slist ; ic && code == NA_IIO_PROVIDER_CODE_OK ; ic = ic->next ){
544 profile = NA_OBJECT_PROFILE( na_object_get_item( instance, ic->data ));
545
546 if( profile ){
547 code = na_ifactory_provider_write_item( writer, writer_data, NA_IFACTORY_OBJECT( profile ), messages );
548
549 } else {
550 g_warning( "%s: profile not found: %s", thisfn, ( const gchar * ) ic->data );
551 }
552 }
553
554 return( code );
555}

683gchar *
684na_object_action_get_new_profile_name( const NAObjectAction *action )
685{
686 int i;
687 gboolean ok = FALSE;
688 gchar *candidate = NULL;
689 guint last_allocated;
690
691 g_return_val_if_fail( NA_IS_OBJECT_ACTION( action ), NULL );
692
693 if( !action->private->dispose_has_run ){
694
695 last_allocated = na_object_get_last_allocated( action );
696
697 for( i = last_allocated + 1 ; !ok ; ++i ){
698 g_free( candidate );
699 candidate = g_strdup_printf( "profile-%d", i );
700
701 if( !na_object_get_item( action, candidate )){
702 ok = TRUE;
703 na_object_set_last_allocated( action, i );
704 }
705 }
706
707 if( !ok ){
708 g_free( candidate );
709 candidate = NULL;
710 }
711 }
712
713 /*g_debug( "returning candidate=%s", candidate );*/
714 return( candidate );
715}
